65QBBC::QBBC( G4int ver, const G4String&)
66{
67 G4DataQuestionaire it(photon, neutronxs);
68 G4cout << "<<< Reference Physics List QBBC " <<G4endl;
69
70 defaultCutValue = 0.7*mm;
71 SetVerboseLevel(ver);
72
73 // EM Physics
74 RegisterPhysics( new G4EmStandardPhysics(ver) );
75
76 // Synchroton Radiation & GN Physics
77 RegisterPhysics( new G4EmExtraPhysics(ver) );
78
79 // Decays
80 RegisterPhysics( new G4DecayPhysics(ver) );
81
82 // Hadron Physics
83 RegisterPhysics( new G4HadronElasticPhysicsXS(ver) );
84
85 RegisterPhysics( new G4StoppingPhysics(ver) );
86
87 RegisterPhysics( new G4IonPhysicsXS(ver) );
88
89 RegisterPhysics( new G4IonElasticPhysics(ver) );
90
91 RegisterPhysics( new G4HadronInelasticQBBC(ver));
92
93 // Neutron tracking cut
94 RegisterPhysics( new G4NeutronTrackingCut(ver) );
95}
