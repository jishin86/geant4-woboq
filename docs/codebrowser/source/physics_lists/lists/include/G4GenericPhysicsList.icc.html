52template<class T> TG4GenericPhysicsList<T>::TG4GenericPhysicsList(G4int ver): T(), messenger(this, "/PhysicsList/")
53{
54 DeclareProperties();
55
56 G4DataQuestionaire it(photon);
57 G4cout << "<<< Geant4 Physics List simulation engine: G4GenericPhysicsList"<<G4endl;
58 G4cout <<G4endl;
59 this->defaultCutValue = 0.7*CLHEP::mm;
60 this->SetVerboseLevel(ver);
61}
62
63template<class T> TG4GenericPhysicsList<T>::TG4GenericPhysicsList(std::vector<G4String>* physConstr, G4int ver): T(), messenger(this, "/PhysicsList/")
64{
65 DeclareProperties();
66
67 G4DataQuestionaire it(photon);
68 G4cout << "<<< Geant4 Physics List simulation engine: G4GenericPhysicsList"<<G4endl;
69 G4cout <<G4endl;
70 this->defaultCutValue = 0.7*CLHEP::mm;
71 this->SetVerboseLevel(ver);
72
73 for(std::vector<G4String>::iterator i=physConstr->begin(); i!=physConstr->end(); i++) RegisterPhysicsConstructor(*i);
74}

76template<class T> TG4GenericPhysicsList<T>::~TG4GenericPhysicsList()
77{
78}

80template<class T> void TG4GenericPhysicsList<T>::SetCuts()
81{
82 if (this->verboseLevel >1){
83 G4cout << "G4GenericPhysicsList::SetCuts:";
84 }
85
86 this->SetCutsWithDefault();
87
88}
89
90template<class T> void TG4GenericPhysicsList<T>::DeclareProperties()
91{
92 messenger.DeclareProperty("defaultCutValue", this->defaultCutValue, "Default Cut Value");
93 messenger.DeclareMethod("SetVerboseLevel", &TG4GenericPhysicsList::SetVerboseLevel, "Verbose Level");
94
95 messenger.DeclareMethod("RegisterPhysics", &TG4GenericPhysicsList::RegisterPhysicsConstructor, "Register Physics Constructor");
96}
