39#ifndef G4BertiniPiKBuilder_h
40#define G4BertiniPiKBuilder_h 1
41
42#include "globals.hh"
43
44#include "G4HadronElasticProcess.hh"
45#include "G4HadronFissionProcess.hh"
46#include "G4HadronCaptureProcess.hh"
47#include "G4NeutronInelasticProcess.hh"
48#include "G4VPiKBuilder.hh"
49#include "G4CascadeInterface.hh"
50
51
52class G4BertiniPiKBuilder : public G4VPiKBuilder
53{
54 public:
55 G4BertiniPiKBuilder();
56 virtual ~G4BertiniPiKBuilder() {}
57
58 virtual void Build(G4HadronElasticProcess *) final override {}
59 virtual void Build(G4PionPlusInelasticProcess * aP) final override;
60 virtual void Build(G4PionMinusInelasticProcess * aP) final override;
61 virtual void Build(G4KaonPlusInelasticProcess * aP) final override;
62 virtual void Build(G4KaonMinusInelasticProcess * aP) final override;
63 virtual void Build(G4KaonZeroLInelasticProcess * aP) final override;
64 virtual void Build(G4KaonZeroSInelasticProcess * aP) final override;
65
66 virtual void SetMinEnergy(G4double aM) final override {theMin = aM;}
67 virtual void SetMaxEnergy(G4double aM) final override {theMax = aM;}
68
69 using G4VPiKBuilder::Build; //Prevent Compilation warning
70
71 private:
72 G4CascadeInterface * theModel;
73 G4double theMin;
74 G4double theMax;
75};
