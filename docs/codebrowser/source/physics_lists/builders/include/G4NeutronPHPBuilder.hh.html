26#ifndef G4NeutronPHPBuilder_h
27#define G4NeutronPHPBuilder_h 1
28
29#include "globals.hh"
30
31#include "G4HadronElasticProcess.hh"
32#include "G4HadronFissionProcess.hh"
33#include "G4HadronCaptureProcess.hh"
34#include "G4NeutronInelasticProcess.hh"
35#include "G4VNeutronBuilder.hh"
36
37#include "G4ParticleHPElasticData.hh"
38#include "G4ParticleHPElastic.hh"
39#include "G4ParticleHPInelastic.hh"
40#include "G4ParticleHPInelasticData.hh"
41#include "G4ParticleHPFission.hh"
42#include "G4ParticleHPFissionData.hh"
43#include "G4ParticleHPCapture.hh"
44#include "G4ParticleHPCaptureData.hh"
45
46class G4NeutronPHPBuilder : public G4VNeutronBuilder
47{
48 public:
49 G4NeutronPHPBuilder();
50 virtual ~G4NeutronPHPBuilder() {}
51
52 public:
53 virtual void Build(G4HadronElasticProcess * aP) final override;
54 virtual void Build(G4HadronFissionProcess * aP) final override;
55 virtual void Build(G4HadronCaptureProcess * aP) final override;
56 virtual void Build(G4NeutronInelasticProcess * aP) final override;
57
58 virtual void SetMinEnergy(G4double aM) final override
59 {
60 theMin=aM;
61 theIMin = theMin;
62 }
63 void SetMinInelasticEnergy(G4double aM)
64 {
65 theIMin=aM;
66 }
67 virtual void SetMaxEnergy(G4double aM) final override
68 {
69 theIMax = aM;
70 theMax=aM;
71 }
72 void SetMaxInelasticEnergy(G4double aM)
73 {
74 theIMax = aM;
75 }
76
77 using G4VNeutronBuilder::Build; //Prevent compiler warning
78
79 private:
80
81 G4double theMin;
82 G4double theIMin;
83 G4double theMax;
84 G4double theIMax;
85
86 G4ParticleHPElastic * theHPElastic;
87 G4ParticleHPElasticData * theHPElasticData;
88 G4ParticleHPInelastic * theHPInelastic;
89 G4ParticleHPInelasticData * theHPInelasticData;
90 G4ParticleHPFission * theHPFission;
91 G4ParticleHPFissionData * theHPFissionData;
92 G4ParticleHPCapture * theHPCapture;
93 G4ParticleHPCaptureData * theHPCaptureData;
94
95};
