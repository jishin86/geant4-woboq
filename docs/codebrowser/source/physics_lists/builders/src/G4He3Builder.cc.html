40 #include "G4ParticleTable.hh"

43
44 void G4He3Builder::Build()
45 {
46 wasActivated = true;
47 std::vector<G4VHe3Builder *>::iterator i;
48 for(i=theModelCollections.begin(); i!=theModelCollections.end(); i++)
49 {
50 (*i)->Build(theHe3Inelastic);
51 }
52 G4ProcessManager * theProcMan = G4He3::He3()->GetProcessManager();
53 theProcMan->AddDiscreteProcess(theHe3Inelastic);
54 }
55
56 G4He3Builder::
57 G4He3Builder(): wasActivated(false)
58 {
59 theHe3Inelastic=new G4He3InelasticProcess;
60 }
61
62 void G4He3Builder::RegisterMe(G4PhysicsBuilderInterface* aB) {
63 auto bld = dynamic_cast<G4VHe3Builder*>(aB);
64 if ( bld != nullptr ) {
65 theModelCollections.push_back(bld);
66 } else {
67 G4PhysicsBuilderInterface::RegisterMe(aB);
68 }
69 }
