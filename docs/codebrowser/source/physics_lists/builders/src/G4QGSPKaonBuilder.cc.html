45G4QGSPKaonBuilder::
46G4QGSPKaonBuilder(G4bool quasiElastic)
47{
48 theMin = 12.0*GeV;
49 theMax = G4HadronicParameters::Instance()->GetMaxEnergy();
50 theModel = new G4TheoFSGenerator("QGSP");
51
52 theStringModel = new G4QGSModel< G4QGSParticipants >;
53 theStringDecay = new G4ExcitedStringDecay(new G4QGSMFragmentation);
54 theStringModel->SetFragmentationModel(theStringDecay);
55
56 theCascade = new G4GeneratorPrecompoundInterface;
57
58 theModel->SetHighEnergyGenerator(theStringModel);
59 if (quasiElastic)
60 {
61 theQuasiElastic=new G4QuasiElasticChannel;
62 theModel->SetQuasiElasticChannel(theQuasiElastic);
63 } else
64 { theQuasiElastic=0;}
65 theModel->SetTransport(theCascade);
66}
67
68G4QGSPKaonBuilder::
69~G4QGSPKaonBuilder()
70{
71 if ( theQuasiElastic ) delete theQuasiElastic;
72 delete theStringDecay;
73 delete theStringModel;
74}

79void G4QGSPKaonBuilder::
80Build(G4KaonPlusInelasticProcess * aP)
81{
82 theModel->SetMinEnergy(theMin);
83 theModel->SetMaxEnergy(theMax);
84 aP->RegisterMe(theModel);
85 aP->AddDataSet(G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4ChipsKaonPlusInelasticXS::Default_Name()));
86}
