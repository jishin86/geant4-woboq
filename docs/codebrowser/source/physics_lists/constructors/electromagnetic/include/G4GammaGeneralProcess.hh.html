185//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
186
187inline G4double
188G4GammaGeneralProcess::ComputeGeneralLambda(size_t idxe, size_t idxt,
189 size_t& idx, G4double e)
190{
191 idxEnergy = idxe;
192 return theHandler->GetVector(idxt, currentCoupleIndex)->Value(e, idx);
193}
194
195//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
196
197inline G4double G4GammaGeneralProcess::GetProbability(size_t idxt, size_t& idx)
198{
199 return (theT[idxt]) ? theHandler->GetVector(idxt,
200 currentCoupleIndex)->Value(preStepKinEnergy, idx) : 1.0;
201}
202
203//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
204
205inline void
206G4GammaGeneralProcess::SelectedProcess(const G4Track& track, G4VProcess* ptr)
207{
208 selectedProc = ptr;
209 const_cast<G4Track*>(&track)->SetCreatorProcess(ptr);
210}
211
212//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
213
214inline G4VParticleChange*
215G4GammaGeneralProcess::SampleSecondaries(const G4Track& track, const G4Step& step,
216 G4VEmProcess* proc)
217{
218 proc->CurrentSetup(currentCouple,preStepKinEnergy);
219 SelectedProcess(track, proc);
220 return proc->PostStepDoIt(track, step);
221}
