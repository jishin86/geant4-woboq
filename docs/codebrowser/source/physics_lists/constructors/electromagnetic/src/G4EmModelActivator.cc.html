121G4EmModelActivator::G4EmModelActivator(const G4String& emphys)
122 : baseName(emphys)
123{
124 theParameters = G4EmParameters::Instance();
125
126 const std::vector<G4String>& regnamesPAI = theParameters->RegionsPAI();
127 if(regnamesPAI.size() > 0)
128 {
129 ActivatePAI();
130 }
131 const std::vector<G4String>& regnamesME = theParameters->RegionsMicroElec();
132 if(regnamesME.size() > 0)
133 {
134 ActivateMicroElec();
135 }
136 const std::vector<G4String>& regnamesMSC = theParameters->RegionsPhysics();
137 if(regnamesMSC.size() > 0)
138 {
139 ActivateEmOptions();
140 }
141}

145void G4EmModelActivator::ActivateEmOptions()
146{
147 const std::vector<G4String>& regnamesPhys = theParameters->RegionsPhysics();
148 G4int nreg = regnamesPhys.size();
149 if(0 == nreg) { return; }
150 G4int verbose = theParameters->Verbose() - 1;
151 if(verbose > 0) {
152 G4cout << "### G4EmModelActivator::ActivateEmOptions for " << nreg << " regions"
153 << G4endl;
154 }
155 const std::vector<G4String>& typesPhys = theParameters->TypesPhysics();
156
157 // start configuration of models
158 const G4ParticleDefinition* elec = G4Electron::Electron();
159 const G4ParticleDefinition* posi = G4Positron::Positron();
160 const G4ParticleDefinition* phot = G4Gamma::Gamma();
161 const G4ParticleDefinition* prot = G4Proton::Proton();
162 G4LossTableManager* man = G4LossTableManager::Instance();
163 G4EmConfigurator* em_config = man->EmConfigurator();
164 G4VAtomDeexcitation* adeexc = man->AtomDeexcitation();
165 G4ParticleTable* table = G4ParticleTable::GetParticleTable();
166 G4VEmModel* mod;
167
168 // high energy limit for low-energy e+- model of msc
169 G4double mscEnergyLimit = G4EmParameters::Instance()->MscEnergyLimit();
170
171 // high energy limit for Livermore and Penelope models
172 G4double highEnergyLimit = 1*GeV;
173
174 // general high energy limit
175 G4double highEnergy = 100*TeV;
176
177 for(G4int i=0; i<nreg; ++i) {
178 G4String reg = regnamesPhys[i];
179 if(verbose > 0) {
180 G4cout << i << ". region <" << reg << ">; type <" << typesPhys[i] << "> "
181 << G4endl;
182 }
183
184 if(baseName == typesPhys[i]) { continue; }
185
186 if("G4EmStandard" == typesPhys[i]) {
187 G4UrbanMscModel* msc = new G4UrbanMscModel();
188 AddStandardScattering(elec, em_config, msc, reg, mscEnergyLimit, highEnergy);
189
190 msc = new G4UrbanMscModel();
191 AddStandardScattering(posi, em_config, msc, reg, mscEnergyLimit, highEnergy);
192
193 } else if("G4EmStandard_opt1" == typesPhys[i] ||
194 "G4EmStandard_opt2" == typesPhys[i]) {
195 G4UrbanMscModel* msc = new G4UrbanMscModel();
196 msc->SetStepLimitType(fMinimal);
197 msc->SetRangeFactor(0.2);
198 AddStandardScattering(elec, em_config, msc, reg, mscEnergyLimit, highEnergy);
199
200 msc = new G4UrbanMscModel();
201 msc->SetStepLimitType(fMinimal);
202 msc->SetRangeFactor(0.2);
203 AddStandardScattering(posi, em_config, msc, reg, mscEnergyLimit, highEnergy);
204
205 } else if("G4EmStandard_opt3" == typesPhys[i]) {
206
207 G4DummyModel* dummy = new G4DummyModel();
208 G4UrbanMscModel* msc = new G4UrbanMscModel();
209 msc->SetStepLimitType(fUseDistanceToBoundary);
210 msc->SetLocked(true);
211 em_config->SetExtraEmModel("e-", "msc", msc, reg);
212 FindOrAddProcess(elec, "CoulombScat");
213 em_config->SetExtraEmModel("e-", "CoulombScat", dummy, reg);
214
215 msc = new G4UrbanMscModel();
216 msc->SetStepLimitType(fUseDistanceToBoundary);
217 msc->SetLocked(true);
218 em_config->SetExtraEmModel("e+", "msc", msc, reg);
219 FindOrAddProcess(posi, "CoulombScat");
220 em_config->SetExtraEmModel("e+", "CoulombScat", dummy, reg);
221
222 msc = new G4UrbanMscModel();
223 msc->SetStepLimitType(fMinimal);
224 msc->SetRangeFactor(0.2);
225 msc->SetLocked(true);
226 em_config->SetExtraEmModel("proton", "msc", msc, reg);
227 FindOrAddProcess(prot, "CoulombScat");
228 em_config->SetExtraEmModel("proton", "CoulombScat", dummy, reg);
229
230 theParameters->SetNumberOfBinsPerDecade(20);
231 if(G4Threading::IsMasterThread()) {
232 theParameters->SetDeexActiveRegion(reg, true, false, false);
233 }
234 theParameters->DefineRegParamForDeex(adeexc);
235 FindOrAddProcess(phot, "Rayl");
236 mod = new G4LivermoreRayleighModel();
237 em_config->SetExtraEmModel("gamma", "Rayl", mod, reg);
238 FindOrAddProcess(phot, "phot");
239 mod = new G4LivermorePhotoElectricModel();
240 em_config->SetExtraEmModel("gamma", "phot", mod, reg);
241 FindOrAddProcess(phot, "compt");
242 mod = new G4KleinNishinaModel();
243 em_config->SetExtraEmModel("gamma", "compt", mod, reg);
244
245 } else if("G4EmStandard_opt4" == typesPhys[i]) {
246 G4VMscModel* msc = GetGSModel();
247 msc->SetRangeFactor(0.08);
248 msc->SetSkin(3);
249 msc->SetStepLimitType(fUseSafetyPlus);
250 AddStandardScattering(elec, em_config, msc, reg, mscEnergyLimit, highEnergy);
251
252 msc = GetGSModel();
253 msc->SetRangeFactor(0.08);
254 msc->SetSkin(3);
255 msc->SetStepLimitType(fUseSafetyPlus);
256 AddStandardScattering(posi, em_config, msc, reg, mscEnergyLimit, highEnergy);
257
258 theParameters->SetNumberOfBinsPerDecade(20);
259 theParameters->SetUseMottCorrection(true);
260 if(G4Threading::IsMasterThread()) {
261 theParameters->SetDeexActiveRegion(reg, true, false, false);
262 }
263 theParameters->DefineRegParamForDeex(adeexc);
264
265 FindOrAddProcess(phot, "Rayl");
266 mod = new G4LivermoreRayleighModel();
267 em_config->SetExtraEmModel("gamma", "Rayl", mod, reg);
268 FindOrAddProcess(phot, "compt");
269 mod = new G4KleinNishinaModel();
270 em_config->SetExtraEmModel("gamma", "compt", mod, reg);
271 mod = new G4LowEPComptonModel();
272 mod->SetHighEnergyLimit(20*MeV);
273 em_config->SetExtraEmModel("gamma", "compt", mod, reg);
274
275 } else if("G4EmStandardGS" == typesPhys[i]) {
276 G4GoudsmitSaundersonMscModel* msc = new G4GoudsmitSaundersonMscModel();
277 msc->SetRangeFactor(0.06);
278 AddStandardScattering(elec, em_config, msc, reg, mscEnergyLimit, highEnergy);
279
280 msc = new G4GoudsmitSaundersonMscModel();
281 msc->SetRangeFactor(0.06);
282 AddStandardScattering(posi, em_config, msc, reg, mscEnergyLimit, highEnergy);
283
284 } else if("G4EmStandardWVI" == typesPhys[i]) {
285 G4WentzelVIModel* msc = new G4WentzelVIModel();
286 AddStandardScattering(elec, em_config, msc, reg, mscEnergyLimit, highEnergy);
287
288 msc = new G4WentzelVIModel();
289 AddStandardScattering(posi, em_config, msc, reg, mscEnergyLimit, highEnergy);
290
291 if(G4Threading::IsMasterThread()) {
292 theParameters->SetDeexActiveRegion(regnamesPhys[i], true, false, false);
293 }
294 theParameters->DefineRegParamForDeex(adeexc);
295
296 } else if("G4EmStandardSS" == typesPhys[i] &&
297 baseName != "G4EmStandard_opt3") {
298 G4EmParticleList emList;
299 for(const auto& particleName : emList.PartNames()) {
300 G4ParticleDefinition* particle = table->FindParticle(particleName);
301 if(particle && 0.0 != particle->GetPDGCharge()) {
302 FindOrAddProcess(particle, "CoulombScat");
303 G4eCoulombScatteringModel* sc = new G4eCoulombScatteringModel();
304 sc->SetPolarAngleLimit(0.0);
305 sc->SetLocked(true);
306 em_config->SetExtraEmModel(particleName, "CoulombScat", sc, reg);
307 if(particleName == "mu+" || particleName == "mu-") {
308 em_config->SetExtraEmModel(particleName, "muMsc",
309 new G4DummyModel(), reg);
310 } else {
311 em_config->SetExtraEmModel(particleName, "msc",
312 new G4DummyModel(), reg);
313 }
314 }
315 }
316 if(G4Threading::IsMasterThread()) {
317 theParameters->SetDeexActiveRegion(regnamesPhys[i], true, true, true);
318 }
319 theParameters->DefineRegParamForDeex(adeexc);
320
321 } else if("G4EmLivermore" == typesPhys[i]) {
322
323 G4VMscModel* msc = GetGSModel();
324 msc->SetRangeFactor(0.08);
325 msc->SetSkin(3);
326 msc->SetStepLimitType(fUseSafetyPlus);
327 AddStandardScattering(elec, em_config, msc, reg, mscEnergyLimit, highEnergy);
328
329 msc = GetGSModel();
330 msc->SetRangeFactor(0.08);
331 msc->SetSkin(3);
332 msc->SetStepLimitType(fUseSafetyPlus);
333 AddStandardScattering(posi, em_config, msc, reg, mscEnergyLimit, highEnergy);
334
335 mod = new G4LivermorePhotoElectricModel();
336 em_config->SetExtraEmModel("gamma", "phot", mod, reg);
337 mod = new G4LivermoreComptonModel();
338 em_config->SetExtraEmModel("gamma", "compt", mod, reg);
339 mod = new G4LivermoreGammaConversionModel();
340 em_config->SetExtraEmModel("gamma", "conv", mod, reg);
341
342 FindOrAddProcess(phot, "Rayl");
343 mod = new G4LivermoreRayleighModel();
344 em_config->SetExtraEmModel("gamma", "Rayl", mod, reg);
345
346 mod = new G4LivermoreIonisationModel();
347 G4UniversalFluctuation* uf = new G4UniversalFluctuation();
348 em_config->SetExtraEmModel("e-", "eIoni", mod, reg, 0.0, 0.1*MeV, uf);
349 mod = new G4LivermoreBremsstrahlungModel();
350 em_config->SetExtraEmModel("e-", "eBrem", mod, reg, 0.0, highEnergyLimit);
351
352 theParameters->SetNumberOfBinsPerDecade(20);
353 theParameters->SetUseMottCorrection(true);
354 if(G4Threading::IsMasterThread()) {
355 theParameters->SetDeexActiveRegion(regnamesPhys[i], true, false, false);
356 }
357 theParameters->DefineRegParamForDeex(adeexc);
358
359 } else if("G4EmPenelope" == typesPhys[i]) {
360
361 G4VMscModel* msc = GetGSModel();
362 msc->SetRangeFactor(0.08);
363 msc->SetSkin(3);
364 msc->SetStepLimitType(fUseSafetyPlus);
365 AddStandardScattering(elec, em_config, msc, reg, mscEnergyLimit, highEnergy);
366
367 msc = GetGSModel();
368 msc->SetRangeFactor(0.08);
369 msc->SetSkin(3);
370 msc->SetStepLimitType(fUseSafetyPlus);
371 AddStandardScattering(posi, em_config, msc, reg, mscEnergyLimit, highEnergy);
372
373 mod = new G4PenelopePhotoElectricModel();
374 em_config->SetExtraEmModel("gamma", "phot", mod, reg);
375 mod = new G4PenelopeComptonModel();
376 em_config->SetExtraEmModel("gamma", "compt", mod, reg);
377 mod = new G4PenelopeGammaConversionModel();
378 em_config->SetExtraEmModel("gamma", "conv", mod, reg);
379
380 FindOrAddProcess(phot, "Rayl");
381 mod = new G4PenelopeRayleighModel();
382 em_config->SetExtraEmModel("gamma", "Rayl", mod, reg);
383
384 mod = new G4PenelopeIonisationModel();
385 G4UniversalFluctuation* uf = new G4UniversalFluctuation();
386 em_config->SetExtraEmModel("e-", "eIoni", mod, reg, 0.0, highEnergyLimit, uf);
387 mod = new G4PenelopeBremsstrahlungModel();
388 em_config->SetExtraEmModel("e-", "eBrem", mod, reg, 0.0, highEnergyLimit);
389
390 mod = new G4PenelopeIonisationModel();
391 uf = new G4UniversalFluctuation();
392 em_config->SetExtraEmModel("e+", "eIoni", mod, reg, 0.0, highEnergyLimit, uf);
393 mod = new G4PenelopeBremsstrahlungModel();
394 em_config->SetExtraEmModel("e+", "eBrem", mod, reg, 0.0, highEnergyLimit);
395 mod = new G4PenelopeAnnihilationModel();
396 em_config->SetExtraEmModel("e+", "annihil", mod, reg, 0.0, highEnergyLimit);
397
398 theParameters->SetNumberOfBinsPerDecade(20);
399 theParameters->SetUseMottCorrection(true);
400 if(G4Threading::IsMasterThread()) {
401 theParameters->SetDeexActiveRegion(regnamesPhys[i], true, false, false);
402 }
403 theParameters->DefineRegParamForDeex(adeexc);
404
405 } else if("G4RadioactiveDecay" == typesPhys[i]) {
406
407 if(G4Threading::IsMasterThread()) {
408 theParameters->SetAugerCascade(true);
409 theParameters->SetDeexcitationIgnoreCut(true);
410 }
411 theParameters->DefineRegParamForDeex(adeexc);
412
413 } else {
414 if(verbose > 0 && G4Threading::IsMasterThread()) {
415 G4cout << "### G4EmModelActivator::ActivateEmOptions WARNING: \n"
416 << " EM Physics configuration name <" << typesPhys[i]
417 << "> is not known - ignored" << G4endl;
418 }
419 }
420 }
421}

425void G4EmModelActivator::ActivatePAI()
426{
427 const std::vector<G4String> regnamesPAI = theParameters->RegionsPAI();
428 G4int nreg = regnamesPAI.size();
429 if(0 == nreg) { return; }
430 G4int verbose = theParameters->Verbose() - 1;
431 if(verbose > 0) {
432 G4cout << "### G4EmModelActivator::ActivatePAI for " << nreg << " regions"
433 << G4endl;
434 }
435 const std::vector<G4String> particlesPAI = theParameters->ParticlesPAI();
436 const std::vector<G4String> typesPAI = theParameters->TypesPAI();
437
438 const std::vector<G4VEnergyLossProcess*>& v = G4LossTableManager::Instance()
439 ->GetEnergyLossProcessVector();
440
441 G4RegionStore* regionStore = G4RegionStore::GetInstance();
442
443 const G4ParticleDefinition* elec = G4Electron::Electron();
444 const G4ParticleDefinition* posi = G4Positron::Positron();
445 const G4ParticleDefinition* mupl = G4MuonPlus::MuonPlus();
446 const G4ParticleDefinition* mumi = G4MuonMinus::MuonMinus();
447 const G4ParticleDefinition* gion = G4GenericIon::GenericIon();
448
449 for(G4int i = 0; i < nreg; ++i) {
450 const G4ParticleDefinition* p = nullptr;
451 if(particlesPAI[i] != "all") {
452 p = G4ParticleTable::GetParticleTable()->FindParticle(particlesPAI[i]);
453 if(!p) {
454 G4cout << "### WARNING: ActivatePAI::FindParticle fails to find "
455 << particlesPAI[i] << G4endl;
456 continue;
457 }
458 }
459 const G4Region* r = regionStore->GetRegion(regnamesPAI[i], false);
460 if(!r) {
461 G4cout << "### WARNING: ActivatePAI::GetRegion fails to find "
462 << regnamesPAI[i] << G4endl;
463 continue;
464 }
465
466 G4String name = "hIoni";
467 if(p == elec || p == posi)
468 { name = "eIoni"; }
469 else if (p == mupl || p == mumi)
470 { name = "muIoni"; }
471 else if (p == gion)
472 { name = "ionIoni"; }
473
474 for(auto proc : v) {
475
476 if(!proc->IsIonisationProcess()) { continue; }
477
478 G4String namep = proc->GetProcessName();
479 if(p) {
480 if(name != namep) { continue; }
481 } else {
482 if(namep != "hIoni" && namep != "muIoni" &&
483 namep != "eIoni" && namep != "ionIoni")
484 { continue; }
485 }
486 G4VEmModel* em = nullptr;
487 G4VEmFluctuationModel* fm = nullptr;
488 if(typesPAI[i] == "PAIphoton" || typesPAI[i] == "pai_photon") {
489 G4PAIPhotModel* mod = new G4PAIPhotModel(p,"PAIPhotModel");
490 em = mod;
491 fm = mod;
492 } else {
493 G4PAIModel* mod = new G4PAIModel(p,"PAIModel");
494 em = mod;
495 fm = mod;
496 }
497 proc->AddEmModel(0, em, fm, r);
498 if(verbose > 0) {
499 G4cout << "### G4EmModelActivator: add <" << typesPAI[i]
500 << "> model for " << particlesPAI[i]
501 << " in the " << regnamesPAI[i] << G4endl;
502 }
503 }
504 }
505}

509void G4EmModelActivator::ActivateMicroElec()
510{
511 const std::vector<G4String> regnamesME = theParameters->RegionsMicroElec();
512 G4int nreg = regnamesME.size();
513 if(0 == nreg)
514 {
515 return;
516 }
517 G4int verbose = theParameters->Verbose() - 1;
518 if(verbose > 0)
519 {
520 G4cout << "### G4EmModelActivator::ActivateMicroElec for " << nreg
521 << " regions" << G4endl;
522 }
523 G4LossTableManager* man = G4LossTableManager::Instance();
524
525 const G4ParticleDefinition* elec = G4Electron::Electron();
526 const G4ParticleDefinition* prot = G4Proton::Proton();
527 const G4ParticleDefinition* gion = G4GenericIon::GenericIon();
528 G4ProcessManager* eman = elec->GetProcessManager();
529 G4ProcessManager* pman = prot->GetProcessManager();
530 G4ProcessManager* iman = gion->GetProcessManager();
531
532 G4bool emsc = HasMsc(eman);
533
534 // MicroElec elastic is not active in the world
535 G4MicroElecElastic* eElasProc =
536 new G4MicroElecElastic("e-G4MicroElecElastic");
537 eman->AddDiscreteProcess(eElasProc);
538
539 G4MicroElecInelastic* eInelProc =
540 new G4MicroElecInelastic("e-G4MicroElecInelastic");
541 eman->AddDiscreteProcess(eInelProc);
542
543 G4MicroElecInelastic* pInelProc =
544 new G4MicroElecInelastic("p_G4MicroElecInelastic");
545 pman->AddDiscreteProcess(pInelProc);
546
547 G4MicroElecInelastic* iInelProc =
548 new G4MicroElecInelastic("ion_G4MicroElecInelastic");
549 iman->AddDiscreteProcess(iInelProc);
550
551 // start configuration of models
552 G4EmConfigurator* em_config = man->EmConfigurator();
553 G4VEmModel* mod;
554
555 // limits for MicroElec applicability
556 G4double elowest = 16.7 * eV;
557 G4double elimel = 100 * MeV;
558 G4double elimin = 9 * MeV;
559 G4double pmin = 50 * keV;
560 G4double pmax = 99.9 * MeV;
561
562 G4LowECapture* ecap = new G4LowECapture(elowest);
563 eman->AddDiscreteProcess(ecap);
564
565 for(G4int i = 0; i < nreg; ++i)
566 {
567
568 G4String reg = regnamesME[i];
569 G4cout << "### MicroElec models are activated for G4Region " << reg
570 << G4endl
571 << " Energy limits for e- elastic: " << elowest/eV << " eV - "
572 << elimel/MeV << " MeV"
573 << G4endl
574 << " Energy limits for e- inelastic: " << elowest/eV << " eV - "
575 << elimin/MeV << " MeV"
576 << G4endl
577 << " Energy limits for hadrons/ions: " << pmin/MeV << " MeV - "
578 << pmax/MeV << " MeV"
579 << G4endl;
580
581 // e-
582 if(emsc)
583 {

713void G4EmModelActivator::AddStandardScattering(const G4ParticleDefinition* part,
714 G4EmConfigurator* em_config,

717 G4double e1, G4double e2)

733 mod->SetActivationLowEnergyLimit(e1);
734 mod->SetLocked(true);
735 em_config->SetExtraEmModel(pname, "CoulombScat", mod, reg, 0.0, e2);
736}

739
740void G4EmModelActivator::FindOrAddProcess(const G4ParticleDefinition* part,
741 const G4String& name)
742{
743 G4ProcessManager* pm = part->GetProcessManager();
744 G4ProcessVector* pv = pm->GetProcessList();
745 G4int nproc = pm->GetProcessListLength();
746 for(G4int i = 0; i<nproc; ++i) {
747 if(((*pv)[i])->GetProcessName() == name) { return; }
748 }
749 if(name == "CoulombScat") {
750 G4CoulombScattering* cs = new G4CoulombScattering();
751 cs->SetEmModel(new G4DummyModel());
752 pm->AddDiscreteProcess(cs);
753 } else if(name == "Rayl") {
754 G4RayleighScattering* rs = new G4RayleighScattering();
755 rs->SetEmModel(new G4DummyModel());
756 pm->AddDiscreteProcess(rs);
757 }
758}
