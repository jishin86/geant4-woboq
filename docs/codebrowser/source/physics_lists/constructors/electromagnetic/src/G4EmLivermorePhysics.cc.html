134G4EmLivermorePhysics::G4EmLivermorePhysics(G4int ver, const G4String&)
135 : G4VPhysicsConstructor("G4EmLivermore"), verbose(ver)
136{
137 G4EmParameters* param = G4EmParameters::Instance();
138 param->SetDefaults();
139 param->SetVerbose(verbose);
140 param->SetMinEnergy(100*eV);
141 param->SetLowestElectronEnergy(100*eV);
142 param->SetNumberOfBinsPerDecade(20);
143 param->ActivateAngularGeneratorForIonisation(true);
144 param->SetStepFunction(0.2, 10*um);
145 param->SetStepFunctionMuHad(0.2, 50*um);
146 param->SetUseMottCorrection(true);
147 param->SetMscStepLimitType(fUseSafetyPlus);
148 param->SetMscSkin(3);
149 param->SetMscRangeFactor(0.08);
150 param->SetMuHadLateralDisplacement(true);
151 //param->SetUseICRU90Data(true);
152 param->SetFluo(true);
153 // param->SetAugerCascade(true);
154 SetPhysicsType(bElectromagnetic);
155}

164void G4EmLivermorePhysics::ConstructParticle()
165{
166 // gamma
167 G4Gamma::Gamma();
168
169 // leptons
170 G4Electron::Electron();
171 G4Positron::Positron();
172 G4MuonPlus::MuonPlus();
173 G4MuonMinus::MuonMinus();
174
175 // mesons
176 G4PionPlus::PionPlusDefinition();
177 G4PionMinus::PionMinusDefinition();
178 G4KaonPlus::KaonPlusDefinition();
179 G4KaonMinus::KaonMinusDefinition();
180
181 // baryons
182 G4Proton::Proton();
183 G4AntiProton::AntiProton();
184
185 // ions
186 G4Deuteron::Deuteron();
187 G4Triton::Triton();
188 G4He3::He3();
189 G4Alpha::Alpha();
190 G4GenericIon::GenericIonDefinition();
191}

195void G4EmLivermorePhysics::ConstructProcess()
196{
197 if(verbose > 1) {
198 G4cout << "### " << GetPhysicsName() << " Construct Processes " << G4endl;
199 }
200 G4PhysicsListHelper* ph = G4PhysicsListHelper::GetPhysicsListHelper();
201 G4LossTableManager* man = G4LossTableManager::Instance();
202
203 // muon & hadron bremsstrahlung and pair production
204 G4MuBremsstrahlung* mub = new G4MuBremsstrahlung();
205 G4MuPairProduction* mup = new G4MuPairProduction();
206 G4hBremsstrahlung* pib = new G4hBremsstrahlung();
207 G4hPairProduction* pip = new G4hPairProduction();
208 G4hBremsstrahlung* kb = new G4hBremsstrahlung();
209 G4hPairProduction* kp = new G4hPairProduction();
210 G4hBremsstrahlung* pb = new G4hBremsstrahlung();
211 G4hPairProduction* pp = new G4hPairProduction();
212 G4ePairProduction* ee = new G4ePairProduction();
213
214 // muon & hadron multiple scattering
215 G4MuMultipleScattering* mumsc = new G4MuMultipleScattering();
216 mumsc->SetEmModel(new G4WentzelVIModel());
217 G4CoulombScattering* muss = new G4CoulombScattering();
218
219 G4hMultipleScattering* pimsc = new G4hMultipleScattering();
220 pimsc->SetEmModel(new G4WentzelVIModel());
221 G4CoulombScattering* piss = new G4CoulombScattering();
222
223 G4hMultipleScattering* kmsc = new G4hMultipleScattering();
224 kmsc->SetEmModel(new G4WentzelVIModel());
225 G4CoulombScattering* kss = new G4CoulombScattering();
226
227 G4hMultipleScattering* hmsc = new G4hMultipleScattering("ionmsc");
228
229 // high energy limit for e+- scattering models
230 G4double highEnergyLimit= G4EmParameters::Instance()->MscEnergyLimit();
231 G4double livEnergyLimit = 1*GeV;
232
233 // nuclear stopping
234 G4NuclearStopping* pnuc = new G4NuclearStopping();
235
236 // Add Livermore EM Processes
237 G4ParticleTable* table = G4ParticleTable::GetParticleTable();
238 for(const auto& particleName : partList.PartNames()) {
239 G4ParticleDefinition* particle = table->FindParticle(particleName);
240 if (!particle) { continue; }
241 if (particleName == "gamma") {
242
243 // photoelectric effect - Livermore model only
244 G4PhotoElectricEffect* thePhotoElectricEffect = new G4PhotoElectricEffect();
245 thePhotoElectricEffect->SetEmModel(new G4LivermorePhotoElectricModel());
246
247 // Compton scattering - Livermore model only
248 G4ComptonScattering* theComptonScattering = new G4ComptonScattering();
249 theComptonScattering->SetEmModel(new G4KleinNishinaModel());
250 G4VEmModel* comptLiv = new G4LivermoreComptonModel();
251 comptLiv->SetHighEnergyLimit(livEnergyLimit);
252 theComptonScattering->AddEmModel(0, comptLiv);
253
254 // gamma conversion
255 G4GammaConversion* theGammaConversion = new G4GammaConversion();
256 G4VEmModel* convLiv = new G4BetheHeitler5DModel();
257 //G4VEmModel* convLiv = new G4LivermoreGammaConversionModel();
258 convLiv->SetHighEnergyLimit(livEnergyLimit);
259 theGammaConversion->SetEmModel(convLiv);
260
261 // default Rayleigh scattering is Livermore
262 G4RayleighScattering* theRayleigh = new G4RayleighScattering();
263
264 ph->RegisterProcess(thePhotoElectricEffect, particle);
265 ph->RegisterProcess(theComptonScattering, particle);
266 ph->RegisterProcess(theGammaConversion, particle);
267 ph->RegisterProcess(theRayleigh, particle);
268
269 } else if (particleName == "e-") {
270
271 // multiple and single scattering
272 G4eMultipleScattering* msc = new G4eMultipleScattering;
273 G4GoudsmitSaundersonMscModel* msc1 = new G4GoudsmitSaundersonMscModel();
274 G4WentzelVIModel* msc2 = new G4WentzelVIModel();
275 msc1->SetHighEnergyLimit(highEnergyLimit);
276 msc2->SetLowEnergyLimit(highEnergyLimit);
277 msc->SetEmModel(msc1);
278 msc->SetEmModel(msc2);
279
280 G4eCoulombScatteringModel* ssm = new G4eCoulombScatteringModel();
281 G4CoulombScattering* ss = new G4CoulombScattering();
282 ss->SetEmModel(ssm);
283 ss->SetMinKinEnergy(highEnergyLimit);
284 ssm->SetLowEnergyLimit(highEnergyLimit);
285 ssm->SetActivationLowEnergyLimit(highEnergyLimit);
286
287 // Ionisation - Livermore should be used only for low energies
288 G4eIonisation* eioni = new G4eIonisation();
289 G4VEmModel* theIoniLiv = new G4LivermoreIonisationModel();
290 theIoniLiv->SetHighEnergyLimit(0.1*MeV);
291 eioni->AddEmModel(0, theIoniLiv, new G4UniversalFluctuation() );
292
293 // Bremsstrahlung
294 G4eBremsstrahlung* brem = new G4eBremsstrahlung();
295 G4SeltzerBergerModel* br1 = new G4SeltzerBergerModel();
296 G4eBremsstrahlungRelModel* br2 = new G4eBremsstrahlungRelModel();
297 br1->SetAngularDistribution(new G4Generator2BS());
298 br2->SetAngularDistribution(new G4Generator2BS());
299 brem->SetEmModel(br1);
300 brem->SetEmModel(br2);
301 br1->SetHighEnergyLimit(GeV);
302
303 // register processes
304 ph->RegisterProcess(msc, particle);
305 ph->RegisterProcess(eioni, particle);
306 ph->RegisterProcess(brem, particle);
307 ph->RegisterProcess(ee, particle);
308 ph->RegisterProcess(ss, particle);
309
310 } else if (particleName == "e+") {
311
312 // multiple and single scattering
313 G4eMultipleScattering* msc = new G4eMultipleScattering;
314 G4GoudsmitSaundersonMscModel* msc1 = new G4GoudsmitSaundersonMscModel();
315 G4WentzelVIModel* msc2 = new G4WentzelVIModel();
316 msc1->SetHighEnergyLimit(highEnergyLimit);
317 msc2->SetLowEnergyLimit(highEnergyLimit);
318 msc->SetEmModel(msc1);
319 msc->SetEmModel(msc2);
320
321 G4eCoulombScatteringModel* ssm = new G4eCoulombScatteringModel();
322 G4CoulombScattering* ss = new G4CoulombScattering();
323 ss->SetEmModel(ssm);
324 ss->SetMinKinEnergy(highEnergyLimit);
325 ssm->SetLowEnergyLimit(highEnergyLimit);
326 ssm->SetActivationLowEnergyLimit(highEnergyLimit);
327
328 // ionisation
329 G4eIonisation* eIoni = new G4eIonisation();
330
331 // Bremsstrahlung from standard
332 G4eBremsstrahlung* brem = new G4eBremsstrahlung();
333 G4SeltzerBergerModel* br1 = new G4SeltzerBergerModel();
334 G4eBremsstrahlungRelModel* br2 = new G4eBremsstrahlungRelModel();
335 br1->SetAngularDistribution(new G4Generator2BS());
336 br2->SetAngularDistribution(new G4Generator2BS());
337 brem->SetEmModel(br1);
338 brem->SetEmModel(br2);
339 br1->SetHighEnergyLimit(GeV);
340
341 // register processes
342 ph->RegisterProcess(msc, particle);
343 ph->RegisterProcess(eIoni, particle);
344 ph->RegisterProcess(brem, particle);
345 ph->RegisterProcess(ee, particle);
346 ph->RegisterProcess(new G4eplusAnnihilation(), particle);
347 ph->RegisterProcess(ss, particle);
348
349 } else if (particleName == "mu+" ||
350 particleName == "mu-" ) {
351
352 G4MuIonisation* muIoni = new G4MuIonisation();
353
354 ph->RegisterProcess(mumsc, particle);
355 ph->RegisterProcess(muIoni, particle);
356 ph->RegisterProcess(mub, particle);
357 ph->RegisterProcess(mup, particle);
358 ph->RegisterProcess(muss, particle);
359
360 } else if (particleName == "alpha" ||
361 particleName == "He3" ) {
362
363 G4hMultipleScattering* msc = new G4hMultipleScattering();
364 G4ionIonisation* ionIoni = new G4ionIonisation();
365
366 ph->RegisterProcess(msc, particle);
367 ph->RegisterProcess(ionIoni, particle);
368 ph->RegisterProcess(pnuc, particle);
369
370 } else if (particleName == "GenericIon") {
371
372 G4ionIonisation* ionIoni = new G4ionIonisation();
373 ionIoni->SetEmModel(new G4IonParametrisedLossModel());
374 ionIoni->SetStepFunction(0.1, 1*um);
375
376 ph->RegisterProcess(hmsc, particle);
377 ph->RegisterProcess(ionIoni, particle);
378 ph->RegisterProcess(pnuc, particle);
379
380 } else if (particleName == "pi+" ||
381 particleName == "pi-" ) {
382
383 G4hIonisation* hIoni = new G4hIonisation();
384
385 ph->RegisterProcess(pimsc, particle);
386 ph->RegisterProcess(hIoni, particle);
387 ph->RegisterProcess(pib, particle);
388 ph->RegisterProcess(pip, particle);
389 ph->RegisterProcess(piss, particle);
390
391 } else if (particleName == "kaon+" ||
392 particleName == "kaon-" ) {
393
394 G4hIonisation* hIoni = new G4hIonisation();
395
396 ph->RegisterProcess(kmsc, particle);
397 ph->RegisterProcess(hIoni, particle);
398 ph->RegisterProcess(kb, particle);
399 ph->RegisterProcess(kp, particle);
400 ph->RegisterProcess(kss, particle);
401
402 } else if (particleName == "proton" ||
403 particleName == "anti_proton") {
404
405 G4hMultipleScattering* pmsc = new G4hMultipleScattering();
406 pmsc->SetEmModel(new G4WentzelVIModel());
407 G4hIonisation* hIoni = new G4hIonisation();
408
409 ph->RegisterProcess(pmsc, particle);
410 ph->RegisterProcess(hIoni, particle);
411 ph->RegisterProcess(pb, particle);
412 ph->RegisterProcess(pp, particle);
413 ph->RegisterProcess(new G4CoulombScattering(), particle);
414
415 } else if (particleName == "B+" ||
416 particleName == "B-" ||
417 particleName == "D+" ||
418 particleName == "D-" ||
419 particleName == "Ds+" ||
420 particleName == "Ds-" ||
421 particleName == "anti_He3" ||
422 particleName == "anti_alpha" ||
423 particleName == "anti_deuteron" ||
424 particleName == "anti_lambda_c+" ||
425 particleName == "anti_omega-" ||
426 particleName == "anti_sigma_c+" ||
427 particleName == "anti_sigma_c++" ||
428 particleName == "anti_sigma+" ||
429 particleName == "anti_sigma-" ||
430 particleName == "anti_triton" ||
431 particleName == "anti_xi_c+" ||
432 particleName == "anti_xi-" ||
433 particleName == "deuteron" ||
434 particleName == "lambda_c+" ||
435 particleName == "omega-" ||
436 particleName == "sigma_c+" ||
437 particleName == "sigma_c++" ||
438 particleName == "sigma+" ||
439 particleName == "sigma-" ||
440 particleName == "tau+" ||
441 particleName == "tau-" ||
442 particleName == "triton" ||
443 particleName == "xi_c+" ||
444 particleName == "xi-" ) {
445
446 ph->RegisterProcess(hmsc, particle);
447 ph->RegisterProcess(new G4hIonisation(), particle);
448 ph->RegisterProcess(pnuc, particle);
449 }
450 }
451
452 // Nuclear stopping
453 pnuc->SetMaxKinEnergy(MeV);
454
455 // Deexcitation
456 //
457 G4VAtomDeexcitation* de = new G4UAtomicDeexcitation();
458 man->SetAtomDeexcitation(de);
459
460 G4EmModelActivator mact(GetPhysicsName());
461}
