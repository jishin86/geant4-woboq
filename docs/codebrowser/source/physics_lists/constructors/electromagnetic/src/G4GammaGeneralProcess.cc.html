83G4GammaGeneralProcess::G4GammaGeneralProcess():
84 G4VEmProcess("GammaGeneralProc", fElectromagnetic),
85 minPEEnergy(50*CLHEP::keV),
86 minEEEnergy(2*CLHEP::electron_mass_c2),
87 minMMEnergy(100*CLHEP::MeV),
88 peLambda(0.0),
89 nLowE(100),
90 nHighE(100),
91 splineFlag(false)
92{
93 thePhotoElectric = theCompton = theConversionEE = theRayleigh = nullptr;
94 theGammaNuclear = nullptr;
95 theConversionMM = nullptr;
96 selectedProc = nullptr;
97
98 idxEnergy = idx0 = idx1 = idx2 = idx3 = 0;
99
100 SetVerboseLevel(1);
101 SetParticle(theGamma);
102 SetProcessSubType(fGammaGeneralProcess);
103}

126void G4GammaGeneralProcess::AddEmProcess(G4VEmProcess* ptr)
127{
128 if(!ptr) { return; }
129 G4int stype = ptr->GetProcessSubType();
130 if(stype == fRayleigh) { theRayleigh = ptr; }
131 else if(stype == fPhotoElectricEffect) { thePhotoElectric = ptr; }

133 else if(stype == fGammaConversion) { theConversionEE = ptr; }

152void G4GammaGeneralProcess::PreparePhysicsTable(const G4ParticleDefinition& part)
153{
154 if(1 < verboseLevel) {
155 G4cout << "G4GammaGeneralProcess::PreparePhysicsTable() for "
156 << GetProcessName()
157 << " and particle " << part.GetParticleName()
158 << G4endl;
159 }
160 SetParticle(&part);
161 currentCouple = nullptr;
162 currentMaterial = nullptr;
163 preStepLambda = 0.0;
164 idxEnergy = idx0 = idx1 = idx2 = idx3 = 0;
165
166 isTheMaster = lManager->IsMaster();
167 if(isTheMaster) { SetVerboseLevel(theParameters->Verbose()); }
168 else { SetVerboseLevel(theParameters->WorkerVerbose()); }
169
170 if(thePhotoElectric) { thePhotoElectric->PreparePhysicsTable(part); }
171 if(theCompton) { theCompton->PreparePhysicsTable(part); }
172 if(theConversionEE) { theConversionEE->PreparePhysicsTable(part); }
173 if(theRayleigh) { theRayleigh->PreparePhysicsTable(part); }
174 if(theGammaNuclear) { theGammaNuclear->PreparePhysicsTable(part); }
175 if(theConversionMM) { theConversionMM->PreparePhysicsTable(part); }
176
177 InitialiseProcess(&part);
178}

182void G4GammaGeneralProcess::InitialiseProcess(const G4ParticleDefinition*)
183{
184 if(isTheMaster) {
185
186 // tables are created and its size is defined only once
187 if(!theHandler) {
188 theHandler = new G4EmDataHandler(nTables);
189 if(theRayleigh) { theT[1] = theT[4] = true; }
190 if(theGammaNuclear) { theT[4] = theT[5] = theT[9] = theT[13] = true; }
191 if(theConversionMM) { theT[14] = true; }
192
193 theHandler->SetMasterProcess(thePhotoElectric);
194 theHandler->SetMasterProcess(theCompton);
195 theHandler->SetMasterProcess(theConversionEE);
196 theHandler->SetMasterProcess(theRayleigh);
197
198 const G4ProductionCutsTable* theCoupleTable=
199 G4ProductionCutsTable::GetProductionCutsTable();
200 size_t numOfCouples = theCoupleTable->GetTableSize();
201
202 G4double mine = theParameters->MinKinEnergy();
203 G4double maxe = theParameters->MaxKinEnergy();
204 G4int nd = theParameters->NumberOfBinsPerDecade();
205 size_t nbin1 =
206 std::max(5, nd*G4lrint(std::log10(minPEEnergy/mine)));
207 size_t nbin2 =
208 std::max(5, nd*G4lrint(std::log10(maxe/minMMEnergy)));
209
210 G4PhysicsVector* vec = nullptr;
211 G4PhysicsLogVector aVector(mine,minPEEnergy,nbin1);
212 G4PhysicsLinearVector bVector(minPEEnergy,minEEEnergy,nLowE);
213 G4PhysicsLinearVector cVector(minEEEnergy,minMMEnergy,nHighE);
214 G4PhysicsLogVector dVector(minMMEnergy,maxe,nbin2);
215 if(splineFlag) {
216 aVector.SetSpline(splineFlag);
217 bVector.SetSpline(splineFlag);
218 cVector.SetSpline(splineFlag);
219 dVector.SetSpline(splineFlag);
220 }
221
222 for(size_t i=0; i<nTables; ++i) {
223 if(theT[i]) {
224 G4PhysicsTable* table = theHandler->MakeTable(i);
225 for(size_t j=0; j<numOfCouples; ++j) {
226 if(i<=1) {
227 vec = new G4PhysicsVector(aVector);
228 } else if(i<=5) {
229 vec = new G4PhysicsVector(bVector);
230 } else if(i<=9) {
231 vec = new G4PhysicsVector(cVector);
232 } else {
233 vec = new G4PhysicsVector(dVector);
234 }
235 G4PhysicsTableHelper::SetPhysicsVector(table, j, vec);
236 }
237 }
238 }
239 }
240 }
241}

245void G4GammaGeneralProcess::BuildPhysicsTable(const G4ParticleDefinition& part)
246{
247 if(1 < verboseLevel) {
248 G4cout << "### G4VEmProcess::BuildPhysicsTable() for "
249 << GetProcessName()
250 << " and particle " << part.GetParticleName()
251 << G4endl;
252 }
253 if(thePhotoElectric) {
254 if(!isTheMaster) {
255 thePhotoElectric->SetEmMasterProcess(theHandler->GetMasterProcess(0));
256 }
257 thePhotoElectric->BuildPhysicsTable(part);
258 }
259 if(theCompton) {
260 if(!isTheMaster) {
261 theCompton->SetEmMasterProcess(theHandler->GetMasterProcess(1));
262 }
263 theCompton->BuildPhysicsTable(part);
264 }
265 if(theConversionEE) {
266 if(!isTheMaster) {
267 theConversionEE->SetEmMasterProcess(theHandler->GetMasterProcess(2));
268 }
269 theConversionEE->BuildPhysicsTable(part);
270 }
271 if(theRayleigh) {
272 if(!isTheMaster) {
273 theRayleigh->SetEmMasterProcess(theHandler->GetMasterProcess(3));
274 }
275 theRayleigh->BuildPhysicsTable(part);
276 }
277 if(theGammaNuclear) { theGammaNuclear->BuildPhysicsTable(part); }
278 if(theConversionMM) { theConversionMM->BuildPhysicsTable(part); }
279
280 if(isTheMaster) {
281 const G4ProductionCutsTable* theCoupleTable=
282 G4ProductionCutsTable::GetProductionCutsTable();
283 size_t numOfCouples = theCoupleTable->GetTableSize();
284
285 G4LossTableBuilder* bld = lManager->GetTableBuilder();
286 const std::vector<G4PhysicsTable*>& tables = theHandler->GetTables();
287
288 G4CrossSectionDataStore* gn = (theGammaNuclear)
289 ? theGammaNuclear->GetCrossSectionDataStore() : nullptr;
290 G4DynamicParticle* dynParticle =
291 new G4DynamicParticle(G4Gamma::Gamma(),G4ThreeVector(1,0,0),1.0);
292
293 G4double sigComp(0.), sigPE(0.), sigConv(0.), sigR(0.),
294 sigN(0.), sigM(0.), val(0.);
295
296 for(size_t i=0; i<numOfCouples; ++i) {
297
298 if (bld->GetFlag(i)) {
299
300 const G4MaterialCutsCouple* couple =
301 theCoupleTable->GetMaterialCutsCouple(i);
302 const G4Material* material = couple->GetMaterial();
303
304 // energy interval 0
305 size_t nn = (*(tables[0]))[i]->GetVectorLength();
306 if(1 < verboseLevel) {
307 G4cout << "======= Zone 0 ======= N= " << nn << G4endl;
308 }
309 for(size_t j=0; j<nn; ++j) {
310 G4double e = (*(tables[0]))[i]->Energy(j);
311 sigComp = (theCompton) ? theCompton->GetLambda(e, couple) : 0.0;
312 sigR = (theRayleigh) ? theRayleigh->GetLambda(e, couple) : 0.0;
313 G4double sum = sigComp + sigR;
314 if(1 < verboseLevel) {
315 G4cout << j << ". E= " << e << " xs= " << sum
316 << " compt= " << sigComp << " Rayl= " << sigR << G4endl;
317 }
318 (*(tables[0]))[i]->PutValue(j, sum);
319 if(theT[1]) {
320 val = (sum > 0.0) ? sigComp/sum : 0.0;
321 (*(tables[1]))[i]->PutValue(j, val);
322 }
323 }
324
325 // energy interval 1
326 nn = (*(tables[2]))[i]->GetVectorLength();
327 if(1 < verboseLevel) {
328 G4cout << "======= Zone 1 ======= N= " << nn << G4endl;
329 }
330 for(size_t j=0; j<nn; ++j) {
331 G4double e = (*(tables[2]))[i]->Energy(j);
332 sigComp = (theCompton) ? theCompton->GetLambda(e, couple) : 0.0;
333 sigR = (theRayleigh) ? theRayleigh->GetLambda(e, couple) : 0.0;
334 sigPE = (thePhotoElectric)
335 ? thePhotoElectric->GetLambda(e, couple) : 0.0;
336 sigN = 0.0;
337 if(gn) {
338 dynParticle->SetKineticEnergy(e);
339 sigN = gn->ComputeCrossSection(dynParticle, material);
340 }
341 G4double sum = sigComp + sigR + sigPE + sigN;
342 if(1 < verboseLevel) {
343 G4cout << j << ". E= " << e << " xs= " << sum
344 << " compt= " << sigComp << " conv= " << sigConv
345 << " PE= " << sigPE << " Rayl= " << sigR
346 << " GN= " << sigN << G4endl;
347 }
348 (*(tables[2]))[i]->PutValue(j, sum);
349 val = (sum > 0.0) ? sigPE/sum : 0.0;
350 (*(tables[3]))[i]->PutValue(j, val);
351 if(theT[4]) {
352 val = (sum > 0.0) ? (sigComp + sigPE)/sum : 0.0;
353 (*(tables[4]))[i]->PutValue(j, val);
354 }
355 if(theT[5]) {
356 val = (sum > 0.0) ? (sigComp + sigPE + sigR)/sum : 0.0;
357 (*(tables[5]))[i]->PutValue(j, val);
358 }
359 }
360
361 // energy interval 2
362 nn = (*(tables[6]))[i]->GetVectorLength();
363 if(1 < verboseLevel) {
364 G4cout << "======= Zone 2 ======= N= " << nn << G4endl;
365 }
366 for(size_t j=0; j<nn; ++j) {
367 G4double e = (*(tables[6]))[i]->Energy(j);
368 sigComp = (theCompton) ? theCompton->GetLambda(e, couple) : 0.0;
369 sigConv = (theConversionEE)
370 ? theConversionEE->GetLambda(e, couple) : 0.0;
371 sigPE = (thePhotoElectric)
372 ? thePhotoElectric->GetLambda(e, couple) : 0.0;
373 sigN = 0.0;
374 if(gn) {
375 dynParticle->SetKineticEnergy(e);
376 sigN = gn->ComputeCrossSection(dynParticle, material);
377 }
378 G4double sum = sigComp + sigConv + sigPE + sigN;
379 if(1 < verboseLevel) {
380 G4cout << j << ". E= " << e << " xs= " << sum
381 << " compt= " << sigComp << " conv= " << sigConv
382 << " PE= " << sigPE
383 << " GN= " << sigN << G4endl;
384 }
385 (*(tables[6]))[i]->PutValue(j, sum);
386 val = (sum > 0.0) ? sigConv/sum : 0.0;
387 (*(tables[7]))[i]->PutValue(j, val);
388 val = (sum > 0.0) ? (sigConv + sigComp)/sum : 0.0;
389 (*(tables[8]))[i]->PutValue(j, val);
390 if(theT[9]) {
391 val = (sum > 0.0) ? (sigConv + sigComp + sigPE)/sum : 0.0;
392 (*(tables[9]))[i]->PutValue(j, val);
393 }
394 }
395
396 // energy interval 3
397 nn = (*(tables[10]))[i]->GetVectorLength();
398 if(1 < verboseLevel) {
399 G4cout << "======= Zone 3 ======= N= " << nn << G4endl;
400 }
401 for(size_t j=0; j<nn; ++j) {
402 G4double e = (*(tables[10]))[i]->Energy(j);
403 sigComp = (theCompton) ? theCompton->GetLambda(e, couple) : 0.0;
404 sigConv = (theConversionEE)
405 ? theConversionEE->GetLambda(e, couple) : 0.0;
406 sigPE = (thePhotoElectric)
407 ? thePhotoElectric->GetLambda(e, couple) : 0.0;
408 sigN = 0.0;
409 if(gn) {
410 dynParticle->SetKineticEnergy(e);
411 sigN = gn->ComputeCrossSection(dynParticle, material);
412 }
413 sigM = 0.0;
414 if(theConversionMM) {
415 val = theConversionMM->ComputeMeanFreePath(e, material);
416 sigM = (val < DBL_MAX) ? 1./val : 0.0;
417 }
418 G4double sum = sigComp + sigConv + sigPE + sigN + sigM;
419 if(1 < verboseLevel) {
420 G4cout << j << ". E= " << e << " xs= " << sum
421 << " compt= " << sigComp << " conv= " << sigConv
422 << " PE= " << sigPE
423 << " GN= " << sigN << G4endl;
424 }
425 (*(tables[10]))[i]->PutValue(j, sum);
426 val = (sum > 0.0) ? 1.0 - sigConv/sum : 1.0;
427 (*(tables[11]))[i]->PutValue(j, val);
428 val = (sum > 0.0) ? 1.0 - (sigConv + sigComp)/sum : 1.0;
429 (*(tables[12]))[i]->PutValue(j, val);
430 if(theT[13]) {
431 val = (sum > 0.0) ? 1.0 - (sigConv + sigComp + sigPE)/sum : 1.0;
432 (*(tables[13]))[i]->PutValue(j, val);
433 }
434 if(theT[14]) {
435 val = (sum > 0.0)
436 ? 1.0 - (sigConv + sigComp + sigPE + sigN)/sum : 1.0;
437 (*(tables[14]))[i]->PutValue(j, val);
438 }
439 }
440 for(size_t k=0; k<nTables; ++k) {
441 if(theT[k] && splineFlag) {
442 (*(tables[k]))[i]->FillSecondDerivatives();
443 }
444 }
445 }
446 }
447 delete dynParticle;
448 }
449
450 if(1 < verboseLevel) {
451 G4cout << "### G4VEmProcess::BuildPhysicsTable() done for "
452 << GetProcessName()
453 << " and particle " << part.GetParticleName()
454 << G4endl;
455 }
456}

467
468G4double G4GammaGeneralProcess::PostStepGetPhysicalInteractionLength(
469 const G4Track& track,
470 G4double previousStepSize,
471 G4ForceCondition* condition)
472{
473 *condition = NotForced;
474 G4double x = DBL_MAX;
475
476 G4double energy = track.GetKineticEnergy();
477 const G4MaterialCutsCouple* couple = track.GetMaterialCutsCouple();
478 const G4Material* mat = couple->GetMaterial();
479
480 // compute mean free path
481 if(mat != currentMaterial || energy != preStepKinEnergy) {
482 currentMaterial = mat;
483 preStepKinEnergy = energy;
484 preStepLambda = TotalCrossSectionPerVolume(energy, couple);
485
486 // zero cross section
487 if(preStepLambda <= 0.0) {
488 theNumberOfInteractionLengthLeft = -1.0;
489 currentInteractionLength = DBL_MAX;
490 }
491 }
492
493 // non-zero cross section
494 if(preStepLambda > 0.0) {
495
496 if (theNumberOfInteractionLengthLeft < 0.0) {
497
498 // beggining of tracking (or just after DoIt of this process)
499 theNumberOfInteractionLengthLeft = -G4Log( G4UniformRand() );
500 theInitialNumberOfInteractionLength = theNumberOfInteractionLengthLeft;
501
502 } else if(currentInteractionLength < DBL_MAX) {
503
504 theNumberOfInteractionLengthLeft -=
505 previousStepSize/currentInteractionLength;
506 theNumberOfInteractionLengthLeft =
507 std::max(theNumberOfInteractionLengthLeft, 0.0);
508 }
509
510 // new mean free path and step limit for the next step
511 currentInteractionLength = 1.0/preStepLambda;
512 x = theNumberOfInteractionLengthLeft * currentInteractionLength;
513 }
514 /*
515 G4cout << "PostStepGetPhysicalInteractionLength: e= " << energy
516 << " idxe= " << idxEnergy << " xs= " << preStepLambda
517 << " x= " << x << G4endl;
518 */
519 return x;
520}

522//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
523
524G4double G4GammaGeneralProcess::TotalCrossSectionPerVolume(G4double energy,
525 const G4MaterialCutsCouple* couple)
526{
527 currentCouple = couple;
528 currentCoupleIndex = couple->GetIndex();
529 G4double cross = 0.0;
530 if(energy < minPEEnergy) {
531 cross = ComputeGeneralLambda(0, 0, idx0, energy);
532 peLambda = (thePhotoElectric)
533 ? thePhotoElectric->GetLambda(energy, couple) : 0.0;
534 cross += peLambda;
535
536 } else if(energy < minEEEnergy) {
537 cross = ComputeGeneralLambda(1, 2, idx1, energy);
538
539 } else if(energy < minMMEnergy) {
540 cross = ComputeGeneralLambda(2, 6, idx2, energy);
541
542 } else {
543 cross = ComputeGeneralLambda(3, 10, idx3, energy);
544 }
545 /*
546 G4cout << "xs= " << cross << " idxE= " << idxEnergy
547 << " idxC= " << currentCoupleIndex
548 << " E= " << energy << G4endl;
549 */
550 return cross;
551}

553//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
554
555G4VParticleChange* G4GammaGeneralProcess::PostStepDoIt(const G4Track& track,
556 const G4Step& step)
557{
558 // In all cases clear number of interaction lengths
559 theNumberOfInteractionLengthLeft = -1.0;
560 G4double q = G4UniformRand();
561 G4double x = preStepLambda*G4UniformRand();
562 G4double p;
563 /*
564 G4cout << "PostStep: preStepLambda= " << preStepLambda << " PE= " << peLambda
565 << " q= " << q << G4endl;
566 */
567 switch (idxEnergy) {
568 case 0:
569 if(x <= peLambda) {
570 return SampleSecondaries(track, step, thePhotoElectric);
571 } else {
572 p = GetProbability(1, idx0);
573 if(x <= peLambda + (preStepLambda - peLambda)*p) {
574 return SampleSecondaries(track, step, theCompton);
575 } else if(theRayleigh) {
576 return SampleSecondaries(track, step, theRayleigh);
577 }
578 }
579 break;
580
581 case 1:
582 p = GetProbability(3, idx1);
583 if(q <= p) {
584 return SampleSecondaries(track, step, thePhotoElectric);
585 }
586 p = GetProbability(4, idx1);
587 if(q <= p) {
588 return SampleSecondaries(track, step, theCompton);
589 }
590 p = GetProbability(5, idx1);
591 if(q <= p) {
592 if(theRayleigh) {
593 return SampleSecondaries(track, step, theRayleigh);
594 }
595 } else if(theGammaNuclear) {
596 return SampleSecondaries(track, step, theGammaNuclear);
597 }
598 break;
599
600 case 2:
601 p = GetProbability(7, idx2);
602 if(q <= p) {
603 return SampleSecondaries(track, step, theConversionEE);
604 }
605 p = GetProbability(8, idx2);
606 if(q <= p) {
607 return SampleSecondaries(track, step, theCompton);
608 }
609 p = GetProbability(9, idx2);
610 if(q <= p) {
611 return SampleSecondaries(track, step, thePhotoElectric);
612 } else if(theGammaNuclear) {
613 return SampleSecondaries(track, step, theGammaNuclear);
614 }
615 break;
616
617 case 3:
618 p = 1.0 - GetProbability(11, idx3);
619 if(q <= p) {
620 return SampleSecondaries(track, step, theConversionEE);
621 }
622 p = 1.0 - GetProbability(12, idx3);
623 if(q <= p) {
624 return SampleSecondaries(track, step, theCompton);
625 }
626 p = 1.0 - GetProbability(13, idx3);
627 if(q <= p) {
628 return SampleSecondaries(track, step, thePhotoElectric);
629 }
630 p = 1.0 - GetProbability(14, idx3);
631 if(q <= p) {
632 if(theGammaNuclear) {
633 return SampleSecondaries(track, step, theGammaNuclear);
634 }
635 } else if(theConversionMM) {
636 SelectedProcess(track, theConversionMM);
637 return theConversionMM->PostStepDoIt(track, step);
638 }
639 break;
640 }
641 // no interaction
642 fParticleChange.InitializeForPostStep(track);

660 const G4String& directory,

663 G4bool yes = true;
664 if(!isTheMaster) { return yes; }
665 if(thePhotoElectric &&

670 { yes = false; }

672 !theConversionEE->StorePhysicsTable(part, directory, ascii))
673 { yes = false; }
674 if(theRayleigh &&
675 !theRayleigh->StorePhysicsTable(part, directory, ascii))
676 { yes = false; }
677
678 for(size_t i=0; i<nTables; ++i) {
679 if(theT[i]) {
680 G4String nam = (0==i || 2==i || 6==i || 10==i)
681 ? "LambdaGeneral" + nameT[i] : "ProbGeneral" + nameT[i];
682 G4String fnam = GetPhysicsTableFileName(part,directory,nam,ascii);
683 if(!theHandler->StorePhysicsTable(i, part, fnam, ascii)) { yes = false; }
684 }
685 }
686 return yes;
687}
