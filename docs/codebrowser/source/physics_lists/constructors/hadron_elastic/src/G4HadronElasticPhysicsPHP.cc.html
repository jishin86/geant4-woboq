62void G4HadronElasticPhysicsPHP::ConstructProcess()
63{
64 G4HadronElasticPhysics::ConstructProcess();
65
66 const G4ParticleDefinition* neutron = G4Neutron::Neutron();
67 G4HadronElastic* he = GetElasticModel(neutron);
68 G4HadronicProcess* hel = GetElasticProcess(neutron);
69 if(he && hel) {
70 he->SetMinEnergy(19.5*MeV);
71 hel->RegisterMe(new G4ParticleHPElastic());
72 hel->AddDataSet(new G4ParticleHPElasticData());
73 }
74
75 if(verbose > 1) {
76 G4cout << "### HadronElasticPhysicsHP is constructed "
77 << G4endl;
78 }
79}
