41
42#include "G4UnknownDecay.hh"
43
44class G4UnknownDecayPhysics : public G4VPhysicsConstructor
45{
46 public:
47 G4UnknownDecayPhysics(G4int ver = 1);
48 G4UnknownDecayPhysics(const G4String& name, G4int ver = 1);

51 public:

53 // each particle type will be instantiated
54 virtual void ConstructParticle();
55

57 // each physics process will be instantiated and

59 virtual void ConstructProcess();

61 virtual G4UnknownDecay* GetDecayProcess() { return fDecayProcess; }
62
63private:
64 static G4ThreadLocal G4UnknownDecay* fDecayProcess;
65 G4int verbose;
66 static G4ThreadLocal G4bool wasActivated;
67};
