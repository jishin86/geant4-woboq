68void G4UnknownDecayPhysics::ConstructProcess()
69{
70 if(wasActivated) { return; }
71 wasActivated = true;
72
73 G4PhysicsListHelper* ph = G4PhysicsListHelper::GetPhysicsListHelper();
74
75 // Add Decay Process
76 fDecayProcess = new G4UnknownDecay();
77 auto myParticleIterator=GetParticleIterator();
78 myParticleIterator->reset();
79 G4ParticleDefinition* particle=0;
80
81 while( (*myParticleIterator)() )
82 {
83 particle = myParticleIterator->value();
84 if( fDecayProcess->IsApplicable(*particle) )
85 {
86 if(verbose > 1) {
87 G4cout << "### Decays for " << particle->GetParticleName() << G4endl;
88 }
89 ph->RegisterProcess(fDecayProcess, particle);
90 }
91 }
92}
