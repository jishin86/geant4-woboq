91G4HadronPhysicsQGSP_BIC::G4HadronPhysicsQGSP_BIC(const G4String& name, G4bool)
92 : G4VPhysicsConstructor(name)
93{
94 QuasiElasticFTF= false; // Use built-in quasi-elastic (not add-on)
95 QuasiElasticQGS= true; // For QGS, it must use it.
96 minQGSP_proton = minQGSP_neutron = minQGSP_pik = 12.0*GeV;
97 maxFTFP_proton = maxFTFP_neutron = maxFTFP_pik = 25.0*GeV;
98 minFTFP_proton = minFTFP_neutron = 9.5*GeV;
99 minFTFP_pik = 4.*GeV;
100 maxBIC_proton = maxBIC_neutron = 9.9*GeV;
101 maxBERT_pik = 5.0*GeV;
102}

110 Others();
111}
112
113void G4HadronPhysicsQGSP_BIC::Neutron()
114{
115 auto neu = new G4NeutronBuilder;
116 AddBuilder(neu);
117 auto qgs = new G4QGSPNeutronBuilder(QuasiElasticQGS);
118 AddBuilder(qgs);
119 qgs->SetMinEnergy(minQGSP_neutron);
120 neu->RegisterMe(qgs);
121 auto ftf = new G4FTFPNeutronBuilder(QuasiElasticFTF);
122 AddBuilder(ftf);
123 ftf->SetMinEnergy(minFTFP_neutron);
124 ftf->SetMaxEnergy(maxFTFP_neutron);
125 neu->RegisterMe(ftf);
126 auto bic = new G4BinaryNeutronBuilder;
127 AddBuilder(bic);
128 bic->SetMaxEnergy(maxBIC_neutron);
129 neu->RegisterMe(bic);
130 neu->Build();
131}
132
133void G4HadronPhysicsQGSP_BIC::Proton()
134{
135 auto pro = new G4ProtonBuilder;
136 AddBuilder(pro);
137 auto qgs = new G4QGSPProtonBuilder(QuasiElasticQGS);
138 AddBuilder(qgs);
139 qgs->SetMinEnergy(minQGSP_proton);
140 pro->RegisterMe(qgs);
141 auto ftf = new G4FTFPProtonBuilder(QuasiElasticFTF);
142 AddBuilder(ftf);
143 ftf->SetMinEnergy(minFTFP_proton);
144 ftf->SetMaxEnergy(maxFTFP_proton);
145 pro->RegisterMe(ftf);
146 auto bic = new G4BinaryProtonBuilder;
147 AddBuilder(bic);
148 bic->SetMaxEnergy(maxBIC_proton);
149 pro->RegisterMe(bic);
150 pro->Build();
151}
152
153void G4HadronPhysicsQGSP_BIC::Pion()
154{
155 auto pik = new G4PiKBuilder;
156 AddBuilder(pik);
157 auto qgs = new G4QGSPPiKBuilder(QuasiElasticQGS);
158 AddBuilder(qgs);
159 qgs->SetMinEnergy(minQGSP_pik);
160 pik->RegisterMe(qgs);
161 auto ftf = new G4FTFPPiKBuilder(QuasiElasticFTF);
162 AddBuilder(ftf);
163 ftf->SetMaxEnergy(maxFTFP_pik);
164 ftf->SetMinEnergy(minFTFP_pik);
165 pik->RegisterMe(ftf);
166 auto bert = new G4BertiniPiKBuilder;
167 AddBuilder(bert);
168 bert->SetMaxEnergy(maxBERT_pik);
169 pik->RegisterMe(bert);
170 pik->Build();
171}

173void G4HadronPhysicsQGSP_BIC::Others()
174{

181 AddBuilder(ftf);
182 abar->RegisterMe(ftf);
183 abar->Build();
184}

213void G4HadronPhysicsQGSP_BIC::ExtraConfiguration()
214{
215 // --- Kaons ---
216 auto xsk = new G4ComponentGGHadronNucleusXsc();
217 G4VCrossSectionDataSet * kaonxs = new G4CrossSectionInelastic(xsk);
218 G4PhysListUtil::FindInelasticProcess(G4KaonMinus::KaonMinus())->AddDataSet(kaonxs);
219 G4PhysListUtil::FindInelasticProcess(G4KaonPlus::KaonPlus())->AddDataSet(kaonxs);
220 G4PhysListUtil::FindInelasticProcess(G4KaonZeroShort::KaonZeroShort())->AddDataSet(kaonxs);
221 G4PhysListUtil::FindInelasticProcess(G4KaonZeroLong::KaonZeroLong())->AddDataSet(kaonxs);
222
223 // --- Neutrons ---
224 const G4ParticleDefinition* neutron = G4Neutron::Neutron();
225 G4HadronicProcess* inel = G4PhysListUtil::FindInelasticProcess(neutron);
226 if(inel) { inel->AddDataSet(new G4NeutronInelasticXS()); }
227 G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
228 if (capture) {
229 capture->RegisterMe(new G4NeutronRadCapture());
230 }
231}
