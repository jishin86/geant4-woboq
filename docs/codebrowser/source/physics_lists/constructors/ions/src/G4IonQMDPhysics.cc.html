86G4IonQMDPhysics::G4IonQMDPhysics(const G4String& nname, G4int ver)
87 : G4VPhysicsConstructor(nname), verbose(ver)
88{
89 eminQMD = 100.*MeV;
90 emaxQMD = 10.*GeV;
91 overlap = 10*MeV;
92 SetPhysicsType(bIons);
93 G4DeexPrecoParameters* param = G4NuclearLevelData::GetInstance()->GetParameters();
94 param->SetDeexChannelsType(fCombined);
95 if(verbose > 1) { G4cout << "### IonPhysics: " << nname << G4endl; }
96}

103void G4IonQMDPhysics::ConstructProcess()
104{
105 G4HadronicInteraction* p =
106 G4HadronicInteractionRegistry::Instance()->FindModel("PRECO");
107 G4PreCompoundModel* thePreCompound = static_cast<G4PreCompoundModel*>(p);
108 if(!thePreCompound) { thePreCompound = new G4PreCompoundModel; }
109
110 G4BinaryLightIonReaction* theIonBC = new G4BinaryLightIonReaction(thePreCompound);
111 theIonBC->SetMaxEnergy(eminQMD + overlap);
112
113 G4double emax = G4HadronicParameters::Instance()->GetMaxEnergy();
114 G4HadronicInteraction* theFTFP = nullptr;
115 if(emax > emaxQMD) {
116 theFTFPBuilder = new G4FTFBuilder("FTFP",thePreCompound);
117 theFTFP = theFTFPBuilder->GetModel();
118 theFTFP->SetMinEnergy(emaxQMD - overlap);
119 theFTFP->SetMaxEnergy(emax);
120 }
121
122 G4QMDReaction* theQMD = new G4QMDReaction();
123 theQMD->SetMinEnergy(eminQMD);
124 theQMD->SetMaxEnergy(emaxQMD);
125
126 G4VCrossSectionDataSet* theNuclNuclData =
127 new G4CrossSectionInelastic( new G4ComponentGGNuclNuclXsc() );
128
129 AddProcess("dInelastic", G4Deuteron::Deuteron(), theIonBC, theQMD, theFTFP, theNuclNuclData);
130 AddProcess("tInelastic", G4Triton::Triton(), theIonBC, theQMD, theFTFP, theNuclNuclData);
131 AddProcess("He3Inelastic", G4He3::He3(), theIonBC, theQMD, theFTFP, theNuclNuclData);
132 AddProcess("alphaInelastic", G4Alpha::Alpha(), theIonBC, theQMD, theFTFP, theNuclNuclData);
133 AddProcess("ionInelastic", G4GenericIon::GenericIon(), theIonBC, theQMD, theFTFP, theNuclNuclData);
134}

136void G4IonQMDPhysics::AddProcess(const G4String& name,
137 G4ParticleDefinition* p,
138 G4BinaryLightIonReaction* BIC,
139 G4QMDReaction* QMD,
140 G4HadronicInteraction* FTFP,
141 G4VCrossSectionDataSet* theNuclNuclData)
142{
143 G4HadronInelasticProcess* hadi = new G4HadronInelasticProcess(name, p);
144 G4ProcessManager* pManager = p->GetProcessManager();
145 pManager->AddDiscreteProcess(hadi);
146
147 hadi->AddDataSet(theNuclNuclData);
148
149 hadi->RegisterMe(BIC);
150 hadi->RegisterMe(QMD);
151 if(FTFP) { hadi->RegisterMe(FTFP); }
152
153 if(verbose > 1) {
154 G4cout << "Register " << hadi->GetProcessName()
155 << " for " << p->GetParticleName() << G4endl
156 << " Binary Cascade for E(MeV)= 0 - "
157 << eminQMD+overlap;
158 G4cout << " QMD for E(MeV)= " << eminQMD << " - " << emaxQMD;
159 if(FTFP) {
160 G4cout << " FTFP for E(MeV)= " << emaxQMD-overlap << " - " << FTFP->GetMaxEnergy();
161 }
162 G4cout << G4endl;
163 }
164}
