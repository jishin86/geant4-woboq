31#ifndef G4FastSimulationPhysics_h
32#define G4FastSimulationPhysics_h 1
33
34#include "G4VPhysicsConstructor.hh"
35#include "globals.hh"
36
37#include <vector>
38
39//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo......
40
41class G4FastSimulationPhysics : public G4VPhysicsConstructor
42{
43public:
44
45 G4FastSimulationPhysics(const G4String& name = "FastSimP");
46 virtual ~G4FastSimulationPhysics();
47
48public:
49 // --------------------------------------
50 // -- Fast simulation activation methods:
51 // --------------------------------------
52 // -- Used to select particles for which fast simulation has to be activated:
53 // ---- Activate fast simulation for one particle with fast simulation attached to mass geometry:
54 void ActivateFastSimulation(const G4String& particleName);
55 // ---- Activate fast simulation for one particle with fast simulation attached to a parallel geometry:
56 void ActivateFastSimulation(const G4String& particleName, const G4String& parallelGeometryName);
57
58 // -- Information about particles under fast simulation:
59 void BeVerbose() { fVerbose = true; }
60
61public:
62
63 // This method is dummy for physics
64 virtual void ConstructParticle();
65
66 // This method will be invoked in the Construct() method.
67 // each physics process will be instantiated and
68 // registered to the process manager of each particle type
69 virtual void ConstructProcess();
70
71private:
72
73 // hide assignment operator
74 G4FastSimulationPhysics & operator=(const G4FastSimulationPhysics &right);
75 G4FastSimulationPhysics(const G4FastSimulationPhysics&);
76
77 // -- Particles under fast simulation:
78 std::vector< G4String > fParticlesUnderFastSimulation;
79 // -- And their related possible parallel geometries:
80 std::vector< G4String > fGeometries;
81
82 // -- Report:
83 G4bool fVerbose;
84
85
86
87};
