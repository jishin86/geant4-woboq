34class G4VBasePhysConstrFactory
35{
36
37public:
38
39 virtual G4VPhysicsConstructor* Instantiate() = 0;

41};

44template <typename T> class G4PhysicsConstructorFactory : public G4VBasePhysConstrFactory
45{
46public:
47
48 G4PhysicsConstructorFactory(const G4String& name)
49 {
50 G4PhysicsConstructorRegistry::Instance()->AddFactory(name, this);
51 }
52
53 virtual G4VPhysicsConstructor* Instantiate()
54 {
55 return new T();
56 }
57};

59#define G4_DECLARE_PHYSCONSTR_FACTORY(physics_constructor) \
60 const G4PhysicsConstructorFactory<physics_constructor>& physics_constructor##Factory = G4PhysicsConstructorFactory<physics_constructor>(#physics_constructor)
61
62// support for phys constructors defined within a namespace
63// a bit tricky due to cpp macro expansion and the use of "::"
64// use G4_DECLARE_PHYSCONSTR_FACTORY_NS( myns::MyProc, myns, MyProc );
65#define G4_DECLARE_PHYSCONSTR_FACTORY_NS( physics_constructor, nsname, pcbase ) \
66 namespace nsname { \
67 const G4PhysicsConstructorFactory<physics_constructor>& pcbase##Factory = G4PhysicsConstructorFactory<physics_constructor>(#physics_constructor); \
68 } \

72// REFERENCE (rather than DECLARE) when the physics constructor if it is part
73// of a static library. No need to include the header (DECLARE needs this
74// to build the code), we just need to make a reference in order to pull
75// the compilation unit static variable from the library and cause it
76// to be initialized (and thus self-register)
77
78// Use REGREF to allow cases where REFERENCE macro is used twice
79#ifndef REGREF
80 #define REGREF 0
81#endif
82
83#define G4_REFERENCE_PHYSCONSTR_FACTORY(physics_constructor) \
84 class physics_constructor; \
85 extern const G4PhysicsConstructorFactory<physics_constructor>& physics_constructor##Factory; \
86 const G4PhysicsConstructorFactory<physics_constructor>& physics_constructor##FactoryRef##REGREF = physics_constructor##Factory
87
88#define G4_REFERENCE_PHYSCONSTR_FACTORY_NS( physics_constructor, nsname, pcbase ) \
89 namespace nsname { \
90 class pcbase; \
91 extern const G4PhysicsConstructorFactory<physics_constructor>& pcbase##Factory; \
92 const G4PhysicsConstructorFactory<physics_constructor>& pcbase##FactoryRef##REGREF = pcbase##Factory; \
93 } \
94 typedef int xyzzy__LINE__
