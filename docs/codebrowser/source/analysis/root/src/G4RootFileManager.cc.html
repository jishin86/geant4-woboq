62G4bool G4RootFileManager::OpenNtupleFiles()
63{
64 auto finalResult = true;
65
66 for ( auto i = 0; i < fNofNtupleFiles; i++ ) {
67
68 auto name = GetNtupleFileName(i);
69
70#ifdef G4VERBOSE
71 if ( fState.GetVerboseL4() )
72 fState.GetVerboseL4()
73 ->Message("create", "main ntuple file", name);
74#endif
75
76 // create new file
77 auto rfile = std::make_shared<tools::wroot::file>(G4cout, name);
78 rfile->add_ziper('Z', tools::compress_buffer);
79 rfile->set_compression(fState.GetCompressionLevel());
80
81 if ( ! rfile->is_open() ) {
82 G4ExceptionDescription description;
83 description << " " << "Cannot open file " << name;
84 G4Exception("G4RootAnalysisManager::OpenFile()",
85 "Analysis_W001", JustWarning, description);
86 finalResult = false;
87 }
88
89 // Do not create directory if extra ntuple files
90 // auto result = CreateNtupleDirectory(rfile);
91 // finalResult = finalResult && result;
92 tools::wroot::directory* directory = &rfile->dir();
93 if ( fNtupleDirectoryName != "" ) {
94 directory = rfile->dir().mkdir(fNtupleDirectoryName);
95 if ( ! directory ) {
96 G4ExceptionDescription description;
97 description << " "
98 << "cannot create directory " << fNtupleDirectoryName;
99 G4Exception("G4RootFileManager::OpenNtupleFiles()",
100 "Analysis_W001", JustWarning, description);
101 directory = &fFile->dir();
102 }
103 }
104
105 fNtupleFiles.push_back(rfile);
106 fMainNtupleDirectories.push_back(directory);
107
108#ifdef G4VERBOSE
109 if ( fState.GetVerboseL1() )
110 fState.GetVerboseL1()
111 ->Message("create", "main ntuple file", name);
112#endif
113
114 }
115
116 return finalResult;
117}

119//_____________________________________________________________________________
120G4bool G4RootFileManager::WriteFile(std::shared_ptr<tools::wroot::file> rfile,
121#ifdef G4VERBOSE
122 const G4String& fileName)
123#else
124 const G4String& /*fileName*/)
125#endif
126{
127 // Do nothing if there is no file
128 if ( ! fIsOpenFile ) return true;
129
130#ifdef G4VERBOSE
131 if ( fState.GetVerboseL4() )
132 fState.GetVerboseL4()->Message("write", "file", fileName);
133#endif
134
135 unsigned int n;
136 auto result = rfile->write(n);
137
138#ifdef G4VERBOSE
139 if ( fState.GetVerboseL1() )
140 fState.GetVerboseL1()->Message("write", "file", fileName, result);
141#endif
142
143 return result;
144}

176G4bool G4RootFileManager::OpenFile(const G4String& fileName)
177{
178 // Keep file name
179 fFileName = fileName;
180 auto name = GetFullFileName();
181
182 // delete previous file if exists
183 //if ( fFile ) delete fFile;
184
185 // create new file
186 fFile = std::make_shared<tools::wroot::file>(G4cout, name);
187 fFile->add_ziper('Z',tools::compress_buffer);
188 fFile->set_compression(fState.GetCompressionLevel());
189
190 if ( ! fFile->is_open() ) {
191 fFile = nullptr;
192 G4ExceptionDescription description;
193 description << " " << "Cannot open file " << fileName;
194 G4Exception("G4RootAnalysisManager::OpenFile()",
195 "Analysis_W001", JustWarning, description);
196 return false;
197 }
198
199 // Create directories
200 if ( ! CreateHistoDirectory() ) return false;
201 if ( ! CreateNtupleDirectory() ) return false;
202
203 // Open ntuple files
204 OpenNtupleFiles();
205
206 fLockFileName = true;
207 fLockHistoDirectoryName = true;
208 fLockNtupleDirectoryName = true;
209
210 fIsOpenFile = true;
211
212 return true;
213}

215//_____________________________________________________________________________
216G4bool G4RootFileManager:: WriteFile()
217{
218 auto finalResult = true;
219
220 auto result = WriteFile(fFile, GetFullFileName());
221 finalResult = finalResult && result;
222
223 auto counter = 0;
224 for ( auto ntupleFile : fNtupleFiles ) {
225 result = WriteFile(ntupleFile, GetNtupleFileName(counter++));
226 finalResult = finalResult && result;
227 }
228 return finalResult;
229}

233{
234 auto finalResult = true;

242 finalResult = finalResult && result;
243 }
244
245 fLockFileName = false;
246 fIsOpenFile = false;
247
248 return finalResult;
249}

251//_____________________________________________________________________________
252G4bool G4RootFileManager::CreateHistoDirectory()
253{
254 if ( fHistoDirectoryName == "" ) {
255 // Do not create a new directory if its name is not set
256 fHistoDirectory = &(fFile->dir());
257 return true;
258 }
259
260#ifdef G4VERBOSE
261 if ( fState.GetVerboseL4() )
262 fState.GetVerboseL4()
263 ->Message("create", "directory for histograms", fHistoDirectoryName);
264#endif
265
266 fHistoDirectory = fFile->dir().mkdir(fHistoDirectoryName);
267 if ( ! fHistoDirectory ) {
268 G4ExceptionDescription description;
269 description << " "
270 << "cannot create directory " << fHistoDirectoryName;
271 G4Exception("G4RootFileManager::CreateHistoDirectory()",
272 "Analysis_W001", JustWarning, description);
273 return false;
274 }
275#ifdef G4VERBOSE
276 else {
277 if ( fState.GetVerboseL2() )
278 fState.GetVerboseL2()
279 ->Message("create", "directory for histograms", fHistoDirectoryName);
280 }
281#endif
282 return true;
283}

319//_____________________________________________________________________________
320std::shared_ptr<tools::wroot::file>
321G4RootFileManager::GetNtupleFile(G4int index) const
322{
323 if ( index==0 && ( ! fNtupleFiles.size() ) ) return fFile;
324
325 if ( index < 0 || index >= G4int(fNtupleFiles.size()) ) {
326 G4String inFunction = "G4RootFileManager::GetNtupleFile()";
327 G4ExceptionDescription description;
328 description << " " << "ntuple file " << index << " does not exist.";
329 G4Exception(inFunction, "Analysis_W011", JustWarning, description);
330 return nullptr;
331 }
332
333 return fNtupleFiles[index];
334}

346 G4Exception(inFunction, "Analysis_W011", JustWarning, description);
347 return nullptr;

350 return fMainNtupleDirectories[index];
