40G4RootNtupleManager::G4RootNtupleManager(const G4AnalysisManagerState& state,
41 G4int nofMainManagers,
42 G4bool rowWise)
43 : G4TNtupleManager<tools::wroot::ntuple>(state),
44 fCreateMode(G4NtupleCreateMode::kUndefined),
45 fFileManager(nullptr),
46 fNtupleDirectory(nullptr),
47 fMainNtupleManagers(),
48 fRowWise(rowWise)
49{
50 for ( G4int i=0; i<nofMainManagers; ++i) {
51 fMainNtupleManagers.push_back(
52 new G4RootMainNtupleManager(this, rowWise, fState));
53 }
54}

64//_____________________________________________________________________________
65void G4RootNtupleManager::SetCreateMode()
66{
67// Set create mode if not yet defined, or reset it if ntuples are defined both
68// before and after open file
69
70#ifdef G4VERBOSE
71 if ( fState.GetVerboseL4() )
72 fState.GetVerboseL4()
73 ->Message("set", "ntuple create mode", "");
74#endif
75
76 G4String createMode;
77 if ( fCreateMode == G4NtupleCreateMode::kUndefined ) {
78 if ( fMainNtupleManagers.size() ) {
79 if ( fFileManager->GetNtupleFile(0) ) {
80 fCreateMode = G4NtupleCreateMode::kMainAfterOpen;
81 createMode = "G4NtupleCreateMode::kMainAfterOpen";
82 } else {
83 fCreateMode = G4NtupleCreateMode::kMainBeforeOpen;
84 createMode = "G4NtupleCreateMode::kMainBeforeOpen";
85 }
86 }
87 else {
88 if ( fNtupleDirectory ) {
89 fCreateMode = G4NtupleCreateMode::kNoMergeAfterOpen;
90 createMode = "G4NtupleCreateMode::kNoMergeAfterOpen";
91 } else {
92 fCreateMode = G4NtupleCreateMode::kNoMergeBeforeOpen;
93 createMode = "G4NtupleCreateMode::kNoMergeBeforeOpen";
94 }
95 }
96 }
97
98 if ( fCreateMode == G4NtupleCreateMode::kNoMergeBeforeOpen &&
99 fFileManager->GetNtupleFile(0) ) {
100 fCreateMode = G4NtupleCreateMode::kNoMergeAfterOpen;
101 createMode = "G4NtupleCreateMode::kNoMergeAfterOpen";
102 }
103
104 if ( fCreateMode == G4NtupleCreateMode::kMainBeforeOpen &&
105 fFileManager->GetNtupleFile(0) ) {
106 fCreateMode = G4NtupleCreateMode::kMainAfterOpen;
107 createMode = "G4NtupleCreateMode::kMainAfterOpen";
108 }
109
110#ifdef G4VERBOSE
111 if ( fState.GetVerboseL2() )
112 fState.GetVerboseL2()
113 ->Message("set", "ntuple create mode", createMode);
114#endif
115}

117//_____________________________________________________________________________
118void G4RootNtupleManager::CreateTNtuple(
119 G4TNtupleDescription<tools::wroot::ntuple>* ntupleDescription)
120{
121 if ( ! fNtupleDirectory ) {
122 G4String inFunction = "G4RootNtupleManager::::CreateTNtuple";
123 G4ExceptionDescription description;
124 description << " "
125 << "Cannot create ntuple. Ntuple directory does not exist." << G4endl;
126 G4Exception(inFunction, "Analysis_W002", JustWarning, description);
127 return;
128 }
129
130 ntupleDescription->fNtuple
131 = new tools::wroot::ntuple(
132 *fNtupleDirectory, ntupleDescription->fNtupleBooking, fRowWise);
133
134 auto basketSize = fFileManager->GetBasketSize();
135 ntupleDescription->fNtuple->set_basket_size(basketSize);
136
137 ntupleDescription->fIsNtupleOwner = false;
138 // ntuple object is deleted automatically when closing a file
139 fNtupleVector.push_back(ntupleDescription->fNtuple);
140}

142//_____________________________________________________________________________
143void G4RootNtupleManager::CreateTNtupleFromBooking(
144 G4TNtupleDescription<tools::wroot::ntuple>* ntupleDescription)
145{
146 if ( fCreateMode == G4NtupleCreateMode::kNoMergeBeforeOpen ) {
147
148 CreateTNtuple(ntupleDescription);
149 }
150
151 if ( fCreateMode == G4NtupleCreateMode::kMainBeforeOpen ) {
152 auto counter = 0;
153 for ( auto manager : fMainNtupleManagers ) {
154 if ( ! manager->GetNtupleVector().size() ) {
155 // Create only once !!
156 manager->SetNtupleFile(fFileManager->GetNtupleFile(counter));
157 manager->SetNtupleDirectory(fFileManager->GetMainNtupleDirectory(counter++));
158 manager->CreateNtuplesFromBooking();
159 }
160 }
161 }
162}

164//_____________________________________________________________________________
165void G4RootNtupleManager::FinishTNtuple(
166 G4TNtupleDescription<tools::wroot::ntuple>* ntupleDescription,
167 G4bool fromBooking)
168{
169// Create main ntuples
170
171 // Set create mode if not yet defined
172 if ( ! fromBooking ) {
173 SetCreateMode();
174 }
175
176 // Create ntuple if file is open
177 if ( fCreateMode == G4NtupleCreateMode::kNoMergeAfterOpen ) {
178 CreateTNtuple(ntupleDescription);
179 }
180
181 // Create main ntuples if file is open
182 if ( fCreateMode == G4NtupleCreateMode::kMainAfterOpen ) {
183 auto counter = 0;
184 for ( auto manager : fMainNtupleManagers ) {
185 auto warn = true;
186 manager->SetNtupleFile(fFileManager->GetNtupleFile(counter));
187 manager->SetNtupleDirectory(fFileManager->GetMainNtupleDirectory(counter++));
188 manager->CreateNtuple(ntupleDescription->fNtupleBooking, warn);
189 }
190 }
191}

193//_____________________________________________________________________________
194G4bool G4RootNtupleManager::Reset(G4bool deleteNtuple)
195{
196 G4TNtupleManager<tools::wroot::ntuple> ::Reset(deleteNtuple);
197 // this will clear ntuple vector
198
199 if ( fCreateMode == G4NtupleCreateMode::kNoMergeAfterOpen ) {
200 // clear also ntuple description vector
201 fNtupleDescriptionVector.clear();
202 }
203
204 auto finalResult = true;
205 for ( auto manager : fMainNtupleManagers ) {
206 auto result = manager->Reset(false);
207 finalResult = result && finalResult;
208 }
209
210 return finalResult;
211}

213//_____________________________________________________________________________
214G4bool G4RootNtupleManager::Merge()
215{
216 auto finalResult = true;
217
218 for ( auto manager : fMainNtupleManagers ) {
219 auto result = manager->Merge();
220 finalResult = result && finalResult;
221 }
222
223 return finalResult;
224}

226//_____________________________________________________________________________
227G4RootMainNtupleManager* G4RootNtupleManager::GetMainNtupleManager(G4int index) const
228{
229 if ( index < 0 || index >= G4int(fMainNtupleManagers.size()) ) {
230 G4String inFunction = "G4RootNtupleManager::::GetMainNtupleManager";
231 G4ExceptionDescription description;
232 description << " " << "main ntuple manager " << index << " does not exist.";
233 G4Exception(inFunction, "Analysis_W011", JustWarning, description);
234 return nullptr;
235 }
236
237 return fMainNtupleManagers[index];
238}

240//_____________________________________________________________________________
241unsigned int G4RootNtupleManager::GetBasketSize() const
242{
243 if ( ! fFileManager ) {
244 G4String inFunction = "G4RootNtupleManager::::GetBasketSize";
245 G4ExceptionDescription description;
246 description << " " << "File manager must be defined first.";
247 G4Exception(inFunction, "Analysis_W011", JustWarning, description);
248 return 0;
249 }
250
251 return fFileManager->GetBasketSize();
252}
