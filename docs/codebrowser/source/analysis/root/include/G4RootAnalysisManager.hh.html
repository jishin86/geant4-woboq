60class G4RootAnalysisManager : public G4ToolsAnalysisManager
61{
62 friend class G4RootMpiAnalysisManager;
63
64 public:
65 explicit G4RootAnalysisManager(G4bool isMaster = true);
66 virtual ~G4RootAnalysisManager();
67
68 // static methods
69 static G4RootAnalysisManager* Instance();
70 static G4bool IsInstance();
71
72 // Access methods
73 tools::wroot::ntuple* GetNtuple() const;
74 tools::wroot::ntuple* GetNtuple(G4int ntupleId) const;
75
76 // Iterators
77 std::vector<tools::wroot::ntuple*>::iterator BeginNtuple();
78 std::vector<tools::wroot::ntuple*>::iterator EndNtuple();
79 std::vector<tools::wroot::ntuple*>::const_iterator BeginConstNtuple() const;
80 std::vector<tools::wroot::ntuple*>::const_iterator EndConstNtuple() const;
81
82 // MT/MPI
83 void SetNtupleMerging(G4bool mergeNtuples,
84 G4int nofReducedNtupleFiles = 0,
85 G4bool rowWise = true,
86 unsigned int basketSize = fgkDefaultBasketSize);
87 void SetNtupleRowWise(G4bool rowWise);
88
89 protected:
90 // virtual methods from base class
91 virtual G4bool OpenFileImpl(const G4String& fileName) override;
92 virtual G4bool WriteImpl() final;
93 virtual G4bool CloseFileImpl(G4bool reset) override;
94 virtual G4bool IsOpenFileImpl() const final;
95 // virtual functions (overriden in MPI implementation)
96 virtual G4bool WriteNtuple();
97 virtual G4bool Reset();

99 private:

101 static constexpr unsigned int fgkDefaultBasketSize = 32000;
102
103 // static data members
104 static G4RootAnalysisManager* fgMasterInstance;
105 static G4ThreadLocal G4RootAnalysisManager* fgInstance;
106
107 // methods
108 void SetNtupleMergingMode(G4bool mergeNtuples,
109 G4int nofNtupleFiles,
110 G4bool rowWise);
111 void ClearNtupleManagers();
112 void CreateNtupleManagers();
113 G4int GetNtupleFileNumber();
114 G4bool ResetNtuple();
115
116 template <typename T>
117 G4bool WriteT(const std::vector<T*>& htVector,
118 const std::vector<G4HnInformation*>& hnVector,
119 tools::wroot::directory* directory,
120 const G4String& hnType);
121 G4bool WriteH1();
122 G4bool WriteH2();
123 G4bool WriteH3();
124 G4bool WriteP1();
125 G4bool WriteP2();
126
127 // data members
128 G4int fNofNtupleFiles;
129 G4bool fNtupleRowWise;
130 G4NtupleMergeMode fNtupleMergeMode;
131 G4RootNtupleManager* fNtupleManager;

133 std::shared_ptr<G4RootFileManager> fFileManager;
134};
