140template <>
141template <>
142inline G4bool G4TNtupleManager<tools::wroot::ntuple>::FillNtupleTColumn(
143 G4int ntupleId, G4int columnId, const std::string& value)
144{
145 if ( fState.GetIsActivation() && ( ! GetActivation(ntupleId) ) ) {
146 //G4cout << "Skipping FillNtupleIColumn for " << ntupleId << G4endl;
147 return false;
148 }
149
150 auto ntuple = GetNtupleInFunction(ntupleId, "FillNtupleTColumn");
151 if ( ! ntuple ) return false;
152
153 auto index = columnId - fFirstNtupleColumnId;
154 if ( index < 0 || index >= G4int(ntuple->columns().size()) ) {
155 G4ExceptionDescription description;
156 description << " " << "ntupleId " << ntupleId
157 << " columnId " << columnId << " does not exist.";
158 G4Exception("G4RootNtupleManager::FillNtupleTColumn()",
159 "Analysis_W011", JustWarning, description);
160 return false;
161 }
162
163 auto icolumn = ntuple->columns()[index];
164 auto column = dynamic_cast<tools::wroot::ntuple::column_string* >(icolumn);
165 if ( ! column ) {
166 G4ExceptionDescription description;
167 description << " Column type does not match: "
168 << " ntupleId " << ntupleId
169 << " columnId " << columnId << " value " << value;
170 G4Exception("G4RootNtupleManager:FillNtupleColumn",
171 "Analysis_W011", JustWarning, description);
172 return false;
173 }
174
175 column->fill(value);
176
177#ifdef G4VERBOSE
178 if ( fState.GetVerboseL4() ) {
179 G4ExceptionDescription description;
180 description << " ntupleId " << ntupleId
181 << " columnId " << columnId << " value " << value;
182 fState.GetVerboseL4()->Message("fill", "ntuple T column", description);
183 }
184#endif
185 return true;
186}
