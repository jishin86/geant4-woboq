34template <typename T>
35inline
36G4bool G4RootAnalysisManager::WriteT(const std::vector<T*>& htVector,
37 const std::vector<G4HnInformation*>& hnVector,
38 tools::wroot::directory* directory,
39 const G4String& hnType)
40{
41 // Do nothing if there is no file
42 if ( ! directory ) return true;
43
44 for ( G4int i=0; i<G4int(htVector.size()); ++i ) {
45 auto info = hnVector[i];
46 auto activation = info->GetActivation();
47 auto name = info->GetName();
48 // skip writing if activation is enabled and H1 is inactivated
49 if ( fState.GetIsActivation() && ( ! activation ) ) continue;
50 auto ht = htVector[i];
51#ifdef G4VERBOSE
52 if ( fState.GetVerboseL3() )
53 fState.GetVerboseL3()->Message("write", hnType, name);
54#endif
55 G4bool result
56 = to(*directory, *ht, name);
57 if ( ! result ) {
58 G4ExceptionDescription description;
59 description << " " << "saving " << hnType << " " << name << " failed";
60 G4Exception("G4RootAnalysisManager::Write()",
61 "Analysis_W022", JustWarning, description);
62 return false;
63 }
64 }
65
66 return true;
67}
