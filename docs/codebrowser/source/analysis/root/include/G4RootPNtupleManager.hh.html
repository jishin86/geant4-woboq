53// Mutex implementation as in pwroot.cpp
54// with replacement tools::mutex -> G4Mutex

56class mutex : public virtual tools::wroot::imutex {
57 typedef tools::wroot::imutex parent;
58public:
59 virtual bool lock() {
60 // G4cout << "!!! Mutex lock" << G4endl;
61 m_mutex.lock();
62 return true;}
63 virtual bool unlock() {
64 m_mutex.unlock();
65 // G4cout << "!!! Mutex unlock" << G4endl;
66 return true; }
67 //virtual bool trylock() {return m_mutex.trylock();}
68public:
69 mutex(G4AutoLock& a_mutex):m_mutex(a_mutex){}
70 virtual ~mutex(){}
71protected:
72 mutex(const mutex& a_from):parent(a_from),m_mutex(a_from.m_mutex){}
73 mutex& operator=(const mutex&){return *this;}
74protected:
75 G4AutoLock& m_mutex;
76};

182template <typename T>
183G4int G4RootPNtupleManager::CreateNtupleTColumn(
184 G4int ntupleId, const G4String& name, std::vector<T>* vector)
185{
186#ifdef G4VERBOSE
187 if ( fState.GetVerboseL4() ) {
188 G4ExceptionDescription description;
189 description << name << " ntupleId " << ntupleId;
190 fState.GetVerboseL4()->Message("create", "pntuple T column", description);
191 }
192#endif
193
194 auto ntupleDescription = GetNtupleDescriptionInFunction(ntupleId, "CreateNtupleTColumn");
195 if ( ! ntupleDescription ) return G4Analysis::kInvalidId;
196
197 // Save column info in booking
198 auto& ntupleBooking = ntupleDescription->fNtupleBooking;
199 auto index = ntupleBooking.columns().size();
200 if ( ! vector )
201 ntupleBooking.template add_column<T>(name);
202 else
203 ntupleBooking.template add_column<T>(name, *vector);
204
205 fLockFirstNtupleColumnId = true;
206
207#ifdef G4VERBOSE
208 if ( fState.GetVerboseL2() ) {
209 G4ExceptionDescription description;
210 description << name << " ntupleId " << ntupleId;
211 fState.GetVerboseL2()->Message("create", "pntuple T column", description);
212 }
213#endif
214
215 return index + fFirstNtupleColumnId;
216
217}

229template <>
230inline G4bool G4RootPNtupleManager::FillNtupleTColumn(
231 G4int ntupleId, G4int columnId, const std::string& value)
232{
233 if ( fState.GetIsActivation() && ( ! GetActivation(ntupleId) ) ) {
234 G4cout << "Skipping FillNtupleIColumn for " << ntupleId << G4endl;
235 return false;
236 }
237
238#ifdef G4VERBOSE
239 if ( fState.GetVerboseL4() ) {
240 G4ExceptionDescription description;
241 description << " ntupleId " << ntupleId
242 << " columnId " << columnId << " value " << value;
243 fState.GetVerboseL4()->Message("fill", "pntuple T column", description);
244 }
245#endif
246
247 auto ntuple = GetNtupleInFunction(ntupleId, "FillNtupleTColumn");
248 if ( ! ntuple ) return false;
249
250 auto index = columnId - fFirstNtupleColumnId;
251 if ( index < 0 || index >= G4int(ntuple->columns().size()) ) {
252 G4ExceptionDescription description;
253 description << " " << "ntupleId " << ntupleId
254 << " columnId " << columnId << " does not exist.";
255 G4Exception("G4RootNtupleManager::FillNtupleTColumn()",
256 "Analysis_W011", JustWarning, description);
257 return false;
258 }
259
260 auto icolumn = ntuple->columns()[index];
261 auto column = dynamic_cast<tools::wroot::base_pntuple::column_string* >(icolumn);
262 if ( ! column ) {
263 G4ExceptionDescription description;
264 description << " Column type does not match: "
265 << " ntupleId " << ntupleId
266 << " columnId " << columnId << " value " << value;
267 G4Exception("G4RootNtupleManager:FillNtupleColumn",
268 "Analysis_W011", JustWarning, description);
269 return false;
270 }
271
272 column->fill(value);
273
274#ifdef G4VERBOSE
275 if ( fState.GetVerboseL4() ) {
276 G4ExceptionDescription description;
277 description << " ntupleId " << ntupleId
278 << " columnId " << columnId << " value " << value;
279 fState.GetVerboseL4()->Message("done fill", "pntuple T column", description);
280 }
281#endif
282 return true;
283}

285//_____________________________________________________________________________
286template <typename T>
287G4bool G4RootPNtupleManager::FillNtupleTColumn(
288 G4int ntupleId, G4int columnId, const T& value)
289{
290 if ( fState.GetIsActivation() && ( ! GetActivation(ntupleId) ) ) {
291 G4cout << "Skipping FillNtupleIColumn for " << ntupleId << G4endl;
292 return false;
293 }
294
295#ifdef G4VERBOSE
296 if ( fState.GetVerboseL4() ) {
297 G4ExceptionDescription description;
298 description << " ntupleId " << ntupleId
299 << " columnId " << columnId << " value " << value;
300 fState.GetVerboseL4()->Message("fill", "pntuple T column", description);
301 }
302#endif
303
304 // get ntuple
305 auto ntuple = GetNtupleInFunction(ntupleId, "FillNtupleTColumn");
306 if ( ! ntuple ) return false;
307
308 // get generic column
309 auto index = columnId - fFirstNtupleColumnId;
310 if ( index < 0 || index >= G4int(ntuple->columns().size()) ) {
311 G4ExceptionDescription description;
312 description << " " << "ntupleId " << ntupleId
313 << " columnId " << columnId << " does not exist.";
314 G4Exception("G4TNtupleManager::FillNtupleTColumn()",
315 "Analysis_W011", JustWarning, description);
316 return false;
317 }
318 auto icolumn = ntuple->columns()[index];
319
320 // get column and check its type
321 auto column = dynamic_cast<tools::wroot::base_pntuple::column<T>* >(icolumn);
322 if ( ! column ) {
323 G4ExceptionDescription description;
324 description << " Column type does not match: "
325 << " ntupleId " << ntupleId
326 << " columnId " << columnId << " value " << value;
327 G4Exception("G4TNtupleManager:FillNtupleTColumn",
328 "Analysis_W011", JustWarning, description);
329 return false;
330 }
331
332 column->fill(value);
333
334#ifdef G4VERBOSE
335 if ( fState.GetVerboseL4() ) {
336 G4ExceptionDescription description;
337 description << " ntupleId " << ntupleId
338 << " columnId " << columnId << " value " << value;
339 fState.GetVerboseL4()->Message("done fill", "pntuple T column", description);
340 }
341#endif
342 return true;
343}
