42G4P2Messenger::G4P2Messenger(G4VAnalysisManager* manager)
43 : G4UImessenger(),
44 fManager(manager),
45 fHelper(nullptr),
46 fDirectory(nullptr),
47 fCreateP2Cmd(nullptr),
48 fSetP2Cmd(nullptr),
49 fSetP2XCmd(nullptr),
50 fSetP2YCmd(nullptr),
51 fSetP2ZCmd(nullptr),
52 fSetP2TitleCmd(nullptr),
53 fSetP2XAxisCmd(nullptr),
54 fSetP2YAxisCmd(nullptr)
55{
56 fHelper = G4Analysis::make_unique<G4AnalysisMessengerHelper>("p2");
57
58 fDirectory = fHelper->CreateHnDirectory();
59
60 CreateP2Cmd();
61
62 SetP2Cmd();
63 fSetP2XCmd = fHelper->CreateSetBinsCommand("x", this);
64 fSetP2YCmd = fHelper->CreateSetBinsCommand("y", this);
65 fSetP2ZCmd = fHelper->CreateSetValuesCommand("z", this);
66
67 fSetP2TitleCmd = fHelper->CreateSetTitleCommand(this);
68 fSetP2XAxisCmd = fHelper->CreateSetAxisCommand("x", this);
69 fSetP2YAxisCmd = fHelper->CreateSetAxisCommand("y", this);
70 fSetP2ZAxisCmd = fHelper->CreateSetAxisCommand("z", this);
71}

103 p2xValMax0->SetDefaultValue(1.);

298void G4P2Messenger::SetNewValue(G4UIcommand* command, G4String newValues)
299{
300 // tokenize parameters in a vector
301 std::vector<G4String> parameters;
302 G4Analysis::Tokenize(newValues, parameters);
303 // check consistency
304 if ( G4int(parameters.size()) != command->GetParameterEntries() ) {
305 // Should never happen but let's check anyway for consistency
306 fHelper->WarnAboutParameters(command, parameters.size());
307 return;
308 }
309
310 if ( command == fCreateP2Cmd.get() ) {
311 auto counter = 0;
312 auto name = parameters[counter++];
313 auto title = parameters[counter++];
314 G4AnalysisMessengerHelper::BinData xdata;
315 fHelper->GetBinData(xdata, parameters, counter);
316 auto xunit = GetUnitValue(xdata.fSunit);
317 G4AnalysisMessengerHelper::BinData ydata;
318 fHelper->GetBinData(ydata, parameters, counter);
319 auto yunit = GetUnitValue(ydata.fSunit);
320 G4AnalysisMessengerHelper::ValueData zdata;
321 fHelper->GetValueData(zdata, parameters, counter);
322 auto zunit = GetUnitValue(zdata.fSunit);
323 fManager->CreateP2(name, title,
324 xdata.fNbins, xdata.fVmin*xunit, xdata.fVmax*xunit,
325 ydata.fNbins, ydata.fVmin*yunit, ydata.fVmax*yunit,
326 zdata.fVmin*zunit, zdata.fVmax*zunit,
327 xdata.fSunit, ydata.fSunit, zdata.fSunit,
328 xdata.fSfcn, ydata.fSfcn, zdata.fSfcn,
329 xdata.fSbinScheme, ydata.fSbinScheme);
330 }
331 else if ( command == fSetP2Cmd.get() ) {
332 auto counter = 0;
333 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
334 G4AnalysisMessengerHelper::BinData xdata;
335 fHelper->GetBinData(xdata, parameters, counter);
336 auto xunit = GetUnitValue(xdata.fSunit);
337 G4AnalysisMessengerHelper::BinData ydata;
338 fHelper->GetBinData(ydata, parameters, counter);
339 auto yunit = GetUnitValue(ydata.fSunit);
340 G4AnalysisMessengerHelper::ValueData zdata;
341 fHelper->GetValueData(zdata, parameters, counter);
342 auto zunit = GetUnitValue(zdata.fSunit);
343 fManager->SetP2(id,
344 xdata.fNbins, xdata.fVmin*xunit, xdata.fVmax*xunit,
345 ydata.fNbins, ydata.fVmin*yunit, ydata.fVmax*yunit,
346 zdata.fVmin*zunit, zdata.fVmax*zunit,
347 xdata.fSunit, ydata.fSunit, zdata.fSunit,
348 xdata.fSfcn, ydata.fSfcn, zdata.fSfcn,
349 xdata.fSbinScheme, ydata.fSbinScheme);
350 }
351 else if ( command == fSetP2XCmd.get() ) {
352 // Only save values
353 auto counter = 0;
354 fXId = G4UIcommand::ConvertToInt(parameters[counter++]);
355 fHelper->GetBinData(fXData, parameters, counter);
356 }
357 else if ( command == fSetP2YCmd.get() ) {
358 // Save values
359 auto counter = 0;
360 fYId = G4UIcommand::ConvertToInt(parameters[counter++]);
361 // Check if setX command was called
362 if ( fXId == -1 || fXId != fYId ) {
363 fHelper->WarnAboutSetCommands();
364 return;
365 }
366 fHelper->GetBinData(fYData, parameters, counter);
367 // Set values
368 // (another set may follow if setZ is also called)
369 auto xunit = GetUnitValue(fXData.fSunit);
370 auto yunit = GetUnitValue(fYData.fSunit);
371 fManager->SetP2(fYId,
372 fXData.fNbins, fXData.fVmin*xunit, fXData.fVmax*xunit,
373 fYData.fNbins, fYData.fVmin*yunit, fYData.fVmax*yunit,
374 0., 0.,
375 fXData.fSunit, fYData.fSunit, "none",
376 fXData.fSfcn, fYData.fSfcn, "none",
377 fXData.fSbinScheme, fYData.fSbinScheme);
378 }
379 else if ( command == fSetP2ZCmd.get() ) {
380 auto counter = 0;
381 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
382 // Check if setX and setY command was called
383 if ( fXId == -1 || fXId != id ||
384 fYId == -1 || fYId != id ) {
385 fHelper->WarnAboutSetCommands();
386 return;
387 }
388 auto xunit = GetUnitValue(fXData.fSunit);
389 auto yunit = GetUnitValue(fYData.fSunit);
390 G4AnalysisMessengerHelper::ValueData zdata;
391 fHelper->GetValueData(zdata, parameters, counter);
392 auto zunit = GetUnitValue(zdata.fSunit);
393 fManager->SetP2(id,
394 fXData.fNbins, fXData.fVmin*xunit, fXData.fVmax*xunit,
395 fYData.fNbins, fYData.fVmin*yunit, fYData.fVmax*yunit,
396 zdata.fVmin*zunit, zdata.fVmax*zunit,
397 fXData.fSunit, fYData.fSunit, zdata.fSunit,
398 fXData.fSfcn, fYData.fSfcn, zdata.fSfcn,
399 fXData.fSbinScheme, fYData.fSbinScheme);
400 fXId = -1;
401 fYId = -1;
402 }
403 else if ( command == fSetP2TitleCmd.get() ) {
404 auto counter = 0;
405 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
406 auto title = parameters[counter++];
407 fManager->SetP2Title(id, title);
408 }
409 else if ( command == fSetP2XAxisCmd.get() ) {
410 auto counter = 0;
411 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
412 auto xaxis = parameters[counter++];
413 fManager->SetP2XAxisTitle(id, xaxis);
414 }
415 else if ( command == fSetP2YAxisCmd.get() ) {
416 auto counter = 0;
417 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
418 auto yaxis = parameters[counter++];
419 fManager->SetP2YAxisTitle(id, yaxis);
420 }
421 else if ( command == fSetP2ZAxisCmd.get() ) {
422 auto counter = 0;
423 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
424 auto zaxis = parameters[counter++];
425 fManager->SetP2ZAxisTitle(id, zaxis);
426 }
