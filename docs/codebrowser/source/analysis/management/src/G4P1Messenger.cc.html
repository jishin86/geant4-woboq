42G4P1Messenger::G4P1Messenger(G4VAnalysisManager* manager)
43 : G4UImessenger(),
44 fManager(manager),
45 fHelper(nullptr),
46 fDirectory(nullptr),
47 fCreateP1Cmd(nullptr),
48 fSetP1Cmd(nullptr),
49 fSetP1TitleCmd(nullptr),
50 fSetP1XAxisCmd(nullptr),
51 fSetP1YAxisCmd(nullptr),
52 fXData()
53{
54 fHelper = G4Analysis::make_unique<G4AnalysisMessengerHelper>("p1");
55
56 fDirectory = fHelper->CreateHnDirectory();
57
58 CreateP1Cmd();
59
60 SetP1Cmd();
61 fSetP1XCmd = fHelper->CreateSetBinsCommand("x", this);
62 fSetP1YCmd = fHelper->CreateSetValuesCommand("y", this);
63
64 fSetP1TitleCmd = fHelper->CreateSetTitleCommand(this);
65 fSetP1XAxisCmd = fHelper->CreateSetAxisCommand("x", this);
66 fSetP1YAxisCmd = fHelper->CreateSetAxisCommand("y", this);
67}

78void G4P1Messenger::CreateP1Cmd()
79{
80 auto p1Name = new G4UIparameter("name", 's', false);
81 p1Name->SetGuidance("Profile name (label)");
82
83 auto p1Title = new G4UIparameter("title", 's', false);
84 p1Title->SetGuidance("Profile title");
85
86 auto p1xNbins0 = new G4UIparameter("xnbins0", 'i', true);
87 p1xNbins0->SetGuidance("Number of x-bins (default = 100)");
88 p1xNbins0->SetGuidance("Can be reset with /analysis/p1/set command");
89 p1xNbins0->SetDefaultValue(100);
90
91 auto p1xValMin0 = new G4UIparameter("xvalMin0", 'd', true);
92 p1xValMin0->SetGuidance("Minimum x-value, expressed in unit (default = 0.)");
93 p1xValMin0->SetGuidance("Can be reset with /analysis/p1/set command");
94 p1xValMin0->SetDefaultValue(0.);
95
96 auto p1xValMax0 = new G4UIparameter("xvalMax0", 'd', true);
97 p1xValMax0->SetGuidance("Maximum x-value, expressed in unit (default = 1.)");
98 p1xValMax0->SetGuidance("Can be reset with /analysis/p1/set command");
99 p1xValMax0->SetDefaultValue(1.);
100
101 auto p1xValUnit0 = new G4UIparameter("xvalUnit0", 's', true);
102 p1xValUnit0->SetGuidance("The unit applied to filled x-values and xvalMin0, xvalMax0");
103 p1xValUnit0->SetDefaultValue("none");
104
105 auto p1xValFcn0 = new G4UIparameter("xvalFcn0", 's', true);
106 G4String fcnxGuidance = "The function applied to filled x-values (log, log10, exp, none).\n";
107 fcnxGuidance += "Note that the unit parameter cannot be omitted in this case,\n";
108 fcnxGuidance += "but none value should be used instead.";
109 p1xValFcn0->SetGuidance(fcnxGuidance);
110 p1xValFcn0->SetParameterCandidates("log log10 exp none");
111 p1xValFcn0->SetDefaultValue("none");
112
113 auto p1xValBinScheme0 = new G4UIparameter("xvalBinScheme0", 's', true);
114 G4String binSchemeGuidance = "The binning scheme (linear, log).\n";
115 p1xValBinScheme0->SetParameterCandidates("linear log");
116 binSchemeGuidance
117 += "Note that the unit and fcn parameters cannot be omitted in this case,\n";
118 binSchemeGuidance += "but none value should be used instead.";
119 p1xValBinScheme0->SetGuidance(binSchemeGuidance);
120 p1xValBinScheme0->SetDefaultValue("linear");
121
122 auto p1yValMin0 = new G4UIparameter("yvalMin0", 'd', true);
123 p1yValMin0->SetGuidance("Minimum y-value, expressed in unit (default = 0.)");
124 p1yValMin0->SetGuidance("Can be reset with /analysis/p1/set command");
125 p1yValMin0->SetDefaultValue(0.);
126
127 auto p1yValMax0 = new G4UIparameter("yvalMax0", 'd', true);
128 p1yValMax0->SetGuidance("Maximum y-value, expressed in unit (default = 1.)");
129 p1yValMax0->SetGuidance("Can be reset with /analysis/p1/set command");
130 p1yValMax0->SetDefaultValue(1.);
131
132 auto p1yValUnit0 = new G4UIparameter("yvalUnit0", 's', true);
133 p1yValUnit0->SetGuidance("The unit applied to filled y-values and yvalMin0, yvalMax0");
134 p1yValUnit0->SetDefaultValue("none");
135
136 auto p1yValFcn0 = new G4UIparameter("yvalFcn0", 's', true);
137 G4String fcnyGuidance = "The function applied to filled y-values (log, log10, exp, none).\n";
138 fcnyGuidance += "Note that the unit parameter cannot be omitted in this case,\n";
139 fcnyGuidance += "but none value should be used instead.";
140 p1yValFcn0->SetGuidance(fcnyGuidance);
141 p1yValFcn0->SetParameterCandidates("log log10 exp none");
142 p1yValFcn0->SetDefaultValue("none");
143
144 fCreateP1Cmd = G4Analysis::make_unique<G4UIcommand>("/analysis/p1/create", this);
145 fCreateP1Cmd->SetGuidance("Create 1D profile");
146 fCreateP1Cmd->SetParameter(p1Name);
147 fCreateP1Cmd->SetParameter(p1Title);
148 fCreateP1Cmd->SetParameter(p1xNbins0);
149 fCreateP1Cmd->SetParameter(p1xValMin0);
150 fCreateP1Cmd->SetParameter(p1xValMax0);
151 fCreateP1Cmd->SetParameter(p1xValUnit0);
152 fCreateP1Cmd->SetParameter(p1xValFcn0);
153 fCreateP1Cmd->SetParameter(p1xValBinScheme0);
154 fCreateP1Cmd->SetParameter(p1yValMin0);
155 fCreateP1Cmd->SetParameter(p1yValMax0);
156 fCreateP1Cmd->SetParameter(p1yValUnit0);
157 fCreateP1Cmd->SetParameter(p1yValFcn0);
158 fCreateP1Cmd->AvailableForStates(G4State_PreInit, G4State_Idle);
159}

163void G4P1Messenger::SetP1Cmd()
164{
165 auto p1Id = new G4UIparameter("id", 'i', false);
166 p1Id->SetGuidance("Profile id");
167 p1Id->SetParameterRange("id>=0");
168
169 auto p1xNbins = new G4UIparameter("xnbins", 'i', false);
170 p1xNbins->SetGuidance("Number of x-bins");
171
172 auto p1xValMin = new G4UIparameter("xvalMin", 'd', false);
173 p1xValMin->SetGuidance("Minimum x-value, expressed in unit");
174
175 auto p1xValMax = new G4UIparameter("xvalMax", 'd', false);
176 p1xValMax->SetGuidance("Maximum x-value, expressed in unit");
177
178 auto p1xValUnit = new G4UIparameter("xvalUnit", 's', true);
179 p1xValUnit->SetGuidance("The unit applied to filled x-values and xvalMin0, xvalMax0");
180 p1xValUnit->SetDefaultValue("none");
181
182 auto p1xValFcn = new G4UIparameter("xvalFcn", 's', true);
183 p1xValFcn->SetParameterCandidates("log log10 exp none");
184 G4String fcnxGuidance = "The function applied to filled x-values (log, log10, exp, none).\n";
185 fcnxGuidance += "Note that the unit parameter cannot be omitted in this case,\n";
186 fcnxGuidance += "but none value should be used instead.";
187 p1xValFcn->SetGuidance(fcnxGuidance);
188 p1xValFcn->SetDefaultValue("none");
189
190 auto p1xValBinScheme = new G4UIparameter("xvalBinScheme", 's', true);
191 G4String binSchemeGuidance = "The binning scheme (linear, log).\n";
192 p1xValBinScheme->SetParameterCandidates("linear log");
193 binSchemeGuidance
194 += "Note that the unit and fcn parameters cannot be omitted in this case,\n";
195 binSchemeGuidance += "but none value should be used instead.";
196 p1xValBinScheme->SetGuidance(binSchemeGuidance);
197 p1xValBinScheme->SetDefaultValue("linear");
198
199 auto p1yValMin = new G4UIparameter("yvalMin", 'd', true);
200 p1yValMin->SetGuidance("Minimum y-value, expressed in unit");
201
202 auto p1yValMax = new G4UIparameter("yvalMax", 'd', true);
203 p1yValMax->SetGuidance("Maximum y-value, expressed in unit");
204
205 auto p1yValUnit = new G4UIparameter("yvalUnit", 's', true);
206 p1yValUnit->SetGuidance("The unit applied to filled y-values and yvalMin0, yvalMax0");
207 p1yValUnit->SetDefaultValue("none");
208
209 auto p1yValFcn = new G4UIparameter("yvalFcn", 's', true);
210 p1yValFcn->SetParameterCandidates("log log10 exp none");
211 G4String fcnyGuidance = "The function applied to filled y-values (log, log10, exp, none).\n";
212 fcnyGuidance += "Note that the unit parameter cannot be omitted in this case,\n";
213 fcnyGuidance += "but none value should be used instead.";
214 p1yValFcn->SetGuidance(fcnyGuidance);
215 p1yValFcn->SetDefaultValue("none");
216
217 fSetP1Cmd = G4Analysis::make_unique<G4UIcommand>("/analysis/p1/set", this);
218 fSetP1Cmd->SetGuidance("Set parameters for the 1D profile of given id:");
219 fSetP1Cmd->SetGuidance(" nbins; xvalMin; xvalMax; xunit; xfunction; xbinScheme");
220 fSetP1Cmd->SetGuidance(" yvalMin; yvalMax; yunit; yfunction");
221 fSetP1Cmd->SetParameter(p1Id);
222 fSetP1Cmd->SetParameter(p1xNbins);
223 fSetP1Cmd->SetParameter(p1xValMin);
224 fSetP1Cmd->SetParameter(p1xValMax);
225 fSetP1Cmd->SetParameter(p1xValUnit);
226 fSetP1Cmd->SetParameter(p1xValFcn);
227 fSetP1Cmd->SetParameter(p1xValBinScheme);
228 fSetP1Cmd->SetParameter(p1yValMin);
229 fSetP1Cmd->SetParameter(p1yValMax);
230 fSetP1Cmd->SetParameter(p1yValUnit);
231 fSetP1Cmd->SetParameter(p1yValFcn);
232 fSetP1Cmd->AvailableForStates(G4State_PreInit, G4State_Idle);
233}

240void G4P1Messenger::SetNewValue(G4UIcommand* command, G4String newValues)
241{
242 // tokenize parameters in a vector
243 std::vector<G4String> parameters;
244 G4Analysis::Tokenize(newValues, parameters);
245 // check consistency
246 if ( G4int(parameters.size()) != command->GetParameterEntries() ) {
247 // Should never happen but let's check anyway for consistency
248 fHelper->WarnAboutParameters(command, parameters.size());
249 return;
250 }
251
252 if ( command == fCreateP1Cmd.get() ) {
253 auto counter = 0;
254 auto name = parameters[counter++];
255 auto title = parameters[counter++];
256 G4AnalysisMessengerHelper::BinData xdata;
257 fHelper->GetBinData(xdata, parameters, counter);
258 auto xunit = GetUnitValue(xdata.fSunit);
259 G4AnalysisMessengerHelper::ValueData ydata;
260 fHelper->GetValueData(ydata, parameters, counter);
261 auto yunit = GetUnitValue(ydata.fSunit);
262 fManager->CreateP1(name, title,
263 xdata.fNbins, xdata.fVmin*xunit, xdata.fVmax*xunit,
264 ydata.fVmin*yunit, ydata.fVmax*yunit,
265 xdata.fSunit, ydata.fSunit,
266 xdata.fSfcn, ydata.fSfcn,
267 xdata.fSbinScheme);
268 }
269 else if ( command == fSetP1Cmd.get() ) {
270 auto counter = 0;
271 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
272 G4AnalysisMessengerHelper::BinData xdata;
273 fHelper->GetBinData(xdata, parameters, counter);
274 auto xunit = GetUnitValue(xdata.fSunit);
275 G4AnalysisMessengerHelper::ValueData ydata;
276 fHelper->GetValueData(ydata, parameters, counter);
277 auto yunit = GetUnitValue(ydata.fSunit);
278 fManager->SetP1(id,
279 xdata.fNbins, xdata.fVmin*xunit, xdata.fVmax*xunit,
280 ydata.fVmin*yunit, ydata.fVmax*yunit,
281 xdata.fSunit, ydata.fSunit,
282 xdata.fSfcn, ydata.fSfcn,
283 xdata.fSbinScheme);
284 }
285 else if ( command == fSetP1XCmd.get() ) {
286 // Save values
287 auto counter = 0;
288 fXId = G4UIcommand::ConvertToInt(parameters[counter++]);
289 fHelper->GetBinData(fXData, parameters, counter);
290 // Set values
291 // (another set may follow if setY is also called)
292 auto xunit = GetUnitValue(fXData.fSunit);
293 fManager->SetP1(fXId,
294 fXData.fNbins, fXData.fVmin*xunit, fXData.fVmax*xunit,
295 0., 0.,
296 fXData.fSunit, "none",
297 fXData.fSfcn, "none",
298 fXData.fSbinScheme);
299 }
300 else if ( command == fSetP1YCmd.get() ) {
301 // Check if setX command was called
302 auto counter = 0;
303 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
304 if ( fXId == -1 || fXId != id ) {
305 fHelper->WarnAboutSetCommands();
306 return;
307 }
308 auto xunit = GetUnitValue(fXData.fSunit);
309 G4AnalysisMessengerHelper::ValueData ydata;
310 fHelper->GetValueData(ydata, parameters, counter);
311 auto yunit = GetUnitValue(ydata.fSunit);
312 fManager->SetP1(id,
313 fXData.fNbins, fXData.fVmin*xunit, fXData.fVmax*xunit,
314 ydata.fVmin*yunit, ydata.fVmax*yunit,
315 fXData.fSunit, ydata.fSunit,
316 fXData.fSfcn, ydata.fSfcn,
317 fXData.fSbinScheme);
318 fXId = -1;
319 }
320 else if ( command == fSetP1TitleCmd.get() ) {
321 auto counter = 0;
322 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
323 auto title = parameters[counter++];
324 fManager->SetP1Title(id, title);
325 }
326 else if ( command == fSetP1XAxisCmd.get() ) {
327 auto counter = 0;
328 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
329 auto xaxis = parameters[counter++];
330 fManager->SetP1XAxisTitle(id, xaxis);
331 }
332 else if ( command == fSetP1YAxisCmd.get() ) {
333 auto counter = 0;
334 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
335 auto yaxis = parameters[counter++];
336 fManager->SetP1YAxisTitle(id, yaxis);
337 }
