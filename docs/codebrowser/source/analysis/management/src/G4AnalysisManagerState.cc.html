35G4AnalysisManagerState::G4AnalysisManagerState(
36 const G4String& type, G4bool isMaster)
37 : fType(type),
38 fIsMaster(isMaster),
39 fIsActivation(false),
40 fVerboseLevel(0),
41 fCompressionLevel(1),
42 fVerboseL1(type,1),
43 fVerboseL2(type,2),
44 fVerboseL3(type,3),
45 fVerboseL4(type,4),
46 fpVerboseL1(0),
47 fpVerboseL2(0),
48 fpVerboseL3(0),
49 fpVerboseL4(0)
50{
51}

57//_____________________________________________________________________________
58void G4AnalysisManagerState::SetVerboseLevel(G4int verboseLevel)
59{
60 if ( verboseLevel == fVerboseLevel || verboseLevel < 0 ) return;
61
62 fVerboseLevel = verboseLevel;
63
64 if ( verboseLevel == 0 ) {
65 fpVerboseL1 = 0;
66 fpVerboseL2 = 0;
67 fpVerboseL3 = 0;
68 fpVerboseL4 = 0;
69 }
70 else if ( verboseLevel == 1 ) {
71 fpVerboseL1 = &fVerboseL1;
72 fpVerboseL2 = 0;
73 fpVerboseL3 = 0;
74 fpVerboseL4 = 0;
75 }
76 else if ( verboseLevel == 2 ) {
77 fpVerboseL1 = &fVerboseL1;
78 fpVerboseL2 = &fVerboseL2;
79 fpVerboseL3 = 0;
80 fpVerboseL4 = 0;
81 }
82 else if ( verboseLevel == 3 ) {
83 fpVerboseL1 = &fVerboseL1;
84 fpVerboseL2 = &fVerboseL2;
85 fpVerboseL3 = &fVerboseL3;
86 fpVerboseL4 = 0;
87 }
88 else {
89 fpVerboseL1 = &fVerboseL1;
90 fpVerboseL2 = &fVerboseL2;
91 fpVerboseL3 = &fVerboseL3;
92 fpVerboseL4 = &fVerboseL4;
93 }
94}
