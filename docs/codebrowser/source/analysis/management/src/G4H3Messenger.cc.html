42G4H3Messenger::G4H3Messenger(G4VAnalysisManager* manager)
43 : G4UImessenger(),
44 fManager(manager),
45 fHelper(nullptr),
46 fDirectory(nullptr),
47 fCreateH3Cmd(nullptr),
48 fSetH3Cmd(nullptr),
49 fSetH3XCmd(nullptr),
50 fSetH3YCmd(nullptr),
51 fSetH3ZCmd(nullptr),
52 fSetH3TitleCmd(nullptr),
53 fSetH3XAxisCmd(nullptr),
54 fSetH3YAxisCmd(nullptr),
55 fSetH3ZAxisCmd(nullptr),
56 fXId(-1),
57 fYId(-1),
58 fXData(),
59 fYData()
60{
61 fHelper = G4Analysis::make_unique<G4AnalysisMessengerHelper>("h3");
62
63 fDirectory = fHelper->CreateHnDirectory();
64
65 CreateH3Cmd();
66
67 SetH3Cmd();
68 fSetH3XCmd = fHelper->CreateSetBinsCommand("x", this);
69 fSetH3YCmd = fHelper->CreateSetBinsCommand("y", this);
70
71 fSetH3TitleCmd = fHelper->CreateSetTitleCommand(this);
72 fSetH3XAxisCmd = fHelper->CreateSetAxisCommand("x", this);
73 fSetH3YAxisCmd = fHelper->CreateSetAxisCommand("y", this);
74 fSetH3ZAxisCmd = fHelper->CreateSetAxisCommand("z", this);
75}

328void G4H3Messenger::SetNewValue(G4UIcommand* command, G4String newValues)
329{
330 // tokenize parameters in a vector
331 std::vector<G4String> parameters;
332 G4Analysis::Tokenize(newValues, parameters);
333 // check consistency
334 if ( G4int(parameters.size()) != command->GetParameterEntries() ) {
335 // Should never happen but let's check anyway for consistency
336 fHelper->WarnAboutParameters(command, parameters.size());
337 return;
338 }
339
340 if ( command == fCreateH3Cmd.get() ) {
341 auto counter = 0;
342 auto name = parameters[counter++];
343 auto title = parameters[counter++];
344 G4AnalysisMessengerHelper::BinData xdata;
345 fHelper->GetBinData(xdata, parameters, counter);
346 auto xunit = GetUnitValue(xdata.fSunit);
347 G4AnalysisMessengerHelper::BinData ydata;
348 fHelper->GetBinData(ydata, parameters, counter);
349 auto yunit = GetUnitValue(ydata.fSunit);
350 G4AnalysisMessengerHelper::BinData zdata;
351 fHelper->GetBinData(zdata, parameters, counter);
352 auto zunit = GetUnitValue(zdata.fSunit);
353 fManager->CreateH3(name, title,
354 xdata.fNbins, xdata.fVmin*xunit, xdata.fVmax*xunit,
355 ydata.fNbins, ydata.fVmin*yunit, ydata.fVmax*yunit,
356 zdata.fNbins, zdata.fVmin*zunit, zdata.fVmax*zunit,
357 xdata.fSunit, ydata.fSunit, zdata.fSunit,
358 xdata.fSfcn, ydata.fSfcn, zdata.fSfcn,
359 xdata.fSbinScheme, ydata.fSbinScheme, zdata.fSbinScheme);
360 }
361 else if ( command == fSetH3Cmd.get() ) {
362 auto counter = 0;
363 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
364 G4AnalysisMessengerHelper::BinData xdata;
365 fHelper->GetBinData(xdata, parameters, counter);
366 auto xunit = GetUnitValue(xdata.fSunit);
367 G4AnalysisMessengerHelper::BinData ydata;
368 fHelper->GetBinData(ydata, parameters, counter);
369 auto yunit = GetUnitValue(ydata.fSunit);
370 G4AnalysisMessengerHelper::BinData zdata;
371 fHelper->GetBinData(zdata, parameters, counter);
372 auto zunit = GetUnitValue(zdata.fSunit);
373 fManager->SetH3(id,
374 xdata.fNbins, xdata.fVmin*xunit, xdata.fVmax*xunit,
375 ydata.fNbins, ydata.fVmin*yunit, ydata.fVmax*yunit,
376 zdata.fNbins, zdata.fVmin*zunit, zdata.fVmax*zunit,
377 xdata.fSunit, ydata.fSunit, zdata.fSunit,
378 xdata.fSfcn, ydata.fSfcn, zdata.fSfcn,
379 xdata.fSbinScheme, ydata.fSbinScheme, zdata.fSbinScheme);
380 }
381 else if ( command == fSetH3XCmd.get() ) {
382 // Only save values
383 auto counter = 0;
384 fXId = G4UIcommand::ConvertToInt(parameters[counter++]);
385 fHelper->GetBinData(fXData, parameters, counter);
386 }
387 else if ( command == fSetH3YCmd.get() ) {
388 // Only save values
389 auto counter = 0;
390 fYId = G4UIcommand::ConvertToInt(parameters[counter++]);
391 fHelper->GetBinData(fYData, parameters, counter);
392 }
393 else if ( command == fSetH3ZCmd.get() ) {
394 auto counter = 0;
395 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
396 // Check if setX and setY command was called
397 if ( fXId == -1 || fXId != id ||
398 fYId == -1 || fYId != id ) {
399 fHelper->WarnAboutSetCommands();
400 return;
401 }
402 auto xunit = GetUnitValue(fXData.fSunit);
403 auto yunit = GetUnitValue(fYData.fSunit);
404 G4AnalysisMessengerHelper::BinData zdata;
405 fHelper->GetBinData(zdata, parameters, counter);
406 auto zunit = GetUnitValue(zdata.fSunit);
407 fManager->SetH3(id,
408 fXData.fNbins, fXData.fVmin*xunit, fXData.fVmax*xunit,
409 fYData.fNbins, fYData.fVmin*yunit, fYData.fVmax*yunit,
410 zdata.fNbins, zdata.fVmin*zunit, zdata.fVmax*zunit,
411 fXData.fSunit, fYData.fSunit, zdata.fSunit,
412 fXData.fSfcn, fYData.fSfcn, zdata.fSfcn,
413 fXData.fSbinScheme, fYData.fSbinScheme, zdata.fSbinScheme);
414 fXId = -1;
415 fYId = -1;
416 }
417 else if ( command == fSetH3TitleCmd.get() ) {
418 auto counter = 0;
419 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
420 auto title = parameters[counter++];
421 fManager->SetH3Title(id, title);
422 }
423 else if ( command == fSetH3XAxisCmd.get() ) {
424 auto counter = 0;
425 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
426 auto xaxis = parameters[counter++];
427 fManager->SetH3XAxisTitle(id, xaxis);
428 }
429 else if ( command == fSetH3YAxisCmd.get() ) {
430 auto counter = 0;
431 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
432 auto yaxis = parameters[counter++];
433 fManager->SetH3YAxisTitle(id, yaxis);
434 }
435 else if ( command == fSetH3ZAxisCmd.get() ) {
436 auto counter = 0;
437 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
438 auto zaxis = parameters[counter++];
439 fManager->SetH3ZAxisTitle(id, zaxis);
440 }
