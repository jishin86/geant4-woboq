27// Base class for File manager.
28//
29// Author: Ivana Hrivnacova, 18/06/2013 (ivana@ipno.in2p3.fr)
30
31#ifndef G4VFileManager_h
32#define G4VFileManager_h 1
33
34#include "G4BaseFileManager.hh"
35#include "globals.hh"
36
37class G4VFileManager : public G4BaseFileManager
38{
39 public:
40 explicit G4VFileManager(const G4AnalysisManagerState& state);
41 virtual ~G4VFileManager();
42
43 // Methods to manipulate files
44 virtual G4bool OpenFile(const G4String& fileName) = 0;
45 virtual G4bool WriteFile() = 0;
46 virtual G4bool CloseFile() = 0;
47
48 // Methods for handling files and directories names
49 //
50 virtual G4bool SetFileName(const G4String& fileName) final;
51
52 void LockHistoDirectoryName();
53 void LockNtupleDirectoryName();
54
55 G4bool SetHistoDirectoryName(const G4String& dirName);
56 G4bool SetNtupleDirectoryName(const G4String& dirName);
57
58 G4bool IsOpenFile() const;
59 G4String GetHistoDirectoryName() const;
60 G4String GetNtupleDirectoryName() const;
61
62 protected:
63 // data members
64 G4bool fIsOpenFile;
65 G4String fHistoDirectoryName;
66 G4String fNtupleDirectoryName;
67 G4bool fLockFileName;
68 G4bool fLockHistoDirectoryName;
69 G4bool fLockNtupleDirectoryName;
70};
71
72// inline functions
73
74inline G4bool G4VFileManager::IsOpenFile() const
75{ return fIsOpenFile; }
76
77inline void G4VFileManager::LockHistoDirectoryName()
78{ fLockHistoDirectoryName = true; }
79
80inline void G4VFileManager::LockNtupleDirectoryName()
81{ fLockNtupleDirectoryName = true; }
82
83
84inline G4String G4VFileManager::GetHistoDirectoryName() const {
85 return fHistoDirectoryName;
86}
87
88inline G4String G4VFileManager::GetNtupleDirectoryName() const {
89 return fNtupleDirectoryName;
90}
91
92#endif
