53template <typename TNTUPLE>
54G4TNtupleDescription<TNTUPLE>*
55G4TNtupleManager<TNTUPLE>::GetNtupleDescriptionInFunction(
56 G4int id, G4String functionName, G4bool warn) const
57{
58 auto index = id - fFirstId;
59 if ( index < 0 || index >= G4int(fNtupleDescriptionVector.size()) ) {
60 if ( warn) {
61 G4String inFunction = "G4TNtupleManager::";
62 inFunction += functionName;
63 G4ExceptionDescription description;
64 description << " " << "ntuple " << id << " does not exist.";
65 G4Exception(inFunction, "Analysis_W011", JustWarning, description);
66 }
67 return nullptr;
68 }
69
70 return fNtupleDescriptionVector[index];
71}

73//_____________________________________________________________________________
74template <typename TNTUPLE>
75TNTUPLE*
76G4TNtupleManager<TNTUPLE>::GetNtupleInFunction(
77 G4int id, G4String functionName, G4bool warn) const
78{
79 auto ntupleDescription = GetNtupleDescriptionInFunction(id, functionName);
80 if ( ! ntupleDescription ) return nullptr;
81
82 if ( ! ntupleDescription->fNtuple ) {
83 if ( warn ) {
84 G4String inFunction = "G4TNtupleManager::";
85 inFunction += functionName;
86 G4ExceptionDescription description;
87 description << " " << "ntupleId " << id << " does not exist.";
88 G4Exception(inFunction, "Analysis_W011", JustWarning, description);
89 }
90 return nullptr;
91 }
92
93 return ntupleDescription->fNtuple;
94}

97template <typename TNTUPLE>
98template <typename T>
99G4int G4TNtupleManager<TNTUPLE>::CreateNtupleTColumn(
100 G4int ntupleId, const G4String& name, std::vector<T>* vector)
101{
102#ifdef G4VERBOSE
103 if ( fState.GetVerboseL4() ) {
104 G4ExceptionDescription description;
105 description << name << " ntupleId " << ntupleId;
106 fState.GetVerboseL4()->Message("create", "ntuple T column", description);
107 }
108#endif
109
110 auto ntupleDescription = GetNtupleDescriptionInFunction(ntupleId, "CreateNtupleTColumn");
111 if ( ! ntupleDescription ) return G4Analysis::kInvalidId;
112
113 // Save column info in booking
114 auto& ntupleBooking = ntupleDescription->fNtupleBooking;
115 auto index = ntupleBooking.columns().size();
116 if ( ! vector )
117 ntupleBooking.template add_column<T>(name);
118 else
119 ntupleBooking.template add_column<T>(name, *vector);
120
121 // Create column if ntuple already exists
122 // CreateTColumnInNtuple<T>(ntupleDescription, name, vector);
123 fLockFirstNtupleColumnId = true;
124
125#ifdef G4VERBOSE
126 if ( fState.GetVerboseL2() ) {
127 G4ExceptionDescription description;
128 description << name << " ntupleId " << ntupleId;
129 fState.GetVerboseL2()->Message("create", "ntuple T column", description);
130 }
131#endif
132
133 return index + fFirstNtupleColumnId;
134
135}

138template <typename TNTUPLE>
139template <typename T>
140G4bool G4TNtupleManager<TNTUPLE>::FillNtupleTColumn(
141 G4int ntupleId, G4int columnId, const T& value)
142{
143 if ( fState.GetIsActivation() && ( ! GetActivation(ntupleId) ) ) {
144 //G4cout << "Skipping FillNtupleIColumn for " << ntupleId << G4endl;
145 return false;
146 }
147
148 // get ntuple
149 auto ntuple = GetNtupleInFunction(ntupleId, "FillNtupleTColumn");
150 if ( ! ntuple ) return false;
151
152 // get generic column
153 auto index = columnId - fFirstNtupleColumnId;
154 if ( index < 0 || index >= G4int(ntuple->columns().size()) ) {
155 G4ExceptionDescription description;
156 description << " " << "ntupleId " << ntupleId
157 << " columnId " << columnId << " does not exist.";
158 G4Exception("G4TNtupleManager::FillNtupleTColumn()",
159 "Analysis_W011", JustWarning, description);
160 return false;
161 }
162 auto icolumn = ntuple->columns()[index];
163
164 // get column and check its type
165 auto column = dynamic_cast<typename TNTUPLE::template column<T>* >(icolumn);
166 if ( ! column ) {
167 G4ExceptionDescription description;
168 description << " Column type does not match: "
169 << " ntupleId " << ntupleId
170 << " columnId " << columnId << " value " << value;
171 G4Exception("G4TNtupleManager:FillNtupleTColumn",
172 "Analysis_W011", JustWarning, description);
173 return false;
174 }
175
176 column->fill(value);
177
178#ifdef G4VERBOSE
179 if ( fState.GetVerboseL4() ) {
180 G4ExceptionDescription description;
181 description << " ntupleId " << ntupleId
182 << " columnId " << columnId << " value " << value;
183 fState.GetVerboseL4()->Message("fill", "ntuple T column", description);
184 }
185#endif
186 return true;
187}

194template <typename TNTUPLE>
195void G4TNtupleManager<TNTUPLE>::CreateNtuplesFromBooking()
196{
197// Create ntuple from ntuple_booking.
198
199 for ( auto ntupleDescription : fNtupleDescriptionVector ) {
200
201 // Do not create ntuple if it is inactivated
202 if ( fState.GetIsActivation() && ( ! ntupleDescription->fActivation ) ) continue;
203
204 // Do not create ntuple if it already exists
205 if ( ntupleDescription->fNtuple ) continue;
206
207#ifdef G4VERBOSE
208 if ( fState.GetVerboseL4() )
209 fState.GetVerboseL4()
210 ->Message("create from booking", "ntuple",
211 ntupleDescription->fNtupleBooking.name());
212#endif
213
214 // create ntuple
215 CreateTNtupleFromBooking(ntupleDescription);
216
217 // finish created ntuple
218 auto fromBooking = true;
219 FinishTNtuple(ntupleDescription, fromBooking);
220
221#ifdef G4VERBOSE
222 if ( fState.GetVerboseL3() )
223 fState.GetVerboseL3()
224 ->Message("create from booking", "ntuple",
225 ntupleDescription->fNtupleBooking.name());
226#endif
227 }
228}

239template <typename TNTUPLE>
240G4bool
241G4TNtupleManager<TNTUPLE>::Reset(G4bool deleteNtuple)
242{
243 for ( auto ntupleDescription : fNtupleDescriptionVector ) {
244 if ( deleteNtuple ) {
245 delete ntupleDescription->fNtuple;
246 }
247 ntupleDescription->fNtuple = 0;
248 }
249
250 fNtupleVector.clear();
251
252 return true;
253}

255//_____________________________________________________________________________
256template <typename TNTUPLE>
257G4int G4TNtupleManager<TNTUPLE>::CreateNtuple(
258 const G4String& name, const G4String& title)
259{
260#ifdef G4VERBOSE
261 if ( fState.GetVerboseL4() )
262 fState.GetVerboseL4()->Message("create", "ntuple", name);
263#endif
264
265 // Create ntuple description
266 auto index = fNtupleDescriptionVector.size();
267 auto ntupleDescription = new G4TNtupleDescription<TNTUPLE>();
268 fNtupleDescriptionVector.push_back(ntupleDescription);
269
270 // Save name & title in ntuple booking
271 ntupleDescription->fNtupleBooking.set_name(name);
272 ntupleDescription->fNtupleBooking.set_title(title);
273
274 // Create ntuple
275 // CreateTNtuple(ntupleDescription, name, title);
276 fLockFirstId = true;
277
278#ifdef G4VERBOSE
279 if ( fState.GetVerboseL2() ) {
280 G4ExceptionDescription description;
281 description << name << " ntupleId " << index + fFirstId;
282 fState.GetVerboseL2()->Message("create", "ntuple", description);
283 }
284#endif
285
286 return index + fFirstId;
287}

322//_____________________________________________________________________________
323template <typename TNTUPLE>
324void G4TNtupleManager<TNTUPLE>::FinishNtuple(
325 G4int ntupleId)
326{
327 auto ntupleDescription = GetNtupleDescriptionInFunction(ntupleId, "FinishNtuple");
328 if ( ! ntupleDescription ) return;
329
330#ifdef G4VERBOSE
331 if ( fState.GetVerboseL4() ) {
332 G4ExceptionDescription description;
333 description << ntupleDescription->fNtupleBooking.name() << " ntupleId " << ntupleId;
334 fState.GetVerboseL4()->Message("finish", "ntuple", description);
335 }
336#endif
337
338 // check that ntuple was really created
339 // auto ntuple = GetNtupleInFunction(ntupleId, "FinishNtuple", false);
340 // if ( ! ntuple ) return;
341 // The test has to be now done in the FinishTNtuple implementations
342
343 auto fromBooking = false;
344 FinishTNtuple(ntupleDescription, fromBooking);
345
346#ifdef G4VERBOSE
347 if ( fState.GetVerboseL2() ) {
348 G4ExceptionDescription description;
349 description << ntupleDescription->fNtupleBooking.name() << " ntupleId " << ntupleId;
350 fState.GetVerboseL2()->Message("finish", "ntuple", description);
351 }
352#endif
353}

386//_____________________________________________________________________________
387template <typename TNTUPLE>
388G4bool G4TNtupleManager<TNTUPLE>::AddNtupleRow(
389 G4int ntupleId)
390{
391 if ( fState.GetIsActivation() && ( ! GetActivation(ntupleId) ) ) {
392 //G4cout << "Skipping AddNtupleRow for " << ntupleId << G4endl;
393 return false;
394 }
395
396#ifdef G4VERBOSE
397 if ( fState.GetVerboseL4() ) {
398 G4ExceptionDescription description;
399 description << " ntupleId " << ntupleId;
400 fState.GetVerboseL4()->Message("add", "ntuple row", description);
401 }
402#endif
403
404 auto ntuple = GetNtupleInFunction(ntupleId, "AddNtupleRow");
405 if ( ! ntuple ) return false;
406
407 auto result = ntuple->add_row();
408 if ( ! result ) {
409 G4ExceptionDescription description;
410 description << " " << " ntupleId " << ntupleId
411 << "adding row has failed.";
412 G4Exception("G4TNtupleManager::AddTNtupleRow()",
413 "Analysis_W002", JustWarning, description);
414 }
415
416#ifdef G4VERBOSE
417 if ( fState.GetVerboseL2() ) {
418 G4ExceptionDescription description;
419 description << " ntupleId " << ntupleId;
420 fState.GetVerboseL2()->Message("add", "ntuple row", description);
421 }
422#endif
423
424 return true;
425}
