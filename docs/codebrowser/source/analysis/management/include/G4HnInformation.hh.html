39// The additional Hn information per dimension
40
41struct G4HnDimensionInformation
42{
43 G4HnDimensionInformation()
44 : fUnitName(),
45 fFcnName(),
46 fUnit(),
47 fFcn(nullptr),
48 fBinScheme(G4BinScheme::kLinear)
49 {}
50
51 G4HnDimensionInformation(
52 const G4String& unitName,
53 const G4String& fcnName,
54 G4double unit,
55 G4Fcn fcn,
56 G4BinScheme binScheme)
57 : fUnitName(unitName),
58 fFcnName(fcnName),
59 fUnit(unit),
60 fFcn(fcn),
61 fBinScheme(binScheme)
62 {}
63
64 G4HnDimensionInformation(const G4HnDimensionInformation& rhs)
65 : fUnitName(rhs.fUnitName),
66 fFcnName(rhs.fFcnName),
67 fUnit(rhs.fUnit),
68 fFcn(rhs.fFcn),
69 fBinScheme(rhs.fBinScheme)
70 {}
71
72 G4HnDimensionInformation& operator=(const G4HnDimensionInformation& rhs)
73 {
74 // check assignment to self
75 if (this == &rhs) return *this;
76
77 fUnitName = rhs.fUnitName;
78 fFcnName = rhs.fFcnName;
79 fUnit = rhs.fUnit;
80 fFcn = rhs.fFcn;
81 fBinScheme = rhs.fBinScheme;
82
83 return *this;
84 }
85
86 //G4String fName;
87 G4String fUnitName;
88 G4String fFcnName;
89 G4double fUnit;
90 G4Fcn fFcn;
91 G4BinScheme fBinScheme;
92};
93
94class G4HnInformation
95{
96 public:
97 G4HnInformation(const G4String& name, G4int nofDimensions)
98 : fName(name),
99 fHnDimensionInformations(),
100 fActivation(true),
101 fAscii(false),
102 fPlotting(false) { fHnDimensionInformations.reserve(nofDimensions); }
103
104 // Deleted default constructor
105 G4HnInformation() = delete;
106
107 // Set methods
108 void AddHnDimensionInformation(
109 const G4HnDimensionInformation& hnDimensionInformation);
110 void AddDimension(
111 const G4String& unitName, const G4String& fcnName, G4BinScheme binScheme);
112 void SetDimension(G4int dimension,
113 const G4String& unitName, const G4String& fcnName, G4BinScheme binScheme);
114 void SetActivation(G4bool activation);
115 void SetAscii(G4bool ascii);
116 void SetPlotting(G4bool plotting);
117
118 // Get methods
119 G4String GetName() const;
120 G4HnDimensionInformation* GetHnDimensionInformation(G4int dimension);
121 G4bool GetActivation() const;
122 G4bool GetAscii() const;
123 G4bool GetPlotting() const;
124
125 private:
126 // Data members
127 G4String fName;
128 std::vector<G4HnDimensionInformation> fHnDimensionInformations;
129 G4bool fActivation;
130 G4bool fAscii;
131 G4bool fPlotting;
132};
133
134// inline functions
135
136inline void G4HnInformation::AddHnDimensionInformation(
137 const G4HnDimensionInformation& hnDimensionInformation)
138{ fHnDimensionInformations.push_back(hnDimensionInformation); }
139
140inline void G4HnInformation::AddDimension(
141 const G4String& unitName, const G4String& fcnName, G4BinScheme binScheme)
142{
143 auto unit = G4Analysis::GetUnitValue(unitName);
144 auto fcn = G4Analysis::GetFunction(fcnName);
145 fHnDimensionInformations.push_back(
146 G4HnDimensionInformation(unitName, fcnName, unit, fcn, binScheme));
147}
148
149inline void G4HnInformation::SetDimension(G4int dimension,
150 const G4String& unitName, const G4String& fcnName, G4BinScheme binScheme)
151{
152 auto info = GetHnDimensionInformation(dimension);
153 auto unit = G4Analysis::GetUnitValue(unitName);
154 auto fcn = G4Analysis::GetFunction(fcnName);
155 info->fUnitName = unitName;
156 info->fFcnName = fcnName;
157 info->fUnit = unit;
158 info->fFcn = fcn;
159 info->fBinScheme = binScheme;
160}
161
162inline void G4HnInformation::SetActivation(G4bool activation)
163{ fActivation = activation; }
164
165inline void G4HnInformation::SetAscii(G4bool ascii)
166{ fAscii = ascii; }
167
168inline void G4HnInformation::SetPlotting(G4bool plotting)
169{ fPlotting = plotting; }
170
171inline G4String G4HnInformation::GetName() const
172{ return fName; }
173
174inline G4HnDimensionInformation* G4HnInformation::GetHnDimensionInformation(G4int dimension)
175{ return &(fHnDimensionInformations[dimension]); }
176
177inline G4bool G4HnInformation::GetActivation() const
178{ return fActivation; }
179
180inline G4bool G4HnInformation::GetAscii() const
181{ return fAscii; }
182
183inline G4bool G4HnInformation::GetPlotting() const
184{ return fPlotting; }

186#endif
