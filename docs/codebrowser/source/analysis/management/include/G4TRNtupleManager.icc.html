33template <typename TNTUPLE>
34G4TRNtupleManager<TNTUPLE>::G4TRNtupleManager(const G4AnalysisManagerState& state)

36 fNtupleDescriptionVector()
37{}

40template <typename TNTUPLE>
41G4TRNtupleManager<TNTUPLE>::~G4TRNtupleManager()
42{
43 for ( auto ntupleDescription : fNtupleDescriptionVector ) {
44 delete ntupleDescription;
45 }
46}

53template <typename TNTUPLE>
54G4TRNtupleDescription<TNTUPLE>*
55G4TRNtupleManager<TNTUPLE>::GetNtupleDescriptionInFunction(
56 G4int id, G4String functionName, G4bool warn) const
57{
58 G4int index = id - fFirstId;
59 if ( index < 0 || index >= G4int(fNtupleDescriptionVector.size()) ) {
60 if ( warn) {
61 G4String inFunction = "G4TRNtupleManager<TNTUPLE>::";
62 inFunction += functionName;
63 G4ExceptionDescription description;
64 description << " " << "ntuple " << id << " does not exist.";
65 G4Exception(inFunction, "Analysis_WR011", JustWarning, description);
66 }
67 return nullptr;
68 }
69
70 return fNtupleDescriptionVector[index];
71}

73//_____________________________________________________________________________
74template <typename TNTUPLE>
75template <typename T>
76G4bool
77G4TRNtupleManager<TNTUPLE>::SetNtupleTColumn(
78 G4int ntupleId, const G4String& columnName, T& value)
79{
80#ifdef G4VERBOSE
81 if ( fState.GetVerboseL4() ) {
82 G4ExceptionDescription description;
83 description << " ntupleId " << ntupleId << " " << columnName;
84 fState.GetVerboseL4()->Message("set", "ntuple I column", description);
85 }
86#endif
87
88 auto ntupleDescription = GetNtupleDescriptionInFunction(ntupleId, "SetNtupleTColumn");
89 if ( ! ntupleDescription ) return false;
90
91 auto ntupleBinding = ntupleDescription->fNtupleBinding;
92 ntupleBinding->add_column(columnName, value);
93
94#ifdef G4VERBOSE
95 if ( fState.GetVerboseL2() ) {
96 G4ExceptionDescription description;
97 description << " ntupleId " << ntupleId << " " << columnName;
98 fState.GetVerboseL2()->Message("set", "ntuple I colum", description, true);
99 }
100#endif
101
102 return true;
103}

148//_____________________________________________________________________________
149template <typename TNTUPLE>
150G4bool G4TRNtupleManager<TNTUPLE>::Reset()
151{
152// Reset ntuples
153
154 for ( auto ntupleDescription : fNtupleDescriptionVector ) {
155 ntupleDescription->fNtuple=0;
156 }
157
158 return true;
159}

169//_____________________________________________________________________________
170template <typename TNTUPLE>
171TNTUPLE*
172G4TRNtupleManager<TNTUPLE>::GetNtuple(
173 G4int ntupleId) const
174{
175 auto rntupleDescription = GetNtupleDescriptionInFunction(ntupleId, "GetNtuple");
176
177 if ( ! rntupleDescription ) return nullptr;
178
179 return rntupleDescription->fNtuple;
180}

182//_____________________________________________________________________________
183template <typename TNTUPLE>
184G4int G4TRNtupleManager<TNTUPLE>::SetNtuple(
185 G4TRNtupleDescription<TNTUPLE>* rntupleDescription)
186{
187 G4int id = fNtupleDescriptionVector.size() + fFirstId;
188
189 fNtupleDescriptionVector.push_back(rntupleDescription);
190
191 return id;
192}

258template <typename TNTUPLE>
259G4bool G4TRNtupleManager<TNTUPLE>::GetNtupleRow(G4int ntupleId)
260{
261#ifdef G4VERBOSE
262 if ( fState.GetVerboseL4() ) {
263 G4ExceptionDescription description;
264 description << " ntupleId " << ntupleId;
265 fState.GetVerboseL4()->Message("get", "ntuple row", description);
266 }
267#endif
268
269 auto ntupleDescription
270 = GetNtupleDescriptionInFunction(ntupleId, "GetNtupleRow");
271 if ( ! ntupleDescription ) return false;
272
273 auto next = GetTNtupleRow(ntupleDescription);
274
275#ifdef G4VERBOSE
276 if ( fState.GetVerboseL2() ) {
277 G4ExceptionDescription description;
278 description << " ntupleId " << ntupleId;
279 fState.GetVerboseL2()->Message("get", "ntuple row", description, true);
280 }
281#endif
282
283 return next;
284}
