35//_____________________________________________________________________________
36template <typename T>
37G4THnManager<T>::G4THnManager(const G4AnalysisManagerState& state,
38 const G4String& hnType)
39 : fState(state),
40 fTVector(),
41 fNameIdMap(),
42 fHnManager(nullptr)
43{
44 fHnManager = std::make_shared<G4HnManager>(hnType, state);
45}

47//_____________________________________________________________________________
48template <typename T>
49G4THnManager<T>::~G4THnManager()
50{
51 for ( auto t : fTVector ) {
52 delete t;
53 }
54}

60//_____________________________________________________________________________
61template <typename T>
62T* G4THnManager<T>::GetTInFunction(G4int id,
63 G4String functionName, G4bool warn,
64 G4bool onlyIfActive) const
65{
66 G4int index = id - fHnManager->GetFirstId();
67 if ( index < 0 || index >= G4int(fTVector.size()) ) {
68 if ( warn) {
69 G4String inFunction = "G4THnManager::";
70 inFunction += functionName;
71 G4ExceptionDescription description;
72 description << " " << "histogram " << id << " does not exist.";
73 G4Exception(inFunction, "Analysis_W011", JustWarning, description);
74 }
75 return nullptr;
76 }
77
78 // Do not return histogram if inactive
79 if ( fState.GetIsActivation() && onlyIfActive && ( ! fHnManager->GetActivation(id) ) ) {
80 return nullptr;
81 }
82
83 return fTVector[index];
84}

86//_____________________________________________________________________________
87template <typename T>
88G4int G4THnManager<T>::RegisterT(T* t, const G4String& name)
89{
90 G4int index = fTVector.size();
91 fTVector.push_back(t);
92
93 fHnManager->SetLockFirstId(true);
94 fNameIdMap[name] = index + fHnManager->GetFirstId();
95 return index + fHnManager->GetFirstId();
96}
97
98//_____________________________________________________________________________
99template <typename T>
100G4int G4THnManager<T>::GetTId(const G4String& name, G4bool warn) const
101{
102 auto it = fNameIdMap.find(name);
103 if ( it == fNameIdMap.end() ) {
104 if ( warn) {
105 G4String inFunction = "G4THnManager::GetH1Id";
106 G4ExceptionDescription description;
107 description << " " << "histogram " << name << " does not exist.";
108 G4Exception(inFunction, "Analysis_W011", JustWarning, description);
109 }
110 return G4Analysis::kInvalidId;
111 }
112 return it->second;
113}

116template <typename T>
117void G4THnManager<T>::AddTVector(const std::vector<T*>& tVector)
118{
119#ifdef G4VERBOSE
120 if ( fState.GetVerboseL4() )
121 fState.GetVerboseL4()->Message("merge", "all " + fHnManager->GetHnType(), "");
122#endif
123 // std::vector<tools::histo::h1d*>::const_iterator itw = h1Vector.begin();
124 // std::vector<tools::histo::h1d*>::iterator it;
125 // for (it = fH1Vector.begin(); it != fH1Vector.end(); it++ ) {
126 // (*it)->add(*(*itw++));
127 // }
128 auto itw = tVector.begin();
129 for ( auto t : fTVector ) {
130 t->add(*(*itw++));
131 }
132#ifdef G4VERBOSE
133 if ( fState.GetVerboseL1() )
134 fState.GetVerboseL1()->Message("merge", "all " + fHnManager->GetHnType(), "");
135#endif
136}

170//_____________________________________________________________________________
171template <typename T>
172G4bool G4THnManager<T>::Reset()
173{
174// Reset histograms and ntuple
175
176 G4bool finalResult = true;
177
178 for ( auto t : fTVector ) {
179 G4bool result = t->reset();
180 if ( ! result ) finalResult = false;
181 }
182
183 return finalResult;
184}

186//_____________________________________________________________________________
187template <typename T>
188G4bool G4THnManager<T>::IsEmpty() const
189{
190 return ! fTVector.size();
191}
