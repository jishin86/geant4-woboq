27// The main manager for Csv analysis.
28// It delegates most of functions to the object specific managers.
29
30// Author: Ivana Hrivnacova, 18/06/2013 (ivana@ipno.in2p3.fr)
31
32#ifndef G4CsvAnalysisManager_h
33#define G4CsvAnalysisManager_h 1
34
35#include "G4ToolsAnalysisManager.hh"
36#include "globals.hh"
37
38#include "tools/wcsv_ntuple"
39
40#include <memory>
41
42class G4CsvFileManager;
43class G4CsvNtupleManager;
44
45class G4CsvAnalysisManager : public G4ToolsAnalysisManager
46{
47 public:
48 explicit G4CsvAnalysisManager(G4bool isMaster = true);
49 ~G4CsvAnalysisManager();
50
51 // static methods
52 static G4CsvAnalysisManager* Instance();
53 static G4bool IsInstance();
54
55 // Access methods
56 tools::wcsv::ntuple* GetNtuple() const;
57 tools::wcsv::ntuple* GetNtuple(G4int ntupleId) const;
58
59 // Iterators
60 std::vector<tools::wcsv::ntuple*>::iterator BeginNtuple();
61 std::vector<tools::wcsv::ntuple*>::iterator EndNtuple();
62 std::vector<tools::wcsv::ntuple*>::const_iterator BeginConstNtuple() const;
63 std::vector<tools::wcsv::ntuple*>::const_iterator EndConstNtuple() const;
64
65 // Csv format specific option
66 void SetIsCommentedHeader(G4bool isCommentedHeader);
67 void SetIsHippoHeader(G4bool isHippoHeader);
68
69 protected:
70 // virtual methods from base class
71 virtual G4bool OpenFileImpl(const G4String& fileName) final;
72 virtual G4bool WriteImpl() final;
73 virtual G4bool CloseFileImpl(G4bool reset) final;
74 virtual G4bool IsOpenFileImpl() const final;
75
76 private:
77 // static data members
78 //
79 static G4CsvAnalysisManager* fgMasterInstance;
80 static G4ThreadLocal G4CsvAnalysisManager* fgInstance;
81
82 // methods
83 //
84 template <typename T>
85 G4bool WriteT(const std::vector<T*>& htVector,
86 const std::vector<G4HnInformation*>& hnVector,
87 const G4String& hnType);
88 G4bool WriteH1();
89 G4bool WriteH2();
90 G4bool WriteH3();
91 G4bool WriteP1();
92 G4bool WriteP2();
93 G4bool CloseNtupleFiles();
94 G4bool Reset();
95
96 // data members
97 G4CsvNtupleManager* fNtupleManager;
98 std::shared_ptr<G4CsvFileManager> fFileManager;
99};

103#endif
