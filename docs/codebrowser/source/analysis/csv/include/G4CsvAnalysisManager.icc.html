33template <>
34inline
35G4bool G4CsvAnalysisManager::WriteT(const std::vector<tools::histo::p1d*>& htVector,
36 const std::vector<G4HnInformation*>& hnVector,
37 const G4String& hnType)
38{
39 for ( G4int i=0; i<G4int(htVector.size()); ++i ) {
40 auto info = hnVector[i];
41 auto activation = info->GetActivation();
42 auto name = info->GetName();
43 // skip writing if activation is enabled and H1 is inactivated
44 if ( fState.GetIsActivation() && ( ! activation ) ) continue;
45 auto ht = htVector[i];
46 auto fileName = fFileManager->GetHnFileName(hnType, name);
47 std::ofstream hnFile(fileName);
48
49 // Do nothing if there is no file
50 if ( ! hnFile.is_open() ) return false;
51
52 auto result
53 = tools::wcsv::pto(hnFile, ht->s_cls(), *ht);
54
55 if ( ! result ) {
56 G4ExceptionDescription description;
57 description << " " << "saving " << hnType << " " << name << " failed";
58 G4Exception("G4CsvAnalysisManager::Write()",
59 "Analysis_W022", JustWarning, description);
60 return false;
61 }
62 hnFile.close();
63#ifdef G4VERBOSE
64 if ( fState.GetVerboseL1() )
65 fState.GetVerboseL1()->Message("write", "file", fileName);
66#endif
67 fFileManager->LockHistoDirectoryName();
68 }
69
70 return true;
71}
