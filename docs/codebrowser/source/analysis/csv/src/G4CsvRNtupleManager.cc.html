46G4bool G4CsvRNtupleManager::GetTNtupleRow(
47 G4TRNtupleDescription<tools::rcsv::ntuple>* ntupleDescription)
48{
49 auto ntuple = ntupleDescription->fNtuple;
50
51 auto isInitialized = ntupleDescription->fIsInitialized;
52 if ( ! isInitialized ) {
53 auto ntupleBinding = ntupleDescription->fNtupleBinding;
54 if ( ! ntuple->initialize(G4cout, *ntupleBinding) ) {
55 G4ExceptionDescription description;
56 description
57 << " "
58 << "Ntuple initialization failed !!";
59 G4Exception("G4CsvRNtuple::GetNtupleRow()",
60 "Analysis_WR021", JustWarning, description);
61 return false;
62 }
63 ntupleDescription->fIsInitialized = true;
64 ntuple->start();
65 }
66
67 auto next = ntuple->next();
68 if ( next ) {
69 if ( ! ntuple->get_row() ) {
70 G4ExceptionDescription description;
71 description
72 << " "
73 << "Ntuple get_row() failed !!";
74 G4Exception("G4CsvRNtuple::GetTNtupleRow()",
75 "Analysis_WR021", JustWarning, description);
76 return false;
77 }
78 }
79
80 return next;
81}
