27// Class template for accumulables handled by Geant4 analysis
28//
29// Author: Ivana Hrivnacova, 07/09/2015 (ivana@ipno.in2p3.fr)

31#ifndef G4Accumulable_h
32#define G4Accumulable_h 1
33
34#include "G4VAccumulable.hh"
35#include "G4MergeMode.hh"
36
37#include "globals.hh"
38
39template <typename T>
40class G4Accumulable : public G4VAccumulable
41{
42 public:
43 G4Accumulable(const G4String& name, T initValue,
44 G4MergeMode mergeMode = G4MergeMode::kAddition);
45 G4Accumulable(T initValue,
46 G4MergeMode mergeMode = G4MergeMode::kAddition);
47 G4Accumulable(const G4Accumulable& rhs);
48 G4Accumulable() = delete;
49 virtual ~G4Accumulable();
50
51 // operators
52 G4Accumulable<T>& operator= (const G4Accumulable<T>& rhs);
53 G4Accumulable<T>& operator+=(const G4Accumulable<T>& rhs);
54 G4Accumulable<T>& operator*=(const G4Accumulable<T>& rhs);
55 G4Accumulable<T> operator++(int); // postfix increment
56 G4Accumulable<T>& operator++(); // prefix increment
57
58 G4Accumulable<T>& operator= (const T& rhs);
59 G4Accumulable<T>& operator+=(const T& rhs);
60 G4Accumulable<T>& operator*=(const T& rhs);
61
62 // methods
63 virtual void Merge(const G4VAccumulable& other) final;
64 virtual void Reset() final;
65
66 // get methods
67 T GetValue() const;
68 G4MergeMode GetMergeMode() const;
69
70 private:
71 // data members
72 T fValue;
73 T fInitValue;
74 G4MergeMode fMergeMode;
75 G4MergeFunction<T> fMergeFunction;
76 };
77
78// inline functions

80#include "G4Accumulable.icc"
