27inline bool is_mode(mode_t a_mode) {
28 return a_mode<=0x006?true:false;
29}
30
31inline bool is_line(mode_t a_mode) {
32 if(a_mode==points()) return true; //0 sz line !
33 if(a_mode==lines()) return true;
34 if(a_mode==line_loop()) return true;
35 if(a_mode==line_strip()) return true;
36 return false;
37}

39inline void cvt_2to3(size_t a_npt,const float* a_xys,float*& a_xyzs) {
40 const float* vpos = a_xys;
41 float x,y;
42 for(size_t i=0;i<a_npt;i++) {
43 x = *vpos;vpos++;
44 y = *vpos;vpos++;
45 *a_xyzs = x;a_xyzs++;
46 *a_xyzs = y;a_xyzs++;
47 *a_xyzs = 0;a_xyzs++;
48 }
49}

52inline void triangle_fan_to_triangles(size_t a_npt,const float* a_fxyzs,float*& a_xyzs) {
53
54 const float* vpos = a_fxyzs;
55
56 float x1 = *vpos;vpos++;
57 float y1 = *vpos;vpos++;
58 float z1 = *vpos;vpos++;
59
60 float x2 = *vpos;vpos++;
61 float y2 = *vpos;vpos++;
62 float z2 = *vpos;vpos++;
63
64 float x3,y3,z3;
65 for(size_t i=2;i<a_npt;i++) {
66 x3 = *vpos;vpos++;
67 y3 = *vpos;vpos++;
68 z3 = *vpos;vpos++;
69 *a_xyzs = x1;a_xyzs++;
70 *a_xyzs = y1;a_xyzs++;
71 *a_xyzs = z1;a_xyzs++;
72
73 *a_xyzs = x2;a_xyzs++;
74 *a_xyzs = y2;a_xyzs++;
75 *a_xyzs = z2;a_xyzs++;
76
77 *a_xyzs = x3;a_xyzs++;
78 *a_xyzs = y3;a_xyzs++;
79 *a_xyzs = z3;a_xyzs++;
80
81 x2 = x3;
82 y2 = y3;
83 z2 = z3;
84 }

88inline void triangle_strip_to_triangles(size_t a_npt,const float* a_fxyzs,float*& a_xyzs) {
89
90 const float* vpos = a_fxyzs;
91
92 float x1 = *vpos;vpos++;
93 float y1 = *vpos;vpos++;
94 float z1 = *vpos;vpos++;
95
96 float x2 = *vpos;vpos++;
97 float y2 = *vpos;vpos++;
98 float z2 = *vpos;vpos++;
99
100 float x3,y3,z3;
101 bool flip = false;
102 for(size_t i=2;i<a_npt;i++) {
103 x3 = *vpos;vpos++;
104 y3 = *vpos;vpos++;
105 z3 = *vpos;vpos++;
106
107 if(flip) {
108 *a_xyzs = x1;a_xyzs++;
109 *a_xyzs = y1;a_xyzs++;
110 *a_xyzs = z1;a_xyzs++;
111
112 *a_xyzs = x3;a_xyzs++;
113 *a_xyzs = y3;a_xyzs++;
114 *a_xyzs = z3;a_xyzs++;
115
116 *a_xyzs = x2;a_xyzs++;
117 *a_xyzs = y2;a_xyzs++;
118 *a_xyzs = z2;a_xyzs++;
119 } else {
120 *a_xyzs = x1;a_xyzs++;
121 *a_xyzs = y1;a_xyzs++;
122 *a_xyzs = z1;a_xyzs++;
123
124 *a_xyzs = x2;a_xyzs++;
125 *a_xyzs = y2;a_xyzs++;
126 *a_xyzs = z2;a_xyzs++;
127
128 *a_xyzs = x3;a_xyzs++;
129 *a_xyzs = y3;a_xyzs++;
130 *a_xyzs = z3;a_xyzs++;
131 }
132
133 x1 = x2;
134 y1 = y2;
135 z1 = z2;
136
137 x2 = x3;
138 y2 = y3;
139 z2 = z3;
140
141 flip = flip?false:true;
142 }
143}
