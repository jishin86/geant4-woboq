11namespace tools {
12namespace wroot {

14class basket_add : public virtual branch::iadd_basket {
15 typedef branch::iadd_basket parent;
16public:
17 virtual bool add_basket(basket& a_basket) {
18 m_mutex.lock();
19 uint32 add_bytes,nout;
20 if(!m_main_branch->add_basket(m_main_file,a_basket,add_bytes,nout)) {
21 m_mutex.unlock();
22 return false;
23 }
24 m_main_branch->set_tot_bytes(m_main_branch->tot_bytes()+add_bytes);
25 m_main_branch->set_zip_bytes(m_main_branch->zip_bytes()+nout);
26 m_mutex.unlock();
27 return true;
28 }
29public:
30 basket_add(imutex& a_mutex,ifile& a_main_file)
31 :m_mutex(a_mutex),m_main_file(a_main_file),m_main_branch(0)
32 {}
33protected:
34 basket_add(const basket_add& a_from):parent()
35 ,m_mutex(a_from.m_mutex),m_main_file(a_from.m_main_file),m_main_branch(0)
36 {}
37 basket_add& operator=(const basket_add&){return *this;}
38public:
39 void set_main_branch(branch* a_main_branch) {m_main_branch = a_main_branch;}
40protected:
41 imutex& m_mutex;
42 ifile& m_main_file;
43 branch* m_main_branch;
44};
