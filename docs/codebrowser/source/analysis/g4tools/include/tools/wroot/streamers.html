30inline bool AttAxis_stream(buffer& a_buffer) {
31 int fNdivisions = 510; //Number of divisions(10000*n3 + 100*n2 + n1)
32 short fAxisColor = 1; //color of the line axis
33 short fLabelColor = 1; //color of labels
34 short fLabelFont = 62; //font for labels
35 float fLabelOffset = 0.005F; //offset of labels
36 float fLabelSize = 0.04F; //size of labels
37 float fTickLength = 0.03F; //length of tick marks
38 float fTitleOffset = 1; //offset of axis title
39 float fTitleSize = 0.04F; //size of axis title
40 short fTitleColor = 1; //color of axis title
41 short fTitleFont = 62; //font for axis title
42
43 // Version 4 streaming (ROOT/v3-00-6).
44 unsigned int beg;
45 if(!a_buffer.write_version(4,beg)) return false;
46
47 if(!a_buffer.write(fNdivisions)) return false;
48 if(!a_buffer.write(fAxisColor)) return false;
49 if(!a_buffer.write(fLabelColor)) return false;
50 if(!a_buffer.write(fLabelFont)) return false;
51 if(!a_buffer.write(fLabelOffset)) return false;
52 if(!a_buffer.write(fLabelSize)) return false;
53 if(!a_buffer.write(fTickLength)) return false;
54 if(!a_buffer.write(fTitleOffset)) return false;
55 if(!a_buffer.write(fTitleSize)) return false;
56 if(!a_buffer.write(fTitleColor)) return false;
57 if(!a_buffer.write(fTitleFont)) return false;
58
59 if(!a_buffer.set_byte_count(beg)) return false;
60 return true;
61}

63inline bool axis_stream(buffer& a_buffer,
64 const histo::axis<double,unsigned int>& a_axis,
65 const std::string& a_name,
66 const std::string& a_title) {
67 // Version 6 streaming (ROOT/v3-00-6).
68
69 unsigned int beg;
70 if(!a_buffer.write_version(6,beg)) return false;
71
72 if(!Named_stream(a_buffer,a_name,a_title)) return false;
73
74 if(!AttAxis_stream(a_buffer)) return false;
75
76 if(!a_buffer.write(a_axis.bins())) return false;
77 if(!a_buffer.write(a_axis.lower_edge())) return false;
78 if(!a_buffer.write(a_axis.upper_edge())) return false;
79
80 // fXbins
81 //if(a_axis.m_fixed) {
82 // std::vector<double> v;
83 // ArrayT<double> dummy(v);
84 // if(!dummy.stream(a_buffer)) return false; //TArrayD
85 //} else {
86 if(!a_buffer.write_array(a_axis.edges())) return false; //TArrayD
87 //}
88
89 if(!a_buffer.write((int)0)) return false; //fFirst
90 if(!a_buffer.write((int)0)) return false; //fLast
91
92 //Bool_t
93 if(!a_buffer.write((unsigned char)0)) return false; //TimeDisplay
94
95 //TString
96 if(!a_buffer.write(std::string())) return false; //TimeFormat
97
98 if(!a_buffer.set_byte_count(beg)) return false;
99
100 return true;
101}

103inline bool List_empty_stream(buffer& a_buffer) {
104 unsigned int beg;
105 if(!a_buffer.write_version(4,beg)) return false;
106 if(!Object_stream(a_buffer)) return false;
107 std::string name;
108 if(!a_buffer.write(name)) return false;
109 int nobjects = 0;
110 if(!a_buffer.write(nobjects)) return false;
111 if(!a_buffer.set_byte_count(beg)) return false;
112 return true;
113}

115template <class HIST>
116inline std::string axis_title(const HIST& a_h,const std::string& a_key) {
117 typedef std::map<std::string,std::string> annotations_t;
118 annotations_t::const_iterator it = a_h.annotations().find(a_key);
119 if(it==a_h.annotations().end()) return std::string();
120 return (*it).second;
121}

123template <class HIST>
124inline bool TH_write_1D(buffer& a_buffer,
125 const HIST& a_h,
126 const std::string& a_name,
127 const std::vector<double>& a_bin_Sw2) {
128
129 if(!a_buffer.write_version(3)) return false;
130
131 if(!Named_stream(a_buffer,a_name,a_h.title())) return false;
132
133 if(!AttLine_stream(a_buffer)) return false;
134 if(!AttFill_stream(a_buffer)) return false;
135 if(!AttMarker_stream(a_buffer)) return false;
136
137 if(!a_buffer.write((int)a_h.get_bins())) return false;
138
139 //fXAxis,fYAxis,fZAxis
140 if(a_h.dimension()==3) {
141
142 {histo::axis<double,unsigned int> haxis(a_h.get_axis(0));
143 if(!axis_stream(a_buffer,haxis,"xaxis",axis_title(a_h,histo::key_axis_x_title()))) return false;}
144
145 {histo::axis<double,unsigned int> haxis(a_h.get_axis(1));
146 if(!axis_stream(a_buffer,haxis,"yaxis",axis_title(a_h,histo::key_axis_y_title()))) return false;}
147
148 {histo::axis<double,unsigned int> haxis(a_h.get_axis(2));
149 if(!axis_stream(a_buffer,haxis,"zaxis",axis_title(a_h,histo::key_axis_z_title()))) return false;}
150
151 } else if(a_h.dimension()==2) {
152
153 {histo::axis<double,unsigned int> haxis(a_h.get_axis(0));
154 if(!axis_stream(a_buffer,haxis,"xaxis",axis_title(a_h,histo::key_axis_x_title()))) return false;}
155
156 {histo::axis<double,unsigned int> haxis(a_h.get_axis(1));
157 if(!axis_stream(a_buffer,haxis,"yaxis",axis_title(a_h,histo::key_axis_y_title()))) return false;}
158
159 {histo::axis<double,unsigned int> dummy;
160 dummy.configure(1,0,1);
161 if(!axis_stream(a_buffer,dummy,"zaxis",axis_title(a_h,histo::key_axis_z_title()))) return false;}
162
163 } else if(a_h.dimension()==1) {
164
165 {histo::axis<double,unsigned int> haxis(a_h.get_axis(0));
166 if(!axis_stream(a_buffer,haxis,"xaxis",axis_title(a_h,histo::key_axis_x_title()))) return false;}
167
168 {histo::axis<double,unsigned int> dummy;
169 dummy.configure(1,0,1);
170 if(!axis_stream(a_buffer,dummy,"yaxis",axis_title(a_h,histo::key_axis_y_title()))) return false;}
171
172 {histo::axis<double,unsigned int> dummy;
173 dummy.configure(1,0,1);
174 if(!axis_stream(a_buffer,dummy,"zaxis",axis_title(a_h,histo::key_axis_z_title()))) return false;}
175
176 } else {
177 return false;
178 }
179
180 if(!a_buffer.write((short)(1000 * 0.25))) return false; //fBarOffset
181 if(!a_buffer.write((short)(1000 * 0.5))) return false; //fBarWidth
182
183 if(!a_buffer.write((double)a_h.all_entries())) return false;
184 if(!a_buffer.write((double)a_h.get_in_range_Sw())) return false; //enforce double in case h1df
185 if(!a_buffer.write((double)a_h.get_in_range_Sw2())) return false; //idem
186
187 {double value;
188 a_h.get_ith_axis_Sxw(0,value);
189 if(!a_buffer.write(value)) return false;}
190
191 {double value;
192 a_h.get_ith_axis_Sx2w(0,value);
193 if(!a_buffer.write(value)) return false;}
194
195 if(!a_buffer.write((double)-1111)) return false; //fMaximum
196 if(!a_buffer.write((double)-1111)) return false; //fMinimum
197 if(!a_buffer.write((double)0)) return false; //NormFactor
198
199 if(!a_buffer.write_array(std::vector<double>())) return false; //fContour TArrayD
200
201 if(!a_buffer.write_array(a_bin_Sw2)) return false; //fSumw2 TArrayD
202
203 // store annotation on fOption
204 // but try to fool CERN-ROOT in order that it does not
205 // understand fOption as.. CERN-ROOT options !
206 //{std::string opt = " "+fAnnotation;
207 // opt[0] = 0; //awfull trick
208 // if(!a_buffer.write(opt)) return false;} //TString fOption
209 {std::string opt;
210 if(!a_buffer.write(opt)) return false;} //TString fOption
211
212 if(!List_empty_stream(a_buffer)) return false; //*TList fFunctions
213
214 return true;
215}

217template <class HIST>
218inline bool TH_write_2D(buffer& a_buffer,
219 const HIST& a_h,
220 const std::string& a_name,
221 const std::vector<double>& a_bin_Sw2) {
222 if(!a_buffer.write_version(3)) return false;
223 if(!TH_write_1D(a_buffer,a_h,a_name,a_bin_Sw2)) return false;
224 if(!a_buffer.write((double)1)) return false; //ScaleFactor
225
226 {double value;
227 a_h.get_ith_axis_Sxw(1,value);
228 if(!a_buffer.write(value)) return false;}
229
230 {double value;
231 a_h.get_ith_axis_Sx2w(1,value);
232 if(!a_buffer.write(value)) return false;}
233
234 if(!a_buffer.write((double)a_h.Sxyw())) return false; //Tsumwxy
235
236 return true;
237}

239template <class HIST>
240inline bool TH_write_3D(buffer& a_buffer,
241 const HIST& a_h,
242 const std::string& a_name,
243 const std::vector<double>& a_bin_Sw2) {
244 if(!a_buffer.write_version(4)) return false;
245 if(!TH_write_1D(a_buffer,a_h,a_name,a_bin_Sw2)) return false;
246 if(!Att3D_stream(a_buffer)) return false;
247
248 {double value;
249 a_h.get_ith_axis_Sxw(1,value);
250 if(!a_buffer.write(value)) return false;} //Tsumwy : Total Sum of weight*Y
251 {double value;
252 a_h.get_ith_axis_Sx2w(1,value);
253 if(!a_buffer.write(value)) return false;} //Tsumwy2 : Total Sum of weight*Y*Y
254 if(!a_buffer.write((double)a_h.Sxyw())) return false; //Tsumwxy : Total Sum of weight*X*Y
255
256 {double value;
257 a_h.get_ith_axis_Sxw(2,value);
258 if(!a_buffer.write(value)) return false;} //Tsumwz : Total Sum of weight*Z
259 {double value;
260 a_h.get_ith_axis_Sx2w(2,value);
261 if(!a_buffer.write(value)) return false;} //Tsumwz2 : Total Sum of weight*Z*Z
262 if(!a_buffer.write((double)a_h.Szxw())) return false; //Tsumwxz : Total Sum of weight*X*Z
263 if(!a_buffer.write((double)a_h.Syzw())) return false; //Tsumwyz : Total Sum of weight*Y*Z
264
265 return true;
266}

268inline bool TH1F_stream(buffer& a_buffer,const histo::h1df& a_h,const std::string& a_name) {
269 if(!a_buffer.write_version(1)) return false;

274 return true;

343inline bool TProfile_stream(buffer& a_buffer,const histo::p1d& a_p,const std::string& a_name){
344 if(!a_buffer.write_version(4)) return false;
345
346 //WARNING : the mapping histo::p1d / TProfile is not obvious.
347 //p1d::m_bin_Svw <---> TProfile::fArray
348 //p1d::m_bin_Sv2w <---> TProfile::fSumw2
349 //p1d::m_bin_Sw <---> TProfile::fBinEntries
350
351 // TH1D_stream(a_buffer,h,a_name) :
352 //if(!a_buffer.write_version(1)) return false;
353 //if(!TH_write_1D(a_buffer,a_h,a_name,a_h.bins_sum_w2())) return false; //fSumw2 TArrayD
354 //if(!a_buffer.write_array(a_h.bins_sum_w())) return false; //fArray TArrayD
355 // but for profile :
356 if(!a_buffer.write_version(1)) return false;
357 if(!TH_write_1D(a_buffer,a_p,a_name,a_p.bins_sum_v2w())) return false; //fSumw2 TArrayD
358 if(!a_buffer.write_array(a_p.bins_sum_vw())) return false; //fArray TArrayD
359
360 //TProfile specific :
361 if(!a_buffer.write_array(a_p.bins_sum_w())) return false; //fBinEntries TArrayD
362
363 int errorMode = 0;
364 if(!a_buffer.write(errorMode)) return false;
365 if(!a_buffer.write(a_p.min_v())) return false;
366 if(!a_buffer.write(a_p.max_v())) return false;
367
368 // version 4 :
369 if(!a_buffer.write(a_p.get_Svw())) return false; //Double_t fTsumwy; //Total Sum of weight*Y
370 if(!a_buffer.write(a_p.get_Sv2w())) return false; //Double_t fTsumwy2; //Total Sum of weight*Y*Y
371
372 return true;
373}
