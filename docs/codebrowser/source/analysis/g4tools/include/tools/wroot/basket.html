33 //G.Barrand : the below test is "too much". Someone

48 if(m_seek_key) {
49 m_out << "tools::wroot::basket::stream :"
50 << " m_seek_key is not null."
51 << std::endl;
52 return false;
53 }
54 if(m_last) {
55 m_out << "tools::wroot::basket::stream :"
56 << " m_last is not null."
57 << std::endl;
58 return false;
59 }
60 if(!m_entry_offset) {
61 m_out << "tools::wroot::basket::stream :"

67 {uint32 _last = m_data.length()+m_key_length;

71 if(m_last>m_buf_size) {
72 const_cast<basket&>(*this).m_buf_size = m_last;
73 }
74
75 char flag = 11;
76 if(m_displacement) flag += 40;
77 if(!_stream_header(a_buffer,m_verbose,flag)) return false;
78
79 if(m_entry_offset && m_nev) {
80 if(!a_buffer.write_array(m_entry_offset,m_nev)) return false;
81 if(m_displacement) {
82 if(!a_buffer.write_array(m_displacement,m_nev)) return false;

84 }
85
86 if(m_data.to_displace()) {
87 if(!const_cast<basket&>(*this).m_data.displace_mapped(m_key_length)) {
88 m_out << "tools::wroot::basket::stream :"
89 << " m_data.displace_mapped() failed."
90 << std::endl;
91 return false;
92 }
93 }
94
95 buffer bref(m_out,a_buffer.byte_swap(),256);
96 if(!_stream_header(bref,m_verbose)) return false; //then header stored twice !
97 //if(bref.length()!=m_key_length) {}
98 if(!bref.write_fast_array(m_data.buf(),m_data.length())) return false;
99 if(!a_buffer.write_fast_array(bref.buf(),bref.length())) return false;
100
101 return true;
102 }

200 bool update(uint32 a_offset) {
201 if(m_entry_offset) {
202 if((m_nev+1)>=m_nev_buf_size) { // for the +1, we follow CERN-ROOT/TBasket logic that wants to store
203 // nev+1 elements for m_entry_offset in the write_on_file() method.
204 uint32 newsize = mx<uint32>(10,2*m_nev_buf_size);
205 if(!realloc<int>(m_entry_offset,newsize,m_nev_buf_size,true)){
206 m_out << "tools::wroot::basket::update : realloc failed." << std::endl;
207 return false;
208 }
209 if(m_displacement) {
210 if(!realloc<int>(m_displacement,newsize,m_nev_buf_size,true)){
211 m_out << "tools::wroot::basket::update : realloc failed." << std::endl;
212 return false;
213 }
214 }
215 m_nev_buf_size = newsize;
216 }
217 m_entry_offset[m_nev] = (int)a_offset;
218 }
219 m_nev++;
220 return true;
221 }

223 bool write_on_file(ifile& a_file,uint16 a_cycle,uint32& a_nbytes) {
224 // write m_data buffer into file.
225 //NOTE : m_data does not contain the key at its head.
226 // At this point m_seek_key should be 0.
227
228 a_nbytes = 0;
229
230 if(m_seek_key) {
231 m_out << "tools::wroot::basket::write_on_file :"
232 << " m_seek_key should be 0."
233 << std::endl;
234 return false;
235 }
236
237 if(m_version>big_file_version_tag()) {
238 } else {
239 m_out << "tools::wroot::basket::write_on_file : "
240 << " we should not pass here (1)."
241 << std::endl;
242 return false;
243/*
244 if(a_file.END()>START_BIG_FILE()) {
245 //GB : enforce m_version>big_file_version_tag() if m_version is still 2 but

247 // write a big m_seek_key on a seek32 and then have
248 // a problem when reading.
249
250 //m_out << "tools::wroot::basket::write_on_file : "
251 // << " WARNING : pos>START_BIG_FILE."
252 // << std::endl;
253
254 m_version += big_file_version_tag();
255 m_key_length += 8;
256
257 if(m_entry_offset) {
258 for(uint32 i=0;i<m_nev;i++) m_entry_offset[i] += 8;
259 if(m_displacement) {
260 //??? Do we have to shift them ?
261 m_out << "tools::wroot::basket::write_on_file : "
262 << " displace logic : m_displacement not null."
263 << std::endl;
264 }
265 }
266
267 }
268*/
269 }
270
271 // Transfer m_entry_offset table at the end of fBuffer. Offsets to fBuffer
272 // are transformed in entry length to optimize compression algorithm.
273 m_last = m_key_length+m_data.length();
274 if(m_entry_offset) {
275 if(!m_data.write_array<int>(m_entry_offset,m_nev+1)) { // for the +1 we follow CERN-ROOT/TBasket logic.
276 delete [] m_entry_offset;
277 m_entry_offset = 0;
278 return false;
279 }
280 delete [] m_entry_offset;
281 m_entry_offset = 0;
282 if(m_displacement) {

297 if(!m_data.displace_mapped(m_key_length)) return false;

313

318 if(kdelete) delete [] kbuf;

326 m_out << "tools::wroot::basket::write_on_file :"

340
341 m_data.pos() = m_data.buf(); //empty m_data.
342
343 a_nbytes = m_key_length + klen;

347 uint32 header_record_size(uint32 a_version) const {
348 // header only.
349 uint32 nbytes = parent::record_size(a_version);
350
351 nbytes += sizeof(short); //version
352 nbytes += sizeof(uint32); //m_buf_size
353 nbytes += sizeof(uint32); //m_nev_buf_size
354 nbytes += sizeof(uint32); //m_nev
355 nbytes += sizeof(uint32); //m_last
356 nbytes += sizeof(char); //flag
357
358 return nbytes;
359 }
360 bool _stream_header(buffer& a_buffer,bool a_verbose,char a_flag = 0) const {
361 {uint32 l = parent::record_size(m_version);
362 if((a_buffer.length()+l)>a_buffer.size()) {
363 if(!a_buffer.expand(a_buffer.size()+l)) return false;
364 }
365 wbuf wb(m_out,a_buffer.byte_swap(),a_buffer.max_pos(),a_buffer.pos());
366 if(!parent::to_buffer(wb,a_verbose)) return false;}
367
368 if(!a_buffer.write_version(2)) return false;
369 if(!a_buffer.write(m_buf_size)) return false;
370 if(!a_buffer.write(m_nev_buf_size)) return false;
371 if(!a_buffer.write(m_nev)) return false;
372 if(!a_buffer.write(m_last)) return false;
373 if(!a_buffer.write(a_flag)) return false;
374 return true;
375 }
