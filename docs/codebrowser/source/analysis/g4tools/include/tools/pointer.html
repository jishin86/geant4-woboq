18inline bool to_pointer(const std::string& a_string,void*& a_value){
19 upointer v = 0;
20 if(::sscanf(a_string.c_str(),upointer_format_x(),&v)!=1) {
21 if(::sscanf(a_string.c_str(),upointer_format(),&v)!=1) {
22 a_value = 0;
23 return false;
24 }
25 }
26 a_value = (void*)v;
27 return true;
28}

42inline bool p2s(const void* a_value,std::string& a_s){
43 char s[512];
44 snpf(s,sizeof(s),upointer_format(),(upointer)a_value);
45 a_s = s;
46 return true;
