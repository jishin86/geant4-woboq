17template <class T>
18inline const T* vec_data(const std::vector<T>& a_vec) {
19#if TARGET_OS_IPHONE || __INTEL_COMPILER || _MSC_VER || (__GNUC__ == 4 && __GNUC_MINOR__ <= 0)
20 if(a_vec.empty()) return 0;
21 const T& vf = a_vec.front();
22 return &vf;
23#else
24 return a_vec.data();
25#endif
26}
