78class parser {
79public:
80 parser():m_top(0){}
81 virtual ~parser(){m_top.clear();}
82protected:
83 parser(const parser& a_from):m_top(a_from.m_top){}
84 parser& operator=(const parser&){return *this;}
85public:
86 bool parse(const std::string& a_s){
87 m_top.clear();
88 tree* prev = &m_top;
89 {tree tmp(0);
90 std::string s;
91 for(std::string::const_iterator it=a_s.begin();;++it) {
92 if(it==a_s.end()) {
93 if(s.size()) {
94 tmp.m_dcl = s;
95 tmp.m_parent = prev;
96 prev->m_sub.push_back(tmp);
97 s.clear();
98 }
99 tmp.clear();
100 break;
101 } else {
102 const char& c = *it;
103 if(c==',') {
104 if(s.size()) {
105 tmp.m_dcl = s;
106 tmp.m_parent = prev;
107 prev->m_sub.push_back(tmp);
108 s.clear();
109 }
110 tmp.clear();
111 } else if(c=='{') {
112 tmp.clear();
113 if(s.size()) {
114 tmp.m_dcl = s;
115 s.clear();
116 }
117 tmp.m_parent = prev;
118 prev->m_sub.push_back(tmp);
119 prev = &(prev->m_sub.back());
120 } else if(c=='}') {
121 if(s.size()) {
122 tmp.m_dcl = s;
123 tmp.m_parent = prev;
124 prev->m_sub.push_back(tmp);
125 s.clear();
126 }
127 tmp.clear();
128 prev = prev->m_parent;
129 if(!prev) return false; //should not happen.
130 } else {
131 s += c;
132 }
133 }
134 }}
135 return true;
136 }
137
138 void dump(std::ostream& a_out) {m_top.dump_tree(a_out,"");}
139protected:
140 tree m_top;
141};

151inline void delete_columns(std::vector<value>& a_vars) {
152 tools_vforit(value,a_vars,it) {
153 if((*it).type()==value::VOID_STAR) {
154 std::vector<value>* vars =
155 (std::vector<value>*)(*it).get_void_star();
156 delete_columns(*vars);
157 delete vars;
158 }
159 }
160 a_vars.clear();
161}
162
163inline void copy_columns(const std::vector<value>& a_from,std::vector<value>& a_to) {
164 std::vector<value>::const_iterator it;
165 for(it=a_from.begin();it!=a_from.end();++it) {
166 if((*it).type()==value::VOID_STAR) {
167 std::vector<value>* vars = new std::vector<value>();
168 value v((void*)vars);
169 v.set_label((*it).label());
170 a_to.push_back(v);
171 std::vector<value>* p =
172 (std::vector<value>*)(*it).get_void_star();
173 copy_columns(*p,*vars);
174 } else {
175 a_to.push_back(*it);
176 }
177 }
178}

180inline void dump_columns(std::ostream& a_out,const std::vector<value>& a_vars,const std::string& a_margin = "") {
181 std::vector<value>::const_iterator it;
182 for(it=a_vars.begin();it!=a_vars.end();++it) {
183 if((*it).type()==value::VOID_STAR) {
184 a_out << a_margin
185 << "ITuple : " << (*it).label() << " : begin "
186 << std::endl;
187 std::vector<value>* vars = (std::vector<value>*)(*it).get_void_star();
188 dump_columns(a_out,*vars,a_margin+" ");
189 //a_out << a_margin
190 // << "ITuple : " << (*it).label() << " : end "
191 // << std::endl;
192 } else {
193 std::string stype;
194 (*it).s_type(stype);
195 std::string sval;
196 (*it).tos(sval);
197
198 a_out << a_margin
199 << stype << " : "
200 << (*it).label() << " : "
201 << sval
202 << std::endl;
203 }

247 bool analyse(columns::tree& a_tree,std::vector<value>& a_stack) {
248 if(a_tree.m_dcl.empty()) { //top
249 //std::cout << "debug : dcl empty" << std::endl;
250 tools_vforit(columns::tree,a_tree.m_sub,it) {
251 if(!analyse(*it,a_stack)) return false;
252 }
253 } else {
254 //std::cout << "debug : dcl not empty" << std::endl;
255 if(is_spaces(a_tree.m_dcl)) return true;
256 value* v = analyse_dcl(a_tree.m_dcl);
257 if(!v) return false;
258 //std::cout << "debug : dcl label " << v->label() << std::endl;
259 if(a_tree.m_sub.size()) {
260 if(v->type()!=value::VOID_STAR) {
261 m_out << "tools::columns::finder::analyse :"
262 << " Expect a VOID_STAR."
263 << std::endl;
264 delete v;
265 return false;
266 }
267 m_cur_type = value::NONE;
268 std::vector<value>* stk = new std::vector<value>();
269 tools_vforit(columns::tree,a_tree.m_sub,it) {
270 if(!analyse(*it,*stk)) {
271 delete v;
272 return false;
273 }
274 }
275 v->set((void*)stk);
276 } else {
277 m_cur_type = v->type();
278 }
279 a_stack.push_back(*v);
280 delete v;
281 }
282 return true;
283 }
284 value* analyse_dcl(const std::string& a_s) {
285 std::vector<std::string> ws;
286 words(a_s,"=",false,ws);
287 if(ws.size()==2) { //<type> <name>=<value>
288 std::vector<std::string> swords;
289 words(ws[0]," ",false,swords);
290 if(swords.size()==2) {
291
292 strip(swords[0]);
293 strip(swords[1]);
294
295 if(swords[0]=="ITuple") {
296
297 //create a value::VOID_STAR :
298 value* v = new value((void*)0);
299 v->set_label(swords[1]);
300 return v;
301
302 } else {
303
304 value::e_type type;
305 if(!s2type(swords[0],type)){
306 m_out << "tools::columns::finder::analyse_dcl :"
307 << " s2type failed for " << sout(swords[0]) << "."
308 << std::endl;
309 return 0;
310 }
311
312 strip(ws[1]);
313 value* v = new_value(type,ws[1]);
314 if(!v) {
315 m_out << "tools::columns::finder::analyse_dcl :"
316 << " syntax error in " << sout(a_s) << "."
317 << " new_value() failed."
318 << std::endl;
319 return 0;
320 }
321 v->set_label(swords[1]);
322 return v;
323
324 }
325
326 } else if(swords.size()==1) {
327 if(m_cur_type==value::NONE) {
328 m_out << "tools::columns::finder::analyse_dcl :"
329 << " (1) current type is NONE."
330 << std::endl;
331 return 0;
332 }
333
334 strip(ws[1]);
335 value* v = new_value(m_cur_type,ws[1]);
336 if(!v) {
337 m_out << "tools::columns::finder::analyse_dcl :"
338 << " syntax error in " << sout(a_s) << "."
339 << " Bad value " << sout(ws[1]) << "."
340 << std::endl;
341 return 0;
342 }
343 v->set_label(swords[0]);
344 return v;
345
346 } else {
347 m_out << "tools::columns::finder::analyse_dcl :"
348 << " syntax error in " << sout(a_s)
349 << ". Case 1."
350 << std::endl;
351 return 0;
352 }
353
354 } else if(ws.size()==1) {
355 //<type> <name>
356 //<type> <name>={
357 std::vector<std::string> swords;
358 words(ws[0]," ",false,swords);
359 if(swords.size()==2) {
360 strip(swords[0]);
361 strip(swords[1]);
362
363 if(swords[0]=="ITuple") {
364
365 //create a value::VOID_STAR :
366 value* v = new value((void*)0);
367 v->set_label(swords[1]);
368 return v;
369
370 } else {
371 value::e_type type;
372 if(!s2type(swords[0],type)){
373 m_out << "tools::columns::finder::analyse_dcl :"
374 << " s2type failed for " << sout(swords[0]) << "."
375 << std::endl;
376 return 0;
377 }
378
379 value* v = new_value(type,"");
380 if(!v) {
381 m_out << "tools::columns::finder::analyse_dcl :"
382 << " (2) syntax error in " << sout(ws[0]) << "."
383 << " Unknown type " << sout(swords[0]) << "."
384 << std::endl;
385 return 0;
386 }
387 v->set_label(swords[1]);
388 return v;
389 }
390
391 } else if(swords.size()==1) {
392
393 if(m_cur_type==value::NONE) {
394 m_out << "tools::columns::finder::analyse_dcl :"
395 << " (1) current type is NONE."
396 << std::endl;
397 return 0;
398 }
399
400 value* v = new value();
401 v->set_type(m_cur_type);
402 v->set_label(swords[0]);
403 return v;
404
405 } else {
406 m_out << "tools::columns::finder::analyse_dcl :"
407 << " syntax error in " << sout(a_s)

414 m_out << "tools::columns::finder::analyse_dcl :"
415 << " syntax error in " << sout(a_s)
416 << ". Case 3."
417 << std::endl;
418 return 0;
