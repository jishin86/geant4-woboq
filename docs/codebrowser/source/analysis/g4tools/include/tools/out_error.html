4#ifndef tools_out_error

7#include "snpf"
8
9#include <ostream>
10
11namespace tools {
12
13inline void out_error(std::ostream& a_out,
14 const char* location, const char* fmt,...) {
15 char s[1024];
16 va_list args;
17 va_start(args,fmt);
18 vsnpf(s,sizeof(s),fmt,args);
19 va_end(args);
20 if(location) {
21 a_out << location << " : " << s << std::endl;
22 } else {
23 a_out << s << std::endl;
24 }
25}
26
27}
