18inline bool compress_buffer(std::ostream& a_out,
19 unsigned int a_level,
20 unsigned int a_srcsize,const char* a_src,
21 unsigned int a_tgtsize,char* a_tgt,
22 unsigned int& a_irep) {
23
24 z_stream stream; // decompression stream
25
26 stream.next_in = (Bytef*)(a_src);
27 stream.avail_in = (uInt)(a_srcsize);
28 stream.next_out = (Bytef*)a_tgt;
29 stream.avail_out = (uInt)(a_tgtsize);
30 stream.zalloc = (alloc_func)0;
31 stream.zfree = (free_func)0;
32 stream.opaque = (voidpf)0;
33
34 int err = deflateInit(&stream,a_level);
35 if(err!=Z_OK) {
36 a_out << "tools::compress_buffer :"
37 << " error in zlib/deflateInit." << std::endl;
38 a_irep = 0;
39 return false;
40 }
41
42 err = deflate(&stream, Z_FINISH);
43 if(err!=Z_STREAM_END) {
44 deflateEnd(&stream);
45 a_out << "tools::compress_buffer :"
46 << " error in zlib/deflate." << std::endl;
47 a_irep = 0;
48 return false;
49 }
50
51 deflateEnd(&stream);
52
53 //a_out << "tools::compress_buffer : ok "
54 // << stream.total_out << std::endl;
55
56 a_irep = stream.total_out;
57
58 return true;
59}

110inline int gunzip_get_byte(char*& a_buffer) {
111 int c = *a_buffer;a_buffer++;
112 return c;
113}

115inline int gunzip_check_header(char*& a_buffer) {
116#define TOOLS_ZLIB_HEAD_CRC 0x02 /* bit 1 set: header CRC present */
117#define TOOLS_ZLIB_EXTRA_FIELD 0x04 /* bit 2 set: extra field present */
118#define TOOLS_ZLIB_ORIG_NAME 0x08 /* bit 3 set: original file name present */
119#define TOOLS_ZLIB_COMMENT 0x10 /* bit 4 set: file comment present */
120#define TOOLS_ZLIB_RESERVED 0xE0 /* bits 5..7: reserved */
121
122 uInt len;
123 int c;
124
125 /* Check the gzip magic header */
126 for (len = 0; len < 2; len++) {
127 c = gunzip_get_byte(a_buffer);
128/*
129 if (c != gz_magic[len]) {
130 if (len != 0) s->stream.avail_in++, s->stream.next_in--;
131 if (c != EOF) {
132 s->stream.avail_in++, s->stream.next_in--;
133 s->transparent = 1;
134 }
135 s->z_err = s->stream.avail_in != 0 ? Z_OK : Z_STREAM_END;
136 return;
137 }
138*/
139 }
140 int method = gunzip_get_byte(a_buffer);
141 int flags = gunzip_get_byte(a_buffer);
142 if (method != Z_DEFLATED || (flags & TOOLS_ZLIB_RESERVED) != 0) {
143 return Z_DATA_ERROR;
144 }
145
146 /* Discard time, xflags and OS code: */
147 for (len = 0; len < 6; len++) (void)gunzip_get_byte(a_buffer);
148
149 if ((flags & TOOLS_ZLIB_EXTRA_FIELD) != 0) { /* skip the extra field */
150 len = (uInt)gunzip_get_byte(a_buffer);
151 len += ((uInt)gunzip_get_byte(a_buffer))<<8;
152 /* len is garbage if EOF but the loop below will quit anyway */
153 while (len-- != 0 && gunzip_get_byte(a_buffer) != EOF) ;
154 }
155 if ((flags & TOOLS_ZLIB_ORIG_NAME) != 0) { /* skip the original file name */
156 while ((c = gunzip_get_byte(a_buffer)) != 0 && c != EOF) ;
157 }
158 if ((flags & TOOLS_ZLIB_COMMENT) != 0) { /* skip the .gz file comment */
159 while ((c = gunzip_get_byte(a_buffer)) != 0 && c != EOF) ;
160 }
161 if ((flags & TOOLS_ZLIB_HEAD_CRC) != 0) { /* skip the header crc */
162 for (len = 0; len < 2; len++) (void)gunzip_get_byte(a_buffer);
163 }
164 //s->z_err = s->z_eof ? Z_DATA_ERROR : Z_OK;
165 return Z_OK;
166
167#undef TOOLS_ZLIB_HEAD_CRC
168#undef TOOLS_ZLIB_EXTRA_FIELD
169#undef TOOLS_ZLIB_ORIG_NAME
170#undef TOOLS_ZLIB_COMMENT
171#undef TOOLS_ZLIB_RESERVED
172}
173#endif //ZLIB_VERNUM <= 0x1140

175inline bool gunzip_buffer(std::ostream& a_out,
176 unsigned int a_srcsize,const char* a_src,
177 unsigned int a_tgtsize,char* a_tgt,
178 unsigned int& a_irep) {
179
180 z_stream stream; // decompression stream
181
182#if ZLIB_VERNUM <= 0x1140
183 char* pos = (char*)a_src;
184 if(gunzip_check_header(pos)!=Z_OK) return false;
185 stream.next_in = (Bytef*)pos;
186 stream.avail_in = (uInt)(a_srcsize-(pos-a_src));
187#else
188 stream.next_in = (Bytef*)a_src;
189 stream.avail_in = (uInt)a_srcsize;
190#endif //ZLIB_VERNUM
191
192 stream.next_out = (Bytef*)a_tgt;
193 stream.avail_out = (uInt)a_tgtsize;
194 stream.zalloc = (alloc_func)0;
195 stream.zfree = (free_func)0;
196 stream.opaque = (voidpf)0;
197
198#if ZLIB_VERNUM <= 0x1140
199 int err = inflateInit2(&stream,-MAX_WBITS);
200#else
201 int err = inflateInit2(&stream,MAX_WBITS+16);
202#endif
203 if (err != Z_OK) {
204 a_out << "tools::gunzip_buffer :"
205 << " error " << err << " in zlib/inflateInit2." << std::endl;
206 return false;
207 }
208
209 err = inflate(&stream, Z_FINISH);
210 if (err != Z_STREAM_END) {
211 inflateEnd(&stream);
212 a_out << "tools::gunzip_buffer :"
213 << " error " << err << " in zlib/inflate." << std::endl;
214 return false;
215 }
216
217 inflateEnd(&stream);
218
219 a_irep = stream.total_out;
220
221 return true;
222}
