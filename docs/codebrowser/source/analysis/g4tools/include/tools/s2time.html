13inline bool s2time(const std::string& a_string,time_t& a_time) {
14 int yy, mm, dd, hh, mi, ss;
15 if(::sscanf(a_string.c_str(),
16 "%d-%d-%d %d:%d:%d",
17 &yy,&mm,&dd,&hh,&mi,&ss)!=6) {a_time = 0;return false;}
18 struct tm tp;
19 tp.tm_year = yy-1900;
20 tp.tm_mon = mm-1;
21 tp.tm_mday = dd;
22 tp.tm_hour = hh;
23 tp.tm_min = mi;
24 tp.tm_sec = ss;
25 tp.tm_isdst = 0;
26 a_time = ::mktime(&tp);
27 return true;
28}
29
30inline bool time2s(std::string& a_s) {
31 time_t d;
32 if(::time(&d)==(time_t)-1) {a_s.clear();return false;}
33 char* _cstr = ::ctime(&d);
34 _cstr[24] = '\0';
35 a_s = _cstr;
36 return true;
37}
