207public:
208 void update_sg() {
209 // have this method public in order to use it in plotter.
210 // This is so because infos_box::height is an output field
211 // needed in plotter to place the box.
212
213 m_back_sep.clear(); //back_area::update_sg done last.
214
215 m_sep.clear();
216
217 if(width.value()<=0) return;
218 if(confine) {
219 if(height.value()<=0) return;
220 }
221
222 {bool empty = true;
223 std::vector<std::string>::const_iterator it;
224 for(it=lstrings.values().begin();it!=lstrings.values().end();++it) {
225 if((*it).size()) {empty = false;break;}
226 }
227 if(empty){
228 for(it=rstrings.values().begin();it!=rstrings.values().end();++it) {
229 if((*it).size()) {empty = false;break;}
230 }
231 }
232 if(empty) return;}
233
234 rgba* mat = new rgba();
235 mat->color = color;
236 m_sep.add(mat);
237
238 if(font==font_hershey()) {
239 draw_style* ds = new draw_style;
240 ds->style.value(draw_style::lines);
241 //ds->line_pattern = line_pattern;
242 ds->line_width = line_width;
243 m_sep.add(ds);
244 }
245
246 ////////////////////////////////////////////////////////////
247 /// left text //////////////////////////////////////////////
248 ////////////////////////////////////////////////////////////
249 base_text* ltext = 0;
250 matrix* ltsf = 0;
251
252 {separator* sep = new separator;
253 m_sep.add(sep);
254 ltsf = new matrix;
255 sep->add(ltsf);
256 if(font==font_hershey()) {
257 text_hershey* text = new text_hershey;
258 ltext = text;
259 text->encoding = encoding;
260 sep->add(text);
261 } else {
262 base_freetype* text = base_freetype::create(m_ttf);
263 //TTNODE* text = new TTNODE;
264 ltext = text;
265 text->font = font;
266 sep->add(text);
267 }
268 ltext->strings = lstrings;
269 ltext->hjust = lhjust;}
270
271 ////////////////////////////////////////////////////////////
272 /// right text /////////////////////////////////////////////
273 ////////////////////////////////////////////////////////////
274 base_text* rtext = 0;
275 matrix* rtsf = 0;
276
277 {separator* sep = new separator;
278 m_sep.add(sep);
279 rtsf = new matrix;
280 sep->add(rtsf);
281 if(font==font_hershey()) {
282 text_hershey* text = new text_hershey;
283 rtext = text;
284 text->encoding = encoding;
285 sep->add(text);
286 } else {
287 base_freetype* text = base_freetype::create(m_ttf);
288 //TTNODE* text = new TTNODE;
289 rtext = text;
290 text->font = font;
291 sep->add(text);
292 }
293 rtext->strings = rstrings;
294 rtext->hjust = rhjust;}
295
296 ////////////////////////////////////////////////////////////
297 /// middle spaces //////////////////////////////////////////
298 ////////////////////////////////////////////////////////////
299 base_text* mtext = 0;
300 {std::string s(num_spaces,' ');
301 if(font==font_hershey()) {
302 text_hershey* text = new text_hershey;
303 mtext = text;
304 text->strings.add(s);
305 text->hjust = left;
306 } else {
307 base_freetype* text = base_freetype::create(m_ttf);
308 //TTNODE* text = new TTNODE;
309 mtext = text;
310 text->strings.add(s);
311 text->hjust = left;
312 text->font = font;
313 }}
314
315 //sf<float> zfront ?
316 float zz = back_visible.value()?0.01f:0;
317
318 float fw = width * wmargin_factor;
319
320 if(confine) {
321 // left right texts may overlap.
322
323 // adjust height :
324 float fh = height * hmargin_factor;
325 float th = fh;
326 {float mn_x,mn_y,mn_z;
327 float mx_x,mx_y,mx_z;
328 ltext->get_bounds(th,mn_x,mn_y,mn_z,mx_x,mx_y,mx_z);
329 float bxh = mx_y-mn_y;
330 // adjust box height :
331 // fh -> bxh then to have fh :
332 if(!bxh) {
333 m_sep.clear();
334 parent::update_sg();
335 delete mtext;
336 return;
337 }
338 th = fh*fh/bxh;}
339
340 ltext->height = th;
341 rtext->height = th;
342
343 {float mn_x,mn_y,mn_z;
344 float mx_x,mx_y,mx_z;
345 ltext->get_bounds(th,mn_x,mn_y,mn_z,mx_x,mx_y,mx_z);
346 float lw = mx_x-mn_x;
347 float xtrans = (fw-lw)*0.5f; //left justified
348 float xx = -(mn_x+mx_x)*0.5F-xtrans;
349 float yy = -(mn_y+mx_y)*0.5F;
350 ltsf->set_translate(xx,yy,zz);}
351
352 {float mn_x,mn_y,mn_z;
353 float mx_x,mx_y,mx_z;
354 rtext->get_bounds(th,mn_x,mn_y,mn_z,mx_x,mx_y,mx_z);
355 rtext->hjust = right;
356 float xx = fw*0.5f;
357 float yy = -(mn_y+mx_y)*0.5F;
358 rtsf->set_translate(xx,yy,zz);}
359
360 parent::update_sg();
361
362 } else {
363 // height of the info box is an output of the below.
364 // The input to compute the geometry is the width of the infos
365 // box, the left/right texts and the space between
366 // left/right texts.
367
368 // from a text height of th=1, and a left/right sep
369 // of num_spaces*th, get width of the box :
370
371 float th = 1;
372
373 float lmn_x,lmn_y,lmn_z;
374 float lmx_x,lmx_y,lmx_z;
375 ltext->get_bounds(th,lmn_x,lmn_y,lmn_z,lmx_x,lmx_y,lmx_z);
376 float lw = (lmx_x<lmn_x)?0:lmx_x-lmn_x;
377
378 float rmn_x,rmn_y,rmn_z;
379 float rmx_x,rmx_y,rmx_z;
380 rtext->get_bounds(th,rmn_x,rmn_y,rmn_z,rmx_x,rmx_y,rmx_z);
381 float rw = (rmx_x<rmn_x)?0:rmx_x-rmn_x;
382
383 float mmn_x,mmn_y,mmn_z;
384 float mmx_x,mmx_y,mmx_z;
385 mtext->get_bounds(th,mmn_x,mmn_y,mmn_z,mmx_x,mmx_y,mmx_z);
386 float mw = (mmx_x<mmn_x)?0:mmx_x-mmn_x;
387
388 float winfos = lw+mw+rw;
389 if(!winfos) {
390 m_sep.clear();
391 height = 1;
392 parent::update_sg();
393 delete mtext;
394 return;
395 }
396
397 // assuming that text size is linear relative
398 // to th, we get the th needed to match infos_box.width :
399 // th -> winfos then to have fw :
400 th = fw*th/winfos;
401
402 // place the left/right texts by rescaling them to
403 // the new th :
404 ltext->height = th;
405 ltext->get_bounds(th,lmn_x,lmn_y,lmn_z,lmx_x,lmx_y,lmx_z);
406 float hinfos = (lmx_x<lmn_x)?0:lmx_y-lmn_y;
407
408 float xx = -fw*0.5F; //left justified.
409 float ty = -ltext->ascent(th)+hinfos*0.5f;
410 ltsf->set_translate(xx,ty,zz);
411
412 rtext->height = th;
413 if(rtext->hjust==right) {
414 xx = fw*0.5F;
415 } else if(rtext->hjust==left) {
416 rtext->get_bounds(th,rmn_x,rmn_y,rmn_z,rmx_x,rmx_y,rmx_z);
417 rw = (rmx_x<rmn_x)?0:rmx_x-rmn_x;
418 xx = fw*0.5F-rw;
419 } else { //center
420 rtext->get_bounds(th,rmn_x,rmn_y,rmn_z,rmx_x,rmx_y,rmx_z);
421 rw = (rmx_x<rmn_x)?0:rmx_x-rmn_x;
422 xx = fw*0.5F-rw*0.5F;
423 }
424 rtsf->set_translate(xx,ty,zz);
425
426 // set infos_box.height :
427 height = hinfos/hmargin_factor;
428
429 //done last because of the upper height setup.
430 parent::update_sg();
431
432 }
433
434 delete mtext;
435
436 }
437protected:
438 const base_freetype& m_ttf;
439
440 separator m_sep;
441};
