16class base_freetype : public base_text, public gstos {
17 TOOLS_HEADER(base_freetype,tools::sg::base_freetype,base_text)
18public:
19 sf_string font;
20 sf_enum<winding_type> front_face; //default is ccw.
21 sf_enum<font_modeling> modeling;
22 typedef unsigned int unichar;
23 typedef std::vector<unichar> uniline;
24 mf_std_vec<unichar> unitext;
25 //mf_string font_dirs; //do we want that ? It is costly. We should have a sg::font node.
26public:
27 virtual const std::vector<field_desc>& node_fields() const {
28 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::base_freetype)
29 static std::vector<field_desc> s_v;
30 if(s_v.empty()) {
31 s_v = parent::node_fields();
32 TOOLS_ADD_FIELD_DESC(font)
33 TOOLS_ADD_FIELD_DESC(front_face)
34 TOOLS_ADD_FIELD_DESC(modeling)
35 TOOLS_ADD_FIELD_DESC(unitext)
36 //TOOLS_ADD_FIELD_DESC(font_dirs)
37 }
38 return s_v;
39 }
40private:
41 void add_fields(){
42 add_field(&font);
43 add_field(&front_face);
44 add_field(&modeling);
45 add_field(&unitext);
46 //add_field(&font_dirs);
47 }
48public:
49 base_freetype()
50 :parent()
51 ,gstos()
52
53 ,font("")
54 ,front_face(winding_ccw)
55 ,modeling(font_filled) //font_outline
56 //,modeling(font_outline)
57 ,unitext()
58 //,font_dirs()
59 {
60 add_fields();
61 }
62 virtual ~base_freetype(){}
63public:
64 base_freetype(const base_freetype& a_from)
65 :parent(a_from)
66 ,gstos(a_from)
67
68 ,font(a_from.font)
69 ,front_face(a_from.front_face)
70 ,modeling(a_from.modeling)
71 ,unitext(a_from.unitext)
72 //,font_dirs(a_from.font_dirs)
73 {
74 add_fields();
75 }
76
77 base_freetype& operator=(const base_freetype& a_from){
78 parent::operator=(a_from);
79 gstos::operator=(a_from);
80 font = a_from.font;
81 front_face = a_from.front_face;
82 modeling = a_from.modeling;
83 unitext = a_from.unitext;
84 //font_dirs = a_from.font_dirs;
85 return *this;
86 }
87
88public:
89 void dump_unitext(std::ostream& a_out) {
90 //unitext.values().size()
91 a_out << "unitext size : " << unitext.values().size() << std::endl;
92 std::vector<uniline>::const_iterator vit;
93 for(vit=unitext.values().begin();vit!=unitext.values().end();++vit) {
94 const uniline& line = *vit;
95 a_out << "beg line :" << std::endl;
96 //a_out << line << std::endl;
97 std::vector<unichar>::const_iterator it;
98 for(it=line.begin();it!=line.end();++it) {
99 a_out << ((unsigned int)*it) << std::endl;
100 }
101 a_out << "end line." << std::endl;
102 }
103 }
104public:
105 static base_freetype* create(const base_freetype& a_ttf) {
106 return safe_cast<node,base_freetype>(*(a_ttf.copy()));
107 }
