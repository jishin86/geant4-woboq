256 pick_t* closest_pick() {
257 if(m_picks.empty()) return 0;
258 //closest point is minimum z !
259 // near -> -1
260 // far -> 1
261
262 // find first pick_t with not empty zs :
263 pick_t* pck = 0;
264 float z;
265 std::vector<pick_t>::const_iterator it;
266 for(it=m_picks.begin();it!=m_picks.end();++it) {
267 if(minimum((*it).zs(),z)) {
268 pck = (pick_t*)&(*it);
269 break;
270 }
271 }
272 if(!pck) return 0;
273 it++;
274 for(;it!=m_picks.end();++it) {
275 float zi;
276 if(minimum((*it).zs(),zi)) {
277 if(zi<=z) {
278 pck = (pick_t*)&(*it);
279 z = zi;
280 }
281 }
282 }
283 return pck;
284 }

514public: //for markers
515 bool is_inside(float a_x,float a_y,float a_z,float a_w) {
516 //std::cout << "debug : tools::sg::pick_action::is_inside :"
517 // << " x " << a_x
518 // << " y " << a_y
519 // << std::endl;
520
521 // In principle we should receive (because of proj x model matrix
522 // mult of world coord points) point in [-1,1]x[-1,1].
523
524 float x,y;
525 to_pick_ndc(a_x,a_y,x,y);
526
527 if(x<-1) return false;
528 if(1<x) return false;
529 if(y<-1) return false;
530 if(1<y) return false;
531
532 //std::cout << "debug : tools::sg::pick_action::is_inside :"
533 // << " inside !"
534 // << std::endl;
535
536 m_zs.push_back(a_z);
537 m_ws.push_back(a_w);
538
539 return true;
540 }
541 bool intersect_line(float a_bx,float a_by,float a_bz,float a_bw,
542 float a_ex,float a_ey,float a_ez,float a_ew) {
543
544 // a_bz, a_ez are used only to compute the intersection point.
545
546 float bx,by;
547 to_pick_ndc(a_bx,a_by,bx,by);
548 float ex,ey;
549 to_pick_ndc(a_ex,a_ey,ex,ey);
550
551 //no check on z is done.
552 float bz = a_bz;
553 float bw = a_bw;
554 float ez = a_ez;
555 float ew = a_ew;
556
557 bool toggle;
558 if(!ortho_clip_line(bx,by,bz,bw, ex,ey,ez,ew, toggle)) return false;
559
560 m_zs.push_back(bz);
561 m_ws.push_back(bw);
562
563 return true;
564 }

567 bool intersect_triangle(float a_1x,float a_1y,float a_1z,float a_1w,
568 float a_2x,float a_2y,float a_2z,float a_2w,
569 float a_3x,float a_3y,float a_3z,float a_3w) {
570 //test a triangle.
571
572 if(is_inside(a_1x,a_1y,a_1z,a_1w)) return true;
573 if(is_inside(a_2x,a_2y,a_2z,a_2w)) return true;
574 if(is_inside(a_3x,a_3y,a_3z,a_3w)) return true;
575
576 // alll points are outside.
577
578 float x1,y1;
579 to_pick_ndc(a_1x,a_1y,x1,y1);
580 float x2,y2;
581 to_pick_ndc(a_2x,a_2y,x2,y2);
582
583 bool toggle;
584 float bx,by,bz,bw,ex,ey,ez,ew;
585
586 bx = x1;
587 by = y1;
588 bz = a_1z;
589 bw = a_1w;
590
591 ex = x2;
592 ey = y2;
593 ez = a_2z;
594 ew = a_2w;
595 if(ortho_clip_line(bx,by,bz,bw, ex,ey,ez,ew, toggle)) {
596 m_zs.push_back(bz);
597 m_ws.push_back(bw);
598 return true;
599 }
600
601 float x3,y3;
602 to_pick_ndc(a_3x,a_3y,x3,y3);
603
604 bx = x2;
605 by = y2;
606 bz = a_2z;
607 bw = a_2w;
608
609 ex = x3;
610 ey = y3;
611 ez = a_3z;
612 ew = a_3w;
613 if(ortho_clip_line(bx,by,bz,bw, ex,ey,ez,ew, toggle)) {
614 m_zs.push_back(bz);
615 m_ws.push_back(bw);
616 return true;
617 }
618
619 bx = x1;
620 by = y1;
621 bz = a_1z;
622 bw = a_1w;
623
624 ex = x3;
625 ey = y3;
626 ez = a_3z;
627 ew = a_3w;
628 if(ortho_clip_line(bx,by,bz,bw, ex,ey,ez,ew, toggle)) {
629 m_zs.push_back(bz);
630 m_ws.push_back(bw);
631 return true;
632 }
633
634 // no intersection with edges.
635 // but the triangle may surround [-1,1]x[-1,1] !
636
637 // test if O=(0,0) is inside the triangle :
638
639 float xo = 0;
640 float yo = 0;
641 //float zo = 0;
642
643 {float cp2 = (x2-x1)*(y3-y1)-(y2-y1)*(x3-x1); //(2-1).cross(3-1)
644 if(cp2==0) return false; // (1,2,3) aligned points.
645 float cp1 = (x2-x1)*(yo-y1)-(y2-y1)*(xo-x1);
646 if(cp1==0) { // O on (1,2). We can't pass here.
647 //m_out << "pick_action::intersect_triangle : case 0." << std::endl;
648 //m_zs.push_back(???);
649 //m_ws.push_back(???);
650 return false;
651 }
652 if((cp1*cp2)<0) return false;} // O 3 not on same side than (1,2)
653
654 {float cp2 = (x3-x2)*(y1-y2)-(y3-y2)*(x1-x2); //(3-2).cross(1-2)
655 if(cp2==0) return false; // (1,2,3) aligned points.
656 float cp1 = (x3-x2)*(yo-y2)-(y3-y2)*(xo-x2);
657 if(cp1==0) { // O on (2,3). We can't pass here.
658 //m_out << "pick_action::intersect_triangle : case 1." << std::endl;
659 //m_zs.push_back(???);
660 //m_ws.push_back(???);
661 return false;
662 }
663 if((cp1*cp2)<0) return false;} // O 1 not on same side than (2,3)
664
665 {float cp2 = (x1-x3)*(y2-y3)-(y1-y3)*(x2-x3); //(1-3).cross(2-3)
666 if(cp2==0) return false; // (1,2,3) aligned points.
667 float cp1 = (x1-x3)*(yo-y3)-(y1-y3)*(xo-x3);
668 if(cp1==0) { // O on (3,1). We can't pass here.
669 //m_out << "pick_action::intersect_triangle : case 2." << std::endl;
