18class base_colormap {
19public:
20 virtual void get_color(float,colorf&) const = 0;
21 virtual void* cast(const std::string&) const = 0;
22public:
23 base_colormap(){}
24 base_colormap(const base_colormap& aFrom)
25 :m_values(aFrom.m_values),m_colors(aFrom.m_colors){}
26 base_colormap& operator=(const base_colormap& aFrom){
27 m_values = aFrom.m_values;
28 m_colors = aFrom.m_colors;
29 return *this;
30 }
31 virtual ~base_colormap(){}
32public:
33 size_t colorn() const {return m_colors.size();}
34 size_t valn() const {return m_values.size();}
35 colorf color(size_t a_index) const {
36 size_t n = m_colors.size();
37 if(a_index>=n) return colorf(0.5F,0.5F,0.5F);
38 return m_colors[a_index];
39 }
40 float value(size_t a_index) const {
41 size_t n = m_values.size();
42 if(a_index>=n) return 0;
43 return m_values[a_index];
44 }
45public:
46 // helper :
47 void set_colors(void(*aGet)(float,colorf&),size_t a_ncell) {
48 m_colors.clear();
49 m_colors.resize(a_ncell);
50 if(!a_ncell) return;
51 float d = 1.0F/(a_ncell-1);
52 for(size_t index=0;index<a_ncell;index++) {
53 aGet(d*index,m_colors[index]);
54 }
55 }
56
57 void set_PAW_coloring() {
58 size_t _valn = m_values.size();
59 if(_valn==1) {
60 m_values[0] = take_log(m_values[0]);
61 } else if(_valn>=2) {
62 //CERN/PAW coloring :
63 if(m_values[0]==0) m_values[0] = 10e-5F;
64 float vmin = take_log(m_values[0]);
65 float vmax = take_log(m_values[_valn-1]);
66 float dv = (vmax-vmin)/(_valn-1);
67 for(size_t count=0;count<_valn;count++) {
68 m_values[count] = vmin + dv * count;
69 }
70 }
71 }
72
73protected:
74 static float take_log(float aVal){
75 if(aVal<=0) {
76 return -FLT_MAX;
77 } else {
78 return flog10(aVal);
79 }
80 }
81
82protected:
83 std::vector<float> m_values;
84 std::vector<colorf> m_colors;
85};

111 static void set_by_value(std::ostream& a_out,
112 const sg::cmaps_t& a_cmaps,
113 const std::string& aString,
114 std::vector<float>& aValues,
115 std::vector<colorf>& aColors) {
116 // The given string is of the format :
117 // [<color name> <value>] <color name>
118 // or :
119 // [<value> <color name>] <value>
120 // For example :
121 // black 10 cyan 50 green 100 orange 200 blue 300 pink 400 red
122 std::vector<std::string> ws;
123 words(aString," ",false,ws);
124 size_t wordn = ws.size();
125 size_t number = wordn/2;
126 if(number<=0) {
127 aValues.clear();
128 aColors.clear();
129 } else if((2*number+1)!=wordn) {
130 a_out << "by_value_colormap::set_by_value :"
131 << " An odd number (" << wordn
132 << " given) of words is expected in " << sout(aString) << "."
133 << std::endl;
134 aValues.clear();
135 aColors.clear();
136 } else {
137 // look if :
138 // <col> <num> <col> ... <num> <col>
139 // or :
140 // <num> <col> <num> ... <col> <num>
141 // FIXME : case of <col> being three floats ?
142 colorf c;
143 if(sg::find_color(a_cmaps,ws[0],c)) {
144 // <col> <num> <col> ... <num> <col>
145 aValues.resize(number);
146 aColors.resize(number+1);
147 for(size_t count=0;count<number;count++) {
148 {const std::string& word = ws[2*count];
149 if(!sg::find_color(a_cmaps,word,aColors[count])) {
150 a_out << "by_value_colormap::set_by_value :"
151 << " in " << sout(aString)
152 << ", " << word << " not a color."
153 << std::endl;
154 aValues.clear();
155 aColors.clear();
156 return;
157 }}
158 {const std::string& word = ws[2*count+1];
159 if(!to(word,aValues[count])) {
160 a_out << "by_value_colormap::set_by_value :"
161 << " in " << sout(aString)
162 << ", " << word << " not a number."
163 << std::endl;
164 aValues.clear();
165 aColors.clear();
166 return;
167 }}
168 }
169 const std::string& word = ws[wordn-1];
170 if(!sg::find_color(a_cmaps,word,aColors[number])) {
171 a_out << "by_value_colormap::set_by_value :"
172 << " in " << sout(aString)
173 << ", " << word << " not a color."
174 << std::endl;
175 aValues.clear();
176 aColors.clear();
177 }
178 } else {
179 // <num> <col> <num> ... <col> <num>
180 aValues.resize(number+1);
181 aColors.resize(number);
182 for(size_t count=0;count<number;count++) {
183 {const std::string& word = ws[2*count];
184 if(!to(word,aValues[count])) {
185 a_out << "by_value_colormap::set_by_value :"
186 << " in " << sout(aString)
187 << ", " << word << " not a number."
188 << std::endl;
189 aValues.clear();
190 aColors.clear();
191 return;
192 }}
193 {const std::string& word = ws[2*count+1];
194 if(!sg::find_color(a_cmaps,word,aColors[count])) {
195 a_out << "by_value_colormap::set_by_value :"
196 << " in " << sout(aString)
197 << ", " << word << " not a color."
198 << std::endl;
199 aValues.clear();
200 aColors.clear();
201 return;
202 }}
203 }
204 {const std::string& word = ws[wordn-1];
205 if(!to(word,aValues[number])) {
206 a_out << "by_value_colormap::set_by_value :"
207 << " in " << sout(aString)
208 << ", " << word << " not a number."
209 << std::endl;
210 aValues.clear();
211 aColors.clear();
212 return;
213 }}
214 }
215 }

218 static void get_by_value(float aValue,
219 const std::vector<float>& aValues,
220 const std::vector<colorf>& aColors,
221 colorf& a_col){
222 // There are aValuen (n) entries in aValues and (n+1) aColors
223 // aColors[0] aValues[0] aColors[1] aValues[1]...
224 // aValues[n-2] aColors[n-1] aValues[n-1] aColors[n]
225 // black 10 cyan 50 green 100 orange 200 blue 300 pink 400 red
226 // valuen = 6
227 // values[0] 10
228 // values[1] 50
229 // values[2] 100
230 // values[3] 200
231 // values[4] 300
232 // values[5] 400
233 //
234 // colors[0] black
235 // colors[1] cyan
236 // colors[2] green
237 // colors[3] orange
238 // colors[4] blue
239 // colors[5] pink
240 // colors[6] red
241 size_t _valn = aValues.size();
242 if(!_valn) {a_col = colorf_black();return;}
243 if(aColors.size()==(_valn+1)) {
244 // col0 val0 col1 val1 col2 val2 col3
245 if(aValue<aValues[0]) {
246 a_col = aColors[0];
247 } else {
248 for(int count=0;count<=int(_valn-2);count++) {
249 if( (aValues[count]<=aValue) && (aValue<aValues[count+1]) ) {
250 a_col = aColors[count+1];
251 return;
252 }
253 }
254 a_col = aColors[_valn];
255 }
256 } else if((aColors.size()+1)==_valn) {
257 // val0 col0 val1 col1 val2 col2 val3
258 for(int count=0;count<=int(_valn-2);count++) {
259 if( (aValues[count]<=aValue) && (aValue<aValues[count+1]) ) {
260 a_col = aColors[count];
261 return;
262 }
263 }
264 if(aValue<aValues[0]) {a_col = aColors[0];return;}
265 if(aValue>=aValues[_valn-1]) {a_col = aColors[aColors.size()-1];return;}
266 a_col = colorf_black();
267 } else {
268 a_col = colorf_black();
269 }
270 }

341 TOOLS_SCLASS(tools::sg::violet_to_red_colormap)

351 violet_to_red_colormap(float a_min,float a_max,size_t a_ncell){
352 set(a_min,a_max,a_ncell);
353 }
354 violet_to_red_colormap(const violet_to_red_colormap& aFrom):base_colormap(aFrom){}
355 violet_to_red_colormap& operator=(const violet_to_red_colormap& aFrom){
356 base_colormap::operator=(aFrom);
357 return *this;
358 }
359 virtual ~violet_to_red_colormap(){}
360public:
361 void set(float a_min,float a_max,size_t a_ncell){
362 m_values.resize(2);
363 m_values[0] = a_min;
364 m_values[1] = a_max;
365 set_colors(get_violet_to_red,a_ncell);
366 }
367protected:
368 static void get_violet_to_red(float a_ratio,colorf& a_col){
369 if(a_ratio<0.0F) a_ratio = 0;
370 if(a_ratio>1.0F) a_ratio = 1;
371 // a_ratio in [0,1]
372 // From ROOT pretty palette.
373 // Initialize with the spectrum Violet->Red
374 // The color model used here is based on the HLS model which
375 // is much more suitable for creating palettes than RGB.
376 // Fixing the saturation and lightness we can scan through the
377 // spectrum of visible light by using "hue" alone.
378 // In Root hue takes values from 0 to 360.
379 float saturation = 1;
380 float lightness = 0.5;
381 float hue_mn = 0;
382 float hue_mx = 280;
383 float hue = hue_mx - a_ratio * (hue_mx-hue_mn);
384 float r,g,b;
385 hls_to_rgb(hue,lightness,saturation,r,g,b);
386 a_col.set_value(r,g,b,1);
387 }
