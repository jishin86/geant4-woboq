19class back_area : public node {
20 TOOLS_NODE(back_area,tools::sg::back_area,node)
21public:
22 static unsigned int corner_top_right() {return 1<<0;}
23 static unsigned int corner_top_left() {return 1<<1;}
24 static unsigned int corner_bottom_right() {return 1<<2;}
25 static unsigned int corner_bottom_left() {return 1<<3;}
26 static unsigned int corner_all() {
27 return corner_top_right()|corner_top_left()|corner_bottom_right()|corner_bottom_left();
28 }
29public:
30 sf<float> width;
31 sf<float> height;
32
33 sf_vec<colorf,float> color;
34
35 sf<bool> gradient;
36 sf_vec<colorf,float> color_top;
37
38 sf<bool> border_visible;
39 sf_vec<colorf,float> border_color;
40 sf<float> border_line_width;
41
42 sf<float> shadow; //in percent of width.
43
44 sf<float> corner_radius; //percent of height.
45 sf<unsigned int> corner_steps;
46 sf<unsigned int> corner_mask;
47public:
48 virtual const std::vector<field_desc>& node_fields() const {
49 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::back_area)

56 TOOLS_ADD_FIELD_DESC(gradient)

58 TOOLS_ADD_FIELD_DESC(border_visible)

67 }
68private:

86public:
87 virtual void render(render_action& a_action) {
88 if(touched()) {
89 update_sg();
90 reset_touched();

92 m_back_sep.render(a_action);
93 }
94 virtual void search(search_action& a_action) {
95 node::search(a_action);
96 if(a_action.done()) return;
97 if(touched()) {
98 update_sg();
99 reset_touched();
100 }
101 m_back_sep.search(a_action);
102 if(a_action.done()) return;
103 }
104 virtual void pick(pick_action& a_action) {
105 if(touched()) {
106 update_sg();
107 reset_touched();
108 }
109 m_back_sep.pick(a_action);
110 }
111public:
112 back_area()
113 :parent()
114 ,width(1)
115 ,height(1)
116 ,color(colorf_white())
117 ,gradient(false)
118 ,color_top(colorf_white())
119 ,border_visible(true)
120 ,border_color(colorf_black())
121 ,border_line_width(1)
122 ,shadow(0)
123 ,corner_radius(0) //in percent of the height.
124 ,corner_steps(12)
125 ,corner_mask(corner_all())
126 {
127 add_fields();
128 }
129 virtual ~back_area(){}
130public:
131 back_area(const back_area& a_from)
132 :parent(a_from)
133 ,width(a_from.width)

142 ,corner_radius(a_from.corner_radius)
143 ,corner_steps(a_from.corner_steps)
144 ,corner_mask(a_from.corner_mask)
145 {
146 add_fields();
147 }
148 back_area& operator=(const back_area& a_from){
149 parent::operator=(a_from);
150 width = a_from.width;
151 height = a_from.height;
152 color = a_from.color;
153 gradient = a_from.gradient;
154 color_top = a_from.color_top;
155 border_visible = a_from.border_visible;
156 border_color = a_from.border_color;
157 border_line_width = a_from.border_line_width;
158 shadow = a_from.shadow;
159 corner_radius = a_from.corner_radius;
160 corner_steps = a_from.corner_steps;
161 corner_mask = a_from.corner_mask;
162 return *this;
163 }
164protected:
165 void update_sg() {
166 m_back_sep.clear();
167
168 if(width.value()<=0) return;
169 if(height.value()<=0) return;
170
171 float xb = -width*0.5f;
172 float xe = width*0.5f;
173 float yb = -height*0.5f;
174
175 float zshadow = -0.05f;
176 float zback = 0;
177 float zborder = 0.01f;
178
179 if(shadow.value()) {
180 float zz = zshadow;
181 float ye = height*0.5f;
182
183 sg::rgba* mat = new sg::rgba();
184 mat->color = colorf_black();
185 m_back_sep.add(mat);
186
187 normal* nm = new normal;
188 //nm->vec.value(); //default is z. ok.
189 m_back_sep.add(nm);
190
191 vertices* vtxs = new vertices;
192 vtxs->mode = gl::triangle_fan();
193 m_back_sep.add(vtxs);
194
195 float dx = width*shadow;
196 float dy = -dx;
197 vtxs->add(xb+dx,yb+dy,zz);
198 vtxs->add(xe+dx,yb+dy,zz);
199 vtxs->add(xe+dx,ye+dy,zz);
200 vtxs->add(xb+dx,ye+dy,zz);
201 }
202
203 {//background :
204 normal* nm = new normal;
205 //nm->vec.value(); //default is z. ok.
206 m_back_sep.add(nm);
207
208 if(gradient.value()) {
209// if(true) {
210 //color gradient from (bottom,color) to (top,color_top)
211
212 atb_vertices* vtxs = new atb_vertices;
213 vtxs->mode = gl::triangle_strip();
214 m_back_sep.add(vtxs);
215
216 float zz = zback;
217
218 unsigned int ncol = 50;
219 float dy = height/ncol;
220 float ye = yb+dy;
221
222 colorf col_beg = color.value();
223 colorf col_end = color_top.value();
224
225 float dr = (col_end.r()-col_beg.r())/ncol;
226 float dg = (col_end.g()-col_beg.g())/ncol;
227 float db = (col_end.b()-col_beg.b())/ncol;
228 float da = (col_end.a()-col_beg.a())/ncol;
229 vec4f dcol(dr,dg,db,da);
230
231 colorf col = col_beg;
232
233 vtxs->add(xb,yb,zz);
234 vtxs->add_color(col);
235
236 vtxs->add(xe,yb,zz);
237 vtxs->add_color(col);
238
239 for(unsigned int index=0;index<ncol;index++) {
240 vtxs->add(xb,ye,zz);
241 vtxs->add(xe,ye,zz);
242
243 vtxs->add_color(col);
244 vtxs->add_color(col);
245
246 ye += dy;
247 col += dcol;
248 }
249
250 } else {
251
252 float zz = zback;
253 float ye = height*0.5f;
254
255 sg::rgba* mat = new sg::rgba();
256 mat->color = color;
257 m_back_sep.add(mat);
258
259 vertices* vtxs = new vertices;
260 vtxs->mode = gl::triangle_fan();
261 m_back_sep.add(vtxs);
262
263 float r = height*corner_radius;
264 if((r>(0.5f*height.value()))||(r>(0.5f*width.value()))) r = 0;
265
266 if((r>0) && corner_steps.value()) {
267 float dangle = fhalf_pi()/float(corner_steps);
268 unsigned int nslice = corner_steps;
269
270 vtxs->add(0,0,zz);
271 vtxs->add(xe,yb+r,zz);
272 vtxs->add(xe,ye-r,zz);
273
274 // top-right :
275 float angle = dangle;
276 if(corner_mask.value() & corner_top_right()) {
277 float xc = xe-r;
278 float yc = ye-r;
279 for(unsigned int i=0;i<nslice;i++,angle+=dangle) {
280 vtxs->add(r*fcos(angle)+xc,r*fsin(angle)+yc,zz);
281 }
282 } else {
283 angle += fhalf_pi();
284 vtxs->add(xe,ye,zz);
285 vtxs->add(xe-r,ye,zz);
286 }
287
288 vtxs->add(xb+r,ye,zz);
289
290 // top-left :
291 if(corner_mask.value() & corner_top_left()) {
292 float xc = xb+r;
293 float yc = ye-r;
294 for(unsigned int i=0;i<nslice;i++,angle+=dangle) {
295 vtxs->add(r*fcos(angle)+xc,r*fsin(angle)+yc,zz);
296 }
297 } else {
298 angle += fhalf_pi();
299 vtxs->add(xb,ye,zz);
300 vtxs->add(xb,ye-r,zz);
301 }
302
303 vtxs->add(xb,yb+r,zz);
304
305 // bottom-left :
306 if(corner_mask.value() & corner_bottom_left()) {
307 float xc = xb+r;
308 float yc = yb+r;
309 for(unsigned int i=0;i<nslice;i++,angle+=dangle) {
310 vtxs->add(r*fcos(angle)+xc,r*fsin(angle)+yc,zz);
311 }
312 } else {
313 angle += fhalf_pi();
314 vtxs->add(xb,yb,zz);
315 vtxs->add(xb+r,yb,zz);
316 }
317
318 vtxs->add(xe-r,yb,zz);
319
320 // bottom-right :
321 if(corner_mask.value() & corner_bottom_right()) {
322 float xc = xe-r;
323 float yc = yb+r;
324 for(unsigned int i=0;i<nslice;i++,angle+=dangle) {
325 vtxs->add(r*fcos(angle)+xc,r*fsin(angle)+yc,zz);
326 }
327 } else {
328 angle += fhalf_pi();
329 vtxs->add(xe,yb,zz);
330 vtxs->add(xe,yb+r,zz);
331 }
332
333 } else {
334 vtxs->add(xb,yb,zz);
335 vtxs->add(xe,yb,zz);
336 vtxs->add(xe,ye,zz);
337 vtxs->add(xb,ye,zz);
338 }
339
340 }}
341
342 if(border_visible.value()){
343 float zz = zborder;
344 float ye = height*0.5f;
345
346 sg::rgba* mat = new sg::rgba();
347 mat->color = border_color;
348 m_back_sep.add(mat);
349
350 draw_style* ds = new draw_style;
351 ds->style = draw_style::lines;
352 ds->line_width = border_line_width;
353 m_back_sep.add(ds);
354
355 vertices* vtxs = new vertices;
356 vtxs->mode = gl::line_strip();
357 m_back_sep.add(vtxs);
358
359 float r = height*corner_radius;
360 if((r>(0.5f*height.value()))||(r>(0.5f*width.value()))) r = 0;
361
362 if((r>0) && corner_steps.value()) {
363 float dangle = fhalf_pi()/float(corner_steps);
364 unsigned int nslice = corner_steps;
365
366 vtxs->add(xe,yb+r,zz);
367 vtxs->add(xe,ye-r,zz);
368
369 // top-right :
370 float angle = dangle;
371 if(corner_mask.value() & corner_top_right()) {
372 float xc = xe-r;
373 float yc = ye-r;
374 for(unsigned int i=0;i<nslice;i++,angle+=dangle) {
375 vtxs->add(r*fcos(angle)+xc,r*fsin(angle)+yc,zz);
376 }
377 } else {
378 angle += fhalf_pi();
379 vtxs->add(xe,ye,zz);
380 vtxs->add(xe-r,ye,zz);
381 }
382
383 vtxs->add(xb+r,ye,zz);
384
385 // top-left :
386 if(corner_mask.value() & corner_top_left()) {
387 float xc = xb+r;
388 float yc = ye-r;
389 for(unsigned int i=0;i<nslice;i++,angle+=dangle) {
390 vtxs->add(r*fcos(angle)+xc,r*fsin(angle)+yc,zz);
391 }
392 } else {
393 angle += fhalf_pi();
394 vtxs->add(xb,ye,zz);
395 vtxs->add(xb,ye-r,zz);
396 }
397
398 vtxs->add(xb,yb+r,zz);
399
400 // bottom-left :
401 if(corner_mask.value() & corner_bottom_left()) {
402 float xc = xb+r;
403 float yc = yb+r;
404 for(unsigned int i=0;i<nslice;i++,angle+=dangle) {
405 vtxs->add(r*fcos(angle)+xc,r*fsin(angle)+yc,zz);
406 }
407 } else {
408 angle += fhalf_pi();
409 vtxs->add(xb,yb,zz);
410 vtxs->add(xb+r,yb,zz);
411 }
412
413 vtxs->add(xe-r,yb,zz);
414
415 // bottom-right :
416 if(corner_mask.value() & corner_bottom_right()) {
417 float xc = xe-r;
418 float yc = yb+r;
419 for(unsigned int i=0;i<nslice;i++,angle+=dangle) {
420 vtxs->add(r*fcos(angle)+xc,r*fsin(angle)+yc,zz);
421 }
422 } else {
423 angle += fhalf_pi();
424 vtxs->add(xe,yb,zz);
425 vtxs->add(xe,yb+r,zz);
426 }
427
428 } else {
429 vtxs->add(xb,yb,zz);
430 vtxs->add(xe,yb,zz);
431 vtxs->add(xe,ye,zz);
432 vtxs->add(xb,ye,zz);
433 vtxs->add(xb,yb,zz);
434 }
435 }
436
437 }
438protected:
439 separator m_back_sep;
440};
