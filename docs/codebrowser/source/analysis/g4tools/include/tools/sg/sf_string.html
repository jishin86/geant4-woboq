17public:
18 virtual bool write(io::iwbuf& a_buffer) {
19 return a_buffer.write_cstr(m_value.c_str());
20 }
21 virtual bool read(io::irbuf& a_buffer) {
22 char* cstr = 0;
23 if(!a_buffer.read_cstr(cstr)) return false;
24 m_value = cstr;
25 str_del(cstr);
26 return true;
27 }

31 }

40 return *this;
