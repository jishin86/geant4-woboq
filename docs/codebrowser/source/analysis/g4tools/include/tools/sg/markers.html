507 virtual void pick(pick_action& a_action) {
508 size_t num = xyzs.size()/3;
509 if(!num) return;
510
511 state& state = a_action.state();
512
513 const std::vector<float>& _xyzs = xyzs.values();
514
515 std::vector<float> pts;
516
517 float sx = size.value()/float(state.m_ww); //in [-1,1]
518 float hsx = sx*0.5f;
519
520 float sy = size.value()/float(state.m_wh); //in [-1,1]
521 float hsy = sy*0.5f;
522
523 float x,y,z,w;
524
525 std::vector<float>::const_iterator it;
526 for(it=_xyzs.begin();it!=_xyzs.end();){
527 x = *it;it++;
528 y = *it;it++;
529 z = *it;it++;
530
531 a_action.project_point(x,y,z,w); //in [-1,1][-1,1]
532
533 _add(pts, x-hsx,y-hsy,z); //in [-1,1][-1,1]
534 _add(pts, x+hsx,y+hsy,z);
535 _add(pts, x+hsx,y-hsy,z);
536 _add(pts, x-hsx,y+hsy,z);
537 }
538
539 a_action.set_matrices_to_identity();
540 a_action.add__lines(*this,pts);
541 a_action.set_matrices_from_state();
542 }
