307 void update_sg(std::ostream& a_out) {
308 //a_out << "debug : tools::axis::update_sg :" << std::endl;
309
310 if(width<=0) {
311 m_line_sep.clear();
312 m_ticks_sep.clear();
313 m_labels_sep.clear();
314 m_mag_sep.clear();
315 m_title_sep.clear();
316 return;
317 }
318
319 // line scene graph :
320 m_line_sep.clear();
321 if(m_line_style.visible) {
322 rgba* mat = new rgba();
323 mat->color = m_line_style.color;
324 m_line_sep.add(mat);
325
326 draw_style* ds = new draw_style;
327 ds->style = draw_style::lines;
328 ds->line_pattern = m_line_style.pattern;
329 ds->line_width = m_line_style.width;
330 m_line_sep.add(ds);
331
332 vertices* vtxs = new vertices;
333 vtxs->mode = gl::line_strip();
334 vtxs->add(0,0,0);
335 vtxs->add(width,0,0);
336 m_line_sep.add(vtxs);
337 }
338
339 // ticks scene graph :
340 if(modeling==tick_modeling_none()) {
341 } else if(modeling==tick_modeling_hplot()) {
342 compute_ticks_HPLOT(a_out);
343 } else {
344 compute_ticks_hippo();
345 }
346
347 m_ticks_sep.clear();
348 if(m_ticks_style.visible) {
349
350 vertices* vtxs = new vertices;
351 vtxs->mode = gl::lines();
352
353 if(modeling==tick_modeling_hplot()) {
354
355 size_t num = m_sub_ticks.size()/4;
356 size_t pos = 0;
357 for(size_t index=0;index<num;index++) {
358 float bx = m_sub_ticks[pos];pos++;
359 float by = m_sub_ticks[pos];pos++;
360 float ex = m_sub_ticks[pos];pos++;
361 float ey = m_sub_ticks[pos];pos++;
362 if(tick_up) {
363 vtxs->add(bx,by,0);
364 vtxs->add(ex,ey,0);
365 } else {
366 vtxs->add(bx,-by,0);
367 vtxs->add(ex,-ey,0);
368 }
369 }
370
371 } else {
372
373 float yy = tick_up ? tick_length.value():-tick_length.value();
374 for(unsigned int index=0;index<tick_number;index++) {
375 float xx = coords.values()[index];
376 vtxs->add(xx,0,0);
377 vtxs->add(xx,yy,0);
378 }
379 }
380
381 if(vtxs->number()) {
382
383 rgba* mat = new rgba();
384 mat->color = m_ticks_style.color;
385 m_ticks_sep.add(mat);
386
387 draw_style* ds = new draw_style;
388 ds->style = draw_style::lines;
389 ds->line_pattern = m_ticks_style.pattern;
390 ds->line_width = m_ticks_style.width;
391 m_ticks_sep.add(ds);
392
393 m_ticks_sep.add(vtxs);
394 } else {
395 delete vtxs;
396 }
397
398 }
399
400 // labels scene graph :
401 m_labels_sep.clear();
402 if(m_labels_style.visible && tick_number ) {
403 rgba* mat = new rgba();
404 mat->color = m_labels_style.color;
405 m_labels_sep.add(mat);
406
407 float text_size = label_height*m_labels_style.scale;
408 std::string font = m_labels_style.font.value();
409
410 /*FIXME : adjust text_size to be sure that the labels do not overlap.
411 {int number = tick_number;
412 if(number>=2) {
413 float min_delta_tick = coords[number-1]-coords[0];
414 float xx_prev = coords[0];
415 for(unsigned int index=1;index<number;index++) {
416 float xx = coords[index];
417 min_delta_tick = SbMinimum(min_delta_tick,xx-xx_prev);
418 xx_prev = xx;
419 }
420 text_size = SbMinimum(text_size,min_delta_tick);
421 }}
422 */
423
424 if(font==font_hershey()) {
425 draw_style* ds = new draw_style;
426 ds->style = draw_style::lines;
427 ds->line_pattern = m_labels_style.line_pattern;
428 ds->line_width = m_labels_style.line_width;
429 m_labels_sep.add(ds);
430 } else {
431 m_labels_sep.add(new normal);
432 }
433
434 vec3f X = m_labels_style.x_orientation.value();
435 vec3f Y = m_labels_style.y_orientation.value();
436 X.normalize();
437 Y.normalize();
438 vec3f Z;X.cross(Y,Z);
439 Z.cross(X,Y);
440 mat4f scale_rot(X.v0(),Y.v0(),Z.v0(),0, //first row
441 X.v1(),Y.v1(),Z.v1(),0,
442 X.v2(),Y.v2(),Z.v2(),0,
443 0,0,0,1);
444 scale_rot.mul_scale(text_size,text_size,1);
445
446 bool bin_center = (m_labels_style.options.value()=="center"?true:false); //gopaw.
447
448 vec3f vec;float xx;
449 {unsigned int number = tick_number;
450 for(unsigned int index=0;index<number;index++) {
451
452 if(bin_center) { //label at the center of the bin :
453 if(index==(number-1)) continue;
454 xx = 0.5f*(coords.values()[index]+coords.values()[index+1]);
455
456 } else { // label on tick :
457 xx = coords.values()[index];
458 }
459
460 vec.set_value(xx,-label_to_axis,0);
461 vec += m_labels_style.translation.value();
462
463 separator* sep = new separator;
464 m_labels_sep.add(sep);
465
466 add_string_opt(*sep,
467 font,
468 m_labels_style.encoding.value(),
469 m_labels_style.smoothing,
470 labels.values()[index],
471 vec[0],vec[1],vec[2],
472 scale_rot,
473 m_labels_style.hjust,
474 m_labels_style.vjust,
475 m_ttf);
476 }}
477 }
478
479 m_mag_sep.clear();
480 if( magnitude.value() && m_mag_style.visible) {
481 rgba* mat = new rgba();
482 mat->color = m_mag_style.color;
483 m_mag_sep.add(mat);
484
485 char string[64];
486 if(magnitude>=0)
487 snpf(string,sizeof(string),"x10+%d",magnitude.value());
488 else
489 snpf(string,sizeof(string),"x10-%d",::abs(magnitude));
490
491 vec3f vec(width*1.03f,0,0);
492 vec += m_mag_style.translation.value();
493
494 float text_size = label_height*0.8f * m_mag_style.scale;
495 std::string font = m_mag_style.font.value();
496
497 if(font==font_hershey()) {
498 draw_style* ds = new draw_style;
499 ds->style = draw_style::lines;
500 ds->line_pattern = m_mag_style.line_pattern;
501 ds->line_width = m_mag_style.line_width;
502 m_mag_sep.add(ds);
503 }
504
505 add_string(m_mag_sep,
506 font,
507 m_mag_style.encoding.value(),
508 m_mag_style.smoothing,
509 string,
510 vec[0],vec[1],vec[2],
511 m_mag_style.x_orientation.value(),
512 m_mag_style.y_orientation.value(),
513 text_size,
514 m_mag_style.hjust,
515 m_mag_style.vjust,
516 m_ttf);
517 }
518
519 // title scene graph :if(update_title) {
520 m_title_sep.clear();
521 if(title.value().size() && m_title_style.visible) {
522 rgba* mat = new rgba();
523 mat->color = m_title_style.color;
524 m_title_sep.add(mat);
525
526 float text_size = title_height*m_title_style.scale;
527 std::string font = m_title_style.font.value();
528
529 if(font==font_hershey()) {
530 draw_style* ds = new draw_style;
531 ds->style = draw_style::lines;
532 ds->line_pattern = m_title_style.line_pattern;
533 ds->line_width = m_title_style.line_width;
534 m_title_sep.add(ds);
535 } else {
536 m_title_sep.add(new normal);
537 }
538
539 float xx = 0; //left
540 if(title_hjust==center) {
541 xx = width/2;
542 } else if(title_hjust==right) {
543 xx = width;
544 }
545
546 vec3f vec(xx,-title_to_axis,0);
547 vec += m_title_style.translation.value();
548
549 //std::cout << "debug : axis : update_title :"
550 // << " pos : " << vec[0] << " " << vec[1] << " " << vec[2]
551 // << std::endl;
552
553 add_string(m_title_sep,
554 font,
555 m_title_style.encoding.value(),
556 m_title_style.smoothing,
557 title.value(),
558 vec[0],vec[1],vec[2],
559 m_title_style.x_orientation.value(),
560 m_title_style.y_orientation.value(),
561 text_size,
562 m_title_style.hjust,
563 m_title_style.vjust,
564 m_ttf);
565 }
566
567 }

569public: //style
570 void reset_style(bool a_geom = false) {
571 //reset fields that are considered as part of the style.
572
573 ////////////////////////////////////////////
574 // we do not touch :
575 ////////////////////////////////////////////
576 //width
577 //minimum_value
578 //maximum_value
579 //labels_enforced
580 //tick_number
581 //magnitude
582 //time_labels
583 //time_format
584 //time_offset
585 //time_offset_is_GMT
586 //labels
587 //values
588 //coords
589 //sub_coords
590
591 ////////////////////////////////////////////
592 divisions = 510;
593 modeling = tick_modeling_hippo();
594 tick_up = true;
595 is_log = false;
596 title.value().clear();
597
598 if(a_geom) {
599 ////////////////////////////////////////////
600 // Take PAW default :
601 float YSIZ = 20; //page height
602 float YMGL = 2; //low y margin (to data frame).
603 float YMGU = 2; //up y margin (to data frame).
604 float VSIZ = 0.28F; //tick label character size.
605 float YVAL = 0.4F; //y distance of x tick label to data frame.
606 float XTIC = 0.3F; //y length of X axis ticks.
607 float YLAB = 0.8F; //y distance of x title to data frame.
608 float ASIZ = 0.28F; // axis title (label) character size.
609
610 float hData = YSIZ-YMGL-YMGU;
611
612 // To map data space to width :
613 float to1 = width/hData;
614
615 float vsiz = VSIZ * to1; //0.0175F
616 float yval = YVAL * to1; //0.025F
617 float xtic = XTIC * to1; //0.01875F
618 float ylab = YLAB * to1; //0.05F
619 float asiz = ASIZ * to1; //0.0175F
620
621 //sf
622 tick_length = xtic;
623 label_to_axis = yval;
624 label_height = vsiz;
625
626 // The axis title is the PAW axis label.
627 // It is right justified at the end of axis
628 // (at end right for XY_X, at end top for XY_Y)
629 title_to_axis = ylab;
630 title_height = asiz;
631 }
632
633 title_hjust = right;
634
635 ////////////////////////////////////////////
636 // setup styles :
637 m_line_style = line_style();
638 m_ticks_style = line_style();
639 m_labels_style = text_style();
640 m_mag_style = text_style();
641 m_title_style = text_style();
642
643 m_line_style.color = colorf_black();
644 m_ticks_style.color = colorf_black();
645
646 m_labels_style.color = colorf_black();
647 m_labels_style.font = font_hershey();
648 m_labels_style.encoding = encoding_PAW();
649
650 m_mag_style.color = colorf_black();
651 m_mag_style.font = font_hershey();
652 m_mag_style.encoding = encoding_PAW();
653
654 m_title_style.color = colorf_black();
655 m_title_style.font = font_hershey();
656 m_title_style.encoding = encoding_PAW();
657 }

659 typedef std::pair<std::string,std::string> style_item_t;
660 typedef std::vector<style_item_t> style_t;
661 bool set_from_style(std::ostream& a_out,const style_t& a_style) {
662 style_t::const_iterator it;
663 for(it=a_style.begin();it!=a_style.end();++it) {
664 const std::string& key = (*it).first;
665 const std::string& sv = (*it).second;
666 //::printf("debug : axis::set_from_style : key \"%s\" \"%s\"\n",key.c_str(),sv.c_str());
667 //if(key=="reset") {}
668
669 // not part of style :
670 //width
671 //minimum_value
672 //maximum_value
673 //labels_enforced
674 //tick_number
675 //magnitude
676 //title
677 //time_labels
678 //time_format
679 //time_offset
680 //time_offset_is_GMT
681 //labels
682 //values
683 //coords
684 //sub_coords
685
686 if(key=="divisions") {
687 unsigned int v;
688 if(!to(sv,v)) {style_failed(a_out,key,sv);return false;}
689 divisions = v;
690 } else if(key=="modeling") {
691 modeling = sv;
692 } else if(key=="is_log") {
693 bool v;
694 if(!to(sv,v)) {style_failed(a_out,key,sv);return false;}
695 is_log = v;
696
697 } else if(key=="tick_up") {
698 bool v;
699 if(!to(sv,v)) {style_failed(a_out,key,sv);return false;}
700 tick_up = v;
701 } else if(key=="tick_length") {
702 float v;
703 if(!to(sv,v)) {style_failed(a_out,key,sv);return false;}
704 tick_length = v;
705
706 } else if(key=="title") {
707 title = sv;
708 } else if(key=="title_to_axis") {
709 float v;
710 if(!to(sv,v)) {style_failed(a_out,key,sv);return false;}
711 title_to_axis = v;

752 void compute_ticks_hippo() {
753 // input fields :
754 // minimum_value
755 // maximum_value
756 // is_log
757 // output fields :
758 // values
759 // coords
760 // sub_coords
761 // labels (if labels_enforced is false)
762 // tick_number
763
764
765 float mn = minimum_value;
766 float mx = maximum_value;
767
768 bool a_is_log = is_log;
769 if(a_is_log) {
770 if((mn<=0) || (mx<=0) ) a_is_log = false;
771 }
772
773 float magxxx,y,yr,startTick,tickSize;
774 float pmag;
775 char pstr[10] = "";
776 char tmp[10];
777
778 unsigned int tick_num = 0;
779 std::vector<float> tick_values;
780 std::vector<std::string> tick_labels;
781
782 // need include <float.h> which does not exist on some system
783 //NUM_FUZZ DBL_EPSILON*4
784 float NUM_FUZZ = 0.01f;
785
786 if (mn >= mx) {
787 if(tick_number) {
788 tick_number.value(0);
789 values.clear();
790 coords.clear();
791 sub_coords.clear();
792 labels.clear();
793 }
794 if(magnitude.value()) magnitude.value(0);
795 m_sub_ticks.clear();
796 return;
797 }
798
799 if (!a_is_log) {
800
801 tickSize = calculate_ticks_hippo(mx-mn,magxxx);
802 startTick = fceil( mn / tickSize) * tickSize;
803
804 if (ffabs(magxxx) <= 3)
805 pmag = 0.0;
806 else
807 pmag = startTick != 0.0 ? ffloor(flog10(ffabs(startTick))) : magxxx;
808
809 snpf(pstr,sizeof(pstr),"%%1.%df",(int)max_of<float>((pmag-magxxx),0.0));
810
811 y = startTick;
812 while (y <= mx*(1.0+NUM_FUZZ)) {
813
814 yr = ffloor(y/fpow(10,magxxx) + 0.5F);
815
816 snpf(tmp,sizeof(tmp),pstr,yr*fpow(10,magxxx-pmag));
817
818 {float val = yr * fpow(10.0,magxxx);
819 if((val>=mn)&&(val<=mx)) { //G.Barrand : add this test.
820 tick_values.push_back(val);
821 tick_labels.push_back(tmp);
822 tick_num++;
823 }}
824
825 y += tickSize;
826 }
827 if (ffabs(magxxx) <= 3.0) magxxx = 0.0;
828
829 } else {
830
831 if (mn <= 0) {
832 if(tick_number) {
833 tick_number.value(0);
834 values.clear();
835 coords.clear();
836 sub_coords.clear();
837 labels.clear();
838 }
839 if(magnitude.value()) magnitude.value(0);
840 m_sub_ticks.clear();
841 return;
842 }
843
844 int nLogTicks;
845 float logTicks[5];
846 float magStep;
847
848 float maghigh = fceil(flog10(mx));
849 float maglow = ffloor(flog10(mn));
850 float magrng = maghigh - maglow;
851
852 if (magrng <=3) {
853 nLogTicks = 3;
854 logTicks[0] = 1.0;
855 logTicks[1] = 2.0;
856 logTicks[2] = 5.0;
857 magStep = 1.0;
858 } else {
859 nLogTicks = 1;
860 logTicks[0] = 1.0;
861 magStep = magrng <= 7 ? 1.0F : 2.0F;
862 }
863
864 pmag = (nLogTicks == 3 && (ffabs(maglow)>3 || ffabs(maghigh)>3)) ?
865 maglow : 0;
866
867 magxxx = maglow;
868 int i = 0;
869 while ((y=logTicks[i]*fpow(10,magxxx)) < mx*(1+NUM_FUZZ)) {
870 if (y >= mn) {
871
872 // be careful: there is a bug in the NeXT (s)printf
873 // routine when you do, eg. printf("%1.0g",0.01);
874 if ((magxxx-pmag) > 4 || (magxxx-pmag) < -3) {
875 ::strcpy(pstr,"%1.0e");
876 } else {
877 snpf(pstr,sizeof(pstr),
878 "%%1.%df",(int)((magxxx-pmag)>0?0.:-(magxxx-pmag)));
879 }
880 snpf(tmp,sizeof(tmp),pstr,y*fpow(10.0,-pmag));
881
882 {float val = flog10(y);
883 if((val>=flog10(mn))&&(val<=flog10(mx))) { //G.Barrand : add this if
884 tick_values.push_back(val);
885 tick_labels.push_back(tmp);
886 tick_num++;
887 }}
888
889 }
890
891 i++;
892 if (i>=nLogTicks) {
893 i = 0;
894 magxxx += magStep;
895 }
896 }
897
898 mn = flog10(mn);
899 mx = flog10(mx);
900 }
901
902 float range = mx - mn;
903
904 // it is assumes that tick_values are ordered min to max.
905 tick_number.value(tick_num);
906 values.clear();
907 coords.clear();
908 for(unsigned int index=0;index<tick_num;index++) {
909 float val = tick_values[index];
910 float coord = width * (val-mn)/range;
911 values.add(val);
912 coords.add(coord);
913 }
914
915 if(labels_enforced) {
916 size_t n = labels.size();
917 if(tick_num>n) {
918 for(size_t index=n;index<tick_num;index++) labels.add("");
919 }
920 } else {
921 labels = tick_labels;
922 }
923
924 magnitude.value((int)pmag);
925
926 sub_coords.clear();
927 m_sub_ticks.clear();
928 }

930 static float calculate_ticks_hippo(float aSize,float& a_mag) {
931 unsigned int MIN_TICKS = 4;
932
933 if (aSize <= 0.0) {
934 //printf ("CalculateTicks : bad value \n");
935 aSize = ffabs(aSize);
936 if (aSize == 0.0) aSize = 1.0;
937 }
938
939 a_mag = ffloor(flog10(aSize));
940 if (aSize/fpow(10.0,a_mag) < MIN_TICKS) (a_mag)--;
941
942 // now fit the max number of ticks into this range
943
944 float tickSize;
945 int tickIndex;
946 static float goodTicks[] = {10.0, 5.0, 4.0, 2.0, 1.0};
947 for(tickIndex = 0;
948 aSize/(tickSize=goodTicks[tickIndex]*fpow(10.0,a_mag))<MIN_TICKS;
949 tickIndex++){}
950
951 if (tickIndex == 0) a_mag++;
952
953 return tickSize;
954 }
