7#include "matrix_action"

10#include "../lina/box3f"

12namespace tools {
13namespace sg {
14
15class bbox_action : public matrix_action,public primitive_visitor {
16protected:
17 virtual bool project(float& a_x,float& a_y,float& a_z,float& a_w) {
18 model_point(a_x,a_y,a_z,a_w);
19 return true;
20 }
21
22 virtual bool add_point(float a_x,float a_y,float a_z,float) {
23 m_box.extend_by(a_x,a_y,a_z);
24 return true;
25 }
26
27 virtual bool add_point(float a_x,float a_y,float a_z,float,
28 float,float,float,float) {
29 m_box.extend_by(a_x,a_y,a_z);
30 return true;
31 }
32
33 virtual bool add_line(float a_bx,float a_by,float a_bz,float,
34 float a_ex,float a_ey,float a_ez,float) {
35 m_box.extend_by(a_bx,a_by,a_bz);
36 m_box.extend_by(a_ex,a_ey,a_ez);
37 return true;
38 }
39
40 virtual bool add_line(float a_bx,float a_by,float a_bz,float a_bw,
41 float,float,float,float,
42 float a_ex,float a_ey,float a_ez,float a_ew,
43 float,float,float,float) {
44 return add_line(a_bx,a_by,a_bz,a_bw,
45 a_ex,a_ey,a_ez,a_ew);
46 }
47
48 virtual bool add_triangle(float a_p1x,float a_p1y,float a_p1z,float,
49 float a_p2x,float a_p2y,float a_p2z,float,
50 float a_p3x,float a_p3y,float a_p3z,float){
51 m_box.extend_by(a_p1x,a_p1y,a_p1z);
52 m_box.extend_by(a_p2x,a_p2y,a_p2z);
53 m_box.extend_by(a_p3x,a_p3y,a_p3z);
54 return true;
55 }
56
57 virtual bool add_triangle(float a_p1x,float a_p1y,float a_p1z,float a_p1w,
58 float,float,float,float,
59 float a_p2x,float a_p2y,float a_p2z,float a_p2w,
60 float,float,float,float,
61 float a_p3x,float a_p3y,float a_p3z,float a_p3w,
62 float,float,float,float){
63 return bbox_action::add_triangle(a_p1x,a_p1y,a_p1z,a_p1w,
64 a_p2x,a_p2y,a_p2z,a_p2w,
65 a_p3x,a_p3y,a_p3z,a_p3w);
66 }
67
68public:
69 bbox_action(std::ostream& a_out)
70 :matrix_action(a_out,0,0)
71 {}
72 virtual ~bbox_action(){}
73public:
74 bbox_action(const bbox_action& a_from)
75 :matrix_action(a_from)
76 ,primitive_visitor(a_from)
77 {}
78 bbox_action& operator=(const bbox_action& a_from){
79 if(&a_from==this) return *this;
80 matrix_action::operator=(a_from);
81 primitive_visitor::operator=(a_from);
82 m_box.make_empty();
83 return *this;
84 }
85public:
86 void reset() {
87 matrix_action::reset();
88 m_box.make_empty();
89 }
90 const box3f& box() const {return m_box;}
91 box3f& box() {return m_box;}
92protected:
93 box3f m_box;
94};
95
96}}
