65 virtual bool write(write_action& a_action) {
66 if(!a_action.beg_node(*this)) return false;
67 if(!write_fields(a_action)) return false;
68 if(!write_children(a_action)) return false;
69 if(!a_action.end_node(*this)) return false;
70 return true;

87 node::operator=(a_from);
88 if(&a_from==this) return *this;

93public:
94 void add(node* a_node) {
95 //WARNING : take ownership of a_node.

99 //WARNING : take ownership of a_node.
100 m_children.insert(m_children.begin(),a_node);

102 void set(unsigned int a_index,node* a_node) {
103 //WARNING : take ownership of a_node.
104 //WARNING : no check is done on a_index.
105 m_children[a_index] = a_node;

107
108 bool replace(const node* a_from,node* a_to,bool a_del){
109 tools_vforit(node*,m_children,it) {
110 if((*it)==a_from) {
111 node* old = *it;
112 (*it) = a_to;
113 if(a_del) delete old;
114 return true;
115 }
116 }
117 return false;
118 }
119
120 void swap(unsigned int a_1,unsigned int a_2){
121 // WARNING : no check is done on a_1,a_2.
122 node* tmp = m_children[a_1];
123 m_children[a_1] = m_children[a_2];
124 m_children[a_2] = tmp;
125 }

152 void* rsearch_from(const node* a_node,
153 const std::string& a_class,
154 bool a_inc_a_node = true) const {
155 bool found = false;
156 tools_vforcrit(node*,m_children,it) {
157 // the below logic permits to test a_node.
158 if(!found) {
159 if(*it==a_node) {
160 found = true;
161 if(!a_inc_a_node) continue; //skip a_node
162 }
163 }
164 if(found) {
165 void* p = (*it)->cast(a_class);
166 if(p) return p;
167 }
168 }
169 return 0;
170 }

172 bool remove(const node* a_node){
173 //NOTE : no delete on a_node is performed.
174 tools_vforit(node*,m_children,it) {
175 if(a_node==(*it)) {
176 m_children.erase(it);
177 return true;
178 }
179 }
180 return false;
181 }
182
183 bool remove_index(unsigned int a_index){
184 //NOTE : no delete on node at a_index is performed.
185 std::vector<node*>::iterator it = m_children.begin();
186 it += a_index;
187 if(it>=m_children.end()) return false;
188 m_children.erase(it);
189 return true;
190 }
191
192 bool delete_from(const node* a_node,bool a_inc_a_node = true){
193 bool found = false;
194 std::vector<node*>::iterator it;
195 for(it=m_children.begin();it!=m_children.end();) {
196 if(!found) {
197 if(*it==a_node) {
198 found = true;
199 if(!a_inc_a_node) {it++;continue;} //skip a_node
200 }
201 }
202 if(found) {
203 node* old = *it;
204 it = m_children.erase(it);
205 delete old;
206 } else {
207 it++;
208 }
209 }
210 return found;
211 }

213 void transfer(group& a_from) {
214 if(&a_from==this) return;
215 clear();
216 m_children.resize(a_from.size());
217 std::vector<node*>::iterator it = m_children.begin();
218 std::vector<node*>::iterator fit = a_from.m_children.begin();
219 for(;fit!=a_from.m_children.end();++it,++fit) {
220 *it = *fit;
221 *fit = 0;
222 }
223 a_from.m_children.clear();
224 }

246 bool insert(unsigned int a_index,node* a_new){ //iv2sg
247 if(a_index==m_children.size()) {
248 m_children.push_back(a_new);
249 return true;
250 }
251 unsigned int index = 0;
252 std::vector<node*>::iterator it;
253 for(it=m_children.begin();it!=m_children.end();++it,index++) {
254 if(index==a_index) {
255 m_children.insert(it,a_new);
256 return true;
257 }
258 }
259 return false;

281 bool position(const node* a_node,unsigned int& a_index) const {
282 a_index = 0;

297 if(a_index>=m_children.size()) return 0;
