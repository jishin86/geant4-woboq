30 TOOLS_SCLASS(tools::sg::node)

32 if(void* p = cmp_cast<node>(this,a_class)) {return p;}
33 else return 0;

35 virtual const std::string& s_cls() const = 0;
36public:

39 virtual unsigned int cls_version() const {return 1;}

46 virtual void render(render_action&) {}
47 virtual void pick(pick_action&) {}
48 virtual void bbox(bbox_action&) {}
49 virtual void search(search_action& a_action) {
50 if(a_action.what()==search_action::search_node_of_class) {
51 if(void* p = cast(a_action.sclass())) {
52 a_action.add_obj(p);
53 if(a_action.stop_at_first()) a_action.set_done(true);
54 }
55 } else if(a_action.what()==search_action::search_path_to_node) {
56 if(this==a_action.node()){
57 a_action.path_push(this); //ending node in the path.
58 a_action.set_done(true);
59 }
60 } else if(a_action.what()==search_action::search_path_to_node_of_class) {
61 if(cast(a_action.sclass())) {
62 search_action::path_t path = a_action.path();
63 path.push_back(this);
64 a_action.add_path(path);
65 if(a_action.stop_at_first()) a_action.set_done(true);
66 }
67 }
68 }
69 virtual void get_matrix(get_matrix_action& a_action) {
70 if(this==a_action.node()){
71 a_action.set_found_model(a_action.model_matrix());
72 a_action.set_done(true);
73 }
74 }
75 virtual bool write(write_action& a_action) {
76 if(!a_action.beg_node(*this)) return false;
77 if(!write_fields(a_action)) return false;
78 if(!a_action.end_node(*this)) return false;
79 return true;
80 }
81 virtual void event(event_action&) {}
82 virtual bool read(read_action& a_action) {return read_fields(a_action);}
83 virtual void is_visible(visible_action&) {}

85 virtual void protocol_one_fields(std::vector<field*>& a_fields) const {a_fields = m_fields;}

105 void add_field(field* a_field) {
106 m_fields.push_back(a_field); //it does not take ownerhship.
107 }
108 bool touched() {
109 tools_vforcit(field*,m_fields,it) {
110 if((*it)->touched()) return true;
111 }
112 return false;
113 }
114 //TOOLS_CLASS_STRING(fields_end) //for write_bsg, sg::rbsg
115 bool write_fields(write_action& a_action) {
116
117 check_fields(a_action.out()); //costly.
118 //dump_field_descs(a_action.out());
119
120 unsigned int index = 0;
121 tools_vforcit(field*,m_fields,it) {
122 if(!(*it)->write(a_action.buffer())) {
123 a_action.out() << "node::write_fields :"
124 << " for field index " << index
125 << " and field class " << (*it)->s_cls()
126 << " of node class " << s_cls()
127 << " : field.write() failed" << "."
128 << std::endl;
129 return false;
130 }
131 index++;
132 }
133 return true;
134 }
135
136 bool read_fields(read_action& a_action) { //used in protocol-2.
137 node_desc rndesc;
138 if(!a_action.get_node_desc(s_cls(),rndesc)) {
139 a_action.out() << "tools::node::read_fields :"
140 << " for node class " << s_cls()
141 << " : read_action.get_node_desc() failed."
142 << std::endl;
143 return false;
144 }
145 //Whatever the current node fields, we must read all rndesc.fields() :
146 tools_vforcit(field_desc,rndesc.fields(),it) {
147 const field_desc& fdesc = *it;
148
149 field* fd = find_field(fdesc);
150 if(!fd) {
151 a_action.out() << "tools::node::read_fields :"
152 << " for node class " << s_cls()
153 << " : field desc name " << fdesc.name()
154 << " : field desc class " << fdesc.cls()
155 << " : field desc offset " << fdesc.offset()
156 << " : field not found."
157 << "."
158 << std::endl;
159//#define TOOLS_NODE_DEBUG_READ_FIELD
160#ifdef TOOLS_NODE_DEBUG_READ_FIELD
161 check_fields(a_action.out()); //costly.
162 dump_field_descs(a_action.out());
163 {a_action.out() << "read field descs of node class " << s_cls() << " :" << std::endl;
164 tools_vforcit(field_desc,rndesc.fields(),itr) {
165 a_action.out() << "name " << (*itr).name()
166 << ", class " << (*itr).cls()
167 << ", offset " << (*itr).offset()
168 << std::endl;
169 }}
170 {a_action.out() << "m_fields of node class " << s_cls() << " :" << std::endl;
171 tools_vforcit(field*,m_fields,itm) {
172 a_action.out() << "field class " << (*itm)->s_cls()
173 << ", found offset " << field_offset(*itm)
174 << std::endl;
175 }}
176 ::exit(0);
177#endif
178 fd = a_action.field_factory().create(fdesc.cls());
179 if(!fd) {
180 a_action.out() << "tools::node::read_fields :"
181 << " for node class " << s_cls()
182 << " : field desc class " << fdesc.cls()
183 << " : can't create generic field."
184 << "."
185 << std::endl;
186 return false;
187 }
188 } /*else {
189 a_action.out() << "debug : inlib::node::read_fields :"
190 << " for node class " << s_cls()
191 << " : field desc class " << fdesc.cls()
192 << " : field desc offset " << fdesc.offset()
193 << " : field found."
194 << "."
195 << std::endl;
196 }*/
197
198 if(!fd->read(a_action.buffer())) {
199 a_action.out() << "tools::node::read_fields :"
200 << " for node class " << s_cls()
201 << " : and field class " << fd->s_cls()
202 << " : field read() failed."
203 << std::endl;
204 return false;
205 }
206 //fd->dump(a_action.out());
207 }
208
209 //NOTE : if some current node fields had not been found
210 // in rndesc.fields(), they catch the default value
211 // of the node fields.
212
213 return true;
214 }

245 field* find_field(const field_desc& a_rdesc) const {
246/*
247 //WARNING : must not use field name since, through inheritance, a node can have two field with same name.
248 // (For exa sg::text has one field "color" and another one coming from back_area).
249 //::printf("debug : find_field : %s ...\n",a_rdesc.name().c_str());
250 tools_vforcit(field*,m_fields,it) {
251 //::printf("debug : find_field : look : %s\n",field_name(*(*it)).c_str());
252 if(field_offset(*it)==a_rdesc.offset()) return (*it);
253 }
254*/
255 const std::vector<field_desc>& fds = node_fields();
256 tools_vforcit(field_desc,fds,it) {
257 if((*it).name()==a_rdesc.name()) {
258 tools_vforcit(field*,m_fields,itf) {
259 //::printf("debug : find_field : look : %s\n",field_name(*(*it)).c_str());
260 if(field_offset(*itf)==(*it).offset()) return (*itf);
261 }
262 }
263 }
264 return 0;
265 }

267 void dump_field_descs(std::ostream& a_out) const {
268 a_out << "field descs of node class " << s_cls() << " :" << std::endl;
269 const std::vector<field_desc>& fds = node_fields();
270 tools_vforcit(field_desc,fds,itd) {
271 a_out << "name " << (*itd).name()
272 << ", class " << (*itd).cls()
273 << ", offset " << (*itd).offset()
274 << std::endl;
275 }
276 }
277
278 void check_fields(std::ostream& a_out) const {
279 const std::vector<field_desc>& fds = node_fields();
280 tools_vforcit(field*,m_fields,it) {
281 bool found = false;
282 tools_vforcit(field_desc,fds,itd) {
283 if( ((*itd).offset()==field_offset(*it)) &&
284 ((*itd).cls()==(*it)->s_cls())
285 ){
286 found = true;
287 break;
288 }
289 }
290 if(!found) {
291 a_out << "tools::sg::node::check_fields :"
292 << " WARNING : node of class " << s_cls()
293 << " has bad fields description."
294 << std::endl;
295 }
296 }
297 }
298private:
299 std::vector<field*> m_fields;
300};
