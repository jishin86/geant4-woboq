4#ifndef tools_sg_holder
5#define tools_sg_holder
6
7// class to manage an object.
8// Only one holder owns the object. In case
9// of holder copy(), the new holder receives a null object.
10
11#include "node"
12#include "sf_string"
13
14namespace tools {
15namespace sg {
16
17template <class T>
18class holder : public node {
19 typedef node parent;
20public:
21 //WARNING : we do not put the T class name in the name of this class.
22 // We put it in the class_name field.
23 // We do that because of the bsg file format in order to be able
24 // to read back the object without having the T class around.
25 TOOLS_T_SCLASS(T,tools::sg::holder)
26public:
27 virtual void* cast(const std::string& a_class) const {
28 if(void* p = cmp_cast< holder<T> >(this,a_class)) {return p;}
29 return parent::cast(a_class);
30 }
31 virtual node* copy() const {return new holder(*this);}
32 virtual const std::string& s_cls() const {return s_class();}
33public:
34 sf_string class_name;
35public:
36 virtual const std::vector<field_desc>& node_fields() const {
37 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::holder)
38 static std::vector<field_desc> s_v;
39 if(s_v.empty()) {
40 s_v = parent::node_fields();
41 TOOLS_ADD_FIELD_DESC(class_name)
42 }
43 return s_v;
44 }
45private:
46 void add_fields(){
47 add_field(&class_name);
48 }
49public:
50 holder(T* a_obj = 0,const std::string& a_name = "")
51 :parent()
52 ,class_name(T::s_class())
53 ,m_obj(a_obj) //it takes a_obj ownership.
54 ,m_name(a_name)
55 {
56 add_fields();
57 }
58
59 virtual ~holder(){delete m_obj;}
60public:
61 //any copy of a holder does NOT own any object (and we empty the name).
62 holder(const holder& a_from)
63 :parent(a_from)
64 ,class_name(std::string())
65 ,m_obj(0)
66 ,m_name()
67 {
68 add_fields();
69 }
70 holder& operator=(const holder& a_from){
71 parent::operator=(a_from);
72 class_name.value().clear();
73 m_obj = 0;
74 m_name.clear();
75 return *this;
76 }
77public:
78 const T* object() const {return m_obj;}
79 T* object() {return m_obj;}
80 const std::string& name() const {return m_name;}
81protected:
82 T* m_obj;
83 std::string m_name;
84};

87template <class T>
88inline holder<T>* cast_holder(node& a_node) {
89 typedef holder<T> h_t;
90 h_t* _h = (h_t*)a_node.cast(h_t::s_class());
91 if(!_h) return 0;
92 return (_h->class_name.value()==T::s_class()?_h:0);
93}
94
95template <class T>
96inline void remove_holders(std::vector<node*>& a_vec){
97 typedef holder<T> h_t;
98
99 std::vector<node*>::iterator it;
100 for(it=a_vec.begin();it!=a_vec.end();) {
101 if(h_t* h = cast_holder<T>(*(*it))){
102 it = a_vec.erase(it);
103 delete h;
104 continue;
105 }
106
107 ++it;
108 }
109}
110
111template <class T>
112inline void remove_holders(std::vector<node*>& a_vec,const std::string& a_name){
113 typedef holder<T> h_t;
114
115 std::vector<node*>::iterator it;
116 for(it=a_vec.begin();it!=a_vec.end();) {
117 if(h_t* h = cast_holder<T>(*(*it))){
118 if(h->name()==a_name) {
119 it = a_vec.erase(it);
120 delete h;
121 continue;
122 }
123 }
124
125 ++it;
126 }
127}
128
129template <class T>
130inline T* find_holder(std::vector<node*>& a_vec,const std::string& a_name) {
131 //return the first named found.
132
133 typedef holder<T> h_t;
134 std::vector<node*>::iterator it;
135 for(it=a_vec.begin();it!=a_vec.end();++it) {
136 if(h_t* h = cast_holder<T>(*(*it))){
137 if(h->name()==a_name) return h->object();
138 }
139 }
140 return 0;
141}
142
143template <class T>
144inline T* find_first_holder(std::vector<node*>& a_vec){
145 //return the first T found.
146
147 typedef holder<T> h_t;
148 std::vector<node*>::iterator it;
149 for(it=a_vec.begin();it!=a_vec.end();++it) {
150 if(h_t* h = cast_holder<T>(*(*it))) return h->object();
151 }
152
153
154 return 0;
155}
