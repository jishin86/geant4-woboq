4#ifndef tools_sg_manager_zb
5#define tools_sg_manager_zb
6
7#include "gl_manager"
8
9#include <map>
10
11namespace tools {
12namespace sg {
13
14class manager_zb : public virtual gl_manager {
15public:
16 virtual bool begin_render(int,int,unsigned int,unsigned int,float,float,float,float,bool = true) {return true;}
17 virtual void end_render() {}
18
19 virtual unsigned int create_texture(const img_byte& a_img) {
20 m_gen_id++; //never return 0.
21 m_gstos[m_gen_id] = a_img;
22 return m_gen_id;
23 }
24
25 virtual unsigned int create_gsto_from_data(size_t,const float*) {return 0;}
26
27 virtual bool is_gsto_id_valid(unsigned int a_id) const {
28 gstos_t::const_iterator it = m_gstos.find(a_id);
29 if(it==m_gstos.end()) return false;
30 return true;
31 }
32 virtual void delete_gsto(unsigned int a_id) {
33 gstos_t::iterator it = m_gstos.find(a_id);
34 if(it!=m_gstos.end()) m_gstos.erase(it);
35 }
36
37 //
38 virtual gsto_mode get_gsto_mode() const {return gsto_memory;}
39 virtual void set_gsto_mode(gsto_mode) {}
40 virtual void available_gsto_modes(std::vector<std::string>& a_v) {a_v.clear();}
41 virtual void available_not_memory_gsto_mode(std::string& a_s) const {a_s.clear();}
42 virtual size_t used_texture_memory() const {return 0;}
43 virtual size_t gstos_size() const {return 0;}
44public:
45 manager_zb():m_gen_id(0){}
46 virtual ~manager_zb(){
47 m_gstos.clear();
48 }
49public:
50 manager_zb(const manager_zb& a_from)
51 :gl_manager(a_from)
52 ,m_gen_id(0)
53 ,m_gstos()
54 {}
55 manager_zb& operator=(const manager_zb& a_from){
56 if(&a_from==this) return *this;
57 m_gen_id = 0;
58 m_gstos.clear();
59 return *this;
60 }
61public:
62 bool find(unsigned int a_id,img_byte& a_img) {
63 gstos_t::iterator it = m_gstos.find(a_id);
64 if(it==m_gstos.end()) return false;
65 a_img = (*it).second;
66 return true;
67 }
68 //void cleanup() {}
69 void delete_gstos() {
70 m_gstos.clear();
71 }
72protected:
73 unsigned int m_gen_id;
74 typedef std::map<unsigned int,img_byte> gstos_t;
75 gstos_t m_gstos; //only textures for the moment.
76};
77
78}}
