4#ifndef tools_sg_gstos
5#define tools_sg_gstos
6
7#include "gl_manager"
8#include <vector>
9#include <ostream>
10
11//#define TOOLS_SG_GSTOS_DEBUG
12
13namespace tools {
14namespace sg {
15
16class gstos {
17protected:
18 gstos() {}
19 virtual ~gstos() {clean_gstos();}
20protected:
21 gstos(const gstos&) {}
22 gstos& operator=(const gstos& a_from) {
23 if(&a_from==this) return *this;
24 clean_gstos();
25 return *this;
26 }
27protected:
28 unsigned int get_tex_id(std::ostream& a_out,gl_manager& a_mgr,const img_byte& a_img) {
29 unsigned int _id = _find(&a_mgr);
30 if(_id && !a_mgr.is_gsto_id_valid(_id)){
31#ifdef TOOLS_SG_GSTOS_DEBUG
32 a_out << "debug : tools::sg::gstos::get_tex_id : " << _id << " not valid." << std::endl;
33#endif
34 clean_gstos(&a_mgr);
35 _id = 0;
36 }
37 if(!_id) {
38#ifdef TOOLS_SG_GSTOS_DEBUG
39 a_out << "debug : tools::sg::gstos::get_tex_id : create_texture ... " << std::endl;
40#endif
41 _id = a_mgr.create_texture(a_img);
42#ifdef TOOLS_SG_GSTOS_DEBUG
43 a_out << "debug : tools::sg::gstos::get_tex_id : create_texture : _id " << _id << "." << std::endl;
44#endif
45 if(!_id) {
46 a_out << "tools::sg::gstos::get_tex_id :"
47 << " gl_manager.create_texture() failed."
48 << std::endl;
49 } else {
50 m_gstos.push_back(std::pair<unsigned int,gl_manager*>(_id,&a_mgr));
51 }
52 }
53 return _id;
54 }
55protected:
56 virtual unsigned int create_gsto(std::ostream&,gl_manager&) {return 0;}
57
58 unsigned int get_gsto_id(std::ostream& a_out,gl_manager& a_mgr){
59 unsigned int _id = _find(&a_mgr);
60 if(_id && !a_mgr.is_gsto_id_valid(_id)){
61#ifdef TOOLS_SG_GSTOS_DEBUG
62 a_out << "debug : tools::sg::gstos::get_gsto_id : " << _id << " not valid." << std::endl;
63#endif
64 clean_gstos(&a_mgr);
65 _id = 0;
66 }
67 if(!_id) {
68#ifdef TOOLS_SG_GSTOS_DEBUG
69 a_out << "debug : tools::sg::gstos::get_gsto_id : create_gsto ..." << std::endl;
70#endif
71 _id = create_gsto(a_out,a_mgr);
72#ifdef TOOLS_SG_GSTOS_DEBUG
73 a_out << "debug : tools::sg::gstos::get_gsto_id : create_gsto : _id " << _id << "." << std::endl;
74#endif
75 if(!_id) {
76 // could be ok if there is no graphical data to load.
77 //a_out << "tools::sg::gstos::get_gsto_id :"
78 // << " create_gsto() failed."
79 // << std::endl;
80 } else {
81 m_gstos.push_back(std::pair<unsigned int,gl_manager*>(_id,&a_mgr));
82 }
83 }
84 return _id;
85 }
86protected:
87 void clean_gstos() {
88 std::vector< std::pair<unsigned int,gl_manager*> >::iterator it;
89 for(it=m_gstos.begin();it!=m_gstos.end();){
90 (*it).second->delete_gsto((*it).first);
91 it = m_gstos.erase(it);
92 }
93 }
94 void clean_gstos(gl_manager* a_mgr) {
95 std::vector< std::pair<unsigned int,gl_manager*> >::iterator it;
96 for(it=m_gstos.begin();it!=m_gstos.end();){
97 if((*it).second==a_mgr) {
98 (*it).second->delete_gsto((*it).first);
99 it = m_gstos.erase(it);
100 } else {
101 it++;
102 }
103 }
104 }
105protected:
106 unsigned int _find(gl_manager* a_mgr) {
107 std::vector< std::pair<unsigned int,gl_manager*> >::iterator it;
108 for(it=m_gstos.begin();it!=m_gstos.end();++it){
109 if((*it).second==a_mgr) return (*it).first;
110 }
111 return 0;
112 }
113
114protected:
115 std::vector< std::pair<unsigned int,gl_manager*> > m_gstos;
116};
117
118}}

120#endif
