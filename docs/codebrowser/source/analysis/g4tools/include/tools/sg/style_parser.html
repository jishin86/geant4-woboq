546 typedef struct {char string[32];sg::hjust value;} hjust_rec;
547 static const hjust_rec* hjust_recs(unsigned int& a_num){
548 static const hjust_rec list[] = { //read only static, then ok.
549 {"left", sg::left},
550 {"center", sg::center},
551 {"right", sg::right}
552 };
553 a_num = sizeof(list)/sizeof(list[0]);
554 return list;
555 }
556public:
557 static bool shjust(const std::string& a_s,sg::hjust& a_v){
558 unsigned int number;
559 const hjust_rec* list = hjust_recs(number);
560 for(unsigned int count=0;count<number;count++) {
561 if(a_s==list[count].string) {
562 a_v = list[count].value;
563 return true;
564 }
565 }
566 a_v = sg::left;
567 return false;
568 }
569 static const char* shjust(sg::hjust a_v){
570 unsigned int number;
571 const hjust_rec* list = hjust_recs(number);
572 for(unsigned int count=0;count<number;count++) {
573 if(a_v==list[count].value) return list[count].string;
574 }
575 return 0;
576 }
