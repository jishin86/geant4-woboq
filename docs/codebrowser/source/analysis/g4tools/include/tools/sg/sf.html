21template <class T>
22class sf : public bsf<T> {
23 typedef bsf<T> parent;
24public:
25 static const std::string& s_class() {
26 static const std::string s_v("tools::sg::sf<"+stype(T())+">");
27 return s_v;
28 }
29 virtual void* cast(const std::string& a_class) const {
30 if(void* p = cmp_cast< sf<T> >(this,a_class)) {return p;}
31 return parent::cast(a_class);
32 }
33 virtual const std::string& s_cls() const {return s_class();}
34public:
35 virtual bool write(io::iwbuf& a_buffer) {
36 return a_buffer.write(parent::m_value);
37 }
38 virtual bool read(io::irbuf& a_buffer) {
39 return a_buffer.read(parent::m_value);
40 }
41 virtual bool dump(std::ostream& a_out) {
42 a_out << parent::m_value << std::endl;
43 return true;
44 }
45 virtual bool s_value(std::string& a_s) const {
46 std::ostringstream strm;
47 strm << parent::m_value;
48 a_s = strm.str();
49 return true;
50 }
51public:
52 sf(){}
53 sf(const T& a_value):parent(a_value){}
54 virtual ~sf(){}
55public:
56 sf(const sf& a_from)
57 :parent(a_from)
58 {}
59 sf& operator=(const sf& a_from){
60 parent::operator=(a_from);
61 return *this;
62 }
63public:
64 sf& operator=(const T& a_value){
65 parent::operator=(a_value);
66 return *this;
67 }
68};
