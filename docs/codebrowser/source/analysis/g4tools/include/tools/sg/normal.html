4#ifndef tools_sg_normal

7#include "node"
8#include "sf_vec"
9#include "render_action"

11#include <vector>
12
13namespace tools {
14namespace sg {
15
16class normal : public node {
17 TOOLS_NODE(normal,tools::sg::normal,node)
18public:
19 sf_vec<vec3f,float> vec;
20public:
21 virtual const std::vector<field_desc>& node_fields() const {
22 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::normal)
23 static std::vector<field_desc> s_v;
24 if(s_v.empty()) {
25 s_v = parent::node_fields();
26 TOOLS_ADD_FIELD_DESC(vec)
27 }
28 return s_v;
29 }
30private:
31 void add_fields(){
32 add_field(&vec);
33 }
34public:
35 virtual void render(render_action& a_action) {
36 a_action.normal(vec.value());
37 }
38public:
39 normal()
40 :parent()
41 ,vec(vec3f(0,0,1))
42 {
43 add_fields();
44 }
45 virtual ~normal(){}
46public:
47 normal(const normal& a_from)
48 :parent(a_from)
49 ,vec(a_from.vec)
50 {
51 add_fields();
52 }
53 normal& operator=(const normal& a_from){
54 parent::operator=(a_from);
55 vec = a_from.vec;
56 return *this;
57 }
58};
59
60}}
