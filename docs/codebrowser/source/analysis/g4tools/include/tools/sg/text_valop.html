17class text_valop : public base_text {
18public:
19 TOOLS_NODE(text_valop,tools::sg::text_valop,base_text)
20public:
21 sf_string encoding;
22 sf_string font;
23public:
24 virtual const std::vector<field_desc>& node_fields() const {
25 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::text_valop)
26 static std::vector<field_desc> s_v;
27 if(s_v.empty()) {
28 s_v = parent::node_fields();
29 TOOLS_ADD_FIELD_DESC(encoding)
30 TOOLS_ADD_FIELD_DESC(font)
31 }
32 return s_v;
33 }
34private:
35 void add_fields(){
36 add_field(&encoding);
37 add_field(&font);
38 }
39public:
40 virtual void render(render_action& a_action) {
41 if(touched()) {
42 update_sg(a_action.out());
43 reset_touched();
44 }
45 m_sep.render(a_action);
46 }
47 virtual void pick(pick_action& a_action) {
48 if(touched()) {
49 update_sg(a_action.out());
50 reset_touched();
51 }
52 nodekit_pick(a_action,m_sep,this);
53 }
54 virtual void bbox(bbox_action& a_action) {
55 if(touched()) {
56 update_sg(a_action.out());
57 reset_touched();
58 }
59 m_sep.bbox(a_action);
60 }
61 virtual void search(search_action& a_action) {
62 parent::search(a_action);
63 if(a_action.done()) return;
64 if(touched()) {
65 update_sg(a_action.out());
66 reset_touched();
67 }
68 if(a_action.do_path()) a_action.path_push(this);
69 m_sep.search(a_action);
70 if(a_action.done()) return;
71 if(a_action.do_path()) a_action.path_pop();
72 }
73public:
74 text_valop(const base_freetype& a_ttf)
75 :parent()
76 ,encoding(encoding_none())
77 ,font()
78 ,m_ttf(a_ttf)
79 {
80 add_fields();
81 }
82 virtual ~text_valop(){}
83public:
84 text_valop(const text_valop& a_from)
85 :parent(a_from)
86 ,encoding(a_from.encoding)
87 ,font(a_from.font)
88 ,m_ttf(a_from.m_ttf)
89 {
90 add_fields();
91 }
92 text_valop& operator=(const text_valop& a_from){
93 parent::operator=(a_from);
94 encoding = a_from.encoding;
95 font = a_from.font;
96 return *this;
97 }
98public: //sg::base_text :
99 virtual void get_bounds(float /*a_height*/,
100 float& a_mn_x,float& a_mn_y,float& a_mn_z,
101 float& a_mx_x,float& a_mx_y,float& a_mx_z) const {
102 text_valop& self = const_cast<text_valop&>(*this);
103 if(self.touched()) {
104 self.update_sg(self.m_out);
105 self.reset_touched();
106 }
107 vec3f mn,mx;
108 mnmx(self.m_out,self.m_sep,mn,mx);
109 a_mn_x = mn[0];a_mn_y = mn[1];a_mn_z = mn[2];
110 a_mx_x = mx[0];a_mx_y = mx[1];a_mx_z = mx[2];
111 //::printf("debug : get_bounds : %g %g %g : %g %g %g\n",a_mn_x,a_mn_y,a_mn_z,a_mx_x,a_mx_y,a_mx_z);
112 }
113 virtual float ascent(float) const {return 1;}
114 virtual float y_advance(float) const {return 1;}
115 virtual float descent(float) const {return 0;}
116 virtual bool truncate(const std::string&,float,float,std::string&) const {return false;}
117protected:
118 void update_sg(std::ostream& a_out) {
119 m_sep.clear();
120 matrix* tsf = new matrix;
121 m_sep.add(tsf);
122 tools_vforcit(std::string,strings.values(),it) {
123 valop* _valop = new valop(valop::STRING,*it);
124 valop2sg v(a_out,m_sep,m_ttf);
125 if(!v.visit(*_valop)) {
126 a_out << "tools::sg::text_valop::upate_sg : valop2sg.visit() failed." << std::endl;
127 m_sep.clear();
128 delete _valop;
129 return;
130 }
131 delete _valop;
132 }
133 vec3f mn,mx;
134 mnmx(a_out,m_sep,mn,mx);
135 float h = mx[1]-mn[1];
136 if(!h) {
137 a_out << "tools::sg::text_valop::upate_sg : valop has null height." << std::endl;
138 m_sep.clear();
139 return;
140 }
141 float scale = height.value()/h;
142 tsf->set_scale(scale,scale,1);
143 }
144protected:
145 const base_freetype& m_ttf;
146 separator m_sep;
147 nostream m_out;
148};
