20class style_colormap : public std::map<unsigned int,style_color> {
21 typedef std::map<unsigned int,style_color> parent;
22protected:
23 typedef std::map<unsigned int,style_color> map_t;
24public:
25 TOOLS_SCLASS(tools::sg::style_colormap)
26public:
27 style_colormap():parent(){
28#ifdef TOOLS_MEM
29 mem::increment(s_class().c_str());
30#endif
31 }
32 virtual ~style_colormap(){
33#ifdef TOOLS_MEM
34 mem::decrement(s_class().c_str());
35#endif
36 }
37public:
38 style_colormap(const style_colormap& a_from):parent(a_from){
39#ifdef TOOLS_MEM
40 mem::increment(s_class().c_str());
41#endif
42 }
43 style_colormap& operator=(const style_colormap& a_from){
44 parent::operator=(a_from);
45 return *this;
46 }
47public:
48 bool get_color(unsigned int a_index,colorf& a_col) const {
49 map_t::const_iterator it = find(a_index);
50 if(it==end()) {
51 a_col.set_value(0.5F,0.5F,0.5F,1);
52 return false;
53 }
54 a_col = (*it).second.second;
55 return true;
56 }
57 bool get_color(const std::string& a_s,colorf& a_col) const {
58 tools_mforcit(unsigned int,style_color,*this,it) {
59 if((*it).second.first==a_s) {
60 a_col = (*it).second.second;
61 return true;
62 }
63 }
64 //we do not set a_col to something.
65 return false;
66 }
67 bool get_string_color(unsigned int a_index,std::string& a_s) const{
68 map_t::const_iterator it = find(a_index);
69 if(it==end()) {a_s.clear();return false;}
70 colorf c = (*it).second.second;
71
72 char ss[192]; //3*64 = 192
73 snpf(ss,sizeof(ss),"%g %g %g",c[0],c[1],c[2]);
74 a_s = std::string(ss);
75 return true;
76 }
77public:
78 void add(const style_color& a_color) {
79 (*this)[uint32((*this).size())] = a_color;
80 }
81};

366 void create_colors_circle(unsigned int a_offset,
367 const std::string& a_name,
368 unsigned char a_rgb[]) {
369 // Create the "circle" colors in the Color Wheel
370 for(unsigned int n=0;n<15;n++) {
371 int colorn = a_offset+n-10;
372 if(colorn<0) continue;
373 if(n>10) {
374 new_TColor(colorn,a_rgb[3*n]/255.,a_rgb[3*n+1]/255.,a_rgb[3*n+2]/255.,a_name+"+",n-10);
375 } else if (n<10) {
376 new_TColor(colorn,a_rgb[3*n]/255.,a_rgb[3*n+1]/255.,a_rgb[3*n+2]/255.,a_name+"-",10-n);
377 } else {
378 new_TColor(colorn,a_rgb[3*n]/255.,a_rgb[3*n+1]/255.,a_rgb[3*n+2]/255.,a_name);
379 }
380 //color->SetTitle(color->AsHexString());
381 }
382 }

530inline bool to_ulong(const std::string& a_s,unsigned long& a_v){
531 a_v = 0L;
532 if(::sscanf(a_s.c_str(),"%lx",&a_v)!=1) {
533 if(::sscanf(a_s.c_str(),"%lu",&a_v)!=1) {
534 a_v = 0L;

536 }

541inline bool find_color(const cmaps_t& a_cmaps,
542 const std::string& a_s,colorf& a_col){
543 //NOTE : if ret false, we do not set a_col to something.
544
545 std::string::size_type pos_slash = a_s.rfind('/');
546 if(pos_slash!=std::string::npos) { //<cmap>/<color name>
547 std::string cmap = a_s.substr(0,pos_slash);
548 std::string cnam = a_s.substr(pos_slash+1,a_s.size()-(pos_slash+1));
549 cmaps_t::const_iterator it = a_cmaps.find(cmap);
550 if(it==a_cmaps.end()) return false;
551 return (*it).second.get_color(cnam,a_col);
552
553 } else {
554
555 if( (a_s.size()==7) && (a_s[0]=='#') ) {
556 // #RRGGBB format :
557 // 1 3 5
558 unsigned long rr,gg,bb;
559
560 {std::string s("0x");
561 s += a_s.substr(1,2);
562 if(!to_ulong(s,rr)) return false;}
563
564 {std::string s("0x");
565 s += a_s.substr(3,2);
566 if(!to_ulong(s,gg)) return false;}
567
568 {std::string s("0x");
569 s += a_s.substr(5,2);
570 if(!to_ulong(s,bb)) return false;}
571
572 a_col = colorf(((float)rr)/255,((float)gg)/255,((float)bb)/255);
573 return true;
574 }
575
576 //a_s of the form : <real:r> <real:g> <real:b>
577 //a_s of the form : <real:r> <real:g> <real:b> <real:a>
578 //a_s of the form : <real:r> <real:g> <real:b> <string:name>
579 //a_s of the form : <real:r> <real:g> <real:b> <real:a> <string:name>
580
581 {std::vector<std::string> ws;
582 words(a_s," ",false,ws);
583 if(ws.size()==3) {
584 float r,g,b;
585 if( to<float>(ws[0],r) &&
586 to<float>(ws[1],g) &&
587 to<float>(ws[2],b) ){
588 a_col = colorf(r,g,b);
589 return true;
590 }
591
592 } else if(ws.size()==4) {
593 float r,g,b,a;
594 if( to<float>(ws[0],r) &&
595 to<float>(ws[1],g) &&
596 to<float>(ws[2],b) ){
597
598 if(to<float>(ws[3],a)) {
599 a_col = colorf(r,g,b,a);
600 } else {
601 a_col = colorf(r,g,b,1);
602 }
603 return true;
604 }
605
606 } else if(ws.size()==5) {
607 float r,g,b,a;
608 if( to<float>(ws[0],r) &&
609 to<float>(ws[1],g) &&
610 to<float>(ws[2],b) &&
611 to<float>(ws[3],a) ){
612 a_col = colorf(r,g,b,a);
613 return true;
614 }
615
616 }}
617
618 {cmaps_t::const_iterator it = a_cmaps.find(style_default_colormap::s_default());
619 if(it!=a_cmaps.end()) {
620 if((*it).second.get_color(a_s,a_col)) return true;
621 }}
622
623 }
624
625 return false;
626}
