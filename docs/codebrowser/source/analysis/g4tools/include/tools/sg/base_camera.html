168 void rotate_around_direction(float a_delta) { //used in exlib/examples/iOS/sg.
169 vec3f dir;
170 orientation.value().mul_vec(vec3f(0,0,-1),dir);

175 void rotate_around_up(float a_delta){
176 vec3f up;
177 orientation.value().mul_vec(vec3f(0,1,0),up);
178 //orientation.value(rotf(up,a_delta) * orientation.value());
179 // must be the below so that rot-cam works for exlib/cbk/[pyrfits,astro,cfitsio] astro setup.
180 // (astro setup change camera orientation).
181 orientation.value(orientation.value() * rotf(up,a_delta));

184 void rotate_around_y_at_focal(float a_delta){
185 //from coin SoGuiExaminerViewerP::rotYWheelMotion.
186 vec3f dir;
187 orientation.value().mul_vec(vec3f(0,0,-1),dir);
188 vec3f focalpoint = position.value() + focal * dir;
189 orientation.value(rotf(vec3f(0,1,0),a_delta) * orientation.value());
190 orientation.value().mul_vec(vec3f(0,0,-1),dir);
191 position = focalpoint - focal * dir;
192 }
193
194 void rotate_around_x_at_focal(float a_delta){
195 //from coin SoGuiExaminerViewerP::rotXWheelMotion.
196 vec3f dir;
197 orientation.value().mul_vec(vec3f(0,0,-1),dir);
198 vec3f focalpoint = position.value() + focal * dir;
199 orientation.value(rotf(vec3f(1,0,0),a_delta) * orientation.value());
200 orientation.value().mul_vec(vec3f(0,0,-1),dir);
201 position = focalpoint - focal * dir;
202 }

204 void rotate_to_dir(const vec3f& a_dir) {
205 //rotate around up so that a_dir is in (dir,up) plane
206
207 //NOTE : it is the invert of orientation which is used
208 // in projection matrix.
209
210 {vec3f dir;
211 orientation.value().mul_vec(vec3f(0,0,-1),dir);
212 vec3f up;
213 orientation.value().mul_vec(vec3f(0,1,0),up);
214 vec3f side;dir.cross(up,side);
215 vec3f v = side * (side.dot(a_dir)) + dir * (dir.dot(a_dir));
216 if(v.normalize()) orientation.value(orientation.value()*rotf(dir,v));}
217
218 //rotate around dir^up so that a_dir matches dir.
219 {vec3f dir;
220 orientation.value().mul_vec(vec3f(0,0,-1),dir);
221 orientation.value(orientation.value()*rotf(dir,a_dir));}
222

236 void pane_to(float a_x,float a_y,float a_z){
237 //translate in view plane so that (a_x,a_y,a_z) is on direction.
238
239 vec3f dir;
240 orientation.value().mul_vec(vec3f(0,0,-1),dir);
241 vec3f up;
242 orientation.value().mul_vec(vec3f(0,1,0),up);
243 vec3f side;dir.cross(up,side);
244
245 vec3f d(a_x,a_y,a_z);
246 d.subtract(position.value());
247
248 vec3f pos = position.value() + side * (side.dot(d)) + up * (up.dot(d));
249 position.value(pos);
250 }

297 void astro_orientation(float a_ra,float a_dec/*,const vec3f& a_center*/) {
298 // a_ra, a_dec are in decimal degrees.
299
300 // Camera default point toward -z with up along +y and +x at right.
301
302 // Arrange so that camera points toward x with up along +z :
303 rotf r(vec3f::s_y(),-fhalf_pi());
304 r *= rotf(vec3f::s_x(),fhalf_pi());
305 // Now -y is at right.
306
307 // Then rotate it so that it points toward given (ra,dec) by keeping up upward +z direction.
308 r *= rotf(vec3f::s_y(),-a_dec*fdeg2rad());
309 r *= rotf(vec3f::s_z(),a_ra*fdeg2rad());
310 orientation = r;
311
312/*
313 position = a_center*0.99f;
314 znear = 0.1f;
315 zfar = 200.0f;
316 focal = (a_center-position).length();
317*/
318/*
319 position = vec3f(0,0,0);
320 znear = 1.0f;
321 zfar = 2000.0f; //2*sky_radius.
322 focal = a_center.length();
323 da = 0.017f/100; //1/100 of a degree
324*/
325 }
326protected:
327 void update_sg(std::ostream& a_out) {
328
329 {const vec4f& v = m_lrbt.value();
330 float l = v[0];
331 float r = v[1];
332 float b = v[2];
333 float t = v[3];
334 float n = znear.value();
335 float f = zfar.value();
336 if(is_type_ortho()) {
337 m_proj.set_ortho(l,r,b,t,n,f);
338 } else {
339 m_proj.set_frustum(l,r,b,t,n,f);
340 }}
341
342 if(orientation.value().quat()!=id_orientation()) //OPTIMIZATION
343 {rotf rinv;
344 if(orientation.value().inverse(rinv)) {
345 mat4f mtx;
346 rinv.value(mtx);
347 m_proj.mul_mtx(mtx,m_tmp);
348 } else {
349 a_out << "base_camera::update_sg :"
350 << " get orientation inverse failed."
351 << std::endl;
352 }}
353
354 m_proj.mul_translate(-position.value()[0],
355 -position.value()[1],
356 -position.value()[2]);
357 }

359 void _mult_matrix(matrix_action& a_action) {
360 float l,r,b,t;
361 get_lrbt(a_action.ww(),a_action.wh(),l,r,b,t);
362 m_lrbt.set_value(l,r,b,t);
363
364 if(touched()||m_lrbt.touched()) {
365 update_sg(a_action.out());
366 reset_touched();
367 m_lrbt.reset_touched();
368 }
369
370 a_action.projection_matrix().mul_mtx(m_proj,m_tmp);
371 }
