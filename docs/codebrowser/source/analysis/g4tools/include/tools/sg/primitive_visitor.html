55 bool add_triangle_fan(size_t a_floatn,const float* a_xyzs,bool a_stop = false){
56 size_t num = a_floatn/3;
57 if(num<3) return false;
58
59 m_mode = gl::triangle_fan();
60
61 float p1x,p1y,p1z,w1=1;
62 float p2x,p2y,p2z,w2=1;
63 float p3x,p3y,p3z,w3=1;
64
65 const float* pos1 = a_xyzs+3*0;
66 p1x = *(pos1+0);
67 p1y = *(pos1+1);
68 p1z = *(pos1+2);
69 project(p1x,p1y,p1z,w1);
70
71 const float* pos2 = a_xyzs+3*1;
72 p2x = *(pos2+0);
73 p2y = *(pos2+1);
74 p2z = *(pos2+2);
75 project(p2x,p2y,p2z,w2);
76
77 for(size_t index=2;index<num;index++) {
78 const float* pos = a_xyzs+3*index;
79 p3x = *(pos+0);
80 p3y = *(pos+1);
81 p3z = *(pos+2);
82 project(p3x,p3y,p3z,w3);
83
84 if(!add_triangle(p1x,p1y,p1z,w1,
85 p2x,p2y,p2z,w2,
86 p3x,p3y,p3z,w3)) {
87 if(a_stop) return false;
88 }
89
90 p2x = p3x;
91 p2y = p3y;
92 p2z = p3z;
93 w2 = w3;
94 }
95 return true;
96 }

98 bool add_triangle_strip(size_t a_floatn,const float* a_xyzs,bool a_stop = false){
99 size_t num = a_floatn/3;
100 if(num<3) return false;
101
102 m_mode = gl::triangle_strip();
103
104 float p1x,p1y,p1z,w1=1;
105 float p2x,p2y,p2z,w2=1;
106 float p3x,p3y,p3z,w3=1;
107
108 const float* pos1 = a_xyzs+3*0;
109 p1x = *(pos1+0);
110 p1y = *(pos1+1);
111 p1z = *(pos1+2);
112 project(p1x,p1y,p1z,w1);
113
114 const float* pos2 = a_xyzs+3*1;
115 p2x = *(pos2+0);
116 p2y = *(pos2+1);
117 p2z = *(pos2+2);
118 project(p2x,p2y,p2z,w2);
119
120 bool flip = false;
121 for(size_t index=2;index<num;index++) {
122 const float* pos = a_xyzs+3*index;
123 p3x = *(pos+0);
124 p3y = *(pos+1);
125 p3z = *(pos+2);
126 project(p3x,p3y,p3z,w3);
127
128 if(flip){
129 if(!add_triangle(p1x,p1y,p1z,w1,
130 p3x,p3y,p3z,w3,
131 p2x,p2y,p2z,w2)) {
132 if(a_stop) return false;
133 }
134 } else {
135 if(!add_triangle(p1x,p1y,p1z,w1,
136 p2x,p2y,p2z,w2,
137 p3x,p3y,p3z,w3)) {
138 if(a_stop) return false;
139 }
140 }
141
142 p1x = p2x;
143 p1y = p2y;
144 p1z = p2z;
145 w1 = w2;
146
147 p2x = p3x;
148 p2y = p3y;
149 p2z = p3z;
150 w2 = w3;
151
152 flip = flip?false:true;
153 }
154 return true;
155 }

225 p3z = *pos;pos++;

561 bool add_primitive(gl::mode_t a_mode,size_t a_floatn,const float* a_xyzs,bool a_stop = false) {
562
563 if(a_mode==gl::points()) {
564 return add_points(a_floatn,a_xyzs,a_stop);
565
566 } else if(a_mode==gl::lines()) {
567 return add_lines(a_floatn,a_xyzs,a_stop);
568
569 } else if(a_mode==gl::line_loop()) {
570 return add_line_loop(a_floatn,a_xyzs,a_stop);
571
572 } else if(a_mode==gl::line_strip()) {
573 return add_line_strip(a_floatn,a_xyzs,a_stop);
574
575 } else if(a_mode==gl::triangles()) {
576 return add_triangles(a_floatn,a_xyzs,a_stop);
577
578 } else if(a_mode==gl::triangle_strip()) {
579 return add_triangle_strip(a_floatn,a_xyzs,a_stop);
580
581 } else if(a_mode==gl::triangle_fan()) {
582 return add_triangle_fan(a_floatn,a_xyzs,a_stop);
583
584 } else {
585 return false;
586 }

645 bool add_triangle_fan_xy(size_t a_floatn,const float* a_xys,
646 bool a_stop = false,
647 bool a_triangle_revert = false){
648 size_t num = a_floatn/2;
649 if(num<3) return false;
650
651 m_mode = gl::triangle_fan();
652
653 float p1x,p1y,p1z,w1=1;
654 float p2x,p2y,p2z,w2=1;
655 float p3x,p3y,p3z,w3=1;
656
657 const float* pos1 = a_xys+2*0;
658 p1x = *(pos1+0);
659 p1y = *(pos1+1);
660 p1z = 0;
661 project(p1x,p1y,p1z,w1);
662
663 const float* pos2 = a_xys+2*1;
664 p2x = *(pos2+0);
665 p2y = *(pos2+1);
666 p2z = 0;
667 project(p2x,p2y,p2z,w2);
668
669 for(size_t index=2;index<num;index++) {
670 const float* pos = a_xys+2*index;
671 p3x = *(pos+0);
672 p3y = *(pos+1);
673 p3z = 0;
674 project(p3x,p3y,p3z,w3);
675
676 if(a_triangle_revert) {
677 if(!add_triangle(p3x,p3y,p3z,w3,
678 p2x,p2y,p2z,w2,
679 p1x,p1y,p1z,w1)) {
