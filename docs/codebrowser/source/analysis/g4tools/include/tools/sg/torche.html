13#include "../colorfs"

15namespace tools {
16namespace sg {
17
18class torche : public node {
19 TOOLS_NODE(torche,tools::sg::torche,node)
20public:
21 sf_vec<colorf,float> color;
22 sf_vec<vec3f,float> direction;
23 sf<bool> on;
24public:
25 virtual const std::vector<field_desc>& node_fields() const {
26 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::torche)
27 static std::vector<field_desc> s_v;
28 if(s_v.empty()) {
29 s_v = parent::node_fields();
30 TOOLS_ADD_FIELD_DESC(color)
31 TOOLS_ADD_FIELD_DESC(direction)
32 TOOLS_ADD_FIELD_DESC(on)
33 }
34 return s_v;
35 }
36private:
37 void add_fields(){
38 add_field(&color);
39 add_field(&direction);
40 add_field(&on);
41 }
42public:
43 virtual void render(render_action& a_action) {
44 if(!on.value()) return;
45
46 state& state = a_action.state();
47
48 if((state.m_light+1)>=a_action.max_lights()) {
49 a_action.out()
50 << "GL_MAX_LIGHTS (" << a_action.max_lights() << ") reached."
51 << std::endl;
52 return;
53 }
54
55 state.m_GL_LIGHTING = true; //for separator
56
57 a_action.enable_light(state.m_light,direction.value(),color.value());
58 state.m_light++;
59 }
60public:
61 torche()
62 :parent()
63 ,color(colorf_white())
64 ,direction(vec3f(0,0,-1))
65 ,on(true)
66 {
67 add_fields();
68 }
69 virtual ~torche(){}
70public:
71 torche(const torche& a_from)
72 :parent(a_from)
73 ,color(a_from.color)
74 ,direction(a_from.direction)
75 ,on(a_from.on)
76 {
77 add_fields();
78 }
79 torche& operator=(const torche& a_from){
80 parent::operator=(a_from);
81 color = a_from.color;
82 direction = a_from.direction;
83 on = a_from.on;
84 return *this;
85 }
86};
87
88}}
