135 virtual void draw_vertex_array_texture(gl::mode_t,size_t a_xyzn,const float* a_xyzs,
136 gstoid a_id,const float* a_tcs) {
137 //::printf("debug : render_zb : 000 : %d\n",a_id);
138
139 img_byte img;
140 if(!m_mgr.find(a_id,img)) return;
141
142 //::printf("debug : render_zb : img : %d %d\n",img.width(),img.height());
143 if(img.is_empty()) return;
144
145 unsigned int imw = img.width();
146 unsigned int imh = img.height();
147 unsigned int imn = img.bpp();
148 if(imn!=3) {
149 m_out << "tools::sg::render_zb::draw_vertex_array_texture :"
150 << " not a 3 bytes per pixel image."
151 << std::endl;
152 return;
153 }
154
155 if(a_xyzn!=12) {
156 m_out << "tools::sg::render_zb::draw_vertex_array_texture :"
157 << " primitive has not four points."
158 << std::endl;
159 return;
160 }
161
162 //::printf("debug : render_zb : tcs : %g %g, %g %g, %g %g, %g %g\n",
163 // a_tcs[0],a_tcs[1],a_tcs[2],a_tcs[3],a_tcs[4],a_tcs[5],a_tcs[6],a_tcs[7]);
164
165 vec3f p1(a_xyzs[0],a_xyzs[ 1],a_xyzs[ 2]);
166 vec3f p2(a_xyzs[3],a_xyzs[ 4],a_xyzs[ 5]);
167 //vec3f p3(a_xyzs[6],a_xyzs[ 7],a_xyzs[ 8]);
168 vec3f p4(a_xyzs[9],a_xyzs[10],a_xyzs[11]);
169
170 vec3f dx = p2-p1;
171 vec3f dy = p4-p1;
172
173 vec2f t1(a_tcs[0],a_tcs[1]);
174 vec2f t2(a_tcs[2],a_tcs[3]);
175 vec2f t3(a_tcs[4],a_tcs[5]);
176 vec2f t4(a_tcs[6],a_tcs[7]);
177 vec2f tdx = t2-t1;
178 vec2f tdy = t4-t1;
179 if(!tdx.length()) {
180 m_out << "tools::sg::render_zb::draw_vertex_array_texture :"
181 << " tdx is null."
182 << std::endl;
183 return;
184 }
185 if(!tdy.length()) {
186 m_out << "tools::sg::render_zb::draw_vertex_array_texture :"
187 << " tdy is null."
188 << std::endl;
189 return;
190 }
191 std::vector<vec2f> poly;
192 poly.push_back(t1);
193 poly.push_back(t2);
194 poly.push_back(t3);
195 poly.push_back(t4);
196 poly.push_back(t1);
197
198 {float r,g,b,tx,ty;
199 vec3f p;
200 vec2f t;
201 unsigned char* pos = img.buffer();
202 for(unsigned int row=0;row<imh;row++) {
203 for(unsigned int col=0;col<imw;col++) {
204 r = (*pos)/255.0f;pos++;
205 g = (*pos)/255.0f;pos++;
206 b = (*pos)/255.0f;pos++;
207
208 tx = float(col)/float(imw-1); //[0,1]
209 ty = float(row)/float(imh-1); //[0,1]
210
211 t.set_value(tx,ty);
212
213 if(!is_inside(t,poly)) continue;
214
215 t = t - t1;
216
217 p = p1+dx*t.x()/tdx.length()+dy*t.y()/tdy.length();
218
219 m_rgba.set_value(r,g,b,1);
220 m_pv.add_one_point(p.x(),p.y(),p.z());
221 }
222 }}

369 static bool get_rgb(void* a_tag,unsigned int a_col,unsigned int a_row,VCol& a_r,VCol& a_g,VCol& a_b){
370 //used with wps.
371 render_zb* rzb = (render_zb*)a_tag;
372
373 zb::buffer::ZPixel pix;
374 if(!rzb->zbuffer().get_clipped_pixel(a_col,rzb->wh()-1-a_row,pix)){
375 rzb->out() << "get_rgb : can't get zbuffer pixel" << std::endl;
376 a_r = 1;
377 a_g = 0;
378 a_b = 0;
379 return false;
380 }
381
382 if(rzb->rcolormap().empty()) rzb->gen_rcmap();
383
384 {rcmap_t::const_iterator it = rzb->rcolormap().find(pix);
385 if(it==rzb->rcolormap().end()) {
386 rzb->out() << "can't find pixel " << pix
387 << " in cmap (sz " << rzb->rcolormap().size() << ")."
388 << std::endl;
389 a_r = 1;
390 a_g = 0;
391 a_b = 0;
392 return false;
393 }
394 a_r = (*it).second.r();
395 a_g = (*it).second.g();
396 a_b = (*it).second.b();}
397
398 return true;
399 }

401 unsigned char* get_rgbas(size_t& a_sz) {
402 if(!m_ww || !m_wh) {a_sz = 0;return 0;}
403 a_sz = 4 * m_ww * m_wh;
404 typedef unsigned char uchar;
405 uchar* rgbas = new uchar[a_sz];
406 if(!rgbas) {a_sz = 0;return 0;}
407 uchar* pos = rgbas;
408 VCol r,g,b;
409 VCol a = 1;
410 for(unsigned int row=0;row<m_wh;row++) {
411 for(unsigned int col=0;col<m_ww;col++) {
412 get_rgb(this,col,m_wh-row-1,r,g,b);
413 *pos = (uchar)(r*255.0F);pos++;
414 *pos = (uchar)(g*255.0F);pos++;
415 *pos = (uchar)(b*255.0F);pos++;
416 *pos = (uchar)(a*255.0F);pos++;
417 }
418 }
419 return rgbas;
420 }

539 unsigned int npix(float a_size) { //a_size must be >1
540 // 0 -> 0
541 // 1 -> 0
542 // 2 -> 1 3x3
543 // 3 -> 1 3x3
544 // 4 -> 2 5x5
545 // 5 -> 2 5x5
546 // 6 -> 3 7x7
547 unsigned int num = (unsigned int)a_size;
548 unsigned int num_2 = num/2;
549 if(2*num_2==num) {num++;num_2 = num/2;}
550 return num_2;
551 }

569 bool _add_triangle(float a_p1x,float a_p1y,float a_p1z,float a_p1w,
570 float a_p2x,float a_p2y,float a_p2z,float a_p2w,
571 float a_p3x,float a_p3y,float a_p3z,float a_p3w,
572 const colorf& a_color){
573
574 float p1x = a_p1x;float p1y = a_p1y;float p1z = a_p1z;//float p1w = a_p1w;
575 float p2x = a_p2x;float p2y = a_p2y;float p2z = a_p2z;//float p2w = a_p2w;
576 float p3x = a_p3x;float p3y = a_p3y;float p3z = a_p3z;//float p3w = a_p3w;
577
578 m_rzb.m_vp_mtx.mul_3f(p1x,p1y,p1z);
579 m_rzb.m_vp_mtx.mul_3f(p2x,p2y,p2z);
580 m_rzb.m_vp_mtx.mul_3f(p3x,p3y,p3z);
581 p1z *= -1;
582 p2z *= -1;
583 p3z *= -1;
584
585 if(m_rzb.m_POLYGON_OFFSET_FILL){
586 // zs are in [-1,1]
587 float epsil = 1e-5f;
588 //float epsil = 1e-4f;
589 p1z -= epsil;
590 p2z -= epsil;
591 p3z -= epsil;
592 }

594 typedef zb::ZZ ZZ; //double
595
596 plane<vec3d> pn(
597 vec3<ZZ>(p1x,p1y,p1z),
598 vec3<ZZ>(p2x,p2y,p2z),
599 vec3<ZZ>(p3x,p3y,p3z)
600 );
601 if(!pn.is_valid()) return true;
602
603 // norm[0]*x+norm[1]*y+norm[2]*z = dist
604 // A*x+B*y+C*z+D = 0
605
606 ZZ C = pn.normal()[2];
607
608 if(m_rzb.m_CULL_FACE){ // check back facing or by the edge :
609 if(m_rzb.m_ccw) {
610 if(C<=0) return true;
611 } else {
612 if(C>=0) return true;
613 }
614 }
615
616 ZZ A = pn.normal()[0];
617 ZZ B = pn.normal()[1];
618 ZZ D = -pn.distance_from_origin();
619
620 //ZZ zmn = mn<ZZ>(mn<ZZ>(p1z,p2z),p3z);
621 //ZZ zmx = mx<ZZ>(mx<ZZ>(p1z,p2z),p3z);
622
623 zb::point list[3];
624 zinit(list[0],p1x,p1y,p1z);
625 zinit(list[1],p2x,p2y,p2z);
626 zinit(list[2],p3x,p3y,p3z);
627
628 m_rzb.m_zb.set_depth_test(m_rzb.m_DEPTH_TEST);
629
630 if(m_rzb.m_light_on) {
631
632 float _p1x = a_p1x;float _p1y = a_p1y;float _p1z = a_p1z;float _p1w = a_p1w;
633 float _p2x = a_p2x;float _p2y = a_p2y;float _p2z = a_p2z;float _p2w = a_p2w;
634 float _p3x = a_p3x;float _p3y = a_p3y;float _p3z = a_p3z;float _p3w = a_p3w;
635
636 _p1x *= _p1w;_p1y *= _p1w;_p1z *= _p1w;
637 _p2x *= _p2w;_p2y *= _p2w;_p2z *= _p2w;
638 _p3x *= _p3w;_p3y *= _p3w;_p3z *= _p3w;
639
640 m_rzb.m_proj_1.mul_4f(_p1x,_p1y,_p1z,_p1w);
641 m_rzb.m_proj_1.mul_4f(_p2x,_p2y,_p2z,_p2w);
642 m_rzb.m_proj_1.mul_4f(_p3x,_p3y,_p3z,_p3w);
643
644 plane<vec3d> _pn(
645 vec3<ZZ>(_p1x,_p1y,_p1z),
646 vec3<ZZ>(_p2x,_p2y,_p2z),
647 vec3<ZZ>(_p3x,_p3y,_p3z)
648 );
649 if(_pn.is_valid()) {
650 vec3f npn(float(_pn.normal().x()),
651 float(_pn.normal().y()),
652 float(_pn.normal().z()));
653 vec3f d = m_rzb.m_light_direction;
654 float dx = m_rzb.m_light_direction.x();
655 float dy = m_rzb.m_light_direction.y();
656 float dz = m_rzb.m_light_direction.z();
657 m_rzb.m_model.mul_3f(dx,dy,dz);
658 d.set_value(dx,dy,dz);
659 if(d.normalize()) {
660 float dot = npn.dot(d);
661 if((-1<=dot)&&(dot<=0)) {
662 dot *= -1;
663
664// colorf c
665// (a_color.r()*dot,a_color.g()*dot,a_color.b()*dot,a_color.a());
666
667 float h,l,s;
668 rgb_to_hls(a_color.r(),a_color.g(),a_color.b(),h,l,s);
669 l *= dot;
670 float r,g,b;
671 hls_to_rgb(h,l,s,r,g,b);
672
673 colorf c(r,g,b,a_color.a());
674
675 m_rzb.m_zb.draw_polygon(3,list,A,B,C,D,m_rzb.get_pix(c));
676 //m_rzb.m_zb.draw_polygon(3,list,A,B,C,D,m_rzb.get_pix(a_color));
677 }
678 }
679 }
680
681 } else {
682 m_rzb.m_zb.draw_polygon(3,list,A,B,C,D,m_rzb.get_pix(a_color));
683 }
684
685 return true;
686 }
