147public:
148 void screen2ndc(int a_x,int a_y, //signed because of wall.
149 float& a_wcx,float& a_wcy,float& a_wcz,float& a_wcw) const {
150 // a proj point in near plane has (z,w) :
151 // ortho -1,1 -> z/w = -1 and xy in [-1,1][-1,1]
152 // persp -n,n -> z/w = -1 and xy in [-n,n][-n,n]
153
154 a_wcx = 2*(float(a_x)/float(m_ww)-0.5f); //in [-1,1]
155 a_wcy = 2*(float(a_y)/float(m_wh)-0.5f);
156 a_wcz = -1;
157
158 if(m_camera_ortho) {
159 a_wcw = 1;
160 } else {
161 float t = m_camera_znear;
162 a_wcx *= t;
163 a_wcy *= t;
164 a_wcz *= t;
165 a_wcw = t;
166 }
167 }
168
169 bool screen2wc(int a_x,int a_y, //signed because of wall.
170 float& a_wcx,float& a_wcy,float& a_wcz) const {
171 mat4f mtx = m_proj;
172 mtx.mul_mtx(m_model);
173 mat4f inv;
174 if(!mtx.invert(inv)) {a_wcx = 0;a_wcy = 0;a_wcz = 0;return false;}
175 float w;
176 screen2ndc(a_x,a_y,a_wcx,a_wcy,a_wcz,w);
177 inv.mul_4f(a_wcx,a_wcy,a_wcz,w);
178 if(w==0.0F) return false;
179 a_wcx /= w;
180 a_wcy /= w;
181 a_wcz /= w;
182 return true;
183 }
184 bool screen2pwc(int a_x,int a_y, //signed because of wall.
185 float& a_wcx,float& a_wcy,float& a_wcz) const {
186 mat4f mtx = m_proj;
187 //mtx.mul_mtx(m_model);
188 mat4f inv;
189 if(!mtx.invert(inv)) {a_wcx = 0;a_wcy = 0;a_wcz = 0;return false;}
190 float w;
191 screen2ndc(a_x,a_y,a_wcx,a_wcy,a_wcz,w);
192 inv.mul_4f(a_wcx,a_wcy,a_wcz,w);
193 if(w==0.0F) return false;
194 a_wcx /= w;
195 a_wcy /= w;
196 a_wcz /= w;
197 return true;
198 }
199 void camera_proj_only(mat4f& a_mtx) const {
200 float l = m_camera_lrbt[0];
201 float r = m_camera_lrbt[1];
202 float b = m_camera_lrbt[2];
203 float t = m_camera_lrbt[3];
204 float n = m_camera_znear;
205 float f = m_camera_zfar;
206 if(m_camera_ortho) {
207 a_mtx.set_ortho(l,r,b,t,n,f);
208 } else {
209 a_mtx.set_frustum(l,r,b,t,n,f);
210 }
211 }
