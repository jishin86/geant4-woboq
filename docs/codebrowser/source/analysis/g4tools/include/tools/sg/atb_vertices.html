50 virtual unsigned int create_gsto(std::ostream&,sg::gl_manager& a_mgr) {
51 //unsigned int npt = xyzs.values().size()/3;
52 //::printf("debug : atb_vertices : %lu : create_gsto : %u\n",this,npt);
53
54 std::vector<float> gsto_data;
55
56 if(rgbas.size()) {
57 if(nms.size()) {
58 if(do_back.value()) {
59 append(gsto_data,xyzs.values());
60 append(gsto_data,nms.values());
61 append(gsto_data,m_back_xyzs);
62 append(gsto_data,m_back_nms);
63 append(gsto_data,rgbas.values());
64 } else {
65 append(gsto_data,xyzs.values());
66 append(gsto_data,nms.values());
67 append(gsto_data,rgbas.values());
68 }
69 if(draw_edges.value()) {
70 // allocate edges :
71 size_t pos_edges = gsto_data.size();
72 append(gsto_data,xyzs.values());
73 append(gsto_data,xyzs.values());
74 float* pxyz = vec_data<float>(xyzs.values());
75 float* pedges = vec_data<float>(gsto_data)+pos_edges;
76 size_t npt = xyzs.values().size()/3;
77 size_t ntri = npt/3;
78 for(size_t itri=0;itri<ntri;itri++) {
79 // first edge :
80 *pedges = *(pxyz+0);pedges++;
81 *pedges = *(pxyz+1);pedges++;
82 *pedges = *(pxyz+2);pedges++;
83
84 *pedges = *(pxyz+3);pedges++;
85 *pedges = *(pxyz+4);pedges++;
86 *pedges = *(pxyz+5);pedges++;
87
88 // second edge :
89 *pedges = *(pxyz+3);pedges++;
90 *pedges = *(pxyz+4);pedges++;
91 *pedges = *(pxyz+5);pedges++;
92
93 *pedges = *(pxyz+6);pedges++;
94 *pedges = *(pxyz+7);pedges++;
95 *pedges = *(pxyz+8);pedges++;
96
97 // third edge :
98 *pedges = *(pxyz+6);pedges++;
99 *pedges = *(pxyz+7);pedges++;
100 *pedges = *(pxyz+8);pedges++;
101
102 *pedges = *(pxyz+0);pedges++;
103 *pedges = *(pxyz+1);pedges++;
104 *pedges = *(pxyz+2);pedges++;
105
106 pxyz += 9;
107 }
108
109 }
110 } else {
111 append(gsto_data,xyzs.values());
112 append(gsto_data,rgbas.values());
113 }
114 } else {
115 if(nms.size()) {
116 append(gsto_data,xyzs.values());
117 append(gsto_data,nms.values());
118 } else {
119 append(gsto_data,xyzs.values());
120 }
121 }
122 return a_mgr.create_gsto_from_data(gsto_data);
123 }

125public:
126 virtual void render(render_action& a_action) {
127 if(touched()) {
128 if(do_back.value()) gen_back();
129 if(draw_edges.value()) gen_edges();
130 reset_touched();
131 }
132 if(xyzs.empty()) return;
133
134 const state& state = a_action.state();
135
136 if(state.m_use_gsto) {
137 unsigned int _id = get_gsto_id(a_action.out(),a_action.gl_manager());
138 if(_id) {
139 a_action.begin_gsto(_id);
140 if(rgbas.size()) {
141 if(nms.size()) {
142 size_t npt = xyzs.values().size()/3;
143 bufpos pos_xyzs = 0;
144 bufpos pos_nms = 0;
145 bufpos pos_back_xyzs = 0;
146 bufpos pos_back_nms = 0;
147 bufpos pos_rgbas = 0;
148 bufpos pos_edges = 0;
149 {size_t sz = npt*3;
150 if(do_back.value()) {
151 pos_xyzs = 0;
152 pos_nms = pos_xyzs+sz*sizeof(float); //bytes
153 pos_back_xyzs = pos_nms+sz*sizeof(float);
154 pos_back_nms = pos_back_xyzs+sz*sizeof(float);
155 pos_rgbas = pos_back_nms+sz*sizeof(float);
156 } else {
157 pos_xyzs = 0;
158 pos_nms = pos_xyzs+sz*sizeof(float);
159 pos_rgbas = pos_nms+sz*sizeof(float);
160 }}
161 if(draw_edges.value()) {
162 pos_edges = pos_rgbas+npt*4*sizeof(float);
163 }
164 if(gl::is_line(mode.value())) {
165 //Same logic as Inventor SoLightModel.model = BASE_COLOR.
166 a_action.set_lighting(false);
167 if(do_back.value()) a_action.draw_gsto_vcn(mode.value(),npt,pos_back_xyzs,pos_rgbas,pos_back_nms);
168 a_action.draw_gsto_vcn(mode.value(),npt,pos_xyzs,pos_rgbas,pos_nms);
169 a_action.set_lighting(state.m_GL_LIGHTING);
170 } else if(mode.value()==gl::triangles()) {
171 if(draw_edges.value()) {
172 a_action.color4f(0,0,0,1);
173 a_action.line_width(1);
174 a_action.draw_gsto_v(gl::lines(),2*npt,pos_edges);
175 //pushes back the filled polygons to avoid z-fighting with lines
176 a_action.set_polygon_offset(true);
177
178 a_action.color4f(state.m_color);
179 a_action.line_width(state.m_line_width);
180 //a_action.set_lighting(state.m_GL_LIGHTING);
181 }
182 if(do_back.value()) a_action.draw_gsto_vcn(mode.value(),npt,pos_back_xyzs,pos_rgbas,pos_back_nms);
183 a_action.draw_gsto_vcn(mode.value(),npt,pos_xyzs,pos_rgbas,pos_nms);
184 a_action.set_polygon_offset(state.m_GL_POLYGON_OFFSET_FILL);
185 } else {
186 if(do_back.value()) a_action.draw_gsto_vcn(mode.value(),npt,pos_back_xyzs,pos_rgbas,pos_back_nms);
187 a_action.draw_gsto_vcn(mode.value(),npt,pos_xyzs,pos_rgbas,pos_nms);
188 }
189
190 } else {
191 size_t npt = xyzs.values().size()/3;
192 bufpos pos_xyzs = 0;
193 bufpos pos_rgbas = npt*3*sizeof(float);
194 if(gl::is_line(mode.value())) {
195 //Same logic as Inventor SoLightModel.model = BASE_COLOR.
196 a_action.set_lighting(false);
197 a_action.draw_gsto_vc(mode.value(),npt,pos_xyzs,pos_rgbas);
198 a_action.set_lighting(state.m_GL_LIGHTING);
199 } else {
200 a_action.draw_gsto_vc(mode.value(),npt,pos_xyzs,pos_rgbas);
201 }
202 }
203 } else { //rgbas.empty()
204 if(nms.size()) {
205 size_t npt = xyzs.values().size()/3;
206 bufpos pos_xyzs = 0;
207 bufpos pos_nms = npt*3*sizeof(float);
208 if(gl::is_line(mode.value())) {
209 //Same logic as Inventor SoLightModel.model = BASE_COLOR.
210 a_action.set_lighting(false);
211 a_action.draw_gsto_vn(mode.value(),npt,pos_xyzs,pos_nms);
212 a_action.set_lighting(state.m_GL_LIGHTING);
213 } else {
214 a_action.draw_gsto_vn(mode.value(),npt,pos_xyzs,pos_nms);
215 }
216 } else {
217 size_t npt = xyzs.values().size()/3;
218 bufpos pos = 0;
219 if(gl::is_line(mode.value())) {
220 //Same logic as Inventor SoLightModel.model = BASE_COLOR.
221 a_action.set_lighting(false);
222 a_action.draw_gsto_v(mode.value(),npt,pos);
223 a_action.set_lighting(state.m_GL_LIGHTING);
224 } else {
225 a_action.draw_gsto_v(mode.value(),npt,pos);
226 }
227 }
228 }
229 a_action.end_gsto();
230 return;
231
232 } else { //!_id
233 // use immediate rendering.
234 }
235
236 } else {
237 clean_gstos(&a_action.gl_manager());
238 }
239
240 // immediate rendering :
241 if(rgbas.size()) {
242 if(nms.size()) {
243 if(gl::is_line(mode.value())) {
244 //Same logic as Inventor SoLightModel.model = BASE_COLOR.
245 a_action.set_lighting(false);
246 if(do_back.value())
247 a_action.draw_vertex_color_normal_array(mode.value(),m_back_xyzs,rgbas.values(),m_back_nms);
248 a_action.draw_vertex_color_normal_array(mode.value(),xyzs.values(),rgbas.values(),nms.values());
249 a_action.set_lighting(state.m_GL_LIGHTING);
250 } else if(mode.value()==gl::triangles()) {
251 if(draw_edges.value()) {
252 a_action.color4f(0,0,0,1);
253 a_action.line_width(1);
254 a_action.draw_vertex_array(gl::lines(),m_edges);
255 a_action.set_polygon_offset(true);
256 a_action.color4f(state.m_color);
257 a_action.line_width(state.m_line_width);
258 }
259 if(do_back.value()) a_action.draw_vertex_color_normal_array(mode.value(),m_back_xyzs,rgbas.values(),m_back_nms);
260 a_action.draw_vertex_color_normal_array(mode.value(),xyzs.values(),rgbas.values(),nms.values());
261 a_action.set_polygon_offset(state.m_GL_POLYGON_OFFSET_FILL);
262 } else {
263 if(do_back.value()) a_action.draw_vertex_color_normal_array(mode.value(),m_back_xyzs,rgbas.values(),m_back_nms);
264 a_action.draw_vertex_color_normal_array(mode.value(),xyzs.values(),rgbas.values(),nms.values());
265 }
266
267
268 } else {
269 if(gl::is_line(mode.value())) {
270 //Same logic as Inventor SoLightModel.model = BASE_COLOR.
271 a_action.set_lighting(false);
272 a_action.draw_vertex_color_array(mode.value(),xyzs.values(),rgbas.values());
273 a_action.set_lighting(state.m_GL_LIGHTING);
274 } else {
275 a_action.draw_vertex_color_array(mode.value(),xyzs.values(),rgbas.values());
276 }
277 }
278
279 } else { //rgbas.empty()
280 if(nms.size()) {
281 if(gl::is_line(mode.value())) {
282 //Same logic as Inventor SoLightModel.model = BASE_COLOR.
283 a_action.set_lighting(false);
284 a_action.draw_vertex_normal_array(mode.value(),xyzs.values(),nms.values());
285 a_action.set_lighting(state.m_GL_LIGHTING);
286 } else {
287 a_action.draw_vertex_normal_array(mode.value(),xyzs.values(),nms.values());
288 }
289 } else {
290 if(gl::is_line(mode.value())) {
291 //Same logic as Inventor SoLightModel.model = BASE_COLOR.
292 a_action.set_lighting(false);
293 a_action.draw_vertex_array(mode.value(),xyzs.values());
294 a_action.set_lighting(state.m_GL_LIGHTING);
295 } else {
296 a_action.draw_vertex_array(mode.value(),xyzs.values());
297 }
298 }

509protected:
510 void gen_back(){
511 m_back_xyzs.clear();
512 m_back_nms.clear();
513
514 clean_gstos(); //must reset for all gl_manager.
515
516 std::vector<float>& _xyzs = xyzs.values();
517 std::vector<float>& _nms = nms.values();
518
519 if(_xyzs.empty()) return;
520
521 m_back_xyzs.resize(_xyzs.size(),0);
522 m_back_nms.resize(_nms.size(),0);
523
524 float epsil = epsilon.value();
525
526 if(mode.value()==gl::triangle_fan()) { //reverse after first point.
527
528 m_back_xyzs[0] = _xyzs[0] - _nms[0] * epsil;
529 m_back_xyzs[1] = _xyzs[1] - _nms[1] * epsil;
530 m_back_xyzs[2] = _xyzs[2] - _nms[2] * epsil;
531
532 {std::vector<float>::const_iterator it = _xyzs.begin()+3;
533 std::vector<float>::const_iterator _end = _xyzs.end();
534 std::vector<float>::const_iterator itn = _nms.begin()+3;
535 std::vector<float>::reverse_iterator it2 = m_back_xyzs.rbegin();
536 for(;it!=_end;it2+=3) {
537 *(it2+2) = *it - *itn * epsil; it++;itn++; //x
538 *(it2+1) = *it - *itn * epsil; it++;itn++; //y
539 *(it2+0) = *it - *itn * epsil; it++;itn++; //z
540 }}
541
542 m_back_nms[0] = _nms[0] * -1.0f;
543 m_back_nms[1] = _nms[1] * -1.0f;
544 m_back_nms[2] = _nms[2] * -1.0f;
545
546 {std::vector<float>::const_iterator it = _nms.begin()+3;
547 std::vector<float>::const_iterator _end = _nms.end();
548 std::vector<float>::reverse_iterator it2 = m_back_nms.rbegin();
549 for(;it!=_end;it2+=3) {
550 *(it2+2) = *it * -1.0f; it++;
551 *(it2+1) = *it * -1.0f; it++;
552 *(it2+0) = *it * -1.0f; it++;
553 }}
554
555 } else {
556
557 {std::vector<float>::const_iterator it = _xyzs.begin();
558 std::vector<float>::const_iterator _end = _xyzs.end();
559 std::vector<float>::const_iterator itn = _nms.begin();
560 std::vector<float>::reverse_iterator it2 = m_back_xyzs.rbegin();
561 for(;it!=_end;it2+=3) {
562 *(it2+2) = *it - *itn * epsil; it++;itn++; //x
563 *(it2+1) = *it - *itn * epsil; it++;itn++; //y
564 *(it2+0) = *it - *itn * epsil; it++;itn++; //z
565 }}
566
567 {std::vector<float>::const_iterator it = _nms.begin();
568 std::vector<float>::const_iterator _end = _nms.end();
