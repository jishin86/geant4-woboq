14namespace tools {
15namespace sg {

17class line_style : public node {
18 TOOLS_NODE(line_style,tools::sg::line_style,node)
19public:
20 sf<bool> visible;
21 sf_vec<colorf,float> color;
22 sf<float> width;
23 sf<lpat> pattern;
24public:
25 virtual const std::vector<field_desc>& node_fields() const {
26 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::line_style)
27 static std::vector<field_desc> s_v;
28 if(s_v.empty()) {
29 s_v = parent::node_fields();
30 TOOLS_ADD_FIELD_DESC(visible)
31 TOOLS_ADD_FIELD_DESC(color)
32 TOOLS_ADD_FIELD_DESC(width)
33 TOOLS_ADD_FIELD_DESC(pattern)
34 }
35 return s_v;
36 }
37private:
38 void add_fields(){
39 add_field(&visible);
40 add_field(&color);
41 add_field(&width);
42 add_field(&pattern);
43 }
44public:
45 line_style()
46 :parent()
47 ,visible(true)
48 ,color(colorf_black())
49 ,width(1)
50 ,pattern(line_solid)
51 {
52 add_fields();
53 }
54 virtual ~line_style(){}
55public:
56 line_style(const line_style& a_from)
57 :parent(a_from)
58 ,visible(a_from.visible)
59 ,color(a_from.color)
60 ,width(a_from.width)
61 ,pattern(a_from.pattern)
62 {
63 add_fields();
64 }
65 line_style& operator=(const line_style& a_from){
66 parent::operator=(a_from);
67
68 visible = a_from.visible;
69 color = a_from.color;
70 width = a_from.width;
71 pattern = a_from.pattern;
72 return *this;
73 }
74public:
75 bool from_string(std::ostream& a_out,const cmaps_t& a_cmaps,const std::string& a_s){
76 style_parser sp;
77
78 sp.visible(visible.value());
79 sp.color(color.value());
80 //sp.transparency(transparency.value());
81 sp.line_width(width.value());
82 sp.line_pattern(pattern.value());
83
84 if(!sp.parse(a_out,a_cmaps,a_s)) {
85 a_out << "tools::sg::line_style::from_string :"
86 << " parse failed."
87 << std::endl;
88 return false;
89 }
90
91 visible.value(sp.visible());
92 color.value(sp.color());
93 //transparency.value(sp.transparency());
94 width.value(sp.line_width());
95 pattern.value(sp.line_pattern());
96
97 return true;
98 }
99
100};

102}}
