17class column_booking {
18public:
19 column_booking(const std::string& a_name,cid a_cid,void* a_user_obj)
20 :m_name(a_name)
21 ,m_cid(a_cid)
22 ,m_user_obj(a_user_obj) //WARNING : not owner.
23 {}
24 virtual ~column_booking() {}
25public:
26 column_booking(const column_booking& a_from)
27 :m_name(a_from.m_name)
28 ,m_cid(a_from.m_cid)
29 ,m_user_obj(a_from.m_user_obj)
30 {}
31 column_booking& operator=(const column_booking& a_from) {
32 if(&a_from==this) return *this;
33 m_name = a_from.m_name;
34 m_cid = a_from.m_cid;
35 m_user_obj = a_from.m_user_obj;
36 return *this;
37 }
38public:
39 const std::string& name() const {return m_name;}
40 cid cls_id() const {return m_cid;}
41 void* user_obj() const {return m_user_obj;}
42 void set_user_obj(void* a_obj) {m_user_obj = a_obj;}
43protected:
44 std::string m_name;
45 cid m_cid;
46 void* m_user_obj;
47};

49class ntuple_booking {
50public:
51 ntuple_booking(const std::string& a_name = "",const std::string& a_title = "")
52 :m_name(a_name)
53 ,m_title(a_title)
54 {}
55 virtual ~ntuple_booking(){}
56public:
57 ntuple_booking(const ntuple_booking& a_from)
58 :m_name(a_from.m_name)
59 ,m_title(a_from.m_title)
60 ,m_columns(a_from.m_columns)
61 {}
62 ntuple_booking& operator=(const ntuple_booking& a_from){
63 m_name = a_from.m_name;
64 m_title = a_from.m_title;
65 m_columns = a_from.m_columns;
66 return *this;
67 }
68public:
69 template <class T>
70 void add_column(const std::string& a_name) {
71 m_columns.push_back(column_booking(a_name,_cid(T()),0));
72 }
73 template <class T>
74 void add_column_vec(const std::string& a_name) {
75 m_columns.push_back(column_booking(a_name,_cid_std_vector<T>(),0));
76 }
77
78 template <class T>
79 void add_column(const std::string& a_name,T& a_user) {
80 m_columns.push_back(column_booking(a_name,_cid(T()),(void*)&a_user));
81 }
82 template <class T>
83 void add_column(const std::string& a_name,std::vector<T>& a_user_vec) {
84 m_columns.push_back(column_booking(a_name,_cid_std_vector<T>(),(void*)&a_user_vec));
85 }
86
87 const std::string& name() const {return m_name;}
88 const std::string& title() const {return m_title;}
89 const std::vector<column_booking>& columns() const {return m_columns;}
90 std::vector<column_booking>& columns() {return m_columns;}
91
92 void set_name(const std::string& a_s) {m_name = a_s;}
93 void set_title(const std::string& a_s) {m_title = a_s;}
94
95 bool has_similar_layout(std::ostream& a_out,const ntuple_booking& a_nbk) {
96 if(m_columns.size()!=a_nbk.m_columns.size()) {
97 a_out << "tools::ntuple_booking::has_similar_layout :"
98 << " bookings have not the same number of columns."
99 << " (" << m_columns.size() << " != " << a_nbk.m_columns.size() << ")."
100 << std::endl;
101 return false;
102 }
103 std::vector<column_booking>::const_iterator ait = a_nbk.m_columns.begin();
104 tools_vforit(tools::column_booking,m_columns,it) {
105 if((*it).name()!=(*ait).name()) {
106 a_out << "tools::ntuple_booking::has_similar_layout :"
107 << " columns don't have same name."
108 << " (" << (*it).name() << " != " << (*ait).name() << ")."
109 << std::endl;
110 return false;
111 }
112 if((*it).cls_id()!=(*ait).cls_id()) {
113 a_out << "tools::ntuple_booking::has_similar_layout :"
114 << " columns don't have same class id."
115 << " (" << (*it).cls_id() << " != " << (*ait).cls_id() << ")."
116 << std::endl;
117 return false;
118 }
119
120 ait++;
121 }
122 return true;
123 }
124
125protected:
126 std::string m_name;
127 std::string m_title;
128 std::vector<column_booking> m_columns;
129};
