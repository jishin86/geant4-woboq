37inline void str_del(char*& a_cstr) {
38 if(a_cstr==NULL) return;
39 ::free(a_cstr);
40#ifdef TOOLS_MEM
41 mem::decrement(s_cstr().c_str());
42#endif
43 a_cstr = NULL;

46inline char* str_new(size_t a_l = 0,char a_char = ' ') {
47 char* s = (char*)::malloc((a_l+1)*sizeof(char));
48 if(s==NULL) return NULL;
49#ifdef TOOLS_MEM
50 mem::increment(s_cstr().c_str());
51#endif
52 char* pos = s;
53 for(size_t c=0;c<a_l;c++,pos++) *pos = a_char;
54 *(s+a_l) = 0;
55 return s;
56}

58inline bool str_cat(char*& a_1,const char a_c) {
59 size_t l1 = ::strlen(a_1);
60 char* s = (char*)::malloc(l1+1+1);
61 if(!s) return false;
62#ifdef TOOLS_MEM
63 mem::increment(s_cstr().c_str());
64#endif
65 ::memcpy(s,a_1,l1);
66 ::memcpy(s+l1,&a_c,1);
67 *(s+l1+1) = 0;
68 ::free(a_1);
69#ifdef TOOLS_MEM
70 mem::decrement(s_cstr().c_str());
71#endif
72 a_1 = s;
73 return true;
74}

76inline bool str_cat(char*& a_1,const char* a_2) {
77 size_t l1 = ::strlen(a_1);
78 size_t l2 = ::strlen(a_2);
79 char* s = (char*)::malloc(l1+l2+1);
80 if(!s) return false;
81#ifdef TOOLS_MEM
82 mem::increment(s_cstr().c_str());
83#endif
84 ::memcpy(s,a_1,l1);
85 ::memcpy(s+l1,a_2,l2);
86 *(s+l1+l2) = 0;
87 ::free(a_1);
88#ifdef TOOLS_MEM
89 mem::decrement(s_cstr().c_str());
90#endif
91 a_1 = s;
92 return true;
93}

95inline void str_rev(char* a_s) {
96 size_t l = ::strlen(a_s);
97 size_t hl = l/2;
98 char* beg = a_s;
99 char* end = a_s+l-1;
100 for(size_t i=0;i<hl;i++) {
101 char c = *end;
102 *end = *beg;
103 *beg = c;
104 beg++;end--;
105 }
106}

108inline char* str_sub(const char* a_s,
109 unsigned int a_pos,
110 unsigned int a_sz = 0) { //0 = take up end.
111 size_t l = ::strlen(a_s);
112 if(a_pos>=l) return 0; //throw std::out_of_range
113 size_t ls;
114 if(a_sz) {
115 ls = (a_sz<(l-a_pos)?a_sz:(l-a_pos)); //min(a_sz,l-a_pos)
116 } else {
117 ls = l-a_pos;
118 }
119 char* s = (char*)::malloc(ls+1);
120 if(!s) return 0;
121#ifdef TOOLS_MEM
122 mem::increment(s_cstr().c_str());
123#endif
124 //abcdefgh l=8
125 //0123456789
126 ::memcpy(s,a_s+a_pos,ls);
127 *(s+ls) = 0;
128 return s;
129}

131inline char* str_rep(const char* a_s,unsigned int a_pos,unsigned int a_sz,const char* a_new) {
132 //not tested yet.
133 size_t las = ::strlen(a_s);
134 if(a_pos>=las) return 0; //throw std::out_of_range
135 if(a_pos+a_sz>las) return 0;
136 size_t lan = ::strlen(a_new);
137 unsigned int num = a_sz<lan?a_sz:(unsigned int)lan;
138 //abcdefghij : l = 10
139 //0123456789
140 // p
141 size_t le = las-(a_pos+a_sz);
142 size_t ls = a_pos+num+le;
143 char* s = (char*)::malloc(ls+1);
144 if(!s) return 0;
145#ifdef TOOLS_MEM
146 mem::increment(s_cstr().c_str());
147#endif
148 ::memcpy(s,a_s,a_pos);
149 ::memcpy(s+a_pos,a_new,num);
150 if(le) ::memcpy(s+a_pos+num,a_s+a_pos+a_sz,le);
151 *(s+ls) = 0;
152 return s;
153}

168inline char* beg_LC_NUMERIC() {
169 char* _sl = ::setlocale(LC_NUMERIC,0);
170 char* old = _sl?str_dup(_sl):0;
171 ::setlocale(LC_NUMERIC,"C");
172 return old;
173}
174inline void end_LC_NUMERIC(char*& a_s) {
175 if(a_s) {
176 ::setlocale(LC_NUMERIC,a_s);
177 str_del(a_s);
178 }
179}
180
181inline bool str_2d(const char* a_s,double& a_v) {
182 char* olcn = beg_LC_NUMERIC();
183
184 char* end;
185 a_v = ::strtod(a_s,&end);
186 if(end==a_s) {
187 a_v = 0;
188 end_LC_NUMERIC(olcn);
189 return false;
190 }
191
192 end_LC_NUMERIC(olcn);
193 return true;
194}
