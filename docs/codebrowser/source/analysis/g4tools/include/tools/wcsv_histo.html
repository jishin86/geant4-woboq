14template <class AXIS>
15inline void axis_to(std::ostream& a_writer,const AXIS& a_axis,char a_hc) {
16 if(a_axis.m_fixed) {
17 a_writer << a_hc
18 << "axis fixed " << a_axis.m_number_of_bins
19 << " " << a_axis.m_minimum_value
20 << " " << a_axis.m_maximum_value
21// << " " << a_axis.m_bin_width
22 << std::endl;
23 } else {
24 a_writer << a_hc << "axis edges";
25 for(unsigned int iedge=0;iedge<a_axis.m_edges.size();iedge++) {
26 a_writer << " " << a_axis.m_edges[iedge];
27 }
28 a_writer << std::endl;
29 }
30}

32template <class ANNOTATION>
33inline void annotations_to(std::ostream& a_writer,const ANNOTATION& a_ans,char a_hc) {
34 typename ANNOTATION::const_iterator it;
35 for(it=a_ans.begin();it!=a_ans.end();++it) {
36 a_writer << a_hc << "annotation " << (*it).first << " " << (*it).second << std::endl;
37 }
38}

40template <class HIST>
41inline bool hto(std::ostream& a_writer,const std::string& a_class,const HIST& a_h,
42 char a_sep = ',',char a_hc = '#',bool a_header = true) {
43 if(a_header) {
44 a_writer << a_hc << "class " << a_class << std::endl;
45 a_writer << a_hc << "title " << a_h.title() << std::endl;
46 a_writer << a_hc << "dimension " << a_h.dimension() << std::endl;
47 for(unsigned int iaxis=0;iaxis<a_h.dimension();iaxis++) axis_to(a_writer,a_h.get_axis(iaxis),a_hc);
48 {const std::vector<typename HIST::coordinate_t>& planes = a_h.in_range_planes_xyw();
49 if(planes.size()) {
50 a_writer << a_hc << "planes_Sxyw";
51 for(unsigned int iplane=0;iplane<planes.size();iplane++) a_writer << " " << planes[iplane];
52 a_writer << std::endl;
53 }}
54 annotations_to(a_writer,a_h.annotations(),a_hc);
55 a_writer << a_hc << "bin_number " << a_h.get_bins() << std::endl;
56 }
57 {a_writer << "entries" << a_sep << "Sw" << a_sep << "Sw2";
58 for(unsigned int iaxis=0;iaxis<a_h.dimension();iaxis++) {
59 a_writer << a_sep << "Sxw" << iaxis << a_sep << "Sx2w" << iaxis;
60 }
61 a_writer << std::endl;}
62
63 typedef typename HIST::coordinate_t coordinate_t;
64 typedef typename HIST::dim_t dim_t;
65 typedef typename HIST::offset_t offset_t;
66 typedef typename HIST::num_entries_t num_entries_t;
67 typedef typename HIST::weight_t weight_t;
68
69 dim_t _dim = a_h.dimension();
70 offset_t _bins = a_h.get_bins();
71
72 const std::vector<num_entries_t>& _bin_entries = a_h.bins_entries();
73 const std::vector<weight_t>& _bin_Sw = a_h.bins_sum_w();
74 const std::vector<weight_t>& _bin_Sw2 = a_h.bins_sum_w2();
75 const std::vector< std::vector<coordinate_t> >& _bin_Sxw = a_h.bins_sum_xw();
76 const std::vector< std::vector<coordinate_t> >& _bin_Sx2w = a_h.bins_sum_x2w();
77
78 for(unsigned int i=0;i<_bins;i++) {
79 a_writer << _bin_entries[i] << a_sep << _bin_Sw[i] << a_sep << _bin_Sw2[i];
80 for(unsigned int iaxis=0;iaxis<_dim;iaxis++) {
81 a_writer << a_sep << _bin_Sxw[i][iaxis] << a_sep << _bin_Sx2w[i][iaxis];
82 }
83 a_writer << std::endl;
84 }
85
86 a_h.not_a_profile(); //trick to be sure to use this function on an histo and not a profile.
87
88 return true;
89}

91template <class PROF>
92inline bool pto(std::ostream& a_writer,const std::string& a_class,const PROF& a_prof,
93 char a_sep = ',',char a_hc = '#',bool a_header = true) {
94 if(a_header) {
95 a_writer << a_hc << "class " << a_class << std::endl;
96 a_writer << a_hc << "title " << a_prof.title() << std::endl;
97 a_writer << a_hc << "dimension " << a_prof.dimension() << std::endl;
98 for(unsigned int iaxis=0;iaxis<a_prof.dimension();iaxis++) axis_to(a_writer,a_prof.get_axis(iaxis),a_hc);
99 {const std::vector<typename PROF::coordinate_t>& planes = a_prof.in_range_planes_xyw();
100 if(planes.size()) {
101 a_writer << a_hc << "planes_Sxyw";
102 for(unsigned int iplane=0;iplane<planes.size();iplane++) a_writer << " " << planes[iplane];
103 a_writer << std::endl;
104 }}
105 annotations_to(a_writer,a_prof.annotations(),a_hc);
106 a_writer << a_hc << "cut_v " << (a_prof.cut_v()?"true":"false") << std::endl;
107 a_writer << a_hc << "min_v " << a_prof.min_v() << std::endl;
108 a_writer << a_hc << "max_v " << a_prof.max_v() << std::endl;
109 a_writer << a_hc << "bin_number " << a_prof.get_bins() << std::endl;
110 }
111 {a_writer << "entries" << a_sep << "Sw" << a_sep << "Sw2" << a_sep << "Svw" << a_sep << "Sv2w";
112 for(unsigned int iaxis=0;iaxis<a_prof.dimension();iaxis++) {
113 a_writer << a_sep << "Sxw" << iaxis << a_sep << "Sx2w" << iaxis;
114 }
115 a_writer << std::endl;}
116
117 typedef typename PROF::coordinate_t coordinate_t;
118 typedef typename PROF::dim_t dim_t;
119 typedef typename PROF::offset_t offset_t;
120 typedef typename PROF::num_entries_t num_entries_t;
121 typedef typename PROF::weight_t weight_t;
122
123 dim_t _dim = a_prof.dimension();
124 offset_t _bins = a_prof.get_bins();
125
126 const std::vector<num_entries_t>& _bin_entries = a_prof.bins_entries();
127 const std::vector<weight_t>& _bin_Sw = a_prof.bins_sum_w();
128 const std::vector<weight_t>& _bin_Sw2 = a_prof.bins_sum_w2();
129 const std::vector< std::vector<coordinate_t> >& _bin_Sxw = a_prof.bins_sum_xw();
130 const std::vector< std::vector<coordinate_t> >& _bin_Sx2w = a_prof.bins_sum_x2w();
131
132 typedef typename PROF::vs_t vs_t;
133 const vs_t& _bin_Svw = a_prof.bins_sum_vw();
134 const vs_t& _bin_Sv2w = a_prof.bins_sum_v2w();
135
136 for(unsigned int i=0;i<_bins;i++) {
137 a_writer << _bin_entries[i] << a_sep << _bin_Sw[i] << a_sep << _bin_Sw2[i]
138 << a_sep << _bin_Svw[i] << a_sep << _bin_Sv2w[i];
139 for(unsigned int iaxis=0;iaxis<_dim;iaxis++) {
140 a_writer << a_sep << _bin_Sxw[i][iaxis] << a_sep << _bin_Sx2w[i][iaxis];
141 }
142 a_writer << std::endl;
143 }
144
145 return true;
146}
