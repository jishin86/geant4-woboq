12enum what { leading, trailing, both };
13
14inline bool strip(std::string& a_string,what a_type = both,char a_char = ' '){
15 //return true = some stripping had been done.
16
17 std::string::size_type l = a_string.length();
18 if(l==0) return false;
19
20 switch ( a_type ) {
21 case leading:{
22 char* pos = (char*)a_string.c_str();
23 for(std::string::size_type i=0;i<l;i++,pos++) {
24 if(*pos!=a_char) {
25 a_string = a_string.substr(i,l-i);
26 return (i?true:false); //i=0 : same string.
27 }
28 }
29 }break;
30 case trailing:{
31 char* pos = (char*)a_string.c_str();
32 pos += (l-1);
33 for(std::string::size_type i=l-1;;i--,pos--) {
34 if(*pos!=a_char) {
35 a_string = a_string.substr(0,i+1);
36 return (i==(l-1)?false:true); //i==l-1 : same string.
37 }
38 }
39 }break;
40 case both:{
41 bool stat_lead = strip(a_string,leading,a_char);
42 bool stat_trail = strip(a_string,trailing,a_char);
43 if(stat_lead) return true;
44 if(stat_trail) return true;
45 }break;
46 //default:break;
47 }
48 return false; //nothing done.
49}

51#ifdef TOOLS_DEPRECATED
52inline std::string strp(const std::string& a_string,what a_type = both,char a_char = ' '){
53 std::string s(a_string);
54 strip(s,a_type,a_char);
55 return s;
56}
57#endif //TOOLS_DEPRECATED

59inline bool strip(std::vector<std::string>& a_strings,what a_type = both,char a_char = ' ') {
60 bool some_done = false;
61 std::vector<std::string>::iterator it;
62 for(it=a_strings.begin();it!=a_strings.end();++it) {
63 if(strip(*it,a_type,a_char)) some_done = true;
64 }
65 return some_done;
66}
