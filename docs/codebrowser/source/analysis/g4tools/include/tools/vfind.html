12template <class T>
13inline T* find_named(const std::vector<T*>& a_vec,const std::string& a_name) {
14 typedef typename std::vector<T*>::const_iterator it_t;
15 it_t it;
16 for(it=a_vec.begin();it!=a_vec.end();++it) {
17 if((*it)->name()==a_name) return *it;
18 }
19 return 0;
20}

38namespace tools {

40template <class T>
41inline T* find_named_case_insensitive(const std::vector<T*>& a_vec,const std::string& a_name) {
42 std::string low_a_name = a_name;
43 tolowercase(low_a_name);
44 std::string low_name;
45 typedef typename std::vector<T*>::const_iterator it_t;
46 it_t it;
47 for(it=a_vec.begin();it!=a_vec.end();++it) {
48 low_name = (*it)->name();
49 tolowercase(low_name);
50 if(low_name==low_a_name) return *it;
51 }
52 return 0;
53}
