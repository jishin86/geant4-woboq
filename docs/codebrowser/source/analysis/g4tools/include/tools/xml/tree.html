20// A tree is :
21// <tree atb1="" atb2="" ...>
22// ...
23// <tree...>
24// </tree>
25// ...
26// <element atb1="" atb2="" ...> value </element>
27// ...
28// </tree>
29//
30// tree.attribute_value(<name>,s)
31// retrieve the value of atb <name> of a <tag>
32// tree.attribute_value(<element>,<name>,s)
33// retrieve the value of an atb <name> of a <element> of a <tree>
34//

320 static bool check_item(tree& a_tree,void* a_tag){
321 empty_visitor* visitor = (empty_visitor*)a_tag;
322
323 {const std::vector<atb>& atbs = a_tree.attributes();
324 size_t atbn = atbs.size();
325 for(size_t index=0;index<atbn;index++) {
326 const std::string& _atb = atbs[index].first;
327 const std::string& atbv = atbs[index].second;
328 if(atbv.empty()) {
329 visitor->f_out << "check_item :"
330 << " for XML item " << sout(a_tree.tag_name())
331 << ", attribute " << sout(_atb) << " has an empty value."
332 << std::endl;
333 visitor->m_status = false;
334 }
335 }}
336
337 {tools_lforcit(ielem*,a_tree.m_childs,it) {
338 if(element* _elem = id_cast<ielem,element>(*(*it))) {
339 const std::vector<atb>& atbs = _elem->attributes();
340 size_t atbn = atbs.size();
341 for(size_t index=0;index<atbn;index++) {
342 const std::string& _atb = atbs[index].first;
343 const std::string& atbv = atbs[index].second;
344 if(atbv.empty()) {
345 visitor->f_out << "ItemM::check_item :"
346 << " for XML item " << sout(a_tree.tag_name())
347 << ", attribute " << sout(_atb) << " has an empty value."
348 << std::endl;
349 visitor->m_status = false;
350 }
351 }
352 }}}
353
354 return true;
355 }

482 bool copy(const tree& a_from,
483 copy_what a_what = copy_all,
484 bool a_clear = true){
485
486 // Copy data (atbs, propertis, children) of a_from onto this.
487 if((a_what==copy_all)||(a_what==copy_attributes)) {
488 if(a_clear) m_atbs.clear();
489 tools_vforcit(atb,a_from.m_atbs,it) m_atbs.push_back(*it);
490 }
491
492 if((a_what==copy_all)||(a_what==copy_elements)) {
493 if(a_clear) delete_sub_elems();
494 tools_lforcit(ielem*,a_from.m_childs,it) {
495 if(element* _elem = id_cast<ielem,element>(*(*it))) {
496 m_childs.push_back(new element(*_elem));
497 }
498 }
499 }
500
501 if((a_what==copy_all)||(a_what==copy_children)) {
502 if(a_clear) delete_sub_trees();
503 tools_lforcit(ielem*,a_from.m_childs,it) {
504 if(tree* _tree = id_cast<ielem,tree>(*(*it))) {
505 //FIXME : could we have mismatch parent/child ?
506 tree* obj = _tree->create_copy(this);
507 if(!obj) {
508 // Something wrong, cleanup this.
509 clear();
510 return false;
511 }
512 add_child(obj);
513 }
514 }
515 }
516
517 //FIXME : m_save
518
519 return true;
520 }

575 tree* find_by_attribute(const std::string& a_atb,const std::string& a_value,
576 bool a_up_down = true,bool a_left_right = true) const {
577 if(a_up_down) {
578 std::string s;
579 attribute_value(a_atb,s);
580 if(s==a_value) return const_cast<tree*>(this);
581
582 if(a_left_right) {
583 tools_lforcit(ielem*,m_childs,it) {
584 if(tree* _tree = id_cast<ielem,tree>(*(*it))) {
585 tree* itemML =
586 _tree->find_by_attribute(a_atb,a_value,a_up_down,a_left_right);
587 if(itemML) return itemML;
588 }
589 }
590 } else {
591 std::list<ielem*>::const_reverse_iterator it;
592 for(it=m_childs.rbegin();it!=m_childs.rend();++it) {
593 if(tree* _tree = id_cast<ielem,tree>(*(*it))) {
594 tree* itemML =
595 _tree->find_by_attribute(a_atb,a_value,a_up_down,a_left_right);
596 if(itemML) return itemML;
597 }}
598 }
599 } else {
600 if(a_left_right) {
601 tools_lforcit(ielem*,m_childs,it) {
602 if(tree* _tree = id_cast<ielem,tree>(*(*it))) {
603 tree* itemML =
604 _tree->find_by_attribute(a_atb,a_value,a_up_down,a_left_right);
605 if(itemML) return itemML;
606 }}
607 } else {
608 std::list<ielem*>::const_reverse_iterator it;
609 for(it=m_childs.rbegin();it!=m_childs.rend();++it) {
610 if(tree* _tree = id_cast<ielem,tree>(*(*it))) {
611 tree* itemML =
612 _tree->find_by_attribute(a_atb,a_value,a_up_down,a_left_right);
613 if(itemML) return itemML;
614 }}
615 }
616
617 std::string s;
618 attribute_value(a_atb,s);
619 if(s==a_value) return const_cast<tree*>(this);
620
621 }
622 return 0;

625 void dump_xml(std::ostream& a_out,const std::string& a_spaces = "") {
626
627 // begin tag :
628 a_out << a_spaces << "<" << m_tag_name;
629 {size_t atbn = m_atbs.size();
630 for(size_t index=0;index<atbn;index++) {
631 a_out << " " << m_atbs[index].first << "="
632 << sout(m_atbs[index].second);
633 }}
634 a_out << ">" << std::endl;
635
636 {tools_lforcit(ielem*,m_childs,it) {
637 if(element* _elem = id_cast<ielem,element>(*(*it))) {
638
639 const std::vector<atb>& atbs = _elem->attributes();
640
641 bool isCallback = false;
642
643 a_out << a_spaces << " <" << _elem->name();
644 size_t atbn = atbs.size();
645 for(size_t index=0;index<atbn;index++) {
646 a_out << " " << atbs[index].first << "="
647 << sout(atbs[index].second);
648 if(atbs[index].first=="exec") isCallback = true;
649 }
650 if(_elem->value().empty()) {
651 a_out << "/>" << std::endl;
652 } else {
653 a_out << ">";
654 std::string value = to_xml(_elem->value());
655 if(isCallback) {
656 if(value.find("\\n\\")==std::string::npos) {
657 a_out << value;
658 } else {
659 a_out << std::endl;
660 replace_(value,"\\n\\","@OnX@");
661 replace_(value,"@OnX@","\\n\\\n");
662 strip(value,trailing,' ');
663 a_out << value;
664 size_t l = value.size();
665 if(l && value[l-1]!='\n') a_out << std::endl;
666 a_out << a_spaces << " ";
667 }
668 } else {
669 a_out << value;
670 }
671 a_out << "</" << _elem->name() << ">" << std::endl;
672 }
673 }}}
674
675 // End tag :
676 a_out << a_spaces << "</" << m_tag_name << ">" << std::endl;
677 }

681 const std::string& a_value,int a_index = 0){
682 if(a_elem.empty()) {
683 size_t linen = m_atbs.size();
684 for(size_t count=0;count<linen;count++) {
685 if(m_atbs[count].first==a_atb) {

687 return true;
688 }
689 }
690 // Not found, add it :
691 m_atbs.push_back(atb(a_atb,a_value));
692 return true;
693 } else {
694 int index = 0;
695 tools_lforcit(ielem*,m_childs,it) {
696 if(element* _elem = id_cast<ielem,element>(*(*it))) {
697 if(a_elem==_elem->name()) {
698 if(index==a_index) {
699 _elem->set_attribute_value(a_atb,a_value);

806 void update_tree(const tree& a_old) {
807 // We map the opened elements in the old tree within this tree.
808 // Algorithm based on the existence of a "label" element.
809 tools_lforit(ielem*,m_childs,it) {
810 if(tree* _tree = id_cast<ielem,tree>(*(*it))) {
811 std::string slabel;
812 if(_tree->element_value("label",slabel)) {
813 // Try to find a same label in the same level :
814 const tree* item =
815 a_old.find_by_element_in_same_level("label",slabel);
816 if(item) {
817 bool sopened;
818 if(item->element_value_boolean("opened",sopened) && sopened) {
819 _tree->set_element_value("opened","true");
820 }
821 if(a_old.number_of_trees()) _tree->update_tree(*item);
822 }
823 }
824 }}
