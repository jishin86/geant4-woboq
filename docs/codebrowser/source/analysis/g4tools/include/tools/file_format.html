111inline std::string ext_fmt(const std::string& a_ext) {
112 if(a_ext==s_ext_fits()) return s_format_fits();
113 if(a_ext==s_ext_aida()) return s_format_aida();
114 if(a_ext==s_ext_root()) return s_format_root();
115 if(a_ext==s_ext_csv()) return s_format_csv();
116 if(a_ext==s_ext_tnt()) return s_format_hippo();
117 if(a_ext==s_ext_hiptxt()) return s_format_hippo();
118 if(a_ext==s_ext_hdf5()) return s_format_hdf5();
119 if(a_ext==s_ext_jpg()) return s_format_jpeg();
120 if(a_ext==s_ext_png()) return s_format_png();
121 if(a_ext==s_ext_fog()) return s_format_fog();
122 if(a_ext==s_ext_dot()) return s_format_dot();
123 if(a_ext==s_ext_dcm()) return s_format_dcm();
124 if(a_ext==s_ext_iv()) return s_format_iv();
125 if(a_ext==s_ext_hiv()) return s_format_iv();
126 if(a_ext==s_ext_wrl()) return s_format_wrl();
127 if(a_ext==s_ext_exsg()) return s_format_exsg();
128 if(a_ext==s_ext_bsg()) return s_format_bsg();
129 if(a_ext==s_ext_scenarios()) return s_format_scenarios();
130 if(a_ext==s_ext_slides()) return s_format_slides();
131 if(a_ext==s_ext_zvid()) return s_format_zvid();
132 if(a_ext==s_ext_gdml()) return s_format_gdml();
133 if(a_ext==s_ext_mac()) return s_format_mac();
134 if(a_ext==s_ext_cmnd()) return s_format_cmnd();
135 if(a_ext==s_ext_jive()) return s_format_jive();
136 if(a_ext==s_ext_heprep()) return s_format_heprep();
137 if(a_ext==s_ext_zheprep()) return s_format_zheprep();
138 if(a_ext==s_ext_lua()) return s_format_lua();
139 if(a_ext==s_ext_py()) return s_format_py();
140 return s_format_guessed();
141}
