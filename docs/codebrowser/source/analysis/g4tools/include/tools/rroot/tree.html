69 bool find_entry(uint64 a_entry,uint32& a_nbytes) {
70 a_nbytes = 0;
71 if(a_entry>=m_entries) return false;
72 int nbytes = 0;
73 //fReadEntry = a_entry;
74 tools_vforit(branch*,m_branches,it) {
75 uint32 n;
76 if(!(*it)->find_entry(m_file,a_entry,n)) return false;
77 nbytes += n;
78 }
79 a_nbytes = nbytes;
80 return true;
81 }

410 static branch* _find_branch(const std::vector<branch*>& a_bs,const std::string& a_name,bool a_recursive) {
411 tools_vforcit(branch*,a_bs,it) {
412 if(rcmp((*it)->name(),a_name)) return *it;
413 if(a_recursive) {
414 branch* br = _find_branch((*it)->branches(),a_name,a_recursive);
415 if(br) return br;
416 }
417 }
418 return 0;
419 }
420
421 static base_leaf* _find_leaf(const std::vector<branch*>& a_bs,const std::string& a_name,bool a_recursive) {
422 tools_vforcit(branch*,a_bs,it) {
423 tools_vforcit(base_leaf*,(*it)->leaves(),itl) {
424 if(rcmp((*itl)->name(),a_name)) return *itl;
425 }
426 if(a_recursive) {
427 base_leaf* lf = _find_leaf((*it)->branches(),a_name,a_recursive);
428 if(lf) return lf;
429 }
430 }
431 return 0;
432 }
