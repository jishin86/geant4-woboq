56template <class T>
57class leaf : public base_leaf {
58public:
59 static const std::string& s_class() {
60 static const std::string s_v("tools::rroot::leaf<"+stype(T())+">");
61 return s_v;
62 }
63public: //iro
64 virtual void* cast(const std::string& a_class) const {
65 if(void* p = cmp_cast< leaf<T> >(this,a_class)) {return p;}
66 return base_leaf::cast(a_class);
67 }
68 virtual const std::string& s_cls() const {return s_class();}
69public:
70 static cid id_class() {return base_leaf_cid()+_cid(T());}
71 virtual void* cast(cid a_class) const {
72 if(void* p = cmp_cast<leaf>(this,a_class)) {return p;}
73 return base_leaf::cast(a_class);
74 }
75public:
76 virtual iro* copy() const {return new leaf<T>(*this);}
77 virtual bool stream(buffer& a_buffer) {
78 short v;
79 unsigned int s,c;
80 if(!a_buffer.read_version(v,s,c)) return false;
81 if(!base_leaf::stream(a_buffer)) return false;
82 if(!a_buffer.read(m_min)) return false;
83 if(!a_buffer.read(m_max)) return false;
84 if(!a_buffer.check_byte_count(s,c,leaf_store_class(T()))) return false;
85 return true;
86 }
87public: //base_leaf
88 virtual bool read_buffer(buffer& a_buffer) {
89 if(m_leaf_count) {
90 leaf<int>* leaf_i = safe_cast<base_leaf, leaf<int> >(*m_leaf_count);
91 if(!leaf_i) {
92 m_out << "tools::rroot::leaf::read_buffer : leaf_count not a leaf<int>." << std::endl;
93 return false;
94 }
95 int len;
96 if(!leaf_i->value(0,len)) {
97 m_out << "tools::rroot::leaf::read_buffer : leaf<int>.value() failed."
98 << " m_leaf_count " << m_leaf_count
99 << " leaf_i " << leaf_i
100 << " Name " << sout(leaf_i->name())
101 << " Size " << leaf_i->num_elem() << std::endl;
102 return false;
103 }
104
105 if (len > leaf_i->get_max()) { //protection.
106 m_out << "tools::rroot::leaf::read_buffer : warning : " << sout(name())
107 << ", len = " << len << " > max = "
108 << leaf_i->get_max() << std::endl;
109 len = leaf_i->get_max();
110 }
111
112 uint32 ndata = len * m_length;
113
114 //if(!ndata) {
115 // delete [] m_value;
116 // m_value = new T[1];
117 // m_size = 0;
118 // return true;
119 //}
120
121 if(ndata>m_size) {
122 delete [] m_value;
123 m_value = new T[ndata];
124 }
125
126 m_size = ndata;
127 if(!a_buffer.read_fast_array(m_value,ndata)) {
128 m_out << "tools::rroot::leaf::read_buffer : \"" << name() << "\" :"
129 << " read_fast_array failed."
130 << std::endl;
131 return false;
132 }
133 return true;
134
135 } else {
136 if(m_length) {
137 if(m_length>m_size) {
138 delete [] m_value;
139 m_value = new T[m_length];
140 }
141 m_size = m_length;
142 if(!a_buffer.read_fast_array<T>(m_value,m_length)) {
143 m_out << "tools::rroot::leaf::read_buffer :"
144 << " read_fast_array failed. m_length " << m_length
145 << std::endl;
146 return false;
147 }
148 return true;
149 } else {
150 m_out << "tools::rroot::leaf::read_buffer :"
151 << " read_fast_array failed. m_length is zero."
152 << std::endl;
153 return false;
154 }
155 }
156 return true;
157 }
158 virtual bool print_value(std::ostream& a_out,uint32 a_index) const {
159 if(!m_value) return false;
160 if(a_index>=m_size) return false;
161 a_out << m_value[a_index];
162 return true;
163 }
164//virtual uint32 num_elem() const {return m_length;}
165 virtual uint32 num_elem() const {return m_size;}
166public:
167 leaf(std::ostream& a_out,ifac& a_fac)
168 :base_leaf(a_out,a_fac)
169 ,m_min(T()),m_max(T())
170 ,m_value(0),m_size(0)
171 {}
172 virtual ~leaf(){
173 delete [] m_value;
174 }
175protected:
176 leaf(const leaf& a_from)
177 :iro(a_from)
178 ,base_leaf(a_from)
179 ,m_min(T()),m_max(T())
180 ,m_value(0),m_size(0)
181 {}
182 leaf& operator=(const leaf&){return *this;}
183public:
184 bool value(uint32 a_index,T& a_value) const {
185 if(!m_value) {a_value = T();return false;}
186 if(a_index>=m_size) {a_value = T();return false;}
187 a_value = m_value[a_index];
188 return true;
189 }
190 bool value(std::vector<T>& a_v) const {
191 if(!m_value) {a_v.clear();return false;}
192 a_v.resize(m_size);
193 for(uint32 index=0;index<m_size;index++) a_v[index] = m_value[index];
194 return true;
195 }
196 T get_max() const {return m_max;}
197 //uint32 size() const {return m_size;}
198protected:
199 T m_min; //Minimum value if leaf range is specified
200 T m_max; //Maximum value if leaf range is specified
201 T* m_value; //!Pointer to data buffer
202 uint32 m_size; //size of m_value array.
203};

205class leaf_string : public base_leaf {
206 static const std::string& s_store_class() {
207 static const std::string s_v("TLeafC");
208 return s_v;
209 }
210public:
211 static const std::string& s_class() {
212 static const std::string s_v("tools::rroot::leaf_string");
213 return s_v;
214 }
215public: //iro
216 virtual void* cast(const std::string& a_class) const {
217 if(void* p = cmp_cast<leaf_string>(this,a_class)) {return p;}
218 return base_leaf::cast(a_class);
219 }
220 virtual const std::string& s_cls() const {return s_class();}
221public:
222 static cid id_class() {return leaf_string_cid();}
223 virtual void* cast(cid a_class) const {
224 if(void* p = cmp_cast<leaf_string>(this,a_class)) {return p;}
225 return base_leaf::cast(a_class);
226 }
227public:
228 virtual iro* copy() const {return new leaf_string(*this);}
229 virtual bool stream(buffer& a_buffer) {
230 short v;
231 unsigned int s,c;
232 if(!a_buffer.read_version(v,s,c)) return false;
233 if(!base_leaf::stream(a_buffer)) return false;
234 if(!a_buffer.read(m_min)) return false;
235 if(!a_buffer.read(m_max)) return false;
236 if(!a_buffer.check_byte_count(s,c,s_store_class())) return false;
237 return true;
238 }
239public: //base_leaf
240 virtual bool read_buffer(buffer& a_buffer) {
241 delete [] m_value;
242 m_value = 0;
243
244 unsigned char lenchar;
245 if(!a_buffer.read(lenchar)) {
246 m_out << "tools::rroot::leaf_string::read_buffer :"
247 << " read(uchar) failed."
248 << std::endl;
249 return false;
250 }
251 uint32 len = 0;
252 if(lenchar < 255) {
253 len = lenchar;
254 } else {
255 if(!a_buffer.read(len)) {
256 m_out << "tools::rroot::leaf_string::read_buffer :"
257 << " read(int) failed."
258 << std::endl;
259 return false;
260 }
261 }
262 if(len) {
263 //if(!m_length) {
264 // m_out << "tools::rroot::leaf_string::read_buffer : m_length is zero." << std::endl;
265 // return false;
266 //}
267 //if(len >= m_length) len = m_length-1;
268
269 m_value = new char[len+1];
270
271 if(!a_buffer.read_fast_array(m_value,len)) {
272 m_out << "tools::rroot::leaf_string::read_buffer :"
273 << " read_fast_array failed."
274 << std::endl;
275 delete [] m_value;
276 m_value = 0;
277 return false;
278 }
279 m_value[len] = 0;
280 } else {
281 m_value = new char[1];
282 m_value[0] = 0;
283 }
284
285 return true;
286 }
287 virtual bool print_value(std::ostream& a_out,uint32) const {
288 if(m_value) a_out << m_value;
289 return true;
290 }
291 virtual uint32 num_elem() const {return 1;}
292public:
293 leaf_string(std::ostream& a_out,ifac& a_fac)
294 :base_leaf(a_out,a_fac)
295 ,m_min(0),m_max(0),m_value(0){}
296 virtual ~leaf_string(){
297 delete [] m_value;
298 }
299protected:
300 leaf_string(const leaf_string& a_from)
301 :iro(a_from),base_leaf(a_from)
302 ,m_min(0),m_max(0),m_value(0){}
303 leaf_string& operator=(const leaf_string&){return *this;}
304public:
305 const char* value() const {return m_value;}
306protected:
307 int m_min;
308 int m_max;
309 char* m_value;
310};

402 virtual bool stream(buffer& a_buffer) {
403 short v;
404 unsigned int s,c;
405 if(!a_buffer.read_version(v,s,c)) return false;
406 if(!base_leaf::stream(a_buffer)) return false;
407 if(!a_buffer.read(fVirtual)) return false;
408 if(!a_buffer.check_byte_count(s,c,s_store_class())) return false;
409 return true;
410 }
411public: //base_leaf
412 virtual bool read_buffer(buffer& a_buffer) {
413 if(!m_obj) {
414 m_out << "tools::rroot::leaf_object::read_buffer : m_obj is null." << std::endl;
415 return false;
416 }
417 std::string fClassName;
418 if (fVirtual) {
419 unsigned char n;
420 if(!a_buffer.read(n)) {
421 m_out << "tools::rroot::leaf_object::read_buffer :"
422 << " read(unsigned char) failed."
423 << std::endl;
424 return false;
425 }
426 char classname[128];
427 if(!a_buffer.read_fast_array(classname,n+1)) {
428 m_out << "tools::rroot::leaf_object::read_buffer :"
429 << " readFastArray failed."
430 << std::endl;
431 return false;
432 }
433 fClassName = classname;
434 }
435 if(m_obj->store_class_name()!=fClassName) {
436 m_out << "tools::rroot::leaf_object::read_buffer : WARNING : class mismatch :"
437 << " fClassName " << sout(fClassName)
438 << ". m_obj.store_class_name() " << sout(m_obj->store_class_name())
439 << std::endl;
440 //return false;
441 }
442 if(!m_obj->stream(a_buffer)) {
443 m_out << "tools::rroot::leaf_object::read_buffer :"
444 << " object stream failed."
445 << " Object store class was " << m_obj->store_class_name() << "."
446 << std::endl;
447 return false;
448 }
449 // in case we had written a null pointer a Zombie object was created
450 // we must delete it
451 //FIXME
452 //if (object->TestBit(kInvalidObject)) {
453 // if (object->GetUniqueID() == 123456789) {
454 // delete object;
455 // object = 0;
456 // }
457 //}
458 return true;
459 }
460 virtual bool print_value(std::ostream&,uint32) const {
461 m_out << m_obj << std::endl;
462 return true;
463 }
464 virtual uint32 num_elem() const {return 0;}
465public:

477protected:
