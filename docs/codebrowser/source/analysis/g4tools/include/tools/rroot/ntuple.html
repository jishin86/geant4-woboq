47 template <class T>
48 class column_ref : public virtual read::icolumn<T> {
49 typedef read::icolumn<T> parent;
50 public:
51 static cid id_class() {return 200+_cid(T())+10000;}
52 public: //icol
53 virtual void* cast(cid a_class) const {
54 if(void* p = cmp_cast<column_ref>(this,a_class)) return p;
55 return parent::cast(a_class);
56 }
57 virtual cid id_cls() const {return id_class();}
58 public: //icol
59 virtual const std::string& name() const {return m_leaf.name();}
60 public: //icolumn<T>
61 virtual bool fetch_entry() const {return _fetch_entry();}
62 virtual bool get_entry(T& a_v) const {
63 if(!_fetch_entry()) {a_v = T();return false;}
64 a_v = m_ref;
65 return true;
66 }
67 public:
68 column_ref(ifile& a_file,branch& a_branch,leaf<T>& a_leaf,int64& a_index,T& a_ref)
69 :m_file(a_file)
70 ,m_branch(a_branch)
71 ,m_leaf(a_leaf)
72 ,m_index(a_index) //WARNING : we keep the ref !
73 ,m_ref(a_ref)
74 {}
75 virtual ~column_ref(){}
76 protected:
77 column_ref(const column_ref& a_from)
78 :read::intuple(a_from)
79 ,parent(a_from)
80 ,m_file(a_from.m_file)
81 ,m_branch(a_from.m_branch)
82 ,m_leaf(a_from.m_leaf)
83 ,m_index(a_from.m_index)
84 ,m_ref(a_from.m_ref)
85 {}
86 column_ref& operator=(const column_ref& a_from){
87 if(&a_from==this) return *this;
88 return *this;
89 }
90 protected:
91 bool _fetch_entry() const {
92 unsigned int n;
93 if(!m_branch.find_entry(m_file,uint32(m_index),n)) {m_ref = T();return false;}
94 if(!m_leaf.num_elem()) {m_ref = T();return true;} //it is ok. It may be a vector from a row_wise ntuple column.
95 return m_leaf.value(0,m_ref);
96 }
97 protected:
98 ifile& m_file;
99 branch& m_branch;
100 leaf<T>& m_leaf;
101 int64& m_index; //WARNING : a ref.
102 T& m_ref;
103 };

420 return *this;
421 }

423 bool _fetch_entry() const {
424 unsigned int n;
425 if(!m_be.find_entry(m_file,uint32(m_index),n)) {m_ref = T();return false;}
426 iro* obj = m_be.object(); //Not owner.
427 if(!obj) {m_ref = T();return false;}
428 RT* v = id_cast<iro,RT>(*obj);
429 if(!v) {m_ref = T();return false;}
430 m_ref = *v; //it assumes a T::operator=(RT)
431 return true;
432 }
433 protected:
434 ifile& m_file;
435 branch_element& m_be;
436 leaf_element& m_leaf;
437 int64& m_index; //WARNING : a ref.

475public:
476 ntuple(tree& a_tree):m_tree(a_tree),m_index(-1){
477#ifdef TOOLS_MEM
478 mem::increment(s_class().c_str());
479#endif
480 }
481 virtual ~ntuple() {
482 safe_clear<read::icol>(m_cols);
483#ifdef TOOLS_MEM
484 mem::decrement(s_class().c_str());
485#endif
486 }
487protected:
488 ntuple(const ntuple& a_from)
489 :read::intuple(a_from),m_tree(a_from.m_tree){
490#ifdef TOOLS_MEM
491 mem::increment(s_class().c_str());
492#endif
493 }
494 ntuple& operator=(const ntuple&){return *this;}
495public:
496 bool initialize(std::ostream& a_out,const ntuple_binding& a_bd = ntuple_binding()) {
497 safe_clear<read::icol>(m_cols);
498
499 std::vector<base_leaf*> leaves;
500 m_tree.find_leaves(leaves);
501 tools_vforcit(base_leaf*,leaves,it) {
502 base_leaf* bl = (*it);
503 if(find_named<read::icol>(m_cols,bl->name())) {
504 a_out << "tools::rroot::ntuple::initialize :"
505 << " column with name " << sout(bl->name())
506 << " already exists."
507 << std::endl;
508 safe_clear<read::icol>(m_cols);
509 return false;
510 }
511 branch* _branch = m_tree.find_leaf_branch(*bl);
512 if(!_branch) {
513 a_out << "tools::rroot::ntuple::initialize :"
514 << " can't find branch of leaf " << sout(bl->name()) << "."
515 << std::endl;
516 safe_clear<read::icol>(m_cols);
517 return false;
518 }
519
520 //a_out << "tools::rroot::ntuple::initialize :"
521 // << " entries " << bl->branch().entry_number() << "."
522 // << std::endl;
523
524#define TOOLS_RROOT_NTUPLE_CREATE_COL(a__type) \
525 if(leaf<a__type>* lf_##a__type = safe_cast<base_leaf, leaf<a__type> >(*bl) ){\
526 cid user_cid;void* user_obj;\
527 a_bd.find_user_obj(bl->name(),user_cid,user_obj);\
528 if(!user_obj) {\
529 column<a__type>* col = new column<a__type>(m_tree.file(),*_branch,*lf_##a__type,m_index);\
530 m_cols.push_back(col);\
531 } else {\
532 const base_leaf* lfc = bl->leaf_count();\
533 if(lfc) {\
534 /*::printf("debug : ntuple : create col leaf count : %s\n",bl->name().c_str());*/\
535 if(user_cid!=_cid_std_vector<a__type>()) {\
536 a_out << "tools::rroot::ntuple::initialize :"\
537 << " for leaf with name " << sout(bl->name())\
538 << ", user variable type is not a std::vector of " << #a__type << "."\
539 << std::endl;\
540 safe_clear<read::icol>(m_cols);\
541 return false;\
542 }\
543 std::vector<a__type>* user_var = (std::vector<a__type>*)user_obj;\
544 std_vector_column_ref<a__type>* col = new std_vector_column_ref<a__type>\
545 (m_tree.file(),*_branch,*lf_##a__type,m_index,*user_var);\
546 m_cols.push_back(col);\
547 } else {\
548 /*::printf("debug : ntuple : create col : %s\n",bl->name().c_str());*/\
549 if(user_cid!=_cid(a__type())) {\
550 a_out << "tools::rroot::ntuple::initialize :"\
551 << " for leaf with name " << sout(bl->name())\
552 << ", user variable type is not a " << #a__type << "."\
553 << std::endl;\
554 safe_clear<read::icol>(m_cols);\
555 return false;\
556 }\
557 a__type* user_var = (a__type*)user_obj;\
558 column_ref<a__type>* col = new column_ref<a__type>(m_tree.file(),*_branch,*lf_##a__type,m_index,*user_var);\
559 m_cols.push_back(col);\
560 }\
561 }\
562 }
563
564 //below types are in sync with wroot/ntuple.
565
566 TOOLS_RROOT_NTUPLE_CREATE_COL(char)
567 else TOOLS_RROOT_NTUPLE_CREATE_COL(short)
568 else TOOLS_RROOT_NTUPLE_CREATE_COL(int)
569 else TOOLS_RROOT_NTUPLE_CREATE_COL(float)
570 else TOOLS_RROOT_NTUPLE_CREATE_COL(double)
571
572 else if(leaf_string* ls = safe_cast<base_leaf, leaf_string >(*bl) ){
573 char sep = '\n';
574 cid user_cid;void* user_obj;
575 if(!a_bd.find_user_obj(bl->name(),user_cid,user_obj)) {
576 column_vector_string* col = new column_vector_string(m_tree.file(),*_branch,*ls,m_index,sep);
577 m_cols.push_back(col);
578 } else if(user_cid==_cid_std_vector<std::string>()) {
579 std::vector<std::string>* user_var = (std::vector<std::string>*)user_obj;
580 if(user_var) {
581 column_vector_string_ref* col = new column_vector_string_ref(m_tree.file(),*_branch,*ls,m_index,*user_var,sep);
582 m_cols.push_back(col);
583 } else {
584 column_vector_string* col = new column_vector_string(m_tree.file(),*_branch,*ls,m_index,sep);
585 m_cols.push_back(col);
586 }
587 } else if(user_cid==_cid(std::string())) {
588 std::string* user_var = (std::string*)user_obj;
589 if(user_var) {
590 column_string_ref* col = new column_string_ref(m_tree.file(),*_branch,*ls,m_index,*user_var);
591 m_cols.push_back(col);
592 } else {
593 column_string* col = new column_string(m_tree.file(),*_branch,*ls,m_index);
594 m_cols.push_back(col);
595 }
596 } else {
597 a_out << "tools::rroot::ntuple::initialize :"
598 << " for leaf with name " << sout(ls->name())
599 << ", user variable type is not a std::string or a std::vector<std::string>."
600 << ". It's class id is " << user_cid << "."
601 << std::endl;
602 safe_clear<read::icol>(m_cols);
603 return false;
604 }
605
606 } else if(leaf_element* le = safe_cast<base_leaf,leaf_element>(*bl) ){
607
608 branch_element* be = safe_cast<branch,branch_element>(*_branch);
609 if(!be) {
610 a_out << "tools::rroot::ntuple::initialize : branch is not a branch_element." << std::endl;
611 safe_clear<read::icol>(m_cols);
612 return false;
613 }
614
615#define TOOLS_RROOT_NTUPLE_CREATE_VEC_COL(a__name,a__type) \
616 if(be->class_name()==a__name) {\
617 /*::printf("debug : ntuple : create vec col : %s\n",bl->name().c_str());*/\
618 typedef a__type el_t;\
619 std::vector<el_t>* user_var = a_bd.find_vector_variable<el_t>(bl->name());\
