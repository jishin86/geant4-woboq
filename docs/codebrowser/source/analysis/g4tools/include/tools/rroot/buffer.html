135 a_class.clear();

141 char s[80];
142 if(!read_string(s, 80)) {
143 m_out << "tools::rroot::read_class_tag :"
144 << " read string." << std::endl;

146 }
147 //m_out << "tools::rroot::read_class_tag :"
148 // << " tag kNewClassTag"
149 // << " class " << s
150 // << std::endl;
151 a_class = s;
152 return true;
153
154 } else if(tag & kClassMask()) {
155 //m_out << "tools::rroot::read_class_tag :"
156 // << " tag & kClassMask"
157 // << " ref " << (uint32)(tag & ~kClassMask)
158 // << " recurse..."
159 // << std::endl;
160
161 unsigned int cl_offset = (tag & ~kClassMask());
162 cl_offset -= kMapOffset();
163 cl_offset -= m_klen;
164 char* old_pos = m_pos;
165 m_pos = m_buffer + cl_offset;
166 //std::string scls;
167 //uint32 ref;
168 if(!read_class_tag(a_class)) return false;
169 m_pos = old_pos;
170
171 return true;
172
173 } else {
174

176 m_out << "tools::rroot::read_class_tag :"
177 << " tag unknown case ! "
178 << tag << " hex " << std::hex << tag

186 bool read_class(std::string& a_class,uint32& a_bcnt,bool& a_is_ref){
187 //m_verbose = true;
188 a_class.clear();
189 a_bcnt = 0;
190 a_is_ref = false;
191
192 //uint32 fVersion = 0; //Buffer format version
193
194 // read byte count and/or tag (older files don't have byte count)
195 uint32 first_int = 0;
196 if(!parent::read(first_int)) return false;
197
198 if(m_verbose) {
199 std::ios::fmtflags old_flags = m_out.flags();
200 m_out << "tools::rroot::read_class :"
201 << " first_int " << std::hex << first_int
202 << std::endl;
203 m_out.flags(old_flags);
204 }
205
206 if(first_int==kNullTag()) { //GB
207 if(m_verbose) {
208 m_out << "tools::rroot::read_class :"
209 << " first_int is kNullTag."
210 << std::endl;
211 }
212 a_bcnt = 0;
213 return true;
214
215 //} else if(first_int==kNewClassTag()) { // class desc follows.
216 } else if(first_int & kByteCountMask()) {
217 // at write :
218 // skip int to store bcnt.
219 // + write class
220 // if(!write((clIdx | Rio_kClassMask))) return false;
221 // or
222 // if(!write(Rio_kNewClassTag)) return false;
223 // + write object
224 // + set byte cnt (cnt|kByteCountMask()) at skipped int.
225
226 if(m_verbose) {
227 m_out << "tools::rroot::read_class :"
228 << " first_int & kByteCountMask."
229 << std::endl;
230 }
231
232 uint32 bef_tag = uint32(m_pos-m_buffer);
233 //fVersion = 1;
234
235 std::string scls;
236 if(!read_class_tag(scls)) return false;
237 if(scls.empty()) {
238 m_out << "tools::rroot::buffer::read_class :"
239 << " read_class_tag did not find a class name."
240 << std::endl;
241 return false;
242 }
243
244 a_class = scls;
245 a_bcnt = (first_int & ~kByteCountMask());
246
247 if(m_verbose) {
248 m_out << "tools::rroot::read_class :"
249 << " kNewClassTag : read class name " << sout(a_class)
250 << " a_bcnt " << a_bcnt
251 << " bef_tag " << bef_tag
252 << "." << std::endl;
253 }
254
255 return true;
256
257 } else {
258 if(m_verbose) {
259 std::ios::fmtflags old_flags = m_out.flags();
260 m_out << "tools::rroot::read_class :"
261 << " first_int " << std::hex << first_int
262 << ". first_int is position toward object."
263 << std::endl;
264 m_out.flags(old_flags);
265 }
266 a_bcnt = first_int; //pos toward object.
267 a_is_ref = true;
268 a_class.clear();
269 return true;
270 }
271
272 return false;

480 bool read_version(short& a_version){
481 // Read class version from I/O buffer.
482 // Is read a short or three shorts.
483 a_version = 0;
484 short version = 0;
485 // not interested in byte count
486 if(!parent::read(version)) return false;
487
488 // if this is a byte count, then skip next short and read version
489 if(version & kByteCountVMask()) {
490 if(!parent::read(version)) return false;
491 if(!parent::read(version)) return false;
492 }
493
494 a_version = version;
495 return true;
496 }

498 bool read_version(short& a_version,uint32& a_start_pos,uint32& a_byte_count){
499 // Read class version from I/O buffer.
500 // Is read one or three shorts.
501 a_version = 0;
502 a_start_pos = 0;
503 a_byte_count = 0;
504
505 short version = 0;
506
507 // before reading object save start position
508 uint32 startpos = (uint32)(m_pos-m_buffer);
509
510 // read byte count (older files don't have byte count)
511 // byte count is packed in two individual shorts, this to be
512 // backward compatible with old files that have at this location
513 // only a single short (i.e. the version)
514 union {
515 unsigned int cnt;
516 short vers[2];
517 } v;
518
519 if(m_byte_swap) {
520 if(!parent::read(v.vers[1])) return false;
521 if(!parent::read(v.vers[0])) return false;
522 } else {
523 if(!parent::read(v.vers[0])) return false;
524 if(!parent::read(v.vers[1])) return false;
525 }
526
527 // no bytecount, backup and read version
528 uint32 bcnt = 0;
529 if(v.cnt & kByteCountMask()) {
530 bcnt = (v.cnt & ~kByteCountMask());
531 } else {
532 m_pos -= sizeof(unsigned int);
533 }
534 if(!parent::read(version)) return false;
535 //printf("Reading version=%d at pos=%d, bytecount=%d\n",
536 //version,*startpos,*bcnt);
537
538 a_version = version;
539 a_start_pos = startpos;
540 a_byte_count = bcnt;
541
542 return true;
543 }

545 bool check_byte_count(uint32 a_start_pos,
546 uint32 a_byte_count,
547 const std::string& a_store_cls){
548 if(!a_byte_count) return true;
549
550 size_t len = a_start_pos + a_byte_count + sizeof(unsigned int);
551
552 size_t diff = size_t(m_pos-m_buffer);
553
554 if(diff==len) return true;
555
556 if(diff<len) {
557 m_out << "tools::rroot::buffer::check_byte_count :"
558 << " object of class " << sout(a_store_cls)
559 << " read too few bytes ("
560 << long_out(long(len-diff)) << " missing)."
561 << std::endl;
562 }
563 if(diff>len) {
564 m_out << "tools::rroot::buffer::check_byte_count :"
565 << " object of class " << sout(a_store_cls)
566 << " read too many bytes ("
567 << long_out(long(diff-len)) << " in excess)." << std::endl;
568 }
569
570 m_out << "tools::rroot::buffer::check_byte_count :"
571 << " " << sout(a_store_cls)
572 << " streamer not in sync with data on file, fix streamer."
573 << std::endl;
574
575 m_pos = m_buffer+len;
576
577 return false;
578 }
579protected:
580 bool read_string(char* a_string,uint32 a_max) {
581 // Read string from I/O buffer. String is read till 0 character is
582 // found or till max-1 characters are read (i.e. string s has max
583 // bytes allocated).
584 int nr = 0;
585 while (nr < int(a_max-1)) {
586 char ch;
587 if(!parent::read(ch)) return false;
588 // stop when 0 read
589 if(ch == 0) break;
590 a_string[nr++] = ch;
591 }
592 a_string[nr] = 0;
593 return true;
594 }

610inline bool dummy_TXxx_pointer_stream(buffer& a_buffer,ifac& a_fac,bool a_del = true) {
611 ifac::args args;
612 iro* obj;
613 bool created;
614 bool status = a_buffer.read_object(a_fac,args,obj,created);
615 if(obj && a_del) {
616 if(created) {
617 if(a_buffer.map_objs()) a_buffer.remove_in_map(obj);
618 delete obj;
619 } else {
620 a_buffer.out() << "dummy_TXxx_pointer_stream : warning : ask to delete an object of class " << sout(obj->s_cls())
621 << " not created here." << std::endl;
622 }
623 }
624 return status;
625}

628inline bool pointer_stream(buffer& a_buffer,
629 ifac& a_fac,ifac::args& a_args,
630 const std::string& a_T_class,
631 T*& a_obj){
632 iro* obj;
633 bool created;
634 if(!a_buffer.read_object(a_fac,a_args,obj,created)) {
635 a_buffer.out() << "tools::rroot::pointer_stream : read_object failed." << std::endl;
636 a_obj = 0;
637 return false;
638 }
639 if(!obj) {
640 a_obj = 0;
641 } else {
642 a_obj = (T*)obj->cast(a_T_class);
643 if(!a_obj) {
644 a_buffer.out() << "tools::rroot::pointer_stream : "
645 << " inlib::cast to " << a_T_class << " failed."
646 << ". Object is a " << obj->s_cls() << "."
647 << std::endl;
648 return false;
649 }
650 }
651 return true;

688template <class T>
689inline bool fixed_array_stream(buffer& a_buffer,int a_n,T*& a_v){
690 delete [] a_v;
691 a_v = 0;
692 char is_array;
693 if(!a_buffer.read(is_array)) return false;
694 if(!is_array) return true;
695 if(!a_n) return true;
696 a_v = new T[a_n];
697 if(!a_buffer.read_fast_array<T>(a_v,a_n)) {
698 delete [] a_v;
699 a_v = 0;
700 return false;
701 }
702 return true;
