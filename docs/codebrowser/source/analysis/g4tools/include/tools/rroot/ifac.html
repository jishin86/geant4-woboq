18class ifac {
19public:
20 typedef std::map<char,void*> args;
21public:
22 virtual ~ifac(){}
23public:
24 virtual iro* create(const std::string& a_class,const args&) = 0;
25 virtual void destroy(iro*&) = 0;
26public:
27 static void* find_args(const args& a_args,char a_key) {
28 std::map<char,void*>::const_iterator it = a_args.find(a_key);
29 if(it==a_args.end()) return 0;
30 return (*it).second;
31 }
32 //static char arg_branch() {return 'B';}
33 static char arg_class() {return 'C';}
34
35 static std::string* arg_class(const args& a_args) {
36 void* p = ifac::find_args(a_args,ifac::arg_class());
37 if(!p) return 0;
38 return (std::string*)p;
39 }
40};
