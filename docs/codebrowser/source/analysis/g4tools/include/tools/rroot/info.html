16class streamer_element : public virtual iro {
17 static const std::string& s_store_class() {
18 static const std::string s_v("TStreamerElement");
19 return s_v;
20 }
21public:
22 static const std::string& s_class() {
23 static const std::string s_v("tools::rroot::streamer_element");
24 return s_v;
25 }
26 static cid id_class() {return streamer_element_cid();}
27public: //iro
28 virtual void* cast(const std::string& a_class) const {
29 if(void* p = cmp_cast<streamer_element>(this,a_class)) return p;
30 return 0;
31 }
32 virtual void* cast(cid a_class) const {
33 if(void* p = cmp_cast<streamer_element>(this,a_class)) {return p;}
34 else return 0;
35 }
36 virtual const std::string& s_cls() const {return s_class();}
37 virtual iro* copy() const {return new streamer_element(*this);}
38
39 virtual bool stream(buffer& a_buffer) {
40 short v;
41 unsigned int s, c;
42 if(!a_buffer.read_version(v,s,c)) return false;
43 if(!Named_stream(a_buffer,fName,fTitle)) return false;
44 if(!a_buffer.read(fType)) return false;
45 if(!a_buffer.read(fSize)) return false;
46 if(!a_buffer.read(fArrayLength)) return false;
47 if(!a_buffer.read(fArrayDim)) return false;
48 if(!a_buffer.read_fast_array<int>(fMaxIndex,5)) return false;
49 if(!a_buffer.read(fTypeName)) return false;
50 return a_buffer.check_byte_count(s,c,s_store_class());
51 }
52public:
53 virtual void out(std::ostream& aOut) const {
54 std::string _fname;
55 fullName(_fname);
56 char s[128];
57 snpf(s,sizeof(s)," %-14s%-15s offset=%3d type=%2d %-20s",
58 fTypeName.c_str(),_fname.c_str(),fOffset,fType,fTitle.c_str());
59 aOut << s << std::endl;
60 }
61public:
62 streamer_element()
63 :fName(),fTitle(),fType(-1)
64 ,fSize(0),fArrayLength(0),fArrayDim(0),fOffset(0)
65 ,fTypeName(){
66#ifdef TOOLS_MEM
67 mem::increment(s_class().c_str());
68#endif
69 for(int i=0;i<5;i++) fMaxIndex[i] = 0;
70 }
71 virtual ~streamer_element(){
72#ifdef TOOLS_MEM
73 mem::decrement(s_class().c_str());
74#endif
75 }
76protected:
77 streamer_element(const streamer_element& a_from)
78 :iro(a_from)
79 ,fName(a_from.fName),fTitle(a_from.fTitle)
80 ,fType(a_from.fType),fSize(a_from.fSize)
81 ,fArrayLength(a_from.fArrayLength)
82 ,fArrayDim(a_from.fArrayDim),fOffset(a_from.fOffset)
83 ,fTypeName(a_from.fTypeName){
84#ifdef TOOLS_MEM
85 mem::increment(s_class().c_str());
86#endif
87 for(int i=0;i<5;i++) fMaxIndex[i] = a_from.fMaxIndex[i];
88 }
89 streamer_element& operator=(const streamer_element& a_from){
90 fName = a_from.fName;
91 fTitle = a_from.fTitle;
92 fType = a_from.fType;
93 fSize = a_from.fSize;
94 fArrayLength = a_from.fArrayLength;
95 fArrayDim = a_from.fArrayDim;
96 fOffset = a_from.fOffset;
97 fTypeName = a_from.fTypeName;
98 for(int i=0;i<5;i++) fMaxIndex[i] = a_from.fMaxIndex[i];
99 return *this;
100 }
101public:
102 virtual void fullName(std::string& a_s) const {
103 a_s = fName;
104 for (int i=0;i<fArrayDim;i++) {
105 char cdim[32];
106 snpf(cdim,sizeof(cdim),"[%d]",fMaxIndex[i]);
107 a_s += cdim;
108 }
109 }
110protected: //Named
111 std::string fName;
112 std::string fTitle;
113protected:
114 int fType; //element type
115 int fSize; //sizeof element
116 int fArrayLength; //cumulative size of all array dims
117 int fArrayDim; //number of array dimensions
118 int fMaxIndex[5]; //Maximum array index for array dimension "dim"
119 int fOffset; //!element offset in class
120 //FIXME Int_t fNewType; //!new element type when reading
121 std::string fTypeName; //Data type name of data member
122};

124class dummy_streamer_element : public streamer_element {
125 typedef streamer_element parent;
126public: //iro
127 virtual iro* copy() const {return new dummy_streamer_element(*this);}
128 virtual bool stream(buffer& a_buffer) {
129 //the below code skips correctly the data in the file.
130 uint32 startpos = a_buffer.length();
131 short v;
132 unsigned int s,c;
133 if(!a_buffer.read_version(v,s,c)) return false;
134
135 if(!parent::stream(a_buffer)) return false;
136
137 a_buffer.set_offset(startpos+c+sizeof(unsigned int));
138 if(!a_buffer.check_byte_count(s,c,"dummy_streamer_element")) return false;
139 return true;
140 }
141public:
142 dummy_streamer_element(){}
143 virtual ~dummy_streamer_element(){}
144protected:
145 dummy_streamer_element(const dummy_streamer_element& a_from):iro(a_from),parent(a_from){}
146 dummy_streamer_element& operator=(const dummy_streamer_element& a_from){
147 parent::operator=(a_from);
148 return *this;
149 }
150};

152class StreamerInfo : public virtual iro {
153 static const std::string& s_store_class() {
154 static const std::string s_v("TStreamerInfo");
155 return s_v;
156 }
157public:
158 static const std::string& s_class() {
159 static const std::string s_v("tools::rroot::StreamerInfo");
160 return s_v;
161 }
162 static cid id_class() {return StreamerInfo_cid();}
163public: //iro
164 virtual void* cast(const std::string& a_class) const {
165 if(void* p = cmp_cast<StreamerInfo>(this,a_class)) return p;
166 return 0;
167 }
168 virtual void* cast(cid a_class) const {
169 if(void* p = cmp_cast<StreamerInfo>(this,a_class)) {return p;}
170 else return 0;
171 }
172 virtual const std::string& s_cls() const {return s_class();}
173 virtual iro* copy() const {return new StreamerInfo(*this);}
174
175 virtual bool stream(buffer& a_buffer) {
176 short v;
177 unsigned int s, c;
178 if(!a_buffer.read_version(v,s,c)) return false;
179
180 if(!Named_stream(a_buffer,m_name,m_title)) return false;
181 if(!a_buffer.read(m_check_sum)) return false;
182 if(!a_buffer.read(m_streamed_class_version)) return false;
183
184 //TObjArray *fElements; //Array of TStreamerElements
185 {obj_array<streamer_element>* obj;
186 ifac::args args;
187 args[ifac::arg_class()] = (void*)&(streamer_element::s_class());
188 if(!pointer_stream(a_buffer,m_fac,args,obj)) {
189 a_buffer.out() << "tools::rroot::StreamerInfo::stream : "
190 << "can't read fElements."
191 << std::endl;
192 return false;
193 }
194 if(obj) m_elements.operator=(*obj);
195 delete obj;}
196
197 return a_buffer.check_byte_count(s,c,s_store_class());
198 }
199public:
200 void out(std::ostream& a_out) const {
201 a_out << "StreamerInfo for class :"
202 << " " << m_name << ", version=" << m_streamed_class_version
203 << std::endl;
204 tools_vforcit(streamer_element*,m_elements,it) (*it)->out(a_out);
205 }
206public:
207 StreamerInfo(ifac& a_fac)
208 :m_fac(a_fac)
209 ,m_name()
210 ,m_title()
211 ,m_check_sum(0)
212 ,m_streamed_class_version(0)
213 ,m_elements(a_fac,true,true)
214 {
215#ifdef TOOLS_MEM
216 mem::increment(s_class().c_str());
217#endif
218 }
219 virtual ~StreamerInfo(){
220#ifdef TOOLS_MEM
221 mem::decrement(s_class().c_str());
222#endif
223 }
224protected:

227 ,m_fac(a_from.m_fac)
228 ,m_name(a_from.m_name)
229 ,m_title(a_from.m_name)
230 ,m_check_sum(a_from.m_check_sum)
231 ,m_streamed_class_version(a_from.m_streamed_class_version)
232 ,m_elements(a_from.m_elements)
233 {
234#ifdef TOOLS_MEM
235 mem::increment(s_class().c_str());
236#endif
237 }
238 StreamerInfo& operator=(const StreamerInfo& a_from){
239 m_name = a_from.m_name;
240 m_title = a_from.m_name;
241 m_check_sum = a_from.m_check_sum;

247protected:

249protected: //Named

252protected:
