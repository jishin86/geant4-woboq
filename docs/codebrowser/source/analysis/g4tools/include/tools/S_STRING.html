7// a few place in which a cpp macro help the readability.
8
9#include <string>
10
11#define TOOLS_CLASS_STRING(a_name)\
12static const std::string& s_##a_name() {\
13 static const std::string s_v(#a_name);\
14 return s_v;\
15}
16
17#define TOOLS_CLASS_STRING_VALUE(a_name,a_value)\
18static const std::string& s_##a_name() {\
19 static const std::string s_v(#a_value);\
20 return s_v;\
21}
22
23#define TOOLS_CLASS_STRING_SVALUE(a_name,a_value)\
24static const std::string& s_##a_name() {\
25 static const std::string s_v(a_value);\
26 return s_v;\
27}

29#define TOOLS_GLOBAL_STRING(a_name)\

33}

35#define TOOLS_GLOBAL_STRING_VALUE(a_name,a_value)\

41#define TOOLS_SCLASS(a_name)\
