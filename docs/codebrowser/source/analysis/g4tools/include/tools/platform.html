13inline bool is_little_endian() {
14 unsigned int i = 1;
15 unsigned char* b = (unsigned char*)&i;
16 // BE = Big Endian, LE = Little Endian.
17 // The Intels x86 are LE.
18 // Mac PPC b[3] is 1 (BE)
19 // Mac Intel b[0] is 1 (LE)
20 // Linux i386 b[0] is 1 (LE)
21 // Linux x86_64 b[0] is 1 (LE)
22 return (b[0]==1?true:false);
23}

62inline unsigned int tex_mem_limit() {
63 // glGet(GL_MAX_TEXTURE_SIZE) :
64 // MacBookPro : it returns 8192.
65 // SGS : it returns 2048.
66 // iPad1 : it returns 2048.
67 // iPod : it returns ?
68 // Nexus 10 : it returns ?
69
70 // 1024*1024 //*3 = 3 145 728
71 // 2048*2048 //*3 = 12 582 912
72 // 4096*4096 //*3 = 50 331 648
73 // 8192*8192 //*3 = 201 326 592
74 // 8192*4096 //*3 = 100 663 296
75 if(small_screen()) {
76 //iOS : Apple says that max 2D tex size is 1024*1024 with two units
77 // texture available. From doc on PowerVR MBX.
78 return 2048*2048*3;
79 } else {
80 //limit = 4096*4096*3; //=8192*2048 //permit to pass ATLAS big image.
81 //permit to pass fete_science_2010/power2/image_0.jpg
82 return 8192*4096*3;
83 }

178inline char* compiler() {
179 static char s_s[128];
180 char* pos = s_s;
181 unsigned int l;
182 toss(compiler_name(),pos,l);pos += l;
183
184#if defined(__clang__)
185 *pos = '_';pos++;
186 tosu<unsigned int>(__clang_major__,pos,l);pos += l;
187 tosu<unsigned int>(__clang_minor__,pos,l);pos += l;
188 tosu<unsigned int>(__clang_patchlevel__,pos,l);pos += l;
189#elif defined(__GNUC__)
190 *pos = '_';pos++;
191 tosu<unsigned int>(__GNUC__,pos,l);pos += l;
192 tosu<unsigned int>(__GNUC_MINOR__,pos,l);pos += l;
193 tosu<unsigned int>(__GNUC_PATCHLEVEL__,pos,l);pos += l;
194#elif defined(_MSC_VER)
195 *pos = '_';pos++;
196 tosu<unsigned int>(_MSC_VER,pos,l);pos += l;
197 //_mfc_
198 //tosu<unsigned int>(_MFC_VER,pos,l);pos += l;
199#elif defined(__alpha)
200#else
201#endif
202 *pos = 0;
203 return s_s;
204}
