15template <class T>
16inline void safe_clear(std::vector<T*>& a_vec){
17 // the below takes into account the case in
18 // which "delete entry" could modify a_vec.
19 typedef typename std::vector<T*>::iterator it_t;
20 while(!a_vec.empty()) {
21 it_t it = a_vec.begin();
22 T* entry = *it;
23 a_vec.erase(it);
24 delete entry;
25 }
26}
27
28template <class T>
29inline void safe_reverse_clear(std::vector<T*>& a_vec){ //used in sg/group.
30 // the below takes into account the case in
31 // which "delete entry" could modify a_vec.
32 typedef typename std::vector<T*>::iterator it_t;
33 while(!a_vec.empty()) {
34 it_t it = a_vec.end();
35 it--;
36 T* entry = *it;
37 a_vec.erase(it);
38 delete entry;
39 }
40}

46template <class T>
47inline void raw_clear(std::vector<T*>& a_vec){
48 typedef typename std::vector<T*>::iterator it_t;
49 for(it_t it = a_vec.begin();it!=a_vec.end();++it) delete *it;

54inline void copy(std::vector<T*>& a_to,const std::vector<T*>& a_from){

213inline void filter(std::vector<T>& a_vec,unsigned int a_min,unsigned int a_max){
214 unsigned int imx = a_vec.size()-1;
215 unsigned int mx = a_max<imx?a_max:imx;
216 unsigned int i = 0;
217 for(unsigned int index=a_min;index<=mx;index++) {
218 a_vec[i] = a_vec[index];i++;
219 }
220 a_vec.resize(i);
221}

339template <class T>
340inline bool min_max(const std::vector<T>& a_vec,T& a_min,T& a_max) {
341 if(a_vec.empty()) {a_min=T();a_max=T();return false;}
342 a_min = *(a_vec.begin());
343 a_max = a_min;
344 typedef typename std::vector<T>::const_iterator it_t;
345 for(it_t it = a_vec.begin();it!=a_vec.end();++it) {
346 a_min = *it<a_min?*it:a_min;
347 a_max = *it>a_max?*it:a_max;
348 }
349 return true;
350}

352template <class T>
353inline bool mean_rms(const std::vector<T>& a_vec,T& a_mean,T& a_rms,T(*a_sqrt)(T),T(*a_fabs)(T)) {
354 if(a_vec.empty()) {a_mean=T();a_rms=T();return false;}
355 T S = T();
356 T S2 = T();
357 typedef typename std::vector<T>::const_iterator it_t;
358 for(it_t it = a_vec.begin();it!=a_vec.end();++it) {
359 S += *it;
360 S2 += (*it) * (*it);
361 }
362 a_mean = S/T(a_vec.size());
363 a_rms = a_sqrt(a_fabs(S2/T(a_vec.size()) - a_mean * a_mean));
364 return true;
365}

370 if(a_vec.empty()) {a_min=T();a_max=T();a_mean=T();a_rms=T();return false;}

372 a_max = a_min;

376 for(it_t it = a_vec.begin();it!=a_vec.end();++it) {
377 a_min = *it<a_min?*it:a_min;
378 a_max = *it>a_max?*it:a_max;
379 S += *it;
380 S2 += (*it) * (*it);

382 a_mean = S/T(a_vec.size());
