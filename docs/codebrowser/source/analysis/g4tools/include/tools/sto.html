23 ){

42template <class T>
43inline bool to(const std::string& a_s,T& a_v,const T& a_def = T()) {
44 if(a_s.empty()) {a_v = a_def;return false;} //for TOOLS_STL istringstream.
45 std::istringstream strm(a_s.c_str());
46 strm >> a_v;
47 if(strm.fail()) {a_v = a_def;return false;}
48 return strm.eof();
49}

55 if(!to(a_s,a_field)) {a_field = old;a_changed=false;return false;}
