60inline bool read_cstring(FILE* a_file,char* a_buff,unsigned int a_lbuf,size_t& a_length) {
61 if(::fgets(a_buff,a_lbuf,a_file)==NULL) {
62 a_length = 0;
63 return false; //EOF
64 }
65
66 size_t l = ::strlen(a_buff);
67 // On Windows, editors when saving binary files,
68 // put \r\n at place of \n ; we then look for \r\n.
69 if( (l>=2) && (a_buff[l-2]=='\r') && (a_buff[l-1]=='\n') ) {
70 a_buff[l-2] = '\0';
71 l -= 2;
72 } else if( (l>=1) && (a_buff[l-1]=='\n') ) {
73 a_buff[l-1] = '\0';
74 l -= 1;
75 }
76
77 a_length = l;
78 return true;
79}

82inline bool read(const std::string& a_file,std::vector<std::string>& a_text){
83 a_text.clear();
84 FILE* file = ::fopen(a_file.c_str(),"rb");
85 if(!file) return false;
86 unsigned int BUFSIZE = 65536;
87 char* buffer = new char[BUFSIZE+1];
88 if(!buffer) {::fclose(file);return false;}
89 while(true) {
90 size_t l;
91 if(!read_cstring(file,buffer,BUFSIZE,l)) break; // EOF.
92 a_text.push_back(buffer);
93 }
94 delete [] buffer;
95 ::fclose(file);
96 return true;
97}

99inline bool read_num(const std::string& a_file,std::vector<std::string>::size_type a_num,std::vector<std::string>& a_text,const std::string& a_cmt = ""){
100 a_text.clear();
101 FILE* file = ::fopen(a_file.c_str(),"rb");
102 if(!file) return false;
103 unsigned int BUFSIZE = 65536;
104 char* buffer = new char[BUFSIZE+1];
105 if(!buffer) {::fclose(file);return false;}
106 while(true) {
107 size_t l;
108 if(!read_cstring(file,buffer,BUFSIZE,l)) break; // EOF.
109 if(a_cmt.size()&&(!strncmp(a_cmt.c_str(),buffer,a_cmt.size()))) continue;
110 if(a_text.size()<a_num) {
111 a_text.push_back(buffer);
112 } else {
113 break;
114 }
115 }
116 delete [] buffer;
117 ::fclose(file);
118 return true;
119}

121inline bool read_bytes(const std::string& a_file,char*& a_buffer,long& a_length){
122 // Returned buffer should be deleted with delete [].
123 FILE* file = ::fopen(a_file.c_str(),"rb");
124 if(!file) {
125 a_buffer = 0;
126 a_length = 0L;
127 return false;
128 }
129 // Get file size :
130 ::fseek(file,0L,SEEK_END);
131 long filesize = ::ftell(file);
132 if(!filesize) {
133 ::fclose(file);
134 a_buffer = new char[1];
135#ifdef TOOLS_MEM
136 mem::increment(s_new().c_str());
137#endif
138 a_length = 0L;
139 return true; //empty file.
140 }
141 // Add one byte to be able to add \n if file contain lines.
142 a_buffer = new char[filesize+1];
143 if(!a_buffer) {
144 ::fclose(file);
145 a_buffer = 0;
146 a_length = 0L;
147 return false;
148 }
149#ifdef TOOLS_MEM
150 mem::increment(s_new().c_str());
151#endif
152 ::rewind(file);
153 size_t nitem = ::fread(a_buffer,(size_t)filesize,(size_t)1,file);
154 if(nitem!=1){
155 ::fclose(file);
156 delete [] a_buffer;
157#ifdef TOOLS_MEM
158 mem::decrement(s_new().c_str());
159#endif
160 a_buffer = 0;
161 a_length = 0L;
162 return false;
163 }
164 ::fclose(file);
165 a_buffer[filesize] = 0;
166 a_length = filesize;
167 return true;
168}

170inline bool write_bytes(const std::string& a_file,const char* a_buffer,size_t a_length){
171 FILE* file = ::fopen(a_file.c_str(),"wb");
172 if(!file) return false;
173 if(!a_length) {
174 ::fclose(file);
175 return true;
176 }
177 size_t nitem = ::fwrite((char*)a_buffer,a_length,(size_t)1,file);
178 ::fclose(file);
179 return (nitem!=1?false:true);
180}

182inline bool num_csv_doubles(const std::string& a_string,char& a_sep,unsigned int& a_number){
183 if(a_string.empty()) {a_sep=0;a_number=0;return true;} //it is ok.
184
185 //guess sep :
186 char sep = 0;
187 {const char* buffer = a_string.c_str();
188 //::printf("debug : |%s|\n",buffer);
189 const char* pos = buffer;
190 const char* mx = buffer+a_string.size();
191 char* end;
192 {double d = ::strtod(pos,&end);(void)d;}
193 //::printf("debug : d first %g\n",d);
194 if(end==pos) {a_sep=0;a_number=0;return false;} //not starting with a number.
195 if(end==mx) {a_sep=0;a_number=1;return true;} //is ok, one number only.
196 sep = *end;}
197
198 //::printf("debug : %d\n",sep);
199
200 unsigned int number = 0;
201
202 const char* buffer = a_string.c_str();
203 //::printf("debug : |%s|\n",buffer);
204 const char* pos = buffer;
205 const char* mx = buffer+a_string.size();
206 while(true) {
207 char* end;
208 {double d = ::strtod(pos,&end);(void)d;}
209 if(end==pos) {
210 if(end>=mx) break;
211 a_sep = sep;
212 a_number = number;
213 return false; //since a number expected.
214 }
215 //::printf("debug : d %g\n",d);
216 number++;
217 if(*end!=sep) {
218 if(end>=mx) break;
219 a_sep = sep;
220 a_number = number;
221 return false; //since a sep is expected.
222 }
223 pos = end+1;
224 if(pos>mx) break;
225 }
226
227 a_sep = sep;
228 a_number = number;
229 return true;
230}

234inline bool is_hippo(const std::string& a_file,bool& a_is){
235 //if((suffix(a_file)=="hiptxt")||(suffix(a_file)=="tnt")) {
236 // a_is = true;
237 // return true;
238 //}
239 // An hippo file as :
240 // one line for title.
241 // one line with n labels separated by \t
242 // data lines with n numbers separated by a char sep (not necessary \t) (at least one data line expected).
243 std::vector<std::string> txt;
244 if(!file::read_num(a_file,3,txt)) {a_is=false;return false;}
245 if(txt.size()<3) {a_is=false;return true;}
246 char csep;
247 unsigned int number;
248 if(!num_csv_doubles(txt[2],csep,number)) {a_is=false;return true;}
249 std::vector<std::string> ws;
250 words(txt[1],"\t",false,ws);
251 if(ws.size()!=number) {a_is=false;return true;}
252 bool all_doubles = true;
253 tools_vforcit(std::string,ws,it) {
254 const char* pos = (*it).c_str();
255 char* end;
256 double d = ::strtod(pos,&end);(void)d;
257 if(end==pos) {all_doubles=false;break;}
258 }
259 a_is = all_doubles?false:true; //all_double=false then we assume the second line is labels of columns.
260 return true;
261}

265inline bool is_csv(const std::vector<std::string>& a_txt,bool& a_is){
266 //a_sep = 0;
267 a_is = false;
268 if(a_txt.empty()) return true;
269
270 //guess sep from first data line :
271 char sep = 0;
272 {const char* buffer = a_txt[0].c_str();
273 //::printf("debug : |%s|\n",buffer);
274 const char* pos = buffer;
275 char* end;
276 {double d = ::strtod(pos,&end);(void)d;}
277 //::printf("debug : d first %g\n",d);
278 if(end==pos) return true; //not starting with a number.
279 //end==mx is ok, one column only.
280 sep = *end;}
281
282 //::printf("debug : %d\n",sep);
283
284 unsigned int first_coln = 0;
285
286 tools_vforcit(std::string,a_txt,it) {
287 const char* buffer = (*it).c_str();
288 //::printf("debug : |%s|\n",buffer);
289 const char* pos = buffer;
290 const char* mx = buffer+(*it).size();
291 unsigned int coln = 0;
292 while(true) {
293 char* end;
294 {double d = ::strtod(pos,&end);(void)d;}
295 if(end==pos) break;
296 //::printf("debug : d %g\n",d);
297 if(*end!=sep) return true;
298 coln++;
299 pos = end+1;
300 if(pos>mx) break;
301 }
302 //::printf("debug : coln %d\n",coln);
303 if(it==a_txt.begin()) {
304 first_coln = coln;
305 } else {
306 if(coln!=first_coln) return true;
307 }
308 }
309
310 //a_sep = sep;
311 a_is = true;
312 return true;
313}

315inline bool is_csv(const std::string& a_file,bool& a_is){
316 // look at suffix. Some file can't be guessed.
317 if(suffix(a_file)=="csv") {a_is = true;return true;}
318 //try to guess :
319 std::vector<std::string> txt;
320 if(!file::read_num(a_file,3,txt,"#")) {a_is=false;return false;}
321 return is_csv(txt,a_is);

508inline bool copy_bytes(const std::string& a_from,const std::string& a_to) {
509 if(a_to==a_from) return true;
510 char* buffer;
511 long length;
512 if(!read_bytes(a_from,buffer,length)) return false;
513 bool status = write_bytes(a_to,buffer,(size_t)length);
514 delete [] buffer;
515#ifdef TOOLS_MEM
516 mem::decrement(s_new().c_str());
517#endif
518 return status;
519}

558class FILE_reader : public virtual file::reader {
559public: //file_reader
560 virtual bool open(const std::string& a_file) {
561 if(m_FILE) return false;
562 m_FILE = ::fopen(a_file.c_str(),"rb");
563 if(!m_FILE) return false;
564 return true;
565 }
566 virtual void close() {
567 if(!m_FILE) return;
568 ::fclose(m_FILE);
569 m_FILE = 0;
570 }
571 virtual bool is_open() const {return m_FILE?true:false;}
572 virtual bool read(char* a_buff,unsigned int a_lbuf,size_t& a_length) {
573 a_length = ::fread(a_buff,1,a_lbuf,m_FILE);
574 return true;
575 }
576 virtual bool get_line(char* a_buff,unsigned int a_lbuf) {
577 return ::fgets(a_buff,a_lbuf,m_FILE)==NULL?false:true;
578 }
579 virtual bool eof() const {
580#if defined(_MSC_VER) && _MSC_VER <= 1400
581 return feof(m_FILE)?true:false;
582#else
583 return ::feof(m_FILE)?true:false;
584#endif
585 }
586public:
587 FILE_reader():m_FILE(0){}
588 virtual ~FILE_reader() {if(m_FILE) ::fclose(m_FILE);}
589protected:
590 FILE_reader(const FILE_reader& a_from)
591 :file::reader(a_from)
592 ,m_FILE(0)
593 {}
594 FILE_reader& operator=(const FILE_reader&){return *this;}
595protected:
596 FILE* m_FILE;
597};
