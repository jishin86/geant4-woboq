42class ntuple : public virtual read::intuple {
43public: //read::intuple
44 virtual void start() {
45 m_reader.clear();
46 m_reader.seekg(0,std::ios::beg);
47 if(m_hippo) {
48 skip_line(m_reader,m_sz);
49 skip_line(m_reader,m_sz);
50 }
51 }
52 virtual bool next() {
53 if(!m_sep) return false; //not inited.
54 if(m_reader.tellg()>=m_sz) return false;
55 // first time we are at bol but else we are at eol.
56 char c;
57 m_reader.get(c);
58 if(c==LF()){
59 if(m_reader.tellg()>=m_sz) {
60 //eof. Tell caller to stop looping on ntuple rows.
61 return false;
62 }
63 //eol. Next char read is going to be at bol.
64 } else {
65 m_reader.putback(c);
66 //bol
67 }
68 // ready for a new row :
69
70 while(skip_comment(m_reader,m_sz)){}
71 if(m_reader.tellg()>=m_sz) return false;
72
73 return _read_line();
74 }

77 return find_named<read::icol>(m_cols,a_name);

218 static bool find_sep(std::ostream& a_out,
219 std::istream& a_reader,bool a_hippo,
220 bool a_verbose,
221 char& a_sep){
222 // analyse first data line to find the char separator.
223
224 a_reader.clear();
225 a_reader.seekg(0,std::ios::end);
226 std::streampos sz = a_reader.tellg();
227 a_reader.seekg(0,std::ios::beg);
228 if(!sz) {
229 a_out << "tools::rcsv::ntuple::find_sep :"
230 << " stream is empty."
231 << std::endl;
232 a_sep = 0;
233 return false;
234 } //file empty.
235 if(a_verbose) a_out << "file size " << sz << std::endl;
236
237 if(a_hippo) { //skip first two lines :
238 if(!skip_line(a_reader,sz)) {a_sep = 0;return false;}
239 if(!skip_line(a_reader,sz)) {a_sep = 0;return false;}
240 } else {
241 while(skip_comment(a_reader,sz)){}
242 }
243 if(a_reader.tellg()>=sz) {a_sep=0;return false;} //no data line.
244
245 // get first data line :
246 std::string sfirst;
247 {char c;
248 while(true) {
249 if(a_reader.tellg()>=sz) break;
250 a_reader.get(c);
251 if((c==CR())||(c==LF())) break;
252 sfirst += c;
253 }}
254 if(sfirst.empty()) {
255 a_out << "tools::rcsv::ntuple::find_set :"
256 << " first datat line is empty."
257 << std::endl;
258 a_sep = 0;
259 return false;
260 }
261 if(a_verbose) a_out << "first data line \"" << sfirst << "\"" << std::endl;
262
263 //guess sep from first data line :
264 std::istringstream strm(sfirst.c_str());
265 double d;
266 strm >> d;
267 std::streampos pos = strm.tellg();
268 if(pos==std::streampos(-1)) {
269 a_out << "tools::rcsv::ntuple::find_sep :"
270 << " first line does not start with a number."
271 << std::endl;
272 a_sep = 0;
273 return false;
274 } //not a number.
275 if(a_verbose) a_out << "first number " << d
276 << " ending at pos " << pos << std::endl;
277 if(pos>=(std::streampos)sfirst.size()) {
278 a_out << "tools::rcsv::ntuple::find_sep :"
279 << " no separator found in first line."
280 << " pos " << pos
281 << " sfirst.size() " << sfirst.size()
282 << std::endl;
283 a_sep = 0;
284 return false;
285 } //no sep.
286
287 strm.get(a_sep);
288
289 return true;
290 }

292public:
293 bool initialize(std::ostream& a_out,
294 char a_sep = 0, //guessed
295 const std::string& a_suffix = "x", //col suffix
296 bool a_verbose = false) {
297 safe_clear<read::icol>(m_cols);
298 m_sep = 0;
299 m_sz = 0;
300
301 if(a_suffix.empty()) {
302 a_out << "tools::rcsv::ntuple::initialize : expect a column suffix." << std::endl;
303 return false;
304 }
305
306 m_reader.clear();
307 m_reader.seekg(0,std::ios::end);
308 m_sz = m_reader.tellg();
309 m_reader.seekg(0,std::ios::beg);
310 if(!m_sz) {
311 a_out << "tools::rcsv::ntuple::initialize :"
312 << " stream is empty."
313 << std::endl;
314 return false; //file empty.
315 }
316 if(a_verbose) a_out << "file size " << m_sz << std::endl;
317
318 std::vector<std::string> labels;
319 if(m_hippo) { //skip first two lines :
320 std::string title;
321 if(!read_line(m_reader,m_sz,title)) {
322 a_out << "tools::rcsv::ntuple::initialize : read_line() failed." << std::endl;
323 m_sz = 0;
324 return false;
325 }
326 std::string s;
327 if(!read_line(m_reader,m_sz,s)) {
328 a_out << "tools::rcsv::ntuple::initialize : (2) read_line() failed." << std::endl;
329 m_sz = 0;
330 return false;
331 }
332 words(s,"\t",false,labels); //false for glast.tnt that has a trailing \t.
333 } else {
334 while(skip_comment(m_reader,m_sz)){}
335 }
336 if(m_reader.tellg()>=m_sz) {
337 a_out << "tools::rcsv::ntuple::initialize : tellg() >= sz." << std::endl;
338 m_sz=0;
339 return false;
340 }
341
342 // get first data line :
343 std::string sfirst;
344 {{char c;
345 while(true) {
346 if(m_reader.tellg()>=m_sz) break;
347 m_reader.get(c);
348 if((c==CR())||(c==LF())) break;
349 sfirst += c;
350 }}
351 if(sfirst.empty()) {
352 a_out << "tools::rcsv::ntuple::initialize :"
353 << " first datat line is empty."
354 << std::endl;
355 m_sz = 0;
356 return false;
357 }}
358 if(a_verbose) a_out << "first data line \"" << sfirst << "\"" << std::endl;
359
360 if(a_sep) {
361 m_sep = a_sep;
362 } else {
363 //guess sep from first data line :
364 std::istringstream strm(sfirst.c_str());
365 double d;
366 strm >> d;
367 std::streampos pos = strm.tellg();
368 if(pos==std::streampos(-1)) {
369 a_out << "tools::rcsv::ntuple::initialize :"
370 << " first line does not start with a number."
371 << std::endl;
372 m_sz = 0;
373 return false;
374 }
375 if(a_verbose) a_out << "first number " << d << " ending at pos " << pos << std::endl;
376 if(pos>=(std::streampos)sfirst.size()) {
377 a_out << "tools::rcsv::ntuple::initialize :"
378 << " no separator found in first line."
379 << std::endl;
380 m_sz = 0;
381 return false;
382 }
383 strm.get(m_sep);
384 }
385 if(a_verbose) a_out << "sep " << (int)m_sep << std::endl;
386
387 // in case sep is ' ', there is an ambiguity with some leading
388 // space in front of first number.
389 if(m_sep==' ') strip(sfirst,leading,' ');
390
391 std::vector<std::string> ws;
392 {std::string sep;
393 sep += m_sep;
394 words(sfirst,sep,m_hippo?false:true,ws);}
395
396 // look if words are numbers :
397 if(a_verbose) a_out << "words " << ws.size() << std::endl;
398 unsigned int index = 0;
399 std::vector<std::string>::iterator it;
400 for(it=ws.begin();it!=ws.end();++it,index++) {
401 if(a_verbose) a_out << "word " << sout(*it) << "" << std::endl;
402/* with glast.tnt there is trailing \t that will induce an extra empty column.
403 if((*it).empty()) {
404 // do not accept :
405 // <num><sep><num><sep><sep><num>...
406 // but accept a trailing <sep> (glast.tnt) :
407 // <num><sep><num>....<sep><num><sep>
408 if(index==(ws.size()-1)) {
409 break;
410 } else {
411 a_out << "tools::rcsv::ntuple::initialize :"
412 << " empty word."
413 << std::endl;
414 safe_clear<read::icol>(m_cols);
415 m_sep = 0;
416 m_sz = 0;
417 return false;
418 }
419 }
420*/
421 std::string name = a_suffix;
422 if(!numas<uint64>(m_cols.size(),name)){}
423 if(m_hippo) {
424 if(index>=labels.size()) {
425 a_out << "tools::rcsv::ntuple::initialize :"
426 << " warning : not enough labels."
427 << std::endl;
428 } else {
429 name = labels[index];
430 }
431 }
432 double d;
433 if(to<double>(*it,d)) {
434 if(a_verbose) a_out << "number " << d << std::endl;
435 create_column<double>(name);
436 } else {
437 time_t time;
438 if(s2time(*it,time)) {
439 create_column<csv_time>(name);
440 } else {
441 std::vector<double> v;
442 std::string vec_sep;vec_sep += m_vec_sep;
443 if(snums<double>(*it,vec_sep,v)&&v.size()) {
444 create_column< std::vector<double> >(name);
445 } else {
446 create_column<std::string>(name);
447 }
448 }
449 }
450 }
451 size_t num = m_cols.size();
452 if(!num) {
453 a_out << "tools::rcsv::ntuple::initialize :"
454 << " zero columns."
455 << std::endl;
456 m_sep = 0;
457 m_sz = 0;
458 return false;
459 }
460
461 return true;
462 }

464 static const std::string& s_cid(cid a_id) {
465
466#define TOOLS_RCSV_NTUPLE_IF_CID(a__name,a__type) \
467 if(a_id==column<a__type>::id_class()) {\
468 static const std::string s_v(#a__name);\
469 return s_v;\
470 }
471
472#define TOOLS_RCSV_NTUPLE_IF_VEC_CID(a__name,a__type) \
473 if(a_id==column< std::vector<a__type> >::id_class()) {\
474 static const std::string s_v(#a__name+std::string("[]"));\
475 return s_v;\
476 }
477
478 TOOLS_RCSV_NTUPLE_IF_CID(char,char)
479 else TOOLS_RCSV_NTUPLE_IF_CID(short,short)
480 else TOOLS_RCSV_NTUPLE_IF_CID(int,int)
481 else TOOLS_RCSV_NTUPLE_IF_CID(int64,int64)
482
483 else TOOLS_RCSV_NTUPLE_IF_CID(float,float)
484 else TOOLS_RCSV_NTUPLE_IF_CID(double,double)
485
486 else TOOLS_RCSV_NTUPLE_IF_CID(uchar,uchar)
487 else TOOLS_RCSV_NTUPLE_IF_CID(ushort,ushort)
488 else TOOLS_RCSV_NTUPLE_IF_CID(uint,uint32) //WARNING
489 else TOOLS_RCSV_NTUPLE_IF_CID(uint64,uint64)
490
491 else TOOLS_RCSV_NTUPLE_IF_CID(bool,bool)
492 else if(a_id==column<std::string>::id_class()) {
493 static const std::string s_v("string");
494 return s_v;
495 }
496
497 else if(a_id==column<csv_time>::id_class()) {
498 static const std::string s_v("time");
499 return s_v;
500 }
501
502 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(char,char)
503 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(short,short)
504 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(int,int)
505 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(int64,int64)
506
507 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(float,float)
508 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(double,double)
509
510 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(uchar,uchar)
511 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(ushort,ushort)
512 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(uint,uint32) //WARNING
513 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(uint64,uint64)
514
515 else TOOLS_RCSV_NTUPLE_IF_VEC_CID(bool,bool)
516 else if(a_id==column< std::vector<std::string> >::id_class()) {
517 static const std::string s_v("string[]");
518 return s_v;
519 }
520
521#undef TOOLS_RCSV_NTUPLE_IF_CID
522#undef TOOLS_RCSV_NTUPLE_IF_VEC_CID
523
524 else {
525 static const std::string s_v("unknown");
526 return s_v;
527 }

636 }
637
638 //a_out << "tools::rroot::ntuple::initialize :"

640 // << std::endl;

708 return false;
709 }

711 } else if(_words[0]=="#title") {
712 if(_words.size()<1) {
713 a_out << "tools::rcsv::ntuple::read_commented_header :"
714 << " syntax error in " << sout(line)
715 << std::endl;

718 if(_words.size()==1) {
719 a_title.clear();
720 } else {
721 std::string::size_type pos = line.find(_words[0]);
722 pos += _words[0].size()+1;
723 a_title = line.substr(pos,line.size()-pos);
724 }
725 } else if((_words[0]=="#separator")) {
726 if(_words.size()!=2) {
727 a_out << "tools::rcsv::ntuple::read_commented_header :"
728 << " syntax error in " << sout(line)

730 return false;

737 return false;
738 }
739 a_sep = (char)uisep;
740 } else if((_words[0]=="#vector_separator")) {
741 if(_words.size()!=2) {
742 a_out << "tools::rcsv::ntuple::read_commented_header :"
743 << " syntax error in " << sout(line)
744 << std::endl;
745 return false;
746 }
747 unsigned int uisep;
748 if(!to(_words[1],uisep)) {
749 a_out << "tools::rcsv::ntuple::read_commented_header :"
750 << " syntax error in " << sout(line)
751 << std::endl;
752 return false;
753 }
754 a_vec_sep = (char)uisep;
755 } else if((_words[0]=="#column")) {
756 if(_words.size()<2) {
757 a_out << "tools::rcsv::ntuple::read_commented_header :"
758 << " syntax error in " << sout(line)
759 << std::endl;
760 return false;
761 }
762 std::string stype = _words[1];
763 std::string label;
764 if(_words.size()==2) {
765 label.clear();
766 } else {
767 std::string::size_type pos = line.find(_words[1]);
768 pos += _words[1].size()+1;
769 label = line.substr(pos,line.size()-pos);
770 }
771 //a_out << "column " << stype << " " << sout(label) << std::endl;
772 a_cols.push_back(col_desc(stype,label));
773 } else {
774 a_out << "tools::rcsv::ntuple::read_commented_header :"
775 << " syntax error in " << sout(line)
776 << ", unknown keyword " << sout(_words[0])
777 << std::endl;
778 //return false;
779 }
780 }
781
782/*
783 a_out << "class " << _class << std::endl;
784 a_out << "title " << _title << std::endl;
785 a_out << "separator " << _separator << std::endl;
786*/
787
788 return true;
789 }
