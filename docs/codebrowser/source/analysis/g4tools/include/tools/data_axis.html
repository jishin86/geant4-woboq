11class data_axis {
12public:
13 data_axis():m_min_value(0),m_max_value(0),m_is_log(false){}
14 virtual ~data_axis(){}
15public:
16 data_axis(const data_axis& a_from)
17 :m_min_value(a_from.m_min_value)
18 ,m_max_value(a_from.m_max_value)
19 ,m_is_log(a_from.m_is_log)

25 return *this;

27public:
28 bool is_log(bool a_v) {if(m_is_log==a_v) return false;m_is_log = a_v;return true;}
29 bool min_value(float a_v) {if(m_min_value==a_v) return false;m_min_value = a_v;return true;}
30 bool max_value(float a_v) {if(m_max_value==a_v) return false;m_max_value = a_v;return true;}
31 float min_value() const {return m_min_value;}
32 float max_value() const {return m_max_value;}
33 bool is_log() const {return m_is_log;}

35 void adjust() { //from hippodraw.
36 int _axis = 0;
37 float step;
38 float mylow, myhigh;
39 int N_NICE = 4;
40 static const float nice[/*N_NICE*/4] = { 1.0,2.0,2.5,5.0 };
41
42 if (m_min_value > m_max_value) {
43 float low = m_min_value;
44 m_min_value = m_max_value;
45 m_max_value = low;
46 } else if (m_min_value == m_max_value) {
47 float value = m_min_value;
48 m_min_value = value - 1;
49 m_max_value = value + 1;
50 return;
51 }
52
53 //if (m_steps <= 0) { //if letting the if and m_steps as a field, twice this function do not give the same result.
54 _axis = 1;
55 unsigned int m_steps = 10;
56 //}
57
58 // Round the "bin width" to a nice number.
59 // If this is being done for an axis (ie m_steps was 0 , then
60 // we don't have to go > *m_max_value.
61 //
62 float w = (m_max_value - m_min_value)/((float)m_steps);
63 float mag = ffloor(flog10(w));
64 int i = 0;
65 do {
66 step = nice[i] * fpow(10.0,mag);
67 mylow = ffloor(m_min_value/step) * step;
68 //myhigh = _axis==1 ? fceil(m_max_value/step) * step : mylow + step * m_steps;
69 myhigh = fceil(m_max_value/step) * step; //quiet Coverity.
70 i++;
71 if (i>=N_NICE) {i = 0;mag++;}
72 }
73 while ( ( (_axis==1) && (myhigh < m_max_value)) ||
74 ( (_axis==0) && (myhigh <= m_max_value)) );
75
76 float range = myhigh - mylow;
77
78 // we now have decided on a range. Try to move
79 // m_min_value/m_max_value a little
80 // to end up on a nice number.
81 //
82 // first check if either end is near 0.0
83 if ( !m_is_log && (m_min_value >= 0.0) &&
84 (( (_axis==1) && (range>=m_max_value) ) ||
85 ( (_axis==0) && (range>m_max_value) )) ) {
86 m_min_value = 0.0;
87 m_max_value = range;
88 return;
89 }
90
91 if ( (( (_axis==1) && (m_max_value<=0.0) ) ||
92 ( (_axis==0) && (m_max_value<0.0) ))
93 && (-range<=m_min_value)) {
94 m_max_value = 0.0;
95 m_min_value = -range;
96 return;
97 }
98
99 // try to round *m_min_value.
100 // correction
101 if( m_is_log && (m_min_value<=0.0)) m_min_value = 1.0;
102
103 i = N_NICE-1;
104 mag = myhigh != 0.0 ? fceil(flog10(ffabs(myhigh))) : fceil(flog10(ffabs(mylow)));
105
106 do {
107 step = nice[i] * fpow(10.0,mag);
108 mylow = ffloor(m_min_value/step) * step;
109 myhigh = mylow + range;
110 i--;
111 if (i<0) {
112 i = N_NICE-1;
113 mag--;
114 }
115 }
116 while (( m_is_log && (mylow <= 0.0) ) ||
117 ( (_axis==1) && (myhigh < m_max_value) ) ||
118 ( (_axis==0) && (myhigh <= m_max_value) ) );
119
120 m_min_value = mylow;
121 m_max_value = myhigh;
122 }
123protected:
124 float m_min_value;
125 float m_max_value;
126 //int m_steps;
127 bool m_is_log;
128};
