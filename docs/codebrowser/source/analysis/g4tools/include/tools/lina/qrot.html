41 qrot& operator*=(const qrot& a_q) {
42 //Multiplies the quaternions.
43 //Note that order is important when combining quaternions with the
44 //multiplication operator.
45 // Formula from <http://www.lboro.ac.uk/departments/ma/gallery/quat/>
46
47 T tx = m_quat.v0();
48 T ty = m_quat.v1();
49 T tz = m_quat.v2();
50 T tw = m_quat.v3();
51
52 T qx = a_q.m_quat.v0();
53 T qy = a_q.m_quat.v1();
54 T qz = a_q.m_quat.v2();
55 T qw = a_q.m_quat.v3();
56
57 m_quat.set_value(qw*tx + qx*tw + qy*tz - qz*ty,
58 qw*ty - qx*tz + qy*tw + qz*tx,
59 qw*tz + qx*ty - qy*tx + qz*tw,
60 qw*tw - qx*tx - qy*ty - qz*tz);
61 m_quat.normalize();
62 return *this;
63 }

77 bool invert(){
78 T length = m_quat.length();
79 if(length==T()) return false;
80
81 // Optimize by doing 1 div and 4 muls instead of 4 divs.
82 T inv = one() / length;
83
84 m_quat.set_value(-m_quat.v0() * inv,
85 -m_quat.v1() * inv,
86 -m_quat.v2() * inv,
87 m_quat.v3() * inv);
88
89 return true;
90 }

92 bool inverse(qrot& a_r) const {
93 //Non-destructively inverses the rotation and returns the result.
94 T length = m_quat.length();
95 if(length==T()) return false;
96
97 // Optimize by doing 1 div and 4 muls instead of 4 divs.
98 T inv = one() / length;
99
100 a_r.m_quat.set_value(-m_quat.v0() * inv,
101 -m_quat.v1() * inv,
102 -m_quat.v2() * inv,
103 m_quat.v3() * inv);
104
105 return true;
106 }

109 bool set_value(const VEC3& a_axis,T a_radians,T(*a_sin)(T),T(*a_cos)(T)) {
110 // Reset rotation with the given axis-of-rotation and rotation angle.
111 // Make sure axis is not the null vector when calling this method.
112 // From <http://www.automation.hut.fi/~jaro/thesis/hyper/node9.html>.
113 if(a_axis.length()==T()) return false;
114 m_quat.v3(a_cos(a_radians/2));
115 T sineval = a_sin(a_radians/2);
116 VEC3 a = a_axis;
117 a.normalize();
118 m_quat.v0(a.v0() * sineval);
119 m_quat.v1(a.v1() * sineval);
120 m_quat.v2(a.v2() * sineval);
121 return true;
122 }
123
124 bool set_value(const VEC3& a_from,const VEC3& a_to,T(*a_sqrt)(T),T(*a_fabs)(T)) {
125 // code taken from coin3d/SbRotation.
126 VEC3 from(a_from);
127 if(from.normalize()==T()) return false;
128 VEC3 to(a_to);
129 if(to.normalize()==T()) return false;
130
131 T dot = from.dot(to);
132 VEC3 crossvec;from.cross(to,crossvec);
133 T crosslen = crossvec.normalize();
134
135 if(crosslen == T()) { // Parallel vectors
136 // Check if they are pointing in the same direction.
137 if (dot > T()) {
138 m_quat.set_value(0,0,0,1);
139 } else {
140 // Ok, so they are parallel and pointing in the opposite direction
141 // of each other.
142 // Try crossing with x axis.
143 VEC3 t;from.cross(VEC3(1,0,0),t);
144 // If not ok, cross with y axis.
145 if(t.normalize() == T()) {
146 from.cross(VEC3(0,1,0),t);
147 t.normalize();
148 }
149 m_quat.set_value(t[0],t[1],t[2],0);
150 }
151 } else { // Vectors are not parallel
152 // The fabs() wrapping is to avoid problems when `dot' "overflows"
153 // a tiny wee bit, which can lead to sqrt() returning NaN.
154 crossvec *= (T)a_sqrt(half() * a_fabs(one() - dot));
155 // The fabs() wrapping is to avoid problems when `dot' "underflows"
156 // a tiny wee bit, which can lead to sqrt() returning NaN.
157 m_quat.set_value(crossvec[0], crossvec[1], crossvec[2],(T)a_sqrt(half()*a_fabs(one()+dot)));
158 }
159
160 return true;
161 }

164 bool value(VEC3& a_axis,T& a_radians,T(*a_sin)(T),T(*a_acos)(T)) const { //WARNING a_acos and NOT a_cos
165 //WARNING : can fail.
166 if( (m_quat.v3()<minus_one()) || (m_quat.v3()> one()) ){ ////???
167 a_axis.set_value(0,0,1);
168 a_radians = 0;
169 return false;
170 }
171
172 a_radians = a_acos(m_quat.v3()) * 2;
173 T sineval = a_sin(a_radians/2);
174
175 if(sineval==T()) { //???
176 a_axis.set_value(0,0,1);
177 a_radians = 0;
178 return false;
179 }
180 a_axis.set_value(m_quat.v0()/sineval,
181 m_quat.v1()/sineval,
182 m_quat.v2()/sineval);
183 return true;
184 }

226 template <class MAT4>
227 void value(MAT4& a_m) const {
228 //Return this rotation in the form of a matrix.
229 const T x = m_quat.v0();
230 const T y = m_quat.v1();
231 const T z = m_quat.v2();
232 const T w = m_quat.v3();
233 // z = w + x * i + y * j + z * k
234
235 // first row :
236 a_m.v00(w*w + x*x - y*y - z*z);
237 a_m.v01(2*x*y - 2*w*z);
238 a_m.v02(2*x*z + 2*w*y);
239 a_m.v03(0);
240
241 // second row :
242 a_m.v10(2*x*y + 2*w*z);
243 a_m.v11(w*w - x*x + y*y - z*z);
244 a_m.v12(2*y*z - 2*w*x);
245 a_m.v13(0);
246
247 // third row :
248 a_m.v20(2*x*z - 2*w*y);
249 a_m.v21(2*y*z + 2*w*x);
250 a_m.v22(w*w - x*x - y*y + z*z);
251 a_m.v23(0);
252
253 // fourth row :
254 a_m.v30(0);
255 a_m.v31(0);
256 a_m.v32(0);
257 a_m.v33(w*w + x*x + y*y + z*z);
258 }

260 void mul_vec(const VEC3& a_in,VEC3& a_out) const {
261 const T x = m_quat.v0();
262 const T y = m_quat.v1();
263 const T z = m_quat.v2();
264 const T w = m_quat.v3();
265
266 // first row :
267 T v0 = (w*w + x*x - y*y - z*z) * a_in.v0()
268 + (2*x*y - 2*w*z) * a_in.v1()
269 + (2*x*z + 2*w*y) * a_in.v2();
270
271 T v1 = (2*x*y + 2*w*z) * a_in.v0()
272 +(w*w - x*x + y*y - z*z) * a_in.v1()
273 + (2*y*z - 2*w*x) * a_in.v2();
274
275 T v2 = (2*x*z - 2*w*y) * a_in.v0()
276 + (2*y*z + 2*w*x) * a_in.v1()
277 +(w*w - x*x - y*y + z*z) * a_in.v2();
278
279 a_out.set_value(v0,v1,v2);
280 }

304 void mul_3(T& a_x,T& a_y,T& a_z) const {
305 const T x = m_quat.v0();
306 const T y = m_quat.v1();
307 const T z = m_quat.v2();
308 const T w = m_quat.v3();
309
310 // first row :
311 T v0 = (w*w + x*x - y*y - z*z) * a_x
312 + (2*x*y - 2*w*z) * a_y
313 + (2*x*z + 2*w*y) * a_z;
314
315 T v1 = (2*x*y + 2*w*z) * a_x
316 +(w*w - x*x + y*y - z*z) * a_y
317 + (2*y*z - 2*w*x) * a_z;
318
319 T v2 = (2*x*z - 2*w*y) * a_x
320 + (2*y*z + 2*w*x) * a_y
321 +(w*w - x*x - y*y + z*z) * a_z;
322
323 a_x = v0;
324 a_y = v1;
325 a_z = v2;
326 }
