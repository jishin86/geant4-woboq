15template <class T>
16class vec4 {
17#ifdef TOOLS_MEM
18public:
19 static const std::string& s_class() {
20 static const std::string s_v("tools::vec4");
21 return s_v;
22 }
23#endif
24protected:
25 static T zero() {return T();}
26 static T minus_one() {return T(-1);}
27public:
28 typedef T elem_t;
29 unsigned int dimension() const {return 4;}
30public:
31 vec4(){
32#ifdef TOOLS_MEM
33 mem::increment(s_class().c_str());
34#endif
35 m_data[0] = T();
36 m_data[1] = T();
37 m_data[2] = T();
38 m_data[3] = T();

40 vec4(const T a_vec[4]) {
41#ifdef TOOLS_MEM
42 mem::increment(s_class().c_str());
43#endif

45 m_data[1] = a_vec[1];
46 m_data[2] = a_vec[2];
47 m_data[3] = a_vec[3];

49 vec4(const T& a0,const T& a1,const T& a2,const T& a3
50#ifdef TOOLS_MEM
51 ,bool a_inc = true
52#endif
53 ) {
54#ifdef TOOLS_MEM
55 if(a_inc) mem::increment(s_class().c_str());
56#endif
57 m_data[0] = a0;
58 m_data[1] = a1;
59 m_data[2] = a2;
60 m_data[3] = a3;
61 }
62 virtual ~vec4() {

64 mem::decrement(s_class().c_str());
65#endif
66 }
67public:
68 vec4(const vec4& a_from){
69#ifdef TOOLS_MEM
70 mem::increment(s_class().c_str());
71#endif
72 m_data[0] = a_from.m_data[0];
73 m_data[1] = a_from.m_data[1];
74 m_data[2] = a_from.m_data[2];
75 m_data[3] = a_from.m_data[3];
76 }
77 vec4& operator=(const vec4& a_from) {
78 m_data[0] = a_from.m_data[0];
79 m_data[1] = a_from.m_data[1];
80 m_data[2] = a_from.m_data[2];
81 m_data[3] = a_from.m_data[3];
82 return *this;
83 }
84public:
85 const T& v0() const { return m_data[0];}
86 const T& v1() const { return m_data[1];}
87 const T& v2() const { return m_data[2];}
88 const T& v3() const { return m_data[3];}
89
90 void v0(const T& a_value) { m_data[0] = a_value;}
91 void v1(const T& a_value) { m_data[1] = a_value;}
92 void v2(const T& a_value) { m_data[2] = a_value;}
93 void v3(const T& a_value) { m_data[3] = a_value;}
94
95 //T x() const {return m_data[0];}
96 //T y() const {return m_data[1];}
97 //T z() const {return m_data[2];}
98
99 void set_value(const T& a0,const T& a1,const T& a2,const T& a3) {
100 m_data[0] = a0;
101 m_data[1] = a1;
102 m_data[2] = a2;
103 m_data[3] = a3;
104 }
105 void set_value(const T aV[4]) {
106 m_data[0] = aV[0];
107 m_data[1] = aV[1];
108 m_data[2] = aV[2];
109 m_data[3] = aV[3];
110 }
111 void value(T& a0,T& a1,T& a2,T& a3) const {
112 a0 = m_data[0];
113 a1 = m_data[1];
114 a2 = m_data[2];
115 a3 = m_data[3];
116 }
117
118 bool set_value(unsigned int a_index,const T& a_value) {
119 if(a_index>=4) return false;
120 m_data[a_index] = a_value;
121 return true;
122 }
123
124 T length(T(*a_sqrt)(T)) const {
125 return a_sqrt(m_data[0]*m_data[0]+m_data[1]*m_data[1]+m_data[2]*m_data[2]+m_data[3]*m_data[3]);
126 }
127
128 T normalize(T(*a_sqrt)(T)) {
129 T norme = length(a_sqrt);
130 if(norme==T()) return T();
131 divide(norme);
132 return norme;
133 }
134
135 bool equal(const vec4& a_vec) const {
136 if(m_data[0]!=a_vec.m_data[0]) return false;
137 if(m_data[1]!=a_vec.m_data[1]) return false;
138 if(m_data[2]!=a_vec.m_data[2]) return false;
139 if(m_data[3]!=a_vec.m_data[3]) return false;
140 return true;
141 }
142 bool equal(const vec4& a_vec,const T& a_epsil) const {
143 T* tp = (T*)m_data;
144 T* ap = (T*)a_vec.m_data;
145 for(unsigned int i=0;i<4;i++,tp++,ap++) {
146 T diff = (*tp) - (*ap);
147 if(diff<zero()) diff *= minus_one();
148 if(diff>=a_epsil) return false;
149 }
150 return true;
151 }
152
153 bool is_proportional(const vec4& a_vec,T& a_factor) const {
154 // If true, then : a_vec = a_factor * this.
155 a_factor = zero();
156 bool first = true;
157 T* tp = (T*)m_data;
158 T* ap = (T*)a_vec.m_data;
159 for(unsigned int i=0;i<4;i++,tp++,ap++) {
160 if( ((*tp)==zero()) && ((*ap)==zero())) {
161 continue;
162 } else if( ((*tp)!=zero()) && ((*ap)==zero())) {
163 return false;
164 } else if( ((*tp)==zero()) && ((*ap)!=zero())) {
165 return false;
166 } else {
167 if(first) {
168 a_factor = (*ap)/(*tp);
169 first = false;
170 } else {
171 if((*ap)!=(*tp)*a_factor) return false;
172 }
173 }
174 }
175 return true;
176 }
177
178 void multiply(const T& a_T) {
179 m_data[0] *= a_T;
180 m_data[1] *= a_T;
181 m_data[2] *= a_T;
182 m_data[3] *= a_T;
183 }
184
185 bool divide(const T& a_T) {
186 if(a_T==T()) return false;
187 m_data[0] /= a_T;
188 m_data[1] /= a_T;
189 m_data[2] /= a_T;
190 m_data[3] /= a_T;
191 return true;
192 }
193
194 void add(const vec4& a_v) {
195 m_data[0] += a_v.m_data[0];
196 m_data[1] += a_v.m_data[1];
197 m_data[2] += a_v.m_data[2];
198 m_data[3] += a_v.m_data[3];
199 }
200
201 void add(const T& a0,const T& a1,const T& a2,const T& a3) {
202 m_data[0] += a0;
203 m_data[1] += a1;
204 m_data[2] += a2;
205 m_data[3] += a3;
206 }
207
208 void subtract(const vec4& a_v) {
209 m_data[0] -= a_v.m_data[0];
210 m_data[1] -= a_v.m_data[1];
211 m_data[2] -= a_v.m_data[2];
212 m_data[3] -= a_v.m_data[3];
213 }
214
215 void subtract(const T& a0,const T& a1,const T& a2,const T& a3) {
216 m_data[0] -= a0;
217 m_data[1] -= a1;
218 m_data[2] -= a2;
219 m_data[3] -= a3;
220 }
221
222public: //operators
223 T& operator[](size_t a_index) {
224 //WARNING : no check on a_index.
225 return m_data[a_index];
226 }
227 const T& operator[](size_t a_index) const {
228 //WARNING : no check on a_index.
229 return m_data[a_index];
230 }
231
232 vec4 operator+(const vec4& a_v) const {
233 return vec4(m_data[0]+a_v.m_data[0],
234 m_data[1]+a_v.m_data[1],
235 m_data[2]+a_v.m_data[2],
236 m_data[3]+a_v.m_data[3]);
237 }
238
239 vec4 operator-(const vec4& a_v) const {
240 return vec4(m_data[0]-a_v.m_data[0],
241 m_data[1]-a_v.m_data[1],
242 m_data[2]-a_v.m_data[2],
243 m_data[3]-a_v.m_data[3]);
244 }
245
246 vec4 operator*(const T& a_v) const {
247 return vec4(m_data[0]*a_v,
248 m_data[1]*a_v,
249 m_data[2]*a_v,
250 m_data[3]*a_v);
251 }
252
253 bool operator==(const vec4& a_v) const {return equal(a_v);}
254 bool operator!=(const vec4& a_v) const {return !operator==(a_v);}
