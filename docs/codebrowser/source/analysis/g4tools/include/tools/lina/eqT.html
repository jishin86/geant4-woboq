9template <class NUMBER,class PREC>
10inline bool numbers_are_equal(const NUMBER& a_left,const NUMBER& a_right,const PREC& a_prec,PREC(*a_fabs)(const NUMBER&)) {
11 NUMBER diff = a_left - a_right;
12 if(a_fabs(diff)>=a_prec) return false;
13 return true;
14}

16template <class NUMBER,class PREC>
17inline bool is_zero(const NUMBER& a_left,const PREC& a_prec,PREC(*a_fabs)(const NUMBER&)) {
18 if(a_fabs(a_left)>=a_prec) return false;
19 return true;
20}

29template <class VEC,class PREC>
30inline bool vectors_are_equal(const VEC& a_1,const VEC& a_2,const PREC& a_prec,PREC(*a_fabs)(PREC)) {
31 if(a_1.size()!=a_2.size()) return false;
32 typedef typename VEC::size_type sz_t;
33 sz_t sz = a_1.size();
34 //bool status = true;
35 for(sz_t index=0;index<sz;index++) {
36 if(!numbers_are_equal(a_1[index],a_2[index],a_prec,a_fabs))
37 //{ ::printf("debug : vectors_are_equals : %lu : %g %g\n",index,a_1[index],a_2[index]);
38 return false;
39 //status = false;
40 //}
41 }
42 return true;
43 //return status;
44}

47inline bool vecvecs_are_equal(const VECVEC& a_1,const VECVEC& a_2,const PREC& a_prec,PREC(*a_fabs)(PREC)) {
48 if(a_1.size()!=a_2.size()) return false;
49 typedef typename VECVEC::size_type sz_t;

52 if(!vectors_are_equal(a_1[index],a_2[index],a_prec,a_fabs)) return false;
53 }
54 return true;
