137 void PS_PAGE_SCALE(float a_width,float a_height,bool a_portrait = true){
138 //NOTE : no check done on [a_width,a_height]<=0
139
140 PS_SCALE(1/METAFILE_SCALE(),1/METAFILE_SCALE());
141 PS_TRANSLATE(m_deviceWidth/20,m_deviceHeight/30);
142
143 float scale;
144 if(m_deviceWidth<=m_deviceHeight) {
145 scale =
146 (a_height<=a_width?m_deviceWidth/a_width:m_deviceWidth/a_height);
147 } else {
148 scale =
149 (a_height<=a_width?m_deviceHeight/a_width:m_deviceHeight/a_height);
150 }
151
152 {float xtra,ytra;
153 if(a_portrait) {
154 xtra = (m_deviceWidth - scale * a_width)/2;
155 ytra = (m_deviceHeight - scale * a_height)/2;
156 } else {
157 PS_TRANSLATE(m_deviceWidth,0);
158 PS_ROTATE(90);
159 xtra = (m_deviceHeight - scale * a_width)/2;
160 ytra = (m_deviceWidth - scale * a_height)/2;
161 }
162 PS_TRANSLATE(xtra,ytra);}
163
164 PS_SCALE(scale,scale);
165 }

199 void PS_IMAGE(Uint a_width,Uint a_height,rgb_nbit a_nbit,rgb_func a_proc,void* a_tag){
200 //NOTE : no check done on [a_width,a_height] being zero.
201
202 Uint row,col;
203 VCol dr,dg,db;
204 Uchar red,green,blue,b;
205
206 PS_SAVE();
207 in_buffer("%d %d scale ", a_width, a_height );
208 bool status = true;
209 if(a_nbit==rgb_bw) { //grey or black_white
210 in_buffer ("/picstr %d string def ",a_width);
211 in_buffer ("%d %d %d ",a_width,a_height,8);
212 in_buffer ("[ %d 0 0 -%d 0 %d ] ",a_width,a_height,a_height);
213 in_buffer ("{ currentfile picstr readhexstring pop } " );
214 PrintFLN ("image " );
215 for ( row = 0; row < a_height; row++ ) {
216 for ( col = 0; col < a_width; col++) {
217 status = a_proc(a_tag,col,row,dr,dg,db)?status:false;
218 VCol fgrey = rgb2grey(dr,dg,db);
219 Uchar grey = (Uchar) ( 255. * fgrey);
220 WriteByte(grey);
221 }
222 }
223 int nbhex = a_width * a_height * 2;
224 PrintFLN ("%%%% nbhex digit :%d ",nbhex);
225 PrintFLN ("%%%% nbhex/record_length :%d ",
226 nbhex/METAFILE_RECORD_LENGTH());
227 PrintFLN ("%%%% nbhex%%record_length :%d ",
228 nbhex%METAFILE_RECORD_LENGTH());
229
230 } else if(a_nbit==rgb_2) {
231 int nbyte2 = (a_width * 3)/4;
232 nbyte2 /=3;
233 nbyte2 *=3;
234 Uint col_max = (nbyte2 * 4)/3;
235 // 2 bit for r and g and b.
236 // rgbs following each other.
237 in_buffer ("/rgbstr %d string def ",nbyte2);
238 in_buffer ("%d %d %d ",col_max,a_height,2);
239 in_buffer ("[ %d 0 0 -%d 0 %d ] ",col_max,a_height,a_height);
240 in_buffer ("{ currentfile rgbstr readhexstring pop } " );
241 in_buffer ("false 3 " );
242 PrintFLN ("colorimage " );
243 for ( row = 0; row < a_height; row++ ) {
244 for ( col = 0; col < col_max; col+=4) {
245 status = a_proc(a_tag,col,row,dr,dg,db)?status:false;
246 red = (Uchar) ( 3. * dr);
247 green = (Uchar) ( 3. * dg);
248 blue = (Uchar) ( 3. * db);
249 b = red;
250 b = (b<<2)+green;
251 b = (b<<2)+blue;
252
253 status = a_proc(a_tag,col+1,row,dr,dg,db)?status:false;
254 red = (Uchar) ( 3. * dr);
255 green = (Uchar) ( 3. * dg);
256 blue = (Uchar) ( 3. * db);
257 b = (b<<2)+red;
258 WriteByte(b);
259
260 b = green;
261 b = (b<<2)+blue;
262
263 status = a_proc(a_tag,col+2,row,dr,dg,db)?status:false;
264 red = (Uchar) ( 3. * dr);
265 green = (Uchar) ( 3. * dg);
266 blue = (Uchar) ( 3. * db);
267 b = (b<<2)+red;
268 b = (b<<2)+green;
269 WriteByte(b);
270 b = blue;
271
272 status = a_proc(a_tag,col+3,row,dr,dg,db)?status:false;
273 red = (Uchar) ( 3. * dr);
274 green = (Uchar) ( 3. * dg);
275 blue = (Uchar) ( 3. * db);
276 b = (b<<2)+red;
277 b = (b<<2)+green;
278 b = (b<<2)+blue;
279 WriteByte(b);
280 }
281 }
282
283 } else if(a_nbit==rgb_4) {
284 int nbyte4 = (a_width * 3)/2;
285 nbyte4 /=3;
286 nbyte4 *=3;
287 Uint col_max = (nbyte4 * 2)/3;
288 // 4 bit for r and g and b.
289 // rgbs following each other.
290 in_buffer ("/rgbstr %d string def ",nbyte4);
291 in_buffer ("%d %d %d ",col_max,a_height,4);
292 in_buffer ("[ %d 0 0 -%d 0 %d ] ",col_max,a_height,a_height);
293 in_buffer ("{ currentfile rgbstr readhexstring pop } " );
294 in_buffer ("false 3 " );
295 PrintFLN ("colorimage " );
296 for ( row = 0; row < a_height; row++ ) {
297 for ( col = 0; col < col_max; col+=2) {
298 status = a_proc(a_tag,col,row,dr,dg,db)?status:false;
299 red = (Uchar) ( 15. * dr);
300 green = (Uchar) ( 15. * dg);
301 in_buffer ("%x%x",red,green);
302 blue = (Uchar) ( 15. * db);
303
304 status = a_proc(a_tag,col+1,row,dr,dg,db)?status:false;
305 red = (Uchar) ( 15. * dr);
306 in_buffer ("%x%x",blue,red);
307 green = (Uchar) ( 15. * dg);
308 blue = (Uchar) ( 15. * db);
309 in_buffer ("%x%x",green,blue);
310 }
311 }
312
313 } else if(a_nbit==rgb_8) {
314 int nbyte8 = a_width * 3;
315 // 8 bit for r and g and b.
316 in_buffer ("/rgbstr %d string def ",nbyte8);
317 in_buffer ("%d %d %d ",a_width,a_height,8);
318 in_buffer ("[ %d 0 0 -%d 0 %d ] ",a_width,a_height,a_height);
319 in_buffer ("{ currentfile rgbstr readhexstring pop } " );
320 in_buffer ("false 3 " );
321 PrintFLN ("colorimage " );
322 for ( row = 0; row < a_height; row++ ) {
323 for ( col = 0; col < a_width; col++) {
324 status = a_proc(a_tag,col,row,dr,dg,db)?status:false;
325 red = (Uchar) ( 255. * dr);
326 WriteByte (red);
327 green = (Uchar) ( 255. * dg);
328 WriteByte (green);
329 blue = (Uchar) ( 255. * db);
330 WriteByte (blue);
331 }
332 }
333 } else {
334 m_out << "PS_IMAGE :"
335 << " unknown rgb nbit " << a_nbit
336 << std::endl;
337 }
338 if(!status) {
339 m_out << "PS_IMAGE :"
340 << " problem to retreive some pixel rgb."
341 << std::endl;
342 }
343 PS_RESTORE();
344 }

364 bool in_buffer(const char* a_format,...){
365 va_list args;
366 va_start(args,a_format);
367 bool status = vsprintf(m_string,2048,a_format,args);
368 va_end(args);
369 if(!status) {
370 m_out << "tools::wps::in_buffer : overflow." << std::endl;
371 return false;
372 }
373
374 size_t length = m_string.size();
375 if(length>METAFILE_RECORD_LENGTH()) {
376 m_out << "tools::wps::in_buffer : overflow." << std::endl;
377 return false;
378 }
379
380 size_t nlength = m_number + length;
381 if(nlength>METAFILE_RECORD_LENGTH()) {
382 m_buffer[m_number] = '\0';
383 if(::fprintf(m_file,"%s\n",(char*)m_buffer)<0) {
384 m_out << "tools::wps::in_buffer : fprintf failed." << std::endl;
385 }
386 m_number = 0;
387 nlength = length;
388 }
389 Uchar* pointer = m_buffer + m_number;
390 ::strcpy((char*)pointer,m_string.c_str());
391 m_number = nlength;
392 return true;
393 }
394
395 bool PrintFLN(const char* a_format,...){
396 va_list args;
397 va_start(args,a_format);
398 bool status = vsprintf(m_string,2048,a_format,args);
399 va_end(args);
400 if(!status) {
401 m_out << "tools::wps::PrintFLN : overflow." << std::endl;
402 return false;
403 }
404
405 // put buffer in file :
406 if(m_number>0) {
407 m_buffer[m_number] = '\0';
408 if(::fprintf(m_file,"%s\n",(char*)m_buffer)<0) {
409 m_out << "tools::wps::PrintFLN : fprintf failed." << std::endl;
410 }
411 m_number = 0;
412 }
413 if(::fprintf(m_file,"%s\n",m_string.c_str())<0) {
414 m_out << "tools::wps::PrintFLN : fprintf failed." << std::endl;
415 }
416
417 return true;
418 }
