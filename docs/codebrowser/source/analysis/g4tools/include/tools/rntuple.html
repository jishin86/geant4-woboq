15namespace tools {
16namespace read {
17
18class icol {
19public:
20 virtual ~icol(){}
21public:
22 virtual void* cast(cid) const = 0;
23 virtual cid id_cls() const = 0;
24public:
25 virtual const std::string& name() const = 0;
26public:
27 virtual void stop() {}
28 virtual bool fetch_entry() const {return false;} //for binded column API.
29};
30
31template <class T>
32class icolumn : public virtual icol {
33public:
34 static cid id_class() {
35 static const T s_v = T(); //do that for T = std::string.
36 return _cid(s_v);
37 }
38public: //icol
39 virtual void* cast(cid a_class) const {
40 if(void* p = cmp_cast<icolumn>(this,a_class)) return p;
41 return 0;
42 }
43 virtual cid id_cls() const {return id_class();}
44public:
45 virtual ~icolumn(){}
46public:
47 virtual bool get_entry(T&) const = 0;
48};
49
50class intuple {
51public:
52 virtual ~intuple(){}
53public:
54 virtual void start() = 0;
55 virtual bool next() = 0;
56 virtual icol* find_icol(const std::string&) = 0;
57 virtual const std::vector<icol*>& columns() const = 0;
58public:
59 virtual void stop() {}
60public:
61 template <class T>
62 icolumn<T>* find_column(const std::string& a_name){
63 read::icol* col = find_icol(a_name);
64 if(!col) return 0;
65 return id_cast<read::icol, icolumn<T> >(*col);
66 }
67};
