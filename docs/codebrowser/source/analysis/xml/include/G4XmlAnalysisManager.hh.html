40#include <memory>

42class G4XmlFileManager;
43class G4XmlNtupleManager;
44
45class G4XmlAnalysisManager : public G4ToolsAnalysisManager
46{
47 public:
48 explicit G4XmlAnalysisManager(G4bool isMaster = true);
49 ~G4XmlAnalysisManager();
50
51 // static methods
52 static G4XmlAnalysisManager* Instance();
53 static G4bool IsInstance();
54
55 // Access methods
56 tools::waxml::ntuple* GetNtuple() const;
57 tools::waxml::ntuple* GetNtuple(G4int ntupleId) const;
58
59 // Iterators
60 std::vector<tools::waxml::ntuple*>::iterator BeginNtuple();
61 std::vector<tools::waxml::ntuple*>::iterator EndNtuple();

64
65 protected:
66 // virtual methods from base class
67 virtual G4bool OpenFileImpl(const G4String& fileName) final;
68 virtual G4bool WriteImpl() final;
69 virtual G4bool CloseFileImpl(G4bool reset) final;
70 virtual G4bool IsOpenFileImpl() const final;
71
72 private:

74 static G4XmlAnalysisManager* fgMasterInstance;
75 static G4ThreadLocal G4XmlAnalysisManager* fgInstance;
76
77 // methods

79 G4bool WriteT(const std::vector<T*>& htVector,
80 const std::vector<G4HnInformation*>& hnVector,
81 const G4String& directoryName,
82 const G4String& hnType);
83 G4bool WriteH1();
84 G4bool WriteH2();
85 G4bool WriteH3();
86 G4bool WriteP1();
87 G4bool WriteP2();
88 G4bool WriteNtuple();
89 G4bool CloseNtupleFiles();
90 G4bool Reset();
91
92 // data members
93 G4XmlNtupleManager* fNtupleManager;
94 std::shared_ptr<G4XmlFileManager> fFileManager;
95};
