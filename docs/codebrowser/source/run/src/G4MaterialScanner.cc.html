50G4MaterialScanner::G4MaterialScanner()
51{
52 theRayShooter = new G4RayShooter();
53 theMessenger = new G4MatScanMessenger(this);
54 theEventManager = G4EventManager::GetEventManager();
55
56 theUserEventAction = 0;
57 theUserStackingAction = 0;
58 theUserTrackingAction = 0;
59 theUserSteppingAction = 0;
60
61 theMatScannerEventAction = 0;
62 theMatScannerStackingAction = 0;
63 theMatScannerTrackingAction = 0;
64 theMatScannerSteppingAction = 0;
65
66 eyePosition = G4ThreeVector(0.,0.,0.);
67 nTheta = 91;
68 thetaMin = 0.*deg;
69 thetaSpan = 90.*deg;
70 nPhi = 37;
71 phiMin = 0.*deg;
72 phiSpan = 360.*deg;
73
74 regionSensitive = false;
75 regionName = "notDefined";
76 theRegion = 0;
77}
78
79G4MaterialScanner::~G4MaterialScanner()
80{
81 delete theRayShooter;
82 delete theMatScannerSteppingAction;
83 delete theMessenger;
84}

86void G4MaterialScanner::Scan()
87{
88 G4StateManager* theStateMan = G4StateManager::GetStateManager();
89 G4ApplicationState currentState = theStateMan->GetCurrentState();
90 if(currentState!=G4State_Idle)
91 {
92 G4cerr << "Illegal application state - Scan() ignored." << G4endl;
93 return;
94 }
95
96 if(!theMatScannerSteppingAction)
97 { theMatScannerSteppingAction = new G4MSSteppingAction(); }
98 StoreUserActions();
99 DoScan();
100 RestoreUserActions();
101}
102
103void G4MaterialScanner::StoreUserActions()
104{
105 theUserEventAction = theEventManager->GetUserEventAction();
106 theUserStackingAction = theEventManager->GetUserStackingAction();
107 theUserTrackingAction = theEventManager->GetUserTrackingAction();
108 theUserSteppingAction = theEventManager->GetUserSteppingAction();
109
110 theEventManager->SetUserAction(theMatScannerEventAction);
111 theEventManager->SetUserAction(theMatScannerStackingAction);
112 theEventManager->SetUserAction(theMatScannerTrackingAction);
113 theEventManager->SetUserAction(theMatScannerSteppingAction);
114
115 G4SDManager* theSDMan = G4SDManager::GetSDMpointerIfExist();
116 if(theSDMan)
117 { theSDMan->Activate("/",false); }
118
119 G4GeometryManager* theGeomMan = G4GeometryManager::GetInstance();
120 theGeomMan->OpenGeometry();
121 theGeomMan->CloseGeometry(true);
122}
123
124void G4MaterialScanner::RestoreUserActions()
125{
126 theEventManager->SetUserAction(theUserEventAction);
127 theEventManager->SetUserAction(theUserStackingAction);
128 theEventManager->SetUserAction(theUserTrackingAction);
129 theEventManager->SetUserAction(theUserSteppingAction);
130
131 G4SDManager* theSDMan = G4SDManager::GetSDMpointerIfExist();
132 if(theSDMan)
133 { theSDMan->Activate("/",true); }
134}

136void G4MaterialScanner::DoScan()
137{
138// Confirm material table is updated
139 G4RunManagerKernel::GetRunManagerKernel()->UpdateRegion();
140
141///// // Make sure Geantino has been initialized
142///// G4ProcessVector* pVector
143///// = G4Geantino::GeantinoDefinition()->GetProcessManager()->GetProcessList();
144///// for (G4int j=0; j < pVector->size(); ++j) {
145///// (*pVector)[j]->BuildPhysicsTable(*(G4Geantino::GeantinoDefinition()));
146///// }
147
148// Close geometry and set the application state
149 G4GeometryManager* geomManager = G4GeometryManager::GetInstance();
150 geomManager->OpenGeometry();
151 geomManager->CloseGeometry(1,0);
152
153 G4ThreeVector center(0,0,0);
154 G4Navigator* navigator =
155 G4TransportationManager::GetTransportationManager()->GetNavigatorForTracking();
156 navigator->LocateGlobalPointAndSetup(center,0,false);
157
158 G4StateManager* theStateMan = G4StateManager::GetStateManager();
159 theStateMan->SetNewState(G4State_GeomClosed);
160
161// Event loop
162 G4int iEvent = 0;
163 for(G4int iTheta=0;iTheta<nTheta;iTheta++)
164 {
165 G4double theta = thetaMin;
166 if(iTheta>0) theta += G4double(iTheta)*thetaSpan/G4double(nTheta-1);
167 G4double aveLength = 0.;
168 G4double aveX0 = 0.;
169 G4double aveLambda = 0.;
170 G4cout << G4endl;
171 G4cout << " Theta(deg) Phi(deg) Length(mm) x0 lambda0" << G4endl;
172 G4cout << G4endl;
173 for(G4int iPhi=0;iPhi<nPhi;iPhi++)
174 {
175 G4Event* anEvent = new G4Event(iEvent++);
176 G4double phi = phiMin;
177 if(iPhi>0) phi += G4double(iPhi)*phiSpan/G4double(nPhi-1);
178 eyeDirection = G4ThreeVector(std::cos(theta)*std::cos(phi),
179 std::cos(theta)*std::sin(phi),
180 std::sin(theta));
181 theRayShooter->Shoot(anEvent,eyePosition,eyeDirection);
182 theMatScannerSteppingAction->Initialize(regionSensitive,theRegion);
183 theEventManager->ProcessOneEvent(anEvent);
184 G4double length = theMatScannerSteppingAction->GetTotalStepLength();
185 G4double x0 = theMatScannerSteppingAction->GetX0();
186 G4double lambda = theMatScannerSteppingAction->GetLambda0();
187
188 G4cout << " "
189 << std::setw(11) << theta/deg << " "
190 << std::setw(11) << phi/deg << " "
191 << std::setw(11) << length/mm << " "
192 << std::setw(11) << x0 << " "
193 << std::setw(11) << lambda << G4endl;
194 aveLength += length/mm;
195 aveX0 += x0;
196 aveLambda += lambda;
197 }
198 if(nPhi>1)
199 {
200 G4cout << G4endl;
201 G4cout << " ave. for theta = " << std::setw(11) << theta/deg << " : "
202 << std::setw(11) << aveLength/nPhi << " "
203 << std::setw(11) << aveX0/nPhi << " "
204 << std::setw(11) << aveLambda/nPhi << G4endl;
205 }
206 }
207
208 theStateMan->SetNewState(G4State_Idle);
209 return;
210}

212G4bool G4MaterialScanner::SetRegionName(const G4String& val)
213{
214 G4Region* aRegion = G4RegionStore::GetInstance()->GetRegion(val);
215 if(aRegion)
216 {
217 theRegion = aRegion;
218 regionName = val;
219 return true;
220 }
221 else
222 {
223 G4cerr << "Region <" << val << "> not found. Command ignored." << G4endl;
224 G4cerr << "Defined regions are : " << G4endl;
225 for(size_t i=0;i<G4RegionStore::GetInstance()->size();i++)
226 { G4cerr << " " << (*(G4RegionStore::GetInstance()))[i]->GetName(); }
227 G4cerr << G4endl;
228 return false;
229 }
