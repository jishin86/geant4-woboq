29#include "G4Run.hh"
30#include "G4Event.hh"
31#include "G4RunManager.hh"
32#include "G4StatAnalysis.hh"
33
34G4Run::G4Run()
35:runID(0),numberOfEvent(0),numberOfEventToBeProcessed(0),HCtable(0),DCtable(0)
36{
37 eventVector = new std::vector<const G4Event*>;
38 // this is for FOM in G4StatAnalysis
39 G4StatAnalysis::ResetCpuClock();
40}
41
42G4Run::~G4Run()
43{
44 // Objects made by local thread should not be deleted by the master thread
45 G4RunManager::RMType rmType = G4RunManager::GetRunManager()->GetRunManagerType();
46 if(rmType != G4RunManager::masterRM)
47 {
48 std::vector<const G4Event*>::iterator itr = eventVector->begin();
49 for(;itr!=eventVector->end();itr++)
50 { delete *itr; }
51 }
52 delete eventVector;
53}
54
55void G4Run::RecordEvent(const G4Event*)
56{ numberOfEvent++; }
57
58void G4Run::Merge(const G4Run* right)
59{
60 numberOfEvent += right->numberOfEvent;
61 std::vector<const G4Event*>::iterator itr = right->eventVector->begin();
62 for(;itr!=right->eventVector->end();itr++)
63 { eventVector->push_back(*itr); }
64}
65
66void G4Run::StoreEvent(G4Event* evt)
67{ eventVector->push_back(evt); }
