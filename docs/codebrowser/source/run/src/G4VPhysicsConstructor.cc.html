80G4ParticleTable::G4PTblDicIterator* G4VPhysicsConstructor::GetParticleIterator() const
81{
82 return (subInstanceManager.offset[g4vpcInstanceID])._aParticleIterator;
83}
84
85G4VPhysicsConstructor::PhysicsBuilder_V G4VPhysicsConstructor::GetBuilders() const
86{
87 const auto& tls = *((subInstanceManager.offset[g4vpcInstanceID])._builders);
88 PhysicsBuilder_V copy(tls.size());
89 int i = 0;
90 for ( const auto& el : tls ) { copy[i++] = el; }
91 return copy;
92}
93
94void G4VPhysicsConstructor::AddBuilder(G4PhysicsBuilderInterface* bld)
95{
96 (subInstanceManager.offset[g4vpcInstanceID])._builders->push_back(bld);
97}
98
99void G4VPhysicsConstructor::TerminateWorker()
100{
101 if ( subInstanceManager.offset[g4vpcInstanceID]._builders != nullptr ) {
102 std::for_each( subInstanceManager.offset[g4vpcInstanceID]._builders->begin() ,
103 subInstanceManager.offset[g4vpcInstanceID]._builders->end() ,
104 [](PhysicsBuilder_V::value_type bld) { delete bld;});
105 subInstanceManager.offset[g4vpcInstanceID]._builders->clear();
106 }
