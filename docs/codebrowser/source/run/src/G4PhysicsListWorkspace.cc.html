36G4PhysicsListWorkspace::G4PhysicsListWorkspace(G4bool verbose)
37 : fVerbose(verbose)
38{
39
40 fpVUPLSIM =
41 &const_cast<G4VUPLManager&>(G4VUserPhysicsList::GetSubInstanceManager());
42 fpVPCSIM =
43 &const_cast<G4VPCManager&>(G4VPhysicsConstructor::GetSubInstanceManager());
44 fpVMPLSIM =
45 &const_cast<G4VMPLManager&>(G4VModularPhysicsList::GetSubInstanceManager());
46
47 // Copy information from master into PolyCone/Gon Sides in this thread.
48 InitialiseWorkspace();
49
50 // Capture its address of ParticleDefinition split-class in this thread
51 fpVUPLOffset = fpVUPLSIM->GetOffset();
52 fpVPCOffset = fpVPCSIM->GetOffset();
53 fpVMPLOffset = fpVMPLSIM->GetOffset();
54}

56G4PhysicsListWorkspace::~G4PhysicsListWorkspace()
57{
58}

60void
61G4PhysicsListWorkspace::UseWorkspace()
62{
63 if( fVerbose )
64 G4cout << "G4PhysicsListWorkspace::UseWorkspace: "
65 << "Copying particles-definition Split-Class - Start " << G4endl;
66
67 // Implementation copied from
68 // G4WorkerThread::BuildGeometryAndPhysicsVector()
69
70 // Physics List related, split classes mechanism:
71 // instantiate sub-instance for this thread
72 fpVUPLSIM->UseWorkArea(fpVUPLOffset);
73 fpVPCSIM->UseWorkArea(fpVPCOffset);
74 fpVMPLSIM->UseWorkArea(fpVMPLOffset);
75}
76
77void G4PhysicsListWorkspace::ReleaseWorkspace()
78{
79 fpVUPLSIM->UseWorkArea(0);
80 fpVPCSIM->UseWorkArea(0);
81 fpVMPLSIM->UseWorkArea(0);
82}

86}

88void
89G4PhysicsListWorkspace::InitialiseWorkspace()
90{
91 if( fVerbose )
92 G4cout << "G4PhysicsListWorkspace::InitialiseWorkspace: "
93 << "Copying particles-definition Split-Class - Start " << G4endl;
94
95 // PhysicsList related, split classes mechanism:
96 // Do *NOT* instantiate sub-instance for this thread,
97 // just copy the contents !!
98 fpVUPLSIM->NewSubInstances();
99 fpVPCSIM->NewSubInstances();
100 // The following line is fundamental! If we call NewSubInstances it will not work
101 // See: https://jira-geant4.kek.jp/browse/DEV-284
102 fpVMPLSIM->WorkerCopySubInstanceArray();
103
104 // Additional initialization if needed - beyond copying memory
105 InitialisePhysicsList();
106
107 if( fVerbose )
108 G4cout << "G4PhysicsListWorkspace::CreateAndUseWorkspace: "
109 << "Copying particles-definition Split-Class - Done!" << G4endl;
110}
111
112void G4PhysicsListWorkspace::DestroyWorkspace()
113{
114 fpVUPLSIM->FreeWorker();
115 fpVPCSIM->FreeWorker();
116 fpVMPLSIM->FreeWorker();
117}
