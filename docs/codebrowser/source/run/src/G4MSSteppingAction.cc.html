40G4MSSteppingAction::G4MSSteppingAction()
41{
42 Initialize(false,0);
43}
44
45G4MSSteppingAction::~G4MSSteppingAction()
46{;}
47
48void G4MSSteppingAction::Initialize(G4bool rSens,G4Region* reg)
49{
50 regionSensitive = rSens;
51 theRegion = reg;
52 length = 0.;
53 x0 = 0.;
54 lambda = 0.;
55}
56
57void G4MSSteppingAction::UserSteppingAction(const G4Step* aStep)
58{
59 G4StepPoint* preStepPoint = aStep->GetPreStepPoint();
60 G4Region* region = preStepPoint->GetPhysicalVolume()->GetLogicalVolume()->GetRegion();
61
62 if(regionSensitive && (region!=theRegion)) return;
63
64 G4double stlen = aStep->GetStepLength();
65 G4Material* material = preStepPoint->GetMaterial();
66 length += stlen;
67 x0 += stlen/(material->GetRadlen());
68 lambda += stlen/(material->GetNuclearInterLength());
69}
