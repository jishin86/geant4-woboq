29#ifndef G4VPersistencyManager_h
30#define G4VPersistencyManager_h 1
31
32#include "globals.hh"
33
34class G4Event;
35class G4Run;
36class G4VPhysicalVolume;
37
38// class description:
39//
40// This is an abstract base class for persistency management. The user's
41// concrete class derived from this class must be a singleton. The user
42// must construct the object of his/her concrete persistency manager at
43// his/her main().
44// The virtual methods of Store() and Retreive() will be invoked from
45// G4RunManager if the persistency manager exists.
46// Even if the user does not use any ODBMS, the user can use this class
47// especially for Store() methods. Writing an ASCII file for storing
48// event information can be delegated to this class, for example.
49//
50
51class G4VPersistencyManager
52{
53 public: // with description
54 static G4VPersistencyManager* GetPersistencyManager();
55 // Static method to return the pointer to the singleton object.
56 // Note that this method does NOT create the singleton object.
57
58 protected:
59 G4VPersistencyManager();
60
61 public:
62 virtual ~G4VPersistencyManager();
63
64 private:
65 static G4ThreadLocal G4VPersistencyManager * fPersistencyManager;
66
67 public: // with description
68 virtual G4bool Store(const G4Event* anEvent)=0;
69 virtual G4bool Store(const G4Run* aRun)=0;
70 virtual G4bool Store(const G4VPhysicalVolume* theWorld)=0;
71 // Stores G4Event, G4Run, and geometry tree characterized by the world volume.
72
73 virtual G4bool Retrieve(G4Event*& anEvent)=0;
74 virtual G4bool Retrieve(G4Run*& aRun)=0;
75 virtual G4bool Retrieve(G4VPhysicalVolume*& theWorld)=0;
76 // Restore G4Event, G4Run, and geometry tree characterized by the world volume.
77
78};
