173G4double G4StatAnalysis::GetFOM() const
174{
175 G4double elapsed_time = this->GetCpuTime();
176 G4double relative_err = this->GetRelativeError();
177 // lambda for equation clarity (will be inlined)
178 auto compute_figure_of_merit = [&] ()
179 {
180 return ( 1.0 / ( relative_err * relative_err ) / elapsed_time );
181 };
182 return (std::fabs(relative_err) > 0.0 && elapsed_time > 0.0)
183 ? compute_figure_of_merit() : ((fHits > 0) ? 1.0 : 0.0);
184}
185
186//----------------------------------------------------------------------------//
187
188G4double G4StatAnalysis::GetRelativeError() const
189{
190 // lambda for equation clarity (will be inlined)
191 auto compute_relative_error = [&] ()
192 {
193 return ( GetStdDev() / GetMean() / std::sqrt((G4double) fHits) );
194 };
195 return (std::fabs(GetMean()) > 0 && fHits > 0)
196 ? compute_relative_error() : ((fHits > 0) ? 1.0 : 0.0);
197}
198
199//----------------------------------------------------------------------------//
200
201G4double G4StatAnalysis::GetStdDev() const
202{
203 return ::sqrt(std::fabs(GetVariance()));
204}
205
206//----------------------------------------------------------------------------//
207
208G4double G4StatAnalysis::GetVariance() const
209{
210 // lambda for equation clarity (will be inlined)
211 auto compute_variance = [&] ()
212 {
213 return ((fSum2 - (std::pow(fSum1, 2.0)/fHits))/(((G4double) fHits) - 1.0));
214 };
215 return (fHits > 1) ? compute_variance() : 0.0;
216}
217
218//----------------------------------------------------------------------------//
219
220G4double G4StatAnalysis::GetCoeffVariation() const
221{
222 // lambda for equation clarity (will be inlined)
223 auto coefficient_of_variation = [&] ()
224 {
225 G4double hits = fHits;
226 return ::sqrt(
227 (hits / (hits-1.0)) *
228 ( (fSum2/(fSum1*fSum1)) - (1.0/hits))
229 );
230 };
231 return (fHits > 1) ? coefficient_of_variation() : 0.0;
232 //return (fHits > 0 && fabs(fSum1) > 0.0)
233 // ? (100.0*GetStdDev()/GetMean()) : 0.0;
234
235}
236
237//----------------------------------------------------------------------------//
238
239G4double G4StatAnalysis::GetEfficiency() const
240{
241 G4double hits = fHits;
242 G4double nzero = fHits - fZero;
243 return (fHits > 0) ? (nzero/hits) : 0.0;
244}
245
246//----------------------------------------------------------------------------//
247
248G4double G4StatAnalysis::GetR2Int() const
249{
250 G4double hits = fHits;
251 return (fHits > 0)
252 ? (fSum2 / (fSum1 * fSum1)) - 1.0/(GetEfficiency() * hits)
253 : 0.0;
254}
255
256//----------------------------------------------------------------------------//
257
258G4double G4StatAnalysis::GetR2Eff() const
259{
260 G4double hits = fHits;
261 return (fHits > 0)
262 ? (1.0 - GetEfficiency()) / (GetEfficiency() * hits)
263 : 0.0;
264}

283//----------------------------------------------------------------------------//
284
285void G4StatAnalysis::Add(const G4double& val, const G4double& weight)
286{
287 fHits += 1;
288 fSum1 += val * weight;
289 fSum2 += val * val * weight;
290 if(std::fabs(val*weight) < std::fabs(GetMean() * std::numeric_limits<double>::epsilon()))
291 fZero += 1;
292}
293
294//----------------------------------------------------------------------------//
295
296void G4StatAnalysis::Rescale(const G4double& factor)
297{
298 fSum1 *= factor;
299 fSum2 *= factor * factor;
300}

302//----------------------------------------------------------------------------//
303
304void G4StatAnalysis::PrintInfo(std::ostream& os, const std::string& tab) const
305{
306 G4int _hits = this->GetHits();
307 G4double _sum = this->GetSum();
308 G4double _sigma = this->GetStdDev();
309 G4double _coeff = this->GetCoeffVariation();
310 G4double _error = this->GetRelativeError();
311 G4double _eff = this->GetEfficiency();
312 G4double _fom = this->GetFOM();
313 G4double _r2int = this->GetR2Int();
314 G4double _r2eff = this->GetR2Eff();
315
316 using std::setprecision;
317 using std::setw;
318 using std::scientific;
319 using std::fixed;
320 using std::left;
321 using std::right;
322 using std::ios;
323
324 std::stringstream ss;
325 ss << tab //<< scientific
326 << setprecision(os.precision()) << right << _sum
327 << left << " [sigma: " << right << _sigma
328 << left << " | error: " << right << _error
329 << left << " | coeff: " << right << _coeff
330 << left << " | eff: " << right << _eff
331 << left << " | fom: " << right << _fom
332 << left << " | r2int: " << right << _r2int
333 << left << " | r2eff: " << right << _r2eff
334 << left << " | hits: " << right << _hits
335 << left << " ]";
336
337 os << ss.str();
338}

342G4double G4StatAnalysis::GetCpuTime() const
343{
344 tms* startTime = GetCpuClock();
345 tms endTime;
346 times(&endTime);
347 return ((endTime.tms_stime - startTime->tms_stime) +
348 (endTime.tms_utime - startTime->tms_utime)) / sysconf(_SC_CLK_TCK);
349}

363G4StatAnalysis::operator/=(const G4double& _val)
364{

366 fSum2 /= (_val*_val);
367 return *this;
368}
