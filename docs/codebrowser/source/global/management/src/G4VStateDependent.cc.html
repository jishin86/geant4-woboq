36#include "G4VStateDependent.hh"
37#include "G4StateManager.hh"
38
39G4VStateDependent::G4VStateDependent(G4bool bottom)
40{
41 G4StateManager * stateManager = G4StateManager::GetStateManager();
42 stateManager->RegisterDependent(this,bottom);
43}
44
45G4VStateDependent::~G4VStateDependent()
46{
47 G4StateManager * stateManager = G4StateManager::GetStateManager();
48 stateManager->DeregisterDependent(this);
49}
50
51G4VStateDependent::G4VStateDependent(const G4VStateDependent &right)
52{
53 *this = right;
54}
55
56G4VStateDependent& G4VStateDependent::operator=(const G4VStateDependent &right)
57{
58 if (&right == this) { return *this; }
59 *this = right;
60 return *this;
61}
62
63G4bool G4VStateDependent::operator==(const G4VStateDependent &right) const
64{
65 return (this == &right);
66}
67
68G4bool G4VStateDependent::operator!=(const G4VStateDependent &right) const
69{
70 return (this != &right);
71}
