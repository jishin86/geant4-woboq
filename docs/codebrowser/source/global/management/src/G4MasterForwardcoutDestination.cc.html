34#include "G4MasterForwardcoutDestination.hh"
35#include "G4AutoLock.hh"
36
37namespace
38{
39 G4Mutex out_mutex = G4MUTEX_INITIALIZER;
40}
41
42G4MasterForwardcoutDestination::~G4MasterForwardcoutDestination()
43{
44}
45
46G4int G4MasterForwardcoutDestination::ReceiveG4cout(const G4String& msg )
47{
48 // If a master destination is set check that we are not in a recursive
49 // situation, send the message to the master, using a lock to serialize calls
50 // Master is probably a (G)UI that is not thread-safe
51
52 if ( masterG4coutDestination && this!=masterG4coutDestination)
53 {
54 G4AutoLock l(&out_mutex);
55 return masterG4coutDestination->ReceiveG4cout_(msg);
56 }
57 return 0;
58}
59
60G4int G4MasterForwardcoutDestination::ReceiveG4cerr(const G4String& msg )
61{
62 if ( masterG4coutDestination && this!=masterG4coutDestination)
63 {
64 G4AutoLock l(&out_mutex);
65 return masterG4coutDestination->ReceiveG4cerr_(msg);
66 }
67 return 0;
68}
