34#include "G4LockcoutDestination.hh"
35#include "G4AutoLock.hh"
36
37namespace
38{
39 G4Mutex out_mutex = G4MUTEX_INITIALIZER;
40}
41
42G4LockcoutDestination::~G4LockcoutDestination()
43{
44}
45
46G4int G4LockcoutDestination::ReceiveG4cout(const G4String& msg )
47{
48 G4AutoLock l(&out_mutex);
49 // Forward call to base class
50 return G4coutDestination::ReceiveG4cout(msg);
51}
52
53G4int G4LockcoutDestination::ReceiveG4cerr(const G4String& msg )
54{
55 G4AutoLock l(&out_mutex);
56 return G4coutDestination::ReceiveG4cerr(msg);
57}
