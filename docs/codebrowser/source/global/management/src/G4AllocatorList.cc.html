66void G4AllocatorList::Destroy(G4int nStat, G4int verboseLevel)
67{
68 std::vector<G4AllocatorBase*>::iterator itr=fList.begin();
69 G4int i=0, j=0;
70 G4double mem=0, tmem=0;
71 if(verboseLevel>0)
72 {
73 G4cout << "================== Deleting memory pools ==================="
74 << G4endl;
75 }
76 for(; itr!=fList.end();++itr)
77 {
78 mem = (*itr)->GetAllocatedSize();
79 if(i<nStat)
80 {
81 i++;
82 tmem += mem;
83 (*itr)->ResetStorage();
84 continue;
85 }
86 j++;
87 tmem += mem;
88 if(verboseLevel>1)
89 {
90 G4cout << "Pool ID '" << (*itr)->GetPoolType() << "', size : "
91 << std::setprecision(3) << mem/1048576
92 << std::setprecision(6) << " MB" << G4endl;
93 }
94 (*itr)->ResetStorage();
95 delete *itr;
96 }
97 if(verboseLevel>0)
98 {
99 G4cout << "Number of memory pools allocated: " << Size()
100 << "; of which, static: " << i << G4endl;
101 G4cout << "Dynamic pools deleted: " << j
102 << " / Total memory freed: " << std::setprecision(2)
103 << tmem/1048576 << std::setprecision(6) << " MB" << G4endl;
104 G4cout << "============================================================"
105 << G4endl;
106 }
107 fList.clear();
108}
109
110G4int G4AllocatorList::Size() const
111{
112 return fList.size();
