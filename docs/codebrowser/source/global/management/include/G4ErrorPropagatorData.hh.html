41#ifndef G4ErrorPropagatorData_HH

43
44#include "globals.hh"

46enum G4ErrorMode { G4ErrorMode_PropForwards = 1,
47 G4ErrorMode_PropBackwards,
48 G4ErrorMode_PropTest };
49
50enum G4ErrorState { G4ErrorState_PreInit = 1,
51 G4ErrorState_Init,
52 G4ErrorState_Propagating,
53 G4ErrorState_TargetCloserThanBoundary,
54 G4ErrorState_StoppedAtTarget };
55
56enum G4ErrorStage { G4ErrorStage_Inflation = 1,
57 G4ErrorStage_Deflation };

59class G4ErrorTarget;
60
61class G4ErrorPropagatorData
62{
63
64public: // with description
65
66 static G4ErrorPropagatorData* GetErrorPropagatorData();
67 // Singleton instance
68
69 // Get and Set methods
70 G4ErrorMode GetMode() const;
71 void SetMode( G4ErrorMode mode );
72
73 G4ErrorState GetState() const;
74 void SetState( G4ErrorState sta );
75
76 G4ErrorStage GetStage() const;
77 void SetStage( G4ErrorStage sta );
78
79 const G4ErrorTarget* GetTarget( G4bool mustExist = 0) const;
80 void SetTarget( const G4ErrorTarget* target );
81
82 static G4int verbose();
83 static void SetVerbose( G4int ver );
84
85
86private:
87
88 G4ErrorPropagatorData();
89 ~G4ErrorPropagatorData();
90 // constructor and destructor are private
91
92private:
93
94 static G4ThreadLocal G4ErrorPropagatorData* fpInstance;
95
96 G4ErrorMode theMode;
97
98 G4ErrorState theState;
99
100 G4ErrorStage theStage;
101
102 G4ErrorTarget* theTarget;

104 static G4ThreadLocal G4int theVerbosity;
105
106};
