33// ---------------- G4VExceptionHandler ----------------
34//
35// Authors: M.Asai - August 2002
36//
37// ------------------------------------------------------------
38//
39// Class description:
40//
41// Abstract base class which need to be notified when G4Exception occurs.
42// The concrete class object derived from this base class will be automatically
43// registered to G4StateManager and the virtual method Notify() will be invoked
44// when G4Exception occurs.
45
46// ------------------------------------------------------------
47
48#ifndef G4VExceptionHandler_h
49#define G4VExceptionHandler_h 1
50
51#include "G4Types.hh"
52#include "G4ExceptionSeverity.hh"
53
54class G4VExceptionHandler
55{
56
57public:
58
59 G4VExceptionHandler();
60 virtual ~G4VExceptionHandler();
61 G4bool operator==(const G4VExceptionHandler &right) const;
62 G4bool operator!=(const G4VExceptionHandler &right) const;
63
64public: // with description
65
66 virtual G4bool Notify(const char* originOfException,
67 const char* exceptionCode,
68 G4ExceptionSeverity severity,
69 const char* description) = 0;
70 // Pure virtual method which will be invoked by G4StateManager when
71 // G4Exception occurs.
72 // If TRUE returned, core dump will be generated, while FALSE returned,
73 // program execution continues.
74
75private:
76
77 G4VExceptionHandler(const G4VExceptionHandler &right);
78 G4VExceptionHandler& operator=(const G4VExceptionHandler &right);
79
80};
81
82#endif
