29// Class G4ReferenceCountedHandle
30//
31// Class description:
32//
33// A class to provide reference counting mechanism.
34// It is a templated class, acting as a smart pointer,
35// wrapping the type to be counted. It performs the reference counting
36// during the life-time of the counted object. When its count reaches zero
37// the counted object is destroyed by explicit call to its destructor.
38// This class provides overloaded operators *() and ->() to allow similar
39// syntax as for the normal "dumb" pointers.
40// The basic rule for the use of this class is that a handle must always
41// be exchanged by reference never dinamically allocated (i.e. never
42// instantiated using 'new').
43// The validity of a smart pointer object can be verified by using the
44// operator !() or operator bool(). I.e.:
45// if( !smartPtrObj ) { ... } // Problem! We must initialize it first!
46// else { ... } // OK!
47// Trying to 'delete' a smart pointer object will generate a compilation
48// error (since we're dealing with objects, not pointers!).
49
50// Author: Radovan Chytracek, CERN (Radovan.Chytracek@cern.ch)
51// Date: November 2001
52// ----------------------------------------------------------------------
53#ifndef _G4REFERENCECOUNTEDHANDLE_H_
54#define _G4REFERENCECOUNTEDHANDLE_H_ 1

56#include "G4Types.hh"
57#include "G4Allocator.hh"

59template <class X> class G4CountedObject;
60
61template <class X>
62class G4ReferenceCountedHandle
63{
64
65public: // with description
66
67 inline G4ReferenceCountedHandle( X* rep = 0 );
68 // Constructor.
69
70 inline G4ReferenceCountedHandle( const G4ReferenceCountedHandle<X>& right );
71 // Copy constructor.
72
73 inline ~G4ReferenceCountedHandle();
74 // Destructor.
75
76 inline G4ReferenceCountedHandle<X>&
77 operator =( const G4ReferenceCountedHandle<X>& right );
78 // Assignment operator by reference.
79
80 inline G4ReferenceCountedHandle<X>& operator =( X* objPtr );
81 // Assignment operator by pointer.
82
83 inline unsigned int Count() const;
84 // Forward to Counter class.
85
86 inline X* operator ->() const;
87 // Operator -> allowing the access to counted object.
88 // The check for 0-ness is left out for performance reasons,
89 // see operator () below.
90 // May be called on initialised smart-pointer only!
91
92 inline G4bool operator !() const;
93 // Validity test operator.
94
95 inline operator bool() const;
96 // Boolean operator.
97
98 inline X* operator ()() const;
99 // Functor operator (for convenience).
100
101 // There is no provision that this class is subclassed.
102 // If it is subclassed & new data members are added then the
103 // following "new" & "delete" will fail and give errors.
104 //
105 inline void* operator new( size_t );
106 // Operator new defined for G4Allocator.
107
108 inline void operator delete( void *pObj );
109 // Operator delete defined for G4Allocator.
110
111private:
112
113 G4CountedObject<X>* fObj;
114 // The object subject to reference counting.
115};

117extern G4GLOB_DLL
118G4Allocator<G4ReferenceCountedHandle<void>>*& aRCHAllocator();

120template <class X>
121class G4CountedObject
122{
123
124 friend class G4ReferenceCountedHandle<X>;
125
126public: // with description
127
128 G4CountedObject( X* pObj = 0 );
129 // Constructor.
130
131 ~G4CountedObject();
132 // Destructor.
133
134 inline void AddRef();
135 // Increase the count.
136
137 inline void Release();
138 // Decrease the count and if zero destroy itself.
139
140 // There is no provision that this class is subclassed.
141 // If it is subclassed & new data members are added then the
142 // following "new" & "delete" will fail and give errors.
143 //
144 inline void* operator new( size_t );
145 // Operator new defined for G4Allocator.
146
147 inline void operator delete( void *pObj );
148 // operator delete defined for G4Allocator.
149
150private:
151
152 unsigned int fCount;
153 // Reference counter.
154 X* fRep;
155 // The counted object.
156};
157
158extern G4GLOB_DLL
159G4Allocator<G4CountedObject<void>>*& aCountedObjectAllocator();
160
161// --------- G4CountedObject<X> Inline function definitions ---------
162
163template <class X>
164G4CountedObject<X>::G4CountedObject( X* pObj )
165 : fCount(0), fRep( pObj )
166{
167 if( pObj != 0 ) fCount = 1;
168}
169
170template <class X>
171G4CountedObject<X>::~G4CountedObject()
172{
173 delete fRep;
174}
175
176template <class X>
177void G4CountedObject<X>::AddRef()
178{
179 ++fCount;
180}
181
182template <class X>
183void G4CountedObject<X>::Release()
184{
185 if( --fCount == 0 ) delete this;
186}
187
188template <class X>
189void* G4CountedObject<X>::operator new( size_t )
190{
191 if (!aCountedObjectAllocator())
192 aCountedObjectAllocator() = new G4Allocator<G4CountedObject<void>>;
193 return( (void *)aCountedObjectAllocator()->MallocSingle() );
194}
195
196template <class X>
197void G4CountedObject<X>::operator delete( void *pObj )
198{
199 aCountedObjectAllocator()->FreeSingle( (G4CountedObject<void>*)pObj );
200}
201
202// --------- G4ReferenceCountedHandle<X> Inline function definitions ---------
203
204template <class X>
205G4ReferenceCountedHandle<X>::
206 G4ReferenceCountedHandle( X* rep )
207 : fObj( 0 )
208{
209 if( rep != 0 )
210 fObj = new G4CountedObject<X>( rep );
211}
212
213template <class X>
214G4ReferenceCountedHandle<X>::
215 G4ReferenceCountedHandle( const G4ReferenceCountedHandle<X>& right )
216 : fObj( right.fObj )
217{
218 fObj->AddRef();
219}
220
221template <class X>
222G4ReferenceCountedHandle<X>::~G4ReferenceCountedHandle()
223{
224 if( fObj ) fObj->Release();
225}
226
227template <class X>
228G4ReferenceCountedHandle<X>& G4ReferenceCountedHandle<X>::
229 operator =( const G4ReferenceCountedHandle<X>& right )
230{
231 if( fObj != right.fObj )
232 {
233 if( fObj )
234 fObj->Release();
235 this->fObj = right.fObj;
236 fObj->AddRef();
237 }
238 return *this;
239}
240
241template <class X>
242G4ReferenceCountedHandle<X>& G4ReferenceCountedHandle<X>::
243 operator =( X* objPtr )
244{
245 if( fObj )
246 fObj->Release();
247 this->fObj = new G4CountedObject<X>( objPtr );
248 return *this;
249}
250
251template <class X>
252unsigned int G4ReferenceCountedHandle<X>::Count() const
253{
254 return( fObj ? fObj->fCount : 0 );
255}
256
257template <class X>
258X* G4ReferenceCountedHandle<X>::operator ->() const
259{
260 return( fObj ? fObj->fRep : 0 );
261}
262
263template <class X>
264G4bool G4ReferenceCountedHandle<X>::operator !() const
265{
266 return( ( !fObj ) ? true : false );
267}
268
269template <class X>
270G4ReferenceCountedHandle<X>::operator bool() const
271{
272 return( ( fObj ) ? true : false );
273}
274
275template <class X>
276X* G4ReferenceCountedHandle<X>::operator ()() const
277{
278 return( fObj ? fObj->fRep : 0 );
279}
280
281template <class X>
282void* G4ReferenceCountedHandle<X>::operator new( size_t )
283{
284 if (!aRCHAllocator())
285 aRCHAllocator() = new G4Allocator<G4ReferenceCountedHandle<void> >;
286 return( (void *)aRCHAllocator()->MallocSingle() );
287}
288
289template <class X>
290void G4ReferenceCountedHandle<X>::operator delete( void *pObj )
291{
292 aRCHAllocator()->FreeSingle( (G4ReferenceCountedHandle<void>*)pObj );
293}

295#endif // _G4REFERENCECOUNTEDHANDLE_H_
