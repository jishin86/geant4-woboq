33inline
34void G4SliceTimer::Start()
35{
36 fValidTimes=false;
37 fStartRealTime=times(&fStartTimes);
38}
39
40inline
41void G4SliceTimer::Stop()
42{
43 fEndRealTime=times(&fEndTimes);
44 fRealElapsed += fEndRealTime-fStartRealTime;
45 fSystemElapsed += fEndTimes.tms_stime-fStartTimes.tms_stime;
46 fUserElapsed += fEndTimes.tms_utime-fStartTimes.tms_utime;
47 fValidTimes=true;
48}
49
50inline
51void G4SliceTimer::Clear()
52{
53 fRealElapsed = 0.;
54 fSystemElapsed = 0.;
55 fUserElapsed = 0.;
56}
57
58inline
59G4bool G4SliceTimer::IsValid() const
60{
61 return fValidTimes;
62}
