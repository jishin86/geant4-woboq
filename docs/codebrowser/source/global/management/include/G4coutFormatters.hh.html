64 // A function that set ups a style for the destination
65 // Example for a style that set to all capital the messages
66 // to G4cerr:
67 // setupStyle_f myStyle = [](G4coutDestination* dest)->G4int {
68 // dest->SetCerrTransformer(
69 // [](G4String& msg){
70 // msg->toUpper();
71 // return true; }
72 // );
73 // };
74 using SetupStyle_f = std::function<G4int(G4coutDestination*)>;
75
76 using String_V=std::vector<G4String>;

78 // Return list of formatter names

81 // Setup style (by name) to destination

83
84 // Set name of the style for the master thread
85 void SetMasterStyle(const G4String& );

91
92 // To be used by user to register by name a new formatter.
