33// Class Description:
34//
35// Utility class providing similar behaviour of vector<G4double>.
36// It includes additional methods for compatibility with Rogue Wave
37// collection.
38//
39
40#ifndef G4DataVector_h
41#define G4DataVector_h 1
42
43#include "globals.hh"
44#include <vector>
45#include "G4ios.hh"
46#include <iostream>
47#include <fstream>
48
49class G4DataVector : public std::vector<G4double>
50{
51
52 public: // with description
53
54 G4DataVector();
55 // Default constructor.
56
57 explicit G4DataVector(size_t cap);
58 // Constructor given a 'capacity' defining the initial number of elements.
59
60 G4DataVector(size_t cap, G4double value);
61 // Constructor given a 'capacity' defining the initial number of elements
62 // and initialising them to 'value'.
63
64 virtual ~G4DataVector();
65 // Empty destructor
66
67 inline void insertAt(size_t, const G4double&);
68 // Insert an element at given position
69
70 inline size_t index(const G4double&);
71 // Returns back index of the element same as given value
72
73 inline G4bool contains(const G4double&) const;
74 // Returns 'true' if it contains the element same as given value
75
76 inline G4bool remove(const G4double&);
77 // Removes the first element same as given value
78
79 inline size_t removeAll(const G4double&);
80 // Remove all elements same as given value
81
82 enum {T_G4DataVector = 100};
83
84 G4bool Store(std::ofstream& fOut, G4bool ascii=false);
85 G4bool Retrieve(std::ifstream& fIn, G4bool ascii=false);
86 // To store/retrieve persistent data to/from file streams.
87
88 friend std::ostream& operator<<(std::ostream&, const G4DataVector&);
89};
