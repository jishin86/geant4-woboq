140inline G4int G4String::strcasecompare(const char* s1, const char* s2) const
141{
142 char* buf1 = new char[strlen(s1)+1];
143 char* buf2 = new char[strlen(s2)+1];
144
145 for (str_size i=0; i<=strlen(s1); i++)
146 { buf1[i] = tolower(char(s1[i])); }
147 for (str_size j=0; j<=strlen(s2); j++)
148 { buf2[j] = tolower(char(s2[j])); }
149
150 G4int res = strcmp(buf1, buf2);
151 delete [] buf1;
152 delete [] buf2;
153 return res;
154}

179inline std::istream&
180G4String::readLine (std::istream& strm, G4bool skipWhite)
181{
182 char tmp[1024];
183 if ( skipWhite )
184 {
185 strm >> std::ws;
186 strm.getline(tmp,1024);
187 *this=tmp;
188 }
189 else
190 {
191 strm.getline(tmp,1024);
192 *this=tmp;
193 }
194 return strm;
195}

242inline G4String G4String::strip (G4int strip_Type, char ch)
243{
244 G4String retVal = *this;
245 if(length()==0) { return retVal; }
246 str_size i=0;
247 switch ( strip_Type ) {
248 case leading:
249 {
250 for(i=0;i<length();i++)
251 { if (std_string::operator[](i) != ch) { break; } }
252 retVal = substr(i,length()-i);
253 }
254 break;
255 case trailing:
256 {
257 G4int j=0;
258 for(j=length()-1;j>=0;j--)
259 { if (std_string::operator[](j) != ch) { break; } }
260 retVal = substr(0,j+1);
261 }
262 break;
263 case both:
264 {
265 for(i=0;i<length();i++)
266 { if (std_string::operator[](i) != ch) { break; } }
267 G4String tmp(substr(i,length()-i));
268 G4int k=0;
269 for(k=tmp.length()-1;k>=0;k--)
270 { if (tmp.std_string::operator[](k) != ch) { break; } }
271 retVal = tmp.substr(0,k+1);
272 }
273 break;
274 default:
275 break;
276 }
277 return retVal;
278}

280inline void G4String::toLower ()
281{
282 for (str_size i=0; i<size();i++)
283 {
284 //GB:HP-UX-aCC,Linux-KCC
285 std_string::operator[](i) = tolower(char(std_string::operator[](i)));
286 //at(i) = tolower(at(i));
287 }
288}

290inline void G4String::toUpper ()
291{
292 for (str_size i=0; i<size();i++)
293 {
294 //GB:HP-UX-aCC,Linux-KCC
295 std_string::operator[](i) = toupper(char(std_string::operator[](i)));
296 //at(i) = toupper(at(i));
297 }
298}

328inline unsigned int G4String::hash( caseCompare ) const
329{
330 const char* str=c_str();
331 unsigned long h = 0;
332 for ( ; *str; ++str)
333 { h = 5*h + *str; }
334
335 return str_size(h);
