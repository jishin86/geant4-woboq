39#ifndef G4UNIFORMRANDPOOL_HH
40#define G4UNIFORMRANDPOOL_HH
41
42#include <algorithm>
43#include <assert.h>
44
45#include "G4Types.hh"
46#include "Randomize.hh"
47
48#define G4UNIFORMRANDPOOL_DEFAULT_POOLSIZE 1024
49#define G4UNIFORMRANDPOOL_TINY_POOLSIZE 128
50#define G4UNIFORMRANDPOOL_SMALL_POOLSIZE 256
51#define G4UNIFORMRANDPOOL_MEDIUM_POOLSIZE 512
52#define G4UNIFORMRANDPOOL_LARGE_POOLSIZE 2048
53#define G4UNIFORMRANDPOOL_HUGE_POOLSIZE 8192
54
55class G4UniformRandPool
56{
57 public:
58
59 G4UniformRandPool();
60 G4UniformRandPool( /*PoolSize_t&*/ G4int ps );
61 ~G4UniformRandPool();
62
63 void Resize( /*PoolSize_t*/ G4int newSize );
64 void GetMany( G4double* rnds , G4int howMany );
65 inline G4double GetOne();
66 inline /*PoolSize_t*/ G4int GetPoolSize() const;
67
68 // These two static methods are used to
69 // simulate the calls of CLHEP::HepRandom
70 //
71 static G4double flat();
72 static void flatArray( G4int howmany, G4double* rnds );
73
74 private:
75
76 void Fill( G4int howmany );
77
78 private:
79
80 /*PoolSize_t*/ G4int size;
81 G4double* buffer;
82 G4int currentIdx;
83};
84
85inline G4double G4UniformRandPool::GetOne()
86{
87 // No more available numbers, re-fill
88 //
89 if ( currentIdx >= /*(unsigned int)*/size )
90 {
91 Fill(/*(unsigned int)*/size);
92 }
93
94 return buffer[currentIdx++];
95}
96
97inline /*PoolSize_t*/ G4int G4UniformRandPool::GetPoolSize() const
98{
99 return size;
100}

102#endif
