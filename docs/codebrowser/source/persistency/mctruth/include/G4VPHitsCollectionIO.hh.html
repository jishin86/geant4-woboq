31#ifndef V_P_HITS_COLLECTION_I_O_HH
32#define V_P_HITS_COLLECTION_I_O_HH 1
33
34#include "G4VHitsCollection.hh"
35
36// Class Description:
37// Abstract base class for storing and retrieving hit collections
38
39class G4VPHitsCollectionIO
40{
41 public: // With description
42 G4VPHitsCollectionIO( std::string detName, std::string colName );
43 // Constructor
44
45 virtual ~G4VPHitsCollectionIO() {};
46 // Destructor
47
48 public: // With description
49 virtual G4bool Store(const G4VHitsCollection*) =0;
50 // Pure virtual method for storing the hit collection.
51 // Each persistency package should implement a concrete method
52 // with this signature.
53
54 virtual G4bool Retrieve(G4VHitsCollection*&) =0;
55 // Pure virtual method for retrieving the hit collection.
56 // Each persistency package should implement a concrete method
57 // with this signature.
58
59 G4bool operator== (const G4VPHitsCollectionIO& right) const;
60 // virtual operator for comparing hit collections with names.
61
62 std::string SDname() { return f_detName; };
63 // Returns the sensitive detector name.
64
65 std::string CollectionName() { return f_colName; };
66 // Returns the hit collection name.
67
68 void SetVerboseLevel(int v) { m_verbose = v; };
69 // Sets the verbose level
70
71 protected:
72 G4int m_verbose;
73 std::string f_detName;
74 std::string f_colName;
75
76}; // End of class G4VPHitsCollectionIO
77
78#endif
