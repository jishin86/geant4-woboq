33
34#include "G4Types.hh"
35#include <string>
36
37// Class Description:
38// Abstract base class for starting and commiting or aborting data transaction
39
40class G4VTransactionManager
41{
42 public: // With description
43 G4VTransactionManager() : m_verbose(0) {};
44 // Constructor
45
46 virtual ~G4VTransactionManager() {};
47 // Destructor
48
49 public: // With description
50 void SetVerboseLevel(int v) { m_verbose = v; };
51 // Set verbose level.
52
53 virtual G4bool SelectReadFile(std::string obj, std::string file)=0;
54 // Set the input file name and open it for the object type "obj".
55
56 virtual G4bool SelectWriteFile(std::string obj, std::string file)=0;
57 // Set the output file name and open it for the object type "obj".
58
59 virtual G4bool StartUpdate()=0;
60 // Start an update transaction for event store and retrieve
61
62 virtual G4bool StartRead()=0;
63 // Start a read-only transaction for event store and retrieve
64
65 virtual void Commit()=0;
66 // commit the transaction for event store and retrieve
67
68 virtual void Abort()=0;
69 // abort the transaction for event store and retrieve
70
71 protected:
72 G4int m_verbose;
73
74}; // End of class G4VTransactionManager
75
76#endif
