29// class G4tgbRotationMatrixMgr
30//
31// Class description:
32//
33// Singleton class to manage the building of transient rotation matrix,
34// as well as the construction of the corresponding G4RotationMatrix's.
35
36// History:
37// - Created. P.Arce, CIEMAT (November 2007)
38// -------------------------------------------------------------------------
39
40#ifndef G4tgbRotationMatrixMgr_h
41#define G4tgbRotationMatrixMgr_h
42
43#include "globals.hh"
44
45#include <iostream>
46#include <map>
47
48#include "G4tgbRotationMatrix.hh"
49
50typedef std::map< G4String, G4tgbRotationMatrix*,
51 std::less<G4String> > G4mstgbrotm;
52typedef std::map< G4String, G4RotationMatrix*,
53 std::less<G4String> > G4msg4rotm;
54
55class G4tgbRotationMatrixMgr
56{
57 public: // with description
58
59 ~G4tgbRotationMatrixMgr();
60
61 static G4tgbRotationMatrixMgr* GetInstance();
62 // Get only instance (if it does not exists, create it)
63
64 G4RotationMatrix* FindOrBuildG4RotMatrix(const G4String& name);
65 // Look for a G4RotationMatrix and if not found create it
66 // from the corresponding G4tgbRotationMatrix
67 G4RotationMatrix* FindG4RotMatrix(const G4String& name);
68 // Look for a G4RotationMatrix and if not found return 0
69
70 G4tgbRotationMatrix* FindOrBuildTgbRotMatrix(const G4String& name);
71 // Look for an G4tgbRotationMatrix and if not found exit
72 G4tgbRotationMatrix* FindTgbRotMatrix(const G4String& name);
73 // Look for an G4tgbRotationMatrix and if not found return 0
74
75 public: // without description
76
77 const G4mstgbrotm GetTgbRotMatList() const { return theTgbRotMats; }
78 const G4msg4rotm& GetG4RotMatList() const { return theG4RotMats; }
79
80 private:
81
82 G4tgbRotationMatrixMgr();
83 void CopyRotMats();
84
85 static G4ThreadLocal G4tgbRotationMatrixMgr* theInstance;
86
87 private:
88
89 G4mstgbrotm theTgbRotMats;
90 G4msg4rotm theG4RotMats;
91};
