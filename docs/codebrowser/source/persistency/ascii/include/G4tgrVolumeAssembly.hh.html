39#ifndef G4tgrVolumeAssembly_h
40#define G4tgrVolumeAssembly_h
41
42#include "globals.hh"
43#include "G4tgrVolume.hh"
44#include "G4ThreeVector.hh"
45
46class G4tgrVolumeAssembly : public G4tgrVolume
47{
48 public: // with description
49
50 G4tgrVolumeAssembly();
51 G4tgrVolumeAssembly( const std::vector<G4String>& wl );
52 ~G4tgrVolumeAssembly();
53
54 virtual G4tgrPlace* AddPlace( const std::vector<G4String>& wl );
55 // Add a position with the data read from a ':place_assembly' tag
56
57 const G4String& GetComponentName(G4int ii) const
58 { return theComponentNames[ii]; }
59 const G4String& GetComponentRM(G4int ii) const
60 { return theComponentRMs[ii]; }
61 G4ThreeVector GetComponentPos(G4int ii) const
62 { return theComponentPos[ii]; }
63 G4int GetNoComponents() const
64 { return theComponentNames.size(); }
65
66 friend std::ostream& operator<<(std::ostream& os,
67 const G4tgrVolumeAssembly& obj);
68 protected:
69
70 std::vector<G4String> theComponentNames;
71 std::vector<G4String> theComponentRMs;
72 std::vector<G4ThreeVector> theComponentPos;
73};
