29// class G4tgrPlace
30//
31// Class description:
32//
33// Abstract base class to describe the position of a G4tgrVolume inside
34// another G4tgrVolume.
35
36// History:
37// - Created. P.Arce, CIEMAT (November 2007)
38// -------------------------------------------------------------------------
39
40#ifndef G4tgrPlace_h
41#define G4tgrPlace_h
42
43#include "globals.hh"
44#include "G4ThreeVector.hh"
45
46class G4tgrVolume;
47
48class G4tgrPlace
49{
50 public: // with description
51
52 G4tgrPlace();
53 virtual ~G4tgrPlace();
54
55 // Accessors
56
57 const G4String& GetParentName() const { return theParentName; }
58 G4tgrVolume* GetVolume() const { return theVolume; }
59 unsigned int GetCopyNo() const { return theCopyNo; }
60 const G4String& GetType() const { return theType; }
61 void SetVolume( G4tgrVolume* vol ) { theVolume = vol; }
62 void SetType( const G4String& typ ){ theType = typ; }
63
64 virtual G4ThreeVector GetPlacement() const;
65
66 protected:
67
68 G4tgrVolume* theVolume;
69 // The detunit to which it belongs
70
71 G4String theParentName;
72 // The parent (by name, as we will allow that a child
73 // is placed in the file before the parent is created)
74
75 unsigned int theCopyNo;
76 // The copy number
77
78 G4String theType;
79 // The type (simple/replica/param)
80};
81
82#endif
