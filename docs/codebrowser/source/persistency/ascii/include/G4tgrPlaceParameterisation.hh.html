29// class G4tgrPlaceParameterisation
30//
31// Class description:
32//
33// Class to descripe the positioning of a G4tgrVolume inside another
34// G4tgrVolume as a parameterised volume. Several types are possible:
35// - Parameterisation of the position and rotation for each copy
36// - Parameterisation also of the dimensions
37// - Parameterisation of the solid type
38// Data is just stored in this class, without any calculation of the
39// positions of each copy
40// :POS_PARAM "volu_name" copyNo "parent_name" "parametrisation_type"
41// number_copies step offset extra_data(n words).
42
43// History:
44// - Created. P.Arce, CIEMAT (November 2007)
45// -------------------------------------------------------------------------
46
47#ifndef G4tgrPlaceParameterisation_h
48#define G4tgrPlaceParameterisation_h
49

51
52#include <vector>
53
54#include "G4tgrPlace.hh"

56class G4tgrPlaceParameterisation : public G4tgrPlace
57{
58 public: // with description
59
60 G4tgrPlaceParameterisation();

63 G4tgrPlaceParameterisation( const std::vector<G4String>& p);
64 // Creates an object passing the parameters
65
66 // Access functions
67
68 const G4String& GetParamType() const { return theParamType; }
69 // GetType returns placement type

72
73 friend std::ostream& operator<<(std::ostream& os,
74 const G4tgrPlaceParameterisation& obj);
75 private:
76
77 G4String theParamType;

79 // Extra data not common to all parameterisations

82 // The rotation matrix (by name, as the rotations
83 // matrices are not yet created)
84};
