53void G4tgrEvaluator::print_error( G4int estatus ) const
54{
55 switch (estatus)
56 {
57 case ERROR_SYNTAX_ERROR:
58 G4cerr << "G4tgrEvaluator: syntax error!" << G4endl;
59 return;
60 default:
61 G4Evaluator::print_error();
62 return;
63 }
64}

66G4double fsin( G4double arg ){ return std::sin(arg); }
67G4double fcos( G4double arg ){ return std::cos(arg); }
68G4double ftan( G4double arg ){ return std::tan(arg); }
69G4double fasin( G4double arg ){ return std::asin(arg); }
70G4double facos( G4double arg ){ return std::acos(arg); }
71G4double fatan( G4double arg ){ return std::atan(arg); }
72G4double fatan2( G4double arg1, G4double arg2 ){ return std::atan2(arg1,arg2); }
73G4double fsinh( G4double arg ){ return std::sinh(arg); }
74G4double fcosh( G4double arg ){ return std::cosh(arg); }
75G4double ftanh( G4double arg ){ return std::tanh(arg); }
76// G4double fasinh( G4double arg ){ return std::asinh(arg); }
77// G4double facosh( G4double arg ){ return std::acosh(arg); }
78// G4double fatanh( G4double arg ){ return std::atanh(arg); }
79G4double fsqrt( G4double arg ){ return std::sqrt(arg); }
80G4double fexp( G4double arg ){ return std::exp(arg); }
81G4double flog( G4double arg ){ return std::log(arg); }
82G4double flog10( G4double arg ){ return std::log10(arg); }
83G4double fpow( G4double arg1, G4double arg2 ){ return std::pow(arg1,arg2); }

87void G4tgrEvaluator::AddCommonFunctions()
88{
89 setFunction("sin", (*fsin));
90 setFunction("cos", (*fcos));
91 setFunction("tan", (*ftan));
92 setFunction("asin", (*fasin));
93 setFunction("acos", (*facos));
94 setFunction("atan", (*fatan));
95 setFunction("atan2", (*fatan2));
96 setFunction("sinh", (*fsinh));
97 setFunction("cosh", (*fcosh));
98 setFunction("tanh", (*ftanh));
99// setFunction("asinh", (*fasinh));
100// setFunction("acosh", (*facosh));
101// setFunction("atanh", (*fatanh));
102 setFunction("sqrt", (*fsqrt));
103 setFunction("exp", (*fexp));
104 setFunction("log", (*flog));
105 setFunction("log10", (*flog10));
106 setFunction("pow", (*fpow));
107}
