38//----------------------------------------------------------------------
39G4tgbIsotope::G4tgbIsotope()
40 : theTgrIsot(0), theG4Isot(0)
41{
42}
43
44//----------------------------------------------------------------------
45G4tgbIsotope::~G4tgbIsotope()
46{
47}
48
49//----------------------------------------------------------------------
50G4tgbIsotope::G4tgbIsotope( G4tgrIsotope* hg )
51{
52 theTgrIsot = hg;
53 theG4Isot = 0;
54}
55
56//----------------------------------------------------------------------
57G4Isotope* G4tgbIsotope::BuildG4Isotope()
58{
59 G4Isotope* isot = 0;
60
61 //-------- if G4Isotope not found, construct it
62 if( theG4Isot == 0 )
63 {
64 //----- construct new G4Isotope
65
66 isot = new G4Isotope(theTgrIsot->GetName(), theTgrIsot->GetZ(),
67 theTgrIsot->GetN(), theTgrIsot->GetA() );
68
69#ifdef G4VERBOSE
70 if( G4tgrMessenger::GetVerboseLevel() >= 2 )
71 {
72 G4cout << " Constructing new G4Isotope "
73 << *isot << G4endl;
74 }
75#endif
76 theG4Isot = isot;
77 }
78 else
79 {
80 isot = theG4Isot;
81 }
82
83 return isot;
84}
