39G4tgrMaterial::G4tgrMaterial()
40 : theName("Material"), theDensity(0.), theNoComponents(0),
41 theMateType("Material"), theIonisationMeanExcitationEnergy(-1.),
42 theState(kStateUndefined), theTemperature(STP_Temperature),
43 thePressure(STP_Pressure)
44{
45}
46
47
48// -------------------------------------------------------------------------
49G4tgrMaterial::~G4tgrMaterial()
50{
51}
52
53
54// -------------------------------------------------------------------------
55void G4tgrMaterial::SetState( G4String val )
56{
57 if( val == "Undefined" ) {
58 theState = kStateUndefined;
59 } else if( val == "Solid" ) {
60 theState = kStateSolid;
61 } else if( val == "Liquid" ) {
62 theState = kStateLiquid;
63 } else if( val == "Gas" ) {
64 theState = kStateGas;
65 } else {
66
67 G4Exception("G4tgrMaterial::SetState", "Wrong state", FatalErrorInArgument,
68 "Only possible states are Undefined/Solid/Liquid/Gas!");
69 }
70}
