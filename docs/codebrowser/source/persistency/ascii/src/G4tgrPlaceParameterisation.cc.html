52//-------------------------------------------------------------
53G4tgrPlaceParameterisation::
54G4tgrPlaceParameterisation( const std::vector<G4String>& wl )
55{
56 theType = "PlaceParam";
57
58 //---------- Check for exact number of words read
59 G4tgrUtils::CheckWLsize( wl, 7, WLSIZE_GE,
60 "G4tgrPlaceParameterisation::ConstructVolume" );
61
62 //---------- the copy No
63 theCopyNo = G4tgrUtils::GetInt( wl[2] )-1;
64
65 //---------- set the parent name
66 theParentName = G4tgrUtils::GetString( wl[3] );
67
68 //---------- set the type
69 theParamType = G4tgrUtils::GetString( wl[4] );
70
71 //---------- set the rotation matrix name
72 theRotMatName = G4tgrUtils::GetString(wl[5]);
73
74 //---------- set the extra data
75 for( size_t ii = 6; ii < wl.size(); ii++)
76 {
77 theExtraData.push_back( G4tgrUtils::GetDouble(wl[ii]) );
78 }
79
80#ifdef G4VERBOSE
81 if( G4tgrMessenger::GetVerboseLevel() >= 1 )
82 {
83 G4cout << " Created " << *this << G4endl;
84 }
85#endif
86}
87

89// -------------------------------------------------------------------------
90std::ostream& operator<<(std::ostream& os,
91 const G4tgrPlaceParameterisation& obj)
92{
93 os << "G4tgrPlaceParameterisation= in " << obj.theParentName
94 << " ParamTyep = " << obj.theParamType
95 << " RotMatName= " << obj.theRotMatName << " EXTRA DATA= ";
96 for( size_t ii = 0; ii < obj.theExtraData.size(); ii++ )
97 {
98 os << obj.theExtraData[ii] << " ";
99 }
100 os << G4endl;
101
102 return os;
103}
