35#include "G4tgrMessenger.hh"
36#include "G4UIdirectory.hh"
37#include "G4UIcmdWithoutParameter.hh"
38#include "G4UIcmdWithAnInteger.hh"
39
40G4ThreadLocal G4int G4tgrMessenger::theVerboseLevel = 0;
41
42
43// --------------------------------------------------------------------
44G4tgrMessenger::G4tgrMessenger()
45{
46 tgDirectory = new G4UIdirectory("/geometry/textInput/");
47 tgDirectory->SetGuidance("Geometry from text file control commands.");
48 verboseCmd = new G4UIcmdWithAnInteger("/geometry/textInput/verbose",this);
49 verboseCmd->SetGuidance("Set Verbose level of geometry text input category.");
50 verboseCmd->SetGuidance(" 0 : Silent");
51 verboseCmd->SetGuidance(" 1 : info verbosity");
52 verboseCmd->SetGuidance(" 2 : debug verbosity");
53 verboseCmd->SetParameterName("level",false);
54 verboseCmd->SetRange("level>=0");
55}
56
57
58// --------------------------------------------------------------------
59G4tgrMessenger::~G4tgrMessenger()
60{
61 delete verboseCmd;
62 delete tgDirectory;
63}
64
65
66// --------------------------------------------------------------------
67G4int G4tgrMessenger::GetVerboseLevel()
68{
69 return theVerboseLevel;
70}
71
72
73// --------------------------------------------------------------------
74void G4tgrMessenger::SetVerboseLevel( G4int verb )
75{
76 theVerboseLevel = verb;
77}
78
79
80// --------------------------------------------------------------------
81void G4tgrMessenger::SetNewValue(G4UIcommand * command, G4String newValues)
82{
83 if( command == verboseCmd )
84 {
85 G4tgrMessenger::SetVerboseLevel(verboseCmd->GetNewIntValue(newValues));
86 }
87}
88
89
90// --------------------------------------------------------------------
91G4String G4tgrMessenger::GetCurrentValue(G4UIcommand * command)
92{
93 G4String cv;
94 if( command == verboseCmd )
95 {
96 cv = verboseCmd->ConvertToString(G4tgrMessenger::GetVerboseLevel());
97 }
98 return cv;
99}
