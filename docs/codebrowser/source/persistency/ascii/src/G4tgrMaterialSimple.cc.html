55//-------------------------------------------------------------
56G4tgrMaterialSimple::G4tgrMaterialSimple(const G4String& matType,
57 const std::vector<G4String>& wl)
58 : name("MaterialSimple")
59{
60 //---------- Check for miminum number of words read
61 G4tgrUtils::CheckWLsize( wl, 5, WLSIZE_EQ,
62 "G4tgrMaterialSimple::G4tgrMaterialSimple");
63
64 theMateType = matType;
65
66 //---------- Fill private data
67 theName = G4tgrUtils::GetString( wl[1] );
68 theZ = G4tgrUtils::GetDouble( wl[2] );
69 theA = G4tgrUtils::GetDouble( wl[3], g/mole);
70 theDensity = G4tgrUtils::GetDouble( wl[4], g/cm3);
71 theNoComponents = 0;
72#ifdef G4VERBOSE
73 if( G4tgrMessenger::GetVerboseLevel() >= 1 )
74 {
75 G4cout << " Created " << *this << G4endl;
76 }
77#endif
78}

82const G4String& G4tgrMaterialSimple::GetComponent(G4int ii) const
83{
84 G4String ErrMessage = "Should never be called for a MaterialSimple - i:"
85 + G4UIcommand::ConvertToString(ii);
86 G4Exception("G4tgrMaterialSimple::GetComponent()",
87 "InvalidCall", FatalException, ErrMessage);
88
89 return name; // dummy, to avoid compilation warnings...
90}
