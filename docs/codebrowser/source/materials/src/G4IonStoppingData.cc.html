78 G4int atomicNumberIon, // Atomic number of ion
79 G4int atomicNumberElem // Atomic number of elemental material
80 )

165G4bool G4IonStoppingData::AddPhysicsVector(
166 G4PhysicsVector* physicsVector, // Physics vector
167 G4int atomicNumberIon, // Atomic number of ion
168 const G4String& matIdentifier // Name of elemental material
169 )
170{
171 if(physicsVector == nullptr) {
172 G4Exception ("G4IonStoppingData::AddPhysicsVector() for material",
173 "mat037", FatalException,
174 "Pointer to vector is null-pointer.");
175 return false;
176 }
177
178 if(matIdentifier.empty()) {
179 G4Exception ("G4IonStoppingData::AddPhysicsVector() for material",
180 "mat038", FatalException, "Invalid name of the material.");
181 return false;
182 }
183
184 if(atomicNumberIon <= 0) {
185 G4Exception ("G4IonStoppingData::AddPhysicsVector() for material",
186 "mat039", FatalException, "Illegal atomic number.");
187 return false;
188 }
189
190 G4IonDEDXKeyMat mkey = std::make_pair(atomicNumberIon, matIdentifier);
191
192 if(dedxMapMaterials.count(mkey) == 1) {
193 G4ExceptionDescription ed;
194 ed << "Vector with Z1 = " << atomicNumberIon << ", mat = "
195 << matIdentifier
196 << "already exists. Remove first before replacing.";
197 G4Exception ("G4IonStoppingData::AddPhysicsVector() for material",
198 "mat040", FatalException, ed);
199 return false;
200 }
201
202 dedxMapMaterials[mkey] = physicsVector;
203
204 return true;
205}

209G4bool G4IonStoppingData::AddPhysicsVector(
210 G4PhysicsVector* physicsVector, // Physics vector
211 G4int atomicNumberIon, // Atomic number of ion
212 G4int atomicNumberElem // Atomic number of elemental material
213 )
214{
215 if(physicsVector == nullptr) {
216 G4Exception ("G4IonStoppingData::AddPhysicsVector() for element", "mat037",
217 FatalException, "Pointer to vector is null-pointer.");
218 return false;
219 }
220
221 if(atomicNumberIon <= 0) {
222 G4Exception ("G4IonStoppingData::AddPhysicsVector() for element", "mat038",
223 FatalException, "Invalid ion number.");
224 return false;
225 }
226
227 if(atomicNumberElem <= 0) {
228 G4Exception ("G4IonStoppingData::AddPhysicsVector() for element", "mat039",
229 FatalException, "Illegal atomic number.");
230 return false;
231 }
232
233 G4IonDEDXKeyElem key = std::make_pair(atomicNumberIon, atomicNumberElem);
234
235 if(dedxMapElements.count(key) == 1) {
236 G4ExceptionDescription ed;
237 ed << "Vector with Z1 = " << atomicNumberIon << ", Z= "
238 << atomicNumberElem
239 << "already exists. Remove first before replacing.";
240 G4Exception ("G4IonStoppingData::AddPhysicsVector() for element", "mat040",
241 FatalException, ed);
242 return false;
243 }
244
245 dedxMapElements[key] = physicsVector;
246
247 return true;
248}

252G4bool G4IonStoppingData::RemovePhysicsVector(
253 G4int atomicNumberIon, // Atomic number of ion
254 const G4String& matIdentifier // Name or chemical formula of material
255 ) {
256
257 G4IonDEDXKeyMat key = std::make_pair(atomicNumberIon, matIdentifier);
258
259 G4IonDEDXMapMat::iterator iter = dedxMapMaterials.find(key);
260
261 if(iter == dedxMapMaterials.end()) {
262 G4Exception ("G4IonStoppingData::RemovePhysicsVector() for material",
263 "mat038", FatalException, "Invalid name of the material.");
264 return false;
265 }
266
267 G4PhysicsVector* physicsVector = (*iter).second;
268
269 // Deleting key of physics vector from material map
270 dedxMapMaterials.erase(key);
271
272 // Deleting physics vector
273 delete physicsVector;
274
275 return true;
276}

307G4bool G4IonStoppingData::BuildPhysicsVector(
308 G4int atomicNumberIon, // Atomic number of ion
309 const G4String& matIdentifier // Name of material
310 ) {
311
312 if( IsApplicable(atomicNumberIon, matIdentifier) ) return true;
313
314 char* path = getenv("G4LEDATA");
315 if ( !path ) {
316 G4Exception("G4IonStoppingData::BuildPhysicsVector()", "mat521",
317 FatalException, "G4LEDATA environment variable not set");
318 return false;
319 }
320
321 std::ostringstream file;
322
323 file << path << "/" << subDir << "/z"
324 << atomicNumberIon << "_" << matIdentifier
325 << ".dat";
326
327 G4String fileName = G4String( file.str().c_str() );
328
329 std::ifstream ifilestream( fileName );
330
331 if ( !ifilestream.is_open() ) return false;
332
333 G4LPhysicsFreeVector* physicsVector = new G4LPhysicsFreeVector();
334
335 if( !physicsVector -> Retrieve(ifilestream, true) ) {
336
337 ifilestream.close();
338 return false;
339 }
340
341 physicsVector -> ScaleVector( MeV, MeV * cm2 /( 0.001 * g) );
342 physicsVector -> SetSpline( true );
343 physicsVector -> FillSecondDerivatives();
344
345 // Retrieved vector is added to material store
346 if( !AddPhysicsVector(physicsVector, atomicNumberIon, matIdentifier) ) {
347 delete physicsVector;
348 ifilestream.close();
349 return false;
350 }
351
352 ifilestream.close();
353 return true;
354}

408void G4IonStoppingData::ClearTable() {
409
410 G4IonDEDXMapMat::iterator iterMat = dedxMapMaterials.begin();
411 G4IonDEDXMapMat::iterator iterMat_end = dedxMapMaterials.end();
412
413 for(;iterMat != iterMat_end; iterMat++) {
414
415 G4PhysicsVector* vec = iterMat -> second;
416
417 if(vec != 0) delete vec;
418 }
419
420 dedxMapMaterials.clear();
421
422 G4IonDEDXMapElem::iterator iterElem = dedxMapElements.begin();
423 G4IonDEDXMapElem::iterator iterElem_end = dedxMapElements.end();
424
425 for(;iterElem != iterElem_end; iterElem++) {
426
427 G4PhysicsVector* vec = iterElem -> second;
428
429 if(vec != 0) delete vec;
430 }
431
432 dedxMapElements.clear();
433}

437void G4IonStoppingData::DumpMap() {
438
439 G4IonDEDXMapMat::iterator iterMat = dedxMapMaterials.begin();
440 G4IonDEDXMapMat::iterator iterMat_end = dedxMapMaterials.end();
441
442 G4cout << std::setw(15) << std::right
443 << "Atomic nmb ion"
444 << std::setw(25) << std::right
445 << "Material name"
446 << G4endl;
447
448 for(;iterMat != iterMat_end; iterMat++) {
449 G4IonDEDXKeyMat key = iterMat -> first;
450 G4PhysicsVector* physicsVector = iterMat -> second;
451
452 G4int atomicNumberIon = key.first;
453 G4String matIdentifier = key.second;
454
455 if(physicsVector != 0) {
456 G4cout << std::setw(15) << std::right
457 << atomicNumberIon
458 << std::setw(25) << std::right
459 << matIdentifier
460 << G4endl;
461 }
462 }
463
464 G4IonDEDXMapElem::iterator iterElem = dedxMapElements.begin();
465 G4IonDEDXMapElem::iterator iterElem_end = dedxMapElements.end();
466
467 G4cout << std::setw(15) << std::right
468 << "Atomic nmb ion"
469 << std::setw(25) << std::right
470 << "Atomic nmb material"
471 << G4endl;
472
473 for(;iterElem != iterElem_end; iterElem++) {
474 G4IonDEDXKeyElem key = iterElem -> first;
475 G4PhysicsVector* physicsVector = iterElem -> second;
476
477 G4int atomicNumberIon = key.first;
478 G4int atomicNumberElem = key.second;
479
480 if(physicsVector != 0) {
481 G4cout << std::setw(15) << std::right
482 << atomicNumberIon
483 << std::setw(25) << std::right
484 << atomicNumberElem
485 << G4endl;
486 }
487 }
