91// Constructor to create a material from scratch
92
93G4Material::G4Material(const G4String& name, G4double z,
94 G4double a, G4double density,
95 G4State state, G4double temp, G4double pressure)
96 : fName(name)
97{
98 InitializePointers();
99
100 if (density < universe_mean_density)
101 {
102 G4cout << " G4Material WARNING:"
103 << " define a material with density=0 is not allowed. \n"
104 << " The material " << name << " will be constructed with the"
105 << " default minimal density: " << universe_mean_density/(g/cm3)
106 << "g/cm3" << G4endl;
107 density = universe_mean_density;
108 }
109
110 fDensity = density;
111 fState = state;
112 fTemp = temp;
113 fPressure = pressure;
114
115 // Initialize theElementVector allocating one
116 // element corresponding to this material
117 maxNbComponents = fNumberOfComponents = fNumberOfElements = 1;
118 fArrayLength = maxNbComponents;
119 theElementVector = new G4ElementVector();
120
121 const std::vector<G4String> elmnames =
122 G4NistManager::Instance()->GetNistElementNames();
123 G4String enam, snam;
124 G4int iz = G4lrint(z);
125 if(iz < (G4int)elmnames.size()) {
126 snam = elmnames[iz];
127 enam = snam;
128 } else {
129 enam = "ELM_" + name;
130 snam = name;
131 }
132 theElementVector->push_back(new G4Element(enam, snam, z, a));
133
134 fMassFractionVector = new G4double[1];
135 fMassFractionVector[0] = 1. ;
136 fMassOfMolecule = a/CLHEP::Avogadro;
137
138 if (fState == kStateUndefined)
139 {
140 if (fDensity > kGasThreshold) { fState = kStateSolid; }
141 else { fState = kStateGas; }
142 }
143
144 ComputeDerivedQuantities();
145}

149// Constructor to create a material from a List of constituents
150// (elements and/or materials) added with AddElement or AddMaterial

152G4Material::G4Material(const G4String& name, G4double density,
153 G4int nComponents,
154 G4State state, G4double temp, G4double pressure)
155 : fName(name)
156{
157 InitializePointers();
158
159 if (density < universe_mean_density)
160 {
161 G4cout << "--- Warning from G4Material::G4Material()"
162 << " define a material with density=0 is not allowed. \n"
163 << " The material " << name << " will be constructed with the"
164 << " default minimal density: " << universe_mean_density/(g/cm3)
165 << "g/cm3" << G4endl;
166 density = universe_mean_density;
167 }
168
169 fDensity = density;
170 fState = state;
171 fTemp = temp;
172 fPressure = pressure;
173
174 maxNbComponents = nComponents;
175 fArrayLength = maxNbComponents;
176 fNumberOfComponents = fNumberOfElements = 0;
177 theElementVector = new G4ElementVector();
178 theElementVector->reserve(maxNbComponents);
179
180 if (fState == kStateUndefined)
181 {
182 if (fDensity > kGasThreshold) { fState = kStateSolid; }
183 else { fState = kStateGas; }
184 }
185}

257void G4Material::InitializePointers()
258{
259 theElementVector = nullptr;
260 fMassFractionVector = nullptr;
261 fAtomsVector = nullptr;
262 fMaterialPropertiesTable = nullptr;
263
264 VecNbOfAtomsPerVolume = nullptr;
265 fBaseMaterial = nullptr;
266
267 fChemicalFormula = "";
268
269 // initilized data members
270 fDensity = 0.0;
271 fState = kStateUndefined;
272 fTemp = 0.0;
273 fPressure = 0.0;
274 maxNbComponents = 0;
275 fArrayLength = 0;
276 fNumberOfComponents = 0;
277 fNumberOfElements = 0;
278 TotNbOfAtomsPerVolume = 0.0;
279 TotNbOfElectPerVolume = 0.0;
280 fRadlen = 0.0;
281 fNuclInterLen = 0.0;
282 fMassOfMolecule = 0.0;
283
284 fIonisation = nullptr;
285 fSandiaTable = nullptr;
286
287 // Store in the static Table of Materials
288 fIndexInTable = theMaterialTable.size();
289 for(size_t i=0; i<fIndexInTable; ++i) {
290 if(theMaterialTable[i]->GetName() == fName) {
291 G4cout << "G4Material WARNING: duplicate name of material "
292 << fName << G4endl;
293 break;
294 }
295 }
296 theMaterialTable.push_back(this);
297}

300
301void G4Material::ComputeDerivedQuantities()
302{
303 // Header routine to compute various properties of material.
304 //
305
306 // Number of atoms per volume (per element), total nb of electrons per volume
307 G4double Zi, Ai;
308 TotNbOfAtomsPerVolume = 0.;
309 if (VecNbOfAtomsPerVolume) { delete [] VecNbOfAtomsPerVolume; }
310 VecNbOfAtomsPerVolume = new G4double[fNumberOfElements];
311 TotNbOfElectPerVolume = 0.;
312 for (G4int i=0; i<fNumberOfElements; ++i) {
313 Zi = (*theElementVector)[i]->GetZ();
314 Ai = (*theElementVector)[i]->GetA();
315 VecNbOfAtomsPerVolume[i] = Avogadro*fDensity*fMassFractionVector[i]/Ai;
316 TotNbOfAtomsPerVolume += VecNbOfAtomsPerVolume[i];
317 TotNbOfElectPerVolume += VecNbOfAtomsPerVolume[i]*Zi;
318 }
319
320 ComputeRadiationLength();
321 ComputeNuclearInterLength();
322
323 if (fIonisation) { delete fIonisation; }
324 fIonisation = new G4IonisParamMat(this);
325 if (fSandiaTable) { delete fSandiaTable; }
326 fSandiaTable = new G4SandiaTable(this);
327}

331void G4Material::CopyPointersOfBaseMaterial()
332{
333 G4double factor = fDensity/fBaseMaterial->GetDensity();
334 TotNbOfAtomsPerVolume = factor*fBaseMaterial->GetTotNbOfAtomsPerVolume();
335 TotNbOfElectPerVolume = factor*fBaseMaterial->GetTotNbOfElectPerVolume();
336
337 if(fState == kStateUndefined) { fState = fBaseMaterial->GetState(); }
338
339 theElementVector =
340 const_cast<G4ElementVector*>(fBaseMaterial->GetElementVector());
341 fMassFractionVector =
342 const_cast<G4double*>(fBaseMaterial->GetFractionVector());
343 fAtomsVector = const_cast<G4int*>(fBaseMaterial->GetAtomsVector());
344
345 const G4double* v = fBaseMaterial->GetVecNbOfAtomsPerVolume();
346 if (VecNbOfAtomsPerVolume) { delete [] VecNbOfAtomsPerVolume; }
347 VecNbOfAtomsPerVolume = new G4double[fNumberOfElements];
348 for (G4int i=0; i<fNumberOfElements; ++i) {
349 VecNbOfAtomsPerVolume[i] = factor*v[i];
350 }
351 fRadlen = fBaseMaterial->GetRadlen()/factor;
352 fNuclInterLen = fBaseMaterial->GetNuclearInterLength()/factor;
353
354 if (fIonisation) { delete fIonisation; }
355 fIonisation = new G4IonisParamMat(this);
356
357 fSandiaTable = fBaseMaterial->GetSandiaTable();
358 fIonisation->SetMeanExcitationEnergy(fBaseMaterial->GetIonisation()->GetMeanExcitationEnergy());
359
360 fMaterialPropertiesTable = fBaseMaterial->GetMaterialPropertiesTable();
361}

365// AddElement -- composition by atom count
366
367void G4Material::AddElement(G4Element* element, G4int nAtoms)
368{
369 // initialization
370 if ( fNumberOfElements == 0 ) {
371 fAtomsVector = new G4int [fArrayLength];
372 fMassFractionVector = new G4double[fArrayLength];
373 }
374
375 // filling ...
376 if ( fNumberOfElements < maxNbComponents ) {
377 theElementVector->push_back(element);
378 fAtomsVector[fNumberOfElements] = nAtoms;
379 fNumberOfComponents = ++fNumberOfElements;
380 } else {
381 G4cout << "G4Material::AddElement ERROR for " << fName << " nElement= "
382 << fNumberOfElements << G4endl;
383 G4Exception ("G4Material::AddElement()", "mat031", FatalException,
384 "Attempt to add more than the declared number of elements.");
385 }
386 // filled.
387 if ( fNumberOfElements == maxNbComponents ) {
388 // compute proportion by mass
389 G4int i=0;
390 G4double Amol = 0.;
391 for (i=0; i<fNumberOfElements; ++i) {
392 G4double w = fAtomsVector[i]*(*theElementVector)[i]->GetA();
393 Amol += w;
394 fMassFractionVector[i] = w;
395 }
396 for (i=0; i<fNumberOfElements; ++i) {
397 fMassFractionVector[i] /= Amol;
398 }
399
400 fMassOfMolecule = Amol/Avogadro;
401 ComputeDerivedQuantities();
402 }
403}

407// AddElement -- composition by fraction of mass
408
409void G4Material::AddElement(G4Element* element, G4double fraction)
410{
411 if(fraction < 0.0 || fraction > 1.0) {
412 G4cout << "G4Material::AddElement ERROR for " << fName << " and "
413 << element->GetName() << " mass fraction= " << fraction
414 << " is wrong " << G4endl;
415 G4Exception ("G4Material::AddElement()", "mat032", FatalException,
416 "Attempt to add element with wrong mass fraction");
417 }
418 // initialization
419 if (fNumberOfComponents == 0) {
420 fMassFractionVector = new G4double[fArrayLength];
421 fAtomsVector = new G4int [fArrayLength];
422 }
423 // filling ...
424 if (fNumberOfComponents < maxNbComponents) {
425 G4int el = 0;
426 // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
427 while ((el<fNumberOfElements)&&(element!=(*theElementVector)[el])) { ++el; }
428 if (el<fNumberOfElements) fMassFractionVector[el] += fraction;
429 else {
430 theElementVector->push_back(element);
431 fMassFractionVector[el] = fraction;
432 ++fNumberOfElements;
433 }
434 ++fNumberOfComponents;
435 } else {
436 G4cout << "G4Material::AddElement ERROR for " << fName << " nElement= "
437 << fNumberOfElements << G4endl;
438 G4Exception ("G4Material::AddElement()", "mat033", FatalException,
439 "Attempt to add more than the declared number of elements.");
440 }
441
442 // filled.
443 if (fNumberOfComponents == maxNbComponents) {
444
445 G4int i=0;
446 G4double Zmol(0.), Amol(0.);
447 // check sum of weights -- OK?
448 G4double wtSum(0.0);
449 for (i=0; i<fNumberOfElements; ++i) {
450 wtSum += fMassFractionVector[i];
451 Zmol += fMassFractionVector[i]*(*theElementVector)[i]->GetZ();
452 Amol += fMassFractionVector[i]*(*theElementVector)[i]->GetA();
453 }
454 if (std::fabs(1.-wtSum) > perThousand) {
455 G4cerr << "WARNING !! for " << fName << " sum of fractional masses "
456 << wtSum << " is not 1 - results may be wrong"
457 << G4endl;
458 }
459 for (i=0; i<fNumberOfElements; ++i) {
460 fAtomsVector[i] =
461 G4lrint(fMassFractionVector[i]*Amol/(*theElementVector)[i]->GetA());
462 }
463
464 ComputeDerivedQuantities();
465 }
466}

470// AddMaterial -- composition by fraction of mass
471
472void G4Material::AddMaterial(G4Material* material, G4double fraction)
473{
474 if(fraction < 0.0 || fraction > 1.0) {
475 G4cout << "G4Material::AddMaterial ERROR for " << fName << " and "
476 << material->GetName() << " mass fraction= " << fraction
477 << " is wrong ";
478 G4Exception ("G4Material::AddMaterial()", "mat034", FatalException,
479 "Attempt to add material with wrong mass fraction");
480 }
481 // initialization
482 if (fNumberOfComponents == 0) {
483 fMassFractionVector = new G4double[fArrayLength];
484 fAtomsVector = new G4int [fArrayLength];
485 }
486
487 G4int nelm = material->GetNumberOfElements();
488
489 // arrays should be extended
490 if(nelm > 1) {
491 G4int nold = fArrayLength;
492 fArrayLength += nelm - 1;
493 G4double* v1 = new G4double[fArrayLength];
494 G4int* i1 = new G4int[fArrayLength];
495 for(G4int i=0; i<nold; ++i) {
496 v1[i] = fMassFractionVector[i];
497 i1[i] = fAtomsVector[i];
498 }
499 delete [] fAtomsVector;
500 delete [] fMassFractionVector;
501 fMassFractionVector = v1;
502 fAtomsVector = i1;
503 }
504
505 // filling ...
506 if (fNumberOfComponents < maxNbComponents) {
507 for (G4int elm=0; elm<nelm; ++elm)
508 {
509 G4Element* element = (*(material->GetElementVector()))[elm];
510 G4int el = 0;
511 // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
512 while ((el<fNumberOfElements)&&(element!=(*theElementVector)[el])) el++;
513 if (el < fNumberOfElements) fMassFractionVector[el] += fraction
514 *(material->GetFractionVector())[elm];
515 else {
516 theElementVector->push_back(element);
517 fMassFractionVector[el] = fraction
518 *(material->GetFractionVector())[elm];
519 ++fNumberOfElements;
520 }
521 }
522 ++fNumberOfComponents;
523 ///store massFraction of material component
524 fMatComponents[material] = fraction;
525
526 } else {
527 G4cout << "G4Material::AddMaterial ERROR for " << fName << " nElement= "
528 << fNumberOfElements << G4endl;
529 G4Exception ("G4Material::AddMaterial()", "mat035", FatalException,
530 "Attempt to add more than the declared number of components.");
531 }
532
533 // filled.
534 if (fNumberOfComponents == maxNbComponents) {
535 G4int i=0;
536 G4double Zmol(0.), Amol(0.);
537 // check sum of weights -- OK?
538 G4double wtSum(0.0);
539 for (i=0; i<fNumberOfElements; ++i) {
540 wtSum += fMassFractionVector[i];
541 Zmol += fMassFractionVector[i]*(*theElementVector)[i]->GetZ();
542 Amol += fMassFractionVector[i]*(*theElementVector)[i]->GetA();
543 }
544 if (std::fabs(1.-wtSum) > perThousand) {
545 G4cout << "G4Material::AddMaterial WARNING !! for " << fName
546 << " sum of fractional masses "
547 << wtSum << " is not 1 - results may be wrong"
548 << G4endl;
549 }
550 for (i=0; i<fNumberOfElements; ++i) {
551 fAtomsVector[i] =
552 G4lrint(fMassFractionVector[i]*Amol/(*theElementVector)[i]->GetA());
553 }
554
555 ComputeDerivedQuantities();
556 }
557}

561void G4Material::ComputeRadiationLength()
562{
563 G4double radinv = 0.0 ;
564 for (G4int i=0;i<fNumberOfElements;++i) {
565 radinv += VecNbOfAtomsPerVolume[i]*((*theElementVector)[i]->GetfRadTsai());
566 }
567 fRadlen = (radinv <= 0.0 ? DBL_MAX : 1./radinv);
568}
569
570//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo......
571
572void G4Material::ComputeNuclearInterLength()
573{
574 static const G4double lambda0 = 35*CLHEP::g/CLHEP::cm2;
575 static const G4double twothird = 2.0/3.0;
576 G4double NILinv = 0.0;
577 for (G4int i=0; i<fNumberOfElements; ++i) {
578 G4int Z = G4lrint( (*theElementVector)[i]->GetZ());
579 G4double A = (*theElementVector)[i]->GetN();
580 if(1 == Z) {
581 NILinv += VecNbOfAtomsPerVolume[i]*A;
582 } else {
583 NILinv += VecNbOfAtomsPerVolume[i]*G4Exp(twothird*G4Log(A));
584 }
585 }
586 NILinv *= amu/lambda0;
587 fNuclInterLen = (NILinv <= 0.0 ? DBL_MAX : 1./NILinv);
588}

606G4Material*
607G4Material::GetMaterial(const G4String& materialName, G4bool warning)
608{
609 // search the material by its name
610 for (size_t J=0 ; J<theMaterialTable.size() ; ++J)
611 {
612 if (theMaterialTable[J]->GetName() == materialName)
613 { return theMaterialTable[J]; }
614 }
615
616 // the material does not exist in the table
617 if (warning) {
618 G4cout << "G4Material::GetMaterial() WARNING: The material: "
619 << materialName
620 << " does not exist in the table. Return NULL pointer."
621 << G4endl;
622 }
623 return nullptr;
624}

671 << " Imean: " << std::setw(7) << std::setprecision(3)
672 << G4BestUnit(material->GetIonisation()->GetMeanExcitationEnergy(),

674 << " temperature: " << std::setw(6) << std::setprecision(2)
675 << (material->fTemp)/kelvin << " K"
676 << " pressure: " << std::setw(6) << std::setprecision(2)
677 << (material->fPressure)/atmosphere << " atm" << "\n";
678
679 for (G4int i=0; i<material->fNumberOfElements; i++) {
680 flux
681 << "\n ---> " << (*(material->theElementVector))[i]
682 << "\n ElmMassFraction: "
683 << std::setw(6)<< std::setprecision(2)
684 << (material->fMassFractionVector[i])/perCent << " %"
685 << " ElmAbundance " << std::setw(6)<< std::setprecision(2)
686 << 100*(material->VecNbOfAtomsPerVolume[i])
687 /(material->TotNbOfAtomsPerVolume)
688 << " % \n";
689 }
690 flux.precision(prec);
691 flux.setf(mode,std::ios::floatfield);
692
693 if(material->IsExtended())
