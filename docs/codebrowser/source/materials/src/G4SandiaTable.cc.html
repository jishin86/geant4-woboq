67G4SandiaTable::G4SandiaTable(G4Material* material)
68 : fMaterial(material)
69{
70 fMatSandiaMatrix = nullptr;
71 fMatSandiaMatrixPAI = nullptr;
72 fPhotoAbsorptionCof = nullptr;
73
74 fMatNbOfIntervals = 0;
75
76 fMaxInterval = 0;
77 fVerbose = 0;
78
79 //build the CumulInterval array
80 if(0 == fCumulInterval[0]) {
81 fCumulInterval[0] = 1;
82
83 for (G4int Z=1; Z<101; ++Z) {
84 fCumulInterval[Z] = fCumulInterval[Z-1] + fNbOfIntervals[Z];
85 }
86 }
87
88 fMaxInterval = 0;
89 fSandiaCofPerAtom.resize(4,0.0);
90 fLowerI1 = false;
91 //compute macroscopic Sandia coefs for a material
92 ComputeMatSandiaMatrix(); // mma
93}

133void
134G4SandiaTable::GetSandiaCofPerAtom(G4int Z, G4double energy,
135 std::vector<G4double>& coeff) const
136{
137#ifdef G4VERBOSE
138 if(Z < 1 || Z > 100) {
139 Z = PrintErrorZ(Z, "GetSandiaCofPerAtom");
140 }
141 if(4 > coeff.size()) {
142 PrintErrorV("GetSandiaCofPerAtom(): input vector is resized");
143 coeff.resize(4);
144 }
145#endif
146 G4double Emin = fSandiaTable[fCumulInterval[Z-1]][0]*CLHEP::keV;
147 //G4double Iopot = fIonizationPotentials[Z]*eV;
148 //if (Emin < Iopot) Emin = Iopot;
149
150 G4int row = 0;
151 if (energy <= Emin) {
152 energy = Emin;
153
154 } else {
155 G4int interval = fNbOfIntervals[Z] - 1;
156 row = fCumulInterval[Z-1] + interval;
157 // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
158 while ((interval>0) && (energy<fSandiaTable[row][0]*CLHEP::keV)) {
159 --interval;
160 row = fCumulInterval[Z-1] + interval;
161 }
162 }
163
164 G4double AoverAvo = Z*amu/fZtoAratio[Z];
165
166 coeff[0]=AoverAvo*funitc[1]*fSandiaTable[row][1];
167 coeff[1]=AoverAvo*funitc[2]*fSandiaTable[row][2];
168 coeff[2]=AoverAvo*funitc[3]*fSandiaTable[row][3];
169 coeff[3]=AoverAvo*funitc[4]*fSandiaTable[row][4];
170}

174void
175G4SandiaTable::GetSandiaCofWater(G4double energy,
176 std::vector<G4double>& coeff) const
177{
178#ifdef G4VERBOSE
179 if(4 > coeff.size()) {
180 PrintErrorV("GetSandiaCofWater: input vector is resized");
181 coeff.resize(4);
182 }
183#endif
184 G4int i = 0;
185 if(energy > fH2OlowerI1[0][0]*CLHEP::keV) {
186 i = fH2OlowerInt - 1;
187 for(; i>0; --i) {
188 if(energy >= fH2OlowerI1[i][0]*CLHEP::keV) { break; }
189 }
190 }
191 coeff[0]=funitc[1]*fH2OlowerI1[i][1];
192 coeff[1]=funitc[2]*fH2OlowerI1[i][2];
193 coeff[2]=funitc[3]*fH2OlowerI1[i][3];
194 coeff[3]=funitc[4]*fH2OlowerI1[i][4];
195}

248void G4SandiaTable::ComputeMatSandiaMatrix()
249{
250 //get list of elements
251 const G4int NbElm = fMaterial->GetNumberOfElements();
252 const G4ElementVector* ElementVector = fMaterial->GetElementVector();
253
254 G4int* Z = new G4int[NbElm]; //Atomic number
255
256 //determine the maximum number of energy-intervals for this material
257 G4int MaxIntervals = 0;
258 G4int elm, z;
259
260 // here we compute only for a mixture, so no waring or exception
261 // if z is out of validity interval
262 for (elm = 0; elm < NbElm; ++elm)
263 {
264 z = G4lrint((*ElementVector)[elm]->GetZ());
265 if(z < 1) { z = 1; }
266 else if(z > 100) { z = 100; }
267 Z[elm] = z;
268 MaxIntervals += fNbOfIntervals[z];
269 }
270
271 //copy the Energy bins in a tmp1 array
272 //(take care of the Ionization Potential of each element)
273 G4double* tmp1 = new G4double[MaxIntervals];
274 G4double IonizationPot;
275 G4int interval1 = 0;
276
277 for (elm = 0; elm < NbElm; ++elm)
278 {
279 z = Z[elm];
280 IonizationPot = fIonizationPotentials[z]*CLHEP::eV;
281 for(G4int row = fCumulInterval[z-1]; row<fCumulInterval[z]; ++row)
282 {
283 tmp1[interval1] = std::max(fSandiaTable[row][0]*CLHEP::keV,
284 IonizationPot);
285 ++interval1;
286 }
287 }
288 //sort the energies in strickly increasing values in a tmp2 array
289 //(eliminate redondances)
290
291 G4double* tmp2 = new G4double[MaxIntervals];
292 G4double Emin;
293 G4int interval2 = 0;
294
295 do
296 {
297 Emin = DBL_MAX;
298
299 for ( G4int i1 = 0; i1 < MaxIntervals; ++i1)
300 {
301 Emin = std::min(Emin, tmp1[i1]); //find the minimum
302 }
303 if (Emin < DBL_MAX) {
304 tmp2[interval2] = Emin;
305 ++interval2;
306 }
307 //copy Emin in tmp2
308 for ( G4int j1 = 0; j1 < MaxIntervals; ++j1)
309 {
310 if (tmp1[j1] <= Emin) { tmp1[j1] = DBL_MAX; } //eliminate from tmp1
311 }
312 // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
313 } while (Emin < DBL_MAX);
314
315 //create the sandia matrix for this material
316
317 fMatSandiaMatrix = new G4OrderedTable();
318 G4int interval;
319
320 for (interval = 0; interval < interval2; ++interval)
321 {
322 fMatSandiaMatrix->push_back( new G4DataVector(5,0.) );
323 }
324
325 //ready to compute the Sandia coefs for the material
326
327 const G4double* NbOfAtomsPerVolume = fMaterial->GetVecNbOfAtomsPerVolume();
328
329 static const G4double prec = 1.e-03*CLHEP::eV;
330 G4double coef, oldsum(0.), newsum(0.);
331 fMatNbOfIntervals = 0;
332
333 for ( interval = 0; interval < interval2; ++interval)
334 {
335 Emin = (*(*fMatSandiaMatrix)[fMatNbOfIntervals])[0] = tmp2[interval];
336
337 for ( G4int k = 1; k < 5; ++k ) {
338 (*(*fMatSandiaMatrix)[fMatNbOfIntervals])[k] = 0.;
339 }
340 newsum = 0.;
341
342 for ( elm = 0; elm < NbElm; elm++ )
343 {
344 GetSandiaCofPerAtom(Z[elm], Emin+prec, fSandiaCofPerAtom);
345
346 for ( G4int j = 1; j < 5; ++j )
347 {
348 coef = NbOfAtomsPerVolume[elm]*fSandiaCofPerAtom[j-1];
349 (*(*fMatSandiaMatrix)[fMatNbOfIntervals])[j] += coef;
350 newsum += std::abs(coef);
351 }
352 }
353 //check for null or redondant intervals
354
355 if (newsum != oldsum) { oldsum = newsum; ++fMatNbOfIntervals;}
356 }
357 delete [] Z;
358 delete [] tmp1;
359 delete [] tmp2;
360
361 if ( fVerbose > 0 )
362 {
363 G4cout<<"G4SandiaTable::ComputeMatSandiaMatrix(), mat = "
364 <<fMaterial->GetName()<<G4endl;
365
366 for( G4int i = 0; i < fMatNbOfIntervals; ++i)
367 {
368 G4cout<<i<<"\t"<<GetSandiaCofForMaterial(i,0)/keV<<" keV \t"
369 << GetSandiaCofForMaterial(i,1)
370 <<"\t"<< GetSandiaCofForMaterial(i,2)
371 <<"\t"<< GetSandiaCofForMaterial(i,3)
372 <<"\t"<< GetSandiaCofForMaterial(i,4)<<G4endl;
373 }
374 }
375}

379// Sandia matrix for PAI models based on vectors ...
380
381void G4SandiaTable::ComputeMatSandiaMatrixPAI()
382{
383 G4int MaxIntervals = 0;
384 G4int elm, c, i, j, jj, k, k1, k2, c1, n1, z;
385
386 const G4int noElm = fMaterial->GetNumberOfElements();
387 const G4ElementVector* ElementVector = fMaterial->GetElementVector();
388
389 std::vector<G4int> Z(noElm); //Atomic number
390
391 for ( elm = 0; elm < noElm; elm++ )
392 {
393 z = G4lrint((*ElementVector)[elm]->GetZ());
394 if(z < 1) { z = 1; }
395 else if(z > 100) { z = 100; }
396 Z[elm] = z;
397 MaxIntervals += fNbOfIntervals[Z[elm]];
398 }
399 fMaxInterval = MaxIntervals + 2;
400
401 if ( fVerbose > 0 )
402 {
403 G4cout<<"G4SandiaTable::ComputeMatSandiaMatrixPAI: fMaxInterval = "
404 <<fMaxInterval<<G4endl;
405 }
406
407 G4DataVector fPhotoAbsorptionCof0(fMaxInterval);
408 G4DataVector fPhotoAbsorptionCof1(fMaxInterval);
409 G4DataVector fPhotoAbsorptionCof2(fMaxInterval);
410 G4DataVector fPhotoAbsorptionCof3(fMaxInterval);
411 G4DataVector fPhotoAbsorptionCof4(fMaxInterval);
412
413 for( c = 0; c < fMaxInterval; ++c ) // just in case
414 {
415 fPhotoAbsorptionCof0[c] = 0.;
416 fPhotoAbsorptionCof1[c] = 0.;
417 fPhotoAbsorptionCof2[c] = 0.;
418 fPhotoAbsorptionCof3[c] = 0.;
419 fPhotoAbsorptionCof4[c] = 0.;
420 }
421 c = 1;
422
423 for(i = 0; i < noElm; ++i)
424 {
425 G4double I1 = fIonizationPotentials[Z[i]]*CLHEP::keV; // I1 in keV
426 n1 = 1;
427
428 for( j = 1; j < Z[i]; ++j ) n1 += fNbOfIntervals[j];
429
430 G4int n2 = n1 + fNbOfIntervals[Z[i]];
431
432 for( k1 = n1; k1 < n2; k1++ )
433 {
434 if( I1 > fSandiaTable[k1][0] )
435 {
436 continue; // no ionization for energies smaller than I1 (first
437 } // ionisation potential)
438 break;
439 }
440 G4int flag = 0;
441
442 for( c1 = 1; c1 < c; c1++ )
443 {
444 if( fPhotoAbsorptionCof0[c1] == I1 ) // this value already has existed
445 {
446 flag = 1;
447 break;
448 }
449 }
450 if(flag == 0)
451 {
452 fPhotoAbsorptionCof0[c] = I1;
453 ++c;
454 }
455 for( k2 = k1; k2 < n2; k2++ )
456 {
457 flag = 0;
458
459 for( c1 = 1; c1 < c; c1++ )
460 {
461 if( fPhotoAbsorptionCof0[c1] == fSandiaTable[k2][0] )
462 {
463 flag = 1;
464 break;
465 }
466 }
467 if(flag == 0)
468 {
469 fPhotoAbsorptionCof0[c] = fSandiaTable[k2][0];
470 ++c;
471 }
472 }
473 } // end for(i)
474 // sort out
475
476 for( i = 1; i < c; ++i )
477 {
478 for( j = i + 1; j < c; ++j )
479 {
480 if( fPhotoAbsorptionCof0[i] > fPhotoAbsorptionCof0[j] )
481 {
482 G4double tmp = fPhotoAbsorptionCof0[i];
483 fPhotoAbsorptionCof0[i] = fPhotoAbsorptionCof0[j];
484 fPhotoAbsorptionCof0[j] = tmp;
485 }
486 }
487 if ( fVerbose > 0)
488 {
489 G4cout<<i<<"\t energy = "<<fPhotoAbsorptionCof0[i]<<G4endl;
490 }
491 }
492 fMaxInterval = c;
493
494 const G4double* fractionW = fMaterial->GetFractionVector();
495
496 if ( fVerbose > 0)
497 {
498 for( i = 0; i < noElm; ++i )
499 G4cout<<i<<" = elN, fraction = "<<fractionW[i]<<G4endl;
500 }
501
502 for( i = 0; i < noElm; ++i )
503 {
504 n1 = 1;
505 G4double I1 = fIonizationPotentials[Z[i]]*keV;
506
507 for( j = 1; j < Z[i]; ++j ) n1 += fNbOfIntervals[j];
508
509 G4int n2 = n1 + fNbOfIntervals[Z[i]] - 1;
510
511 for(k = n1; k < n2; ++k)
512 {
513 G4double B1 = fSandiaTable[k][0];
514 G4double B2 = fSandiaTable[k+1][0];
515
516 for(G4int q = 1; q < fMaxInterval-1; q++)
517 {
518 G4double E1 = fPhotoAbsorptionCof0[q];
519 G4double E2 = fPhotoAbsorptionCof0[q+1];
520
521 if ( fVerbose > 0 )
522 {
523 G4cout<<"k = "<<k<<", q = "<<q<<", B1 = "<<B1<<", B2 = "<<B2
524 <<", E1 = "<<E1<<", E2 = "<<E2<<G4endl;
525 }
526 if( B1 > E1 || B2 < E2 || E1 < I1 )
527 {
528 if ( fVerbose > 0 )
529 {
530 G4cout<<"continue for: B1 = "<<B1<<", B2 = "<<B2<<", E1 = "
531 <<E1<<", E2 = "<<E2<<G4endl;
532 }
533 continue;
534 }
535 fPhotoAbsorptionCof1[q] += fSandiaTable[k][1]*fractionW[i];
536 fPhotoAbsorptionCof2[q] += fSandiaTable[k][2]*fractionW[i];
537 fPhotoAbsorptionCof3[q] += fSandiaTable[k][3]*fractionW[i];
538 fPhotoAbsorptionCof4[q] += fSandiaTable[k][4]*fractionW[i];
539 }
540 }
541 // Last interval
542
543 fPhotoAbsorptionCof1[fMaxInterval-1] += fSandiaTable[k][1]*fractionW[i];
544 fPhotoAbsorptionCof2[fMaxInterval-1] += fSandiaTable[k][2]*fractionW[i];
545 fPhotoAbsorptionCof3[fMaxInterval-1] += fSandiaTable[k][3]*fractionW[i];
546 fPhotoAbsorptionCof4[fMaxInterval-1] += fSandiaTable[k][4]*fractionW[i];
547 } // for(i)
548 c = 0; // Deleting of first intervals where all coefficients = 0
549
550 do
551 {
552 ++c;
553
554 if( fPhotoAbsorptionCof1[c] != 0.0 ||
555 fPhotoAbsorptionCof2[c] != 0.0 ||
556 fPhotoAbsorptionCof3[c] != 0.0 ||
557 fPhotoAbsorptionCof4[c] != 0.0 ) continue;
558
559 if ( fVerbose > 0 )
560 {
561 G4cout<<c<<" = number with zero cofs"<<G4endl;
562 }
563 for( jj = 2; jj < fMaxInterval; ++jj )
564 {
565 fPhotoAbsorptionCof0[jj-1] = fPhotoAbsorptionCof0[jj];
566 fPhotoAbsorptionCof1[jj-1] = fPhotoAbsorptionCof1[jj];
567 fPhotoAbsorptionCof2[jj-1] = fPhotoAbsorptionCof2[jj];
568 fPhotoAbsorptionCof3[jj-1] = fPhotoAbsorptionCof3[jj];
569 fPhotoAbsorptionCof4[jj-1] = fPhotoAbsorptionCof4[jj];
570 }
571 --fMaxInterval;
572 --c;
573 }
574 // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
575 while( c < fMaxInterval - 1 );
576
577 if( fPhotoAbsorptionCof0[fMaxInterval-1] == 0.0 ) fMaxInterval--;
578
579 // create the sandia matrix for this material
580
581 fMatSandiaMatrixPAI = new G4OrderedTable();
582
583 G4double density = fMaterial->GetDensity();
584
585 for (i = 0; i < fMaxInterval; ++i) // -> G4units
586 {
587 fPhotoAbsorptionCof0[i+1] *= funitc[0];
588 fPhotoAbsorptionCof1[i+1] *= funitc[1]*density;
589 fPhotoAbsorptionCof2[i+1] *= funitc[2]*density;
590 fPhotoAbsorptionCof3[i+1] *= funitc[3]*density;
591 fPhotoAbsorptionCof4[i+1] *= funitc[4]*density;
592 }
593 if(fLowerI1)
594 {
595 if( fMaterial->GetName() == "G4_WATER")
596 {
597 fMaxInterval += fH2OlowerInt;
598
599 for (i = 0; i < fMaxInterval; ++i) // init vector table
600 {
601 fMatSandiaMatrixPAI->push_back( new G4DataVector(5,0.) );
602 }
603 for (i = 0; i < fH2OlowerInt; ++i)
604 {
605 (*(*fMatSandiaMatrixPAI)[i])[0] = fH2OlowerI1[i][0];
606 (*(*fMatSandiaMatrixPAI)[i])[1] = fH2OlowerI1[i][1];
607 (*(*fMatSandiaMatrixPAI)[i])[2] = fH2OlowerI1[i][2];
608 (*(*fMatSandiaMatrixPAI)[i])[3] = fH2OlowerI1[i][3];
609 (*(*fMatSandiaMatrixPAI)[i])[4] = fH2OlowerI1[i][4];
610 }
611 for (i = fH2OlowerInt; i < fMaxInterval; ++i)
612 {
613 (*(*fMatSandiaMatrixPAI)[i])[0] = fPhotoAbsorptionCof0[i+1-fH2OlowerInt];
614 (*(*fMatSandiaMatrixPAI)[i])[1] = fPhotoAbsorptionCof1[i+1-fH2OlowerInt];
615 (*(*fMatSandiaMatrixPAI)[i])[2] = fPhotoAbsorptionCof2[i+1-fH2OlowerInt];
616 (*(*fMatSandiaMatrixPAI)[i])[3] = fPhotoAbsorptionCof3[i+1-fH2OlowerInt];
617 (*(*fMatSandiaMatrixPAI)[i])[4] = fPhotoAbsorptionCof4[i+1-fH2OlowerInt];
618 }
619 }
620 }
621 else
622 {
623 for (i = 0; i < fMaxInterval; ++i) // init vector table
624 {
625 fMatSandiaMatrixPAI->push_back( new G4DataVector(5,0.) );
626 }
627 for (i = 0; i < fMaxInterval; ++i)
628 {
629 (*(*fMatSandiaMatrixPAI)[i])[0] = fPhotoAbsorptionCof0[i+1];
630 (*(*fMatSandiaMatrixPAI)[i])[1] = fPhotoAbsorptionCof1[i+1]; // *density;
631 (*(*fMatSandiaMatrixPAI)[i])[2] = fPhotoAbsorptionCof2[i+1]; // *density;

764//
765// SandiaIntervals
766//
767
768G4int G4SandiaTable::SandiaIntervals(G4int Z[], G4int el )
769{
770 G4int c, i, flag = 0, n1 = 1;
771 G4int j, c1, k1, k2;
772 G4double I1;
773 fMaxInterval = 0;
774
775 for( i = 0; i < el; ++i ) fMaxInterval += fNbOfIntervals[ Z[i] ];
776
777 fMaxInterval += 2;
778
779 if( fVerbose > 0 ) {
780 G4cout<<"begin sanInt, fMaxInterval = "<<fMaxInterval<<G4endl;
781 }
782
783 fPhotoAbsorptionCof = new G4double* [fMaxInterval];
784
785 for( i = 0; i < fMaxInterval; ++i ) {
786 fPhotoAbsorptionCof[i] = new G4double[5];
787 }
788 // for(c = 0; c < fIntervalLimit; ++c) // just in case
789
790 for( c = 0; c < fMaxInterval; ++c ) { fPhotoAbsorptionCof[c][0] = 0.; }
791
792 c = 1;
793
794 for( i = 0; i < el; ++i )
795 {
796 I1 = fIonizationPotentials[ Z[i] ]*keV; // First ionization
797 n1 = 1; // potential in keV
798
799 for( j = 1; j < Z[i]; ++j ) n1 += fNbOfIntervals[j];
800
801 G4int n2 = n1 + fNbOfIntervals[Z[i]];
802
803 for( k1 = n1; k1 < n2; k1++ )
804 {
805 if( I1 > fSandiaTable[k1][0] )
806 {
807 continue; // no ionization for energies smaller than I1 (first
808 } // ionisation potential)
809 break;
810 }
811 flag = 0;
812
813 for( c1 = 1; c1 < c; c1++ )
814 {
815 if( fPhotoAbsorptionCof[c1][0] == I1 ) // this value already has existed
816 {
817 flag = 1;
818 break;
819 }
820 }
821 if( flag == 0 )
822 {
823 fPhotoAbsorptionCof[c][0] = I1;
824 ++c;
825 }
826 for( k2 = k1; k2 < n2; k2++ )
827 {
828 flag = 0;
829
830 for( c1 = 1; c1 < c; c1++ )
831 {
832 if( fPhotoAbsorptionCof[c1][0] == fSandiaTable[k2][0] )
833 {
834 flag = 1;
835 break;
836 }
837 }
838 if( flag == 0 )
839 {
840 fPhotoAbsorptionCof[c][0] = fSandiaTable[k2][0];
841 if( fVerbose > 0 ) {
842 G4cout<<"sanInt, c = "<<c<<", E_c = "<<fPhotoAbsorptionCof[c][0]
843 <<G4endl;
844 }
845 ++c;
846 }
847 }
848 } // end for(i)
849
850 SandiaSort(fPhotoAbsorptionCof,c);
851 fMaxInterval = c;
852 if( fVerbose > 0 ) {
853 G4cout<<"end SanInt, fMaxInterval = "<<fMaxInterval<<G4endl;
854 }
855 return c;
856}

858////////////////////////////////////////////////////////////////////////////..
859//
860// SandiaMixing
861//
862
863G4int
864G4SandiaTable::SandiaMixing( G4int Z[],
865 const G4double fractionW[],
866 G4int el,
867 G4int mi )
868{
869 G4int i, j, n1, k, c=1, jj, kk;
870 G4double I1, B1, B2, E1, E2;
871
872 for( i = 0; i < mi; ++i )
873 {
874 for( j = 1; j < 5; ++j ) fPhotoAbsorptionCof[i][j] = 0.;
875 }
876 for( i = 0; i < el; ++i )
877 {
878 n1 = 1;
879 I1 = fIonizationPotentials[Z[i]]*keV;
880
881 for( j = 1; j < Z[i]; ++j ) n1 += fNbOfIntervals[j];
882
883 G4int n2 = n1 + fNbOfIntervals[Z[i]] - 1;
884
885 for( k = n1; k < n2; ++k )
886 {
887 B1 = fSandiaTable[k][0];
888 B2 = fSandiaTable[k+1][0];
889
890 for( c = 1; c < mi-1; ++c )
891 {
892 E1 = fPhotoAbsorptionCof[c][0];
893 E2 = fPhotoAbsorptionCof[c+1][0];
894
895 if( B1 > E1 || B2 < E2 || E1 < I1 ) continue;
896
897 for( j = 1; j < 5; ++j )
898 {
899 fPhotoAbsorptionCof[c][j] += fSandiaTable[k][j]*fractionW[i];
900 if( fVerbose > 0 )
901 {
902 G4cout<<"c="<<c<<"; j="<<j<<"; fST="<<fSandiaTable[k][j]
903 <<"; frW="<<fractionW[i]<<G4endl;
904 }
905 }
906 }
907 }
908 for( j = 1; j < 5; ++j ) // Last interval
909 {
910 fPhotoAbsorptionCof[mi-1][j] += fSandiaTable[k][j]*fractionW[i];
911 if( fVerbose > 0 )
912 {
913 G4cout<<"mi-1="<<mi-1<<"; j="<<j<<"; fST="<<fSandiaTable[k][j]
914 <<"; frW="<<fractionW[i]<<G4endl;
915 }
916 }
917 } // for(i)
918 c = 0; // Deleting of first intervals where all coefficients = 0
919
920 do
921 {
922 ++c;
923
924 if( fPhotoAbsorptionCof[c][1] != 0.0 ||
925 fPhotoAbsorptionCof[c][2] != 0.0 ||
926 fPhotoAbsorptionCof[c][3] != 0.0 ||
927 fPhotoAbsorptionCof[c][4] != 0.0 ) continue;
928
929 for( jj = 2; jj < mi; ++jj )
930 {
931 for( kk = 0; kk < 5; ++kk ) {
932 fPhotoAbsorptionCof[jj-1][kk] = fPhotoAbsorptionCof[jj][kk];
933 }
934 }
935 mi--;
936 c--;
937 }
938 // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
939 while( c < mi - 1 );
940
941 if( fVerbose > 0 ) G4cout<<"end SanMix, mi = "<<mi<<G4endl;
942
943 return mi;
944}

1048////////////////////////////////////////////////////////////////////////////////
1049//
1050// Sandia interval and mixing calculations for materialCutsCouple constructor
1051//
1052
1053void G4SandiaTable::ComputeMatTable()
1054{
1055 G4int MaxIntervals = 0;
1056 G4int elm, c, i, j, jj, k, kk, k1, k2, c1, n1;
1057
1058 const G4int noElm = fMaterial->GetNumberOfElements();
1059 const G4ElementVector* ElementVector = fMaterial->GetElementVector();
1060 G4int* Z = new G4int[noElm]; //Atomic number
1061
1062 for (elm = 0; elm<noElm; ++elm)
1063 {
1064 Z[elm] = (G4int)(*ElementVector)[elm]->GetZ();
1065 MaxIntervals += fNbOfIntervals[Z[elm]];
1066 }
1067 fMaxInterval = 0;
1068
1069 for(i = 0; i < noElm; ++i) fMaxInterval += fNbOfIntervals[Z[i]];
1070
1071 fMaxInterval += 2;
1072
1073 // G4cout<<"fMaxInterval = "<<fMaxInterval<<G4endl;
1074
1075 fPhotoAbsorptionCof = new G4double* [fMaxInterval];
1076
1077 for(i = 0; i < fMaxInterval; ++i)
1078 {
1079 fPhotoAbsorptionCof[i] = new G4double[5];
1080 }
1081
1082 // for(c = 0; c < fIntervalLimit; ++c) // just in case
1083
1084 for(c = 0; c < fMaxInterval; ++c) // just in case
1085 {
1086 fPhotoAbsorptionCof[c][0] = 0.;
1087 }
1088 c = 1;
1089
1090 for(i = 0; i < noElm; ++i)
1091 {
1092 G4double I1 = fIonizationPotentials[Z[i]]*keV; // First ionization
1093 n1 = 1; // potential in keV
1094
1095 for(j = 1; j < Z[i]; ++j)
1096 {
1097 n1 += fNbOfIntervals[j];
1098 }
1099 G4int n2 = n1 + fNbOfIntervals[Z[i]];
1100
1101 for(k1 = n1; k1 < n2; ++k1)
1102 {
1103 if(I1 > fSandiaTable[k1][0])
1104 {
1105 continue; // no ionization for energies smaller than I1 (first
1106 } // ionisation potential)
1107 break;
1108 }
1109 G4int flag = 0;
1110
1111 for(c1 = 1; c1 < c; ++c1)
1112 {
1113 if(fPhotoAbsorptionCof[c1][0] == I1) // this value already has existed
1114 {
1115 flag = 1;
1116 break;
1117 }
1118 }
1119 if(flag == 0)
1120 {
1121 fPhotoAbsorptionCof[c][0] = I1;
1122 ++c;
1123 }
1124 for(k2 = k1; k2 < n2; ++k2)
1125 {
1126 flag = 0;
1127
1128 for(c1 = 1; c1 < c; ++c1)
1129 {
1130 if(fPhotoAbsorptionCof[c1][0] == fSandiaTable[k2][0])
1131 {
1132 flag = 1;
1133 break;
1134 }
1135 }
1136 if(flag == 0)
1137 {
1138 fPhotoAbsorptionCof[c][0] = fSandiaTable[k2][0];
1139 ++c;
1140 }
1141 }
1142 } // end for(i)
1143
1144 SandiaSort(fPhotoAbsorptionCof,c);
1145 fMaxInterval = c;
1146
1147 const G4double* fractionW = fMaterial->GetFractionVector();
1148
1149 for(i = 0; i < fMaxInterval; ++i)
1150 {
1151 for(j = 1; j < 5; ++j) fPhotoAbsorptionCof[i][j] = 0.;
1152 }
1153 for(i = 0; i < noElm; ++i)
1154 {
1155 n1 = 1;
1156 G4double I1 = fIonizationPotentials[Z[i]]*keV;
1157
1158 for(j = 1; j < Z[i]; ++j)
1159 {
1160 n1 += fNbOfIntervals[j];
1161 }
1162 G4int n2 = n1 + fNbOfIntervals[Z[i]] - 1;
1163
1164 for(k = n1; k < n2; ++k)
1165 {
1166 G4double B1 = fSandiaTable[k][0];
1167 G4double B2 = fSandiaTable[k+1][0];
1168 for(G4int q = 1; q < fMaxInterval-1; q++)
1169 {
1170 G4double E1 = fPhotoAbsorptionCof[q][0];
1171 G4double E2 = fPhotoAbsorptionCof[q+1][0];
1172 if(B1 > E1 || B2 < E2 || E1 < I1)
1173 {
1174 continue;
1175 }
1176 for(j = 1; j < 5; ++j)
1177 {
1178 fPhotoAbsorptionCof[q][j] += fSandiaTable[k][j]*fractionW[i];
1179 }
1180 }
1181 }
1182 for(j = 1; j < 5; ++j) // Last interval
1183 {
1184 fPhotoAbsorptionCof[fMaxInterval-1][j] +=
1185 fSandiaTable[k][j]*fractionW[i];
1186 }
1187 } // for(i)
1188
1189 c = 0; // Deleting of first intervals where all coefficients = 0
1190
1191 do
1192 {
1193 ++c;
1194
1195 if( fPhotoAbsorptionCof[c][1] != 0.0 ||
1196 fPhotoAbsorptionCof[c][2] != 0.0 ||
1197 fPhotoAbsorptionCof[c][3] != 0.0 ||
1198 fPhotoAbsorptionCof[c][4] != 0.0 ) continue;
1199
1200 for(jj = 2; jj < fMaxInterval; ++jj)
1201 {
1202 for(kk = 0; kk < 5; ++kk)
1203 {
1204 fPhotoAbsorptionCof[jj-1][kk]= fPhotoAbsorptionCof[jj][kk];
1205 }
1206 }
1207 --fMaxInterval;
1208 --c;
1209 }
1210 // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
1211 while( c < fMaxInterval - 1 );
1212
1213 // create the sandia matrix for this material
1214
1215 --fMaxInterval; // vmg 20.11.10
1216
1217 fMatSandiaMatrix = new G4OrderedTable();
1218
1219 for (i = 0; i < fMaxInterval; ++i)
1220 {
1221 fMatSandiaMatrix->push_back(new G4DataVector(5,0.));
1222 }
1223 for ( i = 0; i < fMaxInterval; ++i )
1224 {
1225 for( j = 0; j < 5; ++j )
1226 {
1227 (*(*fMatSandiaMatrix)[i])[j] = fPhotoAbsorptionCof[i+1][j];
1228 }
1229 }
1230 fMatNbOfIntervals = fMaxInterval;
1231
1232 if ( fVerbose > 0 )
1233 {
1234 G4cout<<"vmg, G4SandiaTable::ComputeMatTable(), mat = "
1235 <<fMaterial->GetName()<<G4endl;
1236
1237 for ( i = 0; i < fMaxInterval; ++i )
1238 {
1239 // G4cout<<i<<"\t"<<(*(*fMatSandiaMatrix)[i])[0]<<" keV \t"
1240 // <<(*(*fMatSandiaMatrix)[i])[1]
1241 // <<"\t"<<(*(*fMatSandiaMatrix)[i])[2]<<"\t"
1242 // <<(*(*fMatSandiaMatrix)[i])[3]
1243 // <<"\t"<<(*(*fMatSandiaMatrix)[i])[4]<<G4endl;
1244
1245 G4cout<<i<<"\t"<<GetSandiaCofForMaterial(i,0)/keV
1246 <<" keV \t"<<this->GetSandiaCofForMaterial(i,1)
1247 <<"\t"<<this->GetSandiaCofForMaterial(i,2)
1248 <<"\t"<<this->GetSandiaCofForMaterial(i,3)
1249 <<"\t"<<this->GetSandiaCofForMaterial(i,4)<<G4endl;
1250 }
1251 }
1252 delete [] Z;
1253 return;
1254}
