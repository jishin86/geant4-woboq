60 : fMaterial(material)
61{
62 fBirks = 0.;
63 fMeanEnergyPerIon = 0.0;
64 twoln10 = 2.*G4Pow::GetInstance()->logZ(10);
65
66 // minimal set of default parameters for density effect
67 fCdensity = 0.0;
68 fD0density = 0.0;
69 fAdjustmentFactor = 1.0;
70 if(fDensityData == nullptr) { fDensityData = new G4DensityEffectData(); }
71
72 // compute parameters
73 ComputeMeanParameters();
74 ComputeDensityEffect();
75 ComputeFluctModel();
76 ComputeIonParameters();
77}

129void G4IonisParamMat::ComputeMeanParameters()
130{
131 // compute mean excitation energy and shell correction vector
132 fTaul = (*(fMaterial->GetElementVector()))[0]->GetIonisation()->GetTaul();
133
134 fMeanExcitationEnergy = 0.;
135 fLogMeanExcEnergy = 0.;
136
137 size_t nElements = fMaterial->GetNumberOfElements();
138 const G4ElementVector* elmVector = fMaterial->GetElementVector();
139 const G4double* nAtomsPerVolume = fMaterial->GetVecNbOfAtomsPerVolume();
140
141 fMeanExcitationEnergy = FindMeanExcitationEnergy(fMaterial);
142
143 // Chemical formula defines mean excitation energy
144 if(fMeanExcitationEnergy > 0.0) {
145 fLogMeanExcEnergy = G4Log(fMeanExcitationEnergy);
146
147 // Compute average
148 } else {
149 for (size_t i=0; i < nElements; i++) {
150 const G4Element* elm = (*elmVector)[i];
151 fLogMeanExcEnergy += nAtomsPerVolume[i]*elm->GetZ()
152 *G4Log(elm->GetIonisation()->GetMeanExcitationEnergy());
153 }
154 fLogMeanExcEnergy /= fMaterial->GetTotNbOfElectPerVolume();
155 fMeanExcitationEnergy = G4Exp(fLogMeanExcEnergy);
156 }
157
158 fShellCorrectionVector = new G4double[3];
159
160 for (G4int j=0; j<=2; j++)
161 {
162 fShellCorrectionVector[j] = 0.;
163
164 for (size_t k=0; k<nElements; k++) {
165 fShellCorrectionVector[j] += nAtomsPerVolume[k]
166 *(((*elmVector)[k])->GetIonisation()->GetShellCorrectionVector())[j];
167 }
168 fShellCorrectionVector[j] *= 2.0/fMaterial->GetTotNbOfElectPerVolume();
169 }
170}

181void G4IonisParamMat::ComputeDensityEffect()
182{
183 G4State State = fMaterial->GetState();
184
185 // Check if density effect data exist in the table
186 // R.M. Sternheimer, Atomic Data and Nuclear Data Tables, 30: 261 (1984)
187 G4int idx = fDensityData->GetIndex(fMaterial->GetName());
188 G4int nelm= fMaterial->GetNumberOfElements();
189 G4int Z0 = ((*(fMaterial->GetElementVector()))[0])->GetZasInt();
190
191 // for simple non-NIST materials
192 G4double corr = 0.0;
193 if(idx < 0 && 1 == nelm) {
194 idx = fDensityData->GetElementIndex(Z0, fMaterial->GetState());
195
196 // Correction for base material or for non-nominal density
197 // Except cases of very different density defined in user code
198 if(idx >= 0) {
199 const G4Material* bmat = fMaterial->GetBaseMaterial();
200 if(bmat) {
201 corr = G4Log(bmat->GetDensity()/fMaterial->GetDensity());
202 } else {
203 G4double dens = G4NistManager::Instance()->GetNominalDensity(Z0);
204 if(dens <= 0.0) { idx = -1; }
205 else {
206 corr = G4Log(dens/fMaterial->GetDensity());
207 }
208 }
209 // 1.0 is an arbitrary empirical limit
210 // parameterisation with very different density is not applicable
211 if(std::abs(corr) > 1.0) { idx = -1; }
212 }
213 }
214
215 //G4cout<<"DensityEffect for "<<fMaterial->GetName()<<" "<< idx << G4endl;
216
217 if(idx >= 0) {
218
219 // Take parameters for the density effect correction from
220 // R.M. Sternheimer et al. Density Effect For The Ionization Loss
221 // of Charged Particles in Various Substances.
222 // Atom. Data Nucl. Data Tabl. 30 (1984) 261-271.
223
224 fCdensity = fDensityData->GetCdensity(idx);
225 fMdensity = fDensityData->GetMdensity(idx);
226 fAdensity = fDensityData->GetAdensity(idx);
227 fX0density = fDensityData->GetX0density(idx);
228 fX1density = fDensityData->GetX1density(idx);
229 fD0density = fDensityData->GetDelta0density(idx);
230 fPlasmaEnergy = fDensityData->GetPlasmaEnergy(idx);
231 fAdjustmentFactor = fDensityData->GetAdjustmentFactor(idx);
232
233 // parameter C is computed and not taken from Sternheimer tables
234 //fCdensity = 1. + 2*G4Log(fMeanExcitationEnergy/fPlasmaEnergy);
235 //G4cout << "IonisParamMat: " << fMaterial->GetName()
236 // << " Cst= " << Cdensity << " C= " << fCdensity << G4endl;
237
238 // correction on nominal density
239 fCdensity += corr;
240 fX0density += corr/twoln10;
241 fX1density += corr/twoln10;
242
243 } else {
244
245 static const G4double Cd2 = 4*pi*hbarc_squared*classic_electr_radius;
246 fPlasmaEnergy = std::sqrt(Cd2*fMaterial->GetTotNbOfElectPerVolume());
247
248 // Compute parameters for the density effect correction in DE/Dx formula.
249 // The parametrization is from R.M. Sternheimer, Phys. Rev.B,3:3681 (1971)
250 G4int icase;
251
252 fCdensity = 1. + 2*G4Log(fMeanExcitationEnergy/fPlasmaEnergy);
253 //
254 // condensed materials
255 //
256 if ((State == kStateSolid)||(State == kStateLiquid)) {
257
258 static const G4double E100eV = 100.*eV;
259 static const G4double ClimiS[] = {3.681 , 5.215 };
260 static const G4double X0valS[] = {1.0 , 1.5 };
261 static const G4double X1valS[] = {2.0 , 3.0 };
262
263 if(fMeanExcitationEnergy < E100eV) { icase = 0; }
264 else { icase = 1; }
265
266 if(fCdensity < ClimiS[icase]) { fX0density = 0.2; }
267 else { fX0density = 0.326*fCdensity - X0valS[icase]; }
268
269 fX1density = X1valS[icase]; fMdensity = 3.0;
270
271 //special: Hydrogen
272 if (1 == nelm && 1 == Z0) {
273 fX0density = 0.425; fX1density = 2.0; fMdensity = 5.949;
274 }
275 } else {
276 //
277 // gases
278 //
279 fMdensity = 3.;
280 fX1density = 4.0;
281 //static const G4double ClimiG[] = {10.,10.5,11.,11.5,12.25,13.804};
282 //static const G4double X0valG[] = {1.6,1.7,1.8,1.9,2.0,2.0};
283 //static const G4double X1valG[] = {4.0,4.0,4.0,4.0,4.0,5.0};
284
285 if(fCdensity < 10.) {
286 fX0density = 1.6;
287 } else if(fCdensity < 11.5) {
288 fX0density = 1.6 + 0.2*(fCdensity - 10.);
289 } else if(fCdensity < 12.25) {
290 fX0density = 1.9 + (fCdensity - 11.5)/7.5;
291 } else if(fCdensity < 13.804) {
292 fX0density = 2.0;
293 fX1density = 4.0 + (fCdensity - 12.25)/1.554;
294 } else {
295 fX0density = 0.326*fCdensity-2.5; fX1density = 5.0;
296 }
297
298 //special: Hydrogen
299 if (1 == nelm && 1 == Z0) {
300 fX0density = 1.837; fX1density = 3.0; fMdensity = 4.754;
301 }
302
303 //special: Helium
304 if (1 == nelm && 2 == Z0) {
305 fX0density = 2.191; fX1density = 3.0; fMdensity = 3.297;
306 }
307 }
308 }
309
310 // change parameters if the gas is not in STP.
311 // For the correction the density(STP) is needed.
312 // Density(STP) is calculated here :
313
314
315 if (State == kStateGas) {
316 G4double Density = fMaterial->GetDensity();
317 G4double Pressure = fMaterial->GetPressure();
318 G4double Temp = fMaterial->GetTemperature();
319
320 G4double DensitySTP = Density*STP_Pressure*Temp/(Pressure*NTP_Temperature);
321
322 G4double ParCorr = G4Log(Density/DensitySTP);
323
324 fCdensity -= ParCorr;
325 fX0density -= ParCorr/twoln10;
326 fX1density -= ParCorr/twoln10;
327 }
328
329 // fAdensity parameter can be fixed for not conductive materials
330 if(0.0 == fD0density) {
331 G4double Xa = fCdensity/twoln10;
332 fAdensity = twoln10*(Xa-fX0density)
333 /std::pow((fX1density-fX0density),fMdensity);
334 }
335 /*
336 G4cout << "G4IonisParamMat: density effect data for <"
337 << fMaterial->GetName()
338 << "> " << G4endl;
339 G4cout << "Eplasma(eV)= " << fPlasmaEnergy/eV
340 << " rho= " << fAdjustmentFactor
341 << " -C= " << fCdensity
342 << " x0= " << fX0density
343 << " x1= " << fX1density
344 << " a= " << fAdensity
345 << " m= " << fMdensity
346 << G4endl;
347 */
348}

352void G4IonisParamMat::ComputeFluctModel()
353{
354 // compute parameters for the energy loss fluctuation model
355 // needs an 'effective Z'
356 G4double Zeff = 0.;
357 for (size_t i=0;i<fMaterial->GetNumberOfElements();i++) {
358 Zeff += (fMaterial->GetFractionVector())[i]
359 *((*(fMaterial->GetElementVector()))[i]->GetZ());
360 }
361 if (Zeff > 2.) { fF2fluct = 2./Zeff; }
362 else { fF2fluct = 0.; }
363
364 fF1fluct = 1. - fF2fluct;
365 fEnergy2fluct = 10.*Zeff*Zeff*eV;
366 fLogEnergy2fluct = G4Log(fEnergy2fluct);
367 fLogEnergy1fluct = (fLogMeanExcEnergy - fF2fluct*fLogEnergy2fluct)
368 /fF1fluct;
369 fEnergy1fluct = G4Exp(fLogEnergy1fluct);
370 fEnergy0fluct = 10.*eV;
371 fRateionexcfluct = 0.4;
372}

376void G4IonisParamMat::ComputeIonParameters()
377{
378 // get elements in the actual material,
379 const G4ElementVector* theElementVector = fMaterial->GetElementVector() ;
380 const G4double* theAtomicNumDensityVector =
381 fMaterial->GetAtomicNumDensityVector() ;
382 const G4int NumberOfElements = fMaterial->GetNumberOfElements() ;
383
384 // loop for the elements in the material
385 // to find out average values Z, vF, lF
386 G4double z(0.0), vF(0.0), lF(0.0), norm(0.0), a23(0.0);
387
388 G4Pow* g4pow = G4Pow::GetInstance();
389 if( 1 == NumberOfElements ) {
390 const G4Element* element = (*theElementVector)[0];
391 z = element->GetZ();
392 vF= element->GetIonisation()->GetFermiVelocity();
393 lF= element->GetIonisation()->GetLFactor();
394 a23 = 1.0/g4pow->A23(element->GetN());
395
396 } else {
397 for (G4int iel=0; iel<NumberOfElements; iel++)
398 {
399 const G4Element* element = (*theElementVector)[iel];
400 const G4double weight = theAtomicNumDensityVector[iel];
401 norm += weight ;
402 z += element->GetZ() * weight;
403 vF += element->GetIonisation()->GetFermiVelocity() * weight;
404 lF += element->GetIonisation()->GetLFactor() * weight;
405 a23 += weight/g4pow->A23(element->GetN());
406 }
407 z /= norm;
408 vF /= norm;
409 lF /= norm;
410 a23 /= norm;
411 }
412 fZeff = z;
413 fLfactor = lF;
414 fFermiEnergy = 25.*keV*vF*vF;
415 fInvA23 = a23;
416}

420void G4IonisParamMat::SetMeanExcitationEnergy(G4double value)
421{
422 if(value == fMeanExcitationEnergy || value <= 0.0) { return; }
423 if (G4NistManager::Instance()->GetVerbose() > 1) {
424 G4cout << "G4Material: Mean excitation energy is changed for "
425 << fMaterial->GetName()
426 << " Iold= " << fMeanExcitationEnergy/eV
427 << "eV; Inew= " << value/eV << " eV;"
428 << G4endl;
429 }
430
431 fMeanExcitationEnergy = value;
432
433 // add corrections to density effect
434 G4double newlog = G4Log(value);
435 G4double corr = 2*(newlog - fLogMeanExcEnergy);
436 fCdensity += corr;
437 fX0density += corr/twoln10;
438 fX1density += corr/twoln10;
439
440 // recompute parameters of fluctuation model
441 fLogMeanExcEnergy = newlog;
442 ComputeFluctModel();
443}
