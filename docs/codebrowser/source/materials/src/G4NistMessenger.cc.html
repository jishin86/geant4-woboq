55G4NistMessenger::G4NistMessenger(G4NistManager* man)
56 : manager(man)
57{
58 matDir = new G4UIdirectory("/material/");
59 matDir->SetGuidance("Commands for materials");
60
61 verCmd = new G4UIcmdWithAnInteger("/material/verbose",this);
62 verCmd->SetGuidance("Set verbose level.");
63
64 nistDir = new G4UIdirectory("/material/nist/");
65 nistDir->SetGuidance("Commands for the nist dataBase");
66
67 prtElmCmd = new G4UIcmdWithAString("/material/nist/printElement",this);
68 prtElmCmd->SetGuidance("print element(s) in dataBase.");
69 prtElmCmd->SetGuidance("symbol = element.");
70 prtElmCmd->SetGuidance("all = all elements.");
71 prtElmCmd->SetParameterName("symbol", true);
72 prtElmCmd->SetDefaultValue("all");
73
74 przElmCmd = new G4UIcmdWithAnInteger("/material/nist/printElementZ",this);
75 przElmCmd->SetGuidance("print element Z in dataBase.");
76 przElmCmd->SetGuidance("0 = all elements.");
77 przElmCmd->SetParameterName("Z", true);
78 przElmCmd->SetDefaultValue(0);
79 przElmCmd->SetRange("0<=Z && Z<108");
80
81 lisMatCmd = new G4UIcmdWithAString("/material/nist/listMaterials",this);
82 lisMatCmd->SetGuidance("Materials in Geant4 dataBase.");
83 lisMatCmd->SetGuidance("simple - simple NIST materials.");
84 lisMatCmd->SetGuidance("compound - compound NIST materials.");
85 lisMatCmd->SetGuidance("hep - HEP materials.");
86 lisMatCmd->SetGuidance("bio - biomedical materials.");
87 lisMatCmd->SetGuidance("all - list of all Geant4 materials.");
88 lisMatCmd->SetParameterName("matlist", true);
89 // lisMatCmd->SetCandidates("simple compound hep bio all");
90 lisMatCmd->SetDefaultValue("all");
91
92 g4Dir = new G4UIdirectory("/material/g4/");
93 g4Dir->SetGuidance("Commands for G4MaterialTable");
94
95 g4ElmCmd = new G4UIcmdWithAString("/material/g4/printElement",this);
96 g4ElmCmd->SetGuidance("print Element from G4ElementTable.");
97 g4ElmCmd->SetGuidance("all - all elements.");
98 g4ElmCmd->SetParameterName("elm", true);
99 g4ElmCmd->SetDefaultValue("all");
100
101 g4MatCmd = new G4UIcmdWithAString("/material/g4/printMaterial",this);
102 g4MatCmd->SetGuidance("print Material from G4MaterialTable.");
103 g4MatCmd->SetGuidance("all - all materials");
104 g4MatCmd->SetParameterName("pmat", true);
105 g4MatCmd->SetDefaultValue("all");
106
107 g4DensCmd = new G4UIcmdWithAString("/material/g4/printDensityEffParam",this);
108 g4DensCmd->SetGuidance("print Material from G4DensityEffectData.");
109 g4DensCmd->SetGuidance("all - all materials");
110 g4DensCmd->SetParameterName("dmat", true);
111 g4DensCmd->SetDefaultValue("all");
112}

133void G4NistMessenger::SetNewValue(G4UIcommand* command, G4String newValue)
134{
135 //G4cout << "G4NistMessenger::SetNewValue <" << newValue << ">" << G4endl;
136 if (command == verCmd)
137 { manager->SetVerbose(verCmd->GetNewIntValue(newValue));}
138
139 else if (command == prtElmCmd)
140 { manager->PrintElement(newValue); }
141
142 else if (command == przElmCmd) {
143 G4int Z = przElmCmd->GetNewIntValue(newValue);
144 if(Z >= 0 && Z < 108) { manager->PrintElement(Z); }
145 }
146 else if (command == lisMatCmd)
147 { manager->ListMaterials(newValue); }
148
149 else if (command == g4ElmCmd)
150 { manager->PrintG4Element(newValue); }
151
152 else if (command == g4MatCmd)
153 { manager->PrintG4Material(newValue); }
154
155 else if (command == g4DensCmd)
156 { G4IonisParamMat::GetDensityEffectData()->PrintData(newValue); }
157}
