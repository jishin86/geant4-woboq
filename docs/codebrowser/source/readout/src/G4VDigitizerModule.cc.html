29#include "G4VDigitizerModule.hh"
30#include "G4VDigiCollection.hh"
31#include "G4DigiManager.hh"
32
33G4VDigitizerModule::G4VDigitizerModule(G4String modName)
34:verboseLevel(0)
35{
36 moduleName = modName;
37 DigiManager = G4DigiManager::GetDMpointer();
38}
39
40G4VDigitizerModule::~G4VDigitizerModule()
41{;}
42
43G4bool G4VDigitizerModule::operator==(const G4VDigitizerModule &right) const
44{ return (moduleName==right.moduleName); }
45
46G4bool G4VDigitizerModule::operator!=(const G4VDigitizerModule &right) const
47{ return (moduleName!=right.moduleName); }
48
49void G4VDigitizerModule::StoreDigiCollection(G4VDigiCollection* aDC)
50{
51 G4String DCnam = moduleName;
52 DCnam += "/";
53 DCnam += aDC->GetName();
54 G4int DCID = DigiManager->GetDigiCollectionID(DCnam);
55 if(DCID>=0) StoreDigiCollection(DCID,aDC);
56}
57
58void G4VDigitizerModule::StoreDigiCollection(G4int DCID,G4VDigiCollection* aDC)
59{
60 DigiManager->SetDigiCollection(DCID,aDC);
61}
