41G4double G4HyperNucleiProperties::GetNuclearMass(G4int A, G4int Z, G4int LL)
42{
43 if (LL==0) return G4NucleiProperties::GetNuclearMass(A, Z);
44
45 G4ParticleTable* pTable = G4ParticleTable::GetParticleTable();
46 if (A < 2 || Z < 0 || Z > A-LL || LL>A ) {
47#ifdef G4VERBOSE
48 if (pTable->GetVerboseLevel()>0) {
49 G4cout << "G4HyperNucleiProperties::GetNuclearMass: "
50 << " Wrong values for A = " << A
51 << " Z = " << Z
52 << " L = " << LL
53 << G4endl;
54 }
55#endif
56 return 0.0;
57 } else if( A==2 ) {
58#ifdef G4VERBOSE
59 if (pTable->GetVerboseLevel()>0) {
60 G4cout << "G4HyperNucleiProperties::GetNuclearMass: "
61 << " No boud state for A = " << A
62 << " Z = " << Z
63 << " L = " << LL
64 << G4endl;
65 }
66#endif
67 return 0.0;
68 }
69
70
71 G4ParticleDefinition* lambda = pTable->FindParticle("lambda");
72 if (lambda ==0) {
73#ifdef G4VERBOSE
74 if (pTable->GetVerboseLevel()>0) {
75 G4cout << "G4HyperNucleiProperties::GetNuclearMass: "
76 << " Lambda is not defined " << G4endl;
77 }
78#endif
79 return 0.0;
80 }
81 const G4double mLL= lambda->GetPDGMass(); // mLambda
82
83 static const G4double b7=25.*MeV;
84 static const G4double b8=10.5; // Slope
85 static const G4double a2=0.13*MeV; // BindingEnergy for d+Lambda(MeV)
86 static const G4double a3=2.2*MeV; // BindingEnergy for (t/He3)+Lamb(MeV)
87 static const G4double eps =0.0001*MeV; // security value (MeV)
88
89 G4double mass = G4NucleiProperties::GetNuclearMass(A-LL, Z);
90 // A non-"strange" nucleus
91 G4double bs=0.;
92 if (A-LL ==2) bs=a2; // for nnL,npL,ppL
93 else if(A-LL ==3) bs=a3; // for 3nL,2npL,n2pL,3pL
94 else if(A-LL >3) bs=b7*std::exp(-b8/(A-LL+1.));
95 mass += LL*(mLL-bs) + eps;
96
97 return mass;
98}

101G4double G4HyperNucleiProperties::GetAtomicMass(G4int A, G4int Z, G4int LL)
102{
103 if (A < 1 || Z < 0 || Z > A-LL || LL > A || LL <0 ) {
104#ifdef G4VERBOSE
105 if (G4ParticleTable::GetParticleTable()->GetVerboseLevel()>0) {
106 G4cout << "G4HyperNucleiProperties::GetAtomicMass: "
107 << " Wrong values for A = " << A
108 << " Z = " << Z
109 << " L = " << LL << G4endl;
110 }
111#endif
112 return 0.0;
113
114 } else {
115 G4double nuclearMass = GetNuclearMass(A, Z, LL);
116 return nuclearMass + Z*electron_mass_c2
117 - 1.433e-5*MeV*std::pow(G4double(Z),2.39);
118 }
