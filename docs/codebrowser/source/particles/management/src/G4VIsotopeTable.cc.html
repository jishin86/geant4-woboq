72G4IsotopeProperty* G4VIsotopeTable::GetIsotopeByIsoLvl(G4int Z, G4int A, G4int level)

74 // temporal implementation
75 if (level==0) return GetIsotope(Z, A, 0.0);
76 else return 0;

80void G4VIsotopeTable::DumpTable(G4int Zmin, G4int Zmax)
81{
82 G4int Z, A;
83 G4int lvl;
84 const G4int MAX_LVL=9;
85 for ( Z =Zmin; Z<=Zmax; Z++){
86 for ( A= Z; A<=3*Z; A++){
87 for ( lvl=0; lvl<=MAX_LVL; lvl++){
88 G4IsotopeProperty* ptr = GetIsotope(Z,A,lvl);
89 if (ptr!=0) ptr->DumpInfo();
90 }
91 }
92 }
