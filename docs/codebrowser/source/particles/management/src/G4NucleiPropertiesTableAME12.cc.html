63G4int G4NucleiPropertiesTableAME12::GetIndex(G4int Z, G4int A)
64{
65 if(A>G4NucleiPropertiesTableAME12::MaxA) {
66 G4Exception("G4NucleiPropertiesTableAME12::GetIndex",
67 "PART201",
68 EventMustBeAborted,"Nucleon number larger than 293");
69 return -1;
70 } else if(A<1) {
71 G4Exception("G4NucleiPropertiesTableAME12::GetIndex",
72 "Illegal arguemntPART201",
73 EventMustBeAborted," Nucleon number is negative");
74 return -1;
75 } else if(Z>A) {
76 G4Exception("G4NucleiPropertiesTableAME12::GetIndex",
77 "PART201",
78 EventMustBeAborted, "Nucleon number smaller than Z");
79 return -1;
80 }
81
82 for (G4int i = shortTable[A-1]; i < shortTable[A] ; i++) {
83 if (indexArray[0][i] == Z ) return i;
84 }
85 return -1;
86}

103G4double G4NucleiPropertiesTableAME12::GetNuclearMass(G4int Z, G4int A)
104{
105 if (!isIntialized) {
106 // calculate electron mass in orbit with binding energy
107 isIntialized = true;
108 for (int iz=1; iz<ZMax; iz+=1){
109 electronMass[iz] = iz*electron_mass_c2
110 - ( 14.4381 * std::pow( G4double(iz) , 2.39 )) *eV
111 - ( 1.55468*1e-6 * std::pow( G4double(iz) , 5.35 ) ) *eV;
112 }
113 }
114
115 G4double nuclearMass = GetAtomicMass(Z,A) - electronMass[Z];
116
117 if (nuclearMass <0.0) nuclearMass = 0.0;
118
119 return nuclearMass;
120
121}
