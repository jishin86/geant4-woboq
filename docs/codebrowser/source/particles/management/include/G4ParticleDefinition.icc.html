106inline
107 G4int G4ParticleDefinition::GetQuarkContent(G4int flavor) const
108{
109 G4int content = 0;
110 if ((flavor>0) && (flavor<=NumberOfQuarkFlavor))
111 {
112 content = theQuarkContent[flavor-1];
113 }
114 else
115 {
116#ifdef G4VERBOSE
117 if (verboseLevel >0)
118 {
119 std::ostringstream message;
120 message << "Invalid Quark Flavor: " << flavor;
121 G4Exception("G4ParticleDefinition::GetQuarkContent()",
122 "InvalidFlavor", JustWarning, message);
123 }
124#endif
125 }
126 return content;
127}
