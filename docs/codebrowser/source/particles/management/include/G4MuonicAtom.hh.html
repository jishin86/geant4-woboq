36#ifndef G4MuonicAtom_h
37#define G4MuonicAtom_h 1
38
39#include "G4Ions.hh"
40
41class G4MuonicAtom : public G4Ions
42{
43 protected:
44 G4MuonicAtom(){};
45
46 public:
47 G4MuonicAtom(
48 const G4String& aName, G4double mass,
49 G4double width, G4double charge,
50 G4int iSpin, G4int iParity,
51 G4int iConjugation, G4int iIsospin,
52 G4int iIsospin3, G4int gParity,
53 const G4String& pType, G4int lepton,
54 G4int baryon, G4int encoding,
55 G4bool stable, G4double lifeTime,
56 G4DecayTable *decaytable, G4bool shortlived,
57 const G4String& subType, G4Ions const* baseion,
58 G4int anti_encoding =0,
59 G4double excitation = 0.0,
60 G4int isomer = 0,
61 G4double DIOLifeTime = -1.0,
62 G4double NCLifeTime = -1.0
63 );
64
65 virtual ~G4MuonicAtom();
66 G4MuonicAtom* MuonicAtomDefinition();
67 G4MuonicAtom* MuonicAtom();
68 G4Ions const* GetBaseIon() const;
69 G4double GetDIOLifeTime() const;
70 void SetDIOLifeTime(G4double lt);
71 G4double GetNCLifeTime() const;
72 void SetNCLifeTime(G4double lt);
73
74 private:
75 G4Ions const* baseIon;
76 G4double fDIOLifeTime;
77 G4double fNCLifeTime;
78};
79

99 G4double G4MuonicAtom::GetDIOLifeTime() const

101 return fDIOLifeTime;
102}
103
104inline
105 void G4MuonicAtom::SetDIOLifeTime(G4double lt)
106{
107 fDIOLifeTime = lt;
108}
109
110inline
111 G4double G4MuonicAtom::GetNCLifeTime() const
112{
113 return fNCLifeTime;
114}

119 fNCLifeTime = lt;
120}
