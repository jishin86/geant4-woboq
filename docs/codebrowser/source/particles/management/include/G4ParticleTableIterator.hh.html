36#ifndef G4ParticleTableIterator_h
37#define G4ParticleTableIterator_h 1
38
39#include <map>
40#include "G4ParticleDefinition.hh"
41
42template < class K, class V > class G4ParticleTableIterator
43{
44 public:
45 typedef std::map<K, V, std::less<K> > Map;
46 G4ParticleTableIterator( Map &adict):
47 it(adict.begin()),
48 mydict(&adict),
49 defined(false),
50 skipIons(true)
51 {}
52
53 G4bool operator++ ()
54 {
55 if(!defined) return false;
56 it++;
57 return it!=mydict->end() ? true : false;
58 }
59
60 G4bool operator()()
61 {
62 if(!defined) {
63 defined=true;
64 it=mydict->begin();
65 } else {
66 it++;
67 }
68 if(it==mydict->end()) return false;
69 if(skipIons){
70 while((static_cast<G4ParticleDefinition*>((*it).second))->IsGeneralIon()){ // Loop checking, 09.08.2015, K.Kurashige
71 it++;
72 if(it==mydict->end()) return false;
73 }
74 }
75 return true;
76 }
77
78 void reset (G4bool ifSkipIon = true) {defined=false; skipIons = ifSkipIon; }
79 K* key() const { return &((*it).first); }
80 V value() const { return (*it).second; }
81
82 private:
83 typename Map::iterator it;
84 Map * mydict;
85 G4bool defined;
86 G4bool skipIons;
87};
88
89#endif
