36#ifndef G4VShortLivedParticle_h
37#define G4VShortLivedParticle_h 1
38
39#include "G4ios.hh"
40#include "globals.hh"
41#include "G4ParticleDefinition.hh"
42
43class G4VShortLivedParticle : public G4ParticleDefinition
44{
45 // A virtual class for short lived particles.
46 // ShortLivedParticle particles will not be tracked by the TrackingManager
47 // So, G4VShortLivedParticle is not derived from G4ParticleWIthCuts
48 private:
49
50 const G4VShortLivedParticle & operator=(const G4VShortLivedParticle &right);
51
52 public:
53
54 G4VShortLivedParticle(const G4String& aName,
55 G4double mass,
56 G4double width,
57 G4double charge,
58 G4int iSpin,
59 G4int iParity,
60 G4int iConjugation,
61 G4int iIsospin,
62 G4int iIsospinZ,
63 G4int gParity,
64 const G4String& pType,
65 G4int lepton,
66 G4int baryon,
67 G4int encoding,
68 G4bool stable,
69 G4double lifetime,
70 G4DecayTable *decaytable);
71
72 virtual ~G4VShortLivedParticle();
73
74 G4bool operator==(const G4VShortLivedParticle &right) const;
75 G4bool operator!=(const G4VShortLivedParticle &right) const;
76
77};

79#endif
