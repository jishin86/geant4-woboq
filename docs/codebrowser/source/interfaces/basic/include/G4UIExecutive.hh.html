27// ====================================================================
28// G4UIExecutive.hh
29//
30// This class helps automatic instantiation of user session
31// according to your environment variable like G4UI_USE_XXX.
32//
33// Usage in main():
34//
35// ...
36// #include "G4UIExecutive.hh"
37//
38// int main(int argc, char** argv)
39// {
40// ...
41// G4UIExecutive* myapp = new G4UIExecutive(argc, argv);
42// if (session->IsGUI())
43// // Do any extra for a GUI session
44//
45// myapp-> SessionStart();
46// ...
47// delete myapp;
48// ...
49//
50// ====================================================================
51#ifndef G4UI_EXECUTIVE_HH
52#define G4UI_EXECUTIVE_HH
53
54#include "G4VUIshell.hh"
55#include <map>
56
57class G4UIsession;
58
59class G4UIExecutive {
60public:
61 G4UIExecutive(G4int argc, char** argv, const G4String& type = "");
62 ~G4UIExecutive();
63
64 G4UIsession* GetSession() const;
65
66 G4bool IsGUI() const;
67
68 void SetPrompt(const G4String& prompt);
69 void SetLsColor(TermColorIndex dirColor, TermColorIndex cmdColor);
70
71 void SessionStart();
72
73private:
74 enum SessionType { kNone, kQt, kXm, kWin32, kWt, kGag, kTcsh, kCsh };
75 SessionType selected;
76
77 G4UIsession* session;
78 G4VUIshell* shell;
79
80 G4bool isGUI;
81
82 std::map<G4String, G4String> sessionMap;
83
84 void SelectSessionByArg(const G4String& stype);
85 void SelectSessionByEnv();
86 void SelectSessionByFile(const G4String& appname);
87 void SelectSessionByBestGuess();
88};
89
90// ====================================================================
91inline G4UIsession* G4UIExecutive::GetSession() const
92{
93 return session;
94}
95
96inline G4bool G4UIExecutive::IsGUI() const
97{
98 return isGUI;
99}
100
101#endif
