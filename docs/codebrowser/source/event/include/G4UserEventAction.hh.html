31#ifndef G4UserEventAction_h
32#define G4UserEventAction_h 1
33
34class G4EventManager;
35class G4Event;
36
37// class description:
38//
39// This is the base class of one of the user's optional action classes.
40// The two methods BeginOfEventAction() and EndOfEventAction() are invoked
41// at the beginning and the end of one event processing. These methods are
42// invoked by G4EventManager.
43// Be aware that BeginOfEventAction() is invoked when a G4Event object is
44// sent to G4EventManager. Thus the primary vertexes/particles have already
45// been made by the primary generator. In case the user wants to do something
46// before generating primaries (i.e., store random number status), do it in
47// the G4VUserPrimaryGeneratorAction concrete class.
48//
49
50class G4UserEventAction
51{
52 public:
53 G4UserEventAction();
54 virtual ~G4UserEventAction();
55 virtual void SetEventManager(G4EventManager* value)
56 { fpEventManager = value; }
57 public: // with description
58 virtual void BeginOfEventAction(const G4Event* anEvent);
59 virtual void EndOfEventAction(const G4Event* anEvent);
60 // Two virtual method the user can override.
61 protected:
62 G4EventManager* fpEventManager;
63};
64
65#endif
