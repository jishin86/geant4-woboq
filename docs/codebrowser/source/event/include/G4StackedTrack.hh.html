33#ifndef G4StackedTrack_h
34#define G4StackedTrack_h 1
35
36class G4VTrajectory;
37class G4Track;
38
39// class description:
40//
41// This class is exclusively used by G4StackManager and G4TrackStack
42// classes for storing a G4Track object.
43
44class G4StackedTrack
45{
46public:
47 G4StackedTrack() : track(0), trajectory(0) {}
48 G4StackedTrack(G4Track* aTrack, G4VTrajectory* aTraj = 0) : track(aTrack), trajectory(aTraj) {}
49 ~G4StackedTrack() {}
50
51private:
52 G4Track* track;
53 G4VTrajectory* trajectory;
54
55public:
56 G4Track* GetTrack() const { return track; }
57 G4VTrajectory* GetTrajectory() const { return trajectory; }
58};

60#endif
