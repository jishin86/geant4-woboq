29// G4TrajectoryContainer
30//
31// Class description:
32//
33// This is a container of G4VTrajectory objects and the object of this
34// container will be associated to G4Event object.
35//
36
37// ********************************************************************
38#ifndef G4TrajectoryContainer_h
39#define G4TrajectoryContainer_h 1
40
41#include <vector>
42
43#include "globals.hh"
44#include "evtdefs.hh"
45#include "G4VTrajectory.hh"
46#include "G4Allocator.hh"
47
48typedef std::vector<G4VTrajectory*> TrajectoryVector;
49
50class G4TrajectoryContainer
51{
52 public:
53 G4TrajectoryContainer();
54 ~G4TrajectoryContainer();
55
56 private:
57 G4TrajectoryContainer(const G4TrajectoryContainer&);
58 G4TrajectoryContainer& operator=(const G4TrajectoryContainer&);
59
60 public:
61 inline void *operator new(size_t);
62 inline void operator delete(void* anEvent);
63
64 G4bool operator==(const G4TrajectoryContainer& right) const;
65 G4bool operator!=(const G4TrajectoryContainer& right) const;
66
67 public:
68 inline size_t size() const { return vect->size(); }
69 inline void push_back(G4VTrajectory* p) { vect->push_back(p); }
70 inline G4int entries() const { return size(); }
71 inline G4bool insert(G4VTrajectory* p) { push_back(p); return true; }
72 inline void clearAndDestroy()
73 {
74 for(size_t i=0;i<size();i++) delete (*vect)[i];
75 vect->clear();
76 }
77 inline G4VTrajectory* operator[](size_t n) { return (*vect)[n]; }
78 inline TrajectoryVector* GetVector() const { return vect; }
79
80 private:
81 TrajectoryVector* vect;
82};
83
84extern G4EVENT_DLL G4Allocator<G4TrajectoryContainer>*& aTrajectoryContainerAllocator();
85
86inline void* G4TrajectoryContainer::operator new(size_t)
87{
88 if (!aTrajectoryContainerAllocator())
89 aTrajectoryContainerAllocator() = new G4Allocator<G4TrajectoryContainer>;
90 return (void*)aTrajectoryContainerAllocator()->MallocSingle();
91}
92
93inline void G4TrajectoryContainer::operator delete(void* aTrajectoryContainer)
94{
95 aTrajectoryContainerAllocator()->FreeSingle((G4TrajectoryContainer*)aTrajectoryContainer);
96}
97
98#endif
