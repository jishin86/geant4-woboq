42//-------------------------------------------------------------
43// Documentation:
44// Stacking action used in the adjoint simulation. It is responsible to kill a primary forward particle before it is traked in the forwrad phase
45// if the last adjoint particle did not reach the adjoint surface. Was needed for the new design where the G4AdjointSimManager is no more an extension
46// of the G4RunManager. If the primary particles is not killed before being tracked in the sensitive geometry, the User Stacking action
47// can be used duiring the forward phase if specified by the method G4AdjointSimManager::UseUserStackingAction(Bool).
48//
49//
50//
51//
52//
53//
54//

56#define G4AdjointStackingAction_h 1
57
58#include "globals.hh"
59#include "G4UserStackingAction.hh"
60
61class G4Track;
62class G4ParticleDefinition;
63class G4AdjointTrackingAction;
64
65class G4AdjointStackingAction : public G4UserStackingAction
66{
67 public:
68 G4AdjointStackingAction(G4AdjointTrackingAction* anAction);
69 virtual ~G4AdjointStackingAction();
70
71 public:
72 virtual G4ClassificationOfNewTrack ClassifyNewTrack(const G4Track* aTrack);
73 virtual void NewStage();
74 virtual void PrepareNewEvent();
75 inline void SetUserFwdStackingAction(G4UserStackingAction* anAction){theFwdStackingAction = anAction;}
76 inline void SetUserAdjointStackingAction(G4UserStackingAction* anAction){theUserAdjointStackingAction = anAction;}
77 inline void SetKillTracks(G4bool aBool){kill_tracks =aBool;}
78 inline void SetAdjointMode(G4bool aBool){adjoint_mode=aBool;}
79
80
81
82 private:
83 G4UserStackingAction* theFwdStackingAction;
84 G4UserStackingAction* theUserAdjointStackingAction;
85 G4bool reclassification_stage,first_reclassification_stage,kill_tracks,adjoint_mode;
86 G4AdjointTrackingAction* theAdjointTrackingAction;
87};

89#endif
