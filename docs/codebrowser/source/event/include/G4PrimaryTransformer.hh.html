29#ifndef G4PromaryTransformer_h
30#define G4PromaryTransformer_h 1
31
32#include "G4TrackVector.hh"
33#include "G4ParticleTable.hh"
34#include "globals.hh"
35class G4Event;
36class G4PrimaryVertex;
37#include "G4PrimaryParticle.hh"
38#include "G4DynamicParticle.hh"
39
40// class description:
41//
42// This class is exclusively used by G4EventManager for the conversion
43// from G4PrimaryVertex/G4PrimaryParticle to G4DynamicParticle/G4Track.
44//
45
46class G4PrimaryTransformer
47{
48 public:
49 G4PrimaryTransformer();
50 virtual ~G4PrimaryTransformer();
51
52 G4TrackVector* GimmePrimaries(G4Event* anEvent, G4int trackIDCounter=0);
53 void CheckUnknown();
54
55 protected:
56 G4TrackVector TV;
57 G4ParticleTable* particleTable;
58 G4int verboseLevel;
59 G4int trackID;
60
61 G4ParticleDefinition* unknown;
62 G4bool unknownParticleDefined;
63 G4ParticleDefinition* opticalphoton;
64 G4bool opticalphotonDefined;
65
66 G4int nWarn;
67
68 public:
69 inline void SetVerboseLevel(G4int vl)
70 { verboseLevel = vl; };
71
72 public: //with description
73 void SetUnknnownParticleDefined(G4bool vl);
74 // By invoking this Set method, the user can alter the treatment of unknown
75 // particle. The ideal place to invoke this method is in the BeginOfRunAction.
76
77 inline G4bool GetUnknownParticleDefined() const
78 { return unknownParticleDefined; }
79
80 protected:
81 void GenerateTracks(G4PrimaryVertex* primaryVertex);
82 void GenerateSingleTrack(G4PrimaryParticle* primaryParticle,
83 G4double x0,G4double y0,G4double z0,G4double t0,G4double wv);
84 void SetDecayProducts(G4PrimaryParticle* mother,
85 G4DynamicParticle* motherDP);
86 G4bool CheckDynamicParticle(G4DynamicParticle*DP);
87
88 protected: //with description
89 // Following two virtual methods are provided to customize the use of PrimaryTransformer
90 // for particle types exotic to Geant4.
91
92 virtual G4ParticleDefinition* GetDefinition(G4PrimaryParticle*pp);
93 // Return appropriate G4ParticleDefinition w.r.t. the primary particle.
94 // If NULL is returned, the particle will not be treated as a track, but its daughters
95 // will be examined in case it has "pre-assigned decay products".
96
97 virtual G4bool IsGoodForTrack(G4ParticleDefinition*pd);
98 // Return true if a primary particle should be converted into a track.
99 // By default, all particles of non-shortlived and shortlived with valid decay
100 // tables are converted.
101};
102
103#endif
