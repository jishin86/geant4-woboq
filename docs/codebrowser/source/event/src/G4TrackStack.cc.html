29#include "G4TrackStack.hh"
30#include "G4SmartTrackStack.hh"
31#include "G4VTrajectory.hh"

33
34G4TrackStack::~G4TrackStack()
35{
36 clearAndDestroy();
37}
38
39void G4TrackStack::clearAndDestroy()
40{
41 for( iterator i = begin(); i != end(); i++ ) {
42 delete (*i).GetTrack();
43 delete (*i).GetTrajectory();
44 }
45 clear();
46}
47
48void G4TrackStack::TransferTo(G4TrackStack* aStack) {
49 for(iterator i = begin(); i != end(); i++) aStack->push_back(*i);
50 clear();
51}
52
53
54void G4TrackStack::TransferTo(G4SmartTrackStack * aStack)
55{
56 while(size()) { aStack->PushToStack(PopFromStack()); }
57}

60G4double G4TrackStack::getTotalEnergy(void) const
61{
62 G4double totalEnergy = 0.0;
63 for (const_iterator i = begin(); i != end(); i++) {
64 totalEnergy += (*i).GetTrack()->GetDynamicParticle()->GetTotalEnergy();
65 }
66 return totalEnergy;
67}
