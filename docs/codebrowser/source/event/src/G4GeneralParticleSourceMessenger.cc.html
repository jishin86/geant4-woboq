86G4GeneralParticleSourceMessenger* G4GeneralParticleSourceMessenger::GetInstance(G4GeneralParticleSource* psc)
87{
88 G4AutoLock l(&creationM);
89 if ( theInstance == 0 ) theInstance = new G4GeneralParticleSourceMessenger(psc);
90 return theInstance;
91}
92
93void G4GeneralParticleSourceMessenger::Destroy() {
94 G4AutoLock l(&creationM);
95 if ( theInstance != 0 ) {
96 delete theInstance;
97 theInstance = 0;
98 }
99}
