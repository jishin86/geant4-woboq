71const std::map<G4String,G4AttDef>* G4TrajectoryPoint::GetAttDefs() const
72{
73 G4bool isNew;
74 std::map<G4String,G4AttDef>* store
75 = G4AttDefStore::GetInstance("G4TrajectoryPoint",isNew);
76 if (isNew) {
77 G4String Pos("Pos");
78 (*store)[Pos] =
79 G4AttDef(Pos, "Position", "Physics","G4BestUnit","G4ThreeVector");
80 }
81 return store;
82}
83
84std::vector<G4AttValue>* G4TrajectoryPoint::CreateAttValues() const
85{
86 std::vector<G4AttValue>* values = new std::vector<G4AttValue>;
87
88 values->push_back(G4AttValue("Pos",G4BestUnit(fPosition,"Length"),""));
89
90#ifdef G4ATTDEBUG
91 G4cout << G4AttCheck(values,GetAttDefs());
92#endif
93
94 return values;
95}
