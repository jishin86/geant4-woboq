48/////////////////////////////////////////////////////////

50/////////////////////////////////////////////////////////
51void G4AdjointTrackingAction::PreUserTrackingAction(const G4Track* aTrack)
52{
53 G4String partType = aTrack->GetParticleDefinition()->GetParticleType();
54 if (partType.contains(G4String("adjoint"))){
55 is_adjoint_tracking_mode =true;
56 theAdjointSteppingAction->SetPrimWeight(aTrack->GetWeight());
57 }
58 else {
59 is_adjoint_tracking_mode =false;
60 if (theUserFwdTrackingAction)
61 theUserFwdTrackingAction->PreUserTrackingAction(aTrack);
62 }
63 theAdjointSteppingAction->SetAdjointTrackingMode(is_adjoint_tracking_mode);
64}
65/////////////////////////////////////////////////////////
66 void G4AdjointTrackingAction::PostUserTrackingAction(const G4Track* aTrack)
67{
68
69 //important to have it here !
70 last_weight = theAdjointSteppingAction->GetLastWeight();
71 last_ekin = theAdjointSteppingAction->GetLastEkin();
72
73
74 if(!is_adjoint_tracking_mode){
75 if (theUserFwdTrackingAction)
76 theUserFwdTrackingAction->PostUserTrackingAction(aTrack);
77 }
78 else if (theAdjointSteppingAction->GetDidAdjParticleReachTheExtSource()){
79 last_pos = theAdjointSteppingAction->GetLastPosition();
80 last_direction = theAdjointSteppingAction->GetLastMomentum();
81 last_direction /=last_direction.mag();
82 last_cos_th = last_direction.z();
83 G4ParticleDefinition* aPartDef= theAdjointSteppingAction->GetLastPartDef();
84 last_fwd_part_name= aPartDef->GetParticleName();
85 last_fwd_part_name.remove(0,4);
86 last_fwd_part_PDGEncoding=G4ParticleTable::GetParticleTable()
87 ->FindParticle(last_fwd_part_name)->GetPDGEncoding();
88 last_ekin = theAdjointSteppingAction->GetLastEkin();
89 last_ekin_nuc = last_ekin;
90 if (aPartDef->GetParticleType() == "adjoint_nucleus") {
91 G4double nb_nuc=double(aPartDef->GetBaryonNumber());
92 last_ekin_nuc /=nb_nuc;
93 }
94
95 last_fwd_part_index=-1;
96 size_t i=0;
97 while(i< pListOfPrimaryFwdParticles->size() && last_fwd_part_index<0) {
98 if ((*pListOfPrimaryFwdParticles)[i]->GetParticleName() == last_fwd_part_name)
99 last_fwd_part_index=i;
100 i++;
101 }
102 //Fill the vectors
103 last_pos_vec.push_back(last_pos);
104 last_direction_vec.push_back(last_direction);
105 last_ekin_vec.push_back(last_ekin);
106 last_ekin_nuc_vec.push_back(last_ekin_nuc);
107 last_cos_th_vec.push_back(last_cos_th);
108 last_weight_vec.push_back(last_weight);
109 //G4cout<<"Last weight "<<last_weight<<std::endl;
110 last_fwd_part_PDGEncoding_vec.push_back(last_fwd_part_PDGEncoding);
111 last_fwd_part_index_vec.push_back(last_fwd_part_index);
112 }
113 else {
114
115 }
116}
117 /////////////////////////////////////////////////////////
118 void G4AdjointTrackingAction::ClearEndOfAdjointTrackInfoVectors()
119 { last_pos_vec.clear();
120 last_direction_vec.clear();
121 last_ekin_vec.clear();
122 last_ekin_nuc_vec.clear();
123 last_cos_th_vec.clear();
124 last_weight_vec.clear();
125 last_fwd_part_PDGEncoding_vec.clear();
126 last_fwd_part_index_vec.clear();
127 }
