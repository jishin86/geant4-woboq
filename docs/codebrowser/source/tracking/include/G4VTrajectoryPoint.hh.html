30// G4VTrajectoryPoint.hh
31//
32// class description:
33// This class is the abstract base class which represents the point
34// which consists of a trajectory.
35// It includes information of a the point.
36//
37// ---------------------------------------------------------------
38
39#ifndef G4VTrajectoryPoint_h
40#define G4VTrajectoryPoint_h 1
41
42#include "globals.hh"
43#include <vector>
44#include <map>
45#include "G4ThreeVector.hh"
46
47class G4AttDef;
48class G4AttValue;
49
50class G4VTrajectoryPoint
51{
52 public: // with description
53
54 // Constructor/Destructor
55 G4VTrajectoryPoint();
56 virtual ~G4VTrajectoryPoint();
57
58 // Operators
59 G4bool operator==(const G4VTrajectoryPoint& right) const;
60
61 // Get/Set functions
62 virtual const G4ThreeVector GetPosition() const = 0;
63
64 // Get method for a vector of auxiliary points
65 virtual const std::vector<G4ThreeVector>* GetAuxiliaryPoints() const
66 { return 0; }
67 // If implemented by a derived class, returns a pointer to a list
68 // of auxiliary points, e.g., intermediate points used during the
69 // calculation of the step that can be used for drawing a smoother
70 // trajectory. The user must test the validity of this pointer.
71
72 // Get method for HEPRep style attribute definitions
73 virtual const std::map<G4String,G4AttDef>* GetAttDefs() const
74 { return 0; }
75 // If implemented by a derived class, returns a pointer to a map of
76 // attribute definitions for the attribute values below. The user
77 // must test the validity of this pointer. See G4Trajectory for an
78 // example of a concrete implementation of this method.
79
80 // Get method for HEPRep style attribute values
81 virtual std::vector<G4AttValue>* CreateAttValues() const
82 { return 0; }
83 // If implemented by a derived class, returns a pointer to a list
84 // of attribute values suitable, e.g., for picking. Each must
85 // refer to an attribute definition in the above map; its name is
86 // the key. The user must test the validity of this pointer (it
87 // must be non-zero and conform to the G4AttDefs, which may be
88 // checked with G4AttCheck) and delete the list after use. See
89 // G4Trajectory for an example of a concrete implementation of this
90 // method and G4VTrajectory::ShowTrajectory for an example of its
91 // use.
92
93};
94
95#endif
