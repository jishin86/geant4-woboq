95inline void G4ParticleChangeForTransport::Initialize(const G4Track& track)
96{
97 // use base class's method at first
98 InitializeStatusChange(track);
99// InitializeLocalEnergyDeposit(track);
100 InitializeSteppingControl(track);
101// InitializeTrueStepLength(track);
102// InitializeSecondaries(track);
103
104 // set Energy/Momentum etc. equal to those of the parent particle
105 const G4DynamicParticle* pParticle = track.GetDynamicParticle();
106// theEnergyChange = pParticle->GetKineticEnergy();
107// theMomentumChange = pParticle->GetMomentumDirection();
108 theVelocityChange = track.GetVelocity();
109 isVelocityChanged = false;
110 thePolarizationChange = pParticle->GetPolarization();
111// theProperTimeChange = pParticle->GetProperTime();
112
113 // set Position/Time etc. equal to those of the parent track
114 // thePositionChange = track.GetPosition();
115 // set TimeChange equal to local time of the parent track
116 theTimeChange = track.GetLocalTime();
117 // set initial Local/Global time of the parent track
118 theLocalTime0 = track.GetLocalTime();
119 theGlobalTime0 = track.GetGlobalTime();
120
121 // set touchable equal to the next touchable of the parent track
122 // not set as for now
123 //theTouchableChange = track.GetNextTouchable();
124
125// So almost nothing is initialized here.
126// theMomentumChange, theProperTimeChange, thePositionChange and theTimeChange
127// are set by G4Transportation::AlongStepDoIt;
128// the others are not needed.
129// Take care when implementing the PostStep related things!
130// (P. Urban)
131}
