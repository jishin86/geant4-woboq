150inline
151 void G4Step::CopyPostToPreStepPoint( )
152 {
153 //This method is called at the beggining of each step
154 *(fpPreStepPoint) = *(fpPostStepPoint);
155 fpPostStepPoint->SetStepStatus(fUndefined);
156
157 // store number of secondaries
158 nSecondaryByLastStep = fSecondary->size();
159}

182// Other member functions
183inline
184 void G4Step::InitializeStep( G4Track* aValue )
185 {
186 // Initialize G4Step attributes
187 fStepLength = 0.;
188 fTotalEnergyDeposit = 0.;
189 fNonIonizingEnergyDeposit = 0.;
190 fpTrack = aValue;
191 fpTrack->SetStepLength(0.);
192
193 nSecondaryByLastStep = 0;
194
195 // Initialize G4StepPoint attributes.
196 // To avoid the circular dependency between G4Track, G4Step
197 // and G4StepPoint, G4Step has to manage the copy actions.
198 fpPreStepPoint->SetPosition(fpTrack->GetPosition());
199 fpPreStepPoint->SetGlobalTime(fpTrack->GetGlobalTime());
200 fpPreStepPoint->SetLocalTime(fpTrack->GetLocalTime());
201 fpPreStepPoint->SetProperTime(fpTrack->GetProperTime());
202 fpPreStepPoint->SetMomentumDirection(fpTrack->GetMomentumDirection());
203 fpPreStepPoint->SetKineticEnergy(fpTrack->GetKineticEnergy());
204 fpPreStepPoint->SetTouchableHandle(fpTrack->GetTouchableHandle());
205 fpPreStepPoint->SetMaterial( fpTrack->GetTouchable()->GetVolume()->GetLogicalVolume()->GetMaterial());
206 fpPreStepPoint->SetMaterialCutsCouple( fpTrack->GetTouchable()->GetVolume()->GetLogicalVolume()->GetMaterialCutsCouple());
207 fpPreStepPoint->SetSensitiveDetector( fpTrack->GetTouchable()->GetVolume()->GetLogicalVolume()->GetSensitiveDetector());
208 fpPreStepPoint->SetPolarization(fpTrack->GetPolarization());
209 fpPreStepPoint->SetSafety(0.);
210 fpPreStepPoint->SetStepStatus(fUndefined);
211 fpPreStepPoint->SetProcessDefinedStep(0);
212 fpPreStepPoint->SetMass(fpTrack->GetDynamicParticle()->GetMass());
213 fpPreStepPoint->SetCharge(fpTrack->GetDynamicParticle()->GetCharge());
214 fpPreStepPoint->SetWeight(fpTrack->GetWeight());
215
216 // Set Velocity
217 // should be placed after SetMaterial for preStep point
218 fpPreStepPoint->SetVelocity(fpTrack->CalculateVelocity());
219
220 (*fpPostStepPoint) = (*fpPreStepPoint);
221 }

223inline
224 void G4Step::UpdateTrack( )
225 {
226 // To avoid the circular dependency between G4Track, G4Step
227 // and G4StepPoint, G4Step has to manage the update actions.
228 // position, time
229 fpTrack->SetPosition(fpPostStepPoint->GetPosition());
230 fpTrack->SetGlobalTime(fpPostStepPoint->GetGlobalTime());
231 fpTrack->SetLocalTime(fpPostStepPoint->GetLocalTime());
232 fpTrack->SetProperTime(fpPostStepPoint->GetProperTime());
233 // energy, momentum, polarization
234 fpTrack->SetMomentumDirection(fpPostStepPoint->GetMomentumDirection());
235 fpTrack->SetKineticEnergy(fpPostStepPoint->GetKineticEnergy());
236 fpTrack->SetPolarization(fpPostStepPoint->GetPolarization());
237 // mass charge
238 G4DynamicParticle* pParticle = (G4DynamicParticle*)(fpTrack->GetDynamicParticle());
239 pParticle->SetMass(fpPostStepPoint->GetMass());
240 pParticle->SetCharge(fpPostStepPoint->GetCharge());
241 // step length
242 fpTrack->SetStepLength(fStepLength);
243 // NextTouchable is updated
244 // (G4Track::Touchable points touchable of Pre-StepPoint)
245 fpTrack->SetNextTouchableHandle(fpPostStepPoint->GetTouchableHandle());
246 fpTrack->SetWeight(fpPostStepPoint->GetWeight());
247
248
249 // set velocity
250 fpTrack->SetVelocity(fpPostStepPoint->GetVelocity());
251}
