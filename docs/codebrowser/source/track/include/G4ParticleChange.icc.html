145inline
146 void G4ParticleChange::ProposeGlobalTime(G4double t)
147{
148 theTimeChange = (t-theGlobalTime0) + theLocalTime0;
149}

151inline
152 G4double G4ParticleChange::GetGlobalTime(G4double timeDelay) const
153{
154 // Convert the time delay to the global time.
155 return theGlobalTime0 + (theTimeChange-theLocalTime0) + timeDelay;
156}

208inline
209 G4ThreeVector G4ParticleChange::CalcMomentum(G4double energy,
210 G4ThreeVector direction,
211 G4double mass
212 ) const
213{
214 G4double tMomentum = std::sqrt(energy*energy + 2*energy*mass);
215 return direction*tMomentum;
216}
