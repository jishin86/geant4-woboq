42// Operators
43
44 inline void* G4Track::operator new(size_t)
45 {
46 if (!aTrackAllocator()) aTrackAllocator() = new G4Allocator<G4Track>;
47 return (void *) aTrackAllocator()->MallocSingle();
48 }
49 // Override "new" for "G4Allocator".
50
51 inline void G4Track::operator delete(void *aTrack)
52 { aTrackAllocator()->FreeSingle((G4Track *) aTrack); }
53 // Override "delete" for "G4Allocator".
54
55 inline G4bool G4Track::operator==( const G4Track& trk)
56 { return (this==&trk); }
57 // Define "==" operator because "G4TrackVector" uses
58 // "RWPtrOrderdVector" which requires this.
