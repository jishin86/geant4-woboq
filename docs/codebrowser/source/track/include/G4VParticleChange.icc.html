229inline void G4VParticleChange::InitializeSecondaries(const G4Track&)
230{
231 // clear secondaries
232 if (theNumberOfSecondaries>0) {
233#ifdef G4VERBOSE
234 if (verboseLevel>0) {
235 G4cerr << "G4VParticleChange::Initialize() Warning ";
236 G4cerr << "theListOfSecondaries is not empty " << G4endl;
237 G4cerr << "All objects in theListOfSecondaries are destroyed!" << G4endl;
238 }
239#endif
240 for (G4int index= 0; index<theNumberOfSecondaries; index++){
241 if ( (*theListOfSecondaries)[index] ){
242 delete (*theListOfSecondaries)[index] ;
243 }
244 }
245 }
246 theNumberOfSecondaries = 0;
247}
248
249//----------------------------------------------------------------
250// methods for handling secondaries
251//
252
253inline void G4VParticleChange::SetNumberOfSecondaries(G4int totSecondaries)
254{
255 // check if tracks still exist in theListOfSecondaries
256 if (theNumberOfSecondaries>0) {
257#ifdef G4VERBOSE
258 if (verboseLevel>0) {
259 G4cerr << "G4VParticleChange::SetNumberOfSecondaries() Warning ";
260 G4cerr << "theListOfSecondaries is not empty ";
261 }
262#endif
263 for (G4int index= 0; index<theNumberOfSecondaries; index++){
264 if ( (*theListOfSecondaries)[index] ){
265 delete (*theListOfSecondaries)[index] ;
266 }
267 }
268 }
269 theNumberOfSecondaries = 0;
270 theSizeOftheListOfSecondaries = totSecondaries;
271
272 // Initialize ListOfSecondaries
273 theListOfSecondaries->Initialize(totSecondaries);
274}

276inline void G4VParticleChange::Initialize(const G4Track& track)
277{
278 InitializeStatusChange(track);
279 InitializeLocalEnergyDeposit(track);
280 InitializeSteppingControl(track);
281 InitializeTrueStepLength(track);
282 InitializeSecondaries(track);
283 InitializeParentWeight(track);
284 InitializeParentGlobalTime(track);
285 InitializeStepInVolumeFlags(track);
286}
