39G4PVReplica::G4PVReplica( const G4String& pName,
40 G4LogicalVolume* pLogical,
41 G4VPhysicalVolume* pMother,
42 const EAxis pAxis,
43 const G4int nReplicas,
44 const G4double width,
45 const G4double offset )
46 : G4VPhysicalVolume(0, G4ThreeVector(), pName, pLogical, pMother),
47 fRegularVolsId(0)
48{
49
50 instanceID = subInstanceManager.CreateSubInstance();
51
52 G4MT_copyNo = -1;
53
54 if ((!pMother) || (!pMother->GetLogicalVolume()))
55 {
56 std::ostringstream message;
57 message << "NULL pointer specified as mother volume." << G4endl
58 << "The world volume cannot be sliced or parameterised !";
59 G4Exception("G4PVReplica::G4PVReplica()", "GeomVol0002",
60 FatalException, message);
61 return;
62 }
63 G4LogicalVolume* motherLogical = pMother->GetLogicalVolume();
64 if (pLogical == motherLogical)
65 {
66 G4Exception("G4PVReplica::G4PVReplica()", "GeomVol0002",
67 FatalException, "Cannot place a volume inside itself!");
68 return;
69 }
70 SetMotherLogical(motherLogical);
71 motherLogical->AddDaughter(this);
72 if (motherLogical->GetNoDaughters() != 1)
73 {
74 std::ostringstream message;
75 message << "Replica or parameterised volume must be the only daughter !"
76 << G4endl
77 << " Mother physical volume: " << pMother->GetName() << G4endl
78 << " Replicated volume: " << pName;
79 G4Exception("G4PVReplica::G4PVReplica()", "GeomVol0002",
80 FatalException, message);
81 return;
82 }
83 CheckAndSetParameters (pAxis, nReplicas, width, offset);
84}

86G4PVReplica::G4PVReplica( const G4String& pName,
87 G4LogicalVolume* pLogical,
88 G4LogicalVolume* pMotherLogical,
89 const EAxis pAxis,
90 const G4int nReplicas,
91 const G4double width,
92 const G4double offset )
93 : G4VPhysicalVolume(0,G4ThreeVector(),pName,pLogical,0), fRegularVolsId(0)
94{
95
96 instanceID = subInstanceManager.CreateSubInstance();
97 G4MT_copyNo = -1;
98
99 if (!pMotherLogical)
100 {
101 std::ostringstream message;
102 message << "NULL pointer specified as mother volume for "
103 << pName << ".";
104 G4Exception("G4PVReplica::G4PVReplica()", "GeomVol0002",
105 FatalException, message);
106 return;
107 }
108 if (pLogical == pMotherLogical)
109 {
110 G4Exception("G4PVReplica::G4PVReplica()", "GeomVol0002",
111 FatalException, "Cannot place a volume inside itself!");
112 return;
113 }
114 pMotherLogical->AddDaughter(this);
115 SetMotherLogical(pMotherLogical);
116 if (pMotherLogical->GetNoDaughters() != 1)
117 {
118 std::ostringstream message;
119 message << "Replica or parameterised volume must be the only daughter !"
120 << G4endl
121 << " Mother logical volume: " << pMotherLogical->GetName()
122 << G4endl
123 << " Replicated volume: " << pName;
124 G4Exception("G4PVReplica::G4PVReplica()", "GeomVol0002",
125 FatalException, message);
126 return;
127 }
128 CheckAndSetParameters (pAxis, nReplicas, width, offset);
129}

131void G4PVReplica::CheckAndSetParameters( const EAxis pAxis,
132 const G4int nReplicas,
133 const G4double width,
134 const G4double offset)
135{
136 if (nReplicas<1)
137 {
138 G4Exception("G4PVReplica::CheckAndSetParameters()", "GeomVol0002",
139 FatalException, "Illegal number of replicas.");
140 }
141 fnReplicas=nReplicas;
142 if (width<0)
143 {
144 G4Exception("G4PVReplica::CheckAndSetParameters()", "GeomVol0002",
145 FatalException, "Width must be positive.");
146 }
147 fwidth = width;
148 foffset = offset;
149 faxis = pAxis;
150
151 // Create rotation matrix for phi axis case & check axis is valid
152 //
153 G4RotationMatrix* pRMat=0;
154 switch (faxis)
155 {
156 case kPhi:
157 pRMat=new G4RotationMatrix();
158 if (!pRMat)
159 {
160 G4Exception("G4PVReplica::CheckAndSetParameters()", "GeomVol0003",
161 FatalException, "Rotation matrix allocation failed.");
162 }
163 SetRotation(pRMat);
164 break;
165 case kRho:
166 case kXAxis:
167 case kYAxis:
168 case kZAxis:
169 case kUndefined:
170 break;
171 default:
172 G4Exception("G4PVReplica::CheckAndSetParameters()", "GeomVol0002",
173 FatalException, "Unknown axis of replication.");
174 break;
175 }
176}

264// This method is similar to the constructor. It is used by each worker
265// thread to achieve the same effect as that of the master thread exept
266// to register the new created instance. This method is invoked explicitly.
267// It does not create a new G4PVReplica instance. It only assigns the value
268// for the fields encapsulated by the class G4ReplicaData.
269//
270void G4PVReplica::InitialiseWorker(G4PVReplica *pMasterObject)
271{
272
273 G4VPhysicalVolume::InitialiseWorker( pMasterObject, 0, G4ThreeVector());
274 subInstanceManager.SlaveCopySubInstanceArray();
275 G4MT_copyNo = -1;
276
277 // This call causes "self-assignment" of the input paramters
278 // Issue reported by DRD since TerminateWorker below can be called
279 // at the same time by another thread
280 // What we need here is the splic-class component of this funciton
281 // that is copied here
282 // CheckAndSetParameters (faxis, fnReplicas, fwidth, foffset);
283
284 // Create rotation matrix for phi axis case & check axis is valid
285 //
286 G4RotationMatrix* pRMat=0;
287 switch (faxis)
288 {
289 case kPhi:
290 pRMat=new G4RotationMatrix();
291 if (!pRMat)
292 {
293 G4Exception("G4PVReplica::InitialiseWorker(...)", "GeomVol0003",
294 FatalException, "Rotation matrix allocation failed.");
295 }
296 SetRotation(pRMat);
297 break;
298 case kRho:
299 case kXAxis:
300 case kYAxis:
301 case kZAxis:
302 case kUndefined:
303 break;
304 default:
305 G4Exception("G4PVReplica::InitialiseWorker(...)", "GeomVol0002",
306 FatalException, "Unknown axis of replication.");
307 break;
308 }
309}

311// This method is similar to the destructor. It is used by each worker
312// thread to achieve the partial effect as that of the master thread.
313// For G4PVReplica instances, it destories the rotation matrix.
314//
315void G4PVReplica::TerminateWorker(G4PVReplica* /*pMasterObject*/)
316{
317 if ( faxis==kPhi )
318 {
319 delete GetRotation();
320 }
