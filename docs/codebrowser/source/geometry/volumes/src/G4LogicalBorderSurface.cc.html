43G4LogicalBorderSurfaceTable *G4LogicalBorderSurface::theBorderSurfaceTable = 0;
44
45//
46// Constructor & destructor
47//
48
49G4LogicalBorderSurface::
50G4LogicalBorderSurface(const G4String& name,
51 G4VPhysicalVolume* vol1,
52 G4VPhysicalVolume* vol2,
53 G4SurfaceProperty* surfaceProperty)
54 : G4LogicalSurface(name, surfaceProperty),
55 Volume1(vol1), Volume2(vol2)
56{
57 if (!theBorderSurfaceTable)
58 {
59 theBorderSurfaceTable = new G4LogicalBorderSurfaceTable;
60 }
61
62 // Store in the table of Surfaces
63 //
64 theBorderSurfaceTable->push_back(this);
65}
66
67G4LogicalBorderSurface::
68G4LogicalBorderSurface(const G4LogicalBorderSurface& right)
69 : G4LogicalSurface(right.GetName(), right.GetSurfaceProperty())
70{
71 if (!theBorderSurfaceTable)
72 {
73 theBorderSurfaceTable = new G4LogicalBorderSurfaceTable;
74 }
75 SetTransitionRadiationSurface(right.GetTransitionRadiationSurface());
76 Volume1 = right.Volume1;
77 Volume2 = right.Volume2;
78 (*theBorderSurfaceTable) = (*right.theBorderSurfaceTable);
79}
80
81G4LogicalBorderSurface::~G4LogicalBorderSurface()
82{
83// delete theBorderSurfaceTable; theBorderSurfaceTable=0;
84}

90G4LogicalBorderSurface&
91G4LogicalBorderSurface::operator=(const G4LogicalBorderSurface &right)
92{
93 if (&right == this) return *this;
94
95 SetSurfaceProperty(right.GetSurfaceProperty());
96 SetName(right.GetName());
97 SetTransitionRadiationSurface(right.GetTransitionRadiationSurface());
98 Volume1 = right.Volume1;
99 Volume2 = right.Volume2;
100 (*theBorderSurfaceTable) = (*right.theBorderSurfaceTable);
101
102 return *this;
103}

121const G4LogicalBorderSurfaceTable* G4LogicalBorderSurface::GetSurfaceTable()
122{
123 if (!theBorderSurfaceTable)
124 {
125 theBorderSurfaceTable = new G4LogicalBorderSurfaceTable;
126 }
127 return theBorderSurfaceTable;
128}
129
130size_t G4LogicalBorderSurface::GetNumberOfBorderSurfaces()
131{
132 if (theBorderSurfaceTable)
133 {
134 return theBorderSurfaceTable->size();
135 }
136 return 0;
137}
138
139G4LogicalBorderSurface*
140G4LogicalBorderSurface::GetSurface(const G4VPhysicalVolume* vol1,
141 const G4VPhysicalVolume* vol2)
142{
143 if (theBorderSurfaceTable)
144 {
145 for (size_t i=0; i<theBorderSurfaceTable->size(); i++)
146 {
147 if( ((*theBorderSurfaceTable)[i]->GetVolume1() == vol1) &&
148 ((*theBorderSurfaceTable)[i]->GetVolume2() == vol2) )
149 return (*theBorderSurfaceTable)[i];
150 }
151 }
152 return 0;
153}

155// Dump info for known surfaces
156//
157void G4LogicalBorderSurface::DumpInfo()
158{
159 G4cout << "***** Surface Table : Nb of Surfaces = "
160 << GetNumberOfBorderSurfaces() << " *****" << G4endl;
161
162 if (theBorderSurfaceTable)
163 {
164 for (size_t i=0; i<theBorderSurfaceTable->size(); i++)
165 {
166 G4LogicalBorderSurface* pBorderSurface = (*theBorderSurfaceTable)[i];
167 G4cout << pBorderSurface->GetName() << " : " << G4endl
168 << " Border of volumes "
169 << pBorderSurface->GetVolume1()->GetName() << " and "
170 << pBorderSurface->GetVolume2()->GetName()
171 << G4endl;
172 }
173 }
174 G4cout << G4endl;
175}
176
177void G4LogicalBorderSurface::CleanSurfaceTable()
178{
179 if (theBorderSurfaceTable)
180 {
181 G4LogicalBorderSurfaceTable::iterator pos;
182 for(pos=theBorderSurfaceTable->begin();
183 pos!=theBorderSurfaceTable->end(); pos++)
184 {
185 if (*pos) { delete *pos; }
186 }
187 theBorderSurfaceTable->clear();
188 }
189 return;
190}
