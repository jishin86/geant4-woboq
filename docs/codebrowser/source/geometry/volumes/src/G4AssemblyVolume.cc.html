46// Default constructor
47//
48G4AssemblyVolume::G4AssemblyVolume()
49 : fAssemblyID( 0 )
50{
51 InstanceCountPlus();
52 SetAssemblyID( GetInstanceCount() );
53 SetImprintsCount( 0 );
54 G4AssemblyStore* aStore = G4AssemblyStore::GetInstance();
55 if (aStore->GetAssembly(fAssemblyID,false))
56 {
57 std::ostringstream message;
58 message << "The assembly has NOT been registered !" << G4endl
59 << " Assembly " << fAssemblyID
60 << " already existing in store !" << G4endl;
61 G4Exception("G4AssemblyVolume::G4AssemblyVolume()", "GeomVol1001",
62 JustWarning, message);
63 }
64 else
65 {
66 aStore->Register(this);
67 }
68}

97// Destructor
98//
99G4AssemblyVolume::~G4AssemblyVolume()
100{
101 unsigned int howmany = fTriplets.size();
102 if( howmany != 0 )
103 {
104 for( unsigned int i = 0; i < howmany; ++i )
105 {
106 G4RotationMatrix* pRotToClean = fTriplets[i].GetRotation();
107 if( pRotToClean != 0 )
108 {
109 delete pRotToClean;
110 }
111 }
112 }
113 fTriplets.clear();
114
115 // No need to delete physical volumes, as these are deleted
116 // by physical-volumes store.
117 //
118 fPVStore.clear();
119 InstanceCountMinus();
120 G4AssemblyStore::GetInstance()->DeRegister(this);
121}

123// Add and place the given volume according to the specified
124// translation and rotation.
125//
126// The rotation matrix passed in can be 0 = identity or an address even of an
127// object on the upper stack frame. During assembly imprint, it creates anyway
128// a new matrix and keeps track of it so it can delete it later at destruction
129// time.
130// This policy has been adopted since user has no control on the way the
131// rotations are combined.
132//
133void G4AssemblyVolume::AddPlacedVolume( G4LogicalVolume* pVolume,
134 G4ThreeVector& translation,
135 G4RotationMatrix* pRotation )
136{
137 G4RotationMatrix* toStore = new G4RotationMatrix;
138
139 if( pRotation != 0 ) { *toStore = *pRotation; }
140
141 G4AssemblyTriplet toAdd( pVolume, translation, toStore );
142 fTriplets.push_back( toAdd );
143}

145// Add and place the given volume according to the specified transformation
146//
147void G4AssemblyVolume::AddPlacedVolume( G4LogicalVolume* pVolume,
148 G4Transform3D& transformation )
149{
150 // Decompose transformation
151 G4Scale3D scale;
152 G4Rotate3D rotation;
153 G4Translate3D translation;
154 transformation.getDecomposition(scale, rotation, translation);
155
156 G4ThreeVector v = translation.getTranslation();
157 G4RotationMatrix* r = new G4RotationMatrix;
158 *r = rotation.getRotation();
159
160 G4bool isReflection = false;
161 if (scale(0,0)*scale(1,1)*scale(2,2) < 0.) { isReflection = true; }
162
163 G4AssemblyTriplet toAdd( pVolume, v, r, isReflection );
164 fTriplets.push_back( toAdd );
165}

206// Create an instance of an assembly volume inside of the specified
207// mother volume. This works analogically to making stamp imprints.
208// This method makes use of the Geant4 affine transformation class.
209// The algorithm is defined as follows:
210//
211// Having rotation matrix Rm and translation vector Tm to be applied
212// inside the mother and rotation matrix Ra and translation vector Ta
213// to be applied inside the assembly itself for each of the participating
214// volumes the resulting transformation is
215//
216// Tfinal = Ta * Tm
217//
218// where Ta and Tm are constructed as
219//
220// -1 -1
221// Ta = Ra * Ta and Tm = Rm * Tm
222//
223// which in words means that we create first the affine transformations
224// by inverse rotation matrices and translations for mother and assembly.
225// The resulting final transformation to be applied to each of the
226// participating volumes is their product.
227//
228// IMPORTANT NOTE!
229// The order of multiplication is reversed when comparing to CLHEP 3D
230// transformation matrix(G4Transform3D class).
231//
232// The rotation matrix passed in can be 0 = identity or an address even of an
233// object on the upper stack frame. During assembly imprint, it creates anyway
234// a new matrix and keeps track of it so it can delete it later at destruction
235// time.
236// This policy has been adopted since user has no control on the way the
237// rotations are combined.
238//
239// If the assembly volume contains assembly (a'), the function is called
240// recursively with composed transformation:
241//
242// Tanew = Ta * Ta'
243//
244void G4AssemblyVolume::MakeImprint( G4AssemblyVolume* pAssembly,
245 G4LogicalVolume* pMotherLV,
246 G4Transform3D& transformation,
247 G4int copyNumBase,
248 G4bool surfCheck )
249{
250 unsigned int numberOfDaughters;
251
252 if( copyNumBase == 0 )
253 {
254 numberOfDaughters = pMotherLV->GetNoDaughters();
255 }
256 else
257 {
258 numberOfDaughters = copyNumBase;
259 }
260
261 // We start from the first available index
262 //
263 numberOfDaughters++;
264
265 ImprintsCountPlus();
266
267 std::vector<G4AssemblyTriplet> triplets = pAssembly->fTriplets;
268
269 for( unsigned int i = 0; i < triplets.size(); i++ )
270 {
271 G4Transform3D Ta( *(triplets[i].GetRotation()),
272 triplets[i].GetTranslation() );
273 if ( triplets[i].IsReflection() ) { Ta = Ta * G4ReflectZ3D(); }
274
275 G4Transform3D Tfinal = transformation * Ta;
276
277 if ( triplets[i].GetVolume() )
278 {
279 // Generate the unique name for the next PV instance
280 // The name has format:
281 //
282 // av_WWW_impr_XXX_YYY_ZZZ
283 // where the fields mean:
284 // WWW - assembly volume instance number
285 // XXX - assembly volume imprint number
286 // YYY - the name of a log. volume we want to make a placement of
287 // ZZZ - the log. volume index inside the assembly volume
288 //
289 std::stringstream pvName;
290 pvName << "av_"
291 << GetAssemblyID()
292 << "_impr_"
293 << GetImprintsCount()
294 << "_"
295 << triplets[i].GetVolume()->GetName().c_str()
296 << "_pv_"
297 << i
298 << std::ends;
299
300 // Generate a new physical volume instance inside a mother
301 // (as we allow 3D transformation use G4ReflectionFactory to
302 // take into account eventual reflection)
303 //
304 G4PhysicalVolumesPair pvPlaced
305 = G4ReflectionFactory::Instance()->Place( Tfinal,
306 pvName.str().c_str(),
307 triplets[i].GetVolume(),
308 pMotherLV,
309 false,
310 numberOfDaughters + i,
311 surfCheck );
312
313 // Register the physical volume created by us so we can delete it later
314 //
315 fPVStore.push_back( pvPlaced.first );
316 if ( pvPlaced.second ) { fPVStore.push_back( pvPlaced.second ); }
317 }
318 else if ( triplets[i].GetAssembly() )
319 {
320 // Place volumes in this assembly with composed transformation
321 //
322 MakeImprint( triplets[i].GetAssembly(), pMotherLV,
323 Tfinal, i*100+copyNumBase, surfCheck );
324 }
325 else
326 {
327 G4Exception("G4AssemblyVolume::MakeImprint(..)",
328 "GeomVol0003", FatalException,
329 "Triplet has no volume and no assembly");
330 }
331 }

334void G4AssemblyVolume::MakeImprint( G4LogicalVolume* pMotherLV,
335 G4ThreeVector& translationInMother,
336 G4RotationMatrix* pRotationInMother,
337 G4int copyNumBase,
338 G4bool surfCheck )
339{
340 // If needed user can specify explicitely the base count from which to start
341 // off for the generation of phys. vol. copy numbers.
342 // The old behaviour is preserved when copyNumBase == 0, e.g. the generated
343 // copy numbers start from the count equal to current number of daughter
344 // volumes before an imprint is made
345
346 // Compose transformation
347 //
348 if( pRotationInMother == 0 )
349 {
350 // Make it by default an indentity matrix
351 //
352 pRotationInMother =
353 const_cast<G4RotationMatrix*>( &G4RotationMatrix::IDENTITY );
354 }
355
356 G4Transform3D transform( *pRotationInMother,
357 translationInMother );
358 MakeImprint(this, pMotherLV, transform, copyNumBase, surfCheck);
359}

372 MakeImprint(this, pMotherLV, transformation, copyNumBase, surfCheck);
