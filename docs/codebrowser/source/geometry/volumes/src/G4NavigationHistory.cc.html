35#include "G4ios.hh"
36#include "G4NavigationHistory.hh"
37
38G4Allocator<G4NavigationHistory>*& aNavigHistoryAllocator()
39{
40 G4ThreadLocalStatic G4Allocator<G4NavigationHistory>* _instance = nullptr;
41 return _instance;
42}
43
44G4NavigationHistory::G4NavigationHistory()
45 : fStackDepth(0)
46{
47 fNavHistory = G4NavigationHistoryPool::GetInstance()->GetLevels();
48 Clear();
49}
50
51G4NavigationHistory::G4NavigationHistory(const G4NavigationHistory &h)
52{
53 fNavHistory = G4NavigationHistoryPool::GetInstance()->GetLevels();
54 if( GetMaxDepth() != h.GetMaxDepth() )
55 {
56 fNavHistory->resize( h.GetMaxDepth() );
57 }
58
59 for ( G4int ilev=h.fStackDepth; ilev>=0; --ilev )
60 {
61 (*fNavHistory)[ilev] = (*h.fNavHistory)[ilev];
62 }
63 fStackDepth=h.fStackDepth;
64}
65
66G4NavigationHistory::~G4NavigationHistory()
67{
68 G4NavigationHistoryPool::GetInstance()->DeRegister(fNavHistory);
69}
70
71std::ostream&
72operator << (std::ostream& os, const G4NavigationHistory& nav)
73{
74 os << "History depth=" << nav.GetDepth() << G4endl;
75 for ( G4int i=0; i<=nav.GetDepth(); i++ )
76 {
77 os << "Level=["<<i<<"]: ";
78 if( nav.GetVolume(i) != 0 )
79 {
80 os << "Phys Name=["<< nav.GetVolume(i)->GetName()
81 << "] Type=[";
82 switch(nav.GetVolumeType(i))
83 {
84 case kNormal:
85 os << "N";
86 break;
87 case kReplica:
88 os << "R" << nav.GetReplicaNo(i);
89 break;
90 case kParameterised:
91 os << "P" << nav.GetReplicaNo(i);
92 break;
93 }
94 os << "]";
95 }
96 else
97 {
98 os << "Phys = <Null>";
99 }
100 os << G4endl;
101 }
102 return os;
103}
