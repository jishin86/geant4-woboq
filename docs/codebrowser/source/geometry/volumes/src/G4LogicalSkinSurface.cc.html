43G4LogicalSkinSurfaceTable *G4LogicalSkinSurface::theSkinSurfaceTable = 0;
44
45//
46// Constructors & destructor
47//
48
49G4LogicalSkinSurface::G4LogicalSkinSurface(const G4String& name,
50 G4LogicalVolume* logicalVolume,
51 G4SurfaceProperty* surfaceProperty)
52 : G4LogicalSurface(name, surfaceProperty),
53 LogVolume(logicalVolume)
54{
55 if (!theSkinSurfaceTable)
56 {
57 theSkinSurfaceTable = new G4LogicalSkinSurfaceTable;
58 }
59 // Store in the table of Surfaces
60 //
61 theSkinSurfaceTable->push_back(this);
62}
63
64G4LogicalSkinSurface::G4LogicalSkinSurface(const G4LogicalSkinSurface& right)
65 : G4LogicalSurface(right.GetName(), right.GetSurfaceProperty())
66{
67 if (!theSkinSurfaceTable)
68 {
69 theSkinSurfaceTable = new G4LogicalSkinSurfaceTable;
70 }
71 SetTransitionRadiationSurface(right.GetTransitionRadiationSurface());
72 LogVolume = right.LogVolume;
73 (*theSkinSurfaceTable) = (*right.theSkinSurfaceTable);
74}
75
76G4LogicalSkinSurface::~G4LogicalSkinSurface()
77{
78// delete theSkinSurfaceTable; theSkinSurfaceTable = 0;
79}

82// Operators

84
85G4LogicalSkinSurface&
86G4LogicalSkinSurface::operator=(const G4LogicalSkinSurface& right)
87{
88 if (&right == this) return *this;
89
90 SetSurfaceProperty(right.GetSurfaceProperty());
91 SetName(right.GetName());
92 SetTransitionRadiationSurface(right.GetTransitionRadiationSurface());
93 LogVolume = right.LogVolume;
94 (*theSkinSurfaceTable) = (*right.theSkinSurfaceTable);
95
96 return *this;
97}

99G4bool
100G4LogicalSkinSurface::operator==(const G4LogicalSkinSurface& right) const

102 return (this == (G4LogicalSkinSurface *) &right);

105G4bool
106G4LogicalSkinSurface::operator!=(const G4LogicalSkinSurface& right) const

115const G4LogicalSkinSurfaceTable* G4LogicalSkinSurface::GetSurfaceTable()
116{
117 if (!theSkinSurfaceTable)
118 {
119 theSkinSurfaceTable = new G4LogicalSkinSurfaceTable;
120 }
121 return theSkinSurfaceTable;
122}
123
124size_t G4LogicalSkinSurface::GetNumberOfSkinSurfaces()
125{
126 if (theSkinSurfaceTable)
127 {
128 return theSkinSurfaceTable->size();
129 }
130 return 0;
131}
132
133G4LogicalSkinSurface*
134G4LogicalSkinSurface::GetSurface(const G4LogicalVolume* vol)
135{
136 if (theSkinSurfaceTable)
137 {
138 for (size_t i=0; i<theSkinSurfaceTable->size(); i++)
139 {
140 if((*theSkinSurfaceTable)[i]->GetLogicalVolume() == vol)
141 return (*theSkinSurfaceTable)[i];
142 }
143 }
144 return 0;
145}

147// Dump info for known surfaces
148//
149void G4LogicalSkinSurface::DumpInfo()
150{
151 G4cout << "***** Skin Surface Table : Nb of Surfaces = "
152 << GetNumberOfSkinSurfaces() << " *****" << G4endl;
153
154 if (theSkinSurfaceTable)
155 {
156 for (size_t i=0; i<theSkinSurfaceTable->size(); i++)
157 {
158 G4LogicalSkinSurface* pSkinSurface = (*theSkinSurfaceTable)[i];
159 G4cout << pSkinSurface->GetName() << " : " << G4endl
160 << " Skin of logical volume "
161 << pSkinSurface->GetLogicalVolume()->GetName()
162 << G4endl;
163 }
164 }
165 G4cout << G4endl;
166}
167
168void G4LogicalSkinSurface::CleanSurfaceTable()
169{
170 if (theSkinSurfaceTable)
171 {
172 G4LogicalSkinSurfaceTable::iterator pos;
173 for(pos=theSkinSurfaceTable->begin();
174 pos!=theSkinSurfaceTable->end(); pos++)
175 {
176 if (*pos) { delete *pos; }
177 }
178 theSkinSurfaceTable->clear();
179 }
180 return;
181}
