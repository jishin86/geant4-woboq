64// There is no provision in case this class is subclassed.
65// If it is subclassed, this will fail and may not give errors!
66//
67inline
68void* G4NavigationLevel::operator new(size_t)
69{
70 if (!aNavigationLevelAllocator())
71 {
72 aNavigationLevelAllocator() = new G4Allocator<G4NavigationLevel>;
73 }
74 return (void *) aNavigationLevelAllocator()->MallocSingle();
75}

89}
