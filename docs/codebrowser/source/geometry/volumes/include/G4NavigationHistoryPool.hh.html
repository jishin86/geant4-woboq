28// class G4NavigationHistoryPool
29//
30// Class description:
31//
32// Thread-local pool for navigation history levels collections being
33// allocated by G4NavigationHistory. Allows for reuse of the vectors
34// allocated according to lifetime of G4NavigationHistory objects.
35
36// History:
37// 07.05.14 G.Cosmo Initial version
38// --------------------------------------------------------------------
39#ifndef G4NAVIGATIONHISTORYPOOL_HH
40#define G4NAVIGATIONHISTORYPOOL_HH
41
42#include <vector>
43
44#include "G4NavigationLevel.hh"
45
46class G4NavigationHistoryPool
47{
48 public: // with description
49
50 static G4NavigationHistoryPool* GetInstance();
51 // Return unique instance of G4NavigationHistoryPool.
52
53 inline std::vector<G4NavigationLevel> * GetNewLevels();
54 // Return the pointer to a new collection of levels being allocated.
55
56 inline std::vector<G4NavigationLevel> * GetLevels();
57 // Return the pointer of the first available collection of levels
58 // If none are available (i.e. empty Free vector) allocate collection.
59
60 inline void DeRegister(std::vector<G4NavigationLevel> * pLevels);
61 // Deactivate levels collection in pool.
62
63 void Clean();
64 // Delete all levels stored in the pool.
65
66 void Print() const;
67 // Print number of entries.
68
69 ~G4NavigationHistoryPool();
70 // Destructor: takes care to delete allocated levels.
71
72 private:
73
74 G4NavigationHistoryPool();
75 // Default constructor.
76
77 inline void Register(std::vector<G4NavigationLevel> * pLevels);
78 // Register levels collection to pool and activate it.
79
80 void Reset();
81 // Set internal vectors content to zero.
82
83 private:
84
85 static G4ThreadLocal G4NavigationHistoryPool* fgInstance;
86
87 std::vector<std::vector<G4NavigationLevel> *> fPool;
88 std::vector<std::vector<G4NavigationLevel> *> fFree;
89};
90
91// ***************************************************************************
92// Register levels collection to pool (add and/or activate)
93// ***************************************************************************
94//
95inline void G4NavigationHistoryPool::
96Register(std::vector<G4NavigationLevel> * pLevels)
97{
98 fPool.push_back(pLevels);
99}
100
101// ***************************************************************************
102// Deactivate levels collection in pool
103// ***************************************************************************
104//
105inline void G4NavigationHistoryPool::
106DeRegister(std::vector<G4NavigationLevel> * pLevels)
107{
108 fFree.push_back(pLevels);
109}
110
111// ***************************************************************************
112// Return the pointer of a new collection of levels allocated
113// ***************************************************************************
114//
115inline std::vector<G4NavigationLevel> * G4NavigationHistoryPool::GetNewLevels()
116{
117 std::vector<G4NavigationLevel> * aLevelVec =
118 new std::vector<G4NavigationLevel>(kHistoryMax);
119 Register(aLevelVec);
120
121 return aLevelVec;
122}
123
124// ***************************************************************************
125// Return the pointer of the first available collection of levels
126// If none are available (i.e. non active) allocate collection
127// ***************************************************************************
128//
129inline std::vector<G4NavigationLevel> * G4NavigationHistoryPool::GetLevels()
130{
131 std::vector<G4NavigationLevel> * levels = 0;
132
133 if (fFree.size() !=0)
134 {
135 levels = fFree.back();
136 fFree.pop_back();
137 }
138 else
139 {
140 levels = GetNewLevels();
141 }
142
143 return levels;
144}
145
146#endif
