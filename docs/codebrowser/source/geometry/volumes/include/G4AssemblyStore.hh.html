27// class G4AssemblyStore
28//
29// Class description:
30//
31// Container for all assemblies, with functionality derived from
32// std::vector<T>. The class is a `singleton', in that only
33// one can exist, and access is provided via the static method
34// G4AssemblyStore::GetInstance().
35//
36// All assemblies should be registered with G4AssemblyStore, and removed on
37// their destruction. The underlying container initially has a capacity of 20.
38//
39// If much additional functionality is added, should consider containment
40// instead of inheritance for std::vector<T>
41//
42// Member data:
43//
44// static G4AssemblyStore*
45// - Pointer to the single G4AssemblyStore
46
47// History:
48// 9.10.18 G.Cosmo Initial version
49// --------------------------------------------------------------------
50#ifndef G4ASSEMBLYSTORE_HH
51#define G4ASSEMBLYSTORE_HH
52
53#include <vector>
54#include "G4Types.hh"
55#include "G4String.hh"
56#include "G4VStoreNotifier.hh"
57
58class G4AssemblyVolume;
59
60class G4AssemblyStore : public std::vector<G4AssemblyVolume*>
61{
62 public: // with description
63
64 static void Register(G4AssemblyVolume* pAssembly);
65 // Add the assembly to the collection.
66 static void DeRegister(G4AssemblyVolume* pAssembly);
67 // Remove the assembly from the collection.
68 static G4AssemblyStore* GetInstance();
69 // Get a ptr to the unique G4AssemblyStore, creating it if necessary.
70 static void SetNotifier(G4VStoreNotifier* pNotifier);
71 // Assign a notifier for allocation/deallocation of assemblies.
72 static void Clean();
73 // Delete all assemblies from the store.
74
75 G4AssemblyVolume* GetAssembly(unsigned int Id, G4bool verbose=true) const;
76 // Returns an assembly through its Id number specification.
77
78 protected:
79
80 G4AssemblyStore();
81 // Protected singleton constructor.
82 virtual ~G4AssemblyStore();
83 // Destructor: takes care to delete allocated assemblies.
84
85 private:
86
87 static G4AssemblyStore* fgInstance;
88 static G4ThreadLocal G4VStoreNotifier* fgNotifier;
89 static G4ThreadLocal G4bool locked;
90};
91
92#endif
