33inline
34G4AssemblyTriplet::G4AssemblyTriplet()
35 : fVolume( 0 ), fRotation( 0 ), fAssembly(0), fIsReflection(false)
36{
37 G4ThreeVector v(0.,0.,0.);
38 fTranslation = v;
39}
40
41inline
42G4AssemblyTriplet::G4AssemblyTriplet( G4LogicalVolume* pVolume,
43 G4ThreeVector& translation,
44 G4RotationMatrix* pRotation,
45 G4bool isReflection )
46 : fVolume( pVolume ), fTranslation( translation ), fRotation( pRotation ),
47 fAssembly( 0 ), fIsReflection(isReflection)
48{
49}
50
51inline
52G4AssemblyTriplet::G4AssemblyTriplet( G4AssemblyVolume* pAssembly,
53 G4ThreeVector& translation,
54 G4RotationMatrix* pRotation,
55 G4bool isReflection )
56 : fVolume( 0 ), fTranslation( translation ), fRotation( pRotation ),
57 fAssembly( pAssembly ), fIsReflection(isReflection)
58{
59}
60
61inline
62G4AssemblyTriplet::G4AssemblyTriplet( const G4AssemblyTriplet& scopy )
63{
64 fVolume = scopy.GetVolume();
65 fRotation = scopy.GetRotation();
66 fTranslation = scopy.GetTranslation();
67 fAssembly = scopy.GetAssembly();
68 fIsReflection = scopy.IsReflection();
69}
70
71inline
72G4AssemblyTriplet::~G4AssemblyTriplet()
73{
74}
75
76inline
77G4LogicalVolume* G4AssemblyTriplet::GetVolume() const
78{
79 return fVolume;
80}
81
82inline
83void G4AssemblyTriplet::SetVolume( G4LogicalVolume* pVolume )
84{
85 if ( fAssembly )
86 {
87 G4Exception("G4AssemblyTriplet::SetVolume()",
88 "GeomVol1001", JustWarning,
89 "There is an assembly already set, it will be ignored.");
90 }
91 fVolume = pVolume;
92 fAssembly = 0;
93}
94
95inline
96G4AssemblyVolume* G4AssemblyTriplet::GetAssembly() const
97{
98 return fAssembly;
99}
100
101inline
102void G4AssemblyTriplet::SetAssembly( G4AssemblyVolume* pAssembly )
103{
104 if ( fVolume )
105 {
106 G4Exception("G4AssemblyTriplet::SetAssembly()",
107 "GeomVol1001", JustWarning,
108 "There is a volume already set, it will be ignored.");
109 }
110 fAssembly = pAssembly;
111 fVolume = 0;
112}
113
114inline
115G4ThreeVector G4AssemblyTriplet::GetTranslation() const
116{
117 return fTranslation;
118}
119
120inline
121void G4AssemblyTriplet::SetTranslation( G4ThreeVector& translation )

123 fTranslation = translation;
124}
125
126inline
127G4RotationMatrix* G4AssemblyTriplet::GetRotation() const
128{
129 return fRotation;
130}
131
132inline
133void G4AssemblyTriplet::SetRotation( G4RotationMatrix* pRotation )
134{
135 fRotation = pRotation;
136}
137
138inline
139G4bool G4AssemblyTriplet::IsReflection() const
140{
141 return fIsReflection;
142}
143
144inline
145G4AssemblyTriplet&
146G4AssemblyTriplet::operator=( const G4AssemblyTriplet& scopy )
147{
148 if( this != &scopy )
149 {
150 fVolume = scopy.GetVolume();
151 fRotation = scopy.GetRotation();
152 fTranslation = scopy.GetTranslation();
153 fAssembly = scopy.GetAssembly();
154 fIsReflection = scopy.IsReflection();
155 }
156
157 return *this;
158}
