32extern G4GEOM_DLL G4Allocator<G4NavigationLevelRep>*& aNavigLevelRepAllocator();
33
34// Constructors
35//--------------
36
37inline
38G4NavigationLevelRep::G4NavigationLevelRep( G4VPhysicalVolume* pPhysVol,
39 const G4AffineTransform& afTransform,
40 EVolume volTp,
41 G4int repNo )
42 : sTransform(afTransform),
43 sPhysicalVolumePtr(pPhysVol),
44 sReplicaNo(repNo),
45 sVolumeType(volTp),
46 fCountRef(1)
47{
48}
49
50inline
51G4NavigationLevelRep::G4NavigationLevelRep()
52 : sTransform(),
53 sPhysicalVolumePtr(0),
54 sReplicaNo(-1),
55 sVolumeType(kReplica),
56 fCountRef(1)
57{
58}
59
60inline
61G4NavigationLevelRep::G4NavigationLevelRep( G4VPhysicalVolume* pPhysVol,
62 const G4AffineTransform& levelAbove,
63 const G4AffineTransform& relativeCurrent,
64 EVolume volTp,
65 G4int repNo )
66 : sPhysicalVolumePtr(pPhysVol),
67 sReplicaNo(repNo),
68 sVolumeType(volTp),
69 fCountRef(1)
70{
71 sTransform.InverseProduct( levelAbove, relativeCurrent );
72}
73
74inline
75G4NavigationLevelRep::G4NavigationLevelRep( G4NavigationLevelRep& right )
76 : sTransform(right.sTransform),
77 sPhysicalVolumePtr(right.sPhysicalVolumePtr),
78 sReplicaNo(right.sReplicaNo),
79 sVolumeType(right.sVolumeType),
80 fCountRef(1)
81{
82}
83
84// Destructor
85//--------------
86
87inline
88G4NavigationLevelRep::~G4NavigationLevelRep()
89{
90#ifdef DEBUG_NAVIG_LEVEL
91 if(fCountRef>0)
92 {
93 G4Exception("G4NavigationLevelRep::~G4NavigationLevelRep()",
94 "GeomVol0003", FatalException,
95 "Deletion of data-level object with positive reference count.");
96 }
97#endif
98}

100// Operators
101// --------------
102
103inline
104G4NavigationLevelRep&
105G4NavigationLevelRep::operator=( const G4NavigationLevelRep &right )
106{
107 if ( &right != this )
108 {
109 sTransform = right.sTransform;
110 sPhysicalVolumePtr = right.sPhysicalVolumePtr;
111 sVolumeType = right.sVolumeType;
112 sReplicaNo = right.sReplicaNo;
113 fCountRef = right.fCountRef;
114 }
115 return *this;
116}

154inline
155void G4NavigationLevelRep::AddAReference()
156{
157 ++fCountRef;
158}
159
160inline
161G4bool G4NavigationLevelRep::RemoveAReference()
162{
163 return( --fCountRef <= 0 );
164}
165
166// There is no provision that this class is subclassed.
167// If it is subclassed & new data members are added then the
168// following "new" & "delete" will fail and give errors.
169//
170inline
171void* G4NavigationLevelRep::operator new(size_t)
172{
173 if (!aNavigLevelRepAllocator())
174 {
175 aNavigLevelRepAllocator() = new G4Allocator<G4NavigationLevelRep>;
176 }
177 return (void *) aNavigLevelRepAllocator()->MallocSingle();
178}
179
180inline
181void G4NavigationLevelRep::operator delete(void *aLevelRep)
182{
183 aNavigLevelRepAllocator()->FreeSingle((G4NavigationLevelRep *) aLevelRep);
184}
