28// ----------------------------------------------------------------------
29// Class G4GeometryCellStep
30//
31// Class description:
32//
33// This class serves to address the "cell" a track previously
34// touched and a "cell" a track is currently in. It is used
35// for scoring and importance sampling in the "mass" geometry as well
36// as in a "parallel" geometry.
37// The "cell" information is available with the GetPreGeometryCell and
38// the GetPostGeometryCell functions.
39// The GetCrossBoundary finction returns true in case the step
40// crosses a boundary in the geometry this G4GeometryCellStep
41// refers to.
42
43// Author: Michael Dressel (Michael.Dressel@cern.ch)
44// ----------------------------------------------------------------------
45#ifndef G4GeometryCellStep_hh
46#define G4GeometryCellStep_hh G4GeometryCellStep_hh
47
48#include "G4GeometryCell.hh"
49
50class G4GeometryCellStep
51{
52public: // with description
53 G4GeometryCellStep(const G4GeometryCell &preCell,
54 const G4GeometryCell &postCell);
55 // initialise pre and post G4GeometryCell
56
57 ~G4GeometryCellStep();
58
59 const G4GeometryCell &GetPreGeometryCell() const;
60 // addressing the "cell" the track previously touched
61
62 const G4GeometryCell &GetPostGeometryCell() const;
63 // addressing the current "cell"
64
65 G4bool GetCrossBoundary() const;
66 // returns true if step crosses boundary of the geometry it refers to
67
68 // the following functions are used by the scoring and importance
69 // system to set the cell information.
70 void SetPreGeometryCell(const G4GeometryCell &preCell);
71
72 void SetPostGeometryCell(const G4GeometryCell &postCell);
73
74 void SetCrossBoundary(G4bool b);
75
76private:
77 G4GeometryCell fPreGeometryCell;
78 G4GeometryCell fPostGeometryCell;
79 G4bool fCrossBoundary;
80
81};
82
83
84
85inline void G4GeometryCellStep::SetPreGeometryCell(const G4GeometryCell &preCell)
86{
87 fPreGeometryCell = preCell;
88}
89
90inline void G4GeometryCellStep::SetPostGeometryCell(const G4GeometryCell &postCell)
91{
92 fPostGeometryCell = postCell;
93}
94
95inline void G4GeometryCellStep::SetCrossBoundary(G4bool b)
96{
97 fCrossBoundary = b;
98}
99
100inline const G4GeometryCell &G4GeometryCellStep::GetPreGeometryCell() const
101{
102 return fPreGeometryCell;
103}
104
105inline const G4GeometryCell &G4GeometryCellStep::GetPostGeometryCell() const
106{
107 return fPostGeometryCell;
108}
109
110inline G4bool G4GeometryCellStep::GetCrossBoundary() const
111{
112 return fCrossBoundary;
113}
114
115
116#endif
