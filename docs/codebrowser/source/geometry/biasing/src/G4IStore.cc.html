104}

106void G4IStore::AddImportanceGeometryCell(G4double importance,
107 const G4GeometryCell &gCell)
108{
109 if (importance < 0 ) {
110 Error("AddImportanceGeometryCell() - Invalid importance value given.");
111 }
112 if (!IsInWorld(gCell.GetPhysicalVolume()) ) {
113 Error("AddImportanceGeometryCell() - Physical volume not found!");
114 }
115 SetInternalIterator(gCell);
116 if (fCurrentIterator!=fGeometryCelli.end()) {
117 Error("AddImportanceGeometryCell() - Region already existing!");
118 }
119 fGeometryCelli[gCell] = importance;
120}

124 G4int aRepNum)

126 AddImportanceGeometryCell(importance,
127 G4GeometryCell(aVolume, aRepNum));
128}

152G4double G4IStore::GetImportance(const G4VPhysicalVolume &aVolume,
153 G4int aRepNum) const
154{
155#ifdef G4MULTITHREADED
156 G4MUTEXLOCK(&G4IStore::IStoreMutex);
157#endif
158 SetInternalIterator(G4GeometryCell(aVolume, aRepNum));
159 G4GeometryCellImportance::const_iterator gCellIterator = fCurrentIterator;
160 if (gCellIterator==fGeometryCelli.end()) {
161 Error("GetImportance() - Region does not exist!");
162 return 0.;
163 }
164 G4double importance_value = (*fCurrentIterator).second;
165#ifdef G4MULTITHREADED
166 G4MUTEXUNLOCK(&G4IStore::IStoreMutex);
167#endif
168 return importance_value;
169 // return (*fCurrentIterator).second;
170}

173G4double G4IStore::GetImportance(const G4GeometryCell &gCell) const
174{
175#ifdef G4MULTITHREADED
176 G4MUTEXLOCK(&G4IStore::IStoreMutex);
177#endif
178 SetInternalIterator(gCell);
179 G4GeometryCellImportance::const_iterator gCellIterator = fCurrentIterator;
180 if (gCellIterator==fGeometryCelli.end()) {
181 std::ostringstream err_mess;
182 err_mess << "GetImportance() - Region does not exist!" << G4endl
183 << "Geometry cell, " << gCell
184 << ", not found in: " << fGeometryCelli << ".";
185 Error(err_mess.str());
186 return 0.;
187 }
188 G4double importance_value = (*fCurrentIterator).second;
189#ifdef G4MULTITHREADED
190 G4MUTEXUNLOCK(&G4IStore::IStoreMutex);
191#endif
192 return importance_value;
193 // return (*fCurrentIterator).second;
194}

196G4bool G4IStore::IsKnown(const G4GeometryCell &gCell) const {
197#ifdef G4MULTITHREADED
198 G4MUTEXLOCK(&G4IStore::IStoreMutex);
199#endif
200 G4bool inWorldKnown(IsInWorld(gCell.GetPhysicalVolume()));
201
202 if ( inWorldKnown ) {
203 SetInternalIterator(gCell);
204 inWorldKnown = (fCurrentIterator!=fGeometryCelli.end());
205 }
206#ifdef G4MULTITHREADED
207 G4MUTEXUNLOCK(&G4IStore::IStoreMutex);
208#endif
209 return inWorldKnown;
210}

212G4bool G4IStore::IsInWorld(const G4VPhysicalVolume &aVolume) const
213{
214 G4bool isIn(true);
215 // G4cout << "G4IStore:: aVolume: " << aVolume.GetName() << G4endl;
216 // G4cout << "G4IStore:: fWorld: " << fWorldVolume->GetName() << G4endl;
217 if (!(aVolume == *fWorldVolume)) {
218 isIn = fWorldVolume->GetLogicalVolume()->IsAncestor(&aVolume);
219 }
220 return isIn;
221}
