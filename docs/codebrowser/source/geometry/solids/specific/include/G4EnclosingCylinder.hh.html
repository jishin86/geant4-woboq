32// G4EnclosingCylinder
33//
34// Class description:
35//
36// Definition of a utility class for quickly deciding if a point
37// is clearly outside a polyhedra or polycone or deciding if
38// a trajectory is clearly going to miss those shapes.
39
40// Author:
41// David C. Williams (davidw@scipp.ucsc.edu)
42// --------------------------------------------------------------------
43#ifndef G4EnclosingCylinder_hh
44#define G4EnclosingCylinder_hh
45
46#include "G4Types.hh"
47#include "geomdefs.hh"
48#include "G4ThreeVector.hh"
49
50class G4ReduciblePolygon;
51
52class G4EnclosingCylinder
53{
54 public: // with description
55
56 G4EnclosingCylinder( const G4ReduciblePolygon *rz,
57 G4bool phiIsOpen,
58 G4double startPhi, G4double totalPhi );
59 ~G4EnclosingCylinder();
60
61 G4bool MustBeOutside( const G4ThreeVector &p ) const;
62 // Decide very rapidly if the point is outside the cylinder.
63 // If one is not certain, return false.
64
65 G4bool ShouldMiss( const G4ThreeVector &p, const G4ThreeVector &v ) const;
66 // Decide very rapidly if the trajectory is going to miss the cylinder.
67 // If one is not sure, return false.
68
69 public: // without description
70
71 G4EnclosingCylinder(__void__&);
72 // Fake default constructor for usage restricted to direct object
73 // persistency for clients requiring preallocation of memory for
74 // persistifiable objects.
75
76 protected:
77
78 G4double radius; // radius of our cylinder
79 G4double zLo, zHi; // z extent
80
81 G4bool phiIsOpen; // true if there is a phi segment
82 G4double startPhi, // for isPhiOpen==true, starting of phi segment
83 totalPhi; // for isPhiOpen==true, size of phi segment
84
85 G4double rx1, ry1,
86 dx1, dy1;
87 G4double rx2, ry2,
88 dx2, dy2;
89
90 G4bool concave; // true, if x/y cross section is concave
91
92};
93
94#endif
