119inline G4ThreeVector G4TwistTubsFlatSide::
120SurfacePoint(G4double phi , G4double rho , G4bool isGlobal )
121{
122 G4ThreeVector SurfPoint (rho*std::cos(phi) , rho*std::sin(phi) , 0);
123
124 if (isGlobal) { return (fRot * SurfPoint + fTrans); }
125 return SurfPoint;
126}
127
128inline
129G4double G4TwistTubsFlatSide::GetBoundaryMin(G4double)
130{
131 G4ThreeVector dphimin = GetCorner(sC0Max1Min);
132 return std::atan2( dphimin.y(), dphimin.x() );
133}
134
135inline
136G4double G4TwistTubsFlatSide::GetBoundaryMax(G4double)
137{
138 G4ThreeVector dphimax = GetCorner(sC0Max1Max);
139 return std::atan2( dphimax.y(), dphimax.x() );
140}
