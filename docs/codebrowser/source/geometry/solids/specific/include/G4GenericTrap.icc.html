52inline
53G4TwoVector G4GenericTrap::GetVertex(G4int index) const
54{
55 if ( index<0 || index >= G4int(fVertices.size()) )
56 {
57 G4Exception ("G4GenericTrap::GetVertex()", "GeomSolids0003",
58 FatalException, "Index outside range.");
59 return G4TwoVector();
60 }
61 return fVertices[index];
62}

74inline
75G4double G4GenericTrap::GetTwistAngle(G4int index) const
76{
77 if ( (index<0) || (index >= G4int(fVertices.size())) )
78 {
79 G4Exception ("G4GenericTrap::GetTwistAngle()", "GeomSolids0003",
80 FatalException, "Index outside range.");
81 return 0.;
82 }
83 return fTwist[index];
84}

96inline
97void G4GenericTrap::SetTwistAngle(G4int index, G4double twist)
98{
99 if ( (index<0) || (index >= G4int(fVertices.size())) )
100 {
101 G4Exception ("G4GenericTrap::SetTwistAngle()", "GeomSolids0003",
102 FatalException, "Index outside range.");
103 }
104 else
105 {
106 fTwist[index]=twist;
107 }
108}
