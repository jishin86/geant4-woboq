147inline
148G4ThreeVector G4TwistTubsSide::ProjectAtPXPZ(const G4ThreeVector &p,
149 G4bool isglobal) const
150{
151 // Get Rho at p.z() on Hyperbolic Surface.
152 G4ThreeVector tmpp;
153 if (isglobal) {
154 tmpp = fRot.inverse()*p - fTrans;
155 } else {
156 tmpp = p;
157 }
158 G4ThreeVector xx(p.x(), p.x() * fKappa * p.z(), p.z());
159 if (isglobal) { return (fRot * xx + fTrans); }
160 return xx;
161}
162
163inline
164G4ThreeVector
165G4TwistTubsSide::SurfacePoint(G4double x, G4double z, G4bool isGlobal)
166{
167 G4ThreeVector SurfPoint( x , x * fKappa * z , z ) ;
168
169 if (isGlobal) { return (fRot * SurfPoint + fTrans); }
170 return SurfPoint;
171}
172
173inline
174G4double G4TwistTubsSide::GetBoundaryMin(G4double)
175{
176 return fAxisMin[0] ; // inner radius at z = 0
177}
178
179inline
180G4double G4TwistTubsSide::GetBoundaryMax(G4double)
181{
182 return fAxisMax[0] ; // outer radius at z = 0
183}
184
185inline
186G4double G4TwistTubsSide::GetSurfaceArea()
187{
188 // approximation only
189 return ( fAxisMax[0] - fAxisMin[0] ) * ( fAxisMax[1] - fAxisMin[1] ) ;
190}
