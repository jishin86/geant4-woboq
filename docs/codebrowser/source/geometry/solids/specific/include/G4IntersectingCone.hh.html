33// G4IntersectingCone
34//
35// Class description:
36//
37// Utility class which calculates the intersection
38// of an arbitrary line with a fixed cone
39
40// Author:
41// David C. Williams (davidw@scipp.ucsc.edu)
42// --------------------------------------------------------------------
43#ifndef G4IntersectingCone_hh
44#define G4IntersectingCone_hh
45
46#include "G4Types.hh"
47#include "geomdefs.hh"
48#include "G4ThreeVector.hh"
49
50class G4IntersectingCone
51{
52 public:
53
54 G4IntersectingCone( const G4double r[2], const G4double z[2] );
55 virtual ~G4IntersectingCone();
56
57 G4int LineHitsCone( const G4ThreeVector &p, const G4ThreeVector &v,
58 G4double *s1, G4double *s2 );
59
60 G4bool HitOn( const G4double r, const G4double z );
61
62 inline G4double RLo() const { return rLo; }
63 inline G4double RHi() const { return rHi; }
64 inline G4double ZLo() const { return zLo; }
65 inline G4double ZHi() const { return zHi; }
66
67 public: // without description
68
69 G4IntersectingCone(__void__&);
70 // Fake default constructor for usage restricted to direct object
71 // persistency for clients requiring preallocation of memory for
72 // persistifiable objects.
73
74
75 protected:
76
77 G4double zLo, zHi, // Z bounds of side
78 rLo, rHi; // R bounds of side
79
80 G4bool type1; // True if cone is type 1
81 // (std::fabs(z1-z2)>std::fabs(r1-r2))
82 G4double A, B; // Cone radius parameter:
83 // type 1: r = A + B*z
84 // type 2: z = A + B*r
85
86 G4int LineHitsCone1( const G4ThreeVector &p, const G4ThreeVector &v,
87 G4double *s1, G4double *s2 );
88 G4int LineHitsCone2( const G4ThreeVector &p, const G4ThreeVector &v,
89 G4double *s1, G4double *s2 );
90};
91
92#endif
