30// Class G4TessellatedSolid
31//
32// Class description:
33//
34// G4TessellatedSolid is a special Geant4 solid defined by a number of
35// facets (UVFacet). It is important that the supplied facets shall form a
36// fully enclose space which is the solid.
37// At the moment only two types of facet can be used for the construction of
38// a G4TessellatedSolid, i.e. the G4TriangularFacet and G4QuadrangularFacet.
39//
40// How to contruct a G4TessellatedSolid:
41//
42// First declare a tessellated solid:
43//
44// G4TessellatedSolid* solidTarget = new G4TessellatedSolid("Solid_name");
45//
46// Define the facets which form the solid
47//
48// G4double targetSiz = 10*cm ;
49// G4TriangularFacet *facet1 = new
50// G4TriangularFacet (G4ThreeVector(-targetSize,-targetSize, 0.0),
51// G4ThreeVector(+targetSize,-targetSize, 0.0),
52// G4ThreeVector( 0.0, 0.0,+targetSize),
53// ABSOLUTE);
54// G4TriangularFacet *facet2 = new
55// G4TriangularFacet (G4ThreeVector(+targetSize,-targetSize, 0.0),
56// G4ThreeVector(+targetSize,+targetSize, 0.0),
57// G4ThreeVector( 0.0, 0.0,+targetSize),
58// ABSOLUTE);
59// G4TriangularFacet *facet3 = new
60// G4TriangularFacet (G4ThreeVector(+targetSize,+targetSize, 0.0),
61// G4ThreeVector(-targetSize,+targetSize, 0.0),
62// G4ThreeVector( 0.0, 0.0,+targetSize),
63// ABSOLUTE);
64// G4TriangularFacet *facet4 = new
65// G4TriangularFacet (G4ThreeVector(-targetSize,+targetSize, 0.0),
66// G4ThreeVector(-targetSize,-targetSize, 0.0),
67// G4ThreeVector( 0.0, 0.0,+targetSize),
68// ABSOLUTE);
69// G4QuadrangularFacet *facet5 = new
70// G4QuadrangularFacet (G4ThreeVector(-targetSize,-targetSize, 0.0),
71// G4ThreeVector(-targetSize,+targetSize, 0.0),
72// G4ThreeVector(+targetSize,+targetSize, 0.0),
73// G4ThreeVector(+targetSize,-targetSize, 0.0),
74// ABSOLUTE);
75//
76// Then add the facets to the solid:
77//
78// solidTarget->AddFacet((UVFacet*) facet1);
79// solidTarget->AddFacet((UVFacet*) facet2);
80// solidTarget->AddFacet((UVFacet*) facet3);
81// solidTarget->AddFacet((UVFacet*) facet4);
82// solidTarget->AddFacet((UVFacet*) facet5);
83//
84// Finally declare the solid is complete:
85//
86// solidTarget->SetSolidClosed(true);
87

119struct G4VertexInfo
120{
121 G4int id;
122 G4double mag2;
123};

127class G4VertexComparator
128{
129public:
130 G4bool operator() (const G4VertexInfo &l, const G4VertexInfo &r) const
131 {
132 return l.mag2 == r.mag2 ? l.id < r.id : l.mag2 < r.mag2;
133 }
134};

325
326inline G4bool G4TessellatedSolid::OutsideOfExtent(const G4ThreeVector &p,
327 G4double tolerance) const
328{
329 return ( p.x() < fMinExtent.x() - tolerance
330 || p.x() > fMaxExtent.x() + tolerance
331 || p.y() < fMinExtent.y() - tolerance
332 || p.y() > fMaxExtent.y() + tolerance
333 || p.z() < fMinExtent.z() - tolerance
334 || p.z() > fMaxExtent.z() + tolerance);
335}
