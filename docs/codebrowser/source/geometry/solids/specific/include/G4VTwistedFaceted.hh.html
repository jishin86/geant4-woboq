200 class LastState // last Inside result
201 {
202 public:
203 LastState()
204 {
205 p.set(kInfinity,kInfinity,kInfinity); inside = kOutside;
206 }
207 ~LastState(){}
208 LastState(const LastState& r) : p(r.p), inside(r.inside){}
209 LastState& operator=(const LastState& r)
210 {
211 if (this == &r) { return *this; }
212 p = r.p; inside = r.inside;
213 return *this;
214 }
215 public:
216 G4ThreeVector p;
217 EInside inside;
218 };

220 class LastVector // last SurfaceNormal result
221 {
222 public:
223 LastVector()
224 {
225 p.set(kInfinity,kInfinity,kInfinity);
226 vec.set(kInfinity,kInfinity,kInfinity);
227 surface = new G4VTwistSurface*[1];
228 }
229 ~LastVector()
230 {
231 delete [] surface;
232 }
233 LastVector(const LastVector& r) : p(r.p), vec(r.vec)
234 {
235 surface = new G4VTwistSurface*[1];
236 surface[0] = r.surface[0];
237 }
238 LastVector& operator=(const LastVector& r)
239 {
240 if (&r == this) { return *this; }
241 p = r.p; vec = r.vec;
242 delete [] surface; surface = new G4VTwistSurface*[1];
243 surface[0] = r.surface[0];
244 return *this;
245 }
246 public:
247 G4ThreeVector p;
248 G4ThreeVector vec;
249 G4VTwistSurface **surface;
250 };

308inline
309G4double G4VTwistedFaceted::GetCubicVolume()
310{
311 if(fCubicVolume != 0.) ;
312 else fCubicVolume = 2 * fDz
313 * ( ( fDx1 + fDx2 ) * fDy1 + ( fDx3 + fDx4 ) * fDy2 );
314 return fCubicVolume;
315}
316
317inline
318G4double G4VTwistedFaceted::GetSurfaceArea()
319{
320 if(fSurfaceArea != 0.) ;
321 else fSurfaceArea = G4VSolid::GetSurfaceArea();
322 return fSurfaceArea;
323}
324
325inline
326G4double G4VTwistedFaceted::GetValueA(G4double phi) const
327{
328 return ( fDx4 + fDx2 + ( fDx4 - fDx2 ) * ( 2 * phi ) / fPhiTwist ) ;
329}
330
331inline
332G4double G4VTwistedFaceted::GetValueD(G4double phi) const
333{
334 return ( fDx3 + fDx1 + ( fDx3 - fDx1 ) * ( 2 * phi ) / fPhiTwist ) ;
335}
336
337inline
338G4double G4VTwistedFaceted::GetValueB(G4double phi) const
339{
340 return ( fDy2 + fDy1 + ( fDy2 - fDy1 ) * ( 2 * phi ) / fPhiTwist ) ;
341}
342
343inline
344G4double G4VTwistedFaceted::Xcoef(G4double u, G4double phi, G4double ftg) const
345{
346 return GetValueA(phi)/2. + (GetValueD(phi)-GetValueA(phi))/4.
347 - u*( ( GetValueD(phi)-GetValueA(phi) ) / ( 2 * GetValueB(phi) ) - ftg );
348}
