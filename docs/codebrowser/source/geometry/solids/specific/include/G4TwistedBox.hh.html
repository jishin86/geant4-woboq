33// G4TwistedBox
34//
35// Class description:
36//
37// A G4TwistedBox is a twisted cuboid of given half lengths pDx,pDy,pDz
38// and twist angle pPhiTwist.
39// The Box is centred on the origin with sides parallel to the x/y/z axes.
40//
41// Member Data:
42//
43// pDx Half-length along x axis
44// pDy Half-length along y asis
45// pDz Half-length along z axis
46// pPhiTwist Twist angle
47
48// Author:
49//
50// 27-Oct-2004 - O.Link (Oliver.Link@cern.ch)
51//
52// --------------------------------------------------------------------
53#ifndef __G4TWISTEDBOX__
54#define __G4TWISTEDBOX__
55
56#include "G4VTwistedFaceted.hh"
57
58class G4TwistedBox : public G4VTwistedFaceted
59{
60 public: // with description
61
62 G4TwistedBox(const G4String& pName,
63 G4double pPhiTwist,
64 G4double pDx,
65 G4double pDy,
66 G4double pDz );
67
68 virtual ~G4TwistedBox();
69
70 // accessors
71
72 inline G4double GetXHalfLength() const { return GetDx1() ; }
73 inline G4double GetYHalfLength() const { return GetDy1() ; }
74 inline G4double GetZHalfLength() const { return GetDz() ; }
75 inline G4double GetPhiTwist() const { return GetTwistAngle() ; }
76
77 G4GeometryType GetEntityType() const;
78
79 G4VSolid* Clone() const;
80
81 std::ostream& StreamInfo(std::ostream& os) const;
82
83 public: // without description
84
85 G4TwistedBox(__void__&);
86 // Fake default constructor for usage restricted to direct object
87 // persistency for clients requiring preallocation of memory for
88 // persistifiable objects.
89
90 G4TwistedBox(const G4TwistedBox& rhs);
91 G4TwistedBox& operator=(const G4TwistedBox& rhs);
92 // Copy constructor and assignment operator.
93};
94
95#endif
