43// ExactZOrder
44//
45// Decide precisely whether a trajectory passes to the left, right, or exactly
46// passes through the z position of a vertex point in our face.
47//
48// Result is only determined within an arbitrary (positive) factor.
49// > 0 to the right
50// < 0 to the left
51// = 0 exactly on top of
52// In 99.9999% of the cases, a trivial calculation is used. In difficult
53// cases, a precise, compliant calculation is relied on.
54//
55inline
56G4double G4PolyPhiFace::ExactZOrder( G4double z,
57 G4double qx, G4double qy, G4double qz,
58 const G4ThreeVector &v,
59 G4double normSign,
60 const G4PolyPhiFaceVertex *vert ) const
61{
62 G4double answer = vert->z - z;
63 if (std::fabs(answer) < kCarTolerance)
64 {
65 G4ThreeVector qa( qx - vert->x + radial.x(),
66 qy - vert->y + radial.y(), qz - vert->z ),
67 qb( qx - vert->x, qy - vert->y, qz - vert->z );
68 G4ThreeVector qacb = qa.cross(qb);
69
70 answer = normSign*qacb.dot(v)*(normal.y()*radial.x()-normal.x()*radial.y());
71 }
72
73 return answer;
74}
