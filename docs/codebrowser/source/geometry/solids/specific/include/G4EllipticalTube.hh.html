32//
33// Class description:
34//
35// Declaration of a CSG volume representing a tube with elliptical
36// cross section (geant3 solid 'ELTU'):
37//
38// G4EllipticalTube( const G4String& name,
39// G4double Dx,
40// G4double Dy,
41// G4double Dz )

79 EInside Inside( const G4ThreeVector& p ) const;

81 G4ThreeVector SurfaceNormal( const G4ThreeVector& p ) const;

96 G4GeometryType GetEntityType() const;

98 G4VSolid* Clone() const;
99
100 std::ostream& StreamInfo(std::ostream& os) const;

102 G4double GetCubicVolume();

105 G4ThreeVector GetPointOnSurface() const;

146
147 private:
148
149 G4double halfTolerance;
150

153 G4double fDz; // half length in Z

155 G4double fCubicVolume; // volume
156 G4double fSurfaceArea; // surface area
157

167 G4double fScratch; // half length of scratching segment squared
168
169 mutable G4bool fRebuildPolyhedron;
170 mutable G4Polyhedron* fpPolyhedron;
171};
