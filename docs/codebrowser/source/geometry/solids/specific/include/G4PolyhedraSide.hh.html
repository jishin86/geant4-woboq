33// G4PolyhedraSide
34//
35// Class description:
36//
37// Class implementing a face that represents one segmented side
38// of a polyhedra:
39//
40// G4PolyhedraSide( const G4PolyhedraSideRZ *prevRZ,
41// const G4PolyhedraSideRZ *tail,
42// const G4PolyhedraSideRZ *head,
43// const G4PolyhedraSideRZ *nextRZ,
44// G4int numSide,
45// G4double phiStart, G4double phiTotal,
46// G4bool phiIsOpen, G4bool isAllBehind=false )
47//
48// Values for r1,z1 and r2,z2 should be specified in clockwise
49// order in (r,z).
50
51// Author:
52// David C. Williams (davidw@scipp.ucsc.edu)
53// --------------------------------------------------------------------
54

72// The class G4PhSideData is introduced to encapsulate the
73// fields of the class G4PolyhedraSide that may not be read-only.
74//
75class G4PhSideData
76{
77 public:
78 void initialize()
79 {
80 fPhi.first = G4ThreeVector(0,0,0);
81 fPhi.second= 0.0;
82 }
83
84 std::pair<G4ThreeVector, G4double> fPhi; // Cached value for phi
85
86};

88// The type G4PhSideManager is introduced to encapsulate the methods used
89// by both the master thread and worker threads to allocate memory space
90// for the fields encapsulated by the class G4PhSideData.
91//
92typedef G4GeomSplitter<G4PhSideData> G4PhSideManager;

163 // A couple internal data structures

165 struct sG4PolyhedraSideVec; // Secret recipe for allowing
166 friend struct sG4PolyhedraSideVec; // protected nested structures
167
168 typedef struct sG4PolyhedraSideEdge
169 {
170 G4ThreeVector normal; // Unit normal to this edge
171 G4ThreeVector corner[2]; // The two corners of this phi edge
172 G4ThreeVector cornNorm[2]; // The normals of these corners
173 } G4PolyhedraSideEdge;
174
175 typedef struct sG4PolyhedraSideVec
176 {
177 G4ThreeVector normal, // Normal (point out of the shape)
178 center, // Point in center of side
179 surfPhi, // Unit vector on surface pointing along phi
180 surfRZ; // Unit vector on surface pointing along R/Z
181 G4PolyhedraSideEdge *edges[2]; // The phi boundary edges to this side
182 // [0]=low phi [1]=high phi
183 G4ThreeVector edgeNorm[2]; // RZ edge normals [i] at {r[i],z[i]}
184 } G4PolyhedraSideVec;
185
186 protected:
187
188 G4bool IntersectSidePlane( const G4ThreeVector &p, const G4ThreeVector &v,
189 const G4PolyhedraSideVec& vec,
190 G4double normSign,
191 G4double surfTolerance,
192 G4double &distance,
193 G4double &distFromSurface );
194
195 G4int LineHitsSegments( const G4ThreeVector &p,
196 const G4ThreeVector &v,
197 G4int *i1, G4int *i2 );
198
199 G4int ClosestPhiSegment( G4double phi );
200
201 G4int PhiSegment( G4double phi );
202
203 G4double GetPhi( const G4ThreeVector& p );
204
205 G4double DistanceToOneSide( const G4ThreeVector &p,
206 const G4PolyhedraSideVec &vec,
207 G4double *normDist );
208
209 G4double DistanceAway( const G4ThreeVector &p,
210 const G4PolyhedraSideVec &vec,
211 G4double *normDist );
212
213 void CopyStuff( const G4PolyhedraSide &source );
214
215 protected:
216
217 G4int numSide; // Number sides
218 G4double r[2], z[2]; // r, z parameters, in specified order
219 G4double startPhi, // Start phi (0 to 2pi), if phiIsOpen
220 deltaPhi, // Delta phi (0 to 2pi), if phiIsOpen
221 endPhi; // End phi (>startPhi), if phiIsOpen
222 G4bool phiIsOpen; // True if there is a phi slice
223 G4bool allBehind; // True if the entire solid is "behind" this face
224
225 G4IntersectingCone *cone; // Our intersecting cone
226
227 G4PolyhedraSideVec *vecs; // Vector set for each facet of our face
228 G4PolyhedraSideEdge *edges; // The edges belong to vecs
229 G4double lenRZ, // RZ length of each side
230 lenPhi[2]; // Phi dimensions of each side
231 G4double edgeNorm; // Normal in RZ/Phi space to each side
232
233 private:
234

237
238 G4int instanceID;
239 // This field is used as instance ID.
240 G4GEOM_DLL static G4PhSideManager subInstanceManager;
241 // This field helps to use the class G4PhSideManager introduced above.
242};
