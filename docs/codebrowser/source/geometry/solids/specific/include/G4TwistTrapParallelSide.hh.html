158inline
159G4double G4TwistTrapParallelSide::GetValueB(G4double phi)
160{
161 return ( fDy2plus1 + fDy2minus1 * ( 2 * phi ) / fPhiTwist ) ;
162}
163
164inline
165G4double G4TwistTrapParallelSide::Xcoef(G4double phi)
166{
167 return GetValueB(phi)/2. ;
168}
169
170inline G4ThreeVector
171G4TwistTrapParallelSide::
172SurfacePoint( G4double phi, G4double u, G4bool isGlobal )
173{
174 // function to calculate a point on the surface, given by parameters phi,u
175
176 G4ThreeVector SurfPoint ( u*std::cos(phi) - Xcoef(phi)*std::sin(phi)
177 + fdeltaX*phi/fPhiTwist,
178 u*std::sin(phi) + Xcoef(phi)*std::cos(phi)
179 + fdeltaY*phi/fPhiTwist,
180 2*fDz*phi/fPhiTwist );
181 if (isGlobal) { return (fRot * SurfPoint + fTrans); }
182 return SurfPoint;
183}
184
185
186inline
187G4double G4TwistTrapParallelSide::GetBoundaryMin(G4double phi)
188{
189 return -(fPhiTwist*(fDx2 + fDx4 - fDy2plus1*fTAlph)
190 + 2*fDx4minus2*phi - 2*fDy2minus1*fTAlph*phi)/(2.*fPhiTwist) ;
191}
192
193inline
194G4double G4TwistTrapParallelSide::GetBoundaryMax(G4double phi)
195{
196 return (fDx2 + fDx4 + fDy2plus1*fTAlph)/ 2.
197 + ((fDx4minus2 + fDy2minus1*fTAlph)*phi)/fPhiTwist ;
198}
199
200inline
201G4double G4TwistTrapParallelSide::GetSurfaceArea()
202{
203 return 2*fDx4plus2*fDz ;
204}
205
206inline
207G4ThreeVector G4TwistTrapParallelSide::NormAng( G4double phi, G4double u )
208{
209 // function to calculate the norm at a given point on the surface
210 // replace a1-d1
211
212 G4ThreeVector nvec(-2*fDz*std::sin(phi) ,
213 2*fDz*std::cos(phi) ,
214 -(fDy2minus1 + fPhiTwist*u + fdeltaY*std::cos(phi)
215 -fdeltaX*std::sin(phi))) ;
216 return nvec.unit();
217}
