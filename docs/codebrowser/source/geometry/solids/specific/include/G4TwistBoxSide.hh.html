62 G4double pDy2, // half y length at +pDz

154//========================================================
155// inline functions
156//========================================================
157
158inline
159G4double G4TwistBoxSide::GetValueA(G4double phi)
160{
161 return ( fDx4plus2 + fDx4minus2 * ( 2 * phi ) / fPhiTwist ) ;
162}
163
164
165inline
166G4double G4TwistBoxSide::GetValueB(G4double phi)
167{
168 return ( fDy2plus1 + fDy2minus1 * ( 2 * phi ) / fPhiTwist ) ;
169}
170
171inline
172G4double G4TwistBoxSide::Xcoef(G4double u, G4double phi)
173{
174
175 return GetValueA(phi)/2. + u*fTAlph ;
176
177}
178
179inline G4ThreeVector
180G4TwistBoxSide::SurfacePoint( G4double phi, G4double u, G4bool isGlobal )
181{
182 // function to calculate a point on the surface, given by parameters phi,u
183
184 G4ThreeVector SurfPoint ( Xcoef(u,phi) * std::cos(phi)
185 - u * std::sin(phi) + fdeltaX*phi/fPhiTwist,
186 Xcoef(u,phi) * std::sin(phi)
187 + u * std::cos(phi) + fdeltaY*phi/fPhiTwist,
188 2*fDz*phi/fPhiTwist );
189
190 if (isGlobal) { return (fRot * SurfPoint + fTrans); }
191 return SurfPoint;
192}
193
194inline
195G4double G4TwistBoxSide::GetBoundaryMin(G4double phi)
196{
197 return -0.5*GetValueB(phi) ;
198}
199
200inline
201G4double G4TwistBoxSide::GetBoundaryMax(G4double phi)
202{
203 return 0.5*GetValueB(phi) ;
204}
205
206inline
207G4double G4TwistBoxSide::GetSurfaceArea()
208{
209 return (fDz*(std::sqrt(16*fDy1*fDy1
210 + (fa1md1 + 4*fDy1*fTAlph)*(fa1md1 + 4*fDy1*fTAlph))
211 + std::sqrt(16*fDy1*fDy1 + (fa2md2 + 4*fDy1*fTAlph)
212 * (fa2md2 + 4*fDy1*fTAlph))))/2. ;
213}
214
215inline
216G4ThreeVector G4TwistBoxSide::NormAng( G4double phi, G4double u )
217{
218 // function to calculate the norm at a given point on the surface
219 // replace a1-d1
220
221 G4ThreeVector nvec( 4*fDz*(std::cos(phi) + fTAlph*std::sin(phi)) ,
222 4*fDz*(-(fTAlph*std::cos(phi)) + std::sin(phi)),
223 (fDx2 + fDx4)*fPhiTwist*fTAlph
224 + 2*fDx4minus2*(-1 + fTAlph*phi)
225 + 2*fPhiTwist*(1 + fTAlph*fTAlph)*u
226 - 2*(fdeltaX - fdeltaY*fTAlph)*std::cos(phi)
227 - 2*(fdeltaY + fdeltaX*fTAlph)*std::sin(phi) );
228 return nvec.unit();
229}
