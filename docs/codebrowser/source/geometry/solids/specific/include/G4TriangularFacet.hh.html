30// Class G4TriangularFacet
31//
32// Class description:
33//
34// The G4TriangularFacet class is used for the contruction of
35// G4TessellatedSolid.
36// It is defined by three fVertices, which shall be supplied in anti-clockwise
37// order looking from the outsider of the solid where it belongs.
38// Its constructor:
39//
40// G4TriangularFacet (const G4ThreeVector Pt0, const G4ThreeVector vt1,
41// const G4ThreeVector vt2, G4FacetVertexType);
42//
43// takes 4 parameters to define the three fVertices:
44// 1) G4FacetvertexType = "ABSOLUTE": in this case Pt0, vt1 and vt2 are
45// the 3 fVertices in anti-clockwise order looking from the outsider.
46// 2) G4FacetvertexType = "RELATIVE": in this case the first vertex is Pt0,
47// the second vertex is Pt0+vt1 and the third vertex is Pt0+vt2, all
48// in anti-clockwise order when looking from the outsider.
49
50// CHANGE HISTORY
51// --------------
52//
53// 31 October 2004, P R Truscott, QinetiQ Ltd, UK - Created.
54// 12 October 2012, M Gayer, CERN, - Reviewed optimized implementation.
55//

129///////////////////////////////////////////////////////////////////////////////
130// Inlined Methods
131///////////////////////////////////////////////////////////////////////////////
132
133inline G4bool G4TriangularFacet::IsDefined () const
134{
135 return fIsDefined;
136}
137
138inline G4int G4TriangularFacet::GetNumberOfVertices () const
139{
140 return 3;
141}
142
143inline G4ThreeVector G4TriangularFacet::GetVertex (G4int i) const
144{
145 G4int indice = fIndices[i];
146 return indice < 0 ? (*fVertices)[i] : (*fVertices)[indice];
147}
148
149inline void G4TriangularFacet::SetVertex (G4int i, const G4ThreeVector &val)
150{
151 (*fVertices)[i] = val;
152}
153
154inline G4ThreeVector G4TriangularFacet::GetCircumcentre () const
155{
156 return fCircumcentre;
157}
158
159inline G4double G4TriangularFacet::GetRadius () const
160{
161 return fRadius;
162}
163
164inline G4int G4TriangularFacet::AllocatedMemory()
165{
166 G4int size = sizeof(*this);
167 size += GetNumberOfVertices() * sizeof(G4ThreeVector);
168 return size;
169}
170
171inline G4int G4TriangularFacet::GetVertexIndex (G4int i) const
172{
173 if (i < 3) return fIndices[i];
174 else return 999999999;
175}
176
177inline void G4TriangularFacet::SetVertexIndex (G4int i, G4int j)
178{
179 fIndices[i] = j;
180}
181
182inline void G4TriangularFacet::SetVertices(std::vector<G4ThreeVector> *v)
183{
184 if (fIndices[0] < 0 && fVertices)
185 {
186 delete fVertices;
187 fVertices = 0;
188 }
189 fVertices = v;
190}
