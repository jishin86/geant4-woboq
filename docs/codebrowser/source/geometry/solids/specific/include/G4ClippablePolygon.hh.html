30// GEANT 4 class header file
31//
32// G4ClippablePolygon
33//
34// Class description:
35//
36// Declaration of a utility class of a polygon that can be
37// clipped by a voxel.
38
39// Author:
40// David C. Williams (davidw@scipp.ucsc.edu)
41// --------------------------------------------------------------------
42#ifndef G4ClippablePolygon_hh
43#define G4ClippablePolygon_hh
44
45#include <vector>
46
47#include "G4Types.hh"
48#include "geomdefs.hh"
49#include "G4ThreeVector.hh"
50
51class G4AffineTransform;
52class G4VoxelLimits;
53
54class G4ClippablePolygon
55{
56 typedef std::vector<G4ThreeVector> G4ThreeVectorList;
57
58 public: // with description
59
60 G4ClippablePolygon();
61 virtual ~G4ClippablePolygon();
62 // Constructor & virtual destructor.
63
64 virtual void AddVertexInOrder( const G4ThreeVector vertex );
65 virtual void ClearAllVertices();
66
67 inline void SetNormal( const G4ThreeVector &newNormal );
68 inline const G4ThreeVector GetNormal() const;
69
70 virtual G4bool Clip( const G4VoxelLimits &voxelLimit );
71
72 virtual G4bool PartialClip( const G4VoxelLimits &voxelLimit,
73 const EAxis IgnoreMe );
74 // Clip, while ignoring the indicated axis.
75
76 virtual void ClipAlongOneAxis( const G4VoxelLimits &voxelLimit,
77 const EAxis axis );
78 // Clip along just one axis, as specified in voxelLimit.
79
80 virtual G4bool GetExtent( const EAxis axis,
81 G4double &min, G4double &max ) const;
82
83 virtual const G4ThreeVector *GetMinPoint( const EAxis axis ) const;
84 // Returns pointer to minimum point along the specified axis.
85 // Take care! Do not use pointer after destroying parent polygon.
86
87 virtual const G4ThreeVector *GetMaxPoint( const EAxis axis ) const;
88 // Returns pointer to maximum point along the specified axis.
89 // Take care! Do not use pointer after destroying parent polygon.
90
91 inline G4int GetNumVertices() const;
92 inline G4bool Empty() const;
93
94 virtual G4bool InFrontOf( const G4ClippablePolygon &other, EAxis axis ) const;
95 // Decide if the polygon is in "front" of another when
96 // viewed along the specified axis. For our purposes here,
97 // it is sufficient to use the minimum extent of the
98 // polygon along the axis to determine this.
99
100 virtual G4bool BehindOf( const G4ClippablePolygon &other, EAxis axis ) const;
101 // Decide if this polygon is behind another.
102 // Remarks in method "InFrontOf" are valid here too.
103
104 virtual G4bool GetPlanerExtent( const G4ThreeVector &pointOnPlane,
105 const G4ThreeVector &planeNormal,
106 G4double &min, G4double &max ) const;
107 // Get min/max distance in or out of a plane.
108
109 protected: // with description
110
111 void ClipToSimpleLimits( G4ThreeVectorList& pPolygon,
112 G4ThreeVectorList& outputPolygon,
113 const G4VoxelLimits& pVoxelLimit );
114 // pVoxelLimits must be only limited along one axis, and either
115 // the maximum along the axis must be +kInfinity, or the minimum
116 // -kInfinity
117
118 protected:
119
120 G4ThreeVectorList vertices;
121 G4ThreeVector normal;
122 G4double kCarTolerance;
123};
124
125#include "G4ClippablePolygon.icc"
126
127#endif
