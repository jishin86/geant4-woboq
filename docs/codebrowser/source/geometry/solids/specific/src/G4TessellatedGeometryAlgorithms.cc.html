50///////////////////////////////////////////////////////////////////////////////
51//
52// IntersectLineAndTriangle2D
53//
54// Determines whether there is an intersection between a line defined
55// by r = p + s.v and a triangle defined by verticies p0, p0+e0 and p0+e1.
56//
57// Here:
58// p = 2D vector
59// s = scaler on [0,infinity)
60// v = 2D vector
61// p0, e0 and e1 are 2D vectors
62// Information about where the intersection occurs is returned in the
63// variable location.
64//
65// This is based on the work of Rickard Holmberg.
66//
67G4bool G4TessellatedGeometryAlgorithms::IntersectLineAndTriangle2D (
68 const G4TwoVector &p, const G4TwoVector &v,
69 const G4TwoVector &p0, const G4TwoVector &e0, const G4TwoVector &e1,
70 G4TwoVector location[2])
71{
72 G4TwoVector loc0[2];
73 G4int e0i = IntersectLineAndLineSegment2D (p,v,p0,e0,loc0);
74 if (e0i == 2)
75 {
76 location[0] = loc0[0];
77 location[1] = loc0[1];
78 return true;
79 }
80
81 G4TwoVector loc1[2];
82 G4int e1i = IntersectLineAndLineSegment2D (p,v,p0,e1,loc1);
83 if (e1i == 2)
84 {
85 location[0] = loc1[0];
86 location[1] = loc1[1];
87 return true;
88 }
89
90 if ((e0i == 1) && (e1i == 1))
91 {
92 if ((loc0[0]-p).mag2() < (loc1[0]-p).mag2())
93 {
94 location[0] = loc0[0];
95 location[1] = loc1[0];
96 }
97 else
98 {
99 location[0] = loc1[0];
100 location[1] = loc0[0];
101 }
102 return true;
103 }
104
105 G4TwoVector p1 = p0 + e0;
106 G4TwoVector DE = e1 - e0;
107 G4TwoVector loc2[2];
108 G4int e2i = IntersectLineAndLineSegment2D (p,v,p1,DE,loc2);
109 if (e2i == 2)
110 {
111 location[0] = loc2[0];
112 location[1] = loc2[1];
113 return true;
114 }
115
116 if ((e0i == 0) && (e1i == 0) && (e2i == 0)) return false;
117
118 if ((e0i == 1) && (e2i == 1))
119 {
120 if ((loc0[0]-p).mag2() < (loc2[0]-p).mag2())
121 {
122 location[0] = loc0[0];
123 location[1] = loc2[0];
124 }
125 else
126 {
127 location[0] = loc2[0];
128 location[1] = loc0[0];
129 }
130 return true;
131 }
132
133 if ((e1i == 1) && (e2i == 1))
134 {
135 if ((loc1[0]-p).mag2() < (loc2[0]-p).mag2())
136 {
137 location[0] = loc1[0];
138 location[1] = loc2[0];
139 }
140 else
141 {
142 location[0] = loc2[0];
143 location[1] = loc1[0];
144 }
145 return true;
146 }
147
148 return false;
149}

151///////////////////////////////////////////////////////////////////////////////
152//
153// IntersectLineAndLineSegment2D
154//
155// Determines whether there is an intersection between a line defined
156// by r = p0 + s.d0 and a line-segment with endpoints p1 and p1+d1.
157// Here:
158// p0 = 2D vector
159// s = scaler on [0,infinity)
160// d0 = 2D vector
161// p1 and d1 are 2D vectors
162//
163// This function returns:
164// 0 - if there is no intersection;
165// 1 - if there is a unique intersection;
166// 2 - if the line and line-segments overlap, and the intersection is a
167// segment itself.
168// Information about where the intersection occurs is returned in the
169// as ??.
170//
171// This is based on the work of Rickard Holmberg as well as material published
172// by Philip J Schneider and David H Eberly, "Geometric Tools for Computer
173// Graphics," ISBN 1-55860-694-0, pp 244-245, 2003.
174//
175G4int G4TessellatedGeometryAlgorithms::IntersectLineAndLineSegment2D (
176 const G4TwoVector &p0, const G4TwoVector &d0,
177 const G4TwoVector &p1, const G4TwoVector &d1,
178 G4TwoVector location[2])
179{
180 G4TwoVector e = p1 - p0;
181 G4double kross = cross(d0,d1);
182 G4double sqrKross = kross * kross;
183 G4double sqrLen0 = d0.mag2();
184 G4double sqrLen1 = d1.mag2();
185 location[0] = G4TwoVector(0.0,0.0);
186 location[1] = G4TwoVector(0.0,0.0);
187
188 if (sqrKross > DBL_EPSILON * DBL_EPSILON * sqrLen0 * sqrLen1)
189 {
190 //
191 // The line and line segment are not parallel. Determine if the intersection
192 // is in positive s where r=p0 + s*d0, and for 0<=t<=1 where r=p1 + t*d1.
193 //
194 G4double ss = cross(e,d1)/kross;
195 if (ss < 0) return 0; // Intersection does not occur for positive ss
196 G4double t = cross(e,d0)/kross;
197 if (t < 0 || t > 1) return 0; // Intersection does not occur on line-segment
198 //
199 // Intersection of lines is a single point on the forward-propagating line
200 // defined by r=p0 + ss*d0, and the line segment defined by r=p1 + t*d1.
201 //
202 location[0] = p0 + ss*d0;
203 return 1;
204 }
205 //
206 // Line and line segment are parallel. Determine whether they overlap or not.
207 //
208 G4double sqrLenE = e.mag2();
209 kross = cross(e,d0);
210 sqrKross = kross * kross;
211 if (sqrKross > DBL_EPSILON * DBL_EPSILON * sqrLen0 * sqrLenE)
212 {
213 return 0; //Lines are different.
214 }
215 //
216 // Lines are the same. Test for overlap.
217 //
218 G4double s0 = d0.dot(e)/sqrLen0;
219 G4double s1 = s0 + d0.dot(d1)/sqrLen0;
220 G4double smin = 0.0;
221 G4double smax = 0.0;
222
223 if (s0 < s1) {smin = s0; smax = s1;}
224 else {smin = s1; smax = s0;}
225
226 if (smax < 0.0) return 0;
227 else if (smin < 0.0)
228 {
229 location[0] = p0;
230 location[1] = p0 + smax*d0;
231 return 2;
232 }
233 else
234 {
235 location[0] = p0 + smin*d0;
236 location[1] = p0 + smax*d0;
237 return 2;
238 }
239}

241///////////////////////////////////////////////////////////////////////////////
242//
243// CrossProduct
244//
245// This is just a ficticious "cross-product" function for two 2D vectors...
246// "ficticious" because such an operation is not relevant to 2D space compared
247// with 3D space.
248//
249G4double G4TessellatedGeometryAlgorithms::cross(const G4TwoVector &v1,
250 const G4TwoVector &v2)
251{
252 return v1.x()*v2.y() - v1.y()*v2.x();
253}
