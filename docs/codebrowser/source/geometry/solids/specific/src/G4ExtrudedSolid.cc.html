73G4ExtrudedSolid::G4ExtrudedSolid( const G4String& pName,
74 const std::vector<G4TwoVector>& polygon,
75 const std::vector<ZSection>& zsections)
76 : G4TessellatedSolid(pName),
77 fNv(polygon.size()),
78 fNz(zsections.size()),
79 fIsConvex(false),
80 fGeometryType("G4ExtrudedSolid"),
81 fSolidType(0)
82{
83 // General constructor
84
85 // First check input parameters
86
87 if (fNv < 3)
88 {
89 std::ostringstream message;
90 message << "Number of vertices in polygon < 3 - " << pName;
91 G4Exception("G4ExtrudedSolid::G4ExtrudedSolid()", "GeomSolids0002",
92 FatalErrorInArgument, message);
93 }
94
95 if (fNz < 2)
96 {
97 std::ostringstream message;
98 message << "Number of z-sides < 2 - " << pName;
99 G4Exception("G4ExtrudedSolid::G4ExtrudedSolid()", "GeomSolids0002",
100 FatalErrorInArgument, message);
101 }
102
103 for ( G4int i=0; i<fNz-1; ++i )
104 {
105 if ( zsections[i].fZ > zsections[i+1].fZ )
106 {
107 std::ostringstream message;
108 message << "Z-sections have to be ordered by z value (z0 < z1 < z2...) - "
109 << pName;
110 G4Exception("G4ExtrudedSolid::G4ExtrudedSolid()", "GeomSolids0002",
111 FatalErrorInArgument, message);
112 }
113 if ( std::fabs( zsections[i+1].fZ - zsections[i].fZ ) < kCarToleranceHalf )
114 {
115 std::ostringstream message;
116 message << "Z-sections with the same z position are not supported - "
117 << pName;
118 G4Exception("G4ExtrudedSolid::G4ExtrudedSolid()", "GeomSolids0001",
119 FatalException, message);
120 }
121 }
122
123 // Copy polygon
124 //
125 fPolygon = polygon;
126
127 // Remove collinear and coincident vertices, if any
128 //
129 std::vector<G4int> removedVertices;
130 G4GeomTools::RemoveRedundantVertices(fPolygon,removedVertices,
131 2*kCarTolerance);
132 if (removedVertices.size() != 0)
133 {
134 G4int nremoved = removedVertices.size();
135 std::ostringstream message;
136 message << "The following "<< nremoved
137 << " vertices have been removed from polygon in " << pName
138 << "\nas collinear or coincident with other vertices: "
139 << removedVertices[0];
140 for (G4int i=1; i<nremoved; ++i) message << ", " << removedVertices[i];
141 G4Exception("G4ExtrudedSolid::G4ExtrudedSolid()", "GeomSolids1001",
142 JustWarning, message);
143 }
144
145 fNv = fPolygon.size();
146 if (fNv < 3)
147 {
148 std::ostringstream message;
149 message << "Number of vertices in polygon after removal < 3 - " << pName;
150 G4Exception("G4ExtrudedSolid::G4ExtrudedSolid()", "GeomSolids0002",
151 FatalErrorInArgument, message);
152 }
153
154 // Check if polygon vertices are defined clockwise
155 // (the area is positive if polygon vertices are defined anti-clockwise)
156 //
157 if (G4GeomTools::PolygonArea(fPolygon) > 0.)
158 {
159 // Polygon vertices are defined anti-clockwise, we revert them
160 // G4Exception("G4ExtrudedSolid::G4ExtrudedSolid()", "GeomSolids1001",
161 // JustWarning,
162 // "Polygon vertices defined anti-clockwise, reverting polygon");
163 std::reverse(fPolygon.begin(),fPolygon.end());
164 }
165
166 // Copy z-sections
167 //
168 fZSections = zsections;
169
170 G4bool result = MakeFacets();
171 if (!result)
172 {
173 std::ostringstream message;
174 message << "Making facets failed - " << pName;
175 G4Exception("G4ExtrudedSolid::G4ExtrudedSolid()", "GeomSolids0003",
176 FatalException, message);
177 }
178 fIsConvex = G4GeomTools::IsConvex(fPolygon);
179
180 ComputeProjectionParameters();
181
182 // Check if the solid is a right prism, if so then set lateral planes
183 //
184 if ((fNz == 2)
185 && (fZSections[0].fScale == 1) && (fZSections[1].fScale == 1)
186 && (fZSections[0].fOffset == G4TwoVector(0,0))
187 && (fZSections[1].fOffset == G4TwoVector(0,0)))
188 {
189 fSolidType = (fIsConvex) ? 1 : 2; // 1 - convex, 2 - non-convex right prism
190 ComputeLateralPlanes();
191 }
192}

349void G4ExtrudedSolid::ComputeProjectionParameters()
350{
351 // Compute parameters for point projections p(z)
352 // to the polygon scale & offset:
353 // scale(z) = k*z + scale0
354 // offset(z) = l*z + offset0
355 // p(z) = scale(z)*p0 + offset(z)
356 // p0 = (p(z) - offset(z))/scale(z);
357 //
358
359 for ( G4int iz=0; iz<fNz-1; ++iz)
360 {
361 G4double z1 = fZSections[iz].fZ;
362 G4double z2 = fZSections[iz+1].fZ;
363 G4double scale1 = fZSections[iz].fScale;
364 G4double scale2 = fZSections[iz+1].fScale;
365 G4TwoVector off1 = fZSections[iz].fOffset;
366 G4TwoVector off2 = fZSections[iz+1].fOffset;
367
368 G4double kscale = (scale2 - scale1)/(z2 - z1);
369 G4double scale0 = scale2 - kscale*(z2 - z1)/2.0;
370 G4TwoVector koff = (off2 - off1)/(z2 - z1);
371 G4TwoVector off0 = off2 - koff*(z2 - z1)/2.0;
372
373 fKScales.push_back(kscale);
374 fScale0s.push_back(scale0);
375 fKOffsets.push_back(koff);
376 fOffset0s.push_back(off0);
377 }
378}

381
382void G4ExtrudedSolid::ComputeLateralPlanes()
383{
384 // Compute lateral planes: a*x + b*y + c*z + d = 0
385 //
386 G4int Nv = fPolygon.size();
387 fPlanes.resize(Nv);
388 for (G4int i=0, k=Nv-1; i<Nv; k=i++)
389 {
390 G4TwoVector norm = (fPolygon[i] - fPolygon[k]).unit();
391 fPlanes[i].a = -norm.y();
392 fPlanes[i].b = norm.x();
393 fPlanes[i].c = 0;
394 fPlanes[i].d = norm.y()*fPolygon[i].x() - norm.x()*fPolygon[i].y();
395 }
396
397 // Compute edge equations: x = k*y + m
398 // and edge lengths
399 //
400 fLines.resize(Nv);
401 fLengths.resize(Nv);
402 for (G4int i=0, k=Nv-1; i<Nv; k=i++)
403 {
404 if (fPolygon[k].y() == fPolygon[i].y())
405 {
406 fLines[i].k = 0;
407 fLines[i].m = fPolygon[i].x();
408 }
409 else
410 {
411 G4double ctg = (fPolygon[k].x()-fPolygon[i].x())/(fPolygon[k].y()-fPolygon[i].y());
412 fLines[i].k = ctg;
413 fLines[i].m = fPolygon[i].x() - ctg*fPolygon[i].y();
414 }
415 fLengths[i] = (fPolygon[i] - fPolygon[k]).mag();
416 }
417}

422{
423 // Shift and scale vertices
424
425 return G4ThreeVector( fPolygon[ind].x() * fZSections[iz].fScale

433G4TwoVector G4ExtrudedSolid::ProjectPoint(const G4ThreeVector& point) const
434{
435 // Project point in the polygon scale
436 // scale(z) = k*z + scale0
437 // offset(z) = l*z + offset0
438 // p(z) = scale(z)*p0 + offset(z)
439 // p0 = (p(z) - offset(z))/scale(z);
440
441 // Select projection (z-segment of the solid) according to p.z()
442 //
443 G4int iz = 0;
444 while ( point.z() > fZSections[iz+1].fZ && iz < fNz-2 ) { ++iz; }
445 // Loop checking, 13.08.2015, G.Cosmo
446
447 G4double z0 = ( fZSections[iz+1].fZ + fZSections[iz].fZ )/2.0;
448 G4TwoVector p2(point.x(), point.y());
449 G4double pscale = fKScales[iz]*(point.z()-z0) + fScale0s[iz];
450 G4TwoVector poffset = fKOffsets[iz]*(point.z()-z0) + fOffset0s[iz];
451
452 // G4cout << point << " projected to "
453 // << iz << "-th z-segment polygon as "
454 // << (p2 - poffset)/pscale << G4endl;
455
456 // pscale is always >0 as it is an interpolation between two
457 // positive scale values
458 //
459 return (p2 - poffset)/pscale;
460}

464G4bool G4ExtrudedSolid::IsSameLine(const G4TwoVector& p,
465 const G4TwoVector& l1,
466 const G4TwoVector& l2) const
467{
468 // Return true if p is on the line through l1, l2
469
470 if ( l1.x() == l2.x() )
471 {
472 return std::fabs(p.x() - l1.x()) < kCarToleranceHalf;
473 }
474 G4double slope= ((l2.y() - l1.y())/(l2.x() - l1.x()));
475 G4double predy= l1.y() + slope *(p.x() - l1.x());
476 G4double dy= p.y() - predy;
477
478 // Calculate perpendicular distance
479 //
480 // G4double perpD= std::fabs(dy) / std::sqrt( 1 + slope * slope );
481 // G4bool simpleComp= (perpD<kCarToleranceHalf);
482
483 // Check perpendicular distance vs tolerance 'directly'
484 //
485 G4bool squareComp = (dy*dy < (1+slope*slope)
486 * kCarToleranceHalf * kCarToleranceHalf);
487
488 // return simpleComp;
489 return squareComp;
490}

494G4bool G4ExtrudedSolid::IsSameLineSegment(const G4TwoVector& p,
495 const G4TwoVector& l1,
496 const G4TwoVector& l2) const
497{
498 // Return true if p is on the line through l1, l2 and lies between
499 // l1 and l2
500
501 if ( p.x() < std::min(l1.x(), l2.x()) - kCarToleranceHalf ||
502 p.x() > std::max(l1.x(), l2.x()) + kCarToleranceHalf ||
503 p.y() < std::min(l1.y(), l2.y()) - kCarToleranceHalf ||
504 p.y() > std::max(l1.y(), l2.y()) + kCarToleranceHalf )
505 {
506 return false;
507 }
508
509 return IsSameLine(p, l1, l2);
510}

514G4bool G4ExtrudedSolid::IsSameSide(const G4TwoVector& p1,
515 const G4TwoVector& p2,
516 const G4TwoVector& l1,
517 const G4TwoVector& l2) const
518{
519 // Return true if p1 and p2 are on the same side of the line through l1, l2
520
521 return ( (p1.x() - l1.x()) * (l2.y() - l1.y())
522 - (l2.x() - l1.x()) * (p1.y() - l1.y()) )
523 * ( (p2.x() - l1.x()) * (l2.y() - l1.y())
524 - (l2.x() - l1.x()) * (p2.y() - l1.y()) ) > 0;
525}

529G4bool G4ExtrudedSolid::IsPointInside(const G4TwoVector& a,
530 const G4TwoVector& b,
531 const G4TwoVector& c,
532 const G4TwoVector& p) const
533{
534 // Return true if p is inside of triangle abc or on its edges,
535 // else returns false
536
537 // Check extent first
538 //
539 if ( ( p.x() < a.x() && p.x() < b.x() && p.x() < c.x() ) ||
540 ( p.x() > a.x() && p.x() > b.x() && p.x() > c.x() ) ||
541 ( p.y() < a.y() && p.y() < b.y() && p.y() < c.y() ) ||
542 ( p.y() > a.y() && p.y() > b.y() && p.y() > c.y() ) ) return false;
543
544 G4bool inside
545 = IsSameSide(p, a, b, c)
546 && IsSameSide(p, b, a, c)
547 && IsSameSide(p, c, a, b);
548
549 G4bool onEdge
550 = IsSameLineSegment(p, a, b)
551 || IsSameLineSegment(p, b, c)
552 || IsSameLineSegment(p, c, a);
553
554 return inside || onEdge;
555}

559G4double
560G4ExtrudedSolid::GetAngle(const G4TwoVector& po,
561 const G4TwoVector& pa,
562 const G4TwoVector& pb) const
563{
564 // Return the angle of the vertex in po
565
566 G4TwoVector t1 = pa - po;
567 G4TwoVector t2 = pb - po;
568
569 G4double result = (std::atan2(t1.y(), t1.x()) - std::atan2(t2.y(), t2.x()));
570
571 if ( result < 0 ) result += 2*pi;
572
573 return result;
574}

578G4VFacet*
579G4ExtrudedSolid::MakeDownFacet(G4int ind1, G4int ind2, G4int ind3) const
580{
581 // Create a triangular facet from the polygon points given by indices
582 // forming the down side ( the normal goes in -z)
583
584 std::vector<G4ThreeVector> vertices;
585 vertices.push_back(GetVertex(0, ind1));
586 vertices.push_back(GetVertex(0, ind2));
587 vertices.push_back(GetVertex(0, ind3));
588
589 // first vertex most left
590 //
591 G4ThreeVector cross
592 = (vertices[1]-vertices[0]).cross(vertices[2]-vertices[1]);
593
594 if ( cross.z() > 0.0 )
595 {
596 // vertices ordered clock wise has to be reordered
597
598 // G4cout << "G4ExtrudedSolid::MakeDownFacet: reordering vertices "
599 // << ind1 << ", " << ind2 << ", " << ind3 << G4endl;
600
601 G4ThreeVector tmp = vertices[1];
602 vertices[1] = vertices[2];
603 vertices[2] = tmp;
604 }
605
606 return new G4TriangularFacet(vertices[0], vertices[1],
607 vertices[2], ABSOLUTE);
608}

612G4VFacet*
613G4ExtrudedSolid::MakeUpFacet(G4int ind1, G4int ind2, G4int ind3) const
614{
615 // Creates a triangular facet from the polygon points given by indices
616 // forming the upper side ( z>0 )
617
618 std::vector<G4ThreeVector> vertices;
619 vertices.push_back(GetVertex(fNz-1, ind1));
620 vertices.push_back(GetVertex(fNz-1, ind2));
621 vertices.push_back(GetVertex(fNz-1, ind3));
622
623 // first vertex most left
624 //
625 G4ThreeVector cross
626 = (vertices[1]-vertices[0]).cross(vertices[2]-vertices[1]);
627
628 if ( cross.z() < 0.0 )
629 {
630 // vertices ordered clock wise has to be reordered
631
632 // G4cout << "G4ExtrudedSolid::MakeUpFacet: reordering vertices "
633 // << ind1 << ", " << ind2 << ", " << ind3 << G4endl;
634
635 G4ThreeVector tmp = vertices[1];
636 vertices[1] = vertices[2];
637 vertices[2] = tmp;
638 }
639
640 return new G4TriangularFacet(vertices[0], vertices[1],
641 vertices[2], ABSOLUTE);
642}

645
646G4bool G4ExtrudedSolid::AddGeneralPolygonFacets()
647{
648 // Decompose polygonal sides in triangular facets
649
650 typedef std::pair < G4TwoVector, G4int > Vertex;
651
652 static const G4double kAngTolerance =
653 G4GeometryTolerance::GetInstance()->GetAngularTolerance();
654
655 // Fill one more vector
656 //
657 std::vector< Vertex > verticesToBeDone;
658 for ( G4int i=0; i<fNv; ++i )
659 {
660 verticesToBeDone.push_back(Vertex(fPolygon[i], i));
661 }
662 std::vector< Vertex > ears;
663
664 std::vector< Vertex >::iterator c1 = verticesToBeDone.begin();
665 std::vector< Vertex >::iterator c2 = c1+1;
666 std::vector< Vertex >::iterator c3 = c1+2;
667 while ( verticesToBeDone.size()>2 ) // Loop checking, 13.08.2015, G.Cosmo
668 {
669
670 // G4cout << "Looking at triangle : "
671 // << c1->second << " " << c2->second
672 // << " " << c3->second << G4endl;
673 //G4cout << "Looking at triangle : "
674 // << c1->first << " " << c2->first
675 // << " " << c3->first << G4endl;
676
677 // skip concave vertices
678 //
679 G4double angle = GetAngle(c2->first, c3->first, c1->first);
680
681 //G4cout << "angle " << angle << G4endl;
682
683 G4int counter = 0;
684 while ( angle >= (pi-kAngTolerance) ) // Loop checking, 13.08.2015, G.Cosmo
685 {
686 // G4cout << "Skipping concave vertex " << c2->second << G4endl;
687
688 // try next three consecutive vertices
689 //
690 c1 = c2;
691 c2 = c3;

693 if ( c3 == verticesToBeDone.end() ) { c3 = verticesToBeDone.begin(); }
694
695 //G4cout << "Looking at triangle : "
696 // << c1->first << " " << c2->first
697 // << " " << c3->first << G4endl;
698
699 angle = GetAngle(c2->first, c3->first, c1->first);
700 //G4cout << "angle " << angle << G4endl;
701
702 counter++;
703
704 if ( counter > fNv) {
705 G4Exception("G4ExtrudedSolid::AddGeneralPolygonFacets",
706 "GeomSolids0003", FatalException,
707 "Triangularisation has failed.");
708 break;
709 }
710 }
711
712 G4bool good = true;
713 std::vector< Vertex >::iterator it;
714 for ( it=verticesToBeDone.begin(); it != verticesToBeDone.end(); ++it )
715 {
716 // skip vertices of tested triangle
717 //
718 if ( it == c1 || it == c2 || it == c3 ) { continue; }
719
720 if ( IsPointInside(c1->first, c2->first, c3->first, it->first) )
721 {
722 // G4cout << "Point " << it->second << " is inside" << G4endl;
723 good = false;
724
725 // try next three consecutive vertices
726 //
727 c1 = c2;
728 c2 = c3;
729 ++c3;
730 if ( c3 == verticesToBeDone.end() ) { c3 = verticesToBeDone.begin(); }
731 break;
732 }
733 // else
734 // { G4cout << "Point " << it->second << " is outside" << G4endl; }
735 }
736 if ( good )
737 {
738 // all points are outside triangle, we can make a facet
739
740 // G4cout << "Found triangle : "
741 // << c1->second << " " << c2->second
742 // << " " << c3->second << G4endl;
743
744 G4bool result;
745 result = AddFacet( MakeDownFacet(c1->second, c2->second, c3->second) );
746 if ( ! result ) { return false; }
747
748 result = AddFacet( MakeUpFacet(c1->second, c2->second, c3->second) );
749 if ( ! result ) { return false; }
750
751 std::vector<G4int> triangle(3);
752 triangle[0] = c1->second;
753 triangle[1] = c2->second;
754 triangle[2] = c3->second;
755 fTriangles.push_back(triangle);
756
757 // remove the ear point from verticesToBeDone
758 //
759 verticesToBeDone.erase(c2);
760 c1 = verticesToBeDone.begin();
761 c2 = c1+1;
762 c3 = c1+2;
763 }
764 }
765 return true;
766}

865
866EInside G4ExtrudedSolid::Inside(const G4ThreeVector &p) const
867{
868 switch (fSolidType)
869 {
870 case 1: // convex right prism
871 {
872 G4double dist = std::max(fZSections[0].fZ-p.z(),p.z()-fZSections[1].fZ);
873 if (dist > kCarToleranceHalf) { return kOutside; }
874
875 G4int np = fPlanes.size();
876 for (G4int i=0; i<np; ++i)
877 {
878 G4double dd = fPlanes[i].a*p.x() + fPlanes[i].b*p.y() + fPlanes[i].d;
879 if (dd > dist) { dist = dd; }
880 }
881 if (dist > kCarToleranceHalf) { return kOutside; }
882 return (dist > -kCarToleranceHalf) ? kSurface : kInside;
883 }
884 case 2: // non-convex right prism
885 {
886 G4double distz = std::max(fZSections[0].fZ-p.z(),p.z()-fZSections[1].fZ);
887 if (distz > kCarToleranceHalf) { return kOutside; }
888
889 G4bool in = PointInPolygon(p);
890 if (distz > -kCarToleranceHalf && in) { return kSurface; }
891
892 G4double dd = DistanceToPolygonSqr(p) - kCarToleranceHalf*kCarToleranceHalf;
893 if (in)
894 {
895 return (dd >= 0) ? kInside : kSurface;
896 }
897 else
898 {
899 return (dd > 0) ? kOutside : kSurface;
900 }
901 }
902 }
903
904 // Override the base class function as it fails in case of concave polygon.
905 // Project the point in the original polygon scale and check if it is inside
906 // for each triangle.
907
908 // Check first if outside extent
909 //
910 if ( p.x() < GetMinXExtent() - kCarToleranceHalf ||
911 p.x() > GetMaxXExtent() + kCarToleranceHalf ||
912 p.y() < GetMinYExtent() - kCarToleranceHalf ||
913 p.y() > GetMaxYExtent() + kCarToleranceHalf ||
914 p.z() < GetMinZExtent() - kCarToleranceHalf ||
915 p.z() > GetMaxZExtent() + kCarToleranceHalf )
916 {
917 // G4cout << "G4ExtrudedSolid::Outside extent: " << p << G4endl;
918 return kOutside;
919 }
920
921 // Project point p(z) to the polygon scale p0
922 //
923 G4TwoVector pscaled = ProjectPoint(p);
924
925 // Check if on surface of polygon
926 //
927 for ( G4int i=0; i<fNv; ++i )
928 {
929 G4int j = (i+1) % fNv;
930 if ( IsSameLineSegment(pscaled, fPolygon[i], fPolygon[j]) )
931 {
932 // G4cout << "G4ExtrudedSolid::Inside return Surface (on polygon) "
933 // << G4endl;
934
935 return kSurface;
936 }
937 }
938
939 // Now check if inside triangles
940 //
941 std::vector< std::vector<G4int> >::const_iterator it = fTriangles.begin();
942 G4bool inside = false;
943 do // Loop checking, 13.08.2015, G.Cosmo
944 {
945 if ( IsPointInside(fPolygon[(*it)[0]], fPolygon[(*it)[1]],
946 fPolygon[(*it)[2]], pscaled) ) { inside = true; }
947 ++it;
948 } while ( (inside == false) && (it != fTriangles.end()) );
949
950 if ( inside )
951 {
952 // Check if on surface of z sides
953 //
954 if ( std::fabs( p.z() - fZSections[0].fZ ) < kCarToleranceHalf ||
955 std::fabs( p.z() - fZSections[fNz-1].fZ ) < kCarToleranceHalf )
956 {
957 // G4cout << "G4ExtrudedSolid::Inside return Surface (on z side)"
958 // << G4endl;
959
960 return kSurface;
961 }
962
963 // G4cout << "G4ExtrudedSolid::Inside return Inside" << G4endl;
964
965 return kInside;
966 }
967
968 // G4cout << "G4ExtrudedSolid::Inside return Outside " << G4endl;
969
970 return kOutside;
971}

974
975G4ThreeVector G4ExtrudedSolid::SurfaceNormal(const G4ThreeVector& p) const
976{
977 G4int nsurf = 0;
978 G4double nx = 0, ny = 0, nz = 0;
979 switch (fSolidType)
980 {
981 case 1: // convex right prism
982 {
983 if (std::abs(p.z() - fZSections[0].fZ) <= kCarToleranceHalf) { nz = -1; ++nsurf; }
984 if (std::abs(p.z() - fZSections[1].fZ) <= kCarToleranceHalf) { nz = 1; ++nsurf; }
985 for (G4int i=0; i<fNv; ++i)
986 {
987 G4double dd = fPlanes[i].a*p.x() + fPlanes[i].b*p.y() + fPlanes[i].d;
988 if (std::abs(dd) > kCarToleranceHalf) continue;
989 nx += fPlanes[i].a;
990 ny += fPlanes[i].b;
991 ++nsurf;
992 }
993 break;
994 }
995 case 2: // non-convex right prism
996 {
997 if (std::abs(p.z() - fZSections[0].fZ) <= kCarToleranceHalf) { nz = -1; ++nsurf; }
998 if (std::abs(p.z() - fZSections[1].fZ) <= kCarToleranceHalf) { nz = 1; ++nsurf; }
999
1000 G4double sqrCarToleranceHalf = kCarToleranceHalf*kCarToleranceHalf;
1001 for (G4int i=0, k=fNv-1; i<fNv; k=i++)
1002 {
1003 G4double ix = p.x() - fPolygon[i].x();
1004 G4double iy = p.y() - fPolygon[i].y();
1005 G4double u = fPlanes[i].a*iy - fPlanes[i].b*ix;
1006 if (u < 0)
1007 {
1008 if (ix*ix + iy*iy > sqrCarToleranceHalf) continue;
1009 }
1010 else if (u > fLengths[i])
1011 {
1012 G4double kx = p.x() - fPolygon[k].x();
1013 G4double ky = p.y() - fPolygon[k].y();
1014 if (kx*kx + ky*ky > sqrCarToleranceHalf) continue;
1015 }
1016 else
1017 {
1018 G4double dd = fPlanes[i].a*p.x() + fPlanes[i].b*p.y() + fPlanes[i].d;
1019 if (dd*dd > sqrCarToleranceHalf) continue;
1020 }
1021 nx += fPlanes[i].a;
1022 ny += fPlanes[i].b;
1023 ++nsurf;
1024 }
1025 break;
1026 }
1027 default:
1028 {
1029 return G4TessellatedSolid::SurfaceNormal(p);
1030 }
1031 }
1032
1033 // Return normal (right prism)
1034 //
1035 if (nsurf == 1)
1036 {
1037 return G4ThreeVector(nx,ny,nz);
1038 }
1039 else if (nsurf != 0) // edge or corner
1040 {
1041 return G4ThreeVector(nx,ny,nz).unit();
1042 }
1043 else
1044 {
1045 // Point is not on the surface, compute approximate normal
1046 //
1047#ifdef G4CSGDEBUG
1048 std::ostringstream message;
1049 G4int oldprc = message.precision(16);
1050 message << "Point p is not on surface (!?) of solid: "
1051 << GetName() << G4endl;
1052 message << "Position:\n";
1053 message << " p.x() = " << p.x()/mm << " mm\n";
1054 message << " p.y() = " << p.y()/mm << " mm\n";
1055 message << " p.z() = " << p.z()/mm << " mm";
1056 G4cout.precision(oldprc) ;
1057 G4Exception("G4TesselatedSolid::SurfaceNormal(p)", "GeomSolids1002",
1058 JustWarning, message );
1059 DumpInfo();
1060#endif
1061 return ApproxSurfaceNormal(p);
1062 }
1063}

1066
1067G4ThreeVector G4ExtrudedSolid::ApproxSurfaceNormal(const G4ThreeVector& p) const
1068{
1069 // This method is valid only for right prisms and
1070 // normally should not be called
1071
1072 if (fSolidType == 1 || fSolidType == 2)
1073 {
1074 // Find distances to z-planes
1075 //
1076 G4double dz0 = fZSections[0].fZ - p.z();
1077 G4double dz1 = p.z() - fZSections[1].fZ;
1078 G4double ddz0 = dz0*dz0;
1079 G4double ddz1 = dz1*dz1;
1080
1081 // Find nearest lateral side and distance to it
1082 //
1083 G4int iside = 0;
1084 G4double dd = DBL_MAX;
1085 for (G4int i=0, k=fNv-1; i<fNv; k=i++)
1086 {
1087 G4double ix = p.x() - fPolygon[i].x();
1088 G4double iy = p.y() - fPolygon[i].y();
1089 G4double u = fPlanes[i].a*iy - fPlanes[i].b*ix;
1090 if (u < 0)
1091 {
1092 G4double tmp = ix*ix + iy*iy;
1093 if (tmp < dd) { dd = tmp; iside = i; }
1094 }
1095 else if (u > fLengths[i])
1096 {
1097 G4double kx = p.x() - fPolygon[k].x();
1098 G4double ky = p.y() - fPolygon[k].y();
1099 G4double tmp = kx*kx + ky*ky;
1100 if (tmp < dd) { dd = tmp; iside = i; }
1101 }
1102 else
1103 {
1104 G4double tmp = fPlanes[i].a*p.x() + fPlanes[i].b*p.y() + fPlanes[i].d;
1105 tmp *= tmp;
1106 if (tmp < dd) { dd = tmp; iside = i; }
1107 }
1108 }
1109
1110 // Find region
1111 //
1112 // 3 | 1 | 3
1113 // ----+-------+----
1114 // 2 | 0 | 2
1115 // ----+-------+----
1116 // 3 | 1 | 3
1117 //
1118 G4int iregion = 0;
1119 if (std::max(dz0,dz1) > 0) iregion = 1;
1120
1121 G4bool in = PointInPolygon(p);
1122 if (!in) iregion += 2;
1123
1124 // Return normal
1125 //
1126 switch (iregion)
1127 {
1128 case 0:
1129 {
1130 if (ddz0 <= ddz1 && ddz0 <= dd) return G4ThreeVector(0, 0,-1);
1131 if (ddz1 <= ddz0 && ddz1 <= dd) return G4ThreeVector(0, 0, 1);
1132 return G4ThreeVector(fPlanes[iside].a,fPlanes[iside].b, 0);
1133 }
1134 case 1:
1135 {
1136 return G4ThreeVector(0, 0, (dz0 > dz1) ? -1 : 1);
1137 }
1138 case 2:
1139 {
1140 return G4ThreeVector(fPlanes[iside].a,fPlanes[iside].b, 0);
1141 }
1142 case 3:
1143 {
1144 G4double dzmax = std::max(dz0,dz1);
1145 if (dzmax*dzmax > dd) return G4ThreeVector(0,0,(dz0 > dz1) ? -1 : 1);
1146 return G4ThreeVector(fPlanes[iside].a,fPlanes[iside].b, 0);
1147 }
1148 }
1149 }
1150 return G4ThreeVector(0,0,0);
1151}

1155G4double G4ExtrudedSolid::DistanceToIn(const G4ThreeVector& p,
1156 const G4ThreeVector& v) const
1157{
1158 G4double z0 = fZSections[0].fZ;
1159 G4double z1 = fZSections[fNz-1].fZ;
1160 if ((p.z() <= z0 + kCarToleranceHalf) && v.z() <= 0) return kInfinity;
1161 if ((p.z() >= z1 - kCarToleranceHalf) && v.z() >= 0) return kInfinity;
1162
1163 switch (fSolidType)
1164 {
1165 case 1: // convex right prism
1166 {
1167 // Intersection with Z planes
1168 //
1169 G4double dz = (z1 - z0)*0.5;
1170 G4double pz = p.z() - dz - z0;
1171
1172 G4double invz = (v.z() == 0) ? DBL_MAX : -1./v.z();
1173 G4double ddz = (invz < 0) ? dz : -dz;
1174 G4double tzmin = (pz + ddz)*invz;
1175 G4double tzmax = (pz - ddz)*invz;
1176
1177 // Intersection with lateral planes
1178 //
1179 G4int np = fPlanes.size();
1180 G4double txmin = tzmin, txmax = tzmax;
1181 for (G4int i=0; i<np; ++i)
1182 {
1183 G4double cosa = fPlanes[i].a*v.x()+fPlanes[i].b*v.y();
1184 G4double dist = fPlanes[i].a*p.x()+fPlanes[i].b*p.y()+fPlanes[i].d;
1185 if (dist >= -kCarToleranceHalf)
1186 {
1187 if (cosa >= 0) { return kInfinity; }
1188 G4double tmp = -dist/cosa;
1189 if (txmin < tmp) { txmin = tmp; }
1190 }
1191 else if (cosa > 0)
1192 {
1193 G4double tmp = -dist/cosa;
1194 if (txmax > tmp) { txmax = tmp; }
1195 }
1196 }
1197
1198 // Find distance
1199 //
1200 G4double tmin = txmin, tmax = txmax;
1201 if (tmax <= tmin + kCarToleranceHalf) // touch or no hit
1202 {
1203 return kInfinity;
1204 }
1205 return (tmin < kCarToleranceHalf) ? 0. : tmin;
1206 }
1207 case 2: // non-convex right prism
1208 {
1209 }
1210 }
1211 return G4TessellatedSolid::DistanceToIn(p,v);
1212}

1254
1255G4double G4ExtrudedSolid::DistanceToOut (const G4ThreeVector &p,
1256 const G4ThreeVector &v,
1257 const G4bool calcNorm,
1258 G4bool *validNorm,
1259 G4ThreeVector *n) const
1260{
1261 G4bool getnorm = calcNorm;
1262 if (getnorm) *validNorm = true;
1263
1264 G4double z0 = fZSections[0].fZ;
1265 G4double z1 = fZSections[fNz-1].fZ;
1266 if ((p.z() <= z0 + kCarToleranceHalf) && v.z() < 0)
1267 {
1268 if (getnorm) n->set(0,0,-1);
1269 return 0;
1270 }
1271 if ((p.z() >= z1 - kCarToleranceHalf) && v.z() > 0)
1272 {
1273 if (getnorm) n->set(0,0,1);
1274 return 0;
1275 }
1276
1277 switch (fSolidType)
1278 {
1279 case 1: // convex right prism
1280 {
1281 // Intersection with Z planes
1282 //
1283 G4double dz = (z1 - z0)*0.5;
1284 G4double pz = p.z() - z1 - z0;
1285
1286 G4double vz = v.z();
1287 G4double tmax = (vz == 0) ? DBL_MAX : (std::copysign(dz,vz) - pz)/vz;
1288 G4int iside = (vz < 0) ? -4 : -2; // little trick: (-4+3)=-1, (-2+3)=+1
1289
1290 // Intersection with lateral planes
1291 //
1292 G4int np = fPlanes.size();
1293 for (G4int i=0; i<np; ++i)
1294 {
1295 G4double cosa = fPlanes[i].a*v.x()+fPlanes[i].b*v.y();
1296 if (cosa > 0)
1297 {
1298 G4double dist = fPlanes[i].a*p.x()+fPlanes[i].b*p.y()+fPlanes[i].d;
1299 if (dist >= -kCarToleranceHalf)
1300 {
1301 if (getnorm) n->set(fPlanes[i].a, fPlanes[i].b, fPlanes[i].c);
1302 return 0;
1303 }
1304 G4double tmp = -dist/cosa;
1305 if (tmax > tmp) { tmax = tmp; iside = i; }
1306 }
1307 }
1308
1309 // Set normal, if required, and return distance
1310 //
1311 if (getnorm)
1312 {
1313 if (iside < 0)
1314 { n->set(0, 0, iside + 3); } // (-4+3)=-1, (-2+3)=+1
1315 else
1316 { n->set(fPlanes[iside].a, fPlanes[iside].b, fPlanes[iside].c); }
1317 }
1318 return tmax;
1319 }
1320 case 2: // non-convex right prism
1321 {
1322 }
1323 }
1324
1325 // Override the base class function to redefine validNorm
1326 // (the solid can be concave)
1327
1328 G4double distOut =
1329 G4TessellatedSolid::DistanceToOut(p, v, calcNorm, validNorm, n);
1330 if (validNorm) { *validNorm = fIsConvex; }
1331
1332 return distOut;
1333}

1336
1337G4double G4ExtrudedSolid::DistanceToOut(const G4ThreeVector &p) const
1338{
1339 switch (fSolidType)
1340 {
1341 case 1: // convex right prism
1342 {
1343 G4double dist = std::max(fZSections[0].fZ-p.z(),p.z()-fZSections[1].fZ);
1344 G4int np = fPlanes.size();
1345 for (G4int i=0; i<np; ++i)
1346 {
1347 G4double dd = fPlanes[i].a*p.x() + fPlanes[i].b*p.y() + fPlanes[i].d;
1348 if (dd > dist) dist = dd;
1349 }
1350 return (dist < 0) ? -dist : 0.;
1351 }
1352 case 2: // non-convex right prism
1353 {
1354 G4double distz = std::max(fZSections[0].fZ-p.z(),p.z()-fZSections[1].fZ);
1355 G4bool in = PointInPolygon(p);
1356 if (distz >= 0 || (!in)) return 0; // point is outside
1357 return std::min(-distz,std::sqrt(DistanceToPolygonSqr(p)));
1358 }
1359 }
1360
1361 // General case: use tessellated solid
1362 return G4TessellatedSolid::DistanceToOut(p);
1363}

1366// Get bounding box
1367
1368void G4ExtrudedSolid::BoundingLimits(G4ThreeVector& pMin,
1369 G4ThreeVector& pMax) const
1370{
1371 G4double xmin0 = kInfinity, xmax0 = -kInfinity;
1372 G4double ymin0 = kInfinity, ymax0 = -kInfinity;
1373
1374 for (G4int i=0; i<GetNofVertices(); ++i)
1375 {
1376 G4double x = fPolygon[i].x();
1377 if (x < xmin0) xmin0 = x;
1378 if (x > xmax0) xmax0 = x;
1379 G4double y = fPolygon[i].y();
1380 if (y < ymin0) ymin0 = y;
1381 if (y > ymax0) ymax0 = y;
1382 }
1383
1384 G4double xmin = kInfinity, xmax = -kInfinity;
1385 G4double ymin = kInfinity, ymax = -kInfinity;
1386
1387 G4int nsect = GetNofZSections();
1388 for (G4int i=0; i<nsect; ++i)
1389 {
1390 ZSection zsect = GetZSection(i);
1391 G4double dx = zsect.fOffset.x();
1392 G4double dy = zsect.fOffset.y();
1393 G4double scale = zsect.fScale;
1394 xmin = std::min(xmin,xmin0*scale+dx);
1395 xmax = std::max(xmax,xmax0*scale+dx);
1396 ymin = std::min(ymin,ymin0*scale+dy);
1397 ymax = std::max(ymax,ymax0*scale+dy);
1398 }
1399
1400 G4double zmin = GetZSection(0).fZ;
1401 G4double zmax = GetZSection(nsect-1).fZ;
1402
1403 pMin.set(xmin,ymin,zmin);
1404 pMax.set(xmax,ymax,zmax);
1405
1406 // Check correctness of the bounding box
1407 //
1408 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
1409 {
1410 std::ostringstream message;
1411 message << "Bad bounding box (min >= max) for solid: "
1412 << GetName() << " !"
1413 << "\npMin = " << pMin
1414 << "\npMax = " << pMax;
1415 G4Exception("G4ExtrudedSolid::BoundingLimits()",
1416 "GeomMgt0001", JustWarning, message);
1417 DumpInfo();
1418 }
1419}

1422// Calculate extent under transform and specified limit
1423
1424G4bool
1425G4ExtrudedSolid::CalculateExtent(const EAxis pAxis,
1426 const G4VoxelLimits& pVoxelLimit,
1427 const G4AffineTransform& pTransform,
1428 G4double& pMin, G4double& pMax) const
1429{
1430 G4ThreeVector bmin, bmax;
1431 G4bool exist;
1432
1433 // Check bounding box (bbox)
1434 //
1435 BoundingLimits(bmin,bmax);
1436 G4BoundingEnvelope bbox(bmin,bmax);
1437#ifdef G4BBOX_EXTENT
1438 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
1439#endif
1440 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
1441 {
1442 return exist = (pMin < pMax) ? true : false;
1443 }
1444
1445 // To find the extent, the base polygon is subdivided in triangles.
1446 // The extent is calculated as cumulative extent of the parts
1447 // formed by extrusion of the triangles
1448 //
1449 G4TwoVectorList triangles;
1450 G4double eminlim = pVoxelLimit.GetMinExtent(pAxis);
1451 G4double emaxlim = pVoxelLimit.GetMaxExtent(pAxis);
1452
1453 // triangulate the base polygon
1454 if (!G4GeomTools::TriangulatePolygon(fPolygon,triangles))
1455 {
1456 std::ostringstream message;
1457 message << "Triangulation of the base polygon has failed for solid: "
1458 << GetName() << " !"
1459 << "\nExtent has been calculated using boundary box";
1460 G4Exception("G4ExtrudedSolid::CalculateExtent()",
1461 "GeomMgt1002",JustWarning,message);
1462 return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
1463 }
1464
1465 // allocate vector lists
1466 G4int nsect = GetNofZSections();
1467 std::vector<const G4ThreeVectorList *> polygons;
1468 polygons.resize(nsect);
1469 for (G4int k=0; k<nsect; ++k) { polygons[k] = new G4ThreeVectorList(3); }
1470
1471 // main loop along triangles
1472 pMin = kInfinity;
1473 pMax = -kInfinity;
1474 G4int ntria = triangles.size()/3;
1475 for (G4int i=0; i<ntria; ++i)
1476 {
1477 G4int i3 = i*3;
1478 for (G4int k=0; k<nsect; ++k) // extrude triangle
1479 {
1480 ZSection zsect = GetZSection(k);
1481 G4double z = zsect.fZ;
1482 G4double dx = zsect.fOffset.x();
1483 G4double dy = zsect.fOffset.y();
1484 G4double scale = zsect.fScale;
1485
1486 G4ThreeVectorList* ptr = const_cast<G4ThreeVectorList*>(polygons[k]);
1487 G4ThreeVectorList::iterator iter = ptr->begin();
1488 G4double x0 = triangles[i3+0].x()*scale+dx;
1489 G4double y0 = triangles[i3+0].y()*scale+dy;
1490 iter->set(x0,y0,z);
1491 iter++;
1492 G4double x1 = triangles[i3+1].x()*scale+dx;
1493 G4double y1 = triangles[i3+1].y()*scale+dy;
1494 iter->set(x1,y1,z);
1495 iter++;
1496 G4double x2 = triangles[i3+2].x()*scale+dx;
1497 G4double y2 = triangles[i3+2].y()*scale+dy;
1498 iter->set(x2,y2,z);
1499 }
1500
1501 // set sub-envelope and adjust extent
1502 G4double emin,emax;
1503 G4BoundingEnvelope benv(polygons);
1504 if (!benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,emin,emax)) continue;
1505 if (emin < pMin) pMin = emin;
1506 if (emax > pMax) pMax = emax;
1507 if (eminlim > pMin && emaxlim < pMax) break; // max possible extent
1508 }
1509 // free memory
1510 for (G4int k=0; k<nsect; ++k) { delete polygons[k]; polygons[k]=0;}
1511 return (pMin < pMax);
1512}
