59//
60// Constructor (GEANT3 style parameters)
61//
62G4Polycone::G4Polycone( const G4String& name,
63 G4double phiStart,
64 G4double phiTotal,
65 G4int numZPlanes,
66 const G4double zPlane[],
67 const G4double rInner[],
68 const G4double rOuter[] )
69 : G4VCSGfaceted( name )
70{
71 //
72 // Some historical ugliness
73 //
74 original_parameters = new G4PolyconeHistorical();
75
76 original_parameters->Start_angle = phiStart;
77 original_parameters->Opening_angle = phiTotal;
78 original_parameters->Num_z_planes = numZPlanes;
79 original_parameters->Z_values = new G4double[numZPlanes];
80 original_parameters->Rmin = new G4double[numZPlanes];
81 original_parameters->Rmax = new G4double[numZPlanes];
82
83 G4int i;
84 for (i=0; i<numZPlanes; i++)
85 {
86 if(rInner[i]>rOuter[i])
87 {
88 DumpInfo();
89 std::ostringstream message;
90 message << "Cannot create a Polycone with rInner > rOuter for the same Z"
91 << G4endl
92 << " rInner > rOuter for the same Z !" << G4endl
93 << " rMin[" << i << "] = " << rInner[i]
94 << " -- rMax[" << i << "] = " << rOuter[i];
95 G4Exception("G4Polycone::G4Polycone()", "GeomSolids0002",
96 FatalErrorInArgument, message);
97 }
98 if (( i < numZPlanes-1) && ( zPlane[i] == zPlane[i+1] ))
99 {
100 if( (rInner[i] > rOuter[i+1])
101 ||(rInner[i+1] > rOuter[i]) )
102 {
103 DumpInfo();
104 std::ostringstream message;
105 message << "Cannot create a Polycone with no contiguous segments."
106 << G4endl
107 << " Segments are not contiguous !" << G4endl
108 << " rMin[" << i << "] = " << rInner[i]
109 << " -- rMax[" << i+1 << "] = " << rOuter[i+1] << G4endl
110 << " rMin[" << i+1 << "] = " << rInner[i+1]
111 << " -- rMax[" << i << "] = " << rOuter[i];
112 G4Exception("G4Polycone::G4Polycone()", "GeomSolids0002",
113 FatalErrorInArgument, message);
114 }
115 }
116 original_parameters->Z_values[i] = zPlane[i];
117 original_parameters->Rmin[i] = rInner[i];
118 original_parameters->Rmax[i] = rOuter[i];
119 }
120
121 //
122 // Build RZ polygon using special PCON/PGON GEANT3 constructor
123 //
124 G4ReduciblePolygon *rz =
125 new G4ReduciblePolygon( rInner, rOuter, zPlane, numZPlanes );
126
127 //
128 // Do the real work
129 //
130 Create( phiStart, phiTotal, rz );
131
132 delete rz;
133}

136//
137// Constructor (generic parameters)
138//
139G4Polycone::G4Polycone( const G4String& name,
140 G4double phiStart,
141 G4double phiTotal,
142 G4int numRZ,
143 const G4double r[],
144 const G4double z[] )
145 : G4VCSGfaceted( name )
146{
147
148 G4ReduciblePolygon *rz = new G4ReduciblePolygon( r, z, numRZ );
149
150 Create( phiStart, phiTotal, rz );
151
152 // Set original_parameters struct for consistency
153 //
154
155 G4bool convertible=SetOriginalParameters(rz);
156
157 if(!convertible)
158 {
159 std::ostringstream message;
160 message << "Polycone " << GetName() << "cannot be converted" << G4endl
161 << "to Polycone with (Rmin,Rmaz,Z) parameters!";
162 G4Exception("G4Polycone::G4Polycone()", "GeomSolids0002",
163 FatalException, message, "Use G4GenericPolycone instead!");
164 }
165 else
166 {
167 G4cout << "INFO: Converting polycone " << GetName() << G4endl
168 << "to optimized polycone with (Rmin,Rmaz,Z) parameters !"
169 << G4endl;
170 }
171 delete rz;
172}

175// Create
176//
177// Generic create routine, called by each constructor after
178// conversion of arguments
179//
180void G4Polycone::Create( G4double phiStart,
181 G4double phiTotal,
182 G4ReduciblePolygon *rz )
183{
184 //
185 // Perform checks of rz values
186 //
187 if (rz->Amin() < 0.0)
188 {
189 std::ostringstream message;
190 message << "Illegal input parameters - " << GetName() << G4endl
191 << " All R values must be >= 0 !";
192 G4Exception("G4Polycone::Create()", "GeomSolids0002",
193 FatalErrorInArgument, message);
194 }
195
196 G4double rzArea = rz->Area();
197 if (rzArea < -kCarTolerance)
198 {
199 rz->ReverseOrder();
200 }
201 else if (rzArea < kCarTolerance)
202 {
203 std::ostringstream message;
204 message << "Illegal input parameters - " << GetName() << G4endl
205 << " R/Z cross section is zero or near zero: " << rzArea;
206 G4Exception("G4Polycone::Create()", "GeomSolids0002",
207 FatalErrorInArgument, message);
208 }
209
210 if ( (!rz->RemoveDuplicateVertices( kCarTolerance ))
211 || (!rz->RemoveRedundantVertices( kCarTolerance )) )
212 {
213 std::ostringstream message;
214 message << "Illegal input parameters - " << GetName() << G4endl
215 << " Too few unique R/Z values !";
216 G4Exception("G4Polycone::Create()", "GeomSolids0002",
217 FatalErrorInArgument, message);
218 }
219
220 if (rz->CrossesItself(1/kInfinity))
221 {
222 std::ostringstream message;
223 message << "Illegal input parameters - " << GetName() << G4endl
224 << " R/Z segments cross !";
225 G4Exception("G4Polycone::Create()", "GeomSolids0002",
226 FatalErrorInArgument, message);
227 }
228
229 numCorner = rz->NumVertices();
230
231 //
232 // Phi opening? Account for some possible roundoff, and interpret
233 // nonsense value as representing no phi opening
234 //
235 if (phiTotal <= 0 || phiTotal > twopi-1E-10)
236 {
237 phiIsOpen = false;
238 startPhi = 0;
239 endPhi = twopi;
240 }
241 else
242 {
243 phiIsOpen = true;
244
245 //
246 // Convert phi into our convention
247 //
248 startPhi = phiStart;
249 while( startPhi < 0 ) // Loop checking, 13.08.2015, G.Cosmo
250 startPhi += twopi;
251
252 endPhi = phiStart+phiTotal;
253 while( endPhi < startPhi ) // Loop checking, 13.08.2015, G.Cosmo
254 endPhi += twopi;
255 }
256
257 //
258 // Allocate corner array.
259 //
260 corners = new G4PolyconeSideRZ[numCorner];
261
262 //
263 // Copy corners
264 //
265 G4ReduciblePolygonIterator iterRZ(rz);
266
267 G4PolyconeSideRZ *next = corners;
268 iterRZ.Begin();
269 do // Loop checking, 13.08.2015, G.Cosmo
270 {
271 next->r = iterRZ.GetA();
272 next->z = iterRZ.GetB();
273 } while( ++next, iterRZ.Next() );
274
275 //
276 // Allocate face pointer array
277 //
278 numFace = phiIsOpen ? numCorner+2 : numCorner;
279 faces = new G4VCSGface*[numFace];
280
281 //
282 // Construct conical faces
283 //
284 // But! Don't construct a face if both points are at zero radius!
285 //
286 G4PolyconeSideRZ *corner = corners,
287 *prev = corners + numCorner-1,
288 *nextNext;
289 G4VCSGface **face = faces;
290 do // Loop checking, 13.08.2015, G.Cosmo
291 {
292 next = corner+1;
293 if (next >= corners+numCorner) next = corners;
294 nextNext = next+1;
295 if (nextNext >= corners+numCorner) nextNext = corners;
296
297 if (corner->r < 1/kInfinity && next->r < 1/kInfinity) continue;
298
299 //
300 // We must decide here if we can dare declare one of our faces
301 // as having a "valid" normal (i.e. allBehind = true). This
302 // is never possible if the face faces "inward" in r.
303 //
304 G4bool allBehind;
305 if (corner->z > next->z)
306 {
307 allBehind = false;
308 }
309 else
310 {
311 //
312 // Otherwise, it is only true if the line passing
313 // through the two points of the segment do not
314 // split the r/z cross section
315 //
316 allBehind = !rz->BisectedBy( corner->r, corner->z,
317 next->r, next->z, kCarTolerance );
318 }
319
320 *face++ = new G4PolyconeSide( prev, corner, next, nextNext,
321 startPhi, endPhi-startPhi, phiIsOpen, allBehind );
322 } while( prev=corner, corner=next, corner > corners );
323
324 if (phiIsOpen)
325 {
326 //
327 // Construct phi open edges
328 //
329 *face++ = new G4PolyPhiFace( rz, startPhi, 0, endPhi );
330 *face++ = new G4PolyPhiFace( rz, endPhi, 0, startPhi );
331 }
332
333 //
334 // We might have dropped a face or two: recalculate numFace
335 //
336 numFace = face-faces;
337
338 //
339 // Make enclosingCylinder
340 //
341 enclosingCylinder =
342 new G4EnclosingCylinder( rz, phiIsOpen, phiStart, phiTotal );
343}

399//
400// CopyStuff
401//
402void G4Polycone::CopyStuff( const G4Polycone &source )
403{
404 //
405 // Simple stuff
406 //
407 startPhi = source.startPhi;
408 endPhi = source.endPhi;
409 phiIsOpen = source.phiIsOpen;
410 numCorner = source.numCorner;
411
412 //
413 // The corner array
414 //
415 corners = new G4PolyconeSideRZ[numCorner];
416
417 G4PolyconeSideRZ *corn = corners,
418 *sourceCorn = source.corners;
419 do // Loop checking, 13.08.2015, G.Cosmo
420 {
421 *corn = *sourceCorn;
422 } while( ++sourceCorn, ++corn < corners+numCorner );
423
424 //
425 // Original parameters
426 //
427 if (source.original_parameters)
428 {
429 original_parameters =
430 new G4PolyconeHistorical( *source.original_parameters );
431 }
432
433 //
434 // Enclosing cylinder
435 //
436 enclosingCylinder = new G4EnclosingCylinder( *source.enclosingCylinder );
437
438 fRebuildPolyhedron = false;
439 fpPolyhedron = 0;
440}

446G4bool G4Polycone::Reset()
447{
448 //
449 // Clear old setup
450 //
451 G4VCSGfaceted::DeleteStuff();
452 delete [] corners;
453 delete enclosingCylinder;
454
455 //
456 // Rebuild polycone
457 //
458 G4ReduciblePolygon *rz =
459 new G4ReduciblePolygon( original_parameters->Rmin,
460 original_parameters->Rmax,
461 original_parameters->Z_values,
462 original_parameters->Num_z_planes );
463 Create( original_parameters->Start_angle,
464 original_parameters->Opening_angle, rz );
465 delete rz;
466
467 return 0;
468}

521//////////////////////////////////////////////////////////////////////////
522//
523// Get bounding box
524
525void G4Polycone::BoundingLimits(G4ThreeVector& pMin,
526 G4ThreeVector& pMax) const
527{
528 G4double rmin = kInfinity, rmax = -kInfinity;
529 G4double zmin = kInfinity, zmax = -kInfinity;
530
531 for (G4int i=0; i<GetNumRZCorner(); ++i)
532 {
533 G4PolyconeSideRZ corner = GetCorner(i);
534 if (corner.r < rmin) rmin = corner.r;
535 if (corner.r > rmax) rmax = corner.r;
536 if (corner.z < zmin) zmin = corner.z;
537 if (corner.z > zmax) zmax = corner.z;
538 }
539
540 if (IsOpen())
541 {
542 G4TwoVector vmin,vmax;
543 G4GeomTools::DiskExtent(rmin,rmax,
544 GetSinStartPhi(),GetCosStartPhi(),
545 GetSinEndPhi(),GetCosEndPhi(),
546 vmin,vmax);
547 pMin.set(vmin.x(),vmin.y(),zmin);
548 pMax.set(vmax.x(),vmax.y(),zmax);
549 }
550 else
551 {
552 pMin.set(-rmax,-rmax, zmin);
553 pMax.set( rmax, rmax, zmax);
554 }
555
556 // Check correctness of the bounding box
557 //
558 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
559 {
560 std::ostringstream message;
561 message << "Bad bounding box (min >= max) for solid: "
562 << GetName() << " !"
563 << "\npMin = " << pMin
564 << "\npMax = " << pMax;
565 G4Exception("G4Polycone::BoundingLimits()", "GeomMgt0001",
566 JustWarning, message);
567 DumpInfo();
568 }
569}

571//////////////////////////////////////////////////////////////////////////
572//
573// Calculate extent under transform and specified limit
574
575G4bool G4Polycone::CalculateExtent(const EAxis pAxis,
576 const G4VoxelLimits& pVoxelLimit,
577 const G4AffineTransform& pTransform,
578 G4double& pMin, G4double& pMax) const
579{
580 G4ThreeVector bmin, bmax;
581 G4bool exist;
582
583 // Check bounding box (bbox)
584 //
585 BoundingLimits(bmin,bmax);
586 G4BoundingEnvelope bbox(bmin,bmax);
587#ifdef G4BBOX_EXTENT
588 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
589#endif
590 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
591 {
592 return exist = (pMin < pMax) ? true : false;
593 }
594
595 // To find the extent, RZ contour of the polycone is subdivided
596 // in triangles. The extent is calculated as cumulative extent of
597 // all sub-polycones formed by rotation of triangles around Z
598 //
599 G4TwoVectorList contourRZ;
600 G4TwoVectorList triangles;
601 std::vector<G4int> iout;
602 G4double eminlim = pVoxelLimit.GetMinExtent(pAxis);
603 G4double emaxlim = pVoxelLimit.GetMaxExtent(pAxis);
604
605 // get RZ contour, ensure anticlockwise order of corners
606 for (G4int i=0; i<GetNumRZCorner(); ++i)
607 {
608 G4PolyconeSideRZ corner = GetCorner(i);
609 contourRZ.push_back(G4TwoVector(corner.r,corner.z));
610 }
611 G4GeomTools::RemoveRedundantVertices(contourRZ,iout,2*kCarTolerance);
612 G4double area = G4GeomTools::PolygonArea(contourRZ);
613 if (area < 0.) std::reverse(contourRZ.begin(),contourRZ.end());
614
615 // triangulate RZ countour
616 if (!G4GeomTools::TriangulatePolygon(contourRZ,triangles))
617 {
618 std::ostringstream message;
619 message << "Triangulation of RZ contour has failed for solid: "
620 << GetName() << " !"
621 << "\nExtent has been calculated using boundary box";
622 G4Exception("G4Polycone::CalculateExtent()",
623 "GeomMgt1002", JustWarning, message);
624 return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
625 }
626
627 // set trigonometric values
628 const G4int NSTEPS = 24; // number of steps for whole circle
629 G4double astep = twopi/NSTEPS; // max angle for one step
630
631 G4double sphi = GetStartPhi();
632 G4double ephi = GetEndPhi();
633 G4double dphi = IsOpen() ? ephi-sphi : twopi;
634 G4int ksteps = (dphi <= astep) ? 1 : (G4int)((dphi-deg)/astep) + 1;
635 G4double ang = dphi/ksteps;
636
637 G4double sinHalf = std::sin(0.5*ang);
638 G4double cosHalf = std::cos(0.5*ang);
639 G4double sinStep = 2.*sinHalf*cosHalf;
640 G4double cosStep = 1. - 2.*sinHalf*sinHalf;
641
642 G4double sinStart = GetSinStartPhi();
643 G4double cosStart = GetCosStartPhi();
644 G4double sinEnd = GetSinEndPhi();
645 G4double cosEnd = GetCosEndPhi();
646
647 // define vectors and arrays
648 std::vector<const G4ThreeVectorList *> polygons;
649 polygons.resize(ksteps+2);
650 G4ThreeVectorList pols[NSTEPS+2];
651 for (G4int k=0; k<ksteps+2; ++k) pols[k].resize(6);
652 for (G4int k=0; k<ksteps+2; ++k) polygons[k] = &pols[k];
653 G4double r0[6],z0[6]; // contour with original edges of triangle
654 G4double r1[6]; // shifted radii of external edges of triangle
655
656 // main loop along triangles
657 pMin = kInfinity;
658 pMax =-kInfinity;
659 G4int ntria = triangles.size()/3;
660 for (G4int i=0; i<ntria; ++i)
661 {
662 G4int i3 = i*3;
663 for (G4int k=0; k<3; ++k)
664 {
665 G4int e0 = i3+k, e1 = (k<2) ? e0+1 : i3;
666 G4int k2 = k*2;
667 // set contour with original edges of triangle
668 r0[k2+0] = triangles[e0].x(); z0[k2+0] = triangles[e0].y();
669 r0[k2+1] = triangles[e1].x(); z0[k2+1] = triangles[e1].y();
670 // set shifted radii
671 r1[k2+0] = r0[k2+0];
672 r1[k2+1] = r0[k2+1];
673 if (z0[k2+1] - z0[k2+0] <= 0) continue;
674 r1[k2+0] /= cosHalf;
675 r1[k2+1] /= cosHalf;
676 }
677
678 // rotate countour, set sequence of 6-sided polygons
679 G4double sinCur = sinStart*cosHalf + cosStart*sinHalf;
680 G4double cosCur = cosStart*cosHalf - sinStart*sinHalf;
681 for (G4int j=0; j<6; ++j) pols[0][j].set(r0[j]*cosStart,r0[j]*sinStart,z0[j]);
682 for (G4int k=1; k<ksteps+1; ++k)
683 {
684 for (G4int j=0; j<6; ++j) pols[k][j].set(r1[j]*cosCur,r1[j]*sinCur,z0[j]);
685 G4double sinTmp = sinCur;
686 sinCur = sinCur*cosStep + cosCur*sinStep;
687 cosCur = cosCur*cosStep - sinTmp*sinStep;
688 }
689 for (G4int j=0; j<6; ++j) pols[ksteps+1][j].set(r0[j]*cosEnd,r0[j]*sinEnd,z0[j]);
690
691 // set sub-envelope and adjust extent
692 G4double emin,emax;
693 G4BoundingEnvelope benv(polygons);
694 if (!benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,emin,emax)) continue;
695 if (emin < pMin) pMin = emin;
696 if (emax > pMax) pMax = emax;
697 if (eminlim > pMin && emaxlim < pMax) return true; // max possible extent
698 }
699 return (pMin < pMax);
700}

728//
729// Stream object contents to an output stream
730//
731std::ostream& G4Polycone::StreamInfo( std::ostream& os ) const
732{
733 G4int oldprc = os.precision(16);
734 os << "-----------------------------------------------------------\n"
735 << " *** Dump for solid - " << GetName() << " ***\n"
736 << " ===================================================\n"
737 << " Solid type: G4Polycone\n"
738 << " Parameters: \n"
739 << " starting phi angle : " << startPhi/degree << " degrees \n"
740 << " ending phi angle : " << endPhi/degree << " degrees \n";
741 G4int i=0;
742
743 G4int numPlanes = original_parameters->Num_z_planes;
744 os << " number of Z planes: " << numPlanes << "\n"
745 << " Z values: \n";

778//
779// GetPointOnCone
780//
781// Auxiliary method for Get Point On Surface
782//
783G4ThreeVector G4Polycone::GetPointOnCone(G4double fRmin1, G4double fRmax1,
784 G4double fRmin2, G4double fRmax2,
785 G4double zOne, G4double zTwo,
786 G4double& totArea) const
787{
788 // declare working variables
789 //
790 G4double Aone, Atwo, Afive, phi, zRand, fDPhi, cosu, sinu;
791 G4double rRand1, rmin, rmax, chose, rone, rtwo, qone, qtwo;
792 G4double fDz=(zTwo-zOne)/2., afDz=std::fabs(fDz);
793 G4ThreeVector point, offset=G4ThreeVector(0.,0.,0.5*(zTwo+zOne));
794 fDPhi = endPhi - startPhi;
795 rone = (fRmax1-fRmax2)/(2.*fDz);
796 rtwo = (fRmin1-fRmin2)/(2.*fDz);
797 if(fRmax1==fRmax2){qone=0.;}
798 else
799 {
800 qone = fDz*(fRmax1+fRmax2)/(fRmax1-fRmax2);
801 }
802 if(fRmin1==fRmin2){qtwo=0.;}
803 else
804 {
805 qtwo = fDz*(fRmin1+fRmin2)/(fRmin1-fRmin2);
806 }
807 Aone = 0.5*fDPhi*(fRmax2 + fRmax1)*(sqr(fRmin1-fRmin2)+sqr(zTwo-zOne));
808 Atwo = 0.5*fDPhi*(fRmin2 + fRmin1)*(sqr(fRmax1-fRmax2)+sqr(zTwo-zOne));
809 Afive = fDz*(fRmax1-fRmin1+fRmax2-fRmin2);
810 totArea = Aone+Atwo+2.*Afive;
811
812 phi = G4RandFlat::shoot(startPhi,endPhi);
813 cosu = std::cos(phi);
814 sinu = std::sin(phi);
815
816
817 if( (startPhi == 0) && (endPhi == twopi) ) { Afive = 0; }
818 chose = G4RandFlat::shoot(0.,Aone+Atwo+2.*Afive);
819 if( (chose >= 0) && (chose < Aone) )
820 {
821 if(fRmax1 != fRmax2)
822 {
823 zRand = G4RandFlat::shoot(-1.*afDz,afDz);
824 point = G4ThreeVector (rone*cosu*(qone-zRand),
825 rone*sinu*(qone-zRand), zRand);
826 }
827 else
828 {
829 point = G4ThreeVector(fRmax1*cosu, fRmax1*sinu,
830 G4RandFlat::shoot(-1.*afDz,afDz));
831
832 }
833 }
834 else if(chose >= Aone && chose < Aone + Atwo)
835 {
836 if(fRmin1 != fRmin2)
837 {
838 zRand = G4RandFlat::shoot(-1.*afDz,afDz);
839 point = G4ThreeVector (rtwo*cosu*(qtwo-zRand),
840 rtwo*sinu*(qtwo-zRand), zRand);
841
842 }
843 else
844 {
845 point = G4ThreeVector(fRmin1*cosu, fRmin1*sinu,
846 G4RandFlat::shoot(-1.*afDz,afDz));
847 }
848 }
849 else if( (chose >= Aone + Atwo + Afive) && (chose < Aone + Atwo + 2.*Afive) )
850 {
851 zRand = G4RandFlat::shoot(-1.*afDz,afDz);
852 rmin = fRmin2-((zRand-fDz)/(2.*fDz))*(fRmin1-fRmin2);
853 rmax = fRmax2-((zRand-fDz)/(2.*fDz))*(fRmax1-fRmax2);
854 rRand1 = std::sqrt(G4RandFlat::shoot()*(sqr(rmax)-sqr(rmin))+sqr(rmin));
855 point = G4ThreeVector (rRand1*std::cos(startPhi),
856 rRand1*std::sin(startPhi), zRand);
857 }
858 else
859 {
860 zRand = G4RandFlat::shoot(-1.*afDz,afDz);
861 rmin = fRmin2-((zRand-fDz)/(2.*fDz))*(fRmin1-fRmin2);
862 rmax = fRmax2-((zRand-fDz)/(2.*fDz))*(fRmax1-fRmax2);
863 rRand1 = std::sqrt(G4RandFlat::shoot()*(sqr(rmax)-sqr(rmin))+sqr(rmin));
864 point = G4ThreeVector (rRand1*std::cos(endPhi),
865 rRand1*std::sin(endPhi), zRand);
866
867 }
868
869 return point+offset;
870}

873//
874// GetPointOnTubs
875//
876// Auxiliary method for GetPoint On Surface
877//
878G4ThreeVector G4Polycone::GetPointOnTubs(G4double fRMin, G4double fRMax,
879 G4double zOne, G4double zTwo,
880 G4double& totArea) const
881{
882 G4double xRand,yRand,zRand,phi,cosphi,sinphi,chose,
883 aOne,aTwo,aFou,rRand,fDz,fSPhi,fDPhi;
884 fDz = std::fabs(0.5*(zTwo-zOne));
885 fSPhi = startPhi;
886 fDPhi = endPhi-startPhi;
887
888 aOne = 2.*fDz*fDPhi*fRMax;
889 aTwo = 2.*fDz*fDPhi*fRMin;
890 aFou = 2.*fDz*(fRMax-fRMin);
891 totArea = aOne+aTwo+2.*aFou;
892 phi = G4RandFlat::shoot(startPhi,endPhi);
893 cosphi = std::cos(phi);
894 sinphi = std::sin(phi);
895 rRand = fRMin + (fRMax-fRMin)*std::sqrt(G4RandFlat::shoot());
896
897 if(startPhi == 0 && endPhi == twopi)
898 aFou = 0;
899
900 chose = G4RandFlat::shoot(0.,aOne+aTwo+2.*aFou);
901 if( (chose >= 0) && (chose < aOne) )
902 {
903 xRand = fRMax*cosphi;
904 yRand = fRMax*sinphi;
905 zRand = G4RandFlat::shoot(-1.*fDz,fDz);
906 return G4ThreeVector(xRand, yRand, zRand+0.5*(zTwo+zOne));
907 }
908 else if( (chose >= aOne) && (chose < aOne + aTwo) )
909 {
910 xRand = fRMin*cosphi;
911 yRand = fRMin*sinphi;
912 zRand = G4RandFlat::shoot(-1.*fDz,fDz);
913 return G4ThreeVector(xRand, yRand, zRand+0.5*(zTwo+zOne));
914 }
915 else if( (chose >= aOne+aTwo) && (chose <aOne+aTwo+aFou) )
916 {
917 xRand = rRand*std::cos(fSPhi+fDPhi);
918 yRand = rRand*std::sin(fSPhi+fDPhi);
919 zRand = G4RandFlat::shoot(-1.*fDz,fDz);
920 return G4ThreeVector(xRand, yRand, zRand+0.5*(zTwo+zOne));
921 }
922
923 // else
924
925 xRand = rRand*std::cos(fSPhi+fDPhi);
926 yRand = rRand*std::sin(fSPhi+fDPhi);
927 zRand = G4RandFlat::shoot(-1.*fDz,fDz);
928 return G4ThreeVector(xRand, yRand, zRand+0.5*(zTwo+zOne));
929}

933// GetPointOnRing
934//
935// Auxiliary method for GetPoint On Surface
936//
937G4ThreeVector G4Polycone::GetPointOnRing(G4double fRMin1, G4double fRMax1,
938 G4double fRMin2,G4double fRMax2,
939 G4double zOne) const
940{
941 G4double xRand,yRand,phi,cosphi,sinphi,rRand1,rRand2,A1,Atot,rCh;
942 phi = G4RandFlat::shoot(startPhi,endPhi);
943 cosphi = std::cos(phi);
944 sinphi = std::sin(phi);
945
946 if(fRMin1==fRMin2)
947 {
948 rRand1 = fRMin1; A1=0.;
949 }
950 else
951 {
952 rRand1 = G4RandFlat::shoot(fRMin1,fRMin2);
953 A1=std::fabs(fRMin2*fRMin2-fRMin1*fRMin1);
954 }
955 if(fRMax1==fRMax2)
956 {
957 rRand2=fRMax1; Atot=A1;
958 }
959 else
960 {
961 rRand2 = G4RandFlat::shoot(fRMax1,fRMax2);
962 Atot = A1+std::fabs(fRMax2*fRMax2-fRMax1*fRMax1);
963 }
964 rCh = G4RandFlat::shoot(0.,Atot);
965
966 if(rCh>A1) { rRand1=rRand2; }
967
968 xRand = rRand1*cosphi;
969 yRand = rRand1*sinphi;
970
971 return G4ThreeVector(xRand, yRand, zOne);
972}

996//
997// GetPointOnSurface
998//
999G4ThreeVector G4Polycone::GetPointOnSurface() const
1000{
1001 G4double Area=0,totArea=0,Achose1=0,Achose2=0,phi,cosphi,sinphi,rRand;
1002 G4int i=0;
1003 G4int numPlanes = original_parameters->Num_z_planes;
1004
1005 phi = G4RandFlat::shoot(startPhi,endPhi);
1006 cosphi = std::cos(phi);
1007 sinphi = std::sin(phi);
1008
1009 rRand = original_parameters->Rmin[0] +
1010 ( (original_parameters->Rmax[0]-original_parameters->Rmin[0])
1011 * std::sqrt(G4RandFlat::shoot()) );
1012
1013 std::vector<G4double> areas; // (numPlanes+1);
1014 std::vector<G4ThreeVector> points; // (numPlanes-1);
1015
1016 areas.push_back(pi*(sqr(original_parameters->Rmax[0])
1017 -sqr(original_parameters->Rmin[0])));
1018
1019 for(i=0; i<numPlanes-1; i++)
1020 {
1021 Area = (original_parameters->Rmin[i]+original_parameters->Rmin[i+1])
1022 * std::sqrt(sqr(original_parameters->Rmin[i]
1023 -original_parameters->Rmin[i+1])+
1024 sqr(original_parameters->Z_values[i+1]
1025 -original_parameters->Z_values[i]));
1026
1027 Area += (original_parameters->Rmax[i]+original_parameters->Rmax[i+1])
1028 * std::sqrt(sqr(original_parameters->Rmax[i]
1029 -original_parameters->Rmax[i+1])+
1030 sqr(original_parameters->Z_values[i+1]
1031 -original_parameters->Z_values[i]));
1032
1033 Area *= 0.5*(endPhi-startPhi);
1034
1035 if(startPhi==0.&& endPhi == twopi)
1036 {
1037 Area += std::fabs(original_parameters->Z_values[i+1]
1038 -original_parameters->Z_values[i])*
1039 (original_parameters->Rmax[i]
1040 +original_parameters->Rmax[i+1]
1041 -original_parameters->Rmin[i]
1042 -original_parameters->Rmin[i+1]);
1043 }
1044 areas.push_back(Area);
1045 totArea += Area;
1046 }
1047
1048 areas.push_back(pi*(sqr(original_parameters->Rmax[numPlanes-1])-
1049 sqr(original_parameters->Rmin[numPlanes-1])));
1050
1051 totArea += (areas[0]+areas[numPlanes]);
1052 G4double chose = G4RandFlat::shoot(0.,totArea);
1053
1054 if( (chose>=0.) && (chose<areas[0]) )
1055 {
1056 return G4ThreeVector(rRand*cosphi, rRand*sinphi,
1057 original_parameters->Z_values[0]);
1058 }
1059
1060 for (i=0; i<numPlanes-1; i++)
1061 {
1062 Achose1 += areas[i];
1063 Achose2 = (Achose1+areas[i+1]);
1064 if(chose>=Achose1 && chose<Achose2)
1065 {
1066 return GetPointOnCut(original_parameters->Rmin[i],
1067 original_parameters->Rmax[i],
1068 original_parameters->Rmin[i+1],
1069 original_parameters->Rmax[i+1],
1070 original_parameters->Z_values[i],
1071 original_parameters->Z_values[i+1], Area);
1072 }
1073 }
1074
1075 rRand = original_parameters->Rmin[numPlanes-1] +
1076 ( (original_parameters->Rmax[numPlanes-1]-original_parameters->Rmin[numPlanes-1])
1077 * std::sqrt(G4RandFlat::shoot()) );
1078
1079 return G4ThreeVector(rRand*cosphi,rRand*sinphi,
1080 original_parameters->Z_values[numPlanes-1]);
1081
1082}

1102{
1103 G4int numPlanes = (G4int)numCorner;
1104 G4bool isConvertible=true;
1105 G4double Zmax=rz->Bmax();
1106 rz->StartWithZMin();
1107
1108 // Prepare vectors for storage
1109 //
1110 std::vector<G4double> Z;
1111 std::vector<G4double> Rmin;
1112 std::vector<G4double> Rmax;
1113
1114 G4int countPlanes=1;
1115 G4int icurr=0;
1116 G4int icurl=0;
1117
1118 // first plane Z=Z[0]
1119 //
1120 Z.push_back(corners[0].z);
1121 G4double Zprev=Z[0];
1122 if (Zprev == corners[1].z)
1123 {
1124 Rmin.push_back(corners[0].r);
1125 Rmax.push_back (corners[1].r);icurr=1;
1126 }
1127 else if (Zprev == corners[numPlanes-1].z)
1128 {
1129 Rmin.push_back(corners[numPlanes-1].r);
1130 Rmax.push_back (corners[0].r);
1131 icurl=numPlanes-1;
1132 }
1133 else
1134 {
1135 Rmin.push_back(corners[0].r);
1136 Rmax.push_back (corners[0].r);
1137 }
1138
1139 // next planes until last
1140 //
1141 G4int inextr=0, inextl=0;
1142 for (G4int i=0; i < numPlanes-2; i++)
1143 {
1144 inextr=1+icurr;
1145 inextl=(icurl <= 0)? numPlanes-1 : icurl-1;
1146
1147 if((corners[inextr].z >= Zmax) & (corners[inextl].z >= Zmax)) { break; }
1148
1149 G4double Zleft = corners[inextl].z;
1150 G4double Zright = corners[inextr].z;
1151 if(Zright > Zleft) // Next plane will be Zleft
1152 {
1153 Z.push_back(Zleft);
1154 countPlanes++;
1155 G4double difZr=corners[inextr].z - corners[icurr].z;
1156 G4double difZl=corners[inextl].z - corners[icurl].z;
1157
1158 if(std::fabs(difZl) < kCarTolerance)
1159 {
1160 if(std::fabs(difZr) < kCarTolerance)
1161 {
1162 Rmin.push_back(corners[inextl].r);
1163 Rmax.push_back(corners[icurr].r);
1164 }
1165 else
1166 {
1167 Rmin.push_back(corners[inextl].r);
1168 Rmax.push_back(corners[icurr].r + (Zleft-corners[icurr].z)/difZr
1169 *(corners[inextr].r - corners[icurr].r));
1170 }
1171 }
1172 else if (difZl >= kCarTolerance)
1173 {
1174 if(std::fabs(difZr) < kCarTolerance)
1175 {
1176 Rmin.push_back(corners[icurl].r);
1177 Rmax.push_back(corners[icurr].r);
1178 }
1179 else
1180 {
1181 Rmin.push_back(corners[icurl].r);
1182 Rmax.push_back(corners[icurr].r + (Zleft-corners[icurr].z)/difZr
1183 *(corners[inextr].r - corners[icurr].r));
1184 }
1185 }
1186 else
1187 {
1188 isConvertible=false; break;
1189 }
1190 icurl=(icurl == 0)? numPlanes-1 : icurl-1;
1191 }
1192 else if(std::fabs(Zright-Zleft)<kCarTolerance) // Zright=Zleft
1193 {
1194 Z.push_back(Zleft);
1195 countPlanes++;
1196 icurr++;
1197
1198 icurl=(icurl == 0)? numPlanes-1 : icurl-1;
1199
1200 Rmin.push_back(corners[inextl].r);
1201 Rmax.push_back(corners[inextr].r);
1202 }
1203 else // Zright<Zleft

1205 Z.push_back(Zright);
1206 countPlanes++;

1209 G4double difZl=corners[inextl].z - corners[icurl].z;
1210 if(std::fabs(difZr) < kCarTolerance)

1222 }
1223 icurr++;
1224 } // plate
1225 else if (difZr >= kCarTolerance)
1226 {
1227 if(std::fabs(difZl) < kCarTolerance)
1228 {
1229 Rmax.push_back(corners[inextr].r);
1230 Rmin.push_back (corners[icurr].r);
1231 }
1232 else
1233 {
1234 Rmax.push_back(corners[inextr].r);
1235 Rmin.push_back (corners[icurl].r+(Zright-corners[icurl].z)/difZl
1236 * (corners[inextl].r - corners[icurl].r));
1237 }
1238 icurr++;
1239 }
1240 else
1241 {
1242 isConvertible=false; break;
1243 }
1244 }
1245 } // end for loop
1246
1247 // last plane Z=Zmax
1248 //
1249 Z.push_back(Zmax);
1250 countPlanes++;
1251 inextr=1+icurr;
1252 inextl=(icurl <= 0)? numPlanes-1 : icurl-1;
1253

1255 Rmin.push_back(corners[inextl].r);

1268 original_parameters->Z_values[j] = Z[j];
1269 original_parameters->Rmax[j] = Rmax[j];
1270 original_parameters->Rmin[j] = Rmin[j];
1271 }
1272 original_parameters->Start_angle = startPhi;
1273 original_parameters->Opening_angle = endPhi-startPhi;
1274 original_parameters->Num_z_planes = countPlanes;
1275
1276 }
1277 else // Set parameters(r,z) with Rmin==0 as convention
1278 {
1279#ifdef G4SPECSDEBUG

1292 {

1294 original_parameters->Rmax[j] = corners[j].r;
1295 original_parameters->Rmin[j] = 0.0;
1296 }
1297 original_parameters->Start_angle = startPhi;
1298 original_parameters->Opening_angle = endPhi-startPhi;
1299 original_parameters->Num_z_planes = numPlanes;
1300 }

1302}
