71///////////////////////////////////////////////////////////////////////////////
72//
73// Definition of triangular facet using absolute vectors to fVertices.
74// From this for first vector is retained to define the facet location and
75// two relative vectors (E0 and E1) define the sides and orientation of
76// the outward surface normal.
77//
78G4TriangularFacet::G4TriangularFacet (const G4ThreeVector &vt0,
79 const G4ThreeVector &vt1,
80 const G4ThreeVector &vt2,
81 G4FacetVertexType vertexType)
82 : fSqrDist(0.)
83{
84 fVertices = new vector<G4ThreeVector>(3);
85
86 SetVertex(0, vt0);
87 if (vertexType == ABSOLUTE)
88 {
89 SetVertex(1, vt1);
90 SetVertex(2, vt2);
91 fE1 = vt1 - vt0;
92 fE2 = vt2 - vt0;
93 }
94 else
95 {
96 SetVertex(1, vt0 + vt1);
97 SetVertex(2, vt0 + vt2);
98 fE1 = vt1;
99 fE2 = vt2;
100 }
101
102 G4ThreeVector E1xE2 = fE1.cross(fE2);
103 fArea = 0.5 * E1xE2.mag();
104 for (G4int i = 0; i < 3; ++i) fIndices[i] = -1;
105
106 fIsDefined = true;
107 G4double delta = kCarTolerance; // Set tolerance for checking
108
109 // Check length of edges
110 //
111 G4double leng1 = fE1.mag();
112 G4double leng2 = (fE2-fE1).mag();
113 G4double leng3 = fE2.mag();
114 if (leng1 <= delta || leng2 <= delta || leng3 <= delta)
115 {
116 fIsDefined = false;
117 }
118
119 // Check min height of triangle
120 //
121 if (fIsDefined)
122 {
123 if (2.*fArea/std::max(std::max(leng1,leng2),leng3) <= delta)
124 {
125 fIsDefined = false;
126 }
127 }
128
129 // Define facet
130 //
131 if (!fIsDefined)
132 {
133 ostringstream message;
134 message << "Facet is too small or too narrow." << G4endl
135 << "Triangle area = " << fArea << G4endl
136 << "P0 = " << GetVertex(0) << G4endl
137 << "P1 = " << GetVertex(1) << G4endl
138 << "P2 = " << GetVertex(2) << G4endl
139 << "Side1 length (P0->P1) = " << leng1 << G4endl
140 << "Side2 length (P1->P2) = " << leng2 << G4endl
141 << "Side3 length (P2->P0) = " << leng3;
142 G4Exception("G4TriangularFacet::G4TriangularFacet()",
143 "GeomSolids1001", JustWarning, message);
144 fSurfaceNormal.set(0,0,0);
145 fA = fB = fC = 0.0;
146 fDet = 0.0;
147 fCircumcentre = vt0 + 0.5*fE1 + 0.5*fE2;
148 fArea = fRadius = 0.0;
149 }
150 else
151 {
152 fSurfaceNormal = E1xE2.unit();
153 fA = fE1.mag2();
154 fB = fE1.dot(fE2);
155 fC = fE2.mag2();
156 fDet = std::fabs(fA*fC - fB*fB);
157
158 fCircumcentre =
159 vt0 + (E1xE2.cross(fE1)*fC + fE2.cross(E1xE2)*fA) / (2.*E1xE2.mag2());
160 fRadius = (fCircumcentre - vt0).mag();
161 }
162}

164///////////////////////////////////////////////////////////////////////////////
165//
166G4TriangularFacet::G4TriangularFacet ()
167 : fSqrDist(0.)
168{
169 fVertices = new vector<G4ThreeVector>(3);
170 G4ThreeVector zero(0,0,0);
171 SetVertex(0, zero);
172 SetVertex(1, zero);
173 SetVertex(2, zero);
174 for (G4int i = 0; i < 3; ++i) fIndices[i] = -1;
175 fIsDefined = false;
176 fSurfaceNormal.set(0,0,0);
177 fA = fB = fC = 0;
178 fE1 = zero;
179 fE2 = zero;
180 fDet = 0.0;
181 fArea = fRadius = 0.0;
182}

194{
195 char *p = (char *) &rhs;
196 copy(p, p + sizeof(*this), (char *)this);
197
198 if (fIndices[0] < 0 && fVertices)
199 {
200 fVertices = new vector<G4ThreeVector>(3);
201 for (G4int i = 0; i < 3; ++i) (*fVertices)[i] = (*rhs.fVertices)[i];

253///////////////////////////////////////////////////////////////////////////////
254//
255// Distance (G4ThreeVector)
256//
257// Determines the vector between p and the closest point on the facet to p.
258// This is based on the algorithm published in "Geometric Tools for Computer
259// Graphics," Philip J Scheider and David H Eberly, Elsevier Science (USA),
260// 2003. at the time of writing, the algorithm is also available in fA
261// technical note "Distance between point and triangle in 3D," by David Eberly
262// at http://www.geometrictools.com/Documentation/DistancePoint3Triangle3.pdf
263//
264// The by-product is the square-distance fSqrDist, which is retained
265// in case needed by the other "Distance" member functions.
266//
267G4ThreeVector G4TriangularFacet::Distance (const G4ThreeVector &p)
268{
269 G4ThreeVector D = GetVertex(0) - p;
270 G4double d = fE1.dot(D);
271 G4double e = fE2.dot(D);
272 G4double f = D.mag2();
273 G4double q = fB*e - fC*d;
274 G4double t = fB*d - fA*e;
275 fSqrDist = 0.;
276
277 if (q+t <= fDet)
278 {
279 if (q < 0.0)
280 {
281 if (t < 0.0)
282 {
283 //
284 // We are in region 4.
285 //
286 if (d < 0.0)
287 {
288 t = 0.0;
289 if (-d >= fA) {q = 1.0; fSqrDist = fA + 2.0*d + f;}
290 else {q = -d/fA; fSqrDist = d*q + f;}
291 }
292 else
293 {
294 q = 0.0;
295 if (e >= 0.0) {t = 0.0; fSqrDist = f;}
296 else if (-e >= fC) {t = 1.0; fSqrDist = fC + 2.0*e + f;}
297 else {t = -e/fC; fSqrDist = e*t + f;}
298 }
299 }
300 else
301 {
302 //
303 // We are in region 3.
304 //
305 q = 0.0;
306 if (e >= 0.0) {t = 0.0; fSqrDist = f;}
307 else if (-e >= fC) {t = 1.0; fSqrDist = fC + 2.0*e + f;}
308 else {t = -e/fC; fSqrDist = e*t + f;}
309 }
310 }
311 else if (t < 0.0)
312 {
313 //
314 // We are in region 5.
315 //
316 t = 0.0;
317 if (d >= 0.0) {q = 0.0; fSqrDist = f;}
318 else if (-d >= fA) {q = 1.0; fSqrDist = fA + 2.0*d + f;}
319 else {q = -d/fA; fSqrDist = d*q + f;}
320 }
321 else
322 {
323 //
324 // We are in region 0.
325 //
326 q = q / fDet;
327 t = t / fDet;
328 fSqrDist = q*(fA*q + fB*t + 2.0*d) + t*(fB*q + fC*t + 2.0*e) + f;
329 }
330 }
331 else
332 {
333 if (q < 0.0)
334 {
335 //
336 // We are in region 2.
337 //
338 G4double tmp0 = fB + d;
339 G4double tmp1 = fC + e;
340 if (tmp1 > tmp0)
341 {
342 G4double numer = tmp1 - tmp0;
343 G4double denom = fA - 2.0*fB + fC;
344 if (numer >= denom) {q = 1.0; t = 0.0; fSqrDist = fA + 2.0*d + f;}
345 else
346 {
347 q = numer/denom;
348 t = 1.0 - q;
349 fSqrDist = q*(fA*q + fB*t +2.0*d) + t*(fB*q + fC*t + 2.0*e) + f;
350 }
351 }
352 else
353 {
354 q = 0.0;
355 if (tmp1 <= 0.0) {t = 1.0; fSqrDist = fC + 2.0*e + f;}
356 else if (e >= 0.0) {t = 0.0; fSqrDist = f;}
357 else {t = -e/fC; fSqrDist = e*t + f;}
358 }
359 }
360 else if (t < 0.0)
361 {
362 //
363 // We are in region 6.
364 //
365 G4double tmp0 = fB + e;
366 G4double tmp1 = fA + d;
367 if (tmp1 > tmp0)
368 {
369 G4double numer = tmp1 - tmp0;
370 G4double denom = fA - 2.0*fB + fC;
371 if (numer >= denom) {t = 1.0; q = 0.0; fSqrDist = fC + 2.0*e + f;}
372 else
373 {
374 t = numer/denom;
375 q = 1.0 - t;
376 fSqrDist = q*(fA*q + fB*t +2.0*d) + t*(fB*q + fC*t + 2.0*e) + f;
377 }
378 }
379 else
380 {
381 t = 0.0;
382 if (tmp1 <= 0.0) {q = 1.0; fSqrDist = fA + 2.0*d + f;}
383 else if (d >= 0.0) {q = 0.0; fSqrDist = f;}
384 else {q = -d/fA; fSqrDist = d*q + f;}
385 }
386 }
387 else
388 //
389 // We are in region 1.
390 //
391 {
392 G4double numer = fC + e - fB - d;
393 if (numer <= 0.0)
394 {
395 q = 0.0;
396 t = 1.0;
397 fSqrDist = fC + 2.0*e + f;
398 }
399 else
400 {
401 G4double denom = fA - 2.0*fB + fC;
402 if (numer >= denom) {q = 1.0; t = 0.0; fSqrDist = fA + 2.0*d + f;}
403 else
404 {
405 q = numer/denom;
406 t = 1.0 - q;
407 fSqrDist = q*(fA*q + fB*t + 2.0*d) + t*(fB*q + fC*t + 2.0*e) + f;
408 }
409 }
410 }
411 }
412 //
413 //
414 // Do fA check for rounding errors in the distance-squared. It appears that
415 // the conventional methods for calculating fSqrDist breaks down when very
416 // near to or at the surface (as required by transport).
417 // We'll therefore also use the magnitude-squared of the vector displacement.
418 // (Note that I've also tried to get around this problem by using the
419 // existing equations for
420 //
421 // fSqrDist = function(fA,fB,fC,d,q,t)
422 //
423 // and use fA more accurate addition process which minimises errors and
424 // breakdown of cummutitivity [where (A+B)+C != A+(B+C)] but this still
425 // doesn't work.
426 // Calculation from u = D + q*fE1 + t*fE2 is less efficient, but appears
427 // more robust.
428 //
429 if (fSqrDist < 0.0) fSqrDist = 0.;
430 G4ThreeVector u = D + q*fE1 + t*fE2;
431 G4double u2 = u.mag2();
432 //
433 // The following (part of the roundoff error check) is from Oliver Merle'q
434 // updates.
435 //
436 if (fSqrDist > u2) fSqrDist = u2;
437
438 return u;
439}

441///////////////////////////////////////////////////////////////////////////////
442//
443// Distance (G4ThreeVector, G4double)
444//
445// Determines the closest distance between point p and the facet. This makes
446// use of G4ThreeVector G4TriangularFacet::Distance, which stores the
447// square of the distance in variable fSqrDist. If approximate methods show
448// the distance is to be greater than minDist, then forget about further
449// computation and return fA very large number.
450//
451G4double G4TriangularFacet::Distance (const G4ThreeVector &p,
452 G4double minDist)
453{
454 //
455 // Start with quicky test to determine if the surface of the sphere enclosing
456 // the triangle is any closer to p than minDist. If not, then don't bother
457 // about more accurate test.
458 //
459 G4double dist = kInfinity;
460 if ((p-fCircumcentre).mag()-fRadius < minDist)
461 {
462 //
463 // It's possible that the triangle is closer than minDist,
464 // so do more accurate assessment.
465 //
466 dist = Distance(p).mag();
467 }
468 return dist;
469}

471///////////////////////////////////////////////////////////////////////////////
472//
473// Distance (G4ThreeVector, G4double, G4bool)
474//
475// Determine the distance to point p. kInfinity is returned if either:
476// (1) outgoing is TRUE and the dot product of the normal vector to the facet
477// and the displacement vector from p to the triangle is negative.
478// (2) outgoing is FALSE and the dot product of the normal vector to the facet
479// and the displacement vector from p to the triangle is positive.
480// If approximate methods show the distance is to be greater than minDist, then
481// forget about further computation and return fA very large number.
482//
483// This method has been heavily modified thanks to the valuable comments and
484// corrections of Rickard Holmberg.
485//
486G4double G4TriangularFacet::Distance (const G4ThreeVector &p,
487 G4double minDist,
488 const G4bool outgoing)
489{
490 //
491 // Start with quicky test to determine if the surface of the sphere enclosing
492 // the triangle is any closer to p than minDist. If not, then don't bother
493 // about more accurate test.
494 //
495 G4double dist = kInfinity;
496 if ((p-fCircumcentre).mag()-fRadius < minDist)
497 {
498 //
499 // It's possible that the triangle is closer than minDist,
500 // so do more accurate assessment.
501 //
502 G4ThreeVector v = Distance(p);
503 G4double dist1 = sqrt(fSqrDist);
504 G4double dir = v.dot(fSurfaceNormal);
505 G4bool wrongSide = (dir > 0.0 && !outgoing) || (dir < 0.0 && outgoing);
506 if (dist1 <= kCarTolerance)
507 {
508 //
509 // Point p is very close to triangle. Check if it's on the wrong side,
510 // in which case return distance of 0.0 otherwise .
511 //
512 if (wrongSide) dist = 0.0;
513 else dist = dist1;
514 }
515 else if (!wrongSide) dist = dist1;
516 }
517 return dist;
518}

520///////////////////////////////////////////////////////////////////////////////
521//
522// Extent
523//
524// Calculates the furthest the triangle extends in fA particular direction
525// defined by the vector axis.
526//
527G4double G4TriangularFacet::Extent (const G4ThreeVector axis)
528{
529 G4double ss = GetVertex(0).dot(axis);
530 G4double sp = GetVertex(1).dot(axis);
531 if (sp > ss) ss = sp;
532 sp = GetVertex(2).dot(axis);
533 if (sp > ss) ss = sp;
534 return ss;
535}

537///////////////////////////////////////////////////////////////////////////////
538//
539// Intersect
540//
541// Member function to find the next intersection when going from p in the
542// direction of v. If:
543// (1) "outgoing" is TRUE, only consider the face if we are going out through
544// the face.
545// (2) "outgoing" is FALSE, only consider the face if we are going in through
546// the face.
547// Member functions returns TRUE if there is an intersection, FALSE otherwise.
548// Sets the distance (distance along w), distFromSurface (orthogonal distance)
549// and normal.
550//
551// Also considers intersections that happen with negative distance for small
552// distances of distFromSurface = 0.5*kCarTolerance in the wrong direction.
553// This is to detect kSurface without doing fA full Inside(p) in
554// G4TessellatedSolid::Distance(p,v) calculation.
555//
556// This member function is thanks the valuable work of Rickard Holmberg. PT.
557// However, "gotos" are the Work of the Devil have been exorcised with
558// extreme prejudice!!
559//
560// IMPORTANT NOTE: These calculations are predicated on v being fA unit
561// vector. If G4TessellatedSolid or other classes call this member function
562// with |v| != 1 then there will be errors.
563//
564G4bool G4TriangularFacet::Intersect (const G4ThreeVector &p,
565 const G4ThreeVector &v,
566 G4bool outgoing,
567 G4double &distance,
568 G4double &distFromSurface,
569 G4ThreeVector &normal)
570{
571 //
572 // Check whether the direction of the facet is consistent with the vector v
573 // and the need to be outgoing or ingoing. If inconsistent, disregard and
574 // return false.
575 //
576 G4double w = v.dot(fSurfaceNormal);
577 if ((outgoing && w < -dirTolerance) || (!outgoing && w > dirTolerance))
578 {
579 distance = kInfinity;
580 distFromSurface = kInfinity;
581 normal.set(0,0,0);
582 return false;
583 }
584 //
585 // Calculate the orthogonal distance from p to the surface containing the
586 // triangle. Then determine if we're on the right or wrong side of the
587 // surface (at fA distance greater than kCarTolerance to be consistent with
588 // "outgoing".
589 //
590 const G4ThreeVector &p0 = GetVertex(0);
591 G4ThreeVector D = p0 - p;
592 distFromSurface = D.dot(fSurfaceNormal);
593 G4bool wrongSide = (outgoing && distFromSurface < -0.5*kCarTolerance) ||
594 (!outgoing && distFromSurface > 0.5*kCarTolerance);
595
596 if (wrongSide)
597 {
598 distance = kInfinity;
599 distFromSurface = kInfinity;
600 normal.set(0,0,0);
601 return false;
602 }
603
604 wrongSide = (outgoing && distFromSurface < 0.0)
605 || (!outgoing && distFromSurface > 0.0);
606 if (wrongSide)
607 {
608 //
609 // We're slightly on the wrong side of the surface. Check if we're close
610 // enough using fA precise distance calculation.
611 //
612 G4ThreeVector u = Distance(p);
613 if (fSqrDist <= kCarTolerance*kCarTolerance)
614 {
615 //
616 // We're very close. Therefore return fA small negative number
617 // to pretend we intersect.
618 //
619 // distance = -0.5*kCarTolerance
620 distance = 0.0;
621 normal = fSurfaceNormal;
622 return true;
623 }
624 else
625 {
626 //
627 // We're close to the surface containing the triangle, but sufficiently
628 // far from the triangle, and on the wrong side compared to the directions
629 // of the surface normal and v. There is no intersection.
630 //
631 distance = kInfinity;
632 distFromSurface = kInfinity;
633 normal.set(0,0,0);
634 return false;
635 }
636 }
637 if (w < dirTolerance && w > -dirTolerance)
638 {
639 //
640 // The ray is within the plane of the triangle. Project the problem into 2D
641 // in the plane of the triangle. First try to create orthogonal unit vectors
642 // mu and nu, where mu is fE1/|fE1|. This is kinda like
643 // the original algorithm due to Rickard Holmberg, but with better
644 // mathematical justification than the original method ... however,
645 // beware Rickard's was less time-consuming.
646 //
647 // Note that vprime is not fA unit vector. We need to keep it unnormalised
648 // since the values of distance along vprime (s0 and s1) for intersection
649 // with the triangle will be used to determine if we cut the plane at the
650 // same time.
651 //
652 G4ThreeVector mu = fE1.unit();
653 G4ThreeVector nu = fSurfaceNormal.cross(mu);
654 G4TwoVector pprime(p.dot(mu), p.dot(nu));
655 G4TwoVector vprime(v.dot(mu), v.dot(nu));
656 G4TwoVector P0prime(p0.dot(mu), p0.dot(nu));
657 G4TwoVector E0prime(fE1.mag(), 0.0);
658 G4TwoVector E1prime(fE2.dot(mu), fE2.dot(nu));
659 G4TwoVector loc[2];
660 if (G4TessellatedGeometryAlgorithms::IntersectLineAndTriangle2D(pprime,
661 vprime, P0prime, E0prime, E1prime, loc))
662 {
663 //
664 // There is an intersection between the line and triangle in 2D.
665 // Now check which part of the line intersects with the plane
666 // containing the triangle in 3D.
667 //
668 G4double vprimemag = vprime.mag();
669 G4double s0 = (loc[0] - pprime).mag()/vprimemag;
670 G4double s1 = (loc[1] - pprime).mag()/vprimemag;
671 G4double normDist0 = fSurfaceNormal.dot(s0*v) - distFromSurface;
672 G4double normDist1 = fSurfaceNormal.dot(s1*v) - distFromSurface;
673
674 if ((normDist0 < 0.0 && normDist1 < 0.0)
675 || (normDist0 > 0.0 && normDist1 > 0.0)
676 || (normDist0 == 0.0 && normDist1 == 0.0) )
677 {
678 distance = kInfinity;
679 distFromSurface = kInfinity;
680 normal.set(0,0,0);
681 return false;
682 }
683 else
684 {
685 G4double dnormDist = normDist1 - normDist0;
686 if (fabs(dnormDist) < DBL_EPSILON)
687 {
688 distance = s0;
689 normal = fSurfaceNormal;
690 if (!outgoing) distFromSurface = -distFromSurface;
691 return true;
692 }
693 else
694 {
695 distance = s0 - normDist0*(s1-s0)/dnormDist;
696 normal = fSurfaceNormal;
697 if (!outgoing) distFromSurface = -distFromSurface;
698 return true;
699 }
700 }
701 }
702 else
703 {
704 distance = kInfinity;
705 distFromSurface = kInfinity;
706 normal.set(0,0,0);
707 return false;
708 }
709 }
710 //
711 //
712 // Use conventional algorithm to determine the whether there is an
713 // intersection. This involves determining the point of intersection of the
714 // line with the plane containing the triangle, and then calculating if the
715 // point is within the triangle.
716 //
717 distance = distFromSurface / w;
718 G4ThreeVector pp = p + v*distance;
719 G4ThreeVector DD = p0 - pp;
720 G4double d = fE1.dot(DD);
721 G4double e = fE2.dot(DD);
722 G4double ss = fB*e - fC*d;
723 G4double t = fB*d - fA*e;
724
725 G4double sTolerance = (fabs(fB)+ fabs(fC) + fabs(d) + fabs(e))*kCarTolerance;
726 G4double tTolerance = (fabs(fA)+ fabs(fB) + fabs(d) + fabs(e))*kCarTolerance;
727 G4double detTolerance = (fabs(fA)+ fabs(fC) + 2*fabs(fB) )*kCarTolerance;
728
729 //if (ss < 0.0 || t < 0.0 || ss+t > fDet)
730 if (ss < -sTolerance || t < -tTolerance || ( ss+t - fDet ) > detTolerance)
731 {
732 //
733 // The intersection is outside of the triangle.
734 //
735 distance = distFromSurface = kInfinity;
736 normal.set(0,0,0);
737 return false;
738 }
739 else
740 {
741 //
742 // There is an intersection. Now we only need to set the surface normal.
743 //
744 normal = fSurfaceNormal;
745 if (!outgoing) distFromSurface = -distFromSurface;
746 return true;
747 }
748}
