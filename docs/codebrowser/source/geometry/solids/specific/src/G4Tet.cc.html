88////////////////////////////////////////////////////////////////////////
89//
90// Constructor - create a tetrahedron
91// This class is implemented separately from general polyhedra,
92// because the simplex geometry can be computed very quickly,
93// which may become important in situations imported from mesh generators,
94// in which a very large number of G4Tets are created.
95// A Tet has all of its geometrical information precomputed
96
97G4Tet::G4Tet(const G4String& pName,
98 G4ThreeVector anchor,
99 G4ThreeVector p2,
100 G4ThreeVector p3,
101 G4ThreeVector p4, G4bool *degeneracyFlag)
102 : G4VSolid(pName), fRebuildPolyhedron(false), fpPolyhedron(0), warningFlag(0)
103{
104 // fV<x><y> is vector from vertex <y> to vertex <x>
105 //
106 G4ThreeVector fV21=p2-anchor;
107 G4ThreeVector fV31=p3-anchor;
108 G4ThreeVector fV41=p4-anchor;
109
110 // make sure this is a correctly oriented set of points for the tetrahedron
111 //
112 G4double signed_vol=fV21.cross(fV31).dot(fV41);
113 if(signed_vol<0.0)
114 {
115 G4ThreeVector temp(p4);
116 p4=p3;
117 p3=temp;
118 temp=fV41;
119 fV41=fV31;
120 fV31=temp;
121 }
122 fCubicVolume = std::fabs(signed_vol) / 6.;
123
124 G4ThreeVector fV24=p2-p4;
125 G4ThreeVector fV43=p4-p3;
126 G4ThreeVector fV32=p3-p2;
127
128 fXMin=std::min(std::min(std::min(anchor.x(), p2.x()),p3.x()),p4.x());
129 fXMax=std::max(std::max(std::max(anchor.x(), p2.x()),p3.x()),p4.x());
130 fYMin=std::min(std::min(std::min(anchor.y(), p2.y()),p3.y()),p4.y());
131 fYMax=std::max(std::max(std::max(anchor.y(), p2.y()),p3.y()),p4.y());
132 fZMin=std::min(std::min(std::min(anchor.z(), p2.z()),p3.z()),p4.z());
133 fZMax=std::max(std::max(std::max(anchor.z(), p2.z()),p3.z()),p4.z());
134
135 fDx=(fXMax-fXMin)*0.5; fDy=(fYMax-fYMin)*0.5; fDz=(fZMax-fZMin)*0.5;
136
137 fMiddle=G4ThreeVector(fXMax+fXMin, fYMax+fYMin, fZMax+fZMin)*0.5;
138 fMaxSize=std::max(std::max(std::max((anchor-fMiddle).mag(),
139 (p2-fMiddle).mag()),
140 (p3-fMiddle).mag()),
141 (p4-fMiddle).mag());
142
143 G4bool degenerate=std::fabs(signed_vol) < 1e-9*fMaxSize*fMaxSize*fMaxSize;
144
145 if(degeneracyFlag) *degeneracyFlag=degenerate;
146 else if (degenerate)
147 {
148 G4Exception("G4Tet::G4Tet()", "GeomSolids0002", FatalException,
149 "Degenerate tetrahedron not allowed.");
150 }
151
152 fTol=1e-9*(std::fabs(fXMin)+std::fabs(fXMax)+std::fabs(fYMin)
153 +std::fabs(fYMax)+std::fabs(fZMin)+std::fabs(fZMax));
154 //fTol=kCarTolerance;
155
156 fAnchor=anchor;
157 fP2=p2;
158 fP3=p3;
159 fP4=p4;
160
161 G4ThreeVector fCenter123=(anchor+p2+p3)*(1.0/3.0); // face center
162 G4ThreeVector fCenter134=(anchor+p4+p3)*(1.0/3.0);
163 G4ThreeVector fCenter142=(anchor+p4+p2)*(1.0/3.0);
164 G4ThreeVector fCenter234=(p2+p3+p4)*(1.0/3.0);
165
166 // compute area of each triangular face by cross product
167 // and sum for total surface area
168
169 G4ThreeVector normal123=fV31.cross(fV21);
170 G4ThreeVector normal134=fV41.cross(fV31);
171 G4ThreeVector normal142=fV21.cross(fV41);
172 G4ThreeVector normal234=fV32.cross(fV43);
173
174 fSurfaceArea=(
175 normal123.mag()+
176 normal134.mag()+
177 normal142.mag()+
178 normal234.mag()
179 )/2.0;
180
181 fNormal123=normal123.unit();
182 fNormal134=normal134.unit();
183 fNormal142=normal142.unit();
184 fNormal234=normal234.unit();
185
186 fCdotN123=fCenter123.dot(fNormal123);
187 fCdotN134=fCenter134.dot(fNormal134);
188 fCdotN142=fCenter142.dot(fNormal142);
189 fCdotN234=fCenter234.dot(fNormal234);
190}

242
243G4Tet& G4Tet::operator = (const G4Tet& rhs)
244{
245 // Check assignment to self
246 //
247 if (this == &rhs) { return *this; }
248
249 // Copy base class data
250 //

252
253 // Copy data
254 //
255 fCubicVolume = rhs.fCubicVolume; fSurfaceArea = rhs.fSurfaceArea;
256 fAnchor = rhs.fAnchor;
257 fP2 = rhs.fP2; fP3 = rhs.fP3; fP4 = rhs.fP4; fMiddle = rhs.fMiddle;
258 fNormal123 = rhs.fNormal123; fNormal142 = rhs.fNormal142;
259 fNormal134 = rhs.fNormal134; fNormal234 = rhs.fNormal234;
260 warningFlag = rhs.warningFlag; fCdotN123 = rhs.fCdotN123;
261 fCdotN142 = rhs.fCdotN142; fCdotN134 = rhs.fCdotN134;
262 fCdotN234 = rhs.fCdotN234; fXMin = rhs.fXMin; fXMax = rhs.fXMax;
263 fYMin = rhs.fYMin; fYMax = rhs.fYMax; fZMin = rhs.fZMin; fZMax = rhs.fZMax;
264 fDx = rhs.fDx; fDy = rhs.fDy; fDz = rhs.fDz; fTol = rhs.fTol;
265 fMaxSize = rhs.fMaxSize;
266 fRebuildPolyhedron = false;
267 delete fpPolyhedron; fpPolyhedron = 0;
268
269 return *this;
270}

298//////////////////////////////////////////////////////////////////////////

300// Get bounding box
301
302void G4Tet::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
303{
304 pMin.set(fXMin,fYMin,fZMin);
305 pMax.set(fXMax,fYMax,fZMax);
306
307 // Check correctness of the bounding box
308 //
309 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
310 {
311 std::ostringstream message;
312 message << "Bad bounding box (min >= max) for solid: "
313 << GetName() << " !"
314 << "\npMin = " << pMin
315 << "\npMax = " << pMax;
316 G4Exception("G4Tet::BoundingLimits()", "GeomMgt0001", JustWarning, message);
317 DumpInfo();
318 }
319}

321//////////////////////////////////////////////////////////////////////////
322//
323// Calculate extent under transform and specified limit
324
325G4bool G4Tet::CalculateExtent(const EAxis pAxis,
326 const G4VoxelLimits& pVoxelLimit,
327 const G4AffineTransform& pTransform,
328 G4double& pMin, G4double& pMax) const
329{
330 G4ThreeVector bmin, bmax;
331
332 // Check bounding box (bbox)
333 //
334 BoundingLimits(bmin,bmax);
335 G4BoundingEnvelope bbox(bmin,bmax);
336
337 // Use simple bounding-box to help in the case of complex 3D meshes
338 //
339 return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
340
341#if 0
342 // Precise extent computation (disabled by default for this shape)
343 //
344 G4bool exist;
345 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
346 {
347 return exist = (pMin < pMax) ? true : false;
348 }
349
350 // Set bounding envelope (benv) and calculate extent
351 //
352 std::vector<G4ThreeVector> vec = GetVertices();
353
354 G4ThreeVectorList anchor(1);
355 anchor[0].set(vec[0].x(),vec[0].y(),vec[0].z());
356
357 G4ThreeVectorList base(3);
358 base[0].set(vec[1].x(),vec[1].y(),vec[1].z());
359 base[1].set(vec[2].x(),vec[2].y(),vec[2].z());
360 base[2].set(vec[3].x(),vec[3].y(),vec[3].z());
361
362 std::vector<const G4ThreeVectorList *> polygons(2);
363 polygons[0] = &anchor;
364 polygons[1] = &base;
365
366 G4BoundingEnvelope benv(bmin,bmax,polygons);
367 return exists = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
368#endif
369}

371/////////////////////////////////////////////////////////////////////////
372//
373// Return whether point inside/outside/on surface, using tolerance
374
375EInside G4Tet::Inside(const G4ThreeVector& p) const
376{
377 G4double r123, r134, r142, r234;
378
379 // this is written to allow if-statement truncation so the outside test
380 // (where most of the world is) can fail very quickly and efficiently
381
382 if ( (r123=p.dot(fNormal123)-fCdotN123) > fTol ||
383 (r134=p.dot(fNormal134)-fCdotN134) > fTol ||
384 (r142=p.dot(fNormal142)-fCdotN142) > fTol ||
385 (r234=p.dot(fNormal234)-fCdotN234) > fTol )
386 {
387 return kOutside; // at least one is out!
388 }
389 else if( (r123 < -fTol)&&(r134 < -fTol)&&(r142 < -fTol)&&(r234 < -fTol) )
390 {
391 return kInside; // all are definitively inside
392 }
393 else
394 {
395 return kSurface; // too close to tell
396 }
397}

399///////////////////////////////////////////////////////////////////////
400//
401// Calculate side nearest to p, and return normal
402// If two sides are equidistant, normal of first side (x/y/z)
403// encountered returned.
404// This assumes that we are looking from the inside!
405
406G4ThreeVector G4Tet::SurfaceNormal( const G4ThreeVector& p) const
407{
408 G4double r123=std::fabs(p.dot(fNormal123)-fCdotN123);
409 G4double r134=std::fabs(p.dot(fNormal134)-fCdotN134);
410 G4double r142=std::fabs(p.dot(fNormal142)-fCdotN142);
411 G4double r234=std::fabs(p.dot(fNormal234)-fCdotN234);
412
413 const G4double delta = 0.5*kCarTolerance;
414 G4ThreeVector sumnorm(0., 0., 0.);
415 G4int noSurfaces=0;
416
417 if (r123 <= delta)
418 {
419 noSurfaces ++;
420 sumnorm= fNormal123;
421 }
422
423 if (r134 <= delta)
424 {
425 noSurfaces ++;
426 sumnorm += fNormal134;
427 }
428
429 if (r142 <= delta)
430 {
431 noSurfaces ++;
432 sumnorm += fNormal142;
433 }
434 if (r234 <= delta)
435 {
436 noSurfaces ++;
437 sumnorm += fNormal234;
438 }
439
440 if( noSurfaces > 0 )
441 {
442 if( noSurfaces == 1 )
443 {
444 return sumnorm;
445 }
446 else
447 {
448 return sumnorm.unit();
449 }
450 }
451 else // Approximative Surface Normal
452 {
453
454 if( (r123<=r134) && (r123<=r142) && (r123<=r234) ) { return fNormal123; }
455 else if ( (r134<=r142) && (r134<=r234) ) { return fNormal134; }
456 else if (r142 <= r234) { return fNormal142; }
457 return fNormal234;
458 }
459}
460///////////////////////////////////////////////////////////////////////////
461//
462// Calculate distance to box from an outside point
463// - return kInfinity if no intersection.
464// All this is very unrolled, for speed.
465
466G4double G4Tet::DistanceToIn(const G4ThreeVector& p,
467 const G4ThreeVector& v) const
468{
469 G4ThreeVector vu(v.unit()), hp;
470 G4double vdotn, t, tmin=kInfinity;
471
472 G4double extraDistance=10.0*fTol; // a little ways into the solid
473
474 vdotn=-vu.dot(fNormal123);
475 if(vdotn > 1e-12)
476 { // this is a candidate face, since it is pointing at us
477 t=(p.dot(fNormal123)-fCdotN123)/vdotn; // # distance to intersection
478 if( (t>=-fTol) && (t<tmin) )
479 { // if not true, we're going away from this face or it's not close
480 hp=p+vu*(t+extraDistance); // a little beyond point of intersection
481 if ( ( hp.dot(fNormal134)-fCdotN134 < 0.0 ) &&
482 ( hp.dot(fNormal142)-fCdotN142 < 0.0 ) &&
483 ( hp.dot(fNormal234)-fCdotN234 < 0.0 ) )
484 {
485 tmin=t;
486 }
487 }
488 }
489
490 vdotn=-vu.dot(fNormal134);
491 if(vdotn > 1e-12)
492 { // # this is a candidate face, since it is pointing at us
493 t=(p.dot(fNormal134)-fCdotN134)/vdotn; // # distance to intersection
494 if( (t>=-fTol) && (t<tmin) )
495 { // if not true, we're going away from this face
496 hp=p+vu*(t+extraDistance); // a little beyond point of intersection
497 if ( ( hp.dot(fNormal123)-fCdotN123 < 0.0 ) &&
498 ( hp.dot(fNormal142)-fCdotN142 < 0.0 ) &&
499 ( hp.dot(fNormal234)-fCdotN234 < 0.0 ) )
500 {
501 tmin=t;
502 }
503 }
504 }
505
506 vdotn=-vu.dot(fNormal142);
507 if(vdotn > 1e-12)
508 { // # this is a candidate face, since it is pointing at us
509 t=(p.dot(fNormal142)-fCdotN142)/vdotn; // # distance to intersection
510 if( (t>=-fTol) && (t<tmin) )
511 { // if not true, we're going away from this face
512 hp=p+vu*(t+extraDistance); // a little beyond point of intersection
513 if ( ( hp.dot(fNormal123)-fCdotN123 < 0.0 ) &&
514 ( hp.dot(fNormal134)-fCdotN134 < 0.0 ) &&
515 ( hp.dot(fNormal234)-fCdotN234 < 0.0 ) )
516 {
517 tmin=t;
518 }
519 }
520 }
521
522 vdotn=-vu.dot(fNormal234);
523 if(vdotn > 1e-12)
524 { // # this is a candidate face, since it is pointing at us
525 t=(p.dot(fNormal234)-fCdotN234)/vdotn; // # distance to intersection
526 if( (t>=-fTol) && (t<tmin) )
527 { // if not true, we're going away from this face
528 hp=p+vu*(t+extraDistance); // a little beyond point of intersection
529 if ( ( hp.dot(fNormal123)-fCdotN123 < 0.0 ) &&
530 ( hp.dot(fNormal134)-fCdotN134 < 0.0 ) &&
531 ( hp.dot(fNormal142)-fCdotN142 < 0.0 ) )
532 {
533 tmin=t;
534 }
535 }
536 }
537
538 return std::max(0.0,tmin);
539}

553/////////////////////////////////////////////////////////////////////////
554//
555// Calcluate distance to surface of box from inside
556// by calculating distances to box's x/y/z planes.
557// Smallest distance is exact distance to exiting.
558
559G4double G4Tet::DistanceToOut( const G4ThreeVector& p,const G4ThreeVector& v,
560 const G4bool calcNorm,
561 G4bool *validNorm, G4ThreeVector *n) const
562{
563 G4ThreeVector vu(v.unit());
564 G4double t1=kInfinity,t2=kInfinity,t3=kInfinity,t4=kInfinity, vdotn, tt;
565
566 vdotn=vu.dot(fNormal123);
567 if(vdotn > 1e-12) // #we're heading towards this face, so it is a candidate
568 {
569 t1=(fCdotN123-p.dot(fNormal123))/vdotn; // # distance to intersection
570 }
571
572 vdotn=vu.dot(fNormal134);
573 if(vdotn > 1e-12) // #we're heading towards this face, so it is a candidate
574 {
575 t2=(fCdotN134-p.dot(fNormal134))/vdotn; // # distance to intersection
576 }
577
578 vdotn=vu.dot(fNormal142);
579 if(vdotn > 1e-12) // #we're heading towards this face, so it is a candidate
580 {
581 t3=(fCdotN142-p.dot(fNormal142))/vdotn; // # distance to intersection
582 }
583
584 vdotn=vu.dot(fNormal234);
585 if(vdotn > 1e-12) // #we're heading towards this face, so it is a candidate
586 {
587 t4=(fCdotN234-p.dot(fNormal234))/vdotn; // # distance to intersection
588 }
589
590 tt=std::min(std::min(std::min(t1,t2),t3),t4);
591
592 if (warningFlag && (tt == kInfinity || tt < -fTol))
593 {
594 DumpInfo();
595 std::ostringstream message;
596 message << "No good intersection found or already outside!?" << G4endl
597 << "p = " << p / mm << "mm" << G4endl
598 << "v = " << v << G4endl
599 << "t1, t2, t3, t4 (mm) "
600 << t1/mm << ", " << t2/mm << ", " << t3/mm << ", " << t4/mm;
601 G4Exception("G4Tet::DistanceToOut(p,v,...)", "GeomSolids1002",
602 JustWarning, message);
603 if(validNorm)
604 {
605 *validNorm=false; // flag normal as meaningless
606 }
607 }
608 else if(calcNorm && n)
609 {
610 G4ThreeVector normal;
611 if(tt==t1) { normal=fNormal123; }
612 else if (tt==t2) { normal=fNormal134; }
613 else if (tt==t3) { normal=fNormal142; }
614 else if (tt==t4) { normal=fNormal234; }
615 *n=normal;
616 if(validNorm) { *validNorm=true; }
617 }
618
619 return std::max(tt,0.0); // avoid tt<0.0 by a tiny bit
620 // if we are right on a face
621}

623////////////////////////////////////////////////////////////////////////////
624//
625// Calculate exact shortest distance to any boundary from inside
626// - If outside return 0
627
628G4double G4Tet::DistanceToOut(const G4ThreeVector& p) const
629{
630 G4double t1,t2,t3,t4;
631 t1=fCdotN123-p.dot(fNormal123); // distance to plane, positive if inside
632 t2=fCdotN134-p.dot(fNormal134); // distance to plane
633 t3=fCdotN142-p.dot(fNormal142); // distance to plane
634 t4=fCdotN234-p.dot(fNormal234); // distance to plane
635
636 // if any one of these is negative, we are outside,
637 // so return zero in that case
638
639 G4double tmin=std::min(std::min(std::min(t1,t2),t3),t4);
640 return (tmin < fTol)? 0:tmin;
641}

688////////////////////////////////////////////////////////////////////////
689//
690// GetPointOnFace
691//
692// Auxiliary method for get point on surface
693
694G4ThreeVector G4Tet::GetPointOnFace(G4ThreeVector p1, G4ThreeVector p2,
695 G4ThreeVector p3, G4double& area) const
696{
697 G4double lambda1,lambda2;
698 G4ThreeVector v, w;
699
700 v = p3 - p1;
701 w = p1 - p2;
702
703 lambda1 = G4RandFlat::shoot(0.,1.);
704 lambda2 = G4RandFlat::shoot(0.,lambda1);
705
706 area = 0.5*(v.cross(w)).mag();
707
708 return (p2 + lambda1*w + lambda2*v);
709}

711////////////////////////////////////////////////////////////////////////////
712//
713// GetPointOnSurface
714
715G4ThreeVector G4Tet::GetPointOnSurface() const
716{
717 G4double chose,aOne,aTwo,aThree,aFour;
718 G4ThreeVector p1, p2, p3, p4;
719
720 p1 = GetPointOnFace(fAnchor,fP2,fP3,aOne);
721 p2 = GetPointOnFace(fAnchor,fP4,fP3,aTwo);
722 p3 = GetPointOnFace(fAnchor,fP4,fP2,aThree);
723 p4 = GetPointOnFace(fP4,fP3,fP2,aFour);
724
725 chose = G4RandFlat::shoot(0.,aOne+aTwo+aThree+aFour);
726 if( (chose>=0.) && (chose <aOne) ) {return p1;}
727 else if( (chose>=aOne) && (chose < aOne+aTwo) ) {return p2;}
728 else if( (chose>=aOne+aTwo) && (chose<aOne+aTwo+aThree) ) {return p3;}
729 return p4;
730}

787// CreatePolyhedron
788
789G4Polyhedron* G4Tet::CreatePolyhedron () const
790{
791 G4Polyhedron *ph=new G4Polyhedron;
792 G4double xyz[4][3];
793 const G4int faces[4][4]={{1,3,2,0},{1,4,3,0},{1,2,4,0},{2,3,4,0}};
794 xyz[0][0]=fAnchor.x(); xyz[0][1]=fAnchor.y(); xyz[0][2]=fAnchor.z();
795 xyz[1][0]=fP2.x(); xyz[1][1]=fP2.y(); xyz[1][2]=fP2.z();
796 xyz[2][0]=fP3.x(); xyz[2][1]=fP3.y(); xyz[2][2]=fP3.z();
797 xyz[3][0]=fP4.x(); xyz[3][1]=fP4.y(); xyz[3][2]=fP4.z();
798
799 ph->createPolyhedron(4,4,xyz,faces);
800
801 return ph;
802}
