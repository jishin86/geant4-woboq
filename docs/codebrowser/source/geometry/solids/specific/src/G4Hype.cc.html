74// Constructor - check parameters, and fills protected data members
75G4Hype::G4Hype(const G4String& pName,
76 G4double newInnerRadius,
77 G4double newOuterRadius,
78 G4double newInnerStereo,
79 G4double newOuterStereo,
80 G4double newHalfLenZ)
81 : G4VSolid(pName), fCubicVolume(0.), fSurfaceArea(0.),
82 fRebuildPolyhedron(false), fpPolyhedron(0)
83{
84 fHalfTol = 0.5*kCarTolerance;
85
86 // Check z-len
87 //
88 if (newHalfLenZ<=0)
89 {
90 std::ostringstream message;
91 message << "Invalid Z half-length - " << GetName() << G4endl
92 << " Invalid Z half-length: "
93 << newHalfLenZ/mm << " mm";
94 G4Exception("G4Hype::G4Hype()", "GeomSolids0002",
95 FatalErrorInArgument, message);
96 }
97 halfLenZ=newHalfLenZ;
98
99 // Check radii
100 //
101 if (newInnerRadius<0 || newOuterRadius<0)
102 {
103 std::ostringstream message;
104 message << "Invalid radii - " << GetName() << G4endl
105 << " Invalid radii ! Inner radius: "
106 << newInnerRadius/mm << " mm" << G4endl
107 << " Outer radius: "
108 << newOuterRadius/mm << " mm";
109 G4Exception("G4Hype::G4Hype()", "GeomSolids0002",
110 FatalErrorInArgument, message);
111 }
112 if (newInnerRadius >= newOuterRadius)
113 {
114 std::ostringstream message;
115 message << "Outer > inner radius - " << GetName() << G4endl
116 << " Invalid radii ! Inner radius: "
117 << newInnerRadius/mm << " mm" << G4endl
118 << " Outer radius: "
119 << newOuterRadius/mm << " mm";
120 G4Exception("G4Hype::G4Hype()", "GeomSolids0002",
121 FatalErrorInArgument, message);
122 }
123
124 innerRadius=newInnerRadius;
125 outerRadius=newOuterRadius;
126
127 innerRadius2=innerRadius*innerRadius;
128 outerRadius2=outerRadius*outerRadius;
129
130 SetInnerStereo( newInnerStereo );
131 SetOuterStereo( newOuterStereo );
132}

177//
178// Assignment operator
179//
180G4Hype& G4Hype::operator = (const G4Hype& rhs)
181{
182 // Check assignment to self
183 //
184 if (this == &rhs) { return *this; }
185
186 // Copy base class data
187 //
188 G4VSolid::operator=(rhs);
189
190 // Copy data
191 //
192 innerRadius = rhs.innerRadius; outerRadius = rhs.outerRadius;
193 halfLenZ = rhs.halfLenZ;
194 innerStereo = rhs.innerStereo; outerStereo = rhs.outerStereo;
195 tanInnerStereo = rhs.tanInnerStereo; tanOuterStereo = rhs.tanOuterStereo;
196 tanInnerStereo2 = rhs.tanInnerStereo2; tanOuterStereo2 = rhs.tanOuterStereo2;
197 innerRadius2 = rhs.innerRadius2; outerRadius2 = rhs.outerRadius2;
198 endInnerRadius2 = rhs.endInnerRadius2; endOuterRadius2 = rhs.endOuterRadius2;
199 endInnerRadius = rhs.endInnerRadius; endOuterRadius = rhs.endOuterRadius;
200 fCubicVolume = rhs.fCubicVolume; fSurfaceArea = rhs.fSurfaceArea;
201 fHalfTol = rhs.fHalfTol;
202 fRebuildPolyhedron = false;
203 delete fpPolyhedron; fpPolyhedron = 0;
204
205 return *this;
206}

264//
265// Decides whether point is inside, outside or on the surface
266//
267EInside G4Hype::Inside(const G4ThreeVector& p) const
268{
269 //
270 // Check z extents: are we outside?
271 //
272 const G4double absZ(std::fabs(p.z()));
273 if (absZ > halfLenZ + fHalfTol) return kOutside;
274
275 //
276 // Check outer radius
277 //
278 const G4double oRad2(HypeOuterRadius2(absZ));
279 const G4double xR2( p.x()*p.x()+p.y()*p.y() );
280
281 if (xR2 > oRad2 + kCarTolerance*endOuterRadius) return kOutside;
282
283 if (xR2 > oRad2 - kCarTolerance*endOuterRadius) return kSurface;
284
285 if (InnerSurfaceExists())
286 {
287 //
288 // Check inner radius
289 //
290 const G4double iRad2(HypeInnerRadius2(absZ));
291
292 if (xR2 < iRad2 - kCarTolerance*endInnerRadius) return kOutside;
293
294 if (xR2 < iRad2 + kCarTolerance*endInnerRadius) return kSurface;
295 }
296
297 //
298 // We are inside in radius, now check endplate surface
299 //
300 if (absZ > halfLenZ - fHalfTol) return kSurface;
301
302 return kInside;
303}

307//
308// return the normal unit vector to the Hyperbolical Surface at a point
309// p on (or nearly on) the surface
310//
311G4ThreeVector G4Hype::SurfaceNormal( const G4ThreeVector& p ) const
312{
313 //
314 // Which of the three or four surfaces are we closest to?
315 //
316 const G4double absZ(std::fabs(p.z()));
317 const G4double distZ(absZ - halfLenZ);
318 const G4double dist2Z(distZ*distZ);
319
320 const G4double xR2( p.x()*p.x()+p.y()*p.y() );
321 const G4double dist2Outer( std::fabs(xR2 - HypeOuterRadius2(absZ)) );
322
323 if (InnerSurfaceExists())
324 {
325 //
326 // Has inner surface: is this closest?
327 //
328 const G4double dist2Inner( std::fabs(xR2 - HypeInnerRadius2(absZ)) );
329 if (dist2Inner < dist2Z && dist2Inner < dist2Outer)
330 return G4ThreeVector( -p.x(), -p.y(), p.z()*tanInnerStereo2 ).unit();
331 }
332
333 //
334 // Do the "endcaps" win?
335 //
336 if (dist2Z < dist2Outer)
337 return G4ThreeVector( 0.0, 0.0, p.z() < 0 ? -1.0 : 1.0 );
338
339
340 //
341 // Outer surface wins
342 //
343 return G4ThreeVector( p.x(), p.y(), -p.z()*tanOuterStereo2 ).unit();
344}

347//
348// Calculate distance to shape from outside, along normalised vector
349// - return kInfinity if no intersection,
350// or intersection distance <= tolerance
351//
352// Calculating the intersection of a line with the surfaces
353// is fairly straight forward. The difficult problem is dealing
354// with the intersections of the surfaces in a consistent manner,
355// and this accounts for the complicated logic.
356//
357G4double G4Hype::DistanceToIn( const G4ThreeVector& p,
358 const G4ThreeVector& v ) const
359{
360 //
361 // Quick test. Beware! This assumes v is a unit vector!
362 //
363 if (std::fabs(p.x()*v.y() - p.y()*v.x()) > endOuterRadius+kCarTolerance)
364 return kInfinity;
365
366 //
367 // Take advantage of z symmetry, and reflect throught the
368 // z=0 plane so that pz is always positive
369 //
370 G4double pz(p.z()), vz(v.z());
371 if (pz < 0)
372 {
373 pz = -pz;
374 vz = -vz;
375 }
376
377 //
378 // We must be very careful if we don't want to
379 // create subtle leaks at the edges where the
380 // hyperbolic surfaces connect to the endplate.
381 // The only reliable way to do so is to make sure
382 // that the decision as to when a track passes
383 // over the edge of one surface is exactly the
384 // same decision as to when a track passes into the
385 // other surface. By "exact", we don't mean algebraicly
386 // exact, but we mean the same machine instructions
387 // should be used.
388 //
389 G4bool couldMissOuter(true),
390 couldMissInner(true),
391 cantMissInnerCylinder(false);
392
393 //
394 // Check endplate intersection
395 //
396 G4double sigz = pz-halfLenZ;
397
398 if (sigz > -fHalfTol) // equivalent to: if (pz > halfLenZ - fHalfTol)
399 {
400 //
401 // We start in front of the endplate (within roundoff)
402 // Correct direction to intersect endplate?
403 //
404 if (vz >= 0)
405 {
406 //
407 // Nope. As long as we are far enough away, we
408 // can't intersect anything
409 //
410 if (sigz > 0) return kInfinity;
411
412 //
413 // Otherwise, we may still hit a hyperbolic surface
414 // if the point is on the hyperbolic surface (within tolerance)
415 //
416 G4double pr2 = p.x()*p.x() + p.y()*p.y();
417 if (pr2 > endOuterRadius2 + kCarTolerance*endOuterRadius)
418 return kInfinity;
419
420 if (InnerSurfaceExists())
421 {
422 if (pr2 < endInnerRadius2 - kCarTolerance*endInnerRadius)
423 return kInfinity;
424 if ( (pr2 < endOuterRadius2 - kCarTolerance*endOuterRadius)
425 && (pr2 > endInnerRadius2 + kCarTolerance*endInnerRadius) )
426 return kInfinity;
427 }
428 else
429 {
430 if (pr2 < endOuterRadius2 - kCarTolerance*endOuterRadius)
431 return kInfinity;
432 }
433 }
434 else
435 {
436 //
437 // Where do we intersect at z = halfLenZ?
438 //
439 G4double q = -sigz/vz;
440 G4double xi = p.x() + q*v.x(),
441 yi = p.y() + q*v.y();
442
443 //
444 // Is this on the endplate? If so, return s, unless
445 // we are on the tolerant surface, in which case return 0
446 //
447 G4double pr2 = xi*xi + yi*yi;
448 if (pr2 <= endOuterRadius2)
449 {
450 if (InnerSurfaceExists())
451 {
452 if (pr2 >= endInnerRadius2) return (sigz < fHalfTol) ? 0 : q;
453 //
454 // This test is sufficient to ensure that the
455 // trajectory cannot miss the inner hyperbolic surface
456 // for z > 0, if the normal is correct.
457 //
458 G4double dot1 = (xi*v.x() + yi*v.y())*endInnerRadius/std::sqrt(pr2);
459 couldMissInner = (dot1 - halfLenZ*tanInnerStereo2*vz <= 0);
460
461 if (pr2 > endInnerRadius2*(1 - 2*DBL_EPSILON) )
462 {
463 //
464 // There is a potential leak if the inner
465 // surface is a cylinder
466 //
467 if ( (innerStereo < DBL_MIN)
468 && ((std::fabs(v.x()) > DBL_MIN) || (std::fabs(v.y()) > DBL_MIN)))
469 cantMissInnerCylinder = true;
470 }
471 }
472 else
473 {
474 return (sigz < fHalfTol) ? 0 : q;
475 }
476 }
477 else
478 {
479 G4double dotR( xi*v.x() + yi*v.y() );
480 if (dotR >= 0)
481 {
482 //
483 // Otherwise, if we are traveling outwards, we know
484 // we must miss the hyperbolic surfaces also, so
485 // we need not bother checking
486 //
487 return kInfinity;
488 }
489 else
490 {
491 //
492 // This test is sufficient to ensure that the
493 // trajectory cannot miss the outer hyperbolic surface
494 // for z > 0, if the normal is correct.
495 //
496 G4double dot1 = dotR*endOuterRadius/std::sqrt(pr2);
497 couldMissOuter = (dot1 - halfLenZ*tanOuterStereo2*vz>= 0);
498 }
499 }
500 }
501 }
502
503 //
504 // Check intersection with outer hyperbolic surface, save
505 // distance to valid intersection into "best".
506 //
507 G4double best = kInfinity;
508
509 G4double q[2];
510 G4int n = IntersectHype( p, v, outerRadius2, tanOuterStereo2, q );
511
512 if (n > 0)
513 {
514 //
515 // Potential intersection: is p on this surface?
516 //
517 if (pz < halfLenZ+fHalfTol)
518 {
519 G4double dr2 = p.x()*p.x() + p.y()*p.y() - HypeOuterRadius2(pz);
520 if (std::fabs(dr2) < kCarTolerance*endOuterRadius)
521 {
522 //
523 // Sure, but make sure we're traveling inwards at
524 // this point
525 //
526 if (p.x()*v.x() + p.y()*v.y() - pz*tanOuterStereo2*vz < 0)
527 return 0;
528 }
529 }
530
531 //
532 // We are now certain that p is not on the tolerant surface.
533 // Accept only position distance q
534 //
535 G4int i;
536 for( i=0; i<n; i++ )
537 {
538 if (q[i] >= 0)
539 {
540 //
541 // Check to make sure this intersection point is
542 // on the surface, but only do so if we haven't
543 // checked the endplate intersection already
544 //
545 G4double zi = pz + q[i]*vz;
546
547 if (zi < -halfLenZ) continue;
548 if (zi > +halfLenZ && couldMissOuter) continue;
549
550 //
551 // Check normal
552 //
553 G4double xi = p.x() + q[i]*v.x(),
554 yi = p.y() + q[i]*v.y();
555
556 if (xi*v.x() + yi*v.y() - zi*tanOuterStereo2*vz > 0) continue;
557
558 best = q[i];
559 break;
560 }
561 }
562 }
563
564 if (!InnerSurfaceExists()) return best;
565
566 //
567 // Check intersection with inner hyperbolic surface
568 //
569 n = IntersectHype( p, v, innerRadius2, tanInnerStereo2, q );
570 if (n == 0)
571 {
572 if (cantMissInnerCylinder) return (sigz < fHalfTol) ? 0 : -sigz/vz;
573
574 return best;
575 }
576
577 //
578 // P on this surface?
579 //
580 if (pz < halfLenZ+fHalfTol)
581 {
582 G4double dr2 = p.x()*p.x() + p.y()*p.y() - HypeInnerRadius2(pz);
583 if (std::fabs(dr2) < kCarTolerance*endInnerRadius)
584 {
585 //
586 // Sure, but make sure we're traveling outwards at
587 // this point
588 //
589 if (p.x()*v.x() + p.y()*v.y() - pz*tanInnerStereo2*vz > 0) return 0;
590 }
591 }
592
593 //
594 // No, so only positive q is valid. Search for a valid intersection
595 // that is closer than the outer intersection (if it exists)
596 //
597 G4int i;
598 for( i=0; i<n; i++ )
599 {
600 if (q[i] > best) break;
601 if (q[i] >= 0)
602 {
603 //
604 // Check to make sure this intersection point is
605 // on the surface, but only do so if we haven't
606 // checked the endplate intersection already
607 //
608 G4double zi = pz + q[i]*vz;
609
610 if (zi < -halfLenZ) continue;
611 if (zi > +halfLenZ && couldMissInner) continue;
612
613 //
614 // Check normal
615 //
616 G4double xi = p.x() + q[i]*v.x(),
617 yi = p.y() + q[i]*v.y();
618
619 if (xi*v.x() + yi*v.y() - zi*tanOuterStereo2*vz < 0) continue;
620
621 best = q[i];
622 break;
623 }
624 }
625
626 //
627 // Done
628 //
629 return best;
630}

633//
634// Calculate distance to shape from outside, along perpendicular direction
635// (if one exists). May be an underestimate.
636//
637// There are five (r,z) regions:
638// 1. a point that is beyond the endcap but within the
639// endcap radii
640// 2. a point with r > outer endcap radius and with
641// a z position that is beyond the cone formed by the
642// normal of the outer hyperbolic surface at the
643// edge at which it meets the endcap.
644// 3. a point that is outside the outer surface and not in (1 or 2)
645// 4. a point that is inside the inner surface and not in (5)
646// 5. a point with radius < inner endcap radius and
647// with a z position beyond the cone formed by the
648// normal of the inner hyperbolic surface at the
649// edge at which it meets the endcap.
650// (regions 4 and 5 only exist if there is an inner surface)
651//
652G4double G4Hype::DistanceToIn(const G4ThreeVector& p) const
653{
654 G4double absZ(std::fabs(p.z()));
655
656 //
657 // Check region
658 //
659 G4double r2 = p.x()*p.x() + p.y()*p.y();
660 G4double r = std::sqrt(r2);
661
662 G4double sigz = absZ - halfLenZ;
663
664 if (r < endOuterRadius)
665 {
666 if (sigz > -fHalfTol)
667 {
668 if (InnerSurfaceExists())
669 {
670 if (r > endInnerRadius)
671 return sigz < fHalfTol ? 0 : sigz; // Region 1
672
673 G4double dr = endInnerRadius - r;
674 if (sigz > dr*tanInnerStereo2)
675 {
676 //
677 // In region 5
678 //
679 G4double answer = std::sqrt( dr*dr + sigz*sigz );
680 return answer < fHalfTol ? 0 : answer;
681 }
682 }
683 else
684 {
685 //
686 // In region 1 (no inner surface)
687 //
688 return sigz < fHalfTol ? 0 : sigz;
689 }
690 }
691 }
692 else
693 {
694 G4double dr = r - endOuterRadius;
695 if (sigz > -dr*tanOuterStereo2)
696 {
697 //
698 // In region 2
699 //
700 G4double answer = std::sqrt( dr*dr + sigz*sigz );
701 return answer < fHalfTol ? 0 : answer;
702 }
703 }
704
705 if (InnerSurfaceExists())
706 {
707 if (r2 < HypeInnerRadius2(absZ)+kCarTolerance*endInnerRadius)
708 {
709 //
710 // In region 4
711 //
712 G4double answer = ApproxDistInside( r,absZ,innerRadius,tanInnerStereo2 );
713 return answer < fHalfTol ? 0 : answer;
714 }
715 }
716
717 //
718 // We are left by elimination with region 3
719 //
720 G4double answer = ApproxDistOutside( r, absZ, outerRadius, tanOuterStereo );
721 return answer < fHalfTol ? 0 : answer;
722}

725//
726// Calculate distance to surface of shape from `inside', allowing for tolerance
727//
728// The situation here is much simplier than DistanceToIn(p,v). For
729// example, there is no need to even check whether an intersection
730// point is inside the boundary of a surface, as long as all surfaces
731// are checked and the smallest distance is used.
732//
733G4double G4Hype::DistanceToOut( const G4ThreeVector& p, const G4ThreeVector& v,
734 const G4bool calcNorm,
735 G4bool *validNorm, G4ThreeVector *norm ) const
736{
737 static const G4ThreeVector normEnd1(0.0,0.0,+1.0);
738 static const G4ThreeVector normEnd2(0.0,0.0,-1.0);
739
740 //
741 // Keep track of closest surface
742 //
743 G4double sBest; // distance to
744 const G4ThreeVector *nBest; // normal vector
745 G4bool vBest; // whether "valid"
746
747 //
748 // Check endplate, taking advantage of symmetry.
749 // Note that the endcap is the only surface which
750 // has a "valid" normal, i.e. is a surface of which
751 // the entire solid is behind.
752 //
753 G4double pz(p.z()), vz(v.z());
754 if (vz < 0)
755 {
756 pz = -pz;
757 vz = -vz;
758 nBest = &normEnd2;
759 }
760 else
761 nBest = &normEnd1;
762
763 //
764 // Possible intercept. Are we on the surface?
765 //
766 if (pz > halfLenZ-fHalfTol)
767 {
768 if (calcNorm) { *norm = *nBest; *validNorm = true; }
769 return 0;
770 }
771
772 //
773 // Nope. Get distance. Beware of zero vz.
774 //
775 sBest = (vz > DBL_MIN) ? (halfLenZ - pz)/vz : kInfinity;
776 vBest = true;
777
778 //
779 // Check outer surface
780 //
781 G4double r2 = p.x()*p.x() + p.y()*p.y();
782
783 G4double q[2];
784 G4int n = IntersectHype( p, v, outerRadius2, tanOuterStereo2, q );
785
786 G4ThreeVector norm1, norm2;
787
788 if (n > 0)
789 {
790 //
791 // We hit somewhere. Are we on the surface?
792 //
793 G4double dr2 = r2 - HypeOuterRadius2(pz);
794 if (std::fabs(dr2) < endOuterRadius*kCarTolerance)
795 {
796 G4ThreeVector normHere( p.x(), p.y(), -p.z()*tanOuterStereo2 );
797 //
798 // Sure. But are we going the right way?
799 //
800 if (normHere.dot(v) > 0)
801 {
802 if (calcNorm) { *norm = normHere.unit(); *validNorm = false; }
803 return 0;
804 }
805 }
806
807 //
808 // Nope. Check closest positive intercept.
809 //
810 G4int i;
811 for( i=0; i<n; i++ )
812 {
813 if (q[i] > sBest) break;
814 if (q[i] > 0)
815 {
816 //
817 // Make sure normal is correct (that this
818 // solution is an outgoing solution)
819 //
820 G4ThreeVector pk(p+q[i]*v);
821 norm1 = G4ThreeVector( pk.x(), pk.y(), -pk.z()*tanOuterStereo2 );
822 if (norm1.dot(v) > 0)
823 {
824 sBest = q[i];
825 nBest = &norm1;
826 vBest = false;
827 break;
828 }
829 }
830 }
831 }
832
833 if (InnerSurfaceExists())
834 {
835 //
836 // Check inner surface
837 //
838 n = IntersectHype( p, v, innerRadius2, tanInnerStereo2, q );
839 if (n > 0)
840 {
841 //
842 // On surface?
843 //
844 G4double dr2 = r2 - HypeInnerRadius2(pz);
845 if (std::fabs(dr2) < endInnerRadius*kCarTolerance)
846 {
847 G4ThreeVector normHere( -p.x(), -p.y(), p.z()*tanInnerStereo2 );
848 if (normHere.dot(v) > 0)
849 {
850 if (calcNorm)
851 {
852 *norm = normHere.unit();
853 *validNorm = false;
854 }
855 return 0;
856 }
857 }
858
859 //
860 // Check closest positive
861 //
862 G4int i;
863 for( i=0; i<n; i++ )
864 {
865 if (q[i] > sBest) break;
866 if (q[i] > 0)
867 {
868 G4ThreeVector pk(p+q[i]*v);
869 norm2 = G4ThreeVector( -pk.x(), -pk.y(), pk.z()*tanInnerStereo2 );
870 if (norm2.dot(v) > 0)
871 {
872 sBest = q[i];
873 nBest = &norm2;
874 vBest = false;
875 break;
876 }
877 }
878 }
879 }
880 }
881
882 //
883 // Done!
884 //
885 if (calcNorm)
886 {
887 *validNorm = vBest;
888
889 if (nBest == &norm1 || nBest == &norm2)
890 *norm = nBest->unit();
891 else
892 *norm = *nBest;
893 }
894
895 return sBest;
896}

899//
900// Calculate distance (<=actual) to closest surface of shape from inside
901//
902// May be an underestimate
903//
904G4double G4Hype::DistanceToOut(const G4ThreeVector& p) const
905{
906 //
907 // Try each surface and remember the closest
908 //
909 G4double absZ(std::fabs(p.z()));
910 G4double r(p.perp());
911
912 G4double sBest = halfLenZ - absZ;
913
914 G4double tryOuter = ApproxDistInside( r, absZ, outerRadius, tanOuterStereo2 );
915 if (tryOuter < sBest)
916 sBest = tryOuter;
917
918 if (InnerSurfaceExists())
919 {
920 G4double tryInner = ApproxDistOutside( r,absZ,innerRadius,tanInnerStereo );
921 if (tryInner < sBest) sBest = tryInner;
922 }
923
924 return sBest < 0.5*kCarTolerance ? 0 : sBest;
925}

928//
929// IntersectHype (static)
930//
931// Decide if and where a line intersects with a hyperbolic
932// surface (of infinite extent)
933//
934// Arguments:
935// p - (in) Point on trajectory
936// v - (in) Vector along trajectory
937// r2 - (in) Square of radius at z = 0
938// tan2phi - (in) std::tan(phi)**2
939// q - (out) Up to two points of intersection, where the
940// intersection point is p + q*v, and if there are
941// two intersections, q[0] < q[1]. May be negative.
942// Returns:
943// The number of intersections. If 0, the trajectory misses.
944//
945//
946// Equation of a line:
947//
948// x = x0 + q*tx y = y0 + q*ty z = z0 + q*tz
949//
950// Equation of a hyperbolic surface:
951//
952// x**2 + y**2 = r**2 + (z*tanPhi)**2
953//
954// Solution is quadratic:
955//
956// a*q**2 + b*q + c = 0
957//
958// where:
959//
960// a = tx**2 + ty**2 - (tz*tanPhi)**2
961//
962// b = 2*( x0*tx + y0*ty - z0*tz*tanPhi**2 )
963//
964// c = x0**2 + y0**2 - r**2 - (z0*tanPhi)**2
965//
966//
967G4int G4Hype::IntersectHype( const G4ThreeVector &p, const G4ThreeVector &v,
968 G4double r2, G4double tan2Phi, G4double ss[2] )
969{
970 G4double x0 = p.x(), y0 = p.y(), z0 = p.z();
971 G4double tx = v.x(), ty = v.y(), tz = v.z();
972
973 G4double a = tx*tx + ty*ty - tz*tz*tan2Phi;
974 G4double b = 2*( x0*tx + y0*ty - z0*tz*tan2Phi );
975 G4double c = x0*x0 + y0*y0 - r2 - z0*z0*tan2Phi;
976
977 if (std::fabs(a) < DBL_MIN)
978 {
979 //
980 // The trajectory is parallel to the asympotic limit of
981 // the surface: single solution
982 //
983 if (std::fabs(b) < DBL_MIN) return 0;
984 // Unless we travel through exact center
985
986 ss[0] = c/b;
987 return 1;
988 }
989
990
991 G4double radical = b*b - 4*a*c;
992
993 if (radical < -DBL_MIN) return 0; // No solution
994
995 if (radical < DBL_MIN)
996 {
997 //
998 // Grazes surface
999 //
1000 ss[0] = -b/a/2.0;
1001 return 1;
1002 }
1003
1004 radical = std::sqrt(radical);
1005
1006 G4double q = -0.5*( b + (b < 0 ? -radical : +radical) );
1007 G4double sa = q/a;
1008 G4double sb = c/q;
1009 if (sa < sb) { ss[0] = sa; ss[1] = sb; } else { ss[0] = sb; ss[1] = sa; }
1010 return 2;
1011}

1014//
1015// ApproxDistOutside (static)
1016//
1017// Find the approximate distance of a point outside
1018// (greater radius) of a hyperbolic surface. The distance
1019// must be an underestimate. It will also be nice (although
1020// not necesary) that the estimate is always finite no
1021// matter how close the point is.
1022//
1023// Our hyperbola approaches the asymptotic limit at z = +/- infinity
1024// to the lines r = z*tanPhi. We call these lines the
1025// asymptotic limit line.
1026//
1027// We need the distance of the 2d point p(r,z) to the
1028// hyperbola r**2 = r0**2 + (z*tanPhi)**2. Find two
1029// points that bracket the true normal and use the
1030// distance to the line that connects these two points.
1031// The first such point is z=p.z. The second point is
1032// the z position on the asymptotic limit line that
1033// contains the normal on the line through the point p.
1034//
1035G4double G4Hype::ApproxDistOutside( G4double pr, G4double pz,
1036 G4double r0, G4double tanPhi )
1037{
1038 if (tanPhi < DBL_MIN) return pr-r0;
1039
1040 G4double tan2Phi = tanPhi*tanPhi;
1041
1042 //
1043 // First point
1044 //
1045 G4double z1 = pz;
1046 G4double r1 = std::sqrt( r0*r0 + z1*z1*tan2Phi );
1047
1048 //
1049 // Second point
1050 //
1051 G4double z2 = (pr*tanPhi + pz)/(1 + tan2Phi);
1052 G4double r2 = std::sqrt( r0*r0 + z2*z2*tan2Phi );
1053
1054 //
1055 // Line between them
1056 //
1057 G4double dr = r2-r1;
1058 G4double dz = z2-z1;
1059
1060 G4double len = std::sqrt(dr*dr + dz*dz);
1061 if (len < DBL_MIN)
1062 {
1063 //
1064 // The two points are the same?? I guess we
1065 // must have really bracketed the normal
1066 //
1067 dr = pr-r1;
1068 dz = pz-z1;
1069 return std::sqrt( dr*dr + dz*dz );
1070 }
1071
1072 //
1073 // Distance
1074 //
1075 return std::fabs((pr-r1)*dz - (pz-z1)*dr)/len;
1076}
1077
1078//
1079// ApproxDistInside (static)
1080//
1081// Find the approximate distance of a point inside
1082// of a hyperbolic surface. The distance
1083// must be an underestimate. It will also be nice (although
1084// not necesary) that the estimate is always finite no
1085// matter how close the point is.
1086//
1087// This estimate uses the distance to a line tangent to
1088// the hyperbolic function. The point of tangent is chosen
1089// by the z position point
1090//
1091// Assumes pr and pz are positive
1092//
1093G4double G4Hype::ApproxDistInside( G4double pr, G4double pz,
1094 G4double r0, G4double tan2Phi )
1095{
1096 if (tan2Phi < DBL_MIN) return r0 - pr;
1097
1098 //
1099 // Corresponding position and normal on hyperbolic
1100 //
1101 G4double rh = std::sqrt( r0*r0 + pz*pz*tan2Phi );
1102
1103 G4double dr = -rh;
1104 G4double dz = pz*tan2Phi;
1105 G4double len = std::sqrt(dr*dr + dz*dz);
1106
1107 //
1108 // Answer
1109 //
1110 return std::fabs((pr-rh)*dr)/len;
1111}

1178//
1179// GetPointOnSurface
1180//
1181G4ThreeVector G4Hype::GetPointOnSurface() const
1182{
1183 G4double xRand, yRand, zRand, r2 , aOne, aTwo, aThree, chose, sinhu;
1184 G4double phi, cosphi, sinphi, rBar2Out, rBar2In, alpha, t, rOut, rIn2, rOut2;
1185
1186 // we use the formula of the area of a surface of revolution to compute
1187 // the areas, using the equation of the hyperbola:
1188 // x^2 + y^2 = (z*tanphi)^2 + r^2
1189
1190 rBar2Out = outerRadius2;
1191 alpha = 2.*pi*rBar2Out*std::cos(outerStereo)/tanOuterStereo;
1192 t = halfLenZ*tanOuterStereo/(outerRadius*std::cos(outerStereo));
1193 t = std::log(t+std::sqrt(sqr(t)+1));
1194 aOne = std::fabs(2.*alpha*(std::sinh(2.*t)/4.+t/2.));
1195
1196
1197 rBar2In = innerRadius2;
1198 alpha = 2.*pi*rBar2In*std::cos(innerStereo)/tanInnerStereo;
1199 t = halfLenZ*tanInnerStereo/(innerRadius*std::cos(innerStereo));
1200 t = std::log(t+std::sqrt(sqr(t)+1));
1201 aTwo = std::fabs(2.*alpha*(std::sinh(2.*t)/4.+t/2.));
1202
1203 aThree = pi*((outerRadius2+sqr(halfLenZ*tanOuterStereo)
1204 -(innerRadius2+sqr(halfLenZ*tanInnerStereo))));
1205
1206 if(outerStereo == 0.) {aOne = std::fabs(2.*pi*outerRadius*2.*halfLenZ);}
1207 if(innerStereo == 0.) {aTwo = std::fabs(2.*pi*innerRadius*2.*halfLenZ);}
1208
1209 phi = G4RandFlat::shoot(0.,2.*pi);
1210 cosphi = std::cos(phi);
1211 sinphi = std::sin(phi);
1212 sinhu = G4RandFlat::shoot(-1.*halfLenZ*tanOuterStereo/outerRadius,
1213 halfLenZ*tanOuterStereo/outerRadius);
1214
1215 chose = G4RandFlat::shoot(0.,aOne+aTwo+2.*aThree);
1216 if(chose>=0. && chose < aOne)
1217 {
1218 if(outerStereo != 0.)
1219 {
1220 zRand = outerRadius*sinhu/tanOuterStereo;
1221 xRand = std::sqrt(sqr(sinhu)+1)*outerRadius*cosphi;
1222 yRand = std::sqrt(sqr(sinhu)+1)*outerRadius*sinphi;
1223 return G4ThreeVector (xRand, yRand, zRand);
1224 }
1225 else
1226 {
1227 return G4ThreeVector(outerRadius*cosphi,outerRadius*sinphi,
1228 G4RandFlat::shoot(-halfLenZ,halfLenZ));
1229 }
1230 }
1231 else if(chose>=aOne && chose<aOne+aTwo)
1232 {
1233 if(innerStereo != 0.)
1234 {
1235 sinhu = G4RandFlat::shoot(-1.*halfLenZ*tanInnerStereo/innerRadius,
1236 halfLenZ*tanInnerStereo/innerRadius);
1237 zRand = innerRadius*sinhu/tanInnerStereo;
1238 xRand = std::sqrt(sqr(sinhu)+1)*innerRadius*cosphi;
1239 yRand = std::sqrt(sqr(sinhu)+1)*innerRadius*sinphi;
1240 return G4ThreeVector (xRand, yRand, zRand);
1241 }
1242 else
1243 {
1244 return G4ThreeVector(innerRadius*cosphi,innerRadius*sinphi,
1245 G4RandFlat::shoot(-1.*halfLenZ,halfLenZ));
1246 }
1247 }
1248 else if(chose>=aOne+aTwo && chose<aOne+aTwo+aThree)
1249 {
1250 rIn2 = innerRadius2+tanInnerStereo2*halfLenZ*halfLenZ;
1251 rOut2 = outerRadius2+tanOuterStereo2*halfLenZ*halfLenZ;
1252 rOut = std::sqrt(rOut2) ;
1253
1254 do // Loop checking, 13.08.2015, G.Cosmo
1255 {
1256 xRand = G4RandFlat::shoot(-rOut,rOut) ;
1257 yRand = G4RandFlat::shoot(-rOut,rOut) ;
1258 r2 = xRand*xRand + yRand*yRand ;
1259 } while ( ! ( r2 >= rIn2 && r2 <= rOut2 ) ) ;
1260
1261 zRand = halfLenZ;
1262 return G4ThreeVector (xRand, yRand, zRand);
1263 }
1264 else
1265 {
1266 rIn2 = innerRadius2+tanInnerStereo2*halfLenZ*halfLenZ;
1267 rOut2 = outerRadius2+tanOuterStereo2*halfLenZ*halfLenZ;
1268 rOut = std::sqrt(rOut2) ;
1269
1270 do // Loop checking, 13.08.2015, G.Cosmo
1271 {
1272 xRand = G4RandFlat::shoot(-rOut,rOut) ;
1273 yRand = G4RandFlat::shoot(-rOut,rOut) ;
1274 r2 = xRand*xRand + yRand*yRand ;
1275 } while ( ! ( r2 >= rIn2 && r2 <= rOut2 ) ) ;
1276
1277 zRand = -1.*halfLenZ;
1278 return G4ThreeVector (xRand, yRand, zRand);
1279 }
1280}
