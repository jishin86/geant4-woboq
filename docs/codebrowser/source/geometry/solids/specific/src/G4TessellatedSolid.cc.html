229///////////////////////////////////////////////////////////////////////////////
230//
231// Add a facet to the facet list.
232// Note that you can add, but you cannot delete.
233//
234G4bool G4TessellatedSolid::AddFacet (G4VFacet *aFacet)
235{
236 // Add the facet to the vector.
237 //
238 if (fSolidClosed)
239 {
240 G4Exception("G4TessellatedSolid::AddFacet()", "GeomSolids1002",
241 JustWarning, "Attempt to add facets when solid is closed.");
242 return false;
243 }
244 else if (aFacet->IsDefined())
245 {
246 set<G4VertexInfo,G4VertexComparator>::iterator begin
247 = fFacetList.begin(), end = fFacetList.end(), pos, it;
248 G4ThreeVector p = aFacet->GetCircumcentre();
249 G4VertexInfo value;
250 value.id = fFacetList.size();
251 value.mag2 = p.x() + p.y() + p.z();
252
253 G4bool found = false;
254 if (!OutsideOfExtent(p, kCarTolerance))
255 {
256 G4double kCarTolerance3 = 3 * kCarTolerance;
257 pos = fFacetList.lower_bound(value);
258
259 it = pos;
260 while (!found && it != end) // Loop checking, 13.08.2015, G.Cosmo
261 {
262 G4int id = (*it).id;
263 G4VFacet *facet = fFacets[id];
264 G4ThreeVector q = facet->GetCircumcentre();
265 if ((found = (facet == aFacet))) break;
266 G4double dif = q.x() + q.y() + q.z() - value.mag2;
267 if (dif > kCarTolerance3) break;
268 it++;
269 }
270
271 if (fFacets.size() > 1)
272 {
273 it = pos;
274 while (!found && it != begin) // Loop checking, 13.08.2015, G.Cosmo
275 {
276 --it;
277 G4int id = (*it).id;
278 G4VFacet *facet = fFacets[id];
279 G4ThreeVector q = facet->GetCircumcentre();
280 found = (facet == aFacet);
281 if (found) break;
282 G4double dif = value.mag2 - (q.x() + q.y() + q.z());
283 if (dif > kCarTolerance3) break;
284 }
285 }
286 }
287
288 if (!found)
289 {
290 fFacets.push_back(aFacet);
291 fFacetList.insert(value);
292 }
293
294 return true;
295 }
296 else
297 {
298 G4Exception("G4TessellatedSolid::AddFacet()", "GeomSolids1002",
299 JustWarning, "Attempt to add facet not properly defined.");
300 aFacet->StreamInfo(G4cout);
301 return false;
302 }
303}

305///////////////////////////////////////////////////////////////////////////////
306//
307G4int G4TessellatedSolid::SetAllUsingStack(const std::vector<G4int> &voxel,
308 const std::vector<G4int> &max,
309 G4bool status, G4SurfBits &checked)
310{
311 vector<G4int> xyz = voxel;
312 stack<vector<G4int> > pos;
313 pos.push(xyz);
314 G4int filled = 0;
315 G4int cc = 0, nz = 0;
316
317 vector<G4int> candidates;
318
319 while (!pos.empty()) // Loop checking, 13.08.2015, G.Cosmo
320 {
321 xyz = pos.top();
322 pos.pop();
323 G4int index = fVoxels.GetVoxelsIndex(xyz);
324 if (!checked[index])
325 {
326 checked.SetBitNumber(index, true);
327 cc++;
328
329 if (fVoxels.IsEmpty(index))
330 {
331 filled++;
332
333 fInsides.SetBitNumber(index, status);
334
335 for (G4int i = 0; i <= 2; ++i)
336 {
337 if (xyz[i] < max[i] - 1)
338 {
339 xyz[i]++;
340 pos.push(xyz);
341 xyz[i]--;
342 }
343
344 if (xyz[i] > 0)
345 {
346 xyz[i]--;
347 pos.push(xyz);
348 xyz[i]++;
349 }
350 }
351 }
352 else
353 {
354 nz++;
355 }
356 }
357 }
358 return filled;
359}

361///////////////////////////////////////////////////////////////////////////////
362//
363void G4TessellatedSolid::PrecalculateInsides()
364{
365 vector<G4int> voxel(3), maxVoxels(3);
366 for (G4int i = 0; i <= 2; ++i) maxVoxels[i] = fVoxels.GetBoundary(i).size();
367 G4int size = maxVoxels[0] * maxVoxels[1] * maxVoxels[2];
368
369 G4SurfBits checked(size-1);
370 fInsides.Clear();
371 fInsides.ResetBitNumber(size-1);
372
373 G4ThreeVector point;
374 for (voxel[2] = 0; voxel[2] < maxVoxels[2] - 1; ++voxel[2])
375 {
376 for (voxel[1] = 0; voxel[1] < maxVoxels[1] - 1; ++voxel[1])
377 {
378 for (voxel[0] = 0; voxel[0] < maxVoxels[0] - 1; ++voxel[0])
379 {
380 G4int index = fVoxels.GetVoxelsIndex(voxel);
381 if (!checked[index] && fVoxels.IsEmpty(index))
382 {
383 for (G4int i = 0; i <= 2; ++i)
384 {
385 point[i] = fVoxels.GetBoundary(i)[voxel[i]];
386 }
387 G4bool inside = (G4bool) (InsideNoVoxels(point) == kInside);
388 SetAllUsingStack(voxel, maxVoxels, inside, checked);
389 }
390 else checked.SetBitNumber(index);
391 }
392 }
393 }

414///////////////////////////////////////////////////////////////////////////////
415//
416// Compute extremeFacets, i.e. find those facets that have surface
417// planes that bound the volume.
418// Note that this is going to reject concaved surfaces as being extreme. Also
419// note that if the vertex is on the facet, displacement is zero, so IsInside
420// returns true. So will this work?? Need non-equality
421// "G4bool inside = displacement < 0.0;"
422// or
423// "G4bool inside = displacement <= -0.5*kCarTolerance"
424// (Notes from PT 13/08/2007).
425//
426void G4TessellatedSolid::SetExtremeFacets()
427{
428 G4int size = fFacets.size();
429 for (G4int j = 0; j < size; ++j)
430 {
431 G4VFacet &facet = *fFacets[j];
432
433 G4bool isExtreme = true;
434 G4int vsize = fVertexList.size();
435 for (G4int i=0; i < vsize; ++i)
436 {
437 if (!facet.IsInside(fVertexList[i]))
438 {
439 isExtreme = false;
440 break;
441 }
442 }
443 if (isExtreme) fExtremeFacets.insert(&facet);
444 }
445}

447///////////////////////////////////////////////////////////////////////////////
448//
449void G4TessellatedSolid::CreateVertexList()
450{
451 // The algorithm:
452 // we will have additional vertexListSorted, where all the items will be
453 // sorted by magnitude of vertice vector.
454 // New candidate for fVertexList - we will determine the position fo first
455 // item which would be within its magnitude - 0.5*kCarTolerance.
456 // We will go trough until we will reach > +0.5 kCarTolerance.
457 // Comparison (q-p).mag() < 0.5*kCarTolerance will be made.
458 // They can be just stored in std::vector, with custom insertion based
459 // on binary search.
460
461 set<G4VertexInfo,G4VertexComparator> vertexListSorted;
462 set<G4VertexInfo,G4VertexComparator>::iterator begin
463 = vertexListSorted.begin(), end = vertexListSorted.end(), pos, it;
464 G4ThreeVector p;
465 G4VertexInfo value;
466
467 fVertexList.clear();
468 G4int size = fFacets.size();
469
470 G4double kCarTolerance24 = kCarTolerance * kCarTolerance / 4.0;
471 G4double kCarTolerance3 = 3 * kCarTolerance;
472 vector<G4int> newIndex(100);
473
474 for (G4int k = 0; k < size; ++k)
475 {
476 G4VFacet &facet = *fFacets[k];
477 G4int max = facet.GetNumberOfVertices();
478
479 for (G4int i = 0; i < max; ++i)
480 {
481 p = facet.GetVertex(i);
482 value.id = fVertexList.size();
483 value.mag2 = p.x() + p.y() + p.z();
484
485 G4bool found = false;
486 G4int id = 0;
487 if (!OutsideOfExtent(p, kCarTolerance))
488 {
489 pos = vertexListSorted.lower_bound(value);
490 it = pos;
491 while (it != end) // Loop checking, 13.08.2015, G.Cosmo
492 {
493 id = (*it).id;
494 G4ThreeVector q = fVertexList[id];
495 G4double dif = (q-p).mag2();
496 found = (dif < kCarTolerance24);
497 if (found) break;
498 dif = q.x() + q.y() + q.z() - value.mag2;
499 if (dif > kCarTolerance3) break;
500 it++;
501 }
502
503 if (!found && (fVertexList.size() > 1))
504 {
505 it = pos;
506 while (it != begin) // Loop checking, 13.08.2015, G.Cosmo
507 {
508 --it;
509 id = (*it).id;
510 G4ThreeVector q = fVertexList[id];
511 G4double dif = (q-p).mag2();
512 found = (dif < kCarTolerance24);
513 if (found) break;
514 dif = value.mag2 - (q.x() + q.y() + q.z());
515 if (dif > kCarTolerance3) break;
516 }
517 }
518 }
519
520 if (!found)
521 {
522#ifdef G4SPECSDEBUG
523 G4cout << p.x() << ":" << p.y() << ":" << p.z() << G4endl;
524 G4cout << "Adding new vertex #" << i << " of facet " << k
525 << " id " << value.id << G4endl;
526 G4cout << "===" << G4endl;
527#endif
528 fVertexList.push_back(p);
529 vertexListSorted.insert(value);
530 begin = vertexListSorted.begin();
531 end = vertexListSorted.end();
532 newIndex[i] = value.id;
533 //
534 // Now update the maximum x, y and z limits of the volume.
535 //
536 if (value.id == 0) fMinExtent = fMaxExtent = p;
537 else
538 {
539 if (p.x() > fMaxExtent.x()) fMaxExtent.setX(p.x());
540 else if (p.x() < fMinExtent.x()) fMinExtent.setX(p.x());
541 if (p.y() > fMaxExtent.y()) fMaxExtent.setY(p.y());
542 else if (p.y() < fMinExtent.y()) fMinExtent.setY(p.y());
543 if (p.z() > fMaxExtent.z()) fMaxExtent.setZ(p.z());
544 else if (p.z() < fMinExtent.z()) fMinExtent.setZ(p.z());
545 }
546 }
547 else
548 {
549#ifdef G4SPECSDEBUG
550 G4cout << p.x() << ":" << p.y() << ":" << p.z() << G4endl;
551 G4cout << "Vertex #" << i << " of facet " << k
552 << " found, redirecting to " << id << G4endl;
553 G4cout << "===" << G4endl;
554#endif
555 newIndex[i] = id;
556 }
557 }
558 // only now it is possible to change vertices pointer
559 //
560 facet.SetVertices(&fVertexList);
561 for (G4int i = 0; i < max; i++)
562 facet.SetVertexIndex(i,newIndex[i]);
563 }
564 vector<G4ThreeVector>(fVertexList).swap(fVertexList);
565
566#ifdef G4SPECSDEBUG
567 G4double previousValue = 0;
568 for (set<G4VertexInfo,G4VertexComparator>::iterator res=
569 vertexListSorted.begin(); res!=vertexListSorted.end(); ++res)
570 {
571 G4int id = (*res).id;
572 G4ThreeVector vec = fVertexList[id];
573 G4double mvalue = vec.x() + vec.y() + vec.z();
574 if (previousValue && (previousValue - 1e-9 > mvalue))
575 G4cout << "Error in CreateVertexList: previousValue " << previousValue
576 << " is smaller than mvalue " << mvalue << G4endl;
577 previousValue = mvalue;
578 }
579#endif
580}

661///////////////////////////////////////////////////////////////////////////////
662//
663EInside G4TessellatedSolid::InsideVoxels(const G4ThreeVector &p) const
664{
665 //
666 // First the simple test - check if we're outside of the X-Y-Z extremes
667 // of the tessellated solid.
668 //
669 if (OutsideOfExtent(p, kCarTolerance))
670 return kOutside;
671
672 vector<G4int> startingVoxel(3);
673 fVoxels.GetVoxel(startingVoxel, p);
674
675 const G4double dirTolerance = 1.0E-14;
676
677 const vector<G4int> &startingCandidates =
678 fVoxels.GetCandidates(startingVoxel);
679 G4int limit = startingCandidates.size();
680 if (limit == 0 && fInsides.GetNbits())
681 {
682 G4int index = fVoxels.GetPointIndex(p);
683 EInside location = fInsides[index] ? kInside : kOutside;
684 return location;
685 }
686
687 G4double minDist = kInfinity;
688
689 for(G4int i = 0; i < limit; ++i)
690 {
691 G4int candidate = startingCandidates[i];
692 G4VFacet &facet = *fFacets[candidate];
693 G4double dist = facet.Distance(p,minDist);
694 if (dist < minDist) minDist = dist;
695 if (dist <= kCarToleranceHalf)
696 return kSurface;
697 }
698
699 // The following is something of an adaptation of the method implemented by
700 // Rickard Holmberg augmented with information from Schneider & Eberly,
701 // "Geometric Tools for Computer Graphics," pp700-701, 2003. In essence,
702 // we're trying to determine whether we're inside the volume by projecting
703 // a few rays and determining if the first surface crossed is has a normal
704 // vector between 0 to pi/2 (out-going) or pi/2 to pi (in-going).
705 // We should also avoid rays which are nearly within the plane of the
706 // tessellated surface, and therefore produce rays randomly.
707 // For the moment, this is a bit over-engineered (belt-braces-and-ducttape).
708 //
709 G4double distOut = kInfinity;
710 G4double distIn = kInfinity;
711 G4double distO = 0.0;
712 G4double distI = 0.0;
713 G4double distFromSurfaceO = 0.0;
714 G4double distFromSurfaceI = 0.0;
715 G4ThreeVector normalO, normalI;
716 G4bool crossingO = false;
717 G4bool crossingI = false;
718 EInside location = kOutside;
719 G4int sm = 0;
720
721 G4bool nearParallel = false;
722 do // Loop checking, 13.08.2015, G.Cosmo
723 {
724 // We loop until we find direction where the vector is not nearly parallel
725 // to the surface of any facet since this causes ambiguities. The usual
726 // case is that the angles should be sufficiently different, but there
727 // are 20 random directions to select from - hopefully sufficient.
728 //
729 distOut = distIn = kInfinity;
730 const G4ThreeVector &v = fRandir[sm];
731 sm++;
732 //
733 // This code could be voxelized by the same algorithm, which is used for
734 // DistanceToOut(). We will traverse through fVoxels. we will call
735 // intersect only for those, which would be candidates and was not
736 // checked before.
737 //
738 G4ThreeVector currentPoint = p;
739 G4ThreeVector direction = v.unit();
740 // G4SurfBits exclusion(fVoxels.GetBitsPerSlice());
741 vector<G4int> curVoxel(3);

766 crossed = true;

806 // the ray is entering the facet or exiting.

845 location = kOutside;

852///////////////////////////////////////////////////////////////////////////////
853//
854EInside G4TessellatedSolid::InsideNoVoxels (const G4ThreeVector &p) const
855{
856 //
857 // First the simple test - check if we're outside of the X-Y-Z extremes
858 // of the tessellated solid.
859 //
860 if (OutsideOfExtent(p, kCarTolerance))
861 return kOutside;
862
863 const G4double dirTolerance = 1.0E-14;
864
865 G4double minDist = kInfinity;
866 //
867 // Check if we are close to a surface
868 //
869 G4int size = fFacets.size();
870 for (G4int i = 0; i < size; ++i)
871 {
872 G4VFacet &facet = *fFacets[i];
873 G4double dist = facet.Distance(p,minDist);
874 if (dist < minDist) minDist = dist;
875 if (dist <= kCarToleranceHalf)
876 {
877 return kSurface;
878 }
879 }
880 //
881 // The following is something of an adaptation of the method implemented by
882 // Rickard Holmberg augmented with information from Schneider & Eberly,
883 // "Geometric Tools for Computer Graphics," pp700-701, 2003. In essence, we're
884 // trying to determine whether we're inside the volume by projecting a few
885 // rays and determining if the first surface crossed is has a normal vector
886 // between 0 to pi/2 (out-going) or pi/2 to pi (in-going). We should also
887 // avoid rays which are nearly within the plane of the tessellated surface,
888 // and therefore produce rays randomly. For the moment, this is a bit
889 // over-engineered (belt-braces-and-ducttape).
890 //
891#if G4SPECSDEBUG
892 G4int nTry = 7;
893#else
894 G4int nTry = 3;
895#endif

897 G4double distIn = kInfinity;
898 G4double distO = 0.0;
899 G4double distI = 0.0;
900 G4double distFromSurfaceO = 0.0;
901 G4double distFromSurfaceI = 0.0;
902 G4ThreeVector normalO(0.0,0.0,0.0);
903 G4ThreeVector normalI(0.0,0.0,0.0);
904 G4bool crossingO = false;
905 G4bool crossingI = false;
906 EInside location = kOutside;
907 EInside locationprime = kOutside;
908 G4int sm = 0;
909
910 for (G4int i=0; i<nTry; ++i)
911 {
912 G4bool nearParallel = false;
913 do // Loop checking, 13.08.2015, G.Cosmo
914 {
915 //
916 // We loop until we find direction where the vector is not nearly parallel
917 // to the surface of any facet since this causes ambiguities. The usual
918 // case is that the angles should be sufficiently different, but there
919 // are 20 random directions to select from - hopefully sufficient.
920 //
921 distOut = distIn = kInfinity;
922 G4ThreeVector v = fRandir[sm];
923 sm++;
924 vector<G4VFacet*>::const_iterator f = fFacets.begin();
925
926 do // Loop checking, 13.08.2015, G.Cosmo
927 {
928 //
929 // Here we loop through the facets to find out if there is an
930 // intersection between the ray and that facet. The test if performed
931 // separately whether the ray is entering the facet or exiting.
932 //
933 crossingO = ((*f)->Intersect(p,v,true,distO,distFromSurfaceO,normalO));
934 crossingI = ((*f)->Intersect(p,v,false,distI,distFromSurfaceI,normalI));
935 if (crossingO || crossingI)
936 {
937 nearParallel = (crossingO && std::fabs(normalO.dot(v))<dirTolerance)
938 || (crossingI && std::fabs(normalI.dot(v))<dirTolerance);
939 if (!nearParallel)
940 {
941 if (crossingO && distO > 0.0 && distO < distOut) distOut = distO;
942 if (crossingI && distI > 0.0 && distI < distIn) distIn = distI;
943 }
944 }
945 } while (!nearParallel && ++f!=fFacets.end());
946 } while (nearParallel && sm!=fMaxTries);
947
948#ifdef G4VERBOSE
949 if (sm == fMaxTries)
950 {
951 //
952 // We've run out of random vector directions. If nTries is set
953 // sufficiently low (nTries <= 0.5*maxTries) then this would indicate
954 // that there is something wrong with geometry.
955 //
956 std::ostringstream message;
957 G4int oldprc = message.precision(16);
958 message << "Cannot determine whether point is inside or outside volume!"
959 << G4endl
960 << "Solid name = " << GetName() << G4endl
961 << "Geometry Type = " << fGeometryType << G4endl
962 << "Number of facets = " << fFacets.size() << G4endl
963 << "Position:" << G4endl << G4endl
964 << "p.x() = " << p.x()/mm << " mm" << G4endl
965 << "p.y() = " << p.y()/mm << " mm" << G4endl
966 << "p.z() = " << p.z()/mm << " mm";
967 message.precision(oldprc);
968 G4Exception("G4TessellatedSolid::Inside()",
969 "GeomSolids1002", JustWarning, message);
970 }
971#endif
972 //
973 // In the next if-then-elseif G4String the logic is as follows:
974 // (1) You don't hit anything so cannot be inside volume, provided volume
975 // constructed correctly!
976 // (2) Distance to inside (ie. nearest facet such that you enter facet) is
977 // shorter than distance to outside (nearest facet such that you exit
978 // facet) - on condition of safety distance - therefore we're outside.
979 // (3) Distance to outside is shorter than distance to inside therefore
980 // we're inside.
981 //
982 if (distIn == kInfinity && distOut == kInfinity)
983 locationprime = kOutside;
984 else if (distIn <= distOut - kCarToleranceHalf)
985 locationprime = kOutside;
986 else if (distOut <= distIn - kCarToleranceHalf)
987 locationprime = kInside;
988
989 if (i == 0) location = locationprime;
990 }
991
992 return location;

1438//
1439G4double
1440G4TessellatedSolid::MinDistanceFacet(const G4ThreeVector &p,
1441 G4bool simple,
1442 G4VFacet * &minFacet) const
1443{
1444 G4double minDist = kInfinity;
1445
1446 G4int size = fVoxels.GetVoxelBoxesSize();
1447 vector<pair<G4int, G4double> > voxelsSorted(size);
1448
1449 pair<G4int, G4double> info;
1450
1451 for (G4int i = 0; i < size; ++i)
1452 {
1453 const G4VoxelBox &voxelBox = fVoxels.GetVoxelBox(i);
1454
1455 G4ThreeVector pointShifted = p - voxelBox.pos;
1456 G4double safety = fVoxels.MinDistanceToBox(pointShifted, voxelBox.hlen);
1457 info.first = i;
1458 info.second = safety;
1459 voxelsSorted[i] = info;
1460 }
1461
1462 std::sort(voxelsSorted.begin(), voxelsSorted.end(),
1463 &G4TessellatedSolid::CompareSortedVoxel);
1464
1465 for (G4int i = 0; i < size; ++i)
1466 {
1467 const pair<G4int,G4double> &inf = voxelsSorted[i];
1468 G4double dist = inf.second;
1469 if (dist > minDist) break;
1470
1471 const vector<G4int> &candidates = fVoxels.GetVoxelBoxCandidates(inf.first);
1472 G4int csize = candidates.size();

1474 {
1475 G4int candidate = candidates[j];
1476 G4VFacet &facet = *fFacets[candidate];
1477 dist = simple ? facet.Distance(p,minDist)
1478 : facet.Distance(p,minDist,false);
1479 if (dist < minDist)
1480 {
1481 minDist = dist;
1482 minFacet = &facet;
1483 }
1484 }
1485 }
1486 return minDist;
1487}

1851///////////////////////////////////////////////////////////////////////////////
1852//
1853// Calculate extent under transform and specified limit
1854//
1855G4bool
1856G4TessellatedSolid::CalculateExtent(const EAxis pAxis,
1857 const G4VoxelLimits& pVoxelLimit,
1858 const G4AffineTransform& pTransform,
1859 G4double& pMin, G4double& pMax) const
1860{
1861 G4ThreeVector bmin, bmax;
1862
1863 // Check bounding box (bbox)
1864 //
1865 BoundingLimits(bmin,bmax);
1866 G4BoundingEnvelope bbox(bmin,bmax);
1867
1868 // Use simple bounding-box to help in the case of complex meshes
1869 //
1870 return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
1871
1872#if 0
1873 // Precise extent computation (disabled by default for this shape)
1874 //
1875 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
1876 {
1877 return (pMin < pMax) ? true : false;
1878 }
1879
1880 // The extent is calculated as cumulative extent of the pyramids
1881 // formed by facets and the center of the bounding box.
1882 //
1883 G4double eminlim = pVoxelLimit.GetMinExtent(pAxis);
1884 G4double emaxlim = pVoxelLimit.GetMaxExtent(pAxis);
1885
1886 G4ThreeVectorList base;
1887 G4ThreeVectorList apex(1);
1888 std::vector<const G4ThreeVectorList *> pyramid(2);
1889 pyramid[0] = &base;
1890 pyramid[1] = &apex;
1891 apex[0] = (bmin+bmax)*0.5;
1892
1893 // main loop along facets
1894 pMin = kInfinity;
1895 pMax = -kInfinity;
1896 for (G4int i=0; i<GetNumberOfFacets(); ++i)
1897 {
1898 G4VFacet* facet = GetFacet(i);
1899 if (std::abs((facet->GetSurfaceNormal()).dot(facet->GetVertex(0)-apex[0]))
1900 < kCarToleranceHalf) continue;
1901
1902 G4int nv = facet->GetNumberOfVertices();
1903 base.resize(nv);
1904 for (G4int k=0; k<nv; ++k) { base[k] = facet->GetVertex(k); }
1905
1906 G4double emin,emax;
1907 G4BoundingEnvelope benv(pyramid);
1908 if (!benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,emin,emax)) continue;
1909 if (emin < pMin) pMin = emin;
1910 if (emax > pMax) pMax = emax;
1911 if (eminlim > pMin && emaxlim < pMax) break; // max possible extent
1912 }
1913 return (pMin < pMax);
1914#endif
1915}
