77//______________________________________________________________________________
78void G4Voxelizer::BuildEmpty()
79{
80 // by reserving the size of candidates, we would avoid reallocation of
81 // the vector which could cause fragmentation
82 //
83 std::vector<G4int> xyz(3), max(3), candidates(fTotalCandidates);
84 const std::vector<G4int> empty(0);
85
86 for (G4int i = 0; i <= 2; i++) max[i] = fBoundaries[i].size();
87 unsigned int size = max[0] * max[1] * max[2];
88
89 fEmpty.Clear();
90 fEmpty.ResetBitNumber(size-1);
91 fEmpty.ResetAllBits(true);
92
93 for (xyz[2] = 0; xyz[2] < max[2]; ++xyz[2])
94 {
95 for (xyz[1] = 0; xyz[1] < max[1]; ++xyz[1])
96 {
97 for (xyz[0] = 0; xyz[0] < max[0]; ++xyz[0])
98 {
99 if (GetCandidatesVoxelArray(xyz, candidates))
100 {
101 G4int index = GetVoxelsIndex(xyz);
102 fEmpty.SetBitNumber(index, false);
103
104 // rather than assigning directly with:
105 // "fCandidates[index] = candidates;", in an effort to ensure that
106 // capacity would be just exact, we rather use following 3 lines
107 //
108 std::vector<G4int> &c = (fCandidates[index] = empty);
109 c.reserve(candidates.size());
110 c.assign(candidates.begin(), candidates.end());
111 }
112 }
113 }
114 }
115#ifdef G4SPECSDEBUG
116 G4cout << "Non-empty voxels count: " << fCandidates.size() << G4endl;
117#endif
118}

120//______________________________________________________________________________
121void G4Voxelizer::BuildVoxelLimits(std::vector<G4VSolid*>& solids,
122 std::vector<G4Transform3D>& transforms)
123{
124 G4Rotate3D rot;
125 G4Translate3D transl ;
126 G4Scale3D scale;
127
128 // "BuildVoxelLimits"'s aim is to store the coordinates of the origin as
129 // well as the half lengths related to the bounding box of each node.
130 // These quantities are stored in the array "fBoxes" (6 different values per
131 // node
132 //
133 if (G4int numNodes = solids.size()) // Number of nodes in "multiUnion"
134 {
135 fBoxes.resize(numNodes); // Array which will store the half lengths
136 fNPerSlice = 1 + (fBoxes.size() - 1) / (8 * sizeof(unsigned int));
137
138 // related to a particular node, but also
139 // the coordinates of its origin
140
141 G4ThreeVector toleranceVector(fTolerance,fTolerance,fTolerance);
142
143 for (G4int i = 0; i < numNodes; ++i)
144 {
145 G4VSolid& solid = *solids[i];
146 G4Transform3D transform = transforms[i];
147 G4ThreeVector min, max;
148
149 solid.BoundingLimits(min, max);
150 if (solid.GetEntityType() == "G4Orb")
151 {
152 G4Orb& orb = *(G4Orb*) &solid;
153 G4ThreeVector orbToleranceVector;
154 G4double tolerance = orb.GetRadialTolerance() / 2.0;
155 orbToleranceVector.set(tolerance,tolerance,tolerance);
156 min -= orbToleranceVector;
157 max += orbToleranceVector;
158 }
159 else
160 {
161 min -= toleranceVector;
162 max += toleranceVector;
163 }
164 TransformLimits(min, max, transform);
165 fBoxes[i].hlen = (max - min) / 2;
166 transform.getDecomposition(scale,rot,transl);
167 fBoxes[i].pos = transl.getTranslation();
168 }
169 fTotalCandidates = fBoxes.size();
170 }
171}

173//______________________________________________________________________________
174void G4Voxelizer::BuildVoxelLimits(std::vector<G4VFacet *> &facets)
175{
176 // "BuildVoxelLimits"'s aim is to store the coordinates of the origin as well
177 // as the half lengths related to the bounding box of each node.
178 // These quantities are stored in the array "fBoxes" (6 different values per
179 // node.
180
181 if (G4int numNodes = facets.size()) // Number of nodes
182 {
183 fBoxes.resize(numNodes); // Array which will store the half lengths
184 fNPerSlice = 1+(fBoxes.size()-1)/(8*sizeof(unsigned int));
185
186 G4ThreeVector toleranceVector(10*fTolerance, 10*fTolerance, 10*fTolerance);
187
188 for (G4int i = 0; i < numNodes; ++i)
189 {
190 G4VFacet &facet = *facets[i];
191 G4ThreeVector min, max;
192 G4ThreeVector x(1,0,0), y(0,1,0), z(0,0,1);
193 G4ThreeVector extent;
194 max.set (facet.Extent(x), facet.Extent(y), facet.Extent(z));
195 min.set (-facet.Extent(-x), -facet.Extent(-y), -facet.Extent(-z));
196 min -= toleranceVector;
197 max += toleranceVector;
198 G4ThreeVector hlen = (max - min) / 2;
199 fBoxes[i].hlen = hlen;
200 fBoxes[i].pos = min + hlen;
201 }
202 fTotalCandidates = fBoxes.size();
203 }
204}

223//______________________________________________________________________________
224void G4Voxelizer::CreateSortedBoundary(std::vector<G4double> &boundary,
225 G4int axis)
226{
227 // "CreateBoundaries"'s aim is to determine the slices induced by the
228 // bounding fBoxes, along each axis. The created boundaries are stored
229 // in the array "boundariesRaw"
230
231 G4int numNodes = fBoxes.size(); // Number of nodes in structure
232
233 // Determination of the boundaries along x, y and z axis
234 //
235 for(G4int i = 0 ; i < numNodes; ++i)
236 {
237 // For each node, the boundaries are created by using the array "fBoxes"
238 // built in method "BuildVoxelLimits"
239 //
240 G4double p = fBoxes[i].pos[axis], d = fBoxes[i].hlen[axis];
241
242 // x boundaries
243 //
244#ifdef G4SPECSDEBUG
245 G4cout << "Boundary " << p - d << " - " << p + d << G4endl;
246#endif
247 boundary[2*i] = p - d;
248 boundary[2*i+1] = p + d;
249 }
250 std::sort(boundary.begin(), boundary.end());
251}

253//______________________________________________________________________________
254void G4Voxelizer::BuildBoundaries()
255{
256 // "SortBoundaries" orders the boundaries along each axis (increasing order)
257 // and also does not take into account redundant boundaries, i.e. if two
258 // boundaries are separated by a distance strictly inferior to "tolerance".
259 // The sorted boundaries are respectively stored in:
260 // * boundaries[0..2]
261
262 // In addition, the number of elements contained in the three latter arrays
263 // are precise thanks to variables: boundariesCountX, boundariesCountY and
264 // boundariesCountZ.
265
266 if (G4int numNodes = fBoxes.size())
267 {
268 const G4double tolerance = fTolerance / 100.0;
269 // Minimal distance to discriminate two boundaries.
270
271 std::vector<G4double> sortedBoundary(2*numNodes);
272
273 G4int considered;
274
275 for (G4int j = 0; j <= 2; ++j)
276 {
277 CreateSortedBoundary(sortedBoundary, j);
278 std::vector<G4double> &boundary = fBoundaries[j];
279 boundary.clear();
280
281 considered = 0;
282
283 for(G4int i = 0 ; i < 2*numNodes; ++i)
284 {
285 G4double newBoundary = sortedBoundary[i];
286#ifdef G4SPECSDEBUG
287 if (j == 0) G4cout << "Examining " << newBoundary << "..." << G4endl;
288#endif
289 G4int size = boundary.size();
290 if(!size || std::abs(boundary[size-1] - newBoundary) > tolerance)
291 {
292 considered++;
293 {
294#ifdef G4SPECSDEBUG
295 if (j == 0) G4cout << "Adding boundary " << newBoundary << "..."
296 << G4endl;
297#endif
298 boundary.push_back(newBoundary);
299 continue;
300 }
301 }
302 // If two successive boundaries are too close from each other,
303 // only the first one is considered
304 }
305
306 G4int n = boundary.size();
307 G4int max = 100000;
308 if (n > max/2)
309 {
310 G4int skip = n / (max /2); // n has to be 2x bigger then 50.000.
311 // therefore only from 100.000 reduced
312 std::vector<G4double> reduced;
313 for (G4int i = 0; i < n; ++i)
314 {
315 // 50 ok for 2k, 1000, 2000
316 G4int size = boundary.size();
317 if (i % skip == 0 || i == 0 || i == size - 1)
318 {
319 // this condition of merging boundaries was wrong,
320 // it did not count with right part, which can be
321 // completely ommited and not included in final consideration.
322 // Now should be OK
323 //
324 reduced.push_back(boundary[i]);
325 }
326 }
327 boundary = reduced;
328 }
329 }
330 }
331}

360//______________________________________________________________________________
361void G4Voxelizer::BuildBitmasks(std::vector<G4double> boundaries[],
362 G4SurfBits bitmasks[], G4bool countsOnly)
363{
364 // "BuildListNodes" stores in the bitmasks solids present in each slice
365 // along an axis.
366
367 G4int numNodes = fBoxes.size();
368 G4int bitsPerSlice = GetBitsPerSlice();
369
370 for (G4int k = 0; k < 3; ++k)
371 {
372 G4int total = 0;
373 std::vector<G4double> &boundary = boundaries[k];
374 G4int voxelsCount = boundary.size() - 1;
375 G4SurfBits &bitmask = bitmasks[k];
376
377 if (!countsOnly)
378 {
379 bitmask.Clear();
380#ifdef G4SPECSDEBUG
381 G4cout << "Allocating bitmask..." << G4endl;
382#endif
383 bitmask.SetBitNumber(voxelsCount*bitsPerSlice-1, false);
384 // it is here so we can set the maximum number of bits. this line
385 // will rellocate the memory and set all to zero
386 }
387 std::vector<G4int> &candidatesCount = fCandidatesCounts[k];
388 candidatesCount.resize(voxelsCount);
389
390 for(G4int i = 0 ; i < voxelsCount; ++i) { candidatesCount[i] = 0; }
391
392 // Loop on the nodes, number of slices per axis
393 //
394 for(G4int j = 0 ; j < numNodes; ++j)
395 {
396 // Determination of the minimum and maximum position along x
397 // of the bounding boxe of each node
398 //
399 G4double p = fBoxes[j].pos[k], d = fBoxes[j].hlen[k];
400
401 G4double min = p - d; // - localTolerance;
402 G4double max = p + d; // + localTolerance;
403
404 G4int i = BinarySearch(boundary, min);
405 if (i < 0) { i = 0; }
406
407 do // Loop checking, 13.08.2015, G.Cosmo
408 {
409 if (!countsOnly)
410 {
411 bitmask.SetBitNumber(i*bitsPerSlice+j);
412 }
413 candidatesCount[i]++;
414 total++;
415 i++;
416 }
417 while (max > boundary[i] && i < voxelsCount);
418 }
419 }
420#ifdef G4SPECSDEBUG
421 G4cout << "Build list nodes completed." << G4endl;
422#endif
423}

477//______________________________________________________________________________
478void G4Voxelizer::BuildBoundingBox(G4ThreeVector& amin,
479 G4ThreeVector& amax,
480 G4double tolerance)
481{
482 for (G4int i = 0; i <= 2; ++i)
483 {
484 G4double min = amin[i];
485 G4double max = amax[i];
486 fBoundingBoxSize[i] = (max - min) / 2 + tolerance * 0.5;
487 fBoundingBoxCenter[i] = min + fBoundingBoxSize[i];
488 }
489 fBoundingBox.SetXHalfLength(fBoundingBoxSize.x());
490 fBoundingBox.SetYHalfLength(fBoundingBoxSize.y());
491 fBoundingBox.SetZHalfLength(fBoundingBoxSize.z());
492}
493
494// algorithm -
495
496// in order to get balanced voxels, merge should always unite those regions,
497// where the number of voxels is least the number.
498// We will keep sorted list (std::set) with all voxels. there will be
499// comparator function between two voxels, which will tell if voxel is less
500// by looking at his right neighbor.
501// First, we will add all the voxels into the tree.
502// We will be pick the first item in the tree, merging it, adding the right
503// merged voxel into the a list for future reduction (fBitmasks will be
504// rebuilded later, therefore they need not to be updated).
505// The merged voxel need to be added to the tree again, so it's position
506// would be updated.
507
508//______________________________________________________________________________
509void G4Voxelizer::SetReductionRatio(G4int maxVoxels,
510 G4ThreeVector &reductionRatio)
511{
512 G4double maxTotal = (G4double) fCandidatesCounts[0].size()
513 * fCandidatesCounts[1].size() * fCandidatesCounts[2].size();
514
515 if (maxVoxels > 0 && maxVoxels < maxTotal)
516 {
517 G4double ratio = (G4double) maxVoxels / maxTotal;
518 ratio = std::pow(ratio, 1./3.);
519 if (ratio > 1) { ratio = 1; }
520 reductionRatio.set(ratio,ratio,ratio);
521 }
522}

679//______________________________________________________________________________
680void G4Voxelizer::BuildReduceVoxels2(std::vector<G4double> boundaries[],
681 G4ThreeVector reductionRatio)
682{
683 for (G4int k = 0; k <= 2; ++k)
684 {
685 std::vector<G4int> &candidatesCount = fCandidatesCounts[k];
686 G4int max = candidatesCount.size();
687 G4int total = 0;
688 for (G4int i = 0; i < max; ++i) total += candidatesCount[i];
689
690 G4double reduction = reductionRatio[k];
691 if (reduction == 0)
692 break;
693
694 G4int destination = (G4int) (reduction * max) + 1;
695 if (destination > 1000) destination = 1000;
696 if (destination < 2) destination = 2;
697 G4double average = ((G4double)total / max) / reduction;
698
699 std::vector<G4int> mergings;
700
701 std::vector<G4double> &boundary = boundaries[k];
702 std::vector<G4double> reducedBoundary(destination);
703
704 G4int sum = 0, cur = 0;
705 for (G4int i = 0; i < max; ++i)
706 {
707 sum += candidatesCount[i];
708 if (sum > average * (cur + 1) || i == 0)
709 {
710 G4double val = boundary[i];
711 reducedBoundary[cur] = val;
712 cur++;
713 if (cur == destination)
714 break;
715 }
716 }
717 reducedBoundary[destination-1] = boundary[max];
718 boundaries[k] = reducedBoundary;
719 }

921void G4Voxelizer::FindComponentsFastest(unsigned int mask,
922 std::vector<G4int> &list, G4int i)
923{
924 for (G4int byte = 0; byte < (G4int) (sizeof(unsigned int)); byte++)
925 {
926 if (G4int maskByte = mask & 0xFF)
927 {
928 for (G4int bit = 0; bit < 8; bit++)
929 {
930 if (maskByte & 1)
931 { list.push_back(8*(sizeof(unsigned int)*i+ byte) + bit); }
932 if (!(maskByte >>= 1)) break;
933 }
934 }
935 mask >>= 8;
936 }

939//______________________________________________________________________________
940void G4Voxelizer::TransformLimits(G4ThreeVector& min, G4ThreeVector& max,
941 const G4Transform3D& transformation) const
942{
943 // The goal of this method is to convert the quantities min and max
944 // (representing the bounding box of a given solid in its local frame)
945 // to the main frame, using "transformation"
946
947 G4ThreeVector vertices[8] = // Detemination of the vertices thanks to
948 { // the extension of each solid:
949 G4ThreeVector(min.x(), min.y(), min.z()), // 1st vertice:
950 G4ThreeVector(min.x(), max.y(), min.z()), // 2nd vertice:
951 G4ThreeVector(max.x(), max.y(), min.z()),
952 G4ThreeVector(max.x(), min.y(), min.z()),
953 G4ThreeVector(min.x(), min.y(), max.z()),
954 G4ThreeVector(min.x(), max.y(), max.z()),
955 G4ThreeVector(max.x(), max.y(), max.z()),
956 G4ThreeVector(max.x(), min.y(), max.z())
957 };
958
959 min.set(kInfinity,kInfinity,kInfinity);
960 max.set(-kInfinity,-kInfinity,-kInfinity);
961
962 // Loop on th vertices
963 G4int limit = sizeof(vertices) / sizeof(G4ThreeVector);
964 for (G4int i = 0 ; i < limit; ++i)
965 {
966 // From local frame to the global one:
967 // Current positions on the three axis:
968 G4ThreeVector current = GetGlobalPoint(transformation, vertices[i]);
969
970 // If need be, replacement of the min & max values:
971 if (current.x() > max.x()) max.setX(current.x());
972 if (current.x() < min.x()) min.setX(current.x());
973
974 if (current.y() > max.y()) max.setY(current.y());
975 if (current.y() < min.y()) min.setY(current.y());
976
977 if (current.z() > max.z()) max.setZ(current.z());
978 if (current.z() < min.z()) min.setZ(current.z());
979 }
980}

1201G4double
1202G4Voxelizer::MinDistanceToBox (const G4ThreeVector &aPoint,
1203 const G4ThreeVector &f)
1204{
1205 // Estimates the isotropic safety from a point outside the current solid to
1206 // any of its surfaces. The algorithm may be accurate or should provide a
1207 // fast underestimate.
1208
1209 G4double safe, safx, safy, safz;
1210 safe = safx = -f.x() + std::abs(aPoint.x());
1211 safy = -f.y() + std::abs(aPoint.y());
1212 if ( safy > safe ) safe = safy;
1213 safz = -f.z() + std::abs(aPoint.z());
1214 if ( safz > safe ) safe = safz;
1215 if (safe < 0.0) return 0.0; // point is inside
1216
1217 G4double safsq = 0.0;
1218 G4int count = 0;
1219 if ( safx > 0 ) { safsq += safx*safx; count++; }
1220 if ( safy > 0 ) { safsq += safy*safy; count++; }
1221 if ( safz > 0 ) { safsq += safz*safz; count++; }
1222 if (count == 1) return safe;
1223 return std::sqrt(safsq);
1224}
