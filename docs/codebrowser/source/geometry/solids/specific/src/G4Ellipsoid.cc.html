66///////////////////////////////////////////////////////////////////////////////
67//
68// constructor - check parameters, convert angles so 0<sphi+dpshi<=2_PI
69// - note if pDPhi>2PI then reset to 2PI
70
71G4Ellipsoid::G4Ellipsoid(const G4String& pName,
72 G4double pxSemiAxis,
73 G4double pySemiAxis,
74 G4double pzSemiAxis,
75 G4double pzBottomCut,
76 G4double pzTopCut)
77 : G4VSolid(pName), fRebuildPolyhedron(false), fpPolyhedron(0),
78 fCubicVolume(0.), fSurfaceArea(0.), zBottomCut(0.), zTopCut(0.)
79{
80 // note: for users that want to use the full ellipsoid it is useful
81 // to include a default for the cuts
82
83 kRadTolerance = G4GeometryTolerance::GetInstance()->GetRadialTolerance();
84
85 halfCarTolerance = kCarTolerance*0.5;
86 halfRadTolerance = kRadTolerance*0.5;
87
88 // Check Semi-Axis
89 if ( (pxSemiAxis<=0.) || (pySemiAxis<=0.) || (pzSemiAxis<=0.) )
90 {
91 std::ostringstream message;
92 message << "Invalid semi-axis - " << GetName();
93 G4Exception("G4Ellipsoid::G4Ellipsoid()", "GeomSolids0002",
94 FatalErrorInArgument, message);
95 }
96 SetSemiAxis(pxSemiAxis, pySemiAxis, pzSemiAxis);
97
98 if ( pzBottomCut == 0 && pzTopCut == 0 )
99 {
100 SetZCuts(-pzSemiAxis, pzSemiAxis);
101 }
102 else if ( (pzBottomCut < pzSemiAxis) && (pzTopCut > -pzSemiAxis)
103 && (pzBottomCut < pzTopCut) )
104 {
105 SetZCuts(pzBottomCut, pzTopCut);
106 }
107 else
108 {
109 std::ostringstream message;
110 message << "Invalid z-coordinate for cutting plane - " << GetName();
111 G4Exception("G4Ellipsoid::G4Ellipsoid()", "GeomSolids0002",
112 FatalErrorInArgument, message);
113 }
114}

196///////////////////////////////////////////////////////////////////////////////
197//
198// Get bounding box
199
200void G4Ellipsoid::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
201{
202 G4double dx = GetSemiAxisMax(0);
203 G4double dy = GetSemiAxisMax(1);
204 G4double dz = GetSemiAxisMax(2);
205 G4double zmin = std::max(-dz,GetZBottomCut());
206 G4double zmax = std::min( dz,GetZTopCut());
207 pMin.set(-dx,-dy,zmin);
208 pMax.set( dx, dy,zmax);
209
210 // Check correctness of the bounding box
211 //
212 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
213 {
214 std::ostringstream message;
215 message << "Bad bounding box (min >= max) for solid: "
216 << GetName() << " !"
217 << "\npMin = " << pMin
218 << "\npMax = " << pMax;
219 G4Exception("G4Ellipsoid::BoundingLimits()", "GeomMgt0001",
220 JustWarning, message);
221 DumpInfo();
222 }
223}

245///////////////////////////////////////////////////////////////////////////////
246//
247// Return whether point inside/outside/on surface
248// Split into radius, phi, theta checks
249// Each check modifies `in', or returns as approprate
250
251EInside G4Ellipsoid::Inside(const G4ThreeVector& p) const
252{
253 G4double rad2oo, // outside surface outer tolerance
254 rad2oi; // outside surface inner tolerance
255 EInside in;
256
257 // check this side of z cut first, because that's fast
258 //
259 if (p.z() < zBottomCut-halfRadTolerance) { return in=kOutside; }
260 if (p.z() > zTopCut+halfRadTolerance) { return in=kOutside; }
261
262 rad2oo= sqr(p.x()/(xSemiAxis+halfRadTolerance))
263 + sqr(p.y()/(ySemiAxis+halfRadTolerance))
264 + sqr(p.z()/(zSemiAxis+halfRadTolerance));
265
266 if (rad2oo > 1.0) { return in=kOutside; }
267
268 rad2oi= sqr(p.x()*(1.0+halfRadTolerance/xSemiAxis)/xSemiAxis)
269 + sqr(p.y()*(1.0+halfRadTolerance/ySemiAxis)/ySemiAxis)
270 + sqr(p.z()*(1.0+halfRadTolerance/zSemiAxis)/zSemiAxis);
271
272 // Check radial surfaces
273 // sets `in' (already checked for rad2oo > 1.0)
274 //
275 if (rad2oi < 1.0)
276 {
277 in = ( (p.z() < zBottomCut+halfRadTolerance)
278 || (p.z() > zTopCut-halfRadTolerance) ) ? kSurface : kInside;
279 if ( rad2oi > 1.0-halfRadTolerance ) { in=kSurface; }
280 }
281 else
282 {
283 in = kSurface;
284 }
285 return in;
286
287}

289///////////////////////////////////////////////////////////////////////////////
290//
291// Return unit normal of surface closest to p not protected against p=0
292
293G4ThreeVector G4Ellipsoid::SurfaceNormal( const G4ThreeVector& p) const
294{
295 G4double distR, distZBottom, distZTop;
296
297 // normal vector with special magnitude: parallel to normal, units 1/length
298 // norm*p == 1.0 if on surface, >1.0 if outside, <1.0 if inside
299 //
300 G4ThreeVector norm(p.x()/(xSemiAxis*xSemiAxis),
301 p.y()/(ySemiAxis*ySemiAxis),
302 p.z()/(zSemiAxis*zSemiAxis));
303 G4double radius = 1.0/norm.mag();
304
305 // approximate distance to curved surface
306 //
307 distR = std::fabs( (p*norm - 1.0) * radius ) / 2.0;
308
309 // Distance to z-cut plane
310 //
311 distZBottom = std::fabs( p.z() - zBottomCut );
312 distZTop = std::fabs( p.z() - zTopCut );
313
314 if ( (distZBottom < distR) || (distZTop < distR) )
315 {
316 return G4ThreeVector(0.,0.,(distZBottom < distZTop) ? -1.0 : 1.0);
317 }
318 return ( norm *= radius );
319}

321///////////////////////////////////////////////////////////////////////////////
322//
323// Calculate distance to shape from outside, along normalised vector
324// - return kInfinity if no intersection, or intersection distance <= tolerance
325//
326
327G4double G4Ellipsoid::DistanceToIn( const G4ThreeVector& p,
328 const G4ThreeVector& v ) const
329{
330 G4double distMin = std::min(xSemiAxis,ySemiAxis);
331 const G4double dRmax = 100.*std::min(distMin,zSemiAxis);
332 distMin= kInfinity;
333
334 // check to see if Z plane is relevant
335 if (p.z() <= zBottomCut+halfCarTolerance)
336 {
337 if (v.z() <= 0.0) { return distMin; }
338 G4double distZ = (zBottomCut - p.z()) / v.z();
339
340 if ( (distZ > -halfRadTolerance) && (Inside(p+distZ*v) != kOutside) )
341 {
342 // early exit since can't intercept curved surface if we reach here
343 if ( std::fabs(distZ) < halfRadTolerance ) { distZ=0.; }
344 return distMin= distZ;
345 }
346 }
347 if (p.z() >= zTopCut-halfCarTolerance)
348 {
349 if (v.z() >= 0.0) { return distMin;}
350 G4double distZ = (zTopCut - p.z()) / v.z();
351 if ( (distZ > -halfRadTolerance) && (Inside(p+distZ*v) != kOutside) )
352 {
353 // early exit since can't intercept curved surface if we reach here
354 if ( std::fabs(distZ) < halfRadTolerance ) { distZ=0.; }
355 return distMin= distZ;
356 }
357 }
358 // if fZCut1 <= p.z() <= fZCut2, then must hit curved surface
359
360 // now check curved surface intercept
361 G4double A,B,C;
362
363 A= sqr(v.x()/xSemiAxis) + sqr(v.y()/ySemiAxis) + sqr(v.z()/zSemiAxis);
364 C= sqr(p.x()/xSemiAxis) + sqr(p.y()/ySemiAxis) + sqr(p.z()/zSemiAxis) - 1.0;
365 B= 2.0 * ( p.x()*v.x()/(xSemiAxis*xSemiAxis)
366 + p.y()*v.y()/(ySemiAxis*ySemiAxis)
367 + p.z()*v.z()/(zSemiAxis*zSemiAxis) );
368
369 C= B*B - 4.0*A*C;
370 if (C > 0.0)
371 {
372 G4double distR= (-B - std::sqrt(C)) / (2.0*A);
373 G4double intZ = p.z()+distR*v.z();
374 if ( (distR > halfRadTolerance)
375 && (intZ >= zBottomCut-halfRadTolerance)
376 && (intZ <= zTopCut+halfRadTolerance) )
377 {
378 distMin = distR;
379 }
380 else if( (distR >- halfRadTolerance)
381 && (intZ >= zBottomCut-halfRadTolerance)
382 && (intZ <= zTopCut+halfRadTolerance) )
383 {
384 // p is on the curved surface, DistanceToIn returns 0 or kInfinity:
385 // DistanceToIn returns 0, if second root is positive (means going inside)
386 // If second root is negative, DistanceToIn returns kInfinity (outside)
387 //
388 distR = (-B + std::sqrt(C) ) / (2.0*A);
389 if(distR>0.) { distMin=0.; }
390 }
391 else
392 {
393 distR= (-B + std::sqrt(C)) / (2.0*A);
394 intZ = p.z()+distR*v.z();
395 if ( (distR > halfRadTolerance)
396 && (intZ >= zBottomCut-halfRadTolerance)
397 && (intZ <= zTopCut+halfRadTolerance) )
398 {
399 G4ThreeVector norm=SurfaceNormal(p);
400 if (norm.dot(v)<0.) { distMin = distR; }
401 }
402 }
403 if ( (distMin!=kInfinity) && (distMin>dRmax) )
404 { // Avoid rounding errors due to precision issues on
405 // 64 bits systems. Split long distances and recompute
406 G4double fTerm = distMin-std::fmod(distMin,dRmax);
407 distMin = fTerm + DistanceToIn(p+fTerm*v,v);
408 }
409 }
410
411 if (std::fabs(distMin)<halfRadTolerance) { distMin=0.; }
412 return distMin;
413}

415///////////////////////////////////////////////////////////////////////////////
416//
417// Calculate distance (<= actual) to closest surface of shape from outside
418// - Return 0 if point inside
419
420G4double G4Ellipsoid::DistanceToIn(const G4ThreeVector& p) const
421{
422 G4double distR, distZ;
423
424 // normal vector: parallel to normal, magnitude 1/(characteristic radius)
425 //
426 G4ThreeVector norm(p.x()/(xSemiAxis*xSemiAxis),
427 p.y()/(ySemiAxis*ySemiAxis),
428 p.z()/(zSemiAxis*zSemiAxis));
429 G4double radius= 1.0/norm.mag();
430
431 // approximate distance to curved surface ( <= actual distance )
432 //
433 distR= (p*norm - 1.0) * radius / 2.0;
434
435 // Distance to z-cut plane
436 //
437 distZ= zBottomCut - p.z();
438 if (distZ < 0.0)
439 {
440 distZ = p.z() - zTopCut;
441 }
442
443 // Distance to closest surface from outside
444 //
445 if (distZ < 0.0)
446 {
447 return (distR < 0.0) ? 0.0 : distR;
448 }
449 else if (distR < 0.0)
450 {
451 return distZ;
452 }
453 else
454 {
455 return (distZ < distR) ? distZ : distR;
456 }
457}

459///////////////////////////////////////////////////////////////////////////////
460//
461// Calculate distance to surface of shape from `inside', allowing for tolerance
462
463G4double G4Ellipsoid::DistanceToOut(const G4ThreeVector& p,
464 const G4ThreeVector& v,
465 const G4bool calcNorm,
466 G4bool *validNorm,
467 G4ThreeVector *n ) const
468{
469 G4double distMin;
470 enum surface_e {kPlaneSurf, kCurvedSurf, kNoSurf} surface;
471
472 distMin= kInfinity;
473 surface= kNoSurf;
474
475 // check to see if Z plane is relevant
476 //
477 if (v.z() < 0.0)
478 {
479 G4double distZ = (zBottomCut - p.z()) / v.z();
480 if (distZ < 0.0)
481 {
482 distZ= 0.0;
483 if (!calcNorm) {return 0.0;}
484 }
485 distMin= distZ;
486 surface= kPlaneSurf;
487 }
488 if (v.z() > 0.0)
489 {
490 G4double distZ = (zTopCut - p.z()) / v.z();
491 if (distZ < 0.0)
492 {
493 distZ= 0.0;
494 if (!calcNorm) {return 0.0;}
495 }
496 distMin= distZ;
497 surface= kPlaneSurf;
498 }
499
500 // normal vector: parallel to normal, magnitude 1/(characteristic radius)
501 //
502 G4ThreeVector nearnorm(p.x()/(xSemiAxis*xSemiAxis),
503 p.y()/(ySemiAxis*ySemiAxis),
504 p.z()/(zSemiAxis*zSemiAxis));
505
506 // now check curved surface intercept
507 //
508 G4double A,B,C;
509
510 A= sqr(v.x()/xSemiAxis) + sqr(v.y()/ySemiAxis) + sqr(v.z()/zSemiAxis);
511 C= (p * nearnorm) - 1.0;
512 B= 2.0 * (v * nearnorm);
513
514 C= B*B - 4.0*A*C;
515 if (C > 0.0)
516 {
517 G4double distR= (-B + std::sqrt(C) ) / (2.0*A);
518 if (distR < 0.0)
519 {
520 distR= 0.0;
521 if (!calcNorm) {return 0.0;}
522 }
523 if (distR < distMin)
524 {
525 distMin= distR;
526 surface= kCurvedSurf;
527 }
528 }
529
530 // set normal if requested
531 //
532 if (calcNorm)
533 {
534 if (surface == kNoSurf)
535 {
536 *validNorm = false;
537 }
538 else
539 {
540 *validNorm = true;
541 switch (surface)
542 {
543 case kPlaneSurf:
544 *n= G4ThreeVector(0.,0.,(v.z() > 0.0 ? 1. : -1.));
545 break;
546 case kCurvedSurf:
547 {
548 G4ThreeVector pexit= p + distMin*v;
549 G4ThreeVector truenorm(pexit.x()/(xSemiAxis*xSemiAxis),
550 pexit.y()/(ySemiAxis*ySemiAxis),
551 pexit.z()/(zSemiAxis*zSemiAxis));
552 truenorm *= 1.0/truenorm.mag();
553 *n= truenorm;
554 } break;
555 default: // Should never reach this case ...
556 DumpInfo();
557 std::ostringstream message;
558 G4int oldprc = message.precision(16);
559 message << "Undefined side for valid surface normal to solid."
560 << G4endl
561 << "Position:" << G4endl
562 << " p.x() = " << p.x()/mm << " mm" << G4endl
563 << " p.y() = " << p.y()/mm << " mm" << G4endl
564 << " p.z() = " << p.z()/mm << " mm" << G4endl
565 << "Direction:" << G4endl << G4endl
566 << " v.x() = " << v.x() << G4endl
567 << " v.y() = " << v.y() << G4endl
568 << " v.z() = " << v.z() << G4endl
569 << "Proposed distance :" << G4endl
570 << " distMin = " << distMin/mm << " mm";
571 message.precision(oldprc);
572 G4Exception("G4Ellipsoid::DistanceToOut(p,v,..)",
573 "GeomSolids1002", JustWarning, message);
574 break;
575 }
576 }
577 }
578
579 return distMin;
580}

582///////////////////////////////////////////////////////////////////////////////
583//
584// Calculate distance (<=actual) to closest surface of shape from inside
585
586G4double G4Ellipsoid::DistanceToOut(const G4ThreeVector& p) const
587{
588 G4double distR, distZ;
589
590#ifdef G4SPECSDEBUG
591 if( Inside(p) == kOutside )
592 {
593 DumpInfo();
594 std::ostringstream message;
595 G4int oldprc = message.precision(16);
596 message << "Point p is outside !?" << G4endl
597 << "Position:" << G4endl
598 << " p.x() = " << p.x()/mm << " mm" << G4endl
599 << " p.y() = " << p.y()/mm << " mm" << G4endl
600 << " p.z() = " << p.z()/mm << " mm";
601 message.precision(oldprc) ;
602 G4Exception("G4Ellipsoid::DistanceToOut(p)", "GeomSolids1002",
603 JustWarning, message);
604 }
605#endif
606
607 // Normal vector: parallel to normal, magnitude 1/(characteristic radius)
608 //
609 G4ThreeVector norm(p.x()/(xSemiAxis*xSemiAxis),
610 p.y()/(ySemiAxis*ySemiAxis),
611 p.z()/(zSemiAxis*zSemiAxis));
612
613 // the following is a safe inlined "radius= min(1.0/norm.mag(),p.mag())
614 //
615 G4double radius= p.mag();
616 G4double tmp= norm.mag();
617 if ( (tmp > 0.0) && (1.0 < radius*tmp) ) {radius = 1.0/tmp;}
618
619 // Approximate distance to curved surface ( <= actual distance )
620 //
621 distR = (1.0 - p*norm) * radius / 2.0;
622
623 // Distance to z-cut plane
624 //
625 distZ = p.z() - zBottomCut;
626 if (distZ < 0.0) {distZ= zTopCut - p.z();}
627
628 // Distance to closest surface from inside
629 //
630 if ( (distZ < 0.0) || (distR < 0.0) )
631 {
632 return 0.0;
633 }
634 else
635 {
636 return (distZ < distR) ? distZ : distR;
637 }
638}

683////////////////////////////////////////////////////////////////////
684//
685// GetPointOnSurface
686
687G4ThreeVector G4Ellipsoid::GetPointOnSurface() const
688{
689 G4double aTop, aBottom, aCurved, chose, xRand, yRand, zRand, phi;
690 G4double cosphi, sinphi, costheta, sintheta, alpha, beta, max1, max2, max3;
691
692 max1 = xSemiAxis > ySemiAxis ? xSemiAxis : ySemiAxis;
693 max1 = max1 > zSemiAxis ? max1 : zSemiAxis;
694 if (max1 == xSemiAxis) { max2 = ySemiAxis; max3 = zSemiAxis; }
695 else if (max1 == ySemiAxis) { max2 = xSemiAxis; max3 = zSemiAxis; }
696 else { max2 = xSemiAxis; max3 = ySemiAxis; }
697
698 phi = G4RandFlat::shoot(0.,twopi);
699
700 cosphi = std::cos(phi); sinphi = std::sin(phi);
701 costheta = G4RandFlat::shoot(zBottomCut,zTopCut)/zSemiAxis;
702 sintheta = std::sqrt(1.-sqr(costheta));
703
704 alpha = 1.-sqr(max2/max1); beta = 1.-sqr(max3/max1);
705
706 aTop = pi*xSemiAxis*ySemiAxis*(1 - sqr(zTopCut/zSemiAxis));
707 aBottom = pi*xSemiAxis*ySemiAxis*(1 - sqr(zBottomCut/zSemiAxis));
708
709 // approximation
710 // from:" http://www.citr.auckland.ac.nz/techreports/2004/CITR-TR-139.pdf"
711 aCurved = 4.*pi*max1*max2*(1.-1./6.*(alpha+beta)-
712 1./120.*(3.*sqr(alpha)+2.*alpha*beta+3.*sqr(beta)));
713
714 aCurved *= 0.5*(1.2*zTopCut/zSemiAxis - 1.2*zBottomCut/zSemiAxis);
715
716 if( ( zTopCut >= zSemiAxis && zBottomCut <= -1.*zSemiAxis )
717 || ( zTopCut == 0 && zBottomCut ==0 ) )
718 {
719 aTop = 0; aBottom = 0;
720 }
721
722 chose = G4RandFlat::shoot(0.,aTop + aBottom + aCurved);
723
724 if(chose < aCurved)
725 {
726 xRand = xSemiAxis*sintheta*cosphi;
727 yRand = ySemiAxis*sintheta*sinphi;
728 zRand = zSemiAxis*costheta;
729 return G4ThreeVector (xRand,yRand,zRand);
730 }
731 else if(chose >= aCurved && chose < aCurved + aTop)
732 {
733 xRand = G4RandFlat::shoot(-1.,1.)*xSemiAxis
734 * std::sqrt(1-sqr(zTopCut/zSemiAxis));
735 yRand = G4RandFlat::shoot(-1.,1.)*ySemiAxis
736 * std::sqrt(1.-sqr(zTopCut/zSemiAxis)-sqr(xRand/xSemiAxis));
737 zRand = zTopCut;
738 return G4ThreeVector (xRand,yRand,zRand);
739 }
740 else
741 {
742 xRand = G4RandFlat::shoot(-1.,1.)*xSemiAxis
743 * std::sqrt(1-sqr(zBottomCut/zSemiAxis));
744 yRand = G4RandFlat::shoot(-1.,1.)*ySemiAxis
745 * std::sqrt(1.-sqr(zBottomCut/zSemiAxis)-sqr(xRand/xSemiAxis));
746 zRand = zBottomCut;
747 return G4ThreeVector (xRand,yRand,zRand);
748 }
749}
