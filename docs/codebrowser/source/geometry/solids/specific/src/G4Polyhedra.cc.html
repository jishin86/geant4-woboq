33// G4Polyhedra.cc
34//
35// Implementation of a CSG polyhedra, as an inherited class of G4VCSGfaceted.
36//
37// To be done:
38// * Cracks: there are probably small cracks in the seams between the
39// phi face (G4PolyPhiFace) and sides (G4PolyhedraSide) that are not
40// entirely leakproof. Also, I am not sure all vertices are leak proof.
41// * Many optimizations are possible, but not implemented.
42// * Visualization needs to be updated outside of this routine.
43//
44// Utility classes:
45// * G4EnclosingCylinder: I decided a quick check of geometry would be a
46// good idea (for CPU speed). If the quick check fails, the regular
47// full-blown G4VCSGfaceted version is invoked.
48// * G4ReduciblePolygon: Really meant as a check of input parameters,
49// this utility class also "converts" the GEANT3-like PGON/PCON
50// arguments into the newer ones.
51// Both these classes are implemented outside this file because they are
52// shared with G4Polycone.
53//

78//
79// Constructor (GEANT3 style parameters)
80//
81// GEANT3 PGON radii are specified in the distance to the norm of each face.
82//
83G4Polyhedra::G4Polyhedra( const G4String& name,
84 G4double phiStart,
85 G4double thePhiTotal,
86 G4int theNumSide,
87 G4int numZPlanes,
88 const G4double zPlane[],
89 const G4double rInner[],
90 const G4double rOuter[] )
91 : G4VCSGfaceted( name ), genericPgon(false)
92{
93 if (theNumSide <= 0)
94 {
95 std::ostringstream message;
96 message << "Solid must have at least one side - " << GetName() << G4endl
97 << " No sides specified !";
98 G4Exception("G4Polyhedra::G4Polyhedra()", "GeomSolids0002",
99 FatalErrorInArgument, message);
100 }
101
102 //
103 // Calculate conversion factor from G3 radius to G4 radius
104 //
105 G4double phiTotal = thePhiTotal;
106 if ( (phiTotal <=0) || (phiTotal >= twopi*(1-DBL_EPSILON)) )
107 { phiTotal = twopi; }
108 G4double convertRad = std::cos(0.5*phiTotal/theNumSide);
109
110 //
111 // Some historical stuff
112 //
113 original_parameters = new G4PolyhedraHistorical;
114
115 original_parameters->numSide = theNumSide;
116 original_parameters->Start_angle = phiStart;
117 original_parameters->Opening_angle = phiTotal;
118 original_parameters->Num_z_planes = numZPlanes;
119 original_parameters->Z_values = new G4double[numZPlanes];
120 original_parameters->Rmin = new G4double[numZPlanes];
121 original_parameters->Rmax = new G4double[numZPlanes];
122
123 G4int i;
124 for (i=0; i<numZPlanes; i++)
125 {
126 if (( i < numZPlanes-1) && ( zPlane[i] == zPlane[i+1] ))
127 {
128 if( (rInner[i] > rOuter[i+1])
129 ||(rInner[i+1] > rOuter[i]) )
130 {
131 DumpInfo();
132 std::ostringstream message;
133 message << "Cannot create a Polyhedra with no contiguous segments."
134 << G4endl
135 << " Segments are not contiguous !" << G4endl
136 << " rMin[" << i << "] = " << rInner[i]
137 << " -- rMax[" << i+1 << "] = " << rOuter[i+1] << G4endl
138 << " rMin[" << i+1 << "] = " << rInner[i+1]
139 << " -- rMax[" << i << "] = " << rOuter[i];
140 G4Exception("G4Polyhedra::G4Polyhedra()", "GeomSolids0002",
141 FatalErrorInArgument, message);
142 }
143 }
144 original_parameters->Z_values[i] = zPlane[i];
145 original_parameters->Rmin[i] = rInner[i]/convertRad;
146 original_parameters->Rmax[i] = rOuter[i]/convertRad;
147 }
148
149
150 //
151 // Build RZ polygon using special PCON/PGON GEANT3 constructor
152 //
153 G4ReduciblePolygon *rz =
154 new G4ReduciblePolygon( rInner, rOuter, zPlane, numZPlanes );
155 rz->ScaleA( 1/convertRad );
156
157 //
158 // Do the real work
159 //
160 Create( phiStart, phiTotal, theNumSide, rz );
161
162 delete rz;
163}

199//
200// Create
201//
202// Generic create routine, called by each constructor
203// after conversion of arguments
204//
205void G4Polyhedra::Create( G4double phiStart,
206 G4double phiTotal,
207 G4int theNumSide,
208 G4ReduciblePolygon *rz )
209{
210 //
211 // Perform checks of rz values
212 //
213 if (rz->Amin() < 0.0)
214 {
215 std::ostringstream message;
216 message << "Illegal input parameters - " << GetName() << G4endl
217 << " All R values must be >= 0 !";
218 G4Exception("G4Polyhedra::Create()", "GeomSolids0002",
219 FatalErrorInArgument, message);
220 }
221
222 G4double rzArea = rz->Area();
223 if (rzArea < -kCarTolerance)
224 {
225 rz->ReverseOrder();
226 }
227 else if (rzArea < kCarTolerance)
228 {
229 std::ostringstream message;
230 message << "Illegal input parameters - " << GetName() << G4endl
231 << " R/Z cross section is zero or near zero: " << rzArea;
232 G4Exception("G4Polyhedra::Create()", "GeomSolids0002",
233 FatalErrorInArgument, message);
234 }
235
236 if ( (!rz->RemoveDuplicateVertices( kCarTolerance ))
237 || (!rz->RemoveRedundantVertices( kCarTolerance )) )
238 {
239 std::ostringstream message;
240 message << "Illegal input parameters - " << GetName() << G4endl
241 << " Too few unique R/Z values !";
242 G4Exception("G4Polyhedra::Create()", "GeomSolids0002",
243 FatalErrorInArgument, message);
244 }
245
246 if (rz->CrossesItself( 1/kInfinity ))
247 {
248 std::ostringstream message;
249 message << "Illegal input parameters - " << GetName() << G4endl
250 << " R/Z segments cross !";
251 G4Exception("G4Polyhedra::Create()", "GeomSolids0002",
252 FatalErrorInArgument, message);
253 }
254
255 numCorner = rz->NumVertices();
256
257
258 startPhi = phiStart;
259 while( startPhi < 0 ) // Loop checking, 13.08.2015, G.Cosmo
260 startPhi += twopi;
261 //
262 // Phi opening? Account for some possible roundoff, and interpret
263 // nonsense value as representing no phi opening
264 //
265 if ( (phiTotal <= 0) || (phiTotal > twopi*(1-DBL_EPSILON)) )
266 {
267 phiIsOpen = false;
268 endPhi = phiStart+twopi;
269 }
270 else
271 {
272 phiIsOpen = true;
273
274 //
275 // Convert phi into our convention
276 //
277 endPhi = phiStart+phiTotal;
278 while( endPhi < startPhi ) // Loop checking, 13.08.2015, G.Cosmo
279 endPhi += twopi;
280 }
281
282 //
283 // Save number sides
284 //
285 numSide = theNumSide;
286
287 //
288 // Allocate corner array.
289 //
290 corners = new G4PolyhedraSideRZ[numCorner];
291
292 //
293 // Copy corners
294 //
295 G4ReduciblePolygonIterator iterRZ(rz);
296
297 G4PolyhedraSideRZ *next = corners;
298 iterRZ.Begin();
299 do // Loop checking, 13.08.2015, G.Cosmo
300 {
301 next->r = iterRZ.GetA();
302 next->z = iterRZ.GetB();
303 } while( ++next, iterRZ.Next() );
304
305 //
306 // Allocate face pointer array
307 //
308 numFace = phiIsOpen ? numCorner+2 : numCorner;
309 faces = new G4VCSGface*[numFace];
310
311 //
312 // Construct side faces
313 //
314 // To do so properly, we need to keep track of four successive RZ
315 // corners.
316 //
317 // But! Don't construct a face if both points are at zero radius!
318 //
319 G4PolyhedraSideRZ *corner = corners,
320 *prev = corners + numCorner-1,
321 *nextNext;
322 G4VCSGface **face = faces;
323 do // Loop checking, 13.08.2015, G.Cosmo
324 {
325 next = corner+1;
326 if (next >= corners+numCorner) next = corners;
327 nextNext = next+1;
328 if (nextNext >= corners+numCorner) nextNext = corners;
329
330 if (corner->r < 1/kInfinity && next->r < 1/kInfinity) continue;
331/*
332 // We must decide here if we can dare declare one of our faces
333 // as having a "valid" normal (i.e. allBehind = true). This
334 // is never possible if the face faces "inward" in r *unless*
335 // we have only one side
336 //
337 G4bool allBehind;
338 if ((corner->z > next->z) && (numSide > 1))
339 {
340 allBehind = false;
341 }
342 else
343 {
344 //
345 // Otherwise, it is only true if the line passing
346 // through the two points of the segment do not
347 // split the r/z cross section
348 //
349 allBehind = !rz->BisectedBy( corner->r, corner->z,
350 next->r, next->z, kCarTolerance );
351 }
352*/
353 *face++ = new G4PolyhedraSide( prev, corner, next, nextNext,
354 numSide, startPhi, endPhi-startPhi, phiIsOpen );
355 } while( prev=corner, corner=next, corner > corners );
356
357 if (phiIsOpen)
358 {
359 //
360 // Construct phi open edges
361 //
362 *face++ = new G4PolyPhiFace( rz, startPhi, phiTotal/numSide, endPhi );
363 *face++ = new G4PolyPhiFace( rz, endPhi, phiTotal/numSide, startPhi );
364 }
365
366 //
367 // We might have dropped a face or two: recalculate numFace
368 //
369 numFace = face-faces;
370
371 //
372 // Make enclosingCylinder
373 //
374 enclosingCylinder =
375 new G4EnclosingCylinder( rz, phiIsOpen, phiStart, phiTotal );
376}

433//
434// CopyStuff
435//
436void G4Polyhedra::CopyStuff( const G4Polyhedra &source )
437{
438 //
439 // Simple stuff
440 //
441 numSide = source.numSide;
442 startPhi = source.startPhi;
443 endPhi = source.endPhi;
444 phiIsOpen = source.phiIsOpen;
445 numCorner = source.numCorner;
446 genericPgon= source.genericPgon;
447
448 //
449 // The corner array
450 //
451 corners = new G4PolyhedraSideRZ[numCorner];
452
453 G4PolyhedraSideRZ *corn = corners,
454 *sourceCorn = source.corners;
455 do // Loop checking, 13.08.2015, G.Cosmo
456 {
457 *corn = *sourceCorn;
458 } while( ++sourceCorn, ++corn < corners+numCorner );
459
460 //
461 // Original parameters
462 //
463 if (source.original_parameters)
464 {
465 original_parameters =
466 new G4PolyhedraHistorical( *source.original_parameters );
467 }
468
469 //
470 // Enclosing cylinder
471 //
472 enclosingCylinder = new G4EnclosingCylinder( *source.enclosingCylinder );
473
474 fRebuildPolyhedron = false;
475 fpPolyhedron = 0;
476}

479//
480// Reset
481//
482// Recalculates and reshapes the solid, given pre-assigned scaled
483// original_parameters.
484//
485G4bool G4Polyhedra::Reset()
486{
487 if (genericPgon)
488 {
489 std::ostringstream message;
490 message << "Solid " << GetName() << " built using generic construct."
491 << G4endl << "Not applicable to the generic construct !";
492 G4Exception("G4Polyhedra::Reset()", "GeomSolids1001",
493 JustWarning, message, "Parameters NOT resetted.");
494 return 1;
495 }
496
497 //
498 // Clear old setup
499 //
500 G4VCSGfaceted::DeleteStuff();
501 delete [] corners;
502 delete enclosingCylinder;
503
504 //
505 // Rebuild polyhedra
506 //
507 G4ReduciblePolygon *rz =
508 new G4ReduciblePolygon( original_parameters->Rmin,
509 original_parameters->Rmax,
510 original_parameters->Z_values,
511 original_parameters->Num_z_planes );
512 Create( original_parameters->Start_angle,
513 original_parameters->Opening_angle,
514 original_parameters->numSide, rz );
515 delete rz;
516
517 return 0;
518}

571//////////////////////////////////////////////////////////////////////////
572//
573// Get bounding box
574
575void G4Polyhedra::BoundingLimits(G4ThreeVector& pMin,
576 G4ThreeVector& pMax) const
577{
578 G4double rmin = kInfinity, rmax = -kInfinity;
579 G4double zmin = kInfinity, zmax = -kInfinity;
580 for (G4int i=0; i<GetNumRZCorner(); ++i)
581 {
582 G4PolyhedraSideRZ corner = GetCorner(i);
583 if (corner.r < rmin) rmin = corner.r;
584 if (corner.r > rmax) rmax = corner.r;
585 if (corner.z < zmin) zmin = corner.z;
586 if (corner.z > zmax) zmax = corner.z;
587 }
588
589 G4double sphi = GetStartPhi();
590 G4double ephi = GetEndPhi();
591 G4double dphi = IsOpen() ? ephi-sphi : twopi;
592 G4int ksteps = GetNumSide();
593 G4double astep = dphi/ksteps;
594 G4double sinStep = std::sin(astep);
595 G4double cosStep = std::cos(astep);
596
597 G4double sinCur = GetSinStartPhi();
598 G4double cosCur = GetCosStartPhi();
599 if (!IsOpen()) rmin = 0;
600 G4double xmin = rmin*cosCur, xmax = xmin;
601 G4double ymin = rmin*sinCur, ymax = ymin;
602 for (G4int k=0; k<ksteps+1; ++k)
603 {
604 G4double x = rmax*cosCur;
605 if (x < xmin) xmin = x;
606 if (x > xmax) xmax = x;
607 G4double y = rmax*sinCur;
608 if (y < ymin) ymin = y;
609 if (y > ymax) ymax = y;
610 if (rmin > 0)
611 {
612 G4double xx = rmin*cosCur;
613 if (xx < xmin) xmin = xx;
614 if (xx > xmax) xmax = xx;
615 G4double yy = rmin*sinCur;
616 if (yy < ymin) ymin = yy;
617 if (yy > ymax) ymax = yy;
618 }
619 G4double sinTmp = sinCur;
620 sinCur = sinCur*cosStep + cosCur*sinStep;
621 cosCur = cosCur*cosStep - sinTmp*sinStep;
622 }
623 pMin.set(xmin,ymin,zmin);
624 pMax.set(xmax,ymax,zmax);
625
626 // Check correctness of the bounding box
627 //
628 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
629 {
630 std::ostringstream message;
631 message << "Bad bounding box (min >= max) for solid: "
632 << GetName() << " !"
633 << "\npMin = " << pMin
634 << "\npMax = " << pMax;
635 G4Exception("G4Polyhedra::BoundingLimits()", "GeomMgt0001",
636 JustWarning, message);
637 DumpInfo();
638 }
639}

641//////////////////////////////////////////////////////////////////////////
642//
643// Calculate extent under transform and specified limit
644
645G4bool G4Polyhedra::CalculateExtent(const EAxis pAxis,
646 const G4VoxelLimits& pVoxelLimit,
647 const G4AffineTransform& pTransform,
648 G4double& pMin, G4double& pMax) const
649{
650 G4ThreeVector bmin, bmax;
651 G4bool exist;
652
653 // Check bounding box (bbox)
654 //
655 BoundingLimits(bmin,bmax);
656 G4BoundingEnvelope bbox(bmin,bmax);
657#ifdef G4BBOX_EXTENT
658 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
659#endif
660 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
661 {
662 return exist = (pMin < pMax) ? true : false;
663 }
664
665 // To find the extent, RZ contour of the polycone is subdivided
666 // in triangles. The extent is calculated as cumulative extent of
667 // all sub-polycones formed by rotation of triangles around Z
668 //
669 G4TwoVectorList contourRZ;
670 G4TwoVectorList triangles;
671 std::vector<G4int> iout;
672 G4double eminlim = pVoxelLimit.GetMinExtent(pAxis);
673 G4double emaxlim = pVoxelLimit.GetMaxExtent(pAxis);
674
675 // get RZ contour, ensure anticlockwise order of corners
676 for (G4int i=0; i<GetNumRZCorner(); ++i)
677 {
678 G4PolyhedraSideRZ corner = GetCorner(i);
679 contourRZ.push_back(G4TwoVector(corner.r,corner.z));
680 }
681 G4GeomTools::RemoveRedundantVertices(contourRZ,iout,2*kCarTolerance);
682 G4double area = G4GeomTools::PolygonArea(contourRZ);
683 if (area < 0.) std::reverse(contourRZ.begin(),contourRZ.end());
684
685 // triangulate RZ countour
686 if (!G4GeomTools::TriangulatePolygon(contourRZ,triangles))
687 {
688 std::ostringstream message;
689 message << "Triangulation of RZ contour has failed for solid: "
690 << GetName() << " !"
691 << "\nExtent has been calculated using boundary box";
692 G4Exception("G4Polyhedra::CalculateExtent()",
693 "GeomMgt1002",JustWarning,message);
694 return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
695 }
696
697 // set trigonometric values
698 G4double sphi = GetStartPhi();
699 G4double ephi = GetEndPhi();
700 G4double dphi = IsOpen() ? ephi-sphi : twopi;
701 G4int ksteps = GetNumSide();
702 G4double astep = dphi/ksteps;
703 G4double sinStep = std::sin(astep);
704 G4double cosStep = std::cos(astep);
705 G4double sinStart = GetSinStartPhi();
706 G4double cosStart = GetCosStartPhi();
707
708 // allocate vector lists
709 std::vector<const G4ThreeVectorList *> polygons;
710 polygons.resize(ksteps+1);
711 for (G4int k=0; k<ksteps+1; ++k) {
712 polygons[k] = new G4ThreeVectorList(3);
713 }
714
715 // main loop along triangles
716 pMin = kInfinity;
717 pMax = -kInfinity;
718 G4int ntria = triangles.size()/3;
719 for (G4int i=0; i<ntria; ++i)
720 {
721 G4double sinCur = sinStart;
722 G4double cosCur = cosStart;
723 G4int i3 = i*3;
724 for (G4int k=0; k<ksteps+1; ++k) // rotate triangle
725 {
726 G4ThreeVectorList* ptr = const_cast<G4ThreeVectorList*>(polygons[k]);
727 G4ThreeVectorList::iterator iter = ptr->begin();
728 iter->set(triangles[i3+0].x()*cosCur,
729 triangles[i3+0].x()*sinCur,
730 triangles[i3+0].y());
731 iter++;
732 iter->set(triangles[i3+1].x()*cosCur,
733 triangles[i3+1].x()*sinCur,
734 triangles[i3+1].y());
735 iter++;
736 iter->set(triangles[i3+2].x()*cosCur,
737 triangles[i3+2].x()*sinCur,
738 triangles[i3+2].y());
739
740 G4double sinTmp = sinCur;
741 sinCur = sinCur*cosStep + cosCur*sinStep;
742 cosCur = cosCur*cosStep - sinTmp*sinStep;
743 }
744
745 // set sub-envelope and adjust extent
746 G4double emin,emax;
747 G4BoundingEnvelope benv(polygons);
748 if (!benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,emin,emax)) continue;
749 if (emin < pMin) pMin = emin;
750 if (emax > pMax) pMax = emax;
751 if (eminlim > pMin && emaxlim < pMax) break; // max possible extent
752 }
753 // free memory
754 for (G4int k=0; k<ksteps+1; ++k) { delete polygons[k]; polygons[k]=0;}
755 return (pMin < pMax);
756}

839//
840// GetPointOnPlane
841//
842// Auxiliary method for get point on surface
843//
844G4ThreeVector
845G4Polyhedra::GetPointOnPlane(G4ThreeVector p0, G4ThreeVector p1,
846 G4ThreeVector p2, G4ThreeVector p3) const
847{
848 G4double lambda1, lambda2, chose,aOne,aTwo;
849 G4ThreeVector t, u, v, w, Area, normal;
850 aOne = 1.;
851 aTwo = 1.;
852
853 t = p1 - p0;
854 u = p2 - p1;
855 v = p3 - p2;
856 w = p0 - p3;
857
858 chose = G4RandFlat::shoot(0.,aOne+aTwo);
859 if( (chose>=0.) && (chose < aOne) )
860 {
861 lambda1 = G4RandFlat::shoot(0.,1.);
862 lambda2 = G4RandFlat::shoot(0.,lambda1);
863 return (p2+lambda1*v+lambda2*w);
864 }
865
866 lambda1 = G4RandFlat::shoot(0.,1.);
867 lambda2 = G4RandFlat::shoot(0.,lambda1);
868 return (p0+lambda1*t+lambda2*u);
869}

891//
892// GetPointOnSurface
893//
894G4ThreeVector G4Polyhedra::GetPointOnSurface() const
895{
896 if( !genericPgon ) // Polyhedra by faces
897 {
898 G4int j, numPlanes = original_parameters->Num_z_planes, Flag=0;
899 G4double chose, totArea=0., Achose1, Achose2,
900 rad1, rad2, sinphi1, sinphi2, cosphi1, cosphi2;
901 G4double a, b, l2, rang, totalPhi, ksi,
902 area, aTop=0., aBottom=0., zVal=0.;
903
904 G4ThreeVector p0, p1, p2, p3;
905 std::vector<G4double> aVector1;
906 std::vector<G4double> aVector2;
907 std::vector<G4double> aVector3;
908
909 totalPhi= (phiIsOpen) ? (endPhi-startPhi) : twopi;
910 ksi = totalPhi/numSide;
911 G4double cosksi = std::cos(ksi/2.);
912
913 // Below we generate the areas relevant to our solid
914 //
915 for(j=0; j<numPlanes-1; j++)
916 {
917 a = original_parameters->Rmax[j+1];
918 b = original_parameters->Rmax[j];
919 l2 = sqr(original_parameters->Z_values[j]
920 -original_parameters->Z_values[j+1]) + sqr(b-a);
921 area = std::sqrt(l2-sqr((a-b)*cosksi))*(a+b)*cosksi;
922 aVector1.push_back(area);
923 }
924
925 for(j=0; j<numPlanes-1; j++)
926 {
927 a = original_parameters->Rmin[j+1];//*cosksi;
928 b = original_parameters->Rmin[j];//*cosksi;
929 l2 = sqr(original_parameters->Z_values[j]
930 -original_parameters->Z_values[j+1]) + sqr(b-a);
931 area = std::sqrt(l2-sqr((a-b)*cosksi))*(a+b)*cosksi;
932 aVector2.push_back(area);
933 }
934
935 for(j=0; j<numPlanes-1; j++)
936 {
937 if(phiIsOpen == true)
938 {
939 aVector3.push_back(0.5*(original_parameters->Rmax[j]
940 -original_parameters->Rmin[j]
941 +original_parameters->Rmax[j+1]
942 -original_parameters->Rmin[j+1])
943 *std::fabs(original_parameters->Z_values[j+1]
944 -original_parameters->Z_values[j]));
945 }
946 else { aVector3.push_back(0.); }
947 }
948
949 for(j=0; j<numPlanes-1; j++)
950 {
951 totArea += numSide*(aVector1[j]+aVector2[j])+2.*aVector3[j];
952 }
953
954 // Must include top and bottom areas
955 //
956 if(original_parameters->Rmax[numPlanes-1] != 0.)
957 {
958 a = original_parameters->Rmax[numPlanes-1];
959 b = original_parameters->Rmin[numPlanes-1];
960 l2 = sqr(a-b);
961 aTop = std::sqrt(l2-sqr((a-b)*cosksi))*(a+b)*cosksi;
962 }
963
964 if(original_parameters->Rmax[0] != 0.)
965 {
966 a = original_parameters->Rmax[0];
967 b = original_parameters->Rmin[0];
968 l2 = sqr(a-b);
969 aBottom = std::sqrt(l2-sqr((a-b)*cosksi))*(a+b)*cosksi;
970 }
971
972 Achose1 = 0.;
973 Achose2 = numSide*(aVector1[0]+aVector2[0])+2.*aVector3[0];
974
975 chose = G4RandFlat::shoot(0.,totArea+aTop+aBottom);
976 if( (chose >= 0.) && (chose < aTop + aBottom) )
977 {
978 chose = G4RandFlat::shoot(startPhi,startPhi+totalPhi);
979 rang = std::floor((chose-startPhi)/ksi-0.01);
980 if(rang<0) { rang=0; }
981 rang = std::fabs(rang);
982 sinphi1 = std::sin(startPhi+rang*ksi);
983 sinphi2 = std::sin(startPhi+(rang+1)*ksi);
984 cosphi1 = std::cos(startPhi+rang*ksi);
985 cosphi2 = std::cos(startPhi+(rang+1)*ksi);
986 chose = G4RandFlat::shoot(0., aTop + aBottom);
987 if(chose>=0. && chose<aTop)
988 {
989 rad1 = original_parameters->Rmin[numPlanes-1];
990 rad2 = original_parameters->Rmax[numPlanes-1];
991 zVal = original_parameters->Z_values[numPlanes-1];
992 }
993 else
994 {
995 rad1 = original_parameters->Rmin[0];
996 rad2 = original_parameters->Rmax[0];
997 zVal = original_parameters->Z_values[0];
998 }
999 p0 = G4ThreeVector(rad1*cosphi1,rad1*sinphi1,zVal);
1000 p1 = G4ThreeVector(rad2*cosphi1,rad2*sinphi1,zVal);
1001 p2 = G4ThreeVector(rad2*cosphi2,rad2*sinphi2,zVal);
1002 p3 = G4ThreeVector(rad1*cosphi2,rad1*sinphi2,zVal);
1003 return GetPointOnPlane(p0,p1,p2,p3);
1004 }
1005 else
1006 {
1007 for (j=0; j<numPlanes-1; j++)
1008 {
1009 if( ((chose >= Achose1) && (chose < Achose2)) || (j == numPlanes-2) )
1010 {
1011 Flag = j; break;
1012 }
1013 Achose1 += numSide*(aVector1[j]+aVector2[j])+2.*aVector3[j];
1014 Achose2 = Achose1 + numSide*(aVector1[j+1]+aVector2[j+1])
1015 + 2.*aVector3[j+1];
1016 }
1017 }
1018
1019 // At this point we have chosen a subsection
1020 // between to adjacent plane cuts...
1021
1022 j = Flag;
1023
1024 totArea = numSide*(aVector1[j]+aVector2[j])+2.*aVector3[j];
1025 chose = G4RandFlat::shoot(0.,totArea);
1026
1027 if( (chose>=0.) && (chose<numSide*aVector1[j]) )
1028 {
1029 chose = G4RandFlat::shoot(startPhi,startPhi+totalPhi);
1030 rang = std::floor((chose-startPhi)/ksi-0.01);
1031 if(rang<0) { rang=0; }
1032 rang = std::fabs(rang);
1033 rad1 = original_parameters->Rmax[j];
1034 rad2 = original_parameters->Rmax[j+1];
1035 sinphi1 = std::sin(startPhi+rang*ksi);
1036 sinphi2 = std::sin(startPhi+(rang+1)*ksi);
1037 cosphi1 = std::cos(startPhi+rang*ksi);
1038 cosphi2 = std::cos(startPhi+(rang+1)*ksi);
1039 zVal = original_parameters->Z_values[j];
1040
1041 p0 = G4ThreeVector(rad1*cosphi1,rad1*sinphi1,zVal);
1042 p1 = G4ThreeVector(rad1*cosphi2,rad1*sinphi2,zVal);
1043
1044 zVal = original_parameters->Z_values[j+1];
1045
1046 p2 = G4ThreeVector(rad2*cosphi2,rad2*sinphi2,zVal);
1047 p3 = G4ThreeVector(rad2*cosphi1,rad2*sinphi1,zVal);
1048 return GetPointOnPlane(p0,p1,p2,p3);
1049 }
1050 else if ( (chose >= numSide*aVector1[j])
1051 && (chose <= numSide*(aVector1[j]+aVector2[j])) )
1052 {
1053 chose = G4RandFlat::shoot(startPhi,startPhi+totalPhi);
1054 rang = std::floor((chose-startPhi)/ksi-0.01);
1055 if(rang<0) { rang=0; }
1056 rang = std::fabs(rang);
1057 rad1 = original_parameters->Rmin[j];
1058 rad2 = original_parameters->Rmin[j+1];
1059 sinphi1 = std::sin(startPhi+rang*ksi);
1060 sinphi2 = std::sin(startPhi+(rang+1)*ksi);
1061 cosphi1 = std::cos(startPhi+rang*ksi);
1062 cosphi2 = std::cos(startPhi+(rang+1)*ksi);
1063 zVal = original_parameters->Z_values[j];
1064
1065 p0 = G4ThreeVector(rad1*cosphi1,rad1*sinphi1,zVal);
1066 p1 = G4ThreeVector(rad1*cosphi2,rad1*sinphi2,zVal);
1067
1068 zVal = original_parameters->Z_values[j+1];
1069
1070 p2 = G4ThreeVector(rad2*cosphi2,rad2*sinphi2,zVal);
1071 p3 = G4ThreeVector(rad2*cosphi1,rad2*sinphi1,zVal);
1072 return GetPointOnPlane(p0,p1,p2,p3);
1073 }
1074
1075 chose = G4RandFlat::shoot(0.,2.2);
1076 if( (chose>=0.) && (chose < 1.) )
1077 {
1078 rang = startPhi;
1079 }
1080 else
1081 {
1082 rang = endPhi;
1083 }
1084
1085 cosphi1 = std::cos(rang); rad1 = original_parameters->Rmin[j];
1086 sinphi1 = std::sin(rang); rad2 = original_parameters->Rmax[j];
1087
1088 p0 = G4ThreeVector(rad1*cosphi1,rad1*sinphi1,
1089 original_parameters->Z_values[j]);
1090 p1 = G4ThreeVector(rad2*cosphi1,rad2*sinphi1,
1091 original_parameters->Z_values[j]);
1092
1093 rad1 = original_parameters->Rmax[j+1];
1094 rad2 = original_parameters->Rmin[j+1];
1095
1096 p2 = G4ThreeVector(rad1*cosphi1,rad1*sinphi1,
1097 original_parameters->Z_values[j+1]);
1098 p3 = G4ThreeVector(rad2*cosphi1,rad2*sinphi1,
1099 original_parameters->Z_values[j+1]);
1100 return GetPointOnPlane(p0,p1,p2,p3);
1101 }
1102 else // Generic polyhedra
1103 {
1104 return GetPointOnSurfaceGeneric();
1105 }
1106}

1108//
1109// CreatePolyhedron
1110//
1111G4Polyhedron* G4Polyhedra::CreatePolyhedron() const
1112{
1113 if (!genericPgon)
1114 {
1115 return new G4PolyhedronPgon( original_parameters->Start_angle,
1116 original_parameters->Opening_angle,
1117 original_parameters->numSide,
1118 original_parameters->Num_z_planes,
1119 original_parameters->Z_values,
1120 original_parameters->Rmin,
1121 original_parameters->Rmax);
1122 }
1123 else
1124 {
1125 // The following code prepares for:
1126 // HepPolyhedron::createPolyhedron(int Nnodes, int Nfaces,
1127 // const double xyz[][3],
1128 // const int faces_vec[][4])
1129 // Here is an extract from the header file HepPolyhedron.h:
1130 /**
1131 * Creates user defined polyhedron.
1132 * This function allows to the user to define arbitrary polyhedron.
1133 * The faces of the polyhedron should be either triangles or planar
1134 * quadrilateral. Nodes of a face are defined by indexes pointing to
1135 * the elements in the xyz array. Numeration of the elements in the
1136 * array starts from 1 (like in fortran). The indexes can be positive
1137 * or negative. Negative sign means that the corresponding edge is
1138 * invisible. The normal of the face should be directed to exterior
1139 * of the polyhedron.
1140 *
1141 * @param Nnodes number of nodes
1142 * @param Nfaces number of faces
1143 * @param xyz nodes
1144 * @param faces_vec faces (quadrilaterals or triangles)
1145 * @return status of the operation - is non-zero in case of problem
1146 */
1147 G4int nNodes;
1148 G4int nFaces;
1149 typedef G4double double3[3];
1150 double3* xyz;
1151 typedef G4int int4[4];
1152 int4* faces_vec;
1153 if (phiIsOpen)
1154 {
1155 // Triangulate open ends. Simple ear-chopping algorithm...
1156 // I'm not sure how robust this algorithm is (J.Allison).
1157 //
1158 std::vector<G4bool> chopped(numCorner, false);
1159 std::vector<G4int*> triQuads;
1160 G4int remaining = numCorner;
1161 G4int iStarter = 0;
1162 while (remaining >= 3) // Loop checking, 13.08.2015, G.Cosmo
1163 {
1164 // Find unchopped corners...
1165 //
1166 G4int A = -1, B = -1, C = -1;
1167 G4int iStepper = iStarter;
1168 do // Loop checking, 13.08.2015, G.Cosmo
1169 {
1170 if (A < 0) { A = iStepper; }
1171 else if (B < 0) { B = iStepper; }
1172 else if (C < 0) { C = iStepper; }
1173 do // Loop checking, 13.08.2015, G.Cosmo
1174 {
1175 if (++iStepper >= numCorner) iStepper = 0;
1176 }
1177 while (chopped[iStepper]);
1178 }
1179 while (C < 0 && iStepper != iStarter);
1180
1181 // Check triangle at B is pointing outward (an "ear").
1182 // Sign of z cross product determines...
1183
1184 G4double BAr = corners[A].r - corners[B].r;
1185 G4double BAz = corners[A].z - corners[B].z;
1186 G4double BCr = corners[C].r - corners[B].r;
1187 G4double BCz = corners[C].z - corners[B].z;
1188 if (BAr * BCz - BAz * BCr < kCarTolerance)
1189 {
1190 G4int* tq = new G4int[3];
1191 tq[0] = A + 1;
1192 tq[1] = B + 1;
1193 tq[2] = C + 1;
1194 triQuads.push_back(tq);
1195 chopped[B] = true;
1196 --remaining;
1197 }
1198 else
1199 {
1200 do // Loop checking, 13.08.2015, G.Cosmo
1201 {
1202 if (++iStarter >= numCorner) { iStarter = 0; }
1203 }
1204 while (chopped[iStarter]);
1205 }
1206 }
1207
1208 // Transfer to faces...
1209
1210 nNodes = (numSide + 1) * numCorner;
1211 nFaces = numSide * numCorner + 2 * triQuads.size();
1212 faces_vec = new int4[nFaces];
1213 G4int iface = 0;
1214 G4int addition = numCorner * numSide;
1215 G4int d = numCorner - 1;
1216 for (G4int iEnd = 0; iEnd < 2; ++iEnd)
1217 {
1218 for (size_t i = 0; i < triQuads.size(); ++i)
1219 {
1220 // Negative for soft/auxiliary/normally invisible edges...
1221 //
1222 G4int a, b, c;
1223 if (iEnd == 0)
1224 {
1225 a = triQuads[i][0];
1226 b = triQuads[i][1];
1227 c = triQuads[i][2];
1228 }
1229 else
1230 {
1231 a = triQuads[i][0] + addition;
1232 b = triQuads[i][2] + addition;
1233 c = triQuads[i][1] + addition;
1234 }
1235 G4int ab = std::abs(b - a);
1236 G4int bc = std::abs(c - b);
1237 G4int ca = std::abs(a - c);
1238 faces_vec[iface][0] = (ab == 1 || ab == d)? a: -a;
1239 faces_vec[iface][1] = (bc == 1 || bc == d)? b: -b;
1240 faces_vec[iface][2] = (ca == 1 || ca == d)? c: -c;
1241 faces_vec[iface][3] = 0;
1242 ++iface;
1243 }
1244 }
1245
1246 // Continue with sides...
1247
1248 xyz = new double3[nNodes];
1249 const G4double dPhi = (endPhi - startPhi) / numSide;
1250 G4double phi = startPhi;
1251 G4int ixyz = 0;
1252 for (G4int iSide = 0; iSide < numSide; ++iSide)
1253 {
1254 for (G4int iCorner = 0; iCorner < numCorner; ++iCorner)
1255 {
1256 xyz[ixyz][0] = corners[iCorner].r * std::cos(phi);
1257 xyz[ixyz][1] = corners[iCorner].r * std::sin(phi);
1258 xyz[ixyz][2] = corners[iCorner].z;
1259 if (iCorner < numCorner - 1)
1260 {
1261 faces_vec[iface][0] = ixyz + 1;
1262 faces_vec[iface][1] = ixyz + numCorner + 1;
1263 faces_vec[iface][2] = ixyz + numCorner + 2;
1264 faces_vec[iface][3] = ixyz + 2;
1265 }
1266 else
1267 {
1268 faces_vec[iface][0] = ixyz + 1;
1269 faces_vec[iface][1] = ixyz + numCorner + 1;
1270 faces_vec[iface][2] = ixyz + 2;
1271 faces_vec[iface][3] = ixyz - numCorner + 2;
1272 }
1273 ++iface;
1274 ++ixyz;
1275 }
1276 phi += dPhi;
1277 }
1278
1279 // Last corners...
1280
1281 for (G4int iCorner = 0; iCorner < numCorner; ++iCorner)
1282 {
1283 xyz[ixyz][0] = corners[iCorner].r * std::cos(phi);
1284 xyz[ixyz][1] = corners[iCorner].r * std::sin(phi);
1285 xyz[ixyz][2] = corners[iCorner].z;
1286 ++ixyz;
1287 }
1288 }
1289 else // !phiIsOpen - i.e., a complete 360 degrees.
1290 {
1291 nNodes = numSide * numCorner;
1292 nFaces = numSide * numCorner;;
1293 xyz = new double3[nNodes];
1294 faces_vec = new int4[nFaces];
1295 // const G4double dPhi = (endPhi - startPhi) / numSide;
1296 const G4double dPhi = twopi / numSide;
1297 // !phiIsOpen endPhi-startPhi = 360 degrees.
1298 G4double phi = startPhi;
1299 G4int ixyz = 0, iface = 0;
1300 for (G4int iSide = 0; iSide < numSide; ++iSide)
1301 {
1302 for (G4int iCorner = 0; iCorner < numCorner; ++iCorner)
1303 {
1304 xyz[ixyz][0] = corners[iCorner].r * std::cos(phi);
1305 xyz[ixyz][1] = corners[iCorner].r * std::sin(phi);
1306 xyz[ixyz][2] = corners[iCorner].z;
1307 if (iSide < numSide - 1)
1308 {
1309 if (iCorner < numCorner - 1)
1310 {
1311 faces_vec[iface][0] = ixyz + 1;
1312 faces_vec[iface][1] = ixyz + numCorner + 1;
1313 faces_vec[iface][2] = ixyz + numCorner + 2;
1314 faces_vec[iface][3] = ixyz + 2;
1315 }
1316 else
1317 {
1318 faces_vec[iface][0] = ixyz + 1;
1319 faces_vec[iface][1] = ixyz + numCorner + 1;
1320 faces_vec[iface][2] = ixyz + 2;
1321 faces_vec[iface][3] = ixyz - numCorner + 2;
1322 }
1323 }
1324 else // Last side joins ends...
1325 {
1326 if (iCorner < numCorner - 1)
1327 {
1328 faces_vec[iface][0] = ixyz + 1;
1329 faces_vec[iface][1] = ixyz + numCorner - nFaces + 1;
1330 faces_vec[iface][2] = ixyz + numCorner - nFaces + 2;
1331 faces_vec[iface][3] = ixyz + 2;
1332 }
1333 else
1334 {
1335 faces_vec[iface][0] = ixyz + 1;
1336 faces_vec[iface][1] = ixyz - nFaces + numCorner + 1;
1337 faces_vec[iface][2] = ixyz - nFaces + 2;
1338 faces_vec[iface][3] = ixyz - numCorner + 2;
1339 }
1340 }
1341 ++ixyz;
1342 ++iface;
1343 }
1344 phi += dPhi;
1345 }
1346 }
1347 G4Polyhedron* polyhedron = new G4Polyhedron;
1348 G4int problem = polyhedron->createPolyhedron(nNodes,nFaces,xyz,faces_vec);
1349 delete [] faces_vec;
1350 delete [] xyz;
1351 if (problem)
1352 {
1353 std::ostringstream message;
1354 message << "Problem creating G4Polyhedron for: " << GetName();
1355 G4Exception("G4Polyhedra::CreatePolyhedron()", "GeomSolids1002",
1356 JustWarning, message);
1357 delete polyhedron;
1358 return 0;
1359 }
1360 else
1361 {
1362 return polyhedron;
1363 }
1364 }
