49
50G4TwistTrapAlphaSide::
51G4TwistTrapAlphaSide(const G4String &name,
52 G4double PhiTwist, // twist angle
53 G4double pDz, // half z lenght
54 G4double pTheta, // direction between end planes
55 G4double pPhi, // by polar and azimutal angles
56 G4double pDy1, // half y length at -pDz
57 G4double pDx1, // half x length at -pDz,-pDy
58 G4double pDx2, // half x length at -pDz,+pDy
59 G4double pDy2, // half y length at +pDz
60 G4double pDx3, // half x length at +pDz,-pDy
61 G4double pDx4, // half x length at +pDz,+pDy
62 G4double pAlph, // tilt angle at +pDz
63 G4double AngleSide // parity
64 )
65 : G4VTwistSurface(name)
66{
67 fAxis[0] = kYAxis; // in local coordinate system
68 fAxis[1] = kZAxis;
69 fAxisMin[0] = -kInfinity ; // Y Axis boundary
70 fAxisMax[0] = kInfinity ; // depends on z !!
71 fAxisMin[1] = -pDz ; // Z Axis boundary
72 fAxisMax[1] = pDz ;
73
74 fDx1 = pDx1 ;
75 fDx2 = pDx2 ;
76 fDx3 = pDx3 ;
77 fDx4 = pDx4 ;
78
79 fDy1 = pDy1 ;
80 fDy2 = pDy2 ;
81
82 fDz = pDz ;
83
84 fAlph = pAlph ;
85 fTAlph = std::tan(fAlph) ;
86
87 fTheta = pTheta ;
88 fPhi = pPhi ;
89
90 // precalculate frequently used parameters
91 fDx4plus2 = fDx4 + fDx2 ;
92 fDx4minus2 = fDx4 - fDx2 ;
93 fDx3plus1 = fDx3 + fDx1 ;
94 fDx3minus1 = fDx3 - fDx1 ;
95 fDy2plus1 = fDy2 + fDy1 ;
96 fDy2minus1 = fDy2 - fDy1 ;
97
98 fa1md1 = 2*fDx2 - 2*fDx1 ;
99 fa2md2 = 2*fDx4 - 2*fDx3 ;
100
101 fPhiTwist = PhiTwist ; // dphi
102 fAngleSide = AngleSide ; // 0,90,180,270 deg
103
104 fdeltaX = 2 * fDz * std::tan(fTheta) * std::cos(fPhi);
105 // dx in surface equation
106 fdeltaY = 2 * fDz * std::tan(fTheta) * std::sin(fPhi);
107 // dy in surface equation
108
109 fRot.rotateZ( AngleSide ) ;
110
111 fTrans.set(0, 0, 0); // No Translation
112 fIsValidNorm = false;
113
114 SetCorners() ;
115 SetBoundaries() ;
116
117}

144G4ThreeVector
145G4TwistTrapAlphaSide::GetNormal(const G4ThreeVector &tmpxx,
146 G4bool isGlobal)
147{
148 // GetNormal returns a normal vector at a surface (or very close
149 // to surface) point at tmpxx.
150 // If isGlobal=true, it returns the normal in global coordinate.
151 //
152
153 G4ThreeVector xx;
154 if (isGlobal)
155 {
156 xx = ComputeLocalPoint(tmpxx);
157 if ((xx - fCurrentNormal.p).mag() < 0.5 * kCarTolerance)
158 {
159 return ComputeGlobalDirection(fCurrentNormal.normal);
160 }
161 }
162 else
163 {
164 xx = tmpxx;
165 if (xx == fCurrentNormal.p)
166 {
167 return fCurrentNormal.normal;
168 }
169 }
170
171 G4double phi ;
172 G4double u ;
173
174 GetPhiUAtX(xx,phi,u) ; // phi,u for point xx close to surface
175
176 G4ThreeVector normal = NormAng(phi,u) ; // the normal vector at phi,u
177
178#ifdef G4TWISTDEBUG
179 G4cout << "normal vector = " << normal << G4endl ;
180 G4cout << "phi = " << phi << " , u = " << u << G4endl ;
181#endif
182
183 if (isGlobal)
184 {
185 fCurrentNormal.normal = ComputeGlobalDirection(normal.unit());
186 }
187 else
188 {
189 fCurrentNormal.normal = normal.unit();
190 }
191
192 return fCurrentNormal.normal;
193}

197
198G4int
199G4TwistTrapAlphaSide::DistanceToSurface(const G4ThreeVector &gp,
200 const G4ThreeVector &gv,
201 G4ThreeVector gxx[],
202 G4double distance[],
203 G4int areacode[],
204 G4bool isvalid[],
205 EValidate validate)
206{
207 static const G4double pihalf = pi/2 ;
208 const G4double ctol = 0.5 * kCarTolerance;
209
210 G4bool IsParallel = false ;
211 G4bool IsConverged = false ;
212
213 G4int nxx = 0 ; // number of physical solutions
214
215 fCurStatWithV.ResetfDone(validate, &gp, &gv);
216
217 if (fCurStatWithV.IsDone())
218 {
219 for (G4int i=0; i<fCurStatWithV.GetNXX(); i++)
220 {
221 gxx[i] = fCurStatWithV.GetXX(i);
222 distance[i] = fCurStatWithV.GetDistance(i);
223 areacode[i] = fCurStatWithV.GetAreacode(i);
224 isvalid[i] = fCurStatWithV.IsValid(i);
225 }
226 return fCurStatWithV.GetNXX();
227 }
228 else // initialise
229 {
230 for (G4int j=0; j<G4VSURFACENXX ; j++)
231 {
232 distance[j] = kInfinity;
233 areacode[j] = sOutside;
234 isvalid[j] = false;
235 gxx[j].set(kInfinity, kInfinity, kInfinity);
236 }
237 }
238
239 G4ThreeVector p = ComputeLocalPoint(gp);
240 G4ThreeVector v = ComputeLocalDirection(gv);
241
242#ifdef G4TWISTDEBUG
243 G4cout << "Local point p = " << p << G4endl ;
244 G4cout << "Local direction v = " << v << G4endl ;
245#endif
246
247 G4double phi,u ; // parameters
248
249 // temporary variables
250
251 G4double tmpdist = kInfinity ;
252 G4ThreeVector tmpxx;
253 G4int tmpareacode = sOutside ;
254 G4bool tmpisvalid = false ;
255
256 std::vector<Intersection> xbuf ;
257 Intersection xbuftmp ;
258
259 // prepare some variables for the intersection finder
260
261 G4double L = 2*fDz ;
262
263 G4double phixz = fPhiTwist * ( p.x() * v.z() - p.z() * v.x() ) ;
264 G4double phiyz = fPhiTwist * ( p.y() * v.z() - p.z() * v.y() ) ;
265
266
267 // special case vz = 0
268
269 if ( v.z() == 0. )
270 {
271 if ( std::fabs(p.z()) <= L ) // intersection possible in z
272 {
273 phi = p.z() * fPhiTwist / L ; // phi is determined by the z-position
274 u = (fDy1*(4*(-(fdeltaY*phi*v.x()) + fPhiTwist*p.y()*v.x()
275 + fdeltaX*phi*v.y() - fPhiTwist*p.x()*v.y())
276 + ((fDx3plus1 + fDx4plus2)*fPhiTwist
277 + 2*(fDx3minus1 + fDx4minus2)*phi)
278 *(v.y()*std::cos(phi) - v.x()*std::sin(phi))))
279 /(fPhiTwist*(4*fDy1* v.x() - (fa1md1 + 4*fDy1*fTAlph)*v.y())
280 *std::cos(phi) + fPhiTwist*(fa1md1*v.x()
281 + 4*fDy1*(fTAlph*v.x() + v.y()))*std::sin(phi));
282 xbuftmp.phi = phi ;
283 xbuftmp.u = u ;
284 xbuftmp.areacode = sOutside ;
285 xbuftmp.distance = kInfinity ;
286 xbuftmp.isvalid = false ;
287
288 xbuf.push_back(xbuftmp) ; // store it to xbuf
289 }
290 else // no intersection possible
291 {
292 distance[0] = kInfinity;
293 gxx[0].set(kInfinity,kInfinity,kInfinity);
294 isvalid[0] = false ;
295 areacode[0] = sOutside ;
296 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0],
297 areacode[0], isvalid[0],
298 0, validate, &gp, &gv);
299 return 0;
300 } // end std::fabs(p.z() <= L
301 } // end v.z() == 0
302 else // general solution for non-zero vz
303 {
304
305 G4double c[8],srd[7],si[7] ;
306
307 c[7] = 57600*
308 fDy1*(fa1md1*phiyz +
309 fDy1*(-4*phixz + 4*fTAlph*phiyz
310 + (fDx3plus1 + fDx4plus2)*fPhiTwist*v.z())) ;
311 c[6] = -57600*
312 fDy1*(4*fDy1*(phiyz + 2*fDz*v.x() + fTAlph*(phixz - 2*fDz*v.y()))
313 - 2*fDy1*(2*fdeltaX + fDx3minus1 + fDx4minus2
314 - 2*fdeltaY*fTAlph)*v.z()
315 + fa1md1*(phixz - 2*fDz*v.y() + fdeltaY*v.z()));
316 c[5] = 4800*
317 fDy1*(fa1md1*(-5*phiyz - 24*fDz*v.x() + 12*fdeltaX*v.z()) +
318 fDy1*(20*phixz - 4*(5*fTAlph*phiyz + 24*fDz*fTAlph*v.x()
319 + 24*fDz*v.y()) + (48*fdeltaY + (fDx3plus1 + fDx4plus2)
320 *fPhiTwist + 48*fdeltaX*fTAlph)*v.z()));
321 c[4] = 4800*
322 fDy1*(fa1md1*(phixz - 10*fDz*v.y() + 5*fdeltaY*v.z())
323 + 2*fDy1*(2*phiyz + 20*fDz*v.x()
324 + (-10*fdeltaX + fDx3minus1 + fDx4minus2)*v.z()
325 + 2*fTAlph*(phixz - 10*fDz*v.y() + 5*fdeltaY*v.z())));
326 c[3] = -96*
327 fDy1*(-(fa1md1*(phiyz + 100*fDz*v.x() - 50*fdeltaX*v.z()))
328 + fDy1*(4*phixz - 400*fDz*v.y()
329 + (200*fdeltaY - (fDx3plus1 + fDx4plus2)*fPhiTwist)*v.z()
330 - 4*fTAlph*(phiyz + 100*fDz*v.x() - 50*fdeltaX*v.z())));
331 c[2] = 32*
332 fDy1*(4*fDy1*(7*fTAlph*phixz + 7*phiyz - 6*fDz*v.x() + 6*fDz*fTAlph*v.y())
333 + 6*fDy1*(2*fdeltaX+fDx3minus1+fDx4minus2-2*fdeltaY*fTAlph)*v.z()
334 + fa1md1*(7*phixz + 6*fDz*v.y() - 3*fdeltaY*v.z()));
335 c[1] = -8*
336 fDy1*(fa1md1*(-9*phiyz - 56*fDz*v.x() + 28*fdeltaX*v.z())
337 + 4*fDy1*(9*phixz - 9*fTAlph*phiyz - 56*fDz*fTAlph*v.x()
338 - 56*fDz*v.y() + 28*(fdeltaY + fdeltaX*fTAlph)*v.z()));
339 c[0] = 72*
340 fDy1*(fa1md1*(2*fDz*v.y() - fdeltaY*v.z())
341 + fDy1*(-8*fDz*v.x() + 8*fDz*fTAlph*v.y()
342 + 4*fdeltaX*v.z() - 4*fdeltaY*fTAlph*v.z()));
343
344#ifdef G4TWISTDEBUG
345 G4cout << "coef = " << c[0] << " "
346 << c[1] << " "
347 << c[2] << " "
348 << c[3] << " "
349 << c[4] << " "
350 << c[5] << " "
351 << c[6] << " "
352 << c[7] << G4endl ;
353#endif
354
355 G4JTPolynomialSolver trapEq ;
356 G4int num = trapEq.FindRoots(c,7,srd,si);
357
358 for (G4int i = 0 ; i<num ; i++ ) // loop over all math solutions
359 {
360 if ( si[i]==0.0 ) // only real solutions
361 {
362#ifdef G4TWISTDEBUG
363 G4cout << "Solution " << i << " : " << srd[i] << G4endl ;
364#endif
365 phi = std::fmod(srd[i] , pihalf) ;
366 u = (fDy1*(4*(phiyz + 2*fDz*phi*v.y() - fdeltaY*phi*v.z())
367 - ((fDx3plus1 + fDx4plus2)*fPhiTwist
368 + 2*(fDx3minus1 + fDx4minus2)*phi)*v.z()*std::sin(phi)))
369 /(fPhiTwist*v.z()*(4*fDy1*std::cos(phi)
370 + (fa1md1 + 4*fDy1*fTAlph)*std::sin(phi)));
371 xbuftmp.phi = phi ;
372 xbuftmp.u = u ;
373 xbuftmp.areacode = sOutside ;
374 xbuftmp.distance = kInfinity ;
375 xbuftmp.isvalid = false ;
376
377 xbuf.push_back(xbuftmp) ; // store it to xbuf
378
379#ifdef G4TWISTDEBUG
380 G4cout << "solution " << i << " = " << phi << " , " << u << G4endl ;
381#endif
382 } // end if real solution
383 } // end loop i
384 } // end general case
385
386 nxx = xbuf.size() ; // save the number of solutions
387
388 G4ThreeVector xxonsurface ; // point on surface
389 G4ThreeVector surfacenormal ; // normal vector
390 G4double deltaX; // distance between intersection point and point on surface
391 G4double theta; // angle between track and surfacenormal
392 G4double factor; // a scaling factor
393 G4int maxint=30; // number of iterations
394
395 for ( size_t k = 0 ; k<xbuf.size() ; k++ )
396 {
397#ifdef G4TWISTDEBUG
398 G4cout << "Solution " << k << " : "
399 << "reconstructed phiR = " << xbuf[k].phi
400 << ", uR = " << xbuf[k].u << G4endl ;
401#endif
402
403 phi = xbuf[k].phi ; // get the stored values for phi and u
404 u = xbuf[k].u ;
405
406 IsConverged = false ; // no convergence at the beginning
407
408 for ( G4int i = 1 ; i<maxint ; i++ )
409 {
410 xxonsurface = SurfacePoint(phi,u) ;
411 surfacenormal = NormAng(phi,u) ;
412
413 tmpdist = DistanceToPlaneWithV(p, v, xxonsurface, surfacenormal, tmpxx);
414 deltaX = ( tmpxx - xxonsurface ).mag() ;
415 theta = std::fabs(std::acos(v*surfacenormal) - pihalf) ;
416 if ( theta < 0.001 )
417 {
418 factor = 50 ;
419 IsParallel = true ;
420 }
421 else
422 {
423 factor = 1 ;
424 }
425
426#ifdef G4TWISTDEBUG
427 G4cout << "Step i = " << i << ", distance = " << tmpdist
428 << ", " << deltaX << G4endl ;
429 G4cout << "X = " << tmpxx << G4endl ;
430#endif
431
432 GetPhiUAtX(tmpxx, phi, u) ;
433 // the new point xx is accepted and phi/u replaced
434
435#ifdef G4TWISTDEBUG
436 G4cout << "approximated phi = " << phi << ", u = " << u << G4endl ;
437#endif
438
439 if ( deltaX <= factor*ctol ) { IsConverged = true ; break ; }
440
441 } // end iterative loop (i)
442
443 if ( std::fabs(tmpdist)<ctol ) { tmpdist = 0 ; }
444
445#ifdef G4TWISTDEBUG
446 G4cout << "refined solution " << phi << " , " << u << G4endl ;
447 G4cout << "distance = " << tmpdist << G4endl ;
448 G4cout << "local X = " << tmpxx << G4endl ;
449#endif
450
451 tmpisvalid = false ; // init
452
453 if ( IsConverged )
454 {
455 if (validate == kValidateWithTol)
456 {
457 tmpareacode = GetAreaCode(tmpxx);
458 if (!IsOutside(tmpareacode))
459 {
460 if (tmpdist >= 0) tmpisvalid = true;
461 }
462 }
463 else if (validate == kValidateWithoutTol)
464 {
465 tmpareacode = GetAreaCode(tmpxx, false);
466 if (IsInside(tmpareacode))
467 {
468 if (tmpdist >= 0) { tmpisvalid = true; }
469 }
470 }
471 else // kDontValidate
472 {
473 G4Exception("G4TwistTrapAlphaSide::DistanceToSurface()",
474 "GeomSolids0001", FatalException,
475 "Feature NOT implemented !");
476 }
477 }
478 else
479 {
480 tmpdist = kInfinity; // no convergence after 10 steps
481 tmpisvalid = false ; // solution is not vaild
482 }
483
484 // store the found values
485 //
486 xbuf[k].xx = tmpxx ;
487 xbuf[k].distance = tmpdist ;
488 xbuf[k].areacode = tmpareacode ;
489 xbuf[k].isvalid = tmpisvalid ;
490
491 } // end loop over physical solutions (variable k)
492
493 std::sort(xbuf.begin() , xbuf.end(), DistanceSort ) ; // sorting
494
495#ifdef G4TWISTDEBUG
496 G4cout << G4endl << "list xbuf after sorting : " << G4endl ;
497 G4cout << G4endl << G4endl ;
498#endif
499
500 // erase identical intersection (within kCarTolerance)
501 //
502 xbuf.erase( std::unique(xbuf.begin(), xbuf.end() , EqualIntersection ),
503 xbuf.end() );
504
505
506 // add guesses
507 //
508 G4int nxxtmp = xbuf.size() ;
509
510 if ( nxxtmp<2 || IsParallel ) // positive end
511 {
512
513#ifdef G4TWISTDEBUG
514 G4cout << "add guess at +z/2 .. " << G4endl ;
515#endif
516
517 phi = fPhiTwist/2 ;
518 u = 0 ;
519
520 xbuftmp.phi = phi ;
521 xbuftmp.u = u ;
522 xbuftmp.areacode = sOutside ;
523 xbuftmp.distance = kInfinity ;
524 xbuftmp.isvalid = false ;
525
526 xbuf.push_back(xbuftmp) ; // store it to xbuf
527
528#ifdef G4TWISTDEBUG
529 G4cout << "add guess at -z/2 .. " << G4endl ;
530#endif
531
532 phi = -fPhiTwist/2 ;
533 u = 0 ;
534
535 xbuftmp.phi = phi ;
536 xbuftmp.u = u ;
537 xbuftmp.areacode = sOutside ;
538 xbuftmp.distance = kInfinity ;
539 xbuftmp.isvalid = false ;
540
541 xbuf.push_back(xbuftmp) ; // store it to xbuf
542
543 for ( size_t k = nxxtmp ; k<xbuf.size() ; k++ )
544 {
545
546#ifdef G4TWISTDEBUG
547 G4cout << "Solution " << k << " : "
548 << "reconstructed phiR = " << xbuf[k].phi
549 << ", uR = " << xbuf[k].u << G4endl ;
550#endif
551
552 phi = xbuf[k].phi ; // get the stored values for phi and u
553 u = xbuf[k].u ;
554
555 IsConverged = false ; // no convergence at the beginning
556
557 for ( G4int i = 1 ; i<maxint ; i++ )
558 {
559 xxonsurface = SurfacePoint(phi,u) ;
560 surfacenormal = NormAng(phi,u) ;
561 tmpdist = DistanceToPlaneWithV(p, v, xxonsurface, surfacenormal, tmpxx);
562 deltaX = ( tmpxx - xxonsurface ).mag();
563 theta = std::fabs(std::acos(v*surfacenormal) - pihalf);
564 if ( theta < 0.001 )
565 {
566 factor = 50 ;
567 }
568 else
569 {
570 factor = 1 ;
571 }
572
573#ifdef G4TWISTDEBUG
574 G4cout << "Step i = " << i << ", distance = " << tmpdist
575 << ", " << deltaX << G4endl
576 << "X = " << tmpxx << G4endl ;
577#endif
578
579 GetPhiUAtX(tmpxx, phi, u) ;
580 // the new point xx is accepted and phi/u replaced
581
582#ifdef G4TWISTDEBUG
583 G4cout << "approximated phi = " << phi << ", u = " << u << G4endl ;
584#endif
585
586 if ( deltaX <= factor*ctol ) { IsConverged = true ; break ; }
587
588 } // end iterative loop (i)
589
590 if ( std::fabs(tmpdist)<ctol ) { tmpdist = 0; }
591
592#ifdef G4TWISTDEBUG
593 G4cout << "refined solution " << phi << " , " << u << G4endl ;
594 G4cout << "distance = " << tmpdist << G4endl ;
595 G4cout << "local X = " << tmpxx << G4endl ;
596#endif
597
598 tmpisvalid = false ; // init
599
600 if ( IsConverged )
601 {
602 if (validate == kValidateWithTol)
603 {
604 tmpareacode = GetAreaCode(tmpxx);
605 if (!IsOutside(tmpareacode))
606 {
607 if (tmpdist >= 0) { tmpisvalid = true; }
608 }
609 }
610 else if (validate == kValidateWithoutTol)
611 {
612 tmpareacode = GetAreaCode(tmpxx, false);
613 if (IsInside(tmpareacode))
614 {
615 if (tmpdist >= 0) { tmpisvalid = true; }
616 }
617 }
618 else // kDontValidate
619 {
620 G4Exception("G4TwistedBoxSide::DistanceToSurface()",
621 "GeomSolids0001", FatalException,
622 "Feature NOT implemented !");
623 }
624 }
625 else
626 {
627 tmpdist = kInfinity; // no convergence after 10 steps
628 tmpisvalid = false ; // solution is not vaild
629 }
630
631 // store the found values
632 //
633 xbuf[k].xx = tmpxx ;
634 xbuf[k].distance = tmpdist ;
635 xbuf[k].areacode = tmpareacode ;
636 xbuf[k].isvalid = tmpisvalid ;
637
638 } // end loop over physical solutions
639 } // end less than 2 solutions
640
641 // sort again
642 std::sort(xbuf.begin() , xbuf.end(), DistanceSort ) ; // sorting
643
644 // erase identical intersection (within kCarTolerance)
645 xbuf.erase( std::unique(xbuf.begin(), xbuf.end() , EqualIntersection ) ,
646 xbuf.end() );
647
648#ifdef G4TWISTDEBUG
649 G4cout << G4endl << "list xbuf after sorting : " << G4endl ;
650 G4cout << G4endl << G4endl ;
651#endif
652
653 nxx = xbuf.size() ; // determine number of solutions again.
654

658 gxx[i] = ComputeGlobalPoint(xbuf[i].xx);

661
662 fCurStatWithV.SetCurrentStatus(i, gxx[i], distance[i], areacode[i],
663 isvalid[i], nxx, validate, &gp, &gv);
664#ifdef G4TWISTDEBUG

678 G4cout << nxx << " possible physical solutions found" << G4endl ;
679 for ( G4int k= 0 ; k< nxx ; k++ )

681 G4cout << "global intersection Point found: " << gxx[k] << G4endl ;
682 G4cout << "distance = " << distance[k] << G4endl ;
683 G4cout << "isvalid = " << isvalid[k] << G4endl ;
684 }
685#endif

691//=====================================================================
692//* DistanceToSurface -------------------------------------------------
693
694G4int
695G4TwistTrapAlphaSide::DistanceToSurface(const G4ThreeVector &gp,
696 G4ThreeVector gxx[],
697 G4double distance[],
698 G4int areacode[])
699{
700 const G4double ctol = 0.5 * kCarTolerance;
701
702 fCurStat.ResetfDone(kDontValidate, &gp);
703
704 if (fCurStat.IsDone())
705 {
706 for (G4int i=0; i<fCurStat.GetNXX(); i++)
707 {
708 gxx[i] = fCurStat.GetXX(i);
709 distance[i] = fCurStat.GetDistance(i);
710 areacode[i] = fCurStat.GetAreacode(i);
711 }
712 return fCurStat.GetNXX();
713 }
714 else // initialize
715 {
716 for (G4int i=0; i<G4VSURFACENXX; i++)
717 {
718 distance[i] = kInfinity;
719 areacode[i] = sOutside;
720 gxx[i].set(kInfinity, kInfinity, kInfinity);
721 }
722 }
723
724 G4ThreeVector p = ComputeLocalPoint(gp);
725 G4ThreeVector xx; // intersection point
726 G4ThreeVector xxonsurface ; // interpolated intersection point
727
728 // the surfacenormal at that surface point
729 //
730 G4double phiR = 0 ;
731 G4double uR = 0 ;
732
733 G4ThreeVector surfacenormal ;
734 G4double deltaX, uMax ;
735 G4double halfphi = 0.5*fPhiTwist ;
736
737 for ( G4int i = 1 ; i<20 ; i++ )
738 {
739 xxonsurface = SurfacePoint(phiR,uR) ;
740 surfacenormal = NormAng(phiR,uR) ;
741 distance[0] = DistanceToPlane(p,xxonsurface,surfacenormal,xx); // new XX
742 deltaX = ( xx - xxonsurface ).mag() ;
743
744#ifdef G4TWISTDEBUG
745 G4cout << "i = " << i << ", distance = " << distance[0]
746 << ", " << deltaX << G4endl
747 << "X = " << xx << G4endl ;
748#endif
749
750 // the new point xx is accepted and phi/psi replaced
751 //
752 GetPhiUAtX(xx, phiR, uR) ;
753
754 if ( deltaX <= ctol ) { break ; }
755 }
756
757 // check validity of solution ( valid phi,psi )
758
759 uMax = GetBoundaryMax(phiR) ;
760
761 if ( phiR > halfphi ) { phiR = halfphi ; }
762 if ( phiR < -halfphi ) { phiR = -halfphi ; }
763 if ( uR > uMax ) { uR = uMax ; }
764 if ( uR < -uMax ) { uR = -uMax ; }
765
766 xxonsurface = SurfacePoint(phiR,uR) ;
767 distance[0] = ( p - xx ).mag() ;
768 if ( distance[0] <= ctol ) { distance[0] = 0 ; }
769
770 // end of validity
771
772#ifdef G4TWISTDEBUG
773 G4cout << "refined solution " << phiR << " , " << uR << " , " << G4endl ;
774 G4cout << "distance = " << distance[0] << G4endl ;
775 G4cout << "X = " << xx << G4endl ;
776#endif
777
778 G4bool isvalid = true;
779 gxx[0] = ComputeGlobalPoint(xx);
780
781#ifdef G4TWISTDEBUG
782 G4cout << "intersection Point found: " << gxx[0] << G4endl ;
783 G4cout << "distance = " << distance[0] << G4endl ;
784#endif
785
786 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
787 isvalid, 1, kDontValidate, &gp);
788 return 1;
789}

793//* GetAreaCode -------------------------------------------------------
794
795G4int
796G4TwistTrapAlphaSide::GetAreaCode(const G4ThreeVector &xx, G4bool withTol)
797{
798 // We must use the function in local coordinate system.
799 // See the description of DistanceToSurface(p,v).
800
801 const G4double ctol = 0.5 * kCarTolerance;
802
803 G4double phi ;
804 G4double yprime ;
805 GetPhiUAtX(xx, phi,yprime ) ;
806
807 G4double fYAxisMax = GetBoundaryMax(phi) ;
808 G4double fYAxisMin = GetBoundaryMin(phi) ;
809
810#ifdef G4TWISTDEBUG
811 G4cout << "GetAreaCode: phi = " << phi << G4endl ;
812 G4cout << "GetAreaCode: yprime = " << yprime << G4endl ;
813 G4cout << "Intervall is " << fYAxisMin << " to " << fYAxisMax << G4endl ;
814#endif
815
816 G4int areacode = sInside;
817
818 if (fAxis[0] == kYAxis && fAxis[1] == kZAxis)
819 {
820 G4int zaxis = 1;
821
822 if (withTol)
823 {
824 G4bool isoutside = false;
825
826 // test boundary of yaxis
827
828 if (yprime < fYAxisMin + ctol)
829 {
830 areacode |= (sAxis0 & (sAxisY | sAxisMin)) | sBoundary;
831 if (yprime <= fYAxisMin - ctol) { isoutside = true; }
832
833 }
834 else if (yprime > fYAxisMax - ctol)
835 {
836 areacode |= (sAxis0 & (sAxisY | sAxisMax)) | sBoundary;
837 if (yprime >= fYAxisMax + ctol) { isoutside = true; }
838 }
839
840 // test boundary of z-axis
841
842 if (xx.z() < fAxisMin[zaxis] + ctol)
843 {
844 areacode |= (sAxis1 & (sAxisZ | sAxisMin));
845
846 if (areacode & sBoundary) // xx is on the corner
847 { areacode |= sCorner; }
848
849 else
850 { areacode |= sBoundary; }
851 if (xx.z() <= fAxisMin[zaxis] - ctol) { isoutside = true; }
852 }
853 else if (xx.z() > fAxisMax[zaxis] - ctol)
854 {
855 areacode |= (sAxis1 & (sAxisZ | sAxisMax));
856
857 if (areacode & sBoundary) // xx is on the corner
858 { areacode |= sCorner; }
859 else
860 { areacode |= sBoundary; }
861 if (xx.z() >= fAxisMax[zaxis] + ctol) { isoutside = true; }
862 }
863
864 // if isoutside = true, clear inside bit.
865 // if not on boundary, add axis information.
866
867 if (isoutside)
868 {
869 G4int tmpareacode = areacode & (~sInside);
870 areacode = tmpareacode;
871 }
872 else if ((areacode & sBoundary) != sBoundary)
873 {
874 areacode |= (sAxis0 & sAxisY) | (sAxis1 & sAxisZ);
875 }
876
877 }
878 else
879 {
880 // boundary of y-axis
881
882 if (yprime < fYAxisMin )
883 {
884 areacode |= (sAxis0 & (sAxisY | sAxisMin)) | sBoundary;
885 }
886 else if (yprime > fYAxisMax)
887 {
888 areacode |= (sAxis0 & (sAxisY | sAxisMax)) | sBoundary;
889 }
890
891 // boundary of z-axis
892
893 if (xx.z() < fAxisMin[zaxis])
894 {
895 areacode |= (sAxis1 & (sAxisZ | sAxisMin));
896 if (areacode & sBoundary) // xx is on the corner
897 { areacode |= sCorner; }
898 else
899 { areacode |= sBoundary; }
900 }
901 else if (xx.z() > fAxisMax[zaxis])
902 {
903 areacode |= (sAxis1 & (sAxisZ | sAxisMax)) ;
904 if (areacode & sBoundary) // xx is on the corner
905 { areacode |= sCorner; }
906 else
907 { areacode |= sBoundary; }
908 }
909
910 if ((areacode & sBoundary) != sBoundary)
911 {
912 areacode |= (sAxis0 & sAxisY) | (sAxis1 & sAxisZ);
913 }
914 }
915 return areacode;
916 }
917 else
918 {
919 G4Exception("G4TwistTrapAlphaSide::GetAreaCode()",
920 "GeomSolids0001", FatalException,
921 "Feature NOT implemented !");
922 }
923 return areacode;
924}

1040//=====================================================================
1041//* GetPhiUAtX --------------------------------------------------------
1042
1043void
1044G4TwistTrapAlphaSide::GetPhiUAtX( G4ThreeVector p, G4double &phi, G4double &u )
1045{
1046 // find closest point XX on surface for a given point p
1047 // X0 is a point on the surface, d is the direction
1048 // ( both for a fixed z = pz)
1049
1050 // phi is given by the z coordinate of p
1051
1052 phi = p.z()/(2*fDz)*fPhiTwist ;
1053 u = (fPhiTwist*(2*fDx1*fDx1 - 2*fDx2*fDx2 - fa1md1*(fDx3 + fDx4)
1054 - 4*(fDx3plus1 + fDx4plus2)*fDy1*fTAlph)
1055 - 2*(2*fDx1*fDx1 - 2*fDx2*fDx2 + fa1md1*(fDx3 + fDx4)
1056 + 4*(fDx3minus1 + fDx4minus2)*fDy1*fTAlph)*phi
1057 - 4*(fa1md1*(fdeltaX*phi - fPhiTwist*p.x())
1058 + 4*fDy1*(fdeltaY*phi + fdeltaX*fTAlph*phi
1059 - fPhiTwist*(fTAlph*p.x() + p.y())))*std::cos(phi)
1060 - 4*(fa1md1*fdeltaY*phi - 4*fdeltaX*fDy1*phi
1061 + 4*fdeltaY*fDy1*fTAlph*phi + 4*fDy1*fPhiTwist*p.x()
1062 - fPhiTwist*(fa1md1 + 4*fDy1*fTAlph)*p.y())*std::sin(phi))
1063 /(fDy1* fPhiTwist*((std::fabs(((fa1md1 + 4*fDy1*fTAlph)*std::cos(phi))
1064 /fDy1 - 4*std::sin(phi)))
1065 *(std::fabs(((fa1md1 + 4*fDy1*fTAlph)*std::cos(phi))
1066 /fDy1 - 4*std::sin(phi)))
1067 + (std::fabs(4*std::cos(phi)
1068 + ((fa1md1 + 4*fDy1*fTAlph)*std::sin(phi))/fDy1))
1069 * (std::fabs(4*std::cos(phi)
1070 + ((fa1md1 + 4*fDy1*fTAlph)*std::sin(phi))/fDy1)))) ;
1071}

1076G4ThreeVector
1077G4TwistTrapAlphaSide::ProjectPoint(const G4ThreeVector &p, G4bool isglobal)
1078{
1079 // Get Rho at p.z() on Hyperbolic Surface.
1080
1081 G4ThreeVector tmpp;
1082 if (isglobal)
1083 {
1084 tmpp = fRot.inverse()*p - fTrans;
1085 }
1086 else
1087 {
1088 tmpp = p;
1089 }
1090
1091 G4double phi ;
1092 G4double u ;
1093
1094 GetPhiUAtX( tmpp, phi, u ) ;
1095 // calculate (phi, u) for a point p close the surface
1096
1097 G4ThreeVector xx = SurfacePoint(phi,u) ;
1098 // transform back to cartesian coordinates
1099
1100 if (isglobal)
1101 {
1102 return (fRot * xx + fTrans);
1103 }
1104 else
1105 {
1106 return xx;
1107 }

1113void
1114G4TwistTrapAlphaSide::GetFacets( G4int k, G4int n, G4double xyz[][3],
1115 G4int faces[][4], G4int iside )
1116{
1117
1118 G4double phi ;
1119 G4double b ;
1120
1121 G4double z, u ; // the two parameters for the surface equation
1122 G4ThreeVector p ; // a point on the surface, given by (z,u)
1123
1124 G4int nnode ;
1125 G4int nface ;
1126
1127 // calculate the (n-1)*(k-1) vertices
1128
1129 for ( G4int i = 0 ; i<n ; i++ )
1130 {
1131 z = -fDz+i*(2.*fDz)/(n-1) ;
1132 phi = z*fPhiTwist/(2*fDz) ;
1133 b = GetValueB(phi) ;
1134
1135 for ( G4int j = 0 ; j<k ; j++ )
1136 {
1137 nnode = GetNode(i,j,k,n,iside) ;
1138 u = -b/2 +j*b/(k-1) ;
1139 p = SurfacePoint(phi,u,true) ; // surface point in global coordinates
1140
1141 xyz[nnode][0] = p.x() ;
1142 xyz[nnode][1] = p.y() ;
1143 xyz[nnode][2] = p.z() ;
1144
1145 if ( i<n-1 && j<k-1 ) // conterclock wise filling
1146 {
1147 nface = GetFace(i,j,k,n,iside) ;
1148 faces[nface][0] = GetEdgeVisibility(i,j,k,n,0,-1)
1149 * (GetNode(i ,j ,k,n,iside)+1) ; // f77 numbering
1150 faces[nface][1] = GetEdgeVisibility(i,j,k,n,1,-1)
1151 * (GetNode(i ,j+1,k,n,iside)+1) ;
1152 faces[nface][2] = GetEdgeVisibility(i,j,k,n,2,-1)
1153 * (GetNode(i+1,j+1,k,n,iside)+1) ;
1154 faces[nface][3] = GetEdgeVisibility(i,j,k,n,3,-1)
1155 * (GetNode(i+1,j ,k,n,iside)+1) ;
1156 }
1157 }
1158 }
