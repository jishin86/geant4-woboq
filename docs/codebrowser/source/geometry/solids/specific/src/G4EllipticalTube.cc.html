160//////////////////////////////////////////////////////////////////////////
161//
162// Check dimensions
163
164void G4EllipticalTube::CheckParameters()
165{
166 // Check dimensions
167 //
168 halfTolerance = 0.5*kCarTolerance; // half tolerance
169 G4double dmin = 2*kCarTolerance;
170 if (fDx < dmin || fDy < dmin || fDz < dmin)
171 {
172 std::ostringstream message;
173 message << "Invalid (too small or negative) dimensions for Solid: "
174 << GetName()
175 << "\n Dx = " << fDx
176 << "\n Dy = " << fDy
177 << "\n Dz = " << fDz;
178 G4Exception("G4EllipticalTube::CheckParameters()", "GeomSolids0002",
179 FatalException, message);
180 }
181
182 // Set pre-calculatated values
183 //
184 halfTolerance = 0.5*kCarTolerance; // half tolerance
185 fRsph = std::sqrt(fDx * fDx + fDy * fDy + fDz * fDz); // radius of surrounding sphere
186 fDDx = fDx * fDx; // X semi-axis squared
187 fDDy = fDy * fDy; // Y semi-axis squared
188
189 fR = std::min(fDx, fDy); // resulting radius, after scaling elipse to circle
190 fSx = fR / fDx; // X scale factor
191 fSy = fR / fDy; // Y scale factor
192
193 fQ1 = 0.5 / fR; // distance approxiamtion dist = Q1 * (x^2 + y^2) - Q2
194 fQ2 = 0.5 * (fR + halfTolerance * halfTolerance / fR);
195 fScratch = 2. * fR * fR * DBL_EPSILON; // scratch within calculation error thickness
196 // fScratch = (B * B / A) * (2. + halfTolerance / A) * halfTolerance; // alternative
197}

210//////////////////////////////////////////////////////////////////////////
211//
212// Calculate extent under transform and specified limit
213
214G4bool
215G4EllipticalTube::CalculateExtent( const EAxis pAxis,
216 const G4VoxelLimits& pVoxelLimit,
217 const G4AffineTransform& pTransform,
218 G4double& pMin, G4double& pMax ) const
219{
220 G4ThreeVector bmin, bmax;
221 G4bool exist;
222
223 // Check bounding box (bbox)
224 //
225 BoundingLimits(bmin,bmax);
226 G4BoundingEnvelope bbox(bmin,bmax);
227#ifdef G4BBOX_EXTENT
228 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit, pTransform, pMin, pMax);
229#endif
230 if (bbox.BoundingBoxVsVoxelLimits(pAxis, pVoxelLimit, pTransform, pMin, pMax))
231 {
232 return exist = (pMin < pMax) ? true : false;
233 }
234
235 G4double dx = fDx;
236 G4double dy = fDy;
237 G4double dz = fDz;
238
239 // Set bounding envelope (benv) and calculate extent
240 //
241 const G4int NSTEPS = 24; // number of steps for whole circle
242 G4double ang = twopi/NSTEPS;
243
244 G4double sinHalf = std::sin(0.5*ang);
245 G4double cosHalf = std::cos(0.5*ang);
246 G4double sinStep = 2.*sinHalf*cosHalf;
247 G4double cosStep = 1. - 2.*sinHalf*sinHalf;
248 G4double sx = dx/cosHalf;
249 G4double sy = dy/cosHalf;
250
251 G4double sinCur = sinHalf;
252 G4double cosCur = cosHalf;
253 G4ThreeVectorList baseA(NSTEPS),baseB(NSTEPS);
254 for (G4int k=0; k<NSTEPS; ++k)
255 {
256 baseA[k].set(sx*cosCur,sy*sinCur,-dz);
257 baseB[k].set(sx*cosCur,sy*sinCur, dz);
258
259 G4double sinTmp = sinCur;
260 sinCur = sinCur*cosStep + cosCur*sinStep;
261 cosCur = cosCur*cosStep - sinTmp*sinStep;
262 }
263
264 std::vector<const G4ThreeVectorList *> polygons(2);
265 polygons[0] = &baseA;
266 polygons[1] = &baseB;
267 G4BoundingEnvelope benv(bmin, bmax, polygons);
268 exist = benv.CalculateExtent(pAxis, pVoxelLimit, pTransform, pMin, pMax);
269 return exist;
270}

272//////////////////////////////////////////////////////////////////////////
273//
274// Determine where is point: inside, outside or on surface
275//
276
277EInside G4EllipticalTube::Inside( const G4ThreeVector& p ) const
278{
279 G4double x = p.x() * fSx;
280 G4double y = p.y() * fSy;
281 G4double distR = fQ1 * (x * x + y * y) - fQ2;
282 G4double distZ = std::abs(p.z()) - fDz;
283 G4double dist = std::max(distR, distZ);
284
285 if (dist > halfTolerance) return kOutside;
286 return (dist > -halfTolerance) ? kSurface : kInside;
287}

289//////////////////////////////////////////////////////////////////////////
290//
291// Return unit normal at surface closest to p
292
293G4ThreeVector G4EllipticalTube::SurfaceNormal( const G4ThreeVector& p ) const
294{
295 G4ThreeVector norm(0, 0, 0);
296 G4int nsurf = 0;
297
298 // check lateral surface
299 G4double x = p.x() * fSx;
300 G4double y = p.y() * fSy;
301 G4double distR = fQ1 * (x * x + y * y) - fQ2;
302 if (std::abs(distR) <= halfTolerance) {
303 norm = G4ThreeVector(p.x() * fDDy, p.y() * fDDx, 0.).unit();
304 ++nsurf;
305 }
306
307 // check lateral bases
308 G4double distZ = std::abs(p.z()) - fDz;
309 if (std::abs(distZ) <= halfTolerance) {
310 norm.setZ(p.z() < 0 ? -1. : 1.);
311 ++nsurf;
312 }
313
314 // return normal
315 if (nsurf == 1) return norm;
316 else if (nsurf > 1) return norm.unit(); // edge
317 else
318 {
319 // Point is not on the surface
320 //
321#ifdef G4SPECDEBUG
322 std::ostringstream message;
323 G4int oldprc = message.precision(16);
324 message << "Point p is not on surface (!?) of solid: "
325 << GetName() << G4endl;
326 message << "Position:\n";
327 message << " p.x() = " << p.x()/mm << " mm\n";
328 message << " p.y() = " << p.y()/mm << " mm\n";
329 message << " p.z() = " << p.z()/mm << " mm";
330 G4cout.precision(oldprc);
331 G4Exception("G4EllipticalTube::SurfaceNormal(p)", "GeomSolids1002",
332 JustWarning, message );
333 DumpInfo();
334#endif
335 return ApproxSurfaceNormal(p);
336 }
337}

339//////////////////////////////////////////////////////////////////////////
340//
341// Find surface nearest to point and return corresponding normal.
342// The algorithm is similar to the algorithm used in Inside().
343// This method normally should not be called.
344
345G4ThreeVector
346G4EllipticalTube::ApproxSurfaceNormal( const G4ThreeVector& p ) const
347{
348 G4double x = p.x() * fSx;
349 G4double y = p.y() * fSy;
350 G4double distR = fQ1 * (x * x + y * y) - fQ2;
351 G4double distZ = std::abs(p.z()) - fDz;
352 if (distR > distZ && (x * x + y * y) > 0)
353 return G4ThreeVector(p.x() * fDDy, p.y() * fDDx, 0.).unit();
354 else
355 return G4ThreeVector(0, 0, (p.z() < 0 ? -1. : 1.));
356}

358//////////////////////////////////////////////////////////////////////////
359//
360// Calculate distance to shape from outside, along normalised vector,
361// return kInfinity if no intersection, or distance < halfTolerance
362
363G4double G4EllipticalTube::DistanceToIn( const G4ThreeVector& p,
364 const G4ThreeVector& v ) const
365{
366 G4double offset = 0.;
367 G4ThreeVector pcur = p;
368
369 // Check if point is flying away
370 //
371 G4double safex = std::abs(pcur.x()) - fDx;
372 G4double safey = std::abs(pcur.y()) - fDy;
373 G4double safez = std::abs(pcur.z()) - fDz;
374
375 if (safez >= -halfTolerance && pcur.z() * v.z() >= 0.) return kInfinity;
376 if (safey >= -halfTolerance && pcur.y() * v.y() >= 0.) return kInfinity;
377 if (safex >= -halfTolerance && pcur.x() * v.x() >= 0.) return kInfinity;
378
379 // Relocate point, if required
380 //
381 G4double Dmax = 32. * fRsph;
382 if (std::max(std::max(safex, safey), safez) > Dmax)
383 {
384 offset = (1. - 1.e-08) * pcur.mag() - 2. * fRsph;
385 pcur += offset * v;
386 G4double dist = DistanceToIn(pcur, v);
387 return (dist == kInfinity) ? kInfinity : dist + offset;
388 }
389
390 // Scale elliptical tube to cylinder
391 //
392 G4double px = pcur.x() * fSx;
393 G4double py = pcur.y() * fSy;
394 G4double pz = pcur.z();
395 G4double vx = v.x() * fSx;
396 G4double vy = v.y() * fSy;
397 G4double vz = v.z();
398
399 // Set coefficients of quadratic equation: A t^2 + 2B t + C = 0
400 //
401 G4double rr = px * px + py * py;
402 G4double A = vx * vx + vy * vy;
403 G4double B = px * vx + py * vy;
404 G4double C = rr - fR * fR;
405 G4double D = B * B - A * C;
406
407 // Check if point is flying away relative to lateral surface
408 //
409 G4double distR = fQ1 * rr - fQ2;
410 G4bool parallelToZ = (A < DBL_EPSILON || std::abs(vz) >= 1.);
411 if (distR >= -halfTolerance && (B >= 0. || parallelToZ)) return kInfinity;
412
413 // Find intersection with Z planes
414 //
415 G4double invz = (vz == 0) ? DBL_MAX : -1./vz;
416 G4double dz = std::copysign(fDz, invz);
417 G4double tzmin = (pz - dz) * invz;
418 G4double tzmax = (pz + dz) * invz;
419
420 // Solve qudratic equation. There are two cases special where D <= 0:
421 // 1) trajectory parallel to Z axis (A = 0, B = 0, C - any, D = 0)
422 // 2) touch (D = 0) or no intersection (D < 0) with lateral surface
423 //
424 if (parallelToZ) return (tzmin < halfTolerance) ? offset : tzmin + offset; // 1)
425 if (D <= A * A * fScratch) return kInfinity; // 2)
426
427 // Find roots of quadratic equation
428 G4double tmp = -B - std::copysign(std::sqrt(D), B);
429 G4double t1 = tmp / A;
430 G4double t2 = C / tmp;
431 G4double trmin = std::min(t1, t2);
432 G4double trmax = std::max(t1, t2);
433
434 // Return distance
435 G4double tin = std::max(tzmin, trmin);
436 G4double tout = std::min(tzmax, trmax);
437
438 if (tout <= tin + halfTolerance) return kInfinity; // touch or no hit
439 return (tin < halfTolerance) ? offset : tin + offset;
440}

442//////////////////////////////////////////////////////////////////////////
443//
444// Estimate distance to the surface from outside,
445// returns 0 if point is inside
446
447G4double G4EllipticalTube::DistanceToIn( const G4ThreeVector& p ) const
448{
449 // safety distance to bounding box
450 G4double distX = std::abs(p.x()) - fDx;
451 G4double distY = std::abs(p.y()) - fDy;
452 G4double distZ = std::abs(p.z()) - fDz;
453 G4double distB = std::max(std::max(distX, distY), distZ);
454 // return (distB < 0) ? 0 : distB;
455
456 // safety distance to lateral surface
457 G4double x = p.x() * fSx;
458 G4double y = p.y() * fSy;
459 G4double distR = std::sqrt(x * x + y * y) - fR;
460
461 // return SafetyToIn
462 G4double dist = std::max(distB, distR);
463 return (dist < 0) ? 0 : dist;
464}

466//////////////////////////////////////////////////////////////////////////
467//
468// Calculate distance to shape from inside and find normal
469// at exit point, if required
470// - when leaving the surface, return 0
471
472G4double G4EllipticalTube::DistanceToOut( const G4ThreeVector& p,
473 const G4ThreeVector& v,
474 const G4bool calcNorm,
475 G4bool *validNorm,
476 G4ThreeVector *n ) const
477{
478 // Check if point flying away relative to Z planes
479 //
480 G4double pz = p.z();
481 G4double vz = v.z();
482 G4double distZ = std::abs(pz) - fDz;
483 if (distZ >= -halfTolerance && pz * vz > 0)
484 {
485 if (calcNorm)
486 {
487 *validNorm = true;
488 n->set(0, 0, (pz < 0) ? -1. : 1.);
489 }
490 return 0.;
491 }
492 G4double tzmax = (vz == 0) ? DBL_MAX : (std::copysign(fDz, vz) - pz) / vz;
493
494 // Scale elliptical tube to cylinder
495 //
496 G4double px = p.x() * fSx;
497 G4double py = p.y() * fSy;
498 G4double vx = v.x() * fSx;
499 G4double vy = v.y() * fSy;
500
501 // Check if point is flying away relative to lateral surface
502 //
503 G4double rr = px * px + py * py;
504 G4double B = px * vx + py * vy;
505 G4double distR = fQ1 * rr - fQ2;
506 if (distR >= -halfTolerance && B > 0.)
507 {
508 if (calcNorm)
509 {
510 *validNorm = true;
511 *n = G4ThreeVector(px * fDDy, py * fDDx, 0.).unit();
512 }
513 return 0.;
514 }
515
516 // Just in case check if point is outside, normally it should never be
517 //
518 if (std::max(distZ, distR) > halfTolerance) {
519#ifdef G4SPECDEBUG
520 std::ostringstream message;
521 G4int oldprc = message.precision(16);
522 message << "Point p is outside (!?) of solid: "
523 << GetName() << G4endl;
524 message << "Position: " << p << G4endl;;
525 message << "Direction: " << v;
526 G4cout.precision(oldprc);
527 G4Exception("G4EllipticalTube::DistanceToOut(p,v)", "GeomSolids1002",
528 JustWarning, message );
529 DumpInfo();
530#endif
531 if (calcNorm)
532 {
533 *validNorm = true;
534 *n = ApproxSurfaceNormal(p);
535 }
536 return 0.;
537 }
538
539 // Set coefficients of quadratic equation: A t^2 + 2B t + C = 0
540 //
541 G4double A = vx * vx + vy * vy;
542 G4double C = rr - fR * fR;
543 G4double D = B * B - A * C;
544
545 // Solve qudratic equation. There are two special cases where D <= 0:
546 // 1) trajectory parallel to Z axis (A = 0, B = 0, C - any, D = 0)
547 // 2) touch (D = 0) or no intersection (D < 0) with lateral surface
548 //
549 G4bool parallelToZ = (A < DBL_EPSILON || std::abs(vz) >= 1.);
550 if (parallelToZ) // 1)
551 {
552 if (calcNorm)
553 {
554 *validNorm = true;
555 n->set(0, 0, (vz < 0) ? -1. : 1.);
556 }
557 return tzmax;
558 }
559 if (D <= A * A * fScratch) // 2)
560 {
561 if (calcNorm)
562 {
563 *validNorm = true;
564 *n = G4ThreeVector(px * fDDy, py * fDDx, 0.).unit();
565 }
566 return 0.;
567 }
568
569 // Find roots of quadratic equation
570 G4double tmp = -B - std::copysign(std::sqrt(D), B);
571 G4double t1 = tmp / A;
572 G4double t2 = C / tmp;
573 G4double trmax = std::max(t1, t2);
574
575 // Return distance
576 G4double tmax = std::min(tzmax, trmax);
577
578 // Set normal, if required, and return distance
579 //
580 if (calcNorm)
581 {
582 *validNorm = true;
583 G4ThreeVector pnew = p + tmax * v;
584 if (tmax == tzmax)
585 n->set(0, 0, (pnew.z() < 0) ? -1. : 1.);
586 else
587 *n = G4ThreeVector(pnew.x() * fDDy, pnew.y() * fDDx, 0.).unit();
588 }
589 return tmax;
590}

594// Estimate distance to the surface from inside,
595// returns 0 if point is outside
596//
597
598G4double G4EllipticalTube::DistanceToOut( const G4ThreeVector& p ) const
599{
600#ifdef G4SPECDEBUG
601 if( Inside(p) == kOutside )
602 {
603 std::ostringstream message;
604 G4int oldprc = message.precision(16);
605 message << "Point p is outside (!?) of solid: " << GetName() << "\n"
606 << "Position:\n"
607 << " p.x() = " << p.x()/mm << " mm\n"
608 << " p.y() = " << p.y()/mm << " mm\n"
609 << " p.z() = " << p.z()/mm << " mm";
610 message.precision(oldprc) ;
611 G4Exception("G4ElliptocalTube::DistanceToOut(p)", "GeomSolids1002",
612 JustWarning, message);
613 DumpInfo();
614 }
615#endif
616 // safety distance to Z-bases
617 G4double distZ = fDz - std::abs(p.z());
618
619 // safety distance lateral surface
620 G4double x = p.x() * fSx;
621 G4double y = p.y() * fSy;
622 G4double distR = fR - std::sqrt(x * x + y * y);
623
624 // return SafetyToOut
625 G4double dist = std::min(distZ, distR);
626 return (dist < 0) ? 0 : dist;
627}

659//////////////////////////////////////////////////////////////////////////

661// Return cached surface area
662
663G4double G4EllipticalTube::GetCachedSurfaceArea() const
664{
665 G4ThreadLocalStatic G4double cached_Dx = 0;
666 G4ThreadLocalStatic G4double cached_Dy = 0;
667 G4ThreadLocalStatic G4double cached_Dz = 0;
668 G4ThreadLocalStatic G4double cached_area = 0;
669 if (cached_Dx != fDx || cached_Dy != fDy || cached_Dz != fDz)
670 {
671 cached_Dx = fDx;
672 cached_Dy = fDy;
673 cached_Dz = fDz;

676 return cached_area;

713//////////////////////////////////////////////////////////////////////////
714//
715// Pick up a random point on the surface
716
717G4ThreeVector G4EllipticalTube::GetPointOnSurface() const
718{
719 // Select surface (0 - base at -Z, 1 - base at +Z, 2 - lateral surface)
720 //
721 G4double sbase = pi * fDx * fDy;
722 G4double ssurf = GetCachedSurfaceArea();
723 G4double select = ssurf * G4UniformRand();
724
725 G4int k = 0;
726 if (select > sbase) k = 1;
727 if (select > 2. * sbase) k = 2;
728
729 // Pick random point on selected surface (rejection sampling)
730 //
731 G4ThreeVector p;
732 switch (k) {
733 case 0: // base at -Z
734 {
735 G4TwoVector rho = G4RandomPointInEllipse(fDx, fDy);
736 p.set(rho.x(), rho.y(), -fDz);
737 break;
738 }
739 case 1: // base at +Z
740 {
741 G4TwoVector rho = G4RandomPointInEllipse(fDx, fDy);
742 p.set(rho.x(), rho.y(), fDz);
743 break;
744 }
745 case 2: // lateral surface
746 {
747 G4TwoVector rho = G4RandomPointOnEllipse(fDx, fDy);
748 p.set(rho.x(), rho.y(), (2. * G4UniformRand() - 1.) * fDz);
749 break;
750 }
751 }
752 return p;
753}
