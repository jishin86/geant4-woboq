71G4TwistedTubs::G4TwistedTubs(const G4String &pname,
72 G4double twistedangle,
73 G4double endinnerrad,
74 G4double endouterrad,
75 G4double halfzlen,
76 G4double dphi)
77 : G4VSolid(pname), fDPhi(dphi),
78 fLowerEndcap(0), fUpperEndcap(0), fLatterTwisted(0),
79 fFormerTwisted(0), fInnerHype(0), fOuterHype(0),
80 fCubicVolume(0.), fSurfaceArea(0.),
81 fRebuildPolyhedron(false), fpPolyhedron(0)
82{
83 if (endinnerrad < DBL_MIN)
84 {
85 G4Exception("G4TwistedTubs::G4TwistedTubs()", "GeomSolids0002",
86 FatalErrorInArgument, "Invalid end-inner-radius!");
87 }
88
89 G4double sinhalftwist = std::sin(0.5 * twistedangle);
90
91 G4double endinnerradX = endinnerrad * sinhalftwist;
92 G4double innerrad = std::sqrt( endinnerrad * endinnerrad
93 - endinnerradX * endinnerradX );
94
95 G4double endouterradX = endouterrad * sinhalftwist;
96 G4double outerrad = std::sqrt( endouterrad * endouterrad
97 - endouterradX * endouterradX );
98
99 // temporary treatment!!
100 SetFields(twistedangle, innerrad, outerrad, -halfzlen, halfzlen);
101 CreateSurfaces();
102}

326//////////////////////////////////////////////////////////////////////////
327//
328// Get bounding box
329
330void G4TwistedTubs::BoundingLimits(G4ThreeVector& pMin,
331 G4ThreeVector& pMax) const
332{
333 G4double maxEndOuterRad = (fEndOuterRadius[0] > fEndOuterRadius[1] ?
334 fEndOuterRadius[0] : fEndOuterRadius[1]);
335 pMin.set(-maxEndOuterRad,-maxEndOuterRad,-fZHalfLength);
336 pMax.set( maxEndOuterRad, maxEndOuterRad, fZHalfLength);
337
338 // Check correctness of the bounding box
339 //
340 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
341 {
342 std::ostringstream message;
343 message << "Bad bounding box (min >= max) for solid: "
344 << GetName() << " !"
345 << "\npMin = " << pMin
346 << "\npMax = " << pMax;
347 G4Exception("G4TwistedTubs::BoundingLimits()", "GeomMgt0001",
348 JustWarning, message);
349 DumpInfo();
350 }
351}

374//=====================================================================
375//* Inside ------------------------------------------------------------
376
377EInside G4TwistedTubs::Inside(const G4ThreeVector& p) const
378{
379
380 const G4double halftol
381 = 0.5 * G4GeometryTolerance::GetInstance()->GetRadialTolerance();
382 // static G4int timerid = -1;
383 // G4Timer timer(timerid, "G4TwistedTubs", "Inside");
384 // timer.Start();
385
386 G4ThreeVector *tmpp;
387 EInside *tmpinside;
388 if (fLastInside.p == p)
389 {
390 return fLastInside.inside;
391 }
392 else
393 {
394 tmpp = const_cast<G4ThreeVector*>(&(fLastInside.p));
395 tmpinside = const_cast<EInside*>(&(fLastInside.inside));
396 tmpp->set(p.x(), p.y(), p.z());
397 }
398
399 EInside outerhypearea = ((G4TwistTubsHypeSide *)fOuterHype)->Inside(p);
400 G4double innerhyperho = ((G4TwistTubsHypeSide *)fInnerHype)->GetRhoAtPZ(p);
401 G4double distanceToOut = p.getRho() - innerhyperho; // +ve: inside
402
403 if ((outerhypearea == kOutside) || (distanceToOut < -halftol))
404 {
405 *tmpinside = kOutside;
406 }
407 else if (outerhypearea == kSurface)
408 {
409 *tmpinside = kSurface;
410 }
411 else
412 {
413 if (distanceToOut <= halftol)
414 {
415 *tmpinside = kSurface;
416 }
417 else
418 {
419 *tmpinside = kInside;
420 }
421 }
422
423 return fLastInside.inside;
424}

426//=====================================================================
427//* SurfaceNormal -----------------------------------------------------
428
429G4ThreeVector G4TwistedTubs::SurfaceNormal(const G4ThreeVector& p) const
430{
431 //
432 // return the normal unit vector to the Hyperbolical Surface at a point
433 // p on (or nearly on) the surface
434 //
435 // Which of the three or four surfaces are we closest to?
436 //
437
438 if (fLastNormal.p == p)
439 {
440 return fLastNormal.vec;
441 }
442 G4ThreeVector *tmpp =
443 const_cast<G4ThreeVector*>(&(fLastNormal.p));
444 G4ThreeVector *tmpnormal =
445 const_cast<G4ThreeVector*>(&(fLastNormal.vec));
446 G4VTwistSurface **tmpsurface =
447 const_cast<G4VTwistSurface**>(fLastNormal.surface);
448 tmpp->set(p.x(), p.y(), p.z());
449
450 G4double distance = kInfinity;
451
452 G4VTwistSurface *surfaces[6];
453 surfaces[0] = fLatterTwisted;
454 surfaces[1] = fFormerTwisted;
455 surfaces[2] = fInnerHype;
456 surfaces[3] = fOuterHype;
457 surfaces[4] = fLowerEndcap;
458 surfaces[5] = fUpperEndcap;
459
460 G4ThreeVector xx;
461 G4ThreeVector bestxx;
462 G4int i;
463 G4int besti = -1;
464 for (i=0; i< 6; i++)
465 {
466 G4double tmpdistance = surfaces[i]->DistanceTo(p, xx);
467 if (tmpdistance < distance)
468 {
469 distance = tmpdistance;
470 bestxx = xx;
471 besti = i;
472 }
473 }
474
475 tmpsurface[0] = surfaces[besti];
476 *tmpnormal = tmpsurface[0]->GetNormal(bestxx, true);
477
478 return fLastNormal.vec;
479}

481//=====================================================================
482//* DistanceToIn (p, v) -----------------------------------------------
483
484G4double G4TwistedTubs::DistanceToIn (const G4ThreeVector& p,
485 const G4ThreeVector& v ) const
486{
487
488 // DistanceToIn (p, v):
489 // Calculate distance to surface of shape from `outside'
490 // along with the v, allowing for tolerance.
491 // The function returns kInfinity if no intersection or
492 // just grazing within tolerance.
493
494 //
495 // checking last value
496 //
497
498 G4ThreeVector *tmpp;
499 G4ThreeVector *tmpv;
500 G4double *tmpdist;
501 if ((fLastDistanceToInWithV.p == p) && (fLastDistanceToInWithV.vec == v))
502 {
503 return fLastDistanceToIn.value;
504 }
505 else
506 {
507 tmpp = const_cast<G4ThreeVector*>(&(fLastDistanceToInWithV.p));
508 tmpv = const_cast<G4ThreeVector*>(&(fLastDistanceToInWithV.vec));
509 tmpdist = const_cast<G4double*>(&(fLastDistanceToInWithV.value));
510 tmpp->set(p.x(), p.y(), p.z());
511 tmpv->set(v.x(), v.y(), v.z());
512 }
513
514 //
515 // Calculate DistanceToIn(p,v)
516 //
517
518 EInside currentside = Inside(p);
519
520 if (currentside == kInside)
521 {
522 }
523 else
524 {
525 if (currentside == kSurface)
526 {
527 // particle is just on a boundary.
528 // If the particle is entering to the volume, return 0.
529 //
530 G4ThreeVector normal = SurfaceNormal(p);
531 if (normal*v < 0)
532 {
533 *tmpdist = 0;
534 return fLastDistanceToInWithV.value;
535 }
536 }
537 }
538
539 // now, we can take smallest positive distance.
540
541 // Initialize
542 //
543 G4double distance = kInfinity;
544
545 // find intersections and choose nearest one.
546 //
547 G4VTwistSurface *surfaces[6];
548 surfaces[0] = fLowerEndcap;
549 surfaces[1] = fUpperEndcap;
550 surfaces[2] = fLatterTwisted;
551 surfaces[3] = fFormerTwisted;
552 surfaces[4] = fInnerHype;
553 surfaces[5] = fOuterHype;
554
555 G4ThreeVector xx;
556 G4ThreeVector bestxx;
557 G4int i;
558 for (i=0; i< 6; i++)
559 {
560 G4double tmpdistance = surfaces[i]->DistanceToIn(p, v, xx);
561 if (tmpdistance < distance)
562 {
563 distance = tmpdistance;
564 bestxx = xx;
565 }
566 }
567 *tmpdist = distance;
568
569 return fLastDistanceToInWithV.value;
570}

572//=====================================================================
573//* DistanceToIn (p) --------------------------------------------------
574
575G4double G4TwistedTubs::DistanceToIn (const G4ThreeVector& p) const
576{
577 // DistanceToIn(p):
578 // Calculate distance to surface of shape from `outside',
579 // allowing for tolerance
580
581 //
582 // checking last value
583 //
584
585 G4ThreeVector *tmpp;
586 G4double *tmpdist;
587 if (fLastDistanceToIn.p == p)
588 {
589 return fLastDistanceToIn.value;
590 }
591 else
592 {
593 tmpp = const_cast<G4ThreeVector*>(&(fLastDistanceToIn.p));
594 tmpdist = const_cast<G4double*>(&(fLastDistanceToIn.value));
595 tmpp->set(p.x(), p.y(), p.z());
596 }
597
598 //
599 // Calculate DistanceToIn(p)
600 //
601
602 EInside currentside = Inside(p);
603
604 switch (currentside)
605 {
606 case (kInside) :
607 {}
608 case (kSurface) :
609 {
610 *tmpdist = 0.;
611 return fLastDistanceToIn.value;
612 }
613 case (kOutside) :
614 {
615 // Initialize
616 G4double distance = kInfinity;
617
618 // find intersections and choose nearest one.
619 G4VTwistSurface *surfaces[6];
620 surfaces[0] = fLowerEndcap;
621 surfaces[1] = fUpperEndcap;
622 surfaces[2] = fLatterTwisted;
623 surfaces[3] = fFormerTwisted;
624 surfaces[4] = fInnerHype;
625 surfaces[5] = fOuterHype;
626
627 G4int i;
628 G4ThreeVector xx;
629 G4ThreeVector bestxx;
630 for (i=0; i< 6; i++)
631 {
632 G4double tmpdistance = surfaces[i]->DistanceTo(p, xx);
633 if (tmpdistance < distance)
634 {
635 distance = tmpdistance;
636 bestxx = xx;
637 }
638 }
639 *tmpdist = distance;
640 return fLastDistanceToIn.value;
641 }
642 default :
643 {
644 G4Exception("G4TwistedTubs::DistanceToIn(p)", "GeomSolids0003",
645 FatalException, "Unknown point location!");
646 }
647 } // switch end
648
649 return kInfinity;
650}

652//=====================================================================
653//* DistanceToOut (p, v) ----------------------------------------------
654
655G4double G4TwistedTubs::DistanceToOut( const G4ThreeVector& p,
656 const G4ThreeVector& v,
657 const G4bool calcNorm,
658 G4bool *validNorm,
659 G4ThreeVector *norm ) const
660{
661 // DistanceToOut (p, v):
662 // Calculate distance to surface of shape from `inside'
663 // along with the v, allowing for tolerance.
664 // The function returns kInfinity if no intersection or
665 // just grazing within tolerance.
666
667 //
668 // checking last value
669 //
670
671 G4ThreeVector *tmpp;
672 G4ThreeVector *tmpv;
673 G4double *tmpdist;
674 if ((fLastDistanceToOutWithV.p == p) && (fLastDistanceToOutWithV.vec == v) )
675 {
676 return fLastDistanceToOutWithV.value;
677 }
678 else
679 {
680 tmpp = const_cast<G4ThreeVector*>(&(fLastDistanceToOutWithV.p));
681 tmpv = const_cast<G4ThreeVector*>(&(fLastDistanceToOutWithV.vec));
682 tmpdist = const_cast<G4double*>(&(fLastDistanceToOutWithV.value));
683 tmpp->set(p.x(), p.y(), p.z());
684 tmpv->set(v.x(), v.y(), v.z());
685 }
686
687 //
688 // Calculate DistanceToOut(p,v)
689 //
690
691 EInside currentside = Inside(p);
692
693 if (currentside == kOutside)
694 {
695 }
696 else
697 {
698 if (currentside == kSurface)
699 {
700 // particle is just on a boundary.
701 // If the particle is exiting from the volume, return 0.
702 //
703 G4ThreeVector normal = SurfaceNormal(p);
704 G4VTwistSurface *blockedsurface = fLastNormal.surface[0];
705 if (normal*v > 0)
706 {
707 if (calcNorm)
708 {
709 *norm = (blockedsurface->GetNormal(p, true));
710 *validNorm = blockedsurface->IsValidNorm();
711 }
712 *tmpdist = 0.;
713 return fLastDistanceToOutWithV.value;
714 }
715 }
716 }
717
718 // now, we can take smallest positive distance.
719
720 // Initialize
721 //
722 G4double distance = kInfinity;
723
724 // find intersections and choose nearest one.
725 //
726 G4VTwistSurface *surfaces[6];
727 surfaces[0] = fLatterTwisted;
728 surfaces[1] = fFormerTwisted;
729 surfaces[2] = fInnerHype;
730 surfaces[3] = fOuterHype;
731 surfaces[4] = fLowerEndcap;
732 surfaces[5] = fUpperEndcap;
733
734 G4int i;
735 G4int besti = -1;
736 G4ThreeVector xx;
737 G4ThreeVector bestxx;
738 for (i=0; i< 6; i++)
739 {
740 G4double tmpdistance = surfaces[i]->DistanceToOut(p, v, xx);
741 if (tmpdistance < distance)
742 {
743 distance = tmpdistance;
744 bestxx = xx;
745 besti = i;
746 }
747 }
748
749 if (calcNorm)
750 {
751 if (besti != -1)
752 {
753 *norm = (surfaces[besti]->GetNormal(p, true));
754 *validNorm = surfaces[besti]->IsValidNorm();
755 }
756 }
757
758 *tmpdist = distance;
759
760 return fLastDistanceToOutWithV.value;
761}

764//=====================================================================
765//* DistanceToOut (p) ----------------------------------------------
766
767G4double G4TwistedTubs::DistanceToOut( const G4ThreeVector& p ) const
768{
769 // DistanceToOut(p):
770 // Calculate distance to surface of shape from `inside',
771 // allowing for tolerance
772
773 //
774 // checking last value
775 //
776
777 G4ThreeVector *tmpp;
778 G4double *tmpdist;
779 if (fLastDistanceToOut.p == p)
780 {
781 return fLastDistanceToOut.value;
782 }
783 else
784 {
785 tmpp = const_cast<G4ThreeVector*>(&(fLastDistanceToOut.p));
786 tmpdist = const_cast<G4double*>(&(fLastDistanceToOut.value));
787 tmpp->set(p.x(), p.y(), p.z());
788 }
789
790 //
791 // Calculate DistanceToOut(p)
792 //
793
794 EInside currentside = Inside(p);
795
796 switch (currentside)
797 {
798 case (kOutside) :
799 {
800 }
801 case (kSurface) :
802 {
803 *tmpdist = 0.;
804 return fLastDistanceToOut.value;
805 }
806 case (kInside) :
807 {
808 // Initialize
809 G4double distance = kInfinity;
810
811 // find intersections and choose nearest one.
812 G4VTwistSurface *surfaces[6];
813 surfaces[0] = fLatterTwisted;
814 surfaces[1] = fFormerTwisted;
815 surfaces[2] = fInnerHype;
816 surfaces[3] = fOuterHype;
817 surfaces[4] = fLowerEndcap;
818 surfaces[5] = fUpperEndcap;
819
820 G4int i;
821 G4ThreeVector xx;
822 G4ThreeVector bestxx;
823 for (i=0; i< 6; i++)
824 {
825 G4double tmpdistance = surfaces[i]->DistanceTo(p, xx);
826 if (tmpdistance < distance)
827 {
828 distance = tmpdistance;
829 bestxx = xx;
830 }
831 }
832 *tmpdist = distance;
833
834 return fLastDistanceToOut.value;
835 }
836 default :
837 {
838 G4Exception("G4TwistedTubs::DistanceToOut(p)", "GeomSolids0003",
839 FatalException, "Unknown point location!");
840 }
841 } // switch end
842
843 return 0;
844}

901//=====================================================================
902//* CreatePolyhedron --------------------------------------------------
903
904G4Polyhedron* G4TwistedTubs::CreatePolyhedron () const
905{
906 // number of meshes
907 //
908 G4double absPhiTwist = std::abs(fPhiTwist);
909 G4double dA = std::max(fDPhi,absPhiTwist);
910 const G4int k =
911 G4int(G4Polyhedron::GetNumberOfRotationSteps() * dA / twopi) + 2;
912 const G4int n =
913 G4int(G4Polyhedron::GetNumberOfRotationSteps() * absPhiTwist / twopi) + 2;
914
915 const G4int nnodes = 4*(k-1)*(n-2) + 2*k*k ;
916 const G4int nfaces = 4*(k-1)*(n-1) + 2*(k-1)*(k-1) ;
917
918 G4Polyhedron *ph=new G4Polyhedron;
919 typedef G4double G4double3[3];
920 typedef G4int G4int4[4];
921 G4double3* xyz = new G4double3[nnodes]; // number of nodes
922 G4int4* faces = new G4int4[nfaces] ; // number of faces
923 fLowerEndcap->GetFacets(k,k,xyz,faces,0) ;
924 fUpperEndcap->GetFacets(k,k,xyz,faces,1) ;
925 fInnerHype->GetFacets(k,n,xyz,faces,2) ;
926 fFormerTwisted->GetFacets(k,n,xyz,faces,3) ;
927 fOuterHype->GetFacets(k,n,xyz,faces,4) ;
928 fLatterTwisted->GetFacets(k,n,xyz,faces,5) ;
929
930 ph->createPolyhedron(nnodes,nfaces,xyz,faces);
931
932 delete[] xyz;
933 delete[] faces;
934
935 return ph;
936}

959
960void G4TwistedTubs::CreateSurfaces()
961{
962 // create 6 surfaces of TwistedTub
963
964 G4ThreeVector x0(0, 0, fEndZ[0]);
965 G4ThreeVector n (0, 0, -1);
966
967 fLowerEndcap = new G4TwistTubsFlatSide("LowerEndcap",
968 fEndInnerRadius, fEndOuterRadius,
969 fDPhi, fEndPhi, fEndZ, -1) ;
970
971 fUpperEndcap = new G4TwistTubsFlatSide("UpperEndcap",
972 fEndInnerRadius, fEndOuterRadius,
973 fDPhi, fEndPhi, fEndZ, 1) ;
974
975 G4RotationMatrix rotHalfDPhi;
976 rotHalfDPhi.rotateZ(0.5*fDPhi);
977
978 fLatterTwisted = new G4TwistTubsSide("LatterTwisted",
979 fEndInnerRadius, fEndOuterRadius,
980 fDPhi, fEndPhi, fEndZ,
981 fInnerRadius, fOuterRadius, fKappa,
982 1 ) ;
983 fFormerTwisted = new G4TwistTubsSide("FormerTwisted",
984 fEndInnerRadius, fEndOuterRadius,
985 fDPhi, fEndPhi, fEndZ,
986 fInnerRadius, fOuterRadius, fKappa,
987 -1 ) ;
988
989 fInnerHype = new G4TwistTubsHypeSide("InnerHype",
990 fEndInnerRadius, fEndOuterRadius,
991 fDPhi, fEndPhi, fEndZ,
992 fInnerRadius, fOuterRadius,fKappa,
993 fTanInnerStereo, fTanOuterStereo, -1) ;
994 fOuterHype = new G4TwistTubsHypeSide("OuterHype",
995 fEndInnerRadius, fEndOuterRadius,
996 fDPhi, fEndPhi, fEndZ,
997 fInnerRadius, fOuterRadius,fKappa,
998 fTanInnerStereo, fTanOuterStereo, 1) ;
999
1000
1001 // set neighbour surfaces
1002 //
1003 fLowerEndcap->SetNeighbours(fInnerHype, fLatterTwisted,
1004 fOuterHype, fFormerTwisted);
1005 fUpperEndcap->SetNeighbours(fInnerHype, fLatterTwisted,
1006 fOuterHype, fFormerTwisted);
1007 fLatterTwisted->SetNeighbours(fInnerHype, fLowerEndcap,
1008 fOuterHype, fUpperEndcap);
1009 fFormerTwisted->SetNeighbours(fInnerHype, fLowerEndcap,
1010 fOuterHype, fUpperEndcap);
1011 fInnerHype->SetNeighbours(fLatterTwisted, fLowerEndcap,
1012 fFormerTwisted, fUpperEndcap);
1013 fOuterHype->SetNeighbours(fLatterTwisted, fLowerEndcap,
1014 fFormerTwisted, fUpperEndcap);

1058G4ThreeVector G4TwistedTubs::GetPointOnSurface() const
1059{

1061 G4double z = G4RandFlat::shoot(fEndZ[0],fEndZ[1]);
1062 G4double phi , phimin, phimax ;
1063 G4double x , xmin, xmax ;
1064 G4double r , rmin, rmax ;
1065
1066 G4double a1 = fOuterHype->GetSurfaceArea() ;
1067 G4double a2 = fInnerHype->GetSurfaceArea() ;
1068 G4double a3 = fLatterTwisted->GetSurfaceArea() ;
1069 G4double a4 = fFormerTwisted->GetSurfaceArea() ;
1070 G4double a5 = fLowerEndcap->GetSurfaceArea() ;
1071 G4double a6 = fUpperEndcap->GetSurfaceArea() ;
1072
1073 G4double chose = G4RandFlat::shoot(0.,a1 + a2 + a3 + a4 + a5 + a6) ;
1074
1075 if(chose < a1)
1076 {
1077
1078 phimin = fOuterHype->GetBoundaryMin(z) ;
1079 phimax = fOuterHype->GetBoundaryMax(z) ;
1080 phi = G4RandFlat::shoot(phimin,phimax) ;
1081
1082 return fOuterHype->SurfacePoint(phi,z,true) ;
1083
1084 }
1085 else if ( (chose >= a1) && (chose < a1 + a2 ) )
1086 {
1087
1088 phimin = fInnerHype->GetBoundaryMin(z) ;
1089 phimax = fInnerHype->GetBoundaryMax(z) ;
1090 phi = G4RandFlat::shoot(phimin,phimax) ;
1091
1092 return fInnerHype->SurfacePoint(phi,z,true) ;
1093
1094 }
1095 else if ( (chose >= a1 + a2 ) && (chose < a1 + a2 + a3 ) )
1096 {
1097
1098 xmin = fLatterTwisted->GetBoundaryMin(z) ;
1099 xmax = fLatterTwisted->GetBoundaryMax(z) ;
1100 x = G4RandFlat::shoot(xmin,xmax) ;
1101
1102 return fLatterTwisted->SurfacePoint(x,z,true) ;
1103
1104 }
1105 else if ( (chose >= a1 + a2 + a3 ) && (chose < a1 + a2 + a3 + a4 ) )
1106 {
1107
1108 xmin = fFormerTwisted->GetBoundaryMin(z) ;
1109 xmax = fFormerTwisted->GetBoundaryMax(z) ;
1110 x = G4RandFlat::shoot(xmin,xmax) ;
1111
1112 return fFormerTwisted->SurfacePoint(x,z,true) ;
1113 }
1114 else if( (chose >= a1 + a2 + a3 + a4 )&&(chose < a1 + a2 + a3 + a4 + a5 ) )
1115 {
1116 rmin = GetEndInnerRadius(0) ;
1117 rmax = GetEndOuterRadius(0) ;
1118 r = std::sqrt(G4RandFlat::shoot()*(sqr(rmax)-sqr(rmin))+sqr(rmin));
1119
1120 phimin = fLowerEndcap->GetBoundaryMin(r) ;
1121 phimax = fLowerEndcap->GetBoundaryMax(r) ;
1122 phi = G4RandFlat::shoot(phimin,phimax) ;
1123
1124 return fLowerEndcap->SurfacePoint(phi,r,true) ;
1125 }
1126 else
1127 {
1128 rmin = GetEndInnerRadius(1) ;
1129 rmax = GetEndOuterRadius(1) ;
1130 r = rmin + (rmax-rmin)*std::sqrt(G4RandFlat::shoot());
1131
1132 phimin = fUpperEndcap->GetBoundaryMin(r) ;
1133 phimax = fUpperEndcap->GetBoundaryMax(r) ;
1134 phi = G4RandFlat::shoot(phimin,phimax) ;
