66//
67// Constructor
68//
69// Values for r1,z1 and r2,z2 should be specified in clockwise
70// order in (r,z).
71//
72G4PolyconeSide::G4PolyconeSide( const G4PolyconeSideRZ *prevRZ,
73 const G4PolyconeSideRZ *tail,
74 const G4PolyconeSideRZ *head,
75 const G4PolyconeSideRZ *nextRZ,
76 G4double thePhiStart,
77 G4double theDeltaPhi,
78 G4bool thePhiIsOpen,
79 G4bool isAllBehind )
80 : ncorners(0), corners(0)
81{
82
83 instanceID = subInstanceManager.CreateSubInstance();
84
85 kCarTolerance = G4GeometryTolerance::GetInstance()->GetSurfaceTolerance();
86 fSurfaceArea = 0.0;
87 G4MT_pcphi.first = G4ThreeVector(0,0,0);
88 G4MT_pcphi.second= 0.0;
89
90 //
91 // Record values
92 //
93 r[0] = tail->r; z[0] = tail->z;
94 r[1] = head->r; z[1] = head->z;
95
96 phiIsOpen = thePhiIsOpen;
97 if (phiIsOpen)
98 {
99 deltaPhi = theDeltaPhi;
100 startPhi = thePhiStart;
101
102 //
103 // Set phi values to our conventions
104 //
105 while (deltaPhi < 0.0) // Loop checking, 13.08.2015, G.Cosmo
106 deltaPhi += twopi;
107 while (startPhi < 0.0) // Loop checking, 13.08.2015, G.Cosmo
108 startPhi += twopi;
109
110 //
111 // Calculate corner coordinates
112 //
113 ncorners = 4;
114 corners = new G4ThreeVector[ncorners];
115
116 corners[0] = G4ThreeVector( tail->r*std::cos(startPhi),
117 tail->r*std::sin(startPhi), tail->z );
118 corners[1] = G4ThreeVector( head->r*std::cos(startPhi),
119 head->r*std::sin(startPhi), head->z );
120 corners[2] = G4ThreeVector( tail->r*std::cos(startPhi+deltaPhi),
121 tail->r*std::sin(startPhi+deltaPhi), tail->z );
122 corners[3] = G4ThreeVector( head->r*std::cos(startPhi+deltaPhi),
123 head->r*std::sin(startPhi+deltaPhi), head->z );
124 }
125 else
126 {
127 deltaPhi = twopi;
128 startPhi = 0.0;
129 }
130
131 allBehind = isAllBehind;
132
133 //
134 // Make our intersecting cone
135 //
136 cone = new G4IntersectingCone( r, z );
137
138 //
139 // Calculate vectors in r,z space
140 //
141 rS = r[1]-r[0]; zS = z[1]-z[0];
142 length = std::sqrt( rS*rS + zS*zS);
143 rS /= length; zS /= length;
144
145 rNorm = +zS;
146 zNorm = -rS;
147
148 G4double lAdj;
149
150 prevRS = r[0]-prevRZ->r;
151 prevZS = z[0]-prevRZ->z;
152 lAdj = std::sqrt( prevRS*prevRS + prevZS*prevZS );
153 prevRS /= lAdj;
154 prevZS /= lAdj;
155
156 rNormEdge[0] = rNorm + prevZS;
157 zNormEdge[0] = zNorm - prevRS;
158 lAdj = std::sqrt( rNormEdge[0]*rNormEdge[0] + zNormEdge[0]*zNormEdge[0] );
159 rNormEdge[0] /= lAdj;
160 zNormEdge[0] /= lAdj;
161
162 nextRS = nextRZ->r-r[1];
163 nextZS = nextRZ->z-z[1];
164 lAdj = std::sqrt( nextRS*nextRS + nextZS*nextZS );
165 nextRS /= lAdj;
166 nextZS /= lAdj;
167
168 rNormEdge[1] = rNorm + nextZS;
169 zNormEdge[1] = zNorm - nextRS;
170 lAdj = std::sqrt( rNormEdge[1]*rNormEdge[1] + zNormEdge[1]*zNormEdge[1] );
171 rNormEdge[1] /= lAdj;
172 zNormEdge[1] /= lAdj;
173}

231//
232// CopyStuff
233//
234void G4PolyconeSide::CopyStuff( const G4PolyconeSide &source )
235{
236 r[0] = source.r[0];
237 r[1] = source.r[1];
238 z[0] = source.z[0];
239 z[1] = source.z[1];
240
241 startPhi = source.startPhi;
242 deltaPhi = source.deltaPhi;
243 phiIsOpen = source.phiIsOpen;
244 allBehind = source.allBehind;
245
246 kCarTolerance = source.kCarTolerance;
247 fSurfaceArea = source.fSurfaceArea;
248
249 cone = new G4IntersectingCone( *source.cone );
250
251 rNorm = source.rNorm;
252 zNorm = source.zNorm;
253 rS = source.rS;
254 zS = source.zS;
255 length = source.length;
256 prevRS = source.prevRS;
257 prevZS = source.prevZS;
258 nextRS = source.nextRS;
259 nextZS = source.nextZS;
260
261 rNormEdge[0] = source.rNormEdge[0];
262 rNormEdge[1] = source.rNormEdge[1];
263 zNormEdge[0] = source.zNormEdge[0];
264 zNormEdge[1] = source.zNormEdge[1];
265
266 if (phiIsOpen)
267 {
268 ncorners = 4;
269 corners = new G4ThreeVector[ncorners];
270
271 corners[0] = source.corners[0];
272 corners[1] = source.corners[1];
273 corners[2] = source.corners[2];
274 corners[3] = source.corners[3];
275 }
276}

279//
280// Intersect
281//
282G4bool G4PolyconeSide::Intersect( const G4ThreeVector &p,
283 const G4ThreeVector &v,
284 G4bool outgoing,
285 G4double surfTolerance,
286 G4double &distance,
287 G4double &distFromSurface,
288 G4ThreeVector &normal,
289 G4bool &isAllBehind )
290{
291 G4double s1, s2;
292 G4double normSign = outgoing ? +1 : -1;
293
294 isAllBehind = allBehind;
295
296 //
297 // Check for two possible intersections
298 //
299 G4int nside = cone->LineHitsCone( p, v, &s1, &s2 );
300 if (nside == 0) return false;
301
302 //
303 // Check the first side first, since it is (supposed to be) closest
304 //
305 G4ThreeVector hit = p + s1*v;
306
307 if (PointOnCone( hit, normSign, p, v, normal ))
308 {
309 //
310 // Good intersection! What about the normal?
311 //
312 if (normSign*v.dot(normal) > 0)
313 {
314 //
315 // We have a valid intersection, but it could very easily
316 // be behind the point. To decide if we tolerate this,
317 // we have to see if the point p is on the surface near
318 // the intersecting point.
319 //
320 // What does it mean exactly for the point p to be "near"
321 // the intersection? It means that if we draw a line from
322 // p to the hit, the line remains entirely within the
323 // tolerance bounds of the cone. To test this, we can
324 // ask if the normal is correct near p.
325 //
326 G4double pr = p.perp();
327 if (pr < DBL_MIN) pr = DBL_MIN;
328 G4ThreeVector pNormal( rNorm*p.x()/pr, rNorm*p.y()/pr, zNorm );
329 if (normSign*v.dot(pNormal) > 0)
330 {
331 //
332 // p and intersection in same hemisphere
333 //
334 G4double distOutside2;
335 distFromSurface = -normSign*DistanceAway( p, false, distOutside2 );
336 if (distOutside2 < surfTolerance*surfTolerance)
337 {
338 if (distFromSurface > -surfTolerance)
339 {
340 //
341 // We are just inside or away from the
342 // surface. Accept *any* value of distance.
343 //
344 distance = s1;
345 return true;
346 }
347 }
348 }
349 else
350 distFromSurface = s1;
351
352 //
353 // Accept positive distances
354 //
355 if (s1 > 0)
356 {
357 distance = s1;
358 return true;
359 }
360 }
361 }
362
363 if (nside==1) return false;
364
365 //
366 // Well, try the second hit
367 //
368 hit = p + s2*v;
369
370 if (PointOnCone( hit, normSign, p, v, normal ))
371 {
372 //
373 // Good intersection! What about the normal?
374 //
375 if (normSign*v.dot(normal) > 0)
376 {
377 G4double pr = p.perp();
378 if (pr < DBL_MIN) pr = DBL_MIN;
379 G4ThreeVector pNormal( rNorm*p.x()/pr, rNorm*p.y()/pr, zNorm );
380 if (normSign*v.dot(pNormal) > 0)
381 {
382 G4double distOutside2;
383 distFromSurface = -normSign*DistanceAway( p, false, distOutside2 );
384 if (distOutside2 < surfTolerance*surfTolerance)
385 {
386 if (distFromSurface > -surfTolerance)
387 {
388 distance = s2;
389 return true;
390 }
391 }
392 }
393 else
394 distFromSurface = s2;
395
396 if (s2 > 0)
397 {
398 distance = s2;
399 return true;
400 }
401 }
402 }
403
404 //
405 // Better luck next time
406 //
407 return false;
408}

411//
412// Distance
413//
414G4double G4PolyconeSide::Distance( const G4ThreeVector &p, G4bool outgoing )
415{
416 G4double normSign = outgoing ? -1 : +1;
417 G4double distFrom, distOut2;
418
419 //
420 // We have two tries for each hemisphere. Try the closest first.
421 //
422 distFrom = normSign*DistanceAway( p, false, distOut2 );
423 if (distFrom > -0.5*kCarTolerance )
424 {
425 //
426 // Good answer
427 //
428 if (distOut2 > 0)
429 return std::sqrt( distFrom*distFrom + distOut2 );
430 else
431 return std::fabs(distFrom);
432 }
433
434 //
435 // Try second side.
436 //
437 distFrom = normSign*DistanceAway( p, true, distOut2 );
438 if (distFrom > -0.5*kCarTolerance)
439 {
440
441 if (distOut2 > 0)
442 return std::sqrt( distFrom*distFrom + distOut2 );
443 else
444 return std::fabs(distFrom);
445 }
446
447 return kInfinity;
448}

451//
452// Inside
453//
454EInside G4PolyconeSide::Inside( const G4ThreeVector &p,
455 G4double tolerance,
456 G4double *bestDistance )
457{
458 G4double distFrom, distOut2, dist2;
459 G4double edgeRZnorm;
460
461 distFrom = DistanceAway( p, distOut2, &edgeRZnorm );
462 dist2 = distFrom*distFrom + distOut2;
463
464 *bestDistance = std::sqrt( dist2);
465
466 // Okay then, inside or out?
467 //
468 if ( (std::fabs(edgeRZnorm) < tolerance)
469 && (distOut2< tolerance*tolerance) )
470 return kSurface;
471 else if (edgeRZnorm < 0)
472 return kInside;
473 else
474 return kOutside;
475}

478//
479// Normal
480//
481G4ThreeVector G4PolyconeSide::Normal( const G4ThreeVector &p,
482 G4double *bestDistance )
483{
484 if (p == G4ThreeVector(0.,0.,0.)) { return p; }
485
486 G4double dFrom, dOut2;
487
488 dFrom = DistanceAway( p, false, dOut2 );
489
490 *bestDistance = std::sqrt( dFrom*dFrom + dOut2 );
491
492 G4double rds = p.perp();
493 if (rds!=0.) { return G4ThreeVector(rNorm*p.x()/rds,rNorm*p.y()/rds,zNorm); }
494 return G4ThreeVector( 0.,0., zNorm ).unit();
495}

498//
499// Extent
500//
501G4double G4PolyconeSide::Extent( const G4ThreeVector axis )
502{
503 if (axis.perp2() < DBL_MIN)
504 {
505 //
506 // Special case
507 //
508 return axis.z() < 0 ? -cone->ZLo() : cone->ZHi();
509 }
510
511 //
512 // Is the axis pointing inside our phi gap?
513 //
514 if (phiIsOpen)
515 {
516 G4double phi = GetPhi(axis);
517 while( phi < startPhi ) // Loop checking, 13.08.2015, G.Cosmo
518 phi += twopi;
519
520 if (phi > deltaPhi+startPhi)
521 {
522 //
523 // Yeah, looks so. Make four three vectors defining the phi
524 // opening
525 //
526 G4double cosP = std::cos(startPhi), sinP = std::sin(startPhi);
527 G4ThreeVector a( r[0]*cosP, r[0]*sinP, z[0] );
528 G4ThreeVector b( r[1]*cosP, r[1]*sinP, z[1] );
529 cosP = std::cos(startPhi+deltaPhi); sinP = std::sin(startPhi+deltaPhi);
530 G4ThreeVector c( r[0]*cosP, r[0]*sinP, z[0] );
531 G4ThreeVector d( r[1]*cosP, r[1]*sinP, z[1] );
532
533 G4double ad = axis.dot(a),
534 bd = axis.dot(b),
535 cd = axis.dot(c),
536 dd = axis.dot(d);
537
538 if (bd > ad) ad = bd;
539 if (cd > ad) ad = cd;
540 if (dd > ad) ad = dd;
541
542 return ad;
543 }
544 }
545
546 //
547 // Check either end
548 //
549 G4double aPerp = axis.perp();
550
551 G4double a = aPerp*r[0] + axis.z()*z[0];
552 G4double b = aPerp*r[1] + axis.z()*z[1];
553
554 if (b > a) a = b;
555
556 return a;
557}

560//
561// CalculateExtent
562//
563// See notes in G4VCSGface
564//
565void G4PolyconeSide::CalculateExtent( const EAxis axis,
566 const G4VoxelLimits &voxelLimit,
567 const G4AffineTransform &transform,
568 G4SolidExtentList &extentList )
569{
570 G4ClippablePolygon polygon;
571
572 //
573 // Here we will approximate (ala G4Cons) and divide our conical section
574 // into segments, like G4Polyhedra. When doing so, the radius
575 // is extented far enough such that the segments always lie
576 // just outside the surface of the conical section we are
577 // approximating.
578 //
579
580 //
581 // Choose phi size of our segment(s) based on constants as
582 // defined in meshdefs.hh
583 //
584 G4int numPhi = (G4int)(deltaPhi/kMeshAngleDefault) + 1;
585 if (numPhi < kMinMeshSections)
586 numPhi = kMinMeshSections;
587 else if (numPhi > kMaxMeshSections)
588 numPhi = kMaxMeshSections;
589
590 G4double sigPhi = deltaPhi/numPhi;
591
592 //
593 // Determine radius factor to keep segments outside
594 //
595 G4double rFudge = 1.0/std::cos(0.5*sigPhi);
596
597 //
598 // Decide which radius to use on each end of the side,
599 // and whether a transition mesh is required
600 //
601 // {r0,z0} - Beginning of this side
602 // {r1,z1} - Ending of this side
603 // {r2,z0} - Beginning of transition piece connecting previous
604 // side (and ends at beginning of this side)
605 //
606 // So, order is 2 --> 0 --> 1.
607 // -------
608 //
609 // r2 < 0 indicates that no transition piece is required
610 //
611 G4double r0, r1, r2, z0, z1;
612
613 r2 = -1; // By default: no transition piece
614
615 if (rNorm < -DBL_MIN)
616 {
617 //
618 // This side faces *inward*, and so our mesh has
619 // the same radius
620 //
621 r1 = r[1];
622 z1 = z[1];
623 z0 = z[0];
624 r0 = r[0];
625
626 r2 = -1;
627
628 if (prevZS > DBL_MIN)
629 {
630 //
631 // The previous side is facing outwards
632 //
633 if ( prevRS*zS - prevZS*rS > 0 )
634 {
635 //
636 // Transition was convex: build transition piece
637 //
638 if (r[0] > DBL_MIN) r2 = r[0]*rFudge;
639 }
640 else
641 {
642 //
643 // Transition was concave: short this side
644 //
645 FindLineIntersect( z0, r0, zS, rS,
646 z0, r0*rFudge, prevZS, prevRS*rFudge, z0, r0 );
647 }
648 }
649
650 if ( nextZS > DBL_MIN && (rS*nextZS - zS*nextRS < 0) )
651 {
652 //
653 // The next side is facing outwards, forming a
654 // concave transition: short this side
655 //
656 FindLineIntersect( z1, r1, zS, rS,
657 z1, r1*rFudge, nextZS, nextRS*rFudge, z1, r1 );
658 }
659 }
660 else if (rNorm > DBL_MIN)
661 {
662 //
663 // This side faces *outward* and is given a boost to
664 // it radius
665 //
666 r0 = r[0]*rFudge;
667 z0 = z[0];
668 r1 = r[1]*rFudge;
669 z1 = z[1];
670
671 if (prevZS < -DBL_MIN)
672 {
673 //
674 // The previous side is facing inwards
675 //
676 if ( prevRS*zS - prevZS*rS > 0 )
677 {
678 //
679 // Transition was convex: build transition piece
680 //
681 if (r[0] > DBL_MIN) r2 = r[0];
682 }
683 else
684 {
685 //
686 // Transition was concave: short this side
687 //
688 FindLineIntersect( z0, r0, zS, rS*rFudge,
689 z0, r[0], prevZS, prevRS, z0, r0 );
690 }
691 }
692
693 if ( nextZS < -DBL_MIN && (rS*nextZS - zS*nextRS < 0) )
694 {
695 //
696 // The next side is facing inwards, forming a
697 // concave transition: short this side
698 //
699 FindLineIntersect( z1, r1, zS, rS*rFudge,
700 z1, r[1], nextZS, nextRS, z1, r1 );
701 }
702 }
703 else
704 {
705 //
706 // This side is perpendicular to the z axis (is a disk)
707 //
708 // Whether or not r0 needs a rFudge factor depends
709 // on the normal of the previous edge. Similar with r1
710 // and the next edge. No transition piece is required.
711 //
712 r0 = r[0];
713 r1 = r[1];
714 z0 = z[0];
715 z1 = z[1];
716
717 if (prevZS > DBL_MIN) r0 *= rFudge;
718 if (nextZS > DBL_MIN) r1 *= rFudge;
719 }
720
721 //
722 // Loop
723 //
724 G4double phi = startPhi,
725 cosPhi = std::cos(phi),
726 sinPhi = std::sin(phi);
727
728 G4ThreeVector v0( r0*cosPhi, r0*sinPhi, z0 ),
729 v1( r1*cosPhi, r1*sinPhi, z1 ),
730 v2, w0, w1, w2;
731 transform.ApplyPointTransform( v0 );
732 transform.ApplyPointTransform( v1 );
733
734 if (r2 >= 0)
735 {
736 v2 = G4ThreeVector( r2*cosPhi, r2*sinPhi, z0 );
737 transform.ApplyPointTransform( v2 );
738 }
739
740 do // Loop checking, 13.08.2015, G.Cosmo
741 {
742 phi += sigPhi;
743 if (numPhi == 1) phi = startPhi+deltaPhi; // Try to avoid roundoff
744 cosPhi = std::cos(phi),
745 sinPhi = std::sin(phi);
746
747 w0 = G4ThreeVector( r0*cosPhi, r0*sinPhi, z0 );
748 w1 = G4ThreeVector( r1*cosPhi, r1*sinPhi, z1 );
749 transform.ApplyPointTransform( w0 );
750 transform.ApplyPointTransform( w1 );
751
752 G4ThreeVector deltaV = r0 > r1 ? w0-v0 : w1-v1;
753
754 //
755 // Build polygon, taking special care to keep the vertices
756 // in order
757 //
758 polygon.ClearAllVertices();
759
760 polygon.AddVertexInOrder( v0 );
761 polygon.AddVertexInOrder( v1 );
762 polygon.AddVertexInOrder( w1 );
763 polygon.AddVertexInOrder( w0 );
764
765 //
766 // Get extent
767 //
768 if (polygon.PartialClip( voxelLimit, axis ))
769 {
770 //
771 // Get dot product of normal with target axis
772 //
773 polygon.SetNormal( deltaV.cross(v1-v0).unit() );
774
775 extentList.AddSurface( polygon );
776 }
777
778 if (r2 >= 0)
779 {
780 //
781 // Repeat, for transition piece
782 //
783 w2 = G4ThreeVector( r2*cosPhi, r2*sinPhi, z0 );
784 transform.ApplyPointTransform( w2 );
785
786 polygon.ClearAllVertices();
787
788 polygon.AddVertexInOrder( v2 );
789 polygon.AddVertexInOrder( v0 );
790 polygon.AddVertexInOrder( w0 );
791 polygon.AddVertexInOrder( w2 );
792
793 if (polygon.PartialClip( voxelLimit, axis ))
794 {
795 polygon.SetNormal( deltaV.cross(v0-v2).unit() );
796
797 extentList.AddSurface( polygon );
798 }
799
800 v2 = w2;
801 }
802
803 //
804 // Next vertex
805 //
806 v0 = w0;
807 v1 = w1;
808 } while( --numPhi > 0 );
809
810 //
811 // We are almost done. But, it is important that we leave no
812 // gaps in the surface of our solid. By using rFudge, however,
813 // we've done exactly that, if we have a phi segment.
814 // Add two additional faces if necessary
815 //
816 if (phiIsOpen && rNorm > DBL_MIN)
817 {
818 cosPhi = std::cos(startPhi);
819 sinPhi = std::sin(startPhi);
820
821 G4ThreeVector a0( r[0]*cosPhi, r[0]*sinPhi, z[0] ),
822 a1( r[1]*cosPhi, r[1]*sinPhi, z[1] ),
823 b0( r0*cosPhi, r0*sinPhi, z[0] ),
824 b1( r1*cosPhi, r1*sinPhi, z[1] );
825
826 transform.ApplyPointTransform( a0 );
827 transform.ApplyPointTransform( a1 );
828 transform.ApplyPointTransform( b0 );
829 transform.ApplyPointTransform( b1 );
830
831 polygon.ClearAllVertices();
832
833 polygon.AddVertexInOrder( a0 );
834 polygon.AddVertexInOrder( a1 );
835 polygon.AddVertexInOrder( b0 );
836 polygon.AddVertexInOrder( b1 );
837
838 if (polygon.PartialClip( voxelLimit , axis))
839 {
840 G4ThreeVector normal( sinPhi, -cosPhi, 0 );
841 polygon.SetNormal( transform.TransformAxis( normal ) );
842
843 extentList.AddSurface( polygon );
844 }
845
846 cosPhi = std::cos(startPhi+deltaPhi);
847 sinPhi = std::sin(startPhi+deltaPhi);
848
849 a0 = G4ThreeVector( r[0]*cosPhi, r[0]*sinPhi, z[0] ),
850 a1 = G4ThreeVector( r[1]*cosPhi, r[1]*sinPhi, z[1] ),
851 b0 = G4ThreeVector( r0*cosPhi, r0*sinPhi, z[0] ),
852 b1 = G4ThreeVector( r1*cosPhi, r1*sinPhi, z[1] );
853 transform.ApplyPointTransform( a0 );
854 transform.ApplyPointTransform( a1 );
855 transform.ApplyPointTransform( b0 );
856 transform.ApplyPointTransform( b1 );
857
858 polygon.ClearAllVertices();
859
860 polygon.AddVertexInOrder( a0 );
861 polygon.AddVertexInOrder( a1 );
862 polygon.AddVertexInOrder( b0 );
863 polygon.AddVertexInOrder( b1 );
864
865 if (polygon.PartialClip( voxelLimit, axis ))
866 {
867 G4ThreeVector normal( -sinPhi, cosPhi, 0 );
868 polygon.SetNormal( transform.TransformAxis( normal ) );
869
870 extentList.AddSurface( polygon );
871 }
872 }
873
874 return;
875}

903//
904// DistanceAway
905//
906// Calculate distance of a point from our conical surface, including the effect
907// of any phi segmentation
908//
909// Arguments:
910// p - (in) Point to check
911// opposite - (in) If true, check opposite hemisphere (see below)
912// distOutside - (out) Additional distance outside the edges of the surface
913// edgeRZnorm - (out) if negative, point is inside
914//
915// return value = distance from the conical plane, if extrapolated beyond edges,
916// signed by whether the point is in inside or outside the shape
917//
918// Notes:
919// * There are two answers, depending on which hemisphere is considered.
920//
921G4double G4PolyconeSide::DistanceAway( const G4ThreeVector &p,
922 G4bool opposite,
923 G4double &distOutside2,
924 G4double *edgeRZnorm )
925{
926 //
927 // Convert our point to r and z
928 //
929 G4double rx = p.perp(), zx = p.z();
930
931 //
932 // Change sign of r if opposite says we should
933 //
934 if (opposite) rx = -rx;
935
936 //
937 // Calculate return value
938 //
939 G4double deltaR = rx - r[0], deltaZ = zx - z[0];
940 G4double answer = deltaR*rNorm + deltaZ*zNorm;
941
942 //
943 // Are we off the surface in r,z space?
944 //
945 G4double q = deltaR*rS + deltaZ*zS;
946 if (q < 0)
947 {
948 distOutside2 = q*q;
949 if (edgeRZnorm) *edgeRZnorm = deltaR*rNormEdge[0] + deltaZ*zNormEdge[0];
950 }
951 else if (q > length)
952 {
953 distOutside2 = sqr( q-length );
954 if (edgeRZnorm)
955 {
956 deltaR = rx - r[1];
957 deltaZ = zx - z[1];
958 *edgeRZnorm = deltaR*rNormEdge[1] + deltaZ*zNormEdge[1];
959 }
960 }
961 else
962 {
963 distOutside2 = 0;
964 if (edgeRZnorm) *edgeRZnorm = answer;
965 }
966
967 if (phiIsOpen)
968 {
969 //
970 // Finally, check phi
971 //
972 G4double phi = GetPhi(p);
973 while( phi < startPhi ) // Loop checking, 13.08.2015, G.Cosmo
974 phi += twopi;
975
976 if (phi > startPhi+deltaPhi)
977 {
978 //
979 // Oops. Are we closer to the start phi or end phi?
980 //
981 G4double d1 = phi-startPhi-deltaPhi;
982 while( phi > startPhi ) // Loop checking, 13.08.2015, G.Cosmo
983 phi -= twopi;
984 G4double d2 = startPhi-phi;
985
986 if (d2 < d1) d1 = d2;
987
988 //
989 // Add result to our distance
990 //
991 G4double dist = d1*rx;
992
993 distOutside2 += dist*dist;
994 if (edgeRZnorm)
995 {
996 *edgeRZnorm = std::max(std::fabs(*edgeRZnorm),std::fabs(dist));
997 }
998 }
999 }
1000
1001 return answer;
1002}

1098//
1099// PointOnCone
1100//
1101// Decide if a point is on a cone and return normal if it is
1102//
1103G4bool G4PolyconeSide::PointOnCone( const G4ThreeVector &hit,
1104 G4double normSign,
1105 const G4ThreeVector &p,
1106 const G4ThreeVector &v,
1107 G4ThreeVector &normal )
1108{
1109 G4double rx = hit.perp();
1110 //
1111 // Check radial/z extent, as appropriate
1112 //
1113 if (!cone->HitOn( rx, hit.z() )) return false;
1114
1115 if (phiIsOpen)
1116 {
1117 G4double phiTolerant = 2.0*kCarTolerance/(rx+kCarTolerance);
1118 //
1119 // Check phi segment. Here we have to be careful
1120 // to use the standard method consistent with
1121 // PolyPhiFace. See PolyPhiFace::InsideEdgesExact
1122 //
1123 G4double phi = GetPhi(hit);
1124 while( phi < startPhi-phiTolerant ) // Loop checking, 13.08.2015, G.Cosmo
1125 phi += twopi;
1126
1127 if (phi > startPhi+deltaPhi+phiTolerant) return false;
1128
1129 if (phi > startPhi+deltaPhi-phiTolerant)
1130 {
1131 //
1132 // Exact treatment
1133 //
1134 G4ThreeVector qx = p + v;
1135 G4ThreeVector qa = qx - corners[2],
1136 qb = qx - corners[3];
1137 G4ThreeVector qacb = qa.cross(qb);
1138
1139 if (normSign*qacb.dot(v) < 0) return false;
1140 }
1141 else if (phi < phiTolerant)
1142 {
1143 G4ThreeVector qx = p + v;
1144 G4ThreeVector qa = qx - corners[1],
1145 qb = qx - corners[0];
1146 G4ThreeVector qacb = qa.cross(qb);
1147
1148 if (normSign*qacb.dot(v) < 0) return false;
1149 }
1150 }
1151
1152 //
1153 // We have a good hit! Calculate normal
1154 //
1155 if (rx < DBL_MIN)
1156 normal = G4ThreeVector( 0, 0, zNorm < 0 ? -1 : 1 );
1157 else
1158 normal = G4ThreeVector( rNorm*hit.x()/rx, rNorm*hit.y()/rx, zNorm );
1159 return true;
1160}

1163//
1164// FindLineIntersect
1165//
1166// Decide the point at which two 2-dimensional lines intersect
1167//
1168// Equation of line: x = x1 + s*tx1
1169// y = y1 + s*ty1
1170//
1171// It is assumed that the lines are *not* parallel
1172//
1173void G4PolyconeSide::FindLineIntersect( G4double x1, G4double y1,
1174 G4double tx1, G4double ty1,
1175 G4double x2, G4double y2,
1176 G4double tx2, G4double ty2,
1177 G4double &x, G4double &y )
1178{
1179 //
1180 // The solution is a simple linear equation
1181 //
1182 G4double deter = tx1*ty2 - tx2*ty1;
1183
1184 G4double s1 = ((x2-x1)*ty2 - tx2*(y2-y1))/deter;
1185 G4double s2 = ((x2-x1)*ty1 - tx1*(y2-y1))/deter;
1186
1187 //
1188 // We want the answer to not depend on which order the
1189 // lines were specified. Take average.
1190 //
1191 x = 0.5*( x1+s1*tx1 + x2+s2*tx2 );
1192 y = 0.5*( y1+s1*ty1 + y2+s2*ty2 );
1193}

1210//
1211// GetPointOnFace
1212//
1213G4ThreeVector G4PolyconeSide::GetPointOnFace()
1214{
1215 G4double x,y,zz;
1216 G4double rr,phi,dz,dr;
1217 dr=r[1]-r[0];dz=z[1]-z[0];
1218 phi=startPhi+deltaPhi*G4UniformRand();
1219 rr=r[0]+dr*G4UniformRand();
1220
1221 x=rr*std::cos(phi);
1222 y=rr*std::sin(phi);
1223
1224 // PolyconeSide has a Ring Form
1225 //
1226 if (dz==0.)
1227 {
1228 zz=z[0];
1229 }
1230 else
1231 {
1232 if(dr==0.) // PolyconeSide has a Tube Form
1233 {
1234 zz = z[0]+dz*G4UniformRand();
1235 }
1236 else
1237 {
1238 zz = z[0]+(rr-r[0])*dz/dr;
1239 }
1240 }
1241
1242 return G4ThreeVector(x,y,zz);
1243}
