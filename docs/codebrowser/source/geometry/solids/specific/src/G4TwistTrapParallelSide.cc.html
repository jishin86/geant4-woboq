50G4TwistTrapParallelSide::G4TwistTrapParallelSide(const G4String &name,
51 G4double PhiTwist, // twist angle
52 G4double pDz, // half z lenght
53 G4double pTheta, // direction between end planes
54 G4double pPhi, // defined by polar and azimutal angles.
55 G4double pDy1, // half y length at -pDz
56 G4double pDx1, // half x length at -pDz,-pDy
57 G4double pDx2, // half x length at -pDz,+pDy
58 G4double pDy2, // half y length at +pDz
59 G4double pDx3, // half x length at +pDz,-pDy
60 G4double pDx4, // half x length at +pDz,+pDy
61 G4double pAlph, // tilt angle at +pDz
62 G4double AngleSide // parity
63 )
64 : G4VTwistSurface(name)
65{
66
67 fAxis[0] = kXAxis; // in local coordinate system
68 fAxis[1] = kZAxis;
69 fAxisMin[0] = -kInfinity ; // X Axis boundary
70 fAxisMax[0] = kInfinity ; // depends on z !!
71 fAxisMin[1] = -pDz ; // Z Axis boundary
72 fAxisMax[1] = pDz ;
73
74 fDx1 = pDx1 ;
75 fDx2 = pDx2 ;
76 fDx3 = pDx3 ;
77 fDx4 = pDx4 ;
78
79 fDy1 = pDy1 ;
80 fDy2 = pDy2 ;
81
82 fDz = pDz ;
83
84 fAlph = pAlph ;
85 fTAlph = std::tan(fAlph) ;
86
87 fTheta = pTheta ;
88 fPhi = pPhi ;
89
90 // precalculate frequently used parameters
91 fDx4plus2 = fDx4 + fDx2 ;
92 fDx4minus2 = fDx4 - fDx2 ;
93 fDx3plus1 = fDx3 + fDx1 ;
94 fDx3minus1 = fDx3 - fDx1 ;
95 fDy2plus1 = fDy2 + fDy1 ;
96 fDy2minus1 = fDy2 - fDy1 ;
97
98 fa1md1 = 2*fDx2 - 2*fDx1 ;
99 fa2md2 = 2*fDx4 - 2*fDx3 ;
100
101 fPhiTwist = PhiTwist ; // dphi
102 fAngleSide = AngleSide ; // 0,90,180,270 deg
103
104 fdeltaX = 2 * fDz * std::tan(fTheta) * std::cos(fPhi) ; // dx in surface equation
105 fdeltaY = 2 * fDz * std::tan(fTheta) * std::sin(fPhi) ; // dy in surface equation
106
107 fRot.rotateZ( AngleSide ) ;
108
109 fTrans.set(0, 0, 0); // No Translation
110 fIsValidNorm = false;
111
112 SetCorners() ;
113 SetBoundaries() ;
114
115}

140
141G4ThreeVector G4TwistTrapParallelSide::GetNormal(const G4ThreeVector &tmpxx,
142 G4bool isGlobal)
143{
144 // GetNormal returns a normal vector at a surface (or very close
145 // to surface) point at tmpxx.
146 // If isGlobal=true, it returns the normal in global coordinate.
147 //
148
149 G4ThreeVector xx;
150 if (isGlobal) {
151 xx = ComputeLocalPoint(tmpxx);
152 if ((xx - fCurrentNormal.p).mag() < 0.5 * kCarTolerance) {
153 return ComputeGlobalDirection(fCurrentNormal.normal);
154 }
155 } else {
156 xx = tmpxx;
157 if (xx == fCurrentNormal.p) {
158 return fCurrentNormal.normal;
159 }
160 }
161
162 G4double phi ;
163 G4double u ;
164
165 GetPhiUAtX(xx,phi,u) ; // phi,u for point xx close to surface
166
167 G4ThreeVector normal = NormAng(phi,u) ; // the normal vector at phi,u
168
169#ifdef G4TWISTDEBUG
170 G4cout << "normal vector = " << normal << G4endl ;
171 G4cout << "phi = " << phi << " , u = " << u << G4endl ;
172#endif
173
174 // normal = normal/normal.mag() ;
175
176 if (isGlobal) {
177 fCurrentNormal.normal = ComputeGlobalDirection(normal.unit());
178 } else {
179 fCurrentNormal.normal = normal.unit();
180 }
181 return fCurrentNormal.normal;
182}

185//* DistanceToSurface -------------------------------------------------
186
187G4int G4TwistTrapParallelSide::DistanceToSurface(const G4ThreeVector &gp,
188 const G4ThreeVector &gv,
189 G4ThreeVector gxx[],
190 G4double distance[],
191 G4int areacode[],
192 G4bool isvalid[],
193 EValidate validate)
194{
195
196 static const G4double pihalf = pi/2 ;
197 const G4double ctol = 0.5 * kCarTolerance;
198
199 G4bool IsParallel = false ;
200 G4bool IsConverged = false ;
201
202 G4int nxx = 0 ; // number of physical solutions
203
204 fCurStatWithV.ResetfDone(validate, &gp, &gv);
205
206 if (fCurStatWithV.IsDone()) {
207 G4int i;
208 for (i=0; i<fCurStatWithV.GetNXX(); i++) {
209 gxx[i] = fCurStatWithV.GetXX(i);
210 distance[i] = fCurStatWithV.GetDistance(i);
211 areacode[i] = fCurStatWithV.GetAreacode(i);
212 isvalid[i] = fCurStatWithV.IsValid(i);
213 }
214 return fCurStatWithV.GetNXX();
215 } else {
216
217 // initialize
218 G4int i;
219 for (i=0; i<G4VSURFACENXX ; i++) {
220 distance[i] = kInfinity;
221 areacode[i] = sOutside;
222 isvalid[i] = false;
223 gxx[i].set(kInfinity, kInfinity, kInfinity);
224 }
225 }
226
227 G4ThreeVector p = ComputeLocalPoint(gp);
228 G4ThreeVector v = ComputeLocalDirection(gv);
229
230#ifdef G4TWISTDEBUG
231 G4cout << "Local point p = " << p << G4endl ;
232 G4cout << "Local direction v = " << v << G4endl ;
233#endif
234
235 G4double phi,u ; // parameters
236
237 // temporary variables
238
239 G4double tmpdist = kInfinity ;
240 G4ThreeVector tmpxx;
241 G4int tmpareacode = sOutside ;
242 G4bool tmpisvalid = false ;
243
244 std::vector<Intersection> xbuf ;
245 Intersection xbuftmp ;
246
247 // prepare some variables for the intersection finder
248
249 G4double L = 2*fDz ;
250
251 G4double phixz = fPhiTwist * ( p.x() * v.z() - p.z() * v.x() ) ;
252 G4double phiyz = fPhiTwist * ( p.y() * v.z() - p.z() * v.y() ) ;
253
254 // special case vz = 0
255
256 if ( v.z() == 0. ) {
257
258 if ( std::fabs(p.z()) <= L ) { // intersection possible in z
259
260 phi = p.z() * fPhiTwist / L ; // phi is determined by the z-position
261
262 u = (2*(fdeltaY*phi*v.x() - fPhiTwist*p.y()*v.x() - fdeltaX*phi*v.y()
263 + fPhiTwist*p.x()*v.y()) + (fDy2plus1*fPhiTwist
264 + 2*fDy2minus1*phi)*(v.x()*std::cos(phi) + v.y()*std::sin(phi)))
265 / (2.* fPhiTwist*(v.y()*std::cos(phi) - v.x()*std::sin(phi)));
266
267 xbuftmp.phi = phi ;
268 xbuftmp.u = u ;
269 xbuftmp.areacode = sOutside ;
270 xbuftmp.distance = kInfinity ;
271 xbuftmp.isvalid = false ;
272
273 xbuf.push_back(xbuftmp) ; // store it to xbuf
274
275 }
276
277 else { // no intersection possible
278
279 distance[0] = kInfinity;
280 gxx[0].set(kInfinity,kInfinity,kInfinity);
281 isvalid[0] = false ;
282 areacode[0] = sOutside ;
283 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0],
284 areacode[0], isvalid[0],
285 0, validate, &gp, &gv);
286
287 return 0;
288
289
290 } // end std::fabs(p.z() <= L
291
292 } // end v.z() == 0
293
294
295 // general solution for non-zero vz
296
297 else {
298
299 G4double c[9],srd[8],si[8] ;
300
301 c[8] = -3600*(-2*phiyz + fDy2plus1*fPhiTwist*v.z()) ;
302 c[7] = -7200*(phixz - 2*fDz*v.y() + (fdeltaY + fDy2minus1)*v.z()) ;
303 c[6] = 120*(-52*phiyz - 120*fDz*v.x() + 60*fdeltaX*v.z() + 11*fDy2plus1*fPhiTwist*v.z()) ;
304 c[5] = 240*(16*phixz - 52*fDz*v.y() + 26*fdeltaY*v.z() + 11*fDy2minus1*v.z()) ;
305 c[4] = 12*(127*phiyz + 640*fDz*v.x() - 320*fdeltaX*v.z() + 4*fDy2plus1*fPhiTwist*v.z()) ;
306 c[3] = -404*phixz + 3048*fDz*v.y() - 1524*fdeltaY*v.z() + 96*fDy2minus1*v.z() ;
307 c[2] = -72*phiyz + 404*(-2*fDz*v.x() + fdeltaX*v.z()) ;
308 c[1] = 12*(phixz - 12*fDz*v.y() + 6*fdeltaY*v.z()) ;
309 c[0] = 24*fDz*v.x() - 12*fdeltaX*v.z() ;
310
311
312#ifdef G4TWISTDEBUG
313 G4cout << "coef = " << c[0] << " "
314 << c[1] << " "
315 << c[2] << " "
316 << c[3] << " "
317 << c[4] << " "
318 << c[5] << " "
319 << c[6] << " "
320 << c[7] << " "
321 << c[8] << G4endl ;
322#endif
323
324 G4JTPolynomialSolver trapEq ;
325 G4int num = trapEq.FindRoots(c,8,srd,si);
326
327
328 for (G4int i = 0 ; i<num ; i++ ) { // loop over all mathematical solutions
329 if ( si[i]==0.0 ) { // only real solutions
330#ifdef G4TWISTDEBUG
331 G4cout << "Solution " << i << " : " << srd[i] << G4endl ;
332#endif
333 phi = std::fmod(srd[i] , pihalf) ;
334
335 u = (1/std::cos(phi)*(2*phixz + 4*fDz*phi*v.x() - 2*fdeltaX*phi*v.z() + (fDy2plus1*fPhiTwist + 2*fDy2minus1*phi)*v.z()* std::sin(phi)))/(2.*fPhiTwist*v.z()) ;
336
337 xbuftmp.phi = phi ;
338 xbuftmp.u = u ;
339 xbuftmp.areacode = sOutside ;
340 xbuftmp.distance = kInfinity ;
341 xbuftmp.isvalid = false ;
342
343 xbuf.push_back(xbuftmp) ; // store it to xbuf
344
345#ifdef G4TWISTDEBUG
346 G4cout << "solution " << i << " = " << phi << " , " << u << G4endl ;
347#endif
348
349 } // end if real solution
350 } // end loop i
351
352 } // end general case
353
354
355 nxx = xbuf.size() ; // save the number of solutions
356
357 G4ThreeVector xxonsurface ; // point on surface
358 G4ThreeVector surfacenormal ; // normal vector
359 G4double deltaX ; // distance between intersection point and point on surface
360 G4double theta ; // angle between track and surfacenormal
361 G4double factor ; // a scaling factor
362 G4int maxint = 30 ; // number of iterations
363
364
365 for ( size_t k = 0 ; k<xbuf.size() ; k++ ) {
366
367#ifdef G4TWISTDEBUG
368 G4cout << "Solution " << k << " : "
369 << "reconstructed phiR = " << xbuf[k].phi
370 << ", uR = " << xbuf[k].u << G4endl ;
371#endif
372
373 phi = xbuf[k].phi ; // get the stored values for phi and u
374 u = xbuf[k].u ;
375
376 IsConverged = false ; // no convergence at the beginning
377
378 for ( G4int i = 1 ; i<maxint ; i++ ) {
379
380 xxonsurface = SurfacePoint(phi,u) ;
381 surfacenormal = NormAng(phi,u) ;
382 tmpdist = DistanceToPlaneWithV(p, v, xxonsurface, surfacenormal, tmpxx);
383 deltaX = ( tmpxx - xxonsurface ).mag() ;
384 theta = std::fabs(std::acos(v*surfacenormal) - pihalf) ;
385 if ( theta < 0.001 ) {
386 factor = 50 ;
387 IsParallel = true ;
388 }
389 else {
390 factor = 1 ;
391 }
392
393#ifdef G4TWISTDEBUG
394 G4cout << "Step i = " << i << ", distance = " << tmpdist << ", " << deltaX << G4endl ;
395 G4cout << "X = " << tmpxx << G4endl ;
396#endif
397
398 GetPhiUAtX(tmpxx, phi, u) ; // the new point xx is accepted and phi/u replaced
399
400#ifdef G4TWISTDEBUG
401 G4cout << "approximated phi = " << phi << ", u = " << u << G4endl ;
402#endif
403
404 if ( deltaX <= factor*ctol ) { IsConverged = true ; break ; }
405
406 } // end iterative loop (i)
407
408
409 // new code 21.09.05 O.Link
410 if ( std::fabs(tmpdist)<ctol ) tmpdist = 0 ;
411
412
413#ifdef G4TWISTDEBUG
414 G4cout << "refined solution " << phi << " , " << u << G4endl ;
415 G4cout << "distance = " << tmpdist << G4endl ;
416 G4cout << "local X = " << tmpxx << G4endl ;
417#endif
418
419 tmpisvalid = false ; // init
420
421 if ( IsConverged ) {
422
423 if (validate == kValidateWithTol) {
424 tmpareacode = GetAreaCode(tmpxx);
425 if (!IsOutside(tmpareacode)) {
426 if (tmpdist >= 0) tmpisvalid = true;
427 }
428 } else if (validate == kValidateWithoutTol) {
429 tmpareacode = GetAreaCode(tmpxx, false);
430 if (IsInside(tmpareacode)) {
431 if (tmpdist >= 0) tmpisvalid = true;
432 }
433 } else { // kDontValidate
434 G4Exception("G4TwistTrapParallelSide::DistanceToSurface()",
435 "GeomSolids0001", FatalException,
436 "Feature NOT implemented !");
437 }
438
439 }
440 else {
441 tmpdist = kInfinity; // no convergence after 10 steps
442 tmpisvalid = false ; // solution is not vaild
443 }
444
445
446 // store the found values
447 xbuf[k].xx = tmpxx ;
448 xbuf[k].distance = tmpdist ;
449 xbuf[k].areacode = tmpareacode ;
450 xbuf[k].isvalid = tmpisvalid ;
451
452
453 } // end loop over physical solutions (variable k)
454
455
456 std::sort(xbuf.begin() , xbuf.end(), DistanceSort ) ; // sorting
457
458#ifdef G4TWISTDEBUG
459 G4cout << G4endl << "list xbuf after sorting : " << G4endl ;
460 G4cout << G4endl << G4endl ;
461#endif
462
463
464 // erase identical intersection (within kCarTolerance)
465 xbuf.erase( std::unique(xbuf.begin(), xbuf.end() , EqualIntersection ) , xbuf.end() ) ;
466
467
468 // add guesses
469
470 G4int nxxtmp = xbuf.size() ;
471
472 if ( nxxtmp<2 || IsParallel ) {
473
474 // positive end
475#ifdef G4TWISTDEBUG
476 G4cout << "add guess at +z/2 .. " << G4endl ;
477#endif
478
479 phi = fPhiTwist/2 ;
480 u = 0 ;
481
482 xbuftmp.phi = phi ;
483 xbuftmp.u = u ;
484 xbuftmp.areacode = sOutside ;
485 xbuftmp.distance = kInfinity ;
486 xbuftmp.isvalid = false ;
487
488 xbuf.push_back(xbuftmp) ; // store it to xbuf
489
490
491#ifdef G4TWISTDEBUG
492 G4cout << "add guess at -z/2 .. " << G4endl ;
493#endif
494
495 phi = -fPhiTwist/2 ;
496 u = 0 ;
497
498 xbuftmp.phi = phi ;
499 xbuftmp.u = u ;
500 xbuftmp.areacode = sOutside ;
501 xbuftmp.distance = kInfinity ;
502 xbuftmp.isvalid = false ;
503
504 xbuf.push_back(xbuftmp) ; // store it to xbuf
505
506 for ( size_t k = nxxtmp ; k<xbuf.size() ; k++ ) {
507
508#ifdef G4TWISTDEBUG
509 G4cout << "Solution " << k << " : "
510 << "reconstructed phiR = " << xbuf[k].phi
511 << ", uR = " << xbuf[k].u << G4endl ;
512#endif
513
514 phi = xbuf[k].phi ; // get the stored values for phi and u
515 u = xbuf[k].u ;
516
517 IsConverged = false ; // no convergence at the beginning
518
519 for ( G4int i = 1 ; i<maxint ; i++ ) {
520
521 xxonsurface = SurfacePoint(phi,u) ;
522 surfacenormal = NormAng(phi,u) ;
523 tmpdist = DistanceToPlaneWithV(p, v, xxonsurface, surfacenormal, tmpxx);
524 deltaX = ( tmpxx - xxonsurface ).mag() ;
525 theta = std::fabs(std::acos(v*surfacenormal) - pihalf) ;
526 if ( theta < 0.001 ) {
527 factor = 50 ;
528 }
529 else {
530 factor = 1 ;
531 }
532
533#ifdef G4TWISTDEBUG
534 G4cout << "Step i = " << i << ", distance = " << tmpdist << ", " << deltaX << G4endl ;
535 G4cout << "X = " << tmpxx << G4endl ;
536#endif
537
538 GetPhiUAtX(tmpxx, phi, u) ; // the new point xx is accepted and phi/u replaced
539
540#ifdef G4TWISTDEBUG
541 G4cout << "approximated phi = " << phi << ", u = " << u << G4endl ;
542#endif
543
544 if ( deltaX <= factor*ctol ) { IsConverged = true ; break ; }
545
546 } // end iterative loop (i)
547
548
549 // new code 21.09.05 O.Link
550 if ( std::fabs(tmpdist)<ctol ) tmpdist = 0 ;
551
552
553#ifdef G4TWISTDEBUG
554 G4cout << "refined solution " << phi << " , " << u << G4endl ;
555 G4cout << "distance = " << tmpdist << G4endl ;
556 G4cout << "local X = " << tmpxx << G4endl ;
557#endif
558
559 tmpisvalid = false ; // init
560
561 if ( IsConverged ) {
562
563 if (validate == kValidateWithTol) {
564 tmpareacode = GetAreaCode(tmpxx);
565 if (!IsOutside(tmpareacode)) {
566 if (tmpdist >= 0) tmpisvalid = true;
567 }
568 } else if (validate == kValidateWithoutTol) {
569 tmpareacode = GetAreaCode(tmpxx, false);
570 if (IsInside(tmpareacode)) {
571 if (tmpdist >= 0) tmpisvalid = true;
572 }
573 } else { // kDontValidate
574 G4Exception("G4TwistedBoxSide::DistanceToSurface()",
575 "GeomSolids0001", FatalException,
576 "Feature NOT implemented !");
577 }
578
579 }
580 else {
581 tmpdist = kInfinity; // no convergence after 10 steps
582 tmpisvalid = false ; // solution is not vaild
583 }
584
585
586 // store the found values
587 xbuf[k].xx = tmpxx ;
588 xbuf[k].distance = tmpdist ;
589 xbuf[k].areacode = tmpareacode ;
590 xbuf[k].isvalid = tmpisvalid ;
591
592
593 } // end loop over physical solutions
594
595
596 } // end less than 2 solutions
597
598
599 // sort again
600 std::sort(xbuf.begin() , xbuf.end(), DistanceSort ) ; // sorting
601
602 // erase identical intersection (within kCarTolerance)
603 xbuf.erase( std::unique(xbuf.begin(), xbuf.end() , EqualIntersection ) , xbuf.end() ) ;
604
605#ifdef G4TWISTDEBUG
606 G4cout << G4endl << "list xbuf after sorting : " << G4endl ;
607 G4cout << G4endl << G4endl ;
608#endif
609
610 nxx = xbuf.size() ; // determine number of solutions again.
611
612 for ( size_t i = 0 ; i<xbuf.size() ; i++ ) {
613
614 distance[i] = xbuf[i].distance;
615 gxx[i] = ComputeGlobalPoint(xbuf[i].xx);
616 areacode[i] = xbuf[i].areacode ;
617 isvalid[i] = xbuf[i].isvalid ;
618
619 fCurStatWithV.SetCurrentStatus(i, gxx[i], distance[i], areacode[i],
620 isvalid[i], nxx, validate, &gp, &gv);
621
622#ifdef G4TWISTDEBUG
623 G4cout << "element Nr. " << i
624 << ", local Intersection = " << xbuf[i].xx
625 << ", distance = " << xbuf[i].distance
626 << ", u = " << xbuf[i].u
627 << ", phi = " << xbuf[i].phi
628 << ", isvalid = " << xbuf[i].isvalid
629 << G4endl ;
630#endif
631
632 } // end for( i ) loop
633
634
635#ifdef G4TWISTDEBUG
636 G4cout << "G4TwistTrapParallelSide finished " << G4endl ;
637 G4cout << nxx << " possible physical solutions found" << G4endl ;
638 for ( G4int k= 0 ; k< nxx ; k++ ) {
639 G4cout << "global intersection Point found: " << gxx[k] << G4endl ;
640 G4cout << "distance = " << distance[k] << G4endl ;
641 G4cout << "isvalid = " << isvalid[k] << G4endl ;
642 }
643#endif
644
645 return nxx ;
646
647}

652//* DistanceToSurface -------------------------------------------------
653
654G4int G4TwistTrapParallelSide::DistanceToSurface(const G4ThreeVector &gp,
655 G4ThreeVector gxx[],
656 G4double distance[],
657 G4int areacode[])
658{
659 // to do
660
661 const G4double ctol = 0.5 * kCarTolerance;
662
663 fCurStat.ResetfDone(kDontValidate, &gp);
664
665 if (fCurStat.IsDone()) {
666 G4int i;
667 for (i=0; i<fCurStat.GetNXX(); i++) {
668 gxx[i] = fCurStat.GetXX(i);
669 distance[i] = fCurStat.GetDistance(i);
670 areacode[i] = fCurStat.GetAreacode(i);
671 }
672 return fCurStat.GetNXX();
673 } else {
674 // initialize
675 G4int i;
676 for (i=0; i<G4VSURFACENXX; i++) {
677 distance[i] = kInfinity;
678 areacode[i] = sOutside;
679 gxx[i].set(kInfinity, kInfinity, kInfinity);
680 }
681 }
682
683 G4ThreeVector p = ComputeLocalPoint(gp);
684 G4ThreeVector xx; // intersection point
685 G4ThreeVector xxonsurface ; // interpolated intersection point
686
687 // the surfacenormal at that surface point
688 G4double phiR = 0 ; //
689 G4double uR = 0 ;
690
691 G4ThreeVector surfacenormal ;
692 G4double deltaX ;
693
694 G4int maxint = 20 ;
695
696 for ( G4int i = 1 ; i<maxint ; i++ ) {
697
698 xxonsurface = SurfacePoint(phiR,uR) ;
699 surfacenormal = NormAng(phiR,uR) ;
700 distance[0] = DistanceToPlane(p, xxonsurface, surfacenormal, xx); // new XX
701 deltaX = ( xx - xxonsurface ).mag() ;
702
703#ifdef G4TWISTDEBUG
704 G4cout << "i = " << i << ", distance = " << distance[0] << ", " << deltaX << G4endl ;
705 G4cout << "X = " << xx << G4endl ;
706#endif
707
708 // the new point xx is accepted and phi/psi replaced
709 GetPhiUAtX(xx, phiR, uR) ;
710
711 if ( deltaX <= ctol ) { break ; }
712
713 }
714
715 // check validity of solution ( valid phi,psi )
716
717 G4double halfphi = 0.5*fPhiTwist ;
718 G4double uMax = GetBoundaryMax(phiR) ;
719 G4double uMin = GetBoundaryMin(phiR) ;
720
721 if ( phiR > halfphi ) phiR = halfphi ;
722 if ( phiR < -halfphi ) phiR = -halfphi ;
723 if ( uR > uMax ) uR = uMax ;
724 if ( uR < uMin ) uR = uMin ;
725
726 xxonsurface = SurfacePoint(phiR,uR) ;
727 distance[0] = ( p - xx ).mag() ;
728 if ( distance[0] <= ctol ) { distance[0] = 0 ; }
729
730 // end of validity
731
732#ifdef G4TWISTDEBUG
733 G4cout << "refined solution " << phiR << " , " << uR << " , " << G4endl ;
734 G4cout << "distance = " << distance[0] << G4endl ;
735 G4cout << "X = " << xx << G4endl ;
736#endif
737
738 G4bool isvalid = true;
739 gxx[0] = ComputeGlobalPoint(xx);
740
741#ifdef G4TWISTDEBUG
742 G4cout << "intersection Point found: " << gxx[0] << G4endl ;
743 G4cout << "distance = " << distance[0] << G4endl ;
744#endif
745
746 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
747 isvalid, 1, kDontValidate, &gp);
748 return 1;
749}

752//=====================================================================
753//* GetAreaCode -------------------------------------------------------
754
755G4int G4TwistTrapParallelSide::GetAreaCode(const G4ThreeVector &xx,
756 G4bool withTol)
757{
758 // We must use the function in local coordinate system.
759 // See the description of DistanceToSurface(p,v).
760
761 const G4double ctol = 0.5 * kCarTolerance;
762
763 G4double phi ;
764 G4double yprime ;
765 GetPhiUAtX(xx, phi,yprime ) ;
766
767 G4double fXAxisMax = GetBoundaryMax(phi) ;
768 G4double fXAxisMin = GetBoundaryMin(phi) ;
769
770#ifdef G4TWISTDEBUG
771 G4cout << "GetAreaCode: phi = " << phi << G4endl ;
772 G4cout << "GetAreaCode: yprime = " << yprime << G4endl ;
773 G4cout << "Intervall is " << fXAxisMin << " to " << fXAxisMax << G4endl ;
774#endif
775
776 G4int areacode = sInside;
777
778 if (fAxis[0] == kXAxis && fAxis[1] == kZAxis) {
779
780 G4int zaxis = 1;
781
782 if (withTol) {
783
784 G4bool isoutside = false;
785
786 // test boundary of xaxis
787
788 if (yprime < fXAxisMin + ctol) {
789 areacode |= (sAxis0 & (sAxisX | sAxisMin)) | sBoundary;
790 if (yprime <= fXAxisMin - ctol) isoutside = true;
791
792 } else if (yprime > fXAxisMax - ctol) {
793 areacode |= (sAxis0 & (sAxisX | sAxisMax)) | sBoundary;
794 if (yprime >= fXAxisMax + ctol) isoutside = true;
795 }
796
797 // test boundary of z-axis
798
799 if (xx.z() < fAxisMin[zaxis] + ctol) {
800 areacode |= (sAxis1 & (sAxisZ | sAxisMin));
801
802 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
803 else areacode |= sBoundary;
804 if (xx.z() <= fAxisMin[zaxis] - ctol) isoutside = true;
805
806 } else if (xx.z() > fAxisMax[zaxis] - ctol) {
807 areacode |= (sAxis1 & (sAxisZ | sAxisMax));
808
809 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
810 else areacode |= sBoundary;
811 if (xx.z() >= fAxisMax[zaxis] + ctol) isoutside = true;
812 }
813
814 // if isoutside = true, clear inside bit.
815 // if not on boundary, add axis information.
816
817 if (isoutside) {
818 G4int tmpareacode = areacode & (~sInside);
819 areacode = tmpareacode;
820 } else if ((areacode & sBoundary) != sBoundary) {
821 areacode |= (sAxis0 & sAxisX) | (sAxis1 & sAxisZ);
822 }
823
824 } else {
825
826 // boundary of y-axis
827
828 if (yprime < fXAxisMin ) {
829 areacode |= (sAxis0 & (sAxisX | sAxisMin)) | sBoundary;
830 } else if (yprime > fXAxisMax) {
831 areacode |= (sAxis0 & (sAxisX | sAxisMax)) | sBoundary;
832 }
833
834 // boundary of z-axis
835
836 if (xx.z() < fAxisMin[zaxis]) {
837 areacode |= (sAxis1 & (sAxisZ | sAxisMin));
838 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
839 else areacode |= sBoundary;
840
841 } else if (xx.z() > fAxisMax[zaxis]) {
842 areacode |= (sAxis1 & (sAxisZ | sAxisMax)) ;
843 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
844 else areacode |= sBoundary;
845 }
846
847 if ((areacode & sBoundary) != sBoundary) {
848 areacode |= (sAxis0 & sAxisX) | (sAxis1 & sAxisZ);
849 }
850 }
851 return areacode;
852 } else {
853 G4Exception("G4TwistTrapParallelSide::GetAreaCode()",
854 "GeomSolids0001", FatalException,
855 "Feature NOT implemented !");
856 }
857 return areacode;
858}

861//* SetCorners() ------------------------------------------------------
862
863void G4TwistTrapParallelSide::SetCorners()
864{
865
866 // Set Corner points in local coodinate.
867
868 if (fAxis[0] == kXAxis && fAxis[1] == kZAxis) {
869
870 G4double x, y, z;
871
872 // corner of Axis0min and Axis1min
873
874 x = -fdeltaX/2. + (-fDx2 + fDy1*fTAlph)*std::cos(fPhiTwist/2.) + fDy1*std::sin(fPhiTwist/2.) ;
875 y = -fdeltaY/2. + fDy1*std::cos(fPhiTwist/2.) + (fDx2 - fDy1*fTAlph)*std::sin(fPhiTwist/2.) ;
876 z = -fDz ;
877
878 SetCorner(sC0Min1Min, x, y, z);
879
880 // corner of Axis0max and Axis1min
881
882 x = -fdeltaX/2. + (fDx2 + fDy1*fTAlph)*std::cos(fPhiTwist/2.) + fDy1*std::sin(fPhiTwist/2.) ;
883 y = -fdeltaY/2. + fDy1*std::cos(fPhiTwist/2.) - (fDx2 + fDy1*fTAlph)*std::sin(fPhiTwist/2.) ;
884 z = -fDz;
885
886 SetCorner(sC0Max1Min, x, y, z);
887
888 // corner of Axis0max and Axis1max
889 x = fdeltaX/2. + (fDx4 + fDy2*fTAlph)*std::cos(fPhiTwist/2.) - fDy2*std::sin(fPhiTwist/2.) ;
890 y = fdeltaY/2. + fDy2*std::cos(fPhiTwist/2.) + (fDx4 + fDy2*fTAlph)*std::sin(fPhiTwist/2.) ;
891 z = fDz ;
892
893 SetCorner(sC0Max1Max, x, y, z);
894
895 // corner of Axis0min and Axis1max
896 x = fdeltaX/2. + (-fDx4 + fDy2*fTAlph)*std::cos(fPhiTwist/2.) - fDy2*std::sin(fPhiTwist/2.) ;
897 y = fdeltaY/2. + fDy2*std::cos(fPhiTwist/2.) + (-fDx4 + fDy2*fTAlph)*std::sin(fPhiTwist/2.) ;
898 z = fDz ;
899
900 SetCorner(sC0Min1Max, x, y, z);
901
902 } else {
903
904 G4Exception("G4TwistTrapParallelSide::SetCorners()",
905 "GeomSolids0001", FatalException,
906 "Method NOT implemented !");
907 }

955//=====================================================================
956//* GetPhiUAtX() ------------------------------------------------------
957
958void
959G4TwistTrapParallelSide::GetPhiUAtX( G4ThreeVector p, G4double &phi, G4double &u)
960{
961 // find closest point XX on surface for a given point p
962 // X0 is a point on the surface, d is the direction ( both for a fixed z = pz)
963
964 // phi is given by the z coordinate of p
965
966 phi = p.z()/(2*fDz)*fPhiTwist ;
967
968 u = ((-(fdeltaX*phi) + fPhiTwist*p.x())* std::cos(phi) + (-(fdeltaY*phi) + fPhiTwist*p.y())*std::sin(phi))/fPhiTwist ;
969
970}

973//* ProjectPoint() ----------------------------------------------------
974
975G4ThreeVector G4TwistTrapParallelSide::ProjectPoint(const G4ThreeVector &p,
976 G4bool isglobal)
977{
978 // Get Rho at p.z() on Hyperbolic Surface.
979 G4ThreeVector tmpp;
980 if (isglobal) {
981 tmpp = fRot.inverse()*p - fTrans;
982 } else {
983 tmpp = p;
984 }
985
986 G4double phi ;
987 G4double u ;
988
989 GetPhiUAtX( tmpp, phi, u ) ; // calculate (phi, u) for a point p close the surface
990
991 G4ThreeVector xx = SurfacePoint(phi,u) ; // transform back to cartesian coordinates
992
993 if (isglobal) {
994 return (fRot * xx + fTrans);
995 } else {
996 return xx;
997 }
998}

1001//* GetFacets() -------------------------------------------------------
1002
1003void G4TwistTrapParallelSide::GetFacets( G4int k, G4int n, G4double xyz[][3],
1004 G4int faces[][4], G4int iside )
1005{
1006
1007 G4double phi ;
1008 G4double z, u ; // the two parameters for the surface equation
1009 G4ThreeVector p ; // a point on the surface, given by (z,u)
1010
1011 G4int nnode ;
1012 G4int nface ;
1013
1014 G4double umin, umax ;
1015
1016 // calculate the (n-1)*(k-1) vertices
1017
1018 G4int i,j ;
1019
1020 for ( i = 0 ; i<n ; i++ ) {
1021
1022 z = -fDz+i*(2.*fDz)/(n-1) ;
1023 phi = z*fPhiTwist/(2*fDz) ;
1024 umin = GetBoundaryMin(phi) ;
1025 umax = GetBoundaryMax(phi) ;
1026
1027 for ( j = 0 ; j<k ; j++ ) {
1028
1029 nnode = GetNode(i,j,k,n,iside) ;
1030 u = umax - j*(umax-umin)/(k-1) ;
1031 p = SurfacePoint(phi,u,true) ; // surface point in global coordinate system
1032
1033 xyz[nnode][0] = p.x() ;
1034 xyz[nnode][1] = p.y() ;
1035 xyz[nnode][2] = p.z() ;
1036
1037 if ( i<n-1 && j<k-1 ) { // conterclock wise filling
1038
1039 nface = GetFace(i,j,k,n,iside) ;
1040 faces[nface][0] = GetEdgeVisibility(i,j,k,n,0,-1) * (GetNode(i ,j ,k,n,iside)+1) ; // fortran numbering
1041 faces[nface][1] = GetEdgeVisibility(i,j,k,n,1,-1) * (GetNode(i ,j+1,k,n,iside)+1) ;
1042 faces[nface][2] = GetEdgeVisibility(i,j,k,n,2,-1) * (GetNode(i+1,j+1,k,n,iside)+1) ;
1043 faces[nface][3] = GetEdgeVisibility(i,j,k,n,3,-1) * (GetNode(i+1,j ,k,n,iside)+1) ;
1044
1045 }
1046 }
1047 }
