61//
62// Constructor: special PGON/PCON case
63//
64G4ReduciblePolygon::G4ReduciblePolygon( const G4double rmin[],
65 const G4double rmax[],
66 const G4double z[], G4int n )
67 : aMin(0.), aMax(0.), bMin(0.), bMax(0.),
68 vertexHead(0)
69{
70 //
71 // Translate
72 //
73 G4double *a = new G4double[n*2];
74 G4double *b = new G4double[n*2];
75
76 G4double *rOut = a + n,
77 *zOut = b + n,
78 *rIn = rOut-1,
79 *zIn = zOut-1;
80
81 G4int i;
82 for( i=0; i < n; i++, rOut++, zOut++, rIn--, zIn-- )
83 {
84 *rOut = rmax[i];
85 *rIn = rmin[i];
86 *zOut = *zIn = z[i];
87 }
88
89 Create( a, b, n*2 );
90
91 delete [] a;
92 delete [] b;
93}

96//
97// Create
98//
99// To be called by constructors, fill in the list and statistics for a new
100// polygon
101//
102void G4ReduciblePolygon::Create( const G4double a[],
103 const G4double b[], G4int n )
104{
105 if (n<3)
106 G4Exception("G4ReduciblePolygon::Create()", "GeomSolids0002",
107 FatalErrorInArgument, "Less than 3 vertices specified.");
108
109 const G4double *anext = a, *bnext = b;
110 ABVertex *prev = 0;
111 do // Loop checking, 13.08.2015, G.Cosmo
112 {
113 ABVertex *newVertex = new ABVertex;
114 newVertex->a = *anext;
115 newVertex->b = *bnext;
116 newVertex->next = 0;
117 if (prev==0)
118 {
119 vertexHead = newVertex;
120 }
121 else
122 {
123 prev->next = newVertex;
124 }
125
126 prev = newVertex;
127 } while( ++anext, ++bnext < b+n );
128
129 numVertices = n;
130
131 CalculateMaxMin();
132}

212//
213// RemoveDuplicateVertices
214//
215// Remove adjacent vertices that are equal. Returns "false" if there
216// is a problem (too few vertices remaining).
217//
218G4bool G4ReduciblePolygon::RemoveDuplicateVertices( G4double tolerance )
219{
220 ABVertex *curr = vertexHead,
221 *prev = 0, *next = 0;
222 while( curr ) // Loop checking, 13.08.2015, G.Cosmo
223 {
224 next = curr->next;
225 if (next == 0) next = vertexHead;
226
227 if (std::fabs(curr->a-next->a) < tolerance &&
228 std::fabs(curr->b-next->b) < tolerance )
229 {
230 //
231 // Duplicate found: do we have > 3 vertices?
232 //
233 if (numVertices <= 3)
234 {
235 CalculateMaxMin();
236 return false;
237 }
238
239 //
240 // Delete
241 //
242 ABVertex *toDelete = curr;
243 curr = curr->next;
244 delete toDelete;
245
246 numVertices--;
247
248 if (prev) prev->next = curr; else vertexHead = curr;
249 }
250 else
251 {
252 prev = curr;
253 curr = curr->next;
254 }
255 }
256
257 //
258 // In principle, this is not needed, but why not just play it safe?
259 //
260 CalculateMaxMin();
261
262 return true;
263}

266//
267// RemoveRedundantVertices
268//
269// Remove any unneeded vertices, i.e. those vertices which
270// are on the line connecting the previous and next vertices.
271//
272G4bool G4ReduciblePolygon::RemoveRedundantVertices( G4double tolerance )
273{
274 //
275 // Under these circumstances, we can quit now!
276 //
277 if (numVertices <= 2) return false;
278
279 G4double tolerance2 = tolerance*tolerance;
280
281 //
282 // Loop over all vertices
283 //
284 ABVertex *curr = vertexHead, *next = 0;
285 while( curr ) // Loop checking, 13.08.2015, G.Cosmo
286 {
287 next = curr->next;
288 if (next == 0) next = vertexHead;
289
290 G4double da = next->a - curr->a,
291 db = next->b - curr->b;
292
293 //
294 // Loop over all subsequent vertices, up to curr
295 //
296 for(;;)
297 {
298 //
299 // Get vertex after next
300 //
301 ABVertex *test = next->next;
302 if (test == 0) test = vertexHead;
303
304 //
305 // If we are back to the original vertex, stop
306 //
307 if (test==curr) break;
308
309 //
310 // Test for parallel line segments
311 //
312 G4double dat = test->a - curr->a,
313 dbt = test->b - curr->b;
314
315 if (std::fabs(dat*db-dbt*da)>tolerance2) break;
316
317 //
318 // Redundant vertex found: do we have > 3 vertices?
319 //
320 if (numVertices <= 3)
321 {
322 CalculateMaxMin();
323 return false;
324 }
325
326 //
327 // Delete vertex pointed to by next. Carefully!
328 //
329 if (curr->next)
330 { // next is not head
331 if (next->next)
332 curr->next = test; // next is not tail
333 else
334 curr->next = 0; // New tail
335 }
336 else
337 vertexHead = test; // New head
338
339 if ((curr != next) && (next != test)) delete next;
340
341 numVertices--;
342
343 //
344 // Replace next by the vertex we just tested,
345 // and keep on going...
346 //
347 next = test;
348 da = dat; db = dbt;
349 }
350 curr = curr->next;
351 }
352
353 //
354 // In principle, this is not needed, but why not just play it safe?
355 //
356 CalculateMaxMin();
357
358 return true;
359}

362//
363// ReverseOrder
364//
365// Reverse the order of the vertices
366//
367void G4ReduciblePolygon::ReverseOrder()
368{
369 //
370 // Loop over all vertices
371 //
372 ABVertex *prev = vertexHead;
373 if (prev==0) return; // No vertices
374
375 ABVertex *curr = prev->next;
376 if (curr==0) return; // Just one vertex
377
378 //
379 // Our new tail
380 //
381 vertexHead->next = 0;
382
383 for(;;)
384 {
385 //
386 // Save pointer to next vertex (in original order)
387 //
388 ABVertex *save = curr->next;
389
390 //
391 // Replace it with a pointer to the previous one
392 // (in original order)
393 //
394 curr->next = prev;
395
396 //
397 // Last vertex?
398 //
399 if (save == 0) break;
400
401 //
402 // Next vertex
403 //
404 prev = curr;
405 curr = save;
406 }
407
408 //
409 // Our new head
410 //
411 vertexHead = curr;
412}

415// StartWithZMin
416//
417// Starting alway with Zmin=bMin
418// This method is used for GenericPolycone
419//
420void G4ReduciblePolygon::StartWithZMin()
421{
422 ABVertex *curr = vertexHead;
423 G4double bcurr = curr->b;
424 ABVertex *prev = curr;
425 while( curr ) // Loop checking, 13.08.2015, G.Cosmo
426 {
427 if(curr->b < bcurr)
428 {
429 bcurr = curr->b;
430 ABVertex *curr1 = curr;
431 while( curr1 ) // Loop checking, 13.08.2015, G.Cosmo
432 {
433 if(curr1->next == 0) { curr1->next = vertexHead; break; }
434 curr1 = curr1->next;
435 }
436 vertexHead = curr;
437 prev->next = 0;
438 }
439 prev = curr;
440 curr = curr->next;
441 }
442}

445//
446// CrossesItself
447//
448// Return "true" if the polygon crosses itself
449//
450// Warning: this routine is not very fast (runs as N**2)
451//
452G4bool G4ReduciblePolygon::CrossesItself( G4double tolerance )
453{
454 G4double tolerance2 = tolerance*tolerance;
455 G4double one = 1.0-tolerance,
456 zero = tolerance;
457 //
458 // Top loop over line segments. By the time we finish
459 // with the second to last segment, we're done.
460 //
461 ABVertex *curr1 = vertexHead, *next1=0;
462 while (curr1->next) // Loop checking, 13.08.2015, G.Cosmo
463 {
464 next1 = curr1->next;
465 G4double da1 = next1->a-curr1->a,
466 db1 = next1->b-curr1->b;
467
468 //
469 // Inner loop over subsequent line segments
470 //
471 ABVertex *curr2 = next1->next;
472 while( curr2 ) // Loop checking, 13.08.2015, G.Cosmo
473 {
474 ABVertex *next2 = curr2->next;
475 if (next2==0) next2 = vertexHead;
476 G4double da2 = next2->a-curr2->a,
477 db2 = next2->b-curr2->b;
478 G4double a12 = curr2->a-curr1->a,
479 b12 = curr2->b-curr1->b;
480
481 //
482 // Calculate intersection of the two lines
483 //
484 G4double deter = da1*db2 - db1*da2;
485 if (std::fabs(deter) > tolerance2)
486 {
487 G4double s1, s2;
488 s1 = (a12*db2-b12*da2)/deter;
489
490 if (s1 >= zero && s1 < one)
491 {
492 s2 = -(da1*b12-db1*a12)/deter;
493 if (s2 >= zero && s2 < one) return true;
494 }
495 }
496 curr2 = curr2->next;
497 }
498 curr1 = next1;
499 }
500 return false;
501}

505//
506// BisectedBy
507//
508// Decide if a line through two points crosses the polygon, within tolerance
509//
510G4bool G4ReduciblePolygon::BisectedBy( G4double a1, G4double b1,
511 G4double a2, G4double b2,
512 G4double tolerance )
513{
514 G4int nNeg = 0, nPos = 0;
515
516 G4double a12 = a2-a1, b12 = b2-b1;
517 G4double len12 = std::sqrt( a12*a12 + b12*b12 );
518 a12 /= len12; b12 /= len12;
519
520 ABVertex *curr = vertexHead;
521 do // Loop checking, 13.08.2015, G.Cosmo
522 {
523 G4double av = curr->a - a1,
524 bv = curr->b - b1;
525
526 G4double cross = av*b12 - bv*a12;
527
528 if (cross < -tolerance)
529 {
530 if (nPos) return true;
531 nNeg++;
532 }
533 else if (cross > tolerance)
534 {
535 if (nNeg) return true;
536 nPos++;
537 }
538 curr = curr->next;
539 } while( curr );
540
541 return false;
542}

546//
547// Area
548//
549// Calculated signed polygon area, where polygons specified in a
550// clockwise manner (where x==a, y==b) have negative area
551//
552// References: [O' Rourke (C)] pp. 18-27; [Gems II] pp. 5-6:
553// "The Area of a Simple Polygon", Jon Rokne.
554//
555G4double G4ReduciblePolygon::Area()
556{
557 G4double answer = 0;
558
559 ABVertex *curr = vertexHead, *next;
560 do // Loop checking, 13.08.2015, G.Cosmo
561 {
562 next = curr->next;
563 if (next==0) next = vertexHead;
564
565 answer += curr->a*next->b - curr->b*next->a;
566 curr = curr->next;
567 } while( curr );
568
569 return 0.5*answer;
570}

587//
588// CalculateMaxMin
589//
590// To be called when the vertices are changed, this
591// routine re-calculates global values
592//
593void G4ReduciblePolygon::CalculateMaxMin()
594{
595 ABVertex *curr = vertexHead;
596 aMin = aMax = curr->a;
597 bMin = bMax = curr->b;
598 curr = curr->next;
599 while( curr ) // Loop checking, 13.08.2015, G.Cosmo
600 {
601 if (curr->a < aMin)
602 aMin = curr->a;
603 else if (curr->a > aMax)
604 aMax = curr->a;
605
606 if (curr->b < bMin)
607 bMin = curr->b;
608 else if (curr->b > bMax)
609 bMax = curr->b;
610
611 curr = curr->next;
612 }
