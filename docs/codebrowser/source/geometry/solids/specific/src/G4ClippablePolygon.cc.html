80//
81// Clip
82//
83G4bool G4ClippablePolygon::Clip( const G4VoxelLimits &voxelLimit )
84{
85 if (voxelLimit.IsLimited()) {
86 ClipAlongOneAxis( voxelLimit, kXAxis );
87 ClipAlongOneAxis( voxelLimit, kYAxis );
88 ClipAlongOneAxis( voxelLimit, kZAxis );
89 }
90
91 return (vertices.size() > 0);
92}

95//
96// PartialClip
97//
98// Clip, while ignoring the indicated axis
99//
100G4bool G4ClippablePolygon::PartialClip( const G4VoxelLimits &voxelLimit,
101 const EAxis IgnoreMe )
102{
103 if (voxelLimit.IsLimited()) {
104 if (IgnoreMe != kXAxis) ClipAlongOneAxis( voxelLimit, kXAxis );
105 if (IgnoreMe != kYAxis) ClipAlongOneAxis( voxelLimit, kYAxis );
106 if (IgnoreMe != kZAxis) ClipAlongOneAxis( voxelLimit, kZAxis );
107 }
108
109 return (vertices.size() > 0);
110}

113//
114// GetExtent
115//
116G4bool G4ClippablePolygon::GetExtent( const EAxis axis,
117 G4double &min,
118 G4double &max ) const
119{
120 //
121 // Okay, how many entries do we have?
122 //
123 G4int noLeft = vertices.size();
124
125 //
126 // Return false if nothing is left
127 //
128 if (noLeft == 0) return false;
129
130 //
131 // Initialize min and max to our first vertex
132 //
133 min = max = vertices[0].operator()( axis );
134
135 //
136 // Compare to the rest
137 //
138 G4int i;
139 for( i=1; i<noLeft; i++ )
140 {
141 G4double component = vertices[i].operator()( axis );
142 if (component < min )
143 min = component;
144 else if (component > max )
145 max = component;
146 }
147
148 return true;
149}

152//
153// GetMinPoint
154//
155// Returns pointer to minimum point along the specified axis.
156// Take care! Do not use pointer after destroying parent polygon.
157//
158const G4ThreeVector *G4ClippablePolygon::GetMinPoint( const EAxis axis ) const
159{
160 G4int noLeft = vertices.size();
161 if (noLeft==0)
162 G4Exception("G4ClippablePolygon::GetMinPoint()",
163 "GeomSolids0002", FatalException, "Empty polygon.");
164
165 const G4ThreeVector *answer = &(vertices[0]);
166 G4double min = answer->operator()(axis);
167
168 G4int i;
169 for( i=1; i<noLeft; i++ )
170 {
171 G4double component = vertices[i].operator()( axis );
172 if (component < min)
173 {
174 answer = &(vertices[i]);
175 min = component;
176 }
177 }
178
179 return answer;
180}

214//
215// InFrontOf
216//
217// Decide if this polygon is in "front" of another when
218// viewed along the specified axis. For our purposes here,
219// it is sufficient to use the minimum extent of the
220// polygon along the axis to determine this.
221//
222// In case the minima of the two polygons are equal,
223// we use a more sophisticated test.
224//
225// Note that it is possible for the two following
226// statements to both return true or both return false:
227// polygon1.InFrontOf(polygon2)
228// polygon2.BehindOf(polygon1)
229//
230G4bool G4ClippablePolygon::InFrontOf( const G4ClippablePolygon &other,
231 EAxis axis ) const
232{
233 //
234 // If things are empty, do something semi-sensible
235 //
236 G4int noLeft = vertices.size();
237 if (noLeft==0) return false;
238
239 if (other.Empty()) return true;
240
241 //
242 // Get minimum of other polygon
243 //
244 const G4ThreeVector *minPointOther = other.GetMinPoint( axis );
245 const G4double minOther = minPointOther->operator()(axis);
246
247 //
248 // Get minimum of this polygon
249 //
250 const G4ThreeVector *minPoint = GetMinPoint( axis );
251 const G4double min = minPoint->operator()(axis);
252
253 //
254 // Easy decision
255 //
256 if (min < minOther-kCarTolerance) return true; // Clear winner
257
258 if (minOther < min-kCarTolerance) return false; // Clear loser
259
260 //
261 // We have a tie (this will not be all that rare since our
262 // polygons are connected)
263 //
264 // Check to see if there is a vertex in the other polygon
265 // that is behind this one (or vice versa)
266 //
267 G4bool answer;
268 G4ThreeVector normalOther = other.GetNormal();
269
270 if (std::fabs(normalOther(axis)) > std::fabs(normal(axis)))
271 {
272 G4double minP, maxP;
273 GetPlanerExtent( *minPointOther, normalOther, minP, maxP );
274
275 answer = (normalOther(axis) > 0) ? (minP < -kCarTolerance)
276 : (maxP > +kCarTolerance);
277 }
278 else
279 {
280 G4double minP, maxP;
281 other.GetPlanerExtent( *minPoint, normal, minP, maxP );
282
283 answer = (normal(axis) > 0) ? (maxP > +kCarTolerance)
284 : (minP < -kCarTolerance);
285 }
286 return answer;
287}

289//
290// BehindOf
291//
292// Decide if this polygon is behind another.
293// See notes in method "InFrontOf"
294//
295G4bool G4ClippablePolygon::BehindOf( const G4ClippablePolygon &other,
296 EAxis axis ) const
297{
298 //
299 // If things are empty, do something semi-sensible
300 //
301 G4int noLeft = vertices.size();
302 if (noLeft==0) return false;
303
304 if (other.Empty()) return true;
305
306 //
307 // Get minimum of other polygon
308 //
309 const G4ThreeVector *maxPointOther = other.GetMaxPoint( axis );
310 const G4double maxOther = maxPointOther->operator()(axis);
311
312 //
313 // Get minimum of this polygon
314 //
315 const G4ThreeVector *maxPoint = GetMaxPoint( axis );
316 const G4double max = maxPoint->operator()(axis);
317
318 //
319 // Easy decision
320 //
321 if (max > maxOther+kCarTolerance) return true; // Clear winner
322
323 if (maxOther > max+kCarTolerance) return false; // Clear loser
324
325 //
326 // We have a tie (this will not be all that rare since our
327 // polygons are connected)
328 //
329 // Check to see if there is a vertex in the other polygon
330 // that is in front of this one (or vice versa)
331 //
332 G4bool answer;
333 G4ThreeVector normalOther = other.GetNormal();
334
335 if (std::fabs(normalOther(axis)) > std::fabs(normal(axis)))
336 {
337 G4double minP, maxP;
338 GetPlanerExtent( *maxPointOther, normalOther, minP, maxP );
339
340 answer = (normalOther(axis) > 0) ? (maxP > +kCarTolerance)
341 : (minP < -kCarTolerance);
342 }
343 else
344 {
345 G4double minP, maxP;
346 other.GetPlanerExtent( *maxPoint, normal, minP, maxP );
347
348 answer = (normal(axis) > 0) ? (minP < -kCarTolerance)
349 : (maxP > +kCarTolerance);
350 }
351 return answer;
352}

355//
356// GetPlanerExtent
357//
358// Get min/max distance in or out of a plane
359//
360G4bool G4ClippablePolygon::GetPlanerExtent( const G4ThreeVector &pointOnPlane,
361 const G4ThreeVector &planeNormal,
362 G4double &min,
363 G4double &max ) const
364{
365 //
366 // Okay, how many entries do we have?
367 //
368 G4int noLeft = vertices.size();
369
370 //
371 // Return false if nothing is left
372 //
373 if (noLeft == 0) return false;
374
375 //
376 // Initialize min and max to our first vertex
377 //
378 min = max = planeNormal.dot(vertices[0]-pointOnPlane);
379
380 //
381 // Compare to the rest
382 //
383 G4int i;
384 for( i=1; i<noLeft; i++ )
385 {
386 G4double component = planeNormal.dot(vertices[i] - pointOnPlane);
387 if (component < min )
388 min = component;
389 else if (component > max )
390 max = component;
391 }
392
393 return true;
394}

397//
398// Clip along just one axis, as specified in voxelLimit
399//
400void G4ClippablePolygon::ClipAlongOneAxis( const G4VoxelLimits &voxelLimit,
401 const EAxis axis )
402{
403 if (!voxelLimit.IsLimited(axis)) return;
404
405 G4ThreeVectorList tempPolygon;
406
407 //
408 // Build a "simple" voxelLimit that includes only the min extent
409 // and apply this to our vertices, producing result in tempPolygon
410 //
411 G4VoxelLimits simpleLimit1;
412 simpleLimit1.AddLimit( axis, voxelLimit.GetMinExtent(axis), kInfinity );
413 ClipToSimpleLimits( vertices, tempPolygon, simpleLimit1 );
414
415 //
416 // If nothing is left from the above clip, we might as well return now
417 // (but with an empty vertices)
418 //
419 if (tempPolygon.size() == 0)
420 {
421 vertices.clear();
422 return;
423 }
424
425 //
426 // Now do the same, but using a "simple" limit that includes only the max
427 // extent. Apply this to out tempPolygon, producing result in vertices.
428 //
429 G4VoxelLimits simpleLimit2;
430 simpleLimit2.AddLimit( axis, -kInfinity, voxelLimit.GetMaxExtent(axis) );
431 ClipToSimpleLimits( tempPolygon, vertices, simpleLimit2 );
432
433 //
434 // If nothing is left, return now
435 //
436 if (vertices.size() == 0) return;
437}

440//
441// pVoxelLimits must be only limited along one axis, and either the maximum
442// along the axis must be +kInfinity, or the minimum -kInfinity
443//
444void G4ClippablePolygon::ClipToSimpleLimits( G4ThreeVectorList& pPolygon,
445 G4ThreeVectorList& outputPolygon,
446 const G4VoxelLimits& pVoxelLimit )
447{
448 G4int i;
449 G4int noVertices=pPolygon.size();
450 G4ThreeVector vEnd,vStart;
451
452 outputPolygon.clear();
453
454 for (i=0;i<noVertices;i++)
455 {
456 vStart=pPolygon[i];
457 if (i==noVertices-1)
458 {
459 vEnd=pPolygon[0];
460 }
461 else
462 {
463 vEnd=pPolygon[i+1];
464 }
465
466 if (pVoxelLimit.Inside(vStart))
467 {
468 if (pVoxelLimit.Inside(vEnd))
469 {
470 // vStart and vEnd inside -> output end point
471 //
472 outputPolygon.push_back(vEnd);
473 }
474 else
475 {
476 // vStart inside, vEnd outside -> output crossing point
477 //
478 pVoxelLimit.ClipToLimits(vStart,vEnd);
479 outputPolygon.push_back(vEnd);
480 }
481 }
482 else
483 {
484 if (pVoxelLimit.Inside(vEnd))
485 {
486 // vStart outside, vEnd inside -> output inside section
487 //
488 pVoxelLimit.ClipToLimits(vStart,vEnd);
489 outputPolygon.push_back(vStart);
490 outputPolygon.push_back(vEnd);
491 }
492 else // Both point outside -> no output
493 {
494 }
495 }
496 }
