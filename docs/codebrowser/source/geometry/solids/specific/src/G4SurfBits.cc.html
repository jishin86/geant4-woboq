41//______________________________________________________________________________
42G4SurfBits::G4SurfBits(unsigned int nBits) : fNBits(nBits)
43{
44 // G4SurfBits constructor. All bits set to 0
45
46 if (fNBits <= 0) fNBits = 0;
47 fNBytes = fNBits ? ((fNBits-1)/8) + 1 : 1;
48 fAllBits = new unsigned char[fNBytes];
49 // this is redundant only with libNew
50 std::memset(fAllBits,0,fNBytes);
51}
52
53//______________________________________________________________________________
54G4SurfBits::G4SurfBits(const G4SurfBits &original) : fNBits(original.fNBits),
55 fNBytes(original.fNBytes)
56{
57 // G4SurfBits copy constructor
58
59 fAllBits = new unsigned char[fNBytes];
60 std::memcpy(fAllBits,original.fAllBits,fNBytes);
61}
62
63//______________________________________________________________________________
64G4SurfBits& G4SurfBits::operator=(const G4SurfBits& rhs)
65{
66 // G4SurfBits assignment operator
67 if (this != &rhs) {
68 // TObject::operator=(rhs);
69 fNBits = rhs.fNBits;
70 fNBytes = rhs.fNBytes;
71 delete [] fAllBits;
72 if (fNBytes != 0) {
73 fAllBits = new unsigned char[fNBytes];
74 std::memcpy(fAllBits,rhs.fAllBits,fNBytes);
75 } else {
76 fAllBits = 0;
77 }
78 }
79 return *this;
80}
81
82//______________________________________________________________________________
83G4SurfBits::~G4SurfBits()
84{
85 // G4SurfBits destructor
86
87 delete [] fAllBits;
88}

90//______________________________________________________________________________
91void G4SurfBits::Clear()
92{
93 // Clear the value.
94
95 delete [] fAllBits;
96 fAllBits = 0;
97 fNBits = 0;
98 fNBytes = 0;
99}
100
101//______________________________________________________________________________
102void G4SurfBits::Compact()
103{
104 // Reduce the storage used by the object to a minimun
105
106 if (!fNBits || !fAllBits) return;
107 unsigned int needed;
108 for(needed=fNBytes-1;
109 needed > 0 && fAllBits[needed]==0; ) { needed--; };
110 needed++;
111
112 if (needed!=fNBytes) {
113 unsigned char *old_location = fAllBits;
114 fAllBits = new unsigned char[needed];
115
116 std::memcpy(fAllBits,old_location,needed);
117 delete [] old_location;
118
119 fNBytes = needed;
120 fNBits = 8*fNBytes;
121 }
122}

124//______________________________________________________________________________
125void G4SurfBits::Output(std::ostream &os) const
126{
127 // Print the value to the std::ostream
128 for(unsigned int i=0; i<fNBytes; ++i) {
129 unsigned char val = fAllBits[fNBytes - 1 - i];
130 for (unsigned int j=0; j<8; ++j) {
131 os << (G4bool)(val&0x80);
132 val <<= 1;
133 }
134 }
135}
136
137//______________________________________________________________________________
138void G4SurfBits::Print() const
139{
140 // Print the list of active bits
141 G4int count = 0;
142 for(unsigned int i=0; i<fNBytes; ++i) {
143 unsigned char val = fAllBits[i];
144 for (unsigned int j=0; j<8; ++j) {
145 if (val & 1) G4cout << " bit:" << count << " = 1" << G4endl;
146 count++;
147 val = val >> 1;
148 }
149 }
150}

152//______________________________________________________________________________
153void G4SurfBits::ResetAllBits(G4bool value)
154{
155 if (fAllBits) std::memset(fAllBits, value ? 0xFF : 0,fNBytes);
156}
157
158//______________________________________________________________________________
159void G4SurfBits::ReserveBytes(unsigned int nbytes)
160{
161 // Reverse each bytes.
162
163 if (nbytes > fNBytes) {
164 // do it in this order to remain exception-safe.
165 unsigned char *newBits=new unsigned char[nbytes];
166 delete[] fAllBits;
167 fNBytes=nbytes;
168 fAllBits=newBits;
169 }
170}
171
172//______________________________________________________________________________
173void G4SurfBits::set(unsigned int nBits, const char *array)
174{
175 // set all the bytes
176 unsigned int nbytes=(nBits+7)>>3;
177
178 ReserveBytes(nbytes);
179
180 fNBits=nBits;
181 std::memcpy(fAllBits, array, nbytes);
182}
183
184//______________________________________________________________________________
185void G4SurfBits::Get(char *array) const
186{
187 // Copy all the byes.
188 std::memcpy(array, fAllBits, (fNBits+7)>>3);
189}
190
191// If we are on a little endian machine, a bitvector represented using
192// any integer type is identical to a bitvector represented using bytes.
193
194//______________________________________________________________________________
195void G4SurfBits::set(unsigned int nBits, const G4int *array)
196{
197 // set all the bytes.
198
199 set(nBits, (const char*)array);
200}
201
202//______________________________________________________________________________
203void G4SurfBits::Get(G4int *array) const
204{
205 // Get all the bytes.
206
207 Get((char*)array);
208}
