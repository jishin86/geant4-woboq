49G4TwistTubsSide::G4TwistTubsSide(const G4String &name,
50 const G4RotationMatrix &rot,
51 const G4ThreeVector &tlate,
52 G4int handedness,
53 const G4double kappa,
54 const EAxis axis0,
55 const EAxis axis1,
56 G4double axis0min,
57 G4double axis1min,
58 G4double axis0max,
59 G4double axis1max)
60 : G4VTwistSurface(name, rot, tlate, handedness, axis0, axis1,
61 axis0min, axis1min, axis0max, axis1max),
62 fKappa(kappa)
63{
64 if (axis0 == kZAxis && axis1 == kXAxis) {
65 G4Exception("G4TwistTubsSide::G4TwistTubsSide()", "GeomSolids0002",
66 FatalErrorInArgument, "Should swap axis0 and axis1!");
67 }
68 fIsValidNorm = false;
69 SetCorners();
70 SetBoundaries();
71}

73G4TwistTubsSide::G4TwistTubsSide(const G4String &name,
74 G4double EndInnerRadius[2],
75 G4double EndOuterRadius[2],
76 G4double DPhi,
77 G4double EndPhi[2],
78 G4double EndZ[2],
79 G4double InnerRadius,
80 G4double OuterRadius,
81 G4double Kappa,
82 G4int handedness)
83 : G4VTwistSurface(name)
84{
85 fHandedness = handedness; // +z = +ve, -z = -ve
86 fAxis[0] = kXAxis; // in local coordinate system
87 fAxis[1] = kZAxis;
88 fAxisMin[0] = InnerRadius; // Inner-hype radius at z=0
89 fAxisMax[0] = OuterRadius; // Outer-hype radius at z=0
90 fAxisMin[1] = EndZ[0];
91 fAxisMax[1] = EndZ[1];
92
93 fKappa = Kappa;
94 fRot.rotateZ( fHandedness > 0
95 ? -0.5*DPhi
96 : 0.5*DPhi );
97 fTrans.set(0, 0, 0);
98 fIsValidNorm = false;
99
100 SetCorners( EndInnerRadius, EndOuterRadius, EndPhi, EndZ) ;
101 SetBoundaries();
102}

124G4ThreeVector G4TwistTubsSide::GetNormal(const G4ThreeVector &tmpxx,
125 G4bool isGlobal)
126{
127 // GetNormal returns a normal vector at a surface (or very close
128 // to surface) point at tmpxx.
129 // If isGlobal=true, it returns the normal in global coordinate.
130 //
131 G4ThreeVector xx;
132 if (isGlobal) {
133 xx = ComputeLocalPoint(tmpxx);
134 if ((xx - fCurrentNormal.p).mag() < 0.5 * kCarTolerance) {
135 return ComputeGlobalDirection(fCurrentNormal.normal);
136 }
137 } else {
138 xx = tmpxx;
139 if (xx == fCurrentNormal.p) {
140 return fCurrentNormal.normal;
141 }
142 }
143
144 G4ThreeVector er(1, fKappa * xx.z(), 0);
145 G4ThreeVector ez(0, fKappa * xx.x(), 1);
146 G4ThreeVector normal = fHandedness*(er.cross(ez));
147
148 if (isGlobal) {
149 fCurrentNormal.normal = ComputeGlobalDirection(normal.unit());
150 } else {
151 fCurrentNormal.normal = normal.unit();
152 }
153 return fCurrentNormal.normal;
154}

159G4int G4TwistTubsSide::DistanceToSurface(const G4ThreeVector &gp,
160 const G4ThreeVector &gv,
161 G4ThreeVector gxx[],
162 G4double distance[],
163 G4int areacode[],
164 G4bool isvalid[],
165 EValidate validate)
166{
167 // Coordinate system:
168 //
169 // The coordinate system is so chosen that the intersection of
170 // the twisted surface with the z=0 plane coincides with the
171 // x-axis.
172 // Rotation matrix from this coordinate system (local system)
173 // to global system is saved in fRot field.
174 // So the (global) particle position and (global) velocity vectors,
175 // p and v, should be rotated fRot.inverse() in order to convert
176 // to local vectors.
177 //
178 // Equation of a twisted surface:
179 //
180 // x(rho(z=0), z) = rho(z=0)
181 // y(rho(z=0), z) = rho(z=0)*K*z
182 // z(rho(z=0), z) = z
183 // with
184 // K = std::tan(fPhiTwist/2)/fZHalfLen
185 //
186 // Equation of a line:
187 //
188 // gxx = p + t*v
189 // with
190 // p = fRot.inverse()*gp
191 // v = fRot.inverse()*gv
192 //
193 // Solution for intersection:
194 //
195 // Required time for crossing is given by solving the
196 // following quadratic equation:
197 //
198 // a*t^2 + b*t + c = 0
199 //
200 // where
201 //
202 // a = K*v_x*v_z
203 // b = K*(v_x*p_z + v_z*p_x) - v_y
204 // c = K*p_x*p_z - p_y
205 //
206 // Out of the possible two solutions you must choose
207 // the one that gives a positive rho(z=0).
208 //
209 //
210
211 fCurStatWithV.ResetfDone(validate, &gp, &gv);
212
213 if (fCurStatWithV.IsDone()) {
214 G4int i;
215 for (i=0; i<fCurStatWithV.GetNXX(); i++) {
216 gxx[i] = fCurStatWithV.GetXX(i);
217 distance[i] = fCurStatWithV.GetDistance(i);
218 areacode[i] = fCurStatWithV.GetAreacode(i);
219 isvalid[i] = fCurStatWithV.IsValid(i);
220 }
221 return fCurStatWithV.GetNXX();
222 } else {
223 // initialize
224 G4int i;
225 for (i=0; i<2; i++) {
226 distance[i] = kInfinity;
227 areacode[i] = sOutside;
228 isvalid[i] = false;
229 gxx[i].set(kInfinity, kInfinity, kInfinity);
230 }
231 }
232
233 G4ThreeVector p = ComputeLocalPoint(gp);
234 G4ThreeVector v = ComputeLocalDirection(gv);
235 G4ThreeVector xx[2];
236
237
238 //
239 // special case!
240 // p is origin or
241 //
242
243 G4double absvz = std::fabs(v.z());
244
245 if ((absvz < DBL_MIN) && (std::fabs(p.x() * v.y() - p.y() * v.x()) < DBL_MIN)) {
246 // no intersection
247
248 isvalid[0] = false;
249 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
250 isvalid[0], 0, validate, &gp, &gv);
251 return 0;
252 }
253
254 //
255 // special case end
256 //
257
258
259 G4double a = fKappa * v.x() * v.z();
260 G4double b = fKappa * (v.x() * p.z() + v.z() * p.x()) - v.y();
261 G4double c = fKappa * p.x() * p.z() - p.y();
262 G4double D = b * b - 4 * a * c; // discriminant
263 G4int vout = 0;
264
265 if (std::fabs(a) < DBL_MIN) {
266 if (std::fabs(b) > DBL_MIN) {
267
268 // single solution
269
270 distance[0] = - c / b;
271 xx[0] = p + distance[0]*v;
272 gxx[0] = ComputeGlobalPoint(xx[0]);
273
274 if (validate == kValidateWithTol) {
275 areacode[0] = GetAreaCode(xx[0]);
276 if (!IsOutside(areacode[0])) {
277 if (distance[0] >= 0) isvalid[0] = true;
278 }
279 } else if (validate == kValidateWithoutTol) {
280 areacode[0] = GetAreaCode(xx[0], false);
281 if (IsInside(areacode[0])) {
282 if (distance[0] >= 0) isvalid[0] = true;
283 }
284 } else { // kDontValidate
285 // we must omit x(rho,z) = rho(z=0) < 0
286 if (xx[0].x() > 0) {
287 areacode[0] = sInside;
288 if (distance[0] >= 0) isvalid[0] = true;
289 } else {
290 distance[0] = kInfinity;
291 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0],
292 areacode[0], isvalid[0],
293 0, validate, &gp, &gv);
294 return vout;
295 }
296 }
297
298 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
299 isvalid[0], 1, validate, &gp, &gv);
300 vout = 1;
301
302 } else {
303 // if a=b=0 , v.y=0 and (v.x=0 && p.x=0) or (v.z=0 && p.z=0) .
304 // if v.x=0 && p.x=0, no intersection unless p is on z-axis
305 // (in that case, v is paralell to surface).
306 // if v.z=0 && p.z=0, no intersection unless p is on x-axis
307 // (in that case, v is paralell to surface).
308 // return distance = infinity.
309
310 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
311 isvalid[0], 0, validate, &gp, &gv);
312 }
313
314 } else if (D > DBL_MIN) {
315
316 // double solutions
317
318 D = std::sqrt(D);
319 G4double factor = 0.5/a;
320 G4double tmpdist[2] = {kInfinity, kInfinity};
321 G4ThreeVector tmpxx[2];
322 G4int tmpareacode[2] = {sOutside, sOutside};
323 G4bool tmpisvalid[2] = {false, false};
324 G4int i;
325
326 for (i=0; i<2; i++) {
327 G4double bminusD = - b - D;
328
329 // protection against round off error
330 //G4double protection = 1.0e-6;
331 G4double protection = 0;
332 if ( b * D < 0 && std::fabs(bminusD / D) < protection ) {
333 G4double acovbb = (a*c)/(b*b);
334 tmpdist[i] = - c/b * ( 1 - acovbb * (1 + 2*acovbb));
335 } else {
336 tmpdist[i] = factor * bminusD;
337 }
338
339 D = -D;
340 tmpxx[i] = p + tmpdist[i]*v;
341
342 if (validate == kValidateWithTol) {
343 tmpareacode[i] = GetAreaCode(tmpxx[i]);
344 if (!IsOutside(tmpareacode[i])) {
345 if (tmpdist[i] >= 0) tmpisvalid[i] = true;
346 continue;
347 }
348 } else if (validate == kValidateWithoutTol) {
349 tmpareacode[i] = GetAreaCode(tmpxx[i], false);
350 if (IsInside(tmpareacode[i])) {
351 if (tmpdist[i] >= 0) tmpisvalid[i] = true;
352 continue;
353 }
354 } else { // kDontValidate
355 // we must choose x(rho,z) = rho(z=0) > 0
356 if (tmpxx[i].x() > 0) {
357 tmpareacode[i] = sInside;
358 if (tmpdist[i] >= 0) tmpisvalid[i] = true;
359 continue;
360 } else {
361 tmpdist[i] = kInfinity;
362 continue;
363 }
364 }
365 }
366
367 if (tmpdist[0] <= tmpdist[1]) {
368 distance[0] = tmpdist[0];
369 distance[1] = tmpdist[1];
370 xx[0] = tmpxx[0];
371 xx[1] = tmpxx[1];
372 gxx[0] = ComputeGlobalPoint(tmpxx[0]);
373 gxx[1] = ComputeGlobalPoint(tmpxx[1]);
374 areacode[0] = tmpareacode[0];
375 areacode[1] = tmpareacode[1];
376 isvalid[0] = tmpisvalid[0];
377 isvalid[1] = tmpisvalid[1];
378 } else {
379 distance[0] = tmpdist[1];
380 distance[1] = tmpdist[0];
381 xx[0] = tmpxx[1];
382 xx[1] = tmpxx[0];
383 gxx[0] = ComputeGlobalPoint(tmpxx[1]);
384 gxx[1] = ComputeGlobalPoint(tmpxx[0]);
385 areacode[0] = tmpareacode[1];
386 areacode[1] = tmpareacode[0];
387 isvalid[0] = tmpisvalid[1];
388 isvalid[1] = tmpisvalid[0];
389 }
390
391 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
392 isvalid[0], 2, validate, &gp, &gv);
393 fCurStatWithV.SetCurrentStatus(1, gxx[1], distance[1], areacode[1],
394 isvalid[1], 2, validate, &gp, &gv);
395
396 // protection against roundoff error
397
398 for (G4int k=0; k<2; k++) {
399 if (!isvalid[k]) continue;
400
401 G4ThreeVector xxonsurface(xx[k].x(), fKappa * std::fabs(xx[k].x())
402 * xx[k].z() , xx[k].z());
403 G4double deltaY = (xx[k] - xxonsurface).mag();
404
405 if ( deltaY > 0.5*kCarTolerance ) {
406
407 G4int maxcount = 10;
408 G4int l;
409 G4double lastdeltaY = deltaY;
410 for (l=0; l<maxcount; l++) {
411 G4ThreeVector surfacenormal = GetNormal(xxonsurface);
412 distance[k] = DistanceToPlaneWithV(p, v, xxonsurface,
413 surfacenormal, xx[k]);
414 deltaY = (xx[k] - xxonsurface).mag();
415 if (deltaY > lastdeltaY) {
416
417 }
418 gxx[k] = ComputeGlobalPoint(xx[k]);
419
420 if (deltaY <= 0.5*kCarTolerance) {
421
422 break;
423 }
424 xxonsurface.set(xx[k].x(),
425 fKappa * std::fabs(xx[k].x()) * xx[k].z(),
426 xx[k].z());
427 }
428 if (l == maxcount) {
429 std::ostringstream message;
430 message << "Exceeded maxloop count!" << G4endl
431 << " maxloop count " << maxcount;
432 G4Exception("G4TwistTubsFlatSide::DistanceToSurface(p,v)",
433 "GeomSolids0003", FatalException, message);
434 }
435 }
436
437 }
438 vout = 2;
439 } else {
440 // if D<0, no solution
441 // if D=0, just grazing the surfaces, return kInfinity
442
443 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
444 isvalid[0], 0, validate, &gp, &gv);
445 }
446
447 return vout;
448}

452
453G4int G4TwistTubsSide::DistanceToSurface(const G4ThreeVector &gp,
454 G4ThreeVector gxx[],
455 G4double distance[],
456 G4int areacode[])
457{
458 fCurStat.ResetfDone(kDontValidate, &gp);
459 G4int i = 0;
460 if (fCurStat.IsDone()) {
461 for (i=0; i<fCurStat.GetNXX(); i++) {
462 gxx[i] = fCurStat.GetXX(i);
463 distance[i] = fCurStat.GetDistance(i);
464 areacode[i] = fCurStat.GetAreacode(i);
465 }
466 return fCurStat.GetNXX();
467 } else {
468 // initialize
469 for (i=0; i<2; i++) {
470 distance[i] = kInfinity;
471 areacode[i] = sOutside;
472 gxx[i].set(kInfinity, kInfinity, kInfinity);
473 }
474 }
475
476 const G4double halftol = 0.5 * kCarTolerance;
477
478 G4ThreeVector p = ComputeLocalPoint(gp);
479 G4ThreeVector xx;
480 G4int parity = (fKappa >= 0 ? 1 : -1);
481
482 //
483 // special case!
484 // If p is on surface, or
485 // p is on z-axis,
486 // return here immediatery.
487 //
488
489 G4ThreeVector lastgxx[2];
490 for (i=0; i<2; i++) {
491 lastgxx[i] = fCurStatWithV.GetXX(i);
492 }
493
494 if ((gp - lastgxx[0]).mag() < halftol
495 || (gp - lastgxx[1]).mag() < halftol) {
496 // last winner, or last poststep point is on the surface.
497 xx = p;
498 distance[0] = 0;
499 gxx[0] = gp;
500
501 G4bool isvalid = true;
502 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
503 isvalid, 1, kDontValidate, &gp);
504 return 1;
505 }
506
507 if (p.getRho() == 0) {
508 // p is on z-axis. Namely, p is on twisted surface (invalid area).
509 // We must return here, however, returning distance to x-minimum
510 // boundary is better than return 0-distance.
511 //
512 G4bool isvalid = true;
513 if (fAxis[0] == kXAxis && fAxis[1] == kZAxis) {
514 distance[0] = DistanceToBoundary(sAxis0 & sAxisMin, xx, p);
515 areacode[0] = sInside;
516 } else {
517 distance[0] = 0;
518 xx.set(0., 0., 0.);
519 }
520 gxx[0] = ComputeGlobalPoint(xx);
521 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
522 isvalid, 0, kDontValidate, &gp);
523 return 1;
524 }
525
526 //
527 // special case end
528 //
529
530 // set corner points of quadrangle try area ...
531
532 G4ThreeVector A; // foot of normal from p to boundary of sAxis0 & sAxisMin
533 G4ThreeVector C; // foot of normal from p to boundary of sAxis0 & sAxisMax
534 G4ThreeVector B; // point on boundary sAxis0 & sAxisMax at z = A.z()
535 G4ThreeVector D; // point on boundary sAxis0 & sAxisMin at z = C.z()
536
537 // G4double distToA; // distance from p to A
538 DistanceToBoundary(sAxis0 & sAxisMin, A, p);
539 // G4double distToC; // distance from p to C
540 DistanceToBoundary(sAxis0 & sAxisMax, C, p);
541
542 // is p.z between a.z and c.z?
543 // p.z must be bracketed a.z and c.z.
544 if (A.z() > C.z()) {
545 if (p.z() > A.z()) {
546 A = GetBoundaryAtPZ(sAxis0 & sAxisMin, p);
547 } else if (p.z() < C.z()) {
548 C = GetBoundaryAtPZ(sAxis0 & sAxisMax, p);
549 }
550 } else {
551 if (p.z() > C.z()) {
552 C = GetBoundaryAtPZ(sAxis0 & sAxisMax, p);
553 } else if (p.z() < A.z()) {
554 A = GetBoundaryAtPZ(sAxis0 & sAxisMin, p);
555 }
556 }
557
558 G4ThreeVector d[2]; // direction vectors of boundary
559 G4ThreeVector x0[2]; // foot of normal from line to p
560 G4int btype[2]; // boundary type
561
562 for (i=0; i<2; i++) {
563 if (i == 0) {
564 GetBoundaryParameters((sAxis0 & sAxisMax), d[i], x0[i], btype[i]);
565 B = x0[i] + ((A.z() - x0[i].z()) / d[i].z()) * d[i];
566 // x0 + t*d , d is direction unit vector.
567 } else {
568 GetBoundaryParameters((sAxis0 & sAxisMin), d[i], x0[i], btype[i]);
569 D = x0[i] + ((C.z() - x0[i].z()) / d[i].z()) * d[i];
570 }
571 }
572
573 // In order to set correct diagonal, swap A and D, C and B if needed.
574 G4ThreeVector pt(p.x(), p.y(), 0.);
575 G4double rc = std::fabs(p.x());
576 G4ThreeVector surfacevector(rc, rc * fKappa * p.z(), 0.);
577 G4int pside = AmIOnLeftSide(pt, surfacevector);
578 G4double test = (A.z() - C.z()) * parity * pside;
579
580 if (test == 0) {
581 if (pside == 0) {
582 // p is on surface.
583 xx = p;
584 distance[0] = 0;
585 gxx[0] = gp;
586
587 G4bool isvalid = true;
588 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
589 isvalid, 1, kDontValidate, &gp);
590 return 1;
591 } else {
592 // A.z = C.z(). return distance to line.
593 d[0] = C - A;
594 distance[0] = DistanceToLine(p, A, d[0], xx);
595 areacode[0] = sInside;
596 gxx[0] = ComputeGlobalPoint(xx);
597 G4bool isvalid = true;
598 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
599 isvalid, 1, kDontValidate, &gp);
600 return 1;
601 }
602
603 } else if (test < 0) {
604
605 // wrong diagonal. vector AC is crossing the surface!
606 // swap A and D, C and B
607 G4ThreeVector tmp;
608 tmp = A;
609 A = D;
610 D = tmp;
611 tmp = C;
612 C = B;
613 B = tmp;
614
615 } else {
616 // correct diagonal. nothing to do.
617 }
618
619 // Now, we chose correct diaglnal.
620 // First try. divide quadrangle into double triangle by diagonal and
621 // calculate distance to both surfaces.
622
623 G4ThreeVector xxacb; // foot of normal from plane ACB to p
624 G4ThreeVector nacb; // normal of plane ACD
625 G4ThreeVector xxcad; // foot of normal from plane CAD to p
626 G4ThreeVector ncad; // normal of plane CAD
627 G4ThreeVector AB(A.x(), A.y(), 0);
628 G4ThreeVector DC(C.x(), C.y(), 0);
629
630 G4double distToACB = G4VTwistSurface::DistanceToPlane(p, A, C-A, AB, xxacb, nacb) * parity;
631 G4double distToCAD = G4VTwistSurface::DistanceToPlane(p, C, C-A, DC, xxcad, ncad) * parity;
632
633 // if calculated distance = 0, return
634
635 if (std::fabs(distToACB) <= halftol || std::fabs(distToCAD) <= halftol) {
636 xx = (std::fabs(distToACB) < std::fabs(distToCAD) ? xxacb : xxcad);
637 areacode[0] = sInside;
638 gxx[0] = ComputeGlobalPoint(xx);
639 distance[0] = 0;
640 G4bool isvalid = true;
641 fCurStat.SetCurrentStatus(0, gxx[0], distance[0] , areacode[0],
642 isvalid, 1, kDontValidate, &gp);
643 return 1;
644 }
645
646 if (distToACB * distToCAD > 0 && distToACB < 0) {
647 // both distToACB and distToCAD are negative.
648 // divide quadrangle into double triangle by diagonal
649 G4ThreeVector normal;
650 distance[0] = DistanceToPlane(p, A, B, C, D, parity, xx, normal);
651 } else {
652 if (distToACB * distToCAD > 0) {
653 // both distToACB and distToCAD are positive.
654 // Take smaller one.
655 if (distToACB <= distToCAD) {
656 distance[0] = distToACB;
657 xx = xxacb;
658 } else {
659 distance[0] = distToCAD;
660 xx = xxcad;
661 }
662 } else {
663 // distToACB * distToCAD is negative.
664 // take positive one
665 if (distToACB > 0) {
666 distance[0] = distToACB;
667 xx = xxacb;
668 } else {
669 distance[0] = distToCAD;
670 xx = xxcad;
671 }
672 }
673
674 }
675 areacode[0] = sInside;
676 gxx[0] = ComputeGlobalPoint(xx);
677 G4bool isvalid = true;
678 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
679 isvalid, 1, kDontValidate, &gp);
680 return 1;

683//=====================================================================
684//* DistanceToPlane ---------------------------------------------------
685
686G4double G4TwistTubsSide::DistanceToPlane(const G4ThreeVector &p,
687 const G4ThreeVector &A,
688 const G4ThreeVector &B,
689 const G4ThreeVector &C,
690 const G4ThreeVector &D,
691 const G4int parity,
692 G4ThreeVector &xx,
693 G4ThreeVector &n)
694{
695 const G4double halftol = 0.5 * kCarTolerance;
696
697 G4ThreeVector M = 0.5*(A + B);
698 G4ThreeVector N = 0.5*(C + D);
699 G4ThreeVector xxanm; // foot of normal from p to plane ANM
700 G4ThreeVector nanm; // normal of plane ANM
701 G4ThreeVector xxcmn; // foot of normal from p to plane CMN
702 G4ThreeVector ncmn; // normal of plane CMN
703
704 G4double distToanm = G4VTwistSurface::DistanceToPlane(p, A, (N - A), (M - A), xxanm, nanm) * parity;
705 G4double distTocmn = G4VTwistSurface::DistanceToPlane(p, C, (M - C), (N - C), xxcmn, ncmn) * parity;
706
707 // if p is behind of both surfaces, abort.
708 if (distToanm * distTocmn > 0 && distToanm < 0) {
709 G4Exception("G4TwistTubsSide::DistanceToPlane()",
710 "GeomSolids0003", FatalException,
711 "Point p is behind the surfaces.");
712 }
713
714 // if p is on surface, return 0.
715 if (std::fabs(distToanm) <= halftol) {
716 xx = xxanm;
717 n = nanm * parity;
718 return 0;
719 } else if (std::fabs(distTocmn) <= halftol) {
720 xx = xxcmn;
721 n = ncmn * parity;
722 return 0;
723 }
724
725 if (distToanm <= distTocmn) {
726 if (distToanm > 0) {
727 // both distanses are positive. take smaller one.
728 xx = xxanm;
729 n = nanm * parity;
730 return distToanm;
731 } else {
732 // take -ve distance and call the function recursively.
733 return DistanceToPlane(p, A, M, N, D, parity, xx, n);
734 }
735 } else {
736 if (distTocmn > 0) {
737 // both distanses are positive. take smaller one.
738 xx = xxcmn;
739 n = ncmn * parity;
740 return distTocmn;
741 } else {
742 // take -ve distance and call the function recursively.
743 return DistanceToPlane(p, C, N, M, B, parity, xx, n);
744 }
745 }
746}

748//=====================================================================
749//* GetAreaCode -------------------------------------------------------
750
751G4int G4TwistTubsSide::GetAreaCode(const G4ThreeVector &xx,
752 G4bool withTol)
753{
754 // We must use the function in local coordinate system.
755 // See the description of DistanceToSurface(p,v).
756
757 const G4double ctol = 0.5 * kCarTolerance;
758 G4int areacode = sInside;
759
760 if (fAxis[0] == kXAxis && fAxis[1] == kZAxis) {
761 G4int xaxis = 0;
762 G4int zaxis = 1;
763
764 if (withTol) {
765
766 G4bool isoutside = false;
767
768 // test boundary of xaxis
769
770 if (xx.x() < fAxisMin[xaxis] + ctol) {
771 areacode |= (sAxis0 & (sAxisX | sAxisMin)) | sBoundary;
772 if (xx.x() <= fAxisMin[xaxis] - ctol) isoutside = true;
773
774 } else if (xx.x() > fAxisMax[xaxis] - ctol) {
775 areacode |= (sAxis0 & (sAxisX | sAxisMax)) | sBoundary;
776 if (xx.x() >= fAxisMax[xaxis] + ctol) isoutside = true;
777 }
778
779 // test boundary of z-axis
780
781 if (xx.z() < fAxisMin[zaxis] + ctol) {
782 areacode |= (sAxis1 & (sAxisZ | sAxisMin));
783
784 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
785 else areacode |= sBoundary;
786 if (xx.z() <= fAxisMin[zaxis] - ctol) isoutside = true;
787
788 } else if (xx.z() > fAxisMax[zaxis] - ctol) {
789 areacode |= (sAxis1 & (sAxisZ | sAxisMax));
790
791 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
792 else areacode |= sBoundary;
793 if (xx.z() >= fAxisMax[zaxis] + ctol) isoutside = true;
794 }
795
796 // if isoutside = true, clear inside bit.
797 // if not on boundary, add axis information.
798
799 if (isoutside) {
800 G4int tmpareacode = areacode & (~sInside);
801 areacode = tmpareacode;
802 } else if ((areacode & sBoundary) != sBoundary) {
803 areacode |= (sAxis0 & sAxisX) | (sAxis1 & sAxisZ);
804 }
805
806 } else {
807
808 // boundary of x-axis
809
810 if (xx.x() < fAxisMin[xaxis] ) {
811 areacode |= (sAxis0 & (sAxisX | sAxisMin)) | sBoundary;
812 } else if (xx.x() > fAxisMax[xaxis]) {
813 areacode |= (sAxis0 & (sAxisX | sAxisMax)) | sBoundary;
814 }
815
816 // boundary of z-axis
817
818 if (xx.z() < fAxisMin[zaxis]) {
819 areacode |= (sAxis1 & (sAxisZ | sAxisMin));
820 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
821 else areacode |= sBoundary;
822
823 } else if (xx.z() > fAxisMax[zaxis]) {
824 areacode |= (sAxis1 & (sAxisZ | sAxisMax)) ;
825 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
826 else areacode |= sBoundary;
827 }
828
829 if ((areacode & sBoundary) != sBoundary) {
830 areacode |= (sAxis0 & sAxisX) | (sAxis1 & sAxisZ);
831 }
832 }
833 return areacode;
834 } else {
835 G4Exception("G4TwistTubsSide::GetAreaCode()",
836 "GeomSolids0001", FatalException,
837 "Feature NOT implemented !");
838 }
839 return areacode;
840}

951
952void G4TwistTubsSide::GetFacets( G4int k, G4int n, G4double xyz[][3],
953 G4int faces[][4], G4int iside )
954{
955
956 G4double z ; // the two parameters for the surface equation
957 G4double x,xmin,xmax ;
958
959 G4ThreeVector p ; // a point on the surface, given by (z,u)
960
961 G4int nnode ;
962 G4int nface ;
963
964 // calculate the (n-1)*(k-1) vertices
965
966 G4int i,j ;
967
968 for ( i = 0 ; i<n ; i++ )
969 {
970
971 z = fAxisMin[1] + i*(fAxisMax[1]-fAxisMin[1])/(n-1) ;
972
973 for ( j = 0 ; j<k ; j++ ) {
974
975 nnode = GetNode(i,j,k,n,iside) ;
976
977 xmin = GetBoundaryMin(z) ;
978 xmax = GetBoundaryMax(z) ;
979
980 if (fHandedness < 0) {
981 x = xmin + j*(xmax-xmin)/(k-1) ;
982 } else {
983 x = xmax - j*(xmax-xmin)/(k-1) ;
984 }
985
986 p = SurfacePoint(x,z,true) ; // surface point in global coord.system
987
988 xyz[nnode][0] = p.x() ;
989 xyz[nnode][1] = p.y() ;
990 xyz[nnode][2] = p.z() ;
991
992 if ( i<n-1 && j<k-1 ) { // clock wise filling
993
994 nface = GetFace(i,j,k,n,iside) ;
995
996 faces[nface][0] = GetEdgeVisibility(i,j,k,n,0,1) * ( GetNode(i ,j ,k,n,iside)+1) ;
997 faces[nface][1] = GetEdgeVisibility(i,j,k,n,1,1) * ( GetNode(i+1,j ,k,n,iside)+1) ;
998 faces[nface][2] = GetEdgeVisibility(i,j,k,n,2,1) * ( GetNode(i+1,j+1,k,n,iside)+1) ;
999 faces[nface][3] = GetEdgeVisibility(i,j,k,n,3,1) * ( GetNode(i ,j+1,k,n,iside)+1) ;
1000
1001 }
1002 }
1003 }
