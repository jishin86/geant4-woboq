39#include "G4PolyconeHistorical.hh"
40
41G4PolyconeHistorical::G4PolyconeHistorical()
42 : Start_angle(0.), Opening_angle(0.), Num_z_planes(0),
43 Z_values(0), Rmin(0), Rmax(0)
44{
45}
46
47G4PolyconeHistorical::
48G4PolyconeHistorical( G4int z_planes )
49 : Start_angle(0.), Opening_angle(0.), Num_z_planes(z_planes)
50{
51 Z_values = new G4double[z_planes];
52 Rmin = new G4double[z_planes];
53 Rmax = new G4double[z_planes];
54
55 for( G4int i = 0; i < z_planes; i++)
56 {
57 Z_values[i] = 0.0;
58 Rmin[i] = 0.0;
59 Rmax[i] = 0.0;
60 }
61}
62
63G4PolyconeHistorical::~G4PolyconeHistorical()
64{
65 delete [] Z_values;
66 delete [] Rmin;
67 delete [] Rmax;
68}
69
70G4PolyconeHistorical::
71G4PolyconeHistorical( const G4PolyconeHistorical &source )
72{
73 Start_angle = source.Start_angle;
74 Opening_angle = source.Opening_angle;
75 Num_z_planes = source.Num_z_planes;
76
77 Z_values = new G4double[Num_z_planes];
78 Rmin = new G4double[Num_z_planes];
79 Rmax = new G4double[Num_z_planes];
80
81 for( G4int i = 0; i < Num_z_planes; i++)
82 {
83 Z_values[i] = source.Z_values[i];
84 Rmin[i] = source.Rmin[i];
85 Rmax[i] = source.Rmax[i];
86 }
87}
88
89G4PolyconeHistorical&
90G4PolyconeHistorical::operator=( const G4PolyconeHistorical& right )
91{
92 if ( &right == this ) return *this;
93
94 Start_angle = right.Start_angle;
95 Opening_angle = right.Opening_angle;
96 Num_z_planes = right.Num_z_planes;
97
98 delete [] Z_values;
99 delete [] Rmin;
100 delete [] Rmax;
101 Z_values = new G4double[Num_z_planes];
102 Rmin = new G4double[Num_z_planes];
103 Rmax = new G4double[Num_z_planes];
104
105 for( G4int i = 0; i < Num_z_planes; i++)
106 {
107 Z_values[i] = right.Z_values[i];
108 Rmin[i] = right.Rmin[i];
109 Rmax[i] = right.Rmax[i];
110 }
111
112 return *this;
113}
