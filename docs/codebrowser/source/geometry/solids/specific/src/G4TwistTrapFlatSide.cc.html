45G4TwistTrapFlatSide::G4TwistTrapFlatSide( const G4String &name,
46 G4double PhiTwist,
47 G4double pDx1,
48 G4double pDx2,
49 G4double pDy,
50 G4double pDz,
51 G4double pAlpha,
52 G4double pPhi,
53 G4double pTheta,
54 G4int handedness)
55
56 : G4VTwistSurface(name)
57{
58 fHandedness = handedness; // +z = +ve, -z = -ve
59
60 fDx1 = pDx1 ;
61 fDx2 = pDx2 ;
62 fDy = pDy ;
63 fDz = pDz ;
64 fAlpha = pAlpha ;
65 fTAlph = std::tan(fAlpha) ;
66 fPhi = pPhi ;
67 fTheta = pTheta ;
68
69 fdeltaX = 2 * fDz * std::tan(fTheta) * std::cos(fPhi) ;
70 // dx in surface equation
71 fdeltaY = 2 * fDz * std::tan(fTheta) * std::sin(fPhi) ;
72 // dy in surface equation
73
74 fPhiTwist = PhiTwist ;
75
76 fCurrentNormal.normal.set( 0, 0, (fHandedness < 0 ? -1 : 1));
77 // Unit vector, in local coordinate system
78 fRot.rotateZ( fHandedness > 0
79 ? 0.5 * fPhiTwist
80 : -0.5 * fPhiTwist );
81
82 fTrans.set(
83 fHandedness > 0 ? 0.5*fdeltaX : -0.5*fdeltaX ,
84 fHandedness > 0 ? 0.5*fdeltaY : -0.5*fdeltaY ,
85 fHandedness > 0 ? fDz : -fDz ) ;
86
87 fIsValidNorm = true;
88
89
90 fAxis[0] = kXAxis ;
91 fAxis[1] = kYAxis ;
92 fAxisMin[0] = kInfinity ; // x-Axis cannot be fixed, because it
93 fAxisMax[0] = kInfinity ; // depends on y
94 fAxisMin[1] = -fDy ; // y - axis
95 fAxisMax[1] = fDy ;
96
97 SetCorners();
98 SetBoundaries();
99}

133//* DistanceToSurface(p, v) -------------------------------------------
134
135G4int G4TwistTrapFlatSide::DistanceToSurface(const G4ThreeVector &gp,
136 const G4ThreeVector &gv,
137 G4ThreeVector gxx[],
138 G4double distance[],
139 G4int areacode[],
140 G4bool isvalid[],
141 EValidate validate)
142{
143 fCurStatWithV.ResetfDone(validate, &gp, &gv);
144
145 if (fCurStatWithV.IsDone()) {
146 G4int i;
147 for (i=0; i<fCurStatWithV.GetNXX(); i++) {
148 gxx[i] = fCurStatWithV.GetXX(i);
149 distance[i] = fCurStatWithV.GetDistance(i);
150 areacode[i] = fCurStatWithV.GetAreacode(i);
151 isvalid[i] = fCurStatWithV.IsValid(i);
152 }
153 return fCurStatWithV.GetNXX();
154 } else {
155 // initialize
156 G4int i;
157 for (i=0; i<2; i++) {
158 distance[i] = kInfinity;
159 areacode[i] = sOutside;
160 isvalid[i] = false;
161 gxx[i].set(kInfinity, kInfinity, kInfinity);
162 }
163 }
164
165 G4ThreeVector p = ComputeLocalPoint(gp);
166 G4ThreeVector v = ComputeLocalDirection(gv);
167
168 //
169 // special case!
170 // if p is on surface, distance = 0.
171 //
172
173 if (std::fabs(p.z()) == 0.) { // if p is on the plane
174 distance[0] = 0;
175 G4ThreeVector xx = p;
176 gxx[0] = ComputeGlobalPoint(xx);
177
178 if (validate == kValidateWithTol) {
179 areacode[0] = GetAreaCode(xx);
180 if (!IsOutside(areacode[0])) {
181 isvalid[0] = true;
182 }
183 } else if (validate == kValidateWithoutTol) {
184 areacode[0] = GetAreaCode(xx, false);
185 if (IsInside(areacode[0])) {
186 isvalid[0] = true;
187 }
188 } else { // kDontValidate
189 areacode[0] = sInside;
190 isvalid[0] = true;
191 }
192
193 return 1;
194 }
195 //
196 // special case end
197 //
198
199 if (v.z() == 0) {
200
201 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
202 isvalid[0], 0, validate, &gp, &gv);
203 return 0;
204 }
205
206 distance[0] = - (p.z() / v.z());
207
208 G4ThreeVector xx = p + distance[0]*v;
209 gxx[0] = ComputeGlobalPoint(xx);
210
211 if (validate == kValidateWithTol) {
212 areacode[0] = GetAreaCode(xx);
213 if (!IsOutside(areacode[0])) {
214 if (distance[0] >= 0) isvalid[0] = true;
215 }
216 } else if (validate == kValidateWithoutTol) {
217 areacode[0] = GetAreaCode(xx, false);
218 if (IsInside(areacode[0])) {
219 if (distance[0] >= 0) isvalid[0] = true;
220 }
221 } else { // kDontValidate
222 areacode[0] = sInside;
223 if (distance[0] >= 0) isvalid[0] = true;
224 }
225
226
227 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
228 isvalid[0], 1, validate, &gp, &gv);
229
230#ifdef G4TWISTDEBUG
231 G4cerr << "ERROR - G4TwistTrapFlatSide::DistanceToSurface(p,v)" << G4endl;
232 G4cerr << " Name : " << GetName() << G4endl;
233 G4cerr << " xx : " << xx << G4endl;
234 G4cerr << " gxx[0] : " << gxx[0] << G4endl;
235 G4cerr << " dist[0] : " << distance[0] << G4endl;
236 G4cerr << " areacode[0] : " << areacode[0] << G4endl;
237 G4cerr << " isvalid[0] : " << isvalid[0] << G4endl;
238#endif
239 return 1;
240}

243//* DistanceToSurface(p) ----------------------------------------------
244
245G4int G4TwistTrapFlatSide::DistanceToSurface(const G4ThreeVector &gp,
246 G4ThreeVector gxx[],
247 G4double distance[],
248 G4int areacode[])
249{
250 // Calculate distance to plane in local coordinate,
251 // then return distance and global intersection points.
252 //
253
254 fCurStat.ResetfDone(kDontValidate, &gp);
255
256 if (fCurStat.IsDone()) {
257 G4int i;
258 for (i=0; i<fCurStat.GetNXX(); i++) {
259 gxx[i] = fCurStat.GetXX(i);
260 distance[i] = fCurStat.GetDistance(i);
261 areacode[i] = fCurStat.GetAreacode(i);
262 }
263 return fCurStat.GetNXX();
264 } else {
265 // initialize
266 G4int i;
267 for (i=0; i<2; i++) {
268 distance[i] = kInfinity;
269 areacode[i] = sOutside;
270 gxx[i].set(kInfinity, kInfinity, kInfinity);
271 }
272 }
273
274 G4ThreeVector p = ComputeLocalPoint(gp);
275 G4ThreeVector xx;
276
277 // The plane is placed on origin with making its normal
278 // parallel to z-axis.
279 if (std::fabs(p.z()) <= 0.5 * kCarTolerance)
280 { // if p is on the plane, return 1
281 distance[0] = 0;
282 xx = p;
283 } else {
284 distance[0] = std::fabs(p.z());
285 xx.set(p.x(), p.y(), 0);
286 }
287
288 gxx[0] = ComputeGlobalPoint(xx);
289 areacode[0] = sInside;
290 G4bool isvalid = true;
291 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
292 isvalid, 1, kDontValidate, &gp);
293 return 1;
294

297G4int G4TwistTrapFlatSide::GetAreaCode(const G4ThreeVector &xx,
298 G4bool withTol)
299{
300
301 static const G4double ctol = 0.5 * kCarTolerance;
302 G4int areacode = sInside;
303
304 if (fAxis[0] == kXAxis && fAxis[1] == kYAxis) {
305
306 G4int yaxis = 1;
307
308 G4double wmax = xAxisMax(xx.y(), fTAlph ) ;
309 G4double wmin = -xAxisMax(xx.y(), -fTAlph ) ;
310
311 if (withTol) {
312
313 G4bool isoutside = false;
314
315 // test boundary of x-axis
316
317 if (xx.x() < wmin + ctol) {
318 areacode |= (sAxis0 & (sAxisX | sAxisMin)) | sBoundary;
319 if (xx.x() <= wmin - ctol) isoutside = true;
320
321 } else if (xx.x() > wmax - ctol) {
322 areacode |= (sAxis0 & (sAxisX | sAxisMax)) | sBoundary;
323 if (xx.x() >= wmax + ctol) isoutside = true;
324 }
325
326 // test boundary of y-axis
327
328 if (xx.y() < fAxisMin[yaxis] + ctol) {
329 areacode |= (sAxis1 & (sAxisY | sAxisMin));
330
331 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
332 else areacode |= sBoundary;
333 if (xx.y() <= fAxisMin[yaxis] - ctol) isoutside = true;
334
335 } else if (xx.y() > fAxisMax[yaxis] - ctol) {
336 areacode |= (sAxis1 & (sAxisY | sAxisMax));
337
338 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
339 else areacode |= sBoundary;
340 if (xx.y() >= fAxisMax[yaxis] + ctol) isoutside = true;
341 }
342
343 // if isoutside = true, clear inside bit.
344 // if not on boundary, add axis information.
345
346 if (isoutside) {
347 G4int tmpareacode = areacode & (~sInside);
348 areacode = tmpareacode;
349 } else if ((areacode & sBoundary) != sBoundary) {
350 areacode |= (sAxis0 & sAxisX) | (sAxis1 & sAxisY);
351 }
352
353 } else {
354
355 // boundary of x-axis
356
357 if (xx.x() < wmin ) {
358 areacode |= (sAxis0 & (sAxisX | sAxisMin)) | sBoundary;
359 } else if (xx.x() > wmax) {
360 areacode |= (sAxis0 & (sAxisX | sAxisMax)) | sBoundary;
361 }
362
363 // boundary of y-axis
364
365 if (xx.y() < fAxisMin[yaxis]) {
366 areacode |= (sAxis1 & (sAxisY | sAxisMin));
367 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
368 else areacode |= sBoundary;
369
370 } else if (xx.y() > fAxisMax[yaxis]) {
371 areacode |= (sAxis1 & (sAxisY | sAxisMax)) ;
372 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
373 else areacode |= sBoundary;
374 }
375
376 if ((areacode & sBoundary) != sBoundary) {
377 areacode |= (sAxis0 & sAxisX) | (sAxis1 & sAxisY);
378 }
379 }
380 return areacode;
381 } else {
382 G4Exception("G4TwistTrapFlatSide::GetAreaCode()",
383 "GeomSolids0001", FatalException,
384 "Feature NOT implemented !");
385 }
386
387 return areacode;
388}

394void G4TwistTrapFlatSide::SetCorners()
395{
396 // Set Corner points in local coodinate.
397
398 if (fAxis[0] == kXAxis && fAxis[1] == kYAxis) {
399
400 G4double x, y, z;
401
402 // corner of Axis0min and Axis1min
403 x = -fDx1 + fDy * fTAlph ;
404 y = -fDy ;
405 z = 0 ;
406 SetCorner(sC0Min1Min, x, y, z);
407
408 // corner of Axis0max and Axis1min
409 x = fDx1 + fDy * fTAlph ;
410 y = -fDy ;
411 z = 0 ;
412 SetCorner(sC0Max1Min, x, y, z);
413
414 // corner of Axis0max and Axis1max
415 x = fDx2 - fDy * fTAlph ;
416 y = fDy ;
417 z = 0 ;
418 SetCorner(sC0Max1Max, x, y, z);
419
420 // corner of Axis0min and Axis1max
421 x = -fDx2 - fDy * fTAlph ;
422 y = fDy ;
423 z = 0 ;
424 SetCorner(sC0Min1Max, x, y, z);
425
426 } else {
427 std::ostringstream message;
428 message << "Feature NOT implemented !" << G4endl
429 << " fAxis[0] = " << fAxis[0] << G4endl
430 << " fAxis[1] = " << fAxis[1];
431 G4Exception("G4TwistTrapFlatSide::SetCorners()",
432 "GeomSolids0001", FatalException, message);
433 }
434}

439void G4TwistTrapFlatSide::SetBoundaries()
440{
441 // Set direction-unit vector of phi-boundary-lines in local coodinate.
442 // Don't call the function twice.
443
444 G4ThreeVector direction ;
445
446 if (fAxis[0] == kXAxis && fAxis[1] == kYAxis) {
447
448 // sAxis0 & sAxisMin
449 direction = - ( GetCorner(sC0Min1Max) - GetCorner(sC0Min1Min) ) ;
450 direction = direction.unit();
451 SetBoundary(sAxis0 & (sAxisX | sAxisMin), direction,
452 GetCorner(sC0Min1Max), sAxisY) ;
453
454 // sAxis0 & sAxisMax
455 direction = GetCorner(sC0Max1Max) - GetCorner(sC0Max1Min) ; // inverse
456 direction = direction.unit();
457 SetBoundary(sAxis0 & (sAxisX | sAxisMax), direction,
458 GetCorner(sC0Max1Min), sAxisY);
459
460 // sAxis1 & sAxisMin
461 direction = GetCorner(sC0Max1Min) - GetCorner(sC0Min1Min);
462 direction = direction.unit();
463 SetBoundary(sAxis1 & (sAxisY | sAxisMin), direction,
464 GetCorner(sC0Min1Min), sAxisX);
465
466 // sAxis1 & sAxisMax
467 direction = - ( GetCorner(sC0Max1Max) - GetCorner(sC0Min1Max) ) ;
468 direction = direction.unit();
469 SetBoundary(sAxis1 & (sAxisY | sAxisMax), direction,
470 GetCorner(sC0Max1Max), sAxisX);
471
472 } else {
473 std::ostringstream message;
474 message << "Feature NOT implemented !" << G4endl
475 << " fAxis[0] = " << fAxis[0] << G4endl
476 << " fAxis[1] = " << fAxis[1];
477 G4Exception("G4TwistTrapFlatSide::SetCorners()",
478 "GeomSolids0001", FatalException, message);
479 }

484
485void G4TwistTrapFlatSide::GetFacets( G4int k, G4int n, G4double xyz[][3],
486 G4int faces[][4], G4int iside )
487{
488
489 G4double x,y ; // the two parameters for the surface equation
490 G4ThreeVector p ; // a point on the surface, given by (z,u)
491
492 G4int nnode ;
493 G4int nface ;
494
495 G4double xmin,xmax ;
496
497 // calculate the (n-1)*(k-1) vertices
498
499 G4int i,j ;
500
501 for ( i = 0 ; i<n ; i++ ) {
502
503 y = -fDy + i*(2*fDy)/(n-1) ;
504
505 for ( j = 0 ; j<k ; j++ ) {
506
507 xmin = GetBoundaryMin(y) ;
508 xmax = GetBoundaryMax(y) ;
509 x = xmin + j*(xmax-xmin)/(k-1) ;
510
511 nnode = GetNode(i,j,k,n,iside) ;
512 p = SurfacePoint(x,y,true) ; // surface point in global coordinate system
513
514 xyz[nnode][0] = p.x() ;
515 xyz[nnode][1] = p.y() ;
516 xyz[nnode][2] = p.z() ;
517
518 if ( i<n-1 && j<k-1 ) {
519
520 nface = GetFace(i,j,k,n,iside) ;
521
522 if (fHandedness < 0) { // lower side
523 faces[nface][0] = GetEdgeVisibility(i,j,k,n,0,1) * ( GetNode(i ,j ,k,n,iside)+1) ;
524 faces[nface][1] = GetEdgeVisibility(i,j,k,n,1,1) * ( GetNode(i+1,j ,k,n,iside)+1) ;
525 faces[nface][2] = GetEdgeVisibility(i,j,k,n,2,1) * ( GetNode(i+1,j+1,k,n,iside)+1) ;
526 faces[nface][3] = GetEdgeVisibility(i,j,k,n,3,1) * ( GetNode(i ,j+1,k,n,iside)+1) ;
527 } else { // upper side
528 faces[nface][0] = GetEdgeVisibility(i,j,k,n,0,-1) * ( GetNode(i ,j ,k,n,iside)+1) ;
529 faces[nface][1] = GetEdgeVisibility(i,j,k,n,1,-1) * ( GetNode(i ,j+1,k,n,iside)+1) ;
530 faces[nface][2] = GetEdgeVisibility(i,j,k,n,2,-1) * ( GetNode(i+1,j+1,k,n,iside)+1) ;
531 faces[nface][3] = GetEdgeVisibility(i,j,k,n,3,-1) * ( GetNode(i+1,j ,k,n,iside)+1) ;
532 }
533
534 }
535 }
