34// Class description:

38// History:

41#ifndef G4UTORUS_HH
42#define G4UTORUS_HH
43
44#include "G4UAdapter.hh"
45
46#if ( defined(G4GEOM_USE_USOLIDS) || defined(G4GEOM_USE_PARTIAL_USOLIDS) )
47
48#include <volumes/UnplacedTorus2.h>
49
50#include "G4Polyhedron.hh"
51
52class G4UTorus : public G4UAdapter<vecgeom::UnplacedTorus2>
53{
54 using Shape_t = vecgeom::UnplacedTorus2;
55 using Base_t = G4UAdapter<vecgeom::UnplacedTorus2>;
56
57 public: // with description
58
59 G4UTorus(const G4String& pName,
60 G4double rmin, G4double rmax, G4double rtor,
61 G4double sphi, G4double dphi);
62 // Constructs a torus with name and geometrical parameters
63
64 ~G4UTorus();
65
66 void ComputeDimensions(G4VPVParameterisation* p,
67 const G4int n,
68 const G4VPhysicalVolume* pRep);
69
70 G4VSolid* Clone() const;
71
72 G4double GetRmin() const;
73 G4double GetRmax() const;
74 G4double GetRtor() const;
75 G4double GetSPhi() const;
76 G4double GetDPhi() const;
77 G4double GetSinStartPhi() const;
78 G4double GetCosStartPhi() const;
79 G4double GetSinEndPhi () const;
80 G4double GetCosEndPhi () const;
81
82 void SetRmin(G4double arg);
83 void SetRmax(G4double arg);
84 void SetRtor(G4double arg);
85 void SetSPhi(G4double arg);
86 void SetDPhi(G4double arg);
87
88 void SetAllParameters(G4double arg1, G4double arg2,
89 G4double arg3, G4double arg4, G4double arg5);
90
91 inline G4GeometryType GetEntityType() const;
92
93 void BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const;
94
95 G4bool CalculateExtent(const EAxis pAxis,
96 const G4VoxelLimits& pVoxelLimit,
97 const G4AffineTransform& pTransform,
98 G4double& pmin, G4double& pmax) const;
99
100 G4Polyhedron* CreatePolyhedron() const;
101
102 public: // without description
103
104 G4UTorus(__void__&);
105 // Fake default constructor for usage restricted to direct object
106 // persistency for clients requiring preallocation of memory for
107 // persistifiable objects.
108
109 G4UTorus(const G4UTorus& rhs);
110 G4UTorus& operator=(const G4UTorus& rhs);
111 // Copy constructor and assignment operator.
112};
113
114// --------------------------------------------------------------------
115// Inline methods
116// --------------------------------------------------------------------
117
118inline G4GeometryType G4UTorus::GetEntityType() const
119{
120 return "G4Torus";
121}
