108}

110inline
111void G4Cons::InitializeTrigonometry()
112{
113 G4double hDPhi = 0.5*fDPhi; // half delta phi
114 G4double cPhi = fSPhi + hDPhi;
115 G4double ePhi = fSPhi + fDPhi;
116
117 sinCPhi = std::sin(cPhi);
118 cosCPhi = std::cos(cPhi);
119 cosHDPhiIT = std::cos(hDPhi - 0.5*kAngTolerance); // inner/outer tol half dphi
120 cosHDPhiOT = std::cos(hDPhi + 0.5*kAngTolerance);
121 sinSPhi = std::sin(fSPhi);
122 cosSPhi = std::cos(fSPhi);
123 sinEPhi = std::sin(ePhi);
124 cosEPhi = std::cos(ePhi);
125}

127inline void G4Cons::CheckSPhiAngle(G4double sPhi)
128{
129 // Ensure fSphi in 0-2PI or -2PI-0 range if shape crosses 0
130
131 if ( sPhi < 0 )
132 {
133 fSPhi = CLHEP::twopi - std::fmod(std::fabs(sPhi),CLHEP::twopi);
134 }
135 else
136 {
137 fSPhi = std::fmod(sPhi,CLHEP::twopi) ;
138 }
139 if ( fSPhi+fDPhi > CLHEP::twopi )
140 {
141 fSPhi -= CLHEP::twopi ;
142 }
143}
144
145inline void G4Cons::CheckDPhiAngle(G4double dPhi)
146{
147 fPhiFullCone = true;
148 if ( dPhi >= CLHEP::twopi-kAngTolerance*0.5 )
149 {
150 fDPhi=CLHEP::twopi;
151 fSPhi=0;
152 }
153 else
154 {
155 fPhiFullCone = false;
156 if ( dPhi > 0 )
157 {
158 fDPhi = dPhi;
159 }
160 else
161 {
162 std::ostringstream message;
163 message << "Invalid dphi." << G4endl
164 << "Negative or zero delta-Phi (" << dPhi << ") in solid: "
165 << GetName();
166 G4Exception("G4Cons::CheckDPhiAngle()", "GeomSolids0002",
167 FatalException, message);
168 }
169 }
170}
171
172inline void G4Cons::CheckPhiAngles(G4double sPhi, G4double dPhi)
173{
174 CheckDPhiAngle(dPhi);
175 if ( (fDPhi<CLHEP::twopi) && (sPhi) ) { CheckSPhiAngle(sPhi); }
176 InitializeTrigonometry();
177}

214inline
215void G4Cons::SetStartPhiAngle ( G4double newSPhi, G4bool compute )
216{
217 // Flag 'compute' can be used to explicitely avoid recomputation of
218 // trigonometry in case SetDeltaPhiAngle() is invoked afterwards
219
220 CheckSPhiAngle(newSPhi);
221 fPhiFullCone = false;
222 if (compute) { InitializeTrigonometry(); }
223 Initialize();
224}
225
226void G4Cons::SetDeltaPhiAngle ( G4double newDPhi )
227{
228 CheckPhiAngles(fSPhi, newDPhi);
229 Initialize();
230}

276inline
277G4double G4Cons::GetCubicVolume()
278{
279 if(fCubicVolume != 0.) {;}
280 else
281 {
282 G4double Rmean, rMean, deltaR, deltar;
283
284 Rmean = 0.5*(fRmax1+fRmax2);
285 deltaR = fRmax1-fRmax2;
286
287 rMean = 0.5*(fRmin1+fRmin2);
288 deltar = fRmin1-fRmin2;
289 fCubicVolume = fDPhi*fDz*(Rmean*Rmean-rMean*rMean
290 +(deltaR*deltaR-deltar*deltar)/12);
291 }
292 return fCubicVolume;
293}
294
295inline
296G4double G4Cons::GetSurfaceArea()
297{
298 if(fSurfaceArea != 0.) {;}
299 else
300 {
301 G4double mmin, mmax, dmin, dmax;
302
303 mmin= (fRmin1+fRmin2)*0.5;
304 mmax= (fRmax1+fRmax2)*0.5;
305 dmin= (fRmin2-fRmin1);
306 dmax= (fRmax2-fRmax1);
307
308 fSurfaceArea = fDPhi*( mmin * std::sqrt(dmin*dmin+4*fDz*fDz)
309 + mmax * std::sqrt(dmax*dmax+4*fDz*fDz)
310 + 0.5*(fRmax1*fRmax1-fRmin1*fRmin1
311 +fRmax2*fRmax2-fRmin2*fRmin2 ));
312 if(!fPhiFullCone)
313 {
314 fSurfaceArea = fSurfaceArea+4*fDz*(mmax-mmin);
315 }
316 }
317 return fSurfaceArea;
318}
