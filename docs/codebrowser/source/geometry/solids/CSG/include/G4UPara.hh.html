42#ifndef G4UPARA_HH
43#define G4UPARA_HH
44
45#include "G4UAdapter.hh"
46
47#if ( defined(G4GEOM_USE_USOLIDS) || defined(G4GEOM_USE_PARTIAL_USOLIDS) )
48
49#include <volumes/UnplacedParallelepiped.h>
50
51#include "G4Polyhedron.hh"
52
53class G4UPara : public G4UAdapter<vecgeom::UnplacedParallelepiped>
54{
55 using Shape_t = vecgeom::UnplacedParallelepiped;
56 using Base_t = G4UAdapter<vecgeom::UnplacedParallelepiped>;
57
58 public: // with description
59
60 G4UPara(const G4String& pName,
61 G4double pDx, G4double pDy, G4double pDz,
62 G4double pAlpha, G4double pTheta, G4double pPhi);
63
64 G4UPara(const G4String& pName,
65 const G4ThreeVector pt[8]);
66
67 ~G4UPara();
68
69 // Accessors
70
71 G4double GetZHalfLength() const;
72 G4double GetYHalfLength() const;
73 G4double GetXHalfLength() const;
74 G4ThreeVector GetSymAxis() const;
75 G4double GetTanAlpha() const;
76
77 // Modifiers
78
79 void SetXHalfLength(G4double val);
80 void SetYHalfLength(G4double val);
81 void SetZHalfLength(G4double val);
82 void SetAlpha(G4double alpha);
83 void SetTanAlpha(G4double val);
84 void SetThetaAndPhi(double pTheta, double pPhi);
85
86 void SetAllParameters(G4double pDx, G4double pDy, G4double pDz,
87 G4double pAlpha, G4double pTheta, G4double pPhi);
88
89 void ComputeDimensions(G4VPVParameterisation* p,
90 const G4int n,
91 const G4VPhysicalVolume* pRep);
92
93 inline G4GeometryType GetEntityType() const;
94
95 void BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const;
96
97 G4bool CalculateExtent(const EAxis pAxis,
98 const G4VoxelLimits& pVoxelLimit,
99 const G4AffineTransform& pTransform,
100 G4double& pMin, G4double& pMax) const;
101
102 G4VSolid* Clone() const;
103
104 G4Polyhedron* CreatePolyhedron () const;
105
106 public: // without description
107
108 G4UPara(__void__&);
109 // Fake default constructor for usage restricted to direct object
110 // persistency for clients requiring preallocation of memory for
111 // persistifiable objects
112
113 G4UPara(const G4UPara& rhs);
114 G4UPara& operator=(const G4UPara& rhs);
115 // Copy constructor and assignment operator
116
117 private:
118
119 void CheckParameters();
120 // Check parameters
121
122 void MakePlanes();
123 // Set side planes
124
125 private:
126
127 G4double fTalpha,fTthetaCphi,fTthetaSphi;
128 struct { G4double a,b,c,d; } fPlanes[4];
129};
130
131// --------------------------------------------------------------------
132// Inline methods
133// --------------------------------------------------------------------
134
135inline G4GeometryType G4UPara::GetEntityType() const
136{
137 return "G4Para";
138}
