76//////////////////////////////////////////////////////////////////////////
77//
78// Constructor - design of trapezoid based on 8 vertices
79
80G4Para::G4Para( const G4String& pName,
81 const G4ThreeVector pt[8] )
82 : G4CSGSolid(pName), halfCarTolerance(0.5*kCarTolerance)
83{
84 // Find dimensions and trigonometric values
85 //
86 fDx = (pt[3].x() - pt[2].x())*0.5;
87 fDy = (pt[2].y() - pt[1].y())*0.5;
88 fDz = pt[7].z();
89 CheckParameters(); // check dimensions
90
91 fTalpha = (pt[2].x() + pt[3].x() - pt[1].x() - pt[0].x())*0.25/fDy;
92 fTthetaCphi = (pt[4].x() + fDy*fTalpha + fDx)/fDz;
93 fTthetaSphi = (pt[4].y() + fDy)/fDz;
94 MakePlanes();
95
96 // Recompute vertices
97 //
98 G4ThreeVector v[8];
99 G4double DyTalpha = fDy*fTalpha;
100 G4double DzTthetaSphi = fDz*fTthetaSphi;
101 G4double DzTthetaCphi = fDz*fTthetaCphi;
102 v[0].set(-DzTthetaCphi-DyTalpha-fDx, -DzTthetaSphi-fDy, -fDz);
103 v[1].set(-DzTthetaCphi-DyTalpha+fDx, -DzTthetaSphi-fDy, -fDz);
104 v[2].set(-DzTthetaCphi+DyTalpha-fDx, -DzTthetaSphi+fDy, -fDz);
105 v[3].set(-DzTthetaCphi+DyTalpha+fDx, -DzTthetaSphi+fDy, -fDz);
106 v[4].set( DzTthetaCphi-DyTalpha-fDx, DzTthetaSphi-fDy, fDz);
107 v[5].set( DzTthetaCphi-DyTalpha+fDx, DzTthetaSphi-fDy, fDz);
108 v[6].set( DzTthetaCphi+DyTalpha-fDx, DzTthetaSphi+fDy, fDz);
109 v[7].set( DzTthetaCphi+DyTalpha+fDx, DzTthetaSphi+fDy, fDz);
110
111 // Compare with original vertices
112 //
113 for (G4int i=0; i<8; ++i)
114 {
115 G4double delx = std::abs(pt[i].x() - v[i].x());
116 G4double dely = std::abs(pt[i].y() - v[i].y());
117 G4double delz = std::abs(pt[i].z() - v[i].z());
118 G4double discrepancy = std::max(std::max(delx,dely),delz);
119 if (discrepancy > 0.1*kCarTolerance)
120 {
121 std::ostringstream message;
122 G4int oldprc = message.precision(16);
123 message << "Invalid vertice coordinates for Solid: " << GetName()
124 << "\nVertix #" << i << ", discrepancy = " << discrepancy
125 << "\n original : " << pt[i]
126 << "\n recomputed : " << v[i];
127 G4cout.precision(oldprc);
128 G4Exception("G4Para::G4Para()", "GeomSolids0002",
129 FatalException, message);
130
131 }
132 }
133}

167//////////////////////////////////////////////////////////////////////////
168//
169// Assignment operator
170
171G4Para& G4Para::operator = (const G4Para& rhs)
172{
173 // Check assignment to self
174 //
175 if (this == &rhs) { return *this; }
176
177 // Copy base class data
178 //
179 G4CSGSolid::operator=(rhs);
180
181 // Copy data
182 //
183 halfCarTolerance = rhs.halfCarTolerance;
184 fDx = rhs.fDx;
185 fDy = rhs.fDy;
186 fDz = rhs.fDz;
187 fTalpha = rhs.fTalpha;
188 fTthetaCphi = rhs.fTthetaCphi;
189 fTthetaSphi = rhs.fTthetaSphi;
190 for (G4int i=0; i<4; ++i) { fPlanes[i] = rhs.fPlanes[i]; }
191
192 return *this;
193}

195//////////////////////////////////////////////////////////////////////////
196//
197// Set all parameters, as for constructor - set and check half-widths
198
199void G4Para::SetAllParameters(G4double pDx, G4double pDy, G4double pDz,
200 G4double pAlpha, G4double pTheta, G4double pPhi)
201{
202 // Reset data of the base class
203 fCubicVolume = 0;
204 fSurfaceArea = 0;
205 fRebuildPolyhedron = true;
206
207 // Set parameters
208 fDx = pDx;
209 fDy = pDy;
210 fDz = pDz;
211 fTalpha = std::tan(pAlpha);
212 fTthetaCphi = std::tan(pTheta)*std::cos(pPhi);
213 fTthetaSphi = std::tan(pTheta)*std::sin(pPhi);
214
215 CheckParameters();
216 MakePlanes();
217}

219//////////////////////////////////////////////////////////////////////////
220//
221// Check dimensions
222
223void G4Para::CheckParameters()
224{
225 if (fDx < 2*kCarTolerance ||
226 fDy < 2*kCarTolerance ||
227 fDz < 2*kCarTolerance)
228 {
229 std::ostringstream message;
230 message << "Invalid (too small or negative) dimensions for Solid: "
231 << GetName()
232 << "\n X - " << fDx
233 << "\n Y - " << fDy
234 << "\n Z - " << fDz;
235 G4Exception("G4Para::CheckParameters()", "GeomSolids0002",
236 FatalException, message);
237 }
238}

240//////////////////////////////////////////////////////////////////////////
241//
242// Set side planes
243
244void G4Para::MakePlanes()
245{
246 G4ThreeVector vx(1, 0, 0);
247 G4ThreeVector vy(fTalpha, 1, 0);
248 G4ThreeVector vz(fTthetaCphi, fTthetaSphi, 1);
249
250 // Set -Y & +Y planes
251 //
252 G4ThreeVector ynorm = (vx.cross(vz)).unit();
253
254 fPlanes[0].a = 0.;
255 fPlanes[0].b = ynorm.y();
256 fPlanes[0].c = ynorm.z();
257 fPlanes[0].d = fPlanes[0].b*fDy; // point (0,fDy,0) is on plane
258
259 fPlanes[1].a = 0.;
260 fPlanes[1].b = -fPlanes[0].b;
261 fPlanes[1].c = -fPlanes[0].c;
262 fPlanes[1].d = fPlanes[0].d;
263
264 // Set -X & +X planes
265 //
266 G4ThreeVector xnorm = (vz.cross(vy)).unit();
267
268 fPlanes[2].a = xnorm.x();
269 fPlanes[2].b = xnorm.y();
270 fPlanes[2].c = xnorm.z();
271 fPlanes[2].d = fPlanes[2].a*fDx; // point (fDx,0,0) is on plane
272
273 fPlanes[3].a = -fPlanes[2].a;
274 fPlanes[3].b = -fPlanes[2].b;
275 fPlanes[3].c = -fPlanes[2].c;
276 fPlanes[3].d = fPlanes[2].d;
277}

279//////////////////////////////////////////////////////////////////////////
280//
281// Get volume
282
283G4double G4Para::GetCubicVolume()
284{
285 // It is like G4Box, since para transformations keep the volume to be const
286 if (fCubicVolume == 0)
287 {
288 fCubicVolume = 8*fDx*fDy*fDz;
289 }
290 return fCubicVolume;
291}

293//////////////////////////////////////////////////////////////////////////
294//
295// Get surface area
296
297G4double G4Para::GetSurfaceArea()
298{
299 if(fSurfaceArea == 0)
300 {
301 G4ThreeVector vx(fDx, 0, 0);
302 G4ThreeVector vy(fDy*fTalpha, fDy, 0);
303 G4ThreeVector vz(fDz*fTthetaCphi, fDz*fTthetaSphi, fDz);
304
305 G4double sxy = fDx*fDy; // (vx.cross(vy)).mag();
306 G4double sxz = (vx.cross(vz)).mag();
307 G4double syz = (vy.cross(vz)).mag();
308
309 fSurfaceArea = 8*(sxy+sxz+syz);
310 }
311 return fSurfaceArea;
312}

326//////////////////////////////////////////////////////////////////////////
327//
328// Get bounding box
329
330void G4Para::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
331{
332 G4double dz = GetZHalfLength();
333 G4double dx = GetXHalfLength();
334 G4double dy = GetYHalfLength();
335
336 G4double x0 = dz*fTthetaCphi;
337 G4double x1 = dy*GetTanAlpha();
338 G4double xmin =
339 std::min(
340 std::min(
341 std::min(-x0-x1-dx,-x0+x1-dx),x0-x1-dx),x0+x1-dx);
342 G4double xmax =
343 std::max(
344 std::max(
345 std::max(-x0-x1+dx,-x0+x1+dx),x0-x1+dx),x0+x1+dx);
346
347 G4double y0 = dz*fTthetaSphi;
348 G4double ymin = std::min(-y0-dy,y0-dy);
349 G4double ymax = std::max(-y0+dy,y0+dy);
350
351 pMin.set(xmin,ymin,-dz);
352 pMax.set(xmax,ymax, dz);
353
354 // Check correctness of the bounding box
355 //
356 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
357 {
358 std::ostringstream message;
359 message << "Bad bounding box (min >= max) for solid: "
360 << GetName() << " !"
361 << "\npMin = " << pMin
362 << "\npMax = " << pMax;
363 G4Exception("G4Para::BoundingLimits()", "GeomMgt0001",
364 JustWarning, message);
365 DumpInfo();
366 }
367}

369//////////////////////////////////////////////////////////////////////////
370//
371// Calculate extent under transform and specified limit
372
373G4bool G4Para::CalculateExtent( const EAxis pAxis,
374 const G4VoxelLimits& pVoxelLimit,
375 const G4AffineTransform& pTransform,
376 G4double& pMin, G4double& pMax ) const
377{
378 G4ThreeVector bmin, bmax;
379 G4bool exist;
380
381 // Check bounding box (bbox)
382 //
383 BoundingLimits(bmin,bmax);
384 G4BoundingEnvelope bbox(bmin,bmax);
385#ifdef G4BBOX_EXTENT
386 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
387#endif
388 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
389 {
390 return exist = (pMin < pMax) ? true : false;
391 }
392
393 // Set bounding envelope (benv) and calculate extent
394 //
395 G4double dz = GetZHalfLength();
396 G4double dx = GetXHalfLength();
397 G4double dy = GetYHalfLength();
398
399 G4double x0 = dz*fTthetaCphi;
400 G4double x1 = dy*GetTanAlpha();
401 G4double y0 = dz*fTthetaSphi;
402
403 G4ThreeVectorList baseA(4), baseB(4);
404 baseA[0].set(-x0-x1-dx,-y0-dy,-dz);
405 baseA[1].set(-x0-x1+dx,-y0-dy,-dz);
406 baseA[2].set(-x0+x1+dx,-y0+dy,-dz);
407 baseA[3].set(-x0+x1-dx,-y0+dy,-dz);
408
409 baseB[0].set(+x0-x1-dx, y0-dy, dz);
410 baseB[1].set(+x0-x1+dx, y0-dy, dz);
411 baseB[2].set(+x0+x1+dx, y0+dy, dz);
412 baseB[3].set(+x0+x1-dx, y0+dy, dz);
413
414 std::vector<const G4ThreeVectorList *> polygons(2);
415 polygons[0] = &baseA;
416 polygons[1] = &baseB;
417
418 G4BoundingEnvelope benv(bmin,bmax,polygons);
419 exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
420 return exist;
421}

423//////////////////////////////////////////////////////////////////////////
424//
425// Determine where is point p, inside/on_surface/outside
426//
427
428EInside G4Para::Inside( const G4ThreeVector& p ) const
429{
430 G4double xx = fPlanes[2].a*p.x()+fPlanes[2].b*p.y()+fPlanes[2].c*p.z();
431 G4double dx = std::abs(xx) + fPlanes[2].d;
432
433 G4double yy = fPlanes[0].b*p.y()+fPlanes[0].c*p.z();
434 G4double dy = std::abs(yy) + fPlanes[0].d;
435 G4double dxy = std::max(dx,dy);
436
437 G4double dz = std::abs(p.z())-fDz;
438 G4double dist = std::max(dxy,dz);
439
440 if (dist > halfCarTolerance) return kOutside;
441 return (dist > -halfCarTolerance) ? kSurface : kInside;
442}

444//////////////////////////////////////////////////////////////////////////
445//
446// Determine side where point is, and return corresponding normal
447
448G4ThreeVector G4Para::SurfaceNormal( const G4ThreeVector& p ) const
449{
450 G4int nsurf = 0; // number of surfaces where p is placed
451
452 // Check Z faces
453 //
454 G4double nz = 0;
455 G4double dz = std::abs(p.z()) - fDz;
456 if (std::abs(dz) <= halfCarTolerance)
457 {
458 nz = (p.z() < 0) ? -1 : 1;
459 ++nsurf;
460 }
461
462 // Check Y faces
463 //
464 G4double ny = 0;
465 G4double yy = fPlanes[0].b*p.y()+fPlanes[0].c*p.z();
466 if (std::abs(fPlanes[0].d + yy) <= halfCarTolerance)
467 {
468 ny = fPlanes[0].b;
469 nz += fPlanes[0].c;
470 ++nsurf;
471 }
472 else if (std::abs(fPlanes[1].d - yy) <= halfCarTolerance)
473 {
474 ny = fPlanes[1].b;
475 nz += fPlanes[1].c;
476 ++nsurf;
477 }
478
479 // Check X faces
480 //
481 G4double nx = 0;
482 G4double xx = fPlanes[2].a*p.x()+fPlanes[2].b*p.y()+fPlanes[2].c*p.z();
483 if (std::abs(fPlanes[2].d + xx) <= halfCarTolerance)
484 {
485 nx = fPlanes[2].a;
486 ny += fPlanes[2].b;
487 nz += fPlanes[2].c;
488 ++nsurf;
489 }
490 else if (std::abs(fPlanes[3].d - xx) <= halfCarTolerance)
491 {
492 nx = fPlanes[3].a;
493 ny += fPlanes[3].b;
494 nz += fPlanes[3].c;
495 ++nsurf;
496 }
497
498 // Return normal
499 //
500 if (nsurf == 1) return G4ThreeVector(nx,ny,nz);
501 else if (nsurf != 0) return G4ThreeVector(nx,ny,nz).unit(); // edge or corner
502 else
503 {
504 // Point is not on the surface
505 //
506#ifdef G4CSGDEBUG
507 std::ostringstream message;
508 G4int oldprc = message.precision(16);
509 message << "Point p is not on surface (!?) of solid: "
510 << GetName() << G4endl;
511 message << "Position:\n";
512 message << " p.x() = " << p.x()/mm << " mm\n";
513 message << " p.y() = " << p.y()/mm << " mm\n";
514 message << " p.z() = " << p.z()/mm << " mm";
515 G4cout.precision(oldprc) ;
516 G4Exception("G4Para::SurfaceNormal(p)", "GeomSolids1002",
517 JustWarning, message );
518 DumpInfo();
519#endif
520 return ApproxSurfaceNormal(p);
521 }
522}

524//////////////////////////////////////////////////////////////////////////
525//
526// Algorithm for SurfaceNormal() following the original specification
527// for points not on the surface
528
529G4ThreeVector G4Para::ApproxSurfaceNormal( const G4ThreeVector& p ) const
530{
531 G4double dist = -DBL_MAX;
532 G4int iside = 0;
533 for (G4int i=0; i<4; ++i)
534 {
535 G4double d = fPlanes[i].a*p.x() +
536 fPlanes[i].b*p.y() +
537 fPlanes[i].c*p.z() + fPlanes[i].d;
538 if (d > dist) { dist = d; iside = i; }
539 }
540
541 G4double distz = std::abs(p.z()) - fDz;
542 if (dist > distz)
543 return G4ThreeVector(fPlanes[iside].a, fPlanes[iside].b, fPlanes[iside].c);
544 else
545 return G4ThreeVector(0, 0, (p.z() < 0) ? -1 : 1);
546}

548//////////////////////////////////////////////////////////////////////////
549//
550// Calculate distance to shape from outside
551// - return kInfinity if no intersection
552
553G4double G4Para::DistanceToIn(const G4ThreeVector& p,
554 const G4ThreeVector& v ) const
555{
556 // Z intersections
557 //
558 if ((std::abs(p.z()) - fDz) >= -halfCarTolerance && p.z()*v.z() >= 0)
559 return kInfinity;
560 G4double invz = (-v.z() == 0) ? DBL_MAX : -1./v.z();
561 G4double dz = (invz < 0) ? fDz : -fDz;
562 G4double tzmin = (p.z() + dz)*invz;
563 G4double tzmax = (p.z() - dz)*invz;
564
565 // Y intersections
566 //
567 G4double tmin0 = tzmin, tmax0 = tzmax;
568 G4double cos0 = fPlanes[0].b*v.y() + fPlanes[0].c*v.z();
569 G4double disy = fPlanes[0].b*p.y() + fPlanes[0].c*p.z();
570 G4double dis0 = fPlanes[0].d + disy;
571 if (dis0 >= -halfCarTolerance)
572 {
573 if (cos0 >= 0) return kInfinity;
574 G4double tmp = -dis0/cos0;
575 if (tmin0 < tmp) tmin0 = tmp;
576 }
577 else if (cos0 > 0)
578 {
579 G4double tmp = -dis0/cos0;
580 if (tmax0 > tmp) tmax0 = tmp;
581 }
582
583 G4double tmin1 = tmin0, tmax1 = tmax0;
584 G4double cos1 = -cos0;
585 G4double dis1 = fPlanes[1].d - disy;
586 if (dis1 >= -halfCarTolerance)
587 {
588 if (cos1 >= 0) return kInfinity;
589 G4double tmp = -dis1/cos1;
590 if (tmin1 < tmp) tmin1 = tmp;
591 }
592 else if (cos1 > 0)
593 {
594 G4double tmp = -dis1/cos1;
595 if (tmax1 > tmp) tmax1 = tmp;
596 }
597
598 // X intersections
599 //
600 G4double tmin2 = tmin1, tmax2 = tmax1;
601 G4double cos2 = fPlanes[2].a*v.x() + fPlanes[2].b*v.y() + fPlanes[2].c*v.z();
602 G4double disx = fPlanes[2].a*p.x() + fPlanes[2].b*p.y() + fPlanes[2].c*p.z();
603 G4double dis2 = fPlanes[2].d + disx;
604 if (dis2 >= -halfCarTolerance)
605 {
606 if (cos2 >= 0) return kInfinity;
607 G4double tmp = -dis2/cos2;
608 if (tmin2 < tmp) tmin2 = tmp;
609 }
610 else if (cos2 > 0)
611 {
612 G4double tmp = -dis2/cos2;
613 if (tmax2 > tmp) tmax2 = tmp;
614 }
615
616 G4double tmin3 = tmin2, tmax3 = tmax2;
617 G4double cos3 = -cos2;
618 G4double dis3 = fPlanes[3].d - disx;
619 if (dis3 >= -halfCarTolerance)
620 {
621 if (cos3 >= 0) return kInfinity;
622 G4double tmp = -dis3/cos3;
623 if (tmin3 < tmp) tmin3 = tmp;
624 }
625 else if (cos3 > 0)
626 {
627 G4double tmp = -dis3/cos3;
628 if (tmax3 > tmp) tmax3 = tmp;
629 }
630
631 // Find distance
632 //
633 G4double tmin = tmin3, tmax = tmax3;
634 if (tmax <= tmin + halfCarTolerance) return kInfinity; // touch or no hit
635 return (tmin < halfCarTolerance ) ? 0. : tmin;
636}

638//////////////////////////////////////////////////////////////////////////
639//
640// Calculate exact shortest distance to any boundary from outside
641// - returns 0 is point inside
642
643G4double G4Para::DistanceToIn( const G4ThreeVector& p ) const
644{
645 G4double xx = fPlanes[2].a*p.x()+fPlanes[2].b*p.y()+fPlanes[2].c*p.z();
646 G4double dx = std::abs(xx) + fPlanes[2].d;
647
648 G4double yy = fPlanes[0].b*p.y()+fPlanes[0].c*p.z();
649 G4double dy = std::abs(yy) + fPlanes[0].d;
650 G4double dxy = std::max(dx,dy);
651
652 G4double dz = std::abs(p.z())-fDz;
653 G4double dist = std::max(dxy,dz);
654
655 return (dist > 0) ? dist : 0.;
656}

658//////////////////////////////////////////////////////////////////////////
659//
660// Calculate distance to surface of shape from inside and, if required,
661// find normal at exit point
662// - when leaving the surface, return 0
663
664G4double G4Para::DistanceToOut(const G4ThreeVector& p, const G4ThreeVector& v,
665 const G4bool calcNorm,
666 G4bool *validNorm, G4ThreeVector *n) const
667{
668 // Z intersections
669 //
670 if ((std::abs(p.z()) - fDz) >= -halfCarTolerance && p.z()*v.z() > 0)
671 {
672 if (calcNorm)
673 {
674 *validNorm = true;
675 n->set(0, 0, (p.z() < 0) ? -1 : 1);
676 }
677 return 0.;
678 }
679 G4double vz = v.z();
680 G4double tmax = (vz == 0) ? DBL_MAX : (std::copysign(fDz,vz) - p.z())/vz;
681 G4int iside = (vz < 0) ? -4 : -2; // little trick: (-4+3)=-1, (-2+3)=+1
682
683 // Y intersections
684 //
685 G4double cos0 = fPlanes[0].b*v.y() + fPlanes[0].c*v.z();
686 if (cos0 > 0)
687 {
688 G4double dis0 = fPlanes[0].b*p.y() + fPlanes[0].c*p.z() + fPlanes[0].d;
689 if (dis0 >= -halfCarTolerance)
690 {
691 if (calcNorm)
692 {
693 *validNorm = true;
694 n->set(0, fPlanes[0].b, fPlanes[0].c);
695 }
696 return 0.;
697 }
698 G4double tmp = -dis0/cos0;
699 if (tmax > tmp) { tmax = tmp; iside = 0; }
700 }
701
702 G4double cos1 = -cos0;
703 if (cos1 > 0)
704 {
705 G4double dis1 = fPlanes[1].b*p.y() + fPlanes[1].c*p.z() + fPlanes[1].d;
706 if (dis1 >= -halfCarTolerance)
707 {
708 if (calcNorm)
709 {
710 *validNorm = true;
711 n->set(0, fPlanes[1].b, fPlanes[1].c);
712 }
713 return 0.;
714 }
715 G4double tmp = -dis1/cos1;
716 if (tmax > tmp) { tmax = tmp; iside = 1; }
717 }
718
719 // X intersections
720 //
721 G4double cos2 = fPlanes[2].a*v.x() + fPlanes[2].b*v.y() + fPlanes[2].c*v.z();
722 if (cos2 > 0)
723 {
724 G4double dis2 = fPlanes[2].a*p.x()+fPlanes[2].b*p.y()+fPlanes[2].c*p.z()+fPlanes[2].d;
725 if (dis2 >= -halfCarTolerance)
726 {
727 if (calcNorm)
728 {
729 *validNorm = true;
730 n->set(fPlanes[2].a, fPlanes[2].b, fPlanes[2].c);
731 }
732 return 0.;
733 }
734 G4double tmp = -dis2/cos2;
735 if (tmax > tmp) { tmax = tmp; iside = 2; }
736 }
737
738 G4double cos3 = -cos2;
739 if (cos3 > 0)
740 {
741 G4double dis3 = fPlanes[3].a*p.x()+fPlanes[3].b*p.y()+fPlanes[3].c*p.z()+fPlanes[3].d;
742 if (dis3 >= -halfCarTolerance)
743 {
744 if (calcNorm)
745 {
746 *validNorm = true;
747 n->set(fPlanes[3].a, fPlanes[3].b, fPlanes[3].c);
748 }
749 return 0.;
750 }
751 G4double tmp = -dis3/cos3;
752 if (tmax > tmp) { tmax = tmp; iside = 3; }
753 }
754
755 // Set normal, if required, and return distance
756 //
757 if (calcNorm)
758 {
759 *validNorm = true;
760 if (iside < 0)
761 n->set(0, 0, iside + 3); // (-4+3)=-1, (-2+3)=+1
762 else
763 n->set(fPlanes[iside].a, fPlanes[iside].b, fPlanes[iside].c);
764 }
765 return tmax;
766}

852//////////////////////////////////////////////////////////////////////////
853//
854// Return a point randomly and uniformly selected on the solid surface
855
856G4ThreeVector G4Para::GetPointOnSurface() const
857{
858 G4double DyTalpha = fDy*fTalpha;
859 G4double DzTthetaSphi = fDz*fTthetaSphi;
860 G4double DzTthetaCphi = fDz*fTthetaCphi;
861
862 // Set vertices
863 //
864 G4ThreeVector pt[8];
865 pt[0].set(-DzTthetaCphi-DyTalpha-fDx, -DzTthetaSphi-fDy, -fDz);
866 pt[1].set(-DzTthetaCphi-DyTalpha+fDx, -DzTthetaSphi-fDy, -fDz);
867 pt[2].set(-DzTthetaCphi+DyTalpha-fDx, -DzTthetaSphi+fDy, -fDz);
868 pt[3].set(-DzTthetaCphi+DyTalpha+fDx, -DzTthetaSphi+fDy, -fDz);
869 pt[4].set( DzTthetaCphi-DyTalpha-fDx, DzTthetaSphi-fDy, fDz);
870 pt[5].set( DzTthetaCphi-DyTalpha+fDx, DzTthetaSphi-fDy, fDz);
871 pt[6].set( DzTthetaCphi+DyTalpha-fDx, DzTthetaSphi+fDy, fDz);
872 pt[7].set( DzTthetaCphi+DyTalpha+fDx, DzTthetaSphi+fDy, fDz);
873
874 // Set areas (-Z, -Y, +Y, -X, +X, +Z)
875 //
876 G4ThreeVector vx(fDx, 0, 0);
877 G4ThreeVector vy(DyTalpha, fDy, 0);
878 G4ThreeVector vz(DzTthetaCphi, DzTthetaSphi, fDz);
879
880 G4double sxy = fDx*fDy; // (vx.cross(vy)).mag();
881 G4double sxz = (vx.cross(vz)).mag();
882 G4double syz = (vy.cross(vz)).mag();
883
884 G4double sface[6] = { sxy, syz, syz, sxz, sxz, sxy };
885 for (G4int i=1; i<6; ++i) { sface[i] += sface[i-1]; }
886
887 // Select face
888 //
889 G4double select = sface[5]*G4UniformRand();
890 G4int k = 5;
891 if (select <= sface[4]) k = 4;
892 if (select <= sface[3]) k = 3;
893 if (select <= sface[2]) k = 2;
894 if (select <= sface[1]) k = 1;
895 if (select <= sface[0]) k = 0;
896
897 // Generate point
898 //
899 G4int ip[6][3] = {{0,1,2}, {0,4,1}, {2,3,6}, {0,2,4}, {1,5,3}, {4,6,5}};
900 G4double u = G4UniformRand();
901 G4double v = G4UniformRand();
902 return (1.-u-v)*pt[ip[k][0]] + u*pt[ip[k][1]] + v*pt[ip[k][2]];
903}
