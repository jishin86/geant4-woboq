89//
90// constructor - check parameters, convert angles so 0<sphi+dpshi<=2_PI
91// - note if pDPhi>2PI then reset to 2PI
92
93G4Sphere::G4Sphere( const G4String& pName,
94 G4double pRmin, G4double pRmax,
95 G4double pSPhi, G4double pDPhi,
96 G4double pSTheta, G4double pDTheta )
97 : G4CSGSolid(pName), fEpsilon(2.e-11), fSPhi(0.0),
98 fFullPhiSphere(true), fFullThetaSphere(true)
99{
100 kAngTolerance = G4GeometryTolerance::GetInstance()->GetAngularTolerance();
101 kRadTolerance = G4GeometryTolerance::GetInstance()->GetRadialTolerance();
102
103 halfCarTolerance = 0.5*kCarTolerance;
104 halfAngTolerance = 0.5*kAngTolerance;
105
106 // Check radii and set radial tolerances
107
108 if ( (pRmin >= pRmax) || (pRmax < 1.1*kRadTolerance) || (pRmin < 0) )
109 {
110 std::ostringstream message;
111 message << "Invalid radii for Solid: " << GetName() << G4endl
112 << " pRmin = " << pRmin << ", pRmax = " << pRmax;
113 G4Exception("G4Sphere::G4Sphere()", "GeomSolids0002",
114 FatalException, message);
115 }
116 fRmin=pRmin; fRmax=pRmax;
117 fRminTolerance = (fRmin) ? std::max( kRadTolerance, fEpsilon*fRmin ) : 0;
118 fRmaxTolerance = std::max( kRadTolerance, fEpsilon*fRmax );
119
120 // Check angles
121
122 CheckPhiAngles(pSPhi, pDPhi);
123 CheckThetaAngles(pSTheta, pDTheta);
124}

227//////////////////////////////////////////////////////////////////////////
228//
229// Get bounding box
230
231void G4Sphere::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
232{
233 G4double rmin = GetInnerRadius();
234 G4double rmax = GetOuterRadius();
235
236 // Find bounding box
237 //
238 if (GetDeltaThetaAngle() >= pi && GetDeltaPhiAngle() >= twopi)
239 {
240 pMin.set(-rmax,-rmax,-rmax);
241 pMax.set( rmax, rmax, rmax);
242 }
243 else
244 {
245 G4double sinStart = GetSinStartTheta();
246 G4double cosStart = GetCosStartTheta();
247 G4double sinEnd = GetSinEndTheta();
248 G4double cosEnd = GetCosEndTheta();
249
250 G4double stheta = GetStartThetaAngle();
251 G4double etheta = stheta + GetDeltaThetaAngle();
252 G4double rhomin = rmin*std::min(sinStart,sinEnd);
253 G4double rhomax = rmax;
254 if (stheta > halfpi) rhomax = rmax*sinStart;
255 if (etheta < halfpi) rhomax = rmax*sinEnd;
256
257 G4TwoVector xymin,xymax;
258 G4GeomTools::DiskExtent(rhomin,rhomax,
259 GetSinStartPhi(),GetCosStartPhi(),
260 GetSinEndPhi(),GetCosEndPhi(),
261 xymin,xymax);
262
263 G4double zmin = std::min(rmin*cosEnd,rmax*cosEnd);
264 G4double zmax = std::max(rmin*cosStart,rmax*cosStart);
265 pMin.set(xymin.x(),xymin.y(),zmin);
266 pMax.set(xymax.x(),xymax.y(),zmax);
267 }
268
269 // Check correctness of the bounding box
270 //
271 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
272 {
273 std::ostringstream message;
274 message << "Bad bounding box (min >= max) for solid: "
275 << GetName() << " !"
276 << "\npMin = " << pMin
277 << "\npMax = " << pMax;
278 G4Exception("G4Sphere::BoundingLimits()", "GeomMgt0001",
279 JustWarning, message);
280 DumpInfo();
281 }
282}

303///////////////////////////////////////////////////////////////////////////
304//
305// Return whether point inside/outside/on surface
306// Split into radius, phi, theta checks
307// Each check modifies 'in', or returns as approprate
308
309EInside G4Sphere::Inside( const G4ThreeVector& p ) const
310{
311 G4double rho,rho2,rad2,tolRMin,tolRMax;
312 G4double pPhi,pTheta;
313 EInside in = kOutside;
314
315 const G4double halfRmaxTolerance = fRmaxTolerance*0.5;
316 const G4double halfRminTolerance = fRminTolerance*0.5;
317 const G4double Rmax_minus = fRmax - halfRmaxTolerance;
318 const G4double Rmin_plus = (fRmin > 0) ? fRmin+halfRminTolerance : 0;
319
320 rho2 = p.x()*p.x() + p.y()*p.y() ;
321 rad2 = rho2 + p.z()*p.z() ;
322
323 // Check radial surfaces. Sets 'in'
324
325 tolRMin = Rmin_plus;
326 tolRMax = Rmax_minus;
327
328 if(rad2 == 0.0)
329 {
330 if (fRmin > 0.0)
331 {
332 return in = kOutside;
333 }
334 if ( (!fFullPhiSphere) || (!fFullThetaSphere) )
335 {
336 return in = kSurface;
337 }
338 else
339 {
340 return in = kInside;
341 }
342 }
343
344 if ( (rad2 <= Rmax_minus*Rmax_minus) && (rad2 >= Rmin_plus*Rmin_plus) )
345 {
346 in = kInside;
347 }
348 else
349 {
350 tolRMax = fRmax + halfRmaxTolerance; // outside case
351 tolRMin = std::max(fRmin-halfRminTolerance, 0.); // outside case
352 if ( (rad2 <= tolRMax*tolRMax) && (rad2 >= tolRMin*tolRMin) )
353 {
354 in = kSurface;
355 }
356 else
357 {
358 return in = kOutside;
359 }
360 }
361
362 // Phi boundaries : Do not check if it has no phi boundary!
363
364 if ( !fFullPhiSphere && rho2 ) // [fDPhi < twopi] and [p.x or p.y]
365 {
366 pPhi = std::atan2(p.y(),p.x()) ;
367
368 if ( pPhi < fSPhi - halfAngTolerance ) { pPhi += twopi; }
369 else if ( pPhi > ePhi + halfAngTolerance ) { pPhi -= twopi; }
370
371 if ( (pPhi < fSPhi - halfAngTolerance)
372 || (pPhi > ePhi + halfAngTolerance) ) { return in = kOutside; }
373
374 else if (in == kInside) // else it's kSurface anyway already
375 {
376 if ( (pPhi < fSPhi + halfAngTolerance)
377 || (pPhi > ePhi - halfAngTolerance) ) { in = kSurface; }
378 }
379 }
380
381 // Theta bondaries
382
383 if ( (rho2 || p.z()) && (!fFullThetaSphere) )
384 {
385 rho = std::sqrt(rho2);
386 pTheta = std::atan2(rho,p.z());
387
388 if ( in == kInside )
389 {
390 if ( ((fSTheta > 0.0) && (pTheta < fSTheta + halfAngTolerance))
391 || ((eTheta < pi) && (pTheta > eTheta - halfAngTolerance)) )
392 {
393 if ( (( (fSTheta>0.0)&&(pTheta>=fSTheta-halfAngTolerance) )
394 || (fSTheta == 0.0) )
395 && ((eTheta==pi)||(pTheta <= eTheta + halfAngTolerance) ) )
396 {
397 in = kSurface;
398 }
399 else
400 {
401 in = kOutside;
402 }
403 }
404 }
405 else
406 {
407 if ( ((fSTheta > 0.0)&&(pTheta < fSTheta - halfAngTolerance))
408 ||((eTheta < pi )&&(pTheta > eTheta + halfAngTolerance)) )
409 {
410 in = kOutside;
411 }
412 }
413 }
414 return in;
415}

417/////////////////////////////////////////////////////////////////////
418//
419// Return unit normal of surface closest to p
420// - note if point on z axis, ignore phi divided sides
421// - unsafe if point close to z axis a rmin=0 - no explicit checks
422
423G4ThreeVector G4Sphere::SurfaceNormal( const G4ThreeVector& p ) const
424{
425 G4int noSurfaces = 0;
426 G4double rho, rho2, radius, pTheta, pPhi=0.;
427 G4double distRMin = kInfinity;
428 G4double distSPhi = kInfinity, distEPhi = kInfinity;
429 G4double distSTheta = kInfinity, distETheta = kInfinity;
430 G4ThreeVector nR, nPs, nPe, nTs, nTe, nZ(0.,0.,1.);
431 G4ThreeVector norm, sumnorm(0.,0.,0.);
432
433 rho2 = p.x()*p.x()+p.y()*p.y();
434 radius = std::sqrt(rho2+p.z()*p.z());
435 rho = std::sqrt(rho2);
436
437 G4double distRMax = std::fabs(radius-fRmax);
438 if (fRmin) distRMin = std::fabs(radius-fRmin);
439
440 if ( rho && !fFullSphere )
441 {
442 pPhi = std::atan2(p.y(),p.x());
443
444 if (pPhi < fSPhi-halfAngTolerance) { pPhi += twopi; }
445 else if (pPhi > ePhi+halfAngTolerance) { pPhi -= twopi; }
446 }
447 if ( !fFullPhiSphere )
448 {
449 if ( rho )
450 {
451 distSPhi = std::fabs( pPhi-fSPhi );
452 distEPhi = std::fabs( pPhi-ePhi );
453 }
454 else if( !fRmin )
455 {
456 distSPhi = 0.;
457 distEPhi = 0.;
458 }
459 nPs = G4ThreeVector(sinSPhi,-cosSPhi,0);
460 nPe = G4ThreeVector(-sinEPhi,cosEPhi,0);
461 }
462 if ( !fFullThetaSphere )
463 {
464 if ( rho )
465 {
466 pTheta = std::atan2(rho,p.z());
467 distSTheta = std::fabs(pTheta-fSTheta);
468 distETheta = std::fabs(pTheta-eTheta);
469
470 nTs = G4ThreeVector(-cosSTheta*p.x()/rho,
471 -cosSTheta*p.y()/rho,
472 sinSTheta );
473
474 nTe = G4ThreeVector( cosETheta*p.x()/rho,
475 cosETheta*p.y()/rho,
476 -sinETheta );
477 }
478 else if( !fRmin )
479 {
480 if ( fSTheta )
481 {
482 distSTheta = 0.;
483 nTs = G4ThreeVector(0.,0.,-1.);
484 }
485 if ( eTheta < pi )
486 {
487 distETheta = 0.;
488 nTe = G4ThreeVector(0.,0.,1.);
489 }
490 }
491 }
492 if( radius ) { nR = G4ThreeVector(p.x()/radius,p.y()/radius,p.z()/radius); }
493
494 if( distRMax <= halfCarTolerance )
495 {
496 noSurfaces ++;
497 sumnorm += nR;
498 }
499 if( fRmin && (distRMin <= halfCarTolerance) )
500 {
501 noSurfaces ++;
502 sumnorm -= nR;
503 }
504 if( !fFullPhiSphere )
505 {
506 if (distSPhi <= halfAngTolerance)
507 {
508 noSurfaces ++;
509 sumnorm += nPs;
510 }
511 if (distEPhi <= halfAngTolerance)
512 {
513 noSurfaces ++;
514 sumnorm += nPe;
515 }
516 }
517 if ( !fFullThetaSphere )
518 {
519 if ((distSTheta <= halfAngTolerance) && (fSTheta > 0.))
520 {
521 noSurfaces ++;
522 if ((radius <= halfCarTolerance) && fFullPhiSphere) { sumnorm += nZ; }
523 else { sumnorm += nTs; }
524 }
525 if ((distETheta <= halfAngTolerance) && (eTheta < pi))
526 {
527 noSurfaces ++;
528 if ((radius <= halfCarTolerance) && fFullPhiSphere) { sumnorm -= nZ; }
529 else { sumnorm += nTe; }
530 if(sumnorm.z() == 0.) { sumnorm += nZ; }
531 }
532 }
533 if ( noSurfaces == 0 )
534 {
535#ifdef G4CSGDEBUG
536 G4Exception("G4Sphere::SurfaceNormal(p)", "GeomSolids1002",
537 JustWarning, "Point p is not on surface !?" );
538#endif
539 norm = ApproxSurfaceNormal(p);
540 }
541 else if ( noSurfaces == 1 ) { norm = sumnorm; }
542 else { norm = sumnorm.unit(); }
543 return norm;
544}

547/////////////////////////////////////////////////////////////////////
548//
549// Algorithm for SurfaceNormal() following the original specification
550// for points not on the surface
551
552G4ThreeVector G4Sphere::ApproxSurfaceNormal( const G4ThreeVector& p ) const
553{
554 ENorm side;
555 G4ThreeVector norm;
556 G4double rho,rho2,radius,pPhi,pTheta;
557 G4double distRMin,distRMax,distSPhi,distEPhi,
558 distSTheta,distETheta,distMin;
559
560 rho2=p.x()*p.x()+p.y()*p.y();
561 radius=std::sqrt(rho2+p.z()*p.z());
562 rho=std::sqrt(rho2);
563
564 //
565 // Distance to r shells
566 //
567
568 distRMax=std::fabs(radius-fRmax);
569 if (fRmin)
570 {
571 distRMin=std::fabs(radius-fRmin);
572
573 if (distRMin<distRMax)
574 {
575 distMin=distRMin;
576 side=kNRMin;
577 }
578 else
579 {
580 distMin=distRMax;
581 side=kNRMax;
582 }
583 }
584 else
585 {
586 distMin=distRMax;
587 side=kNRMax;
588 }
589
590 //
591 // Distance to phi planes
592 //
593 // Protected against (0,0,z)
594
595 pPhi = std::atan2(p.y(),p.x());
596 if (pPhi<0) { pPhi += twopi; }
597
598 if (!fFullPhiSphere && rho)
599 {
600 if (fSPhi<0)
601 {
602 distSPhi=std::fabs(pPhi-(fSPhi+twopi))*rho;
603 }
604 else
605 {
606 distSPhi=std::fabs(pPhi-fSPhi)*rho;
607 }
608
609 distEPhi=std::fabs(pPhi-fSPhi-fDPhi)*rho;
610
611 // Find new minimum
612 //
613 if (distSPhi<distEPhi)
614 {
615 if (distSPhi<distMin)
616 {
617 distMin=distSPhi;
618 side=kNSPhi;
619 }
620 }
621 else
622 {
623 if (distEPhi<distMin)
624 {
625 distMin=distEPhi;
626 side=kNEPhi;
627 }
628 }
629 }
630
631 //
632 // Distance to theta planes
633 //
634
635 if (!fFullThetaSphere && radius)
636 {
637 pTheta=std::atan2(rho,p.z());
638 distSTheta=std::fabs(pTheta-fSTheta)*radius;
639 distETheta=std::fabs(pTheta-fSTheta-fDTheta)*radius;
640
641 // Find new minimum
642 //
643 if (distSTheta<distETheta)
644 {
645 if (distSTheta<distMin)
646 {
647 distMin = distSTheta ;
648 side = kNSTheta ;
649 }
650 }
651 else
652 {
653 if (distETheta<distMin)
654 {
655 distMin = distETheta ;
656 side = kNETheta ;
657 }
658 }
659 }
660
661 switch (side)
662 {
663 case kNRMin: // Inner radius
664 norm=G4ThreeVector(-p.x()/radius,-p.y()/radius,-p.z()/radius);
665 break;
666 case kNRMax: // Outer radius
667 norm=G4ThreeVector(p.x()/radius,p.y()/radius,p.z()/radius);
668 break;
669 case kNSPhi:
670 norm=G4ThreeVector(sinSPhi,-cosSPhi,0);
671 break;
672 case kNEPhi:
673 norm=G4ThreeVector(-sinEPhi,cosEPhi,0);
674 break;
675 case kNSTheta:

1766//
1767// Calculate distance to surface of shape from 'inside', allowing for tolerance
1768// - Only Calc rmax intersection if no valid rmin intersection

1781 const G4double halfRminTolerance = fRminTolerance*0.5;

1786
1787 // Variables for phi intersection:
1788
1789 G4double pDistS,compS,pDistE,compE,sphi2,vphi;
1790
1791 G4double rho2,rad2,pDotV2d,pDotV3d;
1792
1793 G4double xi,yi,zi; // Intersection point
1794
1795 // Theta precals
1796 //
1797 G4double rhoSecTheta;
1798 G4double dist2STheta, dist2ETheta, distTheta;
1799 G4double d2,sd;
1800
1801 // General Precalcs
1802 //
1803 rho2 = p.x()*p.x()+p.y()*p.y();
1804 rad2 = rho2+p.z()*p.z();
1805
1806 pDotV2d = p.x()*v.x()+p.y()*v.y();
1807 pDotV3d = pDotV2d+p.z()*v.z();

1809 // Radial Intersections from G4Sphere::DistanceToIn
1810 //
1811 // Outer spherical shell intersection
1812 // - Only if outside tolerant fRmax
1813 // - Check for if inside and outer G4Sphere heading through solid (-> 0)
1814 // - No intersect -> no intersection with G4Sphere
1815 //
1816 // Shell eqn: x^2+y^2+z^2=RSPH^2
1817 //
1818 // => (px+svx)^2+(py+svy)^2+(pz+svz)^2=R^2
1819 //
1820 // => (px^2+py^2+pz^2) +2sd(pxvx+pyvy+pzvz)+sd^2(vx^2+vy^2+vz^2)=R^2
1821 // => rad2 +2sd(pDotV3d) +sd^2 =R^2

1823 // => sd=-pDotV3d+-std::sqrt(pDotV3d^2-(rad2-R^2))
1824

1826 {
1827 c = rad2 - fRmax*fRmax;
1828
1829 if (c < fRmaxTolerance*fRmax)
1830 {
1831 // Within tolerant Outer radius
1832 //
1833 // The test is
1834 // rad - fRmax < 0.5*kRadTolerance
1835 // => rad < fRmax + 0.5*kRadTol
1836 // => rad2 < (fRmax + 0.5*kRadTol)^2
1837 // => rad2 < fRmax^2 + 2.*0.5*fRmax*kRadTol + 0.25*kRadTol*kRadTol
1838 // => rad2 - fRmax^2 <~ fRmax*kRadTol
1839
1840 d2 = pDotV3d*pDotV3d - c;
1841
1842 if( (c >- fRmaxTolerance*fRmax) // on tolerant surface
1843 && ((pDotV3d >=0) || (d2 < 0)) ) // leaving outside from Rmax
1844 // not re-entering
1845 {
1846 if(calcNorm)
1847 {
1848 *validNorm = true ;
1849 *n = G4ThreeVector(p.x()/fRmax,p.y()/fRmax,p.z()/fRmax) ;
1850 }
1851 return snxt = 0;
1852 }
1853 else
1854 {
1855 snxt = -pDotV3d+std::sqrt(d2); // second root since inside Rmax
1856 side = kRMax ;
1857 }
1858 }
1859
1860 // Inner spherical shell intersection:
1861 // Always first >=0 root, because would have passed
1862 // from outside of Rmin surface .
1863
1864 if (fRmin)
1865 {
1866 c = rad2 - fRmin*fRmin;
1867 d2 = pDotV3d*pDotV3d - c;
1868
1869 if (c >- fRminTolerance*fRmin) // 2.0 * (0.5*kRadTolerance) * fRmin
1870 {
1871 if ( (c < fRminTolerance*fRmin) // leaving from Rmin
1872 && (d2 >= fRminTolerance*fRmin) && (pDotV3d < 0) )
1873 {
1874 if(calcNorm) { *validNorm = false; } // Rmin surface is concave
1875 return snxt = 0 ;
1876 }
1877 else
1878 {
1879 if ( d2 >= 0. )
1880 {
1881 sd = -pDotV3d-std::sqrt(d2);
1882
1883 if ( sd >= 0. ) // Always intersect Rmin first
1884 {
1885 snxt = sd ;
1886 side = kRMin ;
1887 }
1888 }
1889 }
1890 }
1891 }
1892 }
1893
1894 // Theta segment intersection
1895
1896 if ( !fFullThetaSphere )
1897 {
1898 // Intersection with theta surfaces
1899 //
1900 // Known failure cases:
1901 // o Inside tolerance of stheta surface, skim
1902 // ~parallel to cone and Hit & enter etheta surface [& visa versa]
1903 //
1904 // To solve: Check 2nd root of etheta surface in addition to stheta

1906 // o start/end theta is exactly pi/2

1908 // Intersections with cones
1909 //
1910 // Cone equation: x^2+y^2=z^2tan^2(t)
1911 //
1912 // => (px+svx)^2+(py+svy)^2=(pz+svz)^2tan^2(t)

1914 // => (px^2+py^2-pz^2tan^2(t))+2sd(pxvx+pyvy-pzvztan^2(t))
1915 // + sd^2(vx^2+vy^2-vz^2tan^2(t)) = 0
1916 //
1917 // => sd^2(1-vz^2(1+tan^2(t))+2sd(pdotv2d-pzvztan^2(t))

1919 //
1920

2816 costheta = G4RandFlat::shoot(cosETheta,cosSTheta);
2817 sintheta = std::sqrt(1.-sqr(costheta));
2818
2819 if(fFullPhiSphere) { aFiv = 0; }
2820 if(fSTheta == 0) { aThr=0; }
2821 if(eTheta == pi) { aFou = 0; }
2822 if(fSTheta == halfpi) { aThr = pi*(fRmax*fRmax-fRmin*fRmin); }
2823 if(eTheta == halfpi) { aFou = pi*(fRmax*fRmax-fRmin*fRmin); }
2824
2825 chose = G4RandFlat::shoot(0.,aOne+aTwo+aThr+aFou+2.*aFiv);
2826 if( (chose>=0.) && (chose<aOne) )
2827 {
2828 return G4ThreeVector(fRmax*sintheta*cosphi,
2829 fRmax*sintheta*sinphi, fRmax*costheta);
2830 }
2831 else if( (chose>=aOne) && (chose<aOne+aTwo) )
2832 {
2833 return G4ThreeVector(fRmin*sintheta*cosphi,
2834 fRmin*sintheta*sinphi, fRmin*costheta);
2835 }
2836 else if( (chose>=aOne+aTwo) && (chose<aOne+aTwo+aThr) )
2837 {
2838 if (fSTheta != halfpi)
2839 {
2840 zRand = G4RandFlat::shoot(fRmin*cosSTheta,fRmax*cosSTheta);
2841 return G4ThreeVector(tanSTheta*zRand*cosphi,
2842 tanSTheta*zRand*sinphi,zRand);
2843 }
2844 else
2845 {
2846 return G4ThreeVector(rRand*cosphi, rRand*sinphi, 0.);
2847 }
2848 }
2849 else if( (chose>=aOne+aTwo+aThr) && (chose<aOne+aTwo+aThr+aFou) )
2850 {
2851 if(eTheta != halfpi)
2852 {
2853 zRand = G4RandFlat::shoot(fRmin*cosETheta, fRmax*cosETheta);
2854 return G4ThreeVector (tanETheta*zRand*cosphi,
2855 tanETheta*zRand*sinphi,zRand);
2856 }
2857 else
2858 {
2859 return G4ThreeVector(rRand*cosphi, rRand*sinphi, 0.);
2860 }
2861 }
2862 else if( (chose>=aOne+aTwo+aThr+aFou) && (chose<aOne+aTwo+aThr+aFou+aFiv) )
2863 {
2864 return G4ThreeVector(rRand*sintheta*cosSPhi,
2865 rRand*sintheta*sinSPhi,rRand*costheta);
2866 }
2867 else
2868 {
2869 return G4ThreeVector(rRand*sintheta*cosEPhi,
2870 rRand*sintheta*sinEPhi,rRand*costheta);
2871 }
2872}

2898 }
