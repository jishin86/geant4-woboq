79//////////////////////////////////////////////////////////////////////////
80//
81// constructor - check parameters, convert angles so 0<sphi+dpshi<=2_PI
82// - note if pDPhi>2PI then reset to 2PI
83
84G4Cons::G4Cons( const G4String& pName,
85 G4double pRmin1, G4double pRmax1,
86 G4double pRmin2, G4double pRmax2,
87 G4double pDz,
88 G4double pSPhi, G4double pDPhi)
89 : G4CSGSolid(pName), fRmin1(pRmin1), fRmin2(pRmin2),
90 fRmax1(pRmax1), fRmax2(pRmax2), fDz(pDz), fSPhi(0.), fDPhi(0.)
91{
92 kRadTolerance = G4GeometryTolerance::GetInstance()->GetRadialTolerance();
93 kAngTolerance = G4GeometryTolerance::GetInstance()->GetAngularTolerance();
94
95 halfCarTolerance=kCarTolerance*0.5;
96 halfRadTolerance=kRadTolerance*0.5;
97 halfAngTolerance=kAngTolerance*0.5;
98
99 // Check z-len
100 //
101 if ( pDz < 0 )
102 {
103 std::ostringstream message;
104 message << "Invalid Z half-length for Solid: " << GetName() << G4endl
105 << " hZ = " << pDz;
106 G4Exception("G4Cons::G4Cons()", "GeomSolids0002",
107 FatalException, message);
108 }
109
110 // Check radii
111 //
112 if (((pRmin1>=pRmax1) || (pRmin2>=pRmax2) || (pRmin1<0)) && (pRmin2<0))
113 {
114 std::ostringstream message;
115 message << "Invalid values of radii for Solid: " << GetName() << G4endl
116 << " pRmin1 = " << pRmin1 << ", pRmin2 = " << pRmin2
117 << ", pRmax1 = " << pRmax1 << ", pRmax2 = " << pRmax2;
118 G4Exception("G4Cons::G4Cons()", "GeomSolids0002",
119 FatalException, message) ;
120 }
121 if( (pRmin1 == 0.0) && (pRmin2 > 0.0) ) { fRmin1 = 1e3*kRadTolerance ; }
122 if( (pRmin2 == 0.0) && (pRmin1 > 0.0) ) { fRmin2 = 1e3*kRadTolerance ; }
123
124 // Check angles
125 //
126 CheckPhiAngles(pSPhi, pDPhi);
127}

203/////////////////////////////////////////////////////////////////////
204//
205// Return whether point inside/outside/on surface
206
207EInside G4Cons::Inside(const G4ThreeVector& p) const
208{
209 G4double r2, rl, rh, pPhi, tolRMin, tolRMax; // rh2, rl2 ;
210 EInside in;
211
212 if (std::fabs(p.z()) > fDz + halfCarTolerance ) { return in = kOutside; }
213 else if(std::fabs(p.z()) >= fDz - halfCarTolerance ) { in = kSurface; }
214 else { in = kInside; }
215
216 r2 = p.x()*p.x() + p.y()*p.y() ;
217 rl = 0.5*(fRmin2*(p.z() + fDz) + fRmin1*(fDz - p.z()))/fDz ;
218 rh = 0.5*(fRmax2*(p.z()+fDz)+fRmax1*(fDz-p.z()))/fDz;
219
220 // rh2 = rh*rh;
221
222 tolRMin = rl - halfRadTolerance;
223 if ( tolRMin < 0 ) { tolRMin = 0; }
224 tolRMax = rh + halfRadTolerance;
225
226 if ( (r2<tolRMin*tolRMin) || (r2>tolRMax*tolRMax) ) { return in = kOutside; }
227
228 if (rl) { tolRMin = rl + halfRadTolerance; }
229 else { tolRMin = 0.0; }
230 tolRMax = rh - halfRadTolerance;
231
232 if (in == kInside) // else it's kSurface already
233 {
234 if ( (r2 < tolRMin*tolRMin) || (r2 >= tolRMax*tolRMax) ) { in = kSurface; }
235 }
236 if ( !fPhiFullCone && ((p.x() != 0.0) || (p.y() != 0.0)) )
237 {
238 pPhi = std::atan2(p.y(),p.x()) ;
239
240 if ( pPhi < fSPhi - halfAngTolerance ) { pPhi += twopi; }
241 else if ( pPhi > fSPhi + fDPhi + halfAngTolerance ) { pPhi -= twopi; }
242
243 if ( (pPhi < fSPhi - halfAngTolerance) ||
244 (pPhi > fSPhi + fDPhi + halfAngTolerance) ) { return in = kOutside; }
245
246 else if (in == kInside) // else it's kSurface anyway already
247 {
248 if ( (pPhi < fSPhi + halfAngTolerance) ||
249 (pPhi > fSPhi + fDPhi - halfAngTolerance) ) { in = kSurface; }
250 }
251 }
252 else if ( !fPhiFullCone ) { in = kSurface; }
253
254 return in ;
255}

269///////////////////////////////////////////////////////////////////////
270//
271// Get bounding box
272
273void G4Cons::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
274{
275 G4double rmin = std::min(GetInnerRadiusMinusZ(),GetInnerRadiusPlusZ());
276 G4double rmax = std::max(GetOuterRadiusMinusZ(),GetOuterRadiusPlusZ());
277 G4double dz = GetZHalfLength();
278
279 // Find bounding box
280 //
281 if (GetDeltaPhiAngle() < twopi)
282 {
283 G4TwoVector vmin,vmax;
284 G4GeomTools::DiskExtent(rmin,rmax,
285 GetSinStartPhi(),GetCosStartPhi(),
286 GetSinEndPhi(),GetCosEndPhi(),
287 vmin,vmax);
288 pMin.set(vmin.x(),vmin.y(),-dz);
289 pMax.set(vmax.x(),vmax.y(), dz);
290 }
291 else
292 {
293 pMin.set(-rmax,-rmax,-dz);
294 pMax.set( rmax, rmax, dz);
295 }
296
297 // Check correctness of the bounding box
298 //
299 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
300 {
301 std::ostringstream message;
302 message << "Bad bounding box (min >= max) for solid: "
303 << GetName() << " !"
304 << "\npMin = " << pMin
305 << "\npMax = " << pMax;
306 G4Exception("G4Cons::BoundingLimits()", "GeomMgt0001",
307 JustWarning, message);
308 DumpInfo();
309 }
310}

312///////////////////////////////////////////////////////////////////////
313//
314// Calculate extent under transform and specified limit
315
316G4bool G4Cons::CalculateExtent( const EAxis pAxis,
317 const G4VoxelLimits& pVoxelLimit,
318 const G4AffineTransform& pTransform,
319 G4double& pMin,
320 G4double& pMax ) const
321{
322 G4ThreeVector bmin, bmax;
323 G4bool exist;
324
325 // Get bounding box
326 BoundingLimits(bmin,bmax);
327
328 // Check bounding box
329 G4BoundingEnvelope bbox(bmin,bmax);
330#ifdef G4BBOX_EXTENT
331 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
332#endif
333 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
334 {
335 return exist = (pMin < pMax) ? true : false;
336 }
337
338 // Get parameters of the solid
339 G4double rmin1 = GetInnerRadiusMinusZ();
340 G4double rmax1 = GetOuterRadiusMinusZ();
341 G4double rmin2 = GetInnerRadiusPlusZ();
342 G4double rmax2 = GetOuterRadiusPlusZ();
343 G4double dz = GetZHalfLength();
344 G4double dphi = GetDeltaPhiAngle();
345
346 // Find bounding envelope and calculate extent
347 //
348 const G4int NSTEPS = 24; // number of steps for whole circle
349 G4double astep = twopi/NSTEPS; // max angle for one step
350 G4int ksteps = (dphi <= astep) ? 1 : (G4int)((dphi-deg)/astep) + 1;
351 G4double ang = dphi/ksteps;
352
353 G4double sinHalf = std::sin(0.5*ang);
354 G4double cosHalf = std::cos(0.5*ang);
355 G4double sinStep = 2.*sinHalf*cosHalf;
356 G4double cosStep = 1. - 2.*sinHalf*sinHalf;
357 G4double rext1 = rmax1/cosHalf;
358 G4double rext2 = rmax2/cosHalf;
359
360 // bounding envelope for full cone without hole consists of two polygons,
361 // in other cases it is a sequence of quadrilaterals
362 if (rmin1 == 0 && rmin2 == 0 && dphi == twopi)
363 {
364 G4double sinCur = sinHalf;
365 G4double cosCur = cosHalf;
366
367 G4ThreeVectorList baseA(NSTEPS),baseB(NSTEPS);
368 for (G4int k=0; k<NSTEPS; ++k)
369 {
370 baseA[k].set(rext1*cosCur,rext1*sinCur,-dz);
371 baseB[k].set(rext2*cosCur,rext2*sinCur, dz);
372
373 G4double sinTmp = sinCur;
374 sinCur = sinCur*cosStep + cosCur*sinStep;
375 cosCur = cosCur*cosStep - sinTmp*sinStep;
376 }
377 std::vector<const G4ThreeVectorList *> polygons(2);
378 polygons[0] = &baseA;
379 polygons[1] = &baseB;
380 G4BoundingEnvelope benv(bmin,bmax,polygons);
381 exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
382 }
383 else
384 {
385 G4double sinStart = GetSinStartPhi();
386 G4double cosStart = GetCosStartPhi();
387 G4double sinEnd = GetSinEndPhi();
388 G4double cosEnd = GetCosEndPhi();
389 G4double sinCur = sinStart*cosHalf + cosStart*sinHalf;
390 G4double cosCur = cosStart*cosHalf - sinStart*sinHalf;
391
392 // set quadrilaterals
393 G4ThreeVectorList pols[NSTEPS+2];
394 for (G4int k=0; k<ksteps+2; ++k) pols[k].resize(4);
395 pols[0][0].set(rmin2*cosStart,rmin2*sinStart, dz);
396 pols[0][1].set(rmin1*cosStart,rmin1*sinStart,-dz);
397 pols[0][2].set(rmax1*cosStart,rmax1*sinStart,-dz);
398 pols[0][3].set(rmax2*cosStart,rmax2*sinStart, dz);
399 for (G4int k=1; k<ksteps+1; ++k)
400 {
401 pols[k][0].set(rmin2*cosCur,rmin2*sinCur, dz);
402 pols[k][1].set(rmin1*cosCur,rmin1*sinCur,-dz);
403 pols[k][2].set(rext1*cosCur,rext1*sinCur,-dz);
404 pols[k][3].set(rext2*cosCur,rext2*sinCur, dz);
405
406 G4double sinTmp = sinCur;
407 sinCur = sinCur*cosStep + cosCur*sinStep;
408 cosCur = cosCur*cosStep - sinTmp*sinStep;
409 }
410 pols[ksteps+1][0].set(rmin2*cosEnd,rmin2*sinEnd, dz);
411 pols[ksteps+1][1].set(rmin1*cosEnd,rmin1*sinEnd,-dz);
412 pols[ksteps+1][2].set(rmax1*cosEnd,rmax1*sinEnd,-dz);
413 pols[ksteps+1][3].set(rmax2*cosEnd,rmax2*sinEnd, dz);
414
415 // set envelope and calculate extent
416 std::vector<const G4ThreeVectorList *> polygons;
417 polygons.resize(ksteps+2);
418 for (G4int k=0; k<ksteps+2; ++k) polygons[k] = &pols[k];
419 G4BoundingEnvelope benv(bmin,bmax,polygons);
420 exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
421 }
422 return exist;
423}

425////////////////////////////////////////////////////////////////////////
426//
427// Return unit normal of surface closest to p
428// - note if point on z axis, ignore phi divided sides
429// - unsafe if point close to z axis a rmin=0 - no explicit checks
430
431G4ThreeVector G4Cons::SurfaceNormal( const G4ThreeVector& p) const
432{
433 G4int noSurfaces = 0;
434 G4double rho, pPhi;
435 G4double distZ, distRMin, distRMax;
436 G4double distSPhi = kInfinity, distEPhi = kInfinity;
437 G4double tanRMin, secRMin, pRMin, widRMin;
438 G4double tanRMax, secRMax, pRMax, widRMax;
439
440 G4ThreeVector norm, sumnorm(0.,0.,0.), nZ = G4ThreeVector(0.,0.,1.);
441 G4ThreeVector nR, nr(0.,0.,0.), nPs, nPe;
442
443 distZ = std::fabs(std::fabs(p.z()) - fDz);
444 rho = std::sqrt(p.x()*p.x() + p.y()*p.y());
445
446 tanRMin = (fRmin2 - fRmin1)*0.5/fDz;
447 secRMin = std::sqrt(1 + tanRMin*tanRMin);
448 pRMin = rho - p.z()*tanRMin;
449 widRMin = fRmin2 - fDz*tanRMin;
450 distRMin = std::fabs(pRMin - widRMin)/secRMin;
451
452 tanRMax = (fRmax2 - fRmax1)*0.5/fDz;
453 secRMax = std::sqrt(1+tanRMax*tanRMax);
454 pRMax = rho - p.z()*tanRMax;
455 widRMax = fRmax2 - fDz*tanRMax;
456 distRMax = std::fabs(pRMax - widRMax)/secRMax;
457
458 if (!fPhiFullCone) // Protected against (0,0,z)
459 {
460 if ( rho )
461 {
462 pPhi = std::atan2(p.y(),p.x());
463
464 if (pPhi < fSPhi-halfCarTolerance) { pPhi += twopi; }
465 else if (pPhi > fSPhi+fDPhi+halfCarTolerance) { pPhi -= twopi; }
466
467 distSPhi = std::fabs( pPhi - fSPhi );
468 distEPhi = std::fabs( pPhi - fSPhi - fDPhi );
469 }
470 else if( !(fRmin1) || !(fRmin2) )
471 {
472 distSPhi = 0.;
473 distEPhi = 0.;
474 }
475 nPs = G4ThreeVector(std::sin(fSPhi), -std::cos(fSPhi), 0);
476 nPe = G4ThreeVector(-std::sin(fSPhi+fDPhi), std::cos(fSPhi+fDPhi), 0);
477 }
478 if ( rho > halfCarTolerance )
479 {
480 nR = G4ThreeVector(p.x()/rho/secRMax, p.y()/rho/secRMax, -tanRMax/secRMax);
481 if (fRmin1 || fRmin2)
482 {
483 nr = G4ThreeVector(-p.x()/rho/secRMin,-p.y()/rho/secRMin,tanRMin/secRMin);
484 }
485 }
486
487 if( distRMax <= halfCarTolerance )
488 {
489 noSurfaces ++;
490 sumnorm += nR;
491 }
492 if( (fRmin1 || fRmin2) && (distRMin <= halfCarTolerance) )
493 {
494 noSurfaces ++;
495 sumnorm += nr;
496 }
497 if( !fPhiFullCone )
498 {
499 if (distSPhi <= halfAngTolerance)
500 {
501 noSurfaces ++;
502 sumnorm += nPs;
503 }
504 if (distEPhi <= halfAngTolerance)
505 {
506 noSurfaces ++;
507 sumnorm += nPe;
508 }
509 }
510 if (distZ <= halfCarTolerance)
511 {
512 noSurfaces ++;
513 if ( p.z() >= 0.) { sumnorm += nZ; }
514 else { sumnorm -= nZ; }
515 }
516 if ( noSurfaces == 0 )
517 {
518#ifdef G4CSGDEBUG
519 G4Exception("G4Cons::SurfaceNormal(p)", "GeomSolids1002",
520 JustWarning, "Point p is not on surface !?" );
521#endif
522 norm = ApproxSurfaceNormal(p);
523 }
524 else if ( noSurfaces == 1 ) { norm = sumnorm; }
525 else { norm = sumnorm.unit(); }
526
527 return norm ;
528}

530////////////////////////////////////////////////////////////////////////////
531//
532// Algorithm for SurfaceNormal() following the original specification
533// for points not on the surface
534
535G4ThreeVector G4Cons::ApproxSurfaceNormal( const G4ThreeVector& p ) const
536{
537 ENorm side ;
538 G4ThreeVector norm ;
539 G4double rho, phi ;
540 G4double distZ, distRMin, distRMax, distSPhi, distEPhi, distMin ;
541 G4double tanRMin, secRMin, pRMin, widRMin ;
542 G4double tanRMax, secRMax, pRMax, widRMax ;
543
544 distZ = std::fabs(std::fabs(p.z()) - fDz) ;
545 rho = std::sqrt(p.x()*p.x() + p.y()*p.y()) ;
546
547 tanRMin = (fRmin2 - fRmin1)*0.5/fDz ;
548 secRMin = std::sqrt(1 + tanRMin*tanRMin) ;
549 pRMin = rho - p.z()*tanRMin ;
550 widRMin = fRmin2 - fDz*tanRMin ;
551 distRMin = std::fabs(pRMin - widRMin)/secRMin ;
552
553 tanRMax = (fRmax2 - fRmax1)*0.5/fDz ;
554 secRMax = std::sqrt(1+tanRMax*tanRMax) ;
555 pRMax = rho - p.z()*tanRMax ;
556 widRMax = fRmax2 - fDz*tanRMax ;
557 distRMax = std::fabs(pRMax - widRMax)/secRMax ;
558
559 if (distRMin < distRMax) // First minimum
560 {
561 if (distZ < distRMin)
562 {
563 distMin = distZ ;
564 side = kNZ ;
565 }
566 else
567 {
568 distMin = distRMin ;
569 side = kNRMin ;
570 }
571 }
572 else
573 {
574 if (distZ < distRMax)
575 {
576 distMin = distZ ;
577 side = kNZ ;
578 }
579 else
580 {
581 distMin = distRMax ;
582 side = kNRMax ;
583 }
584 }
585 if ( !fPhiFullCone && rho ) // Protected against (0,0,z)
586 {
587 phi = std::atan2(p.y(),p.x()) ;
588
589 if (phi < 0) { phi += twopi; }
590
591 if (fSPhi < 0) { distSPhi = std::fabs(phi - (fSPhi + twopi))*rho; }
592 else { distSPhi = std::fabs(phi - fSPhi)*rho; }
593
594 distEPhi = std::fabs(phi - fSPhi - fDPhi)*rho ;
595
596 // Find new minimum
597
598 if (distSPhi < distEPhi)
599 {
600 if (distSPhi < distMin) { side = kNSPhi; }
601 }
602 else
603 {
604 if (distEPhi < distMin) { side = kNEPhi; }
605 }
606 }
607 switch (side)
608 {
609 case kNRMin: // Inner radius
610 rho *= secRMin ;
611 norm = G4ThreeVector(-p.x()/rho, -p.y()/rho, tanRMin/secRMin) ;
612 break ;
613 case kNRMax: // Outer radius
614 rho *= secRMax ;
615 norm = G4ThreeVector(p.x()/rho, p.y()/rho, -tanRMax/secRMax) ;
616 break ;
617 case kNZ: // +/- dz
618 if (p.z() > 0) { norm = G4ThreeVector(0,0,1); }
619 else { norm = G4ThreeVector(0,0,-1); }
620 break ;
621 case kNSPhi:
622 norm = G4ThreeVector(std::sin(fSPhi), -std::cos(fSPhi), 0) ;
623 break ;
624 case kNEPhi:
625 norm=G4ThreeVector(-std::sin(fSPhi+fDPhi), std::cos(fSPhi+fDPhi), 0) ;
626 break ;
627 default: // Should never reach this case...
628 DumpInfo();
629 G4Exception("G4Cons::ApproxSurfaceNormal()",
630 "GeomSolids1002", JustWarning,
631 "Undefined side for valid surface normal to solid.");
632 break ;
633 }
634 return norm ;
635}

638//
639// Calculate distance to shape from outside, along normalised vector
640// - return kInfinity if no intersection, or intersection distance <= tolerance
641//
642// - Compute the intersection with the z planes
643// - if at valid r, phi, return
644//
645// -> If point is outside cone, compute intersection with rmax1*0.5
646// - if at valid phi,z return
647// - if inside outer cone, handle case when on tolerant outer cone
648// boundary and heading inwards(->0 to in)
649//
650// -> Compute intersection with inner cone, taking largest +ve root
651// - if valid (in z,phi), save intersction
652//
653// -> If phi segmented, compute intersections with phi half planes
654// - return smallest of valid phi intersections and
655// inner radius intersection
656//
657// NOTE:
658// - `if valid' implies tolerant checking of intersection points
659// - z, phi intersection from Tubs

661G4double G4Cons::DistanceToIn( const G4ThreeVector& p,
662 const G4ThreeVector& v ) const

665 const G4double dRmax = 50*(fRmax1+fRmax2);// 100*(Rmax1+Rmax2)/2.

667 G4double tanRMax,secRMax,rMaxAv,rMaxOAv ; // Data for cones
668 G4double tanRMin,secRMin,rMinAv,rMinOAv ;
669 G4double rout,rin ;
670
671 G4double tolORMin,tolORMin2,tolIRMin,tolIRMin2 ; // `generous' radii squared
672 G4double tolORMax2,tolIRMax,tolIRMax2 ;
673 G4double tolODz,tolIDz ;
674
675 G4double Dist,sd,xi,yi,zi,ri=0.,risec,rhoi2,cosPsi ; // Intersection point vars
676
677 G4double t1,t2,t3,b,c,d ; // Quadratic solver variables
678 G4double nt1,nt2,nt3 ;
679 G4double Comp ;
680
681 G4ThreeVector Normal;
682
683 // Cone Precalcs
684
685 tanRMin = (fRmin2 - fRmin1)*0.5/fDz ;
686 secRMin = std::sqrt(1.0 + tanRMin*tanRMin) ;
687 rMinAv = (fRmin1 + fRmin2)*0.5 ;
688
689 if (rMinAv > halfRadTolerance)
690 {
691 rMinOAv = rMinAv - halfRadTolerance ;
692 }
693 else
694 {
695 rMinOAv = 0.0 ;
696 }
697 tanRMax = (fRmax2 - fRmax1)*0.5/fDz ;
698 secRMax = std::sqrt(1.0 + tanRMax*tanRMax) ;
699 rMaxAv = (fRmax1 + fRmax2)*0.5 ;
700 rMaxOAv = rMaxAv + halfRadTolerance ;
701
702 // Intersection with z-surfaces
703
704 tolIDz = fDz - halfCarTolerance ;
705 tolODz = fDz + halfCarTolerance ;
706
707 if (std::fabs(p.z()) >= tolIDz)
708 {
709 if ( p.z()*v.z() < 0 ) // at +Z going in -Z or visa versa
710 {
711 sd = (std::fabs(p.z()) - fDz)/std::fabs(v.z()) ; // Z intersect distance
712
713 if( sd < 0.0 ) { sd = 0.0; } // negative dist -> zero
714
715 xi = p.x() + sd*v.x() ; // Intersection coords
716 yi = p.y() + sd*v.y() ;

718
719 // Check validity of intersection
720 // Calculate (outer) tolerant radi^2 at intersecion
721
722 if (v.z() > 0)
723 {
724 tolORMin = fRmin1 - halfRadTolerance*secRMin ;
725 tolIRMin = fRmin1 + halfRadTolerance*secRMin ;
726 tolIRMax = fRmax1 - halfRadTolerance*secRMin ;
727 // tolORMax2 = (fRmax1 + halfRadTolerance*secRMax)*
728 // (fRmax1 + halfRadTolerance*secRMax) ;
729 }
730 else
731 {
732 tolORMin = fRmin2 - halfRadTolerance*secRMin ;
733 tolIRMin = fRmin2 + halfRadTolerance*secRMin ;
734 tolIRMax = fRmax2 - halfRadTolerance*secRMin ;
735 // tolORMax2 = (fRmax2 + halfRadTolerance*secRMax)*
736 // (fRmax2 + halfRadTolerance*secRMax) ;
737 }
738 if ( tolORMin > 0 )
739 {
740 // tolORMin2 = tolORMin*tolORMin ;
741 tolIRMin2 = tolIRMin*tolIRMin ;
742 }
743 else
744 {
745 // tolORMin2 = 0.0 ;
746 tolIRMin2 = 0.0 ;
747 }
748 if ( tolIRMax > 0 ) { tolIRMax2 = tolIRMax*tolIRMax; }
749 else { tolIRMax2 = 0.0; }
750
751 if ( (tolIRMin2 <= rhoi2) && (rhoi2 <= tolIRMax2) )
752 {
753 if ( !fPhiFullCone && rhoi2 )
754 {
755 // Psi = angle made with central (average) phi of shape
756
757 cosPsi = (xi*cosCPhi + yi*sinCPhi)/std::sqrt(rhoi2) ;
758
759 if (cosPsi >= cosHDPhiIT) { return sd; }
760 }
761 else
762 {
763 return sd;
764 }
765 }
766 }
767 else // On/outside extent, and heading away -> cannot intersect
768 {
769 return snxt ;
770 }
771 }
772
773// ----> Can not intersect z surfaces
774
775
776// Intersection with outer cone (possible return) and
777// inner cone (must also check phi)
778//
779// Intersection point (xi,yi,zi) on line x=p.x+t*v.x etc.
780//
781// Intersects with x^2+y^2=(a*z+b)^2
782//
783// where a=tanRMax or tanRMin
784// b=rMaxAv or rMinAv
785//
786// (vx^2+vy^2-(a*vz)^2)t^2+2t(pxvx+pyvy-a*vz(a*pz+b))+px^2+py^2-(a*pz+b)^2=0 ;
787// t1 t2 t3
788//
789// \--------u-------/ \-----------v----------/ \---------w--------/
790//

793 t2 = p.x()*v.x() + p.y()*v.y() ;
794 t3 = p.x()*p.x() + p.y()*p.y() ;

796 rout = tanRMax*p.z() + rMaxAv ;
797
798 // Outer Cone Intersection
799 // Must be outside/on outer cone for valid intersection
800

803 nt3 = t3 - rout*rout ;
804

806 {

808 c = nt3/nt1;

811 || (rout < 0) )

815

843 }
844 }

847 {

849 { // 64 bits systems. Split long distances and recompute

856 {

864 ri = rMaxAv + zi*tanRMax ;
865 cosPsi = (xi*cosCPhi + yi*sinCPhi)/ri ;
866
867 if ( cosPsi >= cosHDPhiIT ) { return sd; }
868 }
869 }
870 } // end if (sd>0)
871 }
872 }
873 else
874 {
875 // Inside outer cone
876 // check not inside, and heading through G4Cons (-> 0 to in)
877
878 if ( ( t3 > (rin + halfRadTolerance*secRMin)*
879 (rin + halfRadTolerance*secRMin) )
880 && (nt2 < 0) && (d >= 0) && (std::fabs(p.z()) <= tolIDz) )
881 {
882 // Inside cones, delta r -ve, inside z extent

884
885 xi = p.x() ;
886 yi = p.y() ;
887 risec = std::sqrt(xi*xi + yi*yi)*secRMax ;

890 {

895 }

897 else
898 {
899 if ( Normal.dot(v) <= 0 ) { return 0.0; }
900 }
901 }
902 }
903 }

905 {

907 {
908 sd = -0.5*nt3/nt2 ;
909

911 else // sd >= 0, If 'forwards'. Check z intersection
912 {

915 if ((std::fabs(zi) <= tolODz) && (nt2 < 0))

917 // Z ok. Check phi intersection if reqd
918

923 yi = p.y() + sd*v.y() ;

928 }

930 }

1326// Calculate distance (<= actual) to closest surface of shape from outside
1327// - Calculate distance to z, radial planes
1328// - Only to phi planes if outside phi extent
1329// - Return 0 if point inside

1331G4double G4Cons::DistanceToIn(const G4ThreeVector& p) const

1335 G4double tanRMax, secRMax, pRMax ;
1336

1338 safeZ = std::fabs(p.z()) - fDz ;
1339
1340 if ( fRmin1 || fRmin2 )
1341 {
1342 tanRMin = (fRmin2 - fRmin1)*0.5/fDz ;
1343 secRMin = std::sqrt(1.0 + tanRMin*tanRMin) ;
1344 pRMin = tanRMin*p.z() + (fRmin1 + fRmin2)*0.5 ;
1345 safeR1 = (pRMin - rho)/secRMin ;
1346
1347 tanRMax = (fRmax2 - fRmax1)*0.5/fDz ;
1348 secRMax = std::sqrt(1.0 + tanRMax*tanRMax) ;
1349 pRMax = tanRMax*p.z() + (fRmax1 + fRmax2)*0.5 ;
1350 safeR2 = (rho - pRMax)/secRMax ;
1351
1352 if ( safeR1 > safeR2) { safe = safeR1; }
1353 else { safe = safeR2; }
1354 }
1355 else
1356 {
1357 tanRMax = (fRmax2 - fRmax1)*0.5/fDz ;
1358 secRMax = std::sqrt(1.0 + tanRMax*tanRMax) ;

1361 }
1362 if ( safeZ > safe ) { safe = safeZ; }

1370 if ( cosPsi < std::cos(fDPhi*0.5) ) // Point lies outside phi range

1381 }
1382 }
1383 if ( safe < 0.0 ) { safe = 0.0; }
1384
