58////////////////////////////////////////////////////////////////////////
59//
60// Constructor - check & set half widths
61
62G4Box::G4Box(const G4String& pName,
63 G4double pX,
64 G4double pY,
65 G4double pZ)
66 : G4CSGSolid(pName), fDx(pX), fDy(pY), fDz(pZ)
67{
68 delta = 0.5*kCarTolerance;
69 if (pX < 2*kCarTolerance ||
70 pY < 2*kCarTolerance ||
71 pZ < 2*kCarTolerance) // limit to thickness of surfaces
72 {
73 std::ostringstream message;
74 message << "Dimensions too small for Solid: " << GetName() << "!" << G4endl
75 << " hX, hY, hZ = " << pX << ", " << pY << ", " << pZ;
76 G4Exception("G4Box::G4Box()", "GeomSolids0002", FatalException, message);
77 }
78}

107//////////////////////////////////////////////////////////////////////////
108//
109// Assignment operator
110
111G4Box& G4Box::operator = (const G4Box& rhs)
112{
113 // Check assignment to self
114 //
115 if (this == &rhs) { return *this; }
116
117 // Copy base class data
118 //
119 G4CSGSolid::operator=(rhs);
120
121 // Copy data
122 //
123 fDx = rhs.fDx;
124 fDy = rhs.fDy;
125 fDz = rhs.fDz;
126 delta = rhs.delta;
127
128 return *this;
129}

213//////////////////////////////////////////////////////////////////////////
214//
215// Get bounding box
216
217void G4Box::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
218{
219 pMin.set(-fDx,-fDy,-fDz);
220 pMax.set( fDx, fDy, fDz);
221
222 // Check correctness of the bounding box
223 //
224 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
225 {
226 std::ostringstream message;
227 message << "Bad bounding box (min >= max) for solid: "
228 << GetName() << " !"
229 << "\npMin = " << pMin
230 << "\npMax = " << pMax;
231 G4Exception("G4Box::BoundingLimits()", "GeomMgt0001", JustWarning, message);
232 DumpInfo();
233 }
234}

236//////////////////////////////////////////////////////////////////////////
237//
238// Calculate extent under transform and specified limit
239
240G4bool G4Box::CalculateExtent(const EAxis pAxis,
241 const G4VoxelLimits& pVoxelLimit,
242 const G4AffineTransform& pTransform,
243 G4double& pMin, G4double& pMax) const
244{
245 G4ThreeVector bmin, bmax;
246
247 // Get bounding box
248 BoundingLimits(bmin,bmax);
249
250 // Find extent
251 G4BoundingEnvelope bbox(bmin,bmax);
252 return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
253}

255//////////////////////////////////////////////////////////////////////////
256//
257// Return whether point inside/outside/on surface, using tolerance
258
259EInside G4Box::Inside(const G4ThreeVector& p) const
260{
261 G4double dist = std::max(std::max(
262 std::abs(p.x())-fDx,
263 std::abs(p.y())-fDy),
264 std::abs(p.z())-fDz);
265 if (dist > delta) return kOutside;
266 return (dist > -delta) ? kSurface : kInside;
267}

269//////////////////////////////////////////////////////////////////////////
270//
271// Detect the side(s) and return corresponding normal
272
273G4ThreeVector G4Box::SurfaceNormal( const G4ThreeVector& p) const
274{
275 G4ThreeVector norm(0,0,0);
276 G4double px = p.x();
277 if (std::abs(std::abs(px) - fDx) <= delta) norm.setX(px < 0 ? -1. : 1.);
278 G4double py = p.y();
279 if (std::abs(std::abs(py) - fDy) <= delta) norm.setY(py < 0 ? -1. : 1.);
280 G4double pz = p.z();
281 if (std::abs(std::abs(pz) - fDz) <= delta) norm.setZ(pz < 0 ? -1. : 1.);
282
283 G4double nside = norm.mag2(); // number of sides = magnitude squared
284 if (nside == 1)
285 return norm;
286 else if (nside > 1)
287 return norm.unit(); // edge or corner
288 else
289 {
290 // Point is not on the surface
291 //
292#ifdef G4CSGDEBUG
293 std::ostringstream message;
294 G4int oldprc = message.precision(16);
295 message << "Point p is not on surface (!?) of solid: "
296 << GetName() << G4endl;
297 message << "Position:\n";
298 message << " p.x() = " << p.x()/mm << " mm\n";
299 message << " p.y() = " << p.y()/mm << " mm\n";
300 message << " p.z() = " << p.z()/mm << " mm";
301 G4cout.precision(oldprc);
302 G4Exception("G4Box::SurfaceNormal(p)", "GeomSolids1002",
303 JustWarning, message );
304 DumpInfo();
305#endif
306 return ApproxSurfaceNormal(p);
307 }
308}

310//////////////////////////////////////////////////////////////////////////
311//
312// Algorithm for SurfaceNormal() following the original specification
313// for points not on the surface
314
315G4ThreeVector G4Box::ApproxSurfaceNormal(const G4ThreeVector& p) const
316{
317 G4double distx = std::abs(p.x()) - fDx;
318 G4double disty = std::abs(p.y()) - fDy;
319 G4double distz = std::abs(p.z()) - fDz;
320
321 if (distx >= disty && distx >= distz)
322 return G4ThreeVector(std::copysign(1.,p.x()), 0., 0.);
323 if (disty >= distx && disty >= distz)
324 return G4ThreeVector(0., std::copysign(1.,p.y()), 0.);
325 else
326 return G4ThreeVector(0., 0., std::copysign(1.,p.z()));
327}

329//////////////////////////////////////////////////////////////////////////
330//
331// Calculate distance to box from an outside point
332// - return kInfinity if no intersection
333//
334
335G4double G4Box::DistanceToIn(const G4ThreeVector& p,
336 const G4ThreeVector& v) const
337{
338 // Check if point is on the surface and traveling away
339 //
340 if ((std::abs(p.x()) - fDx) >= -delta && p.x()*v.x() >= 0) return kInfinity;
341 if ((std::abs(p.y()) - fDy) >= -delta && p.y()*v.y() >= 0) return kInfinity;
342 if ((std::abs(p.z()) - fDz) >= -delta && p.z()*v.z() >= 0) return kInfinity;
343
344 // Find intersection
345 //
346 G4double invx = (v.x() == 0) ? DBL_MAX : -1./v.x();
347 G4double dx = std::copysign(fDx,invx);
348 G4double txmin = (p.x() - dx)*invx;
349 G4double txmax = (p.x() + dx)*invx;
350
351 G4double invy = (v.y() == 0) ? DBL_MAX : -1./v.y();
352 G4double dy = std::copysign(fDy,invy);
353 G4double tymin = std::max(txmin,(p.y() - dy)*invy);
354 G4double tymax = std::min(txmax,(p.y() + dy)*invy);
355
356 G4double invz = (v.z() == 0) ? DBL_MAX : -1./v.z();
357 G4double dz = std::copysign(fDz,invz);
358 G4double tmin = std::max(tymin,(p.z() - dz)*invz);
359 G4double tmax = std::min(tymax,(p.z() + dz)*invz);
360
361 if (tmax <= tmin + delta) return kInfinity; // touch or no hit
362 return (tmin < delta) ? 0. : tmin;
363}

365//////////////////////////////////////////////////////////////////////////
366//
367// Appoximate distance to box.
368// Returns largest perpendicular distance to the closest x/y/z sides of
369// the box, which is the most fast estimation of the shortest distance to box
370// - If inside return 0
371
372G4double G4Box::DistanceToIn(const G4ThreeVector& p) const
373{
374 G4double dist = std::max(std::max(
375 std::abs(p.x())-fDx,
376 std::abs(p.y())-fDy),
377 std::abs(p.z())-fDz);
378 return (dist > 0) ? dist : 0.;
379}

381//////////////////////////////////////////////////////////////////////////
382//
383// Calculate distance to surface of the box from inside and
384// find normal at exit point, if required
385// - when leaving the surface, return 0
386
387G4double G4Box::DistanceToOut( const G4ThreeVector& p,
388 const G4ThreeVector& v,
389 const G4bool calcNorm,
390 G4bool *validNorm, G4ThreeVector *n) const
391{
392 // Check if point is on the surface and traveling away
393 //
394 if ((std::abs(p.x()) - fDx) >= -delta && p.x()*v.x() > 0)
395 {
396 if (calcNorm)
397 {
398 *validNorm = true;
399 n->set((p.x() < 0) ? -1. : 1., 0., 0.);
400 }
401 return 0.;
402 }
403 if ((std::abs(p.y()) - fDy) >= -delta && p.y()*v.y() > 0)
404 {
405 if (calcNorm)
406 {
407 *validNorm = true;
408 n->set(0., (p.y() < 0) ? -1. : 1., 0.);
409 }
410 return 0.;
411 }
412 if ((std::abs(p.z()) - fDz) >= -delta && p.z()*v.z() > 0)
413 {
414 if (calcNorm)
415 {
416 *validNorm = true;
417 n->set(0., 0., (p.z() < 0) ? -1. : 1.);
418 }
419 return 0.;
420 }
421
422 // Find intersection
423 //
424 G4double vx = v.x();
425 G4double tx = (vx == 0) ? DBL_MAX : (std::copysign(fDx,vx) - p.x())/vx;
426
427 G4double vy = v.y();
428 G4double ty = (vy == 0) ? tx : (std::copysign(fDy,vy) - p.y())/vy;
429 G4double txy = std::min(tx,ty);
430
431 G4double vz = v.z();
432 G4double tz = (vz == 0) ? txy : (std::copysign(fDz,vz) - p.z())/vz;
433 G4double tmax = std::min(txy,tz);
434
435 // Set normal, if required, and return distance
436 //
437 if (calcNorm)
438 {
439 *validNorm = true;
440 if (tmax == tx) n->set((v.x() < 0) ? -1. : 1., 0., 0.);
441 else if (tmax == ty) n->set(0., (v.y() < 0) ? -1. : 1., 0.);
442 else n->set(0., 0., (v.z() < 0) ? -1. : 1.);
443 }
444 return tmax;
445}

447////////////////////////////////////////////////////////////////////////////
448//
449// Calculate exact shortest distance to any boundary from inside
450// - if outside return 0
451
452G4double G4Box::DistanceToOut(const G4ThreeVector& p) const
453{
454#ifdef G4CSGDEBUG
455 if( Inside(p) == kOutside )
456 {
457 std::ostringstream message;
458 G4int oldprc = message.precision(16);
459 message << "Point p is outside (!?) of solid: " << GetName() << G4endl;
460 message << "Position:\n";
461 message << " p.x() = " << p.x()/mm << " mm\n";
462 message << " p.y() = " << p.y()/mm << " mm\n";
463 message << " p.z() = " << p.z()/mm << " mm";
464 G4cout.precision(oldprc);
465 G4Exception("G4Box::DistanceToOut(p)", "GeomSolids1002",
466 JustWarning, message );
467 DumpInfo();
468 }
469#endif
470 G4double dist = std::min(std::min(
471 fDx-std::abs(p.x()),
472 fDy-std::abs(p.y())),
473 fDz-std::abs(p.z()));
474 return (dist > 0) ? dist : 0.;
475}

506//////////////////////////////////////////////////////////////////////////
507//
508// GetPointOnSurface
509//
510// Return a point (G4ThreeVector) randomly and uniformly selected
511// on the solid surface
512
513G4ThreeVector G4Box::GetPointOnSurface() const
514{
515 G4double sxy = fDx*fDy, sxz = fDx*fDz, syz = fDy*fDz;
516 G4double select = (sxy + sxz + syz)*G4UniformRand();
517
518 if (select < sxy)
519 return G4ThreeVector((2.*G4UniformRand() - 1.)*fDx,
520 (2.*G4UniformRand() - 1.)*fDy,
521 (select < 0.5*sxy) ? -fDz : fDz);
522
523 if (select < sxy + sxz)
524 return G4ThreeVector((2.*G4UniformRand() - 1.)*fDx,
525 (select < sxy + 0.5*sxz) ? -fDy : fDy,
526 (2.*G4UniformRand() - 1.)*fDz);
527 else
528 return G4ThreeVector((select < sxy + sxz + 0.5*syz) ? -fDx : fDx,
529 (2.*G4UniformRand() - 1.)*fDy,
530 (2.*G4UniformRand() - 1.)*fDz);
531}
