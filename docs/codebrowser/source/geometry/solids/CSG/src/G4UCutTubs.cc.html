123 return sphi();
124}

129G4double G4UCutTubs::GetSinStartPhi() const
130{
131 return std::sin(GetStartPhiAngle());
132}
133G4double G4UCutTubs::GetCosStartPhi() const
134{
135 return std::cos(GetStartPhiAngle());
136}
137G4double G4UCutTubs::GetSinEndPhi() const
138{
139 return std::sin(GetStartPhiAngle()+GetDeltaPhiAngle());
140}

144}

146{

151{

191//////////////////////////////////////////////////////////////////////////
192//
193// Get bounding box
194
195void G4UCutTubs::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
196{
197 static G4bool checkBBox = true;
198
199 G4double rmin = GetInnerRadius();
200 G4double rmax = GetOuterRadius();
201 G4double dz = GetZHalfLength();
202 G4double dphi = GetDeltaPhiAngle();
203
204 G4double sinSphi = GetSinStartPhi();
205 G4double cosSphi = GetCosStartPhi();
206 G4double sinEphi = GetSinEndPhi();
207 G4double cosEphi = GetCosEndPhi();
208
209 G4ThreeVector norm;
210 G4double mag, topx, topy, dists, diste;
211 G4bool iftop;
212
213 // Find Zmin
214 //
215 G4double zmin;
216 norm = GetLowNorm();
217 mag = std::sqrt(norm.x()*norm.x() + norm.y()*norm.y());
218 topx = (mag == 0) ? 0 : -rmax*norm.x()/mag;
219 topy = (mag == 0) ? 0 : -rmax*norm.y()/mag;
220 dists = sinSphi*topx - cosSphi*topy;
221 diste = -sinEphi*topx + cosEphi*topy;
222 if (dphi > pi)
223 {
224 iftop = true;
225 if (dists > 0 && diste > 0)iftop = false;
226 }
227 else
228 {
229 iftop = false;
230 if (dists <= 0 && diste <= 0) iftop = true;
231 }
232 if (iftop)
233 {
234 zmin = -(norm.x()*topx + norm.y()*topy)/norm.z() - dz;
235 }
236 else
237 {
238 G4double z1 = -rmin*(norm.x()*cosSphi + norm.y()*sinSphi)/norm.z() - dz;
239 G4double z2 = -rmin*(norm.x()*cosEphi + norm.y()*sinEphi)/norm.z() - dz;
240 G4double z3 = -rmax*(norm.x()*cosSphi + norm.y()*sinSphi)/norm.z() - dz;
241 G4double z4 = -rmax*(norm.x()*cosEphi + norm.y()*sinEphi)/norm.z() - dz;
242 zmin = std::min(std::min(std::min(z1,z2),z3),z4);
243 }
244
245 // Find Zmax
246 //
247 G4double zmax;
248 norm = GetHighNorm();
249 mag = std::sqrt(norm.x()*norm.x() + norm.y()*norm.y());
250 topx = (mag == 0) ? 0 : -rmax*norm.x()/mag;
251 topy = (mag == 0) ? 0 : -rmax*norm.y()/mag;
252 dists = sinSphi*topx - cosSphi*topy;
253 diste = -sinEphi*topx + cosEphi*topy;
254 if (dphi > pi)
255 {
256 iftop = true;
257 if (dists > 0 && diste > 0) iftop = false;
258 }
259 else
260 {
261 iftop = false;
262 if (dists <= 0 && diste <= 0) iftop = true;
263 }
264 if (iftop)
265 {
266 zmax = -(norm.x()*topx + norm.y()*topy)/norm.z() + dz;
267 }
268 else
269 {
270 G4double z1 = -rmin*(norm.x()*cosSphi + norm.y()*sinSphi)/norm.z() + dz;
271 G4double z2 = -rmin*(norm.x()*cosEphi + norm.y()*sinEphi)/norm.z() + dz;
272 G4double z3 = -rmax*(norm.x()*cosSphi + norm.y()*sinSphi)/norm.z() + dz;
273 G4double z4 = -rmax*(norm.x()*cosEphi + norm.y()*sinEphi)/norm.z() + dz;
274 zmax = std::max(std::max(std::max(z1,z2),z3),z4);
275 }
276
277 // Find bounding box
278 //
279 if (GetDeltaPhiAngle() < twopi)
280 {
281 G4TwoVector vmin,vmax;
282 G4GeomTools::DiskExtent(rmin,rmax,
283 GetSinStartPhi(),GetCosStartPhi(),
284 GetSinEndPhi(),GetCosEndPhi(),
285 vmin,vmax);
286 pMin.set(vmin.x(),vmin.y(), zmin);
287 pMax.set(vmax.x(),vmax.y(), zmax);
288 }
289 else
290 {
291 pMin.set(-rmax,-rmax, zmin);
292 pMax.set( rmax, rmax, zmax);
293 }
294
295 // Check correctness of the bounding box
296 //
297 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
298 {
299 std::ostringstream message;
300 message << "Bad bounding box (min >= max) for solid: "
301 << GetName() << " !"
302 << "\npMin = " << pMin
303 << "\npMax = " << pMax;
304 G4Exception("G4CUutTubs::BoundingLimits()", "GeomMgt0001",
305 JustWarning, message);
306 StreamInfo(G4cout);
307 }
308
309 // Check consistency of bounding boxes
310 //
311 if (checkBBox)
312 {
313 U3Vector vmin, vmax;
314 Extent(vmin,vmax);
315 if (std::abs(pMin.x()-vmin.x()) > kCarTolerance ||
316 std::abs(pMin.y()-vmin.y()) > kCarTolerance ||
317 std::abs(pMin.z()-vmin.z()) > kCarTolerance ||
318 std::abs(pMax.x()-vmax.x()) > kCarTolerance ||
319 std::abs(pMax.y()-vmax.y()) > kCarTolerance ||
320 std::abs(pMax.z()-vmax.z()) > kCarTolerance)
321 {
322 std::ostringstream message;
323 message << "Inconsistency in bounding boxes for solid: "
324 << GetName() << " !"
325 << "\nBBox min: wrapper = " << pMin << " solid = " << vmin
326 << "\nBBox max: wrapper = " << pMax << " solid = " << vmax;
327 G4Exception("G4UCutTubs::BoundingLimits()", "GeomMgt0001",
328 JustWarning, message);
329 checkBBox = false;
330 }
331 }
332}

334//////////////////////////////////////////////////////////////////////////
335//
336// Calculate extent under transform and specified limit
337
338G4bool
339G4UCutTubs::CalculateExtent(const EAxis pAxis,
340 const G4VoxelLimits& pVoxelLimit,
341 const G4AffineTransform& pTransform,
342 G4double& pMin, G4double& pMax) const
343{
344 G4ThreeVector bmin, bmax;
345 G4bool exist;
346
347 // Get bounding box
348 BoundingLimits(bmin,bmax);
349
350 // Check bounding box
351 G4BoundingEnvelope bbox(bmin,bmax);
352#ifdef G4BBOX_EXTENT
353 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
354#endif
355 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
356 {
357 return exist = (pMin < pMax) ? true : false;
358 }
359
360 // Get parameters of the solid
361 G4double rmin = GetInnerRadius();
362 G4double rmax = GetOuterRadius();
363 G4double dphi = GetDeltaPhiAngle();
364 G4double zmin = bmin.z();
365 G4double zmax = bmax.z();
366
367 // Find bounding envelope and calculate extent
368 //
369 const G4int NSTEPS = 24; // number of steps for whole circle
370 G4double astep = twopi/NSTEPS; // max angle for one step
371 G4int ksteps = (dphi <= astep) ? 1 : (G4int)((dphi-deg)/astep) + 1;
372 G4double ang = dphi/ksteps;
373
374 G4double sinHalf = std::sin(0.5*ang);
375 G4double cosHalf = std::cos(0.5*ang);
376 G4double sinStep = 2.*sinHalf*cosHalf;
377 G4double cosStep = 1. - 2.*sinHalf*sinHalf;
378 G4double rext = rmax/cosHalf;
379
380 // bounding envelope for full cylinder consists of two polygons,
381 // in other cases it is a sequence of quadrilaterals
382 if (rmin == 0 && dphi == twopi)
383 {
384 G4double sinCur = sinHalf;
385 G4double cosCur = cosHalf;
386
387 G4ThreeVectorList baseA(NSTEPS),baseB(NSTEPS);
388 for (G4int k=0; k<NSTEPS; ++k)
389 {
390 baseA[k].set(rext*cosCur,rext*sinCur,zmin);
391 baseB[k].set(rext*cosCur,rext*sinCur,zmax);
392
393 G4double sinTmp = sinCur;
394 sinCur = sinCur*cosStep + cosCur*sinStep;
395 cosCur = cosCur*cosStep - sinTmp*sinStep;
396 }
397 std::vector<const G4ThreeVectorList *> polygons(2);
398 polygons[0] = &baseA;
399 polygons[1] = &baseB;
400 G4BoundingEnvelope benv(bmin,bmax,polygons);
401 exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
402 }
403 else
404 {
405 G4double sinStart = GetSinStartPhi();
406 G4double cosStart = GetCosStartPhi();
407 G4double sinEnd = GetSinEndPhi();
408 G4double cosEnd = GetCosEndPhi();
409 G4double sinCur = sinStart*cosHalf + cosStart*sinHalf;
410 G4double cosCur = cosStart*cosHalf - sinStart*sinHalf;
411
412 // set quadrilaterals
413 G4ThreeVectorList pols[NSTEPS+2];
414 for (G4int k=0; k<ksteps+2; ++k) pols[k].resize(4);
415 pols[0][0].set(rmin*cosStart,rmin*sinStart,zmax);
416 pols[0][1].set(rmin*cosStart,rmin*sinStart,zmin);
417 pols[0][2].set(rmax*cosStart,rmax*sinStart,zmin);
418 pols[0][3].set(rmax*cosStart,rmax*sinStart,zmax);
419 for (G4int k=1; k<ksteps+1; ++k)
420 {
421 pols[k][0].set(rmin*cosCur,rmin*sinCur,zmax);
422 pols[k][1].set(rmin*cosCur,rmin*sinCur,zmin);
423 pols[k][2].set(rext*cosCur,rext*sinCur,zmin);
424 pols[k][3].set(rext*cosCur,rext*sinCur,zmax);
425
426 G4double sinTmp = sinCur;
427 sinCur = sinCur*cosStep + cosCur*sinStep;
428 cosCur = cosCur*cosStep - sinTmp*sinStep;
429 }
430 pols[ksteps+1][0].set(rmin*cosEnd,rmin*sinEnd,zmax);
431 pols[ksteps+1][1].set(rmin*cosEnd,rmin*sinEnd,zmin);
432 pols[ksteps+1][2].set(rmax*cosEnd,rmax*sinEnd,zmin);
433 pols[ksteps+1][3].set(rmax*cosEnd,rmax*sinEnd,zmax);
434
435 // set envelope and calculate extent
436 std::vector<const G4ThreeVectorList *> polygons;
437 polygons.resize(ksteps+2);
438 for (G4int k=0; k<ksteps+2; ++k) polygons[k] = &pols[k];
439 G4BoundingEnvelope benv(bmin,bmax,polygons);
440 exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
441 }
442 return exist;
443}

445///////////////////////////////////////////////////////////////////////////
446//
447// Return real Z coordinate of point on Cutted +/- fDZ plane
448
449G4double G4UCutTubs::GetCutZ(const G4ThreeVector& p) const
450{
451 G4double newz = p.z(); // p.z() should be either +fDz or -fDz
452 G4ThreeVector fLowNorm = GetLowNorm();
453 G4ThreeVector fHighNorm = GetHighNorm();
454
455 if (p.z()<0)
456 {
457 if(fLowNorm.z()!=0.)
458 {
459 newz = -GetZHalfLength()
460 - (p.x()*fLowNorm.x()+p.y()*fLowNorm.y())/fLowNorm.z();
461 }
462 }
463 else
464 {
465 if(fHighNorm.z()!=0.)
466 {
467 newz = GetZHalfLength()
468 - (p.x()*fHighNorm.x()+p.y()*fHighNorm.y())/fHighNorm.z();
469 }
470 }
471 return newz;
472}

474//////////////////////////////////////////////////////////////////////////
475//
476// Create polyhedron for visualization
477//
478G4Polyhedron* G4UCutTubs::CreatePolyhedron() const
479{
480 typedef G4double G4double3[3];
481 typedef G4int G4int4[4];
482
483 G4Polyhedron *ph = new G4Polyhedron;
484 G4Polyhedron *ph1 = new G4PolyhedronTubs(GetInnerRadius(),
485 GetOuterRadius(),
486 GetZHalfLength(),
487 GetStartPhiAngle(),
488 GetDeltaPhiAngle());
489 G4int nn=ph1->GetNoVertices();
490 G4int nf=ph1->GetNoFacets();
491 G4double3* xyz = new G4double3[nn]; // number of nodes
492 G4int4* faces = new G4int4[nf] ; // number of faces
493 G4double fDz = GetZHalfLength();
494
495 for(G4int i=0;i<nn;++i)
496 {
497 xyz[i][0]=ph1->GetVertex(i+1).x();
498 xyz[i][1]=ph1->GetVertex(i+1).y();
499 G4double tmpZ=ph1->GetVertex(i+1).z();
500 if(tmpZ>=fDz-kCarTolerance)
501 {
502 xyz[i][2]=GetCutZ(G4ThreeVector(xyz[i][0],xyz[i][1],fDz));
503 }
504 else if(tmpZ<=-fDz+kCarTolerance)
505 {
506 xyz[i][2]=GetCutZ(G4ThreeVector(xyz[i][0],xyz[i][1],-fDz));
507 }
508 else
509 {
510 xyz[i][2]=tmpZ;
511 }
512 }
513 G4int iNodes[4];
514 G4int *iEdge=0;
515 G4int n;
516 for(G4int i=0;i<nf;++i)
517 {
518 ph1->GetFacet(i+1,n,iNodes,iEdge);
519 for(G4int k=0;k<n;++k)
520 {
521 faces[i][k]=iNodes[k];
522 }
523 for(G4int k=n;k<4;++k)
524 {
525 faces[i][k]=0;
526 }
527 }
528 ph->createPolyhedron(nn,nf,xyz,faces);
529
530 delete [] xyz;
531 delete [] faces;
532 delete ph1;
533
534 return ph;
535}
