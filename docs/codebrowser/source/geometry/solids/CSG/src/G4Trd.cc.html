108//////////////////////////////////////////////////////////////////////////
109//
110// Assignment operator
111
112G4Trd& G4Trd::operator = (const G4Trd& rhs)
113{
114 // Check assignment to self
115 //
116 if (this == &rhs) { return *this; }
117
118 // Copy base class data
119 //
120 G4CSGSolid::operator=(rhs);
121
122 // Copy data
123 //
124 halfCarTolerance = rhs.halfCarTolerance;
125 fDx1 = rhs.fDx1; fDx2 = rhs.fDx2;
126 fDy1 = rhs.fDy1; fDy2 = rhs.fDy2;
127 fDz = rhs.fDz;
128 for (G4int i=0; i<4; ++i) { fPlanes[i] = rhs.fPlanes[i]; }
129
130 return *this;
131}

154//////////////////////////////////////////////////////////////////////////
155//
156// Check dimensions
157
158void G4Trd::CheckParameters()
159{
160 G4double dmin = 2*kCarTolerance;
161 if ((fDx1 < 0 || fDx2 < 0 || fDy1 < 0 || fDy2 < 0 || fDz < dmin) ||
162 (fDx1 < dmin && fDx2 < dmin) ||
163 (fDy1 < dmin && fDy2 < dmin))
164 {
165 std::ostringstream message;
166 message << "Invalid (too small or negative) dimensions for Solid: "
167 << GetName()
168 << "\n X - " << fDx1 << ", " << fDx2
169 << "\n Y - " << fDy1 << ", " << fDy2
170 << "\n Z - " << fDz;
171 G4Exception("G4Trd::CheckParameters()", "GeomSolids0002",
172 FatalException, message);
173 }
174}

176//////////////////////////////////////////////////////////////////////////
177//
178// Set side planes
179
180void G4Trd::MakePlanes()
181{
182 G4double dx = fDx1 - fDx2;
183 G4double dy = fDy1 - fDy2;
184 G4double dz = 2*fDz;
185 G4double magx = std::sqrt(dx*dx + dz*dz);
186 G4double magy = std::sqrt(dy*dy + dz*dz);
187
188 // Set -Y & +Y planes
189 //
190 fPlanes[0].a = 0.;
191 fPlanes[0].b = -dz/magy;
192 fPlanes[0].c = dy/magy;
193 fPlanes[0].d = fPlanes[0].b*fDy1 + fPlanes[0].c*fDz;
194
195 fPlanes[1].a = fPlanes[0].a;
196 fPlanes[1].b = -fPlanes[0].b;
197 fPlanes[1].c = fPlanes[0].c;
198 fPlanes[1].d = fPlanes[0].d;
199
200 // Set -X & +X planes
201 //
202 fPlanes[2].a = -dz/magx;
203 fPlanes[2].b = 0.;
204 fPlanes[2].c = dx/magx;
205 fPlanes[2].d = fPlanes[2].a*fDx1 + fPlanes[2].c*fDz;
206
207 fPlanes[3].a = -fPlanes[2].a;
208 fPlanes[3].b = fPlanes[2].b;
209 fPlanes[3].c = fPlanes[2].c;
210 fPlanes[3].d = fPlanes[2].d;
211}

213//////////////////////////////////////////////////////////////////////////
214//
215// Get volume
216
217G4double G4Trd::GetCubicVolume()
218{
219 if (fCubicVolume == 0)
220 {
221 fCubicVolume = 2*fDz*( (fDx1+fDx2)*(fDy1+fDy2) +
222 (fDx2-fDx1)*(fDy2-fDy1)/3 );
223 }
224 return fCubicVolume;
225}
226
227//////////////////////////////////////////////////////////////////////////
228//
229// Get surface area
230
231G4double G4Trd::GetSurfaceArea()
232{
233 if (fSurfaceArea == 0)
234 {
235 fSurfaceArea =
236 4*(fDx1*fDy1+fDx2*fDy2) +
237 2*(fDy1+fDy2)*std::hypot(fDx1-fDx2,2*fDz) +
238 2*(fDx1+fDx2)*std::hypot(fDy1-fDy2,2*fDz);
239 }
240 return fSurfaceArea;
241}

255//////////////////////////////////////////////////////////////////////////
256//
257// Get bounding box
258
259void G4Trd::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
260{
261 G4double dx1 = GetXHalfLength1();
262 G4double dx2 = GetXHalfLength2();
263 G4double dy1 = GetYHalfLength1();
264 G4double dy2 = GetYHalfLength2();
265 G4double dz = GetZHalfLength();
266
267 G4double xmax = std::max(dx1,dx2);
268 G4double ymax = std::max(dy1,dy2);
269 pMin.set(-xmax,-ymax,-dz);
270 pMax.set( xmax, ymax, dz);
271
272 // Check correctness of the bounding box
273 //
274 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
275 {
276 std::ostringstream message;
277 message << "Bad bounding box (min >= max) for solid: "
278 << GetName() << " !"
279 << "\npMin = " << pMin
280 << "\npMax = " << pMax;
281 G4Exception("G4Trd::BoundingLimits()", "GeomMgt0001", JustWarning, message);
282 DumpInfo();
283 }
284}

286//////////////////////////////////////////////////////////////////////////
287//
288// Calculate extent under transform and specified limit
289
290G4bool G4Trd::CalculateExtent( const EAxis pAxis,
291 const G4VoxelLimits& pVoxelLimit,
292 const G4AffineTransform& pTransform,
293 G4double& pMin, G4double& pMax ) const
294{
295 G4ThreeVector bmin, bmax;
296 G4bool exist;
297
298 // Check bounding box (bbox)
299 //
300 BoundingLimits(bmin,bmax);
301 G4BoundingEnvelope bbox(bmin,bmax);
302#ifdef G4BBOX_EXTENT
303 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
304#endif
305 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
306 {
307 return exist = (pMin < pMax) ? true : false;
308 }
309
310 // Set bounding envelope (benv) and calculate extent
311 //
312 G4double dx1 = GetXHalfLength1();
313 G4double dx2 = GetXHalfLength2();
314 G4double dy1 = GetYHalfLength1();
315 G4double dy2 = GetYHalfLength2();
316 G4double dz = GetZHalfLength();
317
318 G4ThreeVectorList baseA(4), baseB(4);
319 baseA[0].set(-dx1,-dy1,-dz);
320 baseA[1].set( dx1,-dy1,-dz);
321 baseA[2].set( dx1, dy1,-dz);
322 baseA[3].set(-dx1, dy1,-dz);
323 baseB[0].set(-dx2,-dy2, dz);
324 baseB[1].set( dx2,-dy2, dz);
325 baseB[2].set( dx2, dy2, dz);
326 baseB[3].set(-dx2, dy2, dz);
327
328 std::vector<const G4ThreeVectorList *> polygons(2);
329 polygons[0] = &baseA;
330 polygons[1] = &baseB;
331
332 G4BoundingEnvelope benv(bmin,bmax,polygons);
333 exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
334 return exist;
335}

337//////////////////////////////////////////////////////////////////////////
338//
339// Return whether point inside/outside/on surface, using tolerance
340
341EInside G4Trd::Inside( const G4ThreeVector& p ) const
342{
343 G4double dx = fPlanes[3].a*std::abs(p.x())+fPlanes[3].c*p.z()+fPlanes[3].d;
344 G4double dy = fPlanes[1].b*std::abs(p.y())+fPlanes[1].c*p.z()+fPlanes[1].d;
345 G4double dxy = std::max(dx,dy);
346
347 G4double dz = std::abs(p.z())-fDz;
348 G4double dist = std::max(dz,dxy);
349
350 if (dist > halfCarTolerance) return kOutside;
351 return (dist > -halfCarTolerance) ? kSurface : kInside;
352}

354//////////////////////////////////////////////////////////////////////////
355//
356// Determine side where point is, and return corresponding normal
357
358G4ThreeVector G4Trd::SurfaceNormal( const G4ThreeVector& p ) const
359{
360 G4int nsurf = 0; // number of surfaces where p is placed
361
362 // Check Z faces
363 //
364 G4double nz = 0;
365 G4double dz = std::abs(p.z()) - fDz;
366 if (std::abs(dz) <= halfCarTolerance)
367 {
368 nz = (p.z() < 0) ? -1 : 1;
369 ++nsurf;
370 }
371
372 // Check Y faces
373 //
374 G4double ny = 0;
375 G4double dy1 = fPlanes[0].b*p.y();
376 G4double dy2 = fPlanes[0].c*p.z() + fPlanes[0].d;
377 if (std::abs(dy2 + dy1) <= halfCarTolerance)
378 {
379 ny += fPlanes[0].b;
380 nz += fPlanes[0].c;
381 ++nsurf;
382 }
383 if (std::abs(dy2 - dy1) <= halfCarTolerance)
384 {
385 ny += fPlanes[1].b;
386 nz += fPlanes[1].c;
387 ++nsurf;
388 }
389
390 // Check X faces
391 //
392 G4double nx = 0;
393 G4double dx1 = fPlanes[2].a*p.x();
394 G4double dx2 = fPlanes[2].c*p.z() + fPlanes[2].d;
395 if (std::abs(dx2 + dx1) <= halfCarTolerance)
396 {
397 nx += fPlanes[2].a;
398 nz += fPlanes[2].c;
399 ++nsurf;
400 }
401 if (std::abs(dx2 - dx1) <= halfCarTolerance)
402 {
403 nx += fPlanes[3].a;
404 nz += fPlanes[3].c;
405 ++nsurf;
406 }
407
408 // Return normal
409 //
410 if (nsurf == 1) return G4ThreeVector(nx,ny,nz);
411 else if (nsurf != 0) return G4ThreeVector(nx,ny,nz).unit(); // edge or corner
412 else
413 {
414 // Point is not on the surface
415 //
416#ifdef G4CSGDEBUG
417 std::ostringstream message;
418 G4int oldprc = message.precision(16);
419 message << "Point p is not on surface (!?) of solid: "
420 << GetName() << G4endl;
421 message << "Position:\n";
422 message << " p.x() = " << p.x()/mm << " mm\n";
423 message << " p.y() = " << p.y()/mm << " mm\n";
424 message << " p.z() = " << p.z()/mm << " mm";
425 G4cout.precision(oldprc) ;
426 G4Exception("G4Trd::SurfaceNormal(p)", "GeomSolids1002",
427 JustWarning, message );
428 DumpInfo();
429#endif
430 return ApproxSurfaceNormal(p);
431 }
432}

434//////////////////////////////////////////////////////////////////////////
435//
436// Algorithm for SurfaceNormal() following the original specification
437// for points not on the surface
438
439G4ThreeVector G4Trd::ApproxSurfaceNormal( const G4ThreeVector& p ) const
440{
441 G4double dist = -DBL_MAX;
442 G4int iside = 0;
443 for (G4int i=0; i<4; ++i)
444 {
445 G4double d = fPlanes[i].a*p.x() +
446 fPlanes[i].b*p.y() +
447 fPlanes[i].c*p.z() + fPlanes[i].d;
448 if (d > dist) { dist = d; iside = i; }
449 }
450
451 G4double distz = std::abs(p.z()) - fDz;
452 if (dist > distz)
453 return G4ThreeVector(fPlanes[iside].a, fPlanes[iside].b, fPlanes[iside].c);
454 else
455 return G4ThreeVector(0, 0, (p.z() < 0) ? -1 : 1);
456}

458//////////////////////////////////////////////////////////////////////////
459//
460// Calculate distance to shape from outside
461// - return kInfinity if no intersection
462
463G4double G4Trd::DistanceToIn(const G4ThreeVector& p,
464 const G4ThreeVector& v ) const
465{
466 // Z intersections
467 //
468 if ((std::abs(p.z()) - fDz) >= -halfCarTolerance && p.z()*v.z() >= 0)
469 return kInfinity;
470 G4double invz = (-v.z() == 0) ? DBL_MAX : -1./v.z();
471 G4double dz = (invz < 0) ? fDz : -fDz;
472 G4double tzmin = (p.z() + dz)*invz;
473 G4double tzmax = (p.z() - dz)*invz;
474
475 // Y intersections
476 //
477 G4double tmin0 = tzmin, tmax0 = tzmax;
478 G4double ya = fPlanes[0].b*v.y(), yb = fPlanes[0].c*v.z();
479 G4double yc = fPlanes[0].b*p.y(), yd = fPlanes[0].c*p.z()+fPlanes[0].d;
480 G4double cos0 = yb + ya;
481 G4double dis0 = yd + yc;
482 if (dis0 >= -halfCarTolerance)
483 {
484 if (cos0 >= 0) return kInfinity;
485 G4double tmp = -dis0/cos0;
486 if (tmin0 < tmp) tmin0 = tmp;
487 }
488 else if (cos0 > 0)
489 {
490 G4double tmp = -dis0/cos0;
491 if (tmax0 > tmp) tmax0 = tmp;
492 }
493
494 G4double tmin1 = tmin0, tmax1 = tmax0;
495 G4double cos1 = yb - ya;
496 G4double dis1 = yd - yc;
497 if (dis1 >= -halfCarTolerance)
498 {
499 if (cos1 >= 0) return kInfinity;
500 G4double tmp = -dis1/cos1;
501 if (tmin1 < tmp) tmin1 = tmp;
502 }
503 else if (cos1 > 0)
504 {
505 G4double tmp = -dis1/cos1;
506 if (tmax1 > tmp) tmax1 = tmp;
507 }
508
509 // X intersections
510 //
511 G4double tmin2 = tmin1, tmax2 = tmax1;
512 G4double xa = fPlanes[2].a*v.x(), xb = fPlanes[2].c*v.z();
513 G4double xc = fPlanes[2].a*p.x(), xd = fPlanes[2].c*p.z()+fPlanes[2].d;
514 G4double cos2 = xb + xa;
515 G4double dis2 = xd + xc;
516 if (dis2 >= -halfCarTolerance)
517 {
518 if (cos2 >= 0) return kInfinity;
519 G4double tmp = -dis2/cos2;
520 if (tmin2 < tmp) tmin2 = tmp;
521 }
522 else if (cos2 > 0)
523 {
524 G4double tmp = -dis2/cos2;
525 if (tmax2 > tmp) tmax2 = tmp;
526 }
527
528 G4double tmin3 = tmin2, tmax3 = tmax2;
529 G4double cos3 = xb - xa;
530 G4double dis3 = xd - xc;
531 if (dis3 >= -halfCarTolerance)
532 {
533 if (cos3 >= 0) return kInfinity;
534 G4double tmp = -dis3/cos3;
535 if (tmin3 < tmp) tmin3 = tmp;
536 }
537 else if (cos3 > 0)
538 {
539 G4double tmp = -dis3/cos3;
540 if (tmax3 > tmp) tmax3 = tmp;
541 }
542
543 // Find distance
544 //
545 G4double tmin = tmin3, tmax = tmax3;
546 if (tmax <= tmin + halfCarTolerance) return kInfinity; // touch or no hit
547 return (tmin < halfCarTolerance ) ? 0. : tmin;
548}

550//////////////////////////////////////////////////////////////////////////
551//
552// Calculate exact shortest distance to any boundary from outside
553// This is the best fast estimation of the shortest distance to trap
554// - returns 0 if point is inside
555
556G4double G4Trd::DistanceToIn( const G4ThreeVector& p ) const
557{
558 G4double dx = fPlanes[3].a*std::abs(p.x())+fPlanes[3].c*p.z()+fPlanes[3].d;
559 G4double dy = fPlanes[1].b*std::abs(p.y())+fPlanes[1].c*p.z()+fPlanes[1].d;
560 G4double dxy = std::max(dx,dy);
561
562 G4double dz = std::abs(p.z())-fDz;
563 G4double dist = std::max(dz,dxy);
564
565 return (dist > 0) ? dist : 0.;
566}

568//////////////////////////////////////////////////////////////////////////
569//
570// Calculate distance to surface of shape from inside and
571// find normal at exit point, if required
572// - when leaving the surface, return 0
573
574G4double G4Trd::DistanceToOut(const G4ThreeVector& p, const G4ThreeVector& v,
575 const G4bool calcNorm,
576 G4bool *validNorm, G4ThreeVector *n) const
577{
578 // Z intersections
579 //
580 if ((std::abs(p.z()) - fDz) >= -halfCarTolerance && p.z()*v.z() > 0)
581 {
582 if (calcNorm)
583 {
584 *validNorm = true;
585 n->set(0, 0, (p.z() < 0) ? -1 : 1);
586 }
587 return 0;
588 }
589 G4double vz = v.z();
590 G4double tmax = (vz == 0) ? DBL_MAX : (std::copysign(fDz,vz) - p.z())/vz;
591 G4int iside = (vz < 0) ? -4 : -2; // little trick: (-4+3)=-1, (-2+3)=+1
592
593 // Y intersections
594 //
595 G4int i = 0;
596 for ( ; i<2; ++i)
597 {
598 G4double cosa = fPlanes[i].b*v.y() + fPlanes[i].c*v.z();
599 if (cosa > 0)
600 {
601 G4double dist = fPlanes[i].b*p.y()+fPlanes[i].c*p.z()+fPlanes[i].d;
602 if (dist >= -halfCarTolerance)
603 {
604 if (calcNorm)
605 {
606 *validNorm = true;
607 n->set(0, fPlanes[i].b, fPlanes[i].c);
608 }
609 return 0;
610 }
611 G4double tmp = -dist/cosa;
612 if (tmax > tmp) { tmax = tmp; iside = i; }
613 }
614 }
615
616 // X intersections
617 //
618 for ( ; i<4; ++i)
619 {
620 G4double cosa = fPlanes[i].a*v.x()+fPlanes[i].c*v.z();
621 if (cosa > 0)
622 {
623 G4double dist = fPlanes[i].a*p.x()+fPlanes[i].c*p.z()+fPlanes[i].d;
624 if (dist >= -halfCarTolerance)
625 {
626 if (calcNorm)
627 {
628 *validNorm = true;
629 n->set(fPlanes[i].a, fPlanes[i].b, fPlanes[i].c);
630 }
631 return 0;
632 }
633 G4double tmp = -dist/cosa;
634 if (tmax > tmp) { tmax = tmp; iside = i; }
635 }
636 }
637
638 // Set normal, if required, and return distance
639 //
640 if (calcNorm)
641 {
642 *validNorm = true;
643 if (iside < 0)
644 n->set(0, 0, iside + 3); // (-4+3)=-1, (-2+3)=+1
645 else
646 n->set(fPlanes[iside].a, fPlanes[iside].b, fPlanes[iside].c);
647 }
648 return tmax;
649}

673#endif
674 G4double dx = fPlanes[3].a*std::abs(p.x())+fPlanes[3].c*p.z()+fPlanes[3].d;
675 G4double dy = fPlanes[1].b*std::abs(p.y())+fPlanes[1].c*p.z()+fPlanes[1].d;
676 G4double dxy = std::max(dx,dy);
677
678 G4double dz = std::abs(p.z())-fDz;
679 G4double dist = std::max(dz,dxy);
680
681 return (dist < 0) ? -dist : 0.;
682}

725//////////////////////////////////////////////////////////////////////////
726//
727// Return a point randomly and uniformly selected on the solid surface
728
729G4ThreeVector G4Trd::GetPointOnSurface() const
730{
731 // Set vertices
732 //
733 G4ThreeVector pt[8];
734 pt[0].set(-fDx1,-fDy1,-fDz);
735 pt[1].set( fDx1,-fDy1,-fDz);
736 pt[2].set(-fDx1, fDy1,-fDz);
737 pt[3].set( fDx1, fDy1,-fDz);
738 pt[4].set(-fDx2,-fDy2, fDz);
739 pt[5].set( fDx2,-fDy2, fDz);
740 pt[6].set(-fDx2, fDy2, fDz);
741 pt[7].set( fDx2, fDy2, fDz);
742
743 // Set faces (-Z, -Y, +Y, -X, +X, +Z)
744 //
745 G4int iface [6][4] =
746 { {0,1,3,2}, {0,4,5,1}, {2,3,7,6}, {0,2,6,4}, {1,5,7,3}, {4,6,7,5} };
747
748 // Set areas
749 //
750 G4double sxz = (fDy1 + fDy2)*std::hypot(fDx1 - fDx2, 2*fDz);
751 G4double syz = (fDx1 + fDx2)*std::hypot(fDy1 - fDy2, 2*fDz);
752 G4double sface[6] = { 4*fDx1*fDy1, syz, syz, sxz, sxz, 4*fDx2*fDy2 };
753 for (G4int i=1; i<6; ++i) { sface[i] += sface[i-1]; }
754
755 // Select face
756 //
757 G4double select = sface[5]*G4UniformRand();
758 G4int k = 5;
759 if (select <= sface[4]) k = 4;
760 if (select <= sface[3]) k = 3;
761 if (select <= sface[2]) k = 2;
762 if (select <= sface[1]) k = 1;
763 if (select <= sface[0]) k = 0;
764
765 // Select sub-triangle
766 //
767 G4int i0 = iface[k][0];
768 G4int i1 = iface[k][1];
769 G4int i2 = iface[k][2];
770 G4int i3 = iface[k][3];
771 G4double s1 = G4GeomTools::TriangleAreaNormal(pt[i0],pt[i1],pt[i3]).mag();
772 G4double s2 = G4GeomTools::TriangleAreaNormal(pt[i2],pt[i1],pt[i3]).mag();
773 if ((s1+s2)*G4UniformRand() > s1) i0 = i2;
774
775 // Generate point
776 //
777 G4double u = G4UniformRand();
778 G4double v = G4UniformRand();
779 if (u + v > 1.) { u = 1. - u; v = 1. - v; }
780 return (1.-u-v)*pt[i0] + u*pt[i1] + v*pt[i3];
781}
