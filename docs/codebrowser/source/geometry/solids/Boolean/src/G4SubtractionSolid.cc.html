136//////////////////////////////////////////////////////////////////////////
137//
138// Get bounding box
139
140void
141G4SubtractionSolid::BoundingLimits(G4ThreeVector& pMin,
142 G4ThreeVector& pMax) const
143{
144 // Since it is unclear how the shape of the first solid will be changed
145 // after subtraction, just return its original bounding box.
146 //
147 fPtrSolidA->BoundingLimits(pMin,pMax);
148
149 // Check correctness of the bounding box
150 //
151 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
152 {
153 std::ostringstream message;
154 message << "Bad bounding box (min >= max) for solid: "
155 << GetName() << " !"
156 << "\npMin = " << pMin
157 << "\npMax = " << pMax;
158 G4Exception("G4SubtractionSolid::BoundingLimits()", "GeomMgt0001",
159 JustWarning, message);
160 DumpInfo();
161 }
162}

164//////////////////////////////////////////////////////////////////////////
165//
166// Calculate extent under transform and specified limit
167
168G4bool
169G4SubtractionSolid::CalculateExtent( const EAxis pAxis,
170 const G4VoxelLimits& pVoxelLimit,
171 const G4AffineTransform& pTransform,
172 G4double& pMin,
173 G4double& pMax ) const
174{
175 // Since we cannot be sure how much the second solid subtracts
176 // from the first, we must use the first solid's extent!
177

179 pTransform, pMin, pMax );

182/////////////////////////////////////////////////////
183//
184// Touching ? Empty subtraction ?
185
186EInside G4SubtractionSolid::Inside( const G4ThreeVector& p ) const
187{
188 EInside positionA = fPtrSolidA->Inside(p);
189 if (positionA == kOutside) return positionA; // outside A
190
191 EInside positionB = fPtrSolidB->Inside(p);
192 if (positionB == kOutside) return positionA;
193
194 if (positionB == kInside) return kOutside;
195 if (positionA == kInside) return kSurface; // surface B
196
197 // Point is on both surfaces
198 //
199 static const G4double rtol = 1000*kCarTolerance;
200
201 return ((fPtrSolidA->SurfaceNormal(p) -
202 fPtrSolidB->SurfaceNormal(p)).mag2() > rtol) ? kSurface : kOutside;
203}

205//////////////////////////////////////////////////////////////
206//
207// SurfaceNormal
208
209G4ThreeVector
210G4SubtractionSolid::SurfaceNormal( const G4ThreeVector& p ) const
211{
212 G4ThreeVector normal;
213
214 EInside InsideA = fPtrSolidA->Inside(p);
215 EInside InsideB = fPtrSolidB->Inside(p);
216
217 if( InsideA == kOutside )
218 {
219#ifdef G4BOOLDEBUG
220 G4cout << "WARNING - Invalid call [1] in "
221 << "G4SubtractionSolid::SurfaceNormal(p)" << G4endl
222 << " Point p is outside !" << G4endl;
223 G4cout << " p = " << p << G4endl;
224 G4cerr << "WARNING - Invalid call [1] in "
225 << "G4SubtractionSolid::SurfaceNormal(p)" << G4endl
226 << " Point p is outside !" << G4endl;
227 G4cerr << " p = " << p << G4endl;
228#endif
229 normal = fPtrSolidA->SurfaceNormal(p) ;
230 }
231 else if( InsideA == kSurface &&
232 InsideB != kInside )
233 {
234 normal = fPtrSolidA->SurfaceNormal(p) ;
235 }
236 else if( InsideA == kInside &&
237 InsideB != kOutside )
238 {
239 normal = -fPtrSolidB->SurfaceNormal(p) ;
240 }
241 else
242 {
243 if ( fPtrSolidA->DistanceToOut(p) <= fPtrSolidB->DistanceToIn(p) )
244 {
245 normal = fPtrSolidA->SurfaceNormal(p) ;
246 }
247 else
248 {
249 normal = -fPtrSolidB->SurfaceNormal(p) ;
250 }
251#ifdef G4BOOLDEBUG
252 if(Inside(p) == kInside)
253 {
254 G4cout << "WARNING - Invalid call [2] in "
255 << "G4SubtractionSolid::SurfaceNormal(p)" << G4endl
256 << " Point p is inside !" << G4endl;
257 G4cout << " p = " << p << G4endl;
258 G4cerr << "WARNING - Invalid call [2] in "
259 << "G4SubtractionSolid::SurfaceNormal(p)" << G4endl
260 << " Point p is inside !" << G4endl;
261 G4cerr << " p = " << p << G4endl;
262 }
263#endif
264 }
265 return normal;
266}

268/////////////////////////////////////////////////////////////
269//
270// The same algorithm as in DistanceToIn(p)
271
272G4double
273G4SubtractionSolid::DistanceToIn( const G4ThreeVector& p,
274 const G4ThreeVector& v ) const
275{
276 G4double dist = 0.0, dist2 = 0.0, disTmp = 0.0;
277
278#ifdef G4BOOLDEBUG
279 if( Inside(p) == kInside )
280 {
281 G4cout << "WARNING - Invalid call in "
282 << "G4SubtractionSolid::DistanceToIn(p,v)" << G4endl
283 << " Point p is inside !" << G4endl;
284 G4cout << " p = " << p << G4endl;
285 G4cout << " v = " << v << G4endl;
286 G4cerr << "WARNING - Invalid call in "
287 << "G4SubtractionSolid::DistanceToIn(p,v)" << G4endl
288 << " Point p is inside !" << G4endl;
289 G4cerr << " p = " << p << G4endl;
290 G4cerr << " v = " << v << G4endl;
291 }
292#endif
293
294 // if( // ( fPtrSolidA->Inside(p) != kOutside) && // case1:p in both A&B
295 if ( fPtrSolidB->Inside(p) != kOutside ) // start: out of B
296 {
297 dist = fPtrSolidB->DistanceToOut(p,v) ; // ,calcNorm,validNorm,n) ;
298
299 if( fPtrSolidA->Inside(p+dist*v) != kInside )
300 {
301 G4int count1=0;
302 do // Loop checking, 13.08.2015, G.Cosmo
303 {
304 disTmp = fPtrSolidA->DistanceToIn(p+dist*v,v) ;
305
306 if(disTmp == kInfinity)
307 {
308 return kInfinity ;
309 }
310 dist += disTmp ;
311
312 if( Inside(p+dist*v) == kOutside )
313 {
314 disTmp = fPtrSolidB->DistanceToOut(p+dist*v,v) ;
315 dist2 = dist+disTmp;
316 if (dist == dist2) { return dist; } // no progress
317 dist = dist2 ;
318 count1++;
319 if( count1 > 1000 ) // Infinite loop detected
320 {
321 G4String nameB = fPtrSolidB->GetName();
322 if(fPtrSolidB->GetEntityType()=="G4DisplacedSolid")
323 {
324 nameB = (dynamic_cast<G4DisplacedSolid*>(fPtrSolidB))
325 ->GetConstituentMovedSolid()->GetName();
326 }
327 std::ostringstream message;
328 message << "Illegal condition caused by solids: "
329 << fPtrSolidA->GetName() << " and " << nameB << G4endl;
330 message.precision(16);
331 message << "Looping detected in point " << p+dist*v
332 << ", from original point " << p
333 << " and direction " << v << G4endl
334 << "Computed candidate distance: " << dist << "*mm. ";
335 message.precision(6);
336 DumpInfo();
337 G4Exception("G4SubtractionSolid::DistanceToIn(p,v)",
338 "GeomSolids1001", JustWarning, message,
339 "Returning candidate distance.");
340 return dist;
341 }
342 }
343 }
344 while( Inside(p+dist*v) == kOutside ) ;
345 }
346 }
347 else // p outside A, start in A
348 {
349 dist = fPtrSolidA->DistanceToIn(p,v) ;
350
351 if( dist == kInfinity ) // past A, hence past A\B
352 {
353 return kInfinity ;
354 }
355 else
356 {
357 G4int count2=0;
358 while( Inside(p+dist*v) == kOutside ) // pushing loop
359 {
360 disTmp = fPtrSolidB->DistanceToOut(p+dist*v,v) ;
361 dist += disTmp ;
362
363 if( Inside(p+dist*v) == kOutside )
364 {
365 disTmp = fPtrSolidA->DistanceToIn(p+dist*v,v) ;
366
367 if(disTmp == kInfinity) // past A, hence past A\B
368 {
369 return kInfinity ;
370 }
371 dist2 = dist+disTmp;
372 if (dist == dist2) { return dist; } // no progress
373 dist = dist2 ;
374 count2++;
375 if( count2 > 1000 ) // Infinite loop detected
376 {
377 G4String nameB = fPtrSolidB->GetName();
378 if(fPtrSolidB->GetEntityType()=="G4DisplacedSolid")
379 {
380 nameB = (dynamic_cast<G4DisplacedSolid*>(fPtrSolidB))
381 ->GetConstituentMovedSolid()->GetName();
382 }
383 std::ostringstream message;
384 message << "Illegal condition caused by solids: "
385 << fPtrSolidA->GetName() << " and " << nameB << G4endl;
386 message.precision(16);
387 message << "Looping detected in point " << p+dist*v
388 << ", from original point " << p
389 << " and direction " << v << G4endl
390 << "Computed candidate distance: " << dist << "*mm. ";
391 message.precision(6);
392 DumpInfo();
393 G4Exception("G4SubtractionSolid::DistanceToIn(p,v)",
394 "GeomSolids1001", JustWarning, message,
395 "Returning candidate distance.");
396 return dist;
397 }
398 }
399 } // Loop checking, 13.08.2015, G.Cosmo
400 }
401 }
402
403 return dist ;
404}

406////////////////////////////////////////////////////////
407//
408// Approximate nearest distance from the point p to the intersection of
409// two solids. It is usually underestimated from the point of view of
410// isotropic safety
411
412G4double
413G4SubtractionSolid::DistanceToIn( const G4ThreeVector& p ) const
414{
415 G4double dist=0.0;
416
417#ifdef G4BOOLDEBUG
418 if( Inside(p) == kInside )
419 {
420 G4cout << "WARNING - Invalid call in "
421 << "G4SubtractionSolid::DistanceToIn(p)" << G4endl
422 << " Point p is inside !" << G4endl;
423 G4cout << " p = " << p << G4endl;
424 G4cerr << "WARNING - Invalid call in "
425 << "G4SubtractionSolid::DistanceToIn(p)" << G4endl
426 << " Point p is inside !" << G4endl;
427 G4cerr << " p = " << p << G4endl;
428 }
429#endif
430
431 if( ( fPtrSolidA->Inside(p) != kOutside) && // case 1
432 ( fPtrSolidB->Inside(p) != kOutside) )
433 {
434 dist= fPtrSolidB->DistanceToOut(p) ;
435 }
436 else
437 {
438 dist= fPtrSolidA->DistanceToIn(p) ;
439 }
440
441 return dist;
442}

444//////////////////////////////////////////////////////////
445//
446// The same algorithm as DistanceToOut(p)
447
448G4double
449G4SubtractionSolid::DistanceToOut( const G4ThreeVector& p,
450 const G4ThreeVector& v,
451 const G4bool calcNorm,
452 G4bool *validNorm,
453 G4ThreeVector *n ) const
454{
455#ifdef G4BOOLDEBUG
456 if( Inside(p) == kOutside )
457 {
458 G4cout << "Position:" << G4endl << G4endl;
459 G4cout << "p.x() = " << p.x()/mm << " mm" << G4endl;
460 G4cout << "p.y() = " << p.y()/mm << " mm" << G4endl;
461 G4cout << "p.z() = " << p.z()/mm << " mm" << G4endl << G4endl;
462 G4cout << "Direction:" << G4endl << G4endl;
463 G4cout << "v.x() = " << v.x() << G4endl;
464 G4cout << "v.y() = " << v.y() << G4endl;
465 G4cout << "v.z() = " << v.z() << G4endl << G4endl;
466 G4cout << "WARNING - Invalid call in "
467 << "G4SubtractionSolid::DistanceToOut(p,v)" << G4endl
468 << " Point p is outside !" << G4endl;
469 G4cout << " p = " << p << G4endl;
470 G4cout << " v = " << v << G4endl;
471 G4cerr << "WARNING - Invalid call in "
472 << "G4SubtractionSolid::DistanceToOut(p,v)" << G4endl
473 << " Point p is outside !" << G4endl;
474 G4cerr << " p = " << p << G4endl;
475 G4cerr << " v = " << v << G4endl;
476 }
477#endif
478
479 G4double distout;
480 G4double distA = fPtrSolidA->DistanceToOut(p,v,calcNorm,validNorm,n) ;
481 G4double distB = fPtrSolidB->DistanceToIn(p,v) ;
482 if(distB < distA)
483 {
484 if(calcNorm)
485 {
486 *n = -(fPtrSolidB->SurfaceNormal(p+distB*v)) ;
487 *validNorm = false ;
488 }
489 distout= distB ;
490 }
491 else
492 {
493 distout= distA ;
494 }
495 return distout;
496}

498//////////////////////////////////////////////////////////////
499//
500// Inverted algorithm of DistanceToIn(p)
501
502G4double
503G4SubtractionSolid::DistanceToOut( const G4ThreeVector& p ) const
504{
505 G4double dist=0.0;
506
507 if( Inside(p) == kOutside )
508 {
509#ifdef G4BOOLDEBUG
510 G4cout << "WARNING - Invalid call in "
511 << "G4SubtractionSolid::DistanceToOut(p)" << G4endl
512 << " Point p is outside" << G4endl;
513 G4cout << " p = " << p << G4endl;
514 G4cerr << "WARNING - Invalid call in "
515 << "G4SubtractionSolid::DistanceToOut(p)" << G4endl
516 << " Point p is outside" << G4endl;
517 G4cerr << " p = " << p << G4endl;
518#endif
519 }
520 else
521 {
522 dist= std::min(fPtrSolidA->DistanceToOut(p),
523 fPtrSolidB->DistanceToIn(p) ) ;
524 }
525 return dist;
526}

571G4Polyhedron*
572G4SubtractionSolid::CreatePolyhedron () const
573{
574 HepPolyhedronProcessor processor;
575 // Stack components and components of components recursively
576 // See G4BooleanSolid::StackPolyhedron
577 G4Polyhedron* top = StackPolyhedron(processor, this);
578 G4Polyhedron* result = new G4Polyhedron(*top);
579 if (processor.execute(*result)) { return result; }
580 else { return 0; }
581}
