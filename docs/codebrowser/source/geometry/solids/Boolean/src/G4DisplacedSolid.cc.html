50//
51// Constructor for transformation like rotation of frame then translation
52// in new frame. It is similar to 1st constractor in G4PVPlacement
53
54G4DisplacedSolid::G4DisplacedSolid( const G4String& pName,
55 G4VSolid* pSolid ,
56 G4RotationMatrix* rotMatrix,
57 const G4ThreeVector& transVector )
58 : G4VSolid(pName), fRebuildPolyhedron(false), fpPolyhedron(0)
59{
60 if (pSolid->GetEntityType() == "G4DisplacedSolid")
61 {
62 fPtrSolid = ((G4DisplacedSolid*)pSolid)->GetConstituentMovedSolid();
63 G4AffineTransform t1 = ((G4DisplacedSolid*)pSolid)->GetDirectTransform();
64 G4AffineTransform t2 = G4AffineTransform(rotMatrix,transVector);
65 fDirectTransform = new G4AffineTransform(t1*t2);
66 }
67 else
68 {
69 fPtrSolid = pSolid;
70 fDirectTransform = new G4AffineTransform(rotMatrix,transVector);
71 }
72 fPtrTransform = new G4AffineTransform(fDirectTransform->Inverse());
73}

97 }
98 fPtrTransform = new G4AffineTransform(fDirectTransform->Inverse());
99}

159///////////////////////////////////////////////////////////////
160//
161// Assignment operator
162
163G4DisplacedSolid& G4DisplacedSolid::operator = (const G4DisplacedSolid& rhs)
164{
165 // Check assignment to self
166 //
167 if (this == &rhs) { return *this; }
168
169 // Copy base class data
170 //
171 G4VSolid::operator=(rhs);
172
173 // Copy data
174 //
175 fPtrSolid = rhs.fPtrSolid;
176 delete fPtrTransform; delete fDirectTransform;
177 fPtrTransform = new G4AffineTransform(*(rhs.fPtrTransform));
178 fDirectTransform = new G4AffineTransform(*(rhs.fDirectTransform));
179 fRebuildPolyhedron = false;
180 delete fpPolyhedron; fpPolyhedron= 0;
181
182 return *this;
183}
184

189 delete fPtrTransform; fPtrTransform=0;
190 delete fDirectTransform; fDirectTransform=0;
191 }

291//////////////////////////////////////////////////////////////////////////
292//
293// Get bounding box
294
295void G4DisplacedSolid::BoundingLimits(G4ThreeVector& pMin,
296 G4ThreeVector& pMax) const
297{
298 if (!fDirectTransform->IsRotated())
299 {
300 // Special case of pure translation
301 //
302 fPtrSolid->BoundingLimits(pMin,pMax);
303 G4ThreeVector offset = fDirectTransform->NetTranslation();
304 pMin += offset;
305 pMax += offset;
306 }
307 else
308 {
309 // General case, use CalculateExtent() to find bounding box
310 //
311 G4VoxelLimits unLimit;
312 G4double xmin,xmax,ymin,ymax,zmin,zmax;
313 fPtrSolid->CalculateExtent(kXAxis,unLimit,*fDirectTransform,xmin,xmax);
314 fPtrSolid->CalculateExtent(kYAxis,unLimit,*fDirectTransform,ymin,ymax);
315 fPtrSolid->CalculateExtent(kZAxis,unLimit,*fDirectTransform,zmin,zmax);
316 pMin.set(xmin,ymin,zmin);
317 pMax.set(xmax,ymax,zmax);
318 }
319
320 // Check correctness of the bounding box
321 //
322 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
323 {
324 std::ostringstream message;
325 message << "Bad bounding box (min >= max) for solid: "
326 << GetName() << " !"
327 << "\npMin = " << pMin
328 << "\npMax = " << pMax;
329 G4Exception("G4DisplacedSolid::BoundingLimits()", "GeomMgt0001",
330 JustWarning, message);
331 DumpInfo();
332 }
333}

335//////////////////////////////////////////////////////////////////////////
336//
337// Calculate extent under transform and specified limit
338
339G4bool
340G4DisplacedSolid::CalculateExtent( const EAxis pAxis,
341 const G4VoxelLimits& pVoxelLimit,
342 const G4AffineTransform& pTransform,
343 G4double& pMin,
344 G4double& pMax ) const
345{
346 G4AffineTransform sumTransform ;
347 sumTransform.Product(*fDirectTransform,pTransform) ;
348 return fPtrSolid->CalculateExtent(pAxis,pVoxelLimit,sumTransform,pMin,pMax) ;
349}

398//////////////////////////////////////////////////////////
399//
400// The same algorithm as DistanceToOut(p)
401
402G4double
403G4DisplacedSolid::DistanceToOut( const G4ThreeVector& p,
404 const G4ThreeVector& v,
405 const G4bool calcNorm,
406 G4bool *validNorm,
407 G4ThreeVector *n ) const
408{
409 G4ThreeVector solNorm ;
410 G4ThreeVector newPoint = fPtrTransform->TransformPoint(p) ;
411 G4ThreeVector newDirection = fPtrTransform->TransformAxis(v) ;
412 G4double dist = fPtrSolid->DistanceToOut(newPoint,newDirection,
413 calcNorm,validNorm,&solNorm) ;
414 if(calcNorm)
415 {
416 *n = fDirectTransform->TransformAxis(solNorm) ;
417 }
418 return dist ;
419}

513//////////////////////////////////////////////////////////////////////////

515//
516
517G4Polyhedron*
518G4DisplacedSolid::CreatePolyhedron () const
519{
520 G4Polyhedron* polyhedron = fPtrSolid->CreatePolyhedron();
521 if (polyhedron)
522 {
523 polyhedron
524 ->Transform(G4Transform3D(GetObjectRotation(),GetObjectTranslation()));
525 }
526 else
527 {
528 DumpInfo();
529 G4Exception("G4DisplacedSolid::CreatePolyhedron()",
530 "GeomSolids2002", JustWarning,
531 "No G4Polyhedron for displaced solid");
532 }
533 return polyhedron;
534}

540G4Polyhedron* G4DisplacedSolid::GetPolyhedron () const
541{

543 fRebuildPolyhedron ||
544 fpPolyhedron->GetNumberOfRotationStepsAtTimeOfCreation() !=
545 fpPolyhedron->GetNumberOfRotationSteps())
546 {
547 fpPolyhedron = CreatePolyhedron();
548 fRebuildPolyhedron = false;
549 }
550 return fpPolyhedron;
