136//////////////////////////////////////////////////////////////////////////
137//
138// Get bounding box
139
140void
141G4IntersectionSolid::BoundingLimits(G4ThreeVector& pMin,
142 G4ThreeVector& pMax) const
143{
144 G4ThreeVector minA,maxA, minB,maxB;
145 fPtrSolidA->BoundingLimits(minA,maxA);
146 fPtrSolidB->BoundingLimits(minB,maxB);
147
148 pMin.set(std::max(minA.x(),minB.x()),
149 std::max(minA.y(),minB.y()),
150 std::max(minA.z(),minB.z()));
151
152 pMax.set(std::min(maxA.x(),maxB.x()),
153 std::min(maxA.y(),maxB.y()),
154 std::min(maxA.z(),maxB.z()));
155
156 // Check correctness of the bounding box
157 //
158 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
159 {
160 std::ostringstream message;
161 message << "Bad bounding box (min >= max) for solid: "
162 << GetName() << " !"
163 << "\npMin = " << pMin
164 << "\npMax = " << pMax;
165 G4Exception("G4IntersectionSolid::BoundingLimits()", "GeomMgt0001",
166 JustWarning, message);
167 DumpInfo();
168 }
169}

171//////////////////////////////////////////////////////////////////////////
172//
173// Calculate extent under transform and specified limit
174
175G4bool
176G4IntersectionSolid::CalculateExtent(const EAxis pAxis,
177 const G4VoxelLimits& pVoxelLimit,
178 const G4AffineTransform& pTransform,
179 G4double& pMin,
180 G4double& pMax) const
181{
182 G4bool retA, retB, out;
183 G4double minA, minB, maxA, maxB;
184
185 retA = fPtrSolidA
186 ->CalculateExtent( pAxis, pVoxelLimit, pTransform, minA, maxA);
187 retB = fPtrSolidB
188 ->CalculateExtent( pAxis, pVoxelLimit, pTransform, minB, maxB);
189
190 if( retA && retB )
191 {
192 pMin = std::max( minA, minB );
193 pMax = std::min( maxA, maxB );
194 out = (pMax > pMin); // true;
195 }
196 else
197 {
198 out = false;
199 }
200
201 return out; // It exists in this slice only if both exist in it.
202}

204/////////////////////////////////////////////////////
205//
206// Touching ? Empty intersection ?
207
208EInside G4IntersectionSolid::Inside(const G4ThreeVector& p) const
209{
210 EInside positionA = fPtrSolidA->Inside(p);
211 if(positionA == kOutside) return positionA; //outside A
212
213 EInside positionB = fPtrSolidB->Inside(p);
214 if(positionA == kInside) return positionB;
215
216 if(positionB == kOutside) return positionB; // outside B
217 return kSurface; // surface A & B
218}

220//////////////////////////////////////////////////////////////
221//
222
223G4ThreeVector
224G4IntersectionSolid::SurfaceNormal( const G4ThreeVector& p ) const
225{
226 G4ThreeVector normal;
227 EInside insideA, insideB;
228
229 insideA= fPtrSolidA->Inside(p);
230 insideB= fPtrSolidB->Inside(p);
231
232#ifdef G4BOOLDEBUG
233 if( (insideA == kOutside) || (insideB == kOutside) )
234 {
235 G4cout << "WARNING - Invalid call in "
236 << "G4IntersectionSolid::SurfaceNormal(p)" << G4endl
237 << " Point p is outside !" << G4endl;
238 G4cout << " p = " << p << G4endl;
239 G4cerr << "WARNING - Invalid call in "
240 << "G4IntersectionSolid::SurfaceNormal(p)" << G4endl
241 << " Point p is outside !" << G4endl;
242 G4cerr << " p = " << p << G4endl;
243 }
244#endif
245
246 // On the surface of both is difficult ... treat it like on A now!
247 //
248 if( insideA == kSurface )
249 {
250 normal= fPtrSolidA->SurfaceNormal(p) ;
251 }
252 else if( insideB == kSurface )
253 {
254 normal= fPtrSolidB->SurfaceNormal(p) ;
255 }
256 else // We are on neither surface, so we should generate an exception
257 {
258 if(fPtrSolidA->DistanceToOut(p) <= fPtrSolidB->DistanceToOut(p) )
259 normal= fPtrSolidA->SurfaceNormal(p) ;
260 else
261 normal= fPtrSolidB->SurfaceNormal(p) ;
262#ifdef G4BOOLDEBUG
263 G4cout << "WARNING - Invalid call in "
264 << "G4IntersectionSolid::SurfaceNormal(p)" << G4endl
265 << " Point p is out of surface !" << G4endl;
266 G4cout << " p = " << p << G4endl;
267 G4cerr << "WARNING - Invalid call in "
268 << "G4IntersectionSolid::SurfaceNormal(p)" << G4endl
269 << " Point p is out of surface !" << G4endl;
270 G4cerr << " p = " << p << G4endl;
271#endif
272 }
273
274 return normal;
275}

277/////////////////////////////////////////////////////////////
278//
279// The same algorithm as in DistanceToIn(p)
280
281G4double
282G4IntersectionSolid::DistanceToIn( const G4ThreeVector& p,
283 const G4ThreeVector& v ) const
284{
285 G4double dist = 0.0;
286 if( Inside(p) == kInside )
287 {
288#ifdef G4BOOLDEBUG
289 G4cout << "WARNING - Invalid call in "
290 << "G4IntersectionSolid::DistanceToIn(p,v)" << G4endl
291 << " Point p is inside !" << G4endl;
292 G4cout << " p = " << p << G4endl;
293 G4cout << " v = " << v << G4endl;
294 G4cerr << "WARNING - Invalid call in "
295 << "G4IntersectionSolid::DistanceToIn(p,v)" << G4endl
296 << " Point p is inside !" << G4endl;
297 G4cerr << " p = " << p << G4endl;
298 G4cerr << " v = " << v << G4endl;
299#endif
300 }
301 else // if( Inside(p) == kSurface )
302 {
303 EInside wA = fPtrSolidA->Inside(p);
304 EInside wB = fPtrSolidB->Inside(p);
305
306 G4ThreeVector pA = p, pB = p;
307 G4double dA = 0., dA1=0., dA2=0.;
308 G4double dB = 0., dB1=0., dB2=0.;
309 G4bool doA = true, doB = true;
310
311 static const size_t max_trials=10000;
312 for (size_t trial=0; trial<max_trials; ++trial)
313 {
314 if(doA)
315 {
316 // find next valid range for A
317
318 dA1 = 0.;
319
320 if( wA != kInside )
321 {
322 dA1 = fPtrSolidA->DistanceToIn(pA, v);
323
324 if( dA1 == kInfinity ) return kInfinity;
325
326 pA += dA1*v;
327 }
328 dA2 = dA1 + fPtrSolidA->DistanceToOut(pA, v);
329 }
330 dA1 += dA;
331 dA2 += dA;
332
333 if(doB)
334 {
335 // find next valid range for B
336
337 dB1 = 0.;
338 if(wB != kInside)
339 {
340 dB1 = fPtrSolidB->DistanceToIn(pB, v);
341
342 if(dB1 == kInfinity) return kInfinity;
343
344 pB += dB1*v;
345 }
346 dB2 = dB1 + fPtrSolidB->DistanceToOut(pB, v);
347 }
348 dB1 += dB;
349 dB2 += dB;
350
351 // check if they overlap
352
353 if( dA1 < dB1 )
354 {
355 if( dB1 < dA2 ) return dB1;
356
357 dA = dA2;
358 pA = p + dA*v; // continue from here
359 wA = kSurface;
360 doA = true;
361 doB = false;
362 }
363 else
364 {
365 if( dA1 < dB2 ) return dA1;
366
367 dB = dB2;
368 pB = p + dB*v; // continue from here
369 wB = kSurface;
370 doB = true;
371 doA = false;
372 }
373 }
374 }
375#ifdef G4BOOLDEBUG
376 G4Exception("G4IntersectionSolid::DistanceToIn(p,v)",
377 "GeomSolids0001", JustWarning,
378 "Reached maximum number of iterations! Returning zero.");
379#endif
380 return dist ;
381}

383////////////////////////////////////////////////////////
384//
385// Approximate nearest distance from the point p to the intersection of
386// two solids
387
388G4double
389G4IntersectionSolid::DistanceToIn( const G4ThreeVector& p) const
390{
391#ifdef G4BOOLDEBUG
392 if( Inside(p) == kInside )
393 {
394 G4cout << "WARNING - Invalid call in "
395 << "G4IntersectionSolid::DistanceToIn(p)" << G4endl
396 << " Point p is inside !" << G4endl;
397 G4cout << " p = " << p << G4endl;
398 G4cerr << "WARNING - Invalid call in "
399 << "G4IntersectionSolid::DistanceToIn(p)" << G4endl
400 << " Point p is inside !" << G4endl;
401 G4cerr << " p = " << p << G4endl;
402 }
403#endif
404 EInside sideA = fPtrSolidA->Inside(p) ;
405 EInside sideB = fPtrSolidB->Inside(p) ;
406 G4double dist=0.0 ;
407
408 if( sideA != kInside && sideB != kOutside )
409 {
410 dist = fPtrSolidA->DistanceToIn(p) ;
411 }
412 else
413 {
414 if( sideB != kInside && sideA != kOutside )
415 {
416 dist = fPtrSolidB->DistanceToIn(p) ;
417 }
418 else
419 {
420 dist = std::min(fPtrSolidA->DistanceToIn(p),
421 fPtrSolidB->DistanceToIn(p) ) ;
422 }
423 }
424 return dist ;
425}

427//////////////////////////////////////////////////////////
428//
429// The same algorithm as DistanceToOut(p)
430
431G4double
432G4IntersectionSolid::DistanceToOut( const G4ThreeVector& p,
433 const G4ThreeVector& v,
434 const G4bool calcNorm,
435 G4bool *validNorm,
436 G4ThreeVector *n ) const
437{
438 G4bool validNormA, validNormB;
439 G4ThreeVector nA, nB;
440
441#ifdef G4BOOLDEBUG
442 if( Inside(p) == kOutside )
443 {
444 G4cout << "Position:" << G4endl << G4endl;
445 G4cout << "p.x() = " << p.x()/mm << " mm" << G4endl;
446 G4cout << "p.y() = " << p.y()/mm << " mm" << G4endl;
447 G4cout << "p.z() = " << p.z()/mm << " mm" << G4endl << G4endl;
448 G4cout << "Direction:" << G4endl << G4endl;
449 G4cout << "v.x() = " << v.x() << G4endl;
450 G4cout << "v.y() = " << v.y() << G4endl;
451 G4cout << "v.z() = " << v.z() << G4endl << G4endl;
452 G4cout << "WARNING - Invalid call in "
453 << "G4IntersectionSolid::DistanceToOut(p,v)" << G4endl
454 << " Point p is outside !" << G4endl;
455 G4cout << " p = " << p << G4endl;
456 G4cout << " v = " << v << G4endl;
457 G4cerr << "WARNING - Invalid call in "
458 << "G4IntersectionSolid::DistanceToOut(p,v)" << G4endl
459 << " Point p is outside !" << G4endl;
460 G4cerr << " p = " << p << G4endl;
461 G4cerr << " v = " << v << G4endl;
462 }
463#endif
464 G4double distA = fPtrSolidA->DistanceToOut(p,v,calcNorm,&validNormA,&nA) ;
465 G4double distB = fPtrSolidB->DistanceToOut(p,v,calcNorm,&validNormB,&nB) ;
466
467 G4double dist = std::min(distA,distB) ;
468
469 if( calcNorm )
470 {
471 if ( distA < distB )
472 {
473 *validNorm = validNormA;
474 *n = nA;
475 }
476 else
477 {
478 *validNorm = validNormB;
479 *n = nB;
480 }
481 }
482
483 return dist ;
484}

554
555G4Polyhedron*
556G4IntersectionSolid::CreatePolyhedron () const
557{
558 HepPolyhedronProcessor processor;
559 // Stack components and components of components recursively
560 // See G4BooleanSolid::StackPolyhedron
561 G4Polyhedron* top = StackPolyhedron(processor, this);
562 G4Polyhedron* result = new G4Polyhedron(*top);
563 if (processor.execute(*result)) { return result; }
564 else { return 0; }
565}
