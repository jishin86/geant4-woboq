32// G4MultiUnion
33//
34// Class description:
35//
36// An instance of "G4MultiUnion" constitutes a grouping of several solids.
37// The constituent solids are stored with their respective location in an
38// instance of "G4Node". An instance of "G4MultiUnion" is subsequently
39// composed of one or several nodes.
40
41// History:
42// 06.04.17 G.Cosmo - Imported implementation in Geant4 for VecGeom migration
43// 19.10.12 M.Gayer - Original implementation from USolids module
44// --------------------------------------------------------------------

221//______________________________________________________________________________
222inline
223G4ThreeVector G4MultiUnion::GetLocalPoint(const G4Transform3D& trans,
224 const G4ThreeVector& global) const
225{
226 // Returns local point coordinates converted from the global frame defined
227 // by the transformation. This is defined by multiplying the inverse
228 // transformation with the global vector.
229
230 return trans.inverse()*G4Point3D(global);
231}
232
233//______________________________________________________________________________
234inline
235G4ThreeVector G4MultiUnion::GetLocalVector(const G4Transform3D& trans,
236 const G4ThreeVector& global) const
237{
238 // Returns local point coordinates converted from the global frame defined
239 // by the transformation. This is defined by multiplying the inverse
240 // transformation with the global vector.
241
242 G4Rotate3D rot;
243 G4Translate3D transl ;
244 G4Scale3D scale;
245
246 trans.getDecomposition(scale,rot,transl);
247 return rot.inverse()*G4Vector3D(global);
248}
249
250//______________________________________________________________________________
251inline
252G4ThreeVector G4MultiUnion::GetGlobalPoint(const G4Transform3D& trans,
253 const G4ThreeVector& local) const
254{
255 // Returns global point coordinates converted from the local frame defined
256 // by the transformation. This is defined by multiplying this transformation
257 // with the local vector.
258
259 return trans*G4Point3D(local);
260}
261
262//______________________________________________________________________________
263inline
264G4ThreeVector G4MultiUnion::GetGlobalVector(const G4Transform3D& trans,
265 const G4ThreeVector& local) const
266{
267 // Returns vector components converted from the local frame defined by the
268 // transformation to the global one. This is defined by multiplying this
269 // transformation with the local vector while ignoring the translation.
270
271 G4Rotate3D rot;
272 G4Translate3D transl ;
273 G4Scale3D scale;
274
275 trans.getDecomposition(scale,rot,transl);
276 return rot*G4Vector3D(local);
277}
