46//--------------------------------------------------------------------------
47G4VParameterisationTrd::
48G4VParameterisationTrd( EAxis axis, G4int nDiv, G4double width,
49 G4double offset, G4VSolid* msolid,
50 DivisionType divType )
51 : G4VDivisionParameterisation( axis, nDiv, width, offset, divType, msolid ),
52 bDivInTrap(false)
53{
54 G4Trd* msol = (G4Trd*)(msolid);
55 if (msolid->GetEntityType() == "G4ReflectedSolid")
56 {
57 // Get constituent solid
58 G4VSolid* mConstituentSolid
59 = ((G4ReflectedSolid*)msolid)->GetConstituentMovedSolid();
60 msol = (G4Trd*)(mConstituentSolid);
61
62 // Create a new solid with inversed parameters
63 G4Trd* newSolid
64 = new G4Trd(msol->GetName(),
65 msol->GetXHalfLength2(), msol->GetXHalfLength1(),
66 msol->GetYHalfLength2(), msol->GetYHalfLength1(),
67 msol->GetZHalfLength());
68 msol = newSolid;
69 fmotherSolid = newSolid;
70 fReflectedSolid = true;
71 fDeleteSolid = true;
72 }
73}

81G4ParameterisationTrdX::
82G4ParameterisationTrdX( EAxis axis, G4int nDiv,
83 G4double width, G4double offset,
84 G4VSolid* msolid, DivisionType divType )
85 : G4VParameterisationTrd( axis, nDiv, width, offset, msolid, divType )
86{
87 CheckParametersValidity();
88 SetType( "DivisionTrdX" );
89
90 G4Trd* msol = (G4Trd*)(fmotherSolid);
91 if( divType == DivWIDTH )
92 {
93 fnDiv = CalculateNDiv( msol->GetXHalfLength1()+msol->GetXHalfLength2(),
94 width, offset );
95 }
96 else if( divType == DivNDIV )
97 {
98 fwidth = CalculateWidth( msol->GetXHalfLength1()+msol->GetXHalfLength2(),
99 nDiv, offset );
100 }
101
102#ifdef G4DIVDEBUG
103 if( verbose >= 1 )
104 {
105 G4cout << " G4ParameterisationTrdX - ## divisions " << fnDiv << " = "
106 << nDiv << G4endl
107 << " Offset " << foffset << " = " << offset << G4endl
108 << " Width " << fwidth << " = " << width << G4endl;
109 }
110#endif
111
112 G4double mpDx1 = msol->GetXHalfLength1();
113 G4double mpDx2 = msol->GetXHalfLength2();
114 if( std::fabs(mpDx1 - mpDx2) > kCarTolerance )
115 {
116 bDivInTrap = true;
117 }
118}

132//------------------------------------------------------------------------
133void
134G4ParameterisationTrdX::
135ComputeTransformation( const G4int copyNo,
136 G4VPhysicalVolume *physVol ) const
137{
138 G4Trd* msol = (G4Trd*)(fmotherSolid );
139 G4double mdx = ( msol->GetXHalfLength1() + msol->GetXHalfLength2() ) / 2.;
140
141 //----- translation
142 G4ThreeVector origin(0.,0.,0.);
143 G4double posi;
144 if( !bDivInTrap )
145 {
146 posi = -mdx + foffset + (copyNo+0.5)*fwidth;
147 }
148 else
149 {
150 G4double aveHL = (msol->GetXHalfLength1()+msol->GetXHalfLength2())/2.;
151 posi = - aveHL + foffset + (copyNo+0.5)*aveHL/fnDiv*2;
152 }
153 if( faxis == kXAxis )
154 {
155 origin.setX( posi );
156 }
157 else
158 {
159 std::ostringstream message;
160 message << "Only axes along X are allowed ! Axis: " << faxis;
161 G4Exception("G4ParameterisationTrdX::ComputeTransformation()",
162 "GeomDiv0002", FatalException, message);
163 }
164
165#ifdef G4DIVDEBUG
166 if( verbose >= 2 )
167 {
168 G4cout << std::setprecision(8)
169 << " G4ParameterisationTrdX::ComputeTransformation() "
170 << copyNo << G4endl
171 << " Position: " << origin << " - Axis: " << faxis << G4endl;
172 }
173#endif
174
175 //----- set translation
176 physVol->SetTranslation( origin );
177}

180void
181G4ParameterisationTrdX::
182ComputeDimensions( G4Trd& trd, const G4int, const G4VPhysicalVolume* ) const
183{
184 G4Trd* msol = (G4Trd*)(fmotherSolid);
185
186 G4double pDy1 = msol->GetYHalfLength1();
187 G4double pDy2 = msol->GetYHalfLength2();
188 G4double pDz = msol->GetZHalfLength();
189 G4double pDx = fwidth/2. - fhgap;
190
191 trd.SetAllParameters ( pDx, pDx, pDy1, pDy2, pDz );
192
193#ifdef G4DIVDEBUG

217//--------------------------------------------------------------------------
218void
219G4ParameterisationTrdX::ComputeDimensions( G4Trap& trap, const G4int copyNo,
220 const G4VPhysicalVolume* ) const
221{
222 G4Trd* msol = (G4Trd*)(fmotherSolid);
223 G4double pDy1 = msol->GetYHalfLength1();
224 G4double pDy2 = msol->GetYHalfLength2();
225 G4double pDz = msol->GetZHalfLength();
226 G4double pDx1 = msol->GetXHalfLength1()/fnDiv;
227 G4double pDx2 = msol->GetXHalfLength2()/fnDiv;
228
229 G4double cxy1 = -msol->GetXHalfLength1() + foffset
230 + (copyNo+0.5)*pDx1*2;// centre of the side at y=-pDy1
231 G4double cxy2 = -msol->GetXHalfLength2() + foffset
232 + (copyNo+0.5)*pDx2*2;// centre of the side at y=+pDy1
233 G4double alp = std::atan( (cxy2-cxy1)/pDz );
234
235 trap.SetAllParameters ( pDz,
236 0.,
237 0.,
238 pDy1,
239 pDx1,
240 pDx2,
241 alp,
242 pDy2,
243 pDx1 - fhgap,
244 pDx2 - fhgap * pDx2/pDx1,
245 alp);
246
247#ifdef G4DIVDEBUG
248 if( verbose >= 2 )
249 {
250 G4cout << " G4ParameterisationTrdX::ComputeDimensions():"
251 << copyNo << G4endl;
252 trap.DumpInfo();
253 }
254#endif
255}

333//--------------------------------------------------------------------------
334void
335G4ParameterisationTrdY::
336ComputeTransformation( const G4int copyNo, G4VPhysicalVolume* physVol ) const
337{
338 G4Trd* msol = (G4Trd*)(fmotherSolid );
339 G4double mdy = msol->GetYHalfLength1();
340
341 //----- translation
342 G4ThreeVector origin(0.,0.,0.);
343 G4double posi = -mdy + foffset + (copyNo+0.5)*fwidth;
344
345 if( faxis == kYAxis )
346 {
347 origin.setY( posi );
348 }
349 else
350 {
351 std::ostringstream message;
352 message << "Only axes along Y are allowed ! Axis: " << faxis;
353 G4Exception("G4ParameterisationTrdY::ComputeTransformation()",
354 "GeomDiv0002", FatalException, message);
355 }
356
357#ifdef G4DIVDEBUG
358 if( verbose >= 2 )

361 << " G4ParameterisationTrdY::ComputeTransformation " << copyNo
362 << " pos " << origin << " rot mat " << " axis " << faxis << G4endl;
363 }
364#endif
365
366 //----- set translation

370//--------------------------------------------------------------------------
371void
372G4ParameterisationTrdY::
373ComputeDimensions(G4Trd& trd, const G4int, const G4VPhysicalVolume*) const
374{
375 //---- The division along Y of a Trd will result a Trd, only
376 //--- if Y at -Z and +Z are equal, else use the G4Trap version
377 G4Trd* msol = (G4Trd*)(fmotherSolid);
378
379 G4double pDx1 = msol->GetXHalfLength1();
380 G4double pDx2 = msol->GetXHalfLength2();
381 G4double pDz = msol->GetZHalfLength();
382 G4double pDy = fwidth/2. - fhgap;
383
384 trd.SetAllParameters ( pDx1, pDx2, pDy, pDy, pDz );
385
386#ifdef G4DIVDEBUG
387 if( verbose >= 2 )
388 {
389 G4cout << " G4ParameterisationTrdY::ComputeDimensions():" << G4endl;
390 trd.DumpInfo();
391 }
392#endif
393}

396void G4ParameterisationTrdY::CheckParametersValidity()
397{
398 G4VDivisionParameterisation::CheckParametersValidity();

401
402 G4double mpDy1 = msol->GetYHalfLength1();

414 "GeomDiv0001", FatalException, message);
415 }

500//--------------------------------------------------------------------------
501void
502G4ParameterisationTrdZ::
503ComputeDimensions(G4Trd& trd, const G4int copyNo,
504 const G4VPhysicalVolume*) const
505{
506 //---- The division along Z of a Trd will result a Trd
507 G4Trd* msol = (G4Trd*)(fmotherSolid);
508
509 G4double pDx1 = msol->GetXHalfLength1();
510 G4double DDx = (msol->GetXHalfLength2() - msol->GetXHalfLength1() );
511 G4double pDy1 = msol->GetYHalfLength1();
512 G4double DDy = (msol->GetYHalfLength2() - msol->GetYHalfLength1() );
513 G4double pDz = fwidth/2. - fhgap;
514 G4double zLength = 2*msol->GetZHalfLength();
515
516 trd.SetAllParameters( pDx1+DDx*(OffsetZ()+copyNo*fwidth+fhgap)/zLength,
517 pDx1+DDx*(OffsetZ()+(copyNo+1)*fwidth-fhgap)/zLength,
518 pDy1+DDy*(OffsetZ()+copyNo*fwidth+fhgap)/zLength,
519 pDy1+DDy*(OffsetZ()+(copyNo+1)*fwidth-fhgap)/zLength,
520 pDz );
521
522#ifdef G4DIVDEBUG
523 if( verbose >= 1 )
524 {
525 G4cout << " G4ParameterisationTrdZ::ComputeDimensions()"
526 << " - Mother TRD " << G4endl;
527 msol->DumpInfo();
528 G4cout << " - Parameterised TRD: "
529 << copyNo << G4endl;
530 trd.DumpInfo();
531 }
532#endif
