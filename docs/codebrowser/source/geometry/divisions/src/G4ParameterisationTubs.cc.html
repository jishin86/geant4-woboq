45//--------------------------------------------------------------------------
46G4VParameterisationTubs::
47G4VParameterisationTubs( EAxis axis, G4int nDiv, G4double width,
48 G4double offset, G4VSolid* msolid,
49 DivisionType divType )
50 : G4VDivisionParameterisation( axis, nDiv, width, offset, divType, msolid )
51{
52 G4Tubs* msol = (G4Tubs*)(msolid);
53 if (msolid->GetEntityType() == "G4ReflectedSolid")
54 {
55 //----- get constituent solid
56 G4VSolid* mConstituentSolid
57 = ((G4ReflectedSolid*)msolid)->GetConstituentMovedSolid();
58 msol = (G4Tubs*)(mConstituentSolid);
59 fmotherSolid = msol;
60 fReflectedSolid = true;
61 }
62}

69//--------------------------------------------------------------------------
70G4ParameterisationTubsRho::
71G4ParameterisationTubsRho( EAxis axis, G4int nDiv,
72 G4double width, G4double offset,
73 G4VSolid* msolid, DivisionType divType )
74 : G4VParameterisationTubs( axis, nDiv, width, offset, msolid, divType )
75{
76 CheckParametersValidity();
77 SetType( "DivisionTubsRho" );
78
79 G4Tubs* msol = (G4Tubs*)(fmotherSolid);
80 if( divType == DivWIDTH )
81 {
82 fnDiv = CalculateNDiv( msol->GetOuterRadius() - msol->GetInnerRadius(),
83 width, offset );
84 }
85 else if( divType == DivNDIV )
86 {
87 fwidth = CalculateWidth( msol->GetOuterRadius() - msol->GetInnerRadius(),
88 nDiv, offset );
89 }
90
91#ifdef G4DIVDEBUG
92 if( verbose >= 1 )
93 {
94 G4cout << " G4ParameterisationTubsRho - no divisions " << fnDiv << " = "
95 << nDiv << G4endl
96 << " Offset " << foffset << " = " << offset << G4endl
97 << " Width " << fwidth << " = " << width << G4endl
98 << " DivType " << divType << G4endl;
99 }
100#endif
101}

149//--------------------------------------------------------------------------
150void
151G4ParameterisationTubsRho::
152ComputeDimensions( G4Tubs& tubs, const G4int copyNo,
153 const G4VPhysicalVolume* ) const
154{
155 G4Tubs* msol = (G4Tubs*)(fmotherSolid);
156
157 G4double pRMin = msol->GetInnerRadius() + foffset + fwidth * copyNo + fhgap;
158 G4double pRMax = msol->GetInnerRadius() + foffset + fwidth * (copyNo+1) - fhgap;
159 G4double pDz = msol->GetZHalfLength();
160 //- already rotated G4double pSR = foffset + copyNo*fwidth;
161 G4double pSPhi = msol->GetStartPhiAngle();
162 G4double pDPhi = msol->GetDeltaPhiAngle();;
163
164 tubs.SetInnerRadius( pRMin );
165 tubs.SetOuterRadius( pRMax );
166 tubs.SetZHalfLength( pDz );
167 tubs.SetStartPhiAngle( pSPhi, false );
168 tubs.SetDeltaPhiAngle( pDPhi );
169
170#ifdef G4DIVDEBUG
171 if( verbose >= 2 )
172 {
173 G4cout << " G4ParameterisationTubsRho::ComputeDimensions()" << G4endl
174 << " pRMin: " << pRMin << " - pRMax: " << pRMax << G4endl;
175 tubs.DumpInfo();
176 }
177#endif
178}

223//--------------------------------------------------------------------------
224void
225G4ParameterisationTubsPhi::
226ComputeTransformation(const G4int copyNo, G4VPhysicalVolume *physVol) const
227{
228 //----- translation
229 G4ThreeVector origin(0.,0.,0.);
230 //----- set translation
231 physVol->SetTranslation( origin );
232
233 //----- calculate rotation matrix (so that all volumes point to the centre)
234 G4double posi = foffset + copyNo*fwidth;
235
236#ifdef G4DIVDEBUG
237 if( verbose >= 2 )
238 {
239 G4cout << " G4ParameterisationTubsPhi - position: " << posi/deg << G4endl
240 << " copyNo: " << copyNo << " - foffset: " << foffset/deg
241 << " - fwidth: " << fwidth/deg << G4endl;
242 }
243#endif
244
245 ChangeRotMatrix( physVol, -posi );
246
247#ifdef G4DIVDEBUG
248 if( verbose >= 2 )
249 {
250 G4cout << std::setprecision(8) << " G4ParameterisationTubsPhi " << copyNo
251 << G4endl
252 << " Position: " << origin << " - Width: " << fwidth
253 << " - Axis: " << faxis << G4endl;
254 }
255#endif
256}

258//--------------------------------------------------------------------------
259void
260G4ParameterisationTubsPhi::
261ComputeDimensions( G4Tubs& tubs, const G4int,
262 const G4VPhysicalVolume* ) const
263{
264 G4Tubs* msol = (G4Tubs*)(fmotherSolid);
265
266 G4double pRMin = msol->GetInnerRadius();
267 G4double pRMax = msol->GetOuterRadius();
268 G4double pDz = msol->GetZHalfLength();
269 //----- already rotated in 'ComputeTransformation'
270 G4double pSPhi = msol->GetStartPhiAngle() + fhgap;
271 G4double pDPhi = fwidth - 2.*fhgap;
272
273 tubs.SetInnerRadius( pRMin );
274 tubs.SetOuterRadius( pRMax );
275 tubs.SetZHalfLength( pDz );
276 tubs.SetStartPhiAngle( pSPhi, false );
277 tubs.SetDeltaPhiAngle( pDPhi );
278
279#ifdef G4DIVDEBUG
280 if( verbose >= 2 )
281 {
282 G4cout << " G4ParameterisationTubsPhi::ComputeDimensions" << G4endl
283 << " pSPhi: " << pSPhi << " - pDPhi: " << pDPhi << G4endl;
284 tubs.DumpInfo();
285 }
286#endif
287}

334G4ParameterisationTubsZ::

336{
337 //----- set translation: along Z axis

339 G4double posi = - motherTubs->GetZHalfLength() + OffsetZ()
340 + fwidth/2 + copyNo*fwidth;
341 G4ThreeVector origin(0.,0.,posi);
342 physVol->SetTranslation( origin );
343
344 //----- calculate rotation matrix: unit
345
346#ifdef G4DIVDEBUG

348 {

350 << " Position: " << posi << " - copyNo: " << copyNo << G4endl
351 << " foffset " << foffset/deg << " - fwidth " << fwidth/deg
352 << G4endl;
353 }
354#endif
355
356 ChangeRotMatrix( physVol );
357
358#ifdef G4DIVDEBUG
