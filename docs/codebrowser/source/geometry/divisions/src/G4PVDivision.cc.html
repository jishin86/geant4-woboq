45//--------------------------------------------------------------------------
46G4PVDivision::G4PVDivision(const G4String& pName,
47 G4LogicalVolume* pLogical,
48 G4LogicalVolume* pMotherLogical,
49 const EAxis pAxis,
50 const G4int nDivs,
51 const G4double width,
52 const G4double offset )
53 : G4VPhysicalVolume(0,G4ThreeVector(),pName,pLogical,0),
54 fcopyNo(-1)
55{
56 if (!pMotherLogical)
57 {
58 std::ostringstream message;
59 message << "Invalid setup." << G4endl
60 << "NULL pointer specified as mother for volume: " << pName;
61 G4Exception("G4PVDivision::G4PVDivision()", "GeomDiv0002",
62 FatalException, message);
63 return;
64 }
65 if (pLogical == pMotherLogical)
66 {
67 std::ostringstream message;
68 message << "Invalid setup." << G4endl
69 << "Cannot place a volume inside itself! Volume: " << pName;
70 G4Exception("G4PVDivision::G4PVDivision()", "GeomDiv0002",
71 FatalException, message);
72 }
73 pMotherLogical->AddDaughter(this);
74 SetMotherLogical(pMotherLogical);
75 SetParameterisation(pMotherLogical, pAxis, nDivs,
76 width, offset, DivNDIVandWIDTH);
77 CheckAndSetParameters (pAxis, nDivs, width, offset,
78 DivNDIVandWIDTH, pMotherLogical);
79}

147//--------------------------------------------------------------------------
148void
149G4PVDivision::CheckAndSetParameters( const EAxis pAxis,
150 const G4int nDivs,
151 const G4double width,
152 const G4double offset,
153 DivisionType divType,
154 const G4LogicalVolume* pMotherLogical )
155{
156 if( divType == DivWIDTH )
157 {
158 fnReplicas = fparam->GetNoDiv();
159 }
160 else
161 {
162 fnReplicas = nDivs;
163 }
164 if (fnReplicas < 1 )
165 {
166 G4Exception("G4PVDivision::CheckAndSetParameters()", "GeomDiv0002",
167 FatalException, "Illegal number of replicas!");
168 }
169
170 if( divType != DivNDIV)
171 {
172 fwidth = fparam->GetWidth();
173 }
174 else
175 {
176 fwidth = width;
177 }
178 if( fwidth < 0 )
179 {
180 G4Exception("G4PVDivision::CheckAndSetParameters()", "GeomDiv0002",
181 FatalException, "Width must be positive!");
182 }
183
184 foffset = offset;
185 fdivAxis = pAxis;
186
187 //!!!!! axis has to be x/y/z in G4VoxelLimits::GetMinExtent
188 //
189 if( pAxis == kRho || pAxis == kRadial3D || pAxis == kPhi )
190 {
191 faxis = kZAxis;
192 }
193 else
194 {
195 faxis = pAxis;
196 }
197
198 // Create rotation matrix: for phi axis it will be changed
199 // in G4VPVParameterisation::ComputeTransformation, for others
200 // it will stay the unity
201 //
202 G4RotationMatrix *pRMat = new G4RotationMatrix();
203 SetRotation(pRMat);
204
205 switch (faxis)
206 {
207 case kPhi:
208 break;
209 case kRho:
210 case kXAxis:
211 case kYAxis:
212 case kZAxis:
213 break;
214 default:
215 G4Exception("G4PVDivision::CheckAndSetParameters()", "GeomDiv0002",
216 FatalException, "Unknown axis of replication.");
217 break;
218 }
219
220
221 //----- Check that mother solid is of the same type than
222 // daughter solid (otherwise, the corresponding
223 // Parameterisation::ComputeDimension() will not be called)
224 //
225 G4String msolType = pMotherLogical->GetSolid()->GetEntityType();
226 G4String dsolType = GetLogicalVolume()->GetSolid()->GetEntityType();
227 if( msolType != dsolType && ( msolType != "G4Trd" || dsolType != "G4Trap" ) )
228 {
229 std::ostringstream message;
230 message << "Incorrect solid type for division of volume "
231 << GetName() << "." << G4endl
232 << "It is: " << msolType
233 << ", while it should be: " << dsolType << "!";
234 G4Exception("G4PVDivision::CheckAndSetParameters()",
235 "GeomDiv0002", FatalException, message );
236 }
237}

302//TODO: this method should check that the child lv is of the correct type,
303// else the ComputeDimensions will never be called
304void G4PVDivision::SetParameterisation( G4LogicalVolume* motherLogical,
305 const EAxis axis,
306 const G4int nDivs,
307 const G4double width,
308 const G4double offset,
309 DivisionType divType )
310{
311 // Check that solid is compatible with mother solid and axis of division
312 // CheckSolid( solid, motherSolid );
313 // G4cout << " Axis " << axis << G4endl;
314
315 G4VSolid* mSolid = motherLogical->GetSolid();
316 G4String mSolidType = mSolid->GetEntityType();
317
318 // If the solid is a reflected one, update type to its
319 // real constituent solid.
320 //
321 if (mSolidType == "G4ReflectedSolid")
322 {
323 mSolidType = ((G4ReflectedSolid*)mSolid)->GetConstituentMovedSolid()
324 ->GetEntityType();
325 }
326
327 // Parameterisation type depend of mother solid type and axis of division
328 //
329 if( mSolidType == "G4Box" )
330 {
331 switch( axis )
332 {
333 case kXAxis:
334 fparam = new G4ParameterisationBoxX( axis, nDivs, width,
335 offset, mSolid, divType );
336 break;
337 case kYAxis:
338 fparam = new G4ParameterisationBoxY( axis, nDivs, width,
339 offset, mSolid, divType );
340 break;
341 case kZAxis:
342 fparam = new G4ParameterisationBoxZ( axis, nDivs, width,
343 offset, mSolid, divType );
344 break;
345 default:
346 ErrorInAxis( axis, mSolid );
347 break;
348 }
349 }
350 else if( mSolidType == "G4Tubs" )
351 {
352 switch( axis )
353 {
354 case kRho:
355 fparam = new G4ParameterisationTubsRho( axis, nDivs, width,
356 offset, mSolid, divType );
357 break;
358 case kPhi:
359 fparam = new G4ParameterisationTubsPhi( axis, nDivs, width,
360 offset, mSolid, divType );
361 break;
362 case kZAxis:
363 fparam = new G4ParameterisationTubsZ( axis, nDivs, width,
364 offset, mSolid, divType );
365 break;
366 default:
367 ErrorInAxis( axis, mSolid );
368 break;
369 }
370 }
371 else if( mSolidType == "G4Cons" )
372 {
373 switch( axis )
374 {
375 case kRho:
376 fparam = new G4ParameterisationConsRho( axis, nDivs, width,
377 offset, mSolid, divType );
378 break;
379 case kPhi:
380 fparam = new G4ParameterisationConsPhi( axis, nDivs, width,
381 offset, mSolid, divType );
382 break;
383 case kZAxis:
384 fparam = new G4ParameterisationConsZ( axis, nDivs, width,
385 offset, mSolid, divType );
386 break;
387 default:
388 ErrorInAxis( axis, mSolid );
389 break;
390 }
391 }
392 else if( mSolidType == "G4Trd" )
393 {
394 switch( axis )
395 {
396 case kXAxis:
397 fparam = new G4ParameterisationTrdX( axis, nDivs, width,
398 offset, mSolid, divType );
399 break;
400 case kYAxis:
401 fparam = new G4ParameterisationTrdY( axis, nDivs, width,
402 offset, mSolid, divType );
403 break;
404 case kZAxis:
405 fparam = new G4ParameterisationTrdZ( axis, nDivs, width,
406 offset, mSolid, divType );
407 break;
408 default:
409 ErrorInAxis( axis, mSolid );
410 break;
411 }
412 }
413 else if( mSolidType == "G4Para" )
414 {
415 switch( axis )
416 {
417 case kXAxis:
418 fparam = new G4ParameterisationParaX( axis, nDivs, width,
419 offset, mSolid, divType );
420 break;
421 case kYAxis:
422 fparam = new G4ParameterisationParaY( axis, nDivs, width,
423 offset, mSolid, divType );
424 break;
425 case kZAxis:
426 fparam = new G4ParameterisationParaZ( axis, nDivs, width,
427 offset, mSolid, divType );
428 break;
429 default:
430 ErrorInAxis( axis, mSolid );
431 break;
432 }
433 }
434// else if( mSolidType == "G4Trap" )
435// {
436// }
437 else if( mSolidType == "G4Polycone" )
438 {
439 switch( axis )
440 {
441 case kRho:
442 fparam = new G4ParameterisationPolyconeRho( axis, nDivs, width,
443 offset, mSolid, divType );
444 break;
445 case kPhi:
446 fparam = new G4ParameterisationPolyconePhi( axis, nDivs, width,
447 offset, mSolid, divType );
448 break;
449 case kZAxis:
450 fparam = new G4ParameterisationPolyconeZ( axis, nDivs, width,
451 offset, mSolid, divType );
452 break;
453 default:
454 ErrorInAxis( axis, mSolid );
455 break;
456 }
457 }
458 else if( mSolidType == "G4Polyhedra" )
459 {
460 switch( axis )
461 {
462 case kRho:
463 fparam = new G4ParameterisationPolyhedraRho( axis, nDivs, width,
464 offset, mSolid, divType );
465 break;
466 case kPhi:
467 fparam = new G4ParameterisationPolyhedraPhi( axis, nDivs, width,
468 offset, mSolid, divType );
469 break;
470 case kZAxis:
471 fparam = new G4ParameterisationPolyhedraZ( axis, nDivs, width,
472 offset, mSolid, divType );
473 break;
474 default:
475 ErrorInAxis( axis, mSolid );
476 break;
477 }
478 }
479 else
480 {
481 std::ostringstream message;
482 message << "Solid type " << mSolidType << " not supported!" << G4endl
483 << "Divisions for " << mSolidType << " are not implemented.";
484 G4Exception("G4PVDivision::SetParameterisation()", "GeomDiv0001",
485 FatalException, message);
486 }
487}

489//--------------------------------------------------------------------------
490void G4PVDivision::ErrorInAxis( EAxis axis, G4VSolid* solid )
491{
492 G4String error = "Trying to divide solid " + solid->GetName()
493 + " of type " + solid->GetEntityType() + " along axis ";
494 switch( axis )
495 {
496 case kXAxis:
497 error += "X.";
498 break;
499 case kYAxis:
500 error += "Y.";
501 break;
502 case kZAxis:
503 error += "Z.";
504 break;
505 case kRho:
506 error += "Rho.";
507 break;
508 case kRadial3D:
509 error += "Radial3D.";
510 break;
511 case kPhi:
512 error += "Phi.";
513 break;
514 default:
515 break;
516 }
517 G4Exception("G4PVDivision::ErrorInAxis()", "GeomDiv0002",
518 FatalException, error);
519}
