29// class G4ElectroMagneticField
30//
31// Class description:
32//
33// A full Electromagnetic field, containing both electric and magnetic fields.
34// It is an abstract class, and a derived type of this field must be
35// created by the user to describe his/her field configuration.
36
37// We have established a convention for the electromagnetic field components:
38// In the GetValue method, the return values of Bfield will have
39// the following meaning
40// - Components 0, 1 and 2 are the Magnetic Field (x, y, z respectively);
41// - Components 3, 4 and 5 are the Electric field (x, y, z respectively).
42//
43// Note 1: one or the other field could optional, depending on the Equation
44// Note 2: such a convention is required between any field and its
45// corresponding equation of motion.
46//
47// History:
48// - Created: J. Apostolakis, November 12th, 1998
49// - Modified:
50// V. Grichine 8 Nov 2001: Extended "Point" arg to [4] array to add time
51// G. Cosmo 2 Apr 2003: Reorgansised, moved inline methods to .cc
52// J. Apostolakis 5 Nov 2003: Derive directly from G4Field
53// J. Apostolakis 31 Aug 2004: Information on convention for components
54// -------------------------------------------------------------------
55
56#ifndef G4ELECTROMAGNETIC_FIELD_DEF
57#define G4ELECTROMAGNETIC_FIELD_DEF
58
59#include "G4Field.hh"
60
61class G4ElectroMagneticField : public G4Field
62{
63 public: // with description
64
65 G4ElectroMagneticField();
66 virtual ~G4ElectroMagneticField();
67
68 G4ElectroMagneticField(const G4ElectroMagneticField &r);
69 G4ElectroMagneticField& operator = (const G4ElectroMagneticField &p);
70 // Copy constructor & assignment operators.
71
72 virtual void GetFieldValue(const G4double Point[4],
73 G4double *Bfield ) const = 0;
74 // Return as Bfield[0], [1], [2] the magnetic field x, y & z components
75 // and as Bfield[3], [4], [5] the electric field x, y & z components
76
77 virtual G4bool DoesFieldChangeEnergy() const = 0;
78
79 // For field with an electric component this should be true
80 // For pure magnetic field this should be false
81 // Alternative: default safe implementation { return true; }
82
83};
84
85#endif /* G4ELECTROMAGNETIC_FIELD_DEF */
