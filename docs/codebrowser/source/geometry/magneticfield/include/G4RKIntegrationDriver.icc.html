43template <class T>
44G4RKIntegrationDriver<T>::G4RKIntegrationDriver(T* pStepper)
45{
46 RenewStepperAndAdjustImpl(pStepper);
47 fMaxStepBase = 250;
48 fMaxNoSteps = fMaxStepBase / pIntStepper->IntegratorOrder();
49}
50
51// Step failed; compute the size of retrial Step.
52template <class T> G4double G4RKIntegrationDriver<T>::
53ShrinkStepSize(G4double h, G4double error) const
54{
55 if (error > errorConstraintShrink)
56 {
57 return max_stepping_decrease * h;
58 }
59 return GetSafety() * h * std::pow(error, GetPshrnk());
60}
61
62template <class T> G4double G4RKIntegrationDriver<T>::
63ShrinkStepSize2(G4double h, G4double error2) const
64{
65 if (error2 > errorConstraintShrink * errorConstraintShrink)
66 {
67 return max_stepping_decrease * h;
68 }
69 return GetSafety() * h * std::pow(error2, 0.5 * GetPshrnk());
70}
71
72// Compute size of next Step
73template<class T> G4double G4RKIntegrationDriver<T>::
74GrowStepSize(G4double h, G4double error) const
75{
76 if (error < errorConstraintGrow)
77 {
78 return max_stepping_increase * h;
79 }
80 return GetSafety() * h * std::pow(error, GetPgrow());
81}
82
83template<class T> G4double G4RKIntegrationDriver<T>::
84GrowStepSize2(G4double h, G4double error2) const
85{
86 if (error2 < errorConstraintGrow * errorConstraintGrow)
87 {
88 return max_stepping_increase * h;
89 }
90 return GetSafety() * h * std::pow(error2, 0.5 * GetPgrow());
91}
92
93template <class T>
94G4double G4RKIntegrationDriver<T>::
95ComputeNewStepSize( G4double errMaxNorm, // max error (normalised)
96 G4double hstepCurrent ) // current step size
97{
98 if (errMaxNorm > 1)
99 {
100 return ShrinkStepSize(hstepCurrent, errMaxNorm);
101 }
102 else if (errMaxNorm >= 0)
103 {
104 return GrowStepSize(hstepCurrent, errMaxNorm);
105 }
106
107 G4Exception("G4RKIntegrationDriver::ComputeNewStepSize", "GeomField0003",
108 FatalException, "Error is negative!");
109
110 return max_stepping_increase * hstepCurrent;
111}

113template <class T>
114void G4RKIntegrationDriver<T>::
115GetDerivatives( const G4FieldTrack& track, G4double dydx[] ) const
116{
117 G4double y[G4FieldTrack::ncompSVEC];
118 track.DumpToArray(y);
119 pIntStepper->RightHandSide(y, dydx);
120}

122template <class T>
123void G4RKIntegrationDriver<T>::GetDerivatives(const G4FieldTrack& track,
124 G4double dydx[],
125 G4double field[]) const
126{
127 G4double y[G4FieldTrack::ncompSVEC];
128 track.DumpToArray(y);
129 pIntStepper->RightHandSide(y, dydx, field);
130}

132template <class T>
133void G4RKIntegrationDriver<T>::UpdateErrorConstraints()
134{
135 errorConstraintShrink = std::pow(
136 max_stepping_decrease / GetSafety(), 1. / GetPshrnk());
137
138 errorConstraintGrow = std::pow(
139 max_stepping_increase / GetSafety(), 1. / GetPgrow());
140}

142template <class T>
143inline G4double G4RKIntegrationDriver<T>::GetSafety() const

145 return safety;

151 return pshrnk;
152}
153
154template <class T>
155G4double G4RKIntegrationDriver<T>::GetPgrow() const
156{
157 return pgrow;
158}
159
160template <class T>
161void G4RKIntegrationDriver<T>::ReSetParameters(G4double new_safety)
162{
163 safety = new_safety;
164 pshrnk = -1.0 / pIntStepper->IntegratorOrder();
165 pgrow = -1.0 / (1.0 + pIntStepper->IntegratorOrder());
166 UpdateErrorConstraints();
167}
168
169template <class T>
170void G4RKIntegrationDriver<T>::SetSafety(G4double val)
171{
172 safety = val;
173 UpdateErrorConstraints();
174}
175
176template <class T> void G4RKIntegrationDriver<T>::
177RenewStepperAndAdjust(G4MagIntegratorStepper* stepper)
178{
179 T* ourStepper = dynamic_cast<T*>(stepper);
180 if ( ourStepper )
181 {
182 RenewStepperAndAdjustImpl( ourStepper );
183 }
184 else
185 {
186 G4Exception("G4RKIntegrationDriver::RenewStepperAndAdjust()",
187 "GeomField0002", FatalException,
188 "The type of the stepper provided is incorrect for this templated driver");
189 }
190}
191
192template <class T>
193void G4RKIntegrationDriver<T>::RenewStepperAndAdjustImpl(T* stepper)
194{
195 pIntStepper = stepper;
196 ReSetParameters();
197}
