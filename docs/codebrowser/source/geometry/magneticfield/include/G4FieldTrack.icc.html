60inline G4FieldTrack&
61G4FieldTrack::SetCurvePnt(const G4ThreeVector& pPosition,
62 const G4ThreeVector& pMomentum,
63 G4double s_curve )
64{
65 SixVector[0] = pPosition.x();
66 SixVector[1] = pPosition.y();
67 SixVector[2] = pPosition.z();
68
69 SixVector[3] = pMomentum.x();
70 SixVector[4] = pMomentum.y();
71 SixVector[5] = pMomentum.z();
72
73 fMomentumDir = (pMomentum.mag2() > 0.0) ?
74 pMomentum.unit() : G4ThreeVector( 0.0, 0.0, 0.0 );
75
76 fDistanceAlongCurve= s_curve;
77
78 return *this;
79}

184{

187 SixVector[5] = pMomentum.z();

189 if( pMomentum.mag2() > 0.0 )

236inline
237G4FieldTrack & G4FieldTrack::operator = ( const G4FieldTrack& rStVec )
238{
239 if (&rStVec == this) return *this;
240
241 SixVector[0]= rStVec.SixVector[0];
242 SixVector[1]= rStVec.SixVector[1];
243 SixVector[2]= rStVec.SixVector[2];
244 SixVector[3]= rStVec.SixVector[3];
245 SixVector[4]= rStVec.SixVector[4];
246 SixVector[5]= rStVec.SixVector[5];
247 SetCurveLength( rStVec.GetCurveLength() );
248
249 fKineticEnergy= rStVec.fKineticEnergy;
250 fRestMass_c2= rStVec.fRestMass_c2;
251 SetLabTimeOfFlight( rStVec.GetLabTimeOfFlight() );
252 SetProperTimeOfFlight( rStVec.GetProperTimeOfFlight() );
253 SetPolarization( rStVec.GetPolarization() );
254 fMomentumDir= rStVec.fMomentumDir;
255
256 fChargeState= rStVec.fChargeState;
257 // (*fpChargeState)= *(rStVec.fpChargeState);
258 // fpChargeState= rStVec.fpChargeState; // Handles!!
259 return *this;
260}

262inline void
263G4FieldTrack::UpdateFourMomentum( G4double kineticEnergy,
264 const G4ThreeVector& momentumDirection )
265{
266 G4double momentum_mag = std::sqrt(kineticEnergy*kineticEnergy
267 +2.0*fRestMass_c2*kineticEnergy);
268 G4ThreeVector momentumVector= momentum_mag * momentumDirection;
269
270 // SetMomentum( momentumVector ); // Set direction (from unit): used sqrt, div
271 SixVector[3] = momentumVector.x();
272 SixVector[4] = momentumVector.y();
273 SixVector[5] = momentumVector.z();
274
275 fMomentumDir= momentumDirection; // Set directly to avoid inaccuracy.
276 fKineticEnergy= kineticEnergy;
277}

279inline void G4FieldTrack::UpdateState( const G4ThreeVector& position,
280 G4double laboratoryTimeOfFlight,
281 const G4ThreeVector& momentumDirection,
282 G4double kineticEnergy
283 )
284{
285 // SetCurvePnt( position, momentumVector, s_curve=0.0);
286 SetPosition( position);
287 fLabTimeOfFlight= laboratoryTimeOfFlight;
288 fDistanceAlongCurve= 0.0;
289
290 UpdateFourMomentum( kineticEnergy, momentumDirection);
291}
