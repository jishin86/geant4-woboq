29// class G4RKIntegrationDriver
30//
31// Class description:
32//
33// Driver class which controls the integration error of a
34// Runge-Kutta stepper
35
36// History:
37// - Created. D.Sorokin
38// --------------------------------------------------------------------
39
40#ifndef G4RKIntegrationDriver_HH
41#define G4RKIntegrationDriver_HH
42
43#include "G4VIntegrationDriver.hh"
44
45template <class T>
46class G4RKIntegrationDriver : public G4VIntegrationDriver {
47public:
48 G4RKIntegrationDriver(T* stepper);
49
50 G4RKIntegrationDriver(const G4RKIntegrationDriver &) = delete;
51 const G4RKIntegrationDriver& operator =(const G4RKIntegrationDriver &) = delete;
52
53 virtual void GetDerivatives(const G4FieldTrack& track,
54 G4double dydx[]) const override;
55
56 virtual void GetDerivatives(const G4FieldTrack& track,
57 G4double dydx[],
58 G4double field[]) const override;
59
60 // Taking the last step's normalised error, calculate
61 // a step size for the next step.
62 // Do not limit the next step's size within a factor of the
63 // current one.
64 virtual G4double ComputeNewStepSize(G4double errMaxNorm, // normalised error
65 G4double hstepCurrent) override; // current step size
66
67 virtual G4EquationOfMotion* GetEquationOfMotion() override;
68 virtual void SetEquationOfMotion(G4EquationOfMotion* equation) override;
69
70 virtual const T* GetStepper() const override;
71 virtual T* GetStepper() override;
72
73 // Accessors.
74 G4double GetSafety() const;
75 G4double GetPshrnk() const;
76 G4double GetPgrow() const;
77
78 virtual void RenewStepperAndAdjust(G4MagIntegratorStepper* stepper) override;
79
80 // i) sets the exponents (pgrow & pshrnk),
81 // using the current Stepper's order,
82 // ii) sets the safety
83 void ReSetParameters(G4double safety = 0.9);
84 void SetSafety(G4double valS);
85
86 // Modify and Get the Maximum number of Steps that can be
87 // taken for the integration of a single segment -
88 // (ie a single call to AccurateAdvance).
89 G4int GetMaxNoSteps() const;
90 void SetMaxNoSteps(G4int val);
91
92 G4double GetSmallestFraction() const;
93 void SetSmallestFraction(G4double val);
94
95protected:
96 G4double ShrinkStepSize(G4double h, G4double error) const;
97 G4double GrowStepSize(G4double h, G4double error) const;
98
99 G4double ShrinkStepSize2(G4double h, G4double error2) const;
100 G4double GrowStepSize2(G4double h, G4double error2) const;
101
102 void UpdateErrorConstraints();
103
104private:
105 inline void RenewStepperAndAdjustImpl(T* stepper);
106
107 G4int fMaxNoSteps;
108
109 // The (default) maximum number of steps is Base
110 // divided by the order of Stepper
111 G4int fMaxStepBase;
112
113 // Parameters used to grow and shrink trial stepsize.
114 G4double safety;
115 G4double pshrnk; // exponent for shrinking
116 G4double pgrow; // exponent for growth
117
118 // muximum error values for shrinking / growing (optimisation).
119 G4double errorConstraintShrink;
120 G4double errorConstraintGrow;
121
122 T* pIntStepper;
123};
124
125#include "G4RKIntegrationDriver.icc"
126
127#endif
