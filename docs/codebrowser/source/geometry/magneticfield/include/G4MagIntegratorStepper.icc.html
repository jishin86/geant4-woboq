78inline
79void G4MagIntegratorStepper::NormaliseTangentVector( G4double vec[6] )
80{
81 G4double drds2 = vec[3]*vec[3]+vec[4]*vec[4]+vec[5]*vec[5];
82
83 if( std::fabs(drds2 - 1.0) > 1.e-14 )
84 {
85 G4double normx = 1.0 / std::sqrt(drds2);
86 for(G4int i=3;i<6;i++) { vec[i] *= normx; }
87 }
88}
89
90inline
91void G4MagIntegratorStepper::NormalisePolarizationVector( G4double vec[12] )
92{
93 G4double drds2 = vec[9]*vec[9]+vec[10]*vec[10]+vec[11]*vec[11];
94
95 if( drds2 > 0. )
96 {
97 if( std::fabs(drds2 - 1.0) > 1.e-14 )
98 {
99 G4double normx = 1.0 / std::sqrt(drds2);
100 for(G4int i=9;i<12;i++) { vec[i] *= normx; }
101 }
102 }
