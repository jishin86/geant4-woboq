57inline
58void G4VFSALIntegrationStepper::NormaliseTangentVector( G4double vec[6] )
59{
60 G4double drds2 = vec[3]*vec[3]+vec[4]*vec[4]+vec[5]*vec[5];
61
62 if( std::fabs(drds2 - 1.0) > 1.e-14 )
63 {
64 G4double normx = 1.0 / std::sqrt(drds2);
65 for(G4int i=3;i<6;i++) { vec[i] *= normx; }
66 }
67}
68
69inline
70void G4VFSALIntegrationStepper::NormalisePolarizationVector( G4double vec[12] )
71{
72 G4double drds2 = vec[9]*vec[9]+vec[10]*vec[10]+vec[11]*vec[11];
73
74 if( drds2 > 0. )
75 {
76 if( std::fabs(drds2 - 1.0) > 1.e-14 )
77 {
78 G4double normx = 1.0 / std::sqrt(drds2);
79 for(G4int i=9;i<12;i++) { vec[i] *= normx; }
80 }
81 }
