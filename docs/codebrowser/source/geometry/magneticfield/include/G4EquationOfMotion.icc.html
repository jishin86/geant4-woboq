62inline
63void G4EquationOfMotion::RightHandSide(const G4double y[], G4double dydx[]) const
64{
65 G4double Field[G4Field::MAX_NUMBER_OF_COMPONENTS];
66 G4double PositionAndTime[4];
67
68 // Position
69 PositionAndTime[0] = y[0];
70 PositionAndTime[1] = y[1];
71 PositionAndTime[2] = y[2];
72 // Global Time
73 PositionAndTime[3] = y[7]; // See G4FieldTrack::LoadFromArray
74
75 GetFieldValue(PositionAndTime, Field);
76 EvaluateRhsGivenB(y, Field, dydx);
77}
