69// A member that calculates the inverse parabolic through
70// the three points (x,y) and returns the value x that, for the
71// inverse parabolic, corresponds to y=0.
72//
73inline G4double G4ChordFinder::InvParabolic ( const G4double xa, const G4double ya,
74 const G4double xb, const G4double yb,
75 const G4double xc, const G4double yc )
76{ const G4double R = yb/yc,
77 S = yb/ya,
78 T = ya/yc;
79 const G4double Q = (T-1)*(R-1)*(S-1);
80 if (std::fabs(Q) <DBL_MIN ) return DBL_MAX;
81
82 const G4double P = S*(T*(R-T)*(xc-xb) - (1-R)*(xb-xa));
83 return xb + P/Q;
84}
