29// class G4ClassicalRK4
30//
31// Class description:
32//
33// Integrate the equations of the motion of a particle in a magnetic field
34// using the classical 4th Runge-Kutta method.
35
36// History:
37// - Created: J.Apostolakis, V.Grichine - 30.1.97
38// - Moved into G4MagErrorStepper: W.Wander <wwc@mit.edu> - 12/09/97
39// - Fix: D.Sorokin add header guard
40// -------------------------------------------------------------------
41
42#ifndef G4ClassicalRK4_HH
43#define G4ClassicalRK4_HH
44
45#include "G4MagErrorStepper.hh"
46
47class G4ClassicalRK4 : public G4MagErrorStepper
48{
49 public: // with description
50
51 G4ClassicalRK4(G4EquationOfMotion *EquationMotion, G4int numberOfVariables = 6) ;
52
53 ~G4ClassicalRK4() ;
54
55 // A stepper that does not know about errors.
56 // It is used by the MagErrorStepper stepper.
57
58 void DumbStepper( const G4double yIn[],
59 const G4double dydx[],
60 G4double h,
61 G4double yOut[] ) ;
62 // Given values for the variables y[0,..,n-1] and their derivatives
63 // dydx[0,...,n-1] known at x, use the classical 4th Runge-Kutta
64 // method to advance the solution over an interval h and return the
65 // incremented variables as yout[0,...,n-1], which not be a distinct
66 // array from y. The user supplies the routine RightHandSide(x,y,dydx),
67 // which returns derivatives dydx at x. The source is routine rk4 from
68 // NRC p. 712-713 .
69
70 public: // without description
71
72 G4int IntegratorOrder() const { return 4; }
73
74 private:
75
76 void StepWithEst( const G4double yIn[],
77 const G4double dydx[],
78 G4double h,
79 G4double yOut[],
80 G4double& alpha2,
81 G4double& beta2,
82 const G4double B1[],
83 G4double B2[] );
84 // No longer used. Obsolete.
85
86 G4ClassicalRK4(const G4ClassicalRK4&);
87 G4ClassicalRK4& operator=(const G4ClassicalRK4&);
88 // Private copy constructor and assignment operator.
89
90 private:
91
92 // G4int fNumberOfVariables ; // is set default to 6 in constructor
93
94 G4double *dydxm, *dydxt, *yt; // scratch space - not state
95};
96
97#endif
