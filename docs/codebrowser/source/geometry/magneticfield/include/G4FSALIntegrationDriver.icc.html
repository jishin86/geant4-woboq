43template <class T>
44G4FSALIntegrationDriver<T>::
45G4FSALIntegrationDriver ( G4double hminimum, T* pStepper,
46 G4int numComponents, G4int statisticsVerbose )
47 : Base(pStepper),
48 fMinimumStep(hminimum),
49 fSmallestFraction(1e-12),
50 fVerboseLevel(statisticsVerbose),
51 fNoQuickAvanceCalls(0),
52 fNoAccurateAdvanceCalls(0),
53 fNoAccurateAdvanceBadSteps(0),
54 fNoAccurateAdvanceGoodSteps(0)
55{
56 if (numComponents != Base::GetStepper()->GetNumberOfVariables())
57 {
58 std::ostringstream message;
59 message << "Driver's number of integrated components "
60 << numComponents
61 << " != Stepper's number of components "
62 << pStepper->GetNumberOfVariables();
63 G4Exception("G4FSALIntegrationDriver","GeomField0002",
64 FatalException, message);
65 }
66}

81// Runge-Kutta driver with adaptive stepsize control. Integrate starting
82// values at y_current over hstep x2 with accuracy eps.
83// On output ystart is replaced by values at the end of the integration
84// interval. RightHandSide is the right-hand side of ODE system.
85// The source is similar to odeint routine from NRC p.721-722 .
86template <class T>
87G4bool G4FSALIntegrationDriver<T>::
88AccurateAdvance( G4FieldTrack& track, G4double hstep,
89 G4double eps, G4double hinitial )
90{
91 ++fNoAccurateAdvanceCalls;
92
93 if (hstep < GetMinimumStep())
94 {
95 G4double dchord_step = 0, dyerr = 0;
96 G4double dydx[G4FieldTrack::ncompSVEC];
97 Base::GetDerivatives(track, dydx);
98 return QuickAdvance(track, dydx, hstep, Base::UNKNOWN_CURVATURE_RADIUS, dchord_step, dyerr);
99 }
100
101 G4bool succeeded = false;
102
103 G4double hnext, hdid;
104
105 G4double y[G4FieldTrack::ncompSVEC], dydx[G4FieldTrack::ncompSVEC];
106
107 track.DumpToArray(y);
108
109 // hstep somtimes is too small. No need to add large curveLength.
110 G4double curveLength = 0;
111 G4double endCurveLength = hstep;
112
113
114 G4double h = hstep;
115 if (hinitial > perMillion * hstep && hinitial < hstep)
116 {
117 h = hinitial;
118 }
119
120 Base::GetStepper()->RightHandSide(y, dydx);
121
122 for (G4int iter = 0; iter < Base::GetMaxNoSteps(); ++iter)
123 {
124 const G4ThreeVector StartPos =
125 field_utils::makeVector(y, field_utils::Value3D::Position);
126
127 OneGoodStep(y, dydx, curveLength, h, eps, hdid, hnext);
128
129 const G4ThreeVector EndPos =
130 field_utils::makeVector(y, field_utils::Value3D::Position);
131
132 CheckStep(EndPos, StartPos, hdid);
133
134 G4double restCurveLength = endCurveLength - curveLength;
135 if (restCurveLength < GetSmallestFraction() * hstep)
136 {
137 succeeded = true;
138 break;
139 }
140 h = std::min(hnext, restCurveLength);
141 }
142
143
144 if (succeeded)
145 {
146 track.LoadFromArray(y, Base::GetStepper()->GetNumberOfVariables());
147 track.SetCurveLength(track.GetCurveLength() + curveLength);
148 }
149
150 return succeeded;
151}

153// Driver for one Runge-Kutta Step with monitoring of local truncation error
154// to ensure accuracy and adjust stepsize. Input are dependent variable
155// array y[0,...,5] and its derivative dydx[0,...,5] at the
156// starting value of the independent variable x . Also input are stepsize
157// to be attempted htry, and the required accuracy eps. On output y and x
158// are replaced by their new values, hdid is the stepsize that was actually
159// accomplished, and hnext is the estimated next stepsize.
160// This is similar to the function rkqs from the book:
161// Numerical Recipes in C: The Art of Scientific Computing (NRC), Second
162// Edition, by William H. Press, Saul A. Teukolsky, William T.
163// Vetterling, and Brian P. Flannery (Cambridge University Press 1992),
164// 16.2 Adaptive StepSize Control for Runge-Kutta, p. 719
165//
166template <class T>
167void G4FSALIntegrationDriver<T>::
168OneGoodStep(G4double y[],
169 G4double dydx[],
170 G4double& curveLength, // InOut
171 G4double htry,
172 G4double eps_rel_max,
173 G4double& hdid, // Out
174 G4double& hnext) // Out
175{
176 G4double error2 = DBL_MAX;
177
178 G4double yError[G4FieldTrack::ncompSVEC],
179 yOut[G4FieldTrack::ncompSVEC],
180 dydxOut[G4FieldTrack::ncompSVEC];
181
182 // Set stepsize to the initial trial value
183 G4double hstep = htry;
184
185 static G4ThreadLocal G4int tot_no_trials = 0;
186 const G4int max_trials = 100;
187
188 for (G4int iter = 0; iter < max_trials; ++iter)
189 {
190 ++tot_no_trials;
191
192 Base::GetStepper()->Stepper(y, dydx, hstep, yOut, yError, dydxOut);
193 error2 = field_utils::relativeError2(y, yError, hstep, eps_rel_max);
194
195 // Step succeeded.
196 if (error2 <= 1) break;
197
198 hstep = Base::ShrinkStepSize2(hstep, error2);
199 }
200
201 hnext = Base::GrowStepSize2(hstep, error2);
202 curveLength += (hdid = hstep);
203
204 for(G4int k = 0; k < Base::GetStepper()->GetNumberOfVariables(); ++k)
205 {
206 y[k] = yOut[k];
207 dydx[k] = dydxOut[k];
208 }
209}

211template <class T>
212G4bool G4FSALIntegrationDriver<T>::
213QuickAdvance( G4FieldTrack& fieldTrack, const G4double dydxIn[],
214 G4double hstep, G4double /*inverseCurvatureRadius*/,
215 G4double& dchord_step, G4double& dyerr )
216{
217 ++fNoQuickAvanceCalls;
218
219 if (hstep == 0)
220 {
221 std::ostringstream message;
222 message << "Proposed step is zero; hstep = " << hstep << " !";
223 G4Exception("G4FSALIntegrationDriver ::QuickAdvance()",
224 "GeomField1001", JustWarning, message);
225 return true;
226 }
227 if (hstep < 0)
228 {
229 std::ostringstream message;
230 message << "Invalid run condition." << G4endl
231 << "Proposed step is negative; hstep = "
232 << hstep << "." << G4endl
233 << "Requested step cannot be negative! Aborting event.";
234 G4Exception("G4FSALIntegrationDriver ::QuickAdvance()",
235 "GeomField0003", EventMustBeAborted, message);
236 return false;
237 }
238
239 G4double yError[G4FieldTrack::ncompSVEC],
240 yIn[G4FieldTrack::ncompSVEC],
241 yOut[G4FieldTrack::ncompSVEC],
242 dydxOut[G4FieldTrack::ncompSVEC];
243
244 fieldTrack.DumpToArray(yIn);
245
246 Base::GetStepper()->Stepper(yIn, dydxIn, hstep, yOut, yError, dydxOut);
247 dchord_step = Base::GetStepper()->DistChord();
248
249 fieldTrack.LoadFromArray(yOut, Base::GetStepper()->GetNumberOfVariables());
250 fieldTrack.SetCurveLength(fieldTrack.GetCurveLength() + hstep);
251
252 dyerr = field_utils::absoluteError(yOut, yError, hstep);
253
254 return true;
255}

257template <class T>
258void G4FSALIntegrationDriver<T>::SetSmallestFraction(G4double newFraction)
259{
260 if( newFraction > 1.e-16 && newFraction < 1e-8 )
261 {
262 fSmallestFraction = newFraction;
263 }
264 else
265 {
266 std::ostringstream message;
267 message << "Smallest Fraction not changed. " << G4endl
268 << " Proposed value was " << newFraction << G4endl
269 << " Value must be between 1.e-8 and 1.e-16";
270 G4Exception("G4FSALIntegrationDriver::SetSmallestFraction()",
271 "GeomField1001", JustWarning, message);
272 }
273}

275template <class T>
276void G4FSALIntegrationDriver<T>::CheckStep(
277 const G4ThreeVector& posIn, const G4ThreeVector& posOut, G4double hdid)
278{
279 const G4double endPointDist = (posOut - posIn).mag();
280 if (endPointDist >= hdid * (1. + perMillion))
281 {
282 ++fNoAccurateAdvanceBadSteps;
283#ifdef G4DEBUG_FIELD
284 // Issue a warning only for gross differences -
285 // we understand how small difference occur.
286 if (endPointDist >= hdid * (1. + perThousand))
287 {
288 G4Exception("G4FSALIntegrationDriver::CheckStep()",
289 "GeomField1002", JustWarning,
290 "endPointDist >= hdid!");
291 }
292#endif
293 }
294 else
295 {
296 ++fNoAccurateAdvanceGoodSteps;
297 }
298}
