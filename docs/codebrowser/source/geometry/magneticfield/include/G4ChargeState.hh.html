38#ifndef G4ChargeState_HH
39#define G4ChargeState_HH
40
41#include "globals.hh"
42
43class G4ChargeState // Charge & moments
44{
45 public: // without description
46
47 inline G4ChargeState(G4double charge,
48 G4double magnetic_dipole_moment,
49 G4double pdgSpin,
50 G4double electric_dipole_moment = 0.0,
51 G4double magnetic_charge = 0.0);
52
53 inline G4ChargeState( const G4ChargeState& right );
54 inline G4ChargeState& operator = ( const G4ChargeState& right );
55
56 void SetChargeSpinMoments(G4double charge,
57 G4double pdgSpin,
58 G4double magnetic_dipole_moment= DBL_MAX,
59 G4double electric_dipole_moment= DBL_MAX,
60 G4double magnetic_charge= DBL_MAX );
61 // Revise the charge, pdgSpin, and optionally both moments and magnetic charge
62
63 void SetCharge(G4double charge){ fCharge = charge; }
64 G4double GetCharge() const { return fCharge; }
65 // Revise the charge (in units of the positron charge)
66
67
68 // Basic Get / Set methods
69 void SetPDGSpin(G4double spin){ fSpin = spin; }
70 G4double GetPDGSpin() const { return fSpin; }
71
72 void SetMagneticDipoleMoment(G4double moment){ fMagn_dipole = moment; }
73 G4double GetMagneticDipoleMoment() const { return fMagn_dipole; }
74
75 void SetElectricDipoleMoment(G4double moment){ fElec_dipole = moment; }
76 G4double ElectricDipoleMoment() const { return fElec_dipole; }
77
78 void SetMagneticCharge(G4double charge){ fMagneticCharge=charge; }
79 G4double MagneticCharge() const { return fMagneticCharge; }
80
81 // Auxiliary methods to set several properties at once
82
83 inline void SetChargeMdm(G4double charge, G4double mag_dipole_moment);
84 // SetCharge and Magnetic Dipole Moment
85
86 inline void SetChargeMdmSpin(G4double charge,
87 G4double magnetic_dipole_moment,
88 G4double pdgSpin);
89
90 inline void SetChargeSpin(G4double charge,
91 G4double pdgSpin);
92
93 // Revise the charge, spin and all both moments
94 inline void SetChargeDipoleMoments(G4double charge,
95 G4double magnetic_dipole_moment,
96 G4double electric_dipole_moment);
97
98 inline void SetChargesAndMoments(G4double charge,
99 G4double magnetic_dipole_moment,
100 G4double electric_dipole_moment,
101 G4double magnetic_charge );
102
103
104
105
106 public: // Obsolete
107 inline void SetSpin(G4double spin){ SetPDGSpin( spin); }
108 inline G4double GetSpin() const { return GetPDGSpin(); }
109
110 private:
111
112 G4double fCharge;
113 G4double fSpin;
114 G4double fMagn_dipole;
115 G4double fElec_dipole;
116 G4double fMagneticCharge; // for magnetic monopole
117};
118
119// Inline methods implementation
120
121inline G4ChargeState::G4ChargeState(G4double charge,
122 G4double magnetic_dipole_moment,
123 G4double spin,
124 G4double electric_dipole_moment,
125 G4double magnetic_charge)
126{
127 fCharge = charge;
128 fSpin = spin;
129 fMagn_dipole = magnetic_dipole_moment;
130 fElec_dipole = electric_dipole_moment;
131 fMagneticCharge = magnetic_charge;
132}
133
134inline G4ChargeState::G4ChargeState( const G4ChargeState& right )
135{
136 fCharge = right.fCharge;
137 fSpin = right.fSpin;
138 fMagn_dipole = right.fMagn_dipole;
139 fElec_dipole = right.fElec_dipole;
140 fMagneticCharge = right.fMagneticCharge;
141}
142
143inline G4ChargeState& G4ChargeState::operator = ( const G4ChargeState& right )
144{
145 if (&right == this) return *this;
146
147 fCharge = right.fCharge;
148 fSpin = right.fSpin;
149 fMagn_dipole = right.fMagn_dipole;
150 fElec_dipole = right.fElec_dipole;
151 fMagneticCharge = right.fMagneticCharge;
152
153 return *this;
154}
155
156inline void G4ChargeState::SetChargeMdm(G4double charge, G4double mag_dipole_moment)
157{
158 SetCharge( charge );
159 SetMagneticDipoleMoment( mag_dipole_moment);
160}
161
162inline void G4ChargeState::SetChargeMdmSpin(G4double charge,
163 G4double magDipoleMoment,
164 G4double pdgSpin)
165{
166 SetChargeMdm( charge, magDipoleMoment );
167 SetPDGSpin( pdgSpin );
168}
169
170inline void G4ChargeState::SetChargeSpin(G4double charge,
171 G4double pdgSpin)
172{
173 SetCharge( charge );
174 SetPDGSpin( pdgSpin );
175}
176
177inline void
178G4ChargeState::SetChargeDipoleMoments(G4double charge,
179 G4double magneticDM,
180 G4double electricDM)
181{
182 SetChargeMdm( charge, magneticDM );
183 SetElectricDipoleMoment( electricDM );
184}
185
186inline void
187G4ChargeState::SetChargesAndMoments(G4double charge,
188 G4double magneticDM,
189 G4double electricDM,
190 G4double magnetic_charge )
191{
192 SetChargeDipoleMoments( charge, magneticDM, electricDM);
193 SetMagneticCharge( magnetic_charge );
194}
195#endif /* End of ifndef G4ChargeState_HH */
