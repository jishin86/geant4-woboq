41#ifndef G4RKG3_Stepper_hh
42#define G4RKG3_Stepper_hh
43
44#include "G4Types.hh"
45#include "G4MagIntegratorStepper.hh"
46#include "G4ThreeVector.hh"
47
48class G4Mag_EqRhs;
49
50class G4RKG3_Stepper : public G4MagIntegratorStepper
51{
52 public: // with description
53
54 G4RKG3_Stepper(G4Mag_EqRhs *EqRhs);
55 // Integrate over 6 variables only: position & velocity.
56 // Not implemented yet !
57
58 ~G4RKG3_Stepper();
59
60 void Stepper( const G4double yIn[],
61 const G4double dydx[],
62 G4double h,
63 G4double yOut[],
64 G4double yErr[] );
65 // The method which must be provided, even if less efficient.
66
67 G4double DistChord() const ;
68
69 void StepNoErr( const G4double tIn[8],
70 const G4double dydx[6],
71 G4double Step,
72 G4double tOut[8],
73 G4double B[3] );
74 // Integrator RK Stepper from G3 with only two field evaluation per
75 // Step. It is used in propagation initial Step by small substeps
76 // after solution error and delta geometry considerations.
77 // B[3] is magnetic field which is passed from substep to substep.
78
79 void StepWithEst( const G4double tIn[8],
80 const G4double dydx[6],
81 G4double Step,
82 G4double tOut[8],
83 G4double& alpha2,
84 G4double& beta2,
85 const G4double B1[3],
86 G4double B2[3] );
87 // Integrator for RK from G3 with evaluation of error in solution and delta
88 // geometry based on naive similarity with the case of uniform magnetic field.
89 // B1[3] is input and is the first magnetic field values
90 // B2[3] is output and is the final magnetic field values.
91
92 public: // without description
93
94 G4int IntegratorOrder() const { return 4; }
95
96 private:
97
98 G4ThreeVector fyInitial,
99 fyMidPoint,
100 fyFinal;
101 G4ThreeVector fpInitial;
102 G4ThreeVector BfldIn;
103 G4double hStep;
104};
