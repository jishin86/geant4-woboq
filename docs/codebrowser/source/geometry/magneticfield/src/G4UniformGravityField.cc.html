39#include "G4UniformGravityField.hh"
40#include "G4PhysicalConstants.hh"
41#include "G4SystemOfUnits.hh"
42
43// Construct from a 3-vector
44G4UniformGravityField::G4UniformGravityField(const G4ThreeVector FieldVector)
45 : G4Field ( true ) // Gravity flag *on*
46{
47 fFieldComponents[0] = FieldVector.x();
48 fFieldComponents[1] = FieldVector.y();
49 fFieldComponents[2] = FieldVector.z();
50}
51
52// Construct from a double > default = -9.81 m*s^-2
53G4UniformGravityField::G4UniformGravityField(const G4double gy )
54 : G4Field ( true )
55{
56 fFieldComponents[0] = 0.0;
57 fFieldComponents[1] = gy;
58 fFieldComponents[2] = 0.0;
59}
60
61G4Field* G4UniformGravityField::Clone() const
62{
63 return new G4UniformGravityField( G4ThreeVector(fFieldComponents[0],
64 fFieldComponents[1],
65 fFieldComponents[2]) );
66}
67
68G4UniformGravityField::~G4UniformGravityField()
69{
70}
71
72G4UniformGravityField::G4UniformGravityField (const G4UniformGravityField &p)
73 : G4Field(p)
74{
75 for (G4int i=0; i<3; i++)
76 {
77 fFieldComponents[i] = p.fFieldComponents[i];
78 }
79}
80
81G4UniformGravityField&
82G4UniformGravityField::operator = (const G4UniformGravityField &p)
83{
84 if (&p == this) return *this;
85 G4Field::operator=(p);
86 for (G4int i=0; i<3; i++)
87 {
88 fFieldComponents[i] = p.fFieldComponents[i];
89 }
90 return *this;
91}
92
93// -------------------------------------------------------------------
94
95void G4UniformGravityField::GetFieldValue (const G4double [4],
96 G4double *G ) const
97{
98 G[0]= fFieldComponents[0] ;
99 G[1]= fFieldComponents[1] ;
100 G[2]= fFieldComponents[2] ;
101}
