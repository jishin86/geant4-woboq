33G4MagErrorStepper::~G4MagErrorStepper()
34{
35 delete[] yMiddle;
36 delete[] dydxMid;
37 delete[] yInitial;
38 delete[] yOneStep;
39}
40
41void
42G4MagErrorStepper::Stepper( const G4double yInput[],
43 const G4double dydx[],
44 G4double hstep,
45 G4double yOutput[],
46 G4double yError [] )
47{
48 const G4int nvar = this->GetNumberOfVariables() ;
49 const G4int maxvar= GetNumberOfStateVariables();
50
51 G4int i;
52 // correction for Richardson Extrapolation.
53 G4double correction = 1. / ( (1 << IntegratorOrder()) -1 );
54
55 // Saving yInput because yInput and yOutput can be aliases for same array
56
57 for(i=0;i<nvar;i++) yInitial[i]=yInput[i];
58 yInitial[7]= yInput[7]; // Copy the time in case ... even if not really needed
59 yMiddle[7] = yInput[7]; // Copy the time from initial value
60 yOneStep[7] = yInput[7]; // As it contributes to final value of yOutput ?
61 // yOutput[7] = yInput[7]; // -> dumb stepper does it too for RK4
62 for(i=nvar;i<maxvar;i++) yOutput[i]=yInput[i];
63 // yError[7] = 0.0;
64
65 G4double halfStep = hstep * 0.5;
66
67 // Do two half steps
68
69 DumbStepper (yInitial, dydx, halfStep, yMiddle);
70 RightHandSide(yMiddle, dydxMid);
71 DumbStepper (yMiddle, dydxMid, halfStep, yOutput);
72
73 // Store midpoint, chord calculation
74
75 fMidPoint = G4ThreeVector( yMiddle[0], yMiddle[1], yMiddle[2]);
76
77 // Do a full Step
78 DumbStepper(yInitial, dydx, hstep, yOneStep);
79 for(i=0;i<nvar;i++) {
80 yError [i] = yOutput[i] - yOneStep[i] ;
81 yOutput[i] += yError[i]*correction ; // Provides accuracy increased
82 // by 1 order via the
83 // Richardson Extrapolation
84 }
85
86 fInitialPoint = G4ThreeVector( yInitial[0], yInitial[1], yInitial[2]);
87 fFinalPoint = G4ThreeVector( yOutput[0], yOutput[1], yOutput[2]);
88
89 return ;
90}

94G4double
95G4MagErrorStepper::DistChord() const
96{
97 // Estimate the maximum distance from the curve to the chord
98 //
99 // We estimate this using the distance of the midpoint to
100 // chord (the line between
101 //
102 // Method below is good only for angle deviations < 2 pi,
103 // This restriction should not a problem for the Runge cutta methods,
104 // which generally cannot integrate accurately for large angle deviations.
105 G4double distLine, distChord;
106
107 if (fInitialPoint != fFinalPoint) {
108 distLine= G4LineSection::Distline( fMidPoint, fInitialPoint, fFinalPoint );
109 // This is a class method that gives distance of Mid
110 // from the Chord between the Initial and Final points.
111
112 distChord = distLine;
113 }else{
114 distChord = (fMidPoint-fInitialPoint).mag();
115 }
116
117 return distChord;
118}
