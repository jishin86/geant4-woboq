39G4UniformElectricField::G4UniformElectricField(const G4ThreeVector FieldVector )
40{
41 fFieldComponents[0] = 0.0;
42 fFieldComponents[1] = 0.0;
43 fFieldComponents[2] = 0.0;
44 fFieldComponents[3] = FieldVector.x();
45 fFieldComponents[4] = FieldVector.y();
46 fFieldComponents[5] = FieldVector.z();
47}
48
49G4UniformElectricField::G4UniformElectricField(G4double vField,
50 G4double vTheta,
51 G4double vPhi )
52{
53 if ( (vField<0) || (vTheta<0) || (vTheta>pi) || (vPhi<0) || (vPhi>twopi) )
54 {
55 G4Exception("G4UniformElectricField::G4UniformElectricField()",
56 "GeomField0002", FatalException, "Invalid parameters.");
57 }
58
59 fFieldComponents[0] = 0.0;
60 fFieldComponents[1] = 0.0;
61 fFieldComponents[2] = 0.0;
62 fFieldComponents[3] = vField*std::sin(vTheta)*std::cos(vPhi) ;
63 fFieldComponents[4] = vField*std::sin(vTheta)*std::sin(vPhi) ;
64 fFieldComponents[5] = vField*std::cos(vTheta) ;
65}
66
67G4Field* G4UniformElectricField::Clone() const
68{
69 return new G4UniformElectricField( G4ThreeVector(fFieldComponents[3],
70 fFieldComponents[4],
71 fFieldComponents[5]) );
72}
73
74G4UniformElectricField::~G4UniformElectricField()
75{
76}
77
78G4UniformElectricField::G4UniformElectricField (const G4UniformElectricField &p)
79 : G4ElectricField(p)
80{
81 for (G4int i=0; i<6; i++)
82 {
83 fFieldComponents[i] = p.fFieldComponents[i];
84 }
85}
86
87G4UniformElectricField&
88G4UniformElectricField::operator = (const G4UniformElectricField &p)
89{
90 if (&p == this) return *this;
91 G4ElectricField::operator=(p);
92 for (G4int i=0; i<6; i++)
93 {
94 fFieldComponents[i] = p.fFieldComponents[i];
95 }
96 return *this;
97}

101void G4UniformElectricField::GetFieldValue (const G4double[4],
102 G4double *fieldBandE ) const
103{
104 fieldBandE[0]= 0.0;
105 fieldBandE[1]= 0.0;
106 fieldBandE[2]= 0.0;
107 fieldBandE[3]= fFieldComponents[3] ;
108 fieldBandE[4]= fFieldComponents[4] ;
109 fieldBandE[5]= fFieldComponents[5] ;
110}
