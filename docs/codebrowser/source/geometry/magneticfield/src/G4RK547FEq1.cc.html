26// The Butcher table of the Higham & Hall 5(4)7 method is:
27//
28// 0 |
29// 2/9 | 2/9
30// 1/3 | 1/12 1/4
31// 1/2 | 1/8 0 3/8
32// 3/5 | 91/500 -27/100 78/125 8/125
33// 1 | -11/20 27/20 12/5 -36/5 5
34// 1 | 1/12 0 27/32 -4/3 125/96 5/48
35//----------------------------------------------------------------------------
36// 1/12 0 27/32 -4/3 125/96 5/48 0
37// 2/15 0 27/80 -2/15 25/48 1/24 1/10

51void G4RK547FEq1::makeStep(
52 const G4double yInput[],
53 const G4double dydx[],
54 const G4double hstep,
55 G4double yOutput[],
56 G4double* dydxOutput,
57 G4double* yError) const
58{
59 G4double yTemp[G4FieldTrack::ncompSVEC];
60 for (int i = GetNumberOfVariables(); i < GetNumberOfStateVariables(); ++i){
61 yOutput[i] = yTemp[i] = yInput[i];
62 }
63
64 G4double ak2[G4FieldTrack::ncompSVEC],
65 ak3[G4FieldTrack::ncompSVEC],
66 ak4[G4FieldTrack::ncompSVEC],
67 ak5[G4FieldTrack::ncompSVEC],
68 ak6[G4FieldTrack::ncompSVEC];
69
70 const G4double
71 b21 = 2./9.,
72 b31 = 1./12., b32 = 1./4.,
73 b41 = 1./8., b42 = 0., b43 = 3./8.,
74 b51 = 91./500., b52 = -27./100., b53 = 78./125., b54 = 8./125.,
75 b61 = -11./20., b62 = 27./20., b63 = 12./5.,
76 b64 = -36./5., b65 = 5.,
77 b71 = 1./12., b72 = 0., b73 = 27./32.,
78 b74 = -4./3., b75 = 125./96., b76 = 5./48.;
79
80 const G4double
81 dc1 = b71 - 2./15.,
82 dc2 = b72 - 0.,
83 dc3 = b73 - 27./80.,
84 dc4 = b74 + 2./15.,
85 dc5 = b75 - 25./48.,
86 dc6 = b76 - 1./24.,
87 dc7 = 0. - 1./10.;
88
89 //RightHandSide(yInput, dydx);
90 for(int i = 0; i < GetNumberOfVariables(); ++i)
91 yTemp[i] = yInput[i] + hstep * b21 * dydx[i];
92
93 RightHandSide(yTemp, ak2);
94 for(int i = 0; i < GetNumberOfVariables(); ++i)
95 yTemp[i] = yInput[i] + hstep * (b31 * dydx[i] + b32 * ak2[i]);
96
97 RightHandSide(yTemp, ak3);
98 for(int i = 0;i < GetNumberOfVariables(); ++i)
99 yTemp[i] = yInput[i] + hstep * (b41 * dydx[i] + b42 * ak2[i] +
100 b43 * ak3[i]);
101
102 RightHandSide(yTemp, ak4);
103 for(int i = 0; i < GetNumberOfVariables(); ++i)
104 yTemp[i] = yInput[i] + hstep * (b51 * dydx[i] + b52 * ak2[i] +
105 b53 * ak3[i] + b54 * ak4[i]);
106
107 RightHandSide(yTemp, ak5);
108 for(int i = 0; i < GetNumberOfVariables(); ++i)
109 yTemp[i] = yInput[i] + hstep * (b61 * dydx[i] + b62 * ak2[i] +
110 b63 * ak3[i] + b64 * ak4[i] +
111 b65 * ak5[i]);
112
113 RightHandSide(yTemp, ak6);
114 for(int i = 0; i < GetNumberOfVariables(); ++i)
115 yOutput[i] = yInput[i] + hstep * (b71 * dydx[i] + b72 * ak2[i] +
116 b73 * ak3[i] + b74 * ak4[i] +
117 b75 * ak5[i] + b76 * ak6[i]);
118
119 if (dydxOutput && yError) {
120 RightHandSide(yOutput, dydxOutput);
121 for(int i = 0; i < GetNumberOfVariables(); ++i)
122 yError[i] = hstep * (dc1 * dydx[i] + dc2 * ak2[i] + dc3 * ak3[i] +
123 dc4 * ak4[i] + dc5 * ak5[i] + dc6 * ak6[i] +
124 dc7 * dydxOutput[i]);
125 }
126}

128void G4RK547FEq1::Stepper(
129 const G4double yInput[],
130 const G4double dydx[],
131 G4double hstep,
132 G4double yOutput[],
133 G4double yError[])
134{
135 copy(fyIn, yInput);
136 copy(fdydx, dydx);
137 fhstep = hstep;
138
139 makeStep(fyIn, fdydx, fhstep, fyOut, fdydxOut, yError);
140
141 copy(yOutput, fyOut);
142}

144void G4RK547FEq1::Stepper(
145 const G4double yInput[],
146 const G4double dydx[],
147 G4double hstep,
148 G4double yOutput[],
149 G4double yError[],
150 G4double dydxOutput[])
151{
152 copy(fyIn, yInput);
153 copy(fdydx, dydx);
154 fhstep = hstep;
155
156 makeStep(fyIn,fdydx, fhstep, fyOut, fdydxOut, yError);
157
158 copy(yOutput, fyOut);
159 copy(dydxOutput, fdydxOut);
160}

162G4double G4RK547FEq1::DistChord() const
163{
164 G4double yMid[G4FieldTrack::ncompSVEC];
165 makeStep(fyIn, fdydx, fhstep / 2., yMid);
166
167 const G4ThreeVector begin = makeVector(fyIn, Value3D::Position);
168 const G4ThreeVector mid = makeVector(yMid, Value3D::Position);
169 const G4ThreeVector end = makeVector(fyOut, Value3D::Position);
170
171 return G4LineSection::Distline(mid, begin, end);
172}
