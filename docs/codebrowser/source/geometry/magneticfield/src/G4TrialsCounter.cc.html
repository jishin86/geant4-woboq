34#include "G4TrialsCounter.hh"
35#include "G4ios.hh"
36
37G4TrialsCounter::G4TrialsCounter( const G4String& nameStats,
38 const G4String& description,
39 G4bool printOnExit )
40 : fName(nameStats), fDescription(description),
41 fStatsVerbose(printOnExit), fPrinted(false)
42{
43 ClearCounts();
44}
45
46G4TrialsCounter::~G4TrialsCounter()
47{
48 if( (fStatsVerbose) && (!fPrinted) ) { PrintStatistics(); }
49}
50
51void
52G4TrialsCounter::PrintStatistics()
53{
54 // Print Statistics
55 G4cout << "G4TrialsCounter::PrintStatistics()" << G4endl
56 << "Report of counts for " << fDescription << " : " << G4endl;
57 G4cout << "Stats for '" << fName << "' > "
58 << " No-trials= " << fTotalNoTrials
59 << " No-calls= " << fNumberCalls
60 << " Max-trial= " << fmaxTrials
61 << " no-max= " << fNoTimesMaxTrials
62 << G4endl;
63 fPrinted= true;
64}
65
66void G4TrialsCounter::ClearCounts()
67{
68 fTotalNoTrials= 0;
69 fNumberCalls = 0;
70 fmaxTrials = 0; // Maximum --> so only unsigned ints expected
71 fNoTimesMaxTrials=0;
72}
73
74G4int
75G4TrialsCounter::ReturnTotals( G4int& calls, G4int& maxTrials, G4int& numMaxT )
76{
77 calls = fNumberCalls;
78 maxTrials= fmaxTrials;
79 numMaxT = fNoTimesMaxTrials;
80
81 return fTotalNoTrials;
82}
