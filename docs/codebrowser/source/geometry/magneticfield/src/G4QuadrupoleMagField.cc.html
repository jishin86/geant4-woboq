31#include "G4RotationMatrix.hh"

33static G4RotationMatrix IdentityMatrix;
34
35G4QuadrupoleMagField::G4QuadrupoleMagField(G4double pGradient)
36{
37 fGradient = pGradient ;
38 fOrigin = G4ThreeVector( 0.0, 0.0, 0.0) ;
39 fpMatrix = &IdentityMatrix;
40}
41

43/////////////////////////////////////////////////////////////////////////
44
45G4QuadrupoleMagField::G4QuadrupoleMagField(G4double pGradient,
46 G4ThreeVector pOrigin,
47 G4RotationMatrix* pMatrix)
48{
49 fGradient = pGradient ;
50 fOrigin = pOrigin ;
51 fpMatrix = pMatrix ;
52}
53
54G4Field* G4QuadrupoleMagField::Clone() const
55{
56 return new G4QuadrupoleMagField(fGradient, fOrigin, fpMatrix);
57}

63}

65void G4QuadrupoleMagField::GetFieldValue( const G4double y[7],
66 G4double B[3] ) const
67// with displaced origin and rotation
68{
69 G4ThreeVector r_global = G4ThreeVector(
70 y[0] - fOrigin.x(),
71 y[1] - fOrigin.y(),
72 y[2] - fOrigin.z());
73
74 const G4ThreeVector r_local = (*fpMatrix) * r_global;
75 const G4ThreeVector B_local( fGradient * r_local.y(),fGradient * r_local.x(),0);
76 const G4ThreeVector B_global = fpMatrix->inverse() * B_local;
77
78 B[0] = B_global.x() ;
79 B[1] = B_global.y() ;
80 B[2] = B_global.z() ;
81}
