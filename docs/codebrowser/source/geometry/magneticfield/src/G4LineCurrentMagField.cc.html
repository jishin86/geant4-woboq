50void G4LineCurrentMagField::GetFieldValue( const G4double yTrack[7],
51 G4double B[3] ) const
52{
53 // G4double fFieldConstant = 100 ;
54 G4double a = 1.00 ; // mm
55 G4double x = a*yTrack[0], y = a*yTrack[1] ;
56 G4double x2 = x*x, y2 = y*y, r2 = x2 + y2 ;
57 G4double r = std::sqrt(r2+a*a) ;
58 G4double Br = fFieldConstant/r;
59 B[0] = -Br*y/r ;
60 B[1] = Br*x/r ;
61 B[2] = 0 ;
62}
