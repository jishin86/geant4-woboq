30#include "G4CachedMagneticField.hh"

32G4CachedMagneticField::G4CachedMagneticField(G4MagneticField *pMagField,
33 G4double distance)
34 : G4MagneticField(),
35 fLastLocation(DBL_MAX,DBL_MAX,DBL_MAX),
36 fLastValue(DBL_MAX,DBL_MAX,DBL_MAX),
37 fCountCalls(0), fCountEvaluations(0)
38{
39 fpMagneticField= pMagField;
40 fDistanceConst= distance;
41
42 // G4cout << " Cached-B-Field constructor> Distance = " << distance << G4endl;
43 this->ClearCounts();
44}
45
46G4Field* G4CachedMagneticField::Clone() const
47{
48 // Cannot use copy constructor: I need to clone the associated magnetic field
49 G4MagneticField* aF = static_cast<G4MagneticField*>(fpMagneticField->Clone());
50 G4CachedMagneticField* cloned = new G4CachedMagneticField(aF, fDistanceConst);
51
52 cloned->fLastLocation = fLastLocation;
53 cloned->fLastValue = fLastValue;
54 return cloned;
55}

57G4CachedMagneticField::~G4CachedMagneticField()

59}

61void
62G4CachedMagneticField::ReportStatistics()
63{
64 G4cout << " Cached field: " << G4endl
65 << " Number of calls: " << fCountCalls << G4endl
66 << " Number of evaluations : " << fCountEvaluations << G4endl;
67}

69G4CachedMagneticField::G4CachedMagneticField(const G4CachedMagneticField &rightCMF)
70 : G4MagneticField(rightCMF)
71{
72 fpMagneticField= rightCMF.fpMagneticField; // NOTE: sharing pointer here!
73 fDistanceConst = rightCMF.fDistanceConst;
74 fLastLocation = rightCMF.fLastLocation;
75 fLastValue = rightCMF.fLastValue;
76 this->ClearCounts();
77}
78
79G4CachedMagneticField& G4CachedMagneticField::operator = (const G4CachedMagneticField &p)
80{
81 if (&p == this) return *this;
82 G4MagneticField::operator=(p);
83 fpMagneticField= p.fpMagneticField; // NOTE: sharing pointer here!
84 fDistanceConst = p.fDistanceConst;
85 fLastLocation = p.fLastLocation;
86 fLastValue = p.fLastValue;
87 this->ClearCounts();
88 return *this;
89}

91void
92G4CachedMagneticField::GetFieldValue( const G4double Point[4],
93 G4double *Bfield ) const
94{
95 G4ThreeVector newLocation( Point[0], Point[1], Point[2] );
96
97 // G4cout << "Cache-B-field called at " << newLocation << G4endl;
98
99 G4double distSq= (newLocation-fLastLocation).mag2();
100 fCountCalls++;
101 if( distSq < fDistanceConst*fDistanceConst ) {
102 Bfield[0] = fLastValue.x();
103 Bfield[1] = fLastValue.y();
104 Bfield[2] = fLastValue.z();
105 }else{
106 // G4CachedMagneticField* thisNonC= const_cast<G4CachedMagneticField*>(this);
107 fpMagneticField->GetFieldValue( Point, Bfield );
108 // G4cout << " Evaluating. " << G4endl;
109 fCountEvaluations++;
110 // thisNonC->
111 fLastLocation= G4ThreeVector( Point[0], Point[1], Point[2] );
112 // thisNonC->
113 fLastValue= G4ThreeVector( Bfield[0], Bfield[1], Bfield[2] );
114 }
