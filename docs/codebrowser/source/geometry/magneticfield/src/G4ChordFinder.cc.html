77G4ChordFinder::G4ChordFinder( G4MagneticField* theMagField,
78 G4double stepMinimum,
79 G4MagIntegratorStepper* pItsStepper, // nullptr is default
80 G4bool useFSALstepper ) // false by default
81 : fDefaultDeltaChord( 0.25 * mm ), // Constants
82 fDeltaChord( fDefaultDeltaChord ), // Parameters
83 fStatsVerbose(0),
84 // fRegularStepperOwned(nullptr), // Dependent objects
85 fEquation(0)
86{
87 // Construct the Chord Finder
88 // by creating in inverse order the Driver, the Stepper and EqRhs ...
89
90 using NewFsalStepperType = G4RK547FEq1; // or 2 or 3
91 const char* NewFSALStepperName =
92 "G4RK574FEq1> FSAL 4th/5th order 7-stage 'Equilibrium-type' #1.";
93 using RegularStepperType =
94 G4DormandPrince745; // DOPRI5 (MatLab) 5th order embedded method. High efficiency.
95 // G4ClassicalRK4; // The old default
96 // G4CashKarpRKF45; // First embedded method in G4
97 // G4BogackiShampine45; // High efficiency 5th order embedded method
98 // G4NystromRK4; // Nystrom stepper 4th order
99 // G4RK547FEq1; // or 2 or 3
100 const char* RegularStepperName =
101 "G4DormandPrince745 (aka DOPRI5): 5th/4th Order 7-stage embedded stepper";
102 // "BogackiShampine 45 (Embedded 5th/4th Order, 7-stage)";
103 // "Nystrom stepper 4th order";
104
105 // Configurable
106 G4bool forceFSALstepper= false; // Choice - true to enable !!
107 G4bool recallFSALflag = useFSALstepper;
108 useFSALstepper = forceFSALstepper || useFSALstepper;
109
110#ifdef G4DEBUG_FIELD
111 G4cout << "G4ChordFinder 2nd Constructor called. " << G4endl;
112 G4cout << " Parameters: " << G4endl;
113 G4cout << " useFSAL stepper= " << useFSALstepper
114 << " (request = " << recallFSALflag
115 << " force FSAL = " << forceFSALstepper << " )" << G4endl;
116#endif
117
118 // useHigherStepper = forceHigherEffiencyStepper || useHigherStepper;
119
120 G4Mag_EqRhs *pEquation = new G4Mag_UsualEqRhs(theMagField);
121 fEquation = pEquation;
122
123 // G4MagIntegratorStepper* regularStepper = nullptr;
124 // G4VFSALIntegrationStepper* fsalSepper = nullptr; // for new-type FSAL steppers only
125 // NewFsalStepperType* fsalStepper = nullptr;
126 // G4MagIntegratorStepper* oldFSALStepper = nullptr;
127
128 G4bool errorInStepperCreation = false;
129
130 std::ostringstream message; // In case of failure, load with description !
131
132 if( pItsStepper != nullptr )
133 {
134 // Type is not known - so must use old class
135 fIntgrDriver = new G4IntegrationDriver<G4MagIntegratorStepper>(
136 stepMinimum, pItsStepper, pItsStepper->GetNumberOfVariables());
137 }
138 else if ( !useFSALstepper )
139 {
140 // RegularStepperType* regularStepper =nullptr; // To check the exception
141 auto regularStepper = new RegularStepperType(pEquation);
142 // *** ******************
143 //
144 // Alternative - for G4NystromRK4:
145 // = new G4NystromRK4(pEquation, 0.1*millimeter ); // *clhep::millimeter );
146 fRegularStepperOwned = regularStepper;
147
148 if( regularStepper == nullptr )
149 {
150 message << "Stepper instantiation FAILED." << G4endl;
151 message << "G4ChordFinder: Attempted to instantiate "
152 << RegularStepperName << " type stepper " << G4endl;
153 G4Exception("G4ChordFinder::G4ChordFinder()",
154 "GeomField1001", JustWarning, message);
155 errorInStepperCreation = true;
156 }
157 else
158 {
159 fIntgrDriver = new G4IntegrationDriver<G4MagIntegratorStepper>(
160 stepMinimum, regularStepper, regularStepper->GetNumberOfVariables());
161
162 if( fIntgrDriver==nullptr)
163 {
164 message << "Using G4IntegrationDriver with "
165 << RegularStepperName << " type stepper " << G4endl;
166 message << "Driver instantiation FAILED." << G4endl;
167 G4Exception("G4ChordFinder::G4ChordFinder()",
168 "GeomField1001", JustWarning, message);
169 }
170 }
171 }
172 else
173 {
174 auto fsalStepper= new NewFsalStepperType(pEquation);
175 // ******************
176 fNewFSALStepperOwned = fsalStepper;
177
178 if( fsalStepper == nullptr )
179 {
180 message << "Stepper instantiation FAILED." << G4endl;
181 message << "Attempted to instantiate "
182 << NewFSALStepperName << " type stepper " << G4endl;
183 G4Exception("G4ChordFinder::G4ChordFinder()",
184 "GeomField1001", JustWarning, message);
185 errorInStepperCreation = true;
186 }
187 else
188 {
189 fIntgrDriver = new
190 G4FSALIntegrationDriver<NewFsalStepperType>(stepMinimum,
191 fsalStepper,
192 fsalStepper->GetNumberOfVariables() );
193 // ==== Create the driver which knows the class type
194
195 if( fIntgrDriver==nullptr )
196 {
197 message << "Using G4FSALIntegrationDriver with stepper type: "
198 << NewFSALStepperName << G4endl;
199 message << "Integration Driver instantiation FAILED." << G4endl;
200 G4Exception("G4ChordFinder::G4ChordFinder()",
201 "GeomField1001", JustWarning, message);
202 }
203 }
204 }
205
206 // -- Main work is now done
207
208 // Now check that no error occured, and report it if one did.
209
210 // To test failure to create driver
211 // delete fIntgrDriver;
212 // fIntgrDriver= nullptr;
213
214 // Detect and report Error conditions
215 if( errorInStepperCreation || (fIntgrDriver == nullptr ))
216 {
217 std::ostringstream errmsg;
218
219 if( errorInStepperCreation )
220 {
221 errmsg << "ERROR> Failure to create Stepper object." << G4endl
222 << " --------------------------------" << G4endl;
223 }
224 if (fIntgrDriver == nullptr )
225 {
226 errmsg << "ERROR> Failure to create Integration-Driver object." << G4endl
227 << " -------------------------------------------" << G4endl;
228 }
229 const std::string BoolName[2]= { "False", "True" };
230 errmsg << " Configuration: (constructor arguments) " << G4endl
231 << " provided Stepper = " << pItsStepper << G4endl
232 << " use FSAL stepper = " << BoolName[useFSALstepper]
233 << " (request = " << BoolName[recallFSALflag]
234 << " force FSAL = " << BoolName[forceFSALstepper] << " )" << G4endl;
235 errmsg << message.str();
236 errmsg << "Aborting.";
237 G4Exception("G4ChordFinder::G4ChordFinder() - constructor 2",
238 "GeomField0003", FatalException, errmsg);
239 }
240
241 assert( ( pItsStepper != nullptr )
242 || ( fRegularStepperOwned != nullptr )
243 || ( fNewFSALStepperOwned != nullptr )
244 );
245 assert( fIntgrDriver != nullptr );
246}

262G4FieldTrack
263G4ChordFinder::ApproxCurvePointS( const G4FieldTrack& CurveA_PointVelocity,
264 const G4FieldTrack& CurveB_PointVelocity,
265 const G4FieldTrack& ApproxCurveV,
266 const G4ThreeVector& CurrentE_Point,
267 const G4ThreeVector& CurrentF_Point,
268 const G4ThreeVector& PointG,
269 G4bool first, G4double eps_step)
270{
271 // ApproxCurvePointS is 2nd implementation of ApproxCurvePoint.
272 // Use Brent Algorithm (or InvParabolic) when possible.
273 // Given a starting curve point A (CurveA_PointVelocity), curve point B
274 // (CurveB_PointVelocity), a point E which is (generally) not on the curve
275 // and a point F which is on the curve (first approximation), find new
276 // point S on the curve closer to point E.
277 // While advancing towards S utilise 'eps_step' as a measure of the
278 // relative accuracy of each Step.
279
280 G4FieldTrack EndPoint(CurveA_PointVelocity);
281 if(!first){EndPoint= ApproxCurveV;}
282
283 G4ThreeVector Point_A,Point_B;
284 Point_A=CurveA_PointVelocity.GetPosition();
285 Point_B=CurveB_PointVelocity.GetPosition();
286
287 G4double xa,xb,xc,ya,yb,yc;
288
289 // InverseParabolic. AF Intersects (First Part of Curve)
290
291 if(first)
292 {
293 xa=0.;
294 ya=(PointG-Point_A).mag();
295 xb=(Point_A-CurrentF_Point).mag();
296 yb=-(PointG-CurrentF_Point).mag();
297 xc=(Point_A-Point_B).mag();
298 yc=-(CurrentE_Point-Point_B).mag();
299 }
300 else
301 {
302 xa=0.;
303 ya=(Point_A-CurrentE_Point).mag();
304 xb=(Point_A-CurrentF_Point).mag();
305 yb=(PointG-CurrentF_Point).mag();
306 xc=(Point_A-Point_B).mag();
307 yc=-(Point_B-PointG).mag();
308 if(xb==0.)
309 {
310 EndPoint=
311 ApproxCurvePointV(CurveA_PointVelocity, CurveB_PointVelocity,
312 CurrentE_Point, eps_step);
313 return EndPoint;
314 }
315 }
316
317 const G4double tolerance= 1.e-12;
318 if(std::abs(ya)<=tolerance||std::abs(yc)<=tolerance)
319 {
320 ; // What to do for the moment: return the same point as at start
321 // then PropagatorInField will take care
322 }
323 else
324 {
325 G4double test_step = InvParabolic(xa,ya,xb,yb,xc,yc);
326 G4double curve;
327 if(first)
328 {
329 curve=std::abs(EndPoint.GetCurveLength()
330 -ApproxCurveV.GetCurveLength());
331 }
332 else
333 {
334 test_step=(test_step-xb);
335 curve=std::abs(EndPoint.GetCurveLength()
336 -CurveB_PointVelocity.GetCurveLength());
337 xb=(CurrentF_Point-Point_B).mag();
338 }
339
340 if(test_step<=0) { test_step=0.1*xb; }
341 if(test_step>=xb) { test_step=0.5*xb; }
342 if(test_step>=curve){ test_step=0.5*curve; }
343
344 if(curve*(1.+eps_step)<xb) // Similar to ReEstimate Step from
345 { // G4VIntersectionLocator
346 test_step=0.5*curve;
347 }
348
349 fIntgrDriver->AccurateAdvance(EndPoint,test_step, eps_step);
350
351#ifdef G4DEBUG_FIELD
352 // Printing Brent and Linear Approximation
353 //
354 G4cout << "G4ChordFinder::ApproxCurvePointS() - test-step ShF = "
355 << test_step << " EndPoint = " << EndPoint << G4endl;
356
357 // Test Track
358 //
359 G4FieldTrack TestTrack( CurveA_PointVelocity);
360 TestTrack = ApproxCurvePointV( CurveA_PointVelocity,
361 CurveB_PointVelocity,
362 CurrentE_Point, eps_step );
363 G4cout.precision(14);
364 G4cout << "G4ChordFinder::BrentApprox = " << EndPoint << G4endl;
365 G4cout << "G4ChordFinder::LinearApprox= " << TestTrack << G4endl;
366#endif
367 }
368 return EndPoint;
369}

374G4FieldTrack G4ChordFinder::
375ApproxCurvePointV( const G4FieldTrack& CurveA_PointVelocity,
376 const G4FieldTrack& CurveB_PointVelocity,
377 const G4ThreeVector& CurrentE_Point,
378 G4double eps_step)
379{
380 // If r=|AE|/|AB|, and s=true path lenght (AB)
381 // return the point that is r*s along the curve!
382
383 G4FieldTrack Current_PointVelocity = CurveA_PointVelocity;
384
385 G4ThreeVector CurveA_Point= CurveA_PointVelocity.GetPosition();
386 G4ThreeVector CurveB_Point= CurveB_PointVelocity.GetPosition();
387
388 G4ThreeVector ChordAB_Vector= CurveB_Point - CurveA_Point;
389 G4ThreeVector ChordAE_Vector= CurrentE_Point - CurveA_Point;
390
391 G4double ABdist= ChordAB_Vector.mag();
392 G4double curve_length; // A curve length of AB
393 G4double AE_fraction;
394
395 curve_length= CurveB_PointVelocity.GetCurveLength()
396 - CurveA_PointVelocity.GetCurveLength();
397
398 G4double integrationInaccuracyLimit= std::max( perMillion, 0.5*eps_step );
399 if( curve_length < ABdist * (1. - integrationInaccuracyLimit) )
400 {
401#ifdef G4DEBUG_FIELD
402 G4cerr << " Warning in G4ChordFinder::ApproxCurvePoint: "
403 << G4endl
404 << " The two points are further apart than the curve length "
405 << G4endl
406 << " Dist = " << ABdist
407 << " curve length = " << curve_length
408 << " relativeDiff = " << (curve_length-ABdist)/ABdist
409 << G4endl;
410 if( curve_length < ABdist * (1. - 10*eps_step) )
411 {
412 std::ostringstream message;
413 message << "Unphysical curve length." << G4endl
414 << "The size of the above difference exceeds allowed limits."
415 << G4endl
416 << "Aborting.";
417 G4Exception("G4ChordFinder::ApproxCurvePointV()", "GeomField0003",
418 FatalException, message);
419 }
420#endif
421 // Take default corrective action: adjust the maximum curve length.
422 // NOTE: this case only happens for relatively straight paths.
423 // curve_length = ABdist;
424 }
425
426 G4double new_st_length;
427
428 if ( ABdist > 0.0 )
429 {
430 AE_fraction = ChordAE_Vector.mag() / ABdist;
431 }
432 else
433 {
434 AE_fraction = 0.5; // Guess .. ?;
435#ifdef G4DEBUG_FIELD
436 G4cout << "Warning in G4ChordFinder::ApproxCurvePointV():"
437 << " A and B are the same point!" << G4endl
438 << " Chord AB length = " << ChordAE_Vector.mag() << G4endl
439 << G4endl;
440#endif
441 }
442
443 if( (AE_fraction> 1.0 + perMillion) || (AE_fraction< 0.) )
444 {
445#ifdef G4DEBUG_FIELD
446 G4cerr << " G4ChordFinder::ApproxCurvePointV() - Warning:"
447 << " Anomalous condition:AE > AB or AE/AB <= 0 " << G4endl
448 << " AE_fraction = " << AE_fraction << G4endl
449 << " Chord AE length = " << ChordAE_Vector.mag() << G4endl
450 << " Chord AB length = " << ABdist << G4endl << G4endl;
451 G4cerr << " OK if this condition occurs after a recalculation of 'B'"
452 << G4endl << " Otherwise it is an error. " << G4endl ;
453#endif
454 // This course can now result if B has been re-evaluated,
455 // without E being recomputed (1 July 99).
456 // In this case this is not a "real error" - but it is undesired
457 // and we cope with it by a default corrective action ...
458 //
459 AE_fraction = 0.5; // Default value
460 }
461
462 new_st_length= AE_fraction * curve_length;
463
464 if ( AE_fraction > 0.0 )
465 {
466 fIntgrDriver->AccurateAdvance(Current_PointVelocity,
467 new_st_length, eps_step );
468 //
469 // In this case it does not matter if it cannot advance the full distance
470 }
471
472 // If there was a memory of the step_length actually required at the start
473 // of the integration Step, this could be re-used ...
474
475 G4cout.precision(14);
476
477 return Current_PointVelocity;
478}
