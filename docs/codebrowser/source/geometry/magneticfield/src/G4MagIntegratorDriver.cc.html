53G4MagInt_Driver::G4MagInt_Driver( G4double hminimum,
54 G4MagIntegratorStepper *pStepper,
55 G4int numComponents,
56 G4int statisticsVerbose)
57 : fSmallestFraction( 1.0e-12 ),
58 fNoIntegrationVariables(numComponents),
59 fMinNoVars(12),
60 fNoVars( std::max( fNoIntegrationVariables, fMinNoVars )),
61 fStatisticsVerboseLevel(statisticsVerbose),
62 fNoTotalSteps(0), fNoBadSteps(0), fNoSmallSteps(0),
63 fNoInitialSmallSteps(0), fNoCalls(0),
64 fDyerr_max(0.0), fDyerr_mx2(0.0),
65 fDyerrPos_smTot(0.0), fDyerrPos_lgTot(0.0), fDyerrVel_lgTot(0.0),
66 fSumH_sm(0.0), fSumH_lg(0.0),
67 fVerboseLevel(0)
68{
69 // In order to accomodate "Laboratory Time", which is [7], fMinNoVars=8
70 // is required. For proper time of flight and spin, fMinNoVars must be 12
71
72 RenewStepperAndAdjust( pStepper );
73 fMinimumStep= hminimum;
74
75 // The (default) maximum number of steps is Base
76 // divided by the order of Stepper
77 //
78 fMaxStepBase = 250; // Was 5000
79
80 fMaxNoSteps = fMaxStepBase / pIntStepper->IntegratorOrder();
81#ifdef G4DEBUG_FIELD
82 fVerboseLevel=2;
83#endif
84
85 if( (fVerboseLevel > 0) || (fStatisticsVerboseLevel > 1) )
86 {
87 G4cout << "MagIntDriver version: Accur-Adv: "
88 << "invE_nS, QuickAdv-2sqrt with Statistics "
89#ifdef G4FLD_STATS
90 << " enabled "
91#else
92 << " disabled "
93#endif
94 << G4endl;
95 }
96}

112G4bool
113G4MagInt_Driver::AccurateAdvance(G4FieldTrack& y_current,
114 G4double hstep,
115 G4double eps,
116 G4double hinitial )
117{
118 // Runge-Kutta driver with adaptive stepsize control. Integrate starting
119 // values at y_current over hstep x2 with accuracy eps.
120 // On output ystart is replaced by values at the end of the integration
121 // interval. RightHandSide is the right-hand side of ODE system.
122 // The source is similar to odeint routine from NRC p.721-722 .
123
124 G4int nstp, i, no_warnings=0;
125 G4double x, hnext, hdid, h;
126
127#ifdef G4DEBUG_FIELD
128 static G4int dbg=1;
129 static G4int nStpPr=50; // For debug printing of long integrations
130 G4double ySubStepStart[G4FieldTrack::ncompSVEC];
131 G4FieldTrack yFldTrkStart(y_current);
132#endif
133
134 G4double y[G4FieldTrack::ncompSVEC], dydx[G4FieldTrack::ncompSVEC];
135 G4double ystart[G4FieldTrack::ncompSVEC], yEnd[G4FieldTrack::ncompSVEC];
136 G4double x1, x2;
137 G4bool succeeded = true, lastStepSucceeded;
138
139 G4double startCurveLength;
140
141 G4int noFullIntegr=0, noSmallIntegr = 0 ;
142 static G4ThreadLocal G4int noGoodSteps =0 ; // Bad = chord > curve-len
143 const G4int nvar= fNoVars;
144
145 G4FieldTrack yStartFT(y_current);
146
147 // Ensure that hstep > 0
148 //
149 if( hstep <= 0.0 )
150 {
151 if(hstep==0.0)
152 {
153 std::ostringstream message;
154 message << "Proposed step is zero; hstep = " << hstep << " !";
155 G4Exception("G4MagInt_Driver::AccurateAdvance()",
156 "GeomField1001", JustWarning, message);
157 return succeeded;
158 }
159 else
160 {
161 std::ostringstream message;
162 message << "Invalid run condition." << G4endl
163 << "Proposed step is negative; hstep = " << hstep << "." << G4endl
164 << "Requested step cannot be negative! Aborting event.";
165 G4Exception("G4MagInt_Driver::AccurateAdvance()",
166 "GeomField0003", EventMustBeAborted, message);
167 return false;
168 }
169 }
170
171 y_current.DumpToArray( ystart );
172
173 startCurveLength= y_current.GetCurveLength();
174 x1= startCurveLength;
175 x2= x1 + hstep;
176
177 if ( (hinitial > 0.0) && (hinitial < hstep)
178 && (hinitial > perMillion * hstep) )
179 {
180 h = hinitial;
181 }
182 else // Initial Step size "h" defaults to the full interval
183 {
184 h = hstep;
185 }
186
187 x = x1;
188
189 for (i=0;i<nvar;i++) { y[i] = ystart[i]; }
190
191 G4bool lastStep= false;
192 nstp=1;
193
194 do
195 {
196 G4ThreeVector StartPos( y[0], y[1], y[2] );
197
198#ifdef G4DEBUG_FIELD
199 G4double xSubStepStart= x;
200 for (i=0;i<nvar;i++) { ySubStepStart[i] = y[i]; }
201 yFldTrkStart.LoadFromArray(y, fNoIntegrationVariables);
202 yFldTrkStart.SetCurveLength(x);
203#endif
204
205 pIntStepper->RightHandSide( y, dydx );
206 fNoTotalSteps++;
207
208 // Perform the Integration
209 //
210 if( h > fMinimumStep )
211 {
212 OneGoodStep(y,dydx,x,h,eps,hdid,hnext) ;
213 //--------------------------------------
214 lastStepSucceeded= (hdid == h);
215#ifdef G4DEBUG_FIELD
216 if (dbg>2)
217 {
218 PrintStatus( ySubStepStart, xSubStepStart, y, x, h, nstp); // Only
219 }
220#endif
221 }
222 else
223 {
224 G4FieldTrack yFldTrk( G4ThreeVector(0,0,0),
225 G4ThreeVector(0,0,0), 0., 0., 0., 0. );
226 G4double dchord_step, dyerr, dyerr_len; // What to do with these ?
227 yFldTrk.LoadFromArray(y, fNoIntegrationVariables);
228 yFldTrk.SetCurveLength( x );
229
230 QuickAdvance( yFldTrk, dydx, h, UNKNOWN_CURVATURE_RADIUS, dchord_step, dyerr_len );
231 //-----------------------------------------------------
232
233 yFldTrk.DumpToArray(y);
234
235#ifdef G4FLD_STATS
236 fNoSmallSteps++;
237 if ( dyerr_len > fDyerr_max) { fDyerr_max= dyerr_len; }
238 fDyerrPos_smTot += dyerr_len;
239 fSumH_sm += h; // Length total for 'small' steps
240 if (nstp<=1) { fNoInitialSmallSteps++; }
241#endif
242#ifdef G4DEBUG_FIELD
243 if (dbg>1)
244 {
245 if(fNoSmallSteps<2) { PrintStatus(ySubStepStart, x1, y, x, h, -nstp); }
246 G4cout << "Another sub-min step, no " << fNoSmallSteps
247 << " of " << fNoTotalSteps << " this time " << nstp << G4endl;
248 PrintStatus( ySubStepStart, x1, y, x, h, nstp); // Only this
249 G4cout << " dyerr= " << dyerr_len << " relative = " << dyerr_len / h
250 << " epsilon= " << eps << " hstep= " << hstep
251 << " h= " << h << " hmin= " << fMinimumStep << G4endl;
252 }
253#endif
254 if( h == 0.0 )
255 {
256 G4Exception("G4MagInt_Driver::AccurateAdvance()",
257 "GeomField0003", FatalException,
258 "Integration Step became Zero!");
259 }
260 dyerr = dyerr_len / h;
261 hdid= h;
262 x += hdid;
263
264 // Compute suggested new step
265 hnext= ComputeNewStepSize( dyerr/eps, h);
266
267 // .. hnext= ComputeNewStepSize_WithinLimits( dyerr/eps, h);
268 lastStepSucceeded= (dyerr<= eps);
269 }
270
271 if (lastStepSucceeded) { noFullIntegr++; }
272 else { noSmallIntegr++; }
273
274 G4ThreeVector EndPos( y[0], y[1], y[2] );
275
276#ifdef G4DEBUG_FIELD
277 if( (dbg>0) && (dbg<=2) && (nstp>nStpPr))
278 {
279 if( nstp==nStpPr ) { G4cout << "***** Many steps ****" << G4endl; }
280 G4cout << "MagIntDrv: " ;
281 G4cout << "hdid=" << std::setw(12) << hdid << " "
282 << "hnext=" << std::setw(12) << hnext << " "
283 << "hstep=" << std::setw(12) << hstep << " (requested) "
284 << G4endl;
285 PrintStatus( ystart, x1, y, x, h, (nstp==nStpPr) ? -nstp: nstp);
286 }
287#endif
288
289 // Check the endpoint
290 G4double endPointDist= (EndPos-StartPos).mag();
291 if ( endPointDist >= hdid*(1.+perMillion) )
292 {
293 fNoBadSteps++;
294
295 // Issue a warning only for gross differences -
296 // we understand how small difference occur.
297 if ( endPointDist >= hdid*(1.+perThousand) )
298 {
299#ifdef G4DEBUG_FIELD
300 if (dbg)
301 {
302 WarnEndPointTooFar ( endPointDist, hdid, eps, dbg );
303 G4cerr << " Total steps: bad " << fNoBadSteps
304 << " good " << noGoodSteps << " current h= " << hdid
305 << G4endl;
306 PrintStatus( ystart, x1, y, x, hstep, no_warnings?nstp:-nstp);
307 }
308#endif
309 no_warnings++;
310 }
311 }
312 else
313 {
314 noGoodSteps ++;
315 }
316// #endif
317
318 // Avoid numerous small last steps
319 if( (h < eps * hstep) || (h < fSmallestFraction * startCurveLength) )
320 {
321 // No more integration -- the next step will not happen
322 lastStep = true;
323 }
324 else
325 {
326 // Check the proposed next stepsize
327 if(std::fabs(hnext) <= Hmin())
328 {
329#ifdef G4DEBUG_FIELD
330 // If simply a very small interval is being integrated, do not warn
331 if( (x < x2 * (1-eps) ) && // The last step can be small: OK
332 (std::fabs(hstep) > Hmin()) ) // and if we are asked, it's OK
333 {
334 if(dbg>0)
335 {
336 WarnSmallStepSize( hnext, hstep, h, x-x1, nstp );
337 PrintStatus( ystart, x1, y, x, hstep, no_warnings?nstp:-nstp);
338 }
339 no_warnings++;
340 }
341#endif
342 // Make sure that the next step is at least Hmin.
343 h = Hmin();
344 }
345 else
346 {
347 h = hnext;
348 }
349
350 // Ensure that the next step does not overshoot
351 if ( x+h > x2 )
352 { // When stepsize overshoots, decrease it!
353 h = x2 - x ; // Must cope with difficult rounding-error
354 } // issues if hstep << x2
355
356 if ( h == 0.0 )
357 {
358 // Cannot progress - accept this as last step - by default
359 lastStep = true;
360#ifdef G4DEBUG_FIELD
361 if (dbg>2)
362 {
363 int prec= G4cout.precision(12);
364 G4cout << "Warning: G4MagIntegratorDriver::AccurateAdvance"
365 << G4endl
366 << " Integration step 'h' became "
367 << h << " due to roundoff. " << G4endl
368 << " Calculated as difference of x2= "<< x2 << " and x=" << x
369 << " Forcing termination of advance." << G4endl;
370 G4cout.precision(prec);
371 }
372#endif
373 }
374 }
375 } while ( ((nstp++)<=fMaxNoSteps) && (x < x2) && (!lastStep) );
376 // Loop checking, 07.10.2016, J. Apostolakis
377
378 // Have we reached the end ?
379 // --> a better test might be x-x2 > an_epsilon
380
381 succeeded= (x>=x2); // If it was a "forced" last step
382
383 for (i=0;i<nvar;i++) { yEnd[i] = y[i]; }
384
385 // Put back the values.
386 y_current.LoadFromArray( yEnd, fNoIntegrationVariables );
387 y_current.SetCurveLength( x );
388
389 if(nstp > fMaxNoSteps)
390 {
391 no_warnings++;
392 succeeded = false;
393#ifdef G4DEBUG_FIELD
394 if (dbg)
395 {
396 WarnTooManyStep( x1, x2, x ); // Issue WARNING
397 PrintStatus( yEnd, x1, y, x, hstep, -nstp);
398 }
399#endif
400 }
401
402#ifdef G4DEBUG_FIELD
403 if( dbg && no_warnings )
404 {
405 G4cerr << "G4MagIntegratorDriver exit status: no-steps " << nstp << G4endl;
406 PrintStatus( yEnd, x1, y, x, hstep, nstp);
407 }
408#endif
409
410 return succeeded;
411} // end of AccurateAdvance ...........................

415void
416G4MagInt_Driver::WarnSmallStepSize( G4double hnext, G4double hstep,
417 G4double h, G4double xDone,
418 G4int nstp)
419{
420 static G4ThreadLocal G4int noWarningsIssued =0;
421 const G4int maxNoWarnings = 10; // Number of verbose warnings
422 std::ostringstream message;
423 if( (noWarningsIssued < maxNoWarnings) || fVerboseLevel > 10 )
424 {
425 message << "The stepsize for the next iteration, " << hnext
426 << ", is too small - in Step number " << nstp << "." << G4endl
427 << "The minimum for the driver is " << Hmin() << G4endl
428 << "Requested integr. length was " << hstep << " ." << G4endl
429 << "The size of this sub-step was " << h << " ." << G4endl
430 << "The integrations has already gone " << xDone;
431 }
432 else
433 {
434 message << "Too small 'next' step " << hnext
435 << ", step-no: " << nstp << G4endl
436 << ", this sub-step: " << h
437 << ", req_tot_len: " << hstep
438 << ", done: " << xDone << ", min: " << Hmin();
439 }
440 G4Exception("G4MagInt_Driver::WarnSmallStepSize()", "GeomField1001",
441 JustWarning, message);
442 noWarningsIssued++;
443}

464void
465G4MagInt_Driver::WarnEndPointTooFar (G4double endPointDist,
466 G4double h ,
467 G4double eps,
468 G4int dbg)
469{
470 static G4ThreadLocal G4double maxRelError=0.0;
471 G4bool isNewMax, prNewMax;
472
473 isNewMax = endPointDist > (1.0 + maxRelError) * h;
474 prNewMax = endPointDist > (1.0 + 1.05 * maxRelError) * h;
475 if( isNewMax ) { maxRelError= endPointDist / h - 1.0; }
476
477 if( dbg && (h > G4GeometryTolerance::GetInstance()->GetSurfaceTolerance())
478 && ( (dbg>1) || prNewMax || (endPointDist >= h*(1.+eps) ) ) )
479 {
480 static G4ThreadLocal G4int noWarnings = 0;
481 std::ostringstream message;
482 if( (noWarnings ++ < 10) || (dbg>2) )
483 {
484 message << "The integration produced an end-point which " << G4endl
485 << "is further from the start-point than the curve length."
486 << G4endl;
487 }
488 message << " Distance of endpoints = " << endPointDist
489 << ", curve length = " << h << G4endl
490 << " Difference (curveLen-endpDist)= " << (h - endPointDist)
491 << ", relative = " << (h-endPointDist) / h
492 << ", epsilon = " << eps;
493 G4Exception("G4MagInt_Driver::WarnEndPointTooFar()", "GeomField1001",
494 JustWarning, message);
495 }
496}

500void
501G4MagInt_Driver::OneGoodStep( G4double y[], // InOut
502 const G4double dydx[],
503 G4double& x, // InOut
504 G4double htry,
505 G4double eps_rel_max,
506 G4double& hdid, // Out
507 G4double& hnext ) // Out
508
509// Driver for one Runge-Kutta Step with monitoring of local truncation error
510// to ensure accuracy and adjust stepsize. Input are dependent variable
511// array y[0,...,5] and its derivative dydx[0,...,5] at the
512// starting value of the independent variable x . Also input are stepsize
513// to be attempted htry, and the required accuracy eps. On output y and x
514// are replaced by their new values, hdid is the stepsize that was actually
515// accomplished, and hnext is the estimated next stepsize.
516// This is similar to the function rkqs from the book:
517// Numerical Recipes in C: The Art of Scientific Computing (NRC), Second
518// Edition, by William H. Press, Saul A. Teukolsky, William T.
519// Vetterling, and Brian P. Flannery (Cambridge University Press 1992),
520// 16.2 Adaptive StepSize Control for Runge-Kutta, p. 719
521
522{
523 G4double errmax_sq;
524 G4double h, htemp, xnew ;
525
526 G4double yerr[G4FieldTrack::ncompSVEC], ytemp[G4FieldTrack::ncompSVEC];
527
528 h = htry ; // Set stepsize to the initial trial value
529
530 G4double inv_eps_vel_sq = 1.0 / (eps_rel_max*eps_rel_max);
531
532 G4double errpos_sq=0.0; // square of displacement error
533 G4double errvel_sq=0.0; // square of momentum vector difference
534 G4double errspin_sq=0.0; // square of spin vector difference
535
536 G4int iter;
537
538 static G4ThreadLocal G4int tot_no_trials=0;
539 const G4int max_trials=100;
540
541 G4ThreeVector Spin(y[9],y[10],y[11]);
542 G4double spin_mag2 =Spin.mag2() ;
543 G4bool hasSpin= (spin_mag2 > 0.0);
544
545 for (iter=0; iter<max_trials ;iter++)
546 {
547 tot_no_trials++;
548 pIntStepper-> Stepper(y,dydx,h,ytemp,yerr);
549 // *******
550 G4double eps_pos = eps_rel_max * std::max(h, fMinimumStep);
551 G4double inv_eps_pos_sq = 1.0 / (eps_pos*eps_pos);
552
553 // Evaluate accuracy
554 //
555 errpos_sq = sqr(yerr[0]) + sqr(yerr[1]) + sqr(yerr[2]) ;
556 errpos_sq *= inv_eps_pos_sq; // Scale relative to required tolerance
557
558 // Accuracy for momentum
559 G4double magvel_sq= sqr(y[3]) + sqr(y[4]) + sqr(y[5]) ;
560 G4double sumerr_sq = sqr(yerr[3]) + sqr(yerr[4]) + sqr(yerr[5]) ;
561 if( magvel_sq > 0.0 )
562 {
563 errvel_sq = sumerr_sq / magvel_sq;
564 }
565 else
566 {
567 std::ostringstream message;
568 message << "Found case of zero momentum." << G4endl
569 << "- iteration= " << iter << "; h= " << h;
570 G4Exception("G4MagInt_Driver::OneGoodStep()",
571 "GeomField1001", JustWarning, message);
572 errvel_sq = sumerr_sq;
573 }
574 errvel_sq *= inv_eps_vel_sq;
575 errmax_sq = std::max( errpos_sq, errvel_sq ); // Square of maximum error
576
577 if( hasSpin )
578 {
579 // Accuracy for spin
580 errspin_sq = ( sqr(yerr[9]) + sqr(yerr[10]) + sqr(yerr[11]) )
581 / spin_mag2; // ( sqr(y[9]) + sqr(y[10]) + sqr(y[11]) );
582 errspin_sq *= inv_eps_vel_sq;
583 errmax_sq = std::max( errmax_sq, errspin_sq );
584 }
585
586 if ( errmax_sq <= 1.0 ) { break; } // Step succeeded.
587
588 // Step failed; compute the size of retrial Step.
589 htemp = GetSafety()*h* std::pow( errmax_sq, 0.5*GetPshrnk() );
590
591 if (htemp >= 0.1*h) { h = htemp; } // Truncation error too large,
592 else { h = 0.1*h; } // reduce stepsize, but no more
593 // than a factor of 10
594 xnew = x + h;
595 if(xnew == x)
596 {
597 std::ostringstream message;
598 message << "Stepsize underflow in Stepper !" << G4endl
599 << "- Step's start x=" << x << " and end x= " << xnew
600 << " are equal !! " << G4endl
601 << " Due to step-size= " << h
602 << ". Note that input step was " << htry;
603 G4Exception("G4MagInt_Driver::OneGoodStep()",
604 "GeomField1001", JustWarning, message);
605 break;
606 }
607 }
608
609 // Compute size of next Step
610 if (errmax_sq > errcon*errcon)
611 {
612 hnext = GetSafety()*h*std::pow(errmax_sq, 0.5*GetPgrow());
613 }
614 else
615 {
616 hnext = max_stepping_increase*h ; // No more than a factor of 5 increase
617 }
618 x += (hdid = h);
619
620 for(G4int k=0;k<fNoIntegrationVariables;k++) { y[k] = ytemp[k]; }
621
622 return;
623}

627// QuickAdvance just tries one Step - it does not ensure accuracy
628//
629G4bool G4MagInt_Driver::QuickAdvance(
630 G4FieldTrack& y_posvel, // INOUT
631 const G4double dydx[],
632 G4double hstep, // In
633 G4double& dchord_step,
634 G4double& dyerr_pos_sq,
635 G4double& dyerr_mom_rel_sq )
636{
637 G4Exception("G4MagInt_Driver::QuickAdvance()", "GeomField0001",
638 FatalException, "Not yet implemented.");
639
640 // Use the parameters of this method, to please compiler
641 dchord_step = dyerr_pos_sq = hstep * hstep * dydx[0];
642 dyerr_mom_rel_sq = y_posvel.GetPosition().mag2();
643 return true;
644}

646//----------------------------------------------------------------------
647
648G4bool G4MagInt_Driver::QuickAdvance(
649 G4FieldTrack& y_posvel, // INOUT
650 const G4double dydx[],
651 G4double hstep, // In
652 G4double /*inverseCurvatureRadius*/,
653 G4double& dchord_step,
654 G4double& dyerr )
655{
656 G4double dyerr_pos_sq, dyerr_mom_rel_sq;
657 G4double yerr_vec[G4FieldTrack::ncompSVEC],
658 yarrin[G4FieldTrack::ncompSVEC], yarrout[G4FieldTrack::ncompSVEC];
659 G4double s_start;
660 G4double dyerr_mom_sq, vel_mag_sq, inv_vel_mag_sq;
661
662 static G4ThreadLocal G4int no_call=0;
663 no_call ++;
664
665 // Move data into array
666 y_posvel.DumpToArray( yarrin ); // yarrin <== y_posvel
667 s_start = y_posvel.GetCurveLength();
668
669 // Do an Integration Step
670 pIntStepper-> Stepper(yarrin, dydx, hstep, yarrout, yerr_vec) ;
671
672 // Estimate curve-chord distance
673 dchord_step= pIntStepper-> DistChord();
674
675 // Put back the values. yarrout ==> y_posvel
676 y_posvel.LoadFromArray( yarrout, fNoIntegrationVariables );
677 y_posvel.SetCurveLength( s_start + hstep );
678
679#ifdef G4DEBUG_FIELD
680 if(fVerboseLevel>2)
681 {
682 G4cout << "G4MagIntDrv: Quick Advance" << G4endl;
683 PrintStatus( yarrin, s_start, yarrout, s_start+hstep, hstep, 1);
684 }
685#endif
686
687 // A single measure of the error
688 // TO-DO : account for energy, spin, ... ?
689 vel_mag_sq = ( sqr(yarrout[3])+sqr(yarrout[4])+sqr(yarrout[5]) );
690 inv_vel_mag_sq = 1.0 / vel_mag_sq;
691 dyerr_pos_sq = ( sqr(yerr_vec[0])+sqr(yerr_vec[1])+sqr(yerr_vec[2]));
692 dyerr_mom_sq = ( sqr(yerr_vec[3])+sqr(yerr_vec[4])+sqr(yerr_vec[5]));
693 dyerr_mom_rel_sq = dyerr_mom_sq * inv_vel_mag_sq;
694
695 // Calculate also the change in the momentum squared also ???
696 // G4double veloc_square = y_posvel.GetVelocity().mag2();
697 // ...
698
699#ifdef RETURN_A_NEW_STEP_LENGTH
700 // The following step cannot be done here because "eps" is not known.
701 dyerr_len = std::sqrt( dyerr_len_sq );
702 dyerr_len_sq /= eps ;
703
704 // Look at the velocity deviation ?
705 // sqr(yerr_vec[3])+sqr(yerr_vec[4])+sqr(yerr_vec[5]));
706
707 // Set suggested new step
708 hstep= ComputeNewStepSize( dyerr_len, hstep);
709#endif
710
711 if( dyerr_pos_sq > ( dyerr_mom_rel_sq * sqr(hstep) ) )
712 {
713 dyerr = std::sqrt(dyerr_pos_sq);
714 }
715 else
716 {
717 // Scale it to the current step size - for now
718 dyerr = std::sqrt(dyerr_mom_rel_sq) * hstep;
719 }
720
721 return true;
722}

744// This method computes new step sizes - but does not limit changes to
745// within certain factors
746//
747G4double G4MagInt_Driver::
748ComputeNewStepSize(G4double errMaxNorm, // max error (normalised)
749 G4double hstepCurrent) // current step size
750{
751 G4double hnew;
752
753 // Compute size of next Step for a failed step
754 if(errMaxNorm > 1.0 )
755 {
756 // Step failed; compute the size of retrial Step.
757 hnew = GetSafety()*hstepCurrent*std::pow(errMaxNorm,GetPshrnk()) ;
758 }
759 else if(errMaxNorm > 0.0 )
760 {
761 // Compute size of next Step for a successful step
762 hnew = GetSafety()*hstepCurrent*std::pow(errMaxNorm,GetPgrow()) ;
763 }
764 else
765 {
766 // if error estimate is zero (possible) or negative (dubious)
767 hnew = max_stepping_increase * hstepCurrent;
768 }
769
770 return hnew;
771}

773// ---------------------------------------------------------------------------
774
775// This method computes new step sizes limiting changes within certain factors
776//
777// It shares its logic with AccurateAdvance.
778// They are kept separate currently for optimisation.
779//
780G4double
781G4MagInt_Driver::ComputeNewStepSize_WithinLimits(
782 G4double errMaxNorm, // max error (normalised)
783 G4double hstepCurrent) // current step size
784{
785 G4double hnew;
786
787 // Compute size of next Step for a failed step
788 if (errMaxNorm > 1.0 )
789 {
790 // Step failed; compute the size of retrial Step.
791 hnew = GetSafety()*hstepCurrent*std::pow(errMaxNorm,GetPshrnk()) ;
792
793 if (hnew < max_stepping_decrease*hstepCurrent)
794 {
795 hnew = max_stepping_decrease*hstepCurrent ;
796 // reduce stepsize, but no more
797 // than this factor (value= 1/10)
798 }
799 }
800 else
801 {
802 // Compute size of next Step for a successful step
803 if (errMaxNorm > errcon)
804 { hnew = GetSafety()*hstepCurrent*std::pow(errMaxNorm,GetPgrow()); }
805 else // No more than a factor of 5 increase
806 { hnew = max_stepping_increase * hstepCurrent; }
807 }
808 return hnew;
809}
