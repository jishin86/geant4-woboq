35#include "G4ChargeState.hh"
36
37void G4ChargeState::SetChargeSpinMoments(G4double charge,
38 G4double spin,
39 G4double magnetic_dipole_moment,
40 G4double electric_dipole_moment,
41 G4double magnetic_charge )
42 // Revise the charge and potentially all moments.
43 // By default do not change mdm, edm, mag charge.
44{
45 fCharge = charge;
46 fSpin = spin;
47 if( magnetic_dipole_moment < DBL_MAX) fMagn_dipole= magnetic_dipole_moment;
48 if( electric_dipole_moment < DBL_MAX) fElec_dipole= electric_dipole_moment;
49 if( magnetic_charge < DBL_MAX) fMagneticCharge= magnetic_charge;
50}
