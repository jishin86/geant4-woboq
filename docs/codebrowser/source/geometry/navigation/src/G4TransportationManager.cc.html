49G4ThreadLocal G4TransportationManager*
50G4TransportationManager::fTransportationManager=0;

52// ----------------------------------------------------------------------------
53// Constructor
54//
55G4TransportationManager::G4TransportationManager()
56{
57 if (fTransportationManager)
58 {
59 G4Exception("G4TransportationManager::G4TransportationManager()",
60 "GeomNav0002", FatalException,
61 "Only ONE instance of G4TransportationManager is allowed!");
62 }
63
64 // Create the navigator for tracking and activate it; add to collections
65 //
66 G4Navigator* trackingNavigator = new G4Navigator();
67 trackingNavigator->Activate(true);
68 fNavigators.push_back(trackingNavigator);
69 fActiveNavigators.push_back(trackingNavigator);
70 fWorlds.push_back(trackingNavigator->GetWorldVolume()); // NULL registered
71
72 fGeomMessenger = new G4GeometryMessenger(this);
73 fFieldManager = new G4FieldManager(); // deleted by G4FieldManagerStore
74 fPropagatorInField= new G4PropagatorInField(trackingNavigator,fFieldManager);
75 fSafetyHelper = new G4SafetyHelper();
76}
77
78// ----------------------------------------------------------------------------
79// Destructor
80//
81G4TransportationManager::~G4TransportationManager()
82{
83 delete fSafetyHelper;
84 delete fPropagatorInField;
85 delete fGeomMessenger;
86 ClearNavigators();
87 fTransportationManager = 0;
88}

145// ----------------------------------------------------------------------------
146// ClearNavigators()
147//
148// Clear collection of navigators and delete allocated objects.
149// Called only by the class destructor.
150//
151void G4TransportationManager::ClearNavigators()
152{
153 std::vector<G4Navigator*>::iterator pNav;
154 for (pNav=fNavigators.begin(); pNav!=fNavigators.end(); pNav++)
155 {
156 delete *pNav;
157 }
158 fNavigators.clear();
159 fActiveNavigators.clear();
160 fWorlds.clear();
161}

163// ----------------------------------------------------------------------------
164// GetParallelWorld()
165//
166// Provided the name of a world volume, returns the associated world pointer.
167// If not existing, create (allocate) and register it in the collection.
168//
169G4VPhysicalVolume*
170G4TransportationManager::GetParallelWorld( const G4String& worldName )
171{
172 G4VPhysicalVolume* wPV = IsWorldExisting(worldName);
173 if (!wPV)
174 {
175 wPV = GetNavigatorForTracking()->GetWorldVolume();
176 G4LogicalVolume* wLV = wPV->GetLogicalVolume();
177 wLV = new G4LogicalVolume(wLV->GetSolid(), 0,
178 worldName);
179 wPV = new G4PVPlacement (wPV->GetRotation(),
180 wPV->GetTranslation(),
181 wLV, worldName, 0, false, 0);
182 RegisterWorld(wPV);
183 }
184 return wPV;
185}

187// ----------------------------------------------------------------------------
188// GetNavigator()
189//
190// Provided the name of a world volume, returns the associated navigator.
191// If not existing, create it and register it in the collection, throw an
192// exception if the associated parallel world does not exist.
193//
194G4Navigator* G4TransportationManager::GetNavigator( const G4String& worldName )
195{
196 // If already existing, return the stored pointer to the navigator
197 //
198 std::vector<G4Navigator*>::iterator pNav;
199 for (pNav=fNavigators.begin(); pNav!=fNavigators.end(); pNav++)
200 {
201 if ((*pNav)->GetWorldVolume()->GetName() == worldName) { return *pNav; }
202 }
203
204 // Check if world of that name already exists,
205 // create a navigator and register it
206 //
207 G4Navigator* aNavigator = 0;
208 G4VPhysicalVolume* aWorld = IsWorldExisting(worldName);
209 if(aWorld)
210 {
211 aNavigator = new G4Navigator();
212 aNavigator->SetWorldVolume(aWorld);
213 fNavigators.push_back(aNavigator);
214 }
215 else
216 {
217 G4String message
218 = "World volume with name -" + worldName
219 + "- does not exist. Create it first by GetParallelWorld() method!";
220 G4Exception("G4TransportationManager::GetNavigator(name)",
221 "GeomNav0002", FatalException, message);
222 }
223
224 return aNavigator;
225}

227// ----------------------------------------------------------------------------
228// GetNavigator()
229//
230// Provided a pointer to a world volume, returns the associated navigator.
231// Create it in case not existing and add it to the collection.
232// If world volume not existing, issue an exception.
233//
234G4Navigator* G4TransportationManager::GetNavigator( G4VPhysicalVolume* aWorld )
235{
236 std::vector<G4Navigator*>::iterator pNav;
237 for (pNav=fNavigators.begin(); pNav!=fNavigators.end(); pNav++)
238 {
239 if ((*pNav)->GetWorldVolume() == aWorld) { return *pNav; }
240 }
241 G4Navigator* aNavigator = 0;
242 std::vector<G4VPhysicalVolume*>::iterator pWorld =
243 std::find(fWorlds.begin(), fWorlds.end(), aWorld);
244 if (pWorld != fWorlds.end())
245 {
246 aNavigator = new G4Navigator();
247 aNavigator->SetWorldVolume(aWorld);
248 fNavigators.push_back(aNavigator);
249 }
250 else
251 {
252 G4String message
253 = "World volume with name -" + aWorld->GetName()
254 + "- does not exist. Create it first by GetParallelWorld() method!";
255 G4Exception("G4TransportationManager::GetNavigator(pointer)",
256 "GeomNav0002", FatalException, message);
257 }
258
259 return aNavigator;
260}

262// ----------------------------------------------------------------------------
263// DeRegisterNavigator()
264//
265// Provided a pointer to an already allocated navigator object, removes the
266// associated entry in the navigators collection (remove pair) but does not
267// delete the actual pointed object, which is still owned by the caller.
268// The navigator for tracking -cannot- be deregistered.
269//
270void G4TransportationManager::DeRegisterNavigator( G4Navigator* aNavigator )
271{
272 if (aNavigator == fNavigators[0])
273 {
274 G4Exception("G4TransportationManager::DeRegisterNavigator()",
275 "GeomNav0003", FatalException,
276 "The navigator for tracking CANNOT be deregistered!");
277 }
278 std::vector<G4Navigator*>::iterator pNav =
279 std::find(fNavigators.begin(), fNavigators.end(), aNavigator);
280 if (pNav != fNavigators.end())
281 {
282 // Deregister associated world volume
283 //
284 DeRegisterWorld((*pNav)->GetWorldVolume());
285
286 // Deregister the navigator
287 //
288 fNavigators.erase(pNav);
289 }
290 else
291 {
292 G4String message
293 = "Navigator for volume -" + aNavigator->GetWorldVolume()->GetName()
294 + "- not found in memory!";
295 G4Exception("G4TransportationManager::DeRegisterNavigator()",
296 "GeomNav1002", JustWarning, message);
297 }
298}

300// ----------------------------------------------------------------------------
301// ActivateNavigator()
302//
303// Provided a pointer to an already allocated navigator object, set to 'true'
304// the associated activation flag for the navigator in the collection.
305// If the provided navigator is not already registered, issue a warning
306// Return the index of the activated navigator. This index should be used for
307// ComputeStep() method of G4PathFinder.
308//
309G4int G4TransportationManager::ActivateNavigator( G4Navigator* aNavigator )
310{
311 std::vector<G4Navigator*>::iterator pNav =
312 std::find(fNavigators.begin(), fNavigators.end(), aNavigator);
313 if (pNav == fNavigators.end())
314 {
315 G4String message
316 = "Navigator for volume -" + aNavigator->GetWorldVolume()->GetName()
317 + "- not found in memory!";
318 G4Exception("G4TransportationManager::ActivateNavigator()",
319 "GeomNav1002", FatalException, message);
320 return -1;
321 }
322
323 aNavigator->Activate(true);
324 G4int id = 0;
325 std::vector<G4Navigator*>::iterator pActiveNav;
326 for(pActiveNav=fActiveNavigators.begin();
327 pActiveNav!=fActiveNavigators.end(); pActiveNav++)
328 {
329 if (*pActiveNav == aNavigator) { return id; }
330 id++;
331 }
332
333 fActiveNavigators.push_back(aNavigator);
334 return id;
335}
336
337// ----------------------------------------------------------------------------
338// DeActivateNavigator()
339//
340// Provided a pointer to an already allocated navigator object, set to 'false'
341// the associated activation flag in the navigators collection.
342// If the provided navigator is not already registered, issue a warning.
343//
344void G4TransportationManager::DeActivateNavigator( G4Navigator* aNavigator )
345{
346 std::vector<G4Navigator*>::iterator pNav =
347 std::find(fNavigators.begin(), fNavigators.end(), aNavigator);
348 if (pNav != fNavigators.end())
349 {
350 (*pNav)->Activate(false);
351 }
352 else
353 {
354 G4String message
355 = "Navigator for volume -" + aNavigator->GetWorldVolume()->GetName()
356 + "- not found in memory!";
357 G4Exception("G4TransportationManager::DeActivateNavigator()",
358 "GeomNav1002", JustWarning, message);
359 }
360
361 std::vector<G4Navigator*>::iterator pActiveNav =
362 std::find(fActiveNavigators.begin(), fActiveNavigators.end(), aNavigator);
363 if (pActiveNav != fActiveNavigators.end())
364 {
365 fActiveNavigators.erase(pActiveNav);
366 }
367}

369// ----------------------------------------------------------------------------
370// InactivateAll()
371//
372// Inactivate all the navigators except for the tracking one, and clear the
373// store of active navigators.
374//
375void G4TransportationManager::InactivateAll( )
376{
377 std::vector<G4Navigator*>::iterator pNav;
378 for (pNav=fActiveNavigators.begin(); pNav!=fActiveNavigators.end(); pNav++)
379 {
380 (*pNav)->Activate(false);
381 }
382 fActiveNavigators.clear();
383
384 // Restore status for the navigator for tracking
385 //
386 fNavigators[0]->Activate(true);
387 fActiveNavigators.push_back(fNavigators[0]);
388}

390// ----------------------------------------------------------------------------
391// IsWorldExisting()
392//
393// Verify existance or not of an istance of the world volume with
394// same name in the collection. Return the world pointer if existing.
395//
396G4VPhysicalVolume*
397G4TransportationManager::IsWorldExisting ( const G4String& name )
398{
399 std::vector<G4VPhysicalVolume*>::iterator pWorld = fWorlds.begin();
400 if (*pWorld==0) { *pWorld=fNavigators[0]->GetWorldVolume(); }
401
402 for (pWorld=fWorlds.begin(); pWorld!=fWorlds.end(); pWorld++)
403 {
404 if ((*pWorld)->GetName() == name ) { return *pWorld; }
405 }
406 return 0;
407}

409// ----------------------------------------------------------------------------
410// RegisterWorld()
411//
412// Provided a pointer to an already allocated world object, check and add the
413// associated entry in the worlds collection. Return 'true' if registration
414// succeeds and the new entry is created.
415//
416G4bool G4TransportationManager::RegisterWorld( G4VPhysicalVolume* aWorld )
417{
418 G4bool done = false;
419
420 std::vector<G4VPhysicalVolume*>::iterator pWorld =
421 std::find(fWorlds.begin(), fWorlds.end(), aWorld);
422 if (pWorld == fWorlds.end())
423 {
424 fWorlds.push_back(aWorld);
425 done = true;
426 }
427 return done;
428}

430// ----------------------------------------------------------------------------
431// DeRegisterWorld()
432//
433// Provided a pointer to an already allocated world object, removes the
434// associated entry in the worlds collection but does not delete the actual
435// pointed object, which is still owned by the caller.
436//
437void G4TransportationManager::DeRegisterWorld( G4VPhysicalVolume* aWorld )
438{
439 std::vector<G4VPhysicalVolume*>::iterator pWorld =
440 std::find(fWorlds.begin(), fWorlds.end(), aWorld);
441 if (pWorld != fWorlds.end())
442 {
443 fWorlds.erase(pWorld);
444 }
445 else
446 {
447 G4String message
448 = "World volume -" + aWorld->GetName() + "- not found in memory!";
449 G4Exception("G4TransportationManager::DeRegisterWorld()",
450 "GeomNav1002", JustWarning, message);
451 }
452}

454// ----------------------------------------------------------------------------
455// ClearParallelWorlds()
456//
457// Clear collection of navigators and delete allocated objects associated with
458// parallel worlds.
459// Called only by the RunManager when the entire geometry is rebuilt from
460// scratch.
461//
462void G4TransportationManager::ClearParallelWorlds()
463{
464 std::vector<G4Navigator*>::iterator pNav = fNavigators.begin();
465 G4Navigator* trackingNavigator = *pNav;
466 for (pNav=fNavigators.begin(); pNav!=fNavigators.end(); pNav++)
467 {
468 if (*pNav != trackingNavigator) { delete *pNav; }
469 }
470 fNavigators.clear();
471 fActiveNavigators.clear();
472 fWorlds.clear();
473
474 // trackingNavigator->SetWorldVolume(0);
475 fNavigators.push_back(trackingNavigator);
476 fActiveNavigators.push_back(trackingNavigator);
477 // fWorlds.push_back(trackingNavigator->GetWorldVolume()); // NULL registered
478 fWorlds.push_back(0); // NULL registered
479}
