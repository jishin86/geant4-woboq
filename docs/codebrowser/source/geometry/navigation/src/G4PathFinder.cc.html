75// ----------------------------------------------------------------------------
76// Constructor
77//
78G4PathFinder::G4PathFinder()
79 : fEndState( G4ThreeVector(), G4ThreeVector(), 0., 0., 0., 0., 0.),
80 fFieldExertedForce(false),
81 fRelocatedPoint(false),
82 fLastStepNo(-1), fCurrentStepNo(-1),
83 fVerboseLevel(0)
84{
85 fpMultiNavigator= new G4MultiNavigator();
86
87 fpTransportManager= G4TransportationManager::GetTransportationManager();
88 fpFieldPropagator = fpTransportManager->GetPropagatorInField();
89
90 kCarTolerance = G4GeometryTolerance::GetInstance()->GetSurfaceTolerance();
91
92 fNoActiveNavigators= 0;
93 G4ThreeVector Big3Vector( kInfinity, kInfinity, kInfinity );
94 fLastLocatedPosition= Big3Vector;
95 fSafetyLocation= Big3Vector;
96 fPreSafetyLocation= Big3Vector;
97 fPreStepLocation= Big3Vector;
98
99 fPreSafetyMinValue= -1.0;
100 fMinSafety_PreStepPt= -1.0;
101 fMinSafety_atSafLocation= -1.0;
102 fMinStep= -1.0;
103 fTrueMinStep= -1.0;
104 fPreStepCenterRenewed= false;
105 fNewTrack= false;
106 fNoGeometriesLimiting= 0;
107
108 for( G4int num=0; num< fMaxNav; ++num )
109 {
110 fpNavigator[num] = 0;
111 fLimitTruth[num] = false;
112 fLimitedStep[num] = kUndefLimited;
113 fCurrentStepSize[num] = -1.0;
114 fLocatedVolume[num] = 0;
115 fPreSafetyValues[num]= -1.0;
116 fCurrentPreStepSafety[num] = -1.0;
117 fNewSafetyComputed[num]= -1.0;
118 }
119}

132void
133G4PathFinder::EnableParallelNavigation(G4bool enableChoice)
134{
135 G4Navigator *navigatorForPropagation=0, *massNavigator=0;
136
137 massNavigator= fpTransportManager->GetNavigatorForTracking();
138 if( enableChoice )
139 {
140 navigatorForPropagation= fpMultiNavigator;
141
142 // Enable SafetyHelper to use PF
143 //
144 fpTransportManager->GetSafetyHelper()->EnableParallelNavigation(true);
145 }
146 else
147 {
148 navigatorForPropagation= massNavigator;
149
150 // Disable SafetyHelper to use PF
151 //
152 fpTransportManager->GetSafetyHelper()->EnableParallelNavigation(false);
153 }
154 fpFieldPropagator->SetNavigatorForPropagating(navigatorForPropagation);
155}

159G4double
160G4PathFinder::ComputeStep( const G4FieldTrack &InitialFieldTrack,
161 G4double proposedStepLength,
162 G4int navigatorNo,
163 G4int stepNo, // find next step
164 G4double &pNewSafety, // for this geom
165 ELimited &limitedStep,
166 G4FieldTrack &EndState,
167 G4VPhysicalVolume* currentVolume)
168{
169 G4double possibleStep= -1.0;
170
171#ifdef G4DEBUG_PATHFINDER
172 if( fVerboseLevel > 2 )
173 {
174 G4cout << " -------------------------" << G4endl;
175 G4cout << " G4PathFinder::ComputeStep - entered " << G4endl;
176 G4cout << " - stepNo = " << std::setw(4) << stepNo << " "
177 << " navigatorId = " << std::setw(2) << navigatorNo << " "
178 << " proposed step len = " << proposedStepLength << " " << G4endl;
179 G4cout << " PF::ComputeStep requested step "
180 << " from " << InitialFieldTrack.GetPosition()
181 << " dir " << InitialFieldTrack.GetMomentumDirection() << G4endl;
182 }
183#endif
184#ifdef G4VERBOSE
185 if( navigatorNo >= fNoActiveNavigators )
186 {
187 std::ostringstream message;
188 message << "Bad Navigator ID !" << G4endl
189 << " Requested Navigator ID = " << navigatorNo << G4endl
190 << " Number of active navigators = " << fNoActiveNavigators;
191 G4Exception("G4PathFinder::ComputeStep()", "GeomNav0002",
192 FatalException, message);
193 }
194#endif
195
196 if( fNewTrack || (stepNo != fLastStepNo) )
197 {
198 // This is a new track or a new step, so we must make the step
199 // ( else we can simply retrieve its results for this Navigator Id )
200
201 G4FieldTrack currentState= InitialFieldTrack;
202
203 fCurrentStepNo = stepNo;
204
205 // Check whether a process shifted the position
206 // since the last step -- by physics processes
207 //
208 G4ThreeVector newPosition = InitialFieldTrack.GetPosition();
209 G4ThreeVector moveVector= newPosition - fLastLocatedPosition;
210 G4double moveLenSq= moveVector.mag2();
211 if( moveLenSq > kCarTolerance * kCarTolerance )
212 {
213 G4ThreeVector newDirection = InitialFieldTrack.GetMomentumDirection();
214#ifdef G4DEBUG_PATHFINDER
215 if( fVerboseLevel > 2 )
216 {
217 G4double moveLen= std::sqrt( moveLenSq );
218 G4cout << " G4PathFinder::ComputeStep : Point moved since last step "
219 << " -- at step # = " << stepNo << G4endl
220 << " by " << moveLen << " to " << newPosition << G4endl;
221 }
222#endif
223 MovePoint(); // Unintentional changed -- ????
224
225 // Relocate to cope with this move -- else could abort !?
226 //
227 Locate( newPosition, newDirection );
228 }
229
230 // Check whether the particle have an (EM) field force exerting upon it
231 //
232 G4double particleCharge= currentState.GetCharge();
233
234 G4FieldManager* fieldMgr=0;
235 G4bool fieldExertsForce = false ;
236 if( (particleCharge != 0.0) )
237 {
238 fieldMgr= fpFieldPropagator->FindAndSetFieldManager( currentVolume );
239
240 // Protect for case where field manager has no field (= field is zero)
241 //
242 fieldExertsForce = (fieldMgr != 0)
243 && (fieldMgr->GetDetectorField() != 0);
244 }
245 fFieldExertedForce = fieldExertsForce; // Store for use in later calls
246 // referring to this 'step'.
247
248 fNoGeometriesLimiting= -1; // At start of track, no process limited step
249 if( fieldExertsForce )
250 {
251 DoNextCurvedStep( currentState, proposedStepLength, currentVolume );
252 //--------------
253 }else{
254 DoNextLinearStep( currentState, proposedStepLength );
255 //--------------
256 }
257 fLastStepNo= stepNo;
258 fRelocatedPoint= false;
259
260#ifdef G4DEBUG_PATHFINDER
261 if ( (fNoGeometriesLimiting < 0)
262 || (fNoGeometriesLimiting > fNoActiveNavigators) )
263 {
264 std::ostringstream message;
265 message << "Number of geometries limiting the step not set." << G4endl
266 << " Number of geometries limiting step = "
267 << fNoGeometriesLimiting;
268 G4Exception("G4PathFinder::ComputeStep()",
269 "GeomNav0002", FatalException, message);
270 }
271#endif
272 }
273#ifdef G4DEBUG_PATHFINDER
274 else
275 {
276 const G4double checkTolerance = 1.0e-9;
277 if( proposedStepLength < fTrueMinStep * ( 1.0 - checkTolerance) ) // For 2nd+ geometry
278 {
279 std::ostringstream message;
280 message.precision( 12 );
281 message << "Problem in step size request." << G4endl
282 << " Being requested to make a step which is shorter"
283 << " than the minimum Step " << G4endl
284 << " already computed for any Navigator/geometry during"
285 << " this tracking-step: " << G4endl
286 << " This could happen due to an error in process ordering."
287 << G4endl
288 << " Check that all physics processes are registered"
289 << " before all processes with a navigator/geometry."
290 << G4endl
291 << " If using pre-packaged physics list and/or"
292 << " functionality, please report this error."
293 << G4endl << G4endl
294 << " Additional information for problem: " << G4endl
295 << " Steps request/proposed = " << proposedStepLength
296 << G4endl
297 << " MinimumStep (true) = " << fTrueMinStep
298 << G4endl
299 << " MinimumStep (navraw) = " << fMinStep
300 << G4endl
301 << " Navigator raw return value" << G4endl
302 << " Requested step now = " << proposedStepLength
303 << G4endl
304 << " Difference min-req (absolute) = "
305 << fTrueMinStep-proposedStepLength << G4endl
306 << " Relative (to max of two) = "
307 << (fTrueMinStep-proposedStepLength)
308 / std::max(proposedStepLength, fTrueMinStep) << G4endl
309 << " -- Step info> stepNo= " << stepNo
310 << " last= " << fLastStepNo
311 << " newTr= " << fNewTrack << G4endl;
312 G4Exception("G4PathFinder::ComputeStep()",
313 "GeomNav0003", FatalException, message);
314 }
315 else
316 {
317 // This is neither a new track nor a new step -- just another
318 // client accessing information for the current track, step
319 // We will simply retrieve the results of the synchronous
320 // stepping for this Navigator Id below.
321 //
322 if( fVerboseLevel > 1 )
323 {
324 G4cout << " G4P::CS -> Not calling DoNextLinearStep: "
325 << " stepNo= " << stepNo << " last= " << fLastStepNo
326 << " new= " << fNewTrack << " Step already done" << G4endl;
327 }
328 }
329 }
330#endif
331
332 fNewTrack= false;
333
334 // Prepare the information to return
335
336 pNewSafety = fCurrentPreStepSafety[ navigatorNo ];
337 limitedStep = fLimitedStep[ navigatorNo ];
338
339 possibleStep= std::min(proposedStepLength, fCurrentStepSize[ navigatorNo ]);
340 EndState = fEndState; // now corrected for smaller step, if needed
341
342#ifdef G4DEBUG_PATHFINDER
343 if( fVerboseLevel > 0 )
344 {
345 G4cout << " G4PathFinder::ComputeStep returns "
346 << fCurrentStepSize[ navigatorNo ]
347 << " for Navigator " << navigatorNo
348 << " Limited step = " << limitedStep
349 << " Safety(mm) = " << pNewSafety / mm
350 << G4endl;
351 }
352#endif
353
354 return possibleStep;
355}

359void
360G4PathFinder::PrepareNewTrack( const G4ThreeVector& position,
361 const G4ThreeVector& direction,
362 G4VPhysicalVolume* massStartVol)
363{
364 // Key purposes:
365 // - Check and cache set of active navigators
366 // - Reset state for new track
367
368 G4int num=0;
369
370 EnableParallelNavigation(true);
371 // Switch PropagatorInField to use MultiNavigator
372
373 fpTransportManager->GetSafetyHelper()->InitialiseHelper();
374 // Reinitialise state of safety helper -- avoid problems with overlaps
375
376 fNewTrack= true;
377 this->MovePoint(); // Signal further that the last status is wiped
378
379 fpFieldPropagator->PrepareNewTrack(); // Inform field propagator of new track
380
381 // Message the G4NavigatorPanel / Dispatcher to find active navigators
382 //
383 std::vector<G4Navigator*>::iterator pNavigatorIter;
384
385 fNoActiveNavigators= fpTransportManager-> GetNoActiveNavigators();
386 if( fNoActiveNavigators > fMaxNav )
387 {
388 std::ostringstream message;
389 message << "Too many active Navigators / worlds." << G4endl
390 << " Transportation Manager has "
391 << fNoActiveNavigators << " active navigators." << G4endl
392 << " This is more than the number allowed = "
393 << fMaxNav << " !";
394 G4Exception("G4PathFinder::PrepareNewTrack()", "GeomNav0002",
395 FatalException, message);
396 }
397
398 fpMultiNavigator->PrepareNavigators();
399 //------------------------------------
400
401 pNavigatorIter= fpTransportManager->GetActiveNavigatorsIterator();
402 for( num=0; num< fNoActiveNavigators; ++pNavigatorIter,++num )
403 {
404 // Keep information in C-array ... for creating touchables - at least
405
406 fpNavigator[num] = *pNavigatorIter;
407 fLimitTruth[num] = false;
408 fLimitedStep[num] = kDoNot;
409 fCurrentStepSize[num] = 0.0;
410 fLocatedVolume[num] = 0;
411 }
412 fNoGeometriesLimiting= 0; // At start of track, no process limited step
413
414 // In case of one geometry, the tracking will have done the locating!!
415
416 if( fNoActiveNavigators > 1 )
417 {
418 Locate( position, direction, false );
419 }
420 else
421 {
422 // Update state -- depending on the tracking's call to Mass Navigator
423
424 fLastLocatedPosition= position;
425 fLocatedVolume[0]= massStartVol; // This information must be given
426 // by transportation
427 fLimitedStep[0] = kDoNot;
428 fCurrentStepSize[0] = 0.0;
429 }
430
431 // Reset Safety Information -- as in case of overlaps this can cause
432 // inconsistencies ...
433 //
434 fMinSafety_PreStepPt= fPreSafetyMinValue= fMinSafety_atSafLocation= 0.0;
435
436 for( num=0; num< fNoActiveNavigators; ++num )
437 {
438 fPreSafetyValues[num]= 0.0;
439 fNewSafetyComputed[num]= 0.0;
440 fCurrentPreStepSafety[num] = 0.0;
441 }
442
443 // The first location for each Navigator must be non-relative
444 // or else call ResetStackAndState() for each Navigator
445
446 fRelocatedPoint= false;
447}

458void G4PathFinder::ReportMove( const G4ThreeVector& OldVector,
459 const G4ThreeVector& NewVector,
460 const G4String& Quantity ) const
461{
462 G4ThreeVector moveVec = ( NewVector - OldVector );
463
464 std::ostringstream message;
465 message.precision(16);
466 message << "Endpoint moved between value returned by ComputeStep()"
467 << " and call to Locate(). " << G4endl
468 << " Change of " << Quantity << " is "
469 << moveVec.mag() / mm << " mm long" << G4endl
470 << " and its vector is "
471 << (1.0/mm) * moveVec << " mm " << G4endl
472 << " Endpoint of ComputeStep() was " << OldVector << G4endl
473 << " and current position to locate is " << NewVector;
474 G4Exception("G4PathFinder::ReportMove()", "GeomNav1002",
475 JustWarning, message);
476}

478void
479G4PathFinder::Locate( const G4ThreeVector& position,
480 const G4ThreeVector& direction,
481 G4bool relative)
482{
483 // Locate the point in each geometry
484
485 static const G4double movLenTol = 10*kCarTolerance*kCarTolerance;
486
487 std::vector<G4Navigator*>::iterator pNavIter=
488 fpTransportManager->GetActiveNavigatorsIterator();
489
490 G4ThreeVector lastEndPosition= fRelocatedPoint ?
491 fLastLocatedPosition : fEndState.GetPosition();
492 G4ThreeVector moveVec = ( position - lastEndPosition );
493 G4double moveLenSq= moveVec.mag2();
494 if( (!fNewTrack) && ( moveLenSq > movLenTol ) )
495 {
496 ReportMove( lastEndPosition, position,
497 " (End) Position / G4PathFinder::Locate" );
498 }
499 fLastLocatedPosition= position;
500
501#ifdef G4DEBUG_PATHFINDER
502 if( fVerboseLevel > 2 )
503 {
504 G4cout << G4endl;
505 G4cout << " G4PathFinder::Locate : entered " << G4endl;
506 G4cout << " -------------------- -------" << G4endl;
507 G4cout << " Locating at position " << position
508 << " with direction " << direction
509 << " relative= " << relative << G4endl;
510 if ( (fVerboseLevel > 1) || ( moveLenSq > 0.0) )
511 {
512 G4cout << " lastEndPosition = " << lastEndPosition
513 << " moveVec = " << moveVec
514 << " newTr = " << fNewTrack
515 << " relocated = " << fRelocatedPoint << G4endl;
516 }
517
518 G4cout << " Located at " << position ;
519 if( fNoActiveNavigators > 1 ) { G4cout << G4endl; }
520 }
521#endif
522
523 for ( G4int num=0; num< fNoActiveNavigators ; ++pNavIter,++num )
524 {
525 // ... who limited the step ....
526
527 if( fLimitTruth[num] ) { (*pNavIter)->SetGeometricallyLimitedStep(); }
528
529 G4VPhysicalVolume *pLocated=
530 (*pNavIter)->LocateGlobalPointAndSetup( position, &direction,
531 relative,
532 false);
533 // Set the state related to the location
534 //
535 fLocatedVolume[num] = pLocated;
536
537 // Clear state related to the step
538 //
539 fLimitedStep[num] = kDoNot;
540 fCurrentStepSize[num] = 0.0;
541
542#ifdef G4DEBUG_PATHFINDER
543 if( fVerboseLevel > 2 )
544 {
545 G4cout << " - In world " << num << " geomLimStep= " << fLimitTruth[num]
546 << " gives volume= " << pLocated ;
547 if( pLocated )
548 {
549 G4cout << " name = '" << pLocated->GetName() << "'";
550 G4cout << " - CopyNo= " << pLocated->GetCopyNo();
551 }
552 G4cout << G4endl;
553 }
554#endif
555 }
556
557 fRelocatedPoint= false;
558}

560void G4PathFinder::ReLocate( const G4ThreeVector& position )
561{
562 // Locate the point in each geometry
563
564 std::vector<G4Navigator*>::iterator pNavIter=
565 fpTransportManager->GetActiveNavigatorsIterator();
566
567#ifdef G4DEBUG_PATHFINDER
568
569 // Check that this relocation does not violate safety
570 // - at endpoint (computed from start point) AND
571 // - at last safety location (likely just called)
572
573 G4ThreeVector lastEndPosition= fEndState.GetPosition();
574
575 // Calculate end-point safety ...
576 //
577 G4double DistanceStartEnd= (lastEndPosition - fPreStepLocation).mag();
578 G4double endPointSafety_raw = fMinSafety_PreStepPt - DistanceStartEnd;
579 G4double endPointSafety_Est1 = std::max( 0.0, endPointSafety_raw );
580
581 // ... and check move from endpoint against this endpoint safety
582 //
583 G4ThreeVector moveVecEndPos = position - lastEndPosition;
584 G4double moveLenEndPosSq = moveVecEndPos.mag2();
585
586 // Check that move from endpoint of last step is within safety
587 // -- or check against last location or relocation ??
588 //
589 G4ThreeVector moveVecSafety= position - fSafetyLocation;
590 G4double moveLenSafSq= moveVecSafety.mag2();
591
592 G4double distCheckEnd_sq= ( moveLenEndPosSq - endPointSafety_Est1
593 *endPointSafety_Est1 );
594 G4double distCheckSaf_sq= ( moveLenSafSq - fMinSafety_atSafLocation
595 *fMinSafety_atSafLocation );
596
597 G4bool longMoveEnd = distCheckEnd_sq > 0.0;
598 G4bool longMoveSaf = distCheckSaf_sq > 0.0;
599
600 G4double revisedSafety= 0.0;
601
602 if( (!fNewTrack) && ( longMoveEnd && longMoveSaf ) )
603 {
604 // Recompute ComputeSafety for end position
605 //
606 revisedSafety= ComputeSafety(lastEndPosition);
607
608 const G4double kRadTolerance =
609 G4GeometryTolerance::GetInstance()->GetRadialTolerance();
610 const G4double cErrorTolerance=1e-12;
611 // Maximum relative error from roundoff of arithmetic
612
613 G4double distCheckRevisedEnd= moveLenEndPosSq-revisedSafety*revisedSafety;
614
615 G4bool longMoveRevisedEnd= ( distCheckRevisedEnd > 0. ) ;
616
617 G4double moveMinusSafety= 0.0;
618 G4double moveLenEndPosition= std::sqrt( moveLenEndPosSq );
619 moveMinusSafety = moveLenEndPosition - revisedSafety;
620
621 if ( longMoveRevisedEnd && (moveMinusSafety > 0.0 )
622 && ( revisedSafety > 0.0 ) )
623 {
624 // Take into account possibility of roundoff error causing
625 // this apparent move further than safety
626
627 if( fVerboseLevel > 0 )
628 {
629 G4cout << " G4PF:Relocate> Ratio to revised safety is "
630 << std::fabs(moveMinusSafety)/revisedSafety << G4endl;
631 }
632
633 G4double absMoveMinusSafety= std::fabs(moveMinusSafety);
634 G4bool smallRatio= absMoveMinusSafety < kRadTolerance * revisedSafety ;
635 G4double maxCoordPos = std::max(
636 std::max( std::fabs(position.x()),
637 std::fabs(position.y())),
638 std::fabs(position.z()) );
639 G4bool smallValue= absMoveMinusSafety < cErrorTolerance * maxCoordPos;
640 if( ! (smallRatio || smallValue) )
641 {
642 G4cout << " G4PF:Relocate> Ratio to revised safety is "
643 << std::fabs(moveMinusSafety)/revisedSafety << G4endl;
644 G4cout << " Difference of move and safety is not very small."
645 << G4endl;
646 }
647 else
648 {
649 moveMinusSafety = 0.0;
650 longMoveRevisedEnd = false; // Numerical issue -- not too long!
651
652 G4cout << " Difference of move & safety is very small in magnitude, "
653 << absMoveMinusSafety << G4endl;
654 if( smallRatio )
655 {
656 G4cout << " ratio to safety " << revisedSafety
657 << " is " << absMoveMinusSafety / revisedSafety
658 << "smaller than " << kRadTolerance << " of safety ";
659 }
660 else
661 {
662 G4cout << " as fraction " << absMoveMinusSafety / maxCoordPos
663 << " of position vector max-coord " << maxCoordPos
664 << " smaller than " << cErrorTolerance ;
665 }
666 G4cout << " -- reset moveMinusSafety to "
667 << moveMinusSafety << G4endl;
668 }
669 }
670
671 if ( longMoveEnd && longMoveSaf
672 && longMoveRevisedEnd && (moveMinusSafety>0.0) )
673 {
674 std::ostringstream message;
675 message.precision(9);
676 message << "ReLocation is further than end-safety value." << G4endl
677 << " Moved from last endpoint by " << moveLenEndPosition
678 << " compared to end safety (from preStep point) = "
679 << endPointSafety_Est1 << G4endl
680 << " --> last PreSafety Location was " << fPreSafetyLocation
681 << G4endl
682 << " safety value = " << fPreSafetyMinValue << G4endl
683 << " --> last PreStep Location was " << fPreStepLocation
684 << G4endl
685 << " safety value = " << fMinSafety_PreStepPt << G4endl
686 << " --> last EndStep Location was " << lastEndPosition
687 << G4endl
688 << " safety value = " << endPointSafety_Est1
689 << " raw-value = " << endPointSafety_raw << G4endl
690 << " --> Calling again at this endpoint, we get "
691 << revisedSafety << " as safety value." << G4endl
692 << " --> last position for safety " << fSafetyLocation
693 << G4endl
694 << " its safety value = " << fMinSafety_atSafLocation
695 << G4endl
696 << " move from safety location = "
697 << std::sqrt(moveLenSafSq) << G4endl
698 << " again= " << moveVecSafety.mag() << G4endl
699 << " safety - Move-from-end= "
700 << revisedSafety - moveLenEndPosition
701 << " (negative is Bad.)" << G4endl
702 << " Debug: distCheckRevisedEnd = "
703 << distCheckRevisedEnd;
704 ReportMove( lastEndPosition, position, "Position" );
705 G4Exception("G4PathFinder::ReLocate", "GeomNav0003",
706 FatalException, message);
707 }
708 }

823G4double
824G4PathFinder::DoNextLinearStep( const G4FieldTrack &initialState,
825 G4double proposedStepLength )
826{
827 std::vector<G4Navigator*>::iterator pNavigatorIter;

829 G4double minSafety= kInfinity, minStep= kInfinity;

850 // Potential optimisation using Maximum Value of safety!
851 // if( MagSqShift >= sqr(fPreSafetyMaxValue ) ){
852 // MagShift= kInfinity; // Not a useful value -- all will not use/ignore

864 fullSafety = 0.0 ;
865 }
866 else
867 {
868 fullSafety = fPreSafetyMinValue - MagShift;
869 }
870 if( proposedStepLength < fullSafety )

873 // -> so we do not have to move the safety center

875 fPreStepCenterRenewed= false;

877 for( num=0; num< fNoActiveNavigators; ++num )
878 {

886#ifdef G4DEBUG_PATHFINDER

894 }
895#endif
896 }
897 else
898#endif // End of G4PATHFINDER_OPTIMISATION 1
899 {
900 // Move is larger than at least one of the safeties
901 // -> so we must move the safety center!
902
903 fPreStepCenterRenewed= true;
904 pNavigatorIter= fpTransportManager-> GetActiveNavigatorsIterator();
905
906 minStep= kInfinity; // Not proposedStepLength;
907
908 for( num=0; num< fNoActiveNavigators; ++pNavigatorIter,++num )
909 {
910 safety = std::max( 0.0, fPreSafetyValues[num] - MagShift);
911
912#ifdef G4PATHFINDER_OPTIMISATION
913 if( proposedStepLength <= safety ) // Should be just < safety ?
914 {
915 // The Step is guaranteed to be taken
916
917 step= kInfinity; // ComputeStep Would return this
918
919#ifdef G4DEBUG_PATHFINDER
920 G4cout.precision(8);
921 G4cout << "PathFinder::ComputeStep> small proposed step = "
922 << proposedStepLength
923 << " <= safety = " << safety << " for nav " << num
924 << " Step fully taken. " << G4endl;
925#endif
926 }
927 else
928#endif // End of G4PATHFINDER_OPTIMISATION 2
929 {
930#ifdef G4DEBUG_PATHFINDER
931 G4double previousSafety= safety;
932#endif
933 step= (*pNavigatorIter)->ComputeStep( initialPosition,
934 initialDirection,
935 proposedStepLength,
936 safety );
937 minStep = std::min( step, minStep); // OLD ==> can be 'logical' value, ie. kInfinity
938
939#ifdef G4DEBUG_PATHFINDER
940 if( fVerboseLevel > 0)
941 {
942 G4cout.precision(8);
943 G4cout << "PathFinder::ComputeStep> long proposed step = "
944 << proposedStepLength
945 << " > safety = " << previousSafety
946 << " for nav " << num
947 << " . New safety = " << safety << " step= " << step
948 << G4endl;
949 }
950#endif
951 }
952 fCurrentStepSize[num] = step; // Raw value - can be kInfinity
953
954 // TODO: consider whether/how to reduce the proposed step
955 // to the latest minStep value - to reduce calculations.
956 // ( If so, much change 1st minimum above. )
957
958 // Save safety value, must be done for all geometries "together"
959 // (even if not recomputed using call to ComputeStep)
960 // since they share the fPreSafetyLocation
961
962 fPreSafetyValues[num]= safety;
963 fCurrentPreStepSafety[num]= safety;
964
965 minSafety= std::min( safety, minSafety );
966
967#ifdef G4DEBUG_PATHFINDER
968 if( fVerboseLevel > 2 )
969 {
970 G4cout << "G4PathFinder::DoNextLinearStep : Navigator ["
971 << num << "] -- step size " << step << G4endl;
972 }
973#endif
974 }
975
976 // Only change these when safety is recalculated
977 // it is good/relevant only for safety calculations
978
979 fPreSafetyLocation= initialPosition;
980 fPreSafetyMinValue= minSafety;
981 } // end of else for if( proposedStepLength <= fullSafety)
982
983 // For use in Relocation, need PreStep point location, min-safety
984 //
985 fPreStepLocation= initialPosition;
986 fMinSafety_PreStepPt= minSafety;
987
988 fMinStep= minStep;
989
990 if( fMinStep == kInfinity )
991 {
992 minStep = proposedStepLength; // Use this below for endpoint !!
993 }
994 fTrueMinStep = minStep;
995
996 // Set the EndState
997
998 G4ThreeVector endPosition;
999
1000 fEndState= initialState;
1001 endPosition= initialPosition + minStep * initialDirection ;
1002
1003#ifdef G4DEBUG_PATHFINDER
1004 if( fVerboseLevel > 1 )
1005 {
1006 G4int oldPrec= G4cout.precision(14);
1007 G4cout << "G4PathFinder::DoNextLinearStep : "

1010 G4cout.precision(oldPrec);
1011 }
1012#endif
1013

1016
1017 if( fNoActiveNavigators == 1 )
1018 {
1019 G4bool transportLimited = (fMinStep!= kInfinity);
1020 fLimitTruth[IdTransport] = transportLimited;
1021 fLimitedStep[IdTransport] = transportLimited ? kUnique : kDoNot;
1022
1023 // Set fNoGeometriesLimiting - as WhichLimited does
1024 fNoGeometriesLimiting = transportLimited ? 1 : 0;
1025 }
1026 else
1027 {
1028 WhichLimited();
1029 }
1030
1031#ifdef G4DEBUG_PATHFINDER
1032 if( fVerboseLevel > 2 )
1033 {
1034 G4cout << " G4PathFinder::DoNextLinearStep : exits returning "
1035 << minStep << G4endl;
1036 G4cout << " Endpoint values = " << fEndState << G4endl;
1037 G4cout << G4endl;
1038 }
1039#endif
1040
1041 return minStep;

1088 fLimitedStep[ last ] = kUnique;

1172
1173
1174 G4EquationOfMotion* equationOfMotion =
1175 fpFieldPropagator->GetCurrentEquationOfMotion();
1176
1177 equationOfMotion->SetChargeMomentumMass( *(initialState.GetChargeState()),
1178 initialState.GetMomentum().mag(),
1179 initialState.GetRestMass() );

1181#ifdef G4DEBUG_PATHFINDER

1188 }
1189#endif
1190
1191 fPreStepCenterRenewed= true; // Always update PreSafety with PreStep point
1192
1193 if( fNoActiveNavigators > 1 )
1194 {
1195 // Calculate the safety values before making the step
1196
1197 G4double minSafety= kInfinity, safety;
1198 for( numNav=0; numNav < fNoActiveNavigators; ++numNav )
1199 {
1200 safety= fpNavigator[numNav]->ComputeSafety( startPoint, DBL_MAX, false );
1201 fPreSafetyValues[numNav]= safety;
1202 fCurrentPreStepSafety[numNav]= safety;
1203 minSafety = std::min( safety, minSafety );
1204 }

1207
1208 fPreSafetyLocation= startPoint;
1209 fPreSafetyMinValue= minSafety;

1222 //
1223 fEndState= fieldTrack;
1224 fMinStep= minStep;
1225 fTrueMinStep = std::min( minStep, proposedStepLength );
1226
1227 if( fNoActiveNavigators== 1 )
1228 {
1229 // Update the 'PreSafety' sphere - as any ComputeStep was called
1230 // (must be done anyway in field)
1231

1233 fPreSafetyLocation= startPoint;
1234 fPreSafetyMinValue= newSafety;

1237 //

1240 fMinSafety_PreStepPt= newSafety;
1241 }
1242
1243#ifdef G4DEBUG_PATHFINDER

1247 << " initialState = " << initialState << G4endl

1251 << " proposedStepLength " << proposedStepLength
1252 << " safety = " << newSafety << G4endl;
1253 }
1254#endif
1255 G4double currentStepSize; // = 0.0;
1256 if( minStep < proposedStepLength ) // if == , then a boundary found at end ??
1257 {
1258 // Recover the remaining information from MultiNavigator
1259 // especially regarding which Navigator limited the step
1260
1261 G4int noLimited= 0; // No geometries limiting step
1262 for( numNav=0; numNav < fNoActiveNavigators; ++numNav )
1263 {
1264 G4double finalStep, lastPreSafety=0.0, minStepLast;
1265 ELimited didLimit;
1266 G4bool limited;
1267
1268 finalStep= fpMultiNavigator->ObtainFinalStep( numNav, lastPreSafety,
1269 minStepLast, didLimit );
1270
1271 // Calculate the step for this geometry, using the
1272 // final step (the only one which can differ.)
1273
1274 currentStepSize = fTrueMinStep;
1275 G4double diffStep= 0.0;
1276 if( (minStepLast != kInfinity) )
1277 {
1278 diffStep = (finalStep-minStepLast);
1279 if ( std::abs(diffStep) <= toleratedRelativeError * finalStep )
1280 {
1281 diffStep = 0.0;
1282 }
1283 currentStepSize += diffStep;
1284 }
1285 fCurrentStepSize[numNav] = currentStepSize;
1286
1287 // TODO: could refine the way to obtain safeties for > 1 geometries
1288 // - for pre step safety
1289 // notify MultiNavigator about new set of sub-steps
1290 // allow it to return this value in ObtainFinalStep
1291 // instead of lastPreSafety (or as well?)
1292 // - for final step start (available)
1293 // get final Step start from MultiNavigator
1294 // and corresponding safety values
1295 // and/or ALSO calculate ComputeSafety at endpoint
1296 // endSafety= fpNavigator[numNav]->ComputeSafety( endPoint );
1297
1298 fLimitedStep[numNav] = didLimit;
1299 fLimitTruth[numNav] = limited = (didLimit != kDoNot );
1300 if( limited ) { noLimited++; }
1301
1302#ifdef G4DEBUG_PATHFINDER
1303 G4bool StepError= (currentStepSize < 0)
1304 || ( (minStepLast != kInfinity) && (diffStep < 0) ) ;
1305 if( StepError || (fVerboseLevel > 2) )
1306 {
1307 G4String limitedString= LimitedString( fLimitedStep[numNav] );
1308
1309 G4cout << " G4PathFinder::ComputeStep. Geometry " << numNav
1310 << " step= " << fCurrentStepSize[numNav]
1311 << " from final-step= " << finalStep
1312 << " fTrueMinStep= " << fTrueMinStep
1313 << " minStepLast= " << minStepLast
1314 << " limited = " << (fLimitTruth[numNav] ? "YES" : " NO")
1315 << " ";
1316 G4cout << " status = " << limitedString << " #= " << didLimit
1317 << G4endl;
1318
1319 if( StepError )
1320 {
1321 std::ostringstream message;
1322 message << "Incorrect calculation of step size for one navigator"
1323 << G4endl
1324 << " currentStepSize = " << currentStepSize
1325 << ", diffStep= " << diffStep << G4endl
1326 << "ERROR in computing step size for this navigator.";
1327 G4Exception("G4PathFinder::DoNextCurvedStep",
1328 "GeomNav0003", FatalException, message);
1329 }
1330 }
1331#endif
1332 } // for num Navigators
1333
1334 fNoGeometriesLimiting= noLimited; // Save # processes limiting step
1335 }
1336 else if ( (minStep == proposedStepLength)
1337 || (minStep == kInfinity)
1338 || ( std::abs(minStep-proposedStepLength)
1339 < toleratedRelativeError * proposedStepLength ) )
1340 {
1341 // In case the step was not limited, use default responses
1342 // --> all Navigators
1343 // Also avoid problems in case of PathFinder using safety to optimise
1344 // - it is possible that the Navigators were not called
1345 // if the safety was already satisfactory.
1346 // (In that case calling ObtainFinalStep gives invalid results.)
1347
1348 currentStepSize= minStep;
1349 for( numNav=0; numNav < fNoActiveNavigators; ++numNav )
1350 {
1351 fCurrentStepSize[numNav] = minStep;
1352 // Safety for endpoint ?? // Can eventuall improve it -- see TODO above
1353 fLimitedStep[numNav] = kDoNot;
1354 fLimitTruth[numNav] = false;
1355 }
1356 fNoGeometriesLimiting= 0; // Save # processes limiting step
1357 }
1358 else // (minStep > proposedStepLength) and not (minStep == kInfinity)
1359 {
1360 std::ostringstream message;
1361 message << "Incorrect calculation of step size for one navigator." << G4endl
1362 << " currentStepSize = " << minStep << " is larger than "
1363 << " proposed StepSize = " << proposedStepLength << ".";
1364 G4Exception("G4PathFinder::DoNextCurvedStep()",
1365 "GeomNav0003", FatalException, message);
1366 }
1367
1368#ifdef G4DEBUG_PATHFINDER
1369 if( fVerboseLevel > 2 )
1370 {
1371 G4cout << " Exiting G4PathFinder::DoNextCurvedStep " << G4endl;
1372 PrintLimited();
1373 }
1374 G4cout.precision(prc);
1375#endif
1376
1377 return minStep;
1378}
