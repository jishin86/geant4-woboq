29// class G4MultiNavigator
30//
31// Class description:
32//
33// Utility class for polling the navigators of several geometries to
34// identify the next boundary.
35
36// History:
37// - Created. John Apostolakis, November 2006

40#ifndef G4MULTINAVIGATOR_HH
41#define G4MULTINAVIGATOR_HH
42
43#include <iostream>
44
45#include "geomdefs.hh"
46#include "G4ThreeVector.hh"
47#include "G4Navigator.hh"
48
49#include "G4TouchableHistoryHandle.hh"
50
51#include "G4NavigationHistory.hh"
52
53enum ELimited { kDoNot,kUnique,kSharedTransport,kSharedOther,kUndefLimited };
54
55class G4TransportationManager;
56class G4VPhysicalVolume;
57
58class G4MultiNavigator : public G4Navigator
59{
60 public: // with description
61
62 friend std::ostream& operator << (std::ostream &os, const G4Navigator &n);
63
64 G4MultiNavigator();
65 // Constructor - initialisers and setup.
66
67 ~G4MultiNavigator();
68 // Destructor. No actions.
69
70 G4double ComputeStep(const G4ThreeVector &pGlobalPoint,
71 const G4ThreeVector &pDirection,
72 const G4double pCurrentProposedStepLength,
73 G4double &pNewSafety);
74 // Return the distance to the next boundary of any geometry
75
76 G4double ObtainFinalStep( G4int navigatorId,
77 G4double &pNewSafety, // for this geom
78 G4double &minStepLast,
79 ELimited &limitedStep);
80 // Get values for a single geometry
81
82 void PrepareNavigators();
83 // Find which geometries are registered for this particles, and keep info
84 void PrepareNewTrack( const G4ThreeVector position,
85 const G4ThreeVector direction );
86 // Prepare Navigators and locate
87
88 G4VPhysicalVolume* ResetHierarchyAndLocate(const G4ThreeVector &point,
89 const G4ThreeVector &direction,
90 const G4TouchableHistory &h);
91 // Reset the geometrical hierarchy for all geometries.
92 // Use the touchable history for the first (mass) geometry.
93 // Return the volume in the first (mass) geometry.
94 //
95 // Important Note: In order to call this the geometries MUST be closed.
96
97 G4VPhysicalVolume* LocateGlobalPointAndSetup(const G4ThreeVector& point,
98 const G4ThreeVector* direction=0,
99 const G4bool pRelativeSearch=true,
100 const G4bool ignoreDirection=true);
101 // Locate in all geometries.
102 // Return the volume in the first (mass) geometry
103 // Maintain vector of other volumes, to be returned separately
104 //
105 // Important Note: In order to call this the geometry MUST be closed.
106
107 void LocateGlobalPointWithinVolume(const G4ThreeVector& position);
108 // Relocate in all geometries for point that has not changed volume
109 // (ie is within safety in all geometries or is distance less that
110 // along the direction of a computed step.
111
112 G4double ComputeSafety(const G4ThreeVector &globalpoint,
113 const G4double pProposedMaxLength = DBL_MAX,
114 const G4bool keepState = false);
115 // Calculate the isotropic distance to the nearest boundary
116 // in any geometry from the specified point in the global coordinate
117 // system. The geometry must be closed.
118
119 G4TouchableHistoryHandle CreateTouchableHistoryHandle() const;
120 // Returns a reference counted handle to a touchable history.
121
122 virtual G4ThreeVector GetLocalExitNormal(G4bool* obtained); // const
123 virtual G4ThreeVector GetLocalExitNormalAndCheck(const G4ThreeVector &CurrentE_Point,
124 G4bool* obtained); // const
125 virtual G4ThreeVector GetGlobalExitNormal(const G4ThreeVector &CurrentE_Point,
126 G4bool* obtained); // const
127 // Return Exit Surface Normal and validity too.
128 // Can only be called if the Navigator's last Step either
129 // - has just crossed a volume geometrical boundary and relocated, or
130 // - has arrived at a boundary in a ComputeStep
131 // It returns the Normal to the surface pointing out of the volume that
132 // was left behind and/or into the volume that was entered.
133 // Convention:x
134 // The *local* normal is in the coordinate system of the *final* volume.
135 // Restriction:
136 // Normals are not available for replica volumes (returns obtained= false)
137
138 public: // without description
139
140 G4Navigator* GetNavigator(G4int n) const
141 {
142 if( (n>fNoActiveNavigators)||(n<0)){ n=0; }
143 return fpNavigator[n];
144 }
145
146 protected: // with description
147
148 void ResetState();
149 // Utility method to reset the navigator state machine.
150
151 void SetupHierarchy();
152 // Renavigate & reset hierarchy described by current history
153 // o Reset volumes
154 // o Recompute transforms and/or solids of replicated/parameterised
155 // volumes.
156
157 void WhichLimited(); // Flag which processes limited the step
158 void PrintLimited(); // Auxiliary, debugging printing
159 void CheckMassWorld();
160
161 private:
162
163 // STATE Information
164
165 G4int fNoActiveNavigators;
166 static const G4int fMaxNav = 16; // rename to kMaxNoNav ??
167 G4VPhysicalVolume* fLastMassWorld;
168
169 // Global state (retained during stepping for one track
170 G4Navigator* fpNavigator[fMaxNav]; // G4Navigator** fpNavigator;
171
172 // State after a step computation
173 ELimited fLimitedStep[fMaxNav];
174 G4bool fLimitTruth[fMaxNav];
175 G4double fCurrentStepSize[fMaxNav];
176 G4double fNewSafety[ fMaxNav ]; // Safety for starting point
177 G4int fNoLimitingStep; // How many geometries limited the step
178 G4int fIdNavLimiting; // Id of Navigator limiting step (if only one limits)
179
180 // Lowest values - determine step length, and safety
181 G4double fMinStep; // As reported by Navigators. Can be kInfinity
182 G4double fMinSafety;
183 G4double fTrueMinStep; // Corrected in case fMinStep >= proposed
184
185 // State after calling 'locate'
186 G4VPhysicalVolume* fLocatedVolume[fMaxNav];
187 G4ThreeVector fLastLocatedPosition;
188
189 // cache of safety information
190 G4ThreeVector fSafetyLocation; // point where ComputeSafety is called
191 G4double fMinSafety_atSafLocation; // /\ corresponding value of safety
192 G4ThreeVector fPreStepLocation; // point where last ComputeStep called
193 G4double fMinSafety_PreStepPt; // /\ corresponding value of safety
194
195 G4TransportationManager* pTransportManager; // Cache for frequent use
196};
197
198#endif
