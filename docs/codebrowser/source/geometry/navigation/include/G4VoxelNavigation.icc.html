33// ********************************************************************
34// VoxelLocate
35// ********************************************************************
36//
37inline
38G4SmartVoxelNode*
39G4VoxelNavigation::VoxelLocate( G4SmartVoxelHeader* pHead,
40 const G4ThreeVector& localPoint )
41{
42 G4SmartVoxelHeader *targetVoxelHeader=pHead;
43 G4SmartVoxelNode *targetVoxelNode=0;
44 G4SmartVoxelProxy *sampleProxy;
45 EAxis targetHeaderAxis;
46 G4double targetHeaderMin, targetHeaderNodeWidth;
47 G4int targetHeaderNoSlices, targetNodeNo;
48
49 fVoxelDepth = 0;
50
51 while ( !targetVoxelNode )
52 {
53 targetHeaderAxis = targetVoxelHeader->GetAxis();
54 targetHeaderNoSlices = targetVoxelHeader->GetNoSlices();
55 targetHeaderMin = targetVoxelHeader->GetMinExtent();
56 targetHeaderNodeWidth = (targetVoxelHeader->GetMaxExtent()-targetHeaderMin)
57 / targetHeaderNoSlices;
58 targetNodeNo = G4int( (localPoint(targetHeaderAxis)-targetHeaderMin)
59 / targetHeaderNodeWidth);
60 // Rounding protection
61 //
62 if ( targetNodeNo<0 )
63 {
64 targetNodeNo = 0;
65 }
66 else if ( targetNodeNo>=targetHeaderNoSlices )
67 {
68 targetNodeNo = targetHeaderNoSlices-1;
69 }
70 // Stack info for stepping
71 //
72 fVoxelAxisStack[fVoxelDepth] = targetHeaderAxis;
73 fVoxelNoSlicesStack[fVoxelDepth] = targetHeaderNoSlices;
74 fVoxelSliceWidthStack[fVoxelDepth] = targetHeaderNodeWidth;
75 fVoxelNodeNoStack[fVoxelDepth] = targetNodeNo;
76 fVoxelHeaderStack[fVoxelDepth] = targetVoxelHeader;
77 sampleProxy = targetVoxelHeader->GetSlice(targetNodeNo);
78
79 if ( sampleProxy->IsNode() )
80 {
81 targetVoxelNode = sampleProxy->GetNode();
82 }
83 else
84 {
85 targetVoxelHeader = sampleProxy->GetHeader();
86 fVoxelDepth++;
87 }
88 }
89 fVoxelNode = targetVoxelNode;
90 return targetVoxelNode;
91}

93// ********************************************************************
94// LevelLocate
95// ********************************************************************
96//
97inline
98G4bool
99G4VoxelNavigation::LevelLocate( G4NavigationHistory& history,
100 const G4VPhysicalVolume* blockedVol,
101 const G4int,
102 const G4ThreeVector& globalPoint,
103 const G4ThreeVector* globalDirection,
104 const G4bool pLocatedOnEdge,
105 G4ThreeVector& localPoint )
106{
107 G4SmartVoxelHeader *targetVoxelHeader;
108 G4SmartVoxelNode *targetVoxelNode;
109 G4VPhysicalVolume *targetPhysical, *samplePhysical;
110 G4LogicalVolume *targetLogical;
111 G4VSolid *sampleSolid;
112 G4ThreeVector samplePoint;
113 G4int targetNoDaughters;
114
115 targetPhysical = history.GetTopVolume();
116 targetLogical = targetPhysical->GetLogicalVolume();
117 targetVoxelHeader = targetLogical->GetVoxelHeader();
118
119 // Find the voxel containing the point
120 //
121 targetVoxelNode = VoxelLocate(targetVoxelHeader,localPoint);
122
123 targetNoDaughters=targetVoxelNode->GetNoContained();
124 if ( targetNoDaughters==0 ) return false;
125
126 //
127 // Search daughters in volume
128 //
129
130 for ( G4int sampleNo=targetNoDaughters-1; sampleNo>=0; sampleNo-- )
131 {
132 samplePhysical = targetLogical->
133 GetDaughter(targetVoxelNode->GetVolume(sampleNo));
134 if ( samplePhysical!=blockedVol )
135 {
136 // Setup history
137 //
138 history.NewLevel(samplePhysical, kNormal, samplePhysical->GetCopyNo());
139 sampleSolid = samplePhysical->GetLogicalVolume()->GetSolid();
140 samplePoint = history.GetTopTransform().TransformPoint(globalPoint);
141
142 if( G4AuxiliaryNavServices::CheckPointOnSurface(sampleSolid,
143 samplePoint,
144 globalDirection,
145 history.GetTopTransform(),
146 pLocatedOnEdge) )
147 {
148 // Enter this daughter
149 //
150 localPoint = samplePoint;
151 return true;
152 }
153 else
154 {
155 history.BackLevel();
156 }
157 }
158 }
159 return false;
160}
