29// class G4NavigationLogger
30//
31// Class description:
32//
33// Simple utility class for use by navigation systems
34// for verbosity and check-mode.
35
36// History:
37// - Created. Gabriele Cosmo, November 2010
38// --------------------------------------------------------------------
39#ifndef G4NAVIGATIONLOGGER_HH
40#define G4NAVIGATIONLOGGER_HH
41
42#include "G4NavigationHistory.hh"
43#include "G4VPhysicalVolume.hh"
44#include "G4LogicalVolume.hh"
45#include "G4VSolid.hh"
46#include "G4ThreeVector.hh"
47
48class G4NavigationLogger
49{
50 public: // with description
51
52 G4NavigationLogger(const G4String& id);
53 ~G4NavigationLogger();
54
55 void PreComputeStepLog (const G4VPhysicalVolume* motherPhysical,
56 G4double motherSafety,
57 const G4ThreeVector& localPoint) const;
58 // Report about first check - mother safety
59
60 void AlongComputeStepLog(const G4VSolid* sampleSolid,
61 const G4ThreeVector& samplePoint,
62 const G4ThreeVector& sampleDirection,
63 const G4ThreeVector& localDirection,
64 G4double sampleSafety,
65 G4double sampleStep) const;
66 // Report about a candidate daughter
67
68 void CheckDaughterEntryPoint(const G4VSolid* sampleSolid,
69 const G4ThreeVector& samplePoint,
70 const G4ThreeVector& sampleDirection,
71 const G4VSolid* motherSolid,
72 const G4ThreeVector& localPoint,
73 const G4ThreeVector& localDirection,
74 G4double motherStep,
75 G4double sampleStep) const;
76 // Check suspicious distance to a candidate daughter
77
78 void PostComputeStepLog (const G4VSolid* motherSolid,
79 const G4ThreeVector& localPoint,
80 const G4ThreeVector& localDirection,
81 G4double motherStep,
82 G4double motherSafety) const;
83 // Report exit distance from mother
84
85 void ComputeSafetyLog (const G4VSolid* solid,
86 const G4ThreeVector& point,
87 G4double safety,
88 G4bool isMotherVolume, // For labeling
89 G4int banner= -1) const;
90 // Report about safety computation (daughter?)
91
92 void PrintDaughterLog (const G4VSolid* sampleSolid,
93 const G4ThreeVector& samplePoint,
94 G4double sampleSafety,
95 G4bool onlySafety,
96 const G4ThreeVector& sampleDirection,
97 G4double sampleStep ) const;
98 // Report about a new minimum distance to candidate daughter
99
100 G4bool CheckAndReportBadNormal(const G4ThreeVector& unitNormal,
101 const G4ThreeVector& localPoint,
102 const G4ThreeVector& localDirection,
103 G4double step,
104 const G4VSolid* solid,
105 const char* msg ) const;
106 // Report issue with normal from Solid - for ComputeStep()
107
108 G4bool CheckAndReportBadNormal(const G4ThreeVector& unitNormal,
109 const G4ThreeVector& originalNormal,
110 const G4RotationMatrix& rotationM,
111 const char* msg ) const;
112 // Report issue with normal from Rotation - for ComputeStep()
113
114 void ReportOutsideMother(const G4ThreeVector& localPoint,
115 const G4ThreeVector& localDirection,
116 const G4VPhysicalVolume* motherPV,
117 G4double tDist = 30.0*CLHEP::cm ) const;
118 // Report if point wrongly located outside mother volume
119
120 void ReportVolumeAndIntersection( std::ostream& ostrm,
121 const G4ThreeVector& localPoint,
122 const G4ThreeVector& localDirection,
123 const G4VPhysicalVolume* physical ) const;
124 // Auxiliary method to report information about volume & position/direction
125
126 public: // without description
127
128 inline G4int GetVerboseLevel() const { return fVerbose; }
129 inline void SetVerboseLevel(G4int level) { fVerbose = level; }
130
131 inline G4double GetMinTriggerDistance() const { return fMinTriggerDistance; }
132 inline void SetMinTriggerDistance(G4double d) { fMinTriggerDistance= d; }
133 inline G4bool GetReportSoftWarnings() const { return fReportSoftWarnings; }
134 inline void SetReportSoftWarnings(G4bool b) { fReportSoftWarnings = b; }
135 private:
136
137 G4String fId; // Navigation type
138 G4int fVerbose; // Verbosity level
139 G4double fMinTriggerDistance; // Errors beyond this distance are fatal in ReportMother
140 G4bool fReportSoftWarnings; // Flag> Whether to warn about small issues
141};
142
143#endif
