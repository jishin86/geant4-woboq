29// class G4PhantomParameterisation
30//
31// Class description:
32//
33// Describes regular parameterisations: a set of boxes of equal dimension
34// in the x, y and z dimensions. The G4PVParameterised volume using this
35// class must be placed inside a volume that is completely filled by these
36// boxes.
37
38// History:
39// - Created. P. Arce, May 2007
40// *********************************************************************
41

161 private:

166

169
170 protected:
171
172 G4double fVoxelHalfX,fVoxelHalfY,fVoxelHalfZ;
173 // Half dimension of voxels (assume they are boxes).
174 size_t fNoVoxelX,fNoVoxelY,fNoVoxelZ;
175 // Number of voxel in x, y and z dimensions.
176 size_t fNoVoxelXY;
177 // Number of voxels in x times number of voxels in y (for speed-up).
178 size_t fNoVoxel;
179 // Total number of voxels (for speed-up).
180
181 std::vector<G4Material*> fMaterials;
182 // List of materials of the voxels.
183 size_t* fMaterialIndices;
184 // Index in fMaterials that correspond to each voxel.
185
186 G4VSolid* fContainerSolid;
187 // Save as container solid the parent of the voxels.
188 // Check that the voxels fill it completely.
189
190 G4double fContainerWallX, fContainerWallY, fContainerWallZ;
191 // Save position of container wall for speed-up.
192
193 G4double kCarTolerance;
194 // Relative surface tolerance.
195
196 G4bool bSkipEqualMaterials;
197 // Flag to skip surface when two voxel have same material or not
198};
