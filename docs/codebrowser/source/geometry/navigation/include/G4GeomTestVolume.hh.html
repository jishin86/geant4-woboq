31// G4GeomTestVolume
32//
33// Class description:
34//
35// Checks for inconsistencies in the geometric boundaries of a physical
36// volume and the boundaries of all its immediate daughters.
37
38// Author: G.Cosmo, CERN
39// --------------------------------------------------------------------
40#ifndef G4GeomTestVolume_hh
41#define G4GeomTestVolume_hh
42
43#include "G4ThreeVector.hh"
44
45class G4VPhysicalVolume;
46class G4GeomTestLogger;
47
48class G4GeomTestVolume
49{
50 public: // with description
51
52 G4GeomTestVolume( G4VPhysicalVolume *theTarget,
53 G4double theTolerance=0.0, // mm
54 G4int numberOfPoints=10000,
55 G4bool theVerbosity=true);
56 ~G4GeomTestVolume();
57 // Constructor and destructor
58
59 G4double GetTolerance() const;
60 void SetTolerance(G4double tolerance);
61 // Get/Set error tolerance (default set to 0*mm)
62 G4int GetResolution() const;
63 void SetResolution(G4int points);
64 // Get/Set number of points to check (default set to 10000)
65 G4bool GetVerbosity() const;
66 void SetVerbosity(G4bool verbosity);
67 // Get/Set verbosity mode (default set to true)
68 G4int GetErrorsThreshold() const;
69 void SetErrorsThreshold(G4int max);
70 // Get/Set maximum number of errors to report (default set to 1)
71
72 void TestRecursiveOverlap( G4int sLevel=0, G4int depth=-1 );
73 // Activate overlaps check, propagating recursively to the daughters,
74 // with possibility of specifying the initial level in the volume tree
75 // and the depth (default is the whole tree).
76 // Be careful: depending on the complexity of the geometry, this
77 // could require long computational time
78
79 private:
80
81 G4VPhysicalVolume *target; // Target volume
82 G4double tolerance; // Error tolerance
83 G4int resolution; // Number of points to test
84 G4int maxErr; // Maximum number of errors to report
85 G4bool verbosity; // Verbosity level for overlaps check
86};
87
88#endif
