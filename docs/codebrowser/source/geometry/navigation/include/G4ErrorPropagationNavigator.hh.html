33// Class Description:
34//
35// Class for performing double navigation in the detector geometry and
36// on the target surface for error propagation. It overloads ComputeStep()
37// and ComputeSafety() methods.
38
39// History:
40// - Created. P. Arce, September 2004
41// --------------------------------------------------------------------
42
43#ifndef G4ErrorPropagationNavigator_hh
44#define G4ErrorPropagationNavigator_hh
45
46#include "G4Navigator.hh"
47#include "G4ThreeVector.hh"
48
49class G4ErrorPropagationNavigator : public G4Navigator
50{
51 public: // with description
52
53 G4ErrorPropagationNavigator();
54 ~G4ErrorPropagationNavigator();
55
56 G4double ComputeStep (const G4ThreeVector &pGlobalPoint,
57 const G4ThreeVector &pDirection,
58 const G4double pCurrentProposedStepLength,
59 G4double &pNewSafety);
60 // Calls the navigation in the detector geometry and then checks
61 // if the distance to surface is smaller than the proposed step
62
63 G4double ComputeSafety(const G4ThreeVector &globalpoint,
64 const G4double pProposedMaxLength = DBL_MAX,
65 const G4bool keepState = true);
66 // Calls the navigation in the detector geometry and then checks
67 // if the distance to surface is smaller than the proposed safety
68
69 G4ThreeVector GetGlobalExitNormal(const G4ThreeVector& point,
70 G4bool* valid);
71 // Return Exit Surface Normal and validity too. Can only be called if
72 // the Navigator's last Step has crossed a volume geometrical boundary.
73 // Normal points out of the volume exited and/or into the volume entered.
74
75 G4double TargetSafetyFromPoint( const G4ThreeVector &pGlobalpoint );
76 // Isotropic safety for 'Target'
77
78 //-- NOT implemented, as it is difficult to define the coordinate system:
79 // G4ThreeVector GetLocalExitNormal(G4bool* valid);
80 // G4ThreeVector GetLocalExitNormalAndCheck(const G4ThreeVector& point,
81 // G4bool* valid);
82 // Convention:
83 // The *local* normal is in the coordinate system of the *final* volume.
84};
85
86#endif
