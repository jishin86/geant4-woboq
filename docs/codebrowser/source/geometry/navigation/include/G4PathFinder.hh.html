27// class G4PathFinder
28//
29// Class description:
30//
31// This class directs the lock-stepped propagation of a track in the
32// 'mass' and other parallel geometries. It ensures that tracking
33// in a magnetic field sees these parallel geometries at each trial step,
34// and that the earliest boundary limits the step.
35//
36// For the movement in field, it relies on the class G4PropagatorInField
37//
38// History:
39// -------
40// 7.10.05 John Apostolakis, Draft design
41// 26.04.06 John Apostolakis, Revised design and first implementation
42// ---------------------------------------------------------------------------

289inline G4VPhysicalVolume* G4PathFinder::GetLocatedVolume( G4int navId ) const
290{
291 G4VPhysicalVolume* vol=0;
292 if( (navId < fMaxNav) && (navId >=0) ) { vol= fLocatedVolume[navId]; }
293 return vol;
294}

296inline G4int G4PathFinder::SetVerboseLevel(G4int newLevel)
297{
298 G4int old= fVerboseLevel; fVerboseLevel= newLevel; return old;
299}

322inline G4Navigator* G4PathFinder::GetNavigator(G4int n) const
323{
324 if( (n>fNoActiveNavigators)||(n<0)) { n=0; }
325 return fpNavigator[n];

328inline G4double G4PathFinder::ObtainSafety( G4int navId, G4ThreeVector& globalCenterPoint )
329{
330 globalCenterPoint= fSafetyLocation;
331 // navId = std::min( navId, fMaxNav-1 );
332 return fNewSafetyComputed[ navId ];
333}
334
335inline G4double G4PathFinder::LastPreSafety( G4int navId,
336 G4ThreeVector& globalCenterPoint,
337 G4double& minSafety )
338{
339 globalCenterPoint= fPreSafetyLocation;
340 minSafety= fPreSafetyMinValue;
341 // navId = std::min( navId, fMaxNav-1 );
342 return fPreSafetyValues[ navId ];
