82// ********************************************************************
83// SetWorldVolume
84//
85// Sets the world (`topmost') volume
86// ********************************************************************
87//
88inline
89void G4Navigator::SetWorldVolume(G4VPhysicalVolume* pWorld)
90{
91 if ( !(pWorld->GetTranslation()==G4ThreeVector(0,0,0)) )
92 {
93 G4Exception ("G4Navigator::SetWorldVolume()", "GeomNav0002",
94 FatalException, "Volume must be centered on the origin.");
95 }
96 const G4RotationMatrix* rm = pWorld->GetRotation();
97 if ( rm && (!rm->isIdentity()) )
98 {
99 G4Exception ("G4Navigator::SetWorldVolume()", "GeomNav0002",
100 FatalException, "Volume must not be rotated.");
101 }
102 fTopPhysical = pWorld;
103 fHistory.SetFirstEntry(pWorld);
104}

152// ********************************************************************
153// GetDaughtersRegularStructureId
154// ********************************************************************
155//
156inline
157G4int G4Navigator::
158GetDaughtersRegularStructureId(const G4LogicalVolume *pLog) const
159{
160 G4int regId = 0;
161 G4VPhysicalVolume *pVol;
162
163 if ( pLog->GetNoDaughters()==1 )
164 {
165 pVol = pLog->GetDaughter(0);
166 regId = pVol->GetRegularStructureId();
167 }
168 return regId;
169}

276// ********************************************************************
277// LocateGlobalPointAndUpdateTouchableHandle
278// ********************************************************************
279//
280inline
281void G4Navigator::LocateGlobalPointAndUpdateTouchableHandle(
282 const G4ThreeVector& position,
283 const G4ThreeVector& direction,
284 G4TouchableHandle& oldTouchableToUpdate,
285 const G4bool RelativeSearch )
286{
287 G4VPhysicalVolume* pPhysVol;
288 pPhysVol = LocateGlobalPointAndSetup( position,&direction,RelativeSearch );
289 if( fEnteredDaughter || fExitedMother )
290 {
291 oldTouchableToUpdate = CreateTouchableHistory();
292 if( pPhysVol == 0 )
293 {
294 // We want to ensure that the touchable is correct in this case.
295 // The method below should do this and recalculate a lot more ....
296 //
297 oldTouchableToUpdate->UpdateYourself( pPhysVol, &fHistory );
298 }
299 }
300 return;
301}

438// ********************************************************************
439// SeverityOfZeroStepping
440//
441// Reports on severity of error in case Navigator is stuck
442// and is returning zero steps
443// ********************************************************************
444//
445inline
446G4int G4Navigator::SeverityOfZeroStepping( G4int* noZeroSteps ) const
447{
448 G4int severity=0, noZeros= fNumberZeroSteps;
449 if( noZeroSteps) *noZeroSteps = fNumberZeroSteps;
450
451 if( noZeros >= fAbandonThreshold_NoZeroSteps )
452 {
453 severity = 10;
454 }
455 if( noZeros > 0 && noZeros < fActionThreshold_NoZeroSteps )
456 {
457 severity = 5 * noZeros / fActionThreshold_NoZeroSteps;
458 }
459 else if( noZeros == fActionThreshold_NoZeroSteps )
460 {
461 severity = 5;
462 }
463 else if( noZeros >= fAbandonThreshold_NoZeroSteps - 2 )
464 {
465 severity = 9;
466 }
467 else if( noZeros < fAbandonThreshold_NoZeroSteps - 2 )
468 {
469 severity = 5 + 4 * (noZeros-fAbandonThreshold_NoZeroSteps)
470 / fActionThreshold_NoZeroSteps;
471 }
472 return severity;
473}
