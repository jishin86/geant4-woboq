33inline G4bool
34G4AuxiliaryNavServices::
35CheckPointOnSurface( const G4VSolid* sampleSolid,
36 const G4ThreeVector& localPoint,
37 const G4ThreeVector* globalDirection,
38 const G4AffineTransform& sampleTransform,
39 const G4bool locatedOnEdge)
40{
41 G4ThreeVector localDirection, sampleNormal;
42 G4bool enter = false;
43
44 EInside insideSolid = sampleSolid->Inside(localPoint);
45 if ( insideSolid!=kOutside )
46 {
47 G4bool checkDirection= locatedOnEdge && (globalDirection!=0);
48 if( (insideSolid==kSurface) && checkDirection)
49 {
50 // We are probably located on an edge.
51 //
52 localDirection= sampleTransform.TransformAxis(*globalDirection);
53
54 // Check whether we enter the volume
55 //
56 sampleNormal = sampleSolid->SurfaceNormal(localPoint);
57
58 G4double dotProd= sampleNormal.dot(localDirection);
59
60#ifdef G4DEBUG_AUX_NAVIGATION
61 G4cout << " dir.norm= " << dotProd // sampleNormal.dot(localDirection)
62 << " localDirection= " << localDirection
63 << " sampleNormal= " << sampleNormal;
64 if( dotProd == 0.0 )
65 {
66 G4cout << " distanceToIn = "
67 << sampleSolid->DistanceToIn( localPoint, localDirection );
68 }
69#endif
70
71 if ( dotProd <= 0.0 )
72 {
73 if( dotProd == 0.0 )
74 {
75 // We can't decide yet, let's make sure we're entering the solid.
76 // If by a confusion we entered the next solid we find out now
77 // whether to leave or to enter.
78 // This happens when we're on the surface or edge shared by two
79 // solids
80 //
81 G4double distanceToIn =
82 sampleSolid->DistanceToIn( localPoint, localDirection );
83 if( distanceToIn != kInfinity )
84 {
85 enter = true;
86 }
87 }
88 else
89 {
90 enter = true;
91 }
92 }
93 }
94 else
95 {
96 enter = true;
97 }
98 }
99#ifdef G4DEBUG_AUX_NAVIGATION
100 G4cout << " enter = " << (enter ? "true" : "false" ) << G4endl;
101#endif
102
103 return enter;
104}

108inline G4bool
109G4AuxiliaryNavServices::
110CheckPointExiting( const G4VSolid* sampleSolid,
111 const G4ThreeVector& localPoint,
112 const G4ThreeVector* globalDirection,
113 const G4AffineTransform& sampleTransform )
114{
115 if( !globalDirection ) { return false; }
116
117 G4ThreeVector localDirection, sampleNormal;
118 G4bool exiting = false;
119
120 EInside insideSolid = sampleSolid->Inside(localPoint);
121 if( insideSolid==kSurface )
122 {
123 localDirection= sampleTransform.TransformAxis(*globalDirection);
124
125 // Check whether we are exiting the volume
126 //
127 sampleNormal = sampleSolid->SurfaceNormal(localPoint);
128 if ( sampleNormal.dot(localDirection) >= 0 )
129 {
130 if( sampleNormal.dot(localDirection) == 0 )
131 {
132 // We can't decide yet, let's make sure we're entering the solid.
133 // If by a confusion we entered the next solid we find out now
134 // whether to leave or to exiting.
135 // This happens when we're on the surface or edge shared by two
136 // solids
137 //
138 G4double distanceToIn =
139 sampleSolid->DistanceToIn( localPoint, localDirection );
140 if( distanceToIn != kInfinity )
141 {
142 exiting = true;
143 }
144 }
145 else
146 {
147 exiting = true;
148 }
149 }
150 }
151 return exiting;
152}
