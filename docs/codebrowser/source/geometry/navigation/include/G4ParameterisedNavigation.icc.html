33// ********************************************************************
34// IdentifyAndPlaceSolid
35// ********************************************************************
36//
37inline
38G4VSolid* G4ParameterisedNavigation::
39IdentifyAndPlaceSolid( G4int num,
40 G4VPhysicalVolume *apparentPhys, // PhysV or PhysT
41 G4VPVParameterisation *curParam )
42{
43 G4VSolid *sampleSolid;
44
45 sampleSolid = curParam->ComputeSolid(num, apparentPhys);
46 sampleSolid->ComputeDimensions(curParam, num, apparentPhys);
47 curParam->ComputeTransformation(num, apparentPhys);
48
49 return sampleSolid;
50}

52// ********************************************************************
53// ParamVoxelLocate
54// ********************************************************************
55//
56inline
57G4SmartVoxelNode*
58G4ParameterisedNavigation::ParamVoxelLocate( G4SmartVoxelHeader* pHead,
59 const G4ThreeVector& localPoint )
60{
61 // If no parameterisation axis is specified, adopt default
62 // location strategy as for placements
63 //
64 if ( pHead->GetParamAxis()==kUndefined )
65 {
66 fVoxelNode = G4VoxelNavigation::VoxelLocate(pHead,localPoint);
67 }
68 else
69 {
70 G4double targetHeaderMin, targetHeaderNodeWidth;
71 G4int targetHeaderNoSlices, targetNodeNo;
72 EAxis targetHeaderAxis;
73
74 targetHeaderAxis = pHead->GetAxis();
75 targetHeaderNoSlices = pHead->GetNoSlices();
76 targetHeaderMin = pHead->GetMinExtent();
77 targetHeaderNodeWidth = (pHead->GetMaxExtent()-targetHeaderMin)
78 / targetHeaderNoSlices;
79 targetNodeNo = G4int ( (localPoint(targetHeaderAxis)-targetHeaderMin)
80 / targetHeaderNodeWidth );
81 // Rounding protection
82 //
83 if ( targetNodeNo<0 )
84 {
85 targetNodeNo = 0;
86 }
87 else if ( targetNodeNo>=targetHeaderNoSlices )
88 {
89 targetNodeNo = targetHeaderNoSlices-1;
90 }
91 fVoxelAxis = targetHeaderAxis;
92 fVoxelNoSlices = targetHeaderNoSlices;
93 fVoxelSliceWidth = targetHeaderNodeWidth;
94 fVoxelNodeNo = targetNodeNo;
95 fVoxelHeader = pHead;
96 fVoxelNode = pHead->GetSlice(targetNodeNo)->GetNode();
97 }
98 return fVoxelNode;
99}
