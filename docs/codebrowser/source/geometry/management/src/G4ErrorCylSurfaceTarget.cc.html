45G4ErrorCylSurfaceTarget::
46G4ErrorCylSurfaceTarget( const G4double& radius,
47 const G4ThreeVector& trans,
48 const G4RotationMatrix& rotm )
49 : fradius(radius)
50{
51 theType = G4ErrorTarget_CylindricalSurface;
52
53 ftransform = G4AffineTransform( rotm.inverse(), -trans );
54#ifdef G4VERBOSE
55 if(G4ErrorPropagatorData::verbose() >= 2 ) {
56 Dump( " $$$ creating G4ErrorCylSurfaceTarget ");

59}

84//---------------------------------------------------------------------
85
86G4double G4ErrorCylSurfaceTarget::
87GetDistanceFromPoint( const G4ThreeVector& point,
88 const G4ThreeVector& dir ) const
89{
90 if( dir.mag() == 0. )
91 {
92 G4Exception("G4ErrorCylSurfaceTarget::GetDistanceFromPoint()",
93 "GeomMgt0003", FatalException, "Direction is zero !");
94 }
95
96 //----- Get intersection point
97 G4ThreeVector localPoint = ftransform.TransformPoint( point );
98 G4ThreeVector localDir = ftransform.TransformAxis( dir );
99 G4ThreeVector inters = IntersectLocal(localPoint, localDir);
100
101 G4double dist = (localPoint-inters).mag();
102
103#ifdef G4VERBOSE
104 if(G4ErrorPropagatorData::verbose() >= 3 )
105 {
106 G4cout << " G4ErrorCylSurfaceTarget::GetDistanceFromPoint():" << G4endl
107 << " Global point " << point << " dir " << dir << G4endl
108 << " Intersection " << inters << G4endl
109 << " Distance " << dist << G4endl;
110 Dump( " CylSurface: " );
111 }
112#endif
113
114 return dist;
115}

120G4double G4ErrorCylSurfaceTarget::
121GetDistanceFromPoint( const G4ThreeVector& point ) const
122{
123 G4ThreeVector localPoint = ftransform.TransformPoint( point );
124
125#ifdef G4VERBOSE
126 if(G4ErrorPropagatorData::verbose() >= 3 )
127 {
128 G4cout << " G4ErrorCylSurfaceTarget::GetDistanceFromPoint:" << G4endl
129 << " Global point " << point << G4endl
130 << " Distance " << fradius - localPoint.perp() << G4endl;
131 Dump( " CylSurface: " );
132 }
133#endif
134
135 return fradius - localPoint.perp();
136}

138//---------------------------------------------------------------------
139
140G4ThreeVector G4ErrorCylSurfaceTarget::
141IntersectLocal( const G4ThreeVector& localPoint,
142 const G4ThreeVector& localDir ) const
143{
144 G4double eqa = localDir.x()*localDir.x()+localDir.y()*localDir.y();
145 G4double eqb = 2*(localPoint.x()*localDir.x()+localPoint.y()*localDir.y());
146 G4double eqc = -fradius*fradius+localPoint.x()*localPoint.x()
147 +localPoint.y()*localPoint.y();
148 G4int inside = (localPoint.perp() > fradius) ? -1 : 1;
149 G4double lambda;
150
151 if( eqa*inside > 0. )
152 {
153 lambda = (-eqb + std::sqrt(eqb*eqb-4*eqa*eqc) ) / (2.*eqa);
154 }
155 else if( eqa*inside < 0. )
156 {
157 lambda = (-eqb - std::sqrt(eqb*eqb-4*eqa*eqc) ) / (2.*eqa);
158 }
159 else
160 {
161 if( eqb != 0. )
162 {
163 lambda = -eqc/eqb;
164 }
165 else
166 {
167 std::ostringstream message;
168 message << "Intersection not possible !" << G4endl
169 << " Point: " << localPoint << ", direction: "
170 << localDir;
171 Dump( " CylSurface: " );
172 G4Exception("G4ErrorCylSurfaceTarget::IntersectLocal()",
173 "GeomMgt1002", JustWarning, message);
174 lambda = kInfinity;
175 }
176 }
177
178 G4ThreeVector inters = localPoint + lambda*localDir/localDir.mag();
179
180#ifdef G4VERBOSE
181 if(G4ErrorPropagatorData::verbose() >= 4 ) {
182 G4cout << " G4ErrorCylSurfaceTarget::IntersectLocal " << inters << " "
183 << inters.perp() << " localPoint " << localPoint << " localDir "
184 << localDir << G4endl;
185 }
186#endif
187
188 return inters;
189}

191//---------------------------------------------------------------------
192
193G4Plane3D G4ErrorCylSurfaceTarget::
194GetTangentPlane( const G4ThreeVector& point ) const
195{
196 G4ThreeVector localPoint = ftransform.TransformPoint( point );
197
198 // check that point is at cylinder surface
199 //
200 if( std::fabs( localPoint.perp() - fradius )
201 > 1000.*G4GeometryTolerance::GetInstance()->GetSurfaceTolerance() )
202 {
203 std::ostringstream message;
204 message << "Local point not at surface !" << G4endl
205 << " Point: " << point << ", local: " << localPoint
206 << G4endl
207 << " is not at surface, but far away by: "
208 << localPoint.perp() - fradius << " !";
209 G4Exception("G4ErrorCylSurfaceTarget::GetTangentPlane()",
210 "GeomMgt1002", JustWarning, message);
211 }
212
213 G4Normal3D normal = localPoint - ftransform.NetTranslation();
214
215 return G4Plane3D( normal, point );
216}
