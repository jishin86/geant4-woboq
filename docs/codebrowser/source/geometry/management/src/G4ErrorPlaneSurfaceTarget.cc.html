100G4ThreeVector G4ErrorPlaneSurfaceTarget::
101Intersect( const G4ThreeVector& pt, const G4ThreeVector& dir ) const
102{
103 G4double lam = GetDistanceFromPoint( pt, dir );
104 G4Point3D inters = pt + lam * dir;
105
106#ifdef G4VERBOSE
107 if(G4ErrorPropagatorData::verbose() >= 4 )
108 {
109 G4cout << " $$$ creating G4ErrorPlaneSurfaceTarget::Intersect "
110 << inters << G4endl;
111 }
112#endif
113
114 return inters;
115}

117//---------------------------------------------------------------------
118
119G4double G4ErrorPlaneSurfaceTarget::
120GetDistanceFromPoint( const G4ThreeVector& pt, const G4ThreeVector& dir ) const
121{
122 if( std::fabs( dir.mag() -1. ) > 1.E-6 )
123 {
124 std::ostringstream message;
125 message << "Direction is not a unit vector: " << dir << " !";
126 G4Exception("G4ErrorPlaneSurfaceTarget::GetDistanceFromPoint()",
127 "GeomMgt1002", JustWarning, message);
128 }
129 G4double dist = -(a_ * pt.x() + b_ * pt.y() + c_ * pt.z() + d_)
130 / (a_ * dir.x() + b_ * dir.y() + c_ * dir.z() );
131
132#ifdef G4VERBOSE
133 if(G4ErrorPropagatorData::verbose() >= 3 )
134 {
135 G4cout << " G4ErrorPlaneSurfaceTarget::GetDistanceFromPoint()" << G4endl
136 << " Point: " << pt << ", Direction: " << dir << G4endl
137 << " Distance: " << dist << G4endl;
138 }
139#endif
140
141 return dist;
142}
143
144//---------------------------------------------------------------------
145
146G4double G4ErrorPlaneSurfaceTarget::
147GetDistanceFromPoint( const G4ThreeVector& pt ) const
148{
149 G4ThreeVector vec = point() - pt;
150 G4double dist = std::fabs(vec * normal() / normal().mag());
151
152#ifdef G4VERBOSE
153 if(G4ErrorPropagatorData::verbose() >= 3 )
154 {
155 G4cout << " G4ErrorPlaneSurfaceTarget::GetDistanceFromPoint()" << G4endl
156 << " Point: " << pt << G4endl
157 << " Distance: " << dist << G4endl;
158 }
159#endif
160
161 return dist;
162}
