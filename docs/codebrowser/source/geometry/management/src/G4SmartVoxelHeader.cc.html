57// ***************************************************************************
58// Constructor for topmost header, to begin voxel construction at a
59// given logical volume.
60// Constructs target List of volumes, calls "Build and refine" constructor.
61// Assumes all daughters represent single volumes (ie. no divisions
62// or parametric)
63// ***************************************************************************
64//
65G4SmartVoxelHeader::G4SmartVoxelHeader(G4LogicalVolume* pVolume,
66 G4int pSlice)
67 : fminEquivalent(pSlice),
68 fmaxEquivalent(pSlice),
69 fparamAxis(kUndefined)
70{
71 G4int nDaughters = pVolume->GetNoDaughters();
72 G4VoxelLimits limits; // Create `unlimited' limits object
73
74 // Determine whether daughter is replicated
75 //
76 if ((nDaughters!=1) || (!pVolume->GetDaughter(0)->IsReplicated()))
77 {
78 // Daughter not replicated => conventional voxel Build
79 // where each daughters extents are computed
80 //
81 BuildVoxels(pVolume);
82 }
83 else
84 {
85 // Single replicated daughter
86 //
87 BuildReplicaVoxels(pVolume);
88 }
89}

91// ***************************************************************************
92// Protected constructor:
93// builds and refines voxels between specified limits, considering only
94// the physical volumes numbered `pCandidates'. `pSlice' is used to set max
95// and min equivalent slice nos for the header - they apply to the level
96// of the header, not its nodes.
97// ***************************************************************************
98//
99G4SmartVoxelHeader::G4SmartVoxelHeader(G4LogicalVolume* pVolume,
100 const G4VoxelLimits& pLimits,
101 const G4VolumeNosVector* pCandidates,
102 G4int pSlice)
103 : fminEquivalent(pSlice),
104 fmaxEquivalent(pSlice),
105 fparamAxis(kUndefined)
106{
107#ifdef G4GEOMETRY_VOXELDEBUG
108 G4cout << "**** G4SmartVoxelHeader::G4SmartVoxelHeader" << G4endl
109 << " Limits " << pLimits << G4endl
110 << " Candidate #s = " ;
111 for (size_t i=0;i<pCandidates->size();i++)
112 {
113 G4cout << (*pCandidates)[i] << " ";
114 }
115 G4cout << G4endl;
116#endif
117
118 BuildVoxelsWithinLimits(pVolume,pLimits,pCandidates);
119}

121// ***************************************************************************
122// Destructor:
123// deletes all proxies and underlying objects.
124// ***************************************************************************
125//
126G4SmartVoxelHeader::~G4SmartVoxelHeader()
127{
128 // Manually destroy underlying nodes/headers
129 // Delete collected headers and nodes once only
130 //
131 G4int node, proxy, maxNode=fslices.size();
132 G4SmartVoxelProxy *lastProxy=0;
133 G4SmartVoxelNode *dyingNode, *lastNode=0;
134 G4SmartVoxelHeader *dyingHeader, *lastHeader=0;
135
136 for (node=0; node<maxNode; node++)
137 {
138 if (fslices[node]->IsHeader())
139 {
140 dyingHeader = fslices[node]->GetHeader();
141 if (lastHeader!=dyingHeader)
142 {
143 lastHeader = dyingHeader;
144 lastNode = 0;
145 delete dyingHeader;
146 }
147 }
148 else
149 {
150 dyingNode = fslices[node]->GetNode();
151 if (dyingNode!=lastNode)
152 {
153 lastNode=dyingNode;
154 lastHeader=0;
155 delete dyingNode;
156 }
157 }
158 }
159 // Delete proxies
160 //
161 for (proxy=0; proxy<maxNode; proxy++)
162 {
163 if (fslices[proxy]!=lastProxy)
164 {
165 lastProxy = fslices[proxy];
166 delete lastProxy;
167 }
168 }
169 // Don't need to clear slices
170 // fslices.clear();
171}

173// ***************************************************************************
174// Equality operator: returns true if contents are equivalent.
175// Implies a deep search through contained nodes/header.
176// Compares headers' axes,sizes,extents. Returns false if different.
177// For each contained proxy, determines whether node/header, compares and
178// returns if different. Compares and returns if proxied nodes/headers
179// are different.
180// ***************************************************************************
181//
182G4bool G4SmartVoxelHeader::operator == (const G4SmartVoxelHeader& pHead) const
183{
184 if ( (GetAxis() == pHead.GetAxis())
185 && (GetNoSlices() == pHead.GetNoSlices())
186 && (GetMinExtent() == pHead.GetMinExtent())
187 && (GetMaxExtent() == pHead.GetMaxExtent()) )
188 {
189 G4int node, maxNode;
190 G4SmartVoxelProxy *leftProxy, *rightProxy;
191 G4SmartVoxelHeader *leftHeader, *rightHeader;
192 G4SmartVoxelNode *leftNode, *rightNode;
193
194 maxNode=GetNoSlices();
195 for (node=0; node<maxNode; node++)
196 {
197 leftProxy = GetSlice(node);
198 rightProxy = pHead.GetSlice(node);
199 if (leftProxy->IsHeader())
200 {
201 if (rightProxy->IsNode())
202 {
203 return false;
204 }
205 else
206 {
207 leftHeader = leftProxy->GetHeader();
208 rightHeader = rightProxy->GetHeader();
209 if (!(*leftHeader==*rightHeader))
210 {
211 return false;
212 }
213 }
214 }
215 else
216 {
217 if (rightProxy->IsHeader())
218 {
219 return false;
220 }
221 else
222 {
223 leftNode = leftProxy->GetNode();
224 rightNode = rightProxy->GetNode();
225 if (!(*leftNode==*rightNode))
226 {
227 return false;
228 }
229 }
230 }
231 }
232 return true;
233 }
234 else
235 {
236 return false;
237 }
238}

240// ***************************************************************************
241// Builds voxels for daughters specified volume, in NON-REPLICATED case
242// o Create List of target volume nos (all daughters; 0->noDaughters-1)
243// o BuildWithinLimits does Build & also determines mother dimensions.
244// ***************************************************************************
245//
246void G4SmartVoxelHeader::BuildVoxels(G4LogicalVolume* pVolume)
247{
248 G4VoxelLimits limits; // Create `unlimited' limits object
249 G4int nDaughters = pVolume->GetNoDaughters();
250
251 G4VolumeNosVector targetList;
252 targetList.reserve(nDaughters);
253 for (G4int i=0; i<nDaughters; i++)
254 {
255 targetList.push_back(i);
256 }
257 BuildVoxelsWithinLimits(pVolume, limits, &targetList);
258}

260// ***************************************************************************
261// Builds voxels for specified volume containing a single replicated volume.
262// If axis is not specified (i.e. "kUndefined"), 3D voxelisation is applied,
263// and the best axis is determined according to heuristics as for placements.
264// ***************************************************************************
265//
266void G4SmartVoxelHeader::BuildReplicaVoxels(G4LogicalVolume* pVolume)
267{
268 G4VPhysicalVolume *pDaughter=0;
269
270 // Replication data
271 //
272 EAxis axis;
273 G4int nReplicas;
274 G4double width,offset;
275 G4bool consuming;
276
277 // Consistency check: pVolume should contain single replicated volume
278 //
279 if ( (pVolume->GetNoDaughters()==1)
280 && (pVolume->GetDaughter(0)->IsReplicated()==true) )
281 {
282 // Obtain replication data
283 //
284 pDaughter=pVolume->GetDaughter(0);
285 pDaughter->GetReplicationData(axis,nReplicas,width,offset,consuming);
286 fparamAxis = axis;
287 if ( consuming==false )
288 {
289 G4VoxelLimits limits; // Create `unlimited' limits object
290 G4VolumeNosVector targetList;
291 targetList.reserve(nReplicas);
292 for (G4int i=0; i<nReplicas; i++)
293 {
294 targetList.push_back(i);
295 }
296 if (axis != kUndefined)
297 {
298 // Apply voxelisation along the specified axis only
299
300 G4ProxyVector* pSlices=BuildNodes(pVolume,limits,&targetList,axis);
301 faxis = axis;
302 fslices = *pSlices;
303 delete pSlices;
304
305 // Calculate and set min and max extents given our axis
306 //
307 const G4AffineTransform origin;
308 pVolume->GetSolid()->CalculateExtent(faxis, limits, origin,
309 fminExtent, fmaxExtent);
310 // Calculate equivalent nos
311 //
312 BuildEquivalentSliceNos();
313 CollectEquivalentNodes(); // Collect common nodes
314 }
315 else
316 {
317 // Build voxels similarly as for normal placements considering
318 // all three cartesian axes.
319
320 BuildVoxelsWithinLimits(pVolume, limits, &targetList);
321 }
322 }
323 else
324 {
325 // Replication is consuming -> Build voxels directly
326 //
327 // o Cartesian axes - range is -width*nREplicas/2 to +width*nREplicas/2
328 // nReplicas replications result
329 // o Radial axis (rho) = range is 0 to width*nReplicas
330 // nReplicas replications result
331 // o Phi axi - range is offset to offset+width*nReplicas radians
332 //
333 // Equivalent slices no computation & collection not required - all
334 // slices are different
335 //
336 switch (axis)
337 {
338 case kXAxis:
339 case kYAxis:
340 case kZAxis:
341 fminExtent = -width*nReplicas*0.5;
342 fmaxExtent = width*nReplicas*0.5;
343 break;
344 case kRho:
345 fminExtent = offset;
346 fmaxExtent = width*nReplicas+offset;
347 break;
348 case kPhi:
349 fminExtent = offset;
350 fmaxExtent = offset+width*nReplicas;
351 break;
352 default:
353 G4Exception("G4SmartVoxelHeader::BuildReplicaVoxels()",
354 "GeomMgt0002", FatalException, "Illegal axis.");
355 break;
356 }
357 faxis = axis; // Set axis
358 BuildConsumedNodes(nReplicas);
359 if ( (axis==kXAxis) || (axis==kYAxis) || (axis==kZAxis) )
360 {
361 // Sanity check on extent
362 //
363 G4double emin = kInfinity, emax = -kInfinity;
364 G4VoxelLimits limits;
365 G4AffineTransform origin;
366 pVolume->GetSolid()->CalculateExtent(axis, limits, origin, emin, emax);
367 if ( (std::fabs((emin-fminExtent)/fminExtent) +
368 std::fabs((emax-fmaxExtent)/fmaxExtent)) > 0.05)
369 {
370 std::ostringstream message;
371 message << "Sanity check: wrong solid extent." << G4endl
372 << " Replicated geometry, logical volume: "
373 << pVolume->GetName();
374 G4Exception("G4SmartVoxelHeader::BuildReplicaVoxels",
375 "GeomMgt0002", FatalException, message);
376 }
377 }
378 }
379 }
380 else
381 {
382 G4Exception("G4SmartVoxelHeader::BuildReplicaVoxels", "GeomMgt0002",
383 FatalException, "Only one replicated daughter is allowed !");
384 }
385}

387// ***************************************************************************
388// Builds `consumed nodes': nReplicas nodes each containing one replication,
389// numbered in sequence 0->nReplicas-1
390// o Modifies fslices `in place'
391// o faxis,fminExtent,fmaxExtent NOT modified.
392// ***************************************************************************
393//
394void G4SmartVoxelHeader::BuildConsumedNodes(G4int nReplicas)
395{
396 G4int nNode, nVol;
397 G4SmartVoxelNode *pNode;
398 G4SmartVoxelProxy *pProxyNode;
399
400 // Create and fill nodes in temporary G4NodeVector (on stack)
401 //
402 G4NodeVector nodeList;
403 nodeList.reserve(nReplicas);
404 for (nNode=0; nNode<nReplicas; nNode++)
405 {
406 pNode=new G4SmartVoxelNode(nNode);
407 if (!pNode)
408 {
409 G4Exception("G4SmartVoxelHeader::BuildConsumedNodes()", "GeomMgt0003",
410 FatalException, "Node allocation error.");
411 }
412 nodeList.push_back(pNode);
413 }
414 for (nVol=0; nVol<nReplicas; nVol++)
415 {
416 nodeList[nVol]->Insert(nVol); // Insert replication of number
417 } // identical to voxel number
418
419 // Create & fill proxy List `in place' by modifying instance data fslices
420 //
421 fslices.clear();
422 for (nNode=0; nNode<nReplicas; nNode++)
423 {
424 pProxyNode = new G4SmartVoxelProxy(nodeList[nNode]);
425 if (!pProxyNode)
426 {
427 G4Exception("G4SmartVoxelHeader::BuildConsumedNodes()", "GeomMgt0003",
428 FatalException, "Proxy node allocation error.");
429 }
430 fslices.push_back(pProxyNode);
431 }
432}

434// ***************************************************************************
435// Builds and refines voxels between specified limits, considering only
436// the physical volumes numbered `pCandidates'.
437// o Chooses axis
438// o Determines min and max extents (of mother solid) within limits.
439// ***************************************************************************
440//
441void
442G4SmartVoxelHeader::BuildVoxelsWithinLimits(G4LogicalVolume* pVolume,
443 G4VoxelLimits pLimits,
444 const G4VolumeNosVector* pCandidates)
445{
446 // Choose best axis for slicing by:
447 // 1. Trying all unlimited cartesian axes
448 // 2. Select axis which gives greatest no slices
449
450 G4ProxyVector *pGoodSlices=0, *pTestSlices, *tmpSlices;
451 G4double goodSliceScore=kInfinity, testSliceScore;
452 EAxis goodSliceAxis = kXAxis;
453 EAxis testAxis = kXAxis;
454 G4int node, maxNode, iaxis;
455 G4VoxelLimits noLimits;
456
457 // Try all non-limited cartesian axes
458 //
459 for (iaxis=0; iaxis<3; iaxis++)
460 {
461 switch(iaxis)
462 {
463 case 0:
464 testAxis = kXAxis;
465 break;
466 case 1:
467 testAxis = kYAxis;
468 break;
469 case 2:
470 testAxis = kZAxis;
471 break;
472 }
473 if (!pLimits.IsLimited(testAxis))
474 {
475 pTestSlices = BuildNodes(pVolume,pLimits,pCandidates,testAxis);
476 testSliceScore = CalculateQuality(pTestSlices);
477 if ( (!pGoodSlices) || (testSliceScore<goodSliceScore) )
478 {
479 goodSliceAxis = testAxis;
480 goodSliceScore = testSliceScore;
481 tmpSlices = pGoodSlices;
482 pGoodSlices = pTestSlices;
483 pTestSlices = tmpSlices;
484 }
485 if (pTestSlices)
486 {
487 // Destroy pTestSlices and all its contents
488 //
489 maxNode=pTestSlices->size();
490 for (node=0; node<maxNode; node++)
491 {
492 delete (*pTestSlices)[node]->GetNode();
493 }
494 G4SmartVoxelProxy* tmpProx;
495 while (pTestSlices->size()>0) // Loop checking, 06.08.2015, G.Cosmo
496 {
497 tmpProx = pTestSlices->back();
498 pTestSlices->pop_back();
499 for (G4ProxyVector::iterator i=pTestSlices->begin();
500 i!=pTestSlices->end(); )
501 {
502 if (*i==tmpProx)
503 {
504 i = pTestSlices->erase(i);
505 }
506 else
507 {
508 ++i;
509 }
510 }
511 if ( tmpProx ) { delete tmpProx; }
512 }
513 delete pTestSlices;
514 }
515 }
516 }
517 // Check for error case.. when limits already 3d,
518 // so cannot select a new axis
519 //
520 if (!pGoodSlices)
521 {
522 G4Exception("G4SmartVoxelHeader::BuildVoxelsWithinLimits()",
523 "GeomMgt0002", FatalException,
524 "Cannot select more than 3 axis for optimisation.");
525 return;
526 }
527
528 //
529 // We have selected pGoodSlices, with a score testSliceScore
530 //
531
532 // Store chosen axis, slice ptr
533 //
534 fslices=*pGoodSlices; // Set slice information, copy ptrs in collection
535 delete pGoodSlices; // Destroy slices vector, but not contained
536 // proxies or nodes
537 faxis=goodSliceAxis;
538
539#ifdef G4GEOMETRY_VOXELDEBUG
540 G4cout << G4endl << " Volume = " << pVolume->GetName()
541 << G4endl << " Selected axis = " << faxis << G4endl;
542 for (size_t islice=0; islice<fslices.size(); islice++)
543 {
544 G4cout << " Node #" << islice << " = {";
545 for (G4int j=0; j<fslices[islice]->GetNode()->GetNoContained(); j++)
546 {
547 G4cout << " " << fslices[islice]->GetNode()->GetVolume(j);
548 }
549 G4cout << " }" << G4endl;
550 }
551 G4cout << G4endl;
552#endif
553
554 // Calculate and set min and max extents given our axis
555 //
556 G4VSolid* outerSolid = pVolume->GetSolid();
557 const G4AffineTransform origin;
558 if(!outerSolid->CalculateExtent(faxis,pLimits,origin,fminExtent,fmaxExtent))
559 {
560 outerSolid->CalculateExtent(faxis,noLimits,origin,fminExtent,fmaxExtent);
561 }
562
563 // Calculate equivalent nos
564 //
565 BuildEquivalentSliceNos();
566 CollectEquivalentNodes(); // Collect common nodes
567 RefineNodes(pVolume,pLimits); // Refine nodes creating headers
568
569 // No common headers can exist because collapsed by construction
570}
571
572// ***************************************************************************
573// Calculates and stores the minimum and maximum equivalent neighbour
574// values for all slices at our level.
575//
576// Precondition: all slices are nodes.
577// For each potential start of a group of equivalent nodes:
578// o searches forwards in fslices to find group end
579// o loops from start to end setting start and end slices.
580// ***************************************************************************
581//
582void G4SmartVoxelHeader::BuildEquivalentSliceNos()
583{
584 G4int sliceNo, minNo, maxNo, equivNo;
585 G4int maxNode = fslices.size();
586 G4SmartVoxelNode *startNode, *sampleNode;
587 for (sliceNo=0; sliceNo<maxNode; sliceNo++)
588 {
589 minNo = sliceNo;
590
591 // Get first node (see preconditions - will throw exception if a header)
592 //
593 startNode = fslices[minNo]->GetNode();
594
595 // Find max equivalent
596 //
597 for (equivNo=minNo+1; equivNo<maxNode; equivNo++)
598 {
599 sampleNode = fslices[equivNo]->GetNode();
600 if (!((*startNode) == (*sampleNode))) { break; }
601 }
602 maxNo = equivNo-1;
603 if (maxNo != minNo)
604 {
605 // Set min and max nos
606 //
607 for (equivNo=minNo; equivNo<=maxNo; equivNo++)
608 {
609 sampleNode = fslices[equivNo]->GetNode();
610 sampleNode->SetMinEquivalentSliceNo(minNo);
611 sampleNode->SetMaxEquivalentSliceNo(maxNo);
612 }
613 // Advance outer loop to end of equivalent group
614 //
615 sliceNo = maxNo;
616 }
617 }
618}

620// ***************************************************************************
621// Collects common nodes at our level, deleting all but one to save
622// memory, and adjusting stored slice pointers appropriately.
623//
624// Preconditions:
625// o the slices have not previously be "collected"
626// o all of the slices are nodes.
627// ***************************************************************************
628//
629void G4SmartVoxelHeader::CollectEquivalentNodes()
630{
631 G4int sliceNo, maxNo, equivNo;
632 G4int maxNode=fslices.size();
633 G4SmartVoxelNode *equivNode;
634 G4SmartVoxelProxy *equivProxy;
635
636 for (sliceNo=0; sliceNo<maxNode; sliceNo++)
637 {
638 equivProxy=fslices[sliceNo];
639
640 // Assumption (see preconditions): all slices are nodes
641 //
642 equivNode = equivProxy->GetNode();
643 maxNo = equivNode->GetMaxEquivalentSliceNo();
644 if (maxNo != sliceNo)
645 {
646#ifdef G4GEOMETRY_VOXELDEBUG
647 G4cout << "**** G4SmartVoxelHeader::CollectEquivalentNodes" << G4endl
648 << " Collecting Nodes = "
649 << sliceNo << " - " << maxNo << G4endl;
650#endif
651 // Do collection between sliceNo and maxNo inclusive
652 //
653 for (equivNo=sliceNo+1; equivNo<=maxNo; equivNo++)
654 {
655 delete fslices[equivNo]->GetNode();
656 delete fslices[equivNo];
657 fslices[equivNo] = equivProxy;
658 }
659 sliceNo = maxNo;
660 }
661 }
662}
663
664// ***************************************************************************
665// Collects common headers at our level, deleting all but one to save
666// memory, and adjusting stored slice pointers appropriately.
667//
668// Preconditions:
669// o if a header forms part of a range of equivalent slices
670// (ie. GetMaxEquivalentSliceNo()>GetMinEquivalentSliceNo()),
671// it is assumed that all slices in the range are headers.
672// o this will be true if a constant Expression is used to evaluate
673// when to refine nodes.
674// ***************************************************************************
675//
676void G4SmartVoxelHeader::CollectEquivalentHeaders()
677{
678 G4int sliceNo, maxNo, equivNo;
679 G4int maxNode = fslices.size();
680 G4SmartVoxelHeader *equivHeader, *sampleHeader;
681 G4SmartVoxelProxy *equivProxy;
682
683 for (sliceNo=0; sliceNo<maxNode; sliceNo++)
684 {
685 equivProxy = fslices[sliceNo];
686 if (equivProxy->IsHeader())
687 {
688 equivHeader = equivProxy->GetHeader();
689 maxNo = equivHeader->GetMaxEquivalentSliceNo();
690 if (maxNo != sliceNo)
691 {
692 // Attempt collection between sliceNo and maxNo inclusive:
693 // look for common headers. All slices between sliceNo and maxNo
694 // are guaranteed to be headers but may not have equal contents
695 //
696#ifdef G4GEOMETRY_VOXELDEBUG
697 G4cout << "**** G4SmartVoxelHeader::CollectEquivalentHeaders" << G4endl
698 << " Collecting Headers =";
699#endif
700 for (equivNo=sliceNo+1; equivNo<=maxNo; equivNo++)
701 {
702 sampleHeader = fslices[equivNo]->GetHeader();
703 if ( (*sampleHeader) == (*equivHeader) )
704 {
705#ifdef G4GEOMETRY_VOXELDEBUG
706 G4cout << " " << equivNo;
707#endif
708 // Delete sampleHeader + proxy and replace with equivHeader/Proxy
709 //
710 delete sampleHeader;
711 delete fslices[equivNo];
712 fslices[equivNo] = equivProxy;
713 }
714 else
715 {
716 // Not equal. Set this header to be
717 // the current header for comparisons
718 //
719 equivProxy = fslices[equivNo];
720 equivHeader = equivProxy->GetHeader();
721 }
722
723 }
724#ifdef G4GEOMETRY_VOXELDEBUG
725 G4cout << G4endl;
726#endif
727 // Skip past examined slices
728 //
729 sliceNo = maxNo;
730 }
731 }
732 }

735// ***************************************************************************
736// Builds the nodes corresponding to slices between the specified limits
737// and along the specified axis, using candidate volume no.s in the vector
738// pCandidates. If the `daughters' are replicated volumes (ie. the logical
739// volume has a single replicated/parameterised volume for a daughter)
740// the candidate no.s are interpreted as PARAMETERISED volume no.s &
741// PARAMETERISATIONs are applied to compute transformations & solid
742// dimensions appropriately. The volume must be parameterised - ie. has a
743// parameterisation object & non-consuming) - in this case.
744//
745// Returns pointer to built node "structure" (guaranteed non NULL) consisting
746// of G4SmartVoxelNodeProxies referring to G4SmartVoxelNodes.
747// ***************************************************************************
748//
749G4ProxyVector* G4SmartVoxelHeader::BuildNodes(G4LogicalVolume* pVolume,
750 G4VoxelLimits pLimits,
751 const G4VolumeNosVector* pCandidates,
752 EAxis pAxis)
753{
754 G4double motherMinExtent= kInfinity, motherMaxExtent= -kInfinity,
755 targetMinExtent= kInfinity, targetMaxExtent= -kInfinity;
756 G4VPhysicalVolume *pDaughter=0;
757 G4VPVParameterisation *pParam=0;
758 G4VSolid *targetSolid;
759 G4AffineTransform targetTransform;
760 G4bool replicated;
761 G4int nCandidates = pCandidates->size();
762 G4int nVol, nNode, targetVolNo;
763 G4VoxelLimits noLimits;
764
765#ifdef G4GEOMETRY_VOXELDEBUG
766 G4cout << "**** G4SmartVoxelHeader::BuildNodes" << G4endl
767 << " Limits = " << pLimits << G4endl
768 << " Axis = " << pAxis << G4endl
769 << " Candidates = " << nCandidates << G4endl;
770#endif
771
772 // Compute extent of logical volume's solid along this axis
773 // NOTE: results stored locally and not preserved/reused
774 //
775 G4VSolid* outerSolid = pVolume->GetSolid();
776 const G4AffineTransform origin;
777 if( !outerSolid->CalculateExtent(pAxis, pLimits, origin,
778 motherMinExtent, motherMaxExtent) )
779 {
780 outerSolid->CalculateExtent(pAxis, noLimits, origin,
781 motherMinExtent, motherMaxExtent);
782 }
783 G4VolumeExtentVector minExtents(nCandidates,0.);
784 G4VolumeExtentVector maxExtents(nCandidates,0.);
785
786 if ( (pVolume->GetNoDaughters()==1)
787 && (pVolume->GetDaughter(0)->IsReplicated()==true) )
788 {
789 // Replication data not required: only parameterisation object
790 // and volume no. List used
791 //
792 pDaughter = pVolume->GetDaughter(0);
793 pParam = pDaughter->GetParameterisation();
794 if (!pParam)
795 {
796 std::ostringstream message;
797 message << "PANIC! - Missing parameterisation." << G4endl
798 << " Replicated volume with no parameterisation object !";
799 G4Exception("G4SmartVoxelHeader::BuildNodes()", "GeomMgt0003",
800 FatalException, message);
801 return 0;
802 }
803
804 // Setup daughter's transformations
805 //
806 targetTransform = G4AffineTransform(pDaughter->GetRotation(),
807 pDaughter->GetTranslation());
808 replicated = true;
809 }
810 else
811 {
812 replicated = false;
813 }
814
815 // Compute extents
816 //
817 for (nVol=0; nVol<nCandidates; nVol++)
818 {
819 targetVolNo=(*pCandidates)[nVol];
820 if (replicated == false)
821 {
822 pDaughter=pVolume->GetDaughter(targetVolNo);
823
824 // Setup daughter's transformations
825 //
826 targetTransform = G4AffineTransform(pDaughter->GetRotation(),
827 pDaughter->GetTranslation());
828 // Get underlying (and setup) solid
829 //
830 targetSolid = pDaughter->GetLogicalVolume()->GetSolid();
831 }
832 else
833 {
834 // Find solid
835 //
836 targetSolid = pParam->ComputeSolid(targetVolNo,pDaughter);
837
838 // Setup solid
839 //
840 targetSolid->ComputeDimensions(pParam,targetVolNo,pDaughter);
841
842 // Setup transform
843 //
844 pParam->ComputeTransformation(targetVolNo,pDaughter);
845 targetTransform = G4AffineTransform(pDaughter->GetRotation(),
846 pDaughter->GetTranslation());
847 }
848 // Calculate extents
849 //
850 if(!targetSolid->CalculateExtent(pAxis, pLimits, targetTransform,
851 targetMinExtent, targetMaxExtent))
852 {
853 targetSolid->CalculateExtent(pAxis, noLimits, targetTransform,
854 targetMinExtent,targetMaxExtent);
855 }
856 minExtents[nVol] = targetMinExtent;
857 maxExtents[nVol] = targetMaxExtent;
858
859#ifdef G4GEOMETRY_VOXELDEBUG
860 G4cout << "---------------------------------------------------" << G4endl
861 << " Volume = " << pDaughter->GetName() << G4endl
862 << " Min Extent = " << targetMinExtent << G4endl
863 << " Max Extent = " << targetMaxExtent << G4endl
864 << "---------------------------------------------------" << G4endl;
865#endif
866
867 // Check not entirely outside mother when processing toplevel nodes
868 //
869 if ( (!pLimits.IsLimited()) && ((targetMaxExtent<=motherMinExtent)
870 ||(targetMinExtent>=motherMaxExtent)) )
871 {
872 std::ostringstream message;
873 message << "PANIC! - Overlapping daughter with mother volume." << G4endl
874 << " Daughter physical volume "
875 << pDaughter->GetName() << G4endl
876 << " is entirely outside mother logical volume "
877 << pVolume->GetName() << " !!";
878 G4Exception("G4SmartVoxelHeader::BuildNodes()", "GeomMgt0002",
879 FatalException, message);
880 }
881
882#ifdef G4GEOMETRY_VOXELDEBUG
883 // Check for straddling volumes when debugging.
884 // If a volume is >kStraddlePercent percent over the mother
885 // boundary, print a warning.
886 //
887 if (!pLimits.IsLimited())
888 {
889 G4double width;
890 G4int kStraddlePercent=5;
891 width = maxExtents[nVol]-minExtents[nVol];
892 if ( (((motherMinExtent-minExtents[nVol])*100/width) > kStraddlePercent)
893 ||(((maxExtents[nVol]-motherMaxExtent)*100/width) > kStraddlePercent) )
894 {
895 G4cout << "**** G4SmartVoxelHeader::BuildNodes" << G4endl
896 << " WARNING : Daughter # " << nVol
897 << " name = " << pDaughter->GetName() << G4endl
898 << " Crosses mother boundary of logical volume, name = "
899 << pVolume->GetName() << G4endl
900 << " by more than " << kStraddlePercent
901 << "%" << G4endl;
902 }
903 }
904#endif
905 }
906
907 // Extents of all daughters known
908
909 // Calculate minimum slice width, only including volumes inside the limits
910 //
911 G4double minWidth = kInfinity;
912 G4double currentWidth;
913 for (nVol=0; nVol<nCandidates; nVol++)
914 {
915 // currentWidth should -always- be a positive value. Inaccurate computed extent
916 // from the solid or situations of malformed geometries (overlaps) may lead to
917 // negative values and therefore unpredictable crashes !
918 //
919 currentWidth = std::abs(maxExtents[nVol]-minExtents[nVol]);
920 if ( (currentWidth<minWidth)
921 && (maxExtents[nVol]>=pLimits.GetMinExtent(pAxis))
922 && (minExtents[nVol]<=pLimits.GetMaxExtent(pAxis)) )
923 {
924 minWidth = currentWidth;
925 }
926 }
927
928 // No. of Nodes formula - nearest integer to
929 // mother width/half min daughter width +1
930 //
931 G4double noNodesExactD = ((motherMaxExtent-motherMinExtent)*2.0/minWidth)+1.0;
932
933 // Compare with "smartless quality", i.e. the average number of slices
934 // used per contained volume.
935 //
936 G4double smartlessComputed = noNodesExactD / nCandidates;
937 G4double smartlessUser = pVolume->GetSmartless();
938 G4double smartless = (smartlessComputed <= smartlessUser)
939 ? smartlessComputed : smartlessUser;
940 G4double noNodesSmart = smartless*nCandidates;
941 G4int noNodesExactI = G4int(noNodesSmart);
942 G4int noNodes = ((noNodesSmart-noNodesExactI)>=0.5)
943 ? noNodesExactI+1 : noNodesExactI;
944 if( noNodes == 0 ) { noNodes=1; }
945
946#ifdef G4GEOMETRY_VOXELDEBUG
947 G4cout << " Smartless computed = " << smartlessComputed << G4endl
948 << " Smartless volume = " << smartlessUser
949 << " => # Smartless = " << smartless << G4endl;
950 G4cout << " Min width = " << minWidth
951 << " => # Nodes = " << noNodes << G4endl;
952#endif
953
954 if (noNodes>kMaxVoxelNodes)
955 {
956 noNodes=kMaxVoxelNodes;
957#ifdef G4GEOMETRY_VOXELDEBUG
958 G4cout << " Nodes Clipped to = " << kMaxVoxelNodes << G4endl;
959#endif
960 }
961 G4double nodeWidth = (motherMaxExtent-motherMinExtent)/noNodes;
962
963 // Create G4VoxelNodes. Will Add proxies before setting fslices
964 //
965 G4NodeVector* nodeList = new G4NodeVector();
966 if (!nodeList)
967 {
968 G4Exception("G4SmartVoxelHeader::BuildNodes()", "GeomMgt0003",
969 FatalException, "NodeList allocation error.");
970 return 0;
971 }
972 nodeList->reserve(noNodes);
973
974 for (nNode=0; nNode<noNodes; nNode++)
975 {
976 G4SmartVoxelNode *pNode;
977 pNode = new G4SmartVoxelNode(nNode);
978 if (!pNode)
979 {
980 G4Exception("G4SmartVoxelHeader::BuildNodes()", "GeomMgt0003",
981 FatalException, "Node allocation error.");
982 return 0;
983 }
984 nodeList->push_back(pNode);
985 }
986
987 // All nodes created (empty)
988
989 // Fill nodes: Step through extent lists
990 //
991 for (nVol=0; nVol<nCandidates; nVol++)
992 {
993 G4int nodeNo, minContainingNode, maxContainingNode;
994 minContainingNode = G4int((minExtents[nVol]-motherMinExtent)/nodeWidth);
995 maxContainingNode = G4int((maxExtents[nVol]-motherMinExtent)/nodeWidth);
996
997 // Only add nodes that are inside the limits of the axis
998 //
999 if ( (maxContainingNode>=0) && (minContainingNode<noNodes) )
1000 {
1001 // If max extent is on max boundary => maxContainingNode=noNodes:
1002 // should be one less as nodeList has noNodes entries
1003 //
1004 if (maxContainingNode>=noNodes)
1005 {
1006 maxContainingNode = noNodes-1;
1007 }
1008 //
1009 // Protection against protruding volumes
1010 //
1011 if (minContainingNode<0)
1012 {
1013 minContainingNode=0;
1014 }
1015 for (nodeNo=minContainingNode; nodeNo<=maxContainingNode; nodeNo++)
1016 {
1017 (*nodeList)[nodeNo]->Insert((*pCandidates)[nVol]);
1018 }
1019 }
1020 }
1021
1022 // All nodes filled
1023
1024 // Create proxy List : caller has deletion responsibility
1025 // (but we must delete nodeList *itself* - not the contents)
1026 //
1027 G4ProxyVector* proxyList = new G4ProxyVector();
1028 if (!proxyList)
1029 {
1030 G4Exception("G4SmartVoxelHeader::BuildNodes()", "GeomMgt0003",
1031 FatalException, "Proxy list allocation error.");
1032 return 0;
1033 }
1034 proxyList->reserve(noNodes);
1035
1036 //
1037 // Fill proxy List
1038 //
1039 for (nNode=0; nNode<noNodes; nNode++)
1040 {
1041 // Get rid of possible excess capacity in the internal node vector
1042 //
1043 ((*nodeList)[nNode])->Shrink();
1044 G4SmartVoxelProxy* pProxyNode = new G4SmartVoxelProxy((*nodeList)[nNode]);
1045 if (!pProxyNode)
1046 {
1047 G4Exception("G4SmartVoxelHeader::BuildNodes()", "GeomMgt0003",
1048 FatalException, "Proxy node allocation failed.");
1049 return 0;
1050 }
1051 proxyList->push_back(pProxyNode);
1052 }
1053 delete nodeList;
1054 return proxyList;
1055}

1057// ***************************************************************************
1058// Calculate a "quality value" for the specified vector of voxels.
1059// The value returned should be >0 and such that the smaller the number
1060// the higher the quality of the slice.
1061//
1062// Preconditions: pSlice must consist of G4SmartVoxelNodeProxies only
1063// Process:
1064// o Examine each node in turn, summing:
1065// no. of non-empty nodes
1066// no. of volumes in each node
1067// o Calculate Quality=sigma(volumes in nod)/(no. of non-empty nodes)
1068// if all nodes empty, return kInfinity
1069// o Call G4Exception on finding a G4SmartVoxelHeaderProxy
1070// ***************************************************************************
1071//
1072G4double G4SmartVoxelHeader::CalculateQuality(G4ProxyVector *pSlice)
1073{
1074 G4double quality;
1075 G4int nNodes = pSlice->size();
1076 G4int noContained, maxContained=0, sumContained=0, sumNonEmptyNodes=0;
1077 G4SmartVoxelNode *node;
1078
1079 for (G4int i=0; i<nNodes; i++)
1080 {
1081 if ((*pSlice)[i]->IsNode())
1082 {
1083 // Definitely a node. Add info to running totals
1084 //
1085 node = (*pSlice)[i]->GetNode();
1086 noContained = node->GetNoContained();
1087 if (noContained)
1088 {
1089 sumNonEmptyNodes++;
1090 sumContained += noContained;
1091 //
1092 // Calc maxContained for statistics
1093 //
1094 if (noContained>maxContained)
1095 {
1096 maxContained = noContained;
1097 }
1098 }
1099 }
1100 else
1101 {
1102 G4Exception("G4SmartVoxelHeader::CalculateQuality()", "GeomMgt0001",
1103 FatalException, "Not applicable to replicated volumes.");
1104 }
1105 }
1106
1107 // Calculate quality with protection against no non-empty nodes
1108 //
1109 if (sumNonEmptyNodes)
1110 {
1111 quality = sumContained/sumNonEmptyNodes;
1112 }
1113 else
1114 {
1115 quality = kInfinity;
1116 }
1117
1118#ifdef G4GEOMETRY_VOXELDEBUG
1119 G4cout << "**** G4SmartVoxelHeader::CalculateQuality" << G4endl
1120 << " Quality = " << quality << G4endl
1121 << " Nodes = " << nNodes
1122 << " of which " << sumNonEmptyNodes << " non empty" << G4endl
1123 << " Max Contained = " << maxContained << G4endl;
1124#endif
1125
1126 return quality;
1127}

1129// ***************************************************************************
1130// Examined each contained node, refines (creates a replacement additional
1131// dimension of voxels) when there is more than one voxel in the slice.
1132// Does not refine further if already limited in two dimensions (=> this
1133// is the third level of limits)
1134//
1135// Preconditions: slices (nodes) have been built.
1136// ***************************************************************************
1137//
1138void G4SmartVoxelHeader::RefineNodes(G4LogicalVolume* pVolume,
1139 G4VoxelLimits pLimits)
1140{
1141 G4int refinedDepth=0, minVolumes;
1142 G4int maxNode = fslices.size();
1143
1144 if (pLimits.IsXLimited())
1145 {
1146 refinedDepth++;
1147 }
1148 if (pLimits.IsYLimited())
1149 {
1150 refinedDepth++;
1151 }
1152 if (pLimits.IsZLimited())
1153 {
1154 refinedDepth++;
1155 }
1156
1157 // Calculate minimum number of volumes necessary to refine
1158 //
1159 switch (refinedDepth)
1160 {
1161 case 0:
1162 minVolumes=kMinVoxelVolumesLevel2;
1163 break;
1164 case 1:
1165 minVolumes=kMinVoxelVolumesLevel3;
1166 break;
1167 default:
1168 minVolumes=10000; // catch refinedDepth=3 and errors
1169 break;
1170 }
1171
1172 if (refinedDepth<2)
1173 {
1174 G4int targetNo, noContainedDaughters, minNo, maxNo, replaceNo, i;
1175 G4double sliceWidth = (fmaxExtent-fminExtent)/maxNode;
1176 G4VoxelLimits newLimits;
1177 G4SmartVoxelNode* targetNode;
1178 G4SmartVoxelProxy* targetNodeProxy;
1179 G4SmartVoxelHeader* replaceHeader;
1180 G4SmartVoxelProxy* replaceHeaderProxy;
1181 G4VolumeNosVector* targetList;
1182 G4SmartVoxelProxy* lastProxy;
1183
1184 for (targetNo=0; targetNo<maxNode; targetNo++)
1185 {
1186 // Assume all slices are nodes (see preconditions)
1187 //
1188 targetNodeProxy = fslices[targetNo];
1189 targetNode = targetNodeProxy->GetNode();
1190
1191 if (targetNode->GetNoContained() >= minVolumes)
1192 {
1193 noContainedDaughters = targetNode->GetNoContained();
1194 targetList = new G4VolumeNosVector();
1195 if (!targetList)
1196 {
1197 G4Exception("G4SmartVoxelHeader::RefineNodes()",
1198 "GeomMgt0003", FatalException,
1199 "Target volume node list allocation error.");
1200 return;
1201 }
1202 targetList->reserve(noContainedDaughters);
1203 for (i=0; i<noContainedDaughters; i++)
1204 {
1205 targetList->push_back(targetNode->GetVolume(i));
1206 }
1207 minNo = targetNode->GetMinEquivalentSliceNo();
1208 maxNo = targetNode->GetMaxEquivalentSliceNo();
1209
1210#ifdef G4GEOMETRY_VOXELDEBUG
1211 G4cout << "**** G4SmartVoxelHeader::RefineNodes" << G4endl
1212 << " Refining nodes " << minNo
1213 << " - " << maxNo << " inclusive" << G4endl;
1214#endif
1215 if (minNo > maxNo) // Delete node and list to be replaced
1216 { // and avoid further action ...
1217 delete targetNode;
1218 delete targetList;
1219 return;
1220 }
1221
1222 // Delete node proxies at start of collected sets of nodes/headers
1223 //
1224 lastProxy=0;
1225 for (replaceNo=minNo; replaceNo<=maxNo; replaceNo++)
1226 {
1227 if (lastProxy != fslices[replaceNo])
1228 {
1229 lastProxy=fslices[replaceNo];
1230 delete lastProxy;
1231 }
1232 }
1233 // Delete node to be replaced
1234 //
1235 delete targetNode;
1236
1237 // Create new headers + proxies and replace in fslices
1238 //
1239 newLimits = pLimits;
1240 newLimits.AddLimit(faxis,fminExtent+sliceWidth*minNo,
1241 fminExtent+sliceWidth*(maxNo+1));
1242 replaceHeader = new G4SmartVoxelHeader(pVolume,newLimits,
1243 targetList,replaceNo);
1244 if (!replaceHeader)
1245 {
1246 G4Exception("G4SmartVoxelHeader::RefineNodes()", "GeomMgt0003",
1247 FatalException, "Refined VoxelHeader allocation error.");
1248 return;
1249 }
1250 replaceHeader->SetMinEquivalentSliceNo(minNo);
1251 replaceHeader->SetMaxEquivalentSliceNo(maxNo);
1252 replaceHeaderProxy = new G4SmartVoxelProxy(replaceHeader);
1253 if (!replaceHeaderProxy)
1254 {
1255 G4Exception("G4SmartVoxelHeader::RefineNodes()", "GeomMgt0003",
1256 FatalException, "Refined VoxelProxy allocation error.");
1257 return;
1258 }
1259 for (replaceNo=minNo; replaceNo<=maxNo; replaceNo++)
1260 {
1261 fslices[replaceNo] = replaceHeaderProxy;
1262 }
1263 // Finished replacing current `equivalent' group
1264 //
1265 delete targetList;
1266 targetNo=maxNo;
1267 }
1268 }
1269 }
1270}

1272// ***************************************************************************
1273// Returns true if all slices have equal contents.
1274// Preconditions: all equal slices have been collected.
1275// Procedure:
1276// o checks all slice proxy pointers are equal
1277// o returns true if only one slice or all slice proxies pointers equal.
1278// ***************************************************************************
1279//
1280G4bool G4SmartVoxelHeader::AllSlicesEqual() const
1281{
1282 G4int noSlices = fslices.size();
1283 G4SmartVoxelProxy* refProxy;
1284
1285 if (noSlices>1)
1286 {
1287 refProxy=fslices[0];
1288 for (G4int i=1; i<noSlices; i++)
1289 {
1290 if (refProxy!=fslices[i])
1291 {
