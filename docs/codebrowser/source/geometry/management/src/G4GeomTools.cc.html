42///////////////////////////////////////////////////////////////////////
43//
44// Calculate area of a triangle in 2D
45
46G4double G4GeomTools::TriangleArea(G4double Ax, G4double Ay,
47 G4double Bx, G4double By,
48 G4double Cx, G4double Cy)
49{
50 return ((Bx-Ax)*(Cy-Ay) - (By-Ay)*(Cx-Ax))*0.5;
51}
52
53///////////////////////////////////////////////////////////////////////
54//
55// Calculate area of a triangle in 2D
56
57G4double G4GeomTools::TriangleArea(const G4TwoVector& A,
58 const G4TwoVector& B,
59 const G4TwoVector& C)
60{
61 G4double Ax = A.x(), Ay = A.y();
62 return ((B.x()-Ax)*(C.y()-Ay) - (B.y()-Ay)*(C.x()-Ax))*0.5;
63}

65///////////////////////////////////////////////////////////////////////
66//
67// Calculate area of a quadrilateral in 2D
68
69G4double G4GeomTools::QuadArea(const G4TwoVector& A,
70 const G4TwoVector& B,
71 const G4TwoVector& C,
72 const G4TwoVector& D)
73{
74 return ((C.x()-A.x())*(D.y()-B.y()) - (C.y()-A.y())*(D.x()-B.x()))*0.5;
75}

77///////////////////////////////////////////////////////////////////////
78//
79// Calculate area of a polygon in 2D
80
81G4double G4GeomTools::PolygonArea(const G4TwoVectorList& p)
82{
83 G4int n = p.size();
84 if (n < 3) return 0; // degerate polygon
85 G4double area = p[n-1].x()*p[0].y() - p[0].x()*p[n-1].y();
86 for(G4int i=1; i<n; ++i)
87 {
88 area += p[i-1].x()*p[i].y() - p[i].x()*p[i-1].y();
89 }
90 return area*0.5;
91}

93///////////////////////////////////////////////////////////////////////
94//
95// Point inside 2D triangle
96
97G4bool G4GeomTools::PointInTriangle(G4double Ax, G4double Ay,
98 G4double Bx, G4double By,
99 G4double Cx, G4double Cy,
100 G4double Px, G4double Py)
101
102{
103 if ((Bx-Ax)*(Cy-Ay) - (By-Ay)*(Cx-Ax) > 0.)
104 {
105 if ((Ax-Cx)*(Py-Cy) - (Ay-Cy)*(Px-Cx) < 0.) return false;
106 if ((Bx-Ax)*(Py-Ay) - (By-Ay)*(Px-Ax) < 0.) return false;
107 if ((Cx-Bx)*(Py-By) - (Cy-By)*(Px-Bx) < 0.) return false;
108 }
109 else
110 {
111 if ((Ax-Cx)*(Py-Cy) - (Ay-Cy)*(Px-Cx) > 0.) return false;
112 if ((Bx-Ax)*(Py-Ay) - (By-Ay)*(Px-Ax) > 0.) return false;
113 if ((Cx-Bx)*(Py-By) - (Cy-By)*(Px-Bx) > 0.) return false;
114 }
115 return true;
116}

118///////////////////////////////////////////////////////////////////////
119//
120// Point inside 2D triangle
121
122G4bool G4GeomTools::PointInTriangle(const G4TwoVector& A,
123 const G4TwoVector& B,
124 const G4TwoVector& C,
125 const G4TwoVector& P)
126{
127 G4double Ax = A.x(), Ay = A.y();
128 G4double Bx = B.x(), By = B.y();
129 G4double Cx = C.x(), Cy = C.y();
130 G4double Px = P.x(), Py = P.y();
131 if ((Bx-Ax)*(Cy-Ay) - (By-Ay)*(Cx-Ax) > 0.)
132 {
133 if ((Ax-Cx)*(Py-Cy) - (Ay-Cy)*(Px-Cx) < 0.) return false;
134 if ((Bx-Ax)*(Py-Ay) - (By-Ay)*(Px-Ax) < 0.) return false;
135 if ((Cx-Bx)*(Py-By) - (Cy-By)*(Px-Bx) < 0.) return false;
136 }
137 else
138 {
139 if ((Ax-Cx)*(Py-Cy) - (Ay-Cy)*(Px-Cx) > 0.) return false;
140 if ((Bx-Ax)*(Py-Ay) - (By-Ay)*(Px-Ax) > 0.) return false;
141 if ((Cx-Bx)*(Py-By) - (Cy-By)*(Px-Bx) > 0.) return false;
142 }
143 return true;
144}

146///////////////////////////////////////////////////////////////////////
147//
148// Point inside 2D polygon
149
150G4bool G4GeomTools::PointInPolygon(const G4TwoVector& p,
151 const G4TwoVectorList& v)
152{
153 G4int Nv = v.size();
154 G4bool in = false;
155 for (G4int i = 0, k = Nv - 1; i < Nv; k = i++)
156 {
157 if ((v[i].y() > p.y()) != (v[k].y() > p.y()))
158 {
159 G4double ctg = (v[k].x()-v[i].x())/(v[k].y()-v[i].y());
160 in ^= (p.x() < (p.y()-v[i].y())*ctg + v[i].x());
161 }
162 }
163 return in;
164}

166///////////////////////////////////////////////////////////////////////
167//
168// Detemine whether 2D polygon is convex or not
169
170G4bool G4GeomTools::IsConvex(const G4TwoVectorList& polygon)
171{
172 static const G4double kCarTolerance =
173 G4GeometryTolerance::GetInstance()->GetSurfaceTolerance();
174
175 G4bool gotNegative = false;
176 G4bool gotPositive = false;
177 G4int n = polygon.size();
178 if (n <= 0) return false;
179 for (G4int icur=0; icur<n; ++icur)
180 {
181 G4int iprev = (icur == 0) ? n-1 : icur-1;
182 G4int inext = (icur == n-1) ? 0 : icur+1;
183 G4TwoVector e1 = polygon[icur] - polygon[iprev];
184 G4TwoVector e2 = polygon[inext] - polygon[icur];
185 G4double cross = e1.x()*e2.y() - e1.y()*e2.x();
186 if (std::abs(cross) < kCarTolerance) return false;
187 if (cross < 0) gotNegative = true;
188 if (cross > 0) gotPositive = true;
189 if (gotNegative && gotPositive) return false;
190 }
191 return true;
192}

194///////////////////////////////////////////////////////////////////////
195//
196// Triangulate simple polygon
197
198G4bool G4GeomTools::TriangulatePolygon(const G4TwoVectorList& polygon,
199 G4TwoVectorList& result)
200{
201 result.resize(0);
202 std::vector<G4int> triangles;
203 G4bool reply = TriangulatePolygon(polygon,triangles);
204
205 G4int n = triangles.size();
206 for (G4int i=0; i<n; ++i) result.push_back(polygon[triangles[i]]);
207 return reply;
208}

210///////////////////////////////////////////////////////////////////////
211//
212// Triangulation of a simple polygon by "ear clipping"
213
214G4bool G4GeomTools::TriangulatePolygon(const G4TwoVectorList& polygon,
215 std::vector<G4int>& result)
216{
217 result.resize(0);
218
219 // allocate and initialize list of Vertices in polygon
220 //
221 G4int n = polygon.size();
222 if (n < 3) return false;
223
224 // we want a counter-clockwise polygon in V
225 //
226 G4double area = G4GeomTools::PolygonArea(polygon);
227 G4int* V = new G4int[n];
228 if (area > 0.)
229 for (G4int i=0; i<n; ++i) V[i] = i;
230 else
231 for (G4int i=0; i<n; ++i) V[i] = (n-1)-i;
232
233 // Triangulation: remove nv-2 Vertices, creating 1 triangle every time
234 //
235 G4int nv = n;
236 G4int count = 2*nv; // error detection counter
237 for(G4int b=nv-1; nv>2; )
238 {
239 // ERROR: if we loop, it is probably a non-simple polygon
240 if ((count--) <= 0)
241 {
242 delete[] V;
243 if (area < 0.) std::reverse(result.begin(),result.end());
244 return false;
245 }
246
247 // three consecutive vertices in current polygon, <a,b,c>
248 G4int a = (b < nv) ? b : 0; // previous
249 b = (a+1 < nv) ? a+1 : 0; // current
250 G4int c = (b+1 < nv) ? b+1 : 0; // next
251
252 if (CheckSnip(polygon, a,b,c, nv,V))
253 {
254 // output Triangle
255 result.push_back(V[a]);
256 result.push_back(V[b]);
257 result.push_back(V[c]);
258
259 // remove vertex b from remaining polygon
260 nv--;
261 for(G4int i=b; i<nv; ++i) V[i] = V[i+1];
262
263 count = 2*nv; // resest error detection counter
264 }
265 }
266 delete[] V;
267 if (area < 0.) std::reverse(result.begin(),result.end());
268 return true;
269}

271///////////////////////////////////////////////////////////////////////
272//
273// Helper function for "ear clipping" polygon triangulation.
274// Check for a valid snip
275
276G4bool G4GeomTools::CheckSnip(const G4TwoVectorList& contour,
277 G4int a, G4int b, G4int c,
278 G4int n, const G4int* V)
279{
280 static const G4double kCarTolerance =
281 G4GeometryTolerance::GetInstance()->GetSurfaceTolerance();
282
283 // check orientation of Triangle
284 G4double Ax = contour[V[a]].x(), Ay = contour[V[a]].y();
285 G4double Bx = contour[V[b]].x(), By = contour[V[b]].y();
286 G4double Cx = contour[V[c]].x(), Cy = contour[V[c]].y();
287 if ((Bx-Ax)*(Cy-Ay) - (By-Ay)*(Cx-Ax) < kCarTolerance) return false;
288
289 // check that there is no point inside Triangle
290 G4double xmin = std::min(std::min(Ax,Bx),Cx);
291 G4double xmax = std::max(std::max(Ax,Bx),Cx);
292 G4double ymin = std::min(std::min(Ay,By),Cy);
293 G4double ymax = std::max(std::max(Ay,By),Cy);
294 for (G4int i=0; i<n; ++i)
295 {
296 if((i == a) || (i == b) || (i == c)) continue;
297 G4double Px = contour[V[i]].x();
298 if (Px < xmin || Px > xmax) continue;
299 G4double Py = contour[V[i]].y();
300 if (Py < ymin || Py > ymax) continue;
301 if (PointInTriangle(Ax,Ay,Bx,By,Cx,Cy,Px,Py)) return false;
302 }
303 return true;
304}

306///////////////////////////////////////////////////////////////////////
307//
308// Remove collinear and coincident points from 2D polygon
309
310void G4GeomTools::RemoveRedundantVertices(G4TwoVectorList& polygon,
311 std::vector<G4int>& iout,
312 G4double tolerance)
313{
314 iout.resize(0);
315 // set tolerance squared
316 G4double delta = tolerance*tolerance;
317 // set special value to mark vertices for removal
318 G4double removeIt = kInfinity;
319
320 G4int nv = polygon.size();
321
322 // Main loop: check every three consecutive points, if the points
323 // are collinear then mark middle point for removal
324 //
325 G4int icur = 0, iprev = 0, inext = 0, nout = 0;
326 for (G4int i=0; i<nv; ++i)
327 {
328 icur = i; // index of current point
329
330 for (G4int k=1; k<nv+1; ++k) // set index of previous point
331 {
332 iprev = icur - k;
333 if (iprev < 0) iprev += nv;
334 if (polygon[iprev].x() != removeIt) break;
335 }
336
337 for (G4int k=1; k<nv+1; ++k) // set index of next point
338 {
339 inext = icur + k;
340 if (inext >= nv) inext -= nv;
341 if (polygon[inext].x() != removeIt) break;
342 }
343
344 if (iprev == inext) break; // degenerate polygon, stop
345
346 // Calculate parameters of triangle (iprev->icur->inext),
347 // if triangle is too small or too narrow then mark current
348 // point for removal
349 G4TwoVector e1 = polygon[iprev] - polygon[icur];
350 G4TwoVector e2 = polygon[inext] - polygon[icur];
351
352 // Check length of edges, then check height of the triangle
353 G4double leng1 = e1.mag2();
354 G4double leng2 = e2.mag2();
355 G4double leng3 = (e2-e1).mag2();
356 if (leng1 <= delta || leng2 <= delta || leng3 <= delta)
357 {
358 polygon[icur].setX(removeIt); nout++;
359 }
360 else
361 {
362 G4double lmax = std::max(std::max(leng1,leng2),leng3);
363 G4double area = std::abs(e1.x()*e2.y()-e1.y()*e2.x())*0.5;
364 if (area/std::sqrt(lmax) <= std::abs(tolerance))
365 {
366 polygon[icur].setX(removeIt); nout++;
367 }
368 }
369 }
370
371 // Remove marked points
372 //
373 icur = 0;
374 if (nv - nout < 3) // degenerate polygon, remove all points
375 {
376 for (G4int i=0; i<nv; ++i) iout.push_back(i);
377 polygon.resize(0);
378 nv = 0;
379 }
380 for (G4int i=0; i<nv; ++i) // move points, if required
381 {
382 if (polygon[i].x() != removeIt)
383 polygon[icur++] = polygon[i];
384 else
385 iout.push_back(i);
386 }
387 if (icur < nv) polygon.resize(icur);
388 return;
389}

391///////////////////////////////////////////////////////////////////////
392//
393// Find bounding rectangle of a disk sector
394
395G4bool G4GeomTools::DiskExtent(G4double rmin, G4double rmax,
396 G4double startPhi, G4double delPhi,
397 G4TwoVector& pmin, G4TwoVector& pmax)
398{
399 static const G4double kCarTolerance =
400 G4GeometryTolerance::GetInstance()->GetSurfaceTolerance();
401
402 // check parameters
403 //
404 pmin.set(0,0);
405 pmax.set(0,0);
406 if (rmin < 0) return false;
407 if (rmax <= rmin + kCarTolerance) return false;
408 if (delPhi <= 0 + kCarTolerance) return false;
409
410 // calculate extent
411 //
412 pmin.set(-rmax,-rmax);
413 pmax.set( rmax, rmax);
414 if (delPhi >= CLHEP::twopi) return true;
415
416 DiskExtent(rmin,rmax,
417 std::sin(startPhi),std::cos(startPhi),
418 std::sin(startPhi+delPhi),std::cos(startPhi+delPhi),
419 pmin,pmax);
420 return true;
421}

423///////////////////////////////////////////////////////////////////////
424//
425// Find bounding rectangle of a disk sector, fast version.
426// No check of parameters !!!
427
428void G4GeomTools::DiskExtent(G4double rmin, G4double rmax,
429 G4double sinStart, G4double cosStart,
430 G4double sinEnd, G4double cosEnd,
431 G4TwoVector& pmin, G4TwoVector& pmax)
432{
433 static const G4double kCarTolerance =
434 G4GeometryTolerance::GetInstance()->GetSurfaceTolerance();
435
436 // check if 360 degrees
437 //
438 pmin.set(-rmax,-rmax);
439 pmax.set( rmax, rmax);
440
441 if (std::abs(sinEnd-sinStart) < kCarTolerance &&
442 std::abs(cosEnd-cosStart) < kCarTolerance) return;
443
444 // get start and end quadrants
445 //
446 // 1 | 0
447 // ---+---
448 // 3 | 2
449 //
450 G4int icase = (cosEnd < 0) ? 1 : 0;
451 if (sinEnd < 0) icase += 2;
452 if (cosStart < 0) icase += 4;
453 if (sinStart < 0) icase += 8;
454
455 switch (icase)
456 {
457 // start quadrant 0
458 case 0: // start->end : 0->0
459 if (sinEnd < sinStart) break;
460 pmin.set(rmin*cosEnd,rmin*sinStart);
461 pmax.set(rmax*cosStart,rmax*sinEnd );
462 break;
463 case 1: // start->end : 0->1
464 pmin.set(rmax*cosEnd,std::min(rmin*sinStart,rmin*sinEnd));
465 pmax.set(rmax*cosStart,rmax );
466 break;
467 case 2: // start->end : 0->2
468 pmin.set(-rmax,-rmax);
469 pmax.set(std::max(rmax*cosStart,rmax*cosEnd),rmax);
470 break;
471 case 3: // start->end : 0->3
472 pmin.set(-rmax,rmax*sinEnd);
473 pmax.set(rmax*cosStart,rmax);
474 break;
475 // start quadrant 1
476 case 4: // start->end : 1->0
477 pmin.set(-rmax,-rmax);
478 pmax.set(rmax,std::max(rmax*sinStart,rmax*sinEnd));
479 break;
480 case 5: // start->end : 1->1
481 if (sinEnd > sinStart) break;
482 pmin.set(rmax*cosEnd,rmin*sinEnd );
483 pmax.set(rmin*cosStart,rmax*sinStart);
484 break;
485 case 6: // start->end : 1->2
486 pmin.set(-rmax,-rmax);
487 pmax.set(rmax*cosEnd,rmax*sinStart);
488 break;
489 case 7: // start->end : 1->3
490 pmin.set(-rmax,rmax*sinEnd);
491 pmax.set(std::max(rmin*cosStart,rmin*cosEnd),rmax*sinStart);
492 break;
493 // start quadrant 2
494 case 8: // start->end : 2->0
495 pmin.set(std::min(rmin*cosStart,rmin*cosEnd),rmax*sinStart);
496 pmax.set(rmax,rmax*sinEnd);
497 break;
498 case 9: // start->end : 2->1
499 pmin.set(rmax*cosEnd,rmax*sinStart);
500 pmax.set(rmax,rmax);
501 break;
502 case 10: // start->end : 2->2
503 if (sinEnd < sinStart) break;
504 pmin.set(rmin*cosStart,rmax*sinStart);
505 pmax.set(rmax*cosEnd,rmin*sinEnd );
506 break;
507 case 11: // start->end : 2->3
508 pmin.set(-rmax,std::min(rmax*sinStart,rmax*sinEnd));
509 pmax.set(rmax,rmax);
510 break;
511 // start quadrant 3
512 case 12: // start->end : 3->0
513 pmin.set(rmax*cosStart,-rmax);
514 pmax.set(rmax,rmax*sinEnd);
515 break;
516 case 13: // start->end : 3->1
517 pmin.set(std::min(rmax*cosStart,rmax*cosEnd),-rmax);
518 pmax.set(rmax,rmax);
519 break;
520 case 14: // start->end : 3->2
521 pmin.set(rmax*cosStart,-rmax);
522 pmax.set(rmax*cosEnd,std::max(rmin*sinStart,rmin*sinEnd));
523 break;
524 case 15: // start->end : 3->3
525 if (sinEnd > sinStart) break;
526 pmin.set(rmax*cosStart,rmax*sinEnd);
527 pmax.set(rmin*cosEnd,rmin*sinStart);
528 break;
529 }
530 return;
531}

533///////////////////////////////////////////////////////////////////////
534//
535// Compute the circumference (perimeter) of an ellipse
536
537G4double G4GeomTools::EllipsePerimeter(G4double pA, G4double pB)
538{
539 G4double x = std::abs(pA);
540 G4double y = std::abs(pB);
541 G4double a = std::max(x,y);
542 G4double b = std::min(x,y);
543 G4double e = std::sqrt((1. - b/a)*(1. + b/a));
544 return 4. * a * comp_ellint_2(e);
545}

547///////////////////////////////////////////////////////////////////////
548//
549// Compute the lateral surface area of an elliptic cone
550
551G4double G4GeomTools::EllipticConeLateralArea(G4double pA,
552 G4double pB,
553 G4double pH)
554{
555 G4double x = std::abs(pA);
556 G4double y = std::abs(pB);
557 G4double h = std::abs(pH);
558 G4double a = std::max(x,y);
559 G4double b = std::min(x,y);
560 G4double e = std::sqrt((1. - b/a)*(1. + b/a)) / std::hypot(1.,b/h);
561 return 2. * a * std::hypot(b,h) * comp_ellint_2(e);
562}

564///////////////////////////////////////////////////////////////////////
565//
566// Compute Elliptical Integral of the Second Kind
567//
568// The algorithm is based upon Carlson B.C., "Computation of real
569// or complex elliptic integrals", Numerical Algorithms,
570// Volume 10, Issue 1, 1995 (see equations 2.36 - 2.39)
571//
572// The code was adopted from C code at:
573// http://paulbourke.net/geometry/ellipsecirc/
574
575G4double G4GeomTools::comp_ellint_2(G4double e)
576{
577 const G4double eps = 1. / 134217728.; // 1 / 2^27
578
579 G4double a = 1.;
580 G4double b = std::sqrt((1. - e)*(1. + e));
581 if (b == 1.) return CLHEP::halfpi;
582 if (b == 0.) return 1.;
583
584 G4double x = 1.;
585 G4double y = b;
586 G4double S = 0.;
587 G4double M = 1.;
588 while (x - y > eps*y) {
589 G4double tmp = (x + y) * 0.5;
590 y = std::sqrt(x*y);
591 x = tmp;
592 M += M;
593 S += M * (x - y)*(x - y);
594 }
595 return 0.5 * CLHEP::halfpi * ((a + b)*(a + b) - S) / (x + y);
596}

621///////////////////////////////////////////////////////////////////////
622//
623// Calculate area of a polygon in 3D
624
625G4ThreeVector G4GeomTools::PolygonAreaNormal(const G4ThreeVectorList& p)
626{
627 G4int n = p.size();
628 if (n < 3) return G4ThreeVector(0,0,0); // degerate polygon
629 G4ThreeVector normal = p[n-1].cross(p[0]);
630 for(G4int i=1; i<n; ++i)
631 {
632 normal += p[i-1].cross(p[i]);
633 }
634 return normal*0.5;
635}

637///////////////////////////////////////////////////////////////////////
638//
639// Calculate distance between point P and line segment AB in 3D
640
641G4double G4GeomTools::DistancePointSegment(const G4ThreeVector& P,
642 const G4ThreeVector& A,
643 const G4ThreeVector& B)
644{
645 G4ThreeVector AP = P - A;
646 G4ThreeVector AB = B - A;
647
648 G4double u = AP.dot(AB);
649 if (u <= 0) return AP.mag(); // closest point is A
650
651 G4double len2 = AB.mag2();
652 if (u >= len2) return (B-P).mag(); // closest point is B
653
654 return ((u/len2)*AB - AP).mag(); // distance to line
655}

657///////////////////////////////////////////////////////////////////////
658//
659// Find closest point on line segment in 3D
660
661G4ThreeVector
662G4GeomTools::ClosestPointOnSegment(const G4ThreeVector& P,
663 const G4ThreeVector& A,
664 const G4ThreeVector& B)
665{
666 G4ThreeVector AP = P - A;
667 G4ThreeVector AB = B - A;
668
669 G4double u = AP.dot(AB);
670 if (u <= 0) return A; // closest point is A
671
672 G4double len2 = AB.mag2();
673 if (u >= len2) return B; // closest point is B
674
675 G4double t = u/len2;
676 return A + t*AB; // closest point on segment
677}

679///////////////////////////////////////////////////////////////////////
680//
681// Find closest point on triangle in 3D.
682//
683// The implementation is based on the algorithm published in
684// "Geometric Tools for Computer Graphics", Philip J Scheider and
685// David H Eberly, Elsevier Science (USA), 2003.
686//
687// The algorithm is also available at:
688// http://www.geometrictools.com/Documentation/DistancePoint3Triangle3.pdf
689
690G4ThreeVector
691G4GeomTools::ClosestPointOnTriangle(const G4ThreeVector& P,
692 const G4ThreeVector& A,
693 const G4ThreeVector& B,
694 const G4ThreeVector& C)
695{
696 G4ThreeVector diff = A - P;
697 G4ThreeVector edge0 = B - A;
698 G4ThreeVector edge1 = C - A;
699
700 G4double a = edge0.mag2();
701 G4double b = edge0.dot(edge1);
702 G4double c = edge1.mag2();
703 G4double d = diff.dot(edge0);
704 G4double e = diff.dot(edge1);
705
706 G4double det = a*c - b*b;
707 G4double t0 = b*e - c*d;
708 G4double t1 = b*d - a*e;
709
710 /*
711 ^ t1
712 \ 2 |
713 \ |
714 \ | regions
715 \|
716 C
717 |\
718 3 | \ 1
719 | \
720 | 0 \
721 | \
722 ---- A --- B ----> t0
723 | \
724 4 | 5 \ 6
725 | \
726 */
727
728 G4int region = -1;
729 if (t0+t1 <= det)
730 region = (t0 < 0) ? ((t1 < 0) ? 4 : 3) : ((t1 < 0) ? 5 : 0);
731 else
732 region = (t0 < 0) ? 2 : ((t1 < 0) ? 6 : 1);
733
734 switch (region)
735 {
736 case 0: // interior of triangle
737 {
738 G4double invDet = 1./det;
739 return A + (t0*invDet)*edge0 + (t1*invDet)*edge1;
740 }
741 case 1: // edge BC
742 {
743 G4double numer = c + e - b - d;
744 if (numer <= 0) return C;
745 G4double denom = a - 2*b + c;
746 return (numer >= denom) ? B : C + (numer/denom)*(edge0-edge1);
747 }
748 case 2: // edge AC or BC
749 {
750 G4double tmp0 = b + d;
751 G4double tmp1 = c + e;
752 if (tmp1 > tmp0)
753 {
754 G4double numer = tmp1 - tmp0;
755 G4double denom = a - 2*b + c;
756 return (numer >= denom) ? B : C + (numer/denom)*(edge0-edge1);
757 }
758 // same: (e >= 0) ? A : ((-e >= c) ? C : A + (-e/c)*edge1)
759 return (tmp1 <= 0) ? C : (( e >= 0) ? A : A + (-e/c)*edge1);
760 }
761 case 3: // edge AC
762 return (e >= 0) ? A : ((-e >= c) ? C : A + (-e/c)*edge1);
763
764 case 4: // edge AB or AC
765 if (d < 0) return (-d >= a) ? B : A + (-d/a)*edge0;
766 return (e >= 0) ? A : ((-e >= c) ? C : A + (-e/c)*edge1);
767
768 case 5: // edge AB
769 return (d >= 0) ? A : ((-d >= a) ? B : A + (-d/a)*edge0);
770
771 case 6: // edge AB or BC
772 {
773 G4double tmp0 = b + e;
774 G4double tmp1 = a + d;
775 if (tmp1 > tmp0)
776 {
777 G4double numer = tmp1 - tmp0;
778 G4double denom = a - 2*b + c;
779 return (numer >= denom) ? C : B + (numer/denom)*(edge1-edge0);
780 }
781 // same: (d >= 0) ? A : ((-d >= a) ? B : A + (-d/a)*edge0)
782 return (tmp1 <= 0) ? B : (( d >= 0) ? A : A + (-d/a)*edge0);
783 }
784 default: // impossible case
785 return G4ThreeVector(kInfinity,kInfinity,kInfinity);
786 }
787}

789///////////////////////////////////////////////////////////////////////
790//
791// Calculate bounding box of a spherical sector
792
793G4bool
794G4GeomTools::SphereExtent(G4double rmin, G4double rmax,
795 G4double startTheta, G4double delTheta,
796 G4double startPhi, G4double delPhi,
797 G4ThreeVector& pmin, G4ThreeVector& pmax)
798{
799 static const G4double kCarTolerance =
800 G4GeometryTolerance::GetInstance()->GetSurfaceTolerance();
801
802 // check parameters
803 //
804 pmin.set(0,0,0);
805 pmax.set(0,0,0);
806 if (rmin < 0) return false;
807 if (rmax <= rmin + kCarTolerance) return false;
808 if (delTheta <= 0 + kCarTolerance) return false;
809 if (delPhi <= 0 + kCarTolerance) return false;
810
811 G4double stheta = startTheta;
812 G4double dtheta = delTheta;
813 if (stheta < 0 && stheta > CLHEP::pi) return false;
814 if (stheta + dtheta > CLHEP::pi) dtheta = CLHEP::pi - stheta;
815 if (dtheta <= 0 + kCarTolerance) return false;
816
817 // calculate extent
818 //
819 pmin.set(-rmax,-rmax,-rmax);
820 pmax.set( rmax, rmax, rmax);
821 if (dtheta >= CLHEP::pi && delPhi >= CLHEP::twopi) return true;
822
823 G4double etheta = stheta + dtheta;
824 G4double sinStart = std::sin(stheta);
825 G4double cosStart = std::cos(stheta);
826 G4double sinEnd = std::sin(etheta);
827 G4double cosEnd = std::cos(etheta);
828
829 G4double rhomin = rmin*std::min(sinStart,sinEnd);
830 G4double rhomax = rmax;
831 if (stheta > CLHEP::halfpi) rhomax = rmax*sinStart;
832 if (etheta < CLHEP::halfpi) rhomax = rmax*sinEnd;
833
834 G4TwoVector xymin,xymax;
835 DiskExtent(rhomin,rhomax,
836 std::sin(startPhi),std::cos(startPhi),
837 std::sin(startPhi+delPhi),std::cos(startPhi+delPhi),
838 xymin,xymax);
839
840 G4double zmin = std::min(rmin*cosEnd,rmax*cosEnd);
841 G4double zmax = std::max(rmin*cosStart,rmax*cosStart);
842 pmin.set(xymin.x(),xymin.y(),zmin);
843 pmax.set(xymax.x(),xymax.y(),zmax);
844 return true;
845}
