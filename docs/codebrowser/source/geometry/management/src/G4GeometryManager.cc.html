83// ***************************************************************************
84// Closes geometry - performs sanity checks and optionally builds optimisation
85// for placed volumes (always built for replicas & parameterised).
86// NOTE: Currently no sanity checks are performed.
87// Applies to just a specific subtree if a physical volume is specified.
88// ***************************************************************************
89//
90G4bool G4GeometryManager::CloseGeometry(G4bool pOptimise, G4bool verbose,
91 G4VPhysicalVolume* pVolume)
92{
93 if (!fIsClosed)
94 {
95 if (pVolume)
96 {
97 BuildOptimisations(pOptimise, pVolume);
98 }
99 else
100 {
101 BuildOptimisations(pOptimise, verbose);
102 }
103 fIsClosed=true;
104 }
105 return true;
106}
107
108// ***************************************************************************
109// Opens the geometry and removes optimisations (optionally, related to just
110// the specified logical-volume).
111// Applies to just a specific subtree if a physical volume is specified.
112// ***************************************************************************
113//
114void G4GeometryManager::OpenGeometry(G4VPhysicalVolume* pVolume)
115{
116 if (fIsClosed)
117 {
118 if (pVolume)
119 {
120 DeleteOptimisations(pVolume);
121 }
122 else
123 {
124 DeleteOptimisations();
125 }
126 fIsClosed=false;
127 }
128}

162// ***************************************************************************
163// Creates optimisation info. Builds all voxels if allOpts=true
164// otherwise it builds voxels only for replicated volumes.
165// ***************************************************************************
166//
167void G4GeometryManager::BuildOptimisations(G4bool allOpts, G4bool verbose)
168{
169 G4Timer timer;
170 G4Timer allTimer;
171 std::vector<G4SmartVoxelStat> stats;
172 if (verbose) { allTimer.Start(); }
173
174 G4LogicalVolumeStore* Store = G4LogicalVolumeStore::GetInstance();
175 G4LogicalVolume* volume;
176 G4SmartVoxelHeader* head;
177
178 for (size_t n=0; n<Store->size(); n++)
179 {
180 if (verbose) timer.Start();
181 volume=(*Store)[n];
182 // For safety, check if there are any existing voxels and
183 // delete before replacement
184 //
185 head = volume->GetVoxelHeader();
186 delete head;
187 volume->SetVoxelHeader(0);
188 if ( ( (volume->IsToOptimise())
189 && (volume->GetNoDaughters()>=kMinVoxelVolumesLevel1&&allOpts) )
190 || ( (volume->GetNoDaughters()==1)
191 && (volume->GetDaughter(0)->IsReplicated()==true)
192 && (volume->GetDaughter(0)->GetRegularStructureId()!=1) ) )
193 {
194#ifdef G4GEOMETRY_VOXELDEBUG
195 G4cout << "**** G4GeometryManager::BuildOptimisations" << G4endl
196 << " Examining logical volume name = "
197 << volume->GetName() << G4endl;
198#endif
199 head = new G4SmartVoxelHeader(volume);
200 if (head)
201 {
202 volume->SetVoxelHeader(head);
203 }
204 else
205 {
206 std::ostringstream message;
207 message << "VoxelHeader allocation error." << G4endl
208 << "Allocation of new VoxelHeader" << G4endl
209 << " for volume " << volume->GetName() << " failed.";
210 G4Exception("G4GeometryManager::BuildOptimisations()", "GeomMgt0003",
211 FatalException, message);
212 }
213 if (verbose)
214 {
215 timer.Stop();
216 stats.push_back( G4SmartVoxelStat( volume, head,
217 timer.GetSystemElapsed(),
218 timer.GetUserElapsed() ) );
219 }
220 }
221 else
222 {
223 // Don't create voxels for this node
224#ifdef G4GEOMETRY_VOXELDEBUG
225 G4cout << "**** G4GeometryManager::BuildOptimisations" << G4endl
226 << " Skipping logical volume name = " << volume->GetName()
227 << G4endl;
228#endif
229 }
230 }
231 if (verbose)
232 {
233 allTimer.Stop();
234 ReportVoxelStats( stats, allTimer.GetSystemElapsed()
235 + allTimer.GetUserElapsed() );
236 }
237}

239// ***************************************************************************
240// Creates optimisation info for the specified volumes subtree.
241// ***************************************************************************
242//
243void G4GeometryManager::BuildOptimisations(G4bool allOpts,
244 G4VPhysicalVolume* pVolume)
245{
246 if (!pVolume) { return; }
247
248 // Retrieve the mother logical volume, if not NULL,
249 // otherwise apply global optimisation for the world volume
250 //
251 G4LogicalVolume* tVolume = pVolume->GetMotherLogical();
252 if (!tVolume) { return BuildOptimisations(allOpts, false); }
253
254 G4SmartVoxelHeader* head = tVolume->GetVoxelHeader();
255 delete head;
256 tVolume->SetVoxelHeader(0);
257 if ( ( (tVolume->IsToOptimise())
258 && (tVolume->GetNoDaughters()>=kMinVoxelVolumesLevel1&&allOpts) )
259 || ( (tVolume->GetNoDaughters()==1)
260 && (tVolume->GetDaughter(0)->IsReplicated()==true) ) )
261 {
262 head = new G4SmartVoxelHeader(tVolume);
263 if (head)
264 {
265 tVolume->SetVoxelHeader(head);
266 }
267 else
268 {
269 std::ostringstream message;
270 message << "VoxelHeader allocation error." << G4endl
271 << "Allocation of new VoxelHeader" << G4endl
272 << " for volume " << tVolume->GetName() << " failed.";
273 G4Exception("G4GeometryManager::BuildOptimisations()", "GeomMgt0003",
274 FatalException, message);
275 }
276 }
277 else
278 {
279 // Don't create voxels for this node
280#ifdef G4GEOMETRY_VOXELDEBUG
281 G4cout << "**** G4GeometryManager::BuildOptimisations" << G4endl
282 << " Skipping logical volume name = " << tVolume->GetName()
283 << G4endl;
284#endif
285 }
286
287 // Scan recursively the associated logical volume tree
288 //
289 tVolume = pVolume->GetLogicalVolume();
290 if (tVolume->GetNoDaughters())
291 {
292 BuildOptimisations(allOpts, tVolume->GetDaughter(0));
293 }
294}

296// ***************************************************************************
297// Removes all optimisation info.
298// Loops over all logical volumes, deleting non-null voxels pointers,
299// ***************************************************************************
300//
301void G4GeometryManager::DeleteOptimisations()
302{
303 G4LogicalVolume* tVolume = 0;
304 G4LogicalVolumeStore* Store = G4LogicalVolumeStore::GetInstance();
305 for (size_t n=0; n<Store->size(); n++)
306 {
307 tVolume=(*Store)[n];
308 delete tVolume->GetVoxelHeader();
309 tVolume->SetVoxelHeader(0);
310 }
311}
312
313// ***************************************************************************
314// Removes optimisation info for the specified subtree.
315// Scans recursively all daughter volumes, deleting non-null voxels pointers.
316// ***************************************************************************
317//
318void G4GeometryManager::DeleteOptimisations(G4VPhysicalVolume* pVolume)
319{
320 if (!pVolume) { return; }
321
322 // Retrieve the mother logical volume, if not NULL,
323 // otherwise global deletion to world volume.
324 //
325 G4LogicalVolume* tVolume = pVolume->GetMotherLogical();
326 if (!tVolume) { return DeleteOptimisations(); }
327 delete tVolume->GetVoxelHeader();
328 tVolume->SetVoxelHeader(0);
329
330 // Scan recursively the associated logical volume tree
331 //
332 tVolume = pVolume->GetLogicalVolume();
333 if (tVolume->GetNoDaughters())
334 {
335 DeleteOptimisations(tVolume->GetDaughter(0));
336 }
337}

339// ***************************************************************************
340// Sets the maximum extent of the world volume. The operation is allowed only
341// if NO solids have been created already.
342// ***************************************************************************
343//
344void G4GeometryManager::SetWorldMaximumExtent(G4double extent)
345{
346 if (G4SolidStore::GetInstance()->size())
347 {
348 // Sanity check to assure that extent is fixed BEFORE creating
349 // any geometry object (solids in this case)
350 //
351 G4Exception("G4GeometryManager::SetMaximumExtent()",
352 "GeomMgt0003", FatalException,
353 "Extent can be set only BEFORE creating any geometry object!");
354 }
355 G4GeometryTolerance::GetInstance()->SetSurfaceTolerance(extent);
356}

358// ***************************************************************************
359// Reports statistics on voxel optimisation when closing geometry.
360// ***************************************************************************
361//
362void
363G4GeometryManager::ReportVoxelStats( std::vector<G4SmartVoxelStat> & stats,
364 G4double totalCpuTime )
365{
366 G4cout << "G4GeometryManager::ReportVoxelStats -- Voxel Statistics"
367 << G4endl << G4endl;
368
369 //
370 // Get total memory use
371 //
372 G4int i, nStat = stats.size();
373 G4long totalMemory = 0;
374
375 for( i=0;i<nStat;++i ) { totalMemory += stats[i].GetMemoryUse(); }
376
377 G4cout << " Total memory consumed for geometry optimisation: "
378 << totalMemory/1024 << " kByte" << G4endl;
379 G4cout << " Total CPU time elapsed for geometry optimisation: "
380 << std::setprecision(2) << totalCpuTime << " seconds"
381 << std::setprecision(6) << G4endl;
382
383 //
384 // First list: sort by total CPU time
385 //
386 std::sort( stats.begin(), stats.end(), G4SmartVoxelStat::ByCpu() );
387
388 G4int nPrint = nStat > 10 ? 10 : nStat;
389
390 if (nPrint)
391 {
392 G4cout << "\n Voxelisation: top CPU users:" << G4endl;
393 G4cout << " Percent Total CPU System CPU Memory Volume\n"
394 << " ------- ---------- ---------- -------- ----------"
395 << G4endl;
396 // 12345678901.234567890123.234567890123.234567890123k .
397 }
398
399 for(i=0;i<nPrint;++i)
400 {
401 G4double total = stats[i].GetTotalTime();
402 G4double system = stats[i].GetSysTime();
403 G4double perc = 0.0;
404
405 if (system < 0) { system = 0.0; }
406 if ((total < 0) || (totalCpuTime < perMillion))
407 { total = 0; }
408 else
409 { perc = total*100/totalCpuTime; }
410
411 G4cout << std::setprecision(2)
412 << std::setiosflags(std::ios::fixed|std::ios::right)
413 << std::setw(11) << perc
414 << std::setw(13) << total
415 << std::setw(13) << system
416 << std::setw(13) << (stats[i].GetMemoryUse()+512)/1024
417 << "k " << std::setiosflags(std::ios::left)
418 << stats[i].GetVolume()->GetName()
419 << std::resetiosflags(std::ios::floatfield|std::ios::adjustfield)
420 << std::setprecision(6)
421 << G4endl;
422 }
423
424 //
425 // Second list: sort by memory use
426 //
427 std::sort( stats.begin(), stats.end(), G4SmartVoxelStat::ByMemory() );
428
429 if (nPrint)
430 {
431 G4cout << "\n Voxelisation: top memory users:" << G4endl;
432 G4cout << " Percent Memory Heads Nodes Pointers Total CPU Volume\n"
433 << " ------- -------- ------ ------ -------- ---------- ----------"
434 << G4endl;
435 // 12345678901.2345678901k .23456789.23456789.2345678901.234567890123. .
436 }
437
438 for(i=0;i<nPrint;++i)
439 {
440 G4long memory = stats[i].GetMemoryUse();
441 G4double totTime = stats[i].GetTotalTime();
442 if (totTime < 0) { totTime = 0.0; }
443
444 G4cout << std::setprecision(2)
445 << std::setiosflags(std::ios::fixed|std::ios::right)
446 << std::setw(11) << G4double(memory*100)/G4double(totalMemory)
447 << std::setw(11) << memory/1024 << "k "
448 << std::setw( 9) << stats[i].GetNumberHeads()
449 << std::setw( 9) << stats[i].GetNumberNodes()
450 << std::setw(11) << stats[i].GetNumberPointers()
451 << std::setw(13) << totTime << " "
452 << std::setiosflags(std::ios::left)
453 << stats[i].GetVolume()->GetName()
454 << std::resetiosflags(std::ios::floatfield|std::ios::adjustfield)
455 << std::setprecision(6)
456 << G4endl;
457 }
