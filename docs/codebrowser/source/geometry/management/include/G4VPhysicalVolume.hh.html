29// class G4VPhysicalVolume
30//
31// Class description:
32//
33// This is an Abstract Base class for the representation of positioned volume.
34// The volume is placed within a mother volume, relative to its coordinate
35// system. Either a single positioned volume or many positioned volume can
36// be represented by a particular G4VPhysicalVolume.
37
38// History:
39// 15.01.13 G.Cosmo, A.Dotti: Modified for thread-safety for MT
40// 09.11.99 J.Apostolakis: Added GetObjectRotationValue() method & comments
41// 28.08.96 P.Kent: Replaced transform by rotmat + vector
42// 25.07.96 P.Kent: Modified interface for new `Replica' capable geometry
43// 24.07.95 P.Kent: First non-stub version
44// --------------------------------------------------------------------

63 // that are not read-only - they will change during simulation

70 frot = 0;

81class G4VPhysicalVolume
82{
83 public: // with description
84
85 G4VPhysicalVolume(G4RotationMatrix *pRot,
86 const G4ThreeVector &tlate,
87 const G4String &pName,
88 G4LogicalVolume *pLogical,
89 G4VPhysicalVolume *pMother);
90 // Initialise volume, positioned in a frame which is rotated by *pRot,
91 // relative to the coordinate system of the mother volume pMother.
92 // The center of the object is then placed at tlate in the new
93 // coordinates. If pRot=0 the volume is unrotated with respect to its
94 // mother. The physical volume is added to the mother's logical volume.
95 //
96 // Must be called by all subclasses. pMother must point to a valid parent
97 // volume, except in the case of the world/top volume, when it =0.
98 //
99 // The constructor also registers volume with physical volume Store.
100 // Note that the Store may be removed or dynamically built in future
101 // because of memory constraints.
102
103 virtual ~G4VPhysicalVolume();
104 // Destructor, will be subclassed. Removes volume from volume Store.
105
106 inline G4bool operator == (const G4VPhysicalVolume& p) const;
107 // Equality defined by equal addresses only.
108
109 // Access functions
110 //
111 // The following are accessor functions that make a distinction
112 // between whether the rotation/translation is being made for the
113 // frame or the object/volume that is being placed.
114 // (They are the inverse of each other).
115
116 G4RotationMatrix* GetObjectRotation() const; // Obsolete
117 G4RotationMatrix GetObjectRotationValue() const; // Replacement
118 G4ThreeVector GetObjectTranslation() const;
119 // Return the rotation/translation of the Object relative to the mother.
120 const G4RotationMatrix* GetFrameRotation() const;
121 G4ThreeVector GetFrameTranslation() const;
122 // Return the rotation/translation of the Frame used to position
123 // this volume in its mother volume (opposite of object rot/trans).
124
125 // Older access functions, that do not distinguish between frame/object!
126
127 const G4ThreeVector& GetTranslation() const;
128 const G4RotationMatrix* GetRotation() const;
129 // Old access functions, that do not distinguish between frame/object!
130 // They return the translation/rotation of the volume.
131
132 // Set functions
133
134 void SetTranslation(const G4ThreeVector &v);
135 G4RotationMatrix* GetRotation();
136 void SetRotation(G4RotationMatrix*);
137 // NOT INTENDED FOR GENERAL USE.
138 // Non constant versions of above. Used to change transformation
139 // for replication/parameterisation mechanism.
140
141 inline G4LogicalVolume* GetLogicalVolume() const;
142 // Return the associated logical volume.
143 inline void SetLogicalVolume(G4LogicalVolume *pLogical);
144 // Set the logical volume. Must not be called when geometry closed.
145
146 inline G4LogicalVolume* GetMotherLogical() const;
147 // Return the current mother logical volume pointer.
148 inline void SetMotherLogical(G4LogicalVolume *pMother);
149 // Set the mother logical volume. Must not be called when geometry closed.
150
151 inline const G4String& GetName() const;
152 // Return the volume's name.
153 inline void SetName(const G4String& pName);
154 // Set the volume's name.
155
156 inline EVolume VolumeType() const;
157 // Characterise the `type' of volume - normal/replicated/parameterised.
158
159 virtual G4int GetMultiplicity() const;
160 // Returns number of object entities (1 for normal placements,
161 // n for replicas or parameterised).
162
163 // Functions required of subclasses
164
165 virtual G4bool IsMany() const = 0;
166 // Return true if the volume is MANY (not implemented yet).
167 virtual G4int GetCopyNo() const = 0;
168 // Return the volumes copy number.
169 virtual void SetCopyNo(G4int CopyNo) = 0;
170 // Set the volumes copy number.
171 virtual G4bool IsReplicated() const = 0;
172 // Return true if replicated (single object instance represents
173 // many real volumes), else false.
174 virtual G4bool IsParameterised() const = 0;
175 // Return true if parameterised (single object instance represents
176 // many real parameterised volumes), else false.
177 virtual G4VPVParameterisation* GetParameterisation() const = 0;
178 // Return replicas parameterisation object (able to compute dimensions
179 // and transformations of replicas), or NULL if not applicable.
180 virtual void GetReplicationData(EAxis& axis,
181 G4int& nReplicas,
182 G4double& width,
183 G4double& offset,
184 G4bool& consuming) const = 0;
185 // Return replication information. No-op for no replicated volumes.
186 virtual G4bool IsRegularStructure() const = 0;
187 // Returns true if the underlying volume structure is regular.
188 virtual G4int GetRegularStructureId() const = 0;
189 // Returns non-zero code in case the underlying volume structure
190 // is regular, voxel-like. Value is id for structure type.
191 // If non-zero the volume is a candidate for specialised
192 // navigation such as 'nearest neighbour' directly on volumes.
193 virtual G4bool CheckOverlaps(G4int res=1000, G4double tol=0.,
194 G4bool verbose=true, G4int errMax=1);
195 // Verifies if the placed volume is overlapping with existing
196 // daughters or with the mother volume. Provides default resolution
197 // for the number of points to be generated and verified.
198 // Concrete implementation is done and required only for placed and
199 // parameterised volumes. Returns true if the volume is overlapping.
200
201 public: // without description
202
203 G4VPhysicalVolume(__void__&);
204 // Fake default constructor for usage restricted to direct object
205 // persistency for clients requiring preallocation of memory for
206 // persistifiable objects.
207
208 inline G4int GetInstanceID() const;
209 // Returns the instance ID.
210
211 static const G4PVManager& GetSubInstanceManager();
212 // Returns the private data instance manager.
213
214 static void Clean();
215 // Clear memory allocated by sub-instance manager.
216
217 protected:
218
219 void InitialiseWorker(G4VPhysicalVolume *pMasterObject,
220 G4RotationMatrix *pRot, const G4ThreeVector &tlate);
221 // This method is similar to the constructor. It is used by each worker
222 // thread to achieve the partial effect as that of the master thread.
223
224 void TerminateWorker(G4VPhysicalVolume *pMasterObject);
225 // This method is similar to the destructor. It is used by each worker
226 // thread to achieve the partial effect as that of the master thread.
227
228 protected:
229
230 G4int instanceID;
231 // For use in implementing the per-thread data,
232 // It is equivalent to a pointer to a G4PVData object.
233 G4GEOM_DLL static G4PVManager subInstanceManager;
234 // Needed to use G4PVManager for the G4PVData per-thread objects.
235
236 private:
237
238 G4VPhysicalVolume(const G4VPhysicalVolume&);
239 G4VPhysicalVolume& operator=(const G4VPhysicalVolume&);
240 // Private copy constructor and assignment operator.
241
242 private:
243
244 G4LogicalVolume* flogical; // The logical volume representing the
245 // physical and tracking attributes of
246 // the volume
247 G4String fname; // The name of the volume
248 G4LogicalVolume* flmother; // The current mother logical volume
249
250 G4PVData* pvdata; // Shadow pointer for use of object persistency
251};

253// NOTE:
254// The type G4PVManager is introduced to encapsulate the methods used by
255// both the master thread and worker threads to allocate memory space for
256// the fields encapsulated by the class G4PVData. When each thread
257// initializes the value for these fields, it refers to them using a macro
258// definition defined below. For every G4VPhysicalVolume instance, there is
259// a corresponding G4PVData instance. All G4PVData instances are organized
260// by the class G4PVManager as an array.
261// The field "int instanceID" is added to the class G4VPhysicalVolume.
262// The value of this field in each G4VPhysicalVolume instance is the subscript
263// of the corresponding G4PVData instance.
264// In order to use the class G4PVManager, we add a static member in the class
265// G4VPhysicalVolume as follows: "static G4PVManager subInstanceManager;".
266// For the master thread, the array for G4PVData instances grows dynamically
267// along with G4VPhysicalVolume instances are created. For each worker thread,
268// it copies the array of G4PVData instances from the master thread.
269// In addition, it invokes a method similiar to the constructor explicitly
270// to achieve the partial effect for each instance in the array.
271//
