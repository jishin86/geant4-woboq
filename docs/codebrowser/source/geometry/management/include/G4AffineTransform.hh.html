29// class G4AffineTransform
30//
31// Class description:
32//
33// A class for geometric affine transformations [see, eg. Foley & Van Dam]
34// Supports efficient arbitrary rotation & transformation of vectors and the
35// computation of compound & inverse transformations. A `rotation flag' is
36// maintained internally for greater computational efficiency for transforms
37// that do not involve rotation.
38//
39// Interfaces to the CLHEP classes G4ThreeVector & G4RotationMatrix
40//
41// For member function descriptions, see comments by declarations. For
42// additional clarification, also check the `const' declarations for
43// functions & their parameters.
44//
45// Member data:
46//
47// G4double rxx,rxy,rxz;
48// G4double ryx,ryy,ryz; A 3x3 rotation matrix - net rotation
49// G4double rzx,rzy,rzz;
50// G4double tx,ty,tz; Net translation
51
52// History:
53// Paul R C Kent 6 Aug 1996 - initial version
54//
55// 19.09.1996 E.Tcherniaev:
56// - direct access to the protected members of the G4RotationMatrix class
57// replaced by access via public access functions
58// - conversion of the rotation matrix to angle & axis used to get
59// a possibility to remove "friend" from the G4RotationMatrix class
60// 06.05.2018 E.Tcherniaev:
61// - optimized InverseProduct
62// - added methods for inverse transformation: InverseTrasformPoint,
63// InverseTransformAxis, InverseNetRotation, InverseNetTranslation
64// --------------------------------------------------------------------
65#ifndef G4AFFINETRANSFORM_HH
66#define G4AFFINETRANSFORM_HH
67
68#include "G4Types.hh"
69#include "G4ThreeVector.hh"
70#include "G4RotationMatrix.hh"
71#include "G4Transform3D.hh"
72
73class G4AffineTransform
74{
75
76public:
77
78 inline G4AffineTransform();
79
80public: // with description
81
82 inline G4AffineTransform(const G4ThreeVector& tlate);
83 // Translation only: under t'form translate point at origin by tlate
84
85 inline G4AffineTransform(const G4RotationMatrix& rot);
86 // Rotation only: under t'form rotate by rot
87
88 inline G4AffineTransform(const G4RotationMatrix& rot,
89 const G4ThreeVector& tlate);
90 // Under t'form: rotate by rot then translate by tlate
91
92 inline G4AffineTransform(const G4RotationMatrix* rot,
93 const G4ThreeVector& tlate);
94 // Optionally rotate by *rot then translate by tlate - rot may be null
95
96 inline G4AffineTransform(const G4AffineTransform& rhs);
97 // Copy constructor
98
99 inline G4AffineTransform& operator=(const G4AffineTransform& rhs);
100 // Assignment operator
101
102 inline ~G4AffineTransform();
103 // Destructor
104
105 inline G4AffineTransform operator * (const G4AffineTransform& tf) const;
106 // Compound Transforms:
107 // tf2=tf2*tf1 equivalent to tf2*=tf1
108 // Returns compound transformation of self*tf
109
110 inline G4AffineTransform& operator *= (const G4AffineTransform& tf);
111 // (Modifying) Multiplies self by tf; Returns self reference
112 // ie. A=AB for a*=b
113
114
115 inline G4AffineTransform& Product(const G4AffineTransform& tf1,
116 const G4AffineTransform& tf2);
117 // 'Products' for avoiding (potential) temporaries:
118 // c.Product(a,b) equivalent to c=a*b
119 // c.InverseProduct(a*b,b ) equivalent to c=a
120 // (Modifying) Sets self=tf1*tf2; Returns self reference
121
122 inline G4AffineTransform& InverseProduct(const G4AffineTransform& tf1,
123 const G4AffineTransform& tf2);
124 // (Modifying) Sets self=tf1*(tf2^-1); Returns self reference
125
126 inline G4ThreeVector TransformPoint(const G4ThreeVector& vec) const;
127 // Transform the specified point: returns vec*rot+tlate
128
129 inline G4ThreeVector InverseTransformPoint(const G4ThreeVector& vec) const;
130 // Transform the specified point using inverse transformation
131
132 inline G4ThreeVector TransformAxis(const G4ThreeVector& axis) const;
133 // Transform the specified axis: returns vec*rot
134
135 inline G4ThreeVector InverseTransformAxis(const G4ThreeVector& axis) const;
136 // Transform the specified axis using inverse transfromation
137
138 inline void ApplyPointTransform(G4ThreeVector& vec) const;
139 // Transform the specified point (in place): sets vec=vec*rot+tlate
140
141 inline void ApplyAxisTransform(G4ThreeVector& axis) const;
142 // Transform the specified axis (in place): sets axis=axis*rot;
143
144 inline G4AffineTransform Inverse() const;
145 // Return inverse of current transform
146
147 inline G4AffineTransform& Invert();
148 // (Modifying) Sets self=inverse of self; Returns self reference
149
150 inline G4AffineTransform& operator +=(const G4ThreeVector& tlate);
151 inline G4AffineTransform& operator -=(const G4ThreeVector& tlate);
152 // (Modifying) Adjust net translation by given vector;
153 // Returns self reference
154
155 inline G4bool operator == (const G4AffineTransform& tf) const;
156 inline G4bool operator != (const G4AffineTransform& tf) const;
157
158 inline G4double operator [] (const G4int n) const;
159
160 inline G4bool IsRotated() const;
161 // True if transform includes rotation
162
163 inline G4bool IsTranslated() const;
164 // True if transform includes translation
165
166 inline G4RotationMatrix NetRotation() const;
167
168 inline G4RotationMatrix InverseNetRotation() const;
169
170 inline G4ThreeVector NetTranslation() const;
171
172 inline G4ThreeVector InverseNetTranslation() const;
173
174 inline void SetNetRotation(const G4RotationMatrix& rot);
175
176 inline void SetNetTranslation(const G4ThreeVector& tlate);
177
178 inline operator G4Transform3D () const;
179 // Conversion operator (cast) to G4Transform3D
180
181private:
182
183 inline G4AffineTransform(
184 const G4double prxx, const G4double prxy, const G4double prxz,
185 const G4double pryx, const G4double pryy, const G4double pryz,
186 const G4double przx, const G4double przy, const G4double przz,
187 const G4double ptx, const G4double pty, const G4double ptz);
188
189 G4double rxx,rxy,rxz;
190 G4double ryx,ryy,ryz;
191 G4double rzx,rzy,rzz;
192 G4double tx,ty,tz;
193};

199#endif
