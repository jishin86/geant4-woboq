112// ********************************************************************
113// IsDaughter
114// ********************************************************************
115//
116inline
117G4bool G4LogicalVolume::IsDaughter(const G4VPhysicalVolume* p) const
118{
119 G4PhysicalVolumeList::const_iterator i;
120 for ( i=fDaughters.begin(); i!=fDaughters.end(); ++i )
121 {
122 if (**i==*p) return true;
123 }
124 return false;
125}
126
127// ********************************************************************
128// CharacteriseDaughters
129// ********************************************************************
130//
131inline
132EVolume G4LogicalVolume::CharacteriseDaughters() const
133{
134 EVolume type;
135 G4VPhysicalVolume *pVol;
136
137 if ( GetNoDaughters()==1 )
138 {
139 pVol = GetDaughter(0);
140 type = pVol->VolumeType();
141 }
142 else
143 {
144 type = kNormal;
145 }
146 return type;
147}
