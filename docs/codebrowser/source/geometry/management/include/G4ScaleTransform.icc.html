74inline void G4ScaleTransform::Init()
75{
76 if (!((fScale.x()>0) && (fScale.y()>0) && (fScale.z()>0)))
77 {
78 G4Exception("G4ScaleTransform::Init()", "GeomMgt0001",
79 FatalException, "Scale transformation must be positive!");
80 }
81 fIScale.set(1./fScale.x(), 1./fScale.y(), 1./fScale.z());
82 flFactor = std::min(std::min(fIScale.x(), fIScale.y()), fIScale.z());
83 fgFactor = std::min(std::min(fScale.x(), fScale.y()), fScale.z());
84}

185inline G4double
186G4ScaleTransform::TransformDistance(G4double dist,
187 const G4ThreeVector& dir) const
188{
189 G4ThreeVector v(dir.x()*fIScale.x(),
190 dir.y()*fIScale.y(),
191 dir.z()*fIScale.z());
192 G4double scale = std::sqrt(v.dot(v));
193 return ( scale*dist );
194}
195
196inline G4double G4ScaleTransform::TransformDistance(G4double safety) const
197{
198 return ( safety*flFactor );
199}

201inline G4double
202G4ScaleTransform::InverseTransformDistance(G4double dist,
203 const G4ThreeVector& dir) const
204{
205 G4ThreeVector v(dir.x()*fScale.x(),
206 dir.y()*fScale.y(),
207 dir.z()*fScale.z());
208 G4double scale = std::sqrt(v.dot(v));
209 return ( scale*dist );
210}

212inline G4double

216}
