200// ********************************************************************
201// ClearMap
202// ********************************************************************
203//
204inline
205void G4Region::ClearMap()
206{
207 if(!(fMaterialCoupleMap.empty()))
208 {
209 G4MaterialCoupleMap::iterator b = fMaterialCoupleMap.begin();
210 G4MaterialCoupleMap::iterator e = fMaterialCoupleMap.end();
211 fMaterialCoupleMap.erase(b,e);
212 }
213}
214
215// ********************************************************************
216// RegisterMateralCouplePair
217// ********************************************************************
218//
219inline
220void G4Region::RegisterMaterialCouplePair(G4Material* mat,
221 G4MaterialCutsCouple* couple)
222{
223 fMaterialCoupleMap.insert(G4MaterialCouplePair(mat,couple));
224}
225
226// ********************************************************************
227// FindCouple
228// ********************************************************************
229//
230inline
231G4MaterialCutsCouple* G4Region::FindCouple(G4Material* mat)
232{
233 G4MaterialCoupleMap::iterator c = fMaterialCoupleMap.find(mat);
234 G4MaterialCutsCouple* couple = 0;
235 if(c!=fMaterialCoupleMap.end()) couple = (*c).second;
236 return couple;
237}

269// ********************************************************************
270// AddMaterial
271// ********************************************************************
272//
273inline
274void G4Region::AddMaterial(G4Material* aMaterial)
275{
276 G4MaterialList::iterator pos =
277 std::find(fMaterials.begin(),fMaterials.end(),aMaterial);
278 if (pos == fMaterials.end())
279 {
280 fMaterials.push_back(aMaterial);
281 fRegionMod = true;
282 }
283}
