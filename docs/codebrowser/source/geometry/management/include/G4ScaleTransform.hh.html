29// class G4ScaleTransform
30//
31// Class description:
32//
33// A class for geometric scaling transformations.
34// Supports efficient arbitrary transformation of points, vectors and
35// normals and the computation of compound & inverse transformations.
36//
37// Interfaces to the CLHEP class G4ThreeVector
38//
39// For member function descriptions, see comments by declarations. For
40// additional clarification, also check the `const' declarations for
41// functions & their parameters.
42//
43// Member data:
44//
45// G4ThreeVector fScale; // scale transformation
46// G4ThreeVector fIScale; // inverse scale (avoid divisions)
47// G4double flFactor; // factor for conversion to local frame
48// G4double fgFactor; // factor for conversion to global frame
49
50// History:
51// E.Tcherniaev 11 Mar 2016 - added transformations for normal
52// G.Cosmo 18 Feb 2016 - initial version
53//
54// --------------------------------------------------------------------
55#ifndef G4SCALETRANSFORM_HH
56#define G4SCALETRANSFORM_HH
57
58#include "G4Types.hh"
59#include "G4ThreeVector.hh"
60#include "G4Transform3D.hh"
61
62class G4ScaleTransform
63{
64
65 public:
66
67 inline G4ScaleTransform();
68 // Default constructor
69
70 inline G4ScaleTransform(G4double sx, G4double sy, G4double sz);
71 // Constructor with scale parameters on each axis
72
73 inline G4ScaleTransform(const G4ThreeVector& scale);
74 // Constructor taking a 3-vector
75
76 inline G4ScaleTransform(const G4Scale3D& scale);
77 // Constructor taking a Scale3D
78
79 inline G4ScaleTransform(const G4ScaleTransform& right);
80 // Copy constructor
81
82 inline G4ScaleTransform& operator=(const G4ScaleTransform& right);
83 // Assignment operator
84
85 inline void Init();
86 // Update the backed-up inverse scale and special conversion factors
87 // based on the values of the scale. Needed at initialisation and
88 // whenever the scale has changed value
89
90 inline const G4ThreeVector& GetScale() const;
91 inline const G4ThreeVector& GetInvScale() const;
92 // Get reference to the inverse scale transformation
93
94 inline void SetScale(const G4ThreeVector& scale);
95 // Set scale based on vector
96 inline void SetScale(const G4Scale3D& scale);
97 // Set scale based on a G4Scale3D transformation
98 inline void SetScale(G4double sx, G4double sy, G4double sz);
99 // Set scale based on values
100
101 inline void Transform(const G4ThreeVector& global,
102 G4ThreeVector& local) const;
103 inline G4ThreeVector Transform(const G4ThreeVector& global) const;
104 // Transform point from global to local frame
105
106 inline void InverseTransform(const G4ThreeVector& local,
107 G4ThreeVector& global) const;
108 inline G4ThreeVector InverseTransform(const G4ThreeVector& local) const;
109 // Transform point from local to global frame
110
111 inline void TransformNormal(const G4ThreeVector& global,
112 G4ThreeVector& local) const;
113 inline G4ThreeVector TransformNormal(const G4ThreeVector& global) const;
114 // Transform normal from global to local frame
115
116 inline void InverseTransformNormal(const G4ThreeVector& local,
117 G4ThreeVector& global) const;
118 inline G4ThreeVector InverseTransformNormal(const G4ThreeVector& local) const;
119 // Transform normal from local to global frame
120
121 inline G4double TransformDistance(G4double dist,
122 const G4ThreeVector& dir) const;
123 // Transform distance along given direction from global to local frame
124
125 inline G4double TransformDistance(G4double safety) const;
126 // Transform distance from global to local frame (conservative)
127
128 inline G4double InverseTransformDistance(G4double dist,
129 const G4ThreeVector& dir) const;
130 // Transform distance along given direction from local to global frame
131
132 inline G4double InverseTransformDistance(G4double safety) const;
133 // Transform distance from local to global frame (conservative)
134
135 private:
136
137 G4ThreeVector fScale; // scale transformation
138 G4ThreeVector fIScale; // inverse scale (avoid divisions)
139 G4double flFactor, fgFactor; // conversion factors to local/global frames
140
141}; // End class G4ScaleTransform
142
143std::ostream& operator<<(std::ostream& os, const G4ScaleTransform& scale);
144
145#include "G4ScaleTransform.icc"
146
147#endif // G4SCALETRANSFORM_HH
