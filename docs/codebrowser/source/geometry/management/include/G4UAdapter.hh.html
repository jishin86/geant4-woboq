29// class G4UAdapter
30//
31// Class description:
32//
33// Utility class for adapting VecGeom solids API to Geant4 solids.
34// NOTE: Using protected inheritance since the Adapter is supposed to
35// be a G4VSolid "implemented-in-terms-of" the VecGeom UnplacedVolume_t.
36// The choice of protected vs private is due to the fact that we want
37// to propagate functions further down in the inheritance hierarchy.
38
39// Author:
40// 17.05.17 G.Cosmo: Adapted for G4VSolid from original G4USolids bridge
41// class and the USolidsAdapter class in VecGeom.
42// ------------------------------------------------------------------------

66template <class UnplacedVolume_t>
67class G4UAdapter : public G4VSolid, protected UnplacedVolume_t
68{
69 public:
70
71 typedef vecgeom::Vector3D<G4double> U3Vector;
72
73 using UnplacedVolume_t::operator delete;
74 using UnplacedVolume_t::operator new;
75 // VecGeom volumes have special delete/new ("AlignedBase")
76 // and we need to make these functions public again
77
78 G4UAdapter(const G4String& name)
79 : G4VSolid(name), fRebuildPolyhedron(false), fPolyhedron(0)
80 { kHalfTolerance = 0.5*kCarTolerance; }
81
82 template <typename... T>
83 G4UAdapter(const G4String& name, const T &... params)
84 : G4VSolid(name), UnplacedVolume_t(params...),
85 fRebuildPolyhedron(false), fPolyhedron(0)
86 { kHalfTolerance = 0.5*kCarTolerance; }
87
88 virtual ~G4UAdapter();
89
90 G4bool operator==(const G4UAdapter& s) const;
91 // Return true only if addresses are the same.
92
93 virtual G4bool CalculateExtent(const EAxis pAxis,
94 const G4VoxelLimits& pVoxelLimit,
95 const G4AffineTransform& pTransform,
96 G4double& pMin, G4double& pMax) const override;
97 // Calculate the minimum and maximum extent of the solid, when under the
98 // specified transform, and within the specified limits. If the solid
99 // is not intersected by the region, return false, else return true.
100
101 virtual EInside Inside(const G4ThreeVector& p) const override;
102 // Returns kOutside if the point at offset p is outside the shapes
103 // boundaries plus Tolerance/2, kSurface if the point is <= Tolerance/2
104 // from a surface, otherwise kInside.
105
106 virtual G4ThreeVector SurfaceNormal(const G4ThreeVector& p) const override;
107 // Returns the outwards pointing unit normal of the shape for the
108 // surface closest to the point at offset p.
109
110 virtual G4double DistanceToIn(const G4ThreeVector& p,
111 const G4ThreeVector& v) const override;
112 // Return the distance along the normalised vector v to the shape,
113 // from the point at offset p. If there is no intersection, return
114 // kInfinity. The first intersection resulting from `leaving' a
115 // surface/volume is discarded. Hence, it is tolerant of points on
116 // the surface of the shape.
117
118 virtual G4double DistanceToIn(const G4ThreeVector& p) const override;
119 // Calculate the distance to the nearest surface of a shape from an
120 // outside point. The distance can be an underestimate.
121
122 virtual G4double DistanceToOut(const G4ThreeVector& p,
123 const G4ThreeVector& v,
124 const G4bool calcNorm = false,
125 G4bool* validNorm = 0,
126 G4ThreeVector* n = 0) const override;
127 // Return the distance along the normalised vector v to the shape,
128 // from a point at an offset p inside or on the surface of the shape.
129 // Intersections with surfaces, when the point is < Tolerance/2 from a
130 // surface must be ignored.
131 // If calcNorm==true:
132 // validNorm set true if the solid lies entirely behind or on the
133 // exiting surface.
134 // n set to exiting outwards normal vector (undefined Magnitude).
135 // validNorm set to false if the solid does not lie entirely behind
136 // or on the exiting surface
137 // If calcNorm==false:
138 // validNorm and n are unused.
139 //
140 // Must be called as solid.DistanceToOut(p,v) or by specifying all
141 // the parameters.
142
143 virtual G4double DistanceToOut(const G4ThreeVector& p) const override;
144 // Calculate the distance to the nearest surface of a shape from an
145 // inside point. The distance can be an underestimate.
146
147 virtual void ComputeDimensions(G4VPVParameterisation* p,
148 const G4int n,
149 const G4VPhysicalVolume* pRep) override;
150 // Throw exception if ComputeDimensions called from an illegal
151 // derived class.
152
153 virtual G4double GetCubicVolume() override;
154 // Returns an estimation of the solid volume in internal units.
155 // This method may be overloaded by derived classes to compute the
156 // exact geometrical quantity for solids where this is possible,
157 // or anyway to cache the computed value.
158 // Note: the computed value is NOT cached.
159
160 virtual G4double GetSurfaceArea() override;
161 // Return an estimation of the solid surface area in internal units.
162 // This method may be overloaded by derived classes to compute the
163 // exact geometrical quantity for solids where this is possible,
164 // or anyway to cache the computed value.
165 // Note: the computed value is NOT cached.
166
167 virtual G4ThreeVector GetPointOnSurface() const override;
168 // Returns a random point located on the surface of the solid.
169
170 virtual G4GeometryType GetEntityType() const override;
171 // Provide identification of the class of an object.
172 // (required for persistency)
173
174 virtual G4VSolid* Clone() const override;
175 // Returns a pointer of a dynamically allocated copy of the solid.
176 // Returns NULL pointer with warning in case the concrete solid does not
177 // implement this method. The caller has responsibility for ownership.
178
179 virtual std::ostream& StreamInfo(std::ostream& os) const override;
180 // Dumps contents of the solid to a stream.
181
182 virtual void DescribeYourselfTo(G4VGraphicsScene& scene) const override;
183 // A "double dispatch" function which identifies the solid
184 // to the graphics scene for visualization.
185
186 virtual G4VisExtent GetExtent() const override;
187 // Provide extent (bounding box) as possible hint to the graphics view.
188 virtual G4Polyhedron* CreatePolyhedron() const override;
189 // Create Polyhedron used for Visualisation
190 virtual G4Polyhedron* GetPolyhedron() const override;
191 // Smart access function - creates on request and stores for future
192 // access. A null pointer means "not available".
193
194 public: // without description
195
196 G4UAdapter(__void__&);
197 // Fake default constructor for usage restricted to direct object
198 // persistency for clients requiring preallocation of memory for
199 // persistifiable objects.
200
201 G4UAdapter(const G4UAdapter& rhs);
202 G4UAdapter& operator=(const G4UAdapter& rhs);
203 // Copy constructor and assignment operator.
204
205 public: // VecGeom overridden methods
206
207 vecgeom::Precision
208 DistanceToOut(U3Vector const &position, U3Vector const &direction,
209 vecgeom::Precision stepMax = kInfinity) const override
210 {
211 return UnplacedVolume_t::DistanceToOut(position, direction, stepMax);
212 }
213
214 vecgeom::EnumInside
215 Inside(U3Vector const &aPoint) const override
216 {
217 return UnplacedVolume_t::Inside(aPoint);
218 }
219
220 vecgeom::Precision
221 DistanceToIn(U3Vector const &position, U3Vector const &direction,
222 const vecgeom::Precision step_max = kInfinity) const override
223 {
224 return UnplacedVolume_t::DistanceToIn(position, direction, step_max);
225 }
226
227 G4bool Normal(U3Vector const &aPoint, U3Vector &aNormal) const override
228 {
229 return UnplacedVolume_t::Normal(aPoint, aNormal);
230 }
231
232 void Extent(U3Vector &aMin, U3Vector &aMax) const override
233 {
234 return UnplacedVolume_t::Extent(aMin, aMax);
235 }
236
237 U3Vector SamplePointOnSurface() const override
238 {
239 return UnplacedVolume_t::SamplePointOnSurface();
240 }
241
242 protected: // data
243
244 mutable G4bool fRebuildPolyhedron;
245 mutable G4Polyhedron* fPolyhedron;
246
247 G4double kHalfTolerance; // Cached geometrical tolerance
248
249 using UnplacedVolume_t::DistanceToOut;
250 using UnplacedVolume_t::DistanceToIn;
251};

285template <class UnplacedVolume_t>
286G4UAdapter<UnplacedVolume_t>& G4UAdapter<UnplacedVolume_t>::
287operator=(const G4UAdapter& rhs)
288{
289 // Check assignment to self
290 //
291 if (this == &rhs)
292 {
293 return *this;
294 }
295
296 // Copy base class data
297 //
298 G4VSolid::operator=(rhs);
299 UnplacedVolume_t::operator=(rhs);
300
301 // Copy data
302 //
303 fRebuildPolyhedron = false;
304 delete fPolyhedron; fPolyhedron = 0;
305 kHalfTolerance = 0.5*kCarTolerance;
306
307 return *this;
308}

310template <class UnplacedVolume_t>
311EInside G4UAdapter<UnplacedVolume_t>::
312Inside(const G4ThreeVector& p) const
313{
314 U3Vector pt(p.x(), p.y(), p.z());
315 vecgeom::EnumInside in_temp;
316 EInside in = kOutside;
317
318 in_temp = UnplacedVolume_t::Inside(pt);
319
320 if (in_temp == vecgeom::EnumInside::eInside) in = kInside;
321 else if (in_temp == vecgeom::EnumInside::eSurface) in = kSurface;
322
323 return in;
324}

336template <class UnplacedVolume_t>
337G4double G4UAdapter<UnplacedVolume_t>::
338DistanceToIn(const G4ThreeVector& pt, const G4ThreeVector& d) const
339{
340 U3Vector p(pt.x(), pt.y(), pt.z());
341 U3Vector v(d.x(), d.y(), d.z());
342 G4double dist = UnplacedVolume_t::DistanceToIn(p, v, kInfinity);
343
344 // apply Geant4 distance conventions
345 //
346 if (dist < kHalfTolerance) return 0.0;
347 return (dist > kInfinity) ? kInfinity : dist;
348}

350template <class UnplacedVolume_t>
351G4double G4UAdapter<UnplacedVolume_t>::
352DistanceToIn(const G4ThreeVector& pt) const
353{
354 U3Vector p(pt.x(), pt.y(), pt.z());
355 G4double dist = UnplacedVolume_t::SafetyToIn(p);
356
357 // Apply Geant4 convention: convert negative values to zero
358 //
359 if (dist < kHalfTolerance) return 0.0;
360 return (dist > kInfinity) ? kInfinity : dist;
361}

363template <class UnplacedVolume_t>
364G4double G4UAdapter<UnplacedVolume_t>::
365DistanceToOut(const G4ThreeVector& pt, const G4ThreeVector& d,
366 const G4bool calcNorm, G4bool* validNorm,
367 G4ThreeVector* norm) const
368{
369 U3Vector p(pt.x(), pt.y(), pt.z());
370 U3Vector v(d.x(), d.y(), d.z());
371
372 G4double dist = UnplacedVolume_t::DistanceToOut(p, v, kInfinity);
373 if(calcNorm) // *norm=n, but only after calcNorm check and if convex volume
374 {
375 if (UnplacedVolume_t::IsConvex())
376 {
377 U3Vector n, hitpoint = p + dist * v;
378 UnplacedVolume_t::Normal(hitpoint, n);
379 *validNorm = true;
380 norm->set(n.x(), n.y(), n.z());
381 }
382 else
383 {
384 *validNorm = false;
385 }
386 }
387
388 // Apply Geant4 distance conventions
389 //
390 if (dist < kHalfTolerance) return 0.0;
391 return (dist > kInfinity) ? kInfinity : dist;
392}

480template <class UnplacedVolume_t>
481G4bool G4UAdapter<UnplacedVolume_t>::CalculateExtent(const EAxis pAxis,
482 const G4VoxelLimits& pVoxelLimit,
483 const G4AffineTransform& pTransform,
484 G4double& pMin, G4double& pMax) const
485{
486 U3Vector vmin, vmax;
487 UnplacedVolume_t::Extent(vmin,vmax);
488 G4ThreeVector bmin(vmin.x(),vmin.y(),vmin.z());
489 G4ThreeVector bmax(vmax.x(),vmax.y(),vmax.z());
490
491 // Check correctness of the bounding box
492 //
493 if (bmin.x() >= bmax.x() || bmin.y() >= bmax.y() || bmin.z() >= bmax.z())
494 {
495 std::ostringstream message;
496 message << "Bad bounding box (min >= max) for solid: "
497 << GetName() << " - " << GetEntityType() << " !"
498 << "\nmin = " << bmin
499 << "\nmax = " << bmax;
500 G4Exception("G4UAdapter::CalculateExtent()", "GeomMgt0001",
501 JustWarning, message);
502 StreamInfo(G4cout);
503 }
504
505 G4BoundingEnvelope bbox(bmin,bmax);
506 return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
507}

522template <class UnplacedVolume_t>
523G4Polyhedron* G4UAdapter<UnplacedVolume_t>::GetPolyhedron() const
524{
525 if (!fPolyhedron ||
526 fRebuildPolyhedron ||
527 fPolyhedron->GetNumberOfRotationStepsAtTimeOfCreation() !=
528 fPolyhedron->GetNumberOfRotationSteps())
529 {
530 G4AutoLock l(&pMutex);
531 delete fPolyhedron;
532 fPolyhedron = CreatePolyhedron();
533 fRebuildPolyhedron = false;
534 l.unlock();
535 }
536 return fPolyhedron;
537}
