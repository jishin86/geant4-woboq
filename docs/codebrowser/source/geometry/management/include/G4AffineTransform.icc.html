62inline G4AffineTransform::G4AffineTransform( const G4RotationMatrix* rot,
63 const G4ThreeVector& tlate )
64 : tx(tlate.x()),ty(tlate.y()),tz(tlate.z())
65{
66 if (rot)
67 {
68 rxx=rot->xx();rxy=rot->xy();rxz=rot->xz();
69 ryx=rot->yx();ryy=rot->yy();ryz=rot->yz();
70 rzx=rot->zx();rzy=rot->zy();rzz=rot->zz();
71 }
72 else
73 {
74 rxx=1; rxy=0; rxz=0;
75 ryx=0; ryy=1; ryz=0;
76 rzx=0; rzy=0; rzz=1;
77 }
78}

99inline G4AffineTransform&
100G4AffineTransform::operator = (const G4AffineTransform& rhs)
101{
102 if (this == &rhs) { return *this; }
103 rxx = rhs.rxx; rxy = rhs.rxy; rxz = rhs.rxz;
104 ryx = rhs.ryx; ryy = rhs.ryy; ryz = rhs.ryz;
105 rzx = rhs.rzx; rzy = rhs.rzy; rzz = rhs.rzz;
106 tx = rhs.tx; ty = rhs.ty; tz = rhs.tz;
107 return *this;
108}

133inline G4AffineTransform&
134G4AffineTransform::operator *= (const G4AffineTransform& tf)
135{
136 // Use temporaries for `in place' compound transform computation
137 //
138 G4double nrxx=rxx*tf.rxx+rxy*tf.ryx+rxz*tf.rzx;
139 G4double nrxy=rxx*tf.rxy+rxy*tf.ryy+rxz*tf.rzy;
140 G4double nrxz=rxx*tf.rxz+rxy*tf.ryz+rxz*tf.rzz;
141
142 G4double nryx=ryx*tf.rxx+ryy*tf.ryx+ryz*tf.rzx;
143 G4double nryy=ryx*tf.rxy+ryy*tf.ryy+ryz*tf.rzy;
144 G4double nryz=ryx*tf.rxz+ryy*tf.ryz+ryz*tf.rzz;
145
146 G4double nrzx=rzx*tf.rxx+rzy*tf.ryx+rzz*tf.rzx;
147 G4double nrzy=rzx*tf.rxy+rzy*tf.ryy+rzz*tf.rzy;
148 G4double nrzz=rzx*tf.rxz+rzy*tf.ryz+rzz*tf.rzz;
149
150 G4double ntx=tx*tf.rxx+ty*tf.ryx+tz*tf.rzx+tf.tx;
151 G4double nty=tx*tf.rxy+ty*tf.ryy+tz*tf.rzy+tf.ty;
152 G4double ntz=tx*tf.rxz+ty*tf.ryz+tz*tf.rzz+tf.tz;
153
154 tx=ntx; ty=nty; tz=ntz;
155 rxx=nrxx; rxy=nrxy; rxz=nrxz;
156 ryx=nryx; ryy=nryy; ryz=nryz;
157 rzx=nrzx; rzy=nrzy; rzz=nrzz;
158
159 return *this;
160}

162inline G4AffineTransform&
163G4AffineTransform::Product( const G4AffineTransform& tf1,
164 const G4AffineTransform& tf2 )
165{
166 rxx = tf1.rxx*tf2.rxx + tf1.rxy*tf2.ryx + tf1.rxz*tf2.rzx;
167 rxy = tf1.rxx*tf2.rxy + tf1.rxy*tf2.ryy + tf1.rxz*tf2.rzy;
168 rxz = tf1.rxx*tf2.rxz + tf1.rxy*tf2.ryz + tf1.rxz*tf2.rzz;
169
170 ryx = tf1.ryx*tf2.rxx + tf1.ryy*tf2.ryx + tf1.ryz*tf2.rzx;
171 ryy = tf1.ryx*tf2.rxy + tf1.ryy*tf2.ryy + tf1.ryz*tf2.rzy;
172 ryz = tf1.ryx*tf2.rxz + tf1.ryy*tf2.ryz + tf1.ryz*tf2.rzz;
173
174 rzx = tf1.rzx*tf2.rxx + tf1.rzy*tf2.ryx + tf1.rzz*tf2.rzx;
175 rzy = tf1.rzx*tf2.rxy + tf1.rzy*tf2.ryy + tf1.rzz*tf2.rzy;
176 rzz = tf1.rzx*tf2.rxz + tf1.rzy*tf2.ryz + tf1.rzz*tf2.rzz;
177
178 tx = tf1.tx*tf2.rxx + tf1.ty*tf2.ryx + tf1.tz*tf2.rzx + tf2.tx;
179 ty = tf1.tx*tf2.rxy + tf1.ty*tf2.ryy + tf1.tz*tf2.rzy + tf2.ty;
180 tz = tf1.tx*tf2.rxz + tf1.ty*tf2.ryz + tf1.tz*tf2.rzz + tf2.tz;
181
182 return *this;
183}

185inline G4AffineTransform&
186G4AffineTransform::InverseProduct( const G4AffineTransform& tf1,
187 const G4AffineTransform& tf2 )
188{
189 if ((tf2.rxx + tf2.ryy + tf2.rzz) == 3.)
190 {
191 rxx = tf1.rxx;
192 rxy = tf1.rxy;
193 rxz = tf1.rxz;
194
195 ryx = tf1.ryx;
196 ryy = tf1.ryy;
197 ryz = tf1.ryz;
198
199 rzx = tf1.rzx;
200 rzy = tf1.rzy;
201 rzz = tf1.rzz;
202
203 tx = tf1.tx - tf2.tx;
204 ty = tf1.ty - tf2.ty;
205 tz = tf1.tz - tf2.tz;
206 }
207 else
208 {
209 G4double tf1rxx = tf1.rxx, tf1rxy = tf1.rxy, tf1rxz = tf1.rxz;
210 rxx = tf1rxx*tf2.rxx + tf1rxy*tf2.rxy + tf1rxz*tf2.rxz;
211 rxy = tf1rxx*tf2.ryx + tf1rxy*tf2.ryy + tf1rxz*tf2.ryz;
212 rxz = tf1rxx*tf2.rzx + tf1rxy*tf2.rzy + tf1rxz*tf2.rzz;
213
214 G4double tf1ryx = tf1.ryx, tf1ryy = tf1.ryy, tf1ryz = tf1.ryz;
215 ryx = tf1ryx*tf2.rxx + tf1ryy*tf2.rxy + tf1ryz*tf2.rxz;
216 ryy = tf1ryx*tf2.ryx + tf1ryy*tf2.ryy + tf1ryz*tf2.ryz;
217 ryz = tf1ryx*tf2.rzx + tf1ryy*tf2.rzy + tf1ryz*tf2.rzz;
218
219 G4double tf1rzx = tf1.rzx, tf1rzy = tf1.rzy, tf1rzz = tf1.rzz;
220 rzx = tf1rzx*tf2.rxx + tf1rzy*tf2.rxy + tf1rzz*tf2.rxz;
221 rzy = tf1rzx*tf2.ryx + tf1rzy*tf2.ryy + tf1rzz*tf2.ryz;
222 rzz = tf1rzx*tf2.rzx + tf1rzy*tf2.rzy + tf1rzz*tf2.rzz;
223
224 G4double tf1_2tx = tf1.tx - tf2.tx;
225 G4double tf1_2ty = tf1.ty - tf2.ty;
226 G4double tf1_2tz = tf1.tz - tf2.tz;
227 tx = tf1_2tx*tf2.rxx + tf1_2ty*tf2.rxy + tf1_2tz*tf2.rxz;
228 ty = tf1_2tx*tf2.ryx + tf1_2ty*tf2.ryy + tf1_2tz*tf2.ryz;
229 tz = tf1_2tx*tf2.rzx + tf1_2ty*tf2.rzy + tf1_2tz*tf2.rzz;
230 }
231 return *this;
232}

300inline
301G4AffineTransform& G4AffineTransform::Invert()
302{
303 G4double ttx = -tx, tty = -ty, ttz = -tz;
304 tx = ttx*rxx + tty*rxy + ttz*rxz;
305 ty = ttx*ryx + tty*ryy + ttz*ryz;
306 tz = ttx*rzx + tty*rzy + ttz*rzz;
307
308 G4double tmp1=ryx; ryx=rxy; rxy=tmp1;
309 G4double tmp2=rzx; rzx=rxz; rxz=tmp2;
310 G4double tmp3=rzy; rzy=ryz; ryz=tmp3;
311
312 return *this;
313}

438inline
439G4ThreeVector G4AffineTransform::InverseNetTranslation() const
440{
441 G4double ttx = -tx, tty = -ty, ttz = -tz;
442 G4double invtx = ttx*rxx + tty*rxy + ttz*rxz;
443 G4double invty = ttx*ryx + tty*ryy + ttz*ryz;
444 G4double invtz = ttx*rzx + tty*rzy + ttz*rzz;
445 return G4ThreeVector(invtx,invty,invtz);

476inline
477std::ostream& operator << (std::ostream& os, const G4AffineTransform& transf)
478{
479 std::streamsize oldPrec = os.precision(6);
480 G4double DeviationTolerance = 1.0e-05;
481
482 G4double diagDeviation = 0.0;
483 G4double offdDeviationUL = 0.0;
484 G4double offdDeviationDR = 0.0;
485 G4double offdDeviation = 0.0;
486
487 os << " Transformation: " << G4endl;
488
489 // double a = std::max ( 1, 2, 3 ) ;
490
491 G4bool UnitTr = ! transf.IsRotated();
492 diagDeviation = std::max( std::fabs( transf[0] - 1.0 ) , // |rxx - 1|
493 std::fabs( transf[5] - 1.0 ) ); // |ryy - 1|
494 diagDeviation = std::max( diagDeviation,
495 std::fabs( transf[10] - 1.0 ) ); // |rzz - 1|
496
497 offdDeviationUL = std::max( std::fabs( transf[1] ) , // |rxy|
498 std::fabs( transf[2] ) ); // |rxz|
499 offdDeviationUL = std::max( offdDeviationUL,
500 std::fabs( transf[4] ) ); // |ryx|
501
502 offdDeviationDR = std::max( std::fabs( transf[6] ) , // |ryz|
503 std::fabs( transf[8] ) ); // |rzx|
504 offdDeviationDR = std::max( offdDeviationDR,
505 std::fabs( transf[9] ) ); // |rzy|
506 offdDeviation = std::max( offdDeviationUL, offdDeviationDR );
507
508 if( UnitTr || std::max(diagDeviation, offdDeviation) < DeviationTolerance )
509 {
510 os << " UNIT Rotation " << G4endl;
511 }
512 else
513 {
514 os << "rx/x,y,z: "
515 << transf[0] << " " << transf[1] << " " << transf[2] << G4endl
516 << "ry/x,y,z: "
517 << transf[4] << " " << transf[5] << " " << transf[6] << G4endl
518 << "rz/x,y,z: "
519 << transf[8] << " " << transf[9] << " " << transf[10] << G4endl;
520 }
521
522 os << "tr/x,y,z: " << transf[12] << " " << transf[13] << " " << transf[14]
523 << G4endl;
524
525 os.precision(oldPrec);
526
527 return os;
528}
