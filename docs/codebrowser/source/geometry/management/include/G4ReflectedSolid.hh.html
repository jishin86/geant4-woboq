29// class G4ReflectedSolid
30//
31// Class description:
32//
33// A Reflected solid is a solid that has been shifted from its original
34// frame of reference to a new one.
35
36// History:
37//
38// 23.07.01 V.Grichine: created
39// --------------------------------------------------------------------

41#define G4ReflectedSolid_HH
42

44#include "G4ThreeVector.hh"

113 public: // without description
