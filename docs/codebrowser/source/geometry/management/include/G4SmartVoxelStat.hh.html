30// G4SmartVoxelStat
31//
32// Class description:
33//
34// Stores information on the performance of the smart voxel algorithm
35// for an individual logical volume.

37// Author: D.C.Williams, UCSC (davidw@scipp.ucsc.edu)

40#ifndef G4SmartVoxelStat_hh
41#define G4SmartVoxelStat_hh
42
43#include "G4Types.hh"
44#include <functional>
45
46class G4LogicalVolume;
47class G4SmartVoxelHeader;
48
49class G4SmartVoxelStat
50{
51 public: // with description
52
53 G4SmartVoxelStat( const G4LogicalVolume *theVolume,
54 const G4SmartVoxelHeader *theVoxel,
55 G4double theSysTime,
56 G4double theUserTime );
57 // Construct information on one volume's voxels
58
59 const G4LogicalVolume *GetVolume() const;
60 // Return a pointer to the logical volume
61
62 const G4SmartVoxelHeader *GetVoxel() const;
63 // Return a pointer to the voxel header
64
65 G4double GetSysTime() const;
66 // Get amount of system CPU time needed to build voxels
67
68 G4double GetUserTime() const;
69 // Get amount of user CPU time needed to build voxels
70
71 G4double GetTotalTime() const;
72 // Get total amount of CPU time needed to build voxels
73
74 G4long GetNumberHeads() const;
75 // Get number of voxel headers used in the volume
76
77 G4long GetNumberNodes() const;
78 // Get number of voxel slices used in the volume
79
80 G4long GetNumberPointers() const;
81 // Get number of voxel proxy pointers used in the volume
82
83 G4long GetMemoryUse() const;
84 // Get number of bytes needed to store voxel information
85
86
87 protected:
88
89 void CountHeadsAndNodes( const G4SmartVoxelHeader *head );
90
91 const G4LogicalVolume *volume;
92 const G4SmartVoxelHeader *voxel;
93
94 G4double sysTime;
95 G4double userTime;
96
97 G4long heads;
98 G4long nodes;
99 G4long pointers;
100
101
102 public:
103
104 //
105 // Functor objects for sorting
106 //
107 struct ByCpu
108 : public std::binary_function< const G4SmartVoxelStat,
109 const G4SmartVoxelStat, G4bool >
110 {
111 G4bool operator()( const G4SmartVoxelStat &a, const G4SmartVoxelStat &b )
112 {
113 return a.GetTotalTime() > b.GetTotalTime();
114 }
115 };
116
117 struct ByMemory
118 : public std::binary_function< const G4SmartVoxelStat,
119 const G4SmartVoxelStat, G4bool >
120 {
121 G4bool operator()( const G4SmartVoxelStat &a, const G4SmartVoxelStat &b )
122 {
123 return a.GetMemoryUse() > b.GetMemoryUse();
124 }
125 };
126};
127
128#endif
