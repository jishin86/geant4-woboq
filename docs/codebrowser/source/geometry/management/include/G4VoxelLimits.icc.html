69inline
70G4double G4VoxelLimits::GetMaxExtent(const EAxis pAxis) const
71{
72 if (pAxis==kXAxis)
73 {
74 return GetMaxXExtent();
75 }
76 else if (pAxis==kYAxis)
77 {
78 return GetMaxYExtent();
79 }
80 else
81 {
82 assert(pAxis==kZAxis);
83 return GetMaxZExtent();
84 }
85}

105inline
106G4bool G4VoxelLimits::IsXLimited() const
107{
108 return (fxAxisMin==-kInfinity&&fxAxisMax==kInfinity) ? false : true;
109}
110
111inline
112G4bool G4VoxelLimits::IsYLimited() const
113{
114 return (fyAxisMin==-kInfinity&&fyAxisMax==kInfinity) ? false : true;
115}
116
117inline
118G4bool G4VoxelLimits::IsZLimited() const
119{
120 return (fzAxisMin==-kInfinity&&fzAxisMax==kInfinity) ? false : true;
121}
122
123inline
124G4bool G4VoxelLimits::IsLimited() const
125{
126 return (IsXLimited()||IsYLimited()||IsZLimited());
127}
128
129inline
130G4bool G4VoxelLimits::IsLimited(const EAxis pAxis) const
131{
132 if (pAxis==kXAxis)
133 {
134 return IsXLimited();
135 }
136 else if (pAxis==kYAxis)
137 {
138 return IsYLimited();
139 }
140 else
141 {
142 assert(pAxis==kZAxis);
143 return IsZLimited();
144 }
145}
146
147inline
148G4bool G4VoxelLimits::Inside(const G4ThreeVector& pVec) const
149{
150 return ((GetMinXExtent()<=pVec.x()) &&
151 (GetMaxXExtent()>=pVec.x()) &&
152 (GetMinYExtent()<=pVec.y()) &&
153 (GetMaxYExtent()>=pVec.y()) &&
154 (GetMinZExtent()<=pVec.z()) &&
155 (GetMaxZExtent()>=pVec.z()) ) ? true : false;
156}
