119G4UImanager::~G4UImanager()
120{
121 if(bridges)
122 {
123 std::vector<G4UIbridge*>::iterator itr = bridges->begin();
124 for(;itr!=bridges->end();itr++)
125 { delete *itr; }
126 delete bridges;
127 }
128 SetCoutDestination(NULL);
129 histVec.clear();
130 if(saveHistory) historyFile.close();
131 delete CoutMessenger;
132 delete UnitsMessenger;
133 delete UImessenger;
134 delete treeTop;
135 delete aliasList;
136 fUImanagerHasBeenKilled() = true;
137 fUImanager() = NULL;
138 if(commandStack)
139 {
140 commandStack->clear();
141 delete commandStack;
142 }
143 if(threadID >= 0)
144 {
145 if(threadCout) delete threadCout;
146 G4iosFinalization();
147 threadID = -1;
148 }
149}

201G4String G4UImanager::GetCurrentStringValue(const char * aCommand,
202G4int parameterNumber, G4bool reGet)
203{
204 if(reGet || savedCommand == NULL)
205 {
206 savedParameters = GetCurrentValues( aCommand );
207 }
208 G4Tokenizer savedToken( savedParameters );
209 G4String token;
210 for(G4int i_thParameter=0;i_thParameter<parameterNumber;i_thParameter++)
211 {
212 token = savedToken();
213 if( token.isNull() ) return G4String();
214 if( token[(size_t)0] == '"' )
215 {
216 token.append(" ");
217 token.append(savedToken("\""));
218 }
219 }
220 return token;
221}

321{
322 G4String vl = valueList;
323 G4Tokenizer parameterToken(vl);
324 G4String mf = parameterToken();
325 G4String vn = parameterToken();
326 G4String c1 = parameterToken();
327 c1 += " ";
328 c1 += parameterToken();
329 c1 += " ";
330 c1 += parameterToken();
331 const char* t1 = c1;
332 std::istringstream is(t1);
333 G4double d1;
334 G4double d2;

336 is >> d1 >> d2 >> d3;
337 Loop(mf,vn,d1,d2,d3);

340void G4UImanager::Loop(const char * macroFile,const char * variableName,
341 G4double initialValue,G4double finalValue,G4double stepSize)
342{
343 G4String cd;
344 if (stepSize > 0) {
345 for(G4double d=initialValue;d<=finalValue;d+=stepSize)
346 {
347 std::ostringstream os;
348 os << d;
349 cd += os.str();
350 cd += " ";
351 }
352 } else {
353 for(G4double d=initialValue;d>=finalValue;d+=stepSize)
354 {
355 std::ostringstream os;
356 os << d;
357 cd += os.str();
358 cd += " ";
359 }
360 }
361 Foreach(macroFile,variableName,cd);
362}

364void G4UImanager::ForeachS(const char* valueList)
365{
366 G4String vl = valueList;
367 G4Tokenizer parameterToken(vl);
368 G4String mf = parameterToken();
369 G4String vn = parameterToken();
370 G4String c1 = parameterToken();
371 G4String ca;
372 while(!((ca=parameterToken()).isNull()))
373 {
374 c1 += " ";
375 c1 += ca;
376 }
377
378 G4String aliasValue = c1;
379 if(aliasValue(0)=='"')
380 {
381 G4String strippedValue;
382 if(aliasValue(aliasValue.length()-1)=='"')
383 { strippedValue = aliasValue(1,aliasValue.length()-2); }
384 else
385 { strippedValue = aliasValue(1,aliasValue.length()-1); }
386 aliasValue = strippedValue;
387 }
388
389// Foreach(mf,vn,c1);
390 Foreach(mf,vn,aliasValue);
391}

393void G4UImanager::Foreach(const char * macroFile,const char * variableName,
394 const char * candidates)
395{
396 G4String candidatesString = candidates;
397 G4Tokenizer parameterToken( candidatesString );
398 G4String cd;
399 while(!((cd=parameterToken()).isNull()))
400 {
401 G4String vl = variableName;
402 vl += " ";
403 vl += cd;
404 SetAlias(vl);
405 ExecuteMacroFile(FindMacroPath(macroFile));
406 }
407}

410G4String G4UImanager::SolveAlias(const char* aCmd)
411{
412 G4String aCommand = aCmd;
413 G4int ia = aCommand.index("{");
414 G4int iz = aCommand.index("#");
415 while((ia != G4int(std::string::npos))&&((iz==G4int(std::string::npos))||(ia<iz)))
416 {
417 G4int ibx = -1;
418 while(ibx<0)
419 {
420 G4int ib = aCommand.index("}");
421 if( ib == G4int(std::string::npos) )
422 {
423 G4cerr << aCommand << G4endl;
424 for(G4int i=0;i<ia;i++) G4cerr << " ";
425 G4cerr << "^" << G4endl;
426 G4cerr << "Unmatched alias parenthesis -- command ignored" << G4endl;
427 G4String nullStr;
428 return nullStr;
429 }
430 G4String ps = aCommand(ia+1,aCommand.length()-(ia+1));
431 G4int ic = ps.index("{");
432 G4int id = ps.index("}");
433 if(ic!=G4int(std::string::npos) && ic < id)
434 { ia+=ic+1; }
435 else
436 { ibx = ib; }
437 }
438 //--- Here ia represents the position of innermost "{"
439 //--- and ibx represents corresponding "}"
440 G4String subs;
441 if(ia>0) subs = aCommand(0,ia);
442 G4String alis = aCommand(ia+1,ibx-ia-1);
443 G4String rems = aCommand(ibx+1,aCommand.length()-ibx);
444 // G4cout << "<" << subs << "> <" << alis << "> <" << rems << ">" << G4endl;
445 G4String* alVal = aliasList->FindAlias(alis);
446 if(!alVal)
447 {
448 G4cerr << "Alias <" << alis << "> not found -- command ignored" << G4endl;
449 G4String nullStr;
450 return nullStr;
451 }
452 aCommand = subs+(*alVal)+rems;
453 ia = aCommand.index("{");
454 }
455 return aCommand;
456}

465G4int G4UImanager::ApplyCommand(const char * aCmd)
466{
467 G4String aCommand = SolveAlias(aCmd);
468 if(aCommand.isNull()) return fAliasNotFound;
469 if(verboseLevel) G4cout << aCommand << G4endl;
470 G4String commandString;
471 G4String commandParameter;
472
473 G4int i = aCommand.index(" ");
474 if( i != G4int(std::string::npos) )
475 {
476 commandString = aCommand(0,i);
477 commandParameter = aCommand(i+1,aCommand.length()-(i+1));
478 }
479 else
480 {
481 commandString = aCommand;
482 }
483
484 // remove doubled slash
485 G4int len = commandString.length();
486 G4int ll = 0;
487 G4String a1;
488 G4String a2;
489 while(ll<len-1)
490 {
491 if(commandString(ll,2)=="//")
492 {
493 if(ll==0)
494 { commandString.remove(ll,1); }
495 else
496 {
497 a1 = commandString(0,ll);
498 a2 = commandString(ll+1,len-ll-1);
499 commandString = a1+a2;
500 }
501 len--;
502 }
503 else
504 { ll++; }
505 }
506
507 if(isMaster&&bridges)
508 {
509 std::vector<G4UIbridge*>::iterator itr = bridges->begin();
510 for(;itr!=bridges->end();itr++)
511 {
512 G4int leng = (*itr)->DirLength();
513 if(commandString(0,leng)==(*itr)->DirName())
514 { return (*itr)->LocalUI()->ApplyCommand(commandString+" "+commandParameter); }
515 }
516 }
517
518 G4UIcommand * targetCommand = treeTop->FindPath( commandString );
519 if( targetCommand == NULL )
520 {
521 if(ignoreCmdNotFound)
522 {
523 if(stackCommandsForBroadcast)
524 { commandStack->push_back(commandString+" "+commandParameter); }
525 return fCommandSucceeded;
526 }
527 else
528 { return fCommandNotFound; }
529 }
530
531 if(stackCommandsForBroadcast && targetCommand->ToBeBroadcasted())
532 { commandStack->push_back(commandString+" "+commandParameter); }
533
534 if(!(targetCommand->IsAvailable()))
535 { return fIllegalApplicationState; }
536
537 if(saveHistory) historyFile << aCommand << G4endl;
538 if( G4int(histVec.size()) >= maxHistSize )
539 { histVec.erase(histVec.begin()); }
540 histVec.push_back(aCommand);
541
542 targetCommand->ResetFailure();
543 G4int commandFailureCode = targetCommand->DoIt( commandParameter );
544 if(commandFailureCode==0)
545 {
546 G4int additionalFailureCode = targetCommand->IfCommandFailed();
547 if(additionalFailureCode > 0)
548 {
549 G4ExceptionDescription msg;
550 msg << targetCommand->GetFailureDescription() << "\n"
551 << "Error code : " << additionalFailureCode;
552 G4Exception("G4UImanager::ApplyCommand","UIMAN0123",
553 JustWarning,msg);
554 commandFailureCode += additionalFailureCode;
555 }
556 }
557 return commandFailureCode;
558}

594G4UIcommandTree* G4UImanager::FindDirectory(const char* dirName)
595{
596 G4String aDirName = dirName;
597 G4String targetDir = aDirName.strip(G4String::both);
598 if( targetDir( targetDir.length()-1 ) != '/' )
599 { targetDir += "/"; }
600 G4UIcommandTree* comTree = treeTop;
601 if( targetDir == "/" )
602 { return comTree; }
603 G4int idx = 1;
604 while( idx < G4int(targetDir.length())-1 )
605 {
606 G4int i = targetDir.index("/",idx);
607 G4String targetDirString = targetDir(0,i+1);
608 comTree = comTree->GetTree(targetDirString);
609 if( comTree == NULL )
610 { return NULL; }
611 idx = i+1;
612 }
613 return comTree;
614}

617{
618 //G4cout << G4StateManager::GetStateManager()->GetStateString(requestedState) << " <--- " << G4StateManager::GetStateManager()->GetStateString(G4StateManager::GetStateManager()->GetPreviousState()) << G4endl;
619 if(pauseAtBeginOfEvent)

621 if(requestedState==G4State_EventProc &&
622 G4StateManager::GetStateManager()->GetPreviousState()==G4State_GeomClosed)
623 { PauseSession("BeginOfEvent"); }
624 }
625 if(pauseAtEndOfEvent)
626 {
627 if(requestedState==G4State_GeomClosed &&
628 G4StateManager::GetStateManager()->GetPreviousState()==G4State_EventProc)
629 { PauseSession("EndOfEvent"); }
630 }
631 return true;
632}

698void G4UImanager::ParseMacroSearchPath()
699{
700 searchDirs.clear();
701
702 size_t idxfirst = 0;
703 size_t idxend = 0;
704 G4String pathstring = "";
705 while( (idxend = searchPath.index(':', idxfirst)) != G4String::npos) {
706 pathstring = searchPath.substr(idxfirst, idxend-idxfirst);
707 if(pathstring.size() != 0) searchDirs.push_back(pathstring);
708 idxfirst = idxend + 1;
709 }
710
711 pathstring = searchPath.substr(idxfirst, searchPath.size()-idxfirst);
712 if(pathstring.size() != 0) searchDirs.push_back(pathstring);
713}

716static G4bool FileFound(const G4String& fname)
717{
718 G4bool qopen = false;
719 std::ifstream fs;
720 fs.open(fname.c_str(), std::ios::in);
721 if(fs.good()) {
722 fs.close();
723 qopen = true;
724 }
725 return qopen;
726}

728G4String G4UImanager::FindMacroPath(const G4String& fname) const
729{
730 G4String macrofile = fname;
731
732 for (size_t i = 0; i < searchDirs.size(); i++) {
733 G4String fullpath = searchDirs[i] + "/" + fname;
734 if ( FileFound(fullpath) ) {
735 macrofile = fullpath;
736 break;
737 }
738 }
739
740 return macrofile;
741}
