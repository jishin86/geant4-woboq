132void G4UIparameter::SetDefaultUnit(const char * theDefaultUnit)
133{
134 char type = toupper( parameterType );
135 if( type != 'S' )
136 {
137 G4ExceptionDescription ed;
138 ed << "This method can be used only for a string-type parameter that is used to specify a unit.\n"
139 << "This parameter <" << parameterName << "> is defined as ";
140 switch(type) {
141 case 'D': ed <<"double."; break;
142 case 'I': ed <<"integer."; break;
143 case 'B': ed <<"bool."; break;
144 default: ed <<"undefined.";
145 }
146 G4Exception("G4UIparameter::SetDefaultUnit","INTERCOM2010",FatalException,ed);
147 }
148 SetDefaultValue(theDefaultUnit);
149 SetParameterCandidates(G4UIcommand::UnitsList(G4UIcommand::CategoryOf(theDefaultUnit)));
150}

159G4int G4UIparameter::
160CheckNewValue(const char* newValue ) {
161 if( TypeCheck(newValue) == 0) return fParameterUnreadable;
162 if( ! parameterRange.isNull() )
163 { if( RangeCheck(newValue) == 0 ) return fParameterOutOfRange; }
164 if( ! parameterCandidate.isNull() )
165 { if( CandidateCheck(newValue) == 0 ) return fParameterOutOfCandidates; }
166 return 0; // succeeded
167}

169G4int G4UIparameter::
170CandidateCheck(const char* newValue) {
171 G4Tokenizer candidateTokenizer(parameterCandidate);
172 G4String aToken;
173 G4int iToken = 0;
174 while( ! (aToken=candidateTokenizer()).isNull() )
175 {
176 iToken++;
177 if(aToken==newValue) return iToken;
178 }
179 G4cerr << "parameter value (" << newValue
180 << ") is not listed in the candidate List." << G4endl;
181 return 0;
182}

184G4int G4UIparameter::
185RangeCheck(const char* newValue) {
186 yystype result;
187 bp = 0; // reset buffer pointer for G4UIpGetc()
188 std::istringstream is(newValue);
189 char type = toupper( parameterType );
190 switch (type) {
191 case 'D': { is >> newVal.D; } break;
192 case 'I': { is >> newVal.I; } break;
193 default: ;
194 }
195 // PrintToken(); // Print tokens (consumes all tokens)
196 token= Yylex();
197 result = Expression();
198 if( paramERR == 1 ) return 0;
199 if( result.type != CONSTINT) {
200 G4cerr << "Illegal Expression in parameter range." << G4endl;
201 return 0;
202 }
203 if ( result.I ) return 1;
204 G4cerr << "parameter out of range: "<< parameterRange << G4endl;
205 return 0;
206}

209G4int G4UIparameter::
210TypeCheck(const char* newValue)
211{
212 G4String newValueString(newValue);
213 char type = toupper( parameterType );
214 switch(type) {
215 case 'D':
216 if( IsDouble(newValueString.data())== 0) {
217 G4cerr<<newValue<<": double value expected."
218 << G4endl;
219 return 0;
220 } break;
221 case 'I':
222 if( IsInt(newValueString.data(),20)== 0) {
223 G4cerr<<newValue<<": integer expected."
224 << G4endl;
225 return 0;
226 } break;
227 case 'S': break;
228 case 'B':
229 newValueString.toUpper();
230 if ( newValueString == "Y" || newValueString == "N"
231 ||newValueString == "YES" || newValueString == "NO"
232 ||newValueString == "1" || newValueString == "0"
233 ||newValueString == "T" || newValueString == "F"
234 ||newValueString == "TRUE" || newValueString == "FALSE")
235 return 1;
236 else {
237 G4cerr<<newValue<<": bool expected." << G4endl;
238 return 0;
239 }
240 default: ;
241 }
242 return 1;
243}

246G4int G4UIparameter::
247IsInt(const char* buf, short maxDigits) // do not allow any std::ws
248{
249 const char* p= buf;
250 G4int length=0;
251 if( *p == '+' || *p == '-') { ++p; }
252 if( isdigit( (G4int)(*p) )) {
253 while( isdigit( (G4int)(*p) )) { ++p; ++length; }
254 if( *p == '\0' ) {
255 if( length > maxDigits) {
256 G4cerr <<"digit length exceeds"<<G4endl;
257 return 0;
258 }
259 return 1;
260 } else {
261 // G4cerr <<"illegal character after int:"<<buf<<G4endl;
262 }
263 } else {
264 // G4cerr <<"illegal int:"<<buf<<G4endl;
265 }
266 return 0;
267}

278G4int G4UIparameter::
279IsDouble(const char* buf) // see state diagram for this spec.
280{
281 const char* p= buf;
282 switch( *p) {
283 case '+': case '-': ++p;
284 if( isdigit(*p) ) {
285 while( isdigit( (G4int)(*p) )) { ++p; }
286 switch ( *p ) {
287 case '\0': return 1; //break;
288 case 'E': case 'e':
289 return ExpectExponent(++p ); //break;
290 case '.': ++p;
291 if( *p == '\0' ) return 1;
292 if( *p == 'e' || *p =='E' ) return ExpectExponent(++p );
293 if( isdigit(*p) ) {
294 while( isdigit( (G4int)(*p) )) { ++p; }
295 if( *p == '\0' ) return 1;
296 if( *p == 'e' || *p =='E') return ExpectExponent(++p);
297 } else return 0; break;
298 default: return 0;
299 }
300 }
301 if( *p == '.' ) { ++p;
302 if( isdigit(*p) ) {
303 while( isdigit( (G4int)(*p) )) { ++p; }
304 if( *p == '\0' ) return 1;
305 if( *p == 'e' || *p =='E') return ExpectExponent(++p);
306 }
307 }
308 break;
309 case '.': ++p;
310 if( isdigit(*p) ) {
311 while( isdigit( (G4int)(*p) )) { ++p; }
312 if( *p == '\0' ) return 1;
313 if( *p == 'e' || *p =='E' ) return ExpectExponent(++p);
314 } break;
315 default: // digit is expected
316 if( isdigit(*p) ) {
317 while( isdigit( (G4int)(*p) )) { ++p; }
318 if( *p == '\0' ) return 1;
319 if( *p == 'e' || *p =='E') return ExpectExponent(++p);
320 if( *p == '.' ) { ++p;
321 if( *p == '\0' ) return 1;
322 if( *p == 'e' || *p =='E') return ExpectExponent(++p);
323 if( isdigit(*p) ) {
324 while( isdigit( (G4int)(*p) )) { ++p; }
325 if( *p == '\0' ) return 1;
326 if( *p == 'e' || *p =='E') return ExpectExponent(++p);
327 }
328 }
329 }
330 }
331 return 0;
332}

348yystype G4UIparameter::
349LogicalORExpression(void)
350{
351 yystype result;
352 yystype p;
353 p = LogicalANDExpression();
354 if( token != LOGICALOR) return p;
355 if( p.type == CONSTSTRING || p.type == IDENTIFIER ) {
356 G4cerr << "Parameter range: illegal type at '||'" << G4endl;
357 paramERR = 1;
358 }
359 result.I = p.I;
360 while (token == LOGICALOR)
361 {
362 token = Yylex();
363 p = LogicalANDExpression();
364 if( p.type == CONSTSTRING || p.type == IDENTIFIER ) {
365 G4cerr << "Parameter range: illegal type at '||'" <<G4endl;
366 paramERR = 1;
367 }
368 switch (p.type) {
369 case CONSTINT:
370 result.I += p.I;
371 result.type = CONSTINT; break;
372 case CONSTDOUBLE:
373 result.I += (p.D != 0.0);
374 result.type = CONSTINT; break;
375 default:
376 G4cerr << "Parameter range: unknown type"<<G4endl;
377 paramERR = 1;
378 }
379 }
380 return result;
381}

383yystype G4UIparameter::
384LogicalANDExpression(void)
385{
386 yystype result;
387 yystype p;
388 p = EqualityExpression();
389 if( token != LOGICALAND) return p;
390 if( p.type == CONSTSTRING || p.type == IDENTIFIER ) {
391 G4cerr << "Parameter range: illegal type at '&&'" << G4endl;
392 paramERR = 1;
393 }
394 result.I = p.I;
395 while (token == LOGICALAND)
396 {
397 token = Yylex();
398 p = EqualityExpression();
399 if( p.type == CONSTSTRING || p.type == IDENTIFIER ) {
400 G4cerr << "Parameter range: illegal type at '&&'" << G4endl;
401 paramERR = 1;
402 }
403 switch (p.type) {
404 case CONSTINT:
405 result.I *= p.I;
406 result.type = CONSTINT; break;
407 case CONSTDOUBLE:
408 result.I *= (p.D != 0.0);
409 result.type = CONSTINT; break;
410 default:
411 G4cerr << "Parameter range: unknown type."<< G4endl;
412 paramERR = 1;
413 }
414 }
415 return result;
416}

419yystype G4UIparameter::
420EqualityExpression(void)
421{
422 yystype arg1, arg2;
423 G4int operat;
424 yystype result;
425 #ifdef DEBUG
426 G4cerr << " EqualityExpression()" <<G4endl;
427 #endif
428 result = RelationalExpression();
429 if( token==EQ || token==NE ) {
430 operat = token;
431 token = Yylex();
432 arg1 = result;
433 arg2 = RelationalExpression();
434 result.I = Eval2( arg1, operat, arg2 ); // semantic action
435 result.type = CONSTINT;
436 #ifdef DEBUG
437 G4cerr << " return code of Eval2(): " << result.I <<G4endl;
438 #endif
439 } else {
440 if (result.type != CONSTINT && result.type != CONSTDOUBLE) {
441 G4cerr << "Parameter range: error at EqualityExpression"
442 << G4endl;
443 paramERR = 1;
444 }
445 }
446 return result;
447}

450yystype G4UIparameter::
451RelationalExpression(void)
452{
453 yystype arg1, arg2;
454 G4int operat;
455 yystype result;
456 #ifdef DEBUG

473 #ifdef DEBUG

475 #endif
476 return result;

503yystype G4UIparameter::
504UnaryExpression(void)
505{
506 yystype result;
507 yystype p;
508 #ifdef DEBUG
509 G4cerr <<" UnaryExpression"<< G4endl;
510 #endif
511 switch(token) {
512 case '-':
513 token = Yylex();
514 p = UnaryExpression();
515 if (p.type == CONSTINT) {
516 result.I = - p.I;
517 result.type = CONSTINT;
518 }
519 if (p.type == CONSTDOUBLE) {
520 result.D = - p.D;
521 result.type = CONSTDOUBLE;
522 } break;
523 case '+':
524 token = Yylex();
525 result = UnaryExpression(); break;
526 case '!':
527 token = Yylex();
528 G4cerr << "Parameter range error: "
529 << "operator '!' is not supported (sorry)."
530 << G4endl;
531 paramERR = 1;
532 result = UnaryExpression(); break;
533 default:
534 result = PrimaryExpression();
535 }
536 return result;
537}

540yystype G4UIparameter::
541PrimaryExpression(void)
542{
543 yystype result;
544 #ifdef DEBUG
545 G4cerr <<" PrimaryExpression" << G4endl;
546 #endif
547 switch (token) {
548 case IDENTIFIER:
549 result.S = yylval.S;
550 result.type = token;
551 token = Yylex(); break;
552 case CONSTINT:
553 result.I = yylval.I;
554 result.type = token;
555 token= Yylex(); break;
556 case CONSTDOUBLE:
557 result.D = yylval.D;
558 result.type = token;
559 token = Yylex(); break;
560 case '(' :
561 token= Yylex();
562 result = Expression();
563 if( token != ')' ) {
564 G4cerr << " ')' expected" << G4endl;
565 paramERR = 1;
566 }
567 token = Yylex();
568 break;
569 default:
570 return result;
571 }
572 return result; // never executed
573}

577G4int G4UIparameter::
578Eval2(yystype arg1, G4int op, yystype arg2)
579{
580 if( (arg1.type != IDENTIFIER) && (arg2.type != IDENTIFIER)) {
581 G4cerr << parameterName
582 << ": meaningless comparison "
583 << G4int(arg1.type) << " " << G4int(arg2.type) << G4endl;
584 paramERR = 1;
585 }
586 char type = toupper( parameterType );
587 if( arg1.type == IDENTIFIER) {
588 switch (type) {
589 case 'I':
590 if ( arg2.type == CONSTINT ) {
591 return CompareInt( newVal.I, op, arg2.I );
592 } else {
593 G4cerr << "integer operand expected for "
594 << parameterRange << '.'
595 << G4endl;
596 }
597 break;
598 case 'D':
599 if ( arg2.type == CONSTDOUBLE ) {
600 return CompareDouble( newVal.D, op, arg2.D );
601 } else
602 if ( arg2.type == CONSTINT ) { // integral promotion
603 return CompareDouble( newVal.D, op, arg2.I );
604 } break;
605 default: ;
606 }
607 }
608 if( arg2.type == IDENTIFIER) {
609 switch (type) {
610 case 'I':
611 if ( arg1.type == CONSTINT ) {
612 return CompareInt( arg1.I, op, newVal.I );
613 } else {
614 G4cerr << "integer operand expected for "
615 << parameterRange << '.'
616 << G4endl;
617 }
618 break;
619 case 'D':
620 if ( arg1.type == CONSTDOUBLE ) {
621 return CompareDouble( arg1.D, op, newVal.D );
622 } else
623 if ( arg1.type == CONSTINT ) { // integral promotion
624 return CompareDouble( arg1.I, op, newVal.D );
625 } break;
626 default: ;
627 }
628 }
629 G4cerr << "no param name is specified at the param range."<<G4endl;
630 return 0;
631}

633G4int G4UIparameter::
634CompareInt(G4int arg1, G4int op, G4int arg2)
635{
636 G4int result=-1;
637 G4String opr;
638 switch (op) {
639 case GT: result = ( arg1 > arg2); opr= ">" ; break;
640 case GE: result = ( arg1 >= arg2); opr= ">="; break;
641 case LT: result = ( arg1 < arg2); opr= "<" ; break;
642 case LE: result = ( arg1 <= arg2); opr= "<="; break;
643 case EQ: result = ( arg1 == arg2); opr= "=="; break;
644 case NE: result = ( arg1 != arg2); opr= "!="; break;
645 default:
646 G4cerr << "Parameter range: error at CompareInt" << G4endl;
647 paramERR = 1;
648 }
649 #ifdef DEBUG
650 G4cerr << "CompareInt "
651 << arg1 << " " << opr << arg2
652 << " result: " << result
653 << G4endl;
654 #endif
655 return result;
656}

683// --------------------- utility functions --------------------------
684
685tokenNum G4UIparameter::
686Yylex() // reads input and returns token number KR486
687{ // (returns EOF)
688 G4int c;
689 G4String buf;
690
691 while(( c= G4UIpGetc())==' '|| c=='\t' || c== '\n' )
692 ;
693 if (c== EOF)
694 return (tokenNum)EOF; // KR488
695 buf= "";
696 if (isdigit(c) || c== '.') { // I or D
697 do {
698 buf += G4String((unsigned char)c);
699 c=G4UIpGetc();
700 } while (c=='.' || isdigit(c) ||
701 c=='e' || c=='E' || c=='+' || c=='-');
702 G4UIpUngetc(c);
703 const char* t = buf;
704 std::istringstream is(t);
705 if ( IsInt(buf.data(),20) ) {
706 is >> yylval.I;
707 return CONSTINT;
708 } else
709 if ( IsDouble(buf.data()) ) {
710 is >> yylval.D;
711 return CONSTDOUBLE;
712 } else {
713 G4cerr << buf<<": numeric format error."<<G4endl;
714 }
715 }
716 buf="";
717 if (isalpha(c)|| c=='_') { // IDENTIFIER
718 do {
719 buf += G4String((unsigned char)c);
720 } while ((c=G4UIpGetc()) != EOF && (isalnum(c) || c=='_'));
721 G4UIpUngetc(c);
722 if( buf == parameterName ) {
723 yylval.S =buf;
724 return IDENTIFIER;
725 } else {
726 G4cerr << buf << " is not a parameter name."<< G4endl;
727 paramERR = 1;
728 }
729 }
730 switch (c) {
731 case '>': return (tokenNum) Follow('=', GE, GT);
732 case '<': return (tokenNum) Follow('=', LE, LT);
733 case '=': return (tokenNum) Follow('=', EQ, '=');
734 case '!': return (tokenNum) Follow('=', NE, '!');
735 case '|': return (tokenNum) Follow('|', LOGICALOR, '|');
736 case '&': return (tokenNum) Follow('&', LOGICALAND, '&');
737 default:
738 return (tokenNum) c;
739 }
740}

753//------------------ low level routines -----------------------------
754G4int G4UIparameter::
755G4UIpGetc() { // emulation of getc()
756 G4int length = parameterRange.length();
757 if( bp < length)
758 return parameterRange(bp++);
759 else
760 return EOF;
761}
762G4int G4UIparameter::
763G4UIpUngetc(G4int c) { // emulation of ungetc()
764 if (c<0) return -1;
765 if (bp >0 && c == parameterRange(bp-1)) {
766 --bp;
767 } else {
