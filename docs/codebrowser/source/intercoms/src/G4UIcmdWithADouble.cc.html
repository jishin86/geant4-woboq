30#include "G4UIcmdWithADouble.hh"
31
32G4UIcmdWithADouble::G4UIcmdWithADouble
33(const char * theCommandPath,G4UImessenger * theMessenger)
34:G4UIcommand(theCommandPath,theMessenger)
35{
36 G4UIparameter * dblParam = new G4UIparameter('d');
37 SetParameter(dblParam);
38}
39
40G4double G4UIcmdWithADouble::GetNewDoubleValue(const char* paramString)
41{
42 return ConvertToDouble(paramString);
43}
44
45void G4UIcmdWithADouble::SetParameterName
46(const char * theName,G4bool omittable,G4bool currentAsDefault)
47{
48 G4UIparameter * theParam = GetParameter(0);
49 theParam->SetParameterName(theName);
50 theParam->SetOmittable(omittable);
51 theParam->SetCurrentAsDefault(currentAsDefault);
52}
53
54void G4UIcmdWithADouble::SetDefaultValue(G4double defVal)
55{
56 G4UIparameter * theParam = GetParameter(0);
57 theParam->SetDefaultValue(defVal);
58}
