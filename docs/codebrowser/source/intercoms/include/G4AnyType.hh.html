58/**
59 * @class G4AnyType G4AnyType.hh
60 * This class represents any data type. The class only holds a reference to the type and not the value.
61 */
62class G4AnyType {
63public:
64 /** Constructor */
65 G4AnyType():
66 fContent(0) {}
67
68 /** Constructor */
69 template <typename ValueType> G4AnyType(ValueType &value):
70 fContent(new Ref<ValueType>(value)) {}
71
72 /** Copy Constructor */
73 G4AnyType(const G4AnyType &other):
74 fContent(other.fContent ? other.fContent->Clone() : 0) {}
75
76 /** Dtor */
77 ~G4AnyType() {
78 delete fContent;
79 }
80
81 /** bool operator */
82 operator bool() {
83 return !Empty();
84 }
85 /** Modifier */
86 G4AnyType& Swap(G4AnyType& rhs) {
87 std::swap(fContent, rhs.fContent);
88 return *this;
89 }
90 /** Modifier */
91 template <typename ValueType> G4AnyType& operator =(const ValueType& rhs) {
92 G4AnyType(rhs).Swap(*this);
93 return *this;
94 }
95 /** Modifier */
96 G4AnyType& operator =(const G4AnyType& rhs) {
97 G4AnyType(rhs).Swap(*this);
98 return *this;
99 }
100 /** Query */
101 bool Empty() const {
102 return !fContent;
103 }
104 /** Query */
105 const std::type_info& TypeInfo() const {
106 return fContent ? fContent->TypeInfo() : typeid(void);
107 }
108 /** Adress */
109 void* Address() const {
110 return fContent ? fContent->Address() : 0;
111 }
112 /** String conversion */
113 std::string ToString() const {
114 return fContent->ToString();
115 }
116 /** String conversion */
117 void FromString(const std::string& val) {
118 fContent->FromString(val);
119 }
120private:
121 /**
122 * @class Placeholder G4AnyType.h G4AnyType.h
123 */
124 class Placeholder {
125 public:
126 /** Constructor */
127 Placeholder() {}
128 /** Destructor */
129 virtual ~Placeholder() {}
130 /** Query */
131 virtual const std::type_info& TypeInfo() const = 0;
132 /** Query */
133 virtual Placeholder* Clone() const = 0;
134 /** Query */
135 virtual void* Address() const = 0;
136 /** ToString */
137 virtual std::string ToString() const = 0;
138 /** FromString */
139 virtual void FromString(const std::string& val) = 0;
140 };
141
142 template <typename ValueType> class Ref: public Placeholder {
143 public:
144 /** Constructor */
145 Ref(ValueType& value): fRef(value) {}
146 /** Query */
147 virtual const std::type_info& TypeInfo() const {
148 return typeid(ValueType);
149 }
150 /** Clone */
151 virtual Placeholder* Clone() const {
152 return new Ref(fRef);
153 }
154 /** Address */
155 virtual void* Address() const {
156 return (void*) (&fRef);
157 }
158 /** ToString */
159 virtual std::string ToString() const {
160 std::stringstream ss;
161 ss << fRef;
162 return ss.str();
163 }
164 /** FromString */
165 virtual void FromString(const std::string& val) {
166 std::stringstream ss(val);
167 ss >> fRef;
168 }
169 /** representation */
170 ValueType& fRef;
171 };
172 /** representation */
173 template <typename ValueType> friend ValueType* any_cast(G4AnyType*);
174 /** representation */
175 Placeholder* fContent;
176};
177
178/**
179 * Specializations
180 */
181
182template <> inline void G4AnyType::Ref<bool>::FromString(const std::string& val) {
183 fRef = G4UIcommand::ConvertToBool(val.c_str());
184}
185
186template <> inline void G4AnyType::Ref<G4String>::FromString(const std::string& val) {
187 if (val[0] == '"' ) fRef = val.substr(1,val.size()-2);
188 else fRef = val;
189}
190
191template <> inline void G4AnyType::Ref<G4ThreeVector>::FromString(const std::string& val) {
192 fRef = G4UIcommand::ConvertTo3Vector(val.c_str());
193}

196/**
197 * @class G4BadAnyCast G4AnyType.h Reflex/G4AnyType.h
198 * @author K. Henney
199 */
200class G4BadAnyCast: public std::bad_cast {
201public:
202 /** Constructor */
203 G4BadAnyCast() {}
204
205 /** Query */
206 virtual const char* what() const throw() {
207 return "G4BadAnyCast: failed conversion using any_cast";
208 }
209};
210
211/** value */
212template <typename ValueType> ValueType* any_cast(G4AnyType* operand) {
213 return operand && operand->TypeInfo() == typeid(ValueType)
214 ? &static_cast<G4AnyType::Ref<ValueType>*>(operand->fContent)->fRef : 0;
215}
216/** value */
217template <typename ValueType> const ValueType* any_cast(const G4AnyType* operand) {
218 return any_cast<ValueType>(const_cast<G4AnyType*>(operand));
219}
220/** value */
221template <typename ValueType> ValueType any_cast(const G4AnyType& operand) {
222 const ValueType* result = any_cast<ValueType>(&operand);
223 if (!result) {
224 throw G4BadAnyCast();
225 }
226 return *result;
227}
