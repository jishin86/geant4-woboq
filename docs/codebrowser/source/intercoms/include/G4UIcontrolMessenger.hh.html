32#include "G4UImessenger.hh"

34class G4UIdirectory;
35class G4UIcmdWithAString;
36class G4UIcmdWithABool;

41// class description:
42// This class is a concrete class of G4UImessenger which defines
43// commands affecting to the G4UImanager. Commands defined by
44// this messenger are
45// /control/
46// /control/macroPath
47// /control/execute
48// /control/loop
49// /control/foreach
50// /control/suppressAbortion
51// /control/verbose
52// /control/useDoublePrecision
53// /control/saveHistory
54// /control/stopSavingHistory
55// /control/alias
56// /control/unalias
57// /control/listAlias
58// /control/getEnv
59// /control/getVal
60// /control/echo
61// /control/shell
62// /control/manual
63// /control/createHTML
64// /control/maximumStoredHistory
65// /control/if
66// /control/doif
67// /control/add
68// /control/subtract
69// /control/multiply
70// /control/divide
71// /control/strif
72// /control/strdoif
73// /control/ifBatch
74// /control/ifInteractive
75// /control/doifBatch
76// /control/doifInteractive
77
78class G4UIcontrolMessenger : public G4UImessenger
79{
80 public:
81 G4UIcontrolMessenger();
82 ~G4UIcontrolMessenger();
83 void SetNewValue(G4UIcommand * command,G4String newValue);
84 G4String GetCurrentValue(G4UIcommand * command);
85
86 private:
87 G4UIdirectory * controlDirectory;
88 G4UIcmdWithAString * macroPathCommand;
89 G4UIcmdWithAString * ExecuteCommand;
90 G4UIcmdWithAnInteger * suppressAbortionCommand;
91 G4UIcmdWithAnInteger * verboseCommand;
92 G4UIcmdWithABool * doublePrecCommand;
93 G4UIcmdWithAString * historyCommand;
94 G4UIcmdWithoutParameter * stopStoreHistoryCommand;
95 G4UIcommand * aliasCommand;
96 G4UIcmdWithAString * unaliasCommand;
97 G4UIcmdWithoutParameter * listAliasCommand;
98 G4UIcmdWithAString * getEnvCmd;
99 G4UIcommand * getValCmd;
100 G4UIcmdWithAString * echoCmd;
101 G4UIcmdWithAString * shellCommand;
102 G4UIcommand * loopCommand;
103 G4UIcommand * foreachCommand;
104 G4UIcmdWithAString * ManualCommand;
105 G4UIcmdWithAString * HTMLCommand;
106 G4UIcmdWithAnInteger * maxStoredHistCommand;
107 G4UIcommand * ifCommand;
108 G4UIcommand * doifCommand;
109 G4UIcommand * addCommand;
110 G4UIcommand * subtractCommand;
111 G4UIcommand * multiplyCommand;
112 G4UIcommand * divideCommand;
113 G4UIcommand * remainderCommand;
114 G4UIcommand * strifCommand;
115 G4UIcommand * strdoifCommand;
116 G4UIcmdWithAString * ifBatchCommand;
117 G4UIcmdWithAString * ifInteractiveCommand;
118 G4UIcmdWithAString * doifBatchCommand;
119 G4UIcmdWithAString * doifInteractiveCommand;
120};

122#endif
