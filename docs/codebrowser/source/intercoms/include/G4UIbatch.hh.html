27// ====================================================================
28// G4UIbatch.hh
29//
30// This is a concrete class of G4UIsession.
31//
32// This class object is instantiated by G4UImanager at every time
33// when "/control/execute macro_file" command is executed.
34// Also in the case of pure batch mode with a macro file,
35// this class can be used as other ordinary G4UIsession
36// concrete classes, i.e. SessionStart() is invoked in main().
37// ====================================================================
38#ifndef G4UI_BATCH_H
39#define G4UI_BATCH_H 1
40
41#include "G4UIsession.hh"
42#include <fstream>
43
44// ====================================================================
45//
46// class definition
47//
48// ====================================================================
49
50class G4UIbatch : public G4UIsession {
51private:
52 G4UIsession* previousSession;
53
54 std::ifstream macroStream;
55 G4bool isOpened;
56
57 //static G4bool commandFailed;
58
59 // get command from a batch script file
60 G4String ReadCommand();
61 G4int ExecCommand(const G4String& command);
62
63public:
64 G4UIbatch(const char* fileName, G4UIsession* prevSession=0);
65 // "prevSession" must be 0 if this class is constructed
66 // from main().
67
68 ~G4UIbatch();
69
70 G4UIsession* GetPreviousSession() const;
71
72 virtual G4UIsession* SessionStart();
73 virtual void PauseSessionStart(const G4String& Prompt);
74
75};
76
77// ============================================================================
78// inlines
79
80inline G4UIsession* G4UIbatch::GetPreviousSession() const
81{
82 return previousSession;
83}
84
85#endif
