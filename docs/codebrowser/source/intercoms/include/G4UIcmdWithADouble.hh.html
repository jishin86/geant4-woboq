30#ifndef G4UIcmdWithADouble_H
31#define G4UIcmdWithADouble_H 1
32
33#include "G4UIcommand.hh"
34
35// class description:
36// A concrete class of G4UIcommand. The command defined by this class
37// takes a double value.
38// General information of G4UIcommand is given in G4UIcommand.hh.
39
40class G4UIcmdWithADouble : public G4UIcommand
41{
42 public: // with description
43 G4UIcmdWithADouble
44 (const char * theCommandPath,G4UImessenger * theMessenger);
45 // Constructor. The command string with full path directory
46 // and the pointer to the messenger must be given.
47 static G4double GetNewDoubleValue(const char* paramString);
48 // Convert string which represents a double value to a double.
49 void SetParameterName(const char * theName,G4bool omittable,
50 G4bool currentAsDefault=false);
51 // Set the parameter name. The name is used by the range checking.
52 // If "omittable" is set as true, the user of this command can ommit
53 // the value when he/she applies the command. If "omittable" is false,
54 // the user must supply a double value.
55 // "currentAsDefault" flag is valid only if "omittable" is true. If this
56 // flag is true, the current value is used as the default value when the
57 // user ommit the parameter. If this flag is false, the value given by the
58 // next SetDefaultValue() method is used.
59 void SetDefaultValue(G4double defVal);
60 // Set the default value of the parameter. This default value is used
61 // when the user of this command ommits the parameter value, and
62 // "ommitable" is true and "currentAsDefault" is false.
63};
64
65#endif
