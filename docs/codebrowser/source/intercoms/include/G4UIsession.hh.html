30#ifndef G4UIsession_h
31#define G4UIsession_h 1
32
33#include "G4coutDestination.hh"
34#include "globals.hh"
35#include "icomsdefs.hh"
36
37// class description:
38//
39// This is a base class of all (G)UI session.
40// SessionStart() method should be called to start the session.
41//
42
43class G4UIsession : public G4coutDestination
44{
45 // Base class of UI/GUI session
46
47 public:
48 G4UIsession();
49 G4UIsession(G4int iBatch);
50 virtual ~G4UIsession();
51
52 virtual G4UIsession * SessionStart();
53 // This method will be invoked by main().
54 // Optionally, it can be invoked by another session.
55
56 virtual void PauseSessionStart(const G4String& Prompt);
57 // This method will be invoked by G4UImanager
58 // when G4kernel becomes to Pause state.
59
60 virtual G4int ReceiveG4cout(const G4String& coutString);
61 virtual G4int ReceiveG4cerr(const G4String& cerrString);
62 // These two methods will be invoked by G4strstreambuf.
63
64 protected:
65 G4ICOMS_DLL static G4int inSession;
66 G4int ifBatch;
67 public:
68 static G4int InSession() { return inSession; }
69};
