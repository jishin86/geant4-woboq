31inline G4ErrorMatrix::G4ErrorMatrix()
32 : m(0), nrow(0), ncol(0), size(0)
33{
34}
35
36inline G4int G4ErrorMatrix::num_row() const
37{
38 return nrow;
39}
40
41inline G4int G4ErrorMatrix::num_col() const
42{
43 return ncol;
44}
45
46inline G4int G4ErrorMatrix::num_size() const
47{
48 return size;
49}
50
51inline G4double & G4ErrorMatrix::operator()(G4int row, G4int col)
52{
53 return *(m.begin()+(row-1)*ncol+col-1);
54}
55
56inline const G4double & G4ErrorMatrix::operator()(G4int row, G4int col) const
57{
58 return *(m.begin()+(row-1)*ncol+col-1);
59}
60
61inline G4ErrorMatrix::G4ErrorMatrix_row G4ErrorMatrix::operator[] (G4int r)
62{
63 G4ErrorMatrix_row b(*this,r);
64 return b;
65}
66
67inline const G4ErrorMatrix::G4ErrorMatrix_row_const
68G4ErrorMatrix::operator[] (G4int r) const
69{
70 G4ErrorMatrix_row_const b(*this,r);
71 return b;
72}
73
74inline G4double &G4ErrorMatrix::G4ErrorMatrix_row::operator[](G4int c)
75{
76 return *(_a.m.begin()+_r*_a.ncol+c);
77}
78
79inline const G4double &
80G4ErrorMatrix::G4ErrorMatrix_row_const::operator[](G4int c) const
81{
82 return *(_a.m.begin()+_r*_a.ncol+c);
83}
84
85inline G4ErrorMatrix::
86G4ErrorMatrix_row::G4ErrorMatrix_row(G4ErrorMatrix&a, G4int r)
87 : _a(a)
88{
89 _r = r;
90}
91
92inline G4ErrorMatrix::G4ErrorMatrix_row_const::G4ErrorMatrix_row_const
93(const G4ErrorMatrix&a, G4int r)
94 : _a(a)
95{
96 _r = r;
97}
98
99inline G4ErrorMatrix G4ErrorMatrix::inverse(G4int &ierr) const
100{
101 G4ErrorMatrix mTmp(*this);
102 mTmp.invert(ierr);
103 return mTmp;
104}
