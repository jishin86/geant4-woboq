29// Class Description:
30// Abstract interface class for a graphics scene handler.
31// It is a minimal scene handler for the GEANT4 kernel.
32// See G4VSceneHandler for a fuller description. G4VSceneHandler is
33// the full abstract interface to graphics systems.
34

57class G4PhysicalVolumeModel;

59class G4VHit;

61template <typename T> class G4THitsMap;
62class G4Polyline;
63class G4Scale;
64class G4Text;

72class G4VGraphicsScene {
73
74public: // With description
75
76 G4VGraphicsScene();
77 virtual ~G4VGraphicsScene();
78
79 ///////////////////////////////////////////////////////////////////
80 // Methods for adding solids to the scene handler. They
81 // must always be called in the triplet PreAddSolid, AddSolid and
82 // PostAddSolid. The transformation and visualization attributes
83 // must be set by the call to PreAddSolid. A possible default
84 // implementation is to request the solid to provide a G4Polyhedron
85 // or similar primitive - see, for example, G4VSceneHandler in the
86 // Visualization Category.
87
88 virtual void PreAddSolid (const G4Transform3D& objectTransformation,
89 const G4VisAttributes& visAttribs) = 0;
90 // objectTransformation is the transformation in the world
91 // coordinate system of the object about to be added, and
92 // visAttribs is its visualization attributes.
93
94 virtual void PostAddSolid () = 0;
95
96 // From geometry/solids/CSG
97 virtual void AddSolid (const G4Box&) = 0;
98 virtual void AddSolid (const G4Cons&) = 0;
99 virtual void AddSolid (const G4Orb&) = 0;
100 virtual void AddSolid (const G4Para&) = 0;
101 virtual void AddSolid (const G4Sphere&) = 0;
102 virtual void AddSolid (const G4Torus&) = 0;
103 virtual void AddSolid (const G4Trap&) = 0;
104 virtual void AddSolid (const G4Trd&) = 0;
105 virtual void AddSolid (const G4Tubs&) = 0;
106
107 // From geometry/solids/specific
108 virtual void AddSolid (const G4Ellipsoid&) = 0;
109 virtual void AddSolid (const G4Polycone&) = 0;
110 virtual void AddSolid (const G4Polyhedra&) = 0;
111
112 // For solids not above.
113 virtual void AddSolid (const G4VSolid&) = 0;
114
115 ///////////////////////////////////////////////////////////////////
116 // Methods for adding "compound" GEANT4 objects to the scene
117 // handler. These methods may either (a) invoke "user code" that
118 // uses the "user interface", G4VVisManager (see, for example,
119 // G4VSceneHandler in the Visualization Category, which for
120 // trajectories uses G4VTrajectory::DrawTrajectory, via
121 // G4TrajectoriesModel in the Modeling Category) or (b) invoke
122 // AddPrimitives below (between calls to Begin/EndPrimitives) or (c)
123 // use graphics-system-specific code or (d) any combination of the
124 // above.
125
126 virtual void AddCompound (const G4VTrajectory&) = 0;
127 virtual void AddCompound (const G4VHit&) = 0;
128 virtual void AddCompound (const G4VDigi&) = 0;
129 virtual void AddCompound (const G4THitsMap<G4double>&) = 0;
130 virtual void AddCompound (const G4THitsMap<G4StatDouble>&) = 0;
131
132 ///////////////////////////////////////////////////////////////////
133 // Methods for adding graphics primitives to the scene handler. A
134 // sequence of calls to AddPrimitive must be sandwiched between
135 // calls to BeginPrimitives and EndPrimitives. A sequence is any
136 // number of calls that have the same transformation.
137
138 virtual void BeginPrimitives
139 (const G4Transform3D& objectTransformation = G4Transform3D()) = 0;
140 // objectTransformation is the transformation in the world
141 // coordinate system of the object about to be added.
142
143 virtual void EndPrimitives () = 0;
144
145 virtual void BeginPrimitives2D
146 (const G4Transform3D& objectTransformation = G4Transform3D()) = 0;
147
148 virtual void EndPrimitives2D () = 0;
149 // The x,y coordinates of the primitives passed to AddPrimitive are
150 // intrepreted as screen coordinates, -1 < x,y < 1. The
151 // z-coordinate is ignored.
152
153 virtual void AddPrimitive (const G4Polyline&) = 0;
154 virtual void AddPrimitive (const G4Scale&) = 0;
155 virtual void AddPrimitive (const G4Text&) = 0;
156 virtual void AddPrimitive (const G4Circle&) = 0;
157 virtual void AddPrimitive (const G4Square&) = 0;
158 virtual void AddPrimitive (const G4Polymarker&) = 0;
159 virtual void AddPrimitive (const G4Polyhedron&) = 0;
160
161 virtual const G4VisExtent& GetExtent() const;
162 // The concrete class should overload this or a null extent will be returned.
163 // See G4VScenHandler for example.
164
165};
166
167#endif
