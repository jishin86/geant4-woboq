30// G4Polyhedron placed in the real world.

35#ifndef G4PLACEDPOLYHEDRON_HH
36#define G4PLACEDPOLYHEDRON_HH
37
38#include "G4Polyhedron.hh"
39#include "G4Transform3D.hh"
40#include <vector>
41
42class G4PlacedPolyhedron {
43
44public: // With description
45
46 G4PlacedPolyhedron ();
47 G4PlacedPolyhedron (const G4Polyhedron&, const G4Transform3D&);
48
49 // Uses default copy constructor, destructor and assignment.
50
51 G4bool operator == (const G4PlacedPolyhedron& right) const {
52 return this == &right;
53 }
54
55 const G4Polyhedron& GetPolyhedron () const {return fPolyhedron;}
56 const G4Transform3D& GetTransform () const {return fTransform;}
57
58 void SetPolyhedron (const G4Polyhedron& polyhedron) {
59 fPolyhedron = polyhedron;
60 }
61 void SetTransform (const G4Transform3D& transform) {
62 fTransform = transform;
63 }
64
65private:
66
67 G4Polyhedron fPolyhedron;
68 G4Transform3D fTransform;
69
70};
71
72typedef std::vector<G4PlacedPolyhedron> G4PlacedPolyhedronList;
