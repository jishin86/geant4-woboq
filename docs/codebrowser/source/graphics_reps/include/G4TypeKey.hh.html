31#ifndef G4TYPEKEY_HH
32#define G4TYPEKEY_HH
33
34#include "globals.hh"
35#include <ostream>
36
37class G4TypeKey {
38
39public:
40
41 typedef unsigned long Key;
42
43 // Constructor
44 G4TypeKey():fMyKey(0) {}
45
46 // Destructor
47 virtual ~G4TypeKey() {}
48
49 G4bool IsValid() {
50 return (0 == fMyKey ? false : true);
51 }
52
53 // Operators
54 Key operator()() const {return fMyKey;}
55 G4bool operator==(const G4TypeKey& rhs) const {return fMyKey == rhs.fMyKey;}
56 G4bool operator!=(const G4TypeKey& rhs) const {return !operator==(rhs);}
57 G4bool operator<(const G4TypeKey& rhs) const {return fMyKey < rhs.fMyKey;}
58 G4bool operator>(const G4TypeKey& rhs) const {return fMyKey > rhs.fMyKey;}
59
60 friend std::ostream& operator<<(std::ostream& out, const G4TypeKey& key){
61 return out<< key.fMyKey;
62 }
63
64protected:
65
66 Key NextKey() const {
67 static G4ThreadLocal Key *npKey = 0 ;
68 if (!npKey) { npKey = new Key; *npKey = 0; }
69 Key &nKey = *npKey;
70 return ++nKey;
71 }
72
73 Key fMyKey;
74
75};
76
77#endif
