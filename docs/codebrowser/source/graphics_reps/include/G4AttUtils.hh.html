46 // Utility class
47 template <typename T>
48 class HasName {
49 public:
50 HasName(const G4String& name): fName(name) {};
51 bool operator()(const T& value) const
52 {return value.GetName() == fName;}
53 private:
54 G4String fName;
55 };
56
57 }
58
59 // Extract attribute definition with given name
60 template <typename T>
61 G4bool ExtractAttDef(const T& object, const G4String& name, G4AttDef& def)
62 {
63 const std::map<G4String, G4AttDef>* attDefs = object.GetAttDefs();
64
65 std::map<G4String, G4AttDef>::const_iterator iter = attDefs->find(name);
66 if (iter == attDefs->end()) return false;
67
68 def = iter->second;
69
70 return true;
71 }
72
73 // Extract attribute value with given name
74 template <typename T>
75 G4bool ExtractAttValue(const T& object, const G4String& name, G4AttValue& attVal)
76 {
77 std::vector<G4AttValue>* attValues = object.CreateAttValues();
78
79 std::vector<G4AttValue>::iterator iter = std::find_if(attValues->begin(), attValues->end(),
80 HasName<G4AttValue>(name));
81 if (iter == attValues->end()) return false;
82
83 attVal = *iter;
84
85 // Clean up
86 delete attValues;
87
88 return true;
89 }
