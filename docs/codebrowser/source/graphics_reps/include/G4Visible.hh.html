31// Class Description:
32// Base class for all things visible, i.e., which have Vis Attributes.
33//
34// Note: a null pointer implies no attributes. Under those circumstances
35// the visualization system is free to choose some.
36// Class Description - End:
37
38
39#ifndef G4VISIBLE_HH
40#define G4VISIBLE_HH
41
42#include "globals.hh"
43#include <iostream>
44
45class G4VisAttributes;
46
47class G4Visible {
48
49 friend std::ostream& operator << (std::ostream& os, const G4Visible& v);
50
51public: // With description
52
53 G4Visible ();
54 G4Visible (const G4Visible&);
55 G4Visible (const G4VisAttributes*);
56
57 virtual ~G4Visible ();
58
59 G4Visible& operator= (const G4Visible&);
60
61 G4bool operator != (const G4Visible& right) const;
62
63 const G4VisAttributes* GetVisAttributes () const;
64
65 void SetVisAttributes (const G4VisAttributes*);
66 // The G4VisAttributes object is not stored in a G4Visible; only a
67 // reference, a const pointer, is kept. Therefore the
68 // G4VisAttributes object to which it refers must have a life long
69 // enough to satisfy all uses of the G4Visible object. E.g., if the
70 // G4Visible object is created on the heap (using `new') then the
71 // associated G4VisAttributes object would normally also be created
72 // on the heap and managed in the same way.
73
74 void SetVisAttributes (const G4VisAttributes&);
75 // A copy of the G4VisAttributes object is created on the heap to
76 // ensure a long life.
77
78protected:
79
80 const G4VisAttributes* fpVisAttributes;
81 G4bool fAllocatedVisAttributes;
82};

86#endif
