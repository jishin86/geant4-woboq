31#ifndef G4SCALE_HH
32#define G4SCALE_HH
33
34#include "G4VMarker.hh"
35#include "globals.hh"
36
37class G4Scale: public G4VMarker {
38
39public: // With description
40
41 enum Direction {x, y, z};
42
43 G4Scale (G4double length, const G4String& annotation = "",
44 Direction direction = x,
45 G4bool autoPlacing = true,
46 G4double xmid = 0., G4double ymid = 0., G4double zmid = 0.,
47 G4double annotationSize = 12.);
48 // This creates a representation of annotated line in the specified
49 // direction with tick marks at the end. If autoPlacing is true it
50 // is required to be centred at the front, right, bottom corner of
51 // the world space, comfortably outside the existing bounding
52 // box/sphere so that existing objects do not obscure it. Otherwise
53 // it is required to be drawn with mid-point at (xmid, ymid, zmid).
54 // Annotation size is size of text labels in pixels.
55 //
56 // The auto placing algorithm might be:
57 // x = xmin + (1 + comfort) * (xmax - xmin)
58 // y = ymin - comfort * (ymax - ymin)
59 // z = zmin + (1 + comfort) * (zmax - zmin)
60 // if direction == x then (x - length,y,z) to (x,y,z)
61 // if direction == y then (x,y,z) to (x,y + length,z)
62 // if direction == z then (x,y,z - length) to (x,y,z)
63
64 // Uses compiler-generated copy constructor.
65
66 ~G4Scale ();
67
68 G4double GetLength() const;
69 const G4String& GetAnnotation() const;
70 G4double GetAnnotationSize() const;
71 Direction GetDirection() const;
72 G4bool GetAutoPlacing() const;
73 G4double GetXmid() const;
74 G4double GetYmid() const;
75 G4double GetZmid() const;
76
77 static const G4String& GetGuidanceString();
78
79private:
80
81 G4double fLength;
82 G4String fAnnotation;
83 G4double fAnnotationSize;
84 Direction fDirection;
85 G4bool fAutoPlacing;
86 G4double fXmid, fYmid, fZmid;
87 static const G4String GuidanceString;
88};
