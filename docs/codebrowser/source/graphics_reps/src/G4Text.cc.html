31#include "G4Text.hh"
32
33G4Text::G4Text (const G4String& text):
34fText (text),
35fLayout (left),
36fXOffset(0.) , fYOffset(0.)
37{}
38
39G4Text::G4Text (const G4String& text, const G4Point3D& pos):
40G4VMarker (pos),
41fText (text),
42fLayout (left),
43fXOffset(0.) , fYOffset(0.)
44{}
45
46G4Text::G4Text (const G4VMarker& marker):
47G4VMarker (marker),
48fText ("") ,
49fLayout (left) ,
50fXOffset(0.) , fYOffset(0.)
51{}
52
53G4Text::G4Text (const G4Text& text):
54G4VMarker (text),
55fText (text.fText),
56fLayout (text.fLayout),
57fXOffset (text.fXOffset),
58fYOffset (text.fYOffset)
59{}
60
61G4Text::~G4Text () {}
62
63G4Text& G4Text::operator= (const G4Text& rhs)
64{
65 if (&rhs == this) return *this;
66 G4VMarker::operator=(rhs);
67 fText = rhs.fText;
68 fLayout = rhs.fLayout;
69 fXOffset = rhs.fXOffset;
70 fYOffset = rhs.fYOffset;
71 return *this;
72}
73
74std::ostream& operator<< (std::ostream& os, const G4Text& text)
75{
76 os << "G4Text: \"" << text.GetText()
77 << "\"\n layout " << text.GetLayout()
78 << ", offset (" << text.GetXOffset() << ',' << text.GetYOffset() << ")\n"
79 << (const G4VMarker&)text;
80 return os;
81}
82
83std::ostream& operator<< (std::ostream& os, G4Text::Layout layout)
84{
85 if (layout == G4Text::left) os << "left";
86 if (layout == G4Text::centre) os << "centre";
87 if (layout == G4Text::right) os << "right";
88 return os;
89}
