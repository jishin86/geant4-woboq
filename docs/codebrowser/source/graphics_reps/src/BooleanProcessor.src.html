373inline void ExtFace::invert()
374/***********************************************************************
375 * *
376 * Name: ExtFace::invert() Date: 28.02.00 *
377 * Author: E.Chernyaev Revised: *
378 * *
379 * Function: Invert face *
380 * *
381 ***********************************************************************/
382{
383 int iEprev, iEcur, iEnext;
384
385 iEprev = 0; iEcur = iold;
386 while (iEcur > 0) {
387 ExtEdge& edge = edges[iEcur]; //G.Barrand : optimize.
388 edge.invert();
389 iEnext = edge.inext;
390 edge.inext = iEprev;
391 iEprev = iEcur;
392 iEcur = iEnext;
393 }
394 if (iold > 0) iold = iEprev;
395
396 iEprev = 0; iEcur = inew;
397 while (iEcur > 0) {
398 ExtEdge& edge = edges[iEcur]; //G.Barrand : optimize.
399 edge.invert();
400 iEnext = edge.inext;
401 edge.inext = iEprev;
402 iEprev = iEcur;
403 iEcur = iEnext;
404 }
405 if (inew > 0) inew = iEprev;
406
407#ifdef BP_GEANT4 //G.Barrand
408 plane = HVPlane3D(-plane.a(), -plane.b(), -plane.c(), -plane.d());

414void BooleanProcessor::takePolyhedron(const HepPolyhedron & p,
415 double dx, double dy, double dz)
416/***********************************************************************
417 * *
418 * Name: BooleanProcessor::takePolyhedron Date: 16.12.99 *
419 * Author: E.Chernyaev Revised: *
420 * *
421 * Function: Transfer Polyhedron to internal representation *
422 * *
423 ***********************************************************************/
424{
425 int i = 0, k = 0, nnode = 0, iNodes[5] = {0,0,0,0,0}, iVis[4] = {0,0,0,0},
426 iFaces[4] = {0,0,0,0};
427 int dnode = nodes.size() - 1;
428 int dface = faces.size() - 1;
429
430 // S E T N O D E S
431
432 // for (i=1; i <= p.GetNoVertices(); i++) {
433 // nodes.push_back(ExtNode(p.GetVertex(i)));
434 // }
435
436 HVPoint3D ppp;
437 for (i=1; i <= p.GetNoVertices(); i++) {
438#ifdef BP_GEANT4 //G.Barrand
439 ppp = p.GetVertex(i);
440 ppp.setX(ppp.x()+dx);
441 ppp.setY(ppp.y()+dy);
442 ppp.setZ(ppp.z()+dz);
443#else
444 ppp = p.GetVertexFast(i);
445 ppp += HVPoint3D(dx,dy,dz);
446#endif
447 nodes.push_back(ExtNode(ppp));
448 }
449
450 // S E T F A C E S
451
452 for (int iface=1; iface <= p.GetNoFacets(); iface++) {
453 faces.push_back(ExtFace(edges,edges.size()));
454
455 // S E T F A C E N O D E S
456
457 p.GetFacet(iface, nnode, iNodes, iVis, iFaces);
458 for (i=0; i<nnode; i++) {
459 //if (iNodes[i] < 1 || iNodes[i] > p.GetNoVertices()) processor_error = 1;
460 //if (iFaces[i] < 1 || iFaces[i] > p.GetNoFacets()) processor_error = 1;
461
462 if (iNodes[i] < 1 || iNodes[i] > p.GetNoVertices()) { //G.Barrand
463 processor_error = 1;
464#ifdef BP_DEBUG
465 G4cerr
466 << "BooleanProcessor::takePolyhedron : problem 1."
467 << G4endl;
468#endif
469 }
470 if (iFaces[i] < 1 || iFaces[i] > p.GetNoFacets()) { //G.Barrand
471 processor_error = 1;
472#ifdef BP_DEBUG
473 G4cerr
474 << "BooleanProcessor::takePolyhedron : problem 2."
475 << G4endl;
476#endif
477 }
478 iNodes[i] += dnode;
479 iFaces[i] += dface;
480 }
481
482 // S E T E D G E S
483
484 iNodes[nnode] = iNodes[0];
485 faces.back().iedges[3] = 0;
486 for (i=0; i<nnode; i++) {
487 faces.back().iedges[i] = edges.size();
488 edges.push_back(ExtEdge(iNodes[i], iNodes[i+1],
489 iface+dface, iFaces[i], iVis[i]));
490 edges.back().inext = edges.size();
491 }
492 edges.back().inext = 0;
493
494 // S E T F A C E M I N - M A X
495
496 ExtFace& face = faces.back(); //G.Barrand : optimize.
497 for (i=0; i<3; i++) {
498 face.rmin[i] = nodes[iNodes[0]].v[i];
499 face.rmax[i] = nodes[iNodes[0]].v[i];
500 }
501 for (i=1; i<nnode; i++) {
502 ExtNode& node = nodes[iNodes[i]]; //G.Barrand : optimize.
503 for (k=0; k<3; k++) {
504 if (face.rmin[k] > node.v[k])
505 face.rmin[k] = node.v[k];
506 if (face.rmax[k] < node.v[k])
507 face.rmax[k] = node.v[k];
508 }
509 }
510
511 // S E T F A C E P L A N E
512
513 HVNormal3D n = (nodes[iNodes[2]].v-nodes[iNodes[0]].v).cross
514 (nodes[iNodes[3]].v-nodes[iNodes[1]].v);
515 HVPoint3D point(0,0,0);
516
517 for (i=0; i<nnode; i++) { point += nodes[iNodes[i]].v; }
518 if (nnode > 1) point *= (1./nnode);
519 //G.Barrand : faces.back().plane = HVPlane3D(n.unit(), point);
520 faces.back().plane = HVPlane3D(n, point); //G.Barrand
521
522 // S E T R E F E R E N C E T O T H E N E X T F A C E
523
524 faces.back().inext = faces.size();
525 }
526 faces.back().inext = 0;
527}

529double BooleanProcessor::findMinMax()
530/***********************************************************************
531 * *
532 * Name: BooleanProcessor::findMinMax Date: 16.12.99 *
533 * Author: E.Chernyaev Revised: *
534 * *
535 * Function: Find min-max (bounding) boxes for polyhedra *
536 * *
537 ***********************************************************************/
538{
539 if (ifaces1 == 0 || ifaces2 == 0) return 0;
540
541 int i, iface;
542 double rmin1[3], rmax1[3];
543 double rmin2[3], rmax2[3];
544
545 // F I N D B O U N D I N G B O X E S
546
547 for (i=0; i<3; i++) {
548 rmin1[i] = faces[ifaces1].rmin[i];
549 rmax1[i] = faces[ifaces1].rmax[i];
550 rmin2[i] = faces[ifaces2].rmin[i];
551 rmax2[i] = faces[ifaces2].rmax[i];
552 }
553
554 iface = faces[ifaces1].inext;
555 while(iface > 0) {
556 ExtFace& face = faces[iface]; //G.Barrand
557 for (i=0; i<3; i++) {
558 if (rmin1[i] > face.rmin[i]) rmin1[i] = face.rmin[i];
559 if (rmax1[i] < face.rmax[i]) rmax1[i] = face.rmax[i];
560 }
561 iface = face.inext;
562 }
563
564 iface = faces[ifaces2].inext;
565 while(iface > 0) {
566 ExtFace& face = faces[iface]; //G.Barrand
567 for (i=0; i<3; i++) {
568 if (rmin2[i] > face.rmin[i]) rmin2[i] = face.rmin[i];
569 if (rmax2[i] < face.rmax[i]) rmax2[i] = face.rmax[i];
570 }
571 iface = face.inext;
572 }
573
574 // F I N D I N T E R S E C T I O N O F B O U N D I N G B O X E S
575
576 for (i=0; i<3; i++) {
577 rmin[i] = (rmin1[i] > rmin2[i]) ? rmin1[i] : rmin2[i];
578 rmax[i] = (rmax1[i] < rmax2[i]) ? rmax1[i] : rmax2[i];
579 }
580
581 // F I N D T O L E R A N C E
582
583 double del1 = 0;
584 double del2 = 0;
585 for (i=0; i<3; i++) {
586 if ((rmax1[i]-rmin1[i]) > del1) del1 = rmax1[i]-rmin1[i];
587 if ((rmax2[i]-rmin2[i]) > del2) del2 = rmax2[i]-rmin2[i];
588 }
589 return ((del1 < del2) ? del1 : del2) / GRANULARITY;
590}

592void BooleanProcessor::selectOutsideFaces(int & ifaces, int & iout)
593/***********************************************************************
594 * *
595 * Name: BooleanProcessor::selectOutsideFaces Date: 10.01.00 *
596 * Author: E.Chernyaev Revised: *
597 * *
598 * Function: Preselection of outside faces *
599 * *
600 ***********************************************************************/
601{
602 int i, outflag, iface = ifaces, *prev;
603 HVPoint3D mmbox[8] = { HVPoint3D(rmin[0],rmin[1],rmin[2]),
604 HVPoint3D(rmax[0],rmin[1],rmin[2]),
605 HVPoint3D(rmin[0],rmax[1],rmin[2]),
606 HVPoint3D(rmax[0],rmax[1],rmin[2]),
607 HVPoint3D(rmin[0],rmin[1],rmax[2]),
608 HVPoint3D(rmax[0],rmin[1],rmax[2]),
609 HVPoint3D(rmin[0],rmax[1],rmax[2]),
610 HVPoint3D(rmax[0],rmax[1],rmax[2]) };
611 prev = &ifaces;
612 while (iface > 0) {
613
614 // B O U N D I N G B O X vs B O U N D I N G B O X
615
616 outflag = 0;
617 ExtFace& face = faces[iface]; //G.Barrand : optimize.
618 for (i=0; i<3; i++) {
619 if (face.rmin[i] > rmax[i] + del) { outflag = 1; break; }
620 if (face.rmax[i] < rmin[i] - del) { outflag = 1; break; }
621 }
622
623 // B O U N D I N G B O X vs P L A N E
624
625 if (outflag == 0) {
626 int npos = 0, nneg = 0;
627 double d;
628 for (i=0; i<8; i++) {
629 d = face.plane.distance(mmbox[i]); //G.Barrand : optimize
630 if (d > +del) npos++;
631 if (d < -del) nneg++;
632 }
633 if (npos == 8 || nneg == 8) outflag = 1;
634 }
635
636 // U P D A T E L I S T S
637
638 if (outflag == 1) {
639 *prev = face.inext;
640 face.inext = iout;
641 iout = iface;
642 }else{
643 prev = &face.inext;
644 }
645
646 iface = *prev;
647 }
648}

659{
660 int iface = edge.iface1;
661 HVPlane3D plane = faces[edge.iface2].plane;
662 int i, nnode, npos = 0, nneg = 0, nzer = 0;
663 double dd[5];
664
665 // F I N D D I S T A N C E S
666
667 nnode = (faces[iface].iedges[3] == 0) ? 3 : 4;
668 for (i=0; i<nnode; i++) {
669 dd[i] = plane.distance(nodes[edges[faces[iface].iedges[i]].i1].v);
670 if (dd[i] > del) {
671 npos++;
672 }else if (dd[i] < -del) {
673 nneg++;
674 }else{
675 nzer++; dd[i] = 0;
676 }
677 }
678
679 // S O M E S I M P L E C A S E S ( N O I N T E R S E C T I O N )
680
681 if (npos == nnode || nneg == nnode) return OUT_OF_PLANE;
682 if (nzer == 1 && nneg == 0) return OUT_OF_PLANE;
683 if (nzer == 1 && npos == 0) return OUT_OF_PLANE;
684 if (nzer == nnode) return ON_PLANE;
685 if (nzer == 3) return NON_PLANAR_FACE;
686
687 // F I N D I N T E R S E C T I O N
688
689 int ie1 = 0, ie2 = 0, s1 = 0, s2 = 0, status, nint = 0;
690 enum { PLUS_MINUS, MINUS_PLUS, ZERO_ZERO, ZERO_PLUS, ZERO_MINUS };
691
692 dd[nnode] = dd[0];
693 for (i=0; i<nnode; i++) {
694 if (dd[i] > 0) {
695 if (dd[i+1] >= 0) continue;
696 status = PLUS_MINUS;
697 }else if (dd[i] < 0) {
698 if (dd[i+1] <= 0) continue;
699 status = MINUS_PLUS;
700 }else{
701 status = ZERO_ZERO;
702 if (dd[i+1] > 0) status = ZERO_PLUS;
703 if (dd[i+1] < 0) status = ZERO_MINUS;
704 }
705 switch (nint) {
706 case 0:
707 ie1 = i; s1 = status; nint++; break;
708 case 1:
709 ie2 = i; s2 = status; nint++; break;
710 default:
711 return NON_PLANAR_FACE;
712 }
713 }
714 if (nint != 2) return NON_PLANAR_FACE;
715
716 // F O R M I N T E R S E C T I O N S E G M E N T
717
718 if (s1 != ZERO_ZERO && s2 != ZERO_ZERO) {
719 if (s1 == s2) return NON_PLANAR_FACE;
720 int iedge, i1 = 0, i2 = 0, ii[2];
721 double d1 = 0., d2 = 0., d3 = 0.;
722 ii[0] = ie1; ii[1] = ie2;
723 for (i=0; i<2; i++) {
724 iedge = faces[iface].iedges[ii[i]];
725 while (iedge > 0) {
726 i1 = edges[iedge].i1;
727 i2 = edges[iedge].i2;
728
729 d1 = plane.distance(nodes[i1].v);
730 d2 = plane.distance(nodes[i2].v);
731 if (d1 > del) {
732 if (d2 < -del) { ii[i] = nodes.size(); break; } // +-
733 }else if (d1 < -del) {
734 if (d2 > del) { ii[i] = nodes.size(); break; } // -+
735 }else{
736 ii[i] = i1; break; // 0+ or 0-
737 }
738 iedge = edges[iedge].inext;
739 }
740 if (ii[i] == (int)nodes.size()) {
741 d3 = d2-d1; d1 = d1/d3; d2 = d2/d3;
742 nodes.push_back(ExtNode(d2*nodes[i1].v-d1*nodes[i2].v, iedge));
743 }
744 }
745 edge.inext = 0;
746 if (s1 == MINUS_PLUS || s1 == ZERO_PLUS) {
747 edge.i1 = ii[1];
748 edge.i2 = ii[0];
749 }else{
750 edge.i1 = ii[0];
751 edge.i2 = ii[1];
752 }
753 return INTERSECTION;
754 }else{
755 if (npos == nneg) return NON_PLANAR_FACE;
756 edge.inext = (s1 == ZERO_ZERO) ? ie1+1 : ie2+1;
757 if (s1 == ZERO_PLUS || s2 == ZERO_MINUS) {
758 edge.i1 = edges[faces[iface].iedges[ie2]].i1;
759 edge.i2 = edges[faces[iface].iedges[ie1]].i1;
760 }else{
761 edge.i1 = edges[faces[iface].iedges[ie1]].i1;
762 edge.i2 = edges[faces[iface].iedges[ie2]].i1;
763 }
764 return EDGE;
765 }
