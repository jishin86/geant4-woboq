130G4VisAttributes& G4VisAttributes::operator= (const G4VisAttributes& rhs)
131{
132 if (&rhs == this) return *this;
133 fVisible = rhs.fVisible;
134 fDaughtersInvisible = rhs.fDaughtersInvisible;
135 fColour = rhs.fColour;
136 fLineStyle = rhs.fLineStyle;
137 fLineWidth = rhs.fLineWidth;
138 fForceDrawingStyle = rhs.fForceDrawingStyle;
139 fForcedStyle = rhs.fForcedStyle;
140 fForceAuxEdgeVisible = rhs.fForceAuxEdgeVisible;
141 fForcedAuxEdgeVisible = rhs.fForcedAuxEdgeVisible;
142 fForcedLineSegmentsPerCircle = rhs.fForcedLineSegmentsPerCircle;
143 fStartTime = rhs.fStartTime;
144 fEndTime = rhs.fEndTime;
145 // AttValues are created afresh for each object (using the
146 // CreateAttValues message), but deletion is the responsibility of
147 // the creator. So just copy pointer.
148 fAttValues = rhs.fAttValues;
149 // AttDefs, if any, belong to the object from which they were obtained
150 // (with a GetAttDefs message), so just copy pointer.
151 fAttDefs = rhs.fAttDefs;
152 return *this;
153}

204void G4VisAttributes::SetForceLineSegmentsPerCircle (G4int nSegments) {
205 const G4int nSegmentsMin = fMinLineSegmentsPerCircle;
206 if (nSegments > 0 && nSegments < nSegmentsMin) {
207 nSegments = nSegmentsMin;
208 G4cout <<
209 "G4VisAttributes::SetForcedLineSegmentsPerCircle: attempt to set the"
210 "\nnumber of line segements per circle < " << nSegmentsMin
211 << "; forced to " << nSegments << G4endl;
212 }
213 fForcedLineSegmentsPerCircle = nSegments;
214}

216std::ostream& operator << (std::ostream& os, const G4VisAttributes& a)
217{
218 os << "G4VisAttributes: ";
219 if (!a.fVisible) os << "in";
220 os << "visible, daughters ";
221 if (a.fDaughtersInvisible) os << "in";
222 os << "visible, colour: " << a.fColour;
223 os << "\n linestyle: ";
224 switch (a.fLineStyle) {
225 case G4VisAttributes::unbroken:
226 os << "solid"; break;
227 case G4VisAttributes::dashed:
228 os << "dashed"; break;
229 case G4VisAttributes::dotted: os << "dotted"; break;
230 default: os << "unrecognised"; break;
231 }
232 os << ", line width: " << a.fLineWidth;
233 os << ", min line segments per circle: " << a.GetMinLineSegmentsPerCircle();
234 os << "\n drawing style: ";
235 if (a.fForceDrawingStyle) {
236 os << "forced to ";
237 switch (a.fForcedStyle) {
238 case G4VisAttributes::wireframe:
239 os << "wireframe"; break;
240 case G4VisAttributes::solid:
241 os << "solid"; break;
242 default: os << "unrecognised"; break;
243 }
244 }
245 else {
246 os << "not forced";
247 }
248 os << ", auxiliary edge visibility: ";
249 if (a.fForceAuxEdgeVisible) {
250 os << "forced to ";
251 if (!a.fForcedAuxEdgeVisible) {
252 os << "not ";
253 }
254 os << "visible";
255 } else {
256 os << "not forced";
257 }
258 os << "\n line segments per circle: ";
259 if (a.fForcedLineSegmentsPerCircle > 0) {
260 os << "forced to " << a.fForcedLineSegmentsPerCircle;
261 } else {
262 os << "not forced.";
263 }
264 os << "\n time range: (" << a.fStartTime << ',' << a.fEndTime << ')';
265 os << "\n G4AttValue pointer is ";
266 if (a.fAttValues) {
267 os << "non-";
268 }
269 os << "zero";
270 os << ", G4AttDef pointer is ";
271 if (a.fAttDefs) {
272 os << "non-";
273 }
274 os << "zero";
275 return os;
276}

278G4bool G4VisAttributes::operator != (const G4VisAttributes& a) const {
279
280 if (
281 (fVisible != a.fVisible) ||
282 (fDaughtersInvisible != a.fDaughtersInvisible) ||
283 (fColour != a.fColour) ||
284 (fLineStyle != a.fLineStyle) ||
285 (fLineWidth != a.fLineWidth) ||
286 (fForceDrawingStyle != a.fForceDrawingStyle) ||
287 (fForceAuxEdgeVisible!= a.fForceAuxEdgeVisible) ||
288 (fForcedLineSegmentsPerCircle != a.fForcedLineSegmentsPerCircle) ||
289 (fStartTime != a.fStartTime) ||
290 (fEndTime != a.fEndTime) ||
291 (fAttValues != a.fAttValues) ||
292 (fAttDefs != a.fAttDefs)
293 )
294 return true;
295
296 if (fForceDrawingStyle) {
297 if (fForcedStyle != a.fForcedStyle) return true;
298 }
299
300 if (fForceAuxEdgeVisible) {
301 if (fForcedAuxEdgeVisible != a.fForcedAuxEdgeVisible) return true;
302 }
303
304 return false;
305}
