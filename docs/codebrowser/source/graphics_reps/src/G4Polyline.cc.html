37G4Polyline& G4Polyline::transform (const G4Transform3D& transformation) {
38 for (iterator i = begin(); i != end(); ++i) i->transform(transformation);
39 return *this;
40}

43 os << "G4Polyline: ";

45 os << '\n' << (const G4Point3DList&) line;
46 return os;
47}
