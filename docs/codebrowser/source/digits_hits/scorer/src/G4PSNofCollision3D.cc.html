39G4PSNofCollision3D::G4PSNofCollision3D(G4String name,
40 G4int ni, G4int nj, G4int nk,
41 G4int depi, G4int depj, G4int depk)
42 :G4PSNofCollision(name),
43 fDepthi(depi),fDepthj(depj),fDepthk(depk)
44{
45 fNi=ni;
46 fNj=nj;
47 fNk=nk;
48}
49
50G4PSNofCollision3D::~G4PSNofCollision3D()
51{;}
52
53G4int G4PSNofCollision3D::GetIndex(G4Step* aStep)
54{
55 const G4VTouchable* touchable = aStep->GetPreStepPoint()->GetTouchable();
56 G4int i = touchable->GetReplicaNumber(fDepthi);
57 G4int j = touchable->GetReplicaNumber(fDepthj);
58 G4int k = touchable->GetReplicaNumber(fDepthk);
59
60 return i*fNj*fNk+j*fNk+k;
61}
