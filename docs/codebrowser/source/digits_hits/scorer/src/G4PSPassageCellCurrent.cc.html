46G4PSPassageCellCurrent::G4PSPassageCellCurrent(G4String name, G4int depth)

56G4bool G4PSPassageCellCurrent::ProcessHits(G4Step* aStep,G4TouchableHistory*)
57{
58
59 if ( IsPassed(aStep) ) {
60 if(weighted) fCurrent = aStep->GetPreStepPoint()->GetWeight();
61 G4int index = GetIndex(aStep);
62 EvtMap->add(index,fCurrent);
63 }
64
65 return TRUE;
66}
67
68G4bool G4PSPassageCellCurrent::IsPassed(G4Step* aStep){
69 G4bool Passed = FALSE;
70
71 G4bool IsEnter = aStep->GetPreStepPoint()->GetStepStatus() == fGeomBoundary;
72 G4bool IsExit = aStep->GetPostStepPoint()->GetStepStatus() == fGeomBoundary;
73
74 G4int trkid = aStep->GetTrack()->GetTrackID();
75
76 if ( IsEnter &&IsExit ){ // Passed at one step
77 Passed = TRUE;
78 }else if ( IsEnter ){ // Enter a new geometry
79 fCurrentTrkID = trkid; // Resetting the current track.
80 }else if ( IsExit ){ // Exit a current geometry
81 if ( fCurrentTrkID == trkid ) {
82 Passed = TRUE; // if the track is same as entered.
83 }
84 }else{ // Inside geometry
85 if ( fCurrentTrkID == trkid ){ // Adding the track length to current one ,
86 }
87 }
88 return Passed;
89}

91void G4PSPassageCellCurrent::Initialize(G4HCofThisEvent* HCE)
92{
93 fCurrentTrkID = -1;
94
95 EvtMap = new G4THitsMap<G4double>(detector->GetName(),
96 GetName());
97 if ( HCID < 0 ) HCID = GetCollectionID(0);
98 HCE->AddHitsCollection(HCID,EvtMap);
99
100}

114{
