62void G4PSDoseDepositForCylinder3D::SetCylinderSize(G4double dr, G4double dz){
63 cylinderSize.setX(dz); //Z Phi R
64 cylinderSize.setY(twopi); //Z Phi R
65 cylinderSize.setZ(dr); //Z Phi R
66}
67void G4PSDoseDepositForCylinder3D::SetNumberOfSegments(G4int nSeg[3]){
68 nSegment[0] = nSeg[0]; // Z
69 nSegment[1] = nSeg[1]; // Phi
70 nSegment[2] = nSeg[2]; // R
71}
72G4double G4PSDoseDepositForCylinder3D::ComputeVolume(G4Step*, G4int idx){
73 G4double r0 = (cylinderSize[2]/nSegment[2])*(idx);
74 G4double r1 = (cylinderSize[2]/nSegment[2])*(idx+1);
75 G4double dRArea = (r1*r1-r0*r0)*pi;
76
77 // cylinderSize is given in Half Size
78 G4double fullz = cylinderSize[0]/nSegment[0]*2.;
79 G4double phiRatio = 1./nSegment[1];
80 G4double v = dRArea*fullz*phiRatio;
81
82 return v;
83}
