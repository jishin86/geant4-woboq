34////////////////////////////////////////////////////////////////////////////////
35// class description:
36//
37// This is the class of a filter to be associated with a
38// sensitive detector.
39// This class filters steps by partilce definition and kinetic energy.
40//
41// Created: 2005-11-14 Tsukasa ASO.
42//
43///////////////////////////////////////////////////////////////////////////////
44
45G4SDParticleWithEnergyFilter::G4SDParticleWithEnergyFilter(G4String name,
46 G4double elow,
47 G4double ehigh)
48 :G4VSDFilter(name)
49{
50 fParticleFilter = new G4SDParticleFilter(name);
51 fKineticFilter = new G4SDKineticEnergyFilter(name,elow,ehigh);
52}
53
54G4SDParticleWithEnergyFilter::~G4SDParticleWithEnergyFilter()
55{
56 delete fParticleFilter;
57 delete fKineticFilter;
58}
59
60G4bool G4SDParticleWithEnergyFilter::Accept(const G4Step* aStep) const
61{
62 if ( ! fParticleFilter->Accept(aStep) ) return FALSE;
63 if ( ! fKineticFilter->Accept(aStep) ) return FALSE;
64 return TRUE;
65}
66
67void G4SDParticleWithEnergyFilter::add(const G4String& particleName)
68{
69 fParticleFilter->add(particleName);
70}
71
72void G4SDParticleWithEnergyFilter::SetKineticEnergy(G4double elow,
73 G4double ehigh)
74{
75 fKineticFilter->SetKineticEnergy(elow,ehigh);
76}
77
78void G4SDParticleWithEnergyFilter::show(){
79 fParticleFilter->show();
80 fKineticFilter->show();
81}
82
83G4SDParticleWithEnergyFilter::G4SDParticleWithEnergyFilter(const G4SDParticleWithEnergyFilter& rhs)
84 : G4VSDFilter(rhs.filterName)
85{
86 fParticleFilter = new G4SDParticleFilter(*rhs.fParticleFilter);
87 fKineticFilter = new G4SDKineticEnergyFilter(*rhs.fKineticFilter);
88}
89
90
91G4SDParticleWithEnergyFilter& G4SDParticleWithEnergyFilter::operator=(const G4SDParticleWithEnergyFilter& rhs)
92{
93 if ( this == &rhs ) return *this;
94 G4VSDFilter::operator=(rhs);
95 delete fParticleFilter;
96 fParticleFilter = new G4SDParticleFilter(*(rhs.fParticleFilter));
97 delete fKineticFilter;
98 fKineticFilter = new G4SDKineticEnergyFilter(*(rhs.fKineticFilter));
99 return *this;
100}
