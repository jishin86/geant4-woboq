31///////////////////////////////////////////////////////////////////////////////
32// (Description)
33// This is a primitive scorer class for scoring Surface Current.
34// Current version assumes only for G4Box shape, and the surface
35// is defined at the -Z plane of the box.
36// The current is given in the unit of area.
37// e.g. (Number of tracks)/mm2.
38//
39// Surface is defined at the -Z surface.
40// Direction -Z +Z
41// 0 IN || OUT ->|<- | fCurrent_InOut
42// 1 IN ->| | fCurrent_In
43// 2 OUT |<- | fCurrent_Out
44//
45//
46// Created: 2008-08-14 Tsukasa ASO
47// 2010-07-22 Introduce Unit specification.
48///////////////////////////////////////////////////////////////////////////////
49
50
51G4PSFlatSurfaceCurrent3D::G4PSFlatSurfaceCurrent3D(G4String name, G4int direction,
52 G4int ni, G4int nj, G4int nk,
53 G4int di, G4int dj, G4int dk)
54
55 :G4PSFlatSurfaceCurrent(name,direction),
56 fDepthi(di),fDepthj(dj),fDepthk(dk)
57{
58 fNi=ni;
59 fNj=nj;
60 fNk=nk;
61}
62
63G4PSFlatSurfaceCurrent3D::G4PSFlatSurfaceCurrent3D(G4String name, G4int direction,
64 const G4String& unit,
65 G4int ni, G4int nj, G4int nk,
66 G4int di, G4int dj, G4int dk)
67
68 :G4PSFlatSurfaceCurrent(name,direction),
69 fDepthi(di),fDepthj(dj),fDepthk(dk)
70{
71 fNi=ni;
72 fNj=nj;
73 fNk=nk;
74 SetUnit(unit);
75}
76
77G4PSFlatSurfaceCurrent3D::~G4PSFlatSurfaceCurrent3D()
78{;}
79
80G4int G4PSFlatSurfaceCurrent3D::GetIndex(G4Step* aStep)
81{
82 const G4VTouchable* touchable = aStep->GetPreStepPoint()->GetTouchable();
83 G4int i = touchable->GetReplicaNumber(fDepthi);
84 G4int j = touchable->GetReplicaNumber(fDepthj);
85 G4int k = touchable->GetReplicaNumber(fDepthk);
86
87 return i*fNj*fNk+j*fNk+k;
88}
