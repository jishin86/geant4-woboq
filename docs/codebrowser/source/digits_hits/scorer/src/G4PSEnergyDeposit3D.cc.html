64G4int G4PSEnergyDeposit3D::GetIndex(G4Step* aStep)
65{
66 const G4VTouchable* touchable = aStep->GetPreStepPoint()->GetTouchable();
67 G4int i = touchable->GetReplicaNumber(fDepthi);
68 G4int j = touchable->GetReplicaNumber(fDepthj);
69 G4int k = touchable->GetReplicaNumber(fDepthk);
70
71if(i<0||j<0||k<0)
72{
73 G4ExceptionDescription ED;
74 ED << "GetReplicaNumber is negative" << G4endl
75 << "touchable->GetReplicaNumber(fDepthi) returns i,j,k = "
76 << i << "," << j << "," << k << " for volume "
77 << touchable->GetVolume(fDepthi)->GetName() << ","
78 << touchable->GetVolume(fDepthj)->GetName() << ","
79 << touchable->GetVolume(fDepthk)->GetName() << G4endl;
80 G4Exception("G4PSEnergyDeposit3D::GetIndex","DetPS0006",JustWarning,ED);
81}
82
83 return i*fNj*fNk+j*fNk+k;
84}
