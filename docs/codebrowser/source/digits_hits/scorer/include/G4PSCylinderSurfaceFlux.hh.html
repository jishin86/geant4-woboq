29#ifndef G4PSCylinderSurfaceFlux_h
30#define G4PSCylinderSurfaceFlux_h 1
31
32#include "G4VPrimitiveScorer.hh"
33#include "G4THitsMap.hh"
34#include "G4Tubs.hh"
35#include "G4PSDirectionFlag.hh"
36////////////////////////////////////////////////////////////////////////////////
37// (Description)
38// This is a primitive scorer class for scoring Surface Flux.
39// Current version assumes only for G4Tubs shape, and the surface
40// is defined at the inner plane of the tube.
41// The surface flux is given in the unit of area.
42// e.g. sum of 1/cos(T)/mm2, where T is a incident angle of the
43// track on the surface.
44//
45//
46// Surface is defined at the inner surface of the tubs.
47// Direction R R+dR
48// 0 IN || OUT ->|<- | fFlux_InOut
49// 1 IN ->| | fFlux_In
50// 2 OUT |<- | fFlux_Out
51//
52// Created: 2007-03-29 Tsukasa ASO
53// 2010-07-22 Introduce Unit specification.
54// 2010-07-22 Add weighted and divideByArea options
55//
56///////////////////////////////////////////////////////////////////////////////
57
58class G4PSCylinderSurfaceFlux : public G4VPrimitiveScorer
59{
60 public: // with description
61 G4PSCylinderSurfaceFlux(G4String name,G4int direction, G4int depth=0);
62 G4PSCylinderSurfaceFlux(G4String name,G4int direction,
63 const G4String& unit, G4int depth=0);
64 virtual ~G4PSCylinderSurfaceFlux();
65
66 inline void Weighted(G4bool flg=true) { weighted = flg; }
67 // Multiply track weight
68
69 inline void DivideByArea(G4bool flg=true) { divideByArea = flg; }
70 // Divided by Area.
71
72
73 protected: // with description
74 virtual G4bool ProcessHits(G4Step*,G4TouchableHistory*);
75 G4int IsSelectedSurface(G4Step*,G4Tubs*);
76
77 public:
78 virtual void Initialize(G4HCofThisEvent*);
79 virtual void EndOfEvent(G4HCofThisEvent*);
80 virtual void clear();
81 virtual void DrawAll();
82 virtual void PrintAll();
83
84 virtual void SetUnit(const G4String& unit);
85
86 protected:
87 virtual void DefineUnitAndCategory();
88
89 private:
90 G4int HCID;
91 G4int fDirection;
92 G4THitsMap<G4double>* EvtMap;
93 G4bool weighted;
94 G4bool divideByArea;
95};
96
97#endif
