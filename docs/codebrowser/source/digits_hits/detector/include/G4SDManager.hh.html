29#ifndef G4SDManager_h
30#define G4SDManager_h 1
31
32#include "globals.hh"
33#include "G4SDStructure.hh"
34#include "G4HCtable.hh"
35class G4VHitsCollection;
36class G4VSensitiveDetector;
37class G4HCofThisEvent;
38class G4SDmessenger;
39
40#include "G4VSDFilter.hh"
41#include <vector>
42
43// class description:
44//
45// This is a singleton class which manages the sensitive detectors.
46// The user cannot access to the constructor. The pointer of the
47// only existing object can be got via G4SDManager::GetSDMpointer()
48// static method. The first invokation of this static method makes
49// the singleton object.
50//
51
52class G4SDManager
53{
54 public: // with description
55 static G4SDManager* GetSDMpointer();
56 // Returns the pointer to the singleton object.
57 public:
58 static G4SDManager* GetSDMpointerIfExist();
59
60 protected:
61 G4SDManager();
62
63 public:
64 ~G4SDManager();
65
66 public: // with description
67 void AddNewDetector(G4VSensitiveDetector*aSD);
68 // Registors the user's sensitive detector. This method must be invoked
69 // when the user construct his/her sensitive detector.
70 void Activate(G4String dName, G4bool activeFlag);
71 // Activate/inactivate the registered sensitive detector. For the inactivated
72 // detectors, hits collections will not be stored to the G4HCofThisEvent object.
73 G4int GetCollectionID(G4String colName);
74 G4int GetCollectionID(G4VHitsCollection * aHC);
75 // These two methods return the ID number of the sensitive detector.
76
77 public:
78 G4VSensitiveDetector* FindSensitiveDetector(G4String dName, G4bool warning = true);
79 G4HCofThisEvent* PrepareNewEvent();
80 void TerminateCurrentEvent(G4HCofThisEvent* HCE);
81 void AddNewCollection(G4String SDname,G4String DCname);
82
83
84 private:
85 static G4ThreadLocal G4SDManager * fSDManager;
86 G4SDStructure * treeTop;
87 G4int verboseLevel;
88 G4HCtable* HCtable;
89 G4SDmessenger* theMessenger;
90
91 public:
92 inline void SetVerboseLevel(G4int vl)
93 {
94 verboseLevel = vl;
95 treeTop->SetVerboseLevel(vl);
96 }
97 inline G4SDStructure* GetTreeTop() const
98 { return treeTop; }
99 inline void ListTree() const
100 { treeTop->ListTree(); }
101 inline G4int GetCollectionCapacity() const
102 { return HCtable->entries(); }
103 inline G4HCtable* GetHCtable() const
104 { return HCtable; }
105private:
106 //Disable copy constructor and assignment operator
107 G4SDManager( const G4SDManager& );
108 G4SDManager& operator=(const G4SDManager&);
109
110public:
111 void RegisterSDFilter(G4VSDFilter* filter);
112 void DeRegisterSDFilter(G4VSDFilter* filter);
113private:
114 void DestroyFilters();
115 std::vector<G4VSDFilter*> FilterList;
116};
