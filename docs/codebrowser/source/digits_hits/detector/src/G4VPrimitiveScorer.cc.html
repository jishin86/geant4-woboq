44G4int G4VPrimitiveScorer::GetCollectionID(G4int)
45{
46 if(detector)
47 return G4SDManager::GetSDMpointer()
48 ->GetCollectionID(detector->GetName()+"/"+primitiveName);
49 else
50 return -1;
51}

68G4int G4VPrimitiveScorer::GetIndex(G4Step* aStep)
69{
70 G4StepPoint* preStep = aStep->GetPreStepPoint();
71 G4TouchableHistory* th = (G4TouchableHistory*)(preStep->GetTouchable());
72 return th->GetReplicaNumber(indexDepth);
73}
74
75void G4VPrimitiveScorer::CheckAndSetUnit(const G4String& unit,
76 const G4String& category){
77 if ( G4UnitDefinition::GetCategory(unit) == category){
78 unitName = unit;
79 unitValue = G4UnitDefinition::GetValueOf(unit);
80 } else {
81 G4String msg = "Invalid unit ["+unit+"] (Current unit is [" +GetUnit()+"] ) requested for " + GetName();
82 G4Exception("G4VPrimitiveScorer::CheckAndSetUnit","Det0151",JustWarning,msg);
83 }
