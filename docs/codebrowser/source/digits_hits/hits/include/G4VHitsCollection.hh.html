29#ifndef G4VHitsCollection_h
30#define G4VHitsCollection_h 1
31
32class G4VHit;
33#include "globals.hh"
34
35// class description:
36//
37// This is the base class of hits collection. The user is advised to
38// use G4THitsCollection template class in case his/her collection is
39// transient. While, in case the collection is persistent with ODBMS,
40// the concrete collection class can be directly derived from this
41// class.
42// Geant4 kernel will use this class methods.
43
44class G4VHitsCollection
45{
46 public:
47 G4VHitsCollection();
48 G4VHitsCollection(G4String detName,G4String colNam);
49 virtual ~G4VHitsCollection();
50 G4bool operator==(const G4VHitsCollection &right) const;
51
52 virtual void DrawAllHits();
53 virtual void PrintAllHits();
54
55 protected:
56
57 // Collection name
58 G4String collectionName;
59 G4String SDname;
60 // Collection ID
61 G4int colID;
62
63 public:
64 inline G4String& GetName()
65 { return collectionName; }
66 inline G4String& GetSDname()
67 { return SDname; }
68 inline void SetColID(G4int i)
69 { colID = i; }
70 inline G4int GetColID() const
71 { return colID; }
72
73 public:
74 // GetHit and GetSize are given a default implementation here so
75 // that the template G4THitsCollection can be used, but they
76 // are re-implemented G4THitsCollection.
77 virtual G4VHit* GetHit(size_t) const { return nullptr; }
78 virtual size_t GetSize() const { return 0; };
79
80};
81
82#endif
