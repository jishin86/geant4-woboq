29#include "G4HCofThisEvent.hh"

31G4Allocator<G4HCofThisEvent>*& anHCoTHAllocator_G4MT_TLS_()
32{
33 G4ThreadLocalStatic G4Allocator<G4HCofThisEvent>* _instance = nullptr;
34 return _instance;
35}
36
37G4HCofThisEvent::G4HCofThisEvent()
38{ if (!anHCoTHAllocator_G4MT_TLS_()) anHCoTHAllocator_G4MT_TLS_() = new G4Allocator<G4HCofThisEvent> ;
39 HC = new std::vector<G4VHitsCollection*>;
40}
41
42G4HCofThisEvent::G4HCofThisEvent(G4int cap)
43{ if (!anHCoTHAllocator_G4MT_TLS_()) anHCoTHAllocator_G4MT_TLS_() = new G4Allocator<G4HCofThisEvent> ;
44 HC = new std::vector<G4VHitsCollection*>;
45 for(G4int i=0;i<cap;i++)
46 {
47 HC->push_back((G4VHitsCollection*)0);
48 }
49}
50
51G4HCofThisEvent::~G4HCofThisEvent()
52{ if (!anHCoTHAllocator_G4MT_TLS_()) anHCoTHAllocator_G4MT_TLS_() = new G4Allocator<G4HCofThisEvent> ;
53 //HC->clearAndDestroy();
54 for(size_t i=0;i<HC->size();i++)
55 { delete (*HC)[i]; }
56 HC->clear();
57 delete HC;
58}
59
60void G4HCofThisEvent::AddHitsCollection(G4int HCID,G4VHitsCollection * aHC)
61{ if (!anHCoTHAllocator_G4MT_TLS_()) anHCoTHAllocator_G4MT_TLS_() = new G4Allocator<G4HCofThisEvent> ;
62 if(HCID>=0 && HCID<G4int(HC->size()))
63 {
64 aHC->SetColID(HCID);
65 (*HC)[HCID] = aHC;
66 }
67}
68
69
70G4HCofThisEvent::G4HCofThisEvent(const G4HCofThisEvent& rhs)
71{
72 if ( !anHCoTHAllocator_G4MT_TLS_() ) anHCoTHAllocator_G4MT_TLS_() = new G4Allocator<G4HCofThisEvent>;
73 HC = new std::vector<G4VHitsCollection*>(rhs.HC->size());
74 for ( unsigned int i = 0 ; i<rhs.HC->size() ; ++i)
75 *(HC->at(i)) = *(rhs.HC->at(i));
76}
77
78G4HCofThisEvent& G4HCofThisEvent::operator=(const G4HCofThisEvent& rhs)
79{
80 if ( this == &rhs ) return *this;
81 if ( !anHCoTHAllocator_G4MT_TLS_() ) anHCoTHAllocator_G4MT_TLS_() = new G4Allocator<G4HCofThisEvent>;
82 for ( std::vector<G4VHitsCollection*>::const_iterator it = HC->begin() ;
83 it != HC->end() ; ++it )
84 {
85 delete *it;
86 }
87 HC->resize(rhs.HC->size());
88 for ( unsigned int i = 0 ; i<rhs.HC->size() ; ++i)
89 *(HC->at(i)) = *(rhs.HC->at(i));
90 return *this;
91}
