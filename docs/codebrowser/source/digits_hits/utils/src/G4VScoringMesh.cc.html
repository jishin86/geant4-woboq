45G4VScoringMesh::G4VScoringMesh(const G4String& wName)
46 : fWorldName(wName),fCurrentPS(nullptr),fConstructed(false),fActive(true),fShape(undefinedMesh),
47 fRotationMatrix(nullptr), fMFD(new G4MultiFunctionalDetector(wName)),
48 verboseLevel(0),sizeIsSet(false),nMeshIsSet(false),
49 fDrawUnit(""), fDrawUnitValue(1.), fMeshElementLogical(nullptr),
50 fParallelWorldProcess(nullptr), fGeometryHasBeenDestroyed(false)
51{
52 G4SDManager::GetSDMpointer()->AddNewDetector(fMFD);
53
54 fSize[0] = fSize[1] = fSize[2] = 0.;
55 fNSegment[0] = fNSegment[1] = fNSegment[2] = 1;
56 fDivisionAxisNames[0] = fDivisionAxisNames[1] = fDivisionAxisNames[2] = "";
57}

63void G4VScoringMesh::ResetScore() {
64 if(verboseLevel > 9) G4cout << "G4VScoringMesh::ResetScore() is called." << G4endl;
65 for(auto mp : fMap)
66 {
67 if(verboseLevel > 9) G4cout << "G4VScoringMesh::ResetScore()" << mp.first << G4endl;
68 mp.second->clear();
69 }
70}

72void G4VScoringMesh::SetSize(G4double size[3]) {
73 if ( !sizeIsSet ){
74 for(int i = 0; i < 3; i++) fSize[i] = size[i];
75 sizeIsSet = true;
76 }else{
77 G4String message = " The size of scoring mesh can not be changed.";
78 G4Exception("G4VScoringMesh::SetSize()",
79 "DigiHitsUtilsScoreVScoringMesh000", JustWarning,
80 message);
81 }
82}
83G4ThreeVector G4VScoringMesh::GetSize() const {
84 if(sizeIsSet)
85 return G4ThreeVector(fSize[0], fSize[1], fSize[2]);
86 else
87 return G4ThreeVector(0., 0., 0.);
88}

121void G4VScoringMesh::SetPrimitiveScorer(G4VPrimitiveScorer * prs) {
122
123 if(!ReadyForQuantity())
124 {
125 G4cerr << "ERROR : G4VScoringMesh::SetPrimitiveScorer() : "
126 << prs->GetName()
127 << " does not yet have mesh size or number of bins. Set them first." << G4endl
128 << "This Method is ignored." << G4endl;
129 return;
130 }
131 if(verboseLevel > 0) G4cout << "G4VScoringMesh::SetPrimitiveScorer() : "
132 << prs->GetName() << " is registered."
133 << " 3D size: ("
134 << fNSegment[0] << ", "
135 << fNSegment[1] << ", "
136 << fNSegment[2] << ")" << G4endl;
137
138 prs->SetNijk(fNSegment[0], fNSegment[1], fNSegment[2]);
139 fCurrentPS = prs;
140 fMFD->RegisterPrimitive(prs);
141 G4THitsMap<G4StatDouble> * map = new G4THitsMap<G4StatDouble>(fWorldName, prs->GetName());
142 fMap[prs->GetName()] = map;
143}

145void G4VScoringMesh::SetFilter(G4VSDFilter * filter) {
146
147 if(!fCurrentPS) {
148 G4cerr << "ERROR : G4VScoringMesh::SetSDFilter() : a quantity must be defined first. This method is ignored." << G4endl;
149 return;
150 }
151 if(verboseLevel > 0) G4cout << "G4VScoringMesh::SetFilter() : "
152 << filter->GetName()
153 << " is set to "
154 << fCurrentPS->GetName() << G4endl;
155
156 G4VSDFilter* oldFilter = fCurrentPS->GetFilter();
157 if(oldFilter)
158 {
159 G4cout << "WARNING : G4VScoringMesh::SetFilter() : " << oldFilter->GetName()
160 << " is overwritten by " << filter->GetName() << G4endl;
161 }
162 fCurrentPS->SetFilter(filter);
163}

223G4VPrimitiveScorer * G4VScoringMesh::GetPrimitiveScorer(const G4String & name) {
224 if(!fMFD) return nullptr;
225
226 G4int nps = fMFD->GetNumberOfPrimitives();
227 for(G4int i = 0; i < nps; i++) {
228 G4VPrimitiveScorer * prs = fMFD->GetPrimitive(i);
229 if(name == prs->GetName()) return prs;
230 }
231
232 return nullptr;
233}

251 << " "
252 << fRotationMatrix->yx() << " "
253 << fRotationMatrix->yy() << " "
254 << fRotationMatrix->yz() << G4endl
255 << " "
256 << fRotationMatrix->zx() << " "
257 << fRotationMatrix->zy() << " "
258 << fRotationMatrix->zz() << G4endl;
259 }
260
261
262 G4cout << " registered primitve scorers : " << G4endl;
263 G4int nps = fMFD->GetNumberOfPrimitives();
264 G4VPrimitiveScorer * prs;
265 for(int i = 0; i < nps; i++) {
266 prs = fMFD->GetPrimitive(i);
267 G4cout << " " << i << " " << prs->GetName();
268 if(prs->GetFilter() != 0) G4cout << " with " << prs->GetFilter()->GetName();
269 G4cout << G4endl;
270 }

286void G4VScoringMesh::DrawMesh(const G4String& psName,G4VScoreColorMap* colorMap,G4int axflg)
287{
288 fDrawPSName = psName;
289 MeshScoreMap::const_iterator fMapItr = fMap.find(psName);
290 if(fMapItr!=fMap.end()) {
291 fDrawUnit = GetPSUnit(psName);
292 fDrawUnitValue = GetPSUnitValue(psName);
293 Draw(fMapItr->second, colorMap,axflg);
294 } else {
295 G4cerr << "Scorer <" << psName << "> is not defined. Method ignored." << G4endl;
296 }

312void G4VScoringMesh::Accumulate(G4THitsMap<G4double> * map)
313{
314 G4String psName = map->GetName();
315 MeshScoreMap::const_iterator fMapItr = fMap.find(psName);
316 *(fMapItr->second) += *map;
317
318 if(verboseLevel > 9) {
319 G4cout << G4endl;
320 G4cout << "G4VScoringMesh::Accumulate()" << G4endl;
321 G4cout << " PS name : " << psName << G4endl;
322 if(fMapItr == fMap.end()) {
323 G4cout << " " << psName << " was not found." << G4endl;
324 } else {
325 G4cout << " map size : " << map->GetSize() << G4endl;
326 map->PrintAllHits();
327 }
328 G4cout << G4endl;
329 }
330}

352void G4VScoringMesh::Construct(G4VPhysicalVolume* fWorldPhys)
353{
354 if(fConstructed) {
355 if(fGeometryHasBeenDestroyed) {
356 SetupGeometry(fWorldPhys);
357 fGeometryHasBeenDestroyed = false;
358 }
359 if(verboseLevel > 0)
360 G4cout << fWorldPhys->GetName() << " --- All quantities are reset."
361 << G4endl;
362 ResetScore();
363 }
364 else {
365 fConstructed = true;
366 SetupGeometry(fWorldPhys);
367 }
368}
369
370void G4VScoringMesh::WorkerConstruct(G4VPhysicalVolume* fWorldPhys)
371{
372 if(fConstructed) {
373 if(fGeometryHasBeenDestroyed) {
374 fMeshElementLogical->SetSensitiveDetector(fMFD);
375 fGeometryHasBeenDestroyed = false;
376 }
377
378 if(verboseLevel > 0)
379 G4cout << fWorldPhys->GetName() << " --- All quantities are reset." << G4endl;
380 ResetScore();
381
382 } else {
383 fConstructed = true;
384 fMeshElementLogical->SetSensitiveDetector(fMFD);
385 }

388void G4VScoringMesh::Merge(const G4VScoringMesh * scMesh)
389{
390 const MeshScoreMap scMap = scMesh->GetScoreMap();
391
392 MeshScoreMap::const_iterator fMapItr = fMap.begin();
393 MeshScoreMap::const_iterator mapItr = scMap.begin();
394 for(; fMapItr != fMap.end(); fMapItr++) {
395 if(verboseLevel > 9) G4cout << "G4VScoringMesh::Merge()" << fMapItr->first << G4endl;
396 *(fMapItr->second) += *(mapItr->second);
397 mapItr++;
398 }
