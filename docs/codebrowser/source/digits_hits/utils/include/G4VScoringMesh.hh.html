29#ifndef G4VScoringMesh_h
30#define G4VScoringMesh_h 1
31
32#include "globals.hh"
33#include "G4THitsMap.hh"
34#include "G4RotationMatrix.hh"
35#include "G4StatDouble.hh"
36
37class G4VPhysicalVolume;
38class G4LogicalVolume;
39class G4MultiFunctionalDetector;
40class G4VPrimitiveScorer;
41class G4VSDFilter;
42class G4VScoreColorMap;
43class G4ParallelWorldProcess;
44
45#include <map>
46
47enum MeshShape { boxMesh, cylinderMesh, sphereMesh , undefinedMesh = -1};
48typedef G4THitsMap< G4double > EventScore;
49typedef G4THitsMap< G4StatDouble > RunScore;
50typedef std::map< G4String, RunScore* > MeshScoreMap;
51// class description:
52//
53// This class represents a parallel world for interactive scoring purposes.
54//
55
56class G4VScoringMesh
57{
58 public:
59 G4VScoringMesh(const G4String& wName);
60 virtual ~G4VScoringMesh();
61
62 public: // with description
63 // a pure virtual function to construct this mesh geometry
64 void Construct(G4VPhysicalVolume* fWorldPhys);
65
66 void WorkerConstruct(G4VPhysicalVolume* fWorldPhys);
67
68 protected:
69 virtual void SetupGeometry(G4VPhysicalVolume * fWorldPhys) = 0;
70
71 public: // with description
72 // list infomration of this mesh
73 virtual void List() const;
74
75 public: // with description
76 // get the world name
77 inline const G4String& GetWorldName() const
78 { return fWorldName; }
79 // get whether this mesh is active or not
80 inline G4bool IsActive() const
81 { return fActive; }
82 // set an activity of this mesh
83 inline void Activate(G4bool vl = true)
84 { fActive = vl; }
85 // get the shape of this mesh
86 inline MeshShape GetShape() const
87 { return fShape; }
88 // accumulate hits in a registered primitive scorer
89 void Accumulate(G4THitsMap<G4double> * map);
90 void Accumulate(G4THitsMap<G4StatDouble> * map);
91 // merge same kind of meshes
92 void Merge(const G4VScoringMesh * scMesh);
93 // dump information of primitive socrers registered in this mesh
94 void Dump();
95 // draw a projected quantity on a current viewer
96 void DrawMesh(const G4String& psName,G4VScoreColorMap* colorMap,G4int axflg=111);
97 // draw a column of a quantity on a current viewer
98 void DrawMesh(const G4String& psName,G4int idxPlane,G4int iColumn,G4VScoreColorMap* colorMap);
99 // draw a projected quantity on a current viewer
100 virtual void Draw(RunScore * map, G4VScoreColorMap* colorMap, G4int axflg=111) = 0;
101 // draw a column of a quantity on a current viewer
102 virtual void DrawColumn(RunScore * map, G4VScoreColorMap* colorMap,
103 G4int idxProj, G4int idxColumn) = 0;
104 // reset registered primitive scorers
105 void ResetScore();
106
107 // set size of this mesh
108 void SetSize(G4double size[3]);
109 // get size of this mesh
110 G4ThreeVector GetSize() const;
111 // set position of center of this mesh
112 void SetCenterPosition(G4double centerPosition[3]);
113 // get position of center of this mesh
114 G4ThreeVector GetTranslation() const {return fCenterPosition;}
115 // set a rotation angle around the x axis
116 void RotateX(G4double delta);
117 // set a rotation angle around the y axis
118 void RotateY(G4double delta);
119 // set a rotation angle around the z axis
120 void RotateZ(G4double delta);
121 // get a rotation matrix
122 G4RotationMatrix GetRotationMatrix() const {
123 if(fRotationMatrix) return *fRotationMatrix;
124 else return G4RotationMatrix::IDENTITY;
125 }
126 // set number of segments of this mesh
127 void SetNumberOfSegments(G4int nSegment[3]);
128 // get number of segments of this mesh
129 void GetNumberOfSegments(G4int nSegment[3]);
130
131 // register a primitive scorer to the MFD & set it to the current primitive scorer
132 void SetPrimitiveScorer(G4VPrimitiveScorer * ps);
133 // register a filter to a current primtive scorer
134 void SetFilter(G4VSDFilter * filter);
135 // set a primitive scorer to the current one by the name
136 void SetCurrentPrimitiveScorer(const G4String & name);
137 // find registered primitive scorer by the name
138 G4bool FindPrimitiveScorer(const G4String & psname);
139 // get whether current primitive scorer is set or not
140 G4bool IsCurrentPrimitiveScorerNull() {
141 if(fCurrentPS == nullptr) return true;
142 else return false;
143 }
144 // get unit of primitive scorer by the name
145 G4String GetPSUnit(const G4String & psname);
146 // get unit of current primitive scorer
147 G4String GetCurrentPSUnit();
148 // set unit of current primitive scorer
149 void SetCurrentPSUnit(const G4String& unit);
150 // get unit value of primitive scorer by the name
151 G4double GetPSUnitValue(const G4String & psname);
152 // set PS name to be drawn
153 void SetDrawPSName(const G4String & psname) {fDrawPSName = psname;}
154
155 // get axis names of the hierarchical division in the divided order
156 void GetDivisionAxisNames(G4String divisionAxisNames[3]);
157
158 // set current primitive scorer to NULL
159 void SetNullToCurrentPrimitiveScorer() {fCurrentPS = nullptr;}
160 // set verbose level
161 inline void SetVerboseLevel(G4int vl)
162 { verboseLevel = vl; }
163 // get the primitive scorer map
164 inline MeshScoreMap GetScoreMap() const
165 { return fMap; }
166 // get whether this mesh setup has been ready
167 inline G4bool ReadyForQuantity() const
168 { return (sizeIsSet && nMeshIsSet); }
169
170protected:
171 // get registered primitive socrer by the name
172 G4VPrimitiveScorer * GetPrimitiveScorer(const G4String & name);
173
174protected:
175 G4String fWorldName;
176 G4VPrimitiveScorer * fCurrentPS;
177 G4bool fConstructed;
178 G4bool fActive;
179 MeshShape fShape;
180
181 G4double fSize[3];
182 G4ThreeVector fCenterPosition;
183 G4RotationMatrix * fRotationMatrix;
184 G4int fNSegment[3];
185
186 MeshScoreMap fMap;
187 G4MultiFunctionalDetector * fMFD;
188
189 G4int verboseLevel;
190
191 G4bool sizeIsSet;
192 G4bool nMeshIsSet;
193
194 G4String fDrawUnit;
195 G4double fDrawUnitValue;
196 G4String fDrawPSName;
197
198 G4String fDivisionAxisNames[3];
199
200 G4LogicalVolume * fMeshElementLogical;
201
202public:
203 inline void SetMeshElementLogical(G4LogicalVolume* val)
204 { fMeshElementLogical = val; }
205 inline G4LogicalVolume* GetMeshElementLogical() const
206 { return fMeshElementLogical; }
207
208protected:
209 G4ParallelWorldProcess* fParallelWorldProcess;
210 G4bool fGeometryHasBeenDestroyed;
211public:
212 inline void SetParallelWorldProcess(G4ParallelWorldProcess* proc)
213 { fParallelWorldProcess = proc; }
214 inline G4ParallelWorldProcess* GetParallelWorldProcess() const
215 { return fParallelWorldProcess; }
216 inline void GeometryHasBeenDestroyed()
217 {
218 fGeometryHasBeenDestroyed = true;
219 fMeshElementLogical = nullptr;
220 }
221};
222
223#endif
