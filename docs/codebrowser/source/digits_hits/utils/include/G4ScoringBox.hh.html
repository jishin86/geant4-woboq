43{
44public:
45 G4ScoringBox(G4String wName);
46 ~G4ScoringBox();
47
48protected:
49 // construct this mesh
50 virtual void SetupGeometry(G4VPhysicalVolume * fWorldPhys);
51
52public:
53 void List() const;

56 G4int idxProj, G4int idxColumn);

58 // set a direction to segment this mesh
59 void SetSegmentDirection(G4int dir) {fSegmentDirection = dir;}
60
61private:
62 // get replicated position from 3D index (x,y,z)
63 G4ThreeVector GetReplicaPosition(G4int x, G4int y, G4int z);
64 // get 3D index (x,y,z) from sequential index
65 void GetXYZ(G4int index, G4int q[3]) const;

67 G4int GetIndex(G4int x, G4int y, G4int z) const;
68
69private:
70 G4int fSegmentDirection; // =1: x, =2: y, =3: z
71
72};
