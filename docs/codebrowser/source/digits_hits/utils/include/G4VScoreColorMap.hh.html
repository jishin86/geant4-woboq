29#ifndef G4VScoreColorMap_h
30#define G4VScoreColorMap_h 1
31
32#include "globals.hh"
33
34class G4VVisManager;
35
36class G4VScoreColorMap
37{
38 public:
39 G4VScoreColorMap(G4String mName);
40 virtual ~G4VScoreColorMap();
41
42 public:
43 virtual void GetMapColor(G4double val, G4double color[4]) = 0;
44
45 public:
46 inline G4String GetName() const
47 { return fName; }
48 inline void SetFloatingMinMax(G4bool vl=true)
49 { ifFloat = vl; }
50 inline G4bool IfFloatMinMax() const
51 { return ifFloat; }
52 inline void SetMinMax(G4double minVal, G4double maxVal)
53 {
54 if(minVal >= maxVal)
55 { G4cerr << "WARNING: G4VScoreColoMap::SetMinMax() : minimum is larger than or equal to maximum. Verify values you set, ["
56 << minVal << ", " << maxVal << "]" << G4endl;
57 fMinVal = maxVal; fMaxVal = minVal;
58 }
59 else {
60 fMinVal = minVal; fMaxVal = maxVal;
61 }
62 }
63 inline G4double GetMin() const
64 { return fMinVal; }
65 inline G4double GetMax() const
66 { return fMaxVal; }
67
68 // draw a color chart
69 virtual void DrawColorChart(G4int nPoint = 5);
70
71 virtual void DrawColorChartBar(G4int nPoint);
72
73 virtual void DrawColorChartText(G4int nPoint);
74
75
76 void SetPSUnit(G4String & unit) {fPSUnit = unit;}
77 void SetPSName(G4String & psName) {fPSName = psName;}
78
79 protected:
80 G4String fName;
81 G4bool ifFloat;
82 G4double fMinVal;
83 G4double fMaxVal;
84 G4VVisManager * fVisManager;
85 G4String fPSUnit;
86 G4String fPSName;
87};

89#endif
