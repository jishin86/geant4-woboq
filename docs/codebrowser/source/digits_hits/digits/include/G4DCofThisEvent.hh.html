38// class description:
39//
40// This is a class which stores digi collections generated at one event.
41// This class is exclusively constructed by G4DigiManager when the first
42// digi collection of an event is passed to the manager, and this class
43// object is deleted by G4RunManager when a G4Event class object is deleted.
44// Almost all public methods must be used by Geant4 kernel classes and
45// the user should not invoke them. The user can use two const methods,
46// GetDC() and GetNumberOfCollections() for accessing to the stored digi
47// collection(s).
48
49class G4DCofThisEvent
50{
51 public:
52 G4DCofThisEvent();
53 G4DCofThisEvent(G4int cap);
54 ~G4DCofThisEvent();
55 inline void *operator new(size_t);
56 inline void operator delete(void* anDCoTE);
57
58 void AddDigiCollection(G4int DCID,G4VDigiCollection * aDC);
59
60 G4DCofThisEvent(const G4DCofThisEvent&);
61 G4DCofThisEvent& operator=(const G4DCofThisEvent&);
62 private:
63 std::vector<G4VDigiCollection*> * DC;
64
65 public: // with description
66 inline G4VDigiCollection* GetDC(G4int i) const
67 { return (*DC)[i]; }
68 // Returns a pointer to a digi collection. Null will be returned
69 // if the particular collection is not stored at the current event.
70 // The integer argument is ID number which is assigned by G4DigiManager
71 // and the number can be obtained by G4DigiManager::GetDigiCollectionID()
72 // method.
73 inline G4int GetNumberOfCollections() const
74 {
75 G4int n = 0;
76 for(auto dc : *DC) if(dc) n++;
77 return n;
78 }
79 // Returns the number of digi collections which are stored in this class
80 // object.
81 public:
82 inline G4int GetCapacity() const
83 {
84 return DC->size();
85 }
86};

88#if defined G4DIGI_ALLOC_EXPORT
89 extern G4DLLEXPORT G4Allocator<G4DCofThisEvent>*& anDCoTHAllocator_G4MT_TLS_();
90#else
91 extern G4DLLIMPORT G4Allocator<G4DCofThisEvent>*& anDCoTHAllocator_G4MT_TLS_();
92#endif
93
94inline void* G4DCofThisEvent::operator new(size_t)
95{
96 if (!anDCoTHAllocator_G4MT_TLS_())
97 anDCoTHAllocator_G4MT_TLS_() = new G4Allocator<G4DCofThisEvent>;
98 G4Allocator<G4DCofThisEvent>& anDCoTHAllocator =
99 *anDCoTHAllocator_G4MT_TLS_();
100 void* anDCoTH;
101 anDCoTH = (void*)anDCoTHAllocator.MallocSingle();
102 return anDCoTH;
103}
104
105inline void G4DCofThisEvent::operator delete(void* anDCoTH)
106{
107 if (!anDCoTHAllocator_G4MT_TLS_())
108 anDCoTHAllocator_G4MT_TLS_() = new G4Allocator<G4DCofThisEvent>;
109 G4Allocator<G4DCofThisEvent>& anDCoTHAllocator =
110 *anDCoTHAllocator_G4MT_TLS_();
111 anDCoTHAllocator.FreeSingle((G4DCofThisEvent*)anDCoTH);
112}
