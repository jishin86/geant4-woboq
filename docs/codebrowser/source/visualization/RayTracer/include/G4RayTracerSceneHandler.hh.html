31#define G4RAYTRACERSCENEHANDLER_HH
32
33#include "G4VSceneHandler.hh"
34
35#include "G4ModelingParameters.hh"
36#include <map>
37
38class G4RayTracerSceneHandler: public G4VSceneHandler
39{
40public:
41
42 G4RayTracerSceneHandler(G4VGraphicsSystem& system,
43 const G4String& name = "");
44 virtual ~G4RayTracerSceneHandler();
45
46 void BuildVisAttsMap (const G4VSolid& solid);
47 // BuildVisAttsMap is the appropriate name for what it does in this class
48 // but it is actually an override of G4VSceneHandler::RequestPrimitives.
49 void RequestPrimitives (const G4VSolid& solid)
50 {BuildVisAttsMap(solid);}
51
52 // Required pure virtual functions, not used in Ray Tracer.
53 void AddPrimitive(const G4Polyline&){}
54 void AddPrimitive(const G4Text&){}
55 void AddPrimitive(const G4Circle&){}
56 void AddPrimitive(const G4Square&){}
57 void AddPrimitive(const G4Polyhedron&){}
58 void AddPrimitive(const G4Polymarker&){}
59 void AddPrimitive(const G4Scale&){}
60
61 void ClearStore ();
62
63 struct PathLessThan
64 {G4bool operator()
65 (const G4ModelingParameters::PVPointerCopyNoPath&,
66 const G4ModelingParameters::PVPointerCopyNoPath&) const;
67 };
68
69 const std::map
70 <G4ModelingParameters::PVPointerCopyNoPath,G4VisAttributes,PathLessThan>&
71 GetSceneVisAttsMap() const
72 {return fSceneVisAttsMap;}
73
74private:
75
76 static G4int fSceneIdCount; // Counter for RayTracer scene handlers.
77
78 std::map
79 <G4ModelingParameters::PVPointerCopyNoPath,G4VisAttributes,PathLessThan>
80 fSceneVisAttsMap;
81};
