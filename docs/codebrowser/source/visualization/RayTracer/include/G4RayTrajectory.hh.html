31//
32// This is a concrete class of G4VTrajectory which represents a trajectory of a ray.
33// This class is used by G4RayTracer. Objects of this class are created by G4RTTrackingAction.
34// This class does not have concrete implementations of draw and print methods but
35// information stored in this class is used by determining a colour of a pixel of the picture.
36//

40///////////////////

42#ifndef G4RayTrajectory_h
43#define G4RayTrajectory_h 1
44
45class G4Step;
46
47#include "G4VTrajectory.hh"
48#include "G4Allocator.hh"
49#include <stdlib.h>
50#include <vector>
51#include "globals.hh"
52#include "G4Track.hh"
53#include "G4RayTrajectoryPoint.hh"
54
55
56class G4RayTrajectory : public G4VTrajectory
57{
58 public:
59
60 G4RayTrajectory();
61 G4RayTrajectory(G4RayTrajectory & right);
62 virtual ~G4RayTrajectory();
63
64 private:
65
66 G4RayTrajectory& operator= (const G4RayTrajectory&);
67
68 public:
69
70 inline void* operator new(size_t);
71 inline void operator delete(void*);
72 // inline int operator == (const G4RayTrajectory& right){return (this==&right);}
73
74 virtual void AppendStep(const G4Step*);
75 virtual void ShowTrajectory(std::ostream&) const;
76 virtual void DrawTrajectory() const {;}
77 virtual int GetPointEntries() const {return positionRecord->size();}
78 virtual G4VTrajectoryPoint* GetPoint(G4int i) const
79 { return (*positionRecord)[i]; }
80 G4RayTrajectoryPoint* GetPointC(G4int i) const
81 { return (*positionRecord)[i]; }
82 virtual void MergeTrajectory(G4VTrajectory* secondTrajectory);
83
84// Get/Set functions to satisfy pure virtual functions of base class.
85 inline G4int GetTrackID() const { return 0; }
86 inline G4int GetParentID() const { return 0; }
87 inline G4String GetParticleName() const { return ""; }
88 inline G4double GetCharge() const { return 0.; }
89 inline G4int GetPDGEncoding() const { return 0; }
90 inline G4ThreeVector GetInitialMomentum() const { return G4ThreeVector(); }
91
92 private:
93
94 std::vector<G4RayTrajectoryPoint*>* positionRecord;
95};
96
97#if defined G4VIS_ALLOC_EXPORT
98 extern G4DLLEXPORT G4Allocator<G4RayTrajectory>*& rayTrajectoryAllocator();
99#else
100 extern G4DLLIMPORT G4Allocator<G4RayTrajectory>*& rayTrajectoryAllocator();
101#endif
102
103inline void* G4RayTrajectory::operator new(size_t)
104{
105 if(!rayTrajectoryAllocator())
106 { rayTrajectoryAllocator() = new G4Allocator<G4RayTrajectory>; }
107 return (void*)rayTrajectoryAllocator()->MallocSingle();
108}
109
110inline void G4RayTrajectory::operator delete(void* aTrajectory)
111{
112 rayTrajectoryAllocator()->FreeSingle((G4RayTrajectory*)aTrajectory);
113}
