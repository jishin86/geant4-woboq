30#ifndef G4RTSimpleScanner_H
31#define G4RTSimpleScanner_H 1
32
33// class description:
34//
35// G4RTSimpleScanner
36// Provides a simple raster of window coordinates.
37
38#include "G4VRTScanner.hh"
39
40class G4RTSimpleScanner: public G4VRTScanner {
41
42public: // with description
43
44 G4RTSimpleScanner();
45 virtual ~G4RTSimpleScanner();
46
47 // Compiler defaults for copy constructor and assignmemt.
48
49 virtual const G4String& GetGSName() const;
50 // Get name that acts as graphics system name.
51
52 virtual const G4String& GetGSNickname() const;
53 // Get name that acts as graphics system nickname. It is this that
54 // the user specifies on the /vis/open and /vis/sceneHandler/create
55 // commands.
56
57 virtual void Initialize(G4int nRow, G4int nColumn);
58 // Intialises scanner for window with nRow rows and nColumn columns.
59
60 virtual G4bool Coords(G4int& iRow, G4int& iColumn);
61 // Supplies coordinate (iRow,iColumn) and returns false when the
62 // sequence has finished, i.e., on the call *after* suplying the
63 // last valid coordinate.
64
65protected:
66 G4String theGSName, theGSNickname;
67 G4int theNRow, theNColumn, theIRow, theIColumn;
68};

70#endif
