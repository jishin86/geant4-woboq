31#define G4VRTScanner_H 1
32
33// class description:
34//
35// G4VRTScanner
36// Interface class for provider of a sequence of window coordinates.
37
38#include "globals.hh"
39
40class G4VRTScanner {
41
42public: // with description
43
44 G4VRTScanner();
45 virtual ~G4VRTScanner();
46
47 virtual const G4String& GetGSName() const = 0;
48 // Get name that acts as graphics system name.
49
50 virtual const G4String& GetGSNickname() const = 0;
51 // Get name that acts as graphics system nickname. It is this that
52 // the user specifies on the /vis/open and /vis/sceneHandler/create
53 // commands.
54
55 virtual void Initialize(G4int nRow, G4int nColumn) = 0;
56 // Intialises scanner for window with nRow rows and nColumn columns.
57
58 virtual G4bool Coords(G4int& iRow, G4int& iColumn) = 0;
59 // Supplies coordinate (iRow,iColumn) and returns false when the
60 // sequence has finished, i.e., on the call *after* suplying the
61 // last valid coordinate.
62
63 virtual void Draw
64 (unsigned char red, unsigned char green, unsigned char blue);
65 // Draw coloured square at current position.
66
67};
68
69inline
70void G4VRTScanner::Draw(unsigned char, unsigned char, unsigned char) {}
