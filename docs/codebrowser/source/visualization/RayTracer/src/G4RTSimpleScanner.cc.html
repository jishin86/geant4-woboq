30#include "G4RTSimpleScanner.hh"
31
32G4RTSimpleScanner::G4RTSimpleScanner():
33 G4VRTScanner(), theNRow(0), theNColumn(0), theIRow(0), theIColumn(0)
34{
35 theGSName = "RayTracer";
36 theGSNickname = "RayTracer";
37}
38
39G4RTSimpleScanner::~G4RTSimpleScanner(){}
40
41const G4String& G4RTSimpleScanner::GetGSName() const
42{return theGSName;}
43
44const G4String& G4RTSimpleScanner::GetGSNickname() const
45{return theGSNickname;}
46
47void G4RTSimpleScanner::Initialize(G4int nRow, G4int nColumn) {
48 theNRow = nRow;
49 theNColumn = nColumn;
50 theIRow = 0;
51 theIColumn = -1;
52}
53
54G4bool G4RTSimpleScanner::Coords(G4int& iRow, G4int& iColumn)
55{
56 // Increment column and, if necessary, increment row...
57 ++theIColumn;
58 if (theIColumn >= theNColumn) {
59 theIColumn = 0;
60 ++theIRow;
61 }
62
63 // Return if finished...
64 if (theIRow >= theNRow) return false;
65
66 // Return current row and column...
67 iRow = theIRow;
68 iColumn = theIColumn;
69 return true;
70}
