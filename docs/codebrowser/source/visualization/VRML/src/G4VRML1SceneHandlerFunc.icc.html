96void G4VRML1SCENEHANDLER::AddSolid(const G4Tubs& tubs)
97{
98#if defined DEBUG_SCENE_FUNC
99 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
100 G4cout << "***** AddSolid tubs" << "\n" ;
101#endif
102 VRMLBeginModeling () ;
103
104 // set current name
105 fCurrentDEF = "tubs_" + tubs.GetName();
106
107 // Get data
108 const G4double R = tubs.GetOuterRadius(); // outside radius
109 const G4double r = tubs.GetInnerRadius(); // inside radius
110 const G4double dz = tubs.GetZHalfLength(); // half length in z
111 const G4double dp = tubs.GetDeltaPhiAngle(); // angle interval
112
113 // Send data
114 if ( r == 0.0 && dp >= 360. * CLHEP::deg ) {
115 // Send a built-in VRML node (Cylinder)
116 fDest << "Separator {" << "\n";
117 SendMatrixTransformNode( fObjectTransformation );
118
119 fDest << "\t" << "DEF " << fCurrentDEF << " Separator {" << "\n";
120 SendMaterialNode();
121 SendCylinderNode(R, dz * 2);
122 fDest << "\t" << "}" << "\n"; // DEF Separator
123 fDest << "}" << "\n"; // Separator
124 } else {
125 // call AddPrimitive(Polyhedron)
126 RequestPrimitives(tubs);
127 }
128
129 // reset current name to null
130 fCurrentDEF = "";
131}

134void G4VRML1SCENEHANDLER::AddSolid(const G4Cons& cons)

136#if defined DEBUG_SCENE_FUNC
137 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
138 G4cout << "***** AddSolid cons" << "\n" ;
139#endif
140 VRMLBeginModeling () ;
141
142 // set current name
143 fCurrentDEF = "cons_" + cons.GetName();
144
145 // Get data
146 const G4double r1 = cons.GetInnerRadiusMinusZ(); // inside radius at -dz
147 const G4double R1 = cons.GetOuterRadiusMinusZ(); // outside radius at -dz
148 const G4double r2 = cons.GetInnerRadiusPlusZ(); // inside radius at +dz
149 const G4double R2 = cons.GetOuterRadiusPlusZ(); // outside radius at +dz
150 const G4double dz = cons.GetZHalfLength(); // half length in z
151 //const G4double sp = cons.GetStartPhiAngle(); // starting angle
152 const G4double dp = cons.GetDeltaPhiAngle(); // angle width
153
154 // Send data
155 if ( r1 == 0.0 && r2 == 0.0 && R1 == R2 && dp >= 360. * CLHEP::deg) {
156 // Send a built-in VRML node (Cylinder)
157 fDest << "Separator {" << "\n";
158 SendMatrixTransformNode( fObjectTransformation );
159
160 fDest << "\t" << "DEF " << fCurrentDEF << " Separator {" << "\n";
161 SendMaterialNode();
162 SendCylinderNode(R1, dz * 2);
163 fDest << "\t" << "}" << "\n"; // DEF Separator
164 fDest << "}" << "\n"; // Separator
165 } else {
166 // call AddPrimitive(Polyhedron)
167 RequestPrimitives(cons);
168 }
169
170 // reset current name to null
171 fCurrentDEF = "";
172}

240void G4VRML1SCENEHANDLER::AddPrimitive(const G4Polyline& polyline)
241{
242#if defined DEBUG_SCENE_FUNC
243 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
244 G4cout << "***** AddPrimitive polyline" << "\n" ;
245#endif
246 if (fProcessing2D) {
247 static G4bool warned = false;
248 if (!warned) {
249 warned = true;
250 G4Exception
251 ("G4VRML1SCENEHANDLER::AddPrimitive (const G4Polyline&)",
252 "VRML-1001", JustWarning,
253 "2D polylines not implemented. Ignored.");
254 }
255 return;
256 }
257
258 VRMLBeginModeling () ;
259
260 fDest << "Separator {" << "\n";
261
262 SendMatrixTransformNode (fObjectTransformation );
263
264 SendMaterialNode( polyline.GetVisAttributes() );
265
266 fDest << "\t" << "Coordinate3 {" << "\n";
267 fDest << "\t\t" << "point [" << "\n";
268 G4int e, i;
269 for (i = 0, e = polyline.size(); e; i++, e--) {
270 fDest << "\t\t\t";
271 fDest << polyline[i].x() << " ";
272 fDest << polyline[i].y() << " ";
273 fDest << polyline[i].z() << "," << "\n";
274 }
275 fDest << "\t\t" << "]" << "\n";
276 fDest << "\t" << "}" << "\n"; // Coordinate3
277
278 fDest << "\t" << "IndexedLineSet {" << "\n";
279 fDest << "\t\t" << "coordIndex [";
280 for (i = 0, e = polyline.size(); e; i++, e--) {
281 if (i % 10 == 0)
282 fDest << "\n" << "\t\t\t";
283 fDest << i << ", ";
284 }
285 fDest << "-1" << "\n";
286 fDest << "\t\t" << "]" << "\n";
287 fDest << "\t" << "}" << "\n"; // IndexedLineSet
288
289 fDest << "}" << "\n"; // Separator
290}

292void G4VRML1SCENEHANDLER::AddPrimitive(const G4Polyhedron& polyhedron)
293{
294#if defined DEBUG_SCENE_FUNC
295 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
296 G4cout << "***** AddPrimitive(G4Polyhedron)" << "\n";
297#endif
298
299 if (polyhedron.GetNoFacets() == 0) return;
300
301 if (fProcessing2D) {
302 static G4bool warned = false;
303 if (!warned) {
304 warned = true;
305 G4Exception
306 ("G4VRML1SCENEHANDLER::AddPrimitive (const G4Polyhedron&)",
307 "VRML-1002", JustWarning,
308 "2D polyhedra not implemented. Ignored.");
309 }
310 return;
311 }
312
313 VRMLBeginModeling () ;
314
315 fDest << "Separator {" << "\n";
316
317 SendMatrixTransformNode( fObjectTransformation );
318
319 fDest << "\t";
320 if (fCurrentDEF != "")
321 fDest << "DEF " << fCurrentDEF << " ";
322
323 fDest << "Separator {" << "\n";
324
325 //fDest << "\t\t" << "renderCulling ON" << "\n";
326
327 fDest << "\t\t" << "ShapeHints {" << "\n";
328 fDest << "\t\t\t" << "vertexOrdering COUNTERCLOCKWISE" << "\n";
329 fDest << "\t\t\t" << "shapeType SOLID" << "\n";
330 fDest << "\t\t\t" << "faceType CONVEX" << "\n";
331 fDest << "\t\t" << "}" << "\n";
332
333 SendMaterialNode();
334
335 fDest << "\t\t" << "Coordinate3 {" << "\n";
336 fDest << "\t\t\t" << "point [" << "\n";
337 G4int i, j;
338 for (i = 1, j = polyhedron.GetNoVertices(); j; j--, i++) {
339 G4Point3D point = polyhedron.GetVertex(i);
340 fDest << "\t\t\t\t";
341 fDest << point.x() << " ";
342 fDest << point.y() << " ";
343 fDest << point.z() << "," << "\n";
344 }
345 fDest << "\t\t\t" << "]" << "\n";
346 fDest << "\t\t" << "}" << "\n"; // Coordinate3
347
348 fDest << "\t\t" << "IndexedFaceSet {" << "\n";
349 fDest << "\t\t\t" << "coordIndex [" << "\n";
350
351 // facet loop
352 G4int f;
353 for (f = polyhedron.GetNoFacets(); f; f--) {
354
355 // edge loop
356 G4bool notLastEdge;
357 G4int index = -1, edgeFlag = 1;
358 fDest << "\t\t\t\t";
359 do {
360 notLastEdge = polyhedron.GetNextVertexIndex(index, edgeFlag);
361 fDest << index - 1 << ", ";
362 } while (notLastEdge);
363 fDest << "-1," << "\n";
364 }
365 fDest << "\t\t\t" << "]" << "\n";
366 fDest << "\t\t" << "}" << "\n"; // IndexFaceSet
367
368 fDest << "\t" << "}" << "\n"; // (DEF) Separator
369 fDest << "}" << "\n"; // Separator
370
371}

560void G4VRML1SCENEHANDLER::SendMatrixTransformNode(const G4Transform3D& trans)
561{
562 G4Point3D B (0.0, 0.0, 0.0);
563
564 G4Point3D x1 (1.0, 0.0, 0.0);
565 G4Point3D y1 (0.0, 1.0, 0.0);
566
567 G4Vector3D e1 (1.0, 0.0, 0.0);
568 G4Vector3D e2 (0.0, 1.0, 0.0);
569 G4Vector3D e3 (0.0, 0.0, 1.0);
570
571 //----- transformed origin of body coord
572 B.transform(trans);
573
574 //----- transformed base vectors of body coord
575 x1.transform(trans);
576 e1 = x1 - B;
577
578 y1.transform(trans);
579 e2 = y1 - B;
580
581 e3 = e1.cross(e2);
582
583 e1 = e1.unit(); // normalize again for accuracy
584 e2 = e2.unit(); //
585 e3 = e3.unit(); //
586
587 fDest << "\t" << "MatrixTransform {" << "\n";
588 fDest << "\t\t" << "matrix ";
589 fDest << e1.x() << " " << e1.y() << " " << e1.z() << " 0 ";
590 fDest << e2.x() << " " << e2.y() << " " << e2.z() << " 0 ";
591 fDest << e3.x() << " " << e3.y() << " " << e3.z() << " 0 ";
592 fDest << B.x() << " " << B.y() << " " << B.z() << " 1" << "\n";
593 fDest << "\t" << "}" << "\n";
594}

686G4double G4VRML1SCENEHANDLER::GetMarkerHalfSize ( const G4VMarker& mark )
687{
688 //----- return value ( marker radius in 3d units)
689 G4double size = 1.0 ; // initialization
690
691 //----- parameters to calculate 3d size from 2d size
692 const double HALF_SCREEN_SIZE_2D = 300.0 ; // pixels
693 double zoom_factor = fpViewer->GetViewParameters().GetZoomFactor() ;
694 if ( zoom_factor <= 0.0 ) { zoom_factor = 1.0 ; }
695 double extent_radius_3d = GetScene()->GetExtent().GetExtentRadius() ;
696 if ( extent_radius_3d <= 0.0 ) { extent_radius_3d = 1.0 ; }
697
698 //----- get marker radius in 3D units
699 if ( mark.GetWorldSize() > 0.0 ) {
700 // get mark radius in 3D units
701 size = 0.5 * mark.GetWorldSize() ;
702
703 } else if ( mark.GetScreenSize() > 0.0 ) {
704 // local
705 double mark_radius_2d = 0.5 * mark.GetScreenSize() ;
706
707 // get mark radius in 3D units
708 size \
709 = extent_radius_3d * ( mark_radius_2d / HALF_SCREEN_SIZE_2D );
710 size *= zoom_factor ;
711
712 } else {
713 // local
714 double mark_radius_2d \
715 = fpViewer->GetViewParameters().GetDefaultMarker().GetScreenSize();
716 mark_radius_2d *= 0.1 ; // Magic number?
717
718 // get mark radius in 3D units
719 size \
720 = extent_radius_3d * ( mark_radius_2d / HALF_SCREEN_SIZE_2D );
721 size *= zoom_factor ;
722 }
723
724 //----- global rescaling
725 size *= fpViewer->GetViewParameters().GetGlobalMarkerScale();
726
727 //----- return size
728 return size ;
729
730}
