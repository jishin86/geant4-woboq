30// A base class for a scene handler to dump geometry hierarchy.
31// In the derived class, override G4VScenehandler::RequestPrimitives
32// to implement dump of the geometry hierarchy.
33
34#ifndef G4VTREESCENEHANDLER_HH
35#define G4VTREESCENEHANDLER_HH
36
37#include "G4VSceneHandler.hh"
38
39#include "G4PhysicalVolumeModel.hh"
40#include <vector>
41#include <set>
42
43class G4VPhysicalVolume;
44class G4LogicalVolume;
45class G4ModelingParameters;
46
47class G4VTreeSceneHandler: public G4VSceneHandler {
48
49public:
50 G4VTreeSceneHandler(G4VGraphicsSystem& system,
51 const G4String& name);
52 virtual ~G4VTreeSceneHandler ();
53 void PreAddSolid (const G4Transform3D& objectTransformation,
54 const G4VisAttributes&);
55 void PostAddSolid ();
56
57 ////////////////////////////////////////////////////////////////
58 // Functions not used but required by the abstract interface.
59
60 virtual void AddPrimitive (const G4Polyline&) {}
61 virtual void AddPrimitive (const G4Text&) {}
62 virtual void AddPrimitive (const G4Circle&) {}
63 virtual void AddPrimitive (const G4Square&) {}
64 virtual void AddPrimitive (const G4Polyhedron&) {}
65 virtual void AddPrimitive (const G4Polymarker&) {}
66 virtual void AddPrimitive (const G4Scale&) {}
67
68 virtual void BeginModeling();
69 virtual void EndModeling();
70
71protected:
72
73 // In the derived class, override G4VScenehandler::RequestPrimitives
74 // to implement dump of the geometry hierarchy.
75 static G4int fSceneIdCount; // Counter for Tree scene handlers.
76 const G4Transform3D* fpCurrentObjectTransformation;
77 std::set<G4LogicalVolume*> fDrawnLVStore; // Stores encountered LVs.
78};

82#endif
