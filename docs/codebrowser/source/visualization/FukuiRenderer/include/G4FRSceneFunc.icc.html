37//----- Add polyline
38void G4FRSCENEHANDLER::AddPrimitive (const G4Polyline& polyline)
39{
40#if defined DEBUG_FR_SCENE
41 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
42 G4cout << "***** AddPrimitive\n";
43#endif
44 if (fProcessing2D) {
45 static G4bool warned = false;
46 if (!warned) {
47 warned = true;
48 G4Exception
49 ("G4FRSCENEHANDLER::AddPrimitive (const G4Polyline&)",
50 "dawn0001", JustWarning,
51 "2D polylines not implemented. Ignored.");
52 }
53 return;
54 }
55 //----- Initialize Fukui Renderer IF NECESSARY
56 FRBeginModeling();
57
58 //----- local working variables
59 G4int nPoints = polyline.size ();
60 G4int i ;
61 const G4VisAttributes* pVA =
62 fpViewer->GetApplicableVisAttributes ( polyline.GetVisAttributes() );
63
64 //----- skip drawing invisible primitive
65 if( pVA ){
66 if( !(pVA->IsVisible()) ) { return ; }
67 }
68
69 //----- Attributes
70 if(!SendVisAttributes( pVA ) ) {
71 SendStr( FR_COLOR_RGB_RED ); // color
72 }
73
74 //----- send coordinates to Fukui Renderer
75 SendTransformedCoordinates();
76
77 //----- send beginning of polyline
78 SendStr( FR_POLYLINE );
79
80 //----- vertices on polyline
81 for ( i = 0; i < nPoints; i++ ) {
82 SendStrDouble3( FR_PL_VERTEX , \
83 polyline[i].x(), \
84 polyline[i].y(), \
85 polyline[i].z() );
86 }
87
88 //----- send ending of polyline
89 SendStr( FR_END_POLYLINE );
90
91} // G4FRSCENEHANDLER::AddPrimitive (polyline)

94//----- Add text
95void G4FRSCENEHANDLER::AddPrimitive ( const G4Text& text )
96{
97 //-----
98#if defined DEBUG_FR_SCENE
99 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
100 G4cout << "***** AddPrimitive( G4Text )\n";
101#endif
102 //----- Initialize DAWN IF NECESSARY
103 FRBeginModeling();
104
105 //----- send color
106 const G4Color& color = GetTextColor (text) ;
107 SendStrDouble3( FR_COLOR_RGB ,
108 color.GetRed (),
109 color.GetGreen(),
110 color.GetBlue () );
111
112 //----- send body coordinates
113 SendTransformedCoordinates();
114
115 //----- Calc size
116 MarkerSizeType size_type;
117 G4double fontsize = GetMarkerDiameter( text , size_type );
118
119 //----- Calc position
120 const G4Point3D& position = text.GetPosition () ;
121
122 //----- offset
123 G4double x_offset = text.GetXOffset();
124 G4double y_offset = text.GetYOffset();
125
126 //----- get string to be visualized and Calc its length
127 const char* vis_text = text.GetText();
128 const int STR_LENGTH = strlen ( vis_text );
129
130 //----- create buffer and copy the string there
131 int MAX_STR_LENGTH = COMMAND_BUF_SIZE - 100 ;
132 if ( MAX_STR_LENGTH <= 0 ) {
133 if (G4VisManager::GetVerbosity() >= G4VisManager::errors) {
134 G4cout << "ERROR (FukuiRenderer) : Not enough buffer size for data transferring." << G4endl;
135 G4cout << " G4Text Visualization is aborted" << G4endl;
136 }
137 return ;
138 }
139 char* buf = new char [ (MAX_STR_LENGTH + 1) ] ;
140 if ( MAX_STR_LENGTH >= STR_LENGTH ) {
141 strcpy ( buf, vis_text ) ;
142 } else {
143 strncpy ( buf, vis_text, MAX_STR_LENGTH ) ;
144 }
145
146 //----- select string command for 3D drawing
147 char text_command[32];
148 switch (size_type) {
149 case world:
150 strcpy ( text_command, FR_MARK_TEXT_2D );
151 break;
152 case screen:
153 default:
154 strcpy ( text_command, FR_MARK_TEXT_2DS );
155 break;
156 }
157
158 //----- Send string command
159 if (fProcessing2D) {
160 // Map -1<x<1, -1<y<1 to 10<x<200, 53<y<243
161 G4double x_mm = 95. * position.x() + 105.;
162 G4double y_mm = 95. * position.y() + 148.;
163 SendStrDouble3Str( FR_TEXT_2DS,
164 x_mm, y_mm ,
165 fontsize ,
166 buf );
167 } else {
168 SendStrDouble6Str( text_command, \
169 position.x() , position.y() , position.z(),
170 fontsize , x_offset , y_offset ,
171 buf );
172 }
173
174 //----- delete buffer
175 delete [] buf ;
176
177} // G4FRSCENEHANDLER::AddPrimitive ( text )

292//----- Add polyhedron
293void G4FRSCENEHANDLER::AddPrimitive ( const G4Polyhedron& polyhedron )
294{
295 //-----
296#if defined DEBUG_FR_SCENE
297 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
298 G4cout << "***** AddPrimitive( G4Polyhedron )\n";
299#endif
300 if (fProcessing2D) {
301 static G4bool warned = false;
302 if (!warned) {
303 warned = true;
304 G4Exception
305 ("G4FRSCENEHANDLER::AddPrimitive (const G4Polyhedron&)",
306 "dawn0004", JustWarning,
307 "2D polyhedrons not implemented. Ignored.");
308 }
309 return;
310 }
311
312 if (polyhedron.GetNoFacets() == 0) return;
313
314 //----- Initialize Fukui Renderer IF NECESSARY
315 FRBeginModeling();
316
317 //----- Attributes
318 if(!SendVisAttributes( fpViewer->GetApplicableVisAttributes
319 (polyhedron.GetVisAttributes() ) ) ) {
320 SendStr( FR_COLOR_RGB_RED ); // color
321 }
322
323 //----- Coordinates
324 SendTransformedCoordinates();
325
326 //----- Brep data
327
328 //---------- (1) Declare beginning of Brep data
329 SendStr(FR_POLYHEDRON);
330
331 //---------- (2) Vertex block
332 for (G4int i = 1, j = polyhedron.GetNoVertices(); j; j--, i++){
333 G4Point3D point = polyhedron.GetVertex(i);
334 SendStrDouble3( FR_VERTEX, point.x (), point.y (), point.z ());
335 }
336
337 //---------- (3) Facet block
338 for (G4int f = polyhedron.GetNoFacets(); f; f--){
339 G4int notLastEdge;
340 G4int index = -1; // initialization
341 G4int edgeFlag = 1;
342 //G4int preedgeFlag = 1; Not used - comment out to prevent warnings (JA).
343 G4int work[4], i = 0;
344 do {
345 //preedgeFlag = edgeFlag; Not used - comment out to prevent warnings (JA).
346 notLastEdge = polyhedron.GetNextVertexIndex(index, edgeFlag);
347 work[i++] = index;
348 }while (notLastEdge);
349 switch (i){
350 case 3:
351 SendStrInt3(FR_FACET, work[0], work[1], work[2] );
352 break;
353 case 4:
354 SendStrInt4(FR_FACET, work[0], work[1], work[2], work[3] );
355 break;
356 default:
357 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
358 G4cout <<
359 "ERROR G4FRSCENEHANDLER::AddPrimitive(G4Polyhedron)\n";
360 G4PhysicalVolumeModel* pPVModel =
361 dynamic_cast<G4PhysicalVolumeModel*>(fpModel);
362 if (pPVModel)
363 if (G4VisManager::GetVerbosity() >= G4VisManager::errors) {
364 G4cout <<
365 "Volume " << pPVModel->GetCurrentPV()->GetName() <<
366 ", Solid " << pPVModel->GetCurrentLV()->GetSolid()->GetName() <<
367 " (" << pPVModel->GetCurrentLV()->GetSolid()->GetEntityType();
368 G4cout <<
369 "\nG4Polyhedron facet with " << i << " edges" << G4endl;
370 }
371 }
372 }
373
374 //---------- (4) Declare ending of Brep data
375 SendStr(FR_END_POLYHEDRON);
376
377} // G4FRSCENEHANDLER::AddPrimitive (polyhedron)

382{

384//#if defined DEBUG_FR_SCENE
385// G4cout << "***** SendNdiv() (/Ndiv)" << G4endl;
386//#endif
387//////////////////////////////////////////////////
388
389 //----- local
390 G4int num_division = FR_DEFALUT_NDIV_VALUE ;
391
392 //----- number used for dividing a curved surface, Ndiv
393 // if ( GetModel() ) { ?? Why test for model. Can be zero. JA ??
394 const G4VisAttributes* pVisAttribs =
395 fpViewer -> GetApplicableVisAttributes (fpVisAttribs);
396 num_division = GetNoOfSides(pVisAttribs);
397 // } else {
398#if defined DEBUG_FR_SCENE
399 if (G4VisManager::GetVerbosity() >= G4VisManager::errors) {
400 G4cout << "WARNING: GetNoOfSides() failed. " ;
401 G4cout << "The default value " << num_division ;
402 G4cout << " is assigned." << G4endl;
403 }
404#endif
405 //}
406
407 //---------- Error recovery for too small Ndiv
408 num_division = ( num_division < 3 ? 3 : num_division );
409
410//////////////////////////////////////////////////
411//#if defined DEBUG_FR_SCENE
412// G4cout << "Ndiv = " << num_division << G4endl;
413//#endif
414//////////////////////////////////////////////////
415
416 //----- Send resultant Ndiv
417 this->SendStrInt( FR_NDIV, num_division );

716//----- Add para
717void G4FRSCENEHANDLER::AddSolid (const G4Para& para)
718{
719#if defined DEBUG_FR_SCENE
720 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
721 G4cout << "***** AddSolid ( para )\n";
722#endif
723
724 //----- skip drawing invisible primitive
725 if( !IsVisible() ) { return ; }
726
727 //----- Initialize Fukui Renderer IF NECESSARY
728 FRBeginModeling();
729
730 //----- Send Name
731 SendPhysVolName();
732
733 //----- Send Ndiv
734 // SendNdiv();
735
736 //----- Attributes
737 if(!SendVisAttributes
738 ( fpViewer->GetApplicableVisAttributes( fpVisAttribs ) ) ) {
739 SendStr( FR_COLOR_RGB_RED ); // color
740 }
741
742 //----- local
743 const G4double epsilon = 1.0e-5 ;
744
745 //----- parameters preprocessing
746 G4double cosTheta = para.GetSymAxis().z() ;
747 if( cosTheta < epsilon ) {
748 if (G4VisManager::GetVerbosity() >= G4VisManager::errors) {
749 G4cout << "WARNING from FukuiRenderer (DAWN) driver:" << G4endl;
750 G4cout << " Invalid parameter for parallelepiped." << G4endl;
751 G4cout << " Drawing is skipped." << G4endl;
752 }
753 return ;
754 }
755 G4double tanTheta_cosPhi_cosTheta = para.GetSymAxis().x() ;
756 G4double tanTheta_sinPhi_cosTheta = para.GetSymAxis().y() ;
757
758 //----- parameters
759 G4double dx = para.GetXHalfLength ();
760 G4double dy = para.GetYHalfLength ();
761 G4double dz = para.GetZHalfLength ();
762 G4double tanAlpha = para.GetTanAlpha();

775//----- Add trap
776void G4FRSCENEHANDLER::AddSolid (const G4Trap& trap)
777{
778#if defined DEBUG_FR_SCENE
779 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
780 G4cout << "***** AddSolid ( trap )\n";
781#endif
782
783 //----- skip drawing invisible primitive
784 if( !IsVisible() ) { return ; }
785
786 //----- Initialize Fukui Renderer IF NECESSARY
787 FRBeginModeling();
788
789 //----- Send Name
790 SendPhysVolName();
791
792 //----- Send Ndiv
793 // SendNdiv();
794
795 //----- Attributes
796 if(!SendVisAttributes
797 ( fpViewer->GetApplicableVisAttributes( fpVisAttribs ) ) ) {
798 SendStr( FR_COLOR_RGB_GREEN ); // color
799 }
800
801 //----- local
802 const G4double epsilon = 1.0e-5 ;
803
804 //----- parameters preprocessing
805 G4double cosTheta = trap.GetSymAxis().z() ;
806 if( cosTheta < epsilon ) {
807 if (G4VisManager::GetVerbosity() >= G4VisManager::errors) {
808 G4cout << "WARNING from FukuiRenderer (DAWN) driver:" << G4endl;
809 G4cout << " Invalid parameter for trap, 1" << G4endl;
810 G4cout << " Drawing is skipped." << G4endl;
811 }
812 return ;
813 }
814
815 G4double nx = trap.GetSymAxis().x() ;
816 G4double ny = trap.GetSymAxis().y() ;
817
818 //----- parameters (half lengths of box)
819 G4double dz = trap.GetZHalfLength ();
820 G4double theta = std::acos( cosTheta ) ;
821 G4double phi;
822 if ( ny==0. && nx ==0.) {
823 phi = 0.; // std::atan2(0.,0.) gives undefined value of phi
824 } else {
825 phi = std::atan2( ny, nx ) ; if( phi < 0. ) { phi += CLHEP::twopi ; }
826 // -PI < std::atan() < PI
827 }
828/////////////////////////////////////////////////
829// G4double phi = std::atan2( ny, nx ) ;
830// if( phi < 0.0 ) { phi += CLHEP::twopi ; }
831// // -PI < std::atan() < PI
832/////////////////////////////////////////////////
833
834 G4double h1 = trap.GetYHalfLength1 ();
835 G4double bl1 = trap.GetXHalfLength1 ();
836 G4double tl1 = trap.GetXHalfLength2 ();
837 G4double alpha1 = std::atan( trap.GetTanAlpha1()) ;
838 G4double h2 = trap.GetYHalfLength2 ();
839 G4double bl2 = trap.GetXHalfLength3 ();
840 G4double tl2 = trap.GetXHalfLength4 ();
841 G4double alpha2 = std::atan( trap.GetTanAlpha2()) ;
842
843 //----- send coordinates to Fukui Renderer
844 SendTransformedCoordinates();
845
846 //----- Change sign of alpha for compatibility
847 // with the DAWN format
848 G4double alpha_sign = -1.0 ;
849 alpha1 *= alpha_sign ; alpha2 *= alpha_sign ;
850
851 //----- send box to Fukui Renderer
852 SendStrDouble11( FR_TRAP ,
853 dz ,
854 theta ,
855 phi ,
856 h1 ,
857 bl1 ,
858 tl1 ,
859 alpha1 ,
860 h2 ,
861 bl2 ,
862 tl2 ,
863 alpha2 );
864
865} // G4FRSCENEHANDLER::AddSolid (const G4Trap& trap)

1055

1057 SendStrDouble6( FR_BASE_VECTOR , \
1058 (x_unit_vec.x()), (x_unit_vec.y()), (x_unit_vec.z()) , \
1059 (y_unit_vec.x()), (y_unit_vec.y()), (y_unit_vec.z()) ) ;
