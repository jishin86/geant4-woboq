36G4HepRepFileViewer::G4HepRepFileViewer
37(G4VSceneHandler& sceneHandler, const G4String& name):
38 G4VViewer(sceneHandler, sceneHandler.IncrementViewCount(), name) {
39 hepRepXMLWriter = ((G4HepRepFileSceneHandler*)(&sceneHandler))->GetHepRepXMLWriter();
40 // Make changes to view parameters for HepRep...
41 fVP.SetCulling(false);
42 fDefaultVP.SetCulling(false);
43}

45G4HepRepFileViewer::~G4HepRepFileViewer() {
46 ShowView ();
47}

76 hepRepXMLWriter->close();
