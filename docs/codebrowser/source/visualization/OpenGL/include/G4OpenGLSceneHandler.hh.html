58 void AddPrimitive (const G4Polymarker&);
59 void AddPrimitive (const G4Text&);
60 void AddPrimitive (const G4Circle&);
61 void AddPrimitive (const G4Square&);
62 void AddPrimitivesCircle (const std::vector <G4VMarker>&);
63 void AddPrimitivesSquare (const std::vector <G4VMarker>&);
64 void AddPrimitive (const G4Scale&);

67 void PreAddSolid (const G4Transform3D& objectTransformation,
68 const G4VisAttributes&);
69 void AddSolid (const G4Box&);

71 void AddSolid (const G4Tubs&);

88 // enum for /vis/ogl/flushAt.
89 enum FlushAction {
90 endOfEvent,
91 endOfRun,
92 eachPrimitive,
93 NthPrimitive,
94 NthEvent,
95 never
96 };
97 static G4int GetEntitiesFlushInterval()
98 {return fEntitiesFlushInterval;}
99 static FlushAction GetFlushAction()
100 {return fFlushAction;}
101 static void SetEntitiesFlushInterval(G4int interval)
102 {fEntitiesFlushInterval = interval;}
103 static void SetFlushAction(FlushAction action)
104 {fFlushAction = action;}
105
106#ifdef G4OPENGL_VERSION_2
107 private :
108 // vertex vector to be given to the graphic card
109 std::vector<double> fOglVertex;
110 // indices vector to be given to the graphic card
111 std::vector<unsigned short> fOglIndices;
112 // before, drawyType (as GL_QUADS, GL_TRIANGLES...) was
113 // given in glBegin. Now it has to be given in glDrawArray (at the end)
114 GLenum fDrawArrayType;
115 // emulate GL_QUADS behaviour by inverting two last positions
116 bool fEmulate_GL_QUADS;
117 // Try to optimize a bit the pipeline
118 void OptimizeVBOForTrd();
119 void OptimizeVBOForCons(G4int aNoFacet);
120 // emulating glEnd and glBegin
121 void glEndVBO();
122 void glBeginVBO(GLenum type);
123 void drawVBOArray(std::vector<double> vertices);
124
125 // Buffers used to access vertex and indices elements
126#if defined (G4VIS_BUILD_OPENGLWT_DRIVER) || defined (G4VIS_USE_OPENGLWT)
127 Wt::WGLWidget::Buffer fVertexBufferObject;
128 Wt::WGLWidget::Buffer fIndicesBufferObject;
129#else
130 GLuint fVertexBufferObject;
131 GLuint fIndicesBufferObject;
132#endif // G4VIS_BUILD_OPENGLWT_DRIVER
133
134#endif //G4OPENGL_VERSION_2

136protected:
137
138 G4OpenGLSceneHandler (G4VGraphicsSystem& system,
139 G4int id,
140 const G4String& name = "");
141 virtual ~G4OpenGLSceneHandler ();
142
143 void ProcessScene();
144 G4VSolid* CreateSectionSolid ();
145 G4VSolid* CreateCutawaySolid ();
146
147 void ClearAndDestroyAtts(); // Destroys att holders and clears pick map.
148
149#ifdef G4OPENGL_VERSION_2
150 // Special case for VBO, we want to have acces to the VBO drawer everywhere
151 // because instead of OpenGL call which are static, VBO openGL functions :
152 // - Are functions of an WGLWidget object(G4OpenGLImmediateViewer in our case)
153 // - Needs an access to the QGLSHader
154 // - Have to be redefined in a VBO way
155
156 inline void setVboDrawer(G4OpenGLVboDrawer* drawer) {
157 fVboDrawer = drawer;
158 }
159 G4OpenGLVboDrawer* fVboDrawer;
160#endif // G4OPENGL_VERSION_2
161
162 GLuint fPickName;
163 std::map<GLuint, G4AttHolder*> fPickMap; // For picking.
164
165 // Shared code to wait until we make a single glFlush
166 void ScaledFlush () ;
167 // Static so that they apply to all OGL scene handlers...
168 static FlushAction fFlushAction;
169 // Number of entities between flushes
170 static G4int fEntitiesFlushInterval;
171
172 // True if caller of primitives is capable of processing three passes.
173 G4bool fThreePassCapable;
174
175 G4bool fSecondPassForTransparencyRequested;
176 G4bool fSecondPassForTransparency;
177
178 G4bool fThirdPassForNonHiddenMarkersRequested;
179 G4bool fThirdPassForNonHiddenMarkers;
180
181 static const GLubyte fStippleMaskHashed [128];
182 bool fEdgeFlag;
