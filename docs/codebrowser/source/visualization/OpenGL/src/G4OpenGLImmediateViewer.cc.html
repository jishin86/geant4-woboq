38
39G4OpenGLImmediateViewer::G4OpenGLImmediateViewer (G4OpenGLImmediateSceneHandler& scene):
40G4VViewer (scene, -1),
41G4OpenGLViewer (scene)
42{}
43
44void G4OpenGLImmediateViewer::ProcessView ()
45{
46 const G4Planes& cutaways = fVP.GetCutawayPlanes();
47 G4bool cutawayUnion = fVP.IsCutaway() &&
48 fVP.GetCutawayMode() == G4ViewParameters::cutawayUnion;
49 size_t nPasses = cutawayUnion? cutaways.size(): 1;
50 for (size_t i = 0; i < nPasses; ++i) {
51
52 if (cutawayUnion) {
53 double a[4];
54 a[0] = cutaways[i].a();
55 a[1] = cutaways[i].b();
56 a[2] = cutaways[i].c();
57 a[3] = cutaways[i].d();
58 glClipPlane (GL_CLIP_PLANE2, a);
59 glEnable (GL_CLIP_PLANE2);
60 }
61
62 NeedKernelVisit (); // Always need to visit G4 kernel.
63 G4VViewer::ProcessView ();
64
65 if (cutawayUnion) glDisable (GL_CLIP_PLANE2);
66 }
67}

69#endif
