43G4VisCommandsTouchableSet::G4VisCommandsTouchableSet()
44{
45 G4bool omitable;
46 G4UIparameter* parameter;
47
48 fpCommandSetColour = new G4UIcommand
49 ("/vis/touchable/set/colour", this);
50 fpCommandSetColour->SetGuidance("Set colour of current touchable.");
51 fpCommandSetColour->SetGuidance
52 ("Use \"/vis/set/touchable\" to set current touchable.");
53 fpCommandSetColour->SetGuidance(ConvertToColourGuidance());
54 parameter = new G4UIparameter("red", 's', omitable = true);
55 parameter->SetDefaultValue("1.");
56 fpCommandSetColour->SetParameter(parameter);
57 parameter = new G4UIparameter("green", 'd', omitable = true);
58 parameter->SetDefaultValue(1.);
59 fpCommandSetColour->SetParameter(parameter);
60 parameter = new G4UIparameter("blue", 'd', omitable = true);
61 parameter->SetDefaultValue(1.);
62 fpCommandSetColour->SetParameter(parameter);
63 parameter = new G4UIparameter("opacity", 'd', omitable = true);
64 parameter->SetDefaultValue(1.);
65 fpCommandSetColour->SetParameter(parameter);
66
67 fpCommandSetDaughtersInvisible = new G4UIcmdWithABool
68 ("/vis/touchable/set/daughtersInvisible", this);
69 fpCommandSetDaughtersInvisible->SetGuidance
70 ("Daughters of current touchable invisible: true/false.");
71 fpCommandSetDaughtersInvisible->SetGuidance
72 ("Use \"/vis/set/touchable\" to set current touchable.");
73 fpCommandSetDaughtersInvisible->SetParameterName("daughtersInvisible", omitable = true);
74 fpCommandSetDaughtersInvisible->SetDefaultValue(true);
75
76 fpCommandSetForceAuxEdgeVisible = new G4UIcmdWithABool
77 ("/vis/touchable/set/forceAuxEdgeVisible", this);
78 fpCommandSetForceAuxEdgeVisible->SetGuidance
79 ("Force auxiliary (soft) edges of current touchable to be visible: true/false.");
80 fpCommandSetForceAuxEdgeVisible->SetGuidance
81 ("Use \"/vis/set/touchable\" to set current touchable.");
82 fpCommandSetForceAuxEdgeVisible->SetParameterName("forceAuxEdgeVisible", omitable = true);
83 fpCommandSetForceAuxEdgeVisible->SetDefaultValue(true);
84
85 fpCommandSetLineSegmentsPerCircle = new G4UIcmdWithAnInteger
86 ("/vis/touchable/set/lineSegmentsPerCircle", this);
87 fpCommandSetLineSegmentsPerCircle->SetGuidance
88 ("For current touchable, set number of line segments per circle, the"
89 "\nprecision with which a curved line or surface is represented by a"
90 "\npolygon or polyhedron, regardless of the view parameters."
91 "\nNegative to pick up G4Polyhedron default value.");
92 fpCommandSetLineSegmentsPerCircle->SetGuidance
93 ("Use \"/vis/set/touchable\" to set current touchable.");
94 fpCommandSetLineSegmentsPerCircle->SetParameterName("lineSegmentsPerCircle", omitable = true);
95 fpCommandSetLineSegmentsPerCircle->SetDefaultValue(0);
96
97 fpCommandSetForceSolid = new G4UIcmdWithABool
98 ("/vis/touchable/set/forceSolid", this);
99 fpCommandSetForceSolid->SetGuidance
100 ("Force current touchable always to be drawn solid (surface drawing).");
101 fpCommandSetForceSolid->SetGuidance
102 ("Use \"/vis/set/touchable\" to set current touchable.");
103 fpCommandSetForceSolid->SetParameterName("forceSolid", omitable = true);
104 fpCommandSetForceSolid->SetDefaultValue(true);
105
106 fpCommandSetForceWireframe = new G4UIcmdWithABool
107 ("/vis/touchable/set/forceWireframe", this);
108 fpCommandSetForceWireframe->SetGuidance
109 ("Force current touchable always to be drawn as wireframe.");
110 fpCommandSetForceWireframe->SetGuidance
111 ("Use \"/vis/set/touchable\" to set current touchable.");
112 fpCommandSetForceWireframe->SetParameterName("forceWireframe", omitable = true);
113 fpCommandSetForceWireframe->SetDefaultValue(true);
114
115 fpCommandSetLineStyle = new G4UIcmdWithAString
116 ("/vis/touchable/set/lineStyle", this);
117 fpCommandSetLineStyle->SetGuidance("Set line style of current touchable drawing.");
118 fpCommandSetLineStyle->SetGuidance
119 ("Use \"/vis/set/touchable\" to set current touchable.");
120 fpCommandSetLineStyle->SetParameterName("lineStyle", omitable = true);
121 fpCommandSetLineStyle->SetCandidates("unbroken dashed dotted");
122 fpCommandSetLineStyle->SetDefaultValue("unbroken");
123
124 fpCommandSetLineWidth = new G4UIcmdWithADouble
125 ("/vis/touchable/set/lineWidth", this);
126 fpCommandSetLineWidth->SetGuidance("Set line width of current touchable.");
127 fpCommandSetLineWidth->SetGuidance
128 ("Use \"/vis/set/touchable\" to set current touchable.");
129 fpCommandSetLineWidth->SetParameterName("lineWidth", omitable = true);
130 fpCommandSetLineWidth->SetDefaultValue(1.);
131
132 fpCommandSetVisibility = new G4UIcmdWithABool
133 ("/vis/touchable/set/visibility", this);
134 fpCommandSetVisibility->SetGuidance
135 ("Set visibility of current touchable: true/false.");
136 fpCommandSetVisibility->SetGuidance
137 ("Use \"/vis/set/touchable\" to set current touchable.");
138 fpCommandSetVisibility->SetParameterName("visibility", omitable = true);
139 fpCommandSetVisibility->SetDefaultValue(true);
140}

143 delete fpCommandSetVisibility;
144 delete fpCommandSetLineWidth;
145 delete fpCommandSetLineStyle;
146 delete fpCommandSetForceWireframe;
147 delete fpCommandSetForceSolid;
148 delete fpCommandSetLineSegmentsPerCircle;

151 delete fpCommandSetColour;
152}

154G4String G4VisCommandsTouchableSet::GetCurrentValue(G4UIcommand*) {
155 return "";
156}
157
158void G4VisCommandsTouchableSet::SetNewValue
159(G4UIcommand* command,G4String newValue)
160{
161 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
162
163 G4VViewer* currentViewer = fpVisManager->GetCurrentViewer();
164 if (!currentViewer) {
165 if (verbosity >= G4VisManager::errors) {
166 G4cerr <<
167 "ERROR: G4VisCommandsTouchableSet::SetNewValue: no current viewer."
168 << G4endl;
169 }
170 return;
171 }
172
173 G4ViewParameters workingVP = currentViewer->GetViewParameters();
174 G4VisAttributes workingVisAtts;
175
176 if (command == fpCommandSetColour)
177 {
178 G4String redOrString;
179 G4double green, blue, opacity;
180 std::istringstream iss(newValue);
181 iss >> redOrString >> green >> blue >> opacity;
182 G4Colour colour(1,1,1,1); // Default white and opaque.
183 ConvertToColour(colour, redOrString, green, blue, opacity);
184 workingVisAtts.SetColour(colour);
185 workingVP.AddVisAttributesModifier
186 (G4ModelingParameters::VisAttributesModifier
187 (workingVisAtts,
188 G4ModelingParameters::VASColour,
189 fCurrentTouchableProperties.fTouchablePath));
190 }
191
192 else if (command == fpCommandSetDaughtersInvisible) {
193 workingVisAtts.SetDaughtersInvisible(G4UIcommand::ConvertToBool(newValue));
194 workingVP.AddVisAttributesModifier
195 (G4ModelingParameters::VisAttributesModifier
196 (workingVisAtts,
197 G4ModelingParameters::VASDaughtersInvisible,
198 fCurrentTouchableProperties.fTouchablePath));
199 }
200
201 else if (command == fpCommandSetForceAuxEdgeVisible) {
202 workingVisAtts.SetForceAuxEdgeVisible(G4UIcommand::ConvertToBool(newValue));
203 workingVP.AddVisAttributesModifier
204 (G4ModelingParameters::VisAttributesModifier
205 (workingVisAtts,
206 G4ModelingParameters::VASForceAuxEdgeVisible,
207 fCurrentTouchableProperties.fTouchablePath));
208 }
209
210 else if (command == fpCommandSetLineSegmentsPerCircle) {
211 workingVisAtts.SetForceLineSegmentsPerCircle
212 (G4UIcommand::ConvertToInt(newValue));
213 workingVP.AddVisAttributesModifier
214 (G4ModelingParameters::VisAttributesModifier
215 (workingVisAtts,
216 G4ModelingParameters::VASForceLineSegmentsPerCircle,
217 fCurrentTouchableProperties.fTouchablePath));
218 }
219
220 else if (command == fpCommandSetForceSolid) {
221 workingVisAtts.SetForceSolid(G4UIcommand::ConvertToBool(newValue));
222 workingVP.AddVisAttributesModifier
223 (G4ModelingParameters::VisAttributesModifier
224 (workingVisAtts,
225 G4ModelingParameters::VASForceSolid,
226 fCurrentTouchableProperties.fTouchablePath));
227 }
228
229 else if (command == fpCommandSetForceWireframe) {
230 workingVisAtts.SetForceWireframe(G4UIcommand::ConvertToBool(newValue));
231 workingVP.AddVisAttributesModifier
232 (G4ModelingParameters::VisAttributesModifier
233 (workingVisAtts,
234 G4ModelingParameters::VASForceWireframe,
235 fCurrentTouchableProperties.fTouchablePath));
236 }
237
238 else if (command == fpCommandSetLineStyle) {
239 G4VisAttributes::LineStyle lineStyle = G4VisAttributes::unbroken;
240 if (newValue == "dashed") {
241 lineStyle = G4VisAttributes::dashed;
242 } else if (newValue == "dotted") {
243 lineStyle = G4VisAttributes::dotted;
244 }
245 // All other values are "unbroken".
246 workingVisAtts.SetLineStyle(lineStyle);
247 workingVP.AddVisAttributesModifier
248 (G4ModelingParameters::VisAttributesModifier
249 (workingVisAtts,
250 G4ModelingParameters::VASLineStyle,
251 fCurrentTouchableProperties.fTouchablePath));
252 }
253
254 else if (command == fpCommandSetLineWidth) {
255 workingVisAtts.SetLineWidth(G4UIcommand::ConvertToDouble(newValue));
256 workingVP.AddVisAttributesModifier
257 (G4ModelingParameters::VisAttributesModifier
258 (workingVisAtts,
259 G4ModelingParameters::VASLineWidth,
260 fCurrentTouchableProperties.fTouchablePath));
261 }
262
263 else if (command == fpCommandSetVisibility) {
264 workingVisAtts.SetVisibility(G4UIcommand::ConvertToBool(newValue));
265 workingVP.AddVisAttributesModifier
266 (G4ModelingParameters::VisAttributesModifier
267 (workingVisAtts,
268 G4ModelingParameters::VASVisibility,
269 fCurrentTouchableProperties.fTouchablePath));
270 }
271
272 else {
273 if (verbosity >= G4VisManager::errors) {
274 G4cerr <<
275 "ERROR: G4VisCommandsTouchableSet::SetNewValue: unrecognised command."
276 << G4endl;
277 }
278 return;
279 }
280
281 SetViewParameters(currentViewer,workingVP);
282
283 // To update all views
284 G4UImanager::GetUIpointer()->ApplyCommand("/vis/scene/notifyHandlers");
285}
