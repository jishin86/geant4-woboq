82G4VisCommandSetColour::G4VisCommandSetColour ()
83{
84 G4bool omitable;
85 fpCommand = new G4UIcommand("/vis/set/colour", this);
86 fpCommand->SetGuidance
87 ("Defines colour and opacity for future \"/vis/scene/add/\" commands.");
88 fpCommand->SetGuidance
89 ("(Except \"/vis/scene/add/text\" commands - see \"/vis/set/textColour\".)");
90 fpCommand->SetGuidance(ConvertToColourGuidance());
91 fpCommand->SetGuidance("Default: white and opaque.");
92 G4UIparameter* parameter;
93 parameter = new G4UIparameter ("red", 's', omitable = true);
94 parameter->SetGuidance
95 ("Red component or a string, e.g., \"cyan\" (green and blue parameters are ignored).");
96 parameter->SetDefaultValue ("1.");
97 fpCommand->SetParameter (parameter);
98 parameter = new G4UIparameter ("green", 'd', omitable = true);
99 parameter->SetDefaultValue (1.);
100 fpCommand->SetParameter (parameter);
101 parameter = new G4UIparameter ("blue", 'd', omitable = true);
102 parameter->SetDefaultValue (1.);
103 fpCommand->SetParameter (parameter);
104 parameter = new G4UIparameter ("alpha", 'd', omitable = true);
105 parameter->SetDefaultValue (1.);
106 parameter->SetGuidance ("Opacity");
107 fpCommand->SetParameter (parameter);
108}
109
110G4VisCommandSetColour::~G4VisCommandSetColour ()
111{
112 delete fpCommand;
113}
114
115G4String G4VisCommandSetColour::GetCurrentValue (G4UIcommand*)
116{
117 return G4String();
118}
119
120void G4VisCommandSetColour::SetNewValue (G4UIcommand*, G4String newValue)
121{
122 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
123
124 G4String redOrString;
125 G4double green, blue, opacity;
126 std::istringstream iss(newValue);
127 iss >> redOrString >> green >> blue >> opacity;
128
129 ConvertToColour(fCurrentColour, redOrString, green, blue, opacity);
130
131 if (verbosity >= G4VisManager::confirmations) {
132 G4cout <<
133 "Colour for future \"/vis/scene/add/\" commands has been set to "
134 << fCurrentColour <<
135 ".\n(Except \"/vis/scene/add/text\" commands - use \"/vis/set/textColour\".)"
136 << G4endl;
137 }
138}

264void G4VisCommandSetTextLayout::SetNewValue (G4UIcommand*, G4String newValue)
265{
266 G4Text::Layout layout = G4Text::left;
267 if (newValue == "left") layout = G4Text::left;
268 else if (newValue == "centre" || newValue == "center")
269 layout = G4Text::centre;
270 else if (newValue == "right") layout = G4Text::right;
271
272 fCurrentTextLayout = layout;
273
274 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
275 if (verbosity >= G4VisManager::confirmations) {
276 G4cout << "Text layout (for future \"text\" commands) has been set to \""
277 << fCurrentTextLayout << "\"."
278 << G4endl;
279 }

350void G4VisCommandSetTouchable::SetNewValue (G4UIcommand*, G4String newValue)
351{
352 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
353
354 G4ModelingParameters::PVNameCopyNoPath currentTouchablePath;
355
356 // Algorithm from Josuttis p.476.
357 G4String::size_type iBegin, iEnd;
358 iBegin = newValue.find_first_not_of(' ');
359 while (iBegin != G4String::npos) {
360 iEnd = newValue.find_first_of(' ',iBegin);
361 if (iEnd == G4String::npos) {
362 iEnd = newValue.length();
363 }
364 G4String name(newValue.substr(iBegin,iEnd-iBegin));
365 iBegin = newValue.find_first_not_of(' ',iEnd);
366 if (iBegin == G4String::npos) {
367 if (verbosity >= G4VisManager::warnings) {
368 G4cout <<
369 "WARNING: G4VisCommandSetTouchable::SetNewValue"
370 "\n A pair not found. (Did you have an even number of parameters?)"
371 "\n Command ignored."
372 << G4endl;
373 return;
374 }
375 }
376 iEnd = newValue.find_first_of(' ',iBegin);
377 if (iEnd == G4String::npos) {
378 iEnd = newValue.length();
379 }
380 G4int copyNo;
381 std::istringstream iss(newValue.substr(iBegin,iEnd-iBegin));
382 if (!(iss >> copyNo)) {
383 if (verbosity >= G4VisManager::warnings) {
384 G4cout <<
385 "WARNING: G4VisCommandSetTouchable::SetNewValue"
386 "\n Error reading copy number - it was not numeric?"
387 "\n Command ignored."
388 << G4endl;
389 return;
390 }
391 }
392 currentTouchablePath.push_back
393 (G4ModelingParameters::PVNameCopyNo(name,copyNo));
394 iBegin = newValue.find_first_not_of(' ',iEnd);
395 }
396
397 // Check validity
398 G4bool successful = false;
399 G4TransportationManager* transportationManager =
400 G4TransportationManager::GetTransportationManager ();
401 size_t nWorlds = transportationManager->GetNoWorlds();
402 std::vector<G4VPhysicalVolume*>::iterator iterWorld =
403 transportationManager->GetWorldsIterator();
404 for (size_t i = 0; i < nWorlds; ++i, ++iterWorld) {
405 G4PhysicalVolumeModel pvModel (*iterWorld); // Unlimited depth.
406 G4ModelingParameters mp; // Default - no culling.
407 pvModel.SetModelingParameters (&mp);
408 G4TouchablePropertiesScene scene (&pvModel,currentTouchablePath);
409 pvModel.DescribeYourselfTo (scene); // Initiate geometry tree traversal.
410 if (scene.GetFoundTouchableProperties().fpTouchablePV) {
411 successful = true;
412 fCurrentTouchableProperties = scene.GetFoundTouchableProperties();
413 break; // Found, so no need to scan more worlds.
414 }
415 }
416
417 if (successful) {
418 if (verbosity >= G4VisManager::confirmations) {
419 G4cout <<
420 "Current touchable: " << fCurrentTouchableProperties.fTouchablePath
421 << G4endl;
422 return;
423 }
424 } else {
425 if (verbosity >= G4VisManager::warnings) {
426 G4cout <<
427 "WARNING: G4VisCommandSetTouchable::SetNewValue"
428 "\n Touchable not found."
429 << G4endl;
430 return;
431 }
432 }
