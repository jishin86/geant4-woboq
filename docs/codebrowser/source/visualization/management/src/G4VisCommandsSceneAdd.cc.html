127void G4VisCommandSceneAddArrow::SetNewValue (G4UIcommand*, G4String newValue)
128{
129 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
130 G4bool warn(verbosity >= G4VisManager::warnings);
131
132 G4Scene* pScene = fpVisManager->GetCurrentScene();
133 if (!pScene) {
134 if (verbosity >= G4VisManager::errors) {
135 G4cerr << "ERROR: No current scene. Please create one." << G4endl;
136 }
137 return;
138 }
139
140 G4String unitString;
141 G4double x1, y1, z1, x2, y2, z2;
142 std::istringstream is(newValue);
143 is >> x1 >> y1 >> z1 >> x2 >> y2 >> z2 >> unitString;
144 G4double unit = G4UIcommand::ValueOf(unitString);
145 x1 *= unit; y1 *= unit; z1 *= unit;
146 x2 *= unit; y2 *= unit; z2 *= unit;
147
148 // Consult scene for arrow width.
149 const G4VisExtent& sceneExtent = pScene->GetExtent();
150 G4double arrowWidth =
151 0.005 * fCurrentLineWidth * sceneExtent.GetExtentRadius();
152
153 G4VModel* model = new G4ArrowModel
154 (x1, y1, z1, x2, y2, z2,
155 arrowWidth, fCurrentColour, newValue,
156 fCurrentArrow3DLineSegmentsPerCircle);
157
158 const G4String& currentSceneName = pScene -> GetName ();
159 G4bool successful = pScene -> AddRunDurationModel (model, warn);
160 if (successful) {
161 if (verbosity >= G4VisManager::confirmations) {
162 G4cout << "Arrow has been added to scene \""
163 << currentSceneName << "\"."
164 << G4endl;
165 }
166 }
167 else G4VisCommandsSceneAddUnsuccessful(verbosity);
168
169 CheckSceneAndNotifyHandlers (pScene);
170}

235G4VisCommandSceneAddArrow2D::Arrow2D::Arrow2D
236(G4double x1, G4double y1,
237 G4double x2, G4double y2,
238 G4double width, const G4Colour& colour):
239 fWidth(width), fColour(colour)
240{
241 fShaftPolyline.push_back(G4Point3D(x1,y1,0));
242 fShaftPolyline.push_back(G4Point3D(x2,y2,0));
243 G4Vector3D arrowDirection = G4Vector3D(x2-x1,y2-y1,0).unit();
244 G4Vector3D arrowPointLeftDirection(arrowDirection);
245 arrowPointLeftDirection.rotateZ(150.*deg);
246 G4Vector3D arrowPointRightDirection(arrowDirection);
247 arrowPointRightDirection.rotateZ(-150.*deg);
248 fHeadPolyline.push_back(G4Point3D(x2,y2,0)+0.04*arrowPointLeftDirection);
249 fHeadPolyline.push_back(G4Point3D(x2,y2,0));
250 fHeadPolyline.push_back(G4Point3D(x2,y2,0)+0.04*arrowPointRightDirection);
251 G4VisAttributes va;
252 va.SetLineWidth(fWidth);
253 va.SetColour(fColour);
254 fShaftPolyline.SetVisAttributes(va);
255 fHeadPolyline.SetVisAttributes(va);
256}

316void G4VisCommandSceneAddAxes::SetNewValue (G4UIcommand*, G4String newValue) {
317
318 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
319 G4bool warn(verbosity >= G4VisManager::warnings);
320
321 G4Scene* pScene = fpVisManager->GetCurrentScene();
322 if (!pScene) {
323 if (verbosity >= G4VisManager::errors) {
324 G4cerr << "ERROR: No current scene. Please create one." << G4endl;
325 }
326 return;
327 } else {
328 if (pScene->GetExtent().GetExtentRadius() <= 0.) {
329 if (verbosity >= G4VisManager::errors) {
330 G4cerr
331 << "ERROR: Scene has no extent. Add volumes or use \"/vis/scene/add/extent\"."
332 << G4endl;
333 }
334 return;
335 }
336 }
337
338 G4String unitString, colourString, showTextString;
339 G4double x0, y0, z0, length;
340 std::istringstream is (newValue);
341 is >> x0 >> y0 >> z0 >> length >> unitString
342 >> colourString >> showTextString;
343 G4bool showText = G4UIcommand::ConvertToBool(showTextString);
344
345
346 G4double unit = G4UIcommand::ValueOf(unitString);
347 x0 *= unit; y0 *= unit; z0 *= unit;
348 const G4VisExtent& sceneExtent = pScene->GetExtent(); // Existing extent.
349 if (length < 0.) {
350 const G4double lengthMax = 0.5 * sceneExtent.GetExtentRadius();
351 const G4double intLog10Length = std::floor(std::log10(lengthMax));
352 length = std::pow(10,intLog10Length);
353 if (5.*length < lengthMax) length *= 5.;
354 else if (2.*length < lengthMax) length *= 2.;
355 } else {
356 length *= unit;
357 }
358
359 // Consult scene for arrow width...
360 G4double arrowWidth =
361 0.005 * fCurrentLineWidth * sceneExtent.GetExtentRadius();
362 // ...but limit it to length/50.
363 if (arrowWidth > length/50.) arrowWidth = length/50.;
364
365 G4VModel* model = new G4AxesModel
366 (x0, y0, z0, length, arrowWidth, colourString, newValue,
367 showText, fCurrentTextSize);
368
369 G4bool successful = pScene -> AddRunDurationModel (model, warn);
370 const G4String& currentSceneName = pScene -> GetName ();
371 if (successful) {
372 if (verbosity >= G4VisManager::confirmations) {
373 G4cout << "Axes of length " << G4BestUnit(length,"Length")
374 << "have been added to scene \"" << currentSceneName << "\"."
375 << G4endl;
376 }
377 }
378 else G4VisCommandsSceneAddUnsuccessful(verbosity);
379
380 CheckSceneAndNotifyHandlers (pScene);
381}

422void G4VisCommandSceneAddDate::SetNewValue (G4UIcommand*, G4String newValue)
423{
424 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
425 G4bool warn(verbosity >= G4VisManager::warnings);

448 else if (layoutString(0) == 'c') layout = G4Text::centre;

453 new G4CallbackModel<G4VisCommandSceneAddDate::Date>(date);
454 model->SetType("Date");

456 model->SetGlobalDescription("Date");
457 const G4String& currentSceneName = pScene -> GetName ();
458 G4bool successful = pScene -> AddRunDurationModel (model, warn);
459 if (successful) {
460 if (verbosity >= G4VisManager::confirmations) {
461 G4cout << "Date has been added to scene \""

471void G4VisCommandSceneAddDate::Date::operator()
472 (G4VGraphicsScene& sceneHandler, const G4Transform3D&)
473{
474 G4String time;
475 if (fDate == "-") {
476 time = fTimer.GetClockTime();
477 } else {
478 time = fDate;
479 }
480 // Check for \n, starting from back, and erase.
481 std::string::size_type i = time.rfind('\n');
482 if (i != std::string::npos) time.erase(i);
483 G4Text text(time, G4Point3D(fX, fY, 0.));
484 text.SetScreenSize(fSize);
485 text.SetLayout(fLayout);
486 G4VisAttributes textAtts(G4Colour(0.,1.,1));
487 text.SetVisAttributes(textAtts);
488 sceneHandler.BeginPrimitives2D();
489 sceneHandler.AddPrimitive(text);
490 sceneHandler.EndPrimitives2D();
491}

624#ifdef G4MULTITHREADED
625 if (G4Threading::IsMultithreadedApplication()) {
626 runManager = G4MTRunManager::GetMasterRunManager();
627 }
628#endif
629 if (runManager) currentRun = runManager->GetCurrentRun();
630
631 G4VModel* model = fpVisManager->GetCurrentSceneHandler()->GetModel();
632 const G4ModelingParameters* mp = 0;
633 const G4Event* currentEvent = 0;

635 mp = model->GetModelingParameters();
636 currentEvent = mp->GetEvent();
637 } else {
638 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();

640 G4cerr << "ERROR: No model defined for this SceneHandler : "
641 << fpVisManager->GetCurrentSceneHandler()->GetName()
642 << G4endl;

649 if (fpVisManager->GetCurrentScene()->GetRefreshAtEndOfEvent()) {
650 oss << "Run " << runID << " Event " << eventID;
651 } else {
652 G4int nEvents = 0;
653#ifdef G4MULTITHREADED
654 if (G4Threading::G4GetThreadId() != G4Threading::MASTER_ID) {

659 if (state == G4State_EventProc) {

666 currentRun->GetEventVector();

671 // drawing the event ID. But for MT mode there is no way of

674 if (eventID < nEvents - 1) return; // Not last event.
675#endif

677 if (nEvents != 1) oss << 's';
678 oss << ')';

681 text.SetScreenSize(fSize);

1651G4VisCommandSceneAddLogo::G4Logo::G4Logo
1652(G4double height, const G4VisAttributes& visAtts):
1653 fVisAtts(visAtts)
1654{
1655 const G4double& h = height;
1656 const G4double h2 = 0.5 * h; // Half height.
1657 const G4double ri = 0.25 * h; // Inner radius.
1658 const G4double ro = 0.5 * h; // Outer radius.
1659 const G4double ro2 = 0.5 * ro; // Half outer radius.
1660 const G4double w = ro - ri; // Width.
1661 const G4double w2 = 0.5 * w; // Half width.
1662 const G4double d2 = 0.2 * h; // Half depth.
1663 const G4double f1 = 0.05 * h; // left edge of stem of "4".
1664 const G4double f2 = -0.3 * h; // bottom edge of cross of "4".
1665 const G4double e = 1.e-4 * h; // epsilon.
1666 const G4double xt = f1, yt = h2; // Top of slope.
1667 const G4double xb = -h2, yb = f2 + w; // Bottom of slope.
1668 const G4double dx = xt - xb, dy = yt - yb;
1669 const G4double angle = std::atan2(dy,dx);
1670 G4RotationMatrix rm;
1671 rm.rotateZ(angle*rad);
1672 const G4double d = std::sqrt(dx * dx + dy * dy);
1673 const G4double ss = h; // Half height of square subtractor
1674 const G4double y8 = ss; // Choose y of subtractor for outer slope.
1675 const G4double x8 = ((-ss * d - dx * (yt - y8)) / dy) + xt;
1676 G4double y9 = ss; // Choose y of subtractor for inner slope.
1677 G4double x9 = ((-(ss - w) * d - dx * (yt - y8)) / dy) + xt;
1678 // But to get inner, we make a triangle translated by...
1679 const G4double xtr = ss - f1, ytr = -ss - f2 -w;
1680 x9 += xtr; y9 += ytr;
1681
1682 // The idea here is to create a polyhedron for the G and the 4. To do
1683 // this we use Geant4 geometry solids and make boolean operations.
1684 // Note that we do not need to keep the solids. We use local objects,
1685 // which, of course, are deleted on leaving this function. This
1686 // is contrary to the usual requirement for solids that are part of the
1687 // detector for which solids MUST be created on the heap (with "new").
1688 // Finally we invoke CreatePolyhedron, which creates a polyhedron on the heap
1689 // and returns a pointer. It is the user's responsibility to delete,
1690 // which is done in the destructor of this class. Thus the polyhedra,
1691 // created here, remain on the heap for the lifetime of the job.
1692
1693 // G...
1694 G4Tubs tG("tG",ri,ro,d2,0.15*pi,1.85*pi);
1695 G4Box bG("bG",w2,ro2,d2);
1696 G4UnionSolid logoG("logoG",&tG,&bG,G4Translate3D(ri+w2,-ro2,0.));
1697 fpG = logoG.CreatePolyhedron();
1698 fpG->SetVisAttributes(&fVisAtts);
1699 fpG->Transform(G4Translate3D(-0.55*h,0.,0.));
1700
1701 // 4...
1702 G4Box b1("b1",h2,h2,d2);
1703 G4Box bS("bS",ss,ss,d2+e); // Subtractor.
1704 G4Box bS2("bS2",ss,ss,d2+2.*e); // 2nd Subtractor.
1705 G4SubtractionSolid s1("s1",&b1,&bS,G4Translate3D(f1-ss,f2-ss,0.));
1706 G4SubtractionSolid s2("s2",&s1,&bS,G4Translate3D(f1+ss+w,f2-ss,0.));
1707 G4SubtractionSolid s3("s3",&s2,&bS,G4Translate3D(f1+ss+w,f2+ss+w,0.));
1708 G4SubtractionSolid s4
1709 ("s4",&s3,&bS,G4Transform3D(rm,G4ThreeVector(x8,y8,0.)));
1710 G4SubtractionSolid s5 // Triangular hole.
1711 ("s5",&bS,&bS2,G4Transform3D(rm,G4ThreeVector(x9,y9,0.)));
1712 G4SubtractionSolid logo4("logo4",&s4,&s5,G4Translate3D(-xtr,-ytr,0.));
1713 fp4 = logo4.CreatePolyhedron();
1714 /* Experiment with creating own polyhedron...
1715 int nNodes = 4;
1716 int nFaces = 4;
1717 double xyz[][3] = {{0,0,0},{1*m,0,0},{0,1*m,0},{0,0,1*m}};
1718 int faces[][4] = {{1,3,2,0},{1,2,4,0},{1,4,3,0},{2,3,4,0}};
1719 fp4 = new G4Polyhedron();
1720 fp4->createPolyhedron(nNodes,nFaces,xyz,faces);
1721 */
1722 fp4->SetVisAttributes(&fVisAtts);
1723 fp4->Transform(G4Translate3D(0.55*h,0.,0.));
1724}
