30void G4GraphicsSystemList::remove(G4VGraphicsSystem* graphicsSystem) {
31 G4GraphicsSystemListIterator iGS;
32 for (iGS = begin(); iGS != end(); ++iGS) {
33 if (*iGS == graphicsSystem) break;
34 }
35 if (iGS != end()) erase(iGS);
36}
