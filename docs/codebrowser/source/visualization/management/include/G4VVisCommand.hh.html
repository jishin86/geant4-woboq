32#define G4VVISCOMMAND_HH
33
34#include "G4VisManager.hh"
35#include "G4UImessenger.hh"
36#include "G4ThreeVector.hh"
37#include "G4Text.hh"
38#include "G4VisAttributes.hh"
39#include "G4VMarker.hh"
40#include "G4ModelingParameters.hh"
41#include <vector>
42
43class G4UIcommand;
44class G4UIcmdWithAString;
45
46class G4VVisCommand: public G4UImessenger {
47public:
48 // Uses compiler defaults for copy constructor and assignment.
49 G4VVisCommand ();
50 virtual ~G4VVisCommand ();
51 static void SetVisManager (G4VisManager*);
52 static const G4Colour& GetCurrentColour() {return fCurrentColour;}
53 //static G4VMarker::FillStyle GetCurrentFillStyle() {return fCurrentFillStyle;}
54 //static G4VMarker::SizeType GetCurrentSizeType() {return fCurrentSizeType;}
55 static G4double GetCurrentLineWidth() {return fCurrentLineWidth;}
56 //static G4VisAttributes::LineStyle GetCurrentLineStyle() {return fCurrentLineStyle;}
57 static const G4Colour& GetCurrentTextColour() {return fCurrentTextColour;}
58 static G4Text::Layout GetCurrentTextLayout() {return fCurrentTextLayout;}
59 static G4double GetCurrentTextSize() {return fCurrentTextSize;}
60
61protected:
62
63 // Conversion routines augmenting those in G4UIcommand.
64
65 static G4String ConvertToString(G4double x, G4double y,
66 const char * unitName);
67
68 static G4bool ConvertToDoublePair(const G4String& paramString,
69 G4double& xval,
70 G4double& yval);
71 // Return false if problem parsing paramString.
72
73 const G4String& ConvertToColourGuidance();
74 void ConvertToColour
75 (G4Colour& colour,
76 const G4String& redOrString,
77 G4double green,
78 G4double blue,
79 G4double opacity);
80 // Note: colour is supplied by the caller and becomes the default if the
81 // remaining parameters cannot be parsed.
82 // Note: redOrString is either a number or string. If a string it must be
83 // one of the recognised colours.
84 // Thus the arguments can be, for example:
85 // (colour,"red",...,...,0.5): will give the colour red with opacity 0.5 (the
86 // third and fourth arguments are ignored), or
87 // (1.,0.,0.,0.5): this also will be red with opacity 0.5.
88
89 G4bool ProvideValueOfUnit
90 (const G4String& where,
91 const G4String& unit,
92 const G4String& category,
93 G4double& value);
94 // Return false if there's a problem
95
96 // Other utilities.
97 void CheckSceneAndNotifyHandlers (G4Scene* = nullptr);
98
99 // Data members.
100 static G4VisManager* fpVisManager;
101
102 // Error management
103 static G4int fErrorCode;
104
105 // Current quantities for use in appropriate commands
106 static G4int fCurrentArrow3DLineSegmentsPerCircle;
107 static G4Colour fCurrentColour;
108 static G4double fCurrentLineWidth;
109 //static G4VisAttributes::LineStyle fCurrentLineStyle; Not yet used.
110 //static G4VMarker::FillStyle fCurrentFillStyle; Not yet used.
111 //static G4VMarker::SizeType fCurrentSizeType; Not yet used.
112 static G4Colour fCurrentTextColour;
113 static G4Text::Layout fCurrentTextLayout;
114 static G4double fCurrentTextSize;
115 static G4PhysicalVolumeModel::TouchableProperties fCurrentTouchableProperties;
116};
