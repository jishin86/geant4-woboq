42G4PSHitsModel::G4PSHitsModel (const G4String& requestedMapName):
43 fRequestedMapName(requestedMapName)
44{
45 fType = "G4PSHitsModel";
46 fGlobalTag = "G4PSHitsModel for G4THitsMap<G4StatDouble> hits.";
47 fGlobalDescription = fGlobalTag;
48}
49
50void G4PSHitsModel::DescribeYourselfTo (G4VGraphicsScene& sceneHandler)
51{
52 G4ScoringManager* scoringManager =
53 G4ScoringManager::GetScoringManagerIfExist();
54 if (scoringManager) {
55 size_t nMeshes = scoringManager->GetNumberOfMesh();
56 for (size_t iMesh = 0; iMesh < nMeshes; ++iMesh) {
57 G4VScoringMesh* mesh = scoringManager->GetMesh(iMesh);
58 if (mesh && mesh->IsActive()) {
59 MeshScoreMap scoreMap = mesh->GetScoreMap();
60 for(MeshScoreMap::const_iterator i = scoreMap.begin();
61 i != scoreMap.end(); ++i) {
62 const G4String& name = i->first;
63 if (fRequestedMapName == "all" || name == fRequestedMapName) {
64 RunScore* fpCurrentHits = i->second;
65 //G4cout << name << ": " << fpCurrentHits << G4endl;
66 if (fpCurrentHits) sceneHandler.AddCompound(*fpCurrentHits);
67 }
68 }
69 }
70 }
71 }
