31// Class Description:
32//
33// G4VModel is a base class for visualization models. A model is a
34// graphics-system-indepedent description of a Geant4 component.
35// The key fuctionality of a model is to know how to describe itself
36// to a scene handler. A scene is a collection of models.
37// A special case is made for G4PhysicalVolumeModel - a non-null pointer
38// is to be returned by G4PhysicalVolumeModel::GetG4PhysicalVolumeModel().
39
40#ifndef G4VMODEL_HH
41#define G4VMODEL_HH
42
43#include "globals.hh"
44#include "G4VisExtent.hh"
45#include "G4Transform3D.hh"
46
47class G4VGraphicsScene;
48class G4ModelingParameters;
49
50class G4PhysicalVolumeModel; // Special case - see above.
51
52class G4VModel {
53
54public: // With description
55
56 friend std::ostream& operator << (std::ostream& os, const G4VModel&);
57
58 G4VModel
59 (const G4Transform3D& modelTransformation = G4Transform3D(),
60 const G4ModelingParameters* = 0);
61
62 virtual ~G4VModel ();
63
64 virtual void DescribeYourselfTo (G4VGraphicsScene&) = 0;
65 // The main task of a model is to describe itself to the graphics scene.
66
67 const G4ModelingParameters* GetModelingParameters () const;
68
69 const G4String& GetType() const;
70 // The sub-class should set its type, which could be the class
71 // name, or, in the case of G4CallBackModel, it could be the type of
72 // object it is coding.
73
74 virtual G4String GetCurrentDescription () const;
75 // A description which depends on the current state of the model.
76
77 virtual G4String GetCurrentTag () const;
78 // A tag which depends on the current state of the model.
79
80 const G4VisExtent& GetExtent () const;
81 // Extent of visible objects in local coordinate system.
82
83 const G4String& GetGlobalDescription () const;
84 // A description which does not change and lasts the life of the model.
85
86 const G4String& GetGlobalTag () const;
87 // A tag which does not change and lasts the life of the model.
88
89 const G4Transform3D& GetTransformation () const;
90 // Model transformation, i.e., position and orientation of model in
91 // world. It is the responsibility of the model to apply this
92 // transformation before passing items to the graphics scene.
93
94 // Set methods for above...
95 void SetModelingParameters (const G4ModelingParameters*);
96 void SetExtent (const G4VisExtent&);
97 void SetType (const G4String&);
98 void SetGlobalDescription (const G4String&);
99 void SetGlobalTag (const G4String&);
100 void SetTransformation (const G4Transform3D&);
101
102 virtual G4bool Validate (G4bool warn = true);
103 // Validate, but allow internal changes (hence non-const function).
104
105protected:
106
107 G4String fType;
108 G4String fGlobalTag;
109 G4String fGlobalDescription;
110 G4VisExtent fExtent;
111 G4Transform3D fTransform;
112 const G4ModelingParameters* fpMP;
113
114private:
115
116 // Private copy constructor and assigment operator - copying and
117 // assignment not allowed. Keeps CodeWizard happy.
118 G4VModel (const G4VModel&);
119 G4VModel& operator = (const G4VModel&);
120};
121
122#include "G4VModel.icc"
123
124#endif
