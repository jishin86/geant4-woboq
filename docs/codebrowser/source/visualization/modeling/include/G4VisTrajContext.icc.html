103inline void G4VisTrajContext::Print(std::ostream& ostr) const
104{
105 ostr<<"Name: "<<Name()<<G4endl;
106 ostr<<"Line colour "<<GetLineColour()<<G4endl;
107 ostr<<"Draw line ? "<<GetDrawLine()<<G4endl;
108 ostr<<"Line visibile ? "<<GetLineVisible()<<G4endl;
109 ostr<<"Draw auxiliary points ? "<<GetDrawAuxPts()<<G4endl;
110 ostr<<"Auxiliary points type "<<GetAuxPtsType()<<G4endl;
111 ostr<<"Auxiliary points size "<<GetAuxPtsSize()<<G4endl;
112 ostr<<"Auxiliary points fill style "<<GetAuxPtsFillStyle()<<G4endl;
113 ostr<<"Auxiliary points colour "<<GetAuxPtsColour()<<G4endl;
114 ostr<<"Auxiliary points visible ? "<<GetAuxPtsVisible()<<G4endl;
115 ostr<<"Draw step points ? "<<GetDrawStepPts()<<G4endl;
116 ostr<<"Step points type "<<GetStepPtsType()<<G4endl;
117 ostr<<"Step points size "<<GetStepPtsSize()<<G4endl;
118 ostr<<"Step points fill style "<<GetStepPtsFillStyle()<<G4endl;
119 ostr<<"Step points colour "<<GetStepPtsColour()<<G4endl;
120 ostr<<"Step points visible ? "<<GetStepPtsVisible()<<G4endl;
121 ostr<<"Time slice interval "<<G4BestUnit(GetTimeSliceInterval(),"Time")<<G4endl;
122}
