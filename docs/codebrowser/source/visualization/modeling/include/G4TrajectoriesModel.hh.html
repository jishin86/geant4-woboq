31// Class Description:
32//
33// Model which knows how to draw GEANT4 trajectories.
34//
35// For access to base class information, e.g., modeling parameters,
36// use GetModelingParameters() inherited from G4VModel. See Class
37// Description of the base class G4VModel.
38
39#ifndef G4TRAJECTORIESMODEL_HH
40#define G4TRAJECTORIESMODEL_HH
41
42#include "G4VModel.hh"
43
44#include <vector>
45#include <map>
46
47class G4VTrajectory;
48class G4AttDef;
49class G4AttValue;
50
51class G4TrajectoriesModel: public G4VModel {
52
53public: // With description
54
55 G4TrajectoriesModel ();
56
57 virtual ~G4TrajectoriesModel ();
58
59 virtual void DescribeYourselfTo (G4VGraphicsScene&);
60 // The main task of a model is to describe itself to the graphics scene.
61
62 const G4VTrajectory* GetCurrentTrajectory() const
63 {return fpCurrentTrajectory;}
64
65 void SetCurrentTrajectory(const G4VTrajectory* pTraj)
66 {fpCurrentTrajectory = pTraj;}
67
68 void SetRunID(G4int runID)
69 {fRunID = runID;}
70
71 void SetEventID(G4int eventID)
72 {fEventID = eventID;}
73
74 const std::map<G4String,G4AttDef>* GetAttDefs() const;
75 std::vector<G4AttValue>* CreateCurrentAttValues() const;
76
77private:
78
79 const G4VTrajectory* fpCurrentTrajectory;
80 G4int fRunID;
81 G4int fEventID;
82
83};
84
85#endif
