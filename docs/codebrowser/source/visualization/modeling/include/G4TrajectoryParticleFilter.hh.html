33#define G4TRAJECTORYPARTICLEFILTER_HH
34
35#include "G4SmartFilter.hh"
36#include "G4VTrajectory.hh"
37#include <vector>
38
39class G4TrajectoryParticleFilter : public G4SmartFilter<G4VTrajectory> {
40
41public: // With description
42
43 // Construct with filter name
44 G4TrajectoryParticleFilter(const G4String& name = "Unspecified");
45
46 virtual ~G4TrajectoryParticleFilter();
47
48 // Evaluate this trajectory
49 virtual bool Evaluate(const G4VTrajectory&) const;
50
51 // Print configuration
52 virtual void Print(std::ostream& ostr) const;
53
54 // Clear filter
55 virtual void Clear();
56
57 // Configuration function
58 void Add(const G4String& particle);
59
60private:
61
62 // Data member
63 std::vector<G4String> fParticles;
64
65};
