31
32#ifndef G4PSEUDOSCENE_HH
33#define G4PSEUDOSCENE_HH
34
35#include "G4VGraphicsScene.hh"
36#include "G4Box.hh"
37#include "G4Cons.hh"
38#include "G4Orb.hh"
39#include "G4Para.hh"
40#include "G4Sphere.hh"
41#include "G4Torus.hh"
42#include "G4Trap.hh"
43#include "G4Trd.hh"
44#include "G4Tubs.hh"
45#include "G4Ellipsoid.hh"
46#include "G4Polycone.hh"
47#include "G4Polyhedra.hh"
48
49class G4PseudoScene: public G4VGraphicsScene {
50
51public:
52
53 G4PseudoScene(): fpCurrentObjectTransformation(0) {}
54 virtual ~G4PseudoScene () {}
55 void PreAddSolid (const G4Transform3D& objectTransformation,
56 const G4VisAttributes&)
57 {fpCurrentObjectTransformation = &objectTransformation;}
58 void PostAddSolid () {}
59 // From geometry/solids/CSG
60 void AddSolid (const G4Box& solid) {ProcessVolume (solid);}
61 void AddSolid (const G4Cons& solid) {ProcessVolume (solid);}
62 void AddSolid (const G4Orb& solid) {ProcessVolume (solid);}
63 void AddSolid (const G4Para& solid) {ProcessVolume (solid);}
64 void AddSolid (const G4Sphere& solid) {ProcessVolume (solid);}
65 void AddSolid (const G4Torus& solid) {ProcessVolume (solid);}
66 void AddSolid (const G4Trap& solid) {ProcessVolume (solid);}
67 void AddSolid (const G4Trd& solid) {ProcessVolume (solid);}
68 void AddSolid (const G4Tubs& solid) {ProcessVolume (solid);}
69 // From geometry/solids/specific
70 void AddSolid (const G4Ellipsoid& solid) {ProcessVolume (solid);}
71 void AddSolid (const G4Polycone& solid) {ProcessVolume (solid);}
72 void AddSolid (const G4Polyhedra& solid) {ProcessVolume (solid);}
73 // For solids not above.
74 void AddSolid (const G4VSolid& solid) {ProcessVolume (solid);}
75 void AddCompound (const G4VTrajectory&) {}
76 void AddCompound (const G4VHit&) {}
77 void AddCompound (const G4VDigi&) {}
78 void AddCompound (const G4THitsMap<G4double>&) {}
79 void AddCompound (const G4THitsMap<G4StatDouble>&) {}
80 void BeginPrimitives (const G4Transform3D&) {}
81 void EndPrimitives () {}
82 void BeginPrimitives2D (const G4Transform3D&) {}
83 void EndPrimitives2D () {}
84 void AddPrimitive (const G4Polyline&) {}
85 void AddPrimitive (const G4Scale&) {}
86 void AddPrimitive (const G4Text&) {}
87 void AddPrimitive (const G4Circle&) {}
88 void AddPrimitive (const G4Square&) {}
89 void AddPrimitive (const G4Polymarker&) {}
90 void AddPrimitive (const G4Polyhedron&) {}
91
92protected:
93
94 virtual void ProcessVolume (const G4VSolid&) = 0;
95 const G4Transform3D* fpCurrentObjectTransformation;
96};
97
98#endif
