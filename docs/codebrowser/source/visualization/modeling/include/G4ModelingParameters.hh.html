51class G4ModelingParameters {
52
53public: // With description
54
55 // Currently requested drawing style.
56 enum DrawingStyle {
57 wf, // Draw edges - no hidden line removal (wireframe).
58 hlr, // Draw edges - hidden lines removed.
59 hsr, // Draw surfaces - hidden surfaces removed.
60 hlhsr // Draw surfaces and edges - hidden removed.
61 };
62
63 // enums and nested class for communicating a modification to the vis
64 // attributes for a specfic touchable defined by PVNameCopyNoPath.
65 enum VisAttributesSignifier {
66 VASVisibility,
67 VASDaughtersInvisible,
68 VASColour,
69 VASLineStyle,
70 VASLineWidth,
71 VASForceWireframe,
72 VASForceSolid,
73 VASForceAuxEdgeVisible,
74 VASForceLineSegmentsPerCircle
75 };
76
77 class PVNameCopyNo {
78 public:
79 // Normal constructor
80 PVNameCopyNo(G4String name, G4int copyNo)
81 : fName(name), fCopyNo(copyNo) {}
82 const G4String& GetName() const {return fName;}
83 G4int GetCopyNo() const {return fCopyNo;}
84 G4bool operator!=(const PVNameCopyNo&) const;
85 G4bool operator==(const PVNameCopyNo& rhs) const {return !operator!=(rhs);}
86 private:
87 G4String fName;
88 G4int fCopyNo;
89 };
90 typedef std::vector<PVNameCopyNo> PVNameCopyNoPath;
91 typedef PVNameCopyNoPath::const_iterator PVNameCopyNoPathConstIterator;
92
93 class PVPointerCopyNo {
94 public:
95 // Normal constructor
96 PVPointerCopyNo(G4VPhysicalVolume* pPV, G4int copyNo)
97 : fpPV(pPV), fCopyNo(copyNo) {}
98 const G4String& GetName() const;
99 const G4VPhysicalVolume* GetPVPointer() const {return fpPV;}
100 G4int GetCopyNo() const {return fCopyNo;}
101 G4bool operator!=(const PVPointerCopyNo&) const;
102 G4bool operator==(const PVPointerCopyNo& rhs) const {return !operator!=(rhs);}
103 private:
104 G4VPhysicalVolume* fpPV;
105 G4int fCopyNo;
106 };
107 typedef std::vector<PVPointerCopyNo> PVPointerCopyNoPath;
108 typedef PVPointerCopyNoPath::const_iterator PVPointerCopyNoPathConstIterator;
109
110 class VisAttributesModifier {
111 public:
112 VisAttributesModifier
113 (const G4VisAttributes& visAtts,
114 VisAttributesSignifier signifier,
115 const PVNameCopyNoPath& path):
116 fVisAtts(visAtts), fSignifier(signifier), fPVNameCopyNoPath(path) {}
117 const G4VisAttributes& GetVisAttributes() const
118 {return fVisAtts;}
119 VisAttributesSignifier GetVisAttributesSignifier() const
120 {return fSignifier;}
121 const PVNameCopyNoPath& GetPVNameCopyNoPath() const
122 {return fPVNameCopyNoPath;}
123 void SetVisAttributes(const G4VisAttributes& visAtts)
124 {fVisAtts = visAtts;}
125 void SetVisAttributesSignifier(VisAttributesSignifier signifier)
126 {fSignifier = signifier;}
127 void SetPVNameCopyNoPath(const PVNameCopyNoPath& PVNameCopyNoPath)
128 {fPVNameCopyNoPath = PVNameCopyNoPath;}
129 G4bool operator!=(const VisAttributesModifier&) const;
130 G4bool operator==(const VisAttributesModifier& rhs) const
131 {return !operator!=(rhs);}
132 private:
133 G4VisAttributes fVisAtts;
134 VisAttributesSignifier fSignifier;
135 PVNameCopyNoPath fPVNameCopyNoPath;
136 };

207
208private:
209
210 // Data members...
211 G4bool fWarning; // Print warnings if true.
212 const G4VisAttributes* fpDefaultVisAttributes;
213 DrawingStyle fDrawingStyle; // Drawing style.
214 G4bool fCulling; // Culling requested.
215 G4bool fCullInvisible; // Cull (don't Draw) invisible objects.
216 G4bool fDensityCulling; // Density culling requested. If so...
217 G4double fVisibleDensity; // ...density lower than this not drawn.
218 G4bool fCullCovered; // Cull daughters covered by opaque mothers.
219 G4int fCBDAlgorithmNumber; // Colour by density algorithm number.
220 std::vector<G4double> fCBDParameters; // Colour by density parameters.
221 G4double fExplodeFactor; // Explode along radius by this factor...
222 G4Point3D fExplodeCentre; // ...about this centre.
223 G4int fNoOfSides; // ...if polygon approximates circle.
224 G4VSolid* fpSectionSolid; // For generic section (DCUT).
225 G4VSolid* fpCutawaySolid; // For generic cutaways.
226 const G4Event* fpEvent; // Event being processed.
227 std::vector<VisAttributesModifier> fVisAttributesModifiers;
228};
