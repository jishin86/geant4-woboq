71/////////////////////////////////////////////////////////////
72G4ProductionCutsTable::G4ProductionCutsTable()
73 : firstUse(true),verboseLevel(1),fMessenger(nullptr)
74{
75 for(size_t i=0;i< NumberOfG4CutIndex;i++)
76 {
77 rangeCutTable.push_back(new G4CutVectorForAParticle);
78 energyCutTable.push_back(new G4CutVectorForAParticle);
79 rangeDoubleVector[i] = nullptr;
80 energyDoubleVector[i] = nullptr;
81 converters[i] = nullptr;
82 }
83 fG4RegionStore = G4RegionStore::GetInstance();
84 defaultProductionCuts = new G4ProductionCuts();
85
86 // add messenger for UI
87 fMessenger = new G4ProductionCutsTableMessenger(this);
88}

98/////////////////////////////////////////////////////////////
99G4ProductionCutsTable::~G4ProductionCutsTable()
100{
101 if (defaultProductionCuts != nullptr) {
102 delete defaultProductionCuts;
103 defaultProductionCuts =nullptr;
104 }
105
106 for(CoupleTableIterator itr=coupleTable.begin();itr!=coupleTable.end();itr++){
107 delete (*itr);
108 }
109 coupleTable.clear();
110
111 for(size_t i=0;i< NumberOfG4CutIndex;i++){
112 delete rangeCutTable[i];
113 delete energyCutTable[i];
114 delete converters[i];
115 if(rangeDoubleVector[i] != nullptr) delete [] rangeDoubleVector[i];
116 if(energyDoubleVector[i] != nullptr) delete [] energyDoubleVector[i];
117 rangeCutTable[i] = nullptr;
118 energyCutTable[i] = nullptr;
119 converters[i] = nullptr;
120 rangeDoubleVector[i] = nullptr;
121 energyDoubleVector[i] = nullptr;
122 }
123 fG4ProductionCutsTable = nullptr;
124
125 if (fMessenger != nullptr) delete fMessenger;
126 fMessenger = nullptr;
127}

129void G4ProductionCutsTable::UpdateCoupleTable(G4VPhysicalVolume* /*currentWorld*/)
130{
131 if(firstUse){
132 if(G4ParticleTable::GetParticleTable()->FindParticle("gamma")){
133 converters[0] = new G4RToEConvForGamma();
134 converters[0]->SetVerboseLevel(GetVerboseLevel());
135 }
136 if(G4ParticleTable::GetParticleTable()->FindParticle("e-")){
137 converters[1] = new G4RToEConvForElectron();
138 converters[1]->SetVerboseLevel(GetVerboseLevel());
139 }
140 if(G4ParticleTable::GetParticleTable()->FindParticle("e+")){
141 converters[2] = new G4RToEConvForPositron();
142 converters[2]->SetVerboseLevel(GetVerboseLevel());
143 }
144 if(G4ParticleTable::GetParticleTable()->FindParticle("proton")){
145 converters[3] = new G4RToEConvForProton();
146 converters[3]->SetVerboseLevel(GetVerboseLevel());
147 }
148 firstUse = false;
149 }
150
151 // Reset "used" flags of all couples
152 for(CoupleTableIterator CoupleItr=coupleTable.begin();
153 CoupleItr!=coupleTable.end();CoupleItr++)
154 {
155 (*CoupleItr)->SetUseFlag(false);
156 }
157
158 // Update Material-Cut-Couple
159 typedef std::vector<G4Region*>::iterator regionIterator;
160 for(regionIterator rItr=fG4RegionStore->begin();
161 rItr!=fG4RegionStore->end();rItr++)
162 {
163 // Material scan is to be done only for the regions appear in the
164 // current tracking world.
165// if((*rItr)->GetWorldPhysical()!=currentWorld) continue;
166 if((*rItr)->IsInMassGeometry() || (*rItr)->IsInParallelGeometry())
167 {
168
169 G4ProductionCuts* fProductionCut = (*rItr)->GetProductionCuts();
170 std::vector<G4Material*>::const_iterator mItr =
171 (*rItr)->GetMaterialIterator();
172 size_t nMaterial = (*rItr)->GetNumberOfMaterials();
173 (*rItr)->ClearMap();
174
175 for(size_t iMate=0;iMate<nMaterial;iMate++){
176 //check if this material cut couple has already been made
177 G4bool coupleAlreadyDefined = false;
178 G4MaterialCutsCouple* aCouple;
179 for(CoupleTableIterator cItr=coupleTable.begin();
180 cItr!=coupleTable.end();cItr++){
181 if( (*cItr)->GetMaterial()==(*mItr) &&
182 (*cItr)->GetProductionCuts()==fProductionCut){
183 coupleAlreadyDefined = true;
184 aCouple = *cItr;
185 break;
186 }
187 }
188
189 // If this combination is new, cleate and register a couple
190 if(!coupleAlreadyDefined){
191 aCouple = new G4MaterialCutsCouple((*mItr),fProductionCut);
192 coupleTable.push_back(aCouple);
193 aCouple->SetIndex(coupleTable.size()-1);
194 }
195
196 // Register this couple to the region
197 (*rItr)->RegisterMaterialCouplePair((*mItr),aCouple);
198
199 // Set the couple to the proper logical volumes in that region
200 aCouple->SetUseFlag();
201
202 std::vector<G4LogicalVolume*>::iterator rootLVItr
203 = (*rItr)->GetRootLogicalVolumeIterator();
204 size_t nRootLV = (*rItr)->GetNumberOfRootVolumes();
205 for(size_t iLV=0;iLV<nRootLV;iLV++) {
206 //Set the couple to the proper logical volumes in that region
207 G4LogicalVolume* aLV = *rootLVItr;
208 G4Region* aR = *rItr;
209
210 ScanAndSetCouple(aLV,aCouple,aR);
211
212 // Proceed to the next root logical volume in this region
213 rootLVItr++;
214 }
215
216 // Proceed to next material in this region
217 mItr++;
218 }
219 }
220 }
221
222 // Check if sizes of Range/Energy cuts tables are equal to the size of
223 // the couple table
224 // If new couples are made during the previous procedure, nCouple becomes
225 // larger then nTable
226 size_t nCouple = coupleTable.size();
227 size_t nTable = energyCutTable[0]->size();
228 G4bool newCoupleAppears = nCouple>nTable;
229 if(newCoupleAppears) {
230 for(size_t n=nCouple-nTable;n>0;n--) {
231 for(size_t nn=0;nn< NumberOfG4CutIndex;nn++){
232 rangeCutTable[nn]->push_back(-1.);
233 energyCutTable[nn]->push_back(-1.);
234 }
235 }
236 }
237
238 // Update RangeEnergy cuts tables
239 size_t idx = 0;
240 G4Timer timer;
241 if (verboseLevel>2) {
242 timer.Start();
243 }
244 for(CoupleTableIterator cItr=coupleTable.begin();
245 cItr!=coupleTable.end();cItr++){
246 G4ProductionCuts* aCut = (*cItr)->GetProductionCuts();
247 const G4Material* aMat = (*cItr)->GetMaterial();
248 if((*cItr)->IsRecalcNeeded()) {
249 for(size_t ptcl=0;ptcl< NumberOfG4CutIndex;ptcl++){
250 G4double rCut = aCut->GetProductionCut(ptcl);
251 (*(rangeCutTable[ptcl]))[idx] = rCut;
252 // if(converters[ptcl] && (*cItr)->IsUsed())
253 if(converters[ptcl]) {
254 (*(energyCutTable[ptcl]))[idx] = converters[ptcl]->Convert(rCut,aMat);
255 } else {
256 (*(energyCutTable[ptcl]))[idx] = -1.;
257 }
258 }
259 }
260 idx++;
261 }
262 if (verboseLevel>2) {
263 timer.Stop();
264 G4cout << "G4ProductionCutsTable::UpdateCoupleTable "
265 << " elapsed time for calculation of energy cuts " << G4endl;
266 G4cout << timer <<G4endl;
267 }
268
269 // resize Range/Energy cuts double vectors if new couple is made
270 if(newCoupleAppears){
271 for(size_t ix=0;ix<NumberOfG4CutIndex;ix++){
272 G4double* rangeVOld = rangeDoubleVector[ix];
273 G4double* energyVOld = energyDoubleVector[ix];
274 if(rangeVOld) delete [] rangeVOld;
275 if(energyVOld) delete [] energyVOld;
276 rangeDoubleVector[ix] = new G4double[(*(rangeCutTable[ix])).size()];
277 energyDoubleVector[ix] = new G4double[(*(energyCutTable[ix])).size()];
278 }
279 }
280
281 // Update Range/Energy cuts double vectors
282 for(size_t ix=0;ix<NumberOfG4CutIndex;ix++) {
283 for(size_t ixx=0;ixx<(*(rangeCutTable[ix])).size();ixx++) {
284 rangeDoubleVector[ix][ixx] = (*(rangeCutTable[ix]))[ixx];
285 energyDoubleVector[ix][ixx] = (*(energyCutTable[ix]))[ixx];
286 }
287 }
288}

291/////////////////////////////////////////////////////////////
292G4double G4ProductionCutsTable::ConvertRangeToEnergy(
293 const G4ParticleDefinition* particle,
294 const G4Material* material,
295 G4double range
296 )
297{
298 // This method gives energy corresponding to range value
299 // check material
300 if (material ==nullptr) return -1.0;
301
302 // check range
303 if (range ==0.0) return 0.0;
304 if (range <0.0) return -1.0;
305
306 // check particle
307 G4int index = G4ProductionCuts::GetIndex(particle);
308
309 if (index<0) {
310#ifdef G4VERBOSE
311 if (verboseLevel >1) {
312 G4cout << "G4ProductionCutsTable::ConvertRangeToEnergy" ;
313 G4cout << particle->GetParticleName() << " has no cut value " << G4endl;
314 }
315#endif
316 return -1.0;
317 }
318
319 return converters[index]->Convert(range, material);
320
321}

351/////////////////////////////////////////////////////////////
352void G4ProductionCutsTable::ScanAndSetCouple(G4LogicalVolume* aLV,
353 G4MaterialCutsCouple* aCouple,
354 G4Region* aRegion)
355{
356 //Check whether or not this logical volume belongs to the same region
357 if((aRegion!=nullptr) && aLV->GetRegion()!=aRegion) return;
358
359 //Check if this particular volume has a material matched to the couple
360 if(aLV->GetMaterial()==aCouple->GetMaterial()) {
361 aLV->SetMaterialCutsCouple(aCouple);
362 }
363
364 size_t noDaughters = aLV->GetNoDaughters();
365 if(noDaughters==0) return;
366
367 //Loop over daughters with same region
368 for(size_t i=0;i<noDaughters;i++){
369 G4LogicalVolume* daughterLVol = aLV->GetDaughter(i)->GetLogicalVolume();
370 ScanAndSetCouple(daughterLVol,aCouple,aRegion);
371 }
372}

494/////////////////////////////////////////////////////////////
495// Store material information in files under the specified directory.
496//
497G4bool G4ProductionCutsTable::StoreMaterialInfo(const G4String& directory,
498 G4bool ascii)
499{
500 const G4String fileName = directory + "/" + "material.dat";
501 const G4String key = "MATERIAL-V3.0";
502 std::ofstream fOut;
503
504 // open output file //
505 if (!ascii )
506 fOut.open(fileName,std::ios::out|std::ios::binary);
507 else
508 fOut.open(fileName,std::ios::out);
509
510
511 // check if the file has been opened successfully
512 if (!fOut) {
513#ifdef G4VERBOSE
514 if (verboseLevel>0) {
515 G4cerr << "G4ProductionCutsTable::StoreMaterialInfo ";
516 G4cerr << " Can not open file " << fileName << G4endl;
517 }
518#endif
519 G4Exception( "G4ProductionCutsTable::StoreMaterialInfo()",
520 "ProcCuts102",
521 JustWarning, "Can not open file ");
522 return false;
523 }
524
525 const G4MaterialTable* matTable = G4Material::GetMaterialTable();
526 // number of materials in the table
527 G4int numberOfMaterial = matTable->size();
528
529 if (ascii) {
530 /////////////// ASCII mode /////////////////
531 // key word
532 fOut << key << G4endl;
533
534 // number of materials in the table
535 fOut << numberOfMaterial << G4endl;
536
537 fOut.setf(std::ios::scientific);
538
539 // material name and density
540 for (size_t idx=0; static_cast<G4int>(idx)<numberOfMaterial; ++idx){
541 fOut << std::setw(FixedStringLengthForStore)
542 << ((*matTable)[idx])->GetName();
543 fOut << std::setw(FixedStringLengthForStore)
544 << ((*matTable)[idx])->GetDensity()/(g/cm3) << G4endl;
545 }
546
547 fOut.unsetf(std::ios::scientific);
548
549 } else {
550 /////////////// Binary mode /////////////////
551 char temp[FixedStringLengthForStore];
552 size_t i;
553
554 // key word
555 for (i=0; i<FixedStringLengthForStore; ++i){
556 temp[i] = '\0';
557 }
558 for (i=0; i<key.length() && i<FixedStringLengthForStore-1; ++i){
559 temp[i]=key[i];
560 }
561 fOut.write(temp, FixedStringLengthForStore);
562
563 // number of materials in the table
564 fOut.write( (char*)(&numberOfMaterial), sizeof (G4int));
565
566 // material name and density
567 for (size_t imat=0; static_cast<G4int>(imat)<numberOfMaterial; ++imat){
568 G4String name = ((*matTable)[imat])->GetName();
569 G4double density = ((*matTable)[imat])->GetDensity();
570 for (i=0; i<FixedStringLengthForStore; ++i) temp[i] = '\0';
571 for (i=0; i<name.length() && i<FixedStringLengthForStore-1; ++i)
572 temp[i]=name[i];
573 fOut.write(temp, FixedStringLengthForStore);
574 fOut.write( (char*)(&density), sizeof (G4double));
575 }
576 }
577
578 fOut.close();
579 return true;
580}

582/////////////////////////////////////////////////////////////
583// check stored material is consistent with the current detector setup.
584G4bool G4ProductionCutsTable::CheckMaterialInfo(const G4String& directory,
585 G4bool ascii)
586{
587 const G4String fileName = directory + "/" + "material.dat";
588 const G4String key = "MATERIAL-V3.0";
589 std::ifstream fIn;
590
591 // open input file //
592 if (!ascii )
593 fIn.open(fileName,std::ios::in|std::ios::binary);
594 else
595 fIn.open(fileName,std::ios::in);
596
597 // check if the file has been opened successfully
598 if (!fIn) {
599#ifdef G4VERBOSE
600 if (verboseLevel >0) {
601 G4cerr << "G4ProductionCutsTable::CheckMaterialInfo ";
602 G4cerr << " Can not open file " << fileName << G4endl;
603 }
604#endif
605 G4Exception( "G4ProductionCutsTable::CheckMaterialInfo()",
606 "ProcCuts102",
607 JustWarning, "Can not open file ");
608 return false;
609 }
610
611 char temp[FixedStringLengthForStore];
612
613 // key word
614 G4String keyword;
615 if (ascii) {
616 fIn >> keyword;
617 } else {
618 fIn.read(temp, FixedStringLengthForStore);
619 keyword = (const char*)(temp);
620 }
621 if (key!=keyword) {
622#ifdef G4VERBOSE
623 if (verboseLevel >0) {
624 G4cerr << "G4ProductionCutsTable::CheckMaterialInfo ";
625 G4cerr << " Key word in " << fileName << "= " << keyword ;
626 G4cerr <<"( should be "<< key << ")" <<G4endl;
627 }
628#endif
629 G4Exception( "G4ProductionCutsTable::CheckMaterialInfo()",
630 "ProcCuts103",
631 JustWarning, "Bad Data Format");
632 return false;
633 }
634
635 // number of materials in the table
636 G4int nmat;
637 if (ascii) {
638 fIn >> nmat;
639 } else {
640 fIn.read( (char*)(&nmat), sizeof (G4int));
641 }
642 if ((nmat<=0) || (nmat >100000)){
643 G4Exception( "G4ProductionCutsTable::CheckMaterialInfo()",
644 "ProcCuts108",JustWarning,
645 "Number of materials is less than zero or too big");
646 return false;
647 }
648
649 // list of material
650 for (G4int idx=0; idx<nmat ; ++idx){
651 // check eof
652 if(fIn.eof()) {
653#ifdef G4VERBOSE
654 if (verboseLevel >0) {
655 G4cout << "G4ProductionCutsTable::CheckMaterialInfo ";
656 G4cout << " encountered End of File " ;
657 G4cout << " at " << idx+1 << "th material "<< G4endl;
658 }
659#endif
660 fIn.close();
661 return false;
662 }
663
664 // check material name and density
665 char name[FixedStringLengthForStore];
666 double density;
667 if (ascii) {
668 fIn >> name >> density;
669 density *= (g/cm3);
670
671 } else {
672 fIn.read(name, FixedStringLengthForStore);
673 fIn.read((char*)(&density), sizeof (G4double));
674 }
675 if (fIn.fail()) {
676#ifdef G4VERBOSE
677 if (verboseLevel >0) {
678 G4cerr << "G4ProductionCutsTable::CheckMaterialInfo ";
679 G4cerr << " Bad data format ";
