32// Class Description
33// This class is a container for pointers to physics process objects.

35
36#ifndef G4ProcessVector_h
37#define G4ProcessVector_h 1
38
39#include "globals.hh"
40#include "G4ios.hh"
41#include <vector>
42
43class G4VProcess;
44
45class G4ProcessVector
46{
47 public:
48 // Constructors
49 G4ProcessVector();
50 explicit G4ProcessVector(size_t);
51 G4ProcessVector(const G4ProcessVector &);
52
53 // Destructor.
54 virtual ~G4ProcessVector();
55
56 //assignment operator
57 G4ProcessVector & operator=(const G4ProcessVector &right);
58
59 // equal operator
60 G4bool operator==(const G4ProcessVector &right) const;
61
62 public: // With Description
63 // Returns the number of items
64 G4int entries() const;
65 G4int length() const;
66 G4int size() const;
67
68 // Returns the position of the element
69 G4int index(G4VProcess* aProcess) const;
70
71 // Returns "true" if the element exists
72 G4bool contains(G4VProcess* aProcess) const;
73
74 // Insert an element
75 G4bool insert(G4VProcess* aProcess);
76
77 // Insert an element at i-th position
78 G4bool insertAt(G4int i, G4VProcess* aProcess);
79
80 // Remove and returns the i-th element
81 G4VProcess* removeAt(G4int i);
82
83 // Remove and returns the last element
84 G4VProcess* removeLast();
85
86 // Clear the collection by removing all items
87 void clear();
88
89 // returns const reference to the i-th item
90 G4VProcess* const& operator[](G4int i) const;
91 G4VProcess* const& operator()(G4int i) const;
92
93 // returns reference to the i-th item
94 G4VProcess* & operator[](G4int i);
95 G4VProcess* & operator()(G4int i);
96
97 protected:
98 // Use STL Vector
99 typedef std::vector<G4VProcess*> G4ProcVector;
100
101 G4ProcVector * pProcVector;
102};
