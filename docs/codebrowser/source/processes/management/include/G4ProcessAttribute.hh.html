42#ifndef G4ProcessAttribute_h
43#define G4ProcessAttribute_h 1
44
45#include "globals.hh"
46#include "G4ios.hh"
47
48#include "G4ProcessManager.hh"
49
50class G4VProcess;
51
52class G4ProcessAttribute
53{
54 // this class is used by G4ProcessManager ONLY for booking !!!
55 friend class G4ProcessManager;
56 public:
57 G4ProcessAttribute();
58 G4ProcessAttribute(const G4VProcess* aProcess);
59 G4ProcessAttribute(const G4ProcessAttribute &right);
60 // Constructors
61
62 ~G4ProcessAttribute();
63 // Destructor
64
65 G4ProcessAttribute & operator=(const G4ProcessAttribute &right);
66 // Assignment operator
67
68 G4bool operator==(const G4ProcessAttribute &right) const;
69 G4bool operator!=(const G4ProcessAttribute &right) const;
70 // equal / unequal operator
71
72
73 protected:
74 G4VProcess* pProcess;
75 // pointer to G4VProcess
76
77 G4bool isActive;
78 // flag for activation/inactivation
79
80 G4int idxProcessList;
81 // index to a ProcessVector for theProcessList and
82
83 G4int idxProcVector[G4ProcessManager::SizeOfProcVectorArray];
84 // index to ProcessVectors for "Doit"s and "GetPhysicalInteractionLength"s
85 // -1 : not applicable
86
87 G4int ordProcVector[G4ProcessManager::SizeOfProcVectorArray];
88 // ordering parameter
89};
90
91inline
92 G4ProcessAttribute::G4ProcessAttribute(const G4VProcess* aProcess):
93 pProcess((G4VProcess*)aProcess),
94 isActive(true),
95 idxProcessList(-1)
96{
97 for(size_t ii=0; ii<G4ProcessManager::SizeOfProcVectorArray; ii++){
98 idxProcVector[ii]=-1;
99 ordProcVector[ii]=0;
100 }
101}
102
103inline
104 G4bool G4ProcessAttribute::operator==(const G4ProcessAttribute &right) const
105{
106 return this->pProcess == right.pProcess;
107}
108
109inline
110 G4bool G4ProcessAttribute::operator!=(const G4ProcessAttribute &right) const
111{
112 return this->pProcess != right.pProcess;
113}
114
115#endif
