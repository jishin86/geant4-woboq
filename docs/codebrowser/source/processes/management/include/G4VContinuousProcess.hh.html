51#ifndef G4VContinuousProcess_h
52#define G4VContinuousProcess_h 1
53
54#include "globals.hh"
55#include "G4ios.hh"
56
57#include "G4VProcess.hh"
58
59class G4VContinuousProcess : public G4VProcess
60{
61 // Abstract class which defines the public behavior of
62 // Continuous physics interactions.
63 public:
64
65 G4VContinuousProcess(const G4String& ,
66 G4ProcessType aType = fNotDefined );
67 G4VContinuousProcess(G4VContinuousProcess &);
68
69 virtual ~G4VContinuousProcess();
70
71 public: // with description
72 virtual G4double AlongStepGetPhysicalInteractionLength(
73 const G4Track& track,
74 G4double previousStepSize,
75 G4double currentMinimumStep,
76 G4double& proposedSafety,
77 G4GPILSelection* selection
78 );
79
80 virtual G4VParticleChange* AlongStepDoIt(
81 const G4Track& ,
82 const G4Step&
83 );
84
85 // no operation in AtRestDoIt and PostStepDoIt
86 virtual G4double PostStepGetPhysicalInteractionLength(
87 const G4Track&,
88 G4double,
89 G4ForceCondition*
90 ){ return -1.0; };
91
92 virtual G4double AtRestGetPhysicalInteractionLength(
93 const G4Track& ,
94 G4ForceCondition*
95 ) { return -1.0; };
96
97 // no operation in AtRestDoIt and PostStepDoIt
98 virtual G4VParticleChange* AtRestDoIt(
99 const G4Track& ,
100 const G4Step&
101 ) {return 0;};
102
103 virtual G4VParticleChange* PostStepDoIt(
104 const G4Track& ,
105 const G4Step&
106 ) {return 0;};
107
108 protected: // with description
109 virtual G4double GetContinuousStepLimit(const G4Track& aTrack,
110 G4double previousStepSize,
111 G4double currentMinimumStep,
112 G4double& currentSafety
113 )=0;
114 // This pure virtual function is used to calculate step limit
115 // for AlongStep in the derived processes
116
117 private:
118 // this is the returnd value of G4GPILSelection in
119 // the arguments of AlongStepGPIL()
120 G4GPILSelection valueGPILSelection;
121
122 protected: // with description
123 // these two methods are set/get methods for valueGPILSelection
124 void SetGPILSelection(G4GPILSelection selection)
125 { valueGPILSelection = selection;};
126
127 G4GPILSelection GetGPILSelection() const{return valueGPILSelection;};
128
129
130 private:
131 // hide default constructor and assignment operator as private
132 G4VContinuousProcess();
133 G4VContinuousProcess & operator=(const G4VContinuousProcess &right);
134
135};
136
137#endif
