51inline
52 G4int G4ProcessManager::GetProcessIndex(G4VProcess* aProcess) const
53{
54 G4int idx = theProcessList->index(aProcess);
55 if (idx>=numberOfProcesses) idx = -1;

59inline
60 G4int G4ProcessManager::GetProcessVectorId(G4ProcessVectorDoItIndex idx,
61 G4ProcessVectorTypeIndex typ) const
62{
63 if ( idx == idxAtRest ) {
64 if (typ == typeGPIL) { return 0; }
65 else { return 1; }
66 } else if ( idx == idxAlongStep ) {
67 if (typ == typeGPIL) { return 2; }
68 else { return 3; }
69 } else if ( idx == idxPostStep ) {
70 if (typ == typeGPIL) { return 4; }
71 else { return 5; }
72 } else {
73 return -1;
74 }
75}
76
77inline
78 G4ProcessVector* G4ProcessManager::GetProcessVector(
79 G4ProcessVectorDoItIndex idx,
80 G4ProcessVectorTypeIndex typ
81 ) const
82{
83 G4int ivec = GetProcessVectorId(idx, typ);
84 if ( ivec >=0 ) {
85 return theProcVector[ivec];
86 } else {
87 return 0;
88 }
89}

91inline
92 G4ProcessVector* G4ProcessManager::GetAtRestProcessVector(G4ProcessVectorTypeIndex typ) const
93{
94 if (typ == typeGPIL) { return theProcVector[0]; }
95 else { return theProcVector[1]; }
96}

98inline
99 G4ProcessVector* G4ProcessManager::GetAlongStepProcessVector(G4ProcessVectorTypeIndex typ) const
100{
101 if (typ == typeGPIL) { return theProcVector[2]; }
102 else { return theProcVector[3]; }
103}
104
105inline
106 G4ProcessVector* G4ProcessManager::GetPostStepProcessVector(G4ProcessVectorTypeIndex typ) const
107{
108 if (typ == typeGPIL) { return theProcVector[4]; }
109 else { return theProcVector[5]; }
110}
