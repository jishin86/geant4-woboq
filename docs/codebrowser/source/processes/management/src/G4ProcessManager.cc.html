58// ///////////////////////////////////////
59G4ProcessManager::G4ProcessManager(const G4ParticleDefinition* aParticleType):
60 theAttrVector(nullptr),
61 theParticleType(aParticleType),
62 numberOfProcesses(0),
63 theProcessList(nullptr),
64 duringTracking(false),
65 verboseLevel(1)
66{
67 // create the process List
68 theProcessList = new G4ProcessVector();
69 if ( theProcessList == nullptr) {
70 G4Exception( "G4ProcessManager::G4ProcessManager()","ProcMan012",
71 FatalException, "Can not create G4ProcessList ");
72 }
73
74 //create process vector
75 for (G4int i=0; i<SizeOfProcVectorArray; ++i) {
76 theProcVector[i] = new G4ProcessVector();
77 if ( theProcVector[i] == nullptr) {
78 G4Exception( "G4ProcessManager::G4ProcessManager()","ProcMan012",
79 FatalException, "Can not create G4ProcessVector ");
80 }
81 }
82
83 // create Process Attribute vector
84 theAttrVector = new G4ProcessAttrVector();
85
86 // create Process Manager Messenger
87 if (fProcessManagerMessenger == nullptr){
88 fProcessManagerMessenger = new G4ProcessManagerMessenger();
89 }
90
91 for (G4int i=0; i<NDoit; ++i) {
92 isSetOrderingFirstInvoked[i]=false;
93 isSetOrderingLastInvoked[i]=false;
94 }
95
96 // Increment counter of G4ProcessManager objects
97 counterOfObjects+=1;
98}

100// ///////////////////////////////////////
101G4ProcessManager::G4ProcessManager(G4ProcessManager &right):
102 theAttrVector(nullptr),
103 theParticleType(right.theParticleType),
104 numberOfProcesses(0),
105 theProcessList(nullptr),
106 duringTracking(false),
107 verboseLevel(right.verboseLevel)
108{
109#ifdef G4VERBOSE
110 if (GetVerboseLevel() > 2) {
111 G4cout << "G4ProcessManageer:: copy constructor " <<G4endl;
112 }
113#endif
114
115 // create the process List and ProcessAttr Vector
116 theProcessList = new G4ProcessVector();
117 theAttrVector = new G4ProcessAttrVector();
118 if ( ( theProcessList == nullptr) || (theAttrVector == nullptr) ){
119 G4Exception( "G4ProcessManager::G4ProcessManager() [coopy constructor]",
120 "ProcMan011",FatalException, "Can not create G4ProcessList ");
121 }
122
123 for (G4int idx=0; idx < right.numberOfProcesses; idx++) {
124 // copy contents in theProcessList
125 theProcessList->insert((*right.theProcessList)[idx]);
126 // create a G4ProcessAttribute same as source's one
127 G4ProcessAttribute* sAttr = (*right.theAttrVector)[idx];
128 G4ProcessAttribute* dAttr = new G4ProcessAttribute(*sAttr);
129 // adds a G4ProcessAttribute object
130 theAttrVector->push_back(dAttr);
131 numberOfProcesses +=1;
132 }
133
134
135 // fill up theProcVector
136 for (G4int i=0; i<SizeOfProcVectorArray; ++i) {
137 // create i-th ProcessVector in theProcVector
138 theProcVector[i] = new G4ProcessVector();
139 if ( theProcVector[i] == nullptr ) {
140 G4Exception( "G4ProcessManager::G4ProcessManager() [coopy constructor]",
141 "ProcMan011",FatalException, "Can not create G4ProcessVector ");
142 }
143
144 G4ProcessTable* theProcessTable = G4ProcessTable::GetProcessTable();
145 G4ProcessVector* src = right.theProcVector[i];
146 for (G4int j=0; j< src->entries() ; j++){
147 // copy j-th process in i-th ProcessVector
148 theProcVector[i]->insert((*src)[j]);
149 //add aProcess and this ProcessManager into ProcesssTable
150 if ( (*src)[j] != nullptr ) {
151 theProcessTable->Insert((*src)[j], this);
152 }
153 }
154 }
155
156 for (G4int i=0; i<NDoit; ++i) {
157 isSetOrderingFirstInvoked[i]= right.isSetOrderingFirstInvoked[i];
158 isSetOrderingLastInvoked[i] = right.isSetOrderingLastInvoked[i];
159 }
160
161 // Increment counter of G4ProcessManager objects
162 counterOfObjects+=1;
163}

234G4int G4ProcessManager::GetProcessVectorIndex(
235 G4VProcess* aProcess,
236 G4ProcessVectorDoItIndex idx,
237 G4ProcessVectorTypeIndex typ
238 ) const
239{
240 G4int idxVect = -1;
241 G4int idxProc = GetProcessIndex(aProcess);
242 G4int ivec = GetProcessVectorId(idx, typ);
243
244 if ( ( idxProc >=0) && (ivec >=0) ){
245 idxVect = GetAttribute(idxProc)->idxProcVector[ivec];
246 } else {
247#ifdef G4VERBOSE
248 if (verboseLevel>0) {
249 G4cout << " G4ProcessManager::GetProcessVectorIndex:";
250 G4cout << "particle[" << theParticleType->GetParticleName() << "] " ;
251 G4cout << "process[" << aProcess->GetProcessName() << "]" ;
252 G4cout << G4endl;
253 if (idxProc <0) {
254 G4cout << " is not registered yet ";
255 }
256 if (ivec <0) {
257 G4cout << " illegal DoIt Index [= " << G4int(idx) << ","
258 << G4int(typ) << "]";
259 }
260 G4cout << G4endl;
261 }
262#endif
263 }
264 return idxVect;

267/////////////////////////////////////////
268G4ProcessAttribute* G4ProcessManager::GetAttribute(G4int index) const
269{
270 // check index range
271 if ((index<0) || (index>=numberOfProcesses)) {
272#ifdef G4VERBOSE
273 if (GetVerboseLevel()>0) {
274 G4cout << "G4ProcessManager::GetAttribute():";
275 G4cout << " particle[" << theParticleType->GetParticleName() << "]";
276 G4cout << G4endl;
277 G4cout << " index out of range " << G4endl;
278 G4cout << " #processes[" << numberOfProcesses << "]";
279 G4cout << " index [" << index << "]" << G4endl;
280 }
281#endif
282 return nullptr;
283 }
284
285 // check process pointer is not null
286 G4VProcess* aProcess = (*theProcessList)[index];
287 if (aProcess == nullptr) {
288 G4String aErrorMessage("Bad ProcessList: Null Pointer for");
289 aErrorMessage += theParticleType->GetParticleName() ;
290 G4Exception("G4ProcessManager::GetAttribute()","ProcMan012",
291 FatalException,aErrorMessage);
292 return nullptr;
293 }
294
295 //find the process attribute
296 if ( ((*theAttrVector)[index])->idxProcessList == index ){
297 return (*theAttrVector)[index];
298 } else {
299 // !! Error !!
300 // attribute vector index is inconsistent with process List index
301#ifdef G4VERBOSE
302 if (GetVerboseLevel()>0) {
303 G4cout << "G4ProcessManager::GetAttribute():";
304 G4cout << " particle[" << theParticleType->GetParticleName() << "]"
305 << G4endl;
306 G4cout << "Warning:: attribute vector index is inconsistent with process List index"
307 << G4endl;
308 }
309#endif
310 // re-ordering attribute vector
311 G4ProcessAttribute *pAttr = nullptr;
312 G4ProcessAttrVector::iterator itr;
313 for (itr = theAttrVector->begin(); itr!= theAttrVector->end(); ++itr) {
314 if ( (*itr)->idxProcessList == index) {
315 pAttr = (*itr);
316 break;
317 }
318 }
319 return pAttr;
320 }
321}

329// ///////////////////////////////////////
330G4int G4ProcessManager::InsertAt(G4int ip, G4VProcess* process, G4int ivec)
331{
332 G4ProcessVector* pVector = theProcVector[ivec];
333 // check position
334 if ( (ip<0) || (ip > pVector->entries()) ) return -1;
335
336 // insert in pVector
337 pVector->insertAt(ip, process);
338
339 //correct index in ProcessAttributes of processes
340 for (G4int iproc=0; iproc<numberOfProcesses; iproc++) {
341 G4ProcessAttribute* aAttr = (*theAttrVector)[iproc];
342 if (aAttr != nullptr) {
343 if (aAttr->idxProcVector[ivec] >= ip){
344 aAttr->idxProcVector[ivec] += 1;
345 }
346 } else {
347#ifdef G4VERBOSE
348 if (GetVerboseLevel()>0) {
349 G4cout << " G4ProcessManager::InsertAt : No Process Attribute " << G4endl;
350 }
351#endif
352 }
353 }
354 return ip;
355}

357// ///////////////////////////////////////
358G4int G4ProcessManager::RemoveAt(G4int ip, G4VProcess* , G4int ivec)
359{
360 G4ProcessVector* pVector = theProcVector[ivec];
361 // check position
362 if ( (ip<0) || (ip >= pVector->entries()) ) return -1;
363
364 // remove process
365 pVector->removeAt(ip);
366
367 // correct index
368 for(G4int iproc=0; iproc<numberOfProcesses; iproc++) {
369 G4ProcessAttribute* aAttr = (*theAttrVector)[iproc];
370 if (aAttr != nullptr) {
371 if (ip < aAttr->idxProcVector[ivec]) {
372 aAttr->idxProcVector[ivec] -=1;
373 } else if (ip == aAttr->idxProcVector[ivec]) {
374 aAttr->idxProcVector[ivec] = -1;
375 aAttr->ordProcVector[ivec] = ordInActive;
376 }
377 }else {
378#ifdef G4VERBOSE
379 if (GetVerboseLevel()>0) {
380 G4cout << " G4ProcessManager::RemoveAt : No Process Attribute " << G4endl;
381 }
382#endif
383 }
384 }
385 return ip;
386}

388// ///////////////////////////////////////
389G4int G4ProcessManager::FindInsertPosition(G4int ord, G4int ivec)
390{
391 G4ProcessVector* pVector = theProcVector[ivec];
392 G4int ip = pVector->entries();
393 G4int tmp = INT_MAX;
394 if (ord == ordLast) return ip;
395
396 // find insert position
397 for (G4int iproc=0; iproc<numberOfProcesses; iproc++) {
398 G4ProcessAttribute* aAttr = (*theAttrVector)[iproc];
399 if ( (aAttr->ordProcVector[ivec] > ord ) && (tmp > aAttr->ordProcVector[ivec])){
400 tmp = aAttr->ordProcVector[ivec] ;
401 if (ip > aAttr->idxProcVector[ivec]) ip = aAttr->idxProcVector[ivec];
402 }
403 }
404 return ip;
405}

407// ///////////////////////////////////////
408G4int G4ProcessManager::AddProcess(
409 G4VProcess *aProcess,
410 G4int ordAtRestDoIt,
411 G4int ordAlongStepDoIt,
412 G4int ordPostStepDoIt
413 )
414{
415
416 //check the process is applicable to this particle type
417 if ( !aProcess->IsApplicable(*theParticleType) ) {
418#ifdef G4VERBOSE
419 if (GetVerboseLevel()>1) {
420 G4cout << "G4ProcessManager::AddProcess()" << G4endl;
421 G4cout << "This process is not applicable to this particle" << G4endl;
422 }
423#endif
424 return -1;
425 }
426
427#ifdef G4VERBOSE
428 if (GetVerboseLevel()>2) {
429 G4cout << "G4ProcessManager::AddProcess()" << G4endl;
430 }
431#endif
432
433 //add aProcess and this ProcessManager into ProcesssTable
434 G4ProcessTable* theProcessTable = G4ProcessTable::GetProcessTable();
435 theProcessTable->Insert(aProcess, this);
436
437 //add aProcess to process List
438 theProcessList->insert(aProcess);
439 G4int idx = (theProcessList->entries()) - 1;
440
441 // check size of the ProcessVector[0]
442 if (numberOfProcesses != idx){
443 theProcessList->removeLast();
444 G4String anErrorMessage("Bad ProcessList: Inconsistent process List size for ");
445 anErrorMessage += "process[" + aProcess->GetProcessName() + "]";
446 anErrorMessage += " particle[" + theParticleType->GetParticleName() + "]";
447 G4Exception( "G4ProcessManager::AddProcess()","ProcMan012",
448 FatalException,anErrorMessage);
449 return -1;
450 }
451
452 // create ProcessAttribute
453 G4ProcessAttribute* pAttr = new G4ProcessAttribute(aProcess);
454 pAttr->idxProcessList = idx;
455
456 // check if ordering parameter is non-zero
457 if (ordAtRestDoIt==0) ordAtRestDoIt = 1;
458 if (ordAlongStepDoIt==0) ordAlongStepDoIt = 1;
459 if (ordPostStepDoIt==0) ordPostStepDoIt = 1;
460
461 // ordering parameter
462 pAttr->ordProcVector[0] = ordAtRestDoIt;
463 pAttr->ordProcVector[1] = ordAtRestDoIt;
464 pAttr->ordProcVector[2] = ordAlongStepDoIt;
465 pAttr->ordProcVector[3] = ordAlongStepDoIt;
466 pAttr->ordProcVector[4] = ordPostStepDoIt;
467 pAttr->ordProcVector[5] = ordPostStepDoIt;
468
469 // add aProccess in Process vectors
470 for (G4int ivec=1; ivec<SizeOfProcVectorArray; ivec+=2) {
471 if (pAttr->ordProcVector[ivec] < 0 ) {
472 // DoIt is inactive if ordering parameter is negative
473 pAttr->idxProcVector[ivec] = -1;
474
475 } else {
476 //add aProcess in ordering of ordProcVector
477 // G4ProcessVector* pVector = theProcVector[ivec];
478 // find insert position
479 G4int ip = FindInsertPosition(pAttr->ordProcVector[ivec], ivec);
480 // insert
481 InsertAt(ip, aProcess, ivec);
482 // set index in Process Attribute
483 pAttr->idxProcVector[ivec] = ip;
484
485#ifdef G4VERBOSE
486 if (verboseLevel>2) {
487 G4cout << "G4ProcessManager::AddProcess()" << G4endl;
488 G4cout << aProcess->GetProcessName() << " is inserted at "<< ip;
489 G4cout << " in ProcessVetor[" << ivec<< "]";
490 G4cout << " with Ordering parameter = " ;
491 G4cout << pAttr->ordProcVector[ivec] << G4endl;
492 }
493#endif
494 }
495 }
496
497 //add ProcessAttribute to ProcessAttrVector
498 theAttrVector->push_back(pAttr);
499
500 numberOfProcesses += 1;
501
502 // check consistencies between ordering parameters and process
503 CheckOrderingParameters(aProcess);
504
505 CreateGPILvectors();
506
507 // inform process manager pointer to the process
508 aProcess->SetProcessManager(this);
509
510 return idx;
511}

514// ///////////////////////////////////////
515G4VProcess* G4ProcessManager::RemoveProcess(G4int index)
516{
517 //find the process attribute
518 G4ProcessAttribute* pAttr = GetAttribute(index);
519 if (pAttr == nullptr) return nullptr;
520
521 // remove process
522 G4VProcess* removedProcess = (*theProcessList)[index];
523
524 if (!(pAttr->isActive)) { ActivateProcess(index);}
525 // remove process from vectors if the process is active
526 for (G4int ivec=0; ivec<SizeOfProcVectorArray; ivec++) {
527 G4ProcessVector* pVector = theProcVector[ivec];
528 G4int idx = pAttr->idxProcVector[ivec];
529 if ((idx >= 0) && (idx < pVector->entries())) {
530 //remove
531 if (RemoveAt(idx, removedProcess, ivec) <0) {
532 G4String anErrorMessage("Bad index in attribute");
533 anErrorMessage += "for particle[" + theParticleType->GetParticleName() + "] ";
534 anErrorMessage += "process[" + removedProcess->GetProcessName() + "] " ;
535 G4Exception( "G4ProcessManager::RemoveProcess()","Fatal Error",
536 FatalException,anErrorMessage);
537 return nullptr;
538 }
539 } else if (idx<0) {
540 // corresponding DoIt is not active
541 } else {
542 // idx is out of range
543 G4String anErrorMessage("Bad ProcessList : Index is out of range ");
544 anErrorMessage += "for particle[" + theParticleType->GetParticleName() + "] ";
545 anErrorMessage += "process[" + removedProcess->GetProcessName() + "] " ;
546 G4Exception( "G4ProcessManager::RemoveProcess()","ProcMan012",
547 FatalException,anErrorMessage);
548 return nullptr;
549 }
550 }
551 pAttr->isActive = false;
552 // remove from the process List and delete the attribute
553 theProcessList->removeAt(index);
554 G4ProcessAttrVector::iterator itr;
555 for (itr = theAttrVector->begin(); itr!= theAttrVector->end(); ++itr) {
556 if ( (*itr) == pAttr) {
557 theAttrVector->erase(itr);
558 break;
559 }
560 }
561 delete pAttr;
562 numberOfProcesses -= 1;
563
564 // correct index
565 for(G4int i=0; i<numberOfProcesses; i++) {
566 G4ProcessAttribute* aAttr = (*theAttrVector)[i];
567 if (index < aAttr->idxProcessList) aAttr->idxProcessList -=1;
568 }
569
570 CreateGPILvectors();
571
572 //remove aProcess from ProcesssTable
573 G4ProcessTable* theProcessTable = G4ProcessTable::GetProcessTable();
574 theProcessTable->Remove(removedProcess, this);
575
576 return removedProcess;
577}

604// ///////////////////////////////////////
605void G4ProcessManager::SetProcessOrdering(
606 G4VProcess *aProcess,
607 G4ProcessVectorDoItIndex idDoIt,
608 G4int ordDoIt
609 )
610{
611 const G4String aErrorMessage(" G4ProcessManager::SetProcessOrdering");
612
613#ifdef G4VERBOSE
614 if (GetVerboseLevel()>2) {
615 G4cout << aErrorMessage ;
616 G4cout << "particle[" + theParticleType->GetParticleName() +"] " ;
617 G4cout <<"process[" + aProcess->GetProcessName() + "]"<< G4endl;
618 }
619#endif
620
621 // get Process Vector Id
622 G4int ivec = GetProcessVectorId(idDoIt, typeDoIt);
623 if (ivec <0 ) {
624#ifdef G4VERBOSE
625 if (verboseLevel>0) {
626 G4cout << aErrorMessage << G4endl;
627 G4cout << "particle[" << theParticleType->GetParticleName() << "] " ;
628 G4cout << "process[" << aProcess->GetProcessName() << "]"<< G4endl;
629 G4cout << " illegal DoIt Index [= " << G4int(idDoIt) << "]";
630 G4cout << G4endl;
631 }
632#endif
633 return;
634 }
635
636 if (ordDoIt>ordLast) ordDoIt=ordLast;
637 // get attribute
638 G4ProcessAttribute* pAttr = GetAttribute(aProcess);
639 if (pAttr == nullptr) {
640 // can not get process attribute
641 return;
642
643 } else {
644 G4int ip = pAttr->idxProcVector[ivec];
645 // remove a process from the process vector
646 if ( ip >=0 ) {
647 RemoveAt(ip, aProcess, ivec);
648 }
649
650 // set ordering parameter to non-zero
651 if (ordDoIt == 0) ordDoIt = 1;
652 pAttr->ordProcVector[ivec-1] = ordDoIt;
653 pAttr->ordProcVector[ivec] = ordDoIt;
654
655 // insert in process vector if ordDoIt >0
656 if (ordDoIt >0) {
657 // find insert position
658 ip = FindInsertPosition(pAttr->ordProcVector[ivec], ivec);
659 // insert
660 InsertAt(ip, aProcess, ivec);
661 // set index in Process Attribute
662 pAttr->idxProcVector[ivec] = ip;
663#ifdef G4VERBOSE
664 if (verboseLevel>2) {
665 G4cout << aErrorMessage << G4endl;
666 G4cout << "particle[" << theParticleType->GetParticleName() << "] " ;
667 G4cout <<"process[" << aProcess->GetProcessName() << "]"<< G4endl;
668 G4cout << aProcess->GetProcessName() << " is inserted at "<< ip;
669 G4cout << " in ProcessVetor[" << ivec<< "]";
670 G4cout << " with Ordering parameter = " << ordDoIt ;
671 G4cout << G4endl;
672 }
673#endif
674 }
675
676 }
677 // check consistencies between ordering parameters and process
678 CheckOrderingParameters(aProcess);
679
680 // create GPIL vectors
681 CreateGPILvectors();
682}

685// ///////////////////////////////////////
686void G4ProcessManager::SetProcessOrderingToFirst(
687 G4VProcess *aProcess,
688 G4ProcessVectorDoItIndex idDoIt
689 )
690{
691 // get Process Vector Id(
692 G4int ivec = GetProcessVectorId(idDoIt, typeDoIt);
693 if (ivec <0 ) {
694#ifdef G4VERBOSE
695 if (verboseLevel>0) {
696 G4cout << "G4ProcessManager::SetProcessOrdering: ";
697 G4cout << " illegal DoIt Index [= " << G4int(idDoIt) << "]";
698 G4cout << G4endl;
699 }
700#endif
701 return;
702 }
703
704 // get attribute
705 G4ProcessAttribute* pAttr = GetAttribute(aProcess);
706 if (pAttr == nullptr) {
707 return;
708 } else {
709 G4int ip = pAttr->idxProcVector[ivec];
710
711 // remove a process from the process vector
712 if ( ip >=0 ) {
713 RemoveAt(ip, aProcess, ivec);
714 }
715
716 // set ordering parameter to zero
717 pAttr->ordProcVector[ivec] = 0;
718 pAttr->ordProcVector[ivec-1] = 0;
719
720 // insert
721 InsertAt(0, aProcess, ivec);
722
723 // set index in Process Attribute
724 pAttr->idxProcVector[ivec] = 0;
725
726#ifdef G4VERBOSE

755// ///////////////////////////////////////
756void G4ProcessManager::SetProcessOrderingToSecond(
757 G4VProcess *aProcess,
758 G4ProcessVectorDoItIndex idDoIt
759 )
760{
761 const G4String aErrorMessage(" G4ProcessManager::SetProcessOrderingToSecond");
762
763#ifdef G4VERBOSE
764 if (GetVerboseLevel()>2) {
765 G4cout << aErrorMessage ;
766 G4cout << "particle[" << theParticleType->GetParticleName() << "] " ;
767 G4cout <<"process[" << aProcess->GetProcessName() << "]"<< G4endl;
768 }
769#endif
770
771 // get Process Vector Id
772 G4int ivec = GetProcessVectorId(idDoIt, typeDoIt);
773 if (ivec <0 ) {
774#ifdef G4VERBOSE
775 if (verboseLevel>0) {
776 G4cout << aErrorMessage << G4endl;
777 G4cout << "particle[" << theParticleType->GetParticleName() << "] " ;
778 G4cout << "process[" << aProcess->GetProcessName() << "]"<< G4endl;
779 G4cout << " illegal DoIt Index [= " << G4int(idDoIt) << "]";
780 G4cout << G4endl;
781 }
782#endif
783 return;
784 }
785
786 // get attribute
787 G4ProcessAttribute* pAttr = GetAttribute(aProcess);
788 if (pAttr == nullptr) {
789 // can not get process attribute
790 return;
791 } else {
792 G4int ip = pAttr->idxProcVector[ivec];
793 // remove a process from the process vector
794 if ( ip >=0 ) {
795 RemoveAt(ip, aProcess, ivec);
796 }
797 }
798
799 // set ordering parameter to 1
800 pAttr->ordProcVector[ivec-1] = 0;
801 pAttr->ordProcVector[ivec] = 0;
802
803 // find insert position
804 G4ProcessVector* pVector = theProcVector[ivec];
805 G4int ip = pVector->entries();
806 G4int tmp = INT_MAX;
807
808 // find insert position
809 for (G4int iproc=0; iproc<numberOfProcesses; iproc++) {
810 G4ProcessAttribute* aAttr = (*theAttrVector)[iproc];
811 if ( aAttr->idxProcVector[ivec] >= 0 ) {
812 if ( (aAttr->ordProcVector[ivec] !=0 ) &&
813 (tmp >= aAttr->ordProcVector[ivec]) ) {
814 tmp = aAttr->ordProcVector[ivec];
815 if ( ip > aAttr->idxProcVector[ivec] ) {
816 ip = aAttr->idxProcVector[ivec] ;
817 }
818 }
819 }
820 }
821
822 // insert
823 InsertAt(ip, aProcess, ivec);
824
825 // set index in Process Attribute
826 pAttr->idxProcVector[ivec] = ip;
827#ifdef G4VERBOSE
828 if (verboseLevel>2) {
829 G4cout << aErrorMessage << G4endl;
830 G4cout << "particle[" << theParticleType->GetParticleName() << "] " ;
831 G4cout <<"process[" << aProcess->GetProcessName() << "]"<< G4endl;
832 G4cout << aProcess->GetProcessName() << " is inserted at "<< ip;
833 G4cout << " in ProcessVetor[" << ivec<< "]";
834 G4cout << " with Ordering parameter = 1 ";
835 G4cout << G4endl;
836 }
837#endif
838
839 // check consistencies between ordering parameters and process
840 CheckOrderingParameters(aProcess);
841
842 // create GPIL vectors
843 CreateGPILvectors();
844}

1056void G4ProcessManager::CreateGPILvectors()
1057{
1058//-- create GetPhysicalInteractionLength process vectors just as the inverse
1059//-- order of DoIt process vector
1060 for(G4int k=0; k<theProcessList->entries(); k++) {
1061 GetAttribute((*theProcessList)[k])->idxProcVector[0]=-1;
1062 GetAttribute((*theProcessList)[k])->idxProcVector[2]=-1;
1063 GetAttribute((*theProcessList)[k])->idxProcVector[4]=-1;
1064 }
1065
1066 for(G4int i=0; i<SizeOfProcVectorArray; i += 2) {
1067 G4ProcessVector* procGPIL = theProcVector[i];
1068 G4ProcessVector* procDoIt = theProcVector[i+1];
1069 G4int nproc = procDoIt->entries();
1070 procGPIL->clear();
1071 for(G4int j=nproc-1;j>=0;j--) {
1072 G4VProcess* aProc = (*procDoIt)[j];
1073 procGPIL->insert(aProc);
1074 GetAttribute(aProc)->idxProcVector[i] = procGPIL->entries()-1;
1075 }
1076 }
1077
1078}

1154void G4ProcessManager::CheckOrderingParameters(G4VProcess* aProcess) const
1155{
1156 if (aProcess == nullptr) return;
1157 G4ProcessAttribute* pAttr = GetAttribute(aProcess);
1158 if (pAttr == nullptr) {
1159#ifdef G4VERBOSE
1160 if (GetVerboseLevel()>0) {
1161 G4cout << "G4ProcessManager::CheckOrderingParameters ";
1162 G4cout << " process " << aProcess->GetProcessName()
1163 << " has no attribute" << G4endl;
1164 }
1165#endif
1166 return;
1167 }
1168
1169 // check consistencies between ordering parameters and
1170 // validity of DoIt of the Process
1171 G4bool isOK =true;
1172 if ( (pAttr->ordProcVector[0]>=0) && (!aProcess->isAtRestDoItIsEnabled()) ){
1173 #ifdef G4VERBOSE
1174 if (GetVerboseLevel()>0) {
1175 G4cerr << "G4ProcessManager::CheckOrderingParameters ";
1176 G4cerr << "You cannot set ordering parameter ["
1177 << pAttr->ordProcVector[0]
1178 << "] for AtRest DoIt to the process "
1179 << aProcess->GetProcessName() << G4endl;
1180 }
1181#endif
1182 isOK = false;
1183 }
1184
1185 if ( (pAttr->ordProcVector[2]>=0) && (!aProcess->isAlongStepDoItIsEnabled()) ){
1186#ifdef G4VERBOSE
1187 if (GetVerboseLevel()>0) {
1188 G4cerr << "G4ProcessManager::CheckOrderingParameters ";
1189 G4cerr << "You cannot set ordering parameter ["
1190 << pAttr->ordProcVector[2]
1191 << "] for AlongStep DoIt to the process "
1192 << aProcess->GetProcessName() << G4endl;
1193
1194 }
1195#endif
1196 isOK = false;
1197 }
1198
1199 if ( (pAttr->ordProcVector[4]>=0) && (!aProcess->isPostStepDoItIsEnabled()) ) {
1200#ifdef G4VERBOSE
1201 if (GetVerboseLevel()>0) {
1202 G4cerr << "G4ProcessManager::CheckOrderingParameters ";
1203 G4cerr << "You cannot set ordering parameter ["
1204 << pAttr->ordProcVector[4]
1205 << "] for PostStep DoIt to the process"
1206 << aProcess->GetProcessName() << G4endl;
1207 }
1208#endif
1209 isOK = false;
1210 }
1211
1212 if (!isOK) {
1213 G4String msg;
1214 msg = "Invalid ordering parameters are set for ";
1215 msg += aProcess->GetProcessName();
1216 G4Exception( "G4ProcessManager::CheckOrderingParameters ",
1217 "ProcMan013",FatalException, msg);
1218 }
1219
1220 return;
1221}
