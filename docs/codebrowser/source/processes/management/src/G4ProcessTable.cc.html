70// destructor //////////////////////////
71G4ProcessTable::~G4ProcessTable()
72{
73#ifdef G4VERBOSE
74 if (verboseLevel>1){
75 G4cout << "-- G4ProcessTable destructor --" << G4endl;
76 }

86 G4ProcTableVector::iterator idx;
87
88 for (idx=fProcTblVector->begin(); idx!=fProcTblVector->end(); ++idx) {
89 delete (*idx);
90 }
91 fProcTblVector ->clear();
92 delete fProcTblVector;
93 fProcTblVector = nullptr;
94 }
95
96 if ( fProcNameVector != nullptr) {
97 fProcNameVector ->clear();
98 delete fProcNameVector;
99 fProcNameVector = nullptr;
100 }
101 fProcessTable = nullptr;
102}

164//////////////////////////
165G4int G4ProcessTable::Insert(G4VProcess* aProcess,
166 G4ProcessManager* aProcMgr)
167{
168 if ( (aProcess == nullptr) || ( aProcMgr == nullptr ) ){
169#ifdef G4VERBOSE
170 if (verboseLevel>0){
171 G4cout << "G4ProcessTable::Insert : arguments are 0 pointer "
172 <<aProcess <<","<< aProcMgr << G4endl;
173 }
174#endif
175 return -1;
176 }
177
178#ifdef G4VERBOSE
179 if (verboseLevel>1){
180 G4cout << "G4ProcessTable::Insert ";
181 G4cout << " Process[" << aProcess->GetProcessName() << "]";
182 G4cout << " Particle[" << aProcMgr->GetParticleType()->GetParticleName() << "]";
183 G4cout << G4endl;
184 }
185#endif
186
187 G4ProcTableVector::iterator itr;
188 G4int idxTbl=0;
189 G4ProcTblElement* anElement = nullptr;
190 G4bool isFoundInTbl = false;
191 // loop over all elements
192 for (itr=fProcTblVector->begin(); itr!=fProcTblVector->end(); ++itr, ++idxTbl) {
193 anElement = (*itr);
194 // check if this process is included
195 if (aProcess == anElement->GetProcess()) {
196 isFoundInTbl = true;
197
198 // add the process manager into the element
199 // unless this process manager is included
200 if (!anElement->Contains(aProcMgr)) {
201 anElement->Insert(aProcMgr);
202#ifdef G4VERBOSE
203 if (verboseLevel>2){
204 G4cout << " This Process Manager is registered !! " << G4endl;
205 }
206#endif
207 }
208 break;
209 }
210 }
211 // add this process into the table by creating a new element
212 if (!isFoundInTbl) {
213 G4ProcTblElement* newElement = new G4ProcTblElement(aProcess);
214 newElement->Insert(aProcMgr);
215 fProcTblVector->push_back(newElement);
216 // add into name vector
217 G4ProcNameVector::iterator ip;
218 G4bool isFound = false;
219 for (ip=fProcNameVector->begin(); ip!=fProcNameVector->end(); ++ip) {
220 isFound |= (aProcess->GetProcessName() == (*ip));
221 }
222 if (!isFound) {
223 fProcNameVector->push_back(aProcess->GetProcessName() );
224#ifdef G4VERBOSE
225 if (verboseLevel>2){
226 G4cout << " This Process is registered !! " << G4endl;
227 }
228#endif
229 }
230 }
231 return idxTbl;
232}

234//////////////////////////
235G4int G4ProcessTable::Remove( G4VProcess* aProcess,
236 G4ProcessManager* aProcMgr)
237{
238 if ( (aProcess == nullptr) || ( aProcMgr == nullptr ) ){
239#ifdef G4VERBOSE
240 if (verboseLevel>0){
241 G4cout << "G4ProcessTable::Remove : arguments are 0 pointer "<< G4endl;
242 }
243#endif
244 return -1;
245 }
246
247#ifdef G4VERBOSE
248 if (verboseLevel>1){
249 G4cout << "G4ProcessTable::Remove ";
250 G4cout << " Process[" << aProcess->GetProcessName() << "]";
251 G4cout << " Particle[" << aProcMgr->GetParticleType()->GetParticleName() << "]" << G4endl;
252 }
253#endif
254
255 G4ProcTableVector::iterator itr;
256 G4int idxTbl=0;
257 G4ProcTblElement* anElement =nullptr;
258 G4bool isFound = false;
259 // loop over all elements
260 for (itr=fProcTblVector->begin(); itr!=fProcTblVector->end(); ++itr, ++idxTbl) {
261 anElement = (*itr);
262
263 // check if this process is included
264 if (aProcess == anElement->GetProcess()) {
265 isFound = anElement->Contains(aProcMgr);
266 // remove the process manager from the element
267 anElement->Remove(aProcMgr);
268#ifdef G4VERBOSE
269 if (verboseLevel>2){
270 G4cout << " This Process Manager is removed !! " << G4endl;
271 }
272#endif
273 break;
274 }
275 }
276 //
277 if (!isFound) {
278#ifdef G4VERBOSE
279 if (verboseLevel>0){
280 G4cout << " This Process Manager is not registered !! " << G4endl;
281 }
282#endif
283 return -1;
284 }
285 // remove the element if it has no entry
286 if (anElement->Length() == 0){
287 fProcTblVector->erase(itr);
288 delete anElement;
289 // check other prcesses with same name exist or not
290 G4bool isSameName = false;
291 for (itr=fProcTblVector->begin(); itr!=fProcTblVector->end(); ++itr) {
292 anElement = (*itr);
293 if (anElement->GetProcessName() == aProcess->GetProcessName()) {
294 isSameName = true;
295 break;
296 }
297 }
298 // remove from name vector
299 if (!isSameName ) {
300 G4ProcNameVector::iterator i;
301 for (i=fProcNameVector->begin(); i!=fProcNameVector->end(); ++i) {
302 if ( *i == aProcess->GetProcessName() ) {
303 fProcNameVector->erase(i);
304 break;
305 }
306 }
307 }
308#ifdef G4VERBOSE
309 if (verboseLevel>1){
310 G4cout << " This Process is removed !! " << G4endl;
311 }
312#endif
313 }
314 return idxTbl;
315}

317//////////////////////////
318G4VProcess* G4ProcessTable::FindProcess(const G4String& processName,
319 const G4ProcessManager* processManager)
320 const
321{
322 G4ProcTableVector::iterator itr;
323 G4int idxTbl = 0;
324 G4bool isFound = false;
325 G4ProcTblElement* anElement =nullptr;
326 for (itr=fProcTblVector->begin(); itr!=fProcTblVector->end(); ++itr, ++idxTbl) {
327 anElement = (*itr);
328 // check name
329 if ( anElement->GetProcessName() == processName ) {
330 // check if the processManage is included
331 if ( anElement->Contains(processManager) ) {
332 isFound = true;
333 break;
334 }
335 }
336 }
337#ifdef G4VERBOSE
338 if (!isFound && verboseLevel>1){
339 G4cout << " G4ProcessTable::FindProcess :" ;
340 G4cout << " The Process[" << processName << "] is not found ";
341 G4cout << " for " << processManager->GetParticleType()->GetParticleName() << G4endl;
342 }
343#endif
344
345 if (isFound) return anElement->GetProcess();
346 else return nullptr;
347}

349///////////////
350G4ProcessTable::G4ProcTableVector* G4ProcessTable::Find(
351 G4ProcTableVector*,
352 const G4String& processName )
353{
354 tmpTblVector->clear();
355
356 G4ProcTableVector::iterator itr;
357 G4bool isFound = false;
358 G4ProcTblElement* anElement =nullptr;
359 for (itr=fProcTblVector->begin(); itr!=fProcTblVector->end(); ++itr) {
360 anElement = (*itr);
361 // check name
362 if ( anElement->GetProcessName() == processName ) {
363 isFound = true;
364 tmpTblVector->push_back(anElement);
365 }
366 }
367
368 if (!isFound && verboseLevel>0){
369#ifdef G4VERBOSE
370 G4cout << " G4ProcessTable::Find :" ;
371 G4cout << " The Process[" << processName << "] is not found " << G4endl;
372#endif
373 }
374
375 return tmpTblVector;
376
377}

448///////////////
449void G4ProcessTable::SetProcessActivation( const G4String& processName,
450 G4bool fActive )
451{
452#ifdef G4VERBOSE
453 if (verboseLevel>1){
454 G4cout << " G4ProcessTable::SetProcessActivation:" ;
455 G4cout << " The Process[" << processName << "] "<< G4endl;
456 }
457#endif
458
459 G4ProcTableVector* pTblVector = Find(fProcTblVector, processName);
460 G4ProcTableVector::iterator itr;
461 G4ProcTblElement* anElement;
462 // loop over all elements
463 for (itr=pTblVector->begin(); itr!=pTblVector->end(); ++itr) {
464 anElement = (*itr);
465 G4VProcess* process = anElement->GetProcess();
466 for (G4int idx = 0 ; idx < anElement->Length(); idx++) {
467 G4ProcessManager* manager = anElement->GetProcessManager(idx);
468 manager->SetProcessActivation(process, fActive);
469#ifdef G4VERBOSE
470 if (verboseLevel>1){
471 G4cout << " for " << manager->GetParticleType()->GetParticleName();
472 G4cout << " Index = " << manager->GetProcessIndex(process);
473 G4cout << G4endl;
474 }
475#endif
476 }
477 }

572/////////////
573void G4ProcessTable::DumpInfo(G4VProcess* process,
574 G4ParticleDefinition* particle)
575{
576 G4ProcTableVector::iterator itr;
577 G4int idxTbl=0;
578 G4ProcTblElement* anElement =nullptr;
579 G4bool isFoundInTbl = false;
580 G4ProcessManager* manager =nullptr;
581 G4int idx;
582 // loop over all elements
583 for (itr=fProcTblVector->begin(); itr!=fProcTblVector->end(); ++itr, ++idxTbl) {
584 anElement = (*itr);
585 if (process == anElement->GetProcess() ){
586 if (particle != nullptr) {
587 for (idx=0; idx<anElement->Length(); idx++){
588 manager = anElement->GetProcessManager(idx);
589 if (particle == manager->GetParticleType()) {
590 isFoundInTbl = true;
591 break;
592 }
593 }
594 } else {
595 isFoundInTbl = true;
596 }
597 break;
598 }
599 }
600 if (!isFoundInTbl ) return;
601
602 G4int tmpVerbose = process->GetVerboseLevel();
603 process->SetVerboseLevel(verboseLevel);
604 process->DumpInfo();
605 process->SetVerboseLevel(tmpVerbose);
606 if (particle == nullptr) {
607 for (idx=0; idx<anElement->Length(); idx++){
608 manager = anElement->GetProcessManager(idx);
609 G4cout << " for " << manager->GetParticleType()->GetParticleName();
610 G4cout << G4endl;
611#ifdef G4VERBOSE
612 if (verboseLevel >2){
613 tmpVerbose = manager->GetVerboseLevel();
614 manager->SetVerboseLevel(verboseLevel);
615 manager->DumpInfo();
616 manager->SetVerboseLevel(tmpVerbose);
617 }
618#endif
619 }
620 } else {
621 G4cout << " for " << manager->GetParticleType()->GetParticleName();
622 G4cout << G4endl;
623#ifdef G4VERBOSE
624 if (verboseLevel >2){
625 tmpVerbose = manager->GetVerboseLevel();
626 manager->SetVerboseLevel(verboseLevel);
627 manager->DumpInfo();
628 manager->SetVerboseLevel(tmpVerbose);
629 }
630#endif
631 }
