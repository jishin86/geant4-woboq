257G4bool G4RadioactiveDecay::IsApplicable(const G4ParticleDefinition& aParticle)
258{
259 // All particles other than G4Ions are rejected by default
260 if (((const G4Ions*)(&aParticle))->GetExcitationEnergy() > 0.) {
261 return true; // Not ground state - decay
262 }
263
264 if (aParticle.GetParticleName() == "GenericIon") {
265 return true;
266 } else if (!(aParticle.GetParticleType() == "nucleus")
267 || aParticle.GetPDGLifeTime() < 0. ) {
268 return false; // Nuclide is stable - no decay
269 }
270
271 // At this point nuclide must be an unstable ground state
272 // Determine whether it falls into the correct A and Z range
273 G4int A = ((const G4Ions*) (&aParticle))->GetAtomicMass();
274 G4int Z = ((const G4Ions*) (&aParticle))->GetAtomicNumber();
275
276 if (A > theNucleusLimits.GetAMax() || A < theNucleusLimits.GetAMin())
277 {return false;}
278 else if (Z > theNucleusLimits.GetZMax() || Z < theNucleusLimits.GetZMin())
279 {return false;}
280 return true;
281}

479// ConvolveSourceTimeProfile performs the convolution of the source time profile
480// function with a single exponential characterized by a decay constant in the
481// decay chain. The time profile is treated as a set of step functions so that
482// the convolution integral can be done bin-by-bin.
483// This implements Eq. 4.13 of DERA technical note, with SProfile[i] = F(t')
484
485G4double
486G4RadioactiveDecay::ConvolveSourceTimeProfile(const G4double t, const G4double tau)
487{
488 G4double convolvedTime = 0.0;
489 G4int nbin;
490 if ( t > SBin[NSourceBin]) {
491 nbin = NSourceBin;
492 } else {
493 nbin = 0;
494
495 G4int loop = 0;
496 G4ExceptionDescription ed;
497 ed << " While count exceeded " << G4endl;
498 while (t > SBin[nbin]) { // Loop checking, 01.09.2015, D.Wright
499 loop++;
500 if (loop > 1000) {
501 G4Exception("G4RadioactiveDecay::ConvolveSourceTimeProfile()",
502 "HAD_RDM_100", JustWarning, ed);
503 break;
504 }
505 nbin++;
506 }
507 nbin--;
508 }
509
510 // Use expm1 wherever possible to avoid large cancellation errors in
511 // 1 - exp(x) for small x
512 G4double earg = 0.0;
513 if (nbin > 0) {
514 for (G4int i = 0; i < nbin; i++) {
515 earg = (SBin[i+1] - SBin[i])/tau;
516 if (earg < 100.) {
517 convolvedTime += SProfile[i] * std::exp((SBin[i] - t)/tau) *
518 std::expm1(earg);
519 } else {
520 convolvedTime += SProfile[i] *
521 (std::exp(-(t-SBin[i+1])/tau)-std::exp(-(t-SBin[i])/tau));
522 }
523 }
524 }
525 convolvedTime -= SProfile[nbin] * std::expm1((SBin[nbin] - t)/tau);
526 // tau divided out of final result to provide probability of decay in window
527
528 if (convolvedTime < 0.) {
529 G4cout << " Convolved time =: " << convolvedTime << " reset to zero! " << G4endl;
530 G4cout << " t = " << t << " tau = " << tau << G4endl;
531 G4cout << SBin[nbin] << " " << SBin[0] << G4endl;
532 convolvedTime = 0.;
533 }
534#ifdef G4VERBOSE
535 if (GetVerboseLevel() > 1)
536 G4cout << " Convolved time: " << convolvedTime << G4endl;
537#endif
538 return convolvedTime;
539}

603G4double G4RadioactiveDecay::GetMeanLifeTime(const G4Track& theTrack,
604 G4ForceCondition*)
605{
606 // For variance reduction the time is set to 0 so as to force the particle
607 // to decay immediately.
608 // In analogueMC mode it returns the particle's mean-life.
609
610 G4double meanlife = 0.;
611 if (AnalogueMC) {
612 const G4DynamicParticle* theParticle = theTrack.GetDynamicParticle();
613 const G4ParticleDefinition* theParticleDef = theParticle->GetDefinition();
614 G4double theLife = theParticleDef->GetPDGLifeTime();
615#ifdef G4VERBOSE
616 if (GetVerboseLevel() > 2) {
617 G4cout << "G4RadioactiveDecay::GetMeanLifeTime() " << G4endl;
618 G4cout << "KineticEnergy: " << theParticle->GetKineticEnergy()/GeV
619 << " GeV, Mass: " << theParticle->GetMass()/GeV
620 << " GeV, Life time: " << theLife/ns << " ns " << G4endl;
621 }
622#endif
623 if (theParticleDef->GetPDGStable()) {meanlife = DBL_MAX;}
624 else if (theLife < 0.0) {meanlife = DBL_MAX;}
625 else {meanlife = theLife;}
626 // Set meanlife to zero for excited istopes which are not in the
627 // RDM database
628 if (((const G4Ions*)(theParticleDef))->GetExcitationEnergy() > 0. &&
629 meanlife == DBL_MAX) {meanlife = 0.;}
630 }
631#ifdef G4VERBOSE
632 if (GetVerboseLevel() > 1)
633 G4cout << " mean life time: " << meanlife/s << " s " << G4endl;
634#endif
635
636 return meanlife;
637}

657 << " GeV, Mass: " << aMass/GeV << " GeV, tau: " << tau << " ns "
658 << G4endl;
659 }
660#endif
661 G4double pathlength = DBL_MAX;
662 if (tau != -1) {
663 // Ion can decay
664
665 if (tau < -1000.0) {
666 pathlength = DBL_MIN; // nuclide had very short lifetime or wasn't in table
667
668 } else if (tau < 0.0) {
669 G4cout << aParticleDef->GetParticleName() << " has lifetime " << tau << G4endl;
670 G4ExceptionDescription ed;
671 ed << "Ion has negative lifetime " << tau
672 << " but is not stable. Setting mean free path to DBL_MAX" << G4endl;
673 G4Exception("G4RadioactiveDecay::GetMeanFreePath()", "HAD_RDM_011",
674 JustWarning, ed);
675 pathlength = DBL_MAX;
676
677 } else {
678 // Calculate mean free path
679 G4double betaGamma = aParticle->GetTotalMomentum()/aMass;
680 pathlength = c_light*tau*betaGamma;
681
682 if (pathlength < DBL_MIN) {
683 pathlength = DBL_MIN;
684#ifdef G4VERBOSE
685 if (GetVerboseLevel() > 2) {
686 G4cout << "G4Decay::GetMeanFreePath: "
687 << aParticleDef->GetParticleName()
688 << " stops, kinetic energy = "
689 << aParticle->GetKineticEnergy()/keV <<" keV " << G4endl;
690 }
691#endif
692 }
693 }
694 }
695
696#ifdef G4VERBOSE
697 if (GetVerboseLevel() > 1) {
698 G4cout << "mean free path: "<< pathlength/m << " m" << G4endl;
699 }
700#endif
701 return pathlength;
702}

1172CalculateChainsFromParent(const G4ParticleDefinition& theParentNucleus)
1173{
1174 // Use extended Bateman equation to calculate the radioactivities of all
1175 // progeny of theParentNucleus. The coefficients required to do this are
1176 // calculated using the method of P. Truscott (Ph.D. thesis and
1177 // DERA Technical Note DERA/CIS/CIS2/7/36/4/10) 11 January 2000.
1178 // Coefficients are then added to the decay rate table vector
1179
1180 // Create and initialise variables used in the method.
1181 theDecayRateVector.clear();
1182
1183 G4int nGeneration = 0;
1184
1185 std::vector<G4double> taos;
1186
1187 // Dimensionless A coefficients of Eqs. 4.24 and 4.25 of the TN
1188 std::vector<G4double> Acoeffs;
1189
1190 // According to Eq. 4.26 the first coefficient (A_1:1) is -1
1191 Acoeffs.push_back(-1.);
1192
1193 G4int A = ((const G4Ions*)(&theParentNucleus))->GetAtomicMass();
1194 G4int Z = ((const G4Ions*)(&theParentNucleus))->GetAtomicNumber();
1195 G4double E = ((const G4Ions*)(&theParentNucleus))->GetExcitationEnergy();
1196 G4double tao = theParentNucleus.GetPDGLifeTime();
1197 if (tao < 0.) tao = 1e-100;
1198 taos.push_back(tao);
1199 G4int nEntry = 0;
1200
1201 // Fill the decay rate container (G4RadioactiveDecayRatesToDaughter) with
1202 // the parent isotope data
1203 SetDecayRate(Z,A,E,nGeneration,Acoeffs,taos);
1204
1205 // store the decay rate in decay rate vector
1206 theDecayRateVector.push_back(ratesToDaughter);
1207 nEntry++;
1208
1209 // Now start treating the secondary generations.
1210 G4bool stable = false;
1211 G4int i;
1212 G4int j;
1213 G4VDecayChannel* theChannel = 0;
1214 G4NuclearDecay* theNuclearDecayChannel = 0;
1215
1216 G4ITDecay* theITChannel = 0;
1217 G4BetaMinusDecay* theBetaMinusChannel = 0;
1218 G4BetaPlusDecay* theBetaPlusChannel = 0;
1219 G4AlphaDecay* theAlphaChannel = 0;
1220 G4ProtonDecay* theProtonChannel = 0;
1221 G4NeutronDecay* theNeutronChannel = 0;
1222 G4RadioactiveDecayMode theDecayMode;
1223 G4double theBR = 0.0;
1224 G4int AP = 0;
1225 G4int ZP = 0;
1226 G4int AD = 0;
1227 G4int ZD = 0;
1228 G4double EP = 0.;
1229 std::vector<G4double> TP;
1230 std::vector<G4double> RP; // A coefficients of the previous generation
1231 G4ParticleDefinition *theDaughterNucleus;
1232 G4double daughterExcitation;
1233 G4double nearestEnergy = 0.0;
1234 G4int nearestLevelIndex = 0;
1235 G4ParticleDefinition *aParentNucleus;
1236 G4IonTable* theIonTable;
1237 G4DecayTable* parentDecayTable;
1238 G4double theRate;
1239 G4double TaoPlus;
1240 G4int nS = 0; // Running index of first decay in a given generation
1241 G4int nT = nEntry; // Total number of decays accumulated over entire history
1242 const G4int nMode = 9;
1243 G4double brs[nMode];
1244 //
1245 theIonTable =
1246 (G4IonTable*)(G4ParticleTable::GetParticleTable()->GetIonTable());
1247
1248 G4int loop = 0;
1249 G4ExceptionDescription ed;
1250 ed << " While count exceeded " << G4endl;
1251
1252 while (!stable) { /* Loop checking, 01.09.2015, D.Wright */
1253 loop++;
1254 if (loop > 10000) {
1255 G4Exception("G4RadioactiveDecay::CalculateChainsFromParent()", "HAD_RDM_100", JustWarning, ed);
1256 break;
1257 }
1258 nGeneration++;
1259 for (j = nS; j < nT; j++) {
1260 // First time through, get data for parent nuclide
1261 ZP = theDecayRateVector[j].GetZ();
1262 AP = theDecayRateVector[j].GetA();
1263 EP = theDecayRateVector[j].GetE();
1264 RP = theDecayRateVector[j].GetDecayRateC();
1265 TP = theDecayRateVector[j].GetTaos();
1266 if (GetVerboseLevel() > 0) {
1267 G4cout << "G4RadioactiveDecay::CalculateChainsFromParent: daughters of ("
1268 << ZP << ", " << AP << ", " << EP
1269 << ") are being calculated, generation = " << nGeneration
1270 << G4endl;
1271 }
1272// G4cout << " Taus = " << G4endl;
1273// for (G4int ii = 0; ii < TP.size(); ii++) G4cout << TP[ii] << ", " ;
1274// G4cout << G4endl;
1275
1276 aParentNucleus = theIonTable->GetIon(ZP,AP,EP);
1277 parentDecayTable = GetDecayTable(aParentNucleus);
1278
1279 G4DecayTable* summedDecayTable = new G4DecayTable();
1280 // This instance of G4DecayTable is for accumulating BRs and decay
1281 // channels. It will contain one decay channel per type of decay
1282 // (alpha, beta, etc.); its branching ratio will be the sum of all
1283 // branching ratios for that type of decay of the parent. If the
1284 // halflife of a particular channel is longer than some threshold,
1285 // that channel will be inserted specifically and its branching
1286 // ratio will not be included in the above sums.
1287 // This instance is not used to perform actual decays.
1288
1289 for (G4int k = 0; k < nMode; k++) brs[k] = 0.0;
1290
1291 // Go through the decay table and sum all channels having the same decay mode
