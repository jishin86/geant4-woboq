35#ifndef G4NuclearDecay_h
36#define G4NuclearDecay_h 1
37
38#include "G4VDecayChannel.hh"
39#include "G4RadioactiveDecayMode.hh"
40#include "G4IonTable.hh"
41
42
43class G4NuclearDecay : public G4VDecayChannel
44{
45 public:
46 G4NuclearDecay(const G4String& channelName,
47 const G4RadioactiveDecayMode& mode,
48 const G4double& excitation,
49 const G4Ions::G4FloatLevelBase& floatingLevel);
50
51 virtual ~G4NuclearDecay();
52
53 G4RadioactiveDecayMode GetDecayMode() {return theMode;}
54
55 G4double GetDaughterExcitation() {return daughterEx;}
56
57 G4Ions::G4FloatLevelBase GetFloatingLevel() {return floatingLevel;}
58
59 G4ParticleDefinition* GetDaughterNucleus() {return GetDaughter(0);}
60
61 void SetHLThreshold(G4double HLT) {halflifeThreshold = HLT;}
62 G4double GetHLThreshold() {return halflifeThreshold;}
63
64 virtual void DumpNuclearInfo() = 0;
65
66 protected:
67 const G4RadioactiveDecayMode theMode;
68
69 private:
70 // Needed for variance reduction mode
71 const G4double daughterEx;
72 const G4Ions::G4FloatLevelBase floatingLevel;
73 G4double halflifeThreshold;
74};
75#endif
