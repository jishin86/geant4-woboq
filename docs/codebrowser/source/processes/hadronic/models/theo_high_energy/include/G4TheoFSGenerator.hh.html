29// Class Description
30// Final state production model for theoretical models of hadron inelastic
31// scattering in geant4;
32// To be used in your physics list in case you need this physics.
33// In this case you want to register an object of this class with
34// the corresponding process.
35// Note: This class is part of an implementation framework. You need to
36// register corresponding high energy generators and transport codes to
37// fill it with life; decay of strong resonances is done directly,
38// in case there is no residual nucleus.
39// Class Description - End
40
41#ifndef G4TheoFSGenerator_h
42#define G4TheoFSGenerator_h 1
43
44#include "G4VIntraNuclearTransportModel.hh"
45#include "G4QuasiElasticChannel.hh"
46#include "G4HadronicInteraction.hh"
47#include "G4VHighEnergyGenerator.hh"
48#include "G4DecayStrongResonances.hh"
49#include "G4HadFinalState.hh"
50#include "G4QuasiElasticChannel.hh"
51
52class G4TheoFSGenerator : public G4HadronicInteraction
53
54{
55 public:
56 G4TheoFSGenerator(const G4String& name = "TheoFSGenerator");
57 ~G4TheoFSGenerator();
58
59 private:
60 G4TheoFSGenerator(const G4TheoFSGenerator &right);
61 const G4TheoFSGenerator & operator=(const G4TheoFSGenerator &right);
62 int operator==(const G4TheoFSGenerator &right) const;
63 int operator!=(const G4TheoFSGenerator &right) const;
64
65 public:
66 G4HadFinalState * ApplyYourself(const G4HadProjectile & thePrimary, G4Nucleus & theNucleus);
67 void SetTransport(G4VIntraNuclearTransportModel *const value);
68 void SetHighEnergyGenerator(G4VHighEnergyGenerator *const value);
69 void SetQuasiElasticChannel(G4QuasiElasticChannel *const value);
70 virtual std::pair<G4double, G4double> GetEnergyMomentumCheckLevels() const;
71 void ModelDescription(std::ostream& outFile) const;
72
73
74 private:
75 const G4VIntraNuclearTransportModel * GetTransport() const;
76 const G4VHighEnergyGenerator * GetHighEnergyGenerator() const;
77 const G4HadFinalState * GetFinalState() const;
78
79 private:
80 G4VIntraNuclearTransportModel * theTransport;
81 G4VHighEnergyGenerator * theHighEnergyGenerator;
82 G4DecayStrongResonances theDecay;
83 G4HadFinalState * theParticleChange;
84 G4QuasiElasticChannel * theQuasielastic;
85};
86
87inline const G4VIntraNuclearTransportModel * G4TheoFSGenerator::GetTransport() const
88{
89 return theTransport;
90}
91
92inline void G4TheoFSGenerator::SetTransport(G4VIntraNuclearTransportModel *const value)
93{
94 theTransport = value;
95}
96
97inline const G4VHighEnergyGenerator * G4TheoFSGenerator::GetHighEnergyGenerator() const
98{
99 return theHighEnergyGenerator;
100}
101
102inline void G4TheoFSGenerator::SetHighEnergyGenerator(G4VHighEnergyGenerator *const value)
103{
104 theHighEnergyGenerator= value;
105}
106
107inline void G4TheoFSGenerator::SetQuasiElasticChannel(G4QuasiElasticChannel *const value)
108{
109 theQuasielastic = value;
110}
111
112inline const G4HadFinalState * G4TheoFSGenerator::GetFinalState() const
113{
114 return theParticleChange;
115}
116
117#endif
