45#ifndef G4INCLFINALSTATE_H_
46#define G4INCLFINALSTATE_H_
47
48#include "G4INCLParticle.hh"
49#include <string>
50
51namespace G4INCL {
52
53 enum FinalStateValidity {
54 ValidFS,
55 PauliBlockedFS,
56 NoEnergyConservationFS,
57 ParticleBelowFermiFS,
58 ParticleBelowZeroFS
59 };
60
61 /**
62 * Final state of an interaction
63 */
64 class FinalState {
65 public:
66 FinalState();
67 virtual ~FinalState();
68
69 void reset();
70
71 void setTotalEnergyBeforeInteraction(G4double E) { totalEnergyBeforeInteraction = E; };
72 G4double getTotalEnergyBeforeInteraction() const { return totalEnergyBeforeInteraction; };
73
74 void addModifiedParticle(Particle *p);
75 void addOutgoingParticle(Particle *p);
76 void addDestroyedParticle(Particle *p);
77 void addCreatedParticle(Particle *p);
78 void addEnteringParticle(Particle *p);
79
80 ParticleList const &getModifiedParticles() const;
81 ParticleList const &getOutgoingParticles() const;
82 ParticleList const &getDestroyedParticles() const;
83 ParticleList const &getCreatedParticles() const;
84 ParticleList const &getEnteringParticles() const;
85
86 FinalStateValidity getValidity() const { return validity; }
87 void makeValid() { validity = ValidFS; }
88 void makePauliBlocked() { validity = PauliBlockedFS; }
89 void makeNoEnergyConservation() { validity = NoEnergyConservationFS; }
90 void makeParticleBelowFermi() { validity = ParticleBelowFermiFS; }
91 void makeParticleBelowZero() { validity = ParticleBelowZeroFS; }
92
93 std::string print() const;
94
95 private:
96 ParticleList outgoing, created, destroyed, modified, entering;
97 G4double totalEnergyBeforeInteraction;
98 FinalStateValidity validity;
99 };
100
101}
102
103#endif /* G4INCLFINALSTATE_H_ */
