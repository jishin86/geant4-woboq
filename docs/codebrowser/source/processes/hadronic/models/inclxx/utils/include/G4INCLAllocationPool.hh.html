38/** \file G4INCLAllocationPool.hh
39 * \brief Singleton for recycling allocation of instances of a given class
40 *
41 * \date 2nd October 2014
42 * \author Davide Mancusi
43 */
44
45#ifndef G4INCLALLOCATIONPOOL_HH
46#define G4INCLALLOCATIONPOOL_HH
47
48#if defined(INCL_USE_ALLOCATION_POOL) || defined(INCLXX_IN_GEANT4_MODE)
49
50#include <stack>
51#include <new>
52#include <cstddef>
53
54namespace G4INCL {
55
56 template<typename T>
57 class AllocationPool {
58 public:
59 static AllocationPool &getInstance() {
60 if(!theInstance)
61 theInstance = new AllocationPool<T>;
62 return *theInstance;
63 }
64
65 T *getObject() {
66 if(theStack.empty())
67 return static_cast<T*>(::operator new(sizeof(T)));
68 else {
69 T *t = theStack.top();
70 theStack.pop();
71 return t;
72 }
73 }
74
75 void recycleObject(T *t) {
76 theStack.push(t);
77 }
78
79 void clear() {
80 while(!theStack.empty()) { /* Loop checking, 10.07.2015, D.Mancusi */
81 ::operator delete(theStack.top());
82 theStack.pop();
83 }
84 }
85
86 protected:
87 AllocationPool() {}
88 virtual ~AllocationPool() {
89 clear();
90 }
91
92 static G4ThreadLocal AllocationPool *theInstance;
93
94 std::stack<T*> theStack;
95
96 };
97
98 template<typename T>
99 G4ThreadLocal AllocationPool<T> *AllocationPool<T>::theInstance = 0;
100
101}
102
103#define INCL_DECLARE_ALLOCATION_POOL(T) \
104 public: \
105 static void *operator new(size_t /* s */) { \
106 ::G4INCL::AllocationPool<T> &allocator = ::G4INCL::AllocationPool<T>::getInstance(); \
107 return allocator.getObject(); \
108 } \
109 static void operator delete(void *a, size_t /* s */) { \
110 ::G4INCL::AllocationPool<T> &allocator = ::G4INCL::AllocationPool<T>::getInstance(); \
111 allocator.recycleObject(static_cast<T *>(a)); \
112 }
113
114#else // defined(INCL_USE_ALLOCATION_POOL) || defined(INCLXX_IN_GEANT4_MODE)
115#define INCL_DECLARE_ALLOCATION_POOL(T)
116#endif
117
118#endif // G4INCLALLOCATIONPOOL_HH
