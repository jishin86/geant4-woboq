59#ifndef INCL_ROOT_USE

67 struct EventInfo {
68 EventInfo() :
69 nParticles(0),
70 eventBias((Float_t)0.0),
71 nRemnants(0),
72 projectileType(0),
73 At(0),
74 Zt(0),
75 St(0),
76 Ap(0),
77 Zp(0),
78 Sp(0),
79 Ep((Float_t)0.0),
80 impactParameter((Float_t)0.0),
81 nCollisions(0),
82 stoppingTime((Float_t)0.0),
83 EBalance((Float_t)0.0),
84 pLongBalance((Float_t)0.0),
85 pTransBalance((Float_t)0.0),
86 nCascadeParticles(0),
87 transparent(false),
88 forcedCompoundNucleus(false),
89 nucleonAbsorption(false),
90 pionAbsorption(false),
91 nDecays(0),
92 nBlockedCollisions(0),
93 nBlockedDecays(0),
94 effectiveImpactParameter((Float_t)0.0),
95 deltasInside(false),
96 sigmasInside(false),
97 kaonsInside(false),
98 antikaonsInside(false),
99 lambdasInside(false),
100 forcedDeltasInside(false),
101 forcedDeltasOutside(false),
102 forcedPionResonancesOutside(false),
103 absorbedStrangeParticle(false),
104 forcedSigmaOutside(false),
105 forcedStrangeInside(false),
106 emitLambda(0),
107 emitKaon(false),
108 clusterDecay(false),
109 firstCollisionTime((Float_t)0.0),
110 firstCollisionXSec((Float_t)0.0),
111 firstCollisionSpectatorPosition((Float_t)0.0),
112 firstCollisionSpectatorMomentum((Float_t)0.0),
113 firstCollisionIsElastic(false),
114 nReflectionAvatars(0),
115 nCollisionAvatars(0),
116 nDecayAvatars(0),
117 nUnmergedSpectators(0),
118 nEnergyViolationInteraction(0),
119 event(0)
120
121 {
122 std::fill_n(A, maxSizeParticles, 0);
123 std::fill_n(Z, maxSizeParticles, 0);
124 std::fill_n(S, maxSizeParticles, 0);
125 std::fill_n(PDGCode, maxSizeParticles, 0);
126 std::fill_n(ParticleBias, maxSizeParticles, (Float_t)0.0);
127 std::fill_n(EKin, maxSizeParticles, (Float_t)0.0);
128 std::fill_n(px, maxSizeParticles, (Float_t)0.0);
129 std::fill_n(py, maxSizeParticles, (Float_t)0.0);
130 std::fill_n(pz, maxSizeParticles, (Float_t)0.0);
131 std::fill_n(theta, maxSizeParticles, (Float_t)0.0);
132 std::fill_n(phi, maxSizeParticles, (Float_t)0.0);
133 std::fill_n(origin, maxSizeParticles, 0);
134 std::fill_n(emissionTime, maxSizeParticles, (Float_t)0.0);
135 std::fill_n(ARem, maxSizeRemnants, 0);
136 std::fill_n(ZRem, maxSizeRemnants, 0);
137 std::fill_n(SRem, maxSizeRemnants, 0);
138 std::fill_n(EStarRem, maxSizeRemnants, (Float_t)0.0);
139 std::fill_n(JRem, maxSizeRemnants, (Float_t)0.0);
140 std::fill_n(EKinRem, maxSizeRemnants, (Float_t)0.0);
141 std::fill_n(pxRem, maxSizeRemnants, (Float_t)0.0);
142 std::fill_n(pyRem, maxSizeRemnants, (Float_t)0.0);
143 std::fill_n(pzRem, maxSizeRemnants, (Float_t)0.0);
144 std::fill_n(thetaRem, maxSizeRemnants, (Float_t)0.0);
145 std::fill_n(phiRem, maxSizeRemnants, (Float_t)0.0);
146 std::fill_n(jxRem, maxSizeRemnants, (Float_t)0.0);
147 std::fill_n(jyRem, maxSizeRemnants, (Float_t)0.0);
148 std::fill_n(jzRem, maxSizeRemnants, (Float_t)0.0);
149 std::fill_n(EKinPrime, maxSizeParticles, (Float_t)0.0);
150 std::fill_n(pzPrime, maxSizeParticles, (Float_t)0.0);
151 std::fill_n(thetaPrime, maxSizeParticles, (Float_t)0.0);
152 }
153
154 /** \brief Number of the event */
155 static G4ThreadLocal Int_t eventNumber;
156
157 /** \brief Maximum array size for remnants */
158 static const Short_t maxSizeRemnants = 10;
159
160 /** \brief Maximum array size for emitted particles */
161 static const Short_t maxSizeParticles = 1000;
162
163 /** \brief Number of particles in the final state */
164 Short_t nParticles;
165 /** \brief Particle mass number */
166 Short_t A[maxSizeParticles];
167 /** \brief Particle charge number */
168 Short_t Z[maxSizeParticles];
169 /** \brief Particle strangeness number */
170 Short_t S[maxSizeParticles];
171 /** \brief PDG numbering of the particles */
172 Int_t PDGCode[maxSizeParticles];
173 /** \brief Particle weight due to the bias */
174 Float_t ParticleBias[maxSizeParticles];
175 /** \brief Event bias */
176 Float_t eventBias;
177 /** \brief Particle kinetic energy [MeV] */
178 Float_t EKin[maxSizeParticles];
179 /** \brief Particle momentum, x component [MeV/c] */
180 Float_t px[maxSizeParticles];
181 /** \brief Particle momentum, y component [MeV/c] */
182 Float_t py[maxSizeParticles];
183 /** \brief Particle momentum, z component [MeV/c] */
184 Float_t pz[maxSizeParticles];
185 /** \brief Particle momentum polar angle [radians] */
186 Float_t theta[maxSizeParticles];
187 /** \brief Particle momentum azimuthal angle [radians] */
188 Float_t phi[maxSizeParticles];
189 /** \brief Origin of the particle
190 *
191 * Should be -1 for cascade particles, or the number of the remnant for
192 * de-excitation particles. */
193 Short_t origin[maxSizeParticles];
194 /** \brief Emission time [fm/c] */
195 Float_t emissionTime[maxSizeParticles];
196 /** \brief History of the particle
197 *
198 * Condensed information about the de-excitation chain of a particle. For
199 * cascade particles, it is just an empty string. For particles arising
200 * from the de-excitation of a cascade remnant, it is a string of
201 * characters. Each character represents one or more identical steps in
202 * the de-excitation process. The currently defined possible character
203 * values and their meanings are the following:
204 *
205 * e: evaporation product
206 * E: evaporation residue
207 * m: multifragmentation
208 * a: light partner in asymmetric fission or IMF emission
209 * A: heavy partner in asymmetric fission or IMF emission
210 * f: light partner in fission
211 * F: heavy partner in fission
212 * s: saddle-to-scission emission
213 * n: non-statistical emission (decay) */
214 std::vector<std::string> history;
215 /** \brief Number of remnants */
216 Short_t nRemnants;
217 /** \brief Remnant mass number */
218 Short_t ARem[maxSizeRemnants];
219 /** \brief Remnant charge number */
220 Short_t ZRem[maxSizeRemnants];
221 /** \brief Remnant strangeness number */
222 Short_t SRem[maxSizeRemnants];
223 /** \brief Remnant excitation energy [MeV] */
224 Float_t EStarRem[maxSizeRemnants];
225 /** \brief Remnant spin [\f$\hbar\f$] */
226 Float_t JRem[maxSizeRemnants];
227 /** \brief Remnant kinetic energy [MeV] */
228 Float_t EKinRem[maxSizeRemnants];
229 /** \brief Remnant momentum, x component [MeV/c] */
230 Float_t pxRem[maxSizeRemnants];
231 /** \brief Remnant momentum, y component [MeV/c] */
232 Float_t pyRem[maxSizeRemnants];
233 /** \brief Remnant momentum, z component [MeV/c] */
234 Float_t pzRem[maxSizeRemnants];
235 /** \brief Remnant momentum polar angle [radians] */
236 Float_t thetaRem[maxSizeRemnants];
237 /** \brief Remnant momentum azimuthal angle [radians] */
238 Float_t phiRem[maxSizeRemnants];
239 /** \brief Remnant angular momentum, x component [\f$\hbar\f$] */
240 Float_t jxRem[maxSizeRemnants];
241 /** \brief Remnant angular momentum, y component [\f$\hbar\f$] */
242 Float_t jyRem[maxSizeRemnants];
243 /** \brief Remnant angular momentum, z component [\f$\hbar\f$] */
244 Float_t jzRem[maxSizeRemnants];
245 /** \brief Projectile particle type */
246 Int_t projectileType;
247 /** \brief Mass number of the target nucleus */
248 Short_t At;
249 /** \brief Charge number of the target nucleus */
250 Short_t Zt;
251 /** \brief Strangeness number of the target nucleus */
252 Short_t St;
253 /** \brief Mass number of the projectile nucleus */
254 Short_t Ap;
255 /** \brief Charge number of the projectile nucleus */
256 Short_t Zp;
257 /** \brief Strangeness number of the projectile nucleus */
258 Short_t Sp;
259 /** \brief Projectile kinetic energy given as input */
260 Float_t Ep;
261 /** \brief Impact parameter [fm] */
262 Float_t impactParameter;
263 /** \brief Number of accepted two-body collisions */
264 Int_t nCollisions;
265 /** \brief Cascade stopping time [fm/c] */
266 Float_t stoppingTime;
267 /** \brief Energy-conservation balance [MeV] */
268 Float_t EBalance;
269 /** \brief Longitudinal momentum-conservation balance [MeV/c] */
270 Float_t pLongBalance;
271 /** \brief Transverse momentum-conservation balance [MeV/c] */
272 Float_t pTransBalance;
273 /** \brief Number of cascade particles */
274 Short_t nCascadeParticles;
275 /** \brief True if the event is transparent */
276 Bool_t transparent;
277 /** \brief True if the event is a forced CN */
278 Bool_t forcedCompoundNucleus;
279 /** \brief True if the event is a nucleon absorption */
280 Bool_t nucleonAbsorption;
281 /** \brief True if the event is a pion absorption */
282 Bool_t pionAbsorption;
283 /** \brief Number of accepted Delta decays */
284 Int_t nDecays;
285 /** \brief Number of two-body collisions blocked by Pauli or CDPP */
286 Int_t nBlockedCollisions;
287 /** \brief Number of decays blocked by Pauli or CDPP */
288 Int_t nBlockedDecays;
289 /** \brief Effective (Coulomb-distorted) impact parameter [fm] */
290 Float_t effectiveImpactParameter;
291 /** \brief Event involved deltas in the nucleus at the end of the cascade */
292 Bool_t deltasInside;
293 /** \brief Event involved sigmas in the nucleus at the end of the cascade */
294 Bool_t sigmasInside;
295 /** \brief Event involved kaons in the nucleus at the end of the cascade */
296 Bool_t kaonsInside;
297 /** \brief Event involved antikaons in the nucleus at the end of the cascade */
298 Bool_t antikaonsInside;
299 /** \brief Event involved lambdas in the nucleus at the end of the cascade */
300 Bool_t lambdasInside;
301 /** \brief Event involved forced delta decays inside the nucleus */
302 Bool_t forcedDeltasInside;
303 /** \brief Event involved forced delta decays outside the nucleus */
304 Bool_t forcedDeltasOutside;
305 /** \brief Event involved forced eta/omega decays outside the nucleus */
306 Bool_t forcedPionResonancesOutside;
307 /** \brief Event involved forced strange absorbtion inside the nucleus */
308 Bool_t absorbedStrangeParticle;
309 /** \brief Event involved forced Sigma Zero decays outside the nucleus */
310 Bool_t forcedSigmaOutside;
311 /** \brief Event involved forced antiKaon/Sigma absorbtion inside the nucleus */
312 Bool_t forcedStrangeInside;
313 /** \brief Number of forced Lambda emit out of the nucleus */
314 Int_t emitLambda;
315 /** \brief Event involved forced Kaon emission */
316 Bool_t emitKaon;
317 /** \brief Event involved cluster decay */
318 Bool_t clusterDecay;
319 /** \brief Time of the first collision [fm/c] */
320 Float_t firstCollisionTime;
321 /** \brief Cross section of the first collision (mb) */
322 Float_t firstCollisionXSec;
323 /** \brief Position of the spectator on the first collision (fm) */
324 Float_t firstCollisionSpectatorPosition;
325 /** \brief Momentum of the spectator on the first collision (fm) */
326 Float_t firstCollisionSpectatorMomentum;
327 /** \brief True if the first collision was elastic */
328 Bool_t firstCollisionIsElastic;
329 /** \brief Number of reflection avatars */
330 Int_t nReflectionAvatars;
331 /** \brief Number of collision avatars */
332 Int_t nCollisionAvatars;
333 /** \brief Number of decay avatars */
334 Int_t nDecayAvatars;
335 /** \brief Number of dynamical spectators that were merged back into the projectile remnant */
336 Int_t nUnmergedSpectators;
337 /** \brief Number of attempted collisions/decays for which the energy-conservation algorithm failed to find a solution. */
338 Int_t nEnergyViolationInteraction;
339 /** \brief Sequential number of the event in the event loop */
340 Int_t event;
341 /** \brief Particle kinetic energy, in inverse kinematics [MeV] */
342 Float_t EKinPrime[maxSizeParticles];
343 /** \brief Particle momentum, z component, in inverse kinematics [MeV/c] */
344 Float_t pzPrime[maxSizeParticles];
345 /** \brief Particle momentum polar angle, in inverse kinematics [radians] */
346 Float_t thetaPrime[maxSizeParticles];
347
348 /** \brief Reset the EventInfo members */
349 void reset() {
350 nParticles = 0;
351 eventBias = (Float_t)0.0;
352 history.clear();
353 nRemnants = 0;
354 projectileType = 0;
355 At = 0;
356 Zt = 0;
357 St = 0;
358 Ap = 0;
359 Zp = 0;
360 Sp = 0;
361 Ep = (Float_t)0.0;
362 impactParameter = (Float_t)0.0;
363 nCollisions = 0;
364 stoppingTime = (Float_t)0.0;
365 EBalance = (Float_t)0.0;
366 pLongBalance = (Float_t)0.0;
367 pTransBalance = (Float_t)0.0;
368 nCascadeParticles = 0;
369 transparent = false;
370 forcedCompoundNucleus = false;
371 nucleonAbsorption = false;
372 pionAbsorption = false;
373 nDecays = 0;
374 nBlockedCollisions = 0;
375 nBlockedDecays = 0;
376 effectiveImpactParameter = (Float_t)0.0;
377 deltasInside = false;
378 sigmasInside = false;
379 kaonsInside = false;
380 antikaonsInside = false;
381 lambdasInside = false;
382 forcedDeltasInside = false;
383 forcedDeltasOutside = false;
384 forcedPionResonancesOutside = false;
385 absorbedStrangeParticle = false;
386 forcedSigmaOutside = false;
387 forcedStrangeInside = false;
388 emitLambda = 0;
389 emitKaon = false;
390 clusterDecay = false;
391 firstCollisionTime = (Float_t)0.0;
392 firstCollisionXSec = (Float_t)0.0;
393 firstCollisionSpectatorPosition = (Float_t)0.0;
394 firstCollisionSpectatorMomentum = (Float_t)0.0;
395 firstCollisionIsElastic = false;
396 nReflectionAvatars = 0;
397 nCollisionAvatars = 0;
398 nDecayAvatars = 0;
399 nUnmergedSpectators = 0;
400 nEnergyViolationInteraction = 0;
401 event = 0;
402
403 }
404
405 /// \brief Move a remnant to the particle array
406 void remnantToParticle(const G4int remnantIndex);
407
408 /// \brief Fill the variables describing the reaction in inverse kinematics
409 void fillInverseKinematics(const Double_t gamma);
410 };
411}
