45#ifndef G4INCLCROSSSECTIONSTRUNCATEDMULTIPIONS_HH
46#define G4INCLCROSSSECTIONSTRUNCATEDMULTIPIONS_HH
47
48#include "G4INCLCrossSectionsMultiPions.hh"
49#include <limits>
50
51namespace G4INCL {
52 /// \brief Truncated multipion cross sections
53
54 class CrossSectionsTruncatedMultiPions : public CrossSectionsMultiPions {
55 public:
56 CrossSectionsTruncatedMultiPions(const G4int nPi=std::numeric_limits<G4int>::max());
57
58 /** \brief Elastic particle-particle cross section
59 *
60 * The elastic pi-N cross section is calculated by difference:
61 * elastic = total - inelastic(pion production) - delta,
62 * but we need to use the unmodified piNToDelta cross section, or else we
63 * will violate the cross-section sum.
64 *
65 * This modification is only necessary for piN collisions.
66 */
67 virtual G4double elastic(Particle const * const p1, Particle const * const p2);
68
69 /// \brief Cross section for Delta production - piN Channel
70 virtual G4double piNToDelta(Particle const * const p1, Particle const * const p2);
71
72 /// \brief Cross section for X pion production - piN Channel
73 virtual G4double piNToxPiN(const G4int xpi, Particle const * const p1, Particle const * const p2);
74
75 /// \brief Cross section for X pion production - NN Channel
76 virtual G4double NNToxPiNN(const G4int xpi, Particle const * const p1, Particle const * const p2);
77
78 protected:
79
80 /// \brief Maximum number of pions produced in TruncatedMultiPion collisions
81 const G4int nMaxPi;
82
83 };
84}
85
86#endif
