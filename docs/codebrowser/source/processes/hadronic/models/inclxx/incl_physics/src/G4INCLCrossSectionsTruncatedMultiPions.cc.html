54 G4double CrossSectionsTruncatedMultiPions::elastic(Particle const * const p1, Particle const * const p2) {
55// if(!p1->isPion() && !p2->isPion()) {
56 if((p1->isNucleon()||p1->isDelta()) && (p2->isNucleon()||p2->isDelta())){
57 return NNElastic(p1, p2);
58 }
59// else if (p1->isNucleon() || p2->isNucleon()) {
60 else if ((p1->isNucleon() && p2->isPion()) || (p2->isNucleon() && p1->isPion())){
61 G4double pielas = piNTot(p1,p2) - piNIne(p1,p2) - CrossSectionsMultiPions::piNToDelta(p1,p2);
62 if (pielas < 0.)
63 pielas = 0.;
64 return pielas;
65 } else
66 return 0.0;
67 }
68
69 G4double CrossSectionsTruncatedMultiPions::piNToDelta(Particle const * const particle1, Particle const * const particle2) {
70 G4double sum = CrossSectionsMultiPions::piNToDelta(particle1, particle2);
71 if(nMaxPi<=1) {
72 for(G4int i=nMaxPi+1; i<=nMaxPiPiN; ++i)
73 sum += CrossSectionsMultiPions::piNToxPiN(i, particle1, particle2);
74 }
75 return sum;
76 }
77
78 G4double CrossSectionsTruncatedMultiPions::NNToxPiNN(const G4int xpi, Particle const * const particle1, Particle const * const particle2) {
79 if(xpi<nMaxPi)
80 return CrossSectionsMultiPions::NNToxPiNN(xpi, particle1, particle2);
81 else if(xpi==nMaxPi) {
82 G4double sum = 0.;
83 for(G4int i=xpi; i<=nMaxPiNN; ++i)
84 sum += CrossSectionsMultiPions::NNToxPiNN(i, particle1, particle2);
85 return sum;
86 } else
87 return 0.;
88 }
89
90 G4double CrossSectionsTruncatedMultiPions::piNToxPiN(const G4int xpi, Particle const * const particle1, Particle const * const particle2) {
91 if(xpi<nMaxPi)
92 return CrossSectionsMultiPions::piNToxPiN(xpi, particle1, particle2);
93 else if(xpi==nMaxPi) {
94 G4double sum = 0.;
95 for(G4int i=xpi; i<=nMaxPiPiN; ++i)
96 sum += CrossSectionsMultiPions::piNToxPiN(i, particle1, particle2);
97 return sum;
98 } else
99 return 0.;
100 }
