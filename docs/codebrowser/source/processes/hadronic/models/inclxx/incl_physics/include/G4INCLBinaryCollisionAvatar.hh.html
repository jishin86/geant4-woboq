54namespace G4INCL {
55
56 class BinaryCollisionAvatar: public InteractionAvatar {
57 public:
58 BinaryCollisionAvatar(G4double, G4double, G4INCL::Nucleus*, G4INCL::Particle*, G4INCL::Particle*);
59 virtual ~BinaryCollisionAvatar();
60 G4INCL::IChannel* getChannel();
61 ParticleList getParticles() const {
62 ParticleList theParticleList;
63 theParticleList.push_back(particle1);
64 theParticleList.push_back(particle2);
65 return theParticleList;
66 };
67
68 virtual void preInteraction();
69 virtual void postInteraction(FinalState *);
70
71 std::string dump() const;
72
73 static void setCutNN(const G4double c) {
74 cutNN = c;
75 cutNNSquared = cutNN*cutNN;
76 }
77
78 static G4double getCutNN() { return cutNN; }
79
80 static G4double getCutNNSquared() { return cutNNSquared; }
81
82 /// \brief Get the bias
83 static G4double getBias() { return bias; }
84
85 /// \brief Set the bias
86 static void setBias(const G4double b) { bias=b; }
87
88 private:
89 static G4ThreadLocal G4double cutNN;
90 static G4ThreadLocal G4double cutNNSquared;
91 static G4ThreadLocal G4double bias;
92 G4double theCrossSection;
93 G4bool isParticle1Spectator;
94 G4bool isParticle2Spectator;
95 G4bool isElastic;
96
97 G4bool isStrangeProduction;
98
99 INCL_DECLARE_ALLOCATION_POOL(BinaryCollisionAvatar)
100 };
