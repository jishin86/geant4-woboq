58G4PreCompoundEmission::G4PreCompoundEmission()
59{
60 theFragmentsFactory = new G4PreCompoundEmissionFactory();
61 theFragmentsVector =
62 new G4PreCompoundFragmentVector(theFragmentsFactory->GetFragmentVector());
63 g4calc = G4Pow::GetInstance();
64 fNuclData = G4NuclearLevelData::GetInstance();
65 G4DeexPrecoParameters* param = fNuclData->GetParameters();
66 fFermiEnergy = param->GetFermiEnergy();
67 fUseAngularGenerator = param->UseAngularGen();
68}
69
70G4PreCompoundEmission::~G4PreCompoundEmission()
71{
72 delete theFragmentsFactory;
73 delete theFragmentsVector;
74}

76void G4PreCompoundEmission::SetDefaultModel()

100G4ReactionProduct*
101G4PreCompoundEmission::PerformEmission(G4Fragment & aFragment)
102{
103 // Choose a Fragment for emission
104 G4VPreCompoundFragment * thePreFragment =
105 theFragmentsVector->ChooseFragment();
106 if (thePreFragment == nullptr)
107 {
108 G4cout << "G4PreCompoundEmission::PerformEmission : "
109 << "I couldn't choose a fragment\n"
110 << "while trying to de-excite\n"
111 << aFragment << G4endl;
112 throw G4HadronicException(__FILE__, __LINE__, "");
113 }
114
115 //G4cout << "Chosen fragment: " << G4endl;
116 //G4cout << *thePreFragment << G4endl;
117
118 // Kinetic Energy of emitted fragment
119 G4double kinEnergy = thePreFragment->SampleKineticEnergy(aFragment);
120 kinEnergy = std::max(kinEnergy, 0.0);
121
122 // Calculate the fragment momentum (three vector)
123 if(fUseAngularGenerator) {
124 AngularDistribution(thePreFragment,aFragment,kinEnergy);
125 } else {
126 G4double pmag =
127 std::sqrt(kinEnergy*(kinEnergy + 2.0*thePreFragment->GetNuclearMass()));
128 theFinalMomentum = pmag*G4RandomDirection();
129 }
130
131 // Mass of emittef fragment
132 G4double EmittedMass = thePreFragment->GetNuclearMass();
133 // Now we can calculate the four momentum
134 // both options are valid and give the same result but 2nd one is faster
135 G4LorentzVector Emitted4Momentum(theFinalMomentum,EmittedMass + kinEnergy);
136
137 // Perform Lorentz boost
138 G4LorentzVector Rest4Momentum = aFragment.GetMomentum();
139 Emitted4Momentum.boost(Rest4Momentum.boostVector());
140
141 // Set emitted fragment momentum
142 thePreFragment->SetMomentum(Emitted4Momentum);
143
144 // NOW THE RESIDUAL NUCLEUS
145 // ------------------------
146
147 Rest4Momentum -= Emitted4Momentum;
148
149 // Update nucleus parameters:
150 // --------------------------
151
152 // Z and A
153 aFragment.SetZandA_asInt(thePreFragment->GetRestZ(),
154 thePreFragment->GetRestA());
155
156 // Number of excitons
157 aFragment.SetNumberOfParticles(aFragment.GetNumberOfParticles()-
158 thePreFragment->GetA());
159 // Number of charges
160 aFragment.SetNumberOfCharged(aFragment.GetNumberOfCharged()-
161 thePreFragment->GetZ());
162
163 // Update nucleus momentum
164 // A check on consistence of Z, A, and mass will be performed
165 aFragment.SetMomentum(Rest4Momentum);
166
167 // Create a G4ReactionProduct
168 G4ReactionProduct * MyRP = thePreFragment->GetReactionProduct();
169
170 // if(kinEnergy < MeV) {
171 //G4cout << "G4PreCompoundEmission::Fragment emitted" << G4endl;
172 //G4cout << *thePreFragment << G4endl;
173 // }
174 return MyRP;
175}

177void G4PreCompoundEmission::AngularDistribution(
178 G4VPreCompoundFragment* thePreFragment,
179 const G4Fragment& aFragment,
180 G4double ekin)
181{
182 G4int p = aFragment.GetNumberOfParticles();
183 G4int h = aFragment.GetNumberOfHoles();
184 G4double U = aFragment.GetExcitationEnergy();
185
186 // Emission particle separation energy
187 G4double Bemission = thePreFragment->GetBindingEnergy();
188
189 G4double gg = (6.0/pi2)*fNuclData->GetLevelDensity(aFragment.GetZ_asInt(),
190 aFragment.GetA_asInt(),U);
191
192 // Average exciton energy relative to bottom of nuclear well
193 G4double Eav = 2*p*(p+1)/((p+h)*gg);
194
195 // Excitation energy relative to the Fermi Level
196 G4double Uf = std::max(U - (p - h)*fFermiEnergy , 0.0);
197 // G4double Uf = U - KineticEnergyOfEmittedFragment - Bemission;
198
199 G4double w_num = rho(p+1, h, gg, Uf, fFermiEnergy);
200 G4double w_den = rho(p, h, gg, Uf, fFermiEnergy);
201 if (w_num > 0.0 && w_den > 0.0)
202 {
203 Eav *= (w_num/w_den);
204 Eav += - Uf/(p+h) + fFermiEnergy;
205 }
206 else
207 {
208 Eav = fFermiEnergy;
209 }
210
211 // VI + JMQ 19/01/2010 update computation of the parameter an
212 //
213 G4double an = 0.0;
214 G4double Eeff = ekin + Bemission + fFermiEnergy;
215 if(ekin > DBL_MIN && Eeff > DBL_MIN) {
216
217 G4double zeta = std::max(1.0,9.3/std::sqrt(ekin/MeV));
218
219 // This should be the projectile energy. If I would know which is
220 // the projectile (proton, neutron) I could remove the binding energy.
221 // But, what happens if INC precedes precompound? This approximation
222 // seems to work well enough
223 G4double ProjEnergy = aFragment.GetExcitationEnergy();
224
225 an = 3*std::sqrt((ProjEnergy+fFermiEnergy)*Eeff)/(zeta*Eav);
226
227 G4int ne = aFragment.GetNumberOfExcitons() - 1;
228 if ( ne > 1 ) { an /= (G4double)ne; }
229
230 // protection of exponent
231 if ( an > 10. ) { an = 10.; }
232 }
233
234 // sample cosine of theta and not theta as in old versions
235 G4double random = G4UniformRand();
236 G4double cost;
237
238 if(an < 0.1) { cost = 1. - 2*random; }
239 else {
240 G4double exp2an = G4Exp(-2*an);
241 cost = 1. + G4Log(1-random*(1-exp2an))/an;
242 if(cost > 1.) { cost = 1.; }
243 else if(cost < -1.) {cost = -1.; }
244 }
245
246 G4double phi = CLHEP::twopi*G4UniformRand();
247
248 // Calculate the momentum magnitude of emitted fragment
249 G4double pmag =
250 std::sqrt(ekin*(ekin + 2.0*thePreFragment->GetNuclearMass()));
251
252 G4double sint = std::sqrt((1.0-cost)*(1.0+cost));
253
254 theFinalMomentum.set(pmag*std::cos(phi)*sint,pmag*std::sin(phi)*sint,
255 pmag*cost);
256
257 // theta is the angle wrt the incident direction
258 G4ThreeVector theIncidentDirection = aFragment.GetMomentum().vect().unit();
259 theFinalMomentum.rotateUz(theIncidentDirection);
260}

262G4double G4PreCompoundEmission::rho(G4int p, G4int h, G4double gg,
263 G4double E, G4double Ef) const
264{
265 // 25.02.2010 V.Ivanchenko added more protections
266 G4double Aph = (p*p + h*h + p - 3.0*h)/(4.0*gg);
267 // G4double alpha = (p*p + h*h)/(2.0*gg);
268
269 if ( E - Aph < 0.0) { return 0.0; }
270
271 G4double logConst = (p+h)*G4Log(gg)
272 - g4calc->logfactorial(p+h-1) - g4calc->logfactorial(p)
273 - g4calc->logfactorial(h);
274
275 // initialise values using j=0
276
277 G4double t1=1;
278 G4double t2=1;
279 G4double logt3 = (p+h-1) * G4Log(E-Aph) + logConst;
280 const G4double logmax = 200.;
281 if(logt3 > logmax) { logt3 = logmax; }
282 G4double tot = G4Exp( logt3 );
283
284 // and now sum rest of terms
285 // 25.02.2010 V.Ivanchenko change while to for loop and cleanup
286 G4double Eeff = E - Aph;
287 for(G4int j=1; j<=h; ++j)
288 {
289 Eeff -= Ef;
290 if(Eeff < 0.0) { break; }
291 t1 *= -1.;
292 t2 *= (G4double)(h+1-j)/(G4double)j;
293 logt3 = (p+h-1) * G4Log( Eeff) + logConst;
294 if(logt3 > logmax) { logt3 = logmax; }
295 tot += t1*t2*G4Exp(logt3);
296 }
297
298 return tot;
299}
