43std::vector<G4VPreCompoundFragment*> * G4PreCompoundEmissionFactory::
44CreateFragmentVector()
45{
46 std::vector<G4VPreCompoundFragment*> * theFragVector =
47 new std::vector<G4VPreCompoundFragment*>();
48 theFragVector->reserve(6);
49
50 // neutron
51 theFragVector->push_back(new G4PreCompoundNeutron());
52 // proton
53 theFragVector->push_back(new G4PreCompoundProton());
54 // deuterium
55 theFragVector->push_back(new G4PreCompoundDeuteron());
56 // alpha
57 theFragVector->push_back(new G4PreCompoundAlpha());
58 // triton
59 theFragVector->push_back(new G4PreCompoundTriton());
60 // helium3
61 theFragVector->push_back(new G4PreCompoundHe3());
62
63 return theFragVector;
64}
