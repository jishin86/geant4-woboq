35// Compute transition probailities:
36// TransitionProb1 => probability of transition with \Delta N = +2
37// number of excitons will be increased on 2
38// TransitionProb2 => probability of transition with \Delta N = -2
39// number of excitons will be decreased on 2
40// TransitionProb3 => probability of transition with \Delta N = 0
41// number of excitons will be the same
42

44#include "globals.hh"

61
62private:
63
64 G4PreCompoundTransitions(const G4PreCompoundTransitions &);

66 G4bool operator==(const G4PreCompoundTransitions &right) const;

68
69 const G4ParticleDefinition* proton;
