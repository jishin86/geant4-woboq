70G4PreCompoundModel::G4PreCompoundModel(G4ExcitationHandler* ptr)
71 : G4VPreCompoundModel(ptr,"PRECO"),theEmission(nullptr),theTransition(nullptr),
72 useSCO(false),isInitialised(false),isActive(true),minZ(3),minA(5)
73{
74 //G4cout << "### NEW PrecompoundModel " << this << G4endl;
75 if(!ptr) { SetExcitationHandler(new G4ExcitationHandler()); }
76
77 fNuclData = G4NuclearLevelData::GetInstance();
78 proton = G4Proton::Proton();
79 neutron = G4Neutron::Neutron();
80 fLimitEnergy = 0.0;
81}

101void G4PreCompoundModel::InitialiseModel()
102{
103 if(isInitialised) { return; }
104 isInitialised = true;
105
106 //G4cout << "G4PreCompoundModel::InitialiseModel() started" << G4endl;
107
108 G4DeexPrecoParameters* param = fNuclData->GetParameters();
109
110 fLimitEnergy = param->GetPrecoLowEnergy();
111
112 useSCO = param->UseSoftCutoff();
113
114 minZ = param->GetMinZForPreco();
115 minA = param->GetMinAForPreco();
116
117 theEmission = new G4PreCompoundEmission();
118 if(param->UseHETC()) { theEmission->SetHETCModel(); }
119 //else { theEmission->SetDefaultModel(); }
120 theEmission->SetOPTxs(param->GetPrecoModelType());
121
122 if(param->UseGNASH()) { theTransition = new G4GNASHTransitions; }
123 else { theTransition = new G4PreCompoundTransitions(); }
124 theTransition->UseNGB(param->NeverGoBack());
125 theTransition->UseCEMtr(param->UseCEM());
126
127 if(param->PrecoDummy()) { isActive = false; }
128
129 GetExcitationHandler()->Initialise();
130}

134G4HadFinalState*
135G4PreCompoundModel::ApplyYourself(const G4HadProjectile & thePrimary,
136 G4Nucleus & theNucleus)
137{
138 const G4ParticleDefinition* primary = thePrimary.GetDefinition();
139 if(primary != neutron && primary != proton) {
140 G4ExceptionDescription ed;
141 ed << "G4PreCompoundModel is used for ";
142 if(primary) { ed << primary->GetParticleName(); }
143 G4Exception("G4PreCompoundModel::ApplyYourself()","had0033",FatalException,
144 ed,"");
145 return 0;
146 }
147
148 G4int Zp = 0;
149 G4int Ap = 1;
150 if(primary == proton) { Zp = 1; }
151
152 G4double timePrimary=thePrimary.GetGlobalTime();
153
154 G4int A = theNucleus.GetA_asInt();
155 G4int Z = theNucleus.GetZ_asInt();
156
157 //G4cout << "### G4PreCompoundModel::ApplyYourself: A= " << A << " Z= " << Z
158 // << " Ap= " << Ap << " Zp= " << Zp << G4endl;
159 // 4-Momentum
160 G4LorentzVector p = thePrimary.Get4Momentum();
161 G4double mass = G4NucleiProperties::GetNuclearMass(A, Z);
162 p += G4LorentzVector(0.0,0.0,0.0,mass);
163 //G4cout << "Primary 4-mom " << p << " mass= " << mass << G4endl;
164
165 // prepare fragment
166 G4Fragment anInitialState(A + Ap, Z + Zp, p);
167 anInitialState.SetNumberOfExcitedParticle(2, 1);
168 anInitialState.SetNumberOfHoles(1,0);
169 anInitialState.SetCreationTime(thePrimary.GetGlobalTime());
170
171 // call excitation handler
172 G4ReactionProductVector * result = DeExcite(anInitialState);
173
174 // fill particle change
175 theResult.Clear();
176 theResult.SetStatusChange(stopAndKill);
177 for(G4ReactionProductVector::iterator i= result->begin();
178 i != result->end(); ++i)
179 {
180 G4DynamicParticle * aNewDP =
181 new G4DynamicParticle((*i)->GetDefinition(),
182 (*i)->GetTotalEnergy(),
183 (*i)->GetMomentum());
184 G4HadSecondary aNew = G4HadSecondary(aNewDP);
185 G4double time=(*i)->GetFormationTime();
186 if(time < 0.0) { time = 0.0; }
187 aNew.SetTime(timePrimary + time);
188 aNew.SetCreatorModelType((*i)->GetCreatorModel());
189 delete (*i);
190 theResult.AddSecondary(aNew);
191 }
192 delete result;
193
194 //return the filled particle change
195 return &theResult;
196}

200G4ReactionProductVector* G4PreCompoundModel::DeExcite(G4Fragment& aFragment)
201{
202 if(!isInitialised) { InitialiseModel(); }
203
204 G4ReactionProductVector * Result = new G4ReactionProductVector;
205 G4double Eex = aFragment.GetExcitationEnergy();
206 G4int Z = aFragment.GetZ_asInt();
207 G4int A = aFragment.GetA_asInt();
208
209 //G4cout << "### G4PreCompoundModel::DeExcite" << G4endl;
210 //G4cout << aFragment << G4endl;
211
212 // Perform Equilibrium Emission
213 if (!isActive || (Z < minZ && A < minA) || Eex < fLimitEnergy*A) {
214 PerformEquilibriumEmission(aFragment, Result);
215 return Result;
216 }
217
218 // main loop
219 G4int count = 0;
220 static const G4double ldfact = 12.0/CLHEP::pi2;
221 static const G4int countmax = 1000;
222 for (;;) {
223 //G4cout << "### PreCompound loop over fragment" << G4endl;
224 //G4cout << aFragment << G4endl;
225 G4double U = aFragment.GetExcitationEnergy();
226 Z = aFragment.GetZ_asInt();
227 A = aFragment.GetA_asInt();
228 G4int EquilibriumExcitonNumber =
229 G4lrint(std::sqrt(ldfact*U*fNuclData->GetLevelDensity(Z, A, U)));
230 //
231 // G4cout<<"Neq="<<EquilibriumExcitonNumber<<G4endl;
232 //
233 // J. M. Quesada (Jan. 08) equilibrium hole number could be used as preeq.
234 // evap. delimiter (IAEA report)
235
236 // Loop for transitions, it is performed while there are
237 // preequilibrium transitions.
238 G4bool ThereIsTransition = false;
239
240 // G4cout<<"----------------------------------------"<<G4endl;
241 // G4double NP=aFragment.GetNumberOfParticles();
242 // G4double NH=aFragment.GetNumberOfHoles();
243 // G4double NE=aFragment.GetNumberOfExcitons();
244 // G4cout<<" Ex. Energy="<<aFragment.GetExcitationEnergy()<<G4endl;
245 // G4cout<<"N. excitons="<<NE<<" N. Part="<<NP<<"N. Holes ="<<NH<<G4endl;
246 do {
247 ++count;
248 //G4cout<<"transition number .."<<count
249 // <<" n ="<<aFragment.GetNumberOfExcitons()<<G4endl;
250 G4bool go_ahead = false;
251 // soft cutoff criterium as an "ad-hoc" solution to force
252 // increase in evaporation
253 G4int test = aFragment.GetNumberOfExcitons();
254 if (test <= EquilibriumExcitonNumber) { go_ahead=true; }
255
256 //J. M. Quesada (Apr. 08): soft-cutoff switched off by default
257 if (useSCO && go_ahead)
258 {
259 G4double x = G4double(test)/G4double(EquilibriumExcitonNumber) - 1;
260 if( G4UniformRand() < 1.0 - G4Exp(-x*x/0.32) ) { go_ahead = false; }
261 }
262
263 // JMQ: WARNING: CalculateProbability MUST be called prior to Get!!
264 // (O values would be returned otherwise)
265 G4double TotalTransitionProbability =
266 theTransition->CalculateProbability(aFragment);
267 G4double P1 = theTransition->GetTransitionProb1();
268 G4double P2 = theTransition->GetTransitionProb2();
269 G4double P3 = theTransition->GetTransitionProb3();
270 //G4cout<<"#0 P1="<<P1<<" P2="<<P2<<" P3="<<P3<<G4endl;
271
272 //J.M. Quesada (May 2008) Physical criterium (lamdas) PREVAILS over
273 // approximation (critical exciton number)
274 //V.Ivanchenko (May 2011) added check on number of nucleons
275 // to send a fragment to FermiBreakUp
276 if(!go_ahead || P1 <= P2+P3 || Z < minZ || A < minA || U <= fLimitEnergy*A)
277 {
278 //G4cout<<"#4 EquilibriumEmission"<<G4endl;
279 PerformEquilibriumEmission(aFragment,Result);
280 return Result;
281 }
282 else
283 {
284 //
285 // Check if number of excitons is greater than 0
286 // else perform equilibrium emission
287 if (aFragment.GetNumberOfExcitons() <= 0)
288 {
289 PerformEquilibriumEmission(aFragment,Result);
290 return Result;
291 }
292
293 G4double TotalEmissionProbability =
294 theEmission->GetTotalProbability(aFragment);
295 //
296 //G4cout<<"#1 TotalEmissionProbability="<<TotalEmissionProbability
297 // <<" Nex= " <<aFragment.GetNumberOfExcitons()<<G4endl;
298 //J.M.Quesada (May 08) this has already been done in order to decide
299 // what to do (preeq-eq)
300 // Sum of all probabilities
301 G4double TotalProbability = TotalEmissionProbability
302 + TotalTransitionProbability;
303
304 // Select subprocess
305 if (TotalProbability*G4UniformRand() > TotalEmissionProbability)
306 {
307 //G4cout<<"#2 Transition"<<G4endl;
308 // It will be transition to state with a new number of excitons
309 ThereIsTransition = true;
310 // Perform the transition
311 theTransition->PerformTransition(aFragment);
312 }
313 else
314 {
315 //G4cout<<"#3 Emission"<<G4endl;
316 // It will be fragment emission
317 ThereIsTransition = false;
318 Result->push_back(theEmission->PerformEmission(aFragment));
319 }
320 }
321 // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
322 } while (ThereIsTransition); // end of do loop
323
324 // stop if too many iterations
325 if(count >= countmax) {
326 G4ExceptionDescription ed;
327 ed << "G4PreCompoundModel loop over " << countmax << " iterations; "
328 << "current G4Fragment: \n" << aFragment;
329 G4Exception("G4PreCompoundModel::DeExcite()","had0034",JustWarning,
330 ed,"");
331 PerformEquilibriumEmission(aFragment, Result);
332 return Result;
333 }
334 } // end of for (;;) loop
335 return Result;
336}

401void G4PreCompoundModel::ModelDescription(std::ostream& outFile) const
402{
403 outFile
404 << "The GEANT4 precompound model is considered as an extension of the\n"
405 << "hadron kinetic model. It gives a possibility to extend the low energy range\n"
406 << "of the hadron kinetic model for nucleon-nucleus inelastic collision and it \n"
407 << "provides a smooth transition from kinetic stage of reaction described by the\n"
408 << "hadron kinetic model to the equilibrium stage of reaction described by the\n"
409 << "equilibrium deexcitation models.\n"
410 << "The initial information for calculation of pre-compound nuclear stage\n"
411 << "consists of the atomic mass number A, charge Z of residual nucleus, its\n"
412 << "four momentum P0 , excitation energy U and number of excitons n, which equals\n"
413 << "the sum of the number of particles p (from them p_Z are charged) and the number of\n"
414 << "holes h.\n"
415 << "At the preequilibrium stage of reaction, we follow the exciton model approach in ref. [1],\n"
416 << "taking into account the competition among all possible nuclear transitions\n"
417 << "with n = +2, 2, 0 (which are defined by their associated transition probabilities) and\n"
418 << "the emission of neutrons, protons, deuterons, thritium and helium nuclei (also defined by\n"
419 << "their associated emission probabilities according to exciton model)\n"
420 << "\n"
421 << "[1] K.K. Gudima, S.G. Mashnik, V.D. Toneev, Nucl. Phys. A401 329 (1983)\n"
422 << "\n";
423}
