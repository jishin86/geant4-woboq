41#include "G4StringModel.hh"
42#include "G4VParticipants.hh"
43#include "G4ReactionProductVector.hh"
44#include "G4ExcitedString.hh"
45#include "G4ExcitedStringVector.hh"

47#include "G4V3DNucleus.hh"
48#include "G4KineticTrackVector.hh"

50class G4VPartonStringModel : public G4VHighEnergyGenerator
51{
52 public:
53 G4VPartonStringModel(const G4String& modelName = "Parton String Model");
54 virtual ~G4VPartonStringModel();
55
56 private:
57 G4VPartonStringModel(const G4VPartonStringModel &right);
58 const G4VPartonStringModel & operator=(const G4VPartonStringModel &right);
59 int operator==(const G4VPartonStringModel &right) const;
60 int operator!=(const G4VPartonStringModel &right) const;
61
62 public:
63 void SetFragmentationModel(G4VStringFragmentation * aModel);
64 G4KineticTrackVector * Scatter(const G4Nucleus &theNucleus, const G4DynamicParticle &thePrimary);
65 virtual G4V3DNucleus * GetWoundedNucleus() const = 0;
66 virtual void ModelDescription(std::ostream& outFile) const;
67 virtual G4V3DNucleus * GetProjectileNucleus() const;
68
69 protected:
70 virtual void Init(const G4Nucleus &theNucleus, const G4DynamicParticle &thePrimary) = 0;
71 virtual G4ExcitedStringVector * GetStrings() = 0;
72 void SetThisPointer(G4VPartonStringModel * aPointer);
73
74 G4bool EnergyAndMomentumCorrector(G4KineticTrackVector* Output, G4LorentzVector& TotalCollisionMomentum);
75
76 private:
77 G4VStringFragmentation * stringFragmentationModel;
78 G4VPartonStringModel * theThis;
79};
80
81inline void G4VPartonStringModel::SetFragmentationModel(G4VStringFragmentation * aModel)
82{
83 stringFragmentationModel = aModel;
84}
85
86inline void G4VPartonStringModel::SetThisPointer(G4VPartonStringModel * aPointer)
87{
88 theThis=aPointer;
89}
