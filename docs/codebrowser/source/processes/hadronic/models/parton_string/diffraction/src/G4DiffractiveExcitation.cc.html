91G4bool G4DiffractiveExcitation::ExciteParticipants( G4VSplitableHadron* projectile,
92 G4VSplitableHadron* target,
93 G4FTFParameters* theParameters,
94 G4ElasticHNScattering* theElastic ) const {
95
96 #ifdef debugFTFexictation
97 G4cout << G4endl << "FTF ExciteParticipants --------------" << G4endl;
98 #endif
99
100 CommonVariables common;
101
102 // Projectile parameters
103 common.Pprojectile = projectile->Get4Momentum();
104 if ( common.Pprojectile.z() < 0.0 ) return false;
105 common.ProjectilePDGcode = projectile->GetDefinition()->GetPDGEncoding();
106 common.absProjectilePDGcode = std::abs( common.ProjectilePDGcode );
107 common.M0projectile = common.Pprojectile.mag();
108 G4double ProjectileRapidity = common.Pprojectile.rapidity();
109
110 // Target parameters
111 common.Ptarget = target->Get4Momentum();
112 common.TargetPDGcode = target->GetDefinition()->GetPDGEncoding();
113 common.absTargetPDGcode = std::abs( common.TargetPDGcode );
114 common.M0target = common.Ptarget.mag();
115 G4double TargetRapidity = common.Ptarget.rapidity();
116
117 // Kinematical properties of the interactions
118 G4LorentzVector Psum = common.Pprojectile + common.Ptarget; // 4-momentum in Lab.
119 common.S = Psum.mag2();
120 common.SqrtS = std::sqrt( common.S );
121
122 // Check off-shellness of the participants
123 G4bool toBePutOnMassShell = false;
124 common.MminProjectile = common.BrW.GetMinimumMass( projectile->GetDefinition() );
125 if ( common.M0projectile < common.MminProjectile ) {
126 toBePutOnMassShell = true;
127 common.M0projectile = common.BrW.SampleMass( projectile->GetDefinition(),
128 projectile->GetDefinition()->GetPDGMass()
129 + 5.0*projectile->GetDefinition()->GetPDGWidth() );
130 }
131 common.M0projectile2 = common.M0projectile * common.M0projectile;
132 common.ProjectileDiffStateMinMass = theParameters->GetProjMinDiffMass();
133 common.ProjectileNonDiffStateMinMass = theParameters->GetProjMinNonDiffMass();
134 if ( common.M0projectile > common.ProjectileDiffStateMinMass ) {
135 common.ProjectileDiffStateMinMass = common.M0projectile + 220.0*MeV;
136 common.ProjectileNonDiffStateMinMass = common.M0projectile + 220.0*MeV;
137 if ( common.absProjectilePDGcode > 3000 ) { // Strange baryon
138 common.ProjectileDiffStateMinMass += 140.0*MeV;
139 common.ProjectileNonDiffStateMinMass += 140.0*MeV;
140 }
141 }
142 common.MminTarget = common.BrW.GetMinimumMass( target->GetDefinition() );
143 if ( common.M0target < common.MminTarget ) {
144 toBePutOnMassShell = true;
145 common.M0target = common.BrW.SampleMass( target->GetDefinition(),
146 target->GetDefinition()->GetPDGMass()
147 + 5.0*target->GetDefinition()->GetPDGWidth() );
148 }
149 common.M0target2 = common.M0target * common.M0target;
150 common.TargetDiffStateMinMass = theParameters->GetTarMinDiffMass();
151 common.TargetNonDiffStateMinMass = theParameters->GetTarMinNonDiffMass();
152 if ( common.M0target > common.TargetDiffStateMinMass ) {
153 common.TargetDiffStateMinMass = common.M0target + 220.0*MeV;
154 common.TargetNonDiffStateMinMass = common.M0target + 220.0*MeV;
155 if ( common.absTargetPDGcode > 3000 ) { // Strange baryon
156 common.TargetDiffStateMinMass += 140.0*MeV;
157 common.TargetNonDiffStateMinMass += 140.0*MeV;
158 }
159 };
160 #ifdef debugFTFexictation
161 G4cout << "Proj Targ PDGcodes " << common.ProjectilePDGcode << " " << common.TargetPDGcode << G4endl
162 << "M0projectile Y " << common.M0projectile << " " << ProjectileRapidity << G4endl;
163 //G4cout << "M0target Y " << common.M0target << " " << TargetRapidity << G4endl;
164 G4cout << "Pproj " << common.Pprojectile << G4endl << "Ptarget " << common.Ptarget << G4endl;
165 #endif
166
167 // Transform momenta to cms and then rotate parallel to z axis;
168 common.toCms = G4LorentzRotation( -1 * Psum.boostVector() );
169 G4LorentzVector Ptmp = common.toCms * common.Pprojectile;
170 if ( Ptmp.pz() <= 0.0 ) return false; // "String" moving backwards in CMS, abort collision!
171 common.toCms.rotateZ( -1*Ptmp.phi() );
172 common.toCms.rotateY( -1*Ptmp.theta() );
173 common.toLab = common.toCms.inverse();
174 common.Pprojectile.transform( common.toCms );
175 common.Ptarget.transform( common.toCms );
176
177 G4double SumMasses = common.M0projectile + common.M0target; // + 220.0*MeV;
178 #ifdef debugFTFexictation
179 G4cout << "SqrtS " << common.SqrtS << G4endl << "M0pr M0tr SumM " << common.M0projectile
180 << " " << common.M0target << " " << SumMasses << G4endl;
181 #endif
182 if ( common.SqrtS < SumMasses ) return false; // The model cannot work at low energy
183
184 common.PZcms2 = ( sqr( common.S ) + sqr( common.M0projectile2 ) + sqr( common.M0target2 )
185 - 2.0 * ( common.S * ( common.M0projectile2 + common.M0target2 )
186 + common.M0projectile2 * common.M0target2 ) ) / 4.0 / common.S;
187 #ifdef debugFTFexictation
188 G4cout << "PZcms2 after toBePutOnMassShell " << common.PZcms2 << G4endl;
189 #endif
190 if ( common.PZcms2 < 0.0 ) return false; // It can be in an interaction with off-shell nuclear nucleon
191
192 common.PZcms = std::sqrt( common.PZcms2 );
193 if ( toBePutOnMassShell ) {
194 if ( common.Pprojectile.z() > 0.0 ) {
195 common.Pprojectile.setPz( common.PZcms );
196 common.Ptarget.setPz( -common.PZcms );
197 } else {
198 common.Pprojectile.setPz( -common.PZcms );
199 common.Ptarget.setPz( common.PZcms );
200 }
201 common.Pprojectile.setE( std::sqrt( common.M0projectile2
202 + common.Pprojectile.x() * common.Pprojectile.x()
203 + common.Pprojectile.y() * common.Pprojectile.y()
204 + common.PZcms2 ) );
205 common.Ptarget.setE( std::sqrt( common.M0target2
206 + common.Ptarget.x() * common.Ptarget.x()
207 + common.Ptarget.y() * common.Ptarget.y()
208 + common.PZcms2 ) );
209 }
210 #ifdef debugFTFexictation
211 G4cout << "Start --------------------" << G4endl << "Proj M0 Mdif Mndif " << common.M0projectile
212 << " " << common.ProjectileDiffStateMinMass << " " << common.ProjectileNonDiffStateMinMass
213 << G4endl
214 << "Targ M0 Mdif Mndif " << common.M0target << " " << common.TargetDiffStateMinMass
215 << " " << common.TargetNonDiffStateMinMass << G4endl << "SqrtS " << common.SqrtS << G4endl
216 << "Proj CMS " << common.Pprojectile << G4endl << "Targ CMS " << common.Ptarget << G4endl;
217 #endif
218
219 // Check for possible quark exchange
220 ProjectileRapidity = common.Pprojectile.rapidity();
221 TargetRapidity = common.Ptarget.rapidity();
222 G4double QeNoExc = theParameters->GetProcProb( 0, ProjectileRapidity - TargetRapidity );
223 G4double QeExc = theParameters->GetProcProb( 1, ProjectileRapidity - TargetRapidity ) *
224 theParameters->GetProcProb( 4, ProjectileRapidity - TargetRapidity );
225 common.ProbProjectileDiffraction =
226 theParameters->GetProcProb( 2, ProjectileRapidity - TargetRapidity );
227 common.ProbTargetDiffraction =
228 theParameters->GetProcProb( 3, ProjectileRapidity - TargetRapidity );
229 common.ProbOfDiffraction = common.ProbProjectileDiffraction + common.ProbTargetDiffraction;
230 #ifdef debugFTFexictation
231 G4cout << "Proc Probs " << QeNoExc << " " << QeExc << " "
232 << common.ProbProjectileDiffraction << " " << common.ProbTargetDiffraction << G4endl
233 << "ProjectileRapidity " << ProjectileRapidity << G4endl;
234 #endif
235 if ( QeNoExc + QeExc + common.ProbProjectileDiffraction + common.ProbTargetDiffraction > 1.0 ) {
236 QeNoExc = 1.0 - QeExc - common.ProbProjectileDiffraction - common.ProbTargetDiffraction;
237 }
238 if ( QeExc + QeNoExc != 0.0 ) {
239 common.ProbExc = QeExc / ( QeExc + QeNoExc );
240 }
241 if ( 1.0 - QeExc - QeNoExc > 0.0 ) {
242 common.ProbProjectileDiffraction /= ( 1.0 - QeExc - QeNoExc );
243 common.ProbTargetDiffraction /= ( 1.0 - QeExc - QeNoExc );
244 }
245 #ifdef debugFTFexictation
246 G4cout << "Proc Probs " << QeNoExc << " " << QeExc << " "
247 << common.ProbProjectileDiffraction << " " << common.ProbTargetDiffraction << G4endl
248 << "ProjectileRapidity " << ProjectileRapidity << G4endl;
249 #endif
250
251 // Try out charge exchange
252 G4int returnCode = 1;
253 if ( G4UniformRand() < QeExc + QeNoExc ) {
254 returnCode = ExciteParticipants_doChargeExchange( projectile, target, theParameters,
255 theElastic, common );
256 }
257
258 G4bool returnResult = false;
259 if ( returnCode == 0 ) {
260 returnResult = true; // Successfully ended: no need of extra work
261 } else if ( returnCode == 1 ) {
262
263 common.ProbOfDiffraction = common.ProbProjectileDiffraction + common.ProbTargetDiffraction;
264 #ifdef debugFTFexictation
265 G4cout << "Excitation --------------------" << G4endl
266 << "Proj M0 MdMin MndMin " << common.M0projectile << " "
267 << common.ProjectileDiffStateMinMass << " " << common.ProjectileNonDiffStateMinMass
268 << G4endl
269 << "Targ M0 MdMin MndMin " << common.M0target << " " << common.TargetDiffStateMinMass
270 << " " << common.TargetNonDiffStateMinMass << G4endl << "SqrtS " << common.SqrtS
271 << G4endl
272 << "Prob: ProjDiff TargDiff + Sum " << common.ProbProjectileDiffraction << " "
273 << common.ProbTargetDiffraction << " " << common.ProbOfDiffraction << G4endl;
274 #endif
275 if ( common.ProbOfDiffraction != 0.0 ) {
276 common.ProbProjectileDiffraction /= common.ProbOfDiffraction;
277 } else {
278 common.ProbProjectileDiffraction = 0.0;
279 }
280 #ifdef debugFTFexictation
281 G4cout << "Prob: ProjDiff TargDiff + Sum " << common.ProbProjectileDiffraction << " "
282 << common.ProbTargetDiffraction << " " << common.ProbOfDiffraction << G4endl;
283 #endif
284 common.ProjectileDiffStateMinMass2 = sqr( common.ProjectileDiffStateMinMass );
285 common.ProjectileNonDiffStateMinMass2 = sqr( common.ProjectileNonDiffStateMinMass );
286 common.TargetDiffStateMinMass2 = sqr( common.TargetDiffStateMinMass );
287 common.TargetNonDiffStateMinMass2 = sqr( common.TargetNonDiffStateMinMass );
288 // Choose between diffraction and non-diffraction process
289 if ( G4UniformRand() < common.ProbOfDiffraction ) {
290
291 returnResult = ExciteParticipants_doDiffraction( projectile, target, theParameters, common );
292
293 } else {
294
295 returnResult = ExciteParticipants_doNonDiffraction( projectile, target, theParameters, common );
296
297 }
298 if ( returnResult ) {
299 common.Pprojectile += common.Qmomentum;
300 common.Ptarget -= common.Qmomentum;
301 // Transform back and update SplitableHadron Participant.
302 common.Pprojectile.transform( common.toLab );
303 common.Ptarget.transform( common.toLab );
304 #ifdef debugFTFexictation
305 G4cout << "Mproj " << common.Pprojectile.mag() << G4endl << "Mtarg " << common.Ptarget.mag()
306 << G4endl;
307 #endif
308 projectile->Set4Momentum( common.Pprojectile );
309 target->Set4Momentum( common.Ptarget );
310 projectile->IncrementCollisionCount( 1 );
311 target->IncrementCollisionCount( 1 );
312 }
313 }
314
315 return returnResult;
316}

320G4int G4DiffractiveExcitation::
321ExciteParticipants_doChargeExchange( G4VSplitableHadron* projectile,
322 G4VSplitableHadron* target,
323 G4FTFParameters* theParameters,
324 G4ElasticHNScattering* theElastic,
325 G4DiffractiveExcitation::CommonVariables& common ) const {
326 // First of the three utility methods used only by ExciteParticipants:
327 // it does the sampling for the charge-exchange case.
328 // This method returns an integer code - instead of a boolean, with the following meaning:
329 // "0" : successfully ended and nothing else needs to be done;
330 // "1" : successfully completed, but the work needs to be continued;
331 // "99" : unsuccessfully ended, nothing else can be done.
332 G4int returnCode = 99;
333
334 G4double DeltaProbAtQuarkExchange = theParameters->GetDeltaProbAtQuarkExchange();
335 G4ParticleDefinition* TestParticle = 0;
336 G4double MtestPr = 0.0, MtestTr = 0.0;
337
338 #ifdef debugFTFexictation
339 G4cout << "Q exchange --------------------------" << G4endl;
340 #endif
341 G4int NewProjCode = 0, NewTargCode = 0, ProjQ1 = 0, ProjQ2 = 0, ProjQ3 = 0;
342 // Projectile unpacking
343 if ( common.absProjectilePDGcode < 1000 ) { // projectile is meson
344 UnpackMeson( common.ProjectilePDGcode, ProjQ1, ProjQ2 );
345 } else { // projectile is baryon
346 UnpackBaryon( common.ProjectilePDGcode, ProjQ1, ProjQ2, ProjQ3 );
347 }
348 // Target unpacking
349 G4int TargQ1 = 0, TargQ2 = 0, TargQ3 = 0;
350 UnpackBaryon( common.TargetPDGcode, TargQ1, TargQ2, TargQ3 );
351 #ifdef debugFTFexictation
352 G4cout << "Proj Quarks " << ProjQ1 << " " << ProjQ2 << " " << ProjQ3 << G4endl
353 << "Targ Quarks " << TargQ1 << " " << TargQ2 << " " << TargQ3 << G4endl;
354 #endif
355
356 // Sampling of exchanged quarks
357 G4int ProjExchangeQ = 0, TargExchangeQ = 0;
358 if ( common.absProjectilePDGcode < 1000 ) { // projectile is meson
359
360 G4bool isProjQ1Quark = false;
361 ProjExchangeQ = ProjQ2;
362 if ( ProjQ1 > 0 ) { // ProjQ1 is a quark
363 isProjQ1Quark = true;
364 ProjExchangeQ = ProjQ1;
365 }
366 // Exchange of non-identical quarks is allowed
367 G4int NpossibleStates = 0;
368 if ( ProjExchangeQ != TargQ1 ) NpossibleStates++;
369 if ( ProjExchangeQ != TargQ2 ) NpossibleStates++;
370 if ( ProjExchangeQ != TargQ3 ) NpossibleStates++;
371 G4int Nsampled = G4RandFlat::shootInt( G4long( NpossibleStates ) ) + 1;
372 NpossibleStates = 0;
373 if ( ProjExchangeQ != TargQ1 ) {
374 if ( ++NpossibleStates == Nsampled ) {
375 TargExchangeQ = TargQ1; TargQ1 = ProjExchangeQ;
376 isProjQ1Quark ? ProjQ1 = TargExchangeQ : ProjQ2 = TargExchangeQ;
377 }
378 }
379 if ( ProjExchangeQ != TargQ2 ) {
380 if ( ++NpossibleStates == Nsampled ) {
381 TargExchangeQ = TargQ2; TargQ2 = ProjExchangeQ;
382 isProjQ1Quark ? ProjQ1 = TargExchangeQ : ProjQ2 = TargExchangeQ;
383 }
384 }
385 if ( ProjExchangeQ != TargQ3 ) {
386 if ( ++NpossibleStates == Nsampled ) {
387 TargExchangeQ = TargQ3; TargQ3 = ProjExchangeQ;
388 isProjQ1Quark ? ProjQ1 = TargExchangeQ : ProjQ2 = TargExchangeQ;
389 }
390 }
391 #ifdef debugFTFexictation
392 G4cout << "Exchanged Qs in Pr Tr " << ProjExchangeQ << " " << TargExchangeQ << G4endl;
393 #endif
394
395 G4int aProjQ1 = std::abs( ProjQ1 ), aProjQ2 = std::abs( ProjQ2 );
396 G4bool ProjExcited = false;
397 const G4int maxNumberOfAttempts = 50;
398 G4int attempts = 0;
399 while ( attempts++ < maxNumberOfAttempts ) { /* Loop checking, 10.08.2015, A.Ribon */
400
401 // Determination of a new projectile ID which satisfies energy-momentum conservation
402 G4double Ksi = G4UniformRand();
403 if ( aProjQ1 == aProjQ2 ) {
404 if ( aProjQ1 != 3 ) {
405 NewProjCode = 111; // Pi0-meson
406 if ( Ksi < 0.5 ) {
407 NewProjCode = 221; // Eta -meson
408 if ( Ksi < 0.25 ) {
409 NewProjCode = 331; // Eta'-meson
410 }
411 }
412 } else {
413 NewProjCode = 221; // Eta -meson
414 if ( Ksi < 0.5 ) {
415 NewProjCode = 331; // Eta'-meson
416 }
417 }
418 } else {
419 if ( aProjQ1 > aProjQ2 ) {
420 NewProjCode = aProjQ1*100 + aProjQ2*10 + 1;
421 } else {
422 NewProjCode = aProjQ2*100 + aProjQ1*10 + 1;
423 }
424 }
425 #ifdef debugFTFexictation
426 G4cout << "NewProjCode " << NewProjCode << G4endl;
427 #endif
428 ProjExcited = false;
429 if ( G4UniformRand() < 0.5 ) {
430 NewProjCode += 2; // Excited meson (J=1 -> 2*J+1=2+1=3, last digit of the PDG code)
431 ProjExcited = true;
432 }
433 // So far we have used the absolute values of the PDG codes of the two constituent quarks:
434 // now look at their signed values to set properly the signed of the meson's PDG code.
435 G4int value = ProjQ1, absValue = aProjQ1, Qquarks = 0;
436 for ( G4int iQuark = 0; iQuark < 2; iQuark++ ) {
437 if ( iQuark == 1 ) {
438 value = ProjQ2; absValue = aProjQ2;
439 }
440 if ( absValue == 2 ) {
441 Qquarks += value; // u or ubar : u-quark is positively charged +2 (e/3 unit)
442 } else {
443 Qquarks -= value/absValue; // d or dbar or s or sbar : d- or s-quark is negatively charged -1 (e/3 unit)
444 }
445 }
446 if ( Qquarks < 0 ) NewProjCode *= -1;
447 #ifdef debugFTFexictation
448 G4cout << "NewProjCode +2 or 0 " << NewProjCode << G4endl;
449 G4cout<<"+++++++++++++++++++++++++++++++++++++++"<<G4endl;
450 G4cout<<ProjQ1<<" "<<ProjQ2<<" "<<Qquarks<<G4endl;
451 G4cout<<"+++++++++++++++++++++++++++++++++++++++"<<G4endl;
452 #endif
453
454 // Proj
455 TestParticle = G4ParticleTable::GetParticleTable()->FindParticle( NewProjCode );
456 if ( ! TestParticle ) continue;
457 common.MminProjectile = common.BrW.GetMinimumMass( TestParticle );
458 if ( common.SqrtS - common.M0target < common.MminProjectile ) continue;
459 MtestPr = common.BrW.SampleMass( TestParticle, TestParticle->GetPDGMass()
460 + 5.0*TestParticle->GetPDGWidth() );
461 #ifdef debugFTFexictation
462 G4cout << "TestParticle Name " << NewProjCode << " " << TestParticle->GetParticleName()
463 << G4endl
464 << "MtestPart MtestPart0 "<<MtestPr<<" "<<TestParticle->GetPDGMass()<<G4endl
465 << "M0projectile projectile PDGMass " << common.M0projectile << " "
466 << projectile->GetDefinition()->GetPDGMass() << G4endl;
467 #endif
468
469 // Targ
470 NewTargCode = NewNucleonId( TargQ1, TargQ2, TargQ3 );
471 #ifdef debugFTFexictation
472 G4cout << "New TrQ " << TargQ1 << " " << TargQ2 << " " << TargQ3 << G4endl
473 << "NewTargCode " << NewTargCode << G4endl;
474 #endif
475 if ( TargQ1 != TargQ2 && TargQ1 != TargQ3 && TargQ2 != TargQ3 ) { // Lambda or Sigma0 ?
476 if ( G4UniformRand() < 0.5 ) {
477 NewTargCode += 2;
478 } else if ( G4UniformRand() < 0.75 ) {
479 NewTargCode = 3122;
480 }
481 } else if ( TargQ1 == TargQ2 && TargQ1 == TargQ3 ) {
482 NewTargCode += 2; ProjExcited = true; // Create Delta isobar
483 } else if ( target->GetDefinition()->GetPDGiIsospin() == 3 ) { // Delta was the target
484 if ( G4UniformRand() > DeltaProbAtQuarkExchange ) {
485 NewTargCode += 2; ProjExcited = true;
486 }
487 } else if ( ! ProjExcited &&
488 G4UniformRand() < DeltaProbAtQuarkExchange && // Nucleon was the target
489 common.SqrtS > common.M0projectile + // Delta mass
490 G4ParticleTable::GetParticleTable()->FindParticle( 2224 )->GetPDGMass() ) {
491 NewTargCode += 2; // Create Delta isobar
492 }
493 TestParticle = G4ParticleTable::GetParticleTable()->FindParticle( NewTargCode );
494 if ( ! TestParticle ) continue;
495 #ifdef debugFTFexictation
496 G4cout << "New targ " << NewTargCode << " " << TestParticle->GetParticleName() << G4endl;
497 #endif
498 common.MminTarget = common.BrW.GetMinimumMass( TestParticle );
499 if ( common.SqrtS - MtestPr < common.MminTarget ) continue;
500 MtestTr = common.BrW.SampleMass( TestParticle, TestParticle->GetPDGMass()
501 + 5.0*TestParticle->GetPDGWidth() );
502 if ( common.SqrtS > MtestPr + MtestTr ) break;
503
504 } // End of while loop
505 if ( attempts >= maxNumberOfAttempts ) return returnCode; // unsuccessfully ended, nothing else can be done
506
507 if ( MtestPr >= common.Pprojectile.mag() || projectile->GetStatus() != 0 ) {
508 common.M0projectile = MtestPr;
509 }
510 #ifdef debugFTFexictation
511 G4cout << "M0projectile After check " << common.M0projectile << G4endl;
512 #endif
513 common.M0projectile2 = common.M0projectile * common.M0projectile;
514 common.ProjectileDiffStateMinMass = common.M0projectile + 220.0*MeV; // 220 MeV=m_pi+80 MeV
515 common.ProjectileNonDiffStateMinMass = common.M0projectile + 220.0*MeV; // 220 MeV=m_pi+80 MeV
516 if ( MtestTr >= common.Ptarget.mag() || target->GetStatus() != 0 ) {
517 common.M0target = MtestTr;
518 }
519 common.M0target2 = common.M0target * common.M0target;
520 #ifdef debugFTFexictation
521 G4cout << "New targ M0 M0^2 " << common.M0target << " " << common.M0target2 << G4endl;
522 #endif
523 common.TargetDiffStateMinMass = common.M0target + 220.0*MeV; // 220 MeV=m_pi+80 MeV;
524 common.TargetNonDiffStateMinMass = common.M0target + 220.0*MeV; // 220 MeV=m_pi+80 MeV;
525
526 } else { // of the if ( common.absProjectilePDGcode < 1000 ) ; the projectile is baryon now
527
528 // Choose randomly, with equal probability, whether to consider the quarks of the
529 // projectile or target hadron for selecting the flavour of the exchanged quark.
530 G4bool isProjectileExchangedQ = false;
531 G4int firstQ = TargQ1, secondQ = TargQ2, thirdQ = TargQ3;
532 G4int otherFirstQ = ProjQ1, otherSecondQ = ProjQ2, otherThirdQ = ProjQ3;
533 if ( G4UniformRand() < 0.5 ) {
534 isProjectileExchangedQ = true;
535 firstQ = ProjQ1; secondQ = ProjQ2; thirdQ = ProjQ3;
536 otherFirstQ = TargQ1; otherSecondQ = TargQ2; otherThirdQ = TargQ3;
537 }
538 // Choose randomly, with equal probability, which of the three quarks of the
539 // selected (projectile or target) hadron is the exhanged quark.
540 G4int exchangedQ = 0;
541 G4double Ksi = G4UniformRand();
542 if ( Ksi < 0.333333 ) {
543 exchangedQ = firstQ;
544 } else if ( 0.333333 <= Ksi && Ksi < 0.666667 ) {
545 exchangedQ = secondQ;
546 } else {
547 exchangedQ = thirdQ;
548 }
549 #ifdef debugFTFexictation
550 G4cout << "Exchange Qs isProjectile Q " << isProjectileExchangedQ << " " << exchangedQ << " ";
551 #endif
552 // The exchanged quarks (one of the projectile hadron and one of the target hadron)
553 // are always accepted if they have different flavour, else (i.e. when they have the
554 // same flavour) they are accepted only with a specified probability.
555 G4double probSame = theParameters->GetProbOfSameQuarkExchange();
556 const G4int MaxCount = 100;
557 G4int count = 0, otherExchangedQ = 0;
558 do {
559 if ( exchangedQ != otherFirstQ || G4UniformRand() < probSame ) {
560 otherExchangedQ = otherFirstQ; otherFirstQ = exchangedQ; exchangedQ = otherExchangedQ;
561 } else {
562 if ( exchangedQ != otherSecondQ || G4UniformRand() < probSame ) {
563 otherExchangedQ = otherSecondQ; otherSecondQ = exchangedQ; exchangedQ = otherExchangedQ;
564 } else {
565 //ALB if ( exchangedQ != otherThirdQ || G4UniformRand() < probSame ) {
566 otherExchangedQ = otherThirdQ; otherThirdQ = exchangedQ; exchangedQ = otherExchangedQ;
567 //ALB }
568 }
569 }
570 } while ( otherExchangedQ == 0 && ++count < MaxCount );
571 if ( count >= MaxCount ) return returnCode; // All attempts failed: unsuccessfully ended, nothing else can be done
572 // Swap (between projectile and target hadron) the two quarks that have been sampled
573 // as "exchanged" quarks.
574 if ( Ksi < 0.333333 ) {
575 firstQ = exchangedQ;
576 } else if ( 0.333333 <= Ksi && Ksi < 0.666667 ) {
577 secondQ = exchangedQ;
578 } else {
579 thirdQ = exchangedQ;
580 }
581 if ( isProjectileExchangedQ ) {
582 ProjQ1 = firstQ; ProjQ2 = secondQ; ProjQ3 = thirdQ;
583 TargQ1 = otherFirstQ; TargQ2 = otherSecondQ; TargQ3 = otherThirdQ;
584 } else {
585 TargQ1 = firstQ; TargQ2 = secondQ; TargQ3 = thirdQ;
586 ProjQ1 = otherFirstQ; ProjQ2 = otherSecondQ; ProjQ3 = otherThirdQ;
587 }
588 #ifdef debugFTFexictation
589 G4cout << "Exchange Qs Pr Tr " << ( isProjectileExchangedQ ? exchangedQ : otherExchangedQ )
590 << " " << ( isProjectileExchangedQ ? otherExchangedQ : exchangedQ ) << G4endl;
591 #endif
592
593 NewProjCode = NewNucleonId( ProjQ1, ProjQ2, ProjQ3 );
594 NewTargCode = NewNucleonId( TargQ1, TargQ2, TargQ3 );
595 // Decide whether the new projectile hadron is a Delta particle;
596 // then decide whether the new target hadron is a Delta particle.
597 // Notice that a Delta particle has the last PDG digit "4" (because its spin is 3/2),
598 // whereas a nucleon has "2" (because its spin is 1/2).
599 for ( G4int iHadron = 0; iHadron < 2; iHadron++ ) {
600 // First projectile hadron, then target hadron
601 G4int codeQ1 = ProjQ1, codeQ2 = ProjQ2, codeQ3 = ProjQ3, newHadCode = NewProjCode;
602 G4double massConstraint = common.M0target;
603 G4bool isHadronADelta = ( projectile->GetDefinition()->GetPDGiIsospin() == 3 );
604 if ( iHadron == 1 ) { // Target hadron
605 codeQ1 = TargQ1, codeQ2 = TargQ2, codeQ3 = TargQ3, newHadCode = NewTargCode;
606 massConstraint = common.M0projectile;
607 isHadronADelta = ( target->GetDefinition()->GetPDGiIsospin() == 3 );
608 }
609 if ( codeQ1 == codeQ2 && codeQ1 == codeQ3 ) { // The three quarks are the same
610 newHadCode += 2; // Delta++ (uuu) or Delta- (ddd) : spin 3/2, last PDG digit = 4 (so +2 wrt p or n)
611 } else if ( isHadronADelta ) { // Hadron (projectile or target) was Delta
612 if ( G4UniformRand() > DeltaProbAtQuarkExchange ) {
613 newHadCode += 2; // Delta+ (uud) or Delta0 (udd) : spin 3/2, last PDG digit = 4 (so +2 wrt p or n)
614 } else {
615 newHadCode += 0; // No delta (so the last PDG digit remains 2)
616 }
617 } else { // Hadron (projectile or target) was Nucleon
618 if ( G4UniformRand() < DeltaProbAtQuarkExchange &&
619 common.SqrtS > G4ParticleTable::GetParticleTable()->FindParticle( 2224 )->GetPDGMass()
620 + massConstraint ) {
621 newHadCode += 2; // Delta+ (uud) or Delta0 (udd) : spin 3/2, last PDG digit = 4 (so +2 wrt p or n)
622 } else {
623 newHadCode += 0; // No delta (so the last PDG digit remains 2)
624 }
625 }
626 if ( iHadron == 0 ) { // Projectile hadron
627 NewProjCode = newHadCode;
628 } else { // Target hadron
629 NewTargCode = newHadCode;
630 }
631 }
632 #ifdef debugFTFexictation
633 G4cout << "NewProjCode NewTargCode " << NewProjCode << " " << NewTargCode << G4endl;
634 #endif
635
636 if ( common.absProjectilePDGcode == NewProjCode && common.absTargetPDGcode == NewTargCode ) {
637 } // Nothing was changed! It is not right!?
638
639 // Sampling of the masses of the projectile and target nucleons.
640 // Because of energy conservation, the ordering of the sampling matters:
641 // randomly, half of the time we sample first the target nucleon mass and
642 // then the projectile nucleon mass, and the other half of the time we
643 // sample first the projectile nucleon mass and then the target nucleon mass.
644 G4VSplitableHadron* firstHadron = target;
645 G4VSplitableHadron* secondHadron = projectile;
646 G4int firstHadronCode = NewTargCode, secondHadronCode = NewProjCode;
647 G4double massConstraint = common.M0projectile;
648 G4bool isFirstTarget = true;
649 if ( G4UniformRand() < 0.5 ) {
650 // Sample first the projectile nucleon mass, then the target nucleon mass.
651 firstHadron = projectile; secondHadron = target;
652 firstHadronCode = NewProjCode; secondHadronCode = NewTargCode;
653 massConstraint = common.M0target;
654 isFirstTarget = false;
655 }
656 G4double Mtest1st = 0.0, Mtest2nd = 0.0, Mmin1st = 0.0, Mmin2nd = 0.0;
657 for ( int iSamplingCase = 0; iSamplingCase < 2; iSamplingCase++ ) {
658 G4VSplitableHadron* aHadron = firstHadron;
659 G4int aHadronCode = firstHadronCode;
660 if ( iSamplingCase == 1 ) { // Second nucleon mass sampling
661 aHadron = secondHadron; aHadronCode = secondHadronCode; massConstraint = Mtest1st;
662 }
663 G4double MtestHadron = 0.0, MminHadron = 0.0;
664 if ( aHadron->GetStatus() == 1 || aHadron->GetStatus() == 2 ) {
665 TestParticle = G4ParticleTable::GetParticleTable()->FindParticle( aHadronCode );
666 if ( ! TestParticle ) return returnCode; // Not possible to find such a hadron: unsuccessfully ended, nothing else can be done
667 MminHadron = common.BrW.GetMinimumMass( TestParticle );
668 if ( common.SqrtS - massConstraint < MminHadron ) return returnCode; // Kinematically impossible: unsuccessfully ended, nothing else can be done
669 if ( TestParticle->GetPDGWidth() == 0.0 ) {
670 MtestHadron = common.BrW.SampleMass( TestParticle, TestParticle->GetPDGMass() );
671 } else {
672 const G4int maxNumberOfAttempts = 50;
673 G4int attempts = 0;
674 while ( attempts < maxNumberOfAttempts ) {
675 attempts++;
676 MtestHadron = common.BrW.SampleMass( TestParticle, TestParticle->GetPDGMass()
677 + 5.0*TestParticle->GetPDGWidth() );
678 if ( common.SqrtS < MtestHadron + massConstraint ) {
679 continue; // Kinematically unacceptable: try again
680 } else {
681 break; // Kinematically acceptable: the mass sampling is successful
682 }
683 }
684 if ( attempts >= maxNumberOfAttempts ) return returnCode; // All attempts failed: unsuccessfully ended, nothing else can be done
685 }
686 }
687 if ( iSamplingCase == 0 ) {
688 Mtest1st = MtestHadron; Mmin1st = MminHadron;
689 } else {
690 Mtest2nd = MtestHadron; Mmin2nd = MminHadron;
691 }
692 } // End for loop on the two sampling cases (1st and 2nd)
693 if ( isFirstTarget ) {
694 MtestTr = Mtest1st; MtestPr = Mtest2nd;
695 common.MminTarget = Mmin1st; common.MminProjectile = Mmin2nd;
696 } else {
697 MtestTr = Mtest2nd; MtestPr = Mtest1st;
698 common.MminTarget = Mmin2nd; common.MminProjectile = Mmin1st;
699 }
700
701 if ( MtestPr != 0.0 ) {
702 common.M0projectile = MtestPr;
703 common.M0projectile2 = common.M0projectile * common.M0projectile;
704 common.ProjectileDiffStateMinMass = common.M0projectile + 220.0*MeV; // 220 MeV=m_pi+80 MeV
705 common.ProjectileNonDiffStateMinMass = common.M0projectile + 220.0*MeV; // 220 MeV=m_pi+80 MeV
706 }
707 if ( MtestTr != 0.0 ) {
708 common.M0target = MtestTr;
709 common.M0target2 = common.M0target * common.M0target;
710 common.TargetDiffStateMinMass = common.M0target + 220.0*MeV; // 220 MeV=m_pi+80 MeV;
711 common.TargetNonDiffStateMinMass = common.M0target + 220.0*MeV; // 220 MeV=m_pi+80 MeV;
712 }
713
714 } // End of if ( common.absProjectilePDGcode < 1000 )
715
716 // If we assume that final state hadrons after the charge exchange will be
717 // in the ground states, we have to put
718 if ( common.SqrtS < common.M0projectile + common.M0target ) return returnCode; // unsuccessfully ended, nothing else can be done
719
720 common.PZcms2 = ( sqr( common.S ) + sqr( common.M0projectile2 ) + sqr( common.M0target2 )
721 - 2.0 * ( common.S * ( common.M0projectile2 + common.M0target2 )
722 + common.M0projectile2 * common.M0target2 ) ) / 4.0 / common.S;
723 #ifdef debugFTFexictation
724 G4cout << "At the end// NewProjCode " << NewProjCode << G4endl
725 << "At the end// NewTargCode " << NewTargCode << G4endl
726 << "M0pr M0tr SqS " << common.M0projectile << " " << common.M0target << " "
727 << common.SqrtS << G4endl
728 << "M0pr2 M0tr2 SqS " << common.M0projectile2 << " " << common.M0target2 << " "
729 << common.SqrtS << G4endl
730 << "PZcms2 after the change " << common.PZcms2 << G4endl << G4endl;
731 #endif
732 if ( common.PZcms2 < 0.0 ) return returnCode; // It can be if energy is not sufficient for Delta
733 // unsuccessfully ended, nothing else can be done
734 projectile->SetDefinition( G4ParticleTable::GetParticleTable()->FindParticle( NewProjCode ) );
735 target->SetDefinition( G4ParticleTable::GetParticleTable()->FindParticle( NewTargCode ) );
736 common.PZcms = std::sqrt( common.PZcms2 );
737 common.Pprojectile.setPz( common.PZcms );
738 common.Pprojectile.setE( std::sqrt( common.M0projectile2 + common.PZcms2 ) );
739 common.Ptarget.setPz( -common.PZcms );
740 common.Ptarget.setE( std::sqrt( common.M0target2 + common.PZcms2 ) );
741 #ifdef debugFTFexictation
742 G4cout << "Proj Targ and Proj+Targ in CMS" << G4endl << common.Pprojectile << G4endl
743 << common.Ptarget << G4endl << common.Pprojectile + common.Ptarget << G4endl;
744 #endif
745
746 if ( projectile->GetStatus() != 0 ) projectile->SetStatus( 2 );
747 if ( target->GetStatus() != 0 ) target->SetStatus( 2 );
748
749 // Check for possible excitation of the participants
750 if ( common.SqrtS < common.M0projectile + common.TargetDiffStateMinMass ||
751 common.SqrtS < common.ProjectileDiffStateMinMass + common.M0target ||
752 common.ProbOfDiffraction == 0.0 ) common.ProbExc = 0.0;
753
754 if ( G4UniformRand() > common.ProbExc ) { // Make elastic scattering
755 #ifdef debugFTFexictation
756 G4cout << "Make elastic scattering of new hadrons" << G4endl;
757 #endif
758 common.Pprojectile.transform( common.toLab );
759 common.Ptarget.transform( common.toLab );
760 projectile->Set4Momentum( common.Pprojectile );
761 target->Set4Momentum( common.Ptarget );
762 G4bool Result = theElastic->ElasticScattering( projectile, target, theParameters );
763 #ifdef debugFTFexictation
764 G4cout << "Result of el. scatt " << Result << G4endl << "Proj Targ and Proj+Targ in Lab"
765 << G4endl << projectile->Get4Momentum() << G4endl << target->Get4Momentum() << G4endl
766 << projectile->Get4Momentum() + target->Get4Momentum() << " "
767 << (projectile->Get4Momentum() + target->Get4Momentum()).mag() << G4endl;
768 #endif
769 if ( Result ) returnCode = 0; // successfully ended and nothing else needs to be done
770 return returnCode;
771 }
772
773 #ifdef debugFTFexictation
774 G4cout << "Make excitation of new hadrons" << G4endl;
775 #endif
776
777 // Redefinition of ProbOfDiffraction because the probabilities are changed due to quark exchange
778 common.ProbOfDiffraction = common.ProbProjectileDiffraction + common.ProbTargetDiffraction;
779 if ( common.ProbOfDiffraction != 0.0 ) {
780 common.ProbProjectileDiffraction /= common.ProbOfDiffraction;
781 common.ProbTargetDiffraction /= common.ProbOfDiffraction;
782 }
783
784 return returnCode = 1; // successfully completed, but the work needs to be continued
785}

789G4bool G4DiffractiveExcitation::
790ExciteParticipants_doDiffraction( G4VSplitableHadron* projectile, G4VSplitableHadron* target,
791 G4FTFParameters* theParameters,
792 G4DiffractiveExcitation::CommonVariables& common ) const {
793 // Second of the three utility methods used only by ExciteParticipants:
794 // it does the sampling for the diffraction case, either projectile or target diffraction.
795
796 G4bool isProjectileDiffraction = false;
797 if ( G4UniformRand() < common.ProbProjectileDiffraction ) { // projectile diffraction
798 isProjectileDiffraction = true;
799 #ifdef debugFTFexictation
800 G4cout << "projectile diffraction" << G4endl;
801 #endif
802 common.ProjMassT2 = common.ProjectileDiffStateMinMass2;
803 common.ProjMassT = common.ProjectileDiffStateMinMass;
804 common.TargMassT2 = common.M0target2;
805 common.TargMassT = common.M0target;
806 } else { // target diffraction
807 #ifdef debugFTFexictation
808 G4cout << "Target diffraction" << G4endl;
809 #endif
810 common.ProjMassT2 = common.M0projectile2;
811 common.ProjMassT = common.M0projectile;
812 common.TargMassT2 = common.TargetDiffStateMinMass2;
813 common.TargMassT = common.TargetDiffStateMinMass;
814 }
815
816 G4double DiffrAveragePt2 = theParameters->GetAvaragePt2ofElasticScattering()*1.2;
817 G4bool loopCondition = true;
818 G4int whilecount = 0;
819 do { // Generate pt and mass of projectile
820
821 whilecount++;
822 if ( whilecount > 1000 ) {
823 common.Qmomentum = G4LorentzVector( 0.0, 0.0, 0.0, 0.0 );
824 return false; // Ignore this interaction
825 };
826
827 // Check that the interaction is possible
828 if ( common.SqrtS < common.ProjMassT + common.TargMassT ) return false;
829
830 common.PZcms2 = ( sqr( common.S ) + sqr( common.ProjMassT2 ) + sqr( common.TargMassT2 )
831 - 2.0 * ( common.S * ( common.ProjMassT2 + common.TargMassT2 )
832 + common.ProjMassT2 * common.TargMassT2 ) ) / 4.0 / common.S;
833 if ( common.PZcms2 < 0.0 ) return false;
834
835 common.maxPtSquare = common.PZcms2;
836 common.Qmomentum = G4LorentzVector( GaussianPt( DiffrAveragePt2, common.maxPtSquare ), 0 );
837 common.Pt2 = G4ThreeVector( common.Qmomentum.vect() ).mag2();
838 if ( isProjectileDiffraction ) { // projectile diffraction
839 common.ProjMassT2 = common.ProjectileDiffStateMinMass2 + common.Pt2;
840 common.TargMassT2 = common.M0target2 + common.Pt2;
841 } else { // target diffraction
842 common.ProjMassT2 = common.M0projectile2 + common.Pt2;
843 common.TargMassT2 = common.TargetDiffStateMinMass2 + common.Pt2;
844 }
845 common.ProjMassT = std::sqrt( common.ProjMassT2 );
846 common.TargMassT = std::sqrt( common.TargMassT2 );
847 if ( common.SqrtS < common.ProjMassT + common.TargMassT ) continue;
848
849 common.PZcms2 = ( sqr( common.S ) + sqr( common.ProjMassT2 ) + sqr( common.TargMassT2 )
850 - 2.0 * ( common.S * ( common.ProjMassT2 + common.TargMassT2 )
851 + common.ProjMassT2 * common.TargMassT2 ) ) / 4.0 / common.S;
852 if ( common.PZcms2 < 0.0 ) continue;
853
854 common.PZcms = std::sqrt( common.PZcms2 );
855 if ( isProjectileDiffraction ) { // projectile diffraction
856 common.PMinusMin = std::sqrt( common.ProjMassT2 + common.PZcms2 ) - common.PZcms;
857 common.PMinusMax = common.SqrtS - common.TargMassT;
858 common.PMinusNew = ChooseP( common.PMinusMin, common.PMinusMax );
859 common.TMinusNew = common.SqrtS - common.PMinusNew;
860 common.Qminus = common.Ptarget.minus() - common.TMinusNew;
861 common.TPlusNew = common.TargMassT2 / common.TMinusNew;
862 common.Qplus = common.Ptarget.plus() - common.TPlusNew;
863 common.Qmomentum.setPz( (common.Qplus - common.Qminus)/2.0 );
864 common.Qmomentum.setE( (common.Qplus + common.Qminus)/2.0 );
865 loopCondition = ( ( common.Pprojectile + common.Qmomentum ).mag2() <
866 common.ProjectileDiffStateMinMass2 );
867 } else { // target diffraction
868 common.TPlusMin = std::sqrt( common.TargMassT2 + common.PZcms2 ) - common.PZcms;
869 common.TPlusMax = common.SqrtS - common.ProjMassT;
870 common.TPlusNew = ChooseP( common.TPlusMin, common.TPlusMax );
871 common.PPlusNew = common.SqrtS - common.TPlusNew;
872 common.Qplus = common.PPlusNew - common.Pprojectile.plus();
873 common.PMinusNew = common.ProjMassT2 / common.PPlusNew;
874 common.Qminus = common.PMinusNew - common.Pprojectile.minus();
875 common.Qmomentum.setPz( (common.Qplus - common.Qminus)/2.0 );
876 common.Qmomentum.setE( (common.Qplus + common.Qminus)/2.0 );
877 loopCondition = ( ( common.Ptarget - common.Qmomentum ).mag2() <
878 common.TargetDiffStateMinMass2 );
879 }
880
881 } while ( loopCondition ); /* Loop checking, 10.08.2015, A.Ribon */
882 // Repeat the sampling because there was not any excitation
883
884 if ( isProjectileDiffraction ) { // projectile diffraction
885 projectile->SetStatus( 0 );
886 if ( projectile->GetStatus() == 2 ) projectile->SetStatus( 1 );
887 if ( target->GetStatus() == 1 && target->GetSoftCollisionCount() == 0 ) target->SetStatus( 2 );
888 } else { // target diffraction
889 target->SetStatus( 0 );
890 }
891
892 return true;
893}

929 common.maxPtSquare = common.PZcms2;
930 common.Qmomentum = G4LorentzVector( GaussianPt( theParameters->GetAveragePt2(),
931 common.maxPtSquare ), 0 );

935 common.TargMassT2 = common.TargetNonDiffStateMinMass2 + common.Pt2;

941 + common.ProjMassT2 * common.TargMassT2 ) ) / 4.0 / common.S;

961 common.Qmomentum.setPz( (common.Qplus - common.Qminus)/2.0 );
962 common.Qmomentum.setE( (common.Qplus + common.Qminus)/2.0 );
963 #ifdef debugFTFexictation

965 << ( common.Pprojectile + common.Qmomentum ).mag() << " "
966 << common.ProjectileNonDiffStateMinMass << G4endl
967 << ( common.Ptarget - common.Qmomentum ).mag() << " "
968 << common.TargetNonDiffStateMinMass << G4endl;
969 #endif
970
971 } while ( ( common.Pprojectile + common.Qmomentum ).mag2() <
972 common.ProjectileNonDiffStateMinMass2 || // No double Diffraction
973 ( common.Ptarget - common.Qmomentum ).mag2() <
974 common.TargetNonDiffStateMinMass2 ); /* Loop checking, 10.08.2015, A.Ribon */
975
976 projectile->SetStatus( 0 );
977 target->SetStatus( 0 );
978 return true;
979}

1353G4double G4DiffractiveExcitation::GetQuarkFractionOfKink( G4double zmin, G4double zmax ) const {
1354 G4double z, yf;
1355 const G4int maxNumberOfLoops = 10000;
1356 G4int loopCounter = 0;
1357 do {
1358 z = zmin + G4UniformRand() * (zmax - zmin);
1359 yf = z*z + sqr(1.0 - z);
1360 } while ( ( G4UniformRand() > yf ) &&
1361 ++loopCounter < maxNumberOfLoops ); /* Loop checking, 10.08.2015, A.Ribon */
1362 if ( loopCounter >= maxNumberOfLoops ) {
1363 z = 0.5*(zmin + zmax); // Just something acceptable, without any physics consideration.
1364 }
