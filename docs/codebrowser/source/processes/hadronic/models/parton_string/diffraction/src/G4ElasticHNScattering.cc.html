63G4bool G4ElasticHNScattering::ElasticScattering( G4VSplitableHadron* projectile,
64 G4VSplitableHadron* target,
65 G4FTFParameters* theParameters ) const {
66 projectile->IncrementCollisionCount( 1 );
67 target->IncrementCollisionCount( 1 );
68
69 G4SampleResonance BrW;
70
71 // Projectile parameters
72 G4LorentzVector Pprojectile = projectile->Get4Momentum();
73 if ( Pprojectile.z() < 0.0 ) return false;
74 G4bool PutOnMassShell( false );
75 G4double M0projectile = Pprojectile.mag();
76 //if ( M0projectile < projectile->GetDefinition()->GetPDGMass() ) {
77
78 G4double MminProjectile=BrW.GetMinimumMass(projectile->GetDefinition());
79
80 if ( M0projectile < MminProjectile ) {
81 PutOnMassShell = true;
82 M0projectile = projectile->GetDefinition()->GetPDGMass();
83 }
84 G4double M0projectile2 = M0projectile * M0projectile;
85 G4double AveragePt2 = theParameters->GetAvaragePt2ofElasticScattering();
86
87 // Target parameters
88 G4LorentzVector Ptarget = target->Get4Momentum();
89 G4double M0target = Ptarget.mag();
90 //if ( M0target < target->GetDefinition()->GetPDGMass() ) {
91
92 G4double MminTarget=BrW.GetMinimumMass(target->GetDefinition());
93
94 if ( M0target < MminTarget ) {
95 PutOnMassShell = true;
96 M0target = target->GetDefinition()->GetPDGMass();
97 }
98 G4double M0target2 = M0target * M0target;
99
100 // Transform momenta to cms and then rotate parallel to z axis;
101 G4LorentzVector Psum;
102 Psum = Pprojectile + Ptarget;
103 G4LorentzRotation toCms( -1*Psum.boostVector() );
104 G4LorentzVector Ptmp = toCms*Pprojectile;
105 if ( Ptmp.pz() <= 0.0 ) return false;
106 //"String" moving backwards in CMS, abort collision !
107 //G4cout << " abort Collision! " << G4endl;
108 toCms.rotateZ( -1*Ptmp.phi() );
109 toCms.rotateY( -1*Ptmp.theta() );
110 G4LorentzRotation toLab( toCms.inverse() );
111 Pprojectile.transform( toCms );
112 Ptarget.transform( toCms );
113
114 // Putting on mass-on-shell, if needed
115 G4double PZcms2, PZcms;
116 G4double S = Psum.mag2();
117 G4double SqrtS = std::sqrt( S );
118 if ( SqrtS < M0projectile + M0target ) return false;
119
120 PZcms2 = ( S*S + sqr( M0projectile2 ) + sqr( M0target2 )
121 - 2*S*M0projectile2 - 2*S*M0target2 - 2*M0projectile2*M0target2 ) / 4.0 / S;
122
123 if ( PZcms2 < 0.0 ) { // It can be in an interaction with off-shell nuclear nucleon
124 if ( M0projectile > projectile->GetDefinition()->GetPDGMass() ) {
125 // An attempt to de-excite the projectile
126 // It is assumed that the target is in the ground state
127 M0projectile = projectile->GetDefinition()->GetPDGMass();
128 M0projectile2 = M0projectile * M0projectile;
129 PZcms2= ( S*S + sqr( M0projectile2 ) + sqr( M0target2 )
130 - 2*S*M0projectile2 - 2*S*M0target2 - 2*M0projectile2*M0target2 ) / 4.0 / S;
131 if ( PZcms2 < 0.0 ) { return false; } // Nonsuccesful attempt to de-excitate the projectile
132 } else {
133 return false; // The projectile was not excited, but the energy was too low to put
134 // the target nucleon on mass-shell
135 }
136 }
137
138 PZcms = std::sqrt( PZcms2 );
139
140 if ( PutOnMassShell ) {
141 if ( Pprojectile.z() > 0.0 ) {
142 Pprojectile.setPz( PZcms );
143 Ptarget.setPz( -PZcms );
144 } else {
145 Pprojectile.setPz( -PZcms );
146 Ptarget.setPz( PZcms );
147 };
148 Pprojectile.setE( std::sqrt( M0projectile2 + Pprojectile.x() * Pprojectile.x() +
149 Pprojectile.y() * Pprojectile.y() + PZcms2 ) );
150 Ptarget.setE( std::sqrt( M0target2 + Ptarget.x() * Ptarget.x() + Ptarget.y() * Ptarget.y() +
151 PZcms2 ) );
152 }
153
154 G4double maxPtSquare = PZcms2;
155
156 // Now we can calculate the transferred Pt
157 G4double Pt2;
158 G4double ProjMassT2, ProjMassT;
159 G4double TargMassT2, TargMassT;
160 G4LorentzVector Qmomentum;
161
162 const G4int maxNumberOfLoops = 1000;
163 G4int loopCounter = 0;
164 do {
165 Qmomentum = G4LorentzVector( GaussianPt( AveragePt2, maxPtSquare ), 0.0 );
166 Pt2 = G4ThreeVector( Qmomentum.vect() ).mag2();
167 ProjMassT2 = M0projectile2 + Pt2;
168 ProjMassT = std::sqrt( ProjMassT2 );
169 TargMassT2 = M0target2 + Pt2;
170 TargMassT = std::sqrt( TargMassT2 );
171 } while ( ( SqrtS < ProjMassT + TargMassT ) &&
172 ++loopCounter < maxNumberOfLoops ); /* Loop checking, 10.08.2015, A.Ribon */
173 if ( loopCounter >= maxNumberOfLoops ) {
174 return false;
175 }
176
177 PZcms2 = ( S*S + sqr( ProjMassT2 ) + sqr( TargMassT2 )
178 - 2.0*S*ProjMassT2 - 2.0*S*TargMassT2 - 2.0*ProjMassT2*TargMassT2 ) / 4.0 / S;
179
180 if ( PZcms2 < 0.0 ) { PZcms2 = 0.0; }; // to avoid the exactness problem
181 PZcms = std::sqrt( PZcms2 );
182 Pprojectile.setPz( PZcms );
183 Ptarget.setPz( -PZcms );
184 Pprojectile += Qmomentum;
185 Ptarget -= Qmomentum;
186
187 // Transform back and update SplitableHadron Participant.
188 Pprojectile.transform( toLab );
189 Ptarget.transform( toLab );
190
191 // Calculation of the creation time
192 projectile->SetTimeOfCreation( target->GetTimeOfCreation() );
193 projectile->SetPosition( target->GetPosition() );
194
195 // Creation time and position of target nucleon were determined at
196 // ReggeonCascade() of G4FTFModel
197
198 projectile->Set4Momentum( Pprojectile );
199 target->Set4Momentum( Ptarget );
200
201 //projectile->IncrementCollisionCount( 1 );
202 //target->IncrementCollisionCount( 1 );
203
204 return true;
205}

210G4ThreeVector G4ElasticHNScattering::GaussianPt( G4double AveragePt2,
211 G4double maxPtSquare ) const {
212 // @@ this method is used in FTFModel as well. Should go somewhere common!
213 G4double Pt2( 0.0 );
214 if ( AveragePt2 <= 0.0 ) {
215 Pt2 = 0.0;
216 } else {
217 Pt2 = -AveragePt2 * G4Log( 1.0 + G4UniformRand() *
218 ( G4Exp( -maxPtSquare/AveragePt2 ) -1.0 ) );
219 }
220 G4double Pt = std::sqrt( Pt2 );
221 G4double phi = G4UniformRand() * twopi;
222 return G4ThreeVector( Pt * std::cos( phi ), Pt * std::sin( phi ), 0.0 );
223}
