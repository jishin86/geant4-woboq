71 G4VPartonStringModel( modelName ),
72 theExcitation( new G4DiffractiveExcitation() ),
73 theElastic( new G4ElasticHNScattering() ),
74 theAnnihilation( new G4FTFAnnihilation() )
75{
76 G4VPartonStringModel::SetThisPointer( this );
77 // ---> JVY theParameters = 0;
78 theParameters = new G4FTFParameters();
79 //
80 NumberOfInvolvedNucleonsOfTarget = 0;
81 NumberOfInvolvedNucleonsOfProjectile= 0;
82 for ( G4int i = 0; i < 250; i++ ) {
83 TheInvolvedNucleonsOfTarget[i] = 0;
84 TheInvolvedNucleonsOfProjectile[i] = 0;
85 }
86
87 //LowEnergyLimit = 2000.0*MeV;

89
90 HighEnergyInter = true;
91

97

102
103 SetEnergyMomentumCheckLevels( 2.0*perCent, 150.0*MeV );
104}

114G4FTFModel::~G4FTFModel() {
115 // Because FTF model can be called for various particles
116 //
117 // ---> NOTE (JVY): This statement below is no longer true !!!
118 // theParameters must be erased at the end of each call.
119 // Thus the delete is also in G4FTFModel::GetStrings() method.
120 // ---> JVY

123 if ( theExcitation != 0 ) delete theExcitation;
124 if ( theElastic != 0 ) delete theElastic;
125 if ( theAnnihilation != 0 ) delete theAnnihilation;
126
127 // Erasing of strings created at annihilation.
128 if ( theAdditionalString.size() != 0 ) {
129 std::for_each( theAdditionalString.begin(), theAdditionalString.end(),
130 DeleteVSplitableHadron() );
131 }
132 theAdditionalString.clear();
133
134 // Erasing of target involved nucleons.
135 if ( NumberOfInvolvedNucleonsOfTarget != 0 ) {
136 for ( G4int i = 0; i < NumberOfInvolvedNucleonsOfTarget; i++ ) {
137 G4VSplitableHadron* aNucleon = TheInvolvedNucleonsOfTarget[i]->GetSplitableHadron();
138 if ( aNucleon ) delete aNucleon;
139 }
140 }
141
142 // Erasing of projectile involved nucleons.
143 if ( NumberOfInvolvedNucleonsOfProjectile != 0 ) {
144 for ( G4int i = 0; i < NumberOfInvolvedNucleonsOfProjectile; i++ ) {
145 G4VSplitableHadron* aNucleon = TheInvolvedNucleonsOfProjectile[i]->GetSplitableHadron();

147 }
148 }

154void G4FTFModel::Init( const G4Nucleus& aNucleus, const G4DynamicParticle& aProjectile ) {
155
156 theProjectile = aProjectile;
157
158 G4double PlabPerParticle( 0.0 ); // Laboratory momentum Pz per particle/nucleon
159
160 #ifdef debugFTFmodel
161 G4cout << "FTF init Proj Name " << theProjectile.GetDefinition()->GetParticleName() << G4endl
162 << "FTF init Proj Mass " << theProjectile.GetMass()
163 << " " << theProjectile.GetMomentum() << G4endl
164 << "FTF init Proj B Q " << theProjectile.GetDefinition()->GetBaryonNumber()
165 << " " << (G4int) theProjectile.GetDefinition()->GetPDGCharge() << G4endl
166 << "FTF init Target A Z " << aNucleus.GetA_asInt()
167 << " " << aNucleus.GetZ_asInt() << G4endl;
168 #endif
169
170 theParticipants.Clean();
171
172 theParticipants.SetProjectileNucleus( 0 );
173
174 G4LorentzVector tmp( 0.0, 0.0, 0.0, 0.0 );
175 ProjectileResidualMassNumber = 0;
176 ProjectileResidualCharge = 0;
177 ProjectileResidualExcitationEnergy = 0.0;
178 ProjectileResidual4Momentum = tmp;
179
180 TargetResidualMassNumber = aNucleus.GetA_asInt();
181 TargetResidualCharge = aNucleus.GetZ_asInt();
182 TargetResidualExcitationEnergy = 0.0;
183 TargetResidual4Momentum = tmp;
184 G4double TargetResidualMass = G4ParticleTable::GetParticleTable()->GetIonTable()
185 ->GetIonMass( TargetResidualCharge, TargetResidualMassNumber );
186
187 TargetResidual4Momentum.setE( TargetResidualMass );
188
189 if ( std::abs( theProjectile.GetDefinition()->GetBaryonNumber() ) <= 1 ) {
190 // Projectile is a hadron : meson or baryon
191 PlabPerParticle = theProjectile.GetMomentum().z();
192 ProjectileResidualMassNumber = std::abs( theProjectile.GetDefinition()->GetBaryonNumber() );
193 ProjectileResidualCharge = G4int( theProjectile.GetDefinition()->GetPDGCharge() );
194 ProjectileResidualExcitationEnergy = 0.0;
195 //G4double ProjectileResidualMass = theProjectile.GetMass();
196 ProjectileResidual4Momentum.setVect( theProjectile.GetMomentum() );
197 ProjectileResidual4Momentum.setE( theProjectile.GetTotalEnergy() );
198 if ( PlabPerParticle < LowEnergyLimit ) {
199 HighEnergyInter = false;
200 } else {
201 HighEnergyInter = true;
202 }
203 } else {
204 if ( theProjectile.GetDefinition()->GetBaryonNumber() > 1 ) {
205 // Projectile is a nucleus
206 theParticipants.InitProjectileNucleus(theProjectile.GetDefinition()->GetBaryonNumber(),
207 G4int(theProjectile.GetDefinition()->GetPDGCharge()));
208 ProjectileResidualMassNumber = theProjectile.GetDefinition()->GetBaryonNumber();
209 ProjectileResidualCharge = G4int( theProjectile.GetDefinition()->GetPDGCharge() );
210 PlabPerParticle = theProjectile.GetMomentum().z() /
211 theProjectile.GetDefinition()->GetBaryonNumber();
212 if ( PlabPerParticle < LowEnergyLimit ) {
213 HighEnergyInter = false;
214 } else {
215 HighEnergyInter = true;
216 }
217 } else if ( theProjectile.GetDefinition()->GetBaryonNumber() < -1 ) {
218 // Projectile is an anti-nucleus
219 theParticipants.InitProjectileNucleus(
220 std::abs( theProjectile.GetDefinition()->GetBaryonNumber() ),
221 std::abs( G4int( theProjectile.GetDefinition()->GetPDGCharge() ) ) );
222 theParticipants.theProjectileNucleus->StartLoop();
223 G4Nucleon* aNucleon;
224 while ( ( aNucleon = theParticipants.theProjectileNucleus->GetNextNucleon() ) ) { /* Loop checking, 10.08.2015, A.Ribon */
225 if ( aNucleon->GetDefinition() == G4Proton::Proton() ) {
226 aNucleon->SetParticleType( G4AntiProton::AntiProton() );
227 } else if ( aNucleon->GetDefinition() == G4Neutron::Neutron() ) {
228 aNucleon->SetParticleType( G4AntiNeutron::AntiNeutron() );
229 }
230 }
231 ProjectileResidualMassNumber = std::abs( theProjectile.GetDefinition()->GetBaryonNumber() );
232 ProjectileResidualCharge = std::abs( G4int(theProjectile.GetDefinition()->GetPDGCharge()) );
233 PlabPerParticle = theProjectile.GetMomentum().z() /
234 std::abs( theProjectile.GetDefinition()->GetBaryonNumber() );
235 if ( PlabPerParticle < LowEnergyLimit ) {
236 HighEnergyInter = false;
237 } else {
238 HighEnergyInter = true;
239 }
240 }
241
242 G4ThreeVector BoostVector = theProjectile.GetMomentum() / theProjectile.GetTotalEnergy();
243 theParticipants.theProjectileNucleus->DoLorentzBoost( BoostVector );
244 theParticipants.theProjectileNucleus->DoLorentzContraction( BoostVector );
245 ProjectileResidualExcitationEnergy = 0.0;
246 //G4double ProjectileResidualMass = theProjectile.GetMass();
247 ProjectileResidual4Momentum.setVect( theProjectile.GetMomentum() );
248 ProjectileResidual4Momentum.setE( theProjectile.GetTotalEnergy() );
249 }
250
251 // Init target nucleus
252 theParticipants.Init( aNucleus.GetA_asInt(), aNucleus.GetZ_asInt() );
253 //theParticipants.Init( aNucleus.GetA_asInt(), 0 ); // For h+neutron
254
255 /*
256 if ( theParameters != 0 ) delete theParameters;
257 theParameters = new G4FTFParameters( theProjectile.GetDefinition(), aNucleus.GetA_asInt(),
258 aNucleus.GetZ_asInt(), PlabPerParticle );
259 */
260
261 // reset/recalculate everything for the new interaction
262 //
263
264 theParameters->InitForInteraction( theProjectile.GetDefinition(), aNucleus.GetA_asInt(),
265 aNucleus.GetZ_asInt(), PlabPerParticle );
266
267 if ( theAdditionalString.size() != 0 ) {
268 std::for_each( theAdditionalString.begin(), theAdditionalString.end(),
269 DeleteVSplitableHadron() );
270 }
271 theAdditionalString.clear();
272
273 #ifdef debugFTFmodel
274 G4cout << "FTF end of Init" << G4endl << G4endl;
275 #endif
276
277 //if ( std::abs( theProjectile.GetDefinition()->GetBaryonNumber() ) <= 1 &&
278 // aNucleus.GetA_asInt() < 2 ) theParameters->SetProbabilityOfElasticScatt( 0.0 );
279}

284G4ExcitedStringVector* G4FTFModel::GetStrings() {
285
286 #ifdef debugFTFmodel
287 G4cout << "G4FTFModel::GetStrings() " << G4endl;
288 #endif
289
290 G4ExcitedStringVector* theStrings = new G4ExcitedStringVector;
291 theParticipants.GetList( theProjectile, theParameters );
292 StoreInvolvedNucleon();
293
294 G4bool Success( true );
295
296 if ( HighEnergyInter ) {
297 ReggeonCascade();
298
299 #ifdef debugFTFmodel
300 G4cout << "FTF PutOnMassShell " << G4endl;
301 #endif
302
303 Success = PutOnMassShell();
304
305 #ifdef debugFTFmodel
306 G4cout << "FTF PutOnMassShell Success? " << Success << G4endl;
307 #endif
308
309 }
310
311 #ifdef debugFTFmodel
312 G4cout << "FTF ExciteParticipants " << G4endl;
313 #endif
314
315 if ( Success ) Success = ExciteParticipants();
316
317 #ifdef debugFTFmodel
318 G4cout << "FTF ExciteParticipants Success? " << Success << G4endl;
319 #endif
320
321 if ( Success ) {
322
323 #ifdef debugFTFmodel
324 G4cout << "FTF BuildStrings ";
325 #endif
326
327 BuildStrings( theStrings );
328
329 #ifdef debugFTFmodel
330 G4cout << "FTF BuildStrings " << theStrings << " OK" << G4endl
331 << "FTF GetResiduals of Nuclei " << G4endl;
332 #endif
333
334 GetResiduals();
335
336 /*
337 if ( theParameters != 0 ) {
338 delete theParameters;
339 theParameters = 0;
340 }
341 */
342 } else if ( ! GetProjectileNucleus() ) {
343 // Erase the hadron projectile
344 std::vector< G4VSplitableHadron* > primaries;
345 theParticipants.StartLoop();
346 while ( theParticipants.Next() ) { /* Loop checking, 10.08.2015, A.Ribon */
347 const G4InteractionContent& interaction = theParticipants.GetInteraction();
348 // Do not allow for duplicates
349 if ( primaries.end() ==
350 std::find( primaries.begin(), primaries.end(), interaction.GetProjectile() ) ) {
351 primaries.push_back( interaction.GetProjectile() );
352 }
353 }
354 std::for_each( primaries.begin(), primaries.end(), DeleteVSplitableHadron() );
355 primaries.clear();
356 }
357
358 // Cleaning of the memory
359 G4VSplitableHadron* aNucleon = 0;
360
361 // Erase the projectile nucleons
362 for ( G4int i = 0; i < NumberOfInvolvedNucleonsOfProjectile; i++ ) {
363 aNucleon = TheInvolvedNucleonsOfProjectile[i]->GetSplitableHadron();
364 if ( aNucleon ) delete aNucleon;
365 }
366 NumberOfInvolvedNucleonsOfProjectile = 0;
367
368 // Erase the target nucleons
369 for ( G4int i = 0; i < NumberOfInvolvedNucleonsOfTarget; i++ ) {
370 aNucleon = TheInvolvedNucleonsOfTarget[i]->GetSplitableHadron();
371 if ( aNucleon ) delete aNucleon;
372 }
373 NumberOfInvolvedNucleonsOfTarget = 0;
374
375 #ifdef debugFTFmodel
376 G4cout << "End of FTF. Go to fragmentation" << G4endl
377 << "To continue - enter 1, to stop - ^C" << G4endl;
378 //G4int Uzhi; G4cin >> Uzhi;
379 #endif
380
381 theParticipants.Clean();
382
383 return theStrings;
384}

389void G4FTFModel::StoreInvolvedNucleon() {
390 //To store nucleons involved in the interaction
391
392 NumberOfInvolvedNucleonsOfTarget = 0;
393
394 G4V3DNucleus* theTargetNucleus = GetTargetNucleus();
395 theTargetNucleus->StartLoop();
396
397 G4Nucleon* aNucleon;
398 while ( ( aNucleon = theTargetNucleus->GetNextNucleon() ) ) { /* Loop checking, 10.08.2015, A.Ribon */
399 if ( aNucleon->AreYouHit() ) {
400 TheInvolvedNucleonsOfTarget[NumberOfInvolvedNucleonsOfTarget] = aNucleon;
401 NumberOfInvolvedNucleonsOfTarget++;
402 }
403 }
404
405 #ifdef debugFTFmodel
406 G4cout << "G4FTFModel::StoreInvolvedNucleon -------------" << G4endl;
407 G4cout << "NumberOfInvolvedNucleonsOfTarget " << NumberOfInvolvedNucleonsOfTarget
408 << G4endl << G4endl;
409 #endif
410
411 if ( ! GetProjectileNucleus() ) return; // The projectile is a hadron
412
413 // The projectile is a nucleus or an anti-nucleus.
414
415 NumberOfInvolvedNucleonsOfProjectile = 0;
416
417 G4V3DNucleus* theProjectileNucleus = GetProjectileNucleus();
418 theProjectileNucleus->StartLoop();
419
420 G4Nucleon* aProjectileNucleon;
421 while ( ( aProjectileNucleon = theProjectileNucleus->GetNextNucleon() ) ) { /* Loop checking, 10.08.2015, A.Ribon */
422 if ( aProjectileNucleon->AreYouHit() ) {
423 // Projectile nucleon was involved in the interaction.
424 TheInvolvedNucleonsOfProjectile[NumberOfInvolvedNucleonsOfProjectile] = aProjectileNucleon;
425 NumberOfInvolvedNucleonsOfProjectile++;
426 }
427 }
428
429 #ifdef debugFTFmodel
430 G4cout << "NumberOfInvolvedNucleonsOfProjectile " << NumberOfInvolvedNucleonsOfProjectile
431 << G4endl << G4endl;
432 #endif
433 return;
434}

439void G4FTFModel::ReggeonCascade() {
440 // Implementation of the reggeon theory inspired model
441
442 #ifdef debugReggeonCascade
443 G4cout << "G4FTFModel::ReggeonCascade -----------" << G4endl
444 << "theProjectile.GetTotalMomentum() " << theProjectile.GetTotalMomentum() << G4endl
445 << "theProjectile.GetTotalEnergy() " << theProjectile.GetTotalEnergy() << G4endl
446 << "ExcitationE/WN " << theParameters->GetExcitationEnergyPerWoundedNucleon() << G4endl;
447 #endif
448
449 G4int InitNINt = NumberOfInvolvedNucleonsOfTarget;
450
451 // Reggeon cascading in target nucleus
452 for ( G4int InvTN = 0; InvTN < InitNINt; InvTN++ ) {
453 G4Nucleon* aTargetNucleon = TheInvolvedNucleonsOfTarget[ InvTN ];
454
455 G4double CreationTime = aTargetNucleon->GetSplitableHadron()->GetTimeOfCreation();
456
457 G4double XofWoundedNucleon = aTargetNucleon->GetPosition().x();
458 G4double YofWoundedNucleon = aTargetNucleon->GetPosition().y();
459
460 G4V3DNucleus* theTargetNucleus = GetTargetNucleus();
461 theTargetNucleus->StartLoop();
462
463 G4Nucleon* Neighbour(0);
464 while ( ( Neighbour = theTargetNucleus->GetNextNucleon() ) ) { /* Loop checking, 10.08.2015, A.Ribon */
465 if ( ! Neighbour->AreYouHit() ) {
466 G4double impact2 = sqr( XofWoundedNucleon - Neighbour->GetPosition().x() ) +
467 sqr( YofWoundedNucleon - Neighbour->GetPosition().y() );
468
469 if ( G4UniformRand() < theParameters->GetCofNuclearDestruction() *
470 G4Exp( -impact2 / theParameters->GetR2ofNuclearDestruction() )
471 ) {
472 // The neighbour nucleon is involved in the reggeon cascade
473 TheInvolvedNucleonsOfTarget[ NumberOfInvolvedNucleonsOfTarget ] = Neighbour;
474 NumberOfInvolvedNucleonsOfTarget++;
475
476 G4VSplitableHadron* targetSplitable;
477 targetSplitable = new G4DiffractiveSplitableHadron( *Neighbour );
478
479 Neighbour->Hit( targetSplitable );
480 targetSplitable->SetTimeOfCreation( CreationTime );
481 targetSplitable->SetStatus( 3 ); // 2->3
482 }
483 }
484 }
485 }
486
487 #ifdef debugReggeonCascade
488 G4cout << "Final NumberOfInvolvedNucleonsOfTarget "
489 << NumberOfInvolvedNucleonsOfTarget << G4endl << G4endl;
490 #endif
491
492 if ( ! GetProjectileNucleus() ) return;
493
494 // Nucleus-Nucleus Interaction : Destruction of Projectile
495 G4int InitNINp = NumberOfInvolvedNucleonsOfProjectile;
496
497 //for ( G4int InvPN = 0; InvPN < NumberOfInvolvedNucleonsOfProjectile; InvPN++ ) {
498 for ( G4int InvPN = 0; InvPN < InitNINp; InvPN++ ) {
499 G4Nucleon* aProjectileNucleon = TheInvolvedNucleonsOfProjectile[ InvPN ];
500
501 G4double CreationTime = aProjectileNucleon->GetSplitableHadron()->GetTimeOfCreation();
502
503 G4double XofWoundedNucleon = aProjectileNucleon->GetPosition().x();
504 G4double YofWoundedNucleon = aProjectileNucleon->GetPosition().y();
505
506 G4V3DNucleus* theProjectileNucleus = GetProjectileNucleus();
507 theProjectileNucleus->StartLoop();
508
509 G4Nucleon* Neighbour( 0 );
510 while ( ( Neighbour = theProjectileNucleus->GetNextNucleon() ) ) { /* Loop checking, 10.08.2015, A.Ribon */
511 if ( ! Neighbour->AreYouHit() ) {
512 G4double impact2= sqr( XofWoundedNucleon - Neighbour->GetPosition().x() ) +
513 sqr( YofWoundedNucleon - Neighbour->GetPosition().y() );
514
515 if ( G4UniformRand() < theParameters->GetCofNuclearDestructionPr() *
516 G4Exp( -impact2 / theParameters->GetR2ofNuclearDestruction() )
517 ) {
518 // The neighbour nucleon is involved in the reggeon cascade
519 TheInvolvedNucleonsOfProjectile[ NumberOfInvolvedNucleonsOfProjectile ] = Neighbour;
520 NumberOfInvolvedNucleonsOfProjectile++;
521
522 G4VSplitableHadron* projectileSplitable;
523 projectileSplitable = new G4DiffractiveSplitableHadron( *Neighbour );
524
525 Neighbour->Hit( projectileSplitable );
526 projectileSplitable->SetTimeOfCreation( CreationTime );
527 projectileSplitable->SetStatus( 3 );
528 }
529 }
530 }
531 }
532
533 #ifdef debugReggeonCascade
534 G4cout << "NumberOfInvolvedNucleonsOfProjectile "
535 << NumberOfInvolvedNucleonsOfProjectile << G4endl << G4endl;
536 #endif
537}

542G4bool G4FTFModel::PutOnMassShell() {
543
544 G4bool isProjectileNucleus = false;
545 if ( GetProjectileNucleus() ) {
546 isProjectileNucleus = true;
547 }
548
549 #ifdef debugPutOnMassShell
550 G4cout << "PutOnMassShell start " << G4endl;
551 if ( isProjectileNucleus ) {
552 G4cout << "PutOnMassShell for Nucleus_Nucleus " << G4endl;
553 }
554 #endif
555
556 G4LorentzVector Pprojectile( theProjectile.GetMomentum(), theProjectile.GetTotalEnergy() );
557 if ( Pprojectile.z() < 0.0 ) {
558 return false;
559 }
560
561 G4bool isOk = true;
562
563 G4LorentzVector Ptarget( 0.0, 0.0, 0.0, 0.0 );
564 G4LorentzVector PtargetResidual( 0.0, 0.0, 0.0, 0.0 );
565 G4double SumMasses = 0.0;
566 G4V3DNucleus* theTargetNucleus = GetTargetNucleus();
567 G4double TargetResidualMass = 0.0;
568
569 #ifdef debugPutOnMassShell
570 G4cout << "Target : ";
571 #endif
572 isOk = ComputeNucleusProperties( theTargetNucleus, Ptarget, PtargetResidual, SumMasses,
573 TargetResidualExcitationEnergy, TargetResidualMass,
574 TargetResidualMassNumber, TargetResidualCharge );
575 if ( ! isOk ) return false;
576
577 G4double Mprojectile = 0.0;
578 G4double M2projectile = 0.0;
579 G4LorentzVector Pproj( 0.0, 0.0, 0.0, 0.0 );
580 G4LorentzVector PprojResidual( 0.0, 0.0, 0.0, 0.0 );
581 G4V3DNucleus* thePrNucleus = GetProjectileNucleus();
582 G4double PrResidualMass = 0.0;
583
584 if ( ! isProjectileNucleus ) { // hadron-nucleus collision
585 Mprojectile = Pprojectile.mag();
586 M2projectile = Pprojectile.mag2();
587 SumMasses += Mprojectile + 20.0*MeV;
588 } else { // nucleus-nucleus or antinucleus-nucleus collision
589 #ifdef debugPutOnMassShell
590 G4cout << "Projectile : ";
591 #endif
592 isOk = ComputeNucleusProperties( thePrNucleus, Pproj, PprojResidual, SumMasses,
593 ProjectileResidualExcitationEnergy, PrResidualMass,
594 ProjectileResidualMassNumber, ProjectileResidualCharge );
595 if ( ! isOk ) return false;
596 }
597
598 G4LorentzVector Psum = Pprojectile + Ptarget;
599 G4double SqrtS = Psum.mag();
600 G4double S = Psum.mag2();
601
602 #ifdef debugPutOnMassShell
603 G4cout << "Psum " << Psum/GeV << " GeV" << G4endl << "SqrtS " << SqrtS/GeV << " GeV" << G4endl
604 << "SumMasses, PrResidualMass and TargetResidualMass " << SumMasses/GeV << " "
605 << PrResidualMass/GeV << " " << TargetResidualMass/GeV << " GeV" << G4endl;
606 #endif
607
608 if ( SqrtS < SumMasses ) {
609 return false; // It is impossible to simulate after putting nuclear nucleons on mass-shell.
610 }
611
612 // Try to consider also the excitation energy of the residual nucleus, if this is
613 // possible, with the available energy; otherwise, set the excitation energy to zero.
614 G4double savedSumMasses = SumMasses;
615 if ( isProjectileNucleus ) {
616 SumMasses -= std::sqrt( sqr( PrResidualMass ) + PprojResidual.perp2() );
617 SumMasses += std::sqrt( sqr( PrResidualMass + ProjectileResidualExcitationEnergy )
618 + PprojResidual.perp2() );
619 }
620 SumMasses -= std::sqrt( sqr( TargetResidualMass ) + PtargetResidual.perp2() );
621 SumMasses += std::sqrt( sqr( TargetResidualMass + TargetResidualExcitationEnergy )
622 + PtargetResidual.perp2() );
623
624 if ( SqrtS < SumMasses ) {
625 SumMasses = savedSumMasses;
626 if ( isProjectileNucleus ) {
627 ProjectileResidualExcitationEnergy = 0.0;
628 }
629 TargetResidualExcitationEnergy = 0.0;
630 }
631
632 TargetResidualMass += TargetResidualExcitationEnergy;
633 if ( isProjectileNucleus ) {
634 PrResidualMass += ProjectileResidualExcitationEnergy;
635 }
636
637 #ifdef debugPutOnMassShell
638 if ( isProjectileNucleus ) {
639 G4cout << "PrResidualMass ProjResidualExcitationEnergy " << PrResidualMass/GeV << " "
640 << ProjectileResidualExcitationEnergy << " MeV" << G4endl;
641 }
642 G4cout << "TargetResidualMass TargetResidualExcitationEnergy " << TargetResidualMass/GeV << " "
643 << TargetResidualExcitationEnergy << " MeV" << G4endl
644 << "Sum masses " << SumMasses/GeV << G4endl;
645 #endif
646
647 // Sampling of nucleons what can transfer to delta-isobars
648 if ( isProjectileNucleus && thePrNucleus->GetMassNumber() != 1 ) {
649 isOk = GenerateDeltaIsobar( SqrtS, NumberOfInvolvedNucleonsOfProjectile,
650 TheInvolvedNucleonsOfProjectile, SumMasses );
651 }
652 if ( theTargetNucleus->GetMassNumber() != 1 ) {
653 isOk = isOk &&
654 GenerateDeltaIsobar( SqrtS, NumberOfInvolvedNucleonsOfTarget,
655 TheInvolvedNucleonsOfTarget, SumMasses );
656 }
657 if ( ! isOk ) return false;
658
659 // Now we know that it is kinematically possible to produce a final state made
660 // of the involved nucleons (or corresponding delta-isobars) and a residual nucleus.
661 // We have to sample the kinematical variables which will allow to define the 4-momenta
662 // of the final state. The sampled kinematical variables refer to the center-of-mass frame.
663 // Notice that the sampling of the transverse momentum corresponds to take into account
664 // Fermi motion.
665
666 G4LorentzRotation toCms( -1*Psum.boostVector() );
667 G4LorentzVector Ptmp = toCms*Pprojectile;
668 if ( Ptmp.pz() <= 0.0 ) { // "String" moving backwards in c.m.s., abort collision!
669 return false;
670 }
671
672 G4LorentzRotation toLab( toCms.inverse() );
673
674 G4double YprojectileNucleus = 0.0;
675 if ( isProjectileNucleus ) {
676 Ptmp = toCms*Pproj;
677 YprojectileNucleus = Ptmp.rapidity();
678 }
679 Ptmp = toCms*Ptarget;
680 G4double YtargetNucleus = Ptmp.rapidity();
681
682 // Ascribing of the involved nucleons Pt and Xminus
683 G4double DcorP = 0.0;
684 if ( isProjectileNucleus ) {
685 DcorP = theParameters->GetDofNuclearDestruction() / thePrNucleus->GetMassNumber();
686 }
687 G4double DcorT = theParameters->GetDofNuclearDestruction() / theTargetNucleus->GetMassNumber();
688 G4double AveragePt2 = theParameters->GetPt2ofNuclearDestruction();
689 G4double maxPtSquare = theParameters->GetMaxPt2ofNuclearDestruction();
690
691 #ifdef debugPutOnMassShell
692 if ( isProjectileNucleus ) {
693 G4cout << "Y projectileNucleus " << YprojectileNucleus << G4endl;
694 }
695 G4cout << "Y targetNucleus " << YtargetNucleus << G4endl
696 << "Dcor " << theParameters->GetDofNuclearDestruction()
697 << " DcorP DcorT " << DcorP << " " << DcorT << " AveragePt2 " << AveragePt2 << G4endl;
698 #endif
699
700 G4double M2proj = M2projectile; // Initialization needed only for hadron-nucleus collisions
701 G4double WplusProjectile = 0.0;
702 G4double M2target = 0.0;
703 G4double WminusTarget = 0.0;
704 G4int NumberOfTries = 0;
705 G4double ScaleFactor = 1.0;
706 G4bool OuterSuccess = true;
707
708 const G4int maxNumberOfLoops = 1000;
709 G4int loopCounter = 0;
710 do { // while ( ! OuterSuccess )
711 OuterSuccess = true;
712 const G4int maxNumberOfInnerLoops = 10000;
713 do { // while ( SqrtS < Mprojectile + std::sqrt( M2target ) )
714 NumberOfTries++;
715 if ( NumberOfTries == 100*(NumberOfTries/100) ) {
716 // After many tries, it is convenient to reduce the values of DcorP, DcorT and
717 // AveragePt2, so that the sampled momenta (respectively, pz, and pt) of the
718 // involved nucleons (or corresponding delta-isomers) are smaller, and therefore
719 // it is more likely to satisfy the momentum conservation.
720 ScaleFactor /= 2.0;
721 DcorP *= ScaleFactor;
722 DcorT *= ScaleFactor;
723 AveragePt2 *= ScaleFactor;
724 }
725 if ( isProjectileNucleus ) {
726 // Sampling of kinematical properties of projectile nucleons
727 isOk = SamplingNucleonKinematics( AveragePt2, maxPtSquare, DcorP,
728 thePrNucleus, PprojResidual,
729 PrResidualMass, ProjectileResidualMassNumber,
730 NumberOfInvolvedNucleonsOfProjectile,
731 TheInvolvedNucleonsOfProjectile, M2proj );
732 }
733 // Sampling of kinematical properties of target nucleons
734 isOk = isOk &&
735 SamplingNucleonKinematics( AveragePt2, maxPtSquare, DcorT,
736 theTargetNucleus, PtargetResidual,
737 TargetResidualMass, TargetResidualMassNumber,
738 NumberOfInvolvedNucleonsOfTarget,
739 TheInvolvedNucleonsOfTarget, M2target );
740
741 #ifdef debugPutOnMassShell
742 G4cout << "SqrtS, Mp+Mt, Mp, Mt " << SqrtS/GeV << " "
743 << ( std::sqrt( M2proj ) + std::sqrt( M2target) )/GeV << " "
744 << std::sqrt( M2proj )/GeV << " " << std::sqrt( M2target )/GeV << G4endl;
745 #endif
746
747 if ( ! isOk ) return false;
748 } while ( ( SqrtS < std::sqrt( M2proj ) + std::sqrt( M2target ) ) &&
749 NumberOfTries < maxNumberOfInnerLoops ); /* Loop checking, 10.08.2015, A.Ribon */
750 if ( NumberOfTries >= maxNumberOfInnerLoops ) {
751 #ifdef debugPutOnMassShell
752 G4cout << "BAD situation: forced exit of the inner while loop!" << G4endl;
753 #endif
754 return false;
755 }
756 if ( isProjectileNucleus ) {
757 isOk = CheckKinematics( S, SqrtS, M2proj, M2target, YprojectileNucleus, true,
758 NumberOfInvolvedNucleonsOfProjectile,
759 TheInvolvedNucleonsOfProjectile,
760 WminusTarget, WplusProjectile, OuterSuccess );
761 }
762 isOk = isOk &&
763 CheckKinematics( S, SqrtS, M2proj, M2target, YtargetNucleus, false,
764 NumberOfInvolvedNucleonsOfTarget, TheInvolvedNucleonsOfTarget,
765 WminusTarget, WplusProjectile, OuterSuccess );
766 if ( ! isOk ) return false;
767 } while ( ( ! OuterSuccess ) &&
768 ++loopCounter < maxNumberOfLoops ); /* Loop checking, 10.08.2015, A.Ribon */
769 if ( loopCounter >= maxNumberOfLoops ) {
770 #ifdef debugPutOnMassShell
771 G4cout << "BAD situation: forced exit of the while loop!" << G4endl;
772 #endif
773 return false;
774 }
775
776 // Now the sampling is completed, and we can determine the kinematics of the
777 // whole system. This is done first in the center-of-mass frame, and then it is boosted
778 // to the lab frame. The transverse momentum of the residual nucleus is determined as
779 // the recoil of each hadron (nucleon or delta) which is emitted, i.e. in such a way
780 // to conserve (by construction) the transverse momentum.
781
782 if ( ! isProjectileNucleus ) { // hadron-nucleus collision
783
784 G4double Pzprojectile = WplusProjectile/2.0 - M2projectile/2.0/WplusProjectile;
785 G4double Eprojectile = WplusProjectile/2.0 + M2projectile/2.0/WplusProjectile;
786 Pprojectile.setPz( Pzprojectile );
787 Pprojectile.setE( Eprojectile );
788
789 #ifdef debugPutOnMassShell
790 G4cout << "Proj after in CMS " << Pprojectile << G4endl;
791 #endif
792
793 Pprojectile.transform( toLab );
794 theProjectile.SetMomentum( Pprojectile.vect() );
795 theProjectile.SetTotalEnergy( Pprojectile.e() );
796
797 theParticipants.StartLoop();
798 theParticipants.Next();
799 G4VSplitableHadron* primary = theParticipants.GetInteraction().GetProjectile();
800 primary->Set4Momentum( Pprojectile );
801
802 #ifdef debugPutOnMassShell
803 G4cout << "Final proj. mom in Lab. " << primary->Get4Momentum() << G4endl;
804 #endif
805
806 } else { // nucleus-nucleus or antinucleus-nucleus collision
807
808 isOk = FinalizeKinematics( WplusProjectile, true, toLab, PrResidualMass,
809 ProjectileResidualMassNumber, NumberOfInvolvedNucleonsOfProjectile,
810 TheInvolvedNucleonsOfProjectile, ProjectileResidual4Momentum );
811
812 #ifdef debugPutOnMassShell
813 G4cout << "Projectile Residual4Momentum in CMS " << ProjectileResidual4Momentum << G4endl;
814 #endif
815
816 if ( ! isOk ) return false;
817
818 ProjectileResidual4Momentum.transform( toLab );
819
820 #ifdef debugPutOnMassShell
821 G4cout << "Projectile Residual4Momentum in Lab " << ProjectileResidual4Momentum << G4endl;
822 #endif
823
824 }
825
826 isOk = FinalizeKinematics( WminusTarget, false, toLab, TargetResidualMass,
827 TargetResidualMassNumber, NumberOfInvolvedNucleonsOfTarget,
828 TheInvolvedNucleonsOfTarget, TargetResidual4Momentum );
829
830 #ifdef debugPutOnMassShell
831 G4cout << "Target Residual4Momentum in CMS " << TargetResidual4Momentum << G4endl;
832 #endif
833
834 if ( ! isOk ) return false;
835
836 TargetResidual4Momentum.transform( toLab );
837
838 #ifdef debugPutOnMassShell
839 G4cout << "Target Residual4Momentum in Lab " << TargetResidual4Momentum << G4endl;
840 #endif
841
842 return true;
843
844}

1152
1153G4int G4FTFModel::AdjustNucleonsAlgorithm_beforeSampling( G4int interactionCase,
1154 G4VSplitableHadron* SelectedAntiBaryon,
1155 G4Nucleon* ProjectileNucleon,
1156 G4VSplitableHadron* SelectedTargetNucleon,
1157 G4Nucleon* TargetNucleon,
1158 G4bool Annihilation,
1159 G4FTFModel::CommonVariables& common ) {
1160 // First of the three utility methods used only by AdjustNucleons: prepare for sampling.
1161 // This method returns an integer code - instead of a boolean, with the following meaning:
1162 // "0" : successfully ended and nothing else needs to be done (i.e. no sampling);
1163 // "1" : successfully completed, but the work needs to be continued, i.e. try to sample;
1164 // "99" : unsuccessfully ended, nothing else can be done.
1165 G4int returnCode = 99;
1166
1167 G4double ExcitationEnergyPerWoundedNucleon = theParameters->GetExcitationEnergyPerWoundedNucleon();
1168
1169 // some checks and initializations
1170 if ( interactionCase == 1 ) {
1171 common.Psum = SelectedAntiBaryon->Get4Momentum() + TargetResidual4Momentum;
1172 #ifdef debugAdjust
1173 G4cout << "Targ res Init " << TargetResidual4Momentum << G4endl;
1174 #endif
1175 common.Pprojectile = SelectedAntiBaryon->Get4Momentum();
1176 } else if ( interactionCase == 2 ) {
1177 common.Psum = ProjectileResidual4Momentum + SelectedTargetNucleon->Get4Momentum();
1178 common.Pprojectile = ProjectileResidual4Momentum;
1179 } else if ( interactionCase == 3 ) {
1180 common.Psum = ProjectileResidual4Momentum + TargetResidual4Momentum;
1181 common.Pprojectile = ProjectileResidual4Momentum;
1182 }
1183
1184 // transform momenta to cms and then rotate parallel to z axis
1185 common.toCms = G4LorentzRotation( -1*common.Psum.boostVector() );
1186 common.Ptmp = common.toCms * common.Pprojectile;
1187 common.toCms.rotateZ( -1*common.Ptmp.phi() );

1672 / common.XplusResidual;

1674 / common.XminusNucleon

1676 / common.XminusResidual;

1747 - common.Mt2projectileNucleon
1748 / ( 2.0 * common.WplusProjectile * common.XplusNucleon );

1750 + common.Mt2projectileNucleon
1751 / ( 2.0 * common.WplusProjectile * common.XplusNucleon );
1752 YprojectileNucleon = 0.5 * G4Log( ( common.EprojectileNucleon + common.PzprojectileNucleon )
1753 / ( common.EprojectileNucleon - common.PzprojectileNucleon) );

1764 }

1774 + common.Mt2projectileNucleon
1775 / ( 2.0 * common.WplusProjectile * common.XplusNucleon );

1777 / ( common.EprojectileNucleon - common.PzprojectileNucleon ) );

1780 + common.Mt2targetNucleon
1781 / ( 2.0 * common.WminusTarget * common.XminusNucleon );

1783 + common.Mt2targetNucleon
1784 / ( 2.0 * common.WminusTarget * common.XminusNucleon );

1788 std::abs( YprojectileNucleon - common.YprojectileNucleus ) > 2 ||
1789 YprojectileNucleon < YtargetNucleon ) {
1790 OuterSuccess = false;
1791 continue;
1792 }
1793 } // End-if on interactionCase
1794
1795 } while ( ( ! OuterSuccess ) && // Outmost do while loop
1796 ++loopCounter < maxNumberOfLoops ); /* Loop checking, 10.08.2015, A.Ribon */
1797 if ( loopCounter >= maxNumberOfLoops ) {
1798 #ifdef debugAdjust
1799 G4cout << "BAD situation: forced exit of the while loop!" << G4endl;
1800 #endif
1801 return false;
1802 }

2678G4bool G4FTFModel::
2679GenerateDeltaIsobar( const G4double sqrtS, // input parameter
2680 const G4int numberOfInvolvedNucleons, // input parameter
2681 G4Nucleon* involvedNucleons[], // input & output parameter
2682 G4double& sumMasses ) { // input & output parameter
2683
2684 // This method, which is called only by PutOnMassShell, check whether is possible to
2685 // re-interpret some of the involved nucleons as delta-isobars:
2686 // - either by replacing a proton (2212) with a Delta+ (2214),
2687 // - or by replacing a neutron (2112) with a Delta0 (2114).
2688 // The on-shell mass of these delta-isobars is ~1232 MeV, so ~292-294 MeV heavier than
2689 // the corresponding nucleon on-shell mass. However 400.0*MeV is considered to estimate
2690 // the max number of deltas compatible with the available energy.
2691 // The delta-isobars are considered with the same transverse momentum as their
2692 // corresponding nucleons.
2693 // This method assumes that all the parameters have been initialized by the caller;
2694 // the action of this method consists in modifying (eventually) involveNucleons and
2695 // sumMasses. The return value is "false" only in the case that the input parameters
2696 // have unphysical values.
2697
2698 if ( sqrtS < 0.0 || numberOfInvolvedNucleons <= 0 || sumMasses < 0.0 ) return false;
2699
2700 const G4double probDeltaIsobar = 0.05;
2701
2702 G4int maxNumberOfDeltas = G4int( (sqrtS - sumMasses)/(400.0*MeV) );
2703 G4int numberOfDeltas = 0;
2704
2705 for ( G4int i = 0; i < numberOfInvolvedNucleons; i++ ) {
2706 //G4cout << "i maxNumberOfDeltas probDeltaIsobar " << i << " " << maxNumberOfDeltas
2707 // << " " << probDeltaIsobar << G4endl;
2708 if ( G4UniformRand() < probDeltaIsobar && numberOfDeltas < maxNumberOfDeltas ) {
2709 numberOfDeltas++;
2710 if ( ! involvedNucleons[i] ) continue;
2711 G4VSplitableHadron* splitableHadron = involvedNucleons[i]->GetSplitableHadron();
2712 G4double massNuc = std::sqrt( sqr( splitableHadron->GetDefinition()->GetPDGMass() )
2713 + splitableHadron->Get4Momentum().perp2() );
2714 // The absolute value below is needed in the case of an antinucleus.
2715 G4int pdgCode = std::abs( splitableHadron->GetDefinition()->GetPDGEncoding() );
2716 const G4ParticleDefinition* old_def = splitableHadron->GetDefinition();
2717 G4int newPdgCode = pdgCode/10; newPdgCode = newPdgCode*10 + 4; // Delta
2718 if ( splitableHadron->GetDefinition()->GetPDGEncoding() < 0 ) newPdgCode *= -1;
2719 const G4ParticleDefinition* ptr =
2720 G4ParticleTable::GetParticleTable()->FindParticle( newPdgCode );
2721 splitableHadron->SetDefinition( ptr );
2722 G4double massDelta = std::sqrt( sqr( splitableHadron->GetDefinition()->GetPDGMass() )
2723 + splitableHadron->Get4Momentum().perp2() );
2724 //G4cout << i << " " << sqrtS/GeV << " " << sumMasses/GeV << " " << massDelta/GeV
2725 // << " " << massNuc << G4endl;
2726 if ( sqrtS < sumMasses + massDelta - massNuc ) { // Change cannot be accepted!
2727 splitableHadron->SetDefinition( old_def );
2728 break;
2729 } else { // Change is accepted
2730 sumMasses += ( massDelta - massNuc );
2731 }
2732 }
2733 }
2734 //G4cout << "maxNumberOfDeltas numberOfDeltas " << maxNumberOfDeltas << " "
2735 // << numberOfDeltas << G4endl;
2736 return true;
2737}

2742G4bool G4FTFModel::
2743SamplingNucleonKinematics( G4double averagePt2, // input parameter
2744 const G4double maxPt2, // input parameter
2745 G4double dCor, // input parameter
2746 G4V3DNucleus* nucleus, // input parameter
2747 const G4LorentzVector& pResidual, // input parameter
2748 const G4double residualMass, // input parameter
2749 const G4int residualMassNumber, // input parameter
2750 const G4int numberOfInvolvedNucleons, // input parameter
2751 G4Nucleon* involvedNucleons[], // input & output parameter
2752 G4double& mass2 ) { // output parameter
2753
2754 // This method, which is called only by PutOnMassShell, does the sampling of:
2755 // - either the target nucleons: this for any kind of hadronic interactions
2756 // (hadron-nucleus, nucleus-nucleus, antinucleus-nucleus);
2757 // - or the projectile nucleons or antinucleons: this only in the case of
2758 // nucleus-nucleus or antinucleus-nucleus interactions, respectively.
2759 // This method assumes that all the parameters have been initialized by the caller;
2760 // the action of this method consists in changing the properties of the nucleons
2761 // whose pointers are in the vector involvedNucleons, as well as changing the
2762 // variable mass2.
2763
2764 if ( ! nucleus ) return false;
2765
2766 if ( residualMassNumber == 0 && numberOfInvolvedNucleons == 1 ) {
2767 dCor = 0.0;
2768 averagePt2 = 0.0;
2769 }
2770
2771 G4bool success = true;
2772
2773 G4double SumMasses = residualMass;
2774
2775 for ( G4int i = 0; i < numberOfInvolvedNucleons; i++ ) {
2776 G4Nucleon* aNucleon = involvedNucleons[i];
2777 if ( ! aNucleon ) continue;
2778 SumMasses += aNucleon->GetSplitableHadron()->GetDefinition()->GetPDGMass();
2779 }
2780 //
2781
2782 const G4int maxNumberOfLoops = 1000;
2783 G4int loopCounter = 0;
2784 do {
2785
2786 success = true;
2787
2788 // Sampling of nucleon Pt
2789 G4ThreeVector ptSum( 0.0, 0.0, 0.0 );
2790
2791 for ( G4int i = 0; i < numberOfInvolvedNucleons; i++ ) {
2792 G4Nucleon* aNucleon = involvedNucleons[i];
2793 if ( ! aNucleon ) continue;
2794 G4ThreeVector tmpPt = GaussianPt( averagePt2, maxPt2 );
2795 ptSum += tmpPt;
2796 G4LorentzVector tmp( tmpPt.x(), tmpPt.y(), 0.0, 0.0 );
2797 aNucleon->SetMomentum( tmp );
2798 }
2799
2800 G4double deltaPx = ( ptSum.x() - pResidual.x() ) / numberOfInvolvedNucleons;
2801 G4double deltaPy = ( ptSum.y() - pResidual.y() ) / numberOfInvolvedNucleons;
2802
2803 SumMasses = residualMass;
2804 for ( G4int i = 0; i < numberOfInvolvedNucleons; i++ ) {
2805 G4Nucleon* aNucleon = involvedNucleons[i];
2806 if ( ! aNucleon ) continue;
2807 G4double px = aNucleon->Get4Momentum().px() - deltaPx;
2808 G4double py = aNucleon->Get4Momentum().py() - deltaPy;
2809 G4double MtN = std::sqrt( sqr( aNucleon->GetSplitableHadron()->GetDefinition()->GetPDGMass() )
2810 + sqr( px ) + sqr( py ) );
2811 SumMasses += MtN;
2812 G4LorentzVector tmp( px, py, 0.0, MtN);
2813 aNucleon->SetMomentum( tmp );
2814 }
2815
2816 // Sampling X of nucleon
2817 G4double xSum = 0.0;
2818
2819 for ( G4int i = 0; i < numberOfInvolvedNucleons; i++ ) {
2820 G4Nucleon* aNucleon = involvedNucleons[i];
2821 if ( ! aNucleon ) continue;
2822
2823 G4ThreeVector tmpX = GaussianPt( dCor*dCor, 1.0 );
2824 //G4double x = tmpX.x() + aNucleon->GetSplitableHadron()->GetDefinition()->GetPDGMass()/SumMasses;
2825 G4double x = tmpX.x() + aNucleon->Get4Momentum().e()/SumMasses;
2826 if ( x < 0.0 || x > 1.0 ) {
2827 success = false;
2828 break;
2829 }
2830 xSum += x;
2831 // The energy is in the lab (instead of cms) frame but it will not be used.
2832
2833 G4LorentzVector tmp( aNucleon->Get4Momentum().x(), aNucleon->Get4Momentum().y(),
2834 x, aNucleon->Get4Momentum().e() );
2835 aNucleon->SetMomentum( tmp );
2836 }
2837
2838 if ( xSum < 0.0 || xSum > 1.0 ) success = false;
2839
2840 if ( ! success ) continue;
2841
2842 //G4double deltaPx = ( ptSum.x() - pResidual.x() ) / numberOfInvolvedNucleons;
2843 //G4double deltaPy = ( ptSum.y() - pResidual.y() ) / numberOfInvolvedNucleons;
2844 G4double delta = 0.0;
2845 if ( residualMassNumber == 0 ) {
2846 delta = ( xSum - 1.0 ) / numberOfInvolvedNucleons;
2847 } else {
2848 delta = 0.0;
2849 }
2850
2851 xSum = 1.0;
2852 mass2 = 0.0;
2853 for ( G4int i = 0; i < numberOfInvolvedNucleons; i++ ) {
2854 G4Nucleon* aNucleon = involvedNucleons[i];
2855 if ( ! aNucleon ) continue;
2856 G4double x = aNucleon->Get4Momentum().pz() - delta;
2857 xSum -= x;
2858 if ( residualMassNumber == 0 ) {
2859 if ( x <= 0.0 || x > 1.0 ) {
2860 success = false;
2861 break;
2862 }
2863 } else {
2864 if ( x <= 0.0 || x > 1.0 || xSum <= 0.0 || xSum > 1.0 ) {
2865 success = false;
2866 break;
2867 }
2868 }
2869
2870 /*
2871 G4double px = aNucleon->Get4Momentum().px() - deltaPx;
2872 G4double py = aNucleon->Get4Momentum().py() - deltaPy;
2873 mass2 += ( sqr( aNucleon->GetSplitableHadron()->GetDefinition()->GetPDGMass() )
2874 + sqr( px ) + sqr( py ) ) / x;
2875 G4LorentzVector tmp( px, py, x, aNucleon->Get4Momentum().e() );
2876 */
2877
2878 mass2 += sqr( aNucleon->Get4Momentum().e() ) / x;
2879 G4LorentzVector tmp( aNucleon->Get4Momentum().px(), aNucleon->Get4Momentum().py(),
2880 x, aNucleon->Get4Momentum().e() );
2881 aNucleon->SetMomentum( tmp );
2882 }
2883 if ( ! success ) continue;
2884
2885 if ( success && residualMassNumber != 0 ) {
2886 mass2 += ( sqr( residualMass ) + pResidual.perp2() ) / xSum;
2887 //mass2 += sqr( residualMass ) / xSum;
2888 }
2889
2890 #ifdef debugPutOnMassShell
2891 G4cout << "success " << success << G4endl << " Mt " << std::sqrt( mass2 )/GeV << G4endl;
2892 #endif
2893
2894 } while ( ( ! success ) &&
2895 ++loopCounter < maxNumberOfLoops ); /* Loop checking, 10.08.2015, A.Ribon */
2896 if ( loopCounter >= maxNumberOfLoops ) {
2897 return false;
2898 }
2899
2900 return true;
2901}

2906G4bool G4FTFModel::
2907CheckKinematics( const G4double sValue, // input parameter
2908 const G4double sqrtS, // input parameter
2909 const G4double projectileMass2, // input parameter
2910 const G4double targetMass2, // input parameter
2911 const G4double nucleusY, // input parameter
2912 const G4bool isProjectileNucleus, // input parameter
2913 const G4int numberOfInvolvedNucleons, // input parameter
2914 G4Nucleon* involvedNucleons[], // input parameter
2915 G4double& targetWminus, // output parameter
2916 G4double& projectileWplus, // output parameter
2917 G4bool& success ) { // input & output parameter
2918
2919 // This method, which is called only by PutOnMassShell, checks whether the
2920 // kinematics is acceptable or not.
2921 // This method assumes that all the parameters have been initialized by the caller;
2922 // notice that the input boolean parameter isProjectileNucleus is meant to be true
2923 // only in the case of nucleus or antinucleus projectile.
2924 // The action of this method consists in computing targetWminus and projectileWplus
2925 // and setting the parameter success to false in the case that the kinematics should
2926 // be rejeted.
2927
2928 G4double decayMomentum2 = sqr( sValue ) + sqr( projectileMass2 ) + sqr( targetMass2 )
2929 - 2.0*( sValue*( projectileMass2 + targetMass2 )
2930 + projectileMass2*targetMass2 );
2931 targetWminus = ( sValue - projectileMass2 + targetMass2 + std::sqrt( decayMomentum2 ) )
2932 / 2.0 / sqrtS;
2933 projectileWplus = sqrtS - targetMass2/targetWminus;
2934 G4double projectilePz = projectileWplus/2.0 - projectileMass2/2.0/projectileWplus;
2935 G4double projectileE = projectileWplus/2.0 + projectileMass2/2.0/projectileWplus;
2936 G4double projectileY = 0.5 * G4Log( (projectileE + projectilePz)/
2937 (projectileE - projectilePz) );
2938 G4double targetPz = -targetWminus/2.0 + targetMass2/2.0/targetWminus;
2939 G4double targetE = targetWminus/2.0 + targetMass2/2.0/targetWminus;
2940 G4double targetY = 0.5 * G4Log( (targetE + targetPz)/(targetE - targetPz) );
2941
2942 #ifdef debugPutOnMassShell
2943 G4cout << "decayMomentum2 " << decayMomentum2 << G4endl
2944 << "\t targetWminus projectileWplus " << targetWminus << " " << projectileWplus << G4endl
2945 << "\t projectileY targetY " << projectileY << " " << targetY << G4endl;
2946 #endif
2947
2948 for ( G4int i = 0; i < numberOfInvolvedNucleons; i++ ) {
2949 G4Nucleon* aNucleon = involvedNucleons[i];
2950 if ( ! aNucleon ) continue;
2951 G4LorentzVector tmp = aNucleon->Get4Momentum();
2952 G4double mt2 = sqr( tmp.x() ) + sqr( tmp.y() ) +
2953 sqr( aNucleon->GetSplitableHadron()->GetDefinition()->GetPDGMass() );
2954 G4double x = tmp.z();
2955 G4double pz = -targetWminus*x/2.0 + mt2/(2.0*targetWminus*x);
2956 G4double e = targetWminus*x/2.0 + mt2/(2.0*targetWminus*x);
2957 if ( isProjectileNucleus ) {
2958 pz = projectileWplus*x/2.0 - mt2/(2.0*projectileWplus*x);
2959 e = projectileWplus*x/2.0 + mt2/(2.0*projectileWplus*x);
2960 }
2961 G4double nucleonY = 0.5 * G4Log( (e + pz)/(e - pz) );
2962
2963 #ifdef debugPutOnMassShell
2964 G4cout << "i nY pY nY-AY AY " << i << " " << nucleonY << " " << projectileY <<G4endl;
2965 #endif
2966
2967 if ( std::abs( nucleonY - nucleusY ) > 2 ||
2968 ( isProjectileNucleus && targetY > nucleonY ) ||
2969 ( ! isProjectileNucleus && projectileY < nucleonY ) ) {
2970 success = false;
2971 break;
2972 }
2973 }
2974 return true;
2975}

2980G4bool G4FTFModel::
2981FinalizeKinematics( const G4double w, // input parameter
2982 const G4bool isProjectileNucleus, // input parameter
2983 const G4LorentzRotation& boostFromCmsToLab, // input parameter
2984 const G4double residualMass, // input parameter
2985 const G4int residualMassNumber, // input parameter
2986 const G4int numberOfInvolvedNucleons, // input parameter
2987 G4Nucleon* involvedNucleons[], // input & output parameter
2988 G4LorentzVector& residual4Momentum ) { // output parameter
2989
2990 // This method, which is called only by PutOnMassShell, finalizes the kinematics:
2991 // this method is called when we are sure that the sampling of the kinematics is
2992 // acceptable.
2993 // This method assumes that all the parameters have been initialized by the caller;
2994 // notice that the input boolean parameter isProjectileNucleus is meant to be true
2995 // only in the case of nucleus or antinucleus projectile: this information is needed
2996 // because the sign of pz (in the center-of-mass frame) in this case is opposite
2997 // with respect to the case of a normal hadron projectile.
2998 // The action of this method consists in modifying the momenta of the nucleons
2999 // (in the lab frame) and computing the residual 4-momentum (in the center-of-mass
3000 // frame).
3001
3002 G4ThreeVector residual3Momentum( 0.0, 0.0, 1.0 );
3003
3004 for ( G4int i = 0; i < numberOfInvolvedNucleons; i++ ) {
3005 G4Nucleon* aNucleon = involvedNucleons[i];
3006 if ( ! aNucleon ) continue;
3007 G4LorentzVector tmp = aNucleon->Get4Momentum();
3008 residual3Momentum -= tmp.vect();
3009 G4double mt2 = sqr( tmp.x() ) + sqr( tmp.y() ) +
3010 sqr( aNucleon->GetSplitableHadron()->GetDefinition()->GetPDGMass() );
3011 G4double x = tmp.z();
3012 G4double pz = -w * x / 2.0 + mt2 / ( 2.0 * w * x );
3013 G4double e = w * x / 2.0 + mt2 / ( 2.0 * w * x );
3014 // Reverse the sign of pz in the case of nucleus or antinucleus projectile
3015 if ( isProjectileNucleus ) pz *= -1.0;
3016 tmp.setPz( pz );
3017 tmp.setE( e );
3018 tmp.transform( boostFromCmsToLab );
3019 aNucleon->SetMomentum( tmp );
3020 G4VSplitableHadron* splitableHadron = aNucleon->GetSplitableHadron();
3021 splitableHadron->Set4Momentum( tmp );
3022 }
3023
3024 G4double residualMt2 = sqr( residualMass ) + sqr( residual3Momentum.x() )
3025 + sqr( residual3Momentum.y() );
3026
3027 #ifdef debugPutOnMassShell
3028 G4cout << "w residual3Momentum.z() " << w << " " << residual3Momentum.z() << G4endl;
3029 #endif
3030
3031 G4double residualPz = 0.0;
3032 G4double residualE = 0.0;
3033 if ( residualMassNumber != 0 ) {
3034 residualPz = -w * residual3Momentum.z() / 2.0 +
3035 residualMt2 / ( 2.0 * w * residual3Momentum.z() );
3036 residualE = w * residual3Momentum.z() / 2.0 +
3037 residualMt2 / ( 2.0 * w * residual3Momentum.z() );
3038 // Reverse the sign of residualPz in the case of nucleus or antinucleus projectile
3039 if ( isProjectileNucleus ) residualPz *= -1.0;
3040 }
3041
3042 residual4Momentum.setPx( residual3Momentum.x() );
3043 residual4Momentum.setPy( residual3Momentum.y() );
3044 residual4Momentum.setPz( residualPz );
3045 residual4Momentum.setE( residualE );
3046
3047 return true;
3048}
