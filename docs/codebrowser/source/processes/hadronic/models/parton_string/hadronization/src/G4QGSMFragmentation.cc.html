49G4QGSMFragmentation::G4QGSMFragmentation() :
50arho(0.5), aphi(0.), an(-0.5), ala(-0.75), aksi(-1.), alft(0.5)
51{
52 SetStrangenessSuppression((1.0 - 0.16)/2.);
53 SetDiquarkSuppression(0.299);
54 SetDiquarkBreakProbability(0.7);
55
56 //... pspin_meson is probability to create pseudo-scalar meson
57 pspin_meson = 0.25; SetVectorMesonProbability(pspin_meson);
58
59 //... pspin_barion is probability to create 1/2 barion
60 pspin_barion = 0.5; SetSpinThreeHalfBarionProbability(pspin_barion);
61
62 //... vectorMesonMix[] is quark mixing parameters for vector mesons (Variable spin = 3)
63 vectorMesonMix[0] = 0.; //0.5
64 vectorMesonMix[1] = 0.375; //0.5;
65 vectorMesonMix[2] = 0.0;
66 vectorMesonMix[3] = 0.375; //0.5;
67 vectorMesonMix[4] = 1.0;
68 vectorMesonMix[5] = 1.0;
69
70 SetVectorMesonMixings(vectorMesonMix);
71 SetMinMasses();
72}

79G4KineticTrackVector* G4QGSMFragmentation::FragmentString(const G4ExcitedString& theString)
80{
81 #ifdef debug_QGSMfragmentation
82 G4cout<<G4endl<<"QGSM StringFragm: String Mass "
83 <<theString.Get4Momentum().mag()<<" Pz "
84 <<theString.Get4Momentum().pz()
85 <<"------------------------------------"<<G4endl;
86 G4cout<<"String ends Direct "<<theString.GetLeftParton()->GetPDGcode()<<" "
87 <<theString.GetRightParton()->GetPDGcode()<<" "
88 <<theString.GetDirection()<< G4endl;
89 G4cout<<"Left mom "<<theString.GetLeftParton()->Get4Momentum()<<G4endl;
90 G4cout<<"Right mom "<<theString.GetRightParton()->Get4Momentum()<<G4endl;
91 G4cout<<"Check for Fragmentation "<<G4endl;
92 #endif
93
94 // Can no longer modify Parameters for Fragmentation.
95 PastInitPhase=true;
96
97 // Check if string has enough mass to fragment...
98 G4KineticTrackVector * LeftVector=LightFragmentationTest(&theString);
99
100 #ifdef debug_QGSMfragmentation
101 if ( LeftVector != 0 ) G4cout<<"Non fragmentable - the string is converted to one hadron "<<G4endl;
102 #endif
103
104 if ( LeftVector != 0 ) return LeftVector;
105
106 #ifdef debug_QGSMfragmentation
107 G4cout<<"The string will be fragmented. "<<G4endl;
108 #endif
109
110 LeftVector = new G4KineticTrackVector;
111 G4KineticTrackVector * RightVector=new G4KineticTrackVector;
112
113 G4ExcitedString *theStringInCMS=CopyExcited(theString);
114 G4LorentzRotation toCms=theStringInCMS->TransformToAlignedCms();
115
116 G4bool success=false, inner_sucess=true;
117 G4int attempt=0;
118 while ( !success && attempt++ < StringLoopInterrupt ) /* Loop checking, 07.08.2015, A.Ribon */
119 {
120 #ifdef debug_QGSMfragmentation
121 G4cout<<"Loop_toFrag "<<theStringInCMS->GetLeftParton()->GetPDGcode()<<" "
122 <<theStringInCMS->GetRightParton()->GetPDGcode()<<" "
123 <<theStringInCMS->GetDirection()<< G4endl;
124 #endif
125
126 G4FragmentingString *currentString=new G4FragmentingString(*theStringInCMS);
127
128 std::for_each(LeftVector->begin(), LeftVector->end(), DeleteKineticTrack());
129 LeftVector->clear();
130 std::for_each(RightVector->begin(), RightVector->end(), DeleteKineticTrack());
131 RightVector->clear();
132
133 inner_sucess=true; // set false on failure..
134 const G4int maxNumberOfLoops = 1000;
135 G4int loopCounter = -1;
136 while (! StopFragmenting(currentString) && ++loopCounter < maxNumberOfLoops ) /* Loop checking, 07.08.2015, A.Ribon */
137 { // Split current string into hadron + new string
138
139 #ifdef debug_QGSMfragmentation
140 G4cout<<"The string can fragment. "<<G4endl;;
141 #endif
142 G4FragmentingString *newString=0; // used as output from SplitUp...
143 G4KineticTrack * Hadron=Splitup(currentString,newString);
144
145 if ( Hadron != 0 )
146 {
147 #ifdef debug_QGSMfragmentation
148 G4cout<<"Hadron prod at fragm. "<<Hadron->GetDefinition()->GetParticleName()<<G4endl;
149 #endif
150 // To close the production of hadrons at fragmentation stage
151 if ( currentString->GetDecayDirection() > 0 )
152 LeftVector->push_back(Hadron);
153 else
154 RightVector->push_back(Hadron);
155
156 delete currentString;
157 currentString=newString;
158
159 } else {
160
161 #ifdef debug_QGSMfragmentation
162 G4cout<<"abandon ... start from the beginning ---------------"<<G4endl;
163 #endif
164
165 // Abandon ... start from the beginning
166 if (newString) delete newString;
167 inner_sucess=false;
168 break;
169 }
170 }
171 if ( loopCounter >= maxNumberOfLoops ) {
172 inner_sucess=false;
173 }
174
175 // Split current string into 2 final Hadrons
176 #ifdef debug_QGSMfragmentation
177 G4cout<<"Split remaining string into 2 final hadrons."<<G4endl;
178 #endif
179 // To the close production of hadrons at last string decay
180 if ( inner_sucess &&
181 SplitLast(currentString,LeftVector, RightVector) )
182 {
183 success=true;
184 }
185 delete currentString;
186 }
187
188 delete theStringInCMS;
189
190 if ( ! success )
191 {
192 std::for_each(LeftVector->begin(), LeftVector->end(), DeleteKineticTrack());
193 LeftVector->clear();
194 std::for_each(RightVector->begin(), RightVector->end(), DeleteKineticTrack());
195 delete RightVector;
196 return LeftVector;
197 }
198
199 // Join Left- and RightVector into LeftVector in correct order.
200 while(!RightVector->empty()) /* Loop checking, 07.08.2015, A.Ribon */
201 {
202 LeftVector->push_back(RightVector->back());
203 RightVector->erase(RightVector->end()-1);
204 }
205 delete RightVector;
206
207 CalculateHadronTimePosition(theString.Get4Momentum().mag(), LeftVector);
208
209 G4LorentzRotation toObserverFrame(toCms.inverse());
210
211 for(size_t C1 = 0; C1 < LeftVector->size(); C1++)
212 {
213 G4KineticTrack* Hadron = LeftVector->operator[](C1);
214 G4LorentzVector Momentum = Hadron->Get4Momentum();
215 Momentum = toObserverFrame*Momentum;
216 Hadron->Set4Momentum(Momentum);
217 G4LorentzVector Coordinate(Hadron->GetPosition(), Hadron->GetFormationTime());
218 Momentum = toObserverFrame*Coordinate;
219 Hadron->SetFormationTime(Momentum.e());
220 G4ThreeVector aPosition(Momentum.vect());
221 Hadron->SetPosition(theString.GetPosition()+aPosition);
222 }
223 return LeftVector;
224}

228G4double G4QGSMFragmentation::GetLightConeZ(G4double zmin, G4double zmax, G4int PartonEncoding,
229 G4ParticleDefinition* pHadron, G4double , G4double )
230{
231 #ifdef debug_QGSMfragmentation
232 G4cout<<"GetLightConeZ zmin zmax Parton pHadron "<<zmin<<" "<<zmax<<" "<<PartonEncoding<<" "<<pHadron->GetParticleName()<<G4endl;
233 #endif
234
235 G4double z;
236 G4double d1, d2, yf;
237 G4double invD1(0.),invD2(0.), r1(0.),r2(0.),r12(0.);
238
239 G4int absCode = std::abs( PartonEncoding );
240 G4int absHadronCode=std::abs(pHadron->GetPDGEncoding());
241
242 G4int q1, q2, q3;
243 q1 = absHadronCode/1000; q2 = (absHadronCode % 1000)/100; q3 = (absHadronCode % 100)/10;
244
245 G4bool StrangeHadron = (q1 == 3) || (q2 == 3) || (q3 == 3);
246
247 if (absCode < 10)
248 { // A quark fragmentation ----------------------------
249 if (absCode == 1 || absCode == 2)
250 {
251 if (absHadronCode < 1000)
252 { // Meson produced
253 if ( !StrangeHadron ) {d1=2.0; d2 = -arho + alft;} // d1=2.0;
254 else {d1=1.0; d2 = -aphi + alft;} // d1=1.0;
255 }
256 else
257 { // Baryon produced
258 if ( !StrangeHadron ) {d1=0.0; d2 = arho - 2.0*an + alft;}
259 else {d1=0.0; d2 = 2.0*arho - 2.0*an - aphi + alft;}
260 }
261 }
262 else if (absCode == 3)
263 {
264 if (absHadronCode < 1000) {d1=1.0 - aphi; d2 = -arho + alft;} // Meson produced s->K + u/d d1=1.0
265 else {d1=1.0 - aphi; d2 = arho - 2.0*an + alft;} // Baryon produced
266
267 } else throw G4HadronicException(__FILE__, __LINE__, "Unknown PDGencoding in G4QGSMFragmentation::G4LightConeZ");
268
269 #ifdef debug_QGSMfragmentation
270 G4cout<<"d1 d2 "<<d1<<" "<<d2<<G4endl;
271 #endif
272
273 d1+=1.0; d2+=1.0;
274
275 invD1=1./d1; invD2=1./d2;
276
277 const G4int maxNumberOfLoops = 10000;
278 G4int loopCounter = 0;
279 do
280 {
281 r1=G4Pow::GetInstance()->powA(G4UniformRand(),invD1);
282 r2=G4Pow::GetInstance()->powA(G4UniformRand(),invD2);
283 r12=r1+r2;
284 z=r1/r12;
285 } while ( ( (r12 > 1.0) || !((zmin <= z)&&(z <= zmax))) && ++loopCounter < maxNumberOfLoops ); /* Loop checking, 07.08.2015, A.Ribon */
286 if ( loopCounter >= maxNumberOfLoops ) {
287 z = 0.5*(zmin + zmax); // Just a value between zmin and zmax, no physics considerations at all!
288 }
289
290 return z;
291 }
292 else
293 { // A di-quark fragmentation -------------------------
294 if (absCode == 1103 || absCode == 2101 ||
295 absCode == 2203 || absCode == 2103)
296 {
297 if(absHadronCode < 1000) // Meson production
298 {
299 if ( !StrangeHadron ) {d1=1.0; d2= arho - 2.0*an + alft;}
300 else {d1=0.0; d2 = 2.*arho - 2.0*an - aphi + alft;} // d1=1.0;
301 }
302 else // Baryon production
303 {
304 if ( !StrangeHadron ) {d1=2.0*(arho - an); d2= -arho + alft;}
305 else {d1=2.0*(arho - an); d2 =-aphi + alft;}
306 }
307
308 #ifdef debug_QGSMfragmentation
309 G4cout<<"d1 d2 "<<d1<<" "<<d2<<G4endl;
310 #endif
311
312 d1+=1.0; d2+=1.0;
313 invD1=1./d1; invD2=1./d2;
314
315 const G4int maxNumberOfLoops = 10000;
316 G4int loopCounter = 0;
317 do
318 {
319 r1=G4Pow::GetInstance()->powA(G4UniformRand(),invD1);
320 r2=G4Pow::GetInstance()->powA(G4UniformRand(),invD2);
321 r12=r1+r2;
322 z=r1/r12;
323 } while ( ( (r12 > 1.0) || !((zmin <= z)&&(z <= zmax))) && ++loopCounter < maxNumberOfLoops ); /* Loop checking, 07.08.2015, A.Ribon */
324 if ( loopCounter >= maxNumberOfLoops ) {
325 z = 0.5*(zmin + zmax); // Just a value between zmin and zmax, no physics considerations at all!
326 }
327
328 return z;
329 }
330 else if (absCode == 3101 || absCode == 3103 || // For strange d-quarks
331 absCode == 3201 || absCode == 3203)
332 {
333 // For future improvements
334 // if (absHadronCode < 1000) {d1=1.0;} // Meson production
335 // else {d1=2.0;} // Baryon production
336 d2 = (alft - (2.*ala - arho));
337 }
338 else
339 {
340 // if (absHadronCode < 1000) {d1=1.0;} // Meson production
341 // else {d1=2.0;} // Baryon production
342 d2 = (alft - (2.*aksi - arho));
343 }
344
345 const G4int maxNumberOfLoops = 1000;
346 G4int loopCounter = 0;
347 do
348 {
349 z = zmin + G4UniformRand() * (zmax - zmin);
350 d1 = (1. - z);
351 yf = G4Pow::GetInstance()->powA(d1, d2);
352 }
353 while ( (G4UniformRand() > yf) && ++loopCounter < maxNumberOfLoops ); // Loop checking, 07.08.2015, A.Ribon //
354 /* For future improvements
355 d1+=1.0; d2+=1.0;
356 invD1=1./d1; invD2=1./d2;
357 const G4int maxNumberOfLoops = 10000;
358 G4int loopCounter = 0;
359 do
360 {
361 r1=G4Pow::GetInstance()->powA(G4UniformRand(),invD1);
362 r2=G4Pow::GetInstance()->powA(G4UniformRand(),invD2);
363 r12=r1+r2;
364 z=r1/r12;
365 } while( ( (r12 > 1.0) || !((zmin <= z)&&(z <= zmax))) && ++loopCounter < maxNumberOfLoops );
366 */
367 if ( loopCounter >= maxNumberOfLoops ) {
368 z = 0.5*(zmin + zmax); // Just a value between zmin and zmax, no physics considerations at all!
369 }
370
371 return z;
372
373 }
374
375 return z;
376}

379G4LorentzVector * G4QGSMFragmentation::SplitEandP(G4ParticleDefinition * pHadron,
380 G4FragmentingString * string,
381 G4FragmentingString * NewString)
382{
383 G4double HadronMass = pHadron->GetPDGMass();
384
385 SetMinimalStringMass(NewString);
386
387 #ifdef debug_QGSMfragmentation
388 G4cout<<"G4QGSMFragmentation::SplitEandP "<<pHadron->GetParticleName()<<G4endl;
389 G4cout<<"String 4 mom, String M "<<string->Get4Momentum()<<" "<<string->Mass()<<G4endl;
390 G4cout<<"HadM MinimalStringMassLeft StringM hM+sM "<<HadronMass<<" "<<MinimalStringMass<<" "
391 <<string->Mass()<<" "<<HadronMass+MinimalStringMass<<G4endl;
392 #endif
393
394 if (HadronMass + MinimalStringMass > string->Mass())
395 {
396 #ifdef debug_QGSMfragmentation
397 G4cout<<"Mass of the string is not sufficient to produce the hadron!"<<G4endl;
398 #endif
399 return 0;
400 } // have to start all over!
401
402 // calculate and assign hadron transverse momentum component HadronPx andHadronPy
403 G4double StringMT2 = string->MassT2();
404 G4double StringMT = std::sqrt(StringMT2);
405
406 G4LorentzVector String4Momentum = string->Get4Momentum();
407 String4Momentum.setPz(0.);
408 G4ThreeVector StringPt = String4Momentum.vect();
409
410 G4ThreeVector HadronPt , RemSysPt;
411 G4double HadronMassT2, ResidualMassT2;
412
413 //... sample Pt of the hadron
414 G4int attempt=0;
415 do
416 {
417 attempt++; if (attempt > StringLoopInterrupt) return 0;
418
419 HadronPt =SampleQuarkPt() + string->DecayPt();
420 HadronPt.setZ(0);
421 RemSysPt = StringPt - HadronPt;
422
423 HadronMassT2 = sqr(HadronMass) + HadronPt.mag2();
424 ResidualMassT2=sqr(MinimalStringMass) + RemSysPt.mag2();
425
426 } while (std::sqrt(HadronMassT2) + std::sqrt(ResidualMassT2) > StringMT); /* Loop checking, 07.08.2015, A.Ribon */
427
428 //... sample z to define hadron longitudinal momentum and energy
429 //... but first check the available phase space
430
431 G4double Pz2 = (sqr(StringMT2 - HadronMassT2 - ResidualMassT2) -
432 4*HadronMassT2 * ResidualMassT2)/4./StringMT2;
433
434 if ( Pz2 < 0 ) {return 0;} // have to start all over!
435
436 //... then compute allowed z region z_min <= z <= z_max
437
438 G4double Pz = std::sqrt(Pz2);
439 G4double zMin = (std::sqrt(HadronMassT2+Pz2) - Pz)/std::sqrt(StringMT2);
440 G4double zMax = (std::sqrt(HadronMassT2+Pz2) + Pz)/std::sqrt(StringMT2);
441
442 if (zMin >= zMax) return 0; // have to start all over!
443
444 G4double z = GetLightConeZ(zMin, zMax,
445 string->GetDecayParton()->GetPDGEncoding(), pHadron,
446 HadronPt.x(), HadronPt.y());
447
448 //... now compute hadron longitudinal momentum and energy
449 // longitudinal hadron momentum component HadronPz
450
451 HadronPt.setZ( 0.5* string->GetDecayDirection() *
452 (z * string->LightConeDecay() -
453 HadronMassT2/(z * string->LightConeDecay())) );
454 G4double HadronE = 0.5* (z * string->LightConeDecay() +
455 HadronMassT2/(z * string->LightConeDecay()) );
456
457 G4LorentzVector * a4Momentum= new G4LorentzVector(HadronPt,HadronE);
458
459 #ifdef debug_QGSMfragmentation
460 G4cout<<"string->GetDecayDirection() string->LightConeDecay() "
461 <<string->GetDecayDirection()<<" "<<string->LightConeDecay()<<G4endl;
462 G4cout<<"HadronPt,HadronE "<<HadronPt<<" "<<HadronE<<G4endl;
463 G4cout<<"Out of QGSM SplitEandP "<<G4endl;
464 #endif
465
466 return a4Momentum;
467}

472G4bool G4QGSMFragmentation::SplitLast(G4FragmentingString * string,
473 G4KineticTrackVector * LeftVector,
474 G4KineticTrackVector * RightVector)
475{
476 //... perform last cluster decay
477
478 G4ThreeVector ClusterVel =string->Get4Momentum().boostVector();
479 G4double ResidualMass =string->Mass();
480
481 #ifdef debug_QGSMfragmentation
482 G4cout<<"Split last-----------------------------------------"<<G4endl;
483 G4cout<<"StrMass "<<ResidualMass<<" q's "
484 <<string->GetLeftParton()->GetParticleName()<<" "
485 <<string->GetRightParton()->GetParticleName()<<G4endl;
486 #endif
487
488 G4int cClusterInterrupt = 0;
489 G4ParticleDefinition * LeftHadron, * RightHadron;
490 const G4int maxNumberOfLoops = 1000;
491 G4int loopCounter = 0;
492 do
493 {
494 if (cClusterInterrupt++ >= ClusterLoopInterrupt)
495 {
496 return false;
497 }
498
499 G4ParticleDefinition * quark = NULL;
500 string->SetLeftPartonStable(); // to query quark contents..
501
502 if (string->DecayIsQuark() && string->StableIsQuark() )
503 {
504 //... there are quarks on cluster ends
505
506 G4int IsParticle=(string->GetLeftParton()->GetPDGEncoding()>0) ? -1 : +1; // if we have a quark,
507 // we need antiquark or diquark
508 pDefPair QuarkPair = CreatePartonPair(IsParticle);
509 quark = QuarkPair.second;
510
511 LeftHadron= hadronizer->BuildLowSpin(QuarkPair.first, string->GetLeftParton());
512 } else {
513 //... there is a Diquark on cluster ends
514 G4int IsParticle;
515 if ( string->StableIsQuark() ) {
516 IsParticle=(string->GetLeftParton()->GetPDGEncoding()>0) ? -1 : +1;
517 } else {
518 IsParticle=(string->GetLeftParton()->GetPDGEncoding()>0) ? +1 : -1;
519 }
520
521 //G4double ProbSaS = 1.0 - 2.0 * GetStrangeSuppress();
522 //G4double ActualProb = ProbSaS * 1.4;
523 //SetStrangenessSuppression((1.0-ActualProb)/2.0);
524
525 pDefPair QuarkPair = CreatePartonPair(IsParticle,false); // no diquarks wanted
526 //SetStrangenessSuppression((1.0-ProbSaS)/2.0);
527 quark = QuarkPair.second;
528 LeftHadron=hadronizer->BuildLowSpin(QuarkPair.first, string->GetLeftParton());
529 }
530
531 RightHadron = hadronizer->BuildLowSpin(string->GetRightParton(), quark);
532
533 } while ( ( ResidualMass <= LeftHadron->GetPDGMass() + RightHadron->GetPDGMass() )
534 && ++loopCounter < maxNumberOfLoops ); /* Loop checking, 07.08.2015, A.Ribon */
535
536 if ( loopCounter >= maxNumberOfLoops ) {
537 return false;
538 }
539
540 //... compute hadron momenta and energies
541 G4LorentzVector LeftMom, RightMom;
542 G4ThreeVector Pos;
543 Sample4Momentum(&LeftMom , LeftHadron->GetPDGMass() ,
544 &RightMom, RightHadron->GetPDGMass(), ResidualMass);
545 LeftMom.boost(ClusterVel);
546 RightMom.boost(ClusterVel);
547
548 #ifdef debug_QGSMfragmentation
549 G4cout<<LeftHadron->GetParticleName()<<" "<<RightHadron->GetParticleName()<<G4endl;
550 G4cout<<"Left Hadrom P M "<<LeftMom<<" "<<LeftMom.mag()<<G4endl;
551 G4cout<<"Right Hadrom P M "<<RightMom<<" "<<RightMom.mag()<<G4endl;
552 #endif
553
554 LeftVector->push_back(new G4KineticTrack(LeftHadron, 0, Pos, LeftMom));
555 RightVector->push_back(new G4KineticTrack(RightHadron, 0, Pos, RightMom));
556
557 return true;
558
559}

570G4bool G4QGSMFragmentation::StopFragmenting(const G4FragmentingString * const string)
571{
572 SetMinimalStringMass(string);
573 if (string->FourQuarkString())
574 {
575 return G4UniformRand() < G4Exp(-0.0005*(string->Mass() - MinimalStringMass));
576 } else {
577 G4bool Result = G4UniformRand() <
578 G4Exp(-0.66e-6*(string->Mass()*string->Mass() - MinimalStringMass*MinimalStringMass));
579 // G4bool Result = string->Mass() < MinimalStringMass + 150.*MeV*G4UniformRand(); // a'la LUND
580
581 #ifdef debug_QGSMfragmentation
582 G4cout<<"StopFragmenting MinimalStringMass string->Mass() "<<MinimalStringMass<<" "<<string->Mass()<<G4endl;
583 G4cout<<"StopFragmenting - Yes/No "<<Result<<G4endl;
584 #endif
585 return Result;
586 }

592void G4QGSMFragmentation::Sample4Momentum(G4LorentzVector* Mom , G4double Mass ,
593 G4LorentzVector* AntiMom, G4double AntiMass, G4double InitialMass)
594{
595 G4double r_val = sqr(InitialMass*InitialMass - Mass*Mass - AntiMass*AntiMass) - sqr(2.*Mass*AntiMass);
596 G4double Pabs = (r_val > 0.)? std::sqrt(r_val)/(2.*InitialMass) : 0;
597
598 //... sample unit vector
599 G4double pz = 1. - 2.*G4UniformRand();
600 G4double st = std::sqrt(1. - pz * pz)*Pabs;
601 G4double phi = 2.*pi*G4UniformRand();
602 G4double px = st*std::cos(phi);
603 G4double py = st*std::sin(phi);
604 pz *= Pabs;
605
606 Mom->setPx(px); Mom->setPy(py); Mom->setPz(pz);
607 Mom->setE(std::sqrt(Pabs*Pabs + Mass*Mass));
608
609 AntiMom->setPx(-px); AntiMom->setPy(-py); AntiMom->setPz(-pz);
610 AntiMom->setE (std::sqrt(Pabs*Pabs + AntiMass*AntiMass));
611}

615G4ParticleDefinition *G4QGSMFragmentation::DiQuarkSplitup( G4ParticleDefinition* decay,
616 G4ParticleDefinition *&created )
617{
618 //... can Diquark break or not?
619 if (G4UniformRand() < DiquarkBreakProb ) //... Diquark break
620 {
621 G4int stableQuarkEncoding = decay->GetPDGEncoding()/1000;
622 G4int decayQuarkEncoding = (decay->GetPDGEncoding()/100)%10;
623
624 if (G4UniformRand() < 0.5)
625 {
626 G4int Swap = stableQuarkEncoding;
627 stableQuarkEncoding = decayQuarkEncoding;
628 decayQuarkEncoding = Swap;
629 }
630
631 G4int IsParticle=(decayQuarkEncoding>0) ? -1 : +1; // if we have a quark, we need antiquark
632
633 G4double StrSup=GetStrangeSuppress();
634 SetStrangenessSuppression((1.0 - 0.07)/2.);
635 pDefPair QuarkPair = CreatePartonPair(IsParticle,false); // no diquarks wanted
636 SetStrangenessSuppression(StrSup);
637
638 //... Build new Diquark
639 G4int QuarkEncoding=QuarkPair.second->GetPDGEncoding();
640 G4int i10 = std::max(std::abs(QuarkEncoding), std::abs(stableQuarkEncoding));
641 G4int i20 = std::min(std::abs(QuarkEncoding), std::abs(stableQuarkEncoding));
642 G4int spin = (i10 != i20 && G4UniformRand() <= 0.5)? 1 : 3;
643 G4int NewDecayEncoding = -1*IsParticle*(i10 * 1000 + i20 * 100 + spin);
644 created = FindParticle(NewDecayEncoding);
645 G4ParticleDefinition * decayQuark=FindParticle(decayQuarkEncoding);
646 G4ParticleDefinition * had=hadronizer->Build(QuarkPair.first, decayQuark);
647
648 return had;
649
650 } else { //... Diquark does not break
651
652 G4int IsParticle=(decay->GetPDGEncoding()>0) ? +1 : -1; // if we have a diquark, we need quark)
653 G4double StrSup=GetStrangeSuppress(); // for changing s-sbar production
654 SetStrangenessSuppression((1.0 - 0.07)/2.);
655 pDefPair QuarkPair = CreatePartonPair(IsParticle,false); // no diquarks wanted
656 SetStrangenessSuppression(StrSup);
657
658 created = QuarkPair.second;
659
660 G4ParticleDefinition * had=hadronizer->Build(QuarkPair.first, decay);
661 return had;
662 }
663}

667G4KineticTrack * G4QGSMFragmentation::Splitup( G4FragmentingString *string,
668 G4FragmentingString *&newString)
669{
670 #ifdef debug_QGSMfragmentation
671 G4cout<<G4endl;
672 G4cout<<"Start SplitUP (G4VLongitudinalStringDecay) ========================="<<G4endl;
673 G4cout<<"String partons: " <<string->GetLeftParton()->GetPDGEncoding()<<" "
674 <<string->GetRightParton()->GetPDGEncoding()<<" "
675 <<"Direction " <<string->GetDecayDirection()<<G4endl;
676 #endif
677
678 //... random choice of string end to use for creating the hadron (decay)
679 G4int SideOfDecay = (G4UniformRand() < 0.5)? 1: -1;
680 if (SideOfDecay < 0)
681 {
682 string->SetLeftPartonStable();
683 } else
684 {
685 string->SetRightPartonStable();
686 }
687
688 G4ParticleDefinition *newStringEnd;
689 G4ParticleDefinition * HadronDefinition;
690 if (string->DecayIsQuark())
691 {
692 G4double ProbDqADq = GetDiquarkSuppress();
693
694 G4int NumberOfpossibleBaryons = 2;
695
696 if (string->GetLeftParton()->GetParticleSubType() != "quark") NumberOfpossibleBaryons++;
697 if (string->GetRightParton()->GetParticleSubType() != "quark") NumberOfpossibleBaryons++;
698
699 G4double ActualProb = ProbDqADq ;
700 ActualProb *= (1.0-G4Exp(2.0*(1.0 - string->Mass()/(NumberOfpossibleBaryons*1400.0))));
701
702 SetDiquarkSuppression(ActualProb);
703
704 HadronDefinition= QuarkSplitup(string->GetDecayParton(), newStringEnd);
705
706 SetDiquarkSuppression(ProbDqADq);
707 } else {
708 HadronDefinition= DiQuarkSplitup(string->GetDecayParton(), newStringEnd);
709 }
710
711 #ifdef debug_QGSMfragmentation
712 G4cout<<"The parton "<<string->GetDecayParton()->GetPDGEncoding()<<" "
713 <<" produces hadron "<<HadronDefinition->GetParticleName()
714 <<" and is transformed to "<<newStringEnd->GetPDGEncoding()<<G4endl;
715 G4cout<<"The side of the string decay Left/Right (1/-1) "<<SideOfDecay<<G4endl;
716 #endif
717 // create new String from old, ie. keep Left and Right order, but replace decay
718
719 newString=new G4FragmentingString(*string,newStringEnd); // To store possible
720 // quark containt of new string
721
722 #ifdef debug_QGSMfragmentation
723 G4cout<<"An attempt to determine its energy (SplitEandP)"<<G4endl;
724 #endif
725 G4LorentzVector* HadronMomentum=SplitEandP(HadronDefinition, string, newString);
726
727 delete newString; newString=0;
728
729 G4KineticTrack * Hadron =0;
730 if ( HadronMomentum != 0 ) {
731
732 #ifdef debug_QGSMfragmentation
733 G4cout<<"The attempt was successful"<<G4endl;
734 #endif
735 G4ThreeVector Pos;
736 Hadron = new G4KineticTrack(HadronDefinition, 0,Pos, *HadronMomentum);
737
738 newString=new G4FragmentingString(*string,newStringEnd,HadronMomentum);
739
740 delete HadronMomentum;
741 }
742 else
743 {
744 #ifdef debug_QGSMfragmentation
745 G4cout<<"The attempt was not successful !!!"<<G4endl;
746 #endif
747 }
748
749 #ifdef debug_VStringDecay
750 G4cout<<"End SplitUP (G4VLongitudinalStringDecay) ====================="<<G4endl;
751 #endif
752
753 return Hadron;
754}

780//--------------------------------------------------------------------------------------
781void G4QGSMFragmentation::SetMinimalStringMass(const G4FragmentingString * const string)
782{
783 G4double EstimatedMass=0.;
784
785 G4int Qleft =std::abs(string->GetLeftParton()->GetPDGEncoding());
786 G4int Qright=std::abs(string->GetRightParton()->GetPDGEncoding());
787
788 if ((Qleft < 4) && (Qright < 4)) { // Q-Qbar string
789 EstimatedMass=minMassQQbarStr[Qleft-1][Qright-1];
790 MinimalStringMass=EstimatedMass;
791 SetMinimalStringMass2(EstimatedMass);
792 return;
793 }
794
795 if ((Qleft < 4) && (Qright > 1000)) { // Q - DiQ string
796 G4int q1=Qright/1000;
797 G4int q2=(Qright/100)%10;
798 EstimatedMass=minMassQDiQStr[Qleft-1][q1-1][q2-1];
799 MinimalStringMass=EstimatedMass;
800 SetMinimalStringMass2(EstimatedMass);
801 return;
802 }
803
804 if ((Qleft > 1000) && (Qright < 4)) { // DiQ - Q string
805 G4int q1=Qleft/1000;
806 G4int q2=(Qleft/100)%10;
807 EstimatedMass=minMassQDiQStr[Qright-1][q1-1][q2-1];
808 MinimalStringMass=EstimatedMass;
809 SetMinimalStringMass2(EstimatedMass);
810 return;
811 }
812
813 // DiQuark - Anti DiQuark string -----------------
814 G4int Number_of_quarks=0;
815 G4int Number_of_squarks=0;
816
817 G4double StringM=string->Get4Momentum().mag();
818
819 #ifdef debug_QGSMfragmentation
820 // G4cout<<"MinStringMass// Input String mass "<<string->Get4Momentum().mag()<<" Qleft "<<Qleft<<G4endl;
821 #endif
822
823 if ( Qleft > 1000)
824 {
825 Number_of_quarks+=2;
826 G4int q1=Qleft/1000;
827 if ( q1 < 3) {EstimatedMass +=Mass_of_light_quark;}
828 if ( q1 > 2) {EstimatedMass +=Mass_of_heavy_quark; Number_of_squarks++;}
829
830 G4int q2=(Qleft/100)%10;
831 if ( q2 < 3) {EstimatedMass +=Mass_of_light_quark;}
832 if ( q2 > 2) {EstimatedMass +=Mass_of_heavy_quark; Number_of_squarks++;}
833 }
834
835 #ifdef debug_QGSMfragmentation
836 // G4cout<<"Min mass with Qleft "<<Qleft<<" "<<EstimatedMass<<G4endl;
837 #endif
838
839 if ( Qright > 1000)
840 {
841 Number_of_quarks+=2;
842 G4int q1=Qright/1000;
843 if ( q1 < 3) {EstimatedMass +=Mass_of_light_quark;}
844 if ( q1 > 2) {EstimatedMass +=Mass_of_heavy_quark; Number_of_squarks++;}
845
846 G4int q2=(Qright/100)%10;
847 if ( q2 < 3) {EstimatedMass +=Mass_of_light_quark;}
848 if ( q2 > 2) {EstimatedMass +=Mass_of_heavy_quark; Number_of_squarks++;}
849 //EstimatedMass +=Mass_of_string_junction;
850 }
851
852 #ifdef debug_QGSMfragmentation
853 // G4cout<<"Min mass with Qleft and Qright "<<Qright<<" "<<EstimatedMass<<G4endl;
854 // G4cout<<"Number_of_quarks "<<Number_of_quarks<<" Number_of_squarks "<<Number_of_squarks<<G4endl;
855 #endif
856
857 if (Number_of_quarks==4)
858 {
859 if (StringM > 1880.) { // 2*Mn = 1880
860 if (Number_of_squarks==0) {EstimatedMass += 1320.*MeV;} //560+1320=1880=2*Mn
861 else if (Number_of_squarks==1) {EstimatedMass += 1150.*MeV;} //920+1150=2070=M(Lam+N)
862 else if (Number_of_squarks==2) {EstimatedMass += 960.*MeV;} //1280+960=2240= 2*M Lam
863 else if (Number_of_squarks==3) {EstimatedMass += 800.*MeV;} //1640+800=2440=Mxi+Mlam
864 else if (Number_of_squarks==4) {EstimatedMass += 640.*MeV;} //2000+640=2640=2*Mxi
865 else {}
866 }
867 else
868 {
869 if (Number_of_squarks < 3) {EstimatedMass -= 200.*MeV;}
870 else if (Number_of_squarks==3) {EstimatedMass -= 50.*MeV;}
871 else if (Number_of_squarks==4) {EstimatedMass -= 40.*MeV;}
872 else {}
873 }
874 }
875
876 #ifdef debug_QGSMfragmentation
877 // G4cout<<"EstimatedMass -------------------- "<<EstimatedMass <<G4endl;
878 #endif
879
880 MinimalStringMass=EstimatedMass;
881 SetMinimalStringMass2(EstimatedMass);
882}
