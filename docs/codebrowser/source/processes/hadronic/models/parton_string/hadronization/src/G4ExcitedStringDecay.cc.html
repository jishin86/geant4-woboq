87G4KineticTrackVector *G4ExcitedStringDecay::FragmentStrings(const G4ExcitedStringVector * theStrings)
88{
89 G4LorentzVector KTsum(0.,0.,0.,0.);
90
91 #ifdef debug_G4ExcitedStringDecay
92 G4cout<<G4endl;
93 G4cout<<"--------------------------- G4ExcitedStringDecay ----------------------"<<G4endl;
94 G4cout<<"Hadronization of Excited Strings: theStrings->size() "<<theStrings->size()<<G4endl;
95 #endif
96
97 for ( unsigned int astring=0; astring < theStrings->size(); astring++)
98 // for ( unsigned int astring=0; astring < 1; astring++)
99 {
100 // G4cout<<"theStrings->operator[](astring)->IsExcited() "<<" "<<astring<<" "<<theStrings->operator[](astring)->IsExcited()<<G4endl;
101 if ( theStrings->operator[](astring)->IsExcited() )
102 {KTsum+= theStrings->operator[](astring)->Get4Momentum();}
103 else {KTsum+=theStrings->operator[](astring)->GetKineticTrack()->Get4Momentum();}
104 }
105
106 G4LorentzRotation toCms( -1 * KTsum.boostVector() );
107 G4LorentzRotation toLab(toCms.inverse());
108 G4LorentzVector Ptmp;
109 KTsum=G4LorentzVector(0.,0.,0.,0.);
110
111 for ( unsigned int astring=0; astring < theStrings->size(); astring++)
112 // for ( unsigned int astring=0; astring < 1; astring++)
113 {
114 if ( theStrings->operator[](astring)->IsExcited() )
115 {
116 Ptmp=toCms * theStrings->operator[](astring)->GetLeftParton()->Get4Momentum();
117 theStrings->operator[](astring)->GetLeftParton()->Set4Momentum(Ptmp);
118
119 Ptmp=toCms * theStrings->operator[](astring)->GetRightParton()->Get4Momentum();
120 theStrings->operator[](astring)->GetRightParton()->Set4Momentum(Ptmp);
121
122 KTsum+= theStrings->operator[](astring)->Get4Momentum();
123 }
124 else
125 {
126 Ptmp=toCms * theStrings->operator[](astring)->GetKineticTrack()->Get4Momentum();
127 theStrings->operator[](astring)->GetKineticTrack()->Set4Momentum(Ptmp);
128 KTsum+= theStrings->operator[](astring)->GetKineticTrack()->Get4Momentum();
129 }
130 }
131
132 G4SampleResonance BrW;
133 const G4ParticleDefinition* TrackDefinition=0;
134
135 G4KineticTrackVector * theResult = new G4KineticTrackVector;
136 G4int attempts(0);
137 G4bool success=false;
138 G4bool NeedEnergyCorrector=false;
139 do {
140 #ifdef debug_G4ExcitedStringDecay
141 G4cout<<"New try No "<<attempts<<" to hadronize strings"<<G4endl;
142 #endif
143
144 std::for_each(theResult->begin() , theResult->end() , DeleteKineticTrack());
145 theResult->clear();
146
147 attempts++;
148
149 G4LorentzVector KTsecondaries(0.,0.,0.,0.);
150 NeedEnergyCorrector=false;
151
152 for ( unsigned int astring=0; astring < theStrings->size(); astring++)
153 // for ( unsigned int astring=0; astring < 1; astring++) // Proj. Last Str. Decay for FTF
154 // for ( unsigned int astring=1; astring < 2; astring++) // Proj. Last Str. Decay for QGS
155 // for ( unsigned int astring=0; astring < 1; astring++) // For testing purposes
156 {
157 #ifdef debug_G4ExcitedStringDecay
158 G4cout<<"String No "<<astring+1<<" Excited? "<<theStrings->operator[](astring)->IsExcited()<<G4endl;
159
160 G4cout<<"String No "<<astring+1<<" 4Momentum "<<theStrings->operator[](astring)->Get4Momentum()
161 <<" "<<theStrings->operator[](astring)->Get4Momentum().mag()<<G4endl;
162 #endif
163
164 G4KineticTrackVector * generatedKineticTracks = NULL;
165 if ( theStrings->operator[](astring)->IsExcited() )
166 {
167 #ifdef debug_G4ExcitedStringDecay
168 G4cout<<"Fragment String with partons: "
169 <<theStrings->operator[](astring)->GetLeftParton()->GetPDGcode() <<" "
170 <<theStrings->operator[](astring)->GetRightParton()->GetPDGcode()<<" "
171 <<"Direction "<<theStrings->operator[](astring)->GetDirection()<<G4endl;
172 #endif
173 generatedKineticTracks=FragmentString(*theStrings->operator[](astring));
174 #ifdef debug_G4ExcitedStringDecay
175 G4cout<<"(G4ExcitedStringDecay) Number of produced hadrons = "
176 <<generatedKineticTracks->size()<<G4endl;
177 #endif
178 } else {
179 #ifdef debug_G4ExcitedStringDecay
180 G4cout<<" GetTrack from the String"<<G4endl;
181 #endif
182 G4LorentzVector Mom=theStrings->operator[](astring)->GetKineticTrack()->Get4Momentum();
183 G4KineticTrack * aTrack= new G4KineticTrack(
184 theStrings->operator[](astring)->GetKineticTrack()->GetDefinition(),
185 theStrings->operator[](astring)->GetKineticTrack()->GetFormationTime(),
186 G4ThreeVector(0), Mom);
187
188 aTrack->SetPosition(theStrings->operator[](astring)->GetKineticTrack()->GetPosition());
189
190 #ifdef debug_G4ExcitedStringDecay
191 G4cout<<" A particle stored in the track is "<<aTrack->GetDefinition()->GetParticleName()<<G4endl;
192 #endif
193
194 generatedKineticTracks = new G4KineticTrackVector;
195 generatedKineticTracks->push_back(aTrack);
196 }
197
198 if (generatedKineticTracks->size() == 0)
199 {
200 // G4cerr << "G4VPartonStringModel:No KineticTracks produced" << G4endl;
201 // continue;
202 success=false; NeedEnergyCorrector=false; break;
203 }
204
205 G4LorentzVector KTsum1(0.,0.,0.,0.);
206 for ( unsigned int aTrack=0; aTrack<generatedKineticTracks->size();aTrack++)
207 {
208 #ifdef debug_G4ExcitedStringDecay
209 G4cout<<"Prod part No. "<<aTrack+1<<" "
210 <<(*generatedKineticTracks)[aTrack]->GetDefinition()->GetParticleName()<<" "
211 <<(*generatedKineticTracks)[aTrack]->Get4Momentum()
212 <<(*generatedKineticTracks)[aTrack]->Get4Momentum().mag()<<G4endl;
213 #endif
214 // --------------- Sampling mass of unstable hadronic resonances ----------------
215 TrackDefinition = (*generatedKineticTracks)[aTrack]->GetDefinition();
216
217 if(TrackDefinition->IsShortLived())
218 {
219 G4double NewTrackMass = BrW.SampleMass( TrackDefinition,
220 TrackDefinition->GetPDGMass() + 5.0*TrackDefinition->GetPDGWidth() );
221 G4LorentzVector Tmp=G4LorentzVector((*generatedKineticTracks)[aTrack]->Get4Momentum());
222 Tmp.setE(std::sqrt(sqr(NewTrackMass) + Tmp.vect().mag2()));
223
224 (*generatedKineticTracks)[aTrack]->Set4Momentum(Tmp);
225
226 #ifdef debug_G4ExcitedStringDecay
227 G4cout<<"Resonance *** "<<aTrack+1<<" "
228 <<(*generatedKineticTracks)[aTrack]->GetDefinition()->GetParticleName()<<" "
229 <<(*generatedKineticTracks)[aTrack]->Get4Momentum()
230 <<(*generatedKineticTracks)[aTrack]->Get4Momentum().mag()<<G4endl;
231 #endif
232 }
233 //-------------------------------------------------------------------------------
234
235 theResult->push_back(generatedKineticTracks->operator[](aTrack));
236 KTsum1+= (*generatedKineticTracks)[aTrack]->Get4Momentum();
237 }
238 KTsecondaries+=KTsum1;
239
240 #ifdef debug_G4ExcitedStringDecay
241 G4cout << "String secondaries(" <<generatedKineticTracks->size()<< ")"<<G4endl
242 <<"Init string momentum: "<< theStrings->operator[](astring)->Get4Momentum()<<G4endl
243 <<"Final hadrons momentum: "<< KTsum1 << G4endl;
244 #endif
245
246 if ( KTsum1.e() > 0 && std::abs((KTsum1.e()-theStrings->operator[](astring)->Get4Momentum().e()) / KTsum1.e()) > perMillion )
247 {
248 NeedEnergyCorrector=true;
249 }
250
251 #ifdef debug_G4ExcitedStringDecay
252 G4cout<<"NeedEnergyCorrection yes/no "<<NeedEnergyCorrector<<G4endl;
253 #endif
254
255 // clean up
256 delete generatedKineticTracks;
257 success=true;
258 }
259
260 if ( NeedEnergyCorrector ) success=EnergyAndMomentumCorrector(theResult, KTsum);
261 } while (!success && (attempts < 100)); /* Loop checking, 07.08.2015, A.Ribon */
262
263 for ( unsigned int aTrack=0; aTrack<theResult->size();aTrack++)
264 {
265 Ptmp=(*theResult)[aTrack]->Get4Momentum();
266 Ptmp.transform( toLab);
267 (*theResult)[aTrack]->Set4Momentum(Ptmp);
268 }
269
270 #ifdef debug_G4ExcitedStringDecay
271 G4cout<<"End of the strings fragmentation (G4ExcitedStringDecay)"<<G4endl;
272
273 G4LorentzVector KTsum1(0.,0.,0.,0.);
274
275 for ( unsigned int aTrack=0; aTrack<theResult->size();aTrack++)
276 {
277 G4cout << " corrected tracks .. " << (*theResult)[aTrack]->GetDefinition()->GetParticleName()
278 <<" " << (*theResult)[aTrack]->Get4Momentum()
279 <<" " << (*theResult)[aTrack]->Get4Momentum().mag()<< G4endl;
280 KTsum1+= (*theResult)[aTrack]->Get4Momentum();
281 }
282
283 G4cout << "Needcorrector/success " << NeedEnergyCorrector << "/" << success
284 << ", Corrected total 4 momentum " << KTsum1 << G4endl;
285 if ( ! success ) G4cout << "failed to correct E/p" << G4endl;
286
287 G4cout<<"End of the Hadronization (G4ExcitedStringDecay)"<<G4endl;
288 #endif
289
290 if(!success)
291 {
292 if(theResult->size() != 0)
293 {
294 std::for_each(theResult->begin() , theResult->end() , DeleteKineticTrack());
295 theResult->clear();
296 delete theResult; theResult=0;
297 }
298 for ( unsigned int astring=0; astring < theStrings->size(); astring++)
299 // for ( unsigned int astring=0; astring < 1; astring++) // Need more correct. For testing purposes.
300 {
301 if ( theStrings->operator[](astring)->IsExcited() )
302 {
303 Ptmp=theStrings->operator[](astring)->GetLeftParton()->Get4Momentum();
304 Ptmp.transform( toLab);
305 theStrings->operator[](astring)->GetLeftParton()->Set4Momentum(Ptmp);
306
307 Ptmp=theStrings->operator[](astring)->GetRightParton()->Get4Momentum();
308 Ptmp.transform( toLab);
309 theStrings->operator[](astring)->GetRightParton()->Set4Momentum(Ptmp);
310 }
311 else
312 {
313 Ptmp=theStrings->operator[](astring)->GetKineticTrack()->Get4Momentum();
314 Ptmp.transform( toLab);
315 theStrings->operator[](astring)->GetKineticTrack()->Set4Momentum(Ptmp);
316 }
317 }
318 }
319 return theResult;
320}

323G4bool G4ExcitedStringDecay::EnergyAndMomentumCorrector
324 (G4KineticTrackVector* Output, G4LorentzVector& TotalCollisionMom)
325{
326 const int nAttemptScale = 500;
327 const double ErrLimit = 1.E-5;
328 if (Output->empty()) return TRUE;
329 G4LorentzVector SumMom;
330 G4double SumMass = 0;
331 G4double TotalCollisionMass = TotalCollisionMom.m();
332
333 std::vector<G4double> HadronMass; G4double HadronM(0.);
334
335 #ifdef debug_G4ExcitedStringCorr
336 G4cout<<G4endl<<"EnergyAndMomentumCorrector. Number of particles: "<<Output->size()<<G4endl;
337 #endif
338 // Calculate sum hadron 4-momenta and summing hadron mass
339 unsigned int cHadron;
340 for(cHadron = 0; cHadron < Output->size(); cHadron++)
341 {
342 SumMom += Output->operator[](cHadron)->Get4Momentum();
343 HadronM=Output->operator[](cHadron)->Get4Momentum().mag(); HadronMass.push_back(HadronM);
344 SumMass += Output->operator[](cHadron)->Get4Momentum().mag(); //GetDefinition()->GetPDGMass();
345 }
346
347 #ifdef debug_G4ExcitedStringCorr
348 G4cout<<"Sum part mom "<<SumMom<<" "<<SumMom.mag()<<G4endl
349 <<"Sum str mom "<<TotalCollisionMom<<" "<<TotalCollisionMom.mag()<<G4endl;
350 G4cout<<"SumMass TotalCollisionMass "<<SumMass<<" "<<TotalCollisionMass<<G4endl;
351 #endif
352
353 // Cannot correct a single particle
354 if (Output->size() < 2) return FALSE;
355
356 if (SumMass > TotalCollisionMass) return FALSE;
357 SumMass = SumMom.m2();
358 if (SumMass < 0) return FALSE;
359 SumMass = std::sqrt(SumMass);
360
361 // Compute c.m.s. hadron velocity and boost KTV to hadron c.m.s.
362 // G4ThreeVector Beta = -SumMom.boostVector();
363 G4ThreeVector Beta = -TotalCollisionMom.boostVector();
364 Output->Boost(Beta);
365
366 // Scale total c.m.s. hadron energy (hadron system mass).
367 // It should be equal interaction mass
368 G4double Scale = 1;
369 G4int cAttempt = 0;
370 G4double Sum = 0;
371 G4bool success = false;
372 for(cAttempt = 0; cAttempt < nAttemptScale; cAttempt++)
373 {
374 Sum = 0;
375 for(cHadron = 0; cHadron < Output->size(); cHadron++)
376 {
377 HadronM = HadronMass.at(cHadron);
378 G4LorentzVector HadronMom = Output->operator[](cHadron)->Get4Momentum();
379 HadronMom.setVect(Scale*HadronMom.vect());
380 G4double E = std::sqrt(HadronMom.vect().mag2() + sqr(HadronM));
381 //sqr(Output->operator[](cHadron)->GetDefinition()->GetPDGMass()));
382 HadronMom.setE(E);
383 Output->operator[](cHadron)->Set4Momentum(HadronMom);
384 Sum += E;
385 }
386 Scale = TotalCollisionMass/Sum;
387 #ifdef debug_G4ExcitedStringCorr
388 G4cout << "Scale-1=" << Scale -1
389 << ", TotalCollisionMass=" << TotalCollisionMass
390 << ", Sum=" << Sum
391 << G4endl;
392 #endif
393 if (std::fabs(Scale - 1) <= ErrLimit)
394 {
395 success = true;
396 break;
397 }
398 }
399
400 #ifdef debug_G4ExcitedStringCorr
401 if(!success)
402 {
403 G4cout << "G4ExcitedStringDecay::EnergyAndMomentumCorrector - Warning"<<G4endl;
404 G4cout << " Scale not unity at end of iteration loop: "<<TotalCollisionMass<<" "<<Sum<<" "<<Scale<<G4endl;
405 G4cout << " Number of secondaries: " << Output->size() << G4endl;
406 G4cout << " Wanted total energy: " << TotalCollisionMom.e() << G4endl;
407 G4cout << " Increase number of attempts or increase ERRLIMIT"<<G4endl;
408 // throw G4HadronicException(__FILE__, __LINE__, "G4ExcitedStringDecay failed to correct...");
409 }
410 #endif
411
412 // Compute c.m.s. interaction velocity and KTV back boost
413 Beta = TotalCollisionMom.boostVector();
414 Output->Boost(Beta);
415
416 return success;
417}
