158inline
159void G4FragmentingString::LorentzRotate(const G4LorentzRotation & rotation)
160{
161 SetPleft(rotation*Pleft);
162 SetPright(rotation*Pright);
163 Pstring = Pleft+Pright;
164 Ptleft =Pleft.vect(); Ptleft.setZ(0.);
165 Ptright=Pright.vect(); Ptright.setZ(0.);
166 Pplus =Pstring.plus();
167 Pminus=Pstring.minus();
168}
169
170inline
171G4LorentzRotation G4FragmentingString::TransformToCenterOfMass()
172{
173 G4LorentzVector momentum=Pstring;
174 G4LorentzRotation toCMS(-1*momentum.boostVector());
175
176 Pleft *= toCMS;
177 Pright *= toCMS;
178 Pstring *= toCMS;
179 Ptleft =Pleft.vect(); Ptleft.setZ(0.);
180 Ptright=Pright.vect(); Ptright.setZ(0.);
181 Pplus =Pstring.plus();
182 Pminus=Pstring.minus();
183 return toCMS;
184}
185
186inline
187G4LorentzRotation G4FragmentingString::TransformToAlignedCms()
188{
189 G4LorentzVector momentum=Pstring;
190 G4LorentzRotation toAlignedCms(-1*momentum.boostVector());
191
192 momentum= toAlignedCms* Pleft;
193 toAlignedCms.rotateZ(-1*momentum.phi());
194 toAlignedCms.rotateY(-1*momentum.theta());
195
196 Pleft *= toAlignedCms;
197 Pright *= toAlignedCms;
198 Pstring *= toAlignedCms;
199
200 Ptleft = G4ThreeVector(0.,0.,0.);
201 Ptright = G4ThreeVector(0.,0.,0.);
202 Pplus = Pstring.plus();
203 Pminus = Pstring.minus();
204
205 return toAlignedCms;
206}
