26#ifndef G4ASCCrossSection_h
27#define G4ASCCrossSection_h
28
29#include "globals.hh"
30#include "G4VAnnihilationCrossSection.hh"
31
32#include "G4Pow.hh"
33
34
35class G4ASCCrossSection : public G4VAnnihilationCrossSection
36{
37 public:
38 G4ASCCrossSection(G4int, G4int, G4double, G4double, G4double, G4double);
39 G4bool InCharge(G4int aCode, G4int bCode);
40 G4double GetXsec(G4double S);
41 private:
42
43 G4int theCode1;
44 G4int theCode2;
45 G4double theX;
46 G4double theY;
47 G4double theEta;
48 G4double theEps;
49};
50
51
52inline G4bool G4ASCCrossSection::InCharge(G4int aCode, G4int bCode)
53{
54 G4bool result;
55 result = (aCode==theCode1&&bCode==theCode2)||(aCode==theCode2&&bCode==theCode1);
56 return result;
57}
58
59inline G4double G4ASCCrossSection::GetXsec(G4double S)
60{
61 G4double result = theX*G4Pow::GetInstance()->powA(S, theEps) +
62 theY*G4Pow::GetInstance()->powA(S, -theEta);
63 return result;
64}
