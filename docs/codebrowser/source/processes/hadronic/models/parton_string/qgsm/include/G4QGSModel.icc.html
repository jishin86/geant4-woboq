28template<class ParticipantType>
29G4QGSModel<ParticipantType>::G4QGSModel()
30 {
31 G4VPartonStringModel::SetThisPointer(this);
32 SetEnergyMomentumCheckLevels(2*CLHEP::perCent, 150*CLHEP::MeV);
33 }
34template<class ParticipantType>
35G4QGSModel<ParticipantType>::G4QGSModel(const G4QGSModel &right)
36 {
37 G4VPartonStringModel::SetThisPointer(this);
38 std::pair<G4double, G4double> levels=right.GetEnergyMomentumCheckLevels();
39 SetEnergyMomentumCheckLevels(levels.first, levels.second);
40 }

57template<class ParticipantType>
58void G4QGSModel<ParticipantType>::Init(const G4Nucleus & aNucleus, const G4DynamicParticle & aProjectile)
59 {
60// It is assumed that target nucleus is at rest.
61
62 theParticipants.Init(aNucleus.GetA_asInt(),aNucleus.GetZ_asInt()); // Init of a target nucleus
63
64 G4LorentzVector Mom = aProjectile.Get4Momentum(); // Momentum of a projectile
65
66 G4ReactionProduct theProjectile;
67 theProjectile.SetDefinition(aProjectile.GetDefinition());
68 theProjectile.SetTotalEnergy(Mom.e());
69 theProjectile.SetMomentum(Mom.vect());
70
71 theParticipants.BuildInteractions(theProjectile); // Creation of strings and nuclear remnant
72 }
73
74template<class ParticipantType>
75G4ExcitedStringVector * G4QGSModel<ParticipantType>::GetStrings()
76{ // The strings are produced at theParticipants.BuildInteractions(theProjectile)
77
78 G4PartonPair* aPair;
79 G4ExcitedStringVector* theStrings = new G4ExcitedStringVector;
80 G4ExcitedString * aString;
81 while( (aPair = theParticipants.GetNextPartonPair()) ) /* Loop checking, 07.08.2015, A.Ribon */
82 {
83 if (aPair->GetCollisionType() == G4PartonPair::DIFFRACTIVE)
84 {
85 aString = theDiffractiveStringBuilder.BuildString(aPair);
86 }
87 else
88 {
89 aString = theSoftStringBuilder.BuildString(aPair);
90 }
91
92// aString->Boost(theCurrentVelocity);
93 theStrings->push_back(aString);
94 delete aPair;
95 }
96
97 return theStrings;
98}

112template<class ParticipantType>
113void G4QGSModel<ParticipantType>::ModelDescription(std::ostream& outFile) const
114{
115 outFile << "The Quark-Gluon String (QGS) model simulates the interaction\n"
116 << "of protons, neutrons, pions and kaons with nuclei in the\n"
117 << "approximate energy range 20 GeV to 50 TeV. The model handles\n"
118 << "the selection of collision partners, splitting of the nucleons\n"
119 << "into quarks and di-quarks, the formation and excitation of\n"
120 << "quark-gluon strings, string hadronization and diffractive dissociation.\n";
121}
