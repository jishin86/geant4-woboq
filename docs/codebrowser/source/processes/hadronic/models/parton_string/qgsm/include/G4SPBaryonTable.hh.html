26#ifndef G4SPBaryonTable_h
27#define G4SPBaryonTable_h
28
29#include <vector>
30#include "G4SPBaryon.hh"
31
32class G4SPBaryonTable
33{
34 public:
35 struct DeleteSPBaryon{void operator()(G4SPBaryon* aS){delete aS;} };
36
37 ~G4SPBaryonTable() {std::for_each(theBaryons.begin(), theBaryons.end(), G4SPBaryonTable::DeleteSPBaryon());}
38 void insert(G4SPBaryon * aBaryon) { theBaryons.push_back(aBaryon);}
39 G4double length() {return theBaryons.size();}
40
41 const G4SPBaryon * GetBaryon(G4ParticleDefinition * aDefinition);
42
43 private:
44 std::vector<G4SPBaryon *> theBaryons;
45};
46
47inline const G4SPBaryon * G4SPBaryonTable::
48GetBaryon(G4ParticleDefinition * aDefinition)
49{
50 G4SPBaryon * result = 0;
51 for(unsigned int i=0; i<theBaryons.size(); i++)
52 {
53 if(theBaryons[i]->GetDefinition()==aDefinition)
54 {
55 result = theBaryons[i];
56 break;
57 }
58 }
59 return result;
60}
