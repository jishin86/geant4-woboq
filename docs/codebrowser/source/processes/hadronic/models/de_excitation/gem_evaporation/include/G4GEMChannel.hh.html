31#ifndef G4GEMChannel_h
32#define G4GEMChannel_h 1
33
34#include "G4VEvaporationChannel.hh"
35#include "G4GEMProbability.hh"
36#include "G4VLevelDensityParameter.hh"
37#include "G4EvaporationLevelDensityParameter.hh"
38#include "G4NucleiProperties.hh"
39#include "Randomize.hh"
40#include "G4ParticleTable.hh"
41
42class G4Pow;
43class G4PairingCorrection;
44class G4VCoulombBarrier;
45
46class G4GEMChannel : public G4VEvaporationChannel
47{
48public:
49
50 explicit G4GEMChannel(G4int theA, G4int theZ, const G4String & aName,
51 G4GEMProbability * aEmissionStrategy);
52
53 // destructor
54 virtual ~G4GEMChannel();
55
56 virtual G4double GetEmissionProbability(G4Fragment* theNucleus);
57
58 virtual G4Fragment* EmittedFragment(G4Fragment* theNucleus);
59
60 virtual void Dump() const;
61
62 inline void SetLevelDensityParameter(G4VLevelDensityParameter * aLevelDensity)
63 {
64 if (MyOwnLevelDensity) { delete theLevelDensityPtr; }
65 theLevelDensityPtr = aLevelDensity;
66 MyOwnLevelDensity = false;
67 }
68
69private:
70
71 // Samples fragment kinetic energy.
72 G4double SampleKineticEnergy(const G4Fragment & fragment);
73
74 G4GEMChannel(const G4GEMChannel & right) = delete;
75 const G4GEMChannel & operator=(const G4GEMChannel & right) = delete;
76 G4bool operator==(const G4GEMChannel & right) const = delete;
77 G4bool operator!=(const G4GEMChannel & right) const = delete;
78
79 // Data Members ************
80 // This data member define the channel.
81 // They are intializated at object creation (constructor) time.
82
83 // Atomic Number
84 G4int A;
85
86 // Charge
87 G4int Z;
88
89 G4double EvaporatedMass;
90 G4double ResidualMass;
91
92 G4Pow* fG4pow;
93
94 // For evaporation probability calcualtion
95 G4GEMProbability * theEvaporationProbabilityPtr;
96
97 // For Level Density calculation
98 G4bool MyOwnLevelDensity;
99 G4VLevelDensityParameter * theLevelDensityPtr;
100
101 // For Coulomb Barrier calculation
102 G4VCoulombBarrier * theCoulombBarrierPtr;
103 G4double CoulombBarrier;
104
105 G4PairingCorrection* pairingCorrection;
106
107 //---------------------------------------------------
108
109 // These values depend on the nucleus that is being evaporated.
110 // They are calculated through the Initialize method which takes as parameters
111 // the atomic number, charge and excitation energy of nucleus.
112
113 // Residual Atomic Number
114 G4int ResidualA;
115
116 // Residual Charge
117 G4int ResidualZ;
118
119 // Emission Probability
120 G4double EmissionProbability;
121
122 // Maximal Kinetic Energy that can be carried by fragment
123 G4double MaximalKineticEnergy;
124};
125
126
127#endif
