31#include "G4FermiFragment.hh"
32#include "G4NucleiProperties.hh"
33#include "G4PhysicalConstants.hh"
34
35G4FermiFragment::G4FermiFragment(G4int anA, G4int aZ, G4int sp,
36 G4double exc, G4bool stable, G4bool phys):
37 A(anA), Z(aZ), spin(sp), isStable(stable), isPhysical(phys),
38 excitEnergy(exc)
39{
40 cBarrier = nullptr;
41 if(phys) {
42 fragmentMass = G4NucleiProperties::GetNuclearMass(A, Z);
43 if(!stable) { cBarrier = new G4CoulombBarrier(A, Z); }
44 } else {
45 fragmentMass = Z*proton_mass_c2 + (A-Z)*neutron_mass_c2;
46 }
47}
48
49G4FermiFragment::~G4FermiFragment()
50{
51 delete cBarrier;
52}
