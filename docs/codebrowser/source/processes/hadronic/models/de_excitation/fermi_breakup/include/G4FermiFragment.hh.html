31#ifndef G4FermiFragment_h
32#define G4FermiFragment_h 1
33
34#include "globals.hh"
35#include "G4FragmentVector.hh"
36#include "G4CoulombBarrier.hh"
37
38class G4FermiFragment
39{
40public:
41
42 explicit G4FermiFragment(G4int anA, G4int aZ, G4int sp,
43 G4double exc, G4bool stable, G4bool phys = true);
44
45 ~G4FermiFragment();
46
47 inline G4int GetA(void) const
48 {
49 return A;
50 }
51
52 inline G4int GetZ(void) const
53 {
54 return Z;
55 }
56
57 inline G4int GetSpin(void) const
58 {
59 return spin;
60 }
61
62 inline G4double GetExcitationEnergy(void) const
63 {
64 return excitEnergy;
65 }
66
67 inline G4double GetFragmentMass(void) const
68 {
69 return fragmentMass;
70 }
71
72 inline G4double GetTotalEnergy(void) const
73 {
74 return (fragmentMass + excitEnergy);
75 }
76
77 inline G4bool IsStable() const
78 {
79 return isStable;
80 }
81
82 inline G4bool IsPhysical() const
83 {
84 return isPhysical;
85 }
86
87 inline G4double GetCoulombBarrier(G4int Ares, G4int Zres, G4double Eex) const
88 {
89 return cBarrier ? cBarrier->GetCoulombBarrier(Ares, Zres, Eex)
90 *cBarrier->BarrierPenetrationFactor(Eex) : 0.0;
91 }
92
93 inline G4bool operator==(const G4FermiFragment &right) const
94 {
95 return (A == right.A && Z == right.Z &&
96 std::abs(excitEnergy - right.excitEnergy) < 0.0001);
97 }
98
99private:
100
101 G4int A;
102 G4int Z;
103 G4int spin;
104
105 G4bool isStable;
106 G4bool isPhysical;
107
108 G4double excitEnergy;
109 G4double fragmentMass;
110
111 G4CoulombBarrier* cBarrier;
112
113 G4FermiFragment(const G4FermiFragment &right) = delete;
114 const G4FermiFragment & operator=(const G4FermiFragment &right) = delete;
115 G4bool operator!=(const G4FermiFragment &right) const = delete;
116
117};
118
119
120#endif
