65G4Evaporation::G4Evaporation(G4VEvaporationChannel* photoEvaporation)
66 : G4VEvaporation(),fVerbose(0),nChannels(0),minExcitation(0.1*keV),
67 isInitialised(false)
68{
69 if(photoEvaporation) { SetPhotonEvaporation(photoEvaporation); }
70 else { SetPhotonEvaporation(new G4PhotonEvaporation()); }
71
72 channelType = fEvaporation;
73 theChannelFactory = new G4EvaporationFactory(thePhotonEvaporation);
74 InitialiseChannelFactory();
75
76 theTableOfIons = G4ParticleTable::GetParticleTable()->GetIonTable();
77 nist = G4NistManager::Instance();
78 unstableBreakUp = new G4UnstableFragmentBreakUp();
79}
80
81G4Evaporation::~G4Evaporation()
82{
83 delete unstableBreakUp;
84}

86void G4Evaporation::InitialiseChannels()
87{
88 if(isInitialised) { return; }
89
90 G4DeexPrecoParameters* param =
91 G4NuclearLevelData::GetInstance()->GetParameters();
92 minExcitation = param->GetMinExcitation();
93
94 G4DeexChannelType type = param->GetDeexChannelsType();
95 if(type == fCombined) { SetCombinedChannel(); }
96 else if(type == fGEM) { SetGEMChannel(); }
97
98 isInitialised = true;
99}
100
101void G4Evaporation::InitialiseChannelFactory()
102{
103 theChannels = theChannelFactory->GetChannel();
104 nChannels = theChannels->size();
105 probabilities.resize(nChannels, 0.0);
106
107 if(fVerbose > 0) {
108 G4cout << "### G4Evaporation::InitialiseChannelFactory for "
109 << nChannels << " channels " << this << G4endl;
110 }
111 for(size_t i=0; i<nChannels; ++i) {
112 (*theChannels)[i]->SetOPTxs(OPTxs);
113 (*theChannels)[i]->Initialise();
114 }
115}

117void G4Evaporation::SetDefaultChannel()
118{
119 if(fEvaporation != channelType) {
120 channelType = fEvaporation;
121 CleanChannels();
122 delete theChannelFactory;
123 theChannelFactory = new G4EvaporationFactory(thePhotonEvaporation);
124 InitialiseChannelFactory();
125 }
126}
127
128void G4Evaporation::SetGEMChannel()
129{
130 if(fGEM != channelType) {
131 channelType = fCombined;
132 CleanChannels();
133 delete theChannelFactory;
134 theChannelFactory = new G4EvaporationGEMFactory(thePhotonEvaporation);
135 InitialiseChannelFactory();
136 }
137}

139void G4Evaporation::SetCombinedChannel()

144 delete theChannelFactory;
145 theChannelFactory = new G4EvaporationDefaultGEMFactory(thePhotonEvaporation);

150void G4Evaporation::BreakFragment(G4FragmentVector* theResult,
151 G4Fragment* theResidualNucleus)
152{
153 if(!isInitialised) { InitialiseChannels(); }
154
155 G4double totprob, prob, oldprob = 0.0;
156 size_t maxchannel, i;
157
158 G4int Amax = theResidualNucleus->GetA_asInt();
159
160 // Starts loop over evaporated particles, loop is limited by number
161 // of nucleons
162 for(G4int ia=0; ia<Amax; ++ia) {
163
164 // g,n,p and light fragments - evaporation is finished
165 G4int Z = theResidualNucleus->GetZ_asInt();
166 G4int A = theResidualNucleus->GetA_asInt();
167 G4double Eex = theResidualNucleus->GetExcitationEnergy();
168
169 // stop deecitation loop if residual can be deexcited by FBU
170 if(theFBU->IsApplicable(Z, A, Eex)) { break; }
171
172 // check if it is stable, then finish evaporation
173 G4double abun = nist->GetIsotopeAbundance(Z, A);
174 if(fVerbose > 0) {
175 G4cout << "### G4Evaporation::BreakItUp step " << ia << " Z= " << Z
176 << " A= " << A << " Eex(MeV)= "
177 << theResidualNucleus->GetExcitationEnergy()
178 << " aban= " << abun << G4endl;
179 }
180 // stop deecitation loop in the case of a cold stable fragment
181 if(Eex <= minExcitation && abun > 0.0) { break; }
182
183 totprob = 0.0;
184 maxchannel = nChannels;
185 if(fVerbose > 1) {
186 G4cout << "### Evaporation loop #" << ia
187 << " Fragment: " << theResidualNucleus << G4endl;
188 }
189 // loop over evaporation channels
190 for(i=0; i<nChannels; ++i) {
191 prob = (*theChannels)[i]->GetEmissionProbability(theResidualNucleus);
192 if(fVerbose > 1 && prob > 0.0) {
193 G4cout << " Channel# " << i << " prob= " << prob << G4endl;
194 }
195 totprob += prob;
196 probabilities[i] = totprob;
197
198 // if two recent probabilities are near zero stop computations
199 if(i>=8 && prob > 0.0) {
200 if(prob <= totprob*1.e-8 && oldprob <= totprob*1.e-8) {
201 maxchannel = i+1;
202 break;
203 }
204 }
205 oldprob = prob;
206 }
207
208 // photon evaporation in the case of no other channels available
209 // do evaporation chain and reset total probability
210 if(0.0 < totprob && probabilities[0] == totprob) {
211 if(fVerbose > 1) {
212 G4cout << "Start chain of gamma evaporation" << G4endl;
213 }
214 (*theChannels)[0]->BreakUpChain(theResult, theResidualNucleus);
215 totprob = 0.0;
216 }
217
218 // stable fragment - evaporation is finished
219 if(0.0 == totprob) {
220
221 // if fragment is exotic, then force its decay
222 if(0.0 == abun) {
223 if(!unstableBreakUp->BreakUpChain(theResult, theResidualNucleus))
224 { break; }
225 if(fVerbose > 1) { G4cout << "$$$ Decay exotic fragment" << G4endl; }
226 } else {
227 break;
228 }
229 }
230
231 // select channel
232 totprob *= G4UniformRand();
233 // loop over evaporation channels
234 for(i=0; i<maxchannel; ++i) { if(probabilities[i] >= totprob) { break; } }
235
236 if(fVerbose > 1) { G4cout << "Channel # " << i << G4endl; }
237 G4Fragment* frag = (*theChannels)[i]->EmittedFragment(theResidualNucleus);
238 if(fVerbose > 1 && frag) { G4cout << " " << *frag << G4endl; }
239
240 // normaly a fragment should be created
241 if(frag) { theResult->push_back(frag); }
242 else { break; }
243 }
244 // loop is stopped, residual fragment is added to the results
245 theResult->push_back(theResidualNucleus);
246}
