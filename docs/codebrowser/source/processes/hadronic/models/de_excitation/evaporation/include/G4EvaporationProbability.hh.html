32#ifndef G4EvaporationProbability_h
33#define G4EvaporationProbability_h 1
34
35#include "G4VEmissionProbability.hh"
36
37class G4VCoulombBarrier;
38class G4NuclearLevelData;
39
40class G4EvaporationProbability : public G4VEmissionProbability
41{
42public:
43
44 explicit G4EvaporationProbability(G4int anA, G4int aZ,
45 G4double aGamma,
46 G4VCoulombBarrier *);
47
48 virtual ~G4EvaporationProbability();
49
50 // not used for evaporation
51 virtual G4double EmissionProbability(const G4Fragment& fragment,
52 G4double maxKineticEnergy);
53
54 // general method used for evaporation
55 G4double TotalProbability(const G4Fragment& fragment,
56 G4double minKineticEnergy,
57 G4double maxKineticEnergy,
58 G4double CoulombBarrier = 0.0);
59
60 // main method to compute full probability for OPTx > 2
61 virtual G4double ComputeProbability(G4double K, G4double kBarrier);
62
63 // Samples fragment kinetic energy and excitation energy
64 // of the residual nucleaus
65 G4double SampleKineticEnergy(G4double minKineticEnergy,
66 G4double maxKineticEnergy,
67 G4double CoulombBarrier = 0.0);
68
69protected:
70
71 virtual G4double CalcAlphaParam(const G4Fragment & fragment)=0 ;
72
73 virtual G4double CalcBetaParam(const G4Fragment & fragment)=0 ;
74
75private:
76
77 G4double CrossSection(G4double K, G4double CoulombBarrier);
78
79 // Copy constructor
80 G4EvaporationProbability(const G4EvaporationProbability &right) = delete;
81
82 const G4EvaporationProbability & operator=
83 (const G4EvaporationProbability &right) = delete;
84 G4bool operator==(const G4EvaporationProbability &right) const = delete;
85 G4bool operator!=(const G4EvaporationProbability &right) const = delete;
86
87 G4NuclearLevelData* fLevelData;
88
89 G4int fragA;
90 G4int fragZ;
91 G4int resA;
92 G4int resZ;
93 G4int index;
94
95 G4double resA13;
96 G4double muu;
97 G4double partMass;
98 G4double resMass;
99 G4double Mass;
100 G4double U, delta0, delta1, a0;
101
102 // Gamma is A_f(2S_f+1) factor, where A_f is fragment atomic
103 // number and S_f is fragment spin
104 G4double Gamma;
105 G4double pcoeff;
106
107 G4double probmax;
108};
109
110#endif
