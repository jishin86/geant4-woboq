53G4EvaporationChannel::G4EvaporationChannel(G4int anA, G4int aZ,
54 const G4String & aName,
55 G4EvaporationProbability* aprob,
56 G4VCoulombBarrier* barrier):
57 G4VEvaporationChannel(aName),
58 theA(anA),
59 theZ(aZ),
60 theProbability(aprob),
61 theCoulombBarrier(barrier)
62{
63 ResA = ResZ = 0;
64 Mass = CoulombBarrier = MinKinEnergy = MaxKinEnergy = EmissionProbability = 0.0;
65 EvapMass = G4NucleiProperties::GetNuclearMass(theA, theZ);
66 pairingCorrection = G4NuclearLevelData::GetInstance()->GetPairingCorrection();
67}

78G4double G4EvaporationChannel::GetEmissionProbability(G4Fragment* fragment)
79{
80 G4int FragA = fragment->GetA_asInt();
81 G4int FragZ = fragment->GetZ_asInt();
82 ResA = FragA - theA;
83 ResZ = FragZ - theZ;
84
85 G4double FragmentMass = fragment->GetGroundStateMass();
86 G4double ExEnergy = fragment->GetExcitationEnergy();
87 Mass = FragmentMass + ExEnergy;
88 //G4cout << "G4EvaporationChannel::Initialize Z= " << theZ << " A= " << theA
89 // << " FragZ= " << FragZ << " FragA= " << FragA << G4endl;
90 EmissionProbability = 0.0;
91
92 // Only channels which are physically allowed are taken into account
93 if (ResA >= ResZ && ResZ > 0 && ResA >= theA) {
94
95 //Effective excitation energy
96 G4double ResMass = G4NucleiProperties::GetNuclearMass(ResA, ResZ);
97
98 CoulombBarrier = (0 == theZ) ? 0.0 :
99 theCoulombBarrier->GetCoulombBarrier(ResA,ResZ,ExEnergy);
100
101 G4double delta0 =
102 std::max(0.0,pairingCorrection->GetPairingCorrection(FragA,FragZ));
103 G4double delta1 =
104 std::max(0.0,pairingCorrection->GetPairingCorrection(ResA,ResZ));
105 ResMass += delta1;
106 /*
107 G4cout << "ExEnergy= " << ExEnergy << " Ec= " << CoulombBarrier
108 << " delta0= " << delta0 << " delta1= " << delta1
109 << " Free= " << Mass - ResMass - EvapMass
110 << G4endl;
111 */
112 // for OPTxs >0 penetration under the barrier is taken into account
113 // G4double elim = (0 == OPTxs) ? CoulombBarrier : CoulombBarrier*0.5;
114 static const G4double dCB = 3.5*CLHEP::MeV;
115 G4double elim = (0 == OPTxs) ? CoulombBarrier : CoulombBarrier - dCB*theZ;
116 if(ExEnergy >= delta0 && Mass >= ResMass + EvapMass + elim) {
117 G4double xm2 = (Mass - EvapMass)*(Mass - EvapMass);
118 G4double xm = Mass - EvapMass - elim;
119 MinKinEnergy = (0.0 >= elim) ? 0.0 : std::max(0.5*(xm2 - xm*xm)/Mass, 0.0);
120 MaxKinEnergy = std::max(0.5*(xm2 - ResMass*ResMass)/Mass, 0.0);
121 //G4cout << "Emin= " << MinKinEnergy << " Emax= " << MaxKinEnergy
122 // << " xm= " << xm << G4endl;
123 EmissionProbability = theProbability->
124 TotalProbability(*fragment, MinKinEnergy, MaxKinEnergy, CoulombBarrier);
125 }
126 }
127 //G4cout << "G4EvaporationChannel:: probability= "
128 // << EmissionProbability << G4endl;
129 return EmissionProbability;
130}

132G4Fragment* G4EvaporationChannel::EmittedFragment(G4Fragment* theNucleus)
133{
134 G4Fragment* evFragment = nullptr;
135 G4double ekin = 0.0;
136 if(ResA <= 4 &&
137 ((ResA == 4 && ResZ == 2) || (ResA == 3 && ResZ == 2) ||
138 (ResA == 3 && ResZ == 1) || (ResA == 2 && ResZ == 1) ||
139 (ResA == 1 && ResZ == 1) || (ResA == 1 && ResZ == 0) )) {
140 G4double mres = G4NucleiProperties::GetNuclearMass(ResA, ResZ);
141 ekin = 0.5*(Mass*Mass - mres*mres + EvapMass*EvapMass)/Mass - EvapMass;
142 } else {
143 ekin = theProbability->SampleKineticEnergy(MinKinEnergy, MaxKinEnergy,
144 CoulombBarrier);
145 }
146 G4LorentzVector lv0 = theNucleus->GetMomentum();
147 G4LorentzVector lv(std::sqrt(ekin*(ekin + 2.0*EvapMass))*G4RandomDirection(),
148 ekin + EvapMass);
149 lv.boost(lv0.boostVector());
150
151 evFragment = new G4Fragment(theA, theZ, lv);
152 lv0 -= lv;
153 theNucleus->SetZandA_asInt(ResZ, ResA);
154 theNucleus->SetMomentum(lv0);
155
156 return evFragment;
157}
