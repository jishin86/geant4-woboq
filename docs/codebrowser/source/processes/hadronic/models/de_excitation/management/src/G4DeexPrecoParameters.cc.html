54void G4DeexPrecoParameters::SetDefaults()
55{
56#ifdef G4MULTITHREADED
57 G4MUTEXLOCK(&G4DeexPrecoParameters::deexPrecoMutex);
58#endif
59 fStateManager = G4StateManager::GetStateManager();
60 theMessenger = new G4DeexParametersMessenger(this);
61
62 fLevelDensity = 0.075/CLHEP::MeV;
63 fR0 = 1.5*CLHEP::fermi;
64 fTransitionsR0 = 0.6*CLHEP::fermi;
65 fFBUEnergyLimit = 20.0*CLHEP::MeV;
66 fFermiEnergy = 35.0*CLHEP::MeV;
67 fPrecoLowEnergy = 0.1*CLHEP::MeV;
68 fPhenoFactor = 1.0;
69 fMinExcitation = 10*CLHEP::eV;
70 fMaxLifeTime = 1000*CLHEP::second;
71 fMinExPerNucleounForMF = 100*CLHEP::GeV;
72 fMinZForPreco = 3;
73 fMinAForPreco = 5;
74 fPrecoType = 3;
75 fDeexType = 3;
76 fTwoJMAX = 10;
77 fNeverGoBack = false;
78 fUseSoftCutoff = false;
79 fUseCEM = true;
80 fUseGNASH = false;
81 fUseHETC = false;
82 fUseAngularGen = true;
83 fPrecoDummy = false;
84 fCorrelatedGamma = false;
85 fStoreAllLevels = false;
86 fInternalConversion = true;
87 fLD = true;
88 fDeexChannelType = fCombined;
89 fInternalConversionID =
90 G4PhysicsModelCatalog::Register("e-InternalConvertion");
91#ifdef G4MULTITHREADED
92 G4MUTEXUNLOCK(&G4DeexPrecoParameters::deexPrecoMutex);
93#endif
94}

264std::ostream& G4DeexPrecoParameters::StreamInfo(std::ostream& os) const
265{
266 static const G4String namm[4] = {"Evaporation","GEM","Evaporation+GEM","Dummy"};
267 static const G4int nmm[4] = {8, 68, 68, 0};
268 size_t idx = (size_t)fDeexChannelType;
269
270 G4int prec = os.precision(5);
271 os << "=======================================================================" << "\n";
272 os << "====== Pre-compound/De-excitation Physics Parameters ========" << "\n";
273 os << "=======================================================================" << "\n";
274 os << "Type of pre-compound inverse x-section " << fPrecoType << "\n";
275 os << "Pre-compound model active " << (!fPrecoDummy) << "\n";
276 os << "Pre-compound low energy (MeV) "
277 << fPrecoLowEnergy/CLHEP::MeV << "\n";
278 os << "Type of de-excitation inverse x-section " << fDeexType << "\n";
279 os << "Type of de-excitation factory " << namm[idx] << "\n";
280 os << "Number of de-excitation channels " << nmm[idx] << "\n";
281 os << "Min excitation energy (keV) "
282 << fMinExcitation/CLHEP::keV << "\n";
283 os << "Min energy per nucleon for multifragmentation (MeV) "
284 << fMinExPerNucleounForMF/CLHEP::MeV << "\n";
285 os << "Limit excitation energy for Fermi BreakUp (MeV) "
286 << fFBUEnergyLimit/CLHEP::MeV << "\n";
287 os << "Level density (1/MeV) "
288 << fLevelDensity*CLHEP::MeV << "\n";
289 os << "Model of level density flag " << fLD << "\n";
290 os << "Time limit for long lived isomeres (ns) "
291 << fMaxLifeTime/CLHEP::ns << "\n";
292 os << "Internal e- conversion flag "
293 << fInternalConversion << "\n";
294 os << "Store e- internal conversion data " << fStoreAllLevels << "\n";
295 os << "Electron internal conversion ID "
296 << fInternalConversionID << "\n";
297 os << "Correlated gamma emission flag " << fCorrelatedGamma << "\n";
298 os << "Max 2J for sampling of angular correlations " << fTwoJMAX << "\n";
299 os << "=======================================================================" << "\n";
300 os.precision(prec);
301 return os;
302}

304void G4DeexPrecoParameters::Dump() const
305{
306 if (G4Threading::IsMasterThread()) { StreamInfo(G4cout); }
307}

309std::ostream& operator<< (std::ostream& os, const G4DeexPrecoParameters& par)

312}

314G4bool G4DeexPrecoParameters::IsLocked() const
315{
316 return (!G4Threading::IsMasterThread() ||
317 (fStateManager->GetCurrentState() != G4State_PreInit));
318}
