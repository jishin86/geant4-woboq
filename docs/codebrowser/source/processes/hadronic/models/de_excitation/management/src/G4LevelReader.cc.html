55G4LevelReader::G4LevelReader(G4NuclearLevelData* ptr)
56 : fData(ptr),fVerbose(0),fLevelMax(632),fTransMax(145)
57{
58 fAlphaMax = (G4float)1.e15;
59 fParam = fData->GetParameters();
60 fTimeFactor = CLHEP::second/G4Pow::GetInstance()->logZ(2);
61 char* directory = getenv("G4LEVELGAMMADATA");
62 if(directory) {
63 fDirectory = directory;
64 } else {
65 G4Exception("G4LevelReader()","had0707",FatalException,
66 "Environment variable G4LEVELGAMMADATA is not defined");
67 fDirectory = "";
68 }
69 fFile = fDirectory + "/z100.a200";
70 fPol = " ";
71 for(G4int i=0; i<10; ++i) { fICC[i] = 0.0f; }
72 for(G4int i=0; i<nbufmax; ++i) { buffer[i] = ' '; }
73 for(G4int i=0; i<nbuf1; ++i) { buff1[i] = ' '; }
74 for(G4int i=0; i<nbuf2; ++i) { buff2[i] = ' '; }
75 bufp[0] = bufp[1] = bufp[2] = ' ';
76
77 fEnergy = fCurrEnergy = fTrEnergy = fTime = 0.0;
78 fProb = fSpin = fAlpha = fRatio = fNorm1 = 0.0f;
79
80 vTrans.resize(fTransMax,0);
81 vRatio.resize(fTransMax,0.0f);
82 vGammaCumProbability.resize(fTransMax,0.0f);
83 vGammaProbability.resize(fTransMax,0.0f);
84 vShellProbability.resize(fTransMax,nullptr);
85
86 vEnergy.resize(fLevelMax,0.0);
87 vSpin.resize(fLevelMax,0);
88 vLevel.resize(fLevelMax,nullptr);
89}

91G4bool G4LevelReader::ReadData(std::istringstream& stream, G4double& x)

93 stream >> x;
94 return stream.fail() ? false : true;
95}

97G4bool G4LevelReader::ReadDataItem(std::istream& dataFile, G4double& x)
98{
99 x = 0.0;
100 for(G4int i=0; i<nbufmax; ++i) { buffer[i] = ' '; }
101 G4bool okay = true;
102 dataFile >> buffer;
103 if(dataFile.fail()) { okay = false; }
104 else { x = strtod(buffer, 0); }
105
106 return okay;
107}

144const std::vector<G4float>* G4LevelReader::NormalizedICCProbability(G4int Z)
145{
146 std::vector<G4float>* vec = nullptr;
147 G4int LL = 3;
148 G4int M = 5;
149 G4int N = 1;
150 if(Z <= 27) {
151 M = N = 0;
152 if(Z <= 4) {
153 LL = 1;
154 } else if(Z <= 6) {
155 LL = 2;
156 } else if(Z <= 10) {
157 } else if(Z <= 12) {
158 M = 1;
159 } else if(Z <= 17) {
160 M = 2;
161 } else if(Z == 18) {
162 M = 3;
163 } else if(Z <= 20) {
164 M = 3;
165 N = 1;
166 } else {
167 M = 4;
168 N = 1;
169 }
170 if(LL < 3) { for(G4int i=LL+1; i<=4; ++i) { fICC[i] = 0.0f; } }
171 if(M < 5) { for(G4int i=M+4; i<=8; ++i) { fICC[i] = 0.0f; } }
172 if(N < 1) { fICC[9] = 0.0f; }
173 }
174 G4float norm = 0.0f;
175 for(G4int i=0; i<10; ++i) {
176 norm += fICC[i];
177 fICC[i] = norm;
178 }
179 if(norm == 0.0f && fAlpha > 0.0f) {
180 fICC[0] = norm = 1.0f;
181 }
182 if(norm > 0.0f) {
183 norm = 1.0f/norm;
184 vec = new std::vector<G4float>;
185 G4float x;
186 for(G4int i=0; i<10; ++i) {
187 x = fICC[i]*norm;
188 if(x > 0.995f || 9 == i) {
189 vec->push_back(1.0f);
190 break;
191 }
192 vec->push_back(x);
193 }
194 if (fVerbose > 2) {
195 G4int prec = G4cout.precision(3);
196 G4cout << "# InternalConv: ";
197 G4int nn = vec->size();
198 for(G4int i=0; i<nn; ++i) { G4cout << " " << (*vec)[i]; }
199 G4cout << G4endl;
200 G4cout.precision(prec);
201 }
202 }
203 return vec;
204}

206const G4LevelManager*
207G4LevelReader::CreateLevelManager(G4int Z, G4int A)
208{
209 std::ostringstream ss;
210 ss << "/z" << Z << ".a" << A;

214

218const G4LevelManager*
219G4LevelReader::MakeLevelManager(G4int Z, G4int A, const G4String& filename)
220{
221 fFile = filename;
222 std::ifstream infile(filename, std::ios::in);
223 if (!infile.is_open()) {
224 G4ExceptionDescription ed;
225 ed << "User file for Z= " << Z << " A= " << A
226 << " <" << fFile << "> is not opened!";
227 G4Exception("G4LevelReader::MakeLevelManager(..)","had014",
228 FatalException, ed, "");
229 return nullptr;
230 }
231 return LevelManager(Z, A, 0, infile);
232}

234const G4LevelManager*
235G4LevelReader::LevelManager(G4int Z, G4int A, G4int nlev,
236 std::ifstream& infile)
237{
238 // file is not opened
239 if (!infile.is_open()) {
240 if(Z < 6 || fVerbose > 0) {
241 G4ExceptionDescription ed;
242 ed << " for Z= " << Z << " A= " << A
243 << " <" << fFile << "> is not opened!";
244 G4Exception("G4LevelReader::LevelManager(..)","had014",
245 FatalException, ed, "");
246 }
247 return nullptr;
248 }
249 if (fVerbose > 0) {
250 G4cout << "G4LevelReader: open file for Z= "
251 << Z << " A= " << A
252 << " <" << fFile << ">" << G4endl;
253 }
254
255 G4bool allLevels = fParam->StoreICLevelData();
256
257 G4int nlevels = (0 == nlev) ? fLevelMax : nlev;
258 if(fVerbose > 0) {
259 G4cout << "## New isotope Z= " << Z << " A= " << A;
260 if(nlevels < fLevelMax) { G4cout << " Nlevels= " << nlevels; }
261 G4cout << G4endl;
262 }
263 if(nlevels > fLevelMax) {
264 fLevelMax = nlevels;
265 vEnergy.resize(fLevelMax,0.0);
266 vSpin.resize(fLevelMax,0);
267 vLevel.resize(fLevelMax,nullptr);
268 }
269 G4int ntrans(0), i1, i, k, kk;
270 G4int i2; // Level number at which transition ends
271 G4int tnum; // Multipolarity index
272 G4String xf(" ");
273 G4double ener, tener;
274
275 for(i=0; i<nlevels; ++i) {
276 infile >> i1 >> xf; // Level number and floating level
277 //G4cout << "New line: i1= " << i1 << " xf= <" << xf << "> " << G4endl;
278 if(infile.eof()) {
279 if(fVerbose > 1) {
280 G4cout << "### End of file Z= " << Z << " A= " << A
281 << " Nlevels= " << i << G4endl;
282 }
283 break;
284 }
285 if(i1 != i) {
286 G4ExceptionDescription ed;
287 ed << " G4LevelReader: wrong data file for Z= " << Z << " A= " << A
288 << " level #" << i << " has index " << i1 << G4endl;
289 G4Exception("G4LevelReader::LevelManager(..)","had014",
290 JustWarning, ed, "Check G4LEVELGAMMADATA");
291 break;
292 }
293
294 if(!(ReadDataItem(infile,ener) &&
295 ReadDataItem(infile,fTime) &&
296 ReadDataItem(infile,fSpin) &&
297 ReadDataItem(infile,ntrans))) {
298 if(fVerbose > 1) {
299 G4cout << "### End of file Z= " << Z << " A= " << A
300 << " Nlevels= " << i << G4endl;
301 }
302 break;
303 }
304 ener *= CLHEP::keV;
305 for(k=0; k<nfloting; ++k) {
306 if(xf == fFloatingLevels[k]) {
307 break;
308 }
309 }
310 // if a previous level has not transitions it may be ignored
311 if(0 < i) {
312 // protection
313 if(ener < vEnergy[i-1]) {
314 G4cout << "### G4LevelReader: broken level " << i
315 << " E(MeV)= " << ener << " < " << vEnergy[i-1]
316 << " for isotope Z= " << Z << " A= "
317 << A << " level energy increased" << G4endl;
318 ener = vEnergy[i-1];
319 }
320 }
321 vEnergy[i] = ener;
322 if(fTime > 0.0f) { fTime *= fTimeFactor; }
323 if(fSpin > 48.0f) { fSpin = 0.0f; }
324 vSpin[i] = (G4int)(100 + fSpin + fSpin) + k*100000;
325 if(fVerbose > 1) {
326 G4cout << " Level #" << i1 << " E(MeV)= " << ener/CLHEP::MeV
327 << " LTime(s)= " << fTime << " 2S= " << vSpin[i]
328 << " meta= " << vSpin[i]/100000 << " idx= " << i
329 << " ntr= " << ntrans << G4endl;
330 }
331 vLevel[i] = nullptr;
332 if(ntrans == 0 && fTime < 0.0) {
333 vLevel[i] = new G4NucLevel(0, fTime,
334 vTrans,
335 vGammaCumProbability,
336 vGammaProbability,
337 vRatio,
338 vShellProbability);
339 } else if(ntrans > 0) {
340
341 // there are transitions
342 if(ntrans > fTransMax) {
343 fTransMax = ntrans;
344 vTrans.resize(fTransMax);
345 vRatio.resize(fTransMax);
346 vGammaCumProbability.resize(fTransMax);
347 vGammaProbability.resize(fTransMax);
348 vShellProbability.resize(fTransMax);
349 }
350 fNorm1 = 0.0f;
351 for(G4int j=0; j<ntrans; ++j) {
352
353 if(!(ReadDataItem(infile,i2) &&
354 ReadDataItem(infile,tener) &&
355 ReadDataItem(infile,fProb) &&
356 ReadDataItem(infile,tnum) &&
357 ReadDataItem(infile,vRatio[j]) &&
358 ReadDataItem(infile,fAlpha))) {
359 //infile >>i2 >> tener >> fProb >> vTrans[j] >> fRatio >> fAlpha;
360 //if(infile.fail()) {
361 if(fVerbose > 0) {
362 G4cout << "### Fail to read transition j= " << j
363 << " Z= " << Z << " A= " << A << G4endl;
364 }
365 break;
366 }
367 if(i2 >= i) {
368 G4cout << "### G4LevelReader: broken transition " << j
369 << " from level " << i << " to " << i2
370 << " for isotope Z= " << Z << " A= "
371 << A << " - use ground level" << G4endl;
372 i2 = 0;
373 }
374 vTrans[j] = i2*10000 + tnum;
375 fAlpha = std::min(std::max(fAlpha,0.f), fAlphaMax);
376 G4float x = 1.0f + fAlpha;
377 fNorm1 += x*fProb;
378 vGammaCumProbability[j] = fNorm1;
379 vGammaProbability[j] = 1.0f/x;
380 vShellProbability[j] = nullptr;
381 if(fVerbose > 1) {
382 G4int prec = G4cout.precision(4);
383 G4cout << "### Transition #" << j << " to level " << i2
384 << " i2= " << i2 << " Etrans(MeV)= " << tener*CLHEP::keV
385 << " fProb= " << fProb << " MultiP= " << tnum
386 << " fMpRatio= " << fRatio << " fAlpha= " << fAlpha
387 << G4endl;
388 G4cout.precision(prec);
389 }
390 if(fAlpha > 0.0f) {
391 for(k=0; k<10; ++k) {
392 //infile >> fICC[k];
393 if(!ReadDataItem(infile,fICC[k])) {
394 //if(infile.fail()) {
395 if(fVerbose > 0) {
396 G4cout << "### Fail to read convertion coeff k= " << k
397 << " for transition j= " << j
398 << " Z= " << Z << " A= " << A << G4endl;
399 }
400 for(kk=k; kk<10; ++kk) { fICC[kk] = 0.f; }
401 break;
402 }
403 }
404 if(allLevels) {
405 vShellProbability[j] = NormalizedICCProbability(Z);
406 if(!vShellProbability[j]) { vGammaProbability[j] = 1.0f; }
407 }
408 }
409 }
410 if(0.0f < fNorm1) { fNorm1 = 1.0f/fNorm1; }
411 G4int nt = ntrans - 1;
412 for(k=0; k<nt; ++k) {
413 vGammaCumProbability[k] *= fNorm1;
414 if(fVerbose > 2) {
415 G4cout << "Probabilities[" << k
416 << "]= " << vGammaCumProbability[k]
417 << " " << vGammaProbability[k]
418 << " idxTrans= " << vTrans[k]/10000
419 << G4endl;
420 }
421 }
422 vGammaCumProbability[nt] = 1.0f;
423 if(fVerbose > 2) {
424 G4cout << "Probabilities[" << nt << "]= "
425 << vGammaCumProbability[nt]
426 << " " << vGammaProbability[nt]
427 << " IdxTrans= " << vTrans[nt]/10000
428 << G4endl;
429 }
430 if(fVerbose > 1) {
431 G4cout << " New G4NucLevel: Ntrans= " << ntrans
432 << " Time(ns)= " << fTime << G4endl;
433 }
434 vLevel[i] = new G4NucLevel((size_t)ntrans, fTime,
435 vTrans,
436 vGammaCumProbability,
437 vGammaProbability,
438 vRatio,
439 vShellProbability);
440 }
441 }
442 G4LevelManager* lman = nullptr;
443 if(1 <= i) {
444 lman = new G4LevelManager(Z, A, (size_t)i,vEnergy,vSpin,vLevel);
445 if(fVerbose > 0) {
446 G4cout << "=== Reader: new manager for Z= " << Z << " A= " << A
447 << " Nlevels= " << i << " E[0]= "
448 << vEnergy[0]/CLHEP::MeV << " MeV E1= "
449 << vEnergy[i-1]/CLHEP::MeV << " MeV "
450 << G4endl;
451 }
452 }
453
454 return lman;
455}
