572G4double
573G4NuclearLevelData::FindLevel(G4int Z, G4int A, G4double resMass,
574 G4double Mass, G4double partMass, G4double ekin)
575{
576 G4double T = ekin;
577 G4double E2 = (Mass - partMass)*(Mass - partMass);
578 G4double Eex = std::sqrt(E2 - 2.0*Mass*T) - resMass;
579 if(Eex <= GetMaxLevelEnergy(Z, A)) {
580 if(Eex <= 0.0) {
581 Eex = 0.0;
582 T = (E2 - resMass*resMass)*0.5/Mass;
583 } else {
584 const G4LevelManager* man = GetLevelManager(Z, A);
585 if(man != nullptr) {
586 size_t idx = man->NearestLevelIndex(Eex, man->NumberOfTransitions());
587 for(;;) {
588 Eex = (G4double)man->LevelEnergy(idx);
589 T = (E2 - (resMass+Eex)*(resMass+Eex))*0.5/Mass;
590 if(T >= 0.0 || 0 == idx) { break; }
591 --idx;
592 }
593 }
594 }
595 T = std::max(T, 0.0);
596 }
597 return T;
598}
