48#ifndef G4ExcitationHandler_h
49#define G4ExcitationHandler_h 1
50
51#include "globals.hh"
52#include "G4Fragment.hh"
53#include "G4ReactionProductVector.hh"
54#include "G4IonTable.hh"
55#include "G4DeexPrecoParameters.hh"
56
57class G4VMultiFragmentation;
58class G4VFermiBreakUp;
59class G4VEvaporation;
60class G4VEvaporationChannel;
61class G4NistManager;
62
63class G4ExcitationHandler
64{
65public:
66
67 explicit G4ExcitationHandler();
68 ~G4ExcitationHandler();
69
70 G4ReactionProductVector* BreakItUp(const G4Fragment &theInitialState);
71
72 // short model description used for automatic web documentation
73 void ModelDescription(std::ostream& outFile) const;
74
75 void Initialise();
76
77 // user defined sub-models
78 // deletion is responsibility of this handler if isLocal=true
79 void SetEvaporation(G4VEvaporation* ptr, G4bool isLocal=false);
80 void SetMultiFragmentation(G4VMultiFragmentation* ptr);
81 void SetFermiModel(G4VFermiBreakUp* ptr);
82 void SetPhotonEvaporation(G4VEvaporationChannel* ptr);
83 void SetDeexChannelsType(G4DeexChannelType val);
84
85 //======== Obsolete methods to be removed =====
86
87 // parameters of sub-models
88 inline void SetMaxZForFermiBreakUp(G4int aZ);
89 inline void SetMaxAForFermiBreakUp(G4int anA);
90 inline void SetMaxAandZForFermiBreakUp(G4int anA,G4int aZ);
91 inline void SetMinEForMultiFrag(G4double anE);
92
93 // access methods
94 inline G4VEvaporation* GetEvaporation();
95 inline G4VMultiFragmentation* GetMultiFragmentation();
96 inline G4VFermiBreakUp* GetFermiModel();
97 inline G4VEvaporationChannel* GetPhotonEvaporation();
98
99 // for inverse cross section choice
100 inline void SetOPTxs(G4int opt);
101 // for superimposed Coulomb Barrir for inverse cross sections
102 inline void UseSICB();
103
104 //==============================================
105
106private:
107
108 void SetParameters();
109
110 G4ExcitationHandler(const G4ExcitationHandler &right) = delete;
111 const G4ExcitationHandler & operator
112 =(const G4ExcitationHandler &right) = delete;
113 G4bool operator==(const G4ExcitationHandler &right) const = delete;
114 G4bool operator!=(const G4ExcitationHandler &right) const = delete;
115
116 G4VEvaporation* theEvaporation;
117 G4VMultiFragmentation* theMultiFragmentation;
118 G4VFermiBreakUp* theFermiModel;
119 G4VEvaporationChannel* thePhotonEvaporation;
120
121 const G4ParticleDefinition* electron;
122 G4int icID;
123
124 G4int maxZForFermiBreakUp;
125 G4int maxAForFermiBreakUp;
126 G4double minEForMultiFrag;
127 G4double minExcitation;
128
129 G4IonTable* theTableOfIons;
130 G4NistManager* nist;
131
132 G4int fVerbose;
133 G4bool isInitialised;
134 G4bool isEvapLocal;
135 G4bool isActive;
136
137 // list of fragments to store final result
138 std::vector<G4Fragment*> theResults;
139
140 // list of fragments to store intermediate result
141 std::vector<G4Fragment*> results;
142
143 // list of fragments to apply PhotonEvaporation
144 std::vector<G4Fragment*> thePhotoEvapList;
145
146 // list of fragments to apply Evaporation or Fermi Break-Up
147 std::vector<G4Fragment*> theEvapList;
148};

153}
154
155inline void G4ExcitationHandler::SetMaxAForFermiBreakUp(G4int anA)
156{
157 maxAForFermiBreakUp = anA;
158}
159

161{
162 SetMaxAForFermiBreakUp(anA);
163 SetMaxZForFermiBreakUp(aZ);
164}
165
166inline void G4ExcitationHandler::SetMinEForMultiFrag(G4double anE)
167{
168 minEForMultiFrag = anE;
169}
170

179}

184}

188 return thePhotonEvaporation;
189}

192{}
