38G4FissionParameters::G4FissionParameters()
39 : A1(134),A2(141),A3((A1 + A2)*0.5),As(0.0),Sigma1(0.0),Sigma2(0.0),
40 SigmaS(0.),w(0.0)
41{}
42
43G4FissionParameters::~G4FissionParameters()
44{}
45
46void G4FissionParameters::DefineParameters(G4int A, G4int Z, G4double ExEnergy,
47 G4double FissionBarrier)
48{
49 // to avoid usage of units
50 G4double U = ExEnergy/CLHEP::MeV;
51
52 As = A*0.5;
53
54 if (A <= 235) { Sigma2 = 5.6; }
55 else { Sigma2 = 5.6 + 0.096*(A-235); }
56
57 Sigma1 = 0.5*Sigma2;
58
59 //JMQ 310509
60 // if (SigmaS > 20.0) SigmaS = 20.0;
61 // SigmaS*=1.3;
62 //JMQ 301009: retuning (after CEM transition prob.have been chosen as default)
63 SigmaS = 0.8*G4Exp(0.00553*U + 2.1386);
64
65 G4double x1 = (A1-As)/Sigma1;
66 G4double x2 = (A2-As)/Sigma2;
67 G4double FasymAsym = 2*G4Exp(-0.5*x2*x2) + G4Exp(-0.5*x1*x1);
68
69 G4double x3 = (As-A3)/SigmaS;
70 G4double FsymA1A2 = G4Exp(-0.5*x3*x3);
71
72 G4double wa = 0.0;
73 w = 0.0;
74 if (Z >= 90) {
75 if (U <= 16.25) { wa = G4Exp(0.5385*U-9.9564); }
76 else { wa = G4Exp(0.09197*U-2.7003); }
77 } else if (Z == 89) {
78 wa = G4Exp(0.09197*U-1.0808);
79 } else if (Z >= 82) {
80 G4double X = std::max(0.0, FissionBarrier/CLHEP::MeV - 7.5);
81 wa = G4Exp(0.09197*(U-X) - 1.0808);
82 } else { // Z < 82
83 w = 1001.0;
84 }
85
86 if (w == 0.0) {
87 G4double w1 = std::max(1.03*wa - FasymAsym, 0.0001);
88 G4double w2 = std::max(1.0 - FsymA1A2*wa, 0.0001);
89
90 w = w1/w2;
91
92 if (82 <= Z && Z < 89 && A < 227) { w *= G4Exp(0.3*(227-A)); }
93 }
94}
