32#ifndef G4FissionParameters_h
33#define G4FissionParameters_h 1
34
35#include "globals.hh"
36
37class G4FissionParameters
38{
39public:
40
41 G4FissionParameters();
42
43 ~G4FissionParameters();
44
45 void DefineParameters(G4int A, G4int Z, G4double ExEnergy,
46 G4double FissionBarrier);
47
48public:
49
50 inline G4int GetA1(void) const { return A1; }
51 inline G4int GetA2(void) const { return A2; }
52
53 inline G4double GetAs(void) const { return As; }
54 inline G4double GetSigma1(void) const { return Sigma1; }
55 inline G4double GetSigma2(void) const { return Sigma2; }
56 inline G4double GetSigmaS(void) const { return SigmaS; }
57 inline G4double GetW(void) const { return w; }
58
59private:
60
61 G4FissionParameters(const G4FissionParameters &right);
62 const G4FissionParameters & operator=(const G4FissionParameters &right);
63 G4bool operator==(const G4FissionParameters &right) const;
64 G4bool operator!=(const G4FissionParameters &right) const;
65
66 // Mean numbers of the corresponding Gaussians for assymmetric
67 // fission
68 G4int A1;
69 G4int A2;
70 G4double A3;
71
72 // Mean number for symmetric fission
73 G4double As;
74
75 // Dispersions of the corresponding Gaussians for assymmetric
76 // fission
77 G4double Sigma1;
78 G4double Sigma2;
79
80 // Dispersion for symmetric fission
81 G4double SigmaS;
82
83 // Weight which determines the relative contribution of symmetric
84 // and assymmetric components
85 G4double w;
86};
87
88
89#endif
