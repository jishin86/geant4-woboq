31#ifndef G4FissionProbability_h
32#define G4FissionProbability_h 1
33
34#include "G4VEmissionProbability.hh"
35#include "G4VLevelDensityParameter.hh"
36
37class G4PairingCorrection;
38
39class G4FissionProbability : public G4VEmissionProbability
40{
41public:
42 // Default constructor
43 explicit G4FissionProbability();
44
45 ~G4FissionProbability() override;
46
47 G4double EmissionProbability(const G4Fragment & fragment,
48 G4double MaximalKineticEnergy) override;
49
50 inline void
51 SetEvaporationLevelDensityParameter(G4VLevelDensityParameter* aLevelDensity)
52 {
53 if (ownEvapLDP) delete theEvapLDP;
54 theEvapLDP = aLevelDensity;
55 ownEvapLDP = false;
56 }
57
58 inline void
59 SetFissionLevelDensityParameter(G4VLevelDensityParameter * aLevelDensity)
60 {
61 if (ownFissLDP) delete theFissLDP;
62 theFissLDP = aLevelDensity;
63 ownFissLDP = false;
64 }
65
66 // Copy constructor
67 G4FissionProbability(const G4FissionProbability &right) = delete;
68 const G4FissionProbability & operator=(const G4FissionProbability &right) = delete;
69 G4bool operator==(const G4FissionProbability &right) const = delete;
70 G4bool operator!=(const G4FissionProbability &right) const = delete;
71
72private:
73
74 G4VLevelDensityParameter *theEvapLDP;
75 G4VLevelDensityParameter *theFissLDP;
76 G4PairingCorrection* fPairCorr;
77 bool ownEvapLDP;
78 bool ownFissLDP;
79
80
81};

84#endif
