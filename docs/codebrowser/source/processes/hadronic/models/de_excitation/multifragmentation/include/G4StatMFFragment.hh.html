31#ifndef G4StatMFFragment_h
32#define G4StatMFFragment_h 1
33
34#include "G4StatMFParameters.hh"
35#include "G4ThreeVector.hh"
36#include "G4ParticleTable.hh"
37#include "G4IonTable.hh"
38#include "G4Fragment.hh"
39
40class G4StatMFFragment {
41
42public:
43 // Constructor
44 G4StatMFFragment(G4int anA, G4int aZ) :
45 theA(anA),theZ(aZ),
46 _position(0.0,0.0,0.0),
47 _momentum(0.0,0.0,0.0)
48 {}
49
50
51 // Destructor
52 virtual ~G4StatMFFragment() {};
53
54
55private:
56 // Default constructor
57 G4StatMFFragment(){};
58
59 // Copy constructor
60 G4StatMFFragment(const G4StatMFFragment & right);
61
62 // operators
63 G4StatMFFragment & operator=(const G4StatMFFragment & right);
64public:
65 G4bool operator==(const G4StatMFFragment & right) const;
66 G4bool operator!=(const G4StatMFFragment & right) const;
67
68public:
69
70 G4double GetCoulombEnergy(void) const;
71
72 G4double GetEnergy(const G4double T) const;
73
74 G4double GetInvLevelDensity(void) const;
75
76 G4int GetA(void) const {return theA;}
77
78 G4int GetZ(void) const {return theZ;}
79
80 void SetPosition(const G4ThreeVector& aPosition) {_position = aPosition;}
81
82 G4ThreeVector& GetPosition(void) {return _position;}
83
84 void SetMomentum(const G4ThreeVector& aMomentum) {_momentum = aMomentum;}
85
86 G4ThreeVector& GetMomentum(void) {return _momentum;}
87
88 G4Fragment * GetFragment(const G4double T);
89
90 G4double GetNuclearMass(void)
91 {return G4ParticleTable::GetParticleTable()->GetIonTable()
92 ->GetIonMass(theZ, theA);}
93
94
95private:
96
97 G4double CalcExcitationEnergy(const G4double T);
98
99private:
100
101 G4int theA;
102
103 G4int theZ;
104
105 G4ThreeVector _position;
106
107 G4ThreeVector _momentum;
108};

110#endif
