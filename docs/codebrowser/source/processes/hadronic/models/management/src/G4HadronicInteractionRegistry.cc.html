35G4HadronicInteractionRegistry* G4HadronicInteractionRegistry::Instance()
36{
37 if(nullptr == instance) {

55void G4HadronicInteractionRegistry::Clean()
56{
57 size_t nModels = allModels.size();
58 //G4cout << "G4HadronicInteractionRegistry::Clean() start " << nModels
59 // << " " << this << G4endl;
60 for (size_t i=0; i<nModels; ++i) {
61 if( allModels[i] ) {
62 const char* xxx = (allModels[i]->GetModelName()).c_str();
63 G4int len = (allModels[i]->GetModelName()).length();
64 len = std::min(len, 9);
65 const G4String mname = G4String(xxx, len);
66 //G4cout << "G4HadronicInteractionRegistry: delete " << i << " "
67 // << allModels[i] << " " << mname
68 // << " " << this << G4endl;
69 if(mname != "NeutronHP" && mname != "ParticleH") {
70 delete allModels[i];
71 }
72 // G4cout << "done " << this << G4endl;
73 }
74 }
75 allModels.clear();
76 //G4cout <<"G4HadronicInteractionRegistry::Clean() is done "<<G4endl;
77}

79void G4HadronicInteractionRegistry::InitialiseModels()
80{
81 size_t nModels = allModels.size();
82 for (size_t i=0; i<nModels; ++i) {
83 if( allModels[i] ) { allModels[i]->InitialiseModel(); }
84 }
85}
86
87void
88G4HadronicInteractionRegistry::RegisterMe(G4HadronicInteraction * aModel)
89{
90 if(!aModel) { return; }
91 size_t nModels = allModels.size();
92 for (size_t i=0; i<nModels; ++i) {
93 if( aModel == allModels[i] ) { return; }
94 }
95 //G4cout << "Register model <" << aModel->GetModelName()
96 // << "> " << nModels+1 << " " << aModel << G4endl;
97 allModels.push_back(aModel);
98}
99
100void
101G4HadronicInteractionRegistry::RemoveMe(G4HadronicInteraction * aModel)
102{
103 if(!aModel) { return; }
104 size_t nModels = allModels.size();
105 for (size_t i=0; i<nModels; ++i) {
106 if( aModel == allModels[i] ) {
107 //G4cout << "DeRegister model <" << aModel
108 // << "> " << i << G4endl;
109 allModels[i] = 0;
110 return;
111 }
112 }
113}
114
115G4HadronicInteraction*
116G4HadronicInteractionRegistry::FindModel(const G4String& name)
117{
118 G4HadronicInteraction* model = 0;
119
120 size_t nModels = allModels.size();
121 for (size_t i=0; i<nModels; ++i) {
122 G4HadronicInteraction* p = allModels[i];
123 if(p) {
124 if (p->GetModelName() == name) {
125 model = p;
126 break;
127 }
128 }
129 }
130 return model;
131}
132
133std::vector<G4HadronicInteraction*>
134G4HadronicInteractionRegistry::FindAllModels(const G4String& name)
135{
136 std::vector<G4HadronicInteraction*> models;
137
138 size_t nModels = allModels.size();
139 for (size_t i=0; i<nModels; ++i) {
140 G4HadronicInteraction* p = allModels[i];
141 if(p) {
142 if (p->GetModelName() == name) {
143 models.push_back(p);
144 }
145 }
146 }
147 return models;
148}
