27// Implements a *reverse* sorting: std::sort expects a less-than operator
28// which returns true if arg1<arg2. This function returns true if arg1>=arg2.
29//
30// 20091125 M. Kelsey -- Add additional operator() which uses pointers
31// 20110415 M. Kelsey -- Add additional operator() for G4CascadParticle
32// 20130522 M. Kelsey -- Long-standing error: should be ">", not ">="
33
34#ifndef G4PARTICLE_LARGER_EKIN_HH
35#define G4PARTICLE_LARGER_EKIN_HH
36
37#include "G4CascadParticle.hh"
38#include "G4InuclElementaryParticle.hh"
39
40#ifdef G4CASCADE_DEBUG_SORT
41#include "G4ios.hh"
42#endif
43
44class G4ParticleLargerEkin {
45public:
46 G4bool operator() (const G4InuclElementaryParticle& part1,
47 const G4InuclElementaryParticle& part2) {
48#ifdef G4CASCADE_DEBUG_SORT
49 G4cout << "part1 @ " << &part1 << ": " << part1
50 << "part2 @ " << &part2 << ": " << part2
51 << G4endl;
52#endif
53 return (part1.getKineticEnergy() > part2.getKineticEnergy());
54 }
55
56 G4bool operator() (const G4InuclElementaryParticle* part1,
57 const G4InuclElementaryParticle* part2) {
58 return (part1 && part2 && operator()(*part1, *part2));
59 }
60
61 G4bool operator() (const G4CascadParticle& part1,
62 const G4CascadParticle& part2) {
63 return (operator()(part1.getParticle(), part2.getParticle()));
64 }
65
66 G4bool operator() (const G4CascadParticle* part1,
67 const G4CascadParticle* part2) {
68 return (part1 && part2 && operator()(*part1, *part2));
69 }
70};
71
72#endif // G4PARTICLE_LARGER_EKIN_HH
