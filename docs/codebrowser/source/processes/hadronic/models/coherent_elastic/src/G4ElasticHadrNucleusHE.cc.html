75G4ElasticData::G4ElasticData(const G4ParticleDefinition* p,
76 G4int Z, G4double AWeight, G4double* eGeV)
77{
78 hadr = p;
79 massGeV = p->GetPDGMass()/GeV;
80 mass2GeV2= massGeV*massGeV;
81 AtomicWeight = G4lrint(AWeight);
82
83 DefineNucleusParameters(AWeight);
84
85 limitQ2 = 35./(R1*R1); // (GeV/c)^2
86
87 G4double dQ2 = limitQ2/(ONQ2 - 1.);
88
89 TableQ2[0] = 0.0;
90
91 for(G4int ii = 1; ii < ONQ2; ii++)
92 {
93 TableQ2[ii] = TableQ2[ii-1]+dQ2;
94 }
95
96 massA = AWeight*amu_c2/GeV;
97 massA2 = massA*massA;
98
99 for(G4int kk = 0; kk < NENERGY; kk++)
100 {
101 dnkE[kk] = 0;
102 G4double elab = eGeV[kk] + massGeV;
103 G4double plab2= eGeV[kk]*(eGeV[kk] + 2.0*massGeV);
104 G4double Q2m = 4.0*plab2*massA2/(mass2GeV2 + massA2 + 2.*massA2*elab);
105
106 if(Z == 1 && p == G4Proton::Proton()) Q2m *= 0.5;
107
108 maxQ2[kk] = std::min(limitQ2, Q2m);
109 TableCrossSec[ONQ2*kk] = 0.0;
110
111// G4cout<<" kk eGeV[kk] "<<kk<<" "<<eGeV[kk]<<G4endl;
112 }
113}

238G4ElasticHadrNucleusHE::G4ElasticHadrNucleusHE(const G4String& name)
239 : G4HadronElastic(name)
240{
241 //ANDREA->
242 G4AutoLock l(&aMutex);
243 if ( onlyOnceInit ) {
244 for ( int i = 0 ; i< NHADRONS ; ++i) {
245 for (int j = 0 ; j<ZMAX ; ++j ) {
246 SetOfElasticData[i][j]=0;
247 G4MUTEXINIT(eldata_m[i][j]);
248 }
249 }
250 onlyOnceInit = false;
251 }
252 l.unlock();
253 //ANDREA<-
254
255 dQ2 = hMass = hMass2 = hLabMomentum = hLabMomentum2 = MomentumCM = HadrEnergy
256 = R1 = R2 = Pnucl = Aeff = HadrTot = HadrSlope = HadrReIm = TotP = DDSect2
257 = DDSect3 = ConstU = FmaxT = Slope1 = Slope2 = Coeff1 = Coeff2 = MaxTR
258 = Slope0 = Coeff0 = aAIm = aDIm = Dtot11 = 0.0;
259 NumbN = iHadrCode = iHadron = 0;
260
261 verboseLevel = 0;
262 plabLowLimit = 20.0*MeV;
263 lowestEnergyLimit = 0.0;
264 //Description();
265
266 MbToGeV2 = 2.568;
267 sqMbToGeV = 1.602;
268 Fm2ToGeV2 = 25.68;
269 GeV2 = GeV*GeV;
270 protonM = proton_mass_c2/GeV;
271 protonM2 = protonM*protonM;
272
273 BoundaryP[0]=9.0;BoundaryTG[0]=5.0;BoundaryTL[0]=0.;
274 BoundaryP[1]=20.0;BoundaryTG[1]=1.5;BoundaryTL[1]=0.;
275 BoundaryP[2]=5.0; BoundaryTG[2]=1.0;BoundaryTL[2]=1.5;
276 BoundaryP[3]=8.0; BoundaryTG[3]=3.0;BoundaryTL[3]=0.;
277 BoundaryP[4]=7.0; BoundaryTG[4]=3.0;BoundaryTL[4]=0.;
278 BoundaryP[5]=5.0; BoundaryTG[5]=2.0;BoundaryTL[5]=0.;
279 BoundaryP[6]=5.0; BoundaryTG[6]=1.5;BoundaryTL[6]=3.0;
280
281 Binom();
282 // energy in GeV
283 Energy[0] = 0.4;
284 Energy[1] = 0.6;
285 Energy[2] = 0.8;
286 LowEdgeEnergy[0] = 0.0;
287 LowEdgeEnergy[1] = 0.5;
288 LowEdgeEnergy[2] = 0.7;
289 G4double e = 1.0;
290 G4double f = G4Exp(G4Log(10.)*0.1);
291 for(G4int i=3; i<NENERGY; i++) {
292 Energy[i] = e;
293 LowEdgeEnergy[i] = e/f;
294 e *= f*f;
295 }
296 nistManager = G4NistManager::Instance();
297
298 // PDG code for hadrons
299 G4int ic[NHADRONS] = {211,-211,2112,2212,321,-321,130,310,311,-311,
300 3122,3222,3112,3212,3312,3322,3334,
301 -2212,-2112,-3122,-3222,-3112,-3212,-3312,-3322,-3334};
302 // internal index
303 G4int id[NHADRONS] = {2,3,6,0,4,5,4,4,4,5,
304 0,0,0,0,0,0,0,
305 1,7,1,1,1,1,1,1,1};
306
307 G4int id1[NHADRONS] = {3,4,1,0,5,6,5,5,5,6,
308 0,0,0,0,0,0,0,
309 2,2,2,2,2,2,2,2,2};
310
311 for(G4int j=0; j<NHADRONS; j++)
312 {
313 HadronCode[j] = ic[j];
314 HadronType[j] = id[j];
315 HadronType1[j] = id1[j];
316
317 for(G4int k = 0; k < ZMAX; k++) { SetOfElasticData[j][k] = 0; }
318 }
319}

361G4double
362G4ElasticHadrNucleusHE::SampleInvariantT(const G4ParticleDefinition* p,
363 G4double inLabMom,
364 G4int iZ, G4int N)
365{
366 G4int Z = iZ;
367 if(Z >= ZMAX) { Z = ZMAX-1; }
368 G4double plab = inLabMom/GeV; // (GeV/c)
369 G4double Q2 = 0;
370
371 iHadrCode = p->GetPDGEncoding();
372
373 NumbN = N;
374
375 if(verboseLevel > 1)
376 {
377 G4cout<< " G4ElasticHadrNucleusHE::SampleT: "
378 << " for " << p->GetParticleName()
379 << " at Z= " << Z << " A= " << N
380 << " plab(GeV)= " << plab
381 << G4endl;
382 }
383 G4int idx;
384
385 for( idx = 0 ; idx < NHADRONS; idx++)
386 {
387 if(iHadrCode == HadronCode[idx]) break;
388 }
389
390 // Hadron is not in the list
391
392 if( idx >= NHADRONS ) return Q2;
393
394 iHadron = HadronType[idx];
395 iHadrCode = HadronCode[idx];
396
397 if(Z==1)
398 {
399 hMass = p->GetPDGMass()/GeV;
400 hMass2 = hMass*hMass;
401
402 G4double T = sqrt(plab*plab+hMass2)-hMass;
403
404 if(T > 0.4) Q2 = HadronProtonQ2(p, plab);
405
406 if (verboseLevel>1)
407 G4cout<<" Proton : Q2 "<<Q2<<G4endl;
408 }
409 else
410 {
411 G4AutoLock l(&(eldata_m[idx][Z]));//ANDREA
412 G4ElasticData* ElD1 = SetOfElasticData[idx][Z];
413
414 // Construct elastic data
415 if(!ElD1)
416 {
417 G4double AWeight = nistManager->GetAtomicMassAmu(Z);
418 ElD1 = new G4ElasticData(p, Z, AWeight, Energy);
419 SetOfElasticData[idx][Z] = ElD1;
420
421 if(verboseLevel > 1)
422 {
423 G4cout<< " G4ElasticHadrNucleusHE::SampleT: new record " << idx
424 << " for " << p->GetParticleName() << " Z= " << Z
425 << G4endl;
426 }
427 }
428 hMass = ElD1->massGeV;
429 hMass2 = ElD1->mass2GeV2;
430 G4double M = ElD1->massA;
431 G4double M2 = ElD1->massA2;
432 G4double plab2 = plab*plab;
433 G4double Q2max = 4.*plab2*M2/
434 (hMass2 + M2 + 2.*M*std::sqrt(plab2 + hMass2));
435
436 // sample scattering
437 G4double T = sqrt(plab2+hMass2)-hMass;
438
439 if(T > 0.4) Q2 = HadronNucleusQ2_2(ElD1, Z, plab, Q2max);
440
441 if(verboseLevel > 1)
442 G4cout<<" SampleT: Q2(GeV^2)= "<<Q2<< " t/tmax= " << Q2/Q2max <<G4endl;
443 }
444 return Q2*GeV2;
445}

463G4double G4ElasticHadrNucleusHE::
464 HadronNucleusQ2_2(G4ElasticData* pElD, G4int Z,
465 G4double plab, G4double tmax)
466{
467 //ANDREA: Important notice on this function
468 // For MT we are sharing among threads the G4ElasticData classes
469 // The *call* to this function is proteced
470 // with a mutex in the calling function
471 G4double LineFq2[ONQ2];
472
473 G4double Rand = G4UniformRand();
474
475 G4int iNumbQ2 = 0;
476 G4double Q2 = 0.0;
477
478 G4double ptot2 = plab*plab;
479 G4double ekin = std::sqrt(hMass2 + ptot2) - hMass;
480
481 if(verboseLevel > 1)
482 G4cout<<"Q2_2: ekin plab "<<ekin<<" "<<plab<<" tmax "<<tmax<<G4endl;
483
484 // Find closest energy bin
485 G4int NumbOnE;
486 for( NumbOnE = 0; NumbOnE < NENERGY-1; NumbOnE++ )
487 {
488 if( ekin <= LowEdgeEnergy[NumbOnE+1] ) break;
489 }
490 G4double* dNumbQ2 = pElD->TableQ2;
491
492 if(NumbOnE >= NENERGY-1) { NumbOnE = NENERGY-2; }
493 G4int index = NumbOnE*ONQ2;
494
495 // Select kinematics for node energy
496 G4double T = Energy[NumbOnE];
497 hLabMomentum2 = T*(T + 2.*hMass);
498 G4double Q2max = pElD->maxQ2[NumbOnE];
499 G4int length = pElD->dnkE[NumbOnE];
500
501 // Build vector
502 if(length == 0)
503 {
504 R1 = pElD->R1;
505 R2 = pElD->R2;
506 Aeff = pElD->Aeff;
507 Pnucl = pElD->Pnucl;
508 hLabMomentum = std::sqrt(hLabMomentum2);
509
510 DefineHadronValues(Z);
511
512 if(verboseLevel >0)
513 {
514 G4cout<<"1 plab T "<<plab<<" "<<T<<" sigTot B ReIm "
515 <<HadrTot<<" "<<HadrSlope<<" "<<HadrReIm<<G4endl;
516 G4cout<<" R1 R2 Aeff p "<<R1<<" "<<R2<<" "<<Aeff<<" "
517 <<Pnucl<<G4endl;
518 }
519
520 //pElD->CrossSecMaxQ2[NumbOnE] = 1.0;
521
522 if(verboseLevel > 1)
523 G4cout<<" HadrNucleusQ2_2: NumbOnE= " << NumbOnE
524 << " length= " << length
525 << " Q2max= " << Q2max
526 << " ekin= " << ekin <<G4endl;
527
528 pElD->TableCrossSec[index] = 0;
529
530
531 dQ2 = pElD->TableQ2[1]-pElD->TableQ2[0];
532
533 GetHeavyFq2(Z, NumbN, LineFq2); // %%%%%%%%%%%%%%%%%%%%%%%%%
534
535 for(G4int ii=0; ii<ONQ2; ++ii)
536 {
537 //if(verboseLevel > 2)
538 // G4cout<<" ii LineFq2 "<<ii<<" "<<LineFq2[ii]/LineFq2[ONQ2-1]
539 // <<" dF(q2) "<<LineFq2[ii]-LineFq2[ii-1]<<G4endl;
540
541 pElD->TableCrossSec[index+ii] = LineFq2[ii]/LineFq2[ONQ2-1];
542 }
543
544 pElD->dnkE[NumbOnE] = ONQ2;
545 length = ONQ2;
546 }
547
548 G4double* dNumbFQ2 = &(pElD->TableCrossSec[index]);
549
550 for( iNumbQ2 = 1; iNumbQ2<length; ++iNumbQ2)
551 {
552 if(Rand <= pElD->TableCrossSec[index+iNumbQ2]) break;
553 }
554 if(iNumbQ2 >= ONQ2) { iNumbQ2 = ONQ2 - 1; }
555 Q2 = GetQ2_2(iNumbQ2, dNumbQ2, dNumbFQ2, Rand);
556
557 if(tmax < Q2max) Q2 *= tmax/Q2max;
558
559 if(verboseLevel > 1)
560 G4cout<<" HadrNucleusQ2_2(2): Q2= "<<Q2<<" iNumbQ2= " << iNumbQ2
561 << " rand= " << Rand << G4endl;
562
563 return Q2;
564}

566///////////////////////////////////////////////////////////////////////
567//
568// The randomization of one dimensional array
569//
570G4double G4ElasticHadrNucleusHE::GetQ2_2(G4int kk, G4double * Q,
571 G4double * F, G4double ranUni)
572{
573 G4double ranQ2;
574
575 G4double F1 = F[kk-2];
576 G4double F2 = F[kk-1];
577 G4double F3 = F[kk];
578 G4double X1 = Q[kk-2];
579 G4double X2 = Q[kk-1];
580 G4double X3 = Q[kk];
581
582 if(verboseLevel > 2)
583 G4cout << "GetQ2_2 kk= " << kk << " X2= " << X2 << " X3= " << X3
584 << " F2= " << F2 << " F3= " << F3 << " Rndm= " << ranUni << G4endl;
585
586 if(kk == 1 || kk == 0)
587 {
588 F1 = F[0];
589 F2 = F[1];
590 F3 = F[2];
591 X1 = Q[0];
592 X2 = Q[1];
593 X3 = Q[2];
594 }
595
596 G4double F12 = F1*F1;
597 G4double F22 = F2*F2;
598 G4double F32 = F3*F3;
599
600 G4double D0 = F12*F2+F1*F32+F3*F22-F32*F2-F22*F1-F12*F3;
601
602 if(verboseLevel > 2)
603 G4cout << " X1= " << X1 << " F1= " << F1 << " D0= "
604 << D0 << G4endl;
605
606 if(std::abs(D0) < 0.00000001)
607 {
608 ranQ2 = X2 + (ranUni - F2)*(X3 - X2)/(F3 - F2);
609 }
610 else
611 {
612 G4double DA = X1*F2+X3*F1+X2*F3-X3*F2-X1*F3-X2*F1;
613 G4double DB = X2*F12+X1*F32+X3*F22-X2*F32-X3*F12-X1*F22;
614 G4double DC = X3*F2*F12+X2*F1*F32+X1*F3*F22
615 -X1*F2*F32-X2*F3*F12-X3*F1*F22;
616 ranQ2 = (DA*ranUni*ranUni + DB*ranUni + DC)/D0;
617 }
618 return ranQ2; // MeV^2
619}

624G4double G4ElasticHadrNucleusHE::GetHeavyFq2(G4int Z, G4int Nucleus, G4double* LineF)
625{
626 G4int ii, jj, aSimp;
627 G4double curQ2, curSec;
628 G4double curSum = 0.0;
629 G4double totSum = 0.0;
630
631 G4double ddQ2 = dQ2/20;
632 G4double Q2l = 0;
633
634 LineF[0] = 0;
635 for(ii = 1; ii<ONQ2; ii++)
636 {
637 curSum = 0;
638 aSimp = 4;
639
640 for(jj = 0; jj<20; jj++)
641 {
642 curQ2 = Q2l+jj*ddQ2;
643
644 curSec = HadrNucDifferCrSec(Z, Nucleus, curQ2);
645 curSum += curSec*aSimp;
646
647 if(aSimp > 3) aSimp = 2;
648 else aSimp = 4;
649
650 if(jj == 0 && verboseLevel>2)
651 G4cout<<" Q2 "<<curQ2<<" AIm "<<aAIm<<" DIm "<<aDIm
652 <<" Diff "<<curSec<<" totSum "<<totSum<<G4endl;
653 }
654
655 Q2l += dQ2;
656 curSum *= ddQ2/2.3; // $$$$$$$$$$$$$$$$$$$$$$$
657 totSum += curSum;
658
659 LineF[ii] = totSum;
660
661 if (verboseLevel>2)
662 G4cout<<" GetHeavy: Q2 dQ2 totSum "<<Q2l<<" "<<dQ2<<" "<<totSum
663 <<" curSec "
664 <<curSec<<" totSum "<< totSum<<" DTot "
