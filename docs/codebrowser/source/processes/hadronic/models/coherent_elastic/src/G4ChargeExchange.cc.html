55G4ChargeExchange::G4ChargeExchange() : G4HadronicInteraction("Charge Exchange")
56{
57 SetMinEnergy( 0.0*GeV );
58 SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
59
60 lowEnergyRecoilLimit = 100.*keV;
61 lowestEnergyLimit = 1.*MeV;
62
63 theProton = G4Proton::Proton();
64 theNeutron = G4Neutron::Neutron();
65 theAProton = G4AntiProton::AntiProton();
66 theANeutron = G4AntiNeutron::AntiNeutron();
67 thePiPlus = G4PionPlus::PionPlus();
68 thePiMinus = G4PionMinus::PionMinus();
69 thePiZero = G4PionZero::PionZero();
70 theKPlus = G4KaonPlus::KaonPlus();
71 theKMinus = G4KaonMinus::KaonMinus();
72 theK0S = G4KaonZeroShort::KaonZeroShort();
73 theK0L = G4KaonZeroLong::KaonZeroLong();
74 theL = G4Lambda::Lambda();
75 theAntiL = G4AntiLambda::AntiLambda();
76 theSPlus = G4SigmaPlus::SigmaPlus();
77 theASPlus = G4AntiSigmaPlus::AntiSigmaPlus();
78 theSMinus = G4SigmaMinus::SigmaMinus();
79 theASMinus = G4AntiSigmaMinus::AntiSigmaMinus();
80 theS0 = G4SigmaZero::SigmaZero();
81 theAS0 = G4AntiSigmaZero::AntiSigmaZero();
82 theXiMinus = G4XiMinus::XiMinus();
83 theXi0 = G4XiZero::XiZero();
84 theAXiMinus = G4AntiXiMinus::AntiXiMinus();
85 theAXi0 = G4AntiXiZero::AntiXiZero();
86 theOmega = G4OmegaMinus::OmegaMinus();
87 theAOmega = G4AntiOmegaMinus::AntiOmegaMinus();
88 theD = G4Deuteron::Deuteron();
89 theT = G4Triton::Triton();
90 theA = G4Alpha::Alpha();
91 theHe3 = G4He3::He3();
92}

97G4HadFinalState* G4ChargeExchange::ApplyYourself(
98 const G4HadProjectile& aTrack, G4Nucleus& targetNucleus)
99{
100 theParticleChange.Clear();
101 const G4HadProjectile* aParticle = &aTrack;
102 G4double ekin = aParticle->GetKineticEnergy();
103
104 G4int A = targetNucleus.GetA_asInt();
105 G4int Z = targetNucleus.GetZ_asInt();
106
107 if(ekin <= lowestEnergyLimit || A < 3) {
108 theParticleChange.SetEnergyChange(ekin);
109 theParticleChange.SetMomentumChange(aTrack.Get4Momentum().vect().unit());
110 return &theParticleChange;
111 }
112
113 G4double plab = aParticle->GetTotalMomentum();
114
115 if (verboseLevel > 1)
116 G4cout << "G4ChargeExchange::DoIt: Incident particle plab="
117 << plab/GeV << " GeV/c "
118 << " ekin(MeV) = " << ekin/MeV << " "
119 << aParticle->GetDefinition()->GetParticleName() << G4endl;
120
121 // Scattered particle referred to axis of incident particle
122 const G4ParticleDefinition* theParticle = aParticle->GetDefinition();
123
124 G4int N = A - Z;
125 G4int projPDG = theParticle->GetPDGEncoding();
126 if (verboseLevel > 1)
127 G4cout << "G4ChargeExchange for " << theParticle->GetParticleName()
128 << " PDGcode= " << projPDG << " on nucleus Z= " << Z
129 << " A= " << A << " N= " << N
130 << G4endl;
131
132 G4ParticleDefinition * theDef = 0;
133
134 G4double mass2 = G4NucleiProperties::GetNuclearMass((G4double)A, (G4double)Z);
135 G4LorentzVector lv1 = aParticle->Get4Momentum();
136 G4LorentzVector lv0(0.0,0.0,0.0,mass2);
137
138 G4LorentzVector lv = lv0 + lv1;
139 G4ThreeVector bst = lv.boostVector();
140 lv1.boost(-bst);
141 lv0.boost(-bst);
142
143 // Sample final particles
144 G4bool theHyperon = false;
145 G4ParticleDefinition* theRecoil = 0;
146 G4ParticleDefinition* theSecondary = 0;
147
148 if(theParticle == theProton) {
149 theSecondary = theNeutron;
150 Z++;
151 } else if(theParticle == theNeutron) {
152 theSecondary = theProton;
153 Z--;
154 } else if(theParticle == thePiPlus) {
155 theSecondary = thePiZero;
156 Z++;
157 } else if(theParticle == thePiMinus) {
158 theSecondary = thePiZero;
159 Z--;
160 } else if(theParticle == theKPlus) {
161 if(G4UniformRand()<0.5) theSecondary = theK0S;
162 else theSecondary = theK0L;
163 Z++;
164 } else if(theParticle == theKMinus) {
165 if(G4UniformRand()<0.5) theSecondary = theK0S;
166 else theSecondary = theK0L;
167 Z--;
168 } else if(theParticle == theK0S || theParticle == theK0L) {
169 if(G4UniformRand()*A < G4double(Z)) {
170 theSecondary = theKPlus;
171 Z--;
172 } else {
173 theSecondary = theKMinus;
174 Z++;
175 }
176 } else if(theParticle == theANeutron) {
177 theSecondary = theAProton;
178 Z++;
179 } else if(theParticle == theAProton) {
180 theSecondary = theANeutron;
181 Z--;
182 } else if(theParticle == theL) {
183 G4double x = G4UniformRand();
184 if(G4UniformRand()*A < G4double(Z)) {
185 if(x < 0.2) {
186 theSecondary = theS0;
187 } else if (x < 0.4) {
188 theSecondary = theSPlus;
189 Z--;
190 } else if (x < 0.6) {
191 theSecondary = theProton;
192 theRecoil = theL;
193 theHyperon = true;
194 A--;
195 } else if (x < 0.8) {
196 theSecondary = theProton;
197 theRecoil = theS0;
198 theHyperon = true;
199 A--;
200 } else {
201 theSecondary = theNeutron;
202 theRecoil = theSPlus;
203 theHyperon = true;
204 A--;
205 }
206 } else {
207 if(x < 0.2) {
208 theSecondary = theS0;
209 } else if (x < 0.4) {
210 theSecondary = theSMinus;
211 Z++;
212 } else if (x < 0.6) {
213 theSecondary = theNeutron;
214 theRecoil = theL;
215 A--;
216 theHyperon = true;
217 } else if (x < 0.8) {
218 theSecondary = theNeutron;
219 theRecoil = theS0;
220 theHyperon = true;
221 A--;
222 } else {
223 theSecondary = theProton;
224 theRecoil = theSMinus;
225 theHyperon = true;
226 A--;
227 }
228 }
229 }
230
231 if (Z == 1 && A == 2) theDef = theD;
232 else if (Z == 1 && A == 3) theDef = theT;
233 else if (Z == 2 && A == 3) theDef = theHe3;
234 else if (Z == 2 && A == 4) theDef = theA;
235 else {
236 theDef =
237 G4ParticleTable::GetParticleTable()->GetIonTable()->GetIon(Z,A,0.0);
238 }
239 if(!theSecondary) { return &theParticleChange; }
240
241 G4double m11 = theSecondary->GetPDGMass();
242 G4double m21 = theDef->GetPDGMass();
243 if(theRecoil) { m21 += theRecoil->GetPDGMass(); }
244 else { theRecoil = theDef; }
245
246 G4double etot = lv0.e() + lv1.e();
247
248 // kinematiacally impossible
249 if(etot < m11 + m21) {
250 theParticleChange.SetEnergyChange(ekin);
251 theParticleChange.SetMomentumChange(aTrack.Get4Momentum().vect().unit());
252 return &theParticleChange;
253 }
254
255 G4ThreeVector p1 = lv1.vect();
256 G4double e1 = 0.5*etot*(1.0 - (m21*m21 - m11*m11)/(etot*etot));
257 // G4double e2 = etot - e1;
258 G4double ptot = std::sqrt(e1*e1 - m11*m11);
259
260 G4double tmax = 4.0*ptot*ptot;
261 G4double g2 = GeV*GeV;
262
263 G4double t = g2*SampleT(tmax/g2, A);
264
265 if(verboseLevel>1)
266 G4cout <<"## G4ChargeExchange t= " << t << " tmax= " << tmax
267 << " ptot= " << ptot << G4endl;
268
269 // Sampling in CM system
270 G4double phi = G4UniformRand()*twopi;
271 G4double cost = 1. - 2.0*t/tmax;
272 if(std::abs(cost) > 1.0) cost = 1.0;
273 G4double sint = std::sqrt((1.0-cost)*(1.0+cost));
274
275 //if (verboseLevel > 1)
276 // G4cout << "cos(t)=" << cost << " std::sin(t)=" << sint << G4endl;
277
278 G4ThreeVector v1(sint*std::cos(phi),sint*std::sin(phi),cost);
279 v1 *= ptot;
280 G4LorentzVector nlv1(v1.x(),v1.y(),v1.z(),e1);
281 G4LorentzVector nlv0 = lv0 + lv1 - nlv1;
282
283 nlv0.boost(bst);
284 nlv1.boost(bst);
285
286 theParticleChange.SetStatusChange(stopAndKill);
287 theParticleChange.SetEnergyChange(0.0);
288 G4DynamicParticle * aSec = new G4DynamicParticle(theSecondary, nlv1);
289 theParticleChange.AddSecondary(aSec);
290
291 G4double erec = nlv0.e() - m21;
292
293 //G4cout << "erec= " <<erec << " Esec= " << aSec->GetKineticEnergy() << G4endl;
294
295 if(theHyperon) {
296 theParticleChange.SetLocalEnergyDeposit(erec);
297 aSec = new G4DynamicParticle();
298 aSec->SetDefinition(theRecoil);
299 aSec->SetKineticEnergy(0.0);
300 } else if(erec > lowEnergyRecoilLimit) {
301 aSec = new G4DynamicParticle(theRecoil, nlv0);
302 theParticleChange.AddSecondary(aSec);
303 } else {
304 if(erec < 0.0) erec = 0.0;
305 theParticleChange.SetLocalEnergyDeposit(erec);
306 }
307 return &theParticleChange;
308}

310G4double G4ChargeExchange::SampleT(G4double tmax, G4double A)
311{
312 G4double aa, bb, cc, dd;
313 if (A <= 62.) {
314 aa = G4Pow::GetInstance()->powA(A, 1.63);
315 bb = 14.5*G4Pow::GetInstance()->powA(A, 0.66);
316 cc = 1.4*G4Pow::GetInstance()->powA(A, 0.33);
317 dd = 10.;
318 } else {
319 aa = G4Pow::GetInstance()->powA(A, 1.33);
320 bb = 60.*G4Pow::GetInstance()->powA(A, 0.33);
321 cc = 0.4*G4Pow::GetInstance()->powA(A, 0.40);
322 dd = 10.;
323 }
324 G4double x1 = (1.0 - G4Exp(-tmax*bb))*aa/bb;
325 G4double x2 = (1.0 - G4Exp(-tmax*dd))*cc/dd;
326
327 G4double t;
328 G4double y = bb;
329 if(G4UniformRand()*(x1 + x2) < x2) y = dd;
330
331 const G4int maxNumberOfLoops = 10000;
332 G4int loopCounter = 0;
333 do {
334 t = -G4Log(G4UniformRand())/y;
335 } while ( (t > tmax) &&
336 ++loopCounter < maxNumberOfLoops ); /* Loop checking, 10.08.2015, A.Ribon */
337 if ( loopCounter >= maxNumberOfLoops ) {
338 t = 0.0;
339 }
340
341 return t;
342}
