38#ifndef G4HadronElastic_h
39#define G4HadronElastic_h 1
40
41#include "globals.hh"
42#include "G4HadronicInteraction.hh"
43#include "G4HadProjectile.hh"
44#include "G4Nucleus.hh"
45#include "G4NucleiProperties.hh"
46
47class G4ParticleDefinition;
48
49class G4HadronElastic : public G4HadronicInteraction
50{
51public:
52
53 G4HadronElastic(const G4String& name = "hElasticLHEP");
54
55 virtual ~G4HadronElastic();
56
57 // implementation of the G4HadronicInteraction interface
58 virtual G4HadFinalState * ApplyYourself(const G4HadProjectile & aTrack,
59 G4Nucleus & targetNucleus);
60
61 // sample momentum transfer using Lab. momentum
62 virtual G4double SampleInvariantT(const G4ParticleDefinition* p,
63 G4double plab,
64 G4int Z, G4int A);
65
66 inline void SetLowestEnergyLimit(G4double value);
67
68 inline G4double LowestEnergyLimit() const;
69
70 inline G4double ComputeMomentumCMS(const G4ParticleDefinition* p,
71 G4double plab, G4int Z, G4int A);
72
73 virtual void ModelDescription(std::ostream&) const;
74
75private:
76
77 G4ParticleDefinition* theProton;
78 G4ParticleDefinition* theNeutron;
79 G4ParticleDefinition* theDeuteron;
80 G4ParticleDefinition* theAlpha;
81
82 G4double lowestEnergyLimit;
83
84};
85
86inline void G4HadronElastic::SetLowestEnergyLimit(G4double value)
87{
88 lowestEnergyLimit = value;
89}
90
91inline G4double G4HadronElastic::LowestEnergyLimit() const
92{
93 return lowestEnergyLimit;
94}
95
96inline G4double
97G4HadronElastic::ComputeMomentumCMS(const G4ParticleDefinition* p,
98 G4double plab, G4int Z, G4int A)
99{
100 G4double m1 = p->GetPDGMass();
101 G4double m12= m1*m1;
102 G4double mass2 = G4NucleiProperties::GetNuclearMass(A, Z);
103 return plab*mass2/std::sqrt(m12 + mass2*mass2 + 2.*mass2*std::sqrt(m12 + plab*plab));
104}
