28#include "globals.hh"
29#include "G4ParticleDefinition.hh"
30#include "G4ConcreteNNToNNStar.hh"
31#include "G4NNstarBuilder.hh"
32
33G4ThreadLocal G4XNNstarTable *G4ConcreteNNToNNStar::theSigmaTable_G4MT_TLS_ = 0;
34
35G4ConcreteNNToNNStar::G4ConcreteNNToNNStar(const G4ParticleDefinition* aPrimary,
36 const G4ParticleDefinition* bPrimary,
37 const G4ParticleDefinition* aSecondary,
38 const G4ParticleDefinition* bSecondary) :
39 G4ConcreteNNTwoBodyResonance(NULL, NULL, NULL, NULL, NULL, NULL, NULL)
40{
41 if (!theSigmaTable_G4MT_TLS_) theSigmaTable_G4MT_TLS_ = new G4XNNstarTable;
42 G4XNNstarTable &theSigmaTable = *theSigmaTable_G4MT_TLS_;
43 establish_G4MT_TLS_G4ConcreteNNTwoBodyResonance(aPrimary,bPrimary,aSecondary,bSecondary,
44 G4NNstarBuilder(bSecondary->GetParticleName(),theSigmaTable));
45}
46
47G4ConcreteNNToNNStar::~G4ConcreteNNToNNStar()
48{
49 if (theSigmaTable_G4MT_TLS_) delete theSigmaTable_G4MT_TLS_;
50 theSigmaTable_G4MT_TLS_=0;
51}
