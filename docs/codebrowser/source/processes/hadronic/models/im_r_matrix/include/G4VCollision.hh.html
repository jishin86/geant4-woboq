28#ifndef G4VCollision_h
29#define G4VCollision_h
30
31#include "globals.hh"
32#include "G4CollisionVector.hh"
33#include "G4VCrossSectionSource.hh"
34#include "G4VAngularDistribution.hh"
35#include "G4KineticTrackVector.hh"
36
37class G4KineticTrack;
38
39class G4VCollision
40{
41public:
42
43 G4VCollision();
44 void establish_G4MT_TLS_G4VCollision();
45 G4VCollision(void *s1, void *s2, void *s3, void *s4, void *s5, void *s6, void *s7);
46
47 virtual ~G4VCollision();
48
49 G4bool operator==(const G4VCollision &right) const;
50 G4bool operator!=(const G4VCollision &right) const;
51
52 virtual G4double CrossSection(const G4KineticTrack& trk1,
53 const G4KineticTrack& trk2) const;
54
55 virtual G4KineticTrackVector* FinalState(const G4KineticTrack& trk1,
56 const G4KineticTrack& trk2) const = 0;
57
58 virtual G4bool IsInCharge(const G4KineticTrack& trk1,
59 const G4KineticTrack& trk2) const = 0;
60
61 virtual G4String GetName() const = 0;
62
63 virtual void Print() const;
64 virtual void Print(const G4KineticTrack& trk1,
65 const G4KineticTrack& trk2) const;
66protected:
67
68 G4int GetNumberOfPartons(const G4ParticleDefinition * aP) const
69 {
70 G4int result = 0;
71 for(G4int i=0; i<6; i++)
72 {
73 result += aP->GetQuarkContent(i+1);
74 result += aP->GetAntiQuarkContent(i+1);
75 }
76 return result;
77 }
78
79 virtual const G4CollisionVector* GetComponents() const { return 0;}
80
81 virtual const G4VCrossSectionSource* GetCrossSectionSource() const = 0;
82
83 virtual const G4VAngularDistribution* GetAngularDistribution() const = 0;
84
85 virtual const std::vector<G4String>& GetListOfColliders(G4int whichOne) const = 0;
86
87
88private:
89
90 G4VCollision(const G4VCollision &right);
91
92 const G4VCollision& operator=(const G4VCollision &right);
93
94};
95
96#endif
