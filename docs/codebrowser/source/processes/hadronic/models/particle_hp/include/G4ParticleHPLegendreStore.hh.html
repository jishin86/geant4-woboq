29#ifndef G4ParticleHPLegendreStore_h
30#define G4ParticleHPLegendreStore_h 1
31
32#include "G4ParticleHPLegendreTable.hh"
33#include "G4InterpolationManager.hh"
34#include "G4ios.hh"
35#include <fstream>
36
37class G4ParticleHPLegendreStore
38{
39 public:
40
41 G4ParticleHPLegendreStore(G4int n)
42 {
43 theCoeff = new G4ParticleHPLegendreTable[n];
44 nEnergy = n;
45 }
46
47 ~G4ParticleHPLegendreStore()
48 {
49 delete [] theCoeff;
50 }
51
52 inline void Init(G4int i, G4double e, G4int n)
53 {
54 theCoeff[i].Init(e, n);
55 }
56 inline void SetNPoints(G4int n) { nEnergy = n; }
57 inline void SetEnergy(G4int i, G4double energy) { theCoeff[i].SetEnergy(energy); }
58 inline void SetTemperature(G4int i, G4double temp) { theCoeff[i].SetTemperature(temp); }
59 inline void SetCoeff(G4int i, G4int l, G4double coeff) {theCoeff[i].SetCoeff(l, coeff); }
60 inline void SetCoeff(G4int i, G4ParticleHPLegendreTable * theTable)
61 {
62 if(i>nEnergy) throw G4HadronicException(__FILE__, __LINE__, "LegendreTableIndex out of range");
63 theCoeff[i] = *theTable;
64// not here -- see G4ParticleHPPhotonDist.cc line 275
65// delete theTable;
66 }
67
68 inline G4double GetCoeff(G4int i, G4int l) {return theCoeff[i].GetCoeff(l);}
69 inline G4double GetEnergy(G4int i){return theCoeff[i].GetEnergy();}
70 inline G4double GetTemperature(G4int i){return theCoeff[i].GetTemperature();}
71 inline G4int GetNumberOfPoly(G4int i) {return theCoeff[i].GetNumberOfPoly();}
72
73 G4double SampleDiscreteTwoBody (G4double anEnergy);
74 G4double SampleElastic (G4double anEnergy);
75 G4double Sample (G4double energy);
76 G4double SampleMax (G4double energy);
77 G4double Integrate(G4int k, G4double costh);
78
79 void InitInterpolation(std::istream & aDataFile)
80 {
81 theManager.Init(aDataFile);
82 }
83
84 void SetManager(G4InterpolationManager & aManager)
85 {
86 theManager = aManager;
87 }
88
89 private:
90
91 G4int nEnergy;
92 G4ParticleHPLegendreTable * theCoeff;
93 G4InterpolationManager theManager; // interpolate between different Tables
94};
95#endif
