29#ifndef G4ParticleHPNBodyPhaseSpace_h
30#define G4ParticleHPNBodyPhaseSpace_h 1
31
32#include <fstream>
33#include <CLHEP/Units/PhysicalConstants.h>
34
35#include "globals.hh"
36#include "G4Pow.hh"
37#include "G4ios.hh"
38#include "G4Neutron.hh"
39#include "G4VParticleHPEnergyAngular.hh"
40#include "G4ReactionProduct.hh"
41
42class G4ParticleHPNBodyPhaseSpace : public G4VParticleHPEnergyAngular
43{
44 public:
45
46 G4ParticleHPNBodyPhaseSpace(){
47 theTotalMass = 0.0;
48 theTotalCount = 0;
49 }
50 ~G4ParticleHPNBodyPhaseSpace(){}
51
52 public:
53
54 void Init(G4double aMass, G4int aCount)
55 {
56 theTotalMass=aMass;
57 theTotalCount=aCount;
58 }
59
60 void Init(std::istream & aDataFile)
61 {
62 aDataFile >> theTotalMass >> theTotalCount;
63 theTotalMass *= G4Neutron::Neutron()->GetPDGMass();
64 }
65
66 G4ReactionProduct * Sample(G4double anEnergy, G4double massCode, G4double mass);
67
68 private:
69
70 inline G4double Prob(G4double anEnergy, G4double eMax, G4int n)
71 {
72 G4double result;
73 result = std::sqrt(anEnergy)*G4Pow::GetInstance()->powA(eMax-anEnergy, 3.*n/2.-4.);
74 return result;
75 }
76
77 inline G4double C(G4double anEnergy, G4double mass)
78 {
79 G4double result(0);
80 if(theTotalCount==3) result = 4./CLHEP::pi/G4Pow::GetInstance()->powN(GetEmax(anEnergy, mass),2);
81 if(theTotalCount==4) result = 105./32./G4Pow::GetInstance()->powA(GetEmax(anEnergy, mass), 3.5);
82 //if(theTotalCount==5) result = 256./14./CLHEP::pi/G4Pow::GetInstance()->powA(GetEmax(anEnergy, mass), 5.);
83 if(theTotalCount==5) result = 256./14./CLHEP::pi/G4Pow::GetInstance()->powN(GetEmax(anEnergy, mass), 5);
84 return result;
85 }
86
87 inline G4double GetEmax(G4double anEnergy, G4double mass)
88 {
89 G4double result;
90 G4double tMass = GetTarget()->GetMass();
91 G4double pMass = GetProjectileRP()->GetMass();
92 G4double availableEnergy = GetQValue() + anEnergy*tMass/(pMass+tMass);
93 result = availableEnergy*(theTotalMass-mass)/theTotalMass;
94 return result;
95 }
96
97 G4double MeanEnergyOfThisInteraction() {return -1; }
98
99 private:
100
101 G4double theTotalMass;
102 G4int theTotalCount;
103
104};
105#endif
