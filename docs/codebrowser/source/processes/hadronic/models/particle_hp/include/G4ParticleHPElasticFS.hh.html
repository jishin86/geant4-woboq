33#ifndef G4ParticleHPElasticFS_h
34#define G4ParticleHPElasticFS_h 1
35
36#include "globals.hh"
37#include "G4HadProjectile.hh"
38#include "G4HadFinalState.hh"
39#include "G4ParticleHPFinalState.hh"
40#include "G4ParticleHPLegendreStore.hh"
41#include "G4ParticleHPPartial.hh"
42#include "G4ParticleHPFastLegendre.hh"
43#include "G4ParticleHPInterpolator.hh"
44#include "G4ParticleHPNames.hh"
45
46class G4ParticleHPElasticFS : public G4ParticleHPFinalState
47{
48 public:
49
50 G4ParticleHPElasticFS()
51 {
52 hasXsec = false;
53 theCoefficients = 0;
54 theProbArray = 0;
55
56 repFlag = 0;
57 tE_of_repFlag3 = 0.0;
58 targetMass = 0.0;
59 frameFlag = 0;
60 }
61 ~G4ParticleHPElasticFS()
62 {
63 if(theCoefficients!=0) delete theCoefficients;
64 if(theProbArray!=0) delete theProbArray;
65 }
66 void Init (G4double A, G4double Z, G4int M, G4String & dirName, G4String & aFSType, G4ParticleDefinition*);
67 G4HadFinalState * ApplyYourself(const G4HadProjectile & theTrack);
68 G4ParticleHPFinalState * New()
69 {
70 G4ParticleHPElasticFS * theNew = new G4ParticleHPElasticFS;
71 return theNew;
72 }
73
74 private:
75 G4int repFlag; // Legendre coeff(1), or probability array(2), or isotropic(0).
76 // add 3: for Legendre (Low Energy) + Probability (High Energy)
77 G4double tE_of_repFlag3; // transition energy in case of repFlag 3:
78 G4double targetMass; // in neutronmass units.
79 G4int frameFlag; // CMS or Lab system.
80
81 G4ParticleHPLegendreStore * theCoefficients; // the legendre coefficients
82 G4ParticleHPPartial * theProbArray; // the probability array p,costh for energy
83 G4ParticleHPInterpolator theInt; // interpolation
84
85 G4ParticleHPFastLegendre theLegend; // fast look-up for leg-integrals
86};
87#endif
