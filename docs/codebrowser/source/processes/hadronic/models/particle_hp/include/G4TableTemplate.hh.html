33#ifndef G4TABLETEMPLATE_HH
34#define G4TABLETEMPLATE_HH
35
36#include <vector>
37
38#include "globals.hh"
39
40#include "G4FFGDefaultValues.hh"
41
42/** G4TableTemplate is essentially a wrapper around a std::vector designed
43 * to work specifically with pointers.
44 */
45template <class T>
46class G4TableTemplate
47{
48public:
49 /** Default constructor */
50 G4TableTemplate( void );
51 /** Adds a container to the table */
52 void G4AddContainer( T* NewContainer );
53 /** Gets a pointer to the table */
54 G4TableTemplate* G4GetTable( void );
55 /** Retrieve a container from the table */
56 T* G4GetContainer( unsigned int WhichContainer );
57 /** Create a new blank container */
58 T* G4GetNewContainer( void );
59 /** Create a new container that is constructed with a G4int */
60 T* G4GetNewContainer( G4int DefaultValue );
61 /** Get the number of elements in the table */
62 G4long G4GetNumberOfElements( void );
63
64private:
65 std::vector<T*> ContainerTable_;
66
67public:
68 ~G4TableTemplate( void );
69};
70
71template <class T>
72G4TableTemplate<T>::
73G4TableTemplate( void )
74{
75 // Nothing to be initialized
76}
77
78template <class T>
79void G4TableTemplate<T>::
80G4AddContainer( T* NewContainer )
81{
82 ContainerTable_.push_back(NewContainer);
83}
84
85template <class T>
86G4TableTemplate<T>* G4TableTemplate<T>::
87G4GetTable( void )
88{
89 return this;
90}
91
92template <class T>
93T* G4TableTemplate<T>::
94G4GetContainer( unsigned int WhichContainer )
95{
96 if(WhichContainer < ContainerTable_.size())
97 {
98 return ContainerTable_[WhichContainer];
99 }
100
101 return NULL;
102}
103
104template <class T>
105T* G4TableTemplate<T>::
106G4GetNewContainer( void )
107{
108 ContainerTable_.push_back(new T);
109
110 return ContainerTable_.back();
111}
112
113template <class T>
114T* G4TableTemplate<T>::
115G4GetNewContainer( G4int DefaultValue )
116{
117 ContainerTable_.push_back(new T(DefaultValue));
118
119 return ContainerTable_.back();
120}
121
122template <class T>
123G4long G4TableTemplate<T>::
124G4GetNumberOfElements( void )
125{
126 return ContainerTable_.size();
127}
128
129template <class T>
130G4TableTemplate<T>::
131~G4TableTemplate()
132{
133 for(unsigned int i = 0; i < ContainerTable_.size(); i++)
134 {
135 delete ContainerTable_[i];
136 }
137}
138
139#endif /* G4TABLETEMPLATE_HH */
