29#ifndef G4ParticleHPInelasticBaseFS_h
30#define G4ParticleHPInelasticBaseFS_h 1
31
32#include "globals.hh"
33#include "G4HadProjectile.hh"
34#include "G4HadFinalState.hh"
35#include "G4ParticleHPFinalState.hh"
36#include "G4ParticleHPAngular.hh"
37#include "G4ParticleHPEnergyDistribution.hh"
38#include "G4ParticleHPEnAngCorrelation.hh"
39#include "G4ParticleHPPhotonDist.hh"
40#include "G4ParticleHPDeExGammas.hh"
41#include "G4Nucleus.hh"
42
43class G4ParticleHPInelasticBaseFS : public G4ParticleHPFinalState
44{
45 public:
46
47 G4ParticleHPInelasticBaseFS()
48 {
49 hasXsec = true;
50 theXsection = new G4ParticleHPVector;
51
52 theEnergyDistribution = 0;
53 theFinalStatePhotons = 0;
54 theEnergyAngData = 0;
55 theAngularDistribution = 0;
56
57 theNuclearMassDifference = 0.0;
58
59 }
60 virtual ~G4ParticleHPInelasticBaseFS()
61 {
62 delete theXsection;
63 if(theEnergyDistribution!=0) delete theEnergyDistribution;
64 if(theFinalStatePhotons!=0) delete theFinalStatePhotons;
65 if(theEnergyAngData!=0) delete theEnergyAngData;
66 if(theAngularDistribution!=0) delete theAngularDistribution;
67 }
68
69 void Init (G4double A, G4double Z, G4int M, G4String & dirName, G4String & bit, G4ParticleDefinition*);
70 void BaseApply(const G4HadProjectile & theTrack, G4ParticleDefinition ** theDefs, G4int nDef);
71 void InitGammas(G4double AR, G4double ZR);
72 virtual G4HadFinalState * ApplyYourself(const G4HadProjectile & theTrack) = 0;
73 virtual G4ParticleHPFinalState * New() = 0;
74
75 virtual G4double GetXsec(G4double anEnergy)
76 {
77 return std::max(0., theXsection->GetY(anEnergy));
78 }
79 virtual G4ParticleHPVector * GetXsec() { return theXsection; }
80
81 protected:
82
83 G4ParticleHPVector * theXsection;
84 G4ParticleHPEnergyDistribution * theEnergyDistribution;
85 G4ParticleHPAngular * theAngularDistribution;
86 G4ParticleHPEnAngCorrelation * theEnergyAngData;
87
88 G4ParticleHPPhotonDist * theFinalStatePhotons;
89 G4double theNuclearMassDifference;
90 G4ParticleHPDeExGammas theGammas;
91 G4String gammaPath;
92
93 private:
94};
95#endif
