29#ifndef G4ParticleHPThermalScatteringNames_h
30#define G4ParticleHPThermalScatteringNames_h 1
31
32// Class Description
33// Name list of Elements for a high precision (based on evaluated data
34// libraries) description of themal neutron scattering below 4 eV;
35// Based on Thermal neutron scattering files
36// from the evaluated nuclear data files ENDF/B-VI, Release2
37// To be used in your physics list in case you need this physics.
38// In this case you want to register an object of this class with
39// the corresponding process.
40// Class Description - End
41
42// 15-Nov-06 First implementation is done by T. Koi (SLAC/SCCS)
43
44#include "globals.hh"
45#include <map>
46
47class G4ParticleHPThermalScatteringNames
48{
49 public:
50
51 G4ParticleHPThermalScatteringNames();
52
53 ~G4ParticleHPThermalScatteringNames();
54
55 G4bool IsThisThermalElement ( G4String );
56 G4bool IsThisThermalElement ( G4String , G4String );
57 size_t GetSize() { return names.size(); };
58 G4String GetTS_NDL_Name( G4String nameG4Element ) { return names.find ( nameG4Element )->second; };
59 G4String GetTS_NDL_Name( G4String material , G4String element ) { return nist_names.find ( std::pair< G4String , G4String > ( material , element ) )->second; };
60 //G4String GetTS_G4E_Name( G4int i ) { return names[i]->first; };
61
62 //For user prepared thermal files
63 //Name of G4Element , Name of NDL file
64 void AddThermalElement ( G4String , G4String );
65
66 private:
67
68// G4Element NDL name
69 std::map< G4String , G4String > names;
70
71// G4Material G4Element NDL name
72 std::map< std::pair < G4String , G4String > , G4String > nist_names;
73
74};
75
76#endif
