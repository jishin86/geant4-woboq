31#ifndef G4VParticleHPEnergyAngular_h
32#define G4VParticleHPEnergyAngular_h 1
33
34#include <fstream>
35
36#include "G4ios.hh"
37#include "globals.hh"
38#include "G4ReactionProduct.hh"
39#include "G4Cache.hh"
40
41class G4VParticleHPEnergyAngular
42{
43
44 struct toBeCached
45 {
46 G4ReactionProduct* theProjectileRP;
47 G4ReactionProduct* theTarget;
48 G4ReactionProduct* theCMS;
49 toBeCached() : theProjectileRP(0),theTarget(0),theCMS(0) {};
50 };
51
52public:
53
54 G4VParticleHPEnergyAngular()
55 {
56 theQValue=0;
57 toBeCached val;
58 fCache.Put( val );
59 }
60
61 virtual ~G4VParticleHPEnergyAngular()
62 {
63 }
64
65public:
66
67 virtual void Init(std::istream & aDataFile) = 0;
68 virtual G4ReactionProduct * Sample(G4double anEnergy,
69 G4double massCode,
70 G4double mass) = 0;
71 virtual G4double MeanEnergyOfThisInteraction() = 0;
72 // returns value cashed in sample
73
74 void SetProjectileRP(G4ReactionProduct * aIncidentParticleRP)
75 {
76 fCache.Get().theProjectileRP = aIncidentParticleRP;
77 }
78
79 void SetTarget(G4ReactionProduct * aTarget)
80 {
81 fCache.Get().theTarget = aTarget;
82 }
83
84 G4ReactionProduct * GetTarget()
85 {
86 return fCache.Get().theTarget;
87 }
88
89 G4ReactionProduct * GetProjectileRP()
90 {
91 return fCache.Get().theProjectileRP;
92 }
93
94 G4ReactionProduct * GetCMS()
95 {
96 *fCache.Get().theCMS = *fCache.Get().theProjectileRP
97 + *fCache.Get().theTarget;
98 return fCache.Get().theCMS;
99 }
100
101 inline void SetQValue(G4double aValue)
102 {
103 theQValue = aValue;
104 }
105
106 virtual void ClearHistories()
107 {
108 }
109
110protected:
111
112 inline G4double GetQValue() { return theQValue; }
113
114private:
115
116 G4double theQValue;
117
118 G4Cache<toBeCached> fCache;
119};
120#endif
