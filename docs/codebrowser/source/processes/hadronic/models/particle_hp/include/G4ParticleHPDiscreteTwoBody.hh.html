29#ifndef G4ParticleHPDiscreteTwoBody_h
30#define G4ParticleHPDiscreteTwoBody_h 1
31
32//101110 Bug fix in MF=6, LAW=2 case; contribution from E. Mendoza, D. Cano-Ott (CIEMAT)
33
34#include "G4ios.hh"
35#include <fstream>
36#include "globals.hh"
37#include "G4VParticleHPEnergyAngular.hh"
38#include "G4ParticleHPLegendreTable.hh"
39#include "G4ParticleHPInterpolator.hh"
40#include "G4InterpolationManager.hh"
41
42class G4ParticleHPDiscreteTwoBody : public G4VParticleHPEnergyAngular
43{
44 public:
45
46 G4ParticleHPDiscreteTwoBody()
47 {
48 theCoeff = 0;
49 bCheckDiffCoeffRepr = true;
50 if ( getenv( "G4PHP_DO_NOT_CHECK_DIFF_COEFF_REPR" ) ) bCheckDiffCoeffRepr = false;
51 nEnergy = 0;
52 }
53 ~G4ParticleHPDiscreteTwoBody()
54 {
55 if(theCoeff!=0) delete [] theCoeff;
56 }
57
58 void Init(std::istream & aDataFile)
59 {
60 aDataFile >> nEnergy;
61 theManager.Init(aDataFile);
62 theCoeff = new G4ParticleHPLegendreTable[nEnergy];
63 for(G4int i=0; i<nEnergy; i++)
64 {
65 G4double energy;
66 G4int aRep, nCoeff;
67 aDataFile >> energy >> aRep >> nCoeff;
68 //- G4cout << this << " " << i << " G4ParticleHPDiscreteTwoBody READ DATA " << energy << " " << aRep << " " << nCoeff << G4endl;
69 energy*=CLHEP::eV;
70 G4int nPoints=nCoeff;
71 if(aRep>0) nPoints*=2;
72 //theCoeff[i].Init(energy, nPoints);
73
74 theCoeff[i].Init(energy, nPoints-1);
75 theCoeff[i].SetRepresentation(aRep);
76 for(G4int ii=0; ii<nPoints; ii++)
77 {
78 G4double y;
79 aDataFile >> y;
80 theCoeff[i].SetCoeff(ii, y);
81 }
82 }
83 }
84
85 G4ReactionProduct * Sample(G4double anEnergy, G4double massCode, G4double mass);
86 G4double MeanEnergyOfThisInteraction() { return -1; }
87
88 private:
89
90 G4int nEnergy;
91 G4InterpolationManager theManager; // knows the interpolation between stores
92 G4ParticleHPLegendreTable * theCoeff;
93
94 private:
95
96 G4ParticleHPInterpolator theInt;
97
98 G4bool bCheckDiffCoeffRepr; // for example ENDF-VII0_proton/Inelastic/F01/4_9_Beryllium has 0 for energy 7.5E+07 and 12 for energy 1.e+08
99};
100#endif
