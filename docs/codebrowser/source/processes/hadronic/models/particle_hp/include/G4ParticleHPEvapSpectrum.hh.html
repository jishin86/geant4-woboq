29#ifndef G4ParticleHPEvapSpectrum_h
30#define G4ParticleHPEvapSpectrum_h 1
31
32#include <fstream>
33#include <CLHEP/Units/SystemOfUnits.h>
34
35#include "globals.hh"
36#include "G4ios.hh"
37#include "Randomize.hh"
38#include "G4ParticleHPVector.hh"
39#include "G4VParticleHPEDis.hh"
40
41// we will need a List of these .... one per term.
42
43class G4ParticleHPEvapSpectrum : public G4VParticleHPEDis
44{
45 public:
46 G4ParticleHPEvapSpectrum()
47 {
48 }
49 ~G4ParticleHPEvapSpectrum()
50 {
51 }
52
53 inline void Init(std::istream & aDataFile)
54 {
55 theFractionalProb.Init(aDataFile);
56 theThetaDist.Init(aDataFile);
57 theXDist.Init(aDataFile);
58 }
59
60 inline G4double GetFractionalProbability(G4double anEnergy)
61 {
62 return theFractionalProb.GetY(anEnergy);
63 }
64
65 inline G4double Sample(G4double anEnergy)
66 {
67 // when this is called, theFractionalProb was used, and 'k' is sorted out already.
68 G4double x = theXDist.Sample();
69 G4double theta = theThetaDist.GetY(anEnergy);
70 G4double result = x*theta;
71 return result*CLHEP::eV;
72 }
73
74 private:
75
76 G4ParticleHPVector theFractionalProb;
77
78 G4ParticleHPVector theThetaDist;
79 G4ParticleHPVector theXDist;
80
81};
82
83#endif
