64G4ParticleHPBGGNucleonInelasticXS::G4ParticleHPBGGNucleonInelasticXS(const G4ParticleDefinition* p)

79G4bool G4ParticleHPBGGNucleonInelasticXS::IsElementApplicable(const G4DynamicParticle* dp,
80 G4int Z,
81 const G4Material*)
82{
83 if ( dp->GetKineticEnergy() < fLowEnergyLimitForHPN ) return false;
84 return (1 < Z);
85}
86
87//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo......
88
89G4bool G4ParticleHPBGGNucleonInelasticXS::IsIsoApplicable(const G4DynamicParticle* dp,
90 G4int Z, G4int A,
91 const G4Element*,
92 const G4Material*)
93{
94 if ( dp->GetKineticEnergy() < fLowEnergyLimitForHPN ) return false;
95 return (1 == Z && 2 >= A);
96}
