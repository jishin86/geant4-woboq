42G4double G4ParticleHPKallbachMannSyst::Sample(G4double anEnergy)
43{
44 G4double result;
45
46 G4double zero = GetKallbachZero(anEnergy);
47 if(zero>1) zero=1.;
48 if(zero<-1)zero=-1.;
49 G4double max = Kallbach(zero, anEnergy);
50 G4double upper = Kallbach(1., anEnergy);
51 G4double lower = Kallbach(-1., anEnergy);
52 if(upper>max) max=upper;
53 if(lower>max) max=lower;
54 G4double value, random;
55
56 G4int icounter=0;
57 G4int icounter_max=1024;
58 do
59 {
60 icounter++;
61 if ( icounter > icounter_max ) {
62 G4cout << "Loop-counter exceeded the threshold value at " << __LINE__ << "th line of " << __FILE__ << "." << G4endl;
63 break;
64 }
65 result = 2.*G4UniformRand()-1;
66 value = Kallbach(result, anEnergy)/max;
67 random = G4UniformRand();
68 }
69 while(random>value); // Loop checking, 11.05.2015, T. Koi
70
71 return result;
72}

74G4double G4ParticleHPKallbachMannSyst::Kallbach(G4double cosTh, G4double anEnergy)
75{
76 // Kallbach-Mann systematics without normalization.
77 G4double result;
78 G4double theX = A(anEnergy)*cosTh;
79 // We need to use here std::exp (and not G4Exp) to avoid underflow/overflow problems
80 // (observed with the physics list QGSP_BIC_AllHP in the version G4 10.5).
81 result = 0.5*(std::exp( theX)*(1+theCompoundFraction)
82 +std::exp(-theX)*(1-theCompoundFraction));
83 return result;
84}

86G4double G4ParticleHPKallbachMannSyst::GetKallbachZero(G4double anEnergy)
87{
88 G4double result;
89 //delta 2.0e-16 in not good.
90 //delta 4.0e-16 is OK
91 //safety factor of 2
92 G4double delta = 8.0e-16;
93 if ( std::abs (theCompoundFraction - 1 ) < delta ) {
94 theCompoundFraction = 1.0-delta;
95 }
96 result = 0.5 * (1./A(anEnergy)) * G4Log((1-theCompoundFraction)/(1+theCompoundFraction));
97 return result;
98}

100G4double G4ParticleHPKallbachMannSyst::A(G4double anEnergy)
101{
102 G4double result;
103 G4double C1 = 0.04/MeV;
104 G4double C2 = 1.8E-6/(MeV*MeV*MeV);
105 G4double C3 = 6.7E-7/(MeV*MeV*MeV*MeV);
106
107 G4double epsa = anEnergy*theTargetMass/(theTargetMass+theIncidentMass);
108 G4int Ac = theTargetA+1;
109 G4int Nc = Ac - theTargetZ;
110 G4int AA = theTargetA;
111 G4int ZA = theTargetZ;
112 G4double ea = epsa+SeparationEnergy(Ac, Nc, AA, ZA);
113 G4double Et1 = 130*MeV;
114 G4double R1 = std::min(ea, Et1);
115 // theProductEnergy is still in CMS!!!
116 G4double epsb = theProductEnergy*(theProductMass+theResidualMass)/theResidualMass;
117 G4int AB = theResidualA;
118 G4int ZB = theResidualZ;
119 G4double eb = epsb+SeparationEnergy(Ac, Nc, AB, ZB );
120 G4double X1 = R1*eb/ea;
121 G4double Et3 = 41*MeV;
122 G4double R3 = std::min(ea, Et3);
123 G4double X3 = R3*eb/ea;
124 G4double Ma = 1;
125 G4double mb(0);
126 G4int productA = theTargetA+1-theResidualA;
127 G4int productZ = theTargetZ-theResidualZ;
128 if(productZ==0)
129 {
130 mb = 0.5;
131 }
132 else if(productZ==1)
133 {
134 mb = 1;
135 }
136 else if(productZ==2)
137 {
138 mb = 2;
139 if(productA==3) mb=1;
140 }
141 else
142 {
143 throw G4HadronicException(__FILE__, __LINE__, "Severe error in the sampling of Kallbach-Mann Systematics");
144 }
145
146 result = C1*X1 + C2*G4Pow::GetInstance()->powN(X1, 3) + C3*Ma*mb*G4Pow::GetInstance()->powN(X3, 4);
147 return result;
148}

150G4double G4ParticleHPKallbachMannSyst::SeparationEnergy(G4int Ac, G4int Nc, G4int AA, G4int ZA)
151{
152 G4double result;
153 G4int NA = AA-ZA;
154 G4int Zc = Ac-Nc;
155 result = 15.68*(Ac-AA);
156 result += -28.07*((Nc-Zc)*(Nc-Zc)/Ac - (NA-ZA)*(NA-ZA)/AA);
157 result += -18.56*(G4Pow::GetInstance()->A23(G4double(Ac)) - G4Pow::GetInstance()->A23(G4double(AA)));
158 result += 33.22*((Nc-Zc)*(Nc-Zc)/G4Pow::GetInstance()->powA(G4double(Ac), 4./3.) - (NA-ZA)*(NA-ZA)/G4Pow::GetInstance()->powA(G4double(AA), 4./3.));
159 result += -0.717*(Zc*Zc/G4Pow::GetInstance()->A13(G4double(Ac))-ZA*ZA/G4Pow::GetInstance()->A13(G4double(AA)));
160 result += 1.211*(Zc*Zc/Ac-ZA*ZA/AA);
161 G4double totalBinding(0);
162 G4int productA = theTargetA+1-theResidualA;
163 G4int productZ = theTargetZ-theResidualZ;
164 if(productZ==0&&productA==1) totalBinding=0;
165 if(productZ==1&&productA==1) totalBinding=0;
166 if(productZ==1&&productA==2) totalBinding=2.22;
167 if(productZ==1&&productA==3) totalBinding=8.48;
168 if(productZ==2&&productA==3) totalBinding=7.72;
169 if(productZ==2&&productA==4) totalBinding=28.3;
170 result += -totalBinding;
171 result *= MeV;
172 return result;
173}
