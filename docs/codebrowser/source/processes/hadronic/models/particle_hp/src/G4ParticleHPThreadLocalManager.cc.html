58void G4ParticleHPThreadLocalManager::OpenReactionWhiteBoard()
59{
60 if ( RWB )
61 {
62 G4cout << "Warning: G4ParticleHPReactionWhiteBoard is tried doubly opening" << G4endl;
63 return;
64 }
65
66 RWB = new G4ParticleHPReactionWhiteBoard();
67}
68
69G4ParticleHPReactionWhiteBoard*
70G4ParticleHPThreadLocalManager::GetReactionWhiteBoard()
71{
72 if ( !RWB )
73 {
74 G4cout << "Warning: try to access G4ParticleHPReactionWhiteBoard before opening" << G4endl;
75 RWB = new G4ParticleHPReactionWhiteBoard();
76 }
77 return RWB;
78}
79
80void G4ParticleHPThreadLocalManager::CloseReactionWhiteBoard()
81{
82 delete RWB;
83 RWB=0;
84}
