197G4int G4ParticleHPInelasticCompFS::SelectExitChannel(G4double eKinetic)
198{
199
200// it = 0 has without Photon
201 G4double running[50];
202 running[0] = 0;
203 unsigned int i;
204 for(i=0; i<50; i++)
205 {
206 if(i!=0) running[i]=running[i-1];
207 if(theXsection[i] != 0)
208 {
209 running[i] += std::max(0., theXsection[i]->GetXsec(eKinetic));
210 }
211 }
212 G4double random = G4UniformRand();
213 G4double sum = running[49];
214 G4int it = 50;
215 if(0!=sum)
216 {
217 G4int i0;
218 for(i0=0; i0<50; i0++)
219 {
220 it = i0;
221 // G4cout << " SelectExitChannel " << it << " " << random << " " << running[i0]/sum << " " << running[i0] << G4endl; //GDEB
222 if(random < running[i0]/sum) break;
223 }
224 }
225//debug: it = 1;
226// G4cout << " SelectExitChannel " << it << " " << sum << G4endl; //GDEB
227 return it;
228}

232void G4ParticleHPInelasticCompFS::CompositeApply(const G4HadProjectile & theTrack, G4ParticleDefinition * aDefinition)
233{
234
235// prepare neutron
236 if ( theResult.Get() == NULL ) theResult.Put( new G4HadFinalState );
237 theResult.Get()->Clear();
238 G4double eKinetic = theTrack.GetKineticEnergy();
239 const G4HadProjectile *hadProjectile = &theTrack;
240 G4ReactionProduct incidReactionProduct( const_cast<G4ParticleDefinition *>(hadProjectile->GetDefinition()) ); // incidReactionProduct
241 incidReactionProduct.SetMomentum( hadProjectile->Get4Momentum().vect() );
242 incidReactionProduct.SetKineticEnergy( eKinetic );
243
244// prepare target
245 G4int i;
246 for(i=0; i<50; i++)
247 { if(theXsection[i] != 0) { break; } }
248
249 G4double targetMass=0;
250 G4double eps = 0.0001;
251 targetMass = G4NucleiProperties::GetNuclearMass(static_cast<G4int>(theBaseA+eps), static_cast<G4int>(theBaseZ+eps));
252#ifdef G4PHPDEBUG
253 if( getenv("G4ParticleHPDebug")) G4cout <<this <<" G4ParticleHPInelasticCompFS::CompositeApply A " <<theBaseA <<" Z " <<theBaseZ <<" incident " <<hadProjectile->GetDefinition()->GetParticleName() <<G4endl;
254#endif
255// if(theEnergyAngData[i]!=0)
256// targetMass = theEnergyAngData[i]->GetTargetMass();
257// else if(theAngularDistribution[i]!=0)
258// targetMass = theAngularDistribution[i]->GetTargetMass();
259// else if(theFinalStatePhotons[50]!=0)
260// targetMass = theFinalStatePhotons[50]->GetTargetMass();
261 G4ReactionProduct theTarget;
262 G4Nucleus aNucleus;
263 //G4ThreeVector neuVelo = (1./hadProjectile->GetDefinition()->GetPDGMass())*incidReactionProduct.GetMomentum();
264 //theTarget = aNucleus.GetBiasedThermalNucleus( targetMass/hadProjectile->GetDefinition()->GetPDGMass() , neuVelo, theTrack.GetMaterial()->GetTemperature());
265 //G4Nucleus::GetBiasedThermalNucleus requests normalization of mass and velocity in neutron mass
266 G4ThreeVector neuVelo = ( 1./G4Neutron::Neutron()->GetPDGMass() )*incidReactionProduct.GetMomentum();
267 theTarget = aNucleus.GetBiasedThermalNucleus( targetMass/G4Neutron::Neutron()->GetPDGMass()
268 , neuVelo, theTrack.GetMaterial()->GetTemperature() );
269
270 theTarget.SetDefinition( G4IonTable::GetIonTable()->GetIon( G4int(theBaseZ), G4int(theBaseA) , 0.0 ) ); //XX
271
272// prepare the residual mass
273 G4double residualMass=0;
274 G4double residualZ = theBaseZ + theProjectile->GetPDGCharge() - aDefinition->GetPDGCharge();
275 G4double residualA = theBaseA + theProjectile->GetBaryonNumber() - aDefinition->GetBaryonNumber();
276 residualMass = G4NucleiProperties::GetNuclearMass(static_cast<G4int>(residualA+eps), static_cast<G4int>(residualZ+eps));
277
278// prepare energy in target rest frame
279 G4ReactionProduct boosted;
280 boosted.Lorentz(incidReactionProduct, theTarget);
281 eKinetic = boosted.GetKineticEnergy();
282// G4double momentumInCMS = boosted.GetTotalMomentum();
283
284// select exit channel for composite FS class.
285 G4int it = SelectExitChannel( eKinetic );
286
287// set target and neutron in the relevant exit channel
288 InitDistributionInitialState(incidReactionProduct, theTarget, it);
289
290 //---------------------------------------------------------------------//
291 //Hook for NRESP71MODEL
292 if ( G4ParticleHPManager::GetInstance()->GetUseNRESP71Model() ) {
293 if ( (G4int)(theBaseZ+0.1) == 6 ) // If the reaction is with Carbon...
294 {
295 if ( theProjectile == G4Neutron::Definition() ) {
296 if ( use_nresp71_model( aDefinition , it , theTarget , boosted ) ) return;
297 }
298 }
299 }
300 //---------------------------------------------------------------------//
301
302 G4ReactionProductVector * thePhotons = 0;
303 G4ReactionProductVector * theParticles = 0;
304 G4ReactionProduct aHadron;
305 aHadron.SetDefinition(aDefinition); // what if only cross-sections exist ==> Na 23 11 @@@@
306 G4double availableEnergy = incidReactionProduct.GetKineticEnergy() + incidReactionProduct.GetMass() - aHadron.GetMass() +
307 (targetMass - residualMass);
308//080730c
309 if ( availableEnergy < 0 )
310 {
311 //G4cout << "080730c Adjust availavleEnergy " << G4endl;
312 availableEnergy = 0;
313 }
314 G4int nothingWasKnownOnHadron = 0;
315 G4int dummy;
316 G4double eGamm = 0;
317 G4int iLevel=it-1;
318
319// TK without photon has it = 0
320 if( 50 == it )
321 {
322
323// TK Excitation level is not determined
324 iLevel=-1;
325 aHadron.SetKineticEnergy(availableEnergy*residualMass/
326 (aHadron.GetMass()+residualMass));
327
328 //aHadron.SetMomentum(incidReactionProduct.GetMomentum()*(1./incidReactionProduct.GetTotalMomentum())*
329 // std::sqrt(aHadron.GetTotalEnergy()*aHadron.GetTotalEnergy()-
330 // aHadron.GetMass()*aHadron.GetMass()));
331
332 //TK add safty 100909
333 G4double p2 = ( aHadron.GetTotalEnergy()*aHadron.GetTotalEnergy() - aHadron.GetMass()*aHadron.GetMass() );
334 G4double p = 0.0;
335 if ( p2 > 0.0 ) p = std::sqrt( p );
336
337 aHadron.SetMomentum(incidReactionProduct.GetMomentum()*(1./incidReactionProduct.GetTotalMomentum())*p );
338
339 }
340 else
341 {
342 while ( iLevel!=-1 && theGammas.GetLevel(iLevel) == 0 ) { iLevel--; } // Loop checking, 11.05.2015, T. Koi
343 }
344
345
346 if ( theAngularDistribution[it] != 0 ) // MF4
347 {
348 if(theEnergyDistribution[it]!=0) // MF5
349 {
350 //************************************************************
351 /*
352 aHadron.SetKineticEnergy(theEnergyDistribution[it]->Sample(eKinetic, dummy));
353 G4double eSecN = aHadron.GetKineticEnergy();
354 */
355 //************************************************************
356 //EMendoza --> maximum allowable energy should be taken into account.
357 G4double dqi = 0.0;
358 if ( QI[it] < 0 || 849 < QI[it] ) dqi = QI[it]; //For backword compatibility QI introduced since G4NDL3.15
359 G4double MaxEne=eKinetic+dqi;
360 G4double eSecN;
361
362 G4int icounter=0;
363 G4int icounter_max=1024;
364 do {
365 icounter++;
366 if ( icounter > icounter_max ) {
367 G4cout << "Loop-counter exceeded the threshold value at " << __LINE__ << "th line of " << __FILE__ << "." << G4endl;
368 break;
369 }
370 eSecN=theEnergyDistribution[it]->Sample(eKinetic, dummy);
371 }while(eSecN>MaxEne); // Loop checking, 11.05.2015, T. Koi
372 aHadron.SetKineticEnergy(eSecN);
373 //************************************************************
374 eGamm = eKinetic-eSecN;
375 for(iLevel=theGammas.GetNumberOfLevels()-1; iLevel>=0; iLevel--)
376 {
377 if(theGammas.GetLevelEnergy(iLevel)<eGamm) break;
378 }
379 G4double random = 2*G4UniformRand();
380 iLevel+=G4int(random);
381 if(iLevel>theGammas.GetNumberOfLevels()-1)iLevel = theGammas.GetNumberOfLevels()-1;
382 }
383 else
384 {
385 G4double eExcitation = 0;
386 if(iLevel>=0) eExcitation = theGammas.GetLevel(iLevel)->GetLevelEnergy();
387 while (eKinetic-eExcitation < 0 && iLevel>0) // Loop checking, 11.05.2015, T. Koi
388 {
389 iLevel--;
390 eExcitation = theGammas.GetLevel(iLevel)->GetLevelEnergy();
391 }
392 //110610TK BEGIN
393 //Use QI value for calculating excitation energy of residual.
394 G4bool useQI=false;
395 G4double dqi = QI[it];
396 if ( dqi < 0 || 849 < dqi ) useQI = true; //Former libraies does not have values of this range
397
398 if ( useQI )
399 {
400 // QI introudced since G4NDL3.15
401 eExcitation = -QI[it];
402 //Re-evluate iLevel based on this eExcitation
403 iLevel = 0;
404 G4bool find = false;
405 G4int imaxEx = 0;
406 G4double level_tolerance = 1.0*CLHEP::keV;
407
408 while( theGammas.GetLevel(iLevel+1) != 0 ) // Loop checking, 11.05.2015, T. Koi
409 {
410 G4double maxEx = 0.0;
411 if ( maxEx < theGammas.GetLevel(iLevel)->GetLevelEnergy() )
412 {
413 maxEx = theGammas.GetLevel(iLevel)->GetLevelEnergy();
414 imaxEx = iLevel;
415 }
416
417 // Fix bug 1789 DHW - first if-branch added because gamma data come from ENSDF
418 // and do not necessarily match the excitations used in ENDF-B.VII
419 // Compromise solution: use 1 keV tolerance suggested by T. Koi
420 if (std::abs(eExcitation - theGammas.GetLevel(iLevel)->GetLevelEnergy() ) < level_tolerance) {
421 find = true;
422 break;
423
424 } else if (eExcitation < theGammas.GetLevel(iLevel)->GetLevelEnergy() ) {
425 find = true;
426 iLevel--;
427 // very small eExcitation, iLevel becomes -1, this is protected below.
428 if ( iLevel == -1 ) iLevel = 0; // But cause energy trouble.
429 break;
430 }
431 iLevel++;
432 }
433 // In case, cannot find proper level, then use the maximum level.
434 if ( !find ) iLevel = imaxEx;
435 }
436 //110610TK END
437
438 if(getenv("G4ParticleHPDebug") && eKinetic-eExcitation < 0)
439 {
440 throw G4HadronicException(__FILE__, __LINE__, "SEVERE: InelasticCompFS: Consistency of data not good enough, please file report");
441 }
442 if(eKinetic-eExcitation < 0) eExcitation = 0;
443 if(iLevel!= -1) aHadron.SetKineticEnergy(eKinetic - eExcitation);
444
445 }
446 theAngularDistribution[it]->SampleAndUpdate(aHadron);
447
448 if( theFinalStatePhotons[it] == 0 )
449 {
450 //G4cout << "110610 USE Gamma Level" << G4endl;
451// TK comment Most n,n* eneter to this
452 thePhotons = theGammas.GetDecayGammas(iLevel);
453 eGamm -= theGammas.GetLevelEnergy(iLevel);
454 if(eGamm>0) // @ ok for now, but really needs an efficient way of correllated sampling @
455 {
456 G4ReactionProduct * theRestEnergy = new G4ReactionProduct;
457 theRestEnergy->SetDefinition(G4Gamma::Gamma());
458 theRestEnergy->SetKineticEnergy(eGamm);
459 G4double costh = 2.*G4UniformRand()-1.;
460 G4double phi = CLHEP::twopi*G4UniformRand();
461 theRestEnergy->SetMomentum(eGamm*std::sin(std::acos(costh))*std::cos(phi),
462 eGamm*std::sin(std::acos(costh))*std::sin(phi),
463 eGamm*costh);
464 if(thePhotons == 0) { thePhotons = new G4ReactionProductVector; }
465 thePhotons->push_back(theRestEnergy);
466 }
467 }
468 }
469 else if(theEnergyAngData[it] != 0) // MF6
470 {
471
472 theParticles = theEnergyAngData[it]->Sample(eKinetic);
473
474 //141017 Fix BEGIN
475 //Adjust A and Z in the case of miss much between selected data and target nucleus
476 if ( theParticles != NULL ) {
477 G4int sumA = 0;
478 G4int sumZ = 0;
479 G4int maxA = 0;
480 G4int jAtMaxA = 0;
481 for ( G4int j = 0 ; j != (G4int)theParticles->size() ; j++ ) {
482 if ( theParticles->at(j)->GetDefinition()->GetBaryonNumber() > maxA ) {
483 maxA = theParticles->at(j)->GetDefinition()->GetBaryonNumber();
484 jAtMaxA = j;
485 }
486 sumA += theParticles->at(j)->GetDefinition()->GetBaryonNumber();
487 sumZ += G4int( theParticles->at(j)->GetDefinition()->GetPDGCharge() + eps );
488 }
489 G4int dA = (G4int)theBaseA + hadProjectile->GetDefinition()->GetBaryonNumber() - sumA;
490 G4int dZ = (G4int)theBaseZ + G4int( hadProjectile->GetDefinition()->GetPDGCharge() + eps ) - sumZ;
491 if ( dA < 0 || dZ < 0 ) {
492 G4int newA = theParticles->at(jAtMaxA)->GetDefinition()->GetBaryonNumber() + dA ;
493 G4int newZ = G4int( theParticles->at(jAtMaxA)->GetDefinition()->GetPDGCharge() + eps ) + dZ;
494 G4ParticleDefinition* pd = G4IonTable::GetIonTable()->GetIon ( newZ , newA );
495 theParticles->at( jAtMaxA )->SetDefinition( pd );
496 }
497 }
498 //141017 Fix END
499
500 }
501 else
502 {
503 // @@@ what to do, if we have photon data, but no info on the hadron itself
504 nothingWasKnownOnHadron = 1;
505 }
506
507 //G4cout << "theFinalStatePhotons it " << it << G4endl;
508 //G4cout << "theFinalStatePhotons[it] " << theFinalStatePhotons[it] << G4endl;
509 //G4cout << "theFinalStatePhotons it " << it << G4endl;
510 //G4cout << "theFinalStatePhotons[it] " << theFinalStatePhotons[it] << G4endl;
511 //G4cout << "thePhotons " << thePhotons << G4endl;
512
513 if ( theFinalStatePhotons[it] != 0 )
514 {
515 // the photon distributions are in the Nucleus rest frame.
516 // TK residual rest frame
517 G4ReactionProduct boosted_tmp;
518 boosted_tmp.Lorentz(incidReactionProduct, theTarget);
519 G4double anEnergy = boosted_tmp.GetKineticEnergy();
520 thePhotons = theFinalStatePhotons[it]->GetPhotons(anEnergy);
521 G4double aBaseEnergy = theFinalStatePhotons[it]->GetLevelEnergy();
522 G4double testEnergy = 0;
523 if(thePhotons!=0 && thePhotons->size()!=0)
524 { aBaseEnergy-=thePhotons->operator[](0)->GetTotalEnergy(); }
525 if(theFinalStatePhotons[it]->NeedsCascade())
526 {
527 while(aBaseEnergy>0.01*CLHEP::keV) // Loop checking, 11.05.2015, T. Koi
528 {
529 // cascade down the levels
530 G4bool foundMatchingLevel = false;
531 G4int closest = 2;
532 G4double deltaEold = -1;
533 for(G4int j=1; j<it; j++)
534 {
535 if(theFinalStatePhotons[j]!=0)
536 {
537 testEnergy = theFinalStatePhotons[j]->GetLevelEnergy();
538 }
539 else
540 {
541 testEnergy = 0;
542 }
543 G4double deltaE = std::abs(testEnergy-aBaseEnergy);
544 if(deltaE<0.1*CLHEP::keV)
545 {
546 G4ReactionProductVector * theNext =
547 theFinalStatePhotons[j]->GetPhotons(anEnergy);
548 if ( thePhotons != NULL ) thePhotons->push_back(theNext->operator[](0));
549 aBaseEnergy = testEnergy-theNext->operator[](0)->GetTotalEnergy();
550 delete theNext;
551 foundMatchingLevel = true;
552 break; // ===>
553 }
554 if(theFinalStatePhotons[j]!=0 && ( deltaE<deltaEold||deltaEold<0.) )
555 {
556 closest = j;
557 deltaEold = deltaE;
558 }
559 } // <=== the break goes here.
560 if(!foundMatchingLevel)
561 {
562 G4ReactionProductVector * theNext =

812void G4ParticleHPInelasticCompFS::two_body_reaction ( G4DynamicParticle* proj, G4DynamicParticle* targ, G4DynamicParticle* hadron, G4double mu )
813{
814
815// Target rest flame
816// 4vector in targ rest frame;
817// targ could have excitation energy (photon energy will be emiited) tricky but,,,
818
819 G4LorentzVector before = proj->Get4Momentum() + targ->Get4Momentum();
820
821 G4ThreeVector p3_proj = proj->GetMomentum();
822 G4ThreeVector d = p3_proj.unit();
823 G4RotationMatrix rot;
824 G4RotationMatrix rot1;
825 rot1.setPhi( CLHEP::pi/2 + d.phi() );
826 G4RotationMatrix rot2;
827 rot2.setTheta( d.theta() );
828 rot=rot2*rot1;
829 proj->SetMomentum( rot*p3_proj );
830
831// Now proj only has pz component;
832
833// mu in CM system
834
835 //Valid only for neutron incidence
836 G4DynamicParticle* residual = new G4DynamicParticle ( G4IonTable::GetIonTable()->GetIon ( (G4int)( targ->GetDefinition()->GetPDGCharge() - hadron->GetDefinition()->GetPDGCharge() ) , (G4int)(targ->GetDefinition()->GetBaryonNumber() - hadron->GetDefinition()->GetBaryonNumber()+1) , 0 ) , G4ThreeVector(0) );
837
838 G4double Q = proj->GetDefinition()->GetPDGMass() + targ->GetDefinition()->GetPDGMass()
839 - ( hadron->GetDefinition()->GetPDGMass() + residual->GetDefinition()->GetPDGMass() );
840
841 // Non Relativistic Case
842 G4double A = targ->GetDefinition()->GetPDGMass() / proj->GetDefinition()->GetPDGMass();
843 G4double AA = hadron->GetDefinition()->GetPDGMass() / proj->GetDefinition()->GetPDGMass();
844 G4double E1 = proj->GetKineticEnergy();
845
846// 101111
847// In _nat_ data (Q+E1) could become negative value, following line is safty for this case.
848 //if ( (Q+E1) < 0 )
849 if ( ( 1 + (1+A)/A*Q/E1 ) < 0 )
850 {
851// 1.0e-6 eV is additional safty for numeric precision
852 Q = -( A/(1+A)*E1 ) + 1.0e-6*CLHEP::eV;
853 }
854
855 G4double beta = std::sqrt ( A*(A+1-AA)/AA*( 1 + (1+A)/A*Q/E1 ) );
856 G4double gamma = AA/(A+1-AA)*beta;
857 G4double E3 = AA/G4Pow::GetInstance()->powN((1+A),2)*(beta*beta+1+2*beta*mu)*E1;
858 G4double omega3 = (1+beta*mu)/std::sqrt(beta*beta+1+2*beta*mu);
859 if ( omega3 > 1.0 ) omega3 = 1.0;
860
861 G4double E4 = (A+1-AA)/G4Pow::GetInstance()->powN((1+A),2)*(gamma*gamma+1-2*gamma*mu)*E1;
862 G4double omega4 = (1-gamma*mu)/std::sqrt(gamma*gamma+1-2*gamma*mu);
863 if ( omega4 > 1.0 ) omega4 = 1.0;
864
865 hadron->SetKineticEnergy ( E3 );
866
867 G4double M = hadron->GetDefinition()->GetPDGMass();
868 G4double pmag = std::sqrt ((E3+M)*(E3+M)-M*M) ;
869 G4ThreeVector p ( 0 , pmag*std::sqrt(1-omega3*omega3), pmag*omega3 );
870
871 G4double M4 = residual->GetDefinition()->GetPDGMass();
872 G4double pmag4 = std::sqrt ((E4+M4)*(E4+M4)-M4*M4) ;
873 G4ThreeVector p4 ( 0 , -pmag4*std::sqrt(1-omega4*omega4), pmag4*omega4 );
874
875// Rotate to orginal target rest flame.
876 p *= rot.inverse();
877 hadron->SetMomentum( p );
878// Now hadron had 4 momentum in target rest flame
879
880// TypeA
881 p4 *= rot.inverse();
882 residual->SetMomentum ( p4 );
883
884//TypeB1
885 //residual->Set4Momentum ( p4_residual );
886//TypeB2
887 //residual->SetMomentum ( p4_residual.v() );
888
889// Type A make difference in Momenutum
890// Type B1 make difference in Mass of residual
891// Type B2 make difference in total energy.
892
893 delete residual;
894
