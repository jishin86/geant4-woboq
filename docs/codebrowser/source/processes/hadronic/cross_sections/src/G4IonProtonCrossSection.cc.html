40#include "G4SystemOfUnits.hh"

46using namespace std;
47
48G4IonProtonCrossSection::G4IonProtonCrossSection()
49 : G4VCrossSectionDataSet("InvProtonXS")
50{
51 theForward = new G4ParticleInelasticXS(G4Proton::Proton());
52}
53
54G4IonProtonCrossSection::~G4IonProtonCrossSection()
55{}
56
57G4bool
58G4IonProtonCrossSection::IsElementApplicable(const G4DynamicParticle*,
59 G4int Z, const G4Material*)
60{
61 return (1 == Z);
62}
63
64G4double
65G4IonProtonCrossSection::GetElementCrossSection(
66 const G4DynamicParticle* dp,
67 G4int, const G4Material*)
68{
69 const G4ParticleDefinition* p = dp->GetDefinition();
70 G4double e = dp->GetKineticEnergy()*CLHEP::proton_mass_c2/p->GetPDGMass();
71 G4int Z = p->GetAtomicNumber();
72 G4int A = p->GetAtomicMass();
73 return theForward->IsoCrossSection(e, Z, A);
74}
75
76void G4IonProtonCrossSection::BuildPhysicsTable(const G4ParticleDefinition& part)
77{
78 theForward->BuildPhysicsTable(part);
79}
80
81void
82G4IonProtonCrossSection::CrossSectionDescription(std::ostream& outFile) const
83{
84 outFile << "G4IonProtonCrossSection calculates the inelastic cross section\n"
85 << "for any ion projectile with Z >=2 only on hydrogen target.\n"
86 << "It uses the inverse kinematics and the G4ParticleInelasticXS\n"
87 << "cross section.\n";
88}
