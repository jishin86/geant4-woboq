1260// Method to calculate cross sections for all processes.
1261// To facilitate comparison with the original Fortran source, and because
1262// of the interdependence of the elastic and inelastic cross section
1263// calculations, this has not been split into separate processes.
1264
1265void
1266G4HadronCrossSections::CalcScatteringCrossSections(
1267 const G4DynamicParticle* aParticle,
1268 G4int ZZ, G4int AA)
1269{
1270 G4double sigel, sigin, sigtot;
1271 G4double xsecel, xsecin=0;
1272 xsecel = 0;
1273 G4double dx, dy, rc, rca, b;
1274 G4double crel, crin;
1275 G4double xspiel, xspiin;
1276
1277 G4int ipart = GetParticleCode(aParticle);
1278
1279 // Protection in the case "IsApplicable" is not invoked
1280 if ( ipart <= 0 ) {
1281 G4ExceptionDescription ed;
1282 ed << aParticle->GetDefinition()->GetParticleName() << " NOT allowed ! ";
1283 G4Exception( "G4HadronCrossSections::CalcScatteringCrossSections(...)",
1284 "hadXsec001", JustWarning, ed );
1285 return;
1286 }
1287
1288 prevAA = AA;
1289 prevZZ = ZZ;
1290 prevParticleDefinition = aParticle->GetDefinition();
1291 prevKineticEnergy = aParticle->GetKineticEnergy();
1292
1293 if (verboseLevel > 1) {
1294 G4cout << "G4HadronCrossSections::ScatteringCrossSections: "
1295 << aParticle->GetDefinition()->GetParticleName()
1296 << " E(MeV)= " << aParticle->GetKineticEnergy()/MeV
1297 << " Z= " << ZZ << " A= " << AA << G4endl;
1298 }
1299
1300 // Ions...
1301
1302 if (ipart >= 30 && ipart <= 32) {
1303
1304 G4double apart=0;
1305 if (ipart == 30) apart = g4pow->Z13(2);
1306 else if (ipart == 31) apart = g4pow->Z13(3);
1307 else if (ipart == 32) apart = g4pow->Z13(4);
1308
1309 G4double term = apart + g4pow->Z13(AA);
1310 sigin = 49.*term*term;
1311 // Convert cross section from mb to default units
1312 siginelastic = sigin*millibarn;
1313 if(prevKineticEnergy < 6*MeV) siginelastic = 0;
1314 sigelastic = 0.;
1315 return;
1316 }
1317
1318 G4double ek = prevKineticEnergy/GeV;
1319
1320// Low energy neutrons...
1321
1322 if (ipart == 16 && ek <= 0.0327) {
1323
1324 // G4int je2 = NELAB;
1325 // for (G4int j = 2; j <= NELAB; j++) {
1326 // if (ek < elab[j - 1]) {
1327 // je2 = j;
1328 // break;
1329 // }
1330 // }
1331 // G4int je1 = je2 -1;
1332 // je1 = je1 - 1; // For array indexing
1333 // je2 = je2 - 1; // For array indexing
1334
1335 G4int je1 = 0;
1336 G4int je2 = NELAB - 1;
1337 do {
1338 G4int midBin = (je1 + je2)/2;
1339 if (ek < elab[midBin])
1340 je2 = midBin;
1341 else
1342 je1 = midBin;
1343 } while (je2 - je1 > 1); /* Loop checking, 08.01.2016, W. Pokorski */
1344
1345 G4double delab = elab[je2] - elab[je1];
1346
1347 // G4int ja2 = NCNLW;
1348 // for (G4int jj = 2; jj <= NCNLW; jj++) {
1349 // if (a < cnlwat[jj - 1]) {
1350 // ja2 = jj;
1351 // break;
1352 // }
1353 // }
1354 // G4int ja1 = ja2 - 1;
1355 // ja1 = ja1 - 1; // For array indexing
1356 // ja2 = ja2 - 1; // For array indexing
1357
1358 G4int ja1 = 0;
1359 G4int ja2 = NCNLW - 1;
1360 do {
1361 G4int midBin = (ja1 + ja2)/2;
1362 if (AA < cnlwat[midBin])
1363 ja2 = midBin;
1364 else
1365 ja1 = midBin;
1366 } while (ja2 - ja1 > 1); /* Loop checking, 08.01.2016, W. Pokorski */
1367
1368 G4double dnlwat = cnlwat[ja2] - cnlwat[ja1];
1369
1370// Elastic cross section:
1371 // E interpolation or extrapolation at JA1
1372 dy = cnlwel[ja1][je2] - cnlwel[ja1][je1];
1373 G4double rce = dy/delab;
1374 // A interpolation or extrapolation at JE1
1375 dy = cnlwel[ja2][je1] - cnlwel[ja1][je1];
1376 rca = dy/dnlwat;
1377 b = cnlwel[ja1][je1] - rce*elab[je1] - rca*cnlwat[ja1];
1378 sigelastic = rce*ek + rca*AA + b;
1379// Inelastic cross section:
1380 // E interpolation or extrapolation at JA1
1381 dy = cnlwin[ja1][je2] - cnlwin[ja1][je1];
1382 rce = dy/delab;
1383 // A interpolation or extrapolation at JE1
1384 dy = cnlwin[ja2][je1] - cnlwin[ja1][je1];
1385 rca = dy/dnlwat;
1386 b = cnlwin[ja1][je1] - rce*elab[je1] - rca*cnlwat[ja1];
1387 siginelastic = rce*ek + rca*AA + b;
1388 // Convert cross sections from mb to default units
1389 sigelastic = sigelastic*millibarn;
1390 siginelastic = siginelastic*millibarn;
1391 return;
1392 }
1393
1394// Remaining particles...
1395
1396// Get momentum bin
1397 G4double p = aParticle->GetTotalMomentum()/GeV;
1398
1399 // G4int j = TSIZE - 1;
1400 // for (G4int i = 2; i <= TSIZE; i++) {
1401 // if (p < plab[i]) {
1402 // j = i - 1;
1403 // break;
1404 // }
1405 // }
1406 // j = j - 1; // For array indexing
1407
1408 G4int je1 = 0;
1409 G4int je2 = TSIZE - 1;
1410 do {
1411 G4int midBin = (je1 + je2)/2;
1412 if (p < plab[midBin])
1413 je2 = midBin;
1414 else
1415 je1 = midBin;
1416 } while (je2 - je1 > 1); /* Loop checking, 08.01.2016, W. Pokorski */
1417
1418 G4int ipart1 = ipart - 1; // For array indexing
1419
1420// Get cross sections for scattering on free protons
1421 dx = plab[je2] - plab[je1];
1422// Elastic cross section
1423 dy = csel[ipart1][je2] - csel[ipart1][je1];
1424 rc = dy/dx;
1425 b = csel[ipart1][je1] - rc*plab[je1];
1426 sigel = rc*p + b;
1427// Inelastic cross section
1428 dy = csin[ipart1][je2] - csin[ipart1][je1];
1429 rc = dy/dx;
1430 b = csin[ipart1][je1] - rc*plab[je1];
1431 sigin = rc*p + b;
1432 if (verboseLevel > 1) {
1433 G4cout << "sigel " << sigel << G4endl;
1434 G4cout << "sigin " << sigin << G4endl;
1435 }
1436 G4double alph = alpha[ipart1];
1437 if (ipart < 14) {
1438 dy = alphac[je2] - alphac[je1];
1439 rc = dy/dx;
1440 b = alphac[je1] - rc*plab[je1];
1441 G4double corfac = rc*p + b;
1442 alph = alph*corfac;
1443 G4int ipart3 = ipart2[ipart - 7];
1444 ipart3 = ipart3 - 1; // For array indexing
1445 // Elastic cross section
1446 dy = csel[ipart3][je2] - csel[ipart3][je1];
1447 rc = dy/dx;
1448 b = csel[ipart3][je1] - rc*plab[je1];
1449 xsecel = rc*p + b;
1450 // Inelastic cross section
1451 dy = csin[ipart3][je2] - csin[ipart3][je1];
1452 rc = dy/dx;
1453 b = csin[ipart3][je1] - rc*plab[je1];
1454 xsecin = rc*p + b;
1455 }
1456
1457// A-dependence from parameterization...
1458
1459 if (AA >= 2) {
1460
1461 crel = 1.;
1462 crin = 1.;
1463
1464 G4int i = 2;
1465 if (AA < 50) i = 1;
1466 else if (AA > 100) i = 3;
1467
1468// Protons and neutrons
1469 if (ipart == 14 || ipart == 16) {
1470 dy = cspnel[i - 1][je2] - cspnel[i - 1][je1];
1471 rc = dy/dx;
1472 b = cspnel[i - 1][je1] - rc*plab[je1];
1473 xsecel = rc*p + b;
1474 dy = cspnin[i - 1][je2] - cspnin[i - 1][je1];
1475 rc = dy/dx;
1476 b = cspnin[i - 1][je1] - rc*plab[je1];
1477 xsecin = rc*p + b;
1478 // The following is a first-order correction to Gheisha (GHESIG)
1479 // behaviour, where for the lighter elements the inelastic cross
1480 // section is not realistic for particles in the first momentum
1481 // bin. In the first momentum bin, it better to interpolate
1482 // in K.E. Subject to further improvements.
1483 if (correctInelasticNearZero && je1 == 0 && i <= 3) {
1484 G4double m0 = aParticle->GetMass()/GeV;
1485 G4double T = std::sqrt(m0*m0 + p*p) - m0;
1486 dx = std::sqrt(m0*m0 + plab[1]*plab[1]) - m0;
1487 rc = dy/dx;
1488 xsecin = rc*T + b;
1489 }
1490
1491 if (sigel >= 0.001)
1492 crel = xsecel/(0.36*sigel*G4Pow::GetInstance()->powA(G4double(csa[i]), 1.17));
1493 sigtot = sigel + sigin;
1494 if (sigtot >= 0.001)
1495 crin = xsecin/(sigtot*G4Pow::GetInstance()->powA(G4double(csa[i]), alph));
1496 }
1497
1498 else if (ipart < 15) {
1499// Calculate correction factors (crel, crin) from values
1500// on Al, Cu, Pb. Note that data is only available for pions and protons.
1501 G4double wgch = 0.5;
1502 if (AA < 20) { wgch = 0.5 + 0.5*G4Exp(-G4double(AA - 1)); }
1503 sigel = wgch*sigel + (1. - wgch)*xsecel;
1504 sigin = wgch*sigin + (1. - wgch)*xsecin;
1505
1506// This section not for kaons
1507 if (ipart < 10) {
1508 dy = cspiel[i - 1][je2] - cspiel[i - 1][je1];
1509 rc = dy/dx;
1510 b = cspiel[i - 1][je1] - rc*plab[je1];
1511 xspiel = rc*p + b;
1512 dy = cspiin[i - 1][je2] - cspiin[i - 1][je1];
1513 rc = dy/dx;
1514 b = cspiin[i - 1][je1] - rc*plab[je1];
1515 xspiin = rc*p + b;
1516 if (verboseLevel > 1) {
1517 G4cout << "xspiel " << xspiel << G4endl;
1518 G4cout << "xspiin " << xspiin << G4endl;
1519 }
1520 if (sigel >= 0.001)
1521 crel = xspiel/(0.36*sigel*G4Pow::GetInstance()->powA(G4double(csa[i]),1.17));
1522 sigtot = sigel + sigin;
1523 if (sigtot >= 0.001)
1524 crin = xspiin/(sigtot*G4Pow::GetInstance()->powA(G4double(csa[i]), alph));
1525 }
1526 }
1527
1528// Apply correction factors
1529 sigin = crin*(sigin + sigel)*g4pow->powZ(AA, alph);
1530 sigel = crel*0.36*sigel*g4pow->powZ(AA, 1.17);
1531 sigel = sigel*partel[ipart1];
1532 sigin = sigin*partin[ipart1];
1533 }
1534
1535// Correction factor for high (p > 100 GeV/c) energies:
1536 G4double corh = 1.;

1538

1540 sigin = corh*sigin;
1541 // Convert cross section from mb to default units
1542 sigelastic = sigel*millibarn;
1543 siginelastic = sigin*millibarn;
1544
1545 return;
