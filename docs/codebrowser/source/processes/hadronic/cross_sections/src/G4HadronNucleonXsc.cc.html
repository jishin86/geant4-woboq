115G4bool
116G4HadronNucleonXsc::IsIsoApplicable(const G4DynamicParticle* aDP, G4int Z)
117{
118 G4bool applicable = false;
119 G4double kineticEnergy = aDP->GetKineticEnergy();
120
121 const G4ParticleDefinition* theParticle = aDP->GetDefinition();
122
123 if ( ( kineticEnergy >= fLowerLimit &&
124 Z > 1 && // >= He
125 ( theParticle == theAProton ||
126 theParticle == theGamma ||
127 theParticle == theKPlus ||
128 theParticle == theKMinus ||
129 theParticle == theSMinus) ) ||
130
131 ( kineticEnergy >= 0.1*fLowerLimit &&
132 Z > 1 && // >= He
133 ( theParticle == theProton ||
134 theParticle == theNeutron ||
135 theParticle == thePiPlus ||
136 theParticle == thePiMinus ) ) ) applicable = true;
137
138 return applicable;
139}

508//////////////////////////////////////////////////////////////////////////////
509//
510// Returns hadron-nucleon cross-section based on N. Starkov parametrisation of
511// data from mainly http://wwwppds.ihep.su:8001/c5-6A.html database
512
513G4double G4HadronNucleonXsc::HadronNucleonXscNS(
514 const G4ParticleDefinition* theParticle,
515 const G4ParticleDefinition* nucleon, G4double ekin)
516{
517 /*
518 G4cout<< "HadronNucleonXscNS: Ekin(GeV)= " << ekin/GeV << " "
519 << theParticle->GetParticleName() << " + "
520 << nucleon->GetParticleName() << G4endl;
521 */
522 if(theParticle == theAProton || theParticle == theANeutron) {
523 return HadronNucleonXscPDG(theParticle, nucleon, ekin);
524 }
525
526 G4double pM = theParticle->GetPDGMass();
527
528 ///////////////////////////// strange hyperons
529 if(pM > CLHEP::GeV) {
530 if( theParticle == theL || theParticle == theSPlus ||
531 theParticle == theSMinus || theParticle == theS0) {
532 fTotalXsc = HadronNucleonXscNS( theProton, nucleon, ekin);
533 fTotalXsc *= fHypTotXscCof;
534 fInelasticXsc *= fHypTotXscCof;
535 fElasticXsc *= fHypTotXscCof;
536 return fTotalXsc;
537 } else if( theParticle == theAntiL || theParticle == theASPlus ||
538 theParticle == theASMinus || theParticle == theAS0) {
539 fTotalXsc = HadronNucleonXscNS( theProton, nucleon, ekin);
540 fTotalXsc *= fHypTotXscCof;
541 fInelasticXsc *= fHypTotXscCof;
542 fElasticXsc *= fHypTotXscCof;
543 return fTotalXsc;
544 } else if( theParticle == theXiMinus || theParticle == theXi0) {
545 fTotalXsc = HadronNucleonXscNS( theProton, nucleon, ekin);
546 G4double cof2 = fHypTotXscCof*fHypTotXscCof;
547 fTotalXsc *= cof2;
548 fInelasticXsc *= cof2;
549 fElasticXsc *= cof2;
550 return fTotalXsc;
551 } else if( theParticle == theAXiMinus || theParticle == theAXi0) {
552 fTotalXsc = HadronNucleonXscNS( theProton, nucleon, ekin);
553 G4double cof2 = fHypTotXscCof*fHypTotXscCof;
554 fTotalXsc *= cof2;
555 fInelasticXsc *= cof2;
556 fElasticXsc *= cof2;
557 return fTotalXsc;
558 } else if( theParticle == theOmega ) {
559 fTotalXsc = HadronNucleonXscNS( theProton, nucleon, ekin);
560 G4double cof3 = fHypTotXscCof*fHypTotXscCof*fHypTotXscCof;
561 fTotalXsc *= cof3;
562 fInelasticXsc *= cof3;
563 fElasticXsc *= cof3;
564 return fTotalXsc;
565 } else if( theParticle == theAOmega ) {
566 fTotalXsc = HadronNucleonXscNS( theProton, nucleon, ekin);
567 G4double cof3 = fHypTotXscCof*fHypTotXscCof*fHypTotXscCof;
568 fTotalXsc *= cof3;
569 fInelasticXsc *= cof3;
570 fElasticXsc *= cof3;
571 return fTotalXsc;
572 } else {
573 return HadronNucleonXscPDG( theProton, nucleon, ekin);
574 }
575 }
576
577 G4double tM = nucleon->GetPDGMass();
578 G4double pE = ekin + pM;
579 G4double pLab = std::sqrt(ekin*(ekin + 2*pM));
580
581 G4double sMand = CalcMandelstamS(ekin, pM, tM)*invGeV2;
582
583 pLab *= invGeV;
584 pE *= invGeV;
585
586 if(pLab >= 10.) {
587 fTotalXsc = HadronNucleonXscPDG(theParticle, nucleon, ekin)/CLHEP::millibarn;
588 } else { fTotalXsc = 0.0; }
589 //G4cout << "Stot(mb)= " << fTotalXsc << " pLab= " << pLab
590 // << " Smand= " << sMand <<G4endl;
591 G4double logP = G4Log(pLab);
592
593 G4bool proton = (nucleon == theProton);
594 G4bool neutron = (nucleon == theNeutron);
595
596 if( theParticle == theNeutron)
597 {
598 if( pLab >= 373.)
599 {
600 fElasticXsc = 6.5 + 0.308*G4Exp(G4Log(G4Log(sMand/400.)*1.65))
601 + 9.19*G4Exp(-G4Log(sMand)*0.458);
602 }
603 else if( pLab >= 100.)
604 {
605 fElasticXsc = 5.53 + 0.308*G4Exp(G4Log(G4Log(sMand/28.9))*1.1)
606 + 9.19*G4Exp(-G4Log(sMand)*0.458);
607 }
608 else if( pLab >= 10.)
609 {
610 fElasticXsc = 6 + 20/( (logP-0.182)*(logP-0.182) + 1.0 );
611 }
612 else // pLab < 10 GeV/c
613 {
614 if( neutron ) // nn to be pp
615 {
616 G4double x = G4Log(pLab/0.73);
617 if( pLab < 0.4 )
618 {
619 fTotalXsc = 23 + 50*std::sqrt(g4calc->powN(-x, 7));
620 fElasticXsc = fTotalXsc;
621 }
622 else if( pLab < 0.73 )
623 {
624 fTotalXsc = 23 + 50*std::sqrt(g4calc->powN(-x, 7));
625 fElasticXsc = fTotalXsc;
626 }
627 else if( pLab < 1.05 )
628 {
629 fTotalXsc = 23 + 40*x*x;
630 fElasticXsc = 23 + 20*x*x;
631 }
632 else // 1.05 - 10 GeV/c
633 {
634 fTotalXsc = 39.0+75*(pLab - 1.2)/(g4calc->powN(pLab,3) + 0.15);
635 fElasticXsc = 6 + 20/( (logP-0.182)*(logP-0.182) + 1.0 );
636 }
637 }
638 if( proton ) // pn to be np
639 {
640 if( pLab < 0.02 )
641 {
642 fTotalXsc = 4100+30*G4Exp(G4Log(G4Log(1.3/pLab))*3.6);
643 fElasticXsc = fTotalXsc;
644 }
645 else if( pLab < 0.8 )
646 {
647 fTotalXsc = 33+30*g4calc->powN(G4Log(pLab/1.3),4);
648 fElasticXsc = fTotalXsc;
649 }
650 else if( pLab < 1.4 )
651 {
652 fTotalXsc = 33+30*g4calc->powN(G4Log(pLab/0.95),2);
653 G4double x = G4Log(0.511/pLab);
654 fElasticXsc = 6 + 52/( x*x + 1.6 );
655 }
656 else // 1.4 < pLab < 10. )
657 {
658 fTotalXsc = 33.3 + 20.8*(pLab*pLab - 1.35)
659 /(std::sqrt(g4calc->powN(pLab,5)) + 0.95);
660 fElasticXsc = 6 + 20/( (logP-0.182)*(logP-0.182) + 1.0 );
661 }
662 }
663 }
664 }
665 ////// proton //////////////////////////////////////////////
666 else if(theParticle == theProton)
667 {
668 if( pLab >= 373.) // pdg due to TOTEM data
669 {
670 fElasticXsc = 6.5 + 0.308*G4Exp(G4Log(G4Log(sMand/400.))*1.65)
671 + 9.19*G4Exp(-G4Log(sMand)*0.458);
672 }
673 else if( pLab >= 100.)
674 {
675 fElasticXsc = 5.53 + 0.308*G4Exp(G4Log(G4Log(sMand/28.9))*1.1)
676 + 9.19*G4Exp(-G4Log(sMand)*0.458);
677 }
678 else if( pLab >= 10.)
679 {
680 fElasticXsc = 6. + 20./( (logP-0.182)*(logP-0.182) + 1.0 );
681 }
682 else
683 {
684 // pp
685 if( proton )
686 {
687 if( pLab < 0.73 )
688 {
689 fTotalXsc = 23 + 50*std::sqrt(g4calc->powN(G4Log(0.73/pLab),7));
690 fElasticXsc = fTotalXsc;
691 }
692 else if( pLab < 1.05 )
693 {
694 G4double x = G4Log(pLab/0.73);
695 fTotalXsc = 23 + 40*x*x;
696 fElasticXsc = 23 + 20*x*x;
697 }
698 else // 1.05 - 10 GeV/c
699 {
700 fTotalXsc = 39.0+75*(pLab - 1.2)/(g4calc->powN(pLab,3) + 0.15);
701 fElasticXsc = 6. + 20./( (logP-0.182)*(logP-0.182) + 1.0 );
702 }
703 }
704 else if( neutron ) // pn to be np
705 {

1457 const G4ParticleDefinition* nucleon,
1458 G4double ekin)
1459{
1460 G4double tR = 0.895*CLHEP::fermi;
1461 G4double pR = 0.5*CLHEP::fermi;
1462
1463 if ( theParticle == theProton ) pR = 0.895*fermi;
1464 else if( theParticle == thePiPlus ) pR = 0.663*fermi;
1465 else if( theParticle == theKPlus ) pR = 0.340*fermi;
1466
1467 G4double pZ = theParticle->GetPDGCharge();
1468 G4double tZ = nucleon->GetPDGCharge();
1469
1470 G4double pM = theParticle->GetPDGMass();
1471 G4double tM = nucleon->GetPDGMass();
1472
1473 G4double pElab = ekin + pM;
1474
1475 G4double totEcm = std::sqrt(pM*pM + tM*tM + 2.*pElab*tM);
1476
1477 G4double totTcm = totEcm - pM -tM;
1478
1479 G4double bC = fine_structure_const*hbarc*pZ*tZ/(2.*(pR + tR));
1480
1481 //G4cout<<"##CB: ekin = "<<ekin<<"; pElab= " << pElab
1482 // <<"; bC = "<<bC<<"; Tcm = "<< totTcm <<G4endl;
1483
1484 G4double ratio = (totTcm > bC) ? 1. - bC/totTcm : 0.0;
1485
1486 // G4cout <<"ratio = "<<ratio<<G4endl;
