1558 G4int Z, G4int A,
1559 const G4Isotope*,
1560 const G4Element*,
1561 const G4Material* mat)
1562{
1563 // if not deuterium, go back to old style
1564 if (!(Z == 1 && A == 2)) return GetElementCrossSection(aPart, Z, mat);
1565
1566 // Otherwise, follow a similar routine
1567 const G4double Energy = aPart->GetKineticEnergy()/MeV;
1568 if (Energy < THmin) return 0.;
1569 G4double sigma = 0.;
1570
1571 // init the XS table if need be
1572 if (deuteron_GDR == NULL) {

1584
