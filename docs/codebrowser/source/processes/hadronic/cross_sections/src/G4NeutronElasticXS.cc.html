114G4double
115G4NeutronElasticXS::GetElementCrossSection(const G4DynamicParticle* aParticle,
116 G4int ZZ, const G4Material*)
117{
118 G4double xs = 0.0;
119 G4double ekin = aParticle->GetKineticEnergy();
120
121 G4int Z = (ZZ >= MAXZEL) ? MAXZEL - 1 : ZZ;
122
123 G4PhysicsVector* pv = data[Z];
124 // G4cout << "G4NeutronElasticXS::GetCrossSection e= " << ekin
125 // << " Z= " << Z << G4endl;
126
127 // element was not initialised
128 if(!pv) { return xs; }
129
130 if(ekin <= pv->Energy(0)) { return (*pv)[0]; }
131
132 if(ekin <= pv->GetMaxEnergy()) {
133 xs = pv->Value(ekin);
134 } else if(1 == Z) {
135 fNucleon->GetHadronNucleonXscNS(aParticle, proton);
136 xs = coeff[1]*fNucleon->GetElasticHadronNucleonXsc();
137 } else {
138 G4int Amean =
139 G4lrint(G4NistManager::Instance()->GetAtomicMassAmu(Z));
140 ggXsection->GetIsoCrossSection(aParticle, Z, Amean);
141 xs = coeff[Z]*ggXsection->GetElasticGlauberGribovXsc();
142 }
143
144 if(verboseLevel > 0){
145 G4cout << "Z= " << Z << " Ekin(MeV)= " << ekin/CLHEP::MeV
146 << ", nElmXSel(bn)= " << xs/CLHEP::barn
147 << G4endl;
148 }
149 return xs;
150}

152void
153G4NeutronElasticXS::BuildPhysicsTable(const G4ParticleDefinition& p)
154{
155 if(verboseLevel > 0){
156 G4cout << "G4NeutronElasticXS::BuildPhysicsTable for "
157 << p.GetParticleName() << G4endl;
158 }
159 if(p.GetParticleName() != "neutron") {
160 G4ExceptionDescription ed;
161 ed << p.GetParticleName() << " is a wrong particle type -"
162 << " only neutron is allowed";
163 G4Exception("G4NeutronElasticXS::BuildPhysicsTable(..)","had012",
164 FatalException, ed, "");
165 return;
166 }
167 if(!ggXsection) { ggXsection = new G4ComponentGGHadronNucleusXsc(); }
168 if(!fNucleon) { fNucleon = new G4HadronNucleonXsc(); }
169
170 if(!data[1]) {
171#ifdef G4MULTITHREADED
172 G4MUTEXLOCK(&neutronElasticXSMutex);
173 if(!data[1]) {
174#endif
175 isMaster = true;
176#ifdef G4MULTITHREADED
177 }
178 G4MUTEXUNLOCK(&neutronElasticXSMutex);
179#endif
180 }
181
182 // it is possible re-initialisation for the second run
183 if(isMaster) {
184
185 // check environment variable
186 // Build the complete string identifying the file with the data set
187 char* path = getenv("G4PARTICLEXSDATA");
188
189 G4DynamicParticle* dynParticle =
190 new G4DynamicParticle(G4Neutron::Neutron(),G4ThreeVector(1,0,0),1);
191
192 // Access to elements
193 const G4ElementTable* theElmTable = G4Element::GetElementTable();
194 size_t numOfElm = G4Element::GetNumberOfElements();
195 for(size_t i=0; i<numOfElm; ++i) {
196 G4int Z = ((*theElmTable)[i])->GetZasInt();
197 if(Z >= MAXZEL) { Z = MAXZEL-1; }
198 //G4cout << "Z= " << Z << G4endl;
199 // Initialisation
200 if(!data[Z]) { Initialise(Z, dynParticle, path); }
201 }
202 delete dynParticle;
203 }
204}

206void
207G4NeutronElasticXS::Initialise(G4int Z, G4DynamicParticle* dp,
208 const char* p)
209{
210 if(data[Z]) { return; }
211 const char* path = p;
212 if(!p) {
213 // check environment variable
214 // Build the complete string identifying the file with the data set
215 path = getenv("G4PARTICLEXSDATA");
216 if (!path) {
217 G4Exception("G4NeutronElasticXS::Initialise(..)","had013",
218 FatalException,
219 "Environment variable G4PARTICLEXSDATA is not defined");
220 return;
221 }
222 }
223
224 // upload data from file
225 data[Z] = new G4PhysicsLogVector();
226
227 std::ostringstream ost;
228 ost << path << "/neutron/el" << Z ;
229 std::ifstream filein(ost.str().c_str());
230 if (!(filein)) {
231 G4ExceptionDescription ed;
232 ed << "Data file <" << ost.str().c_str()
233 << "> is not opened!";
234 G4Exception("G4NeutronElasticXS::Initialise(..)","had014",
235 FatalException, ed, "Check G4PARTICLEXSDATA");
236 return;
237 }else{
238 if(verboseLevel > 1) {
239 G4cout << "file " << ost.str()
240 << " is opened by G4NeutronElasticXS" << G4endl;
241 }
242
243 // retrieve data from DB
244 if(!data[Z]->Retrieve(filein, true)) {
245 G4ExceptionDescription ed;
246 ed << "Data file <" << ost.str().c_str()
247 << "> is not retrieved!";
248 G4Exception("G4NeutronElasticXS::Initialise(..)","had015",
249 FatalException, ed, "Check G4PARTICLEXSDATA");
250 return;
251 }
252
253 // smooth transition
254 G4double sig1 = (*(data[Z]))[data[Z]->GetVectorLength()-1];
255 dp->SetKineticEnergy(data[Z]->GetMaxEnergy());
256 G4double sig2 = 0.0;
257 if(1 == Z) {
258 fNucleon->GetHadronNucleonXscNS(dp, proton);
259 sig2 = fNucleon->GetElasticHadronNucleonXsc();
260 } else {
261 G4int Amean =
262 G4lrint(G4NistManager::Instance()->GetAtomicMassAmu(Z));
263 ggXsection->GetIsoCrossSection(dp, Z, Amean);
264 sig2 = ggXsection->GetElasticGlauberGribovXsc();
265 }
266 if(sig2 > 0.) { coeff[Z] = sig1/sig2; }
267 }
