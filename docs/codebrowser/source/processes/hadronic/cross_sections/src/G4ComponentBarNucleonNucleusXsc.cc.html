226void G4ComponentBarNucleonNucleusXsc::ComputeCrossSections(
227 const G4ParticleDefinition* aParticle, G4double kineticEnergy, G4int ZZ)
228{
229 G4int Z = std::min(ZZ, 92);
230 size_t it = 0;
231 size_t itmax = theZ.size() - 1;
232 for(; it <= itmax; ++it) { if(Z <= theZ[it]) { break; } }
233 if( it > itmax ) { it = itmax; }
234 G4int Z1, Z2;
235 G4double x1, x2, xt1, xt2;
236
237 std::vector<G4PiData*>* theData = (aParticle == theNeutron) ? &thePimData : &thePipData;
238
239 if( theZ[it] == Z )
240 {
241 fInelasticXsc = (*theData)[it]->ReactionXSection(kineticEnergy);
242 fTotalXsc = (*theData)[it]->TotalXSection(kineticEnergy);
243 }
244 else
245 {
246 if(0 == it) { it = 1; }
247 x1 = (*theData)[it-1]->ReactionXSection(kineticEnergy);
248 xt1 = (*theData)[it-1]->TotalXSection(kineticEnergy);
249 Z1 = theZ[it-1];
250 x2 = (*theData)[it]->ReactionXSection(kineticEnergy);
251 xt2 = (*theData)[it]->TotalXSection(kineticEnergy);
252 Z2 = theZ[it];
253
254 fInelasticXsc = Interpolate(Z1, Z2, Z, x1, x2);
255 fTotalXsc = Interpolate(Z1, Z2, Z, xt1, xt2);
256 }
257
258 fElasticXsc = std::max(fTotalXsc - fInelasticXsc, 0.0);
259}

261/////////////////////////////////////////////////////////////////////////////
262//
263
264G4double G4ComponentBarNucleonNucleusXsc::
265Interpolate(G4int Z1, G4int Z2, G4int Z, G4double x1, G4double x2)
266{
267 // for tabulated data, cross section scales with A^(2/3)
268 G4double r1 = x1* A75[Z] / A75[Z1];
269 G4double r2 = x2* A75[Z] / A75[Z2];
270 G4double alp1 = (theA[Z] - theA[Z1]);
271 G4double alp2 = (theA[Z2] - theA[Z]);
272 G4double result = (r1*alp2 + r2*alp1)/(alp1 + alp2);
273 // G4cout << "x1/2, z1/2 z" <<x1<<" "<<x2<<" "<<Z1<<" "<<Z2<<" "<<Z<<G4endl;
274 // G4cout << "res1/2 " << r1 <<" " << r2 <<" " << result<< G4endl;
275 return result;
276}

287}
