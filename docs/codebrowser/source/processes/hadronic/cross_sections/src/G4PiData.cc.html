34///////////////////////////////////////////////////////////////////////
35
36G4PiData::G4PiData(const G4double * aT, const G4double * aIn,
37 const G4double * anE, G4int nP)
38{
39 G4int i=0;
40
41 for( i = 0; i < nP; i++ )
42 {
43 std::pair<G4double, G4double> x;
44 x.first=aT[i]*millibarn;
45 x.second=aIn[i]*millibarn;
46 std::pair<G4double, std::pair<G4double, G4double > > aP;
47 aP.first=anE[i]*GeV;
48 aP.second=x;
49 push_back(aP);
50 }
51}

55G4bool G4PiData::AppliesTo(G4double kineticEnergy)
56{
57 return (kineticEnergy<=back().first);
58}

60//////////////////////////////////////////////////////////////////////////
61
62G4double G4PiData::ReactionXSection(G4double kineticEnergy)
63{
64 G4double result = 0;
65 G4PiData::iterator it=begin();
66 while(it!=end()&&kineticEnergy>(*it).first) {it++;} /* Loop checking, 08.01.2016, W. Pokorski */
67 if(it==end())
68 {
69 throw G4HadronicException(__FILE__, __LINE__,
70 "G4PiData::ReactionXSection: used outside validity range");
71 }
72 if(it==begin()) it++;
73 G4double x1,x2,e1,e2;
74 e1=(*(it-1)).first;
75 x1=(*(it-1)).second.second;
76 e2=(*(it)).first;
77 x2=(*(it)).second.second;
78 result = std::max(0., x1 + (kineticEnergy-e1)*(x2-x1)/(e2-e1));
79 return result;
80}

82////////////////////////////////////////////////////////////////////////////
83
84G4double G4PiData::ElasticXSection(G4double kineticEnergy)
85{
86 G4double result = 0;
87 G4PiData::iterator it=begin();
88 while(it!=end()&&kineticEnergy>(*it).first) {it++;} /* Loop checking, 08.01.2016, W. Pokorski */
89 if(it==end())
90 {
91 throw G4HadronicException(__FILE__, __LINE__,
92 "G4PiData::ElasticXSection: used outside validity range");
93 }
94 if(it==begin()) it++;
95 G4double x1,x2,e1,e2;
96 e1=(*(it-1)).first;
97 x1=(*(it-1)).second.first - (*(it-1)).second.second;
98 e2=(*(it)).first;
99 x2=(*(it)).second.first - (*(it)).second.second;
100 result = std::max(0., x1 + (kineticEnergy-e1)*(x2-x1)/(e2-e1));
101 return result;
102}
