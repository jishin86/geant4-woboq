57G4BGGNucleonElasticXS::G4BGGNucleonElasticXS(const G4ParticleDefinition* p)
58 : G4VCrossSectionDataSet("Barashenkov-Glauber")
59{
60 verboseLevel = 0;
61 fGlauberEnergy = 91.*GeV;
62 fPDGEnergy = 5*GeV;
63 fLowEnergy = 14.*MeV;
64 fSAIDLowEnergyLimit = 1*MeV;
65 fSAIDHighEnergyLimit = 1.3*GeV;
66 fLowestXSection = millibarn;
67 for (G4int i = 0; i < 93; ++i) {
68 theGlauberFac[i] = 0.0;
69 theCoulombFac[i] = 0.0;
70 theA[i] = 1;
71 }
72 fNucleon = nullptr;
73 fGlauber = nullptr;
74 fHadron = nullptr;
75 fSAID = nullptr;
76 particle = p;
77 theProton= G4Proton::Proton();
78 isProton = (theProton == p) ? true : false;
79 isInitialized = false;
80 SetForAllAtomsAndEnergies(true);
81}

111G4double
112G4BGGNucleonElasticXS::GetElementCrossSection(const G4DynamicParticle* dp,
113 G4int ZZ, const G4Material*)
114{
115 // this method should be called only for Z > 1
116
117 G4double cross = 0.0;
118 G4int Z = std::min(ZZ, 92);
119 G4double ekin = dp->GetKineticEnergy();
120 if(1 == Z) {
121 cross = 1.0115*GetIsoCrossSection(dp,1,1);
122 } else {
123 if(ekin <= fLowEnergy) {
124 cross = theCoulombFac[Z]*CoulombFactor(ekin, Z);
125 } else if(ekin > fGlauberEnergy) {
126 cross = theGlauberFac[Z]*fGlauber->GetElasticGlauberGribov(dp, Z, theA[Z]);
127 } else {
128 cross = fNucleon->GetElasticCrossSection(dp, Z);
129 }
130 }
131 if(verboseLevel > 1) {
132 G4cout << "G4BGGNucleonElasticXS::GetElementCrossSection for "
133 << dp->GetDefinition()->GetParticleName()
134 << " Ekin(GeV)= " << dp->GetKineticEnergy()/CLHEP::GeV
135 << " in nucleus Z= " << Z << " A= " << theA[Z]
136 << " XS(b)= " << cross/barn
137 << G4endl;
138 }
139 return cross;
140}

144G4double
145G4BGGNucleonElasticXS::GetIsoCrossSection(const G4DynamicParticle* dp,
146 G4int Z, G4int A,
147 const G4Isotope*,
148 const G4Element*,
149 const G4Material*)
150{
151 // this method should be called only for Z = 1
152
153 G4double cross = 0.0;
154 if(1 == Z) {
155 G4double ekin = std::max(dp->GetKineticEnergy(), fLowEnergy);
156 fHadron->HadronNucleonXscNS(dp->GetDefinition(), theProton, ekin);
157 cross = fHadron->GetElasticHadronNucleonXsc();
158 }
159 cross *= A;
160 if(verboseLevel > 1) {
161 G4cout << "G4BGGNucleonElasticXS::GetIsoCrossSection for "
162 << dp->GetDefinition()->GetParticleName()
163 << " Ekin(GeV)= " << dp->GetKineticEnergy()/CLHEP::GeV
164 << " in nucleus Z= " << Z << " A= " << A
165 << " XS(b)= " << cross/barn
166 << G4endl;
167 }
168 return cross;
169}

173void G4BGGNucleonElasticXS::BuildPhysicsTable(const G4ParticleDefinition& p)
174{
175 if(&p == theProton || &p == G4Neutron::Neutron()) {
176 particle = &p;
177
178 } else {
179 G4cout << "### G4BGGNucleonElasticXS WARNING: is not applicable to "
180 << p.GetParticleName()
181 << G4endl;
182 throw G4HadronicException(__FILE__, __LINE__,
183 "G4BGGNucleonElasticXS::BuildPhysicsTable is used for wrong particle");
184 return;
185 }
186
187 if(isInitialized) { return; }
188 isInitialized = true;
189
190 fNucleon = (G4NucleonNuclearCrossSection*)G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4NucleonNuclearCrossSection::Default_Name());
191 fGlauber = new G4ComponentGGHadronNucleusXsc();
192 fHadron = new G4HadronNucleonXsc();
193
194 fNucleon->BuildPhysicsTable(*particle);
195 fGlauber->BuildPhysicsTable(*particle);
196
197 G4ThreeVector mom(0.0,0.0,1.0);
198 G4DynamicParticle dp(particle, mom, fGlauberEnergy);
199
200 G4NistManager* nist = G4NistManager::Instance();
201
202 G4double csup, csdn;
203 G4int A;
204
205 if(verboseLevel > 0) {
206 G4cout << "### G4BGGNucleonElasticXS::Initialise for "
207 << particle->GetParticleName() << G4endl;
208 }
209
210 for(G4int iz=2; iz<93; iz++) {
211
212 A = G4lrint(nist->GetAtomicMassAmu(iz));
213 theA[iz] = A;
214
215 csup = fGlauber->GetElasticGlauberGribov(&dp, iz, A);
216 csdn = fNucleon->GetElasticCrossSection(&dp, iz);
217
218 theGlauberFac[iz] = csdn/csup;
219 if(verboseLevel > 0) {
220 G4cout << "Z= " << iz << " A= " << A
221 << " factor= " << theGlauberFac[iz] << G4endl;
222 }
223 }
224
225 theCoulombFac[0] = theCoulombFac[1] = 1.0;
226 dp.SetKineticEnergy(fLowEnergy);
227 for(G4int iz=2; iz<93; iz++) {
228 theCoulombFac[iz] =
229 fNucleon->GetElasticCrossSection(&dp, iz)/CoulombFactor(fLowEnergy, iz);
230 if(verboseLevel > 0) {
231 G4cout << "Z= " << iz << " A= " << theA[iz]
232 << " factor= " << theCoulombFac[iz] << G4endl;
233 }
234 }
235}

239G4double G4BGGNucleonElasticXS::CoulombFactor(G4double kinEnergy, G4int Z)
240{
241 G4double res= 1.0;
242
243 // from G4ProtonInelasticCrossSection
244 if(isProton) {
245
246 if (Z <= 1) { return kinEnergy*kinEnergy; }
247
248 static const G4double llog10 = G4Log(10.);
249 G4double elog = G4Log(kinEnergy/GeV)/llog10;
250 G4double aa = theA[Z];
251
252 G4double ff1 = 5.6 - 0.016*aa; // slope of the drop at medium energies.
253 G4double ff2 = 1.37 + 1.37/aa; // start of the slope.
254 G4double ff3 = 0.8 + 18./aa - 0.002*aa; // stephight
255 res = 1.0 + ff3*(1.0 - (1.0/(1+G4Exp(-ff1*(elog + ff2)))));
256
257 ff1 = 8. - 8./aa - 0.008*aa; // slope of the rise
258 ff2 = 2.34 - 5.4/aa - 0.0028*aa; // start of the rise
259 res /= (1.0 + G4Exp(-ff1*(elog + ff2)));
260
261 }
262 return res;
263}
