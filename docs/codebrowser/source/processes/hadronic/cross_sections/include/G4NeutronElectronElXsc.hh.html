96////////////////////////////////////////////////////////////////////
97//
98// return Wentzel atom screening correction for neutron-electron scattering
99
100inline G4double G4NeutronElectronElXsc::CalculateAm( G4double momentum)
101{
102 G4double k = momentum/CLHEP::hbarc;
103 G4double ch = 1.13;
104 G4double zn = 1.77*k*CLHEP::Bohr_radius;
105 G4double zn2 = zn*zn;
106 fAm = ch/zn2;
107
108 return fAm;
109}

111////////////////////////////////////////////////////
112//
113// Slow electron (Tkin << me_c2) in the neutron rest frame
114
115inline G4double G4NeutronElectronElXsc::
116GetElementNonRelXsc(const G4DynamicParticle* aPart, G4int ZZ,
117 const G4Material*)
118{
119 G4double result(0.), te(0.), momentum(0.);
120
121 te = aPart->GetKineticEnergy()*fme/fM;
122 momentum = sqrt( te*(te + 2.*fme) );
123 fAm = CalculateAm(momentum);
124
125 result = 1. + log(1. +1./fAm);
126 result *= fCofXsc; //*energy;
127 result *= ZZ; // incoherent sum over all element electrons
128
129 return result;
130}
