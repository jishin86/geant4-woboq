148inline G4double
149G4VComponentCrossSection::GetTotalElementCrossSection(
150 const G4ParticleDefinition* p,
151 G4double kinEnergy,
152 const G4Element* elm)
153{

155 elm->GetZasInt(),elm->GetN());
156}

169G4VComponentCrossSection::GetElasticElementCrossSection(
