26// Calculation of the total, elastic and inelastic cross-sections
27// based on parametrisations of (proton, pion, kaon, photon) nucleon
28// cross-sections and the hadron-nucleous cross-section model in
29// the framework of Glauber-Gribov approach
30//
31// 14.03.07 V. Grichine - first implementation
32// 04.11.11 V. Grichine - update for kaon-(p,n) xsc, vector spline
33// 21.02.12 V. Grichine - update for pion-(p,n) xsc, NS fit++, vector spline
34// 30.07.18 V. Ivanchenko - general clean-up
35// 30.09.18 V. Grichine hyperon-nucleon xsc first implementation
36
37

39#define G4HadronNucleonXsc_h

51
52 G4HadronNucleonXsc ();
53 ~G4HadronNucleonXsc ();

55 // Xsc parametrisations return total x-section
56 G4double HadronNucleonXscPDG2005(const G4ParticleDefinition* theParticle,
57 const G4ParticleDefinition* nucleon, G4double ekin);
58
59 G4double HadronNucleonXscPDG(const G4ParticleDefinition* theParticle,
60 const G4ParticleDefinition* nucleon, G4double ekin);
61
62 G4double HadronNucleonXscNS(const G4ParticleDefinition* theParticle,
63 const G4ParticleDefinition* nucleon, G4double ekin);
64
65 G4double KaonNucleonXscNS(const G4ParticleDefinition* theParticle,
66 const G4ParticleDefinition* nucleon, G4double ekin);
67
68 G4double KaonNucleonXscGG(const G4ParticleDefinition* theParticle,
69 const G4ParticleDefinition* nucleon, G4double ekin);
70
71 G4double KaonNucleonXscVG(const G4ParticleDefinition* theParticle,
72 const G4ParticleDefinition* nucleon, G4double ekin);
73
74 G4double HadronNucleonXscVU(const G4ParticleDefinition* theParticle,
75 const G4ParticleDefinition* nucleon, G4double ekin);

77 G4double HadronNucleonXscEL(const G4ParticleDefinition* theParticle,
78 const G4ParticleDefinition* nucleon, G4double ekin);
79
80 G4double CoulombBarrier(const G4ParticleDefinition* theParticle,
81 const G4ParticleDefinition* nucleon, G4double ekin);

83 // Xsc for G4DynamicParticle projectile
84 inline G4double GetHadronNucleonXscEL(const G4DynamicParticle* dp,
85 const G4ParticleDefinition* p)
86 { return HadronNucleonXscEL(dp->GetDefinition(), p, dp->GetKineticEnergy()); }
87
88 inline G4double GetHadronNucleonXscPDG(const G4DynamicParticle* dp,
89 const G4ParticleDefinition* p)
90 { return HadronNucleonXscPDG(dp->GetDefinition(), p, dp->GetKineticEnergy()); }
91
92 inline G4double GetHadronNucleonXscNS(const G4DynamicParticle* dp,
93 const G4ParticleDefinition* p)
94 { return HadronNucleonXscNS(dp->GetDefinition(), p, dp->GetKineticEnergy()); }
95
96 inline G4double GetKaonNucleonXscGG(const G4DynamicParticle* dp,
97 const G4ParticleDefinition* p)
98 { return KaonNucleonXscGG(dp->GetDefinition(), p, dp->GetKineticEnergy()); }
99
100 inline G4double GetHadronNucleonXscVU(const G4DynamicParticle* dp,
101 const G4ParticleDefinition* p)
102 { return HadronNucleonXscVU(dp->GetDefinition(), p, dp->GetKineticEnergy()); }
103
104 inline G4double GetCoulombBarrier(const G4DynamicParticle* dp,
105 const G4ParticleDefinition* p)
106 { return CoulombBarrier(dp->GetDefinition(), p, dp->GetKineticEnergy()); }
107
108 // Xsc access
109 inline G4double GetTotalHadronNucleonXsc() const { return fTotalXsc; };
110 inline G4double GetElasticHadronNucleonXsc() const { return fElasticXsc; };
111 inline G4double GetInelasticHadronNucleonXsc() const { return fInelasticXsc; };
112
113 // obsolete methods
114 G4bool IsApplicable(const G4DynamicParticle* aDP, const G4Element*);
115
116 G4bool IsIsoApplicable(const G4DynamicParticle* aDP, G4int Z);
117
118 void DumpPhysicsTable(const G4ParticleDefinition&)
119 {G4cout << "G4HadronNucleonXsc: uses parametrisation"<<G4endl;}
120
121 void CrossSectionDescription(std::ostream&) const;
122
123private:
124
125 G4double ComputeKaonNucleonXsc(const G4ParticleDefinition* theParticle,
126 const G4ParticleDefinition* nucleon, G4double ekin);
127
128
129 inline G4double CalcMandelstamS(G4double ekin1, G4double mass1, G4double mass2)
130 { return mass1*mass1 + mass2*mass2 + 2*mass2*(ekin1 + mass1); }
131
132 inline G4double CalculateEcmValue(G4double ekin1, G4double mass1, G4double mass2)
133 { return std::sqrt(CalcMandelstamS(ekin1, mass1, mass2)); };

144 const G4ParticleDefinition* theANeutron;

146 const G4ParticleDefinition* thePiMinus;

149 const G4ParticleDefinition* theKMinus;
150 const G4ParticleDefinition* theK0S;

166 const G4ParticleDefinition* theD;
167 const G4ParticleDefinition* theT;
168 const G4ParticleDefinition* theA;
169 const G4ParticleDefinition* theHe3;
170};
