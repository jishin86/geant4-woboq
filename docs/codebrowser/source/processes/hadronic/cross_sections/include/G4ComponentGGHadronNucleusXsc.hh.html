239/////////////////////////////////////////////////////////////////////
240//
241// return correction at Tkin = 90*GeV GG -> Barashenkov tot xsc, when it
242// is available, else return 1.0
243
244inline G4double G4ComponentGGHadronNucleusXsc::GetParticleBarCorTot(
245 const G4ParticleDefinition* theParticle, G4int ZZ)
246{
247 G4double cor = 1.0;
248 G4int z = std::min(92, std::max(ZZ, 1));
249 if( theParticle == theProton ) cor = fProtonBarCorrectionTot[z];
250 else if( theParticle == theNeutron) cor = fNeutronBarCorrectionTot[z];
251 else if( theParticle == thePiPlus ) cor = fPionPlusBarCorrectionTot[z];
252 else if( theParticle == thePiMinus) cor = fPionMinusBarCorrectionTot[z];
253 return cor;
254}
255

271 return cor;
272}
