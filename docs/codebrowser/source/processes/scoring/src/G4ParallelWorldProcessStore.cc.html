33G4ThreadLocal G4ParallelWorldProcessStore* G4ParallelWorldProcessStore::fInstance=0;
34
35G4ParallelWorldProcessStore* G4ParallelWorldProcessStore::GetInstance()
36{
37 if(!fInstance)
38 { fInstance = new G4ParallelWorldProcessStore(); }
39 return fInstance;
40}
41
42G4ParallelWorldProcessStore* G4ParallelWorldProcessStore::GetInstanceIfExist()
43{
44 return fInstance;
45}

47G4ParallelWorldProcessStore::G4ParallelWorldProcessStore()
48{;}
49
50G4ParallelWorldProcessStore::~G4ParallelWorldProcessStore()
51{
52 Clear();
53 fInstance = 0;
54}
55
56void G4ParallelWorldProcessStore::SetParallelWorld(G4ParallelWorldProcess* proc,
57 G4String parallelWorldName)
58{
59 std::map<G4ParallelWorldProcess*,G4String>::iterator itr;
60 for(itr=fInstance->begin();itr!=fInstance->end();itr++)
61 {
62 if(itr->first==proc)
63 {
64 if(itr->second==parallelWorldName)
65 { // already registered
66 return;
67 }
68 else
69 { // insonsistent !
70 G4ExceptionDescription ED;
71 ED << "G4ParallelWorldProcess (" << proc << ") has the world volume ("
72 << itr->second << "). It is inconsistent with (" << parallelWorldName << ").";
73 G4Exception("G4ParallelWorldProcessStore::SetParallelWorld","ProcScore0101",
74 FatalException,ED);
75 }
76 }
77 }
78 (*fInstance)[proc] = parallelWorldName;
79}
80
81void G4ParallelWorldProcessStore::UpdateWorlds()
82{
83 std::map<G4ParallelWorldProcess*,G4String>::iterator itr;
84 for(itr=fInstance->begin();itr!=fInstance->end();itr++)
85 { (itr->first)->SetParallelWorld(itr->second); }
86}
87
88G4ParallelWorldProcess* G4ParallelWorldProcessStore::GetProcess(G4String parallelWorldName)
89{
90 std::map<G4ParallelWorldProcess*,G4String>::iterator itr;
91 for(itr=fInstance->begin();itr!=fInstance->end();itr++)
92 { if(itr->second==parallelWorldName) return itr->first; }
93 return 0;
94}
95
96void G4ParallelWorldProcessStore::Clear()
97{ fInstance->clear(); }
