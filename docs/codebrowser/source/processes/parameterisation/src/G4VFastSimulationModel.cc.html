41#include "G4VFastSimulationModel.hh"
42#include "G4FastSimulationManager.hh"
43
44// ----------------------
45// -- Simple constructor:
46// ----------------------
47G4VFastSimulationModel::G4VFastSimulationModel(const G4String& aName)
48 : theModelName(aName) {}
49
50// ----------------------------------------------------------------------------------------------
51// -- Constructor with automatic G4FastSimulationManager constructed if needed fo given envelope:
52// ----------------------------------------------------------------------------------------------
53G4VFastSimulationModel::G4VFastSimulationModel(const G4String& aName,
54 G4Envelope* anEnvelope,
55 G4bool IsUnique)
56 : theModelName(aName)
57{
58 // Retrieves the Fast Simulation Manager ou creates one if needed.
59 G4FastSimulationManager* theFastSimulationManager;
60 if ((theFastSimulationManager = anEnvelope->GetFastSimulationManager()) == 0)
61 theFastSimulationManager = new G4FastSimulationManager(anEnvelope,IsUnique);
62 // adds this model to the Fast Simulation Manager.
63 theFastSimulationManager->AddFastSimulationModel(this);
64}
