31
32#include "G4PhononTrackMap.hh"
33#include "G4ThreeVector.hh"
34#include "G4Track.hh"
35#include <map>
36
37namespace { const G4ThreeVector nullVec(0.,0.,0.); } // For convenience below
38
39// Singleton instance, must be created at first request
40
41G4ThreadLocal G4PhononTrackMap* G4PhononTrackMap::theTrackMap = 0;
42
43
44// Pseudo-constructor creates singleton instance
45
46G4PhononTrackMap* G4PhononTrackMap::GetPhononTrackMap() {
47 if (!theTrackMap) theTrackMap = new G4PhononTrackMap;
48 return theTrackMap;
49}
50
51void G4PhononTrackMap::Clear() {
52 theMap.clear(); // Remove all entries from map
53}
54
55
56
57// Check if specified track is already loaded
58
59G4bool G4PhononTrackMap::Find(const G4Track* track) const {
60 return (!track || theMap.find(track) != theMap.end());
61}
62
63
64// Remove specified track from map (used by EndTracking)
65
66void G4PhononTrackMap::RemoveTrack(const G4Track* track) {
67 TrkIDKmap::iterator entry = theMap.find(track);
68 if (entry != theMap.end()) theMap.erase(entry);
69}
70
71
72// Update the wavevector for specified track, add track if non-existent
73
74void G4PhononTrackMap::SetK(const G4Track* track, const G4ThreeVector& K) {
75 if (track) theMap[track] = K;
76}
77
78
79// Access current wavevector for specified track (NULL if doesn't exist)
80
81const G4ThreeVector& G4PhononTrackMap::GetK(const G4Track* track) const {
82 TrkIDKmap::const_iterator entry = theMap.find(track);
83 return (entry != theMap.end()) ? entry->second : nullVec;
84}
