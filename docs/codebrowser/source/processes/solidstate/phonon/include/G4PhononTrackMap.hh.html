26/// \file processes/phonon/include/G4PhononTrackMap.hh
27/// \brief Definition of the G4PhononTrackMap base class
28//
29//
30// 20131111 Move implementation of Clear() to .cc file
31
32#ifndef G4PhononTrackMap_h
33#define G4PhononTrackMap_h 1
34
35#include "G4ThreeVector.hh"
36#include <map>
37
38class G4Track;
39
40class G4PhononTrackMap {
41public:
42 typedef std::map<const G4Track*, G4ThreeVector> TrkIDKmap;
43 static G4ThreadLocal G4PhononTrackMap* theTrackMap;
44
45public:
46 static G4PhononTrackMap* GetPhononTrackMap(); // Synonyms for access
47 static G4PhononTrackMap* GetInstance() { return GetPhononTrackMap(); }
48
49 // Update the wavevector for specified track, add track if not found
50 void SetK(const G4Track* track, const G4ThreeVector& K);
51 void SetK(const G4Track& track, const G4ThreeVector& K) { SetK(&track, K); }
52
53 // Access current wavevector for specified track (NULL if doesn't exist)
54 const G4ThreeVector& GetK(const G4Track* track) const;
55 const G4ThreeVector& GetK(const G4Track& track) const { return GetK(&track); }
56
57 // Check if specified track is already loaded
58 G4bool Find(const G4Track* track) const;
59 G4bool Find(const G4Track& track) const { return Find(&track); }
60
61 // Remove specified track from map (used by EndTracking)
62 void RemoveTrack(const G4Track* track);
63
64 void Clear(); // Remove all entries from map
65
66private:
67 TrkIDKmap theMap; // Associate track ID numbers with vectors
68
69private:
70 G4PhononTrackMap() { Clear(); } // Ensure map is empty
71 ~G4PhononTrackMap() {;}
72};
73#endif /* G4PhononTrackMap_h */
