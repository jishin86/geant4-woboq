66G4bool G4hPairProduction::IsApplicable(const G4ParticleDefinition& p)
67{
68 return (p.GetPDGCharge() != 0.0 && p.GetPDGMass() > 110.0*MeV);
69}

72
73void G4hPairProduction::InitialiseEnergyLossProcess(
74 const G4ParticleDefinition* part,
75 const G4ParticleDefinition*)
76{
77 if (!isInitialised) {
78 isInitialised = true;
79
80 theParticle = part;
81 if (!EmModel()) { SetEmModel(new G4hPairProductionModel(part)); }
82
83 G4double limit = part->GetPDGMass()*8.;
84 if(limit > lowestKinEnergy) { lowestKinEnergy = limit; }
85
86 G4VEmFluctuationModel* fm = nullptr;
87 G4EmParameters* param = G4EmParameters::Instance();
88 EmModel()->SetLowEnergyLimit(param->MinKinEnergy());
89 EmModel()->SetHighEnergyLimit(param->MaxKinEnergy());
90 AddEmModel(1, EmModel(), fm);
91 }
92}
