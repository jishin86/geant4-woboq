70G4mplIonisationModel::G4mplIonisationModel(G4double mCharge, const G4String& nam)
71 : G4VEmModel(nam),G4VEmFluctuationModel(nam),
72 magCharge(mCharge),
73 twoln10(log(100.0)),
74 betalow(0.01),
75 betalim(0.1),
76 beta2lim(betalim*betalim),
77 bg2lim(beta2lim*(1.0 + beta2lim))
78{
79 nmpl = G4int(abs(magCharge) * 2 * fine_structure_const + 0.5);
80 if(nmpl > 6) { nmpl = 6; }
81 else if(nmpl < 1) { nmpl = 1; }
82 pi_hbarc2_over_mc2 = pi * hbarc * hbarc / electron_mass_c2;
83 chargeSquare = magCharge * magCharge;
84 dedxlim = 45.*nmpl*nmpl*GeV*cm2/g;
85 fParticleChange = nullptr;
86 monopole = nullptr;
87 mass = 0.0;
88}

99void G4mplIonisationModel::SetParticle(const G4ParticleDefinition* p)
100{
101 monopole = p;
102 mass = monopole->GetPDGMass();
103 G4double emin =
104 std::min(LowEnergyLimit(),0.1*mass*(1./sqrt(1. - betalow*betalow) - 1.));
105 G4double emax =
106 std::max(HighEnergyLimit(),10.*mass*(1./sqrt(1. - beta2lim) - 1.));
107 SetLowEnergyLimit(emin);
108 SetHighEnergyLimit(emax);
109}

113void G4mplIonisationModel::Initialise(const G4ParticleDefinition* p,
114 const G4DataVector&)
115{
116 if(!monopole) { SetParticle(p); }
117 if(!fParticleChange) { fParticleChange = GetParticleChangeForLoss(); }
118 if(IsMaster()) {
119 if(!dedx0) { dedx0 = new std::vector<G4double>; }
120 G4ProductionCutsTable* theCoupleTable=
121 G4ProductionCutsTable::GetProductionCutsTable();
122 G4int numOfCouples = theCoupleTable->GetTableSize();
123 G4int n = dedx0->size();
124 if(n < numOfCouples) { dedx0->resize(numOfCouples); }
125
126 // initialise vector
127 for(G4int i=0; i<numOfCouples; ++i) {
128
129 const G4Material* material =
130 theCoupleTable->GetMaterialCutsCouple(i)->GetMaterial();
131 G4double eDensity = material->GetElectronDensity();
132 G4double vF = electron_Compton_length*pow(3.*pi*pi*eDensity,0.3333333333);
133 (*dedx0)[i] = pi_hbarc2_over_mc2*eDensity*nmpl*nmpl*
134 (G4Log(2.*vF/fine_structure_const) - 0.5)/vF;
135 }
136 }
137}

141G4double G4mplIonisationModel::ComputeDEDXPerVolume(const G4Material* material,
142 const G4ParticleDefinition* p,
143 G4double kineticEnergy,
144 G4double)
145{
146 if(!monopole) { SetParticle(p); }
147 G4double tau = kineticEnergy / mass;
148 G4double gam = tau + 1.0;
149 G4double bg2 = tau * (tau + 2.0);
150 G4double beta2 = bg2 / (gam * gam);
151 G4double beta = sqrt(beta2);
152
153 // low-energy asymptotic formula
154 //G4double dedx = dedxlim*beta*material->GetDensity();
155 G4double dedx = (*dedx0)[CurrentCouple()->GetIndex()]*beta;
156
157 // above asymptotic
158 if(beta > betalow) {
159
160 // high energy
161 if(beta >= betalim) {
162 dedx = ComputeDEDXAhlen(material, bg2);
163
164 } else {
165
166 //G4double dedx1 = dedxlim*betalow*material->GetDensity();
167 G4double dedx1 = (*dedx0)[CurrentCouple()->GetIndex()]*betalow;
168 G4double dedx2 = ComputeDEDXAhlen(material, bg2lim);
169
170 // extrapolation between two formula
171 G4double kapa2 = beta - betalow;
172 G4double kapa1 = betalim - beta;
173 dedx = (kapa1*dedx1 + kapa2*dedx2)/(kapa1 + kapa2);
174 }
175 }
176 return dedx;
177}

179//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
180
181G4double G4mplIonisationModel::ComputeDEDXAhlen(const G4Material* material,
182 G4double bg2)
183{
184 G4double eDensity = material->GetElectronDensity();
185 G4double eexc = material->GetIonisation()->GetMeanExcitationEnergy();
186 G4double cden = material->GetIonisation()->GetCdensity();
187 G4double mden = material->GetIonisation()->GetMdensity();
188 G4double aden = material->GetIonisation()->GetAdensity();
189 G4double x0den = material->GetIonisation()->GetX0density();
190 G4double x1den = material->GetIonisation()->GetX1density();
191
192 // Ahlen's formula for nonconductors, [1]p157, f(5.7)
193 G4double dedx = log(2.0 * electron_mass_c2 * bg2 / eexc) - 0.5;
194
195 // Kazama et al. cross-section correction
196 G4double k = 0.406;
197 if(nmpl > 1) k = 0.346;
198
199 // Bloch correction
200 const G4double B[7] = { 0.0, 0.248, 0.672, 1.022, 1.243, 1.464, 1.685};
201
202 dedx += 0.5 * k - B[nmpl];
203
204 // density effect correction
205 G4double deltam;
206 G4double x = log(bg2) / twoln10;
207 if ( x >= x0den ) {
208 deltam = twoln10 * x - cden;
209 if ( x < x1den ) deltam += aden * pow((x1den-x), mden);
210 dedx -= 0.5 * deltam;
211 }
212
213 // now compute the total ionization loss
214 dedx *= pi_hbarc2_over_mc2 * eDensity * nmpl * nmpl;
215
216 if (dedx < 0.0) dedx = 0.;
217 return dedx;
218}

231G4double G4mplIonisationModel::SampleFluctuations(
232 const G4MaterialCutsCouple* couple,
233 const G4DynamicParticle* dp,
234 G4double tmax,
235 G4double length,
236 G4double meanLoss)
237{
238 G4double siga = Dispersion(couple->GetMaterial(),dp,tmax,length);
239 G4double loss = meanLoss;
240 siga = sqrt(siga);
241 G4double twomeanLoss = meanLoss + meanLoss;
242
243 if(twomeanLoss < siga) {
244 G4double x;
245 do {
246 loss = twomeanLoss*G4UniformRand();
247 x = (loss - meanLoss)/siga;
248 // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
249 } while (1.0 - 0.5*x*x < G4UniformRand());
250 } else {
251 do {
252 loss = G4RandGauss::shoot(meanLoss,siga);
253 // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
254 } while (0.0 > loss || loss > twomeanLoss);
255 }
256 return loss;
257}

261G4double G4mplIonisationModel::Dispersion(const G4Material* material,
262 const G4DynamicParticle* dp,
263 G4double tmax,
264 G4double length)
265{
266 G4double siga = 0.0;
267 G4double tau = dp->GetKineticEnergy()/mass;
268 if(tau > 0.0) {
269 G4double electronDensity = material->GetElectronDensity();
270 G4double gam = tau + 1.0;
271 G4double invbeta2 = (gam*gam)/(tau * (tau+2.0));
272 siga = (invbeta2 - 0.5) * twopi_mc2_rcl2 * tmax * length
273 * electronDensity * chargeSquare;
274 }
275 return siga;
276}
