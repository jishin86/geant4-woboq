39// Class Description:
40// * calculates the differential cross section (ME squared, without phase space)
41// incomming positron (along positive z direction) scatters at an electron at rest
42// * phi denotes the angle between the scattering plane and the incoming PRF X-axis
43// * all stokes vectors refer to spins in the Global System (X,Y,Z)
44//
45#ifndef G4PolarizedBhabhaCrossSection_h
46#define G4PolarizedBhabhaCrossSection_h 1
47
48#include "G4StokesVector.hh"
49#include "G4VPolarizedCrossSection.hh"
50
51class G4PolarizedBhabhaCrossSection : public G4VPolarizedCrossSection
52{
53public:
54 G4PolarizedBhabhaCrossSection();
55 virtual ~G4PolarizedBhabhaCrossSection();
56public:
57 void Initialize(G4double x, G4double y, G4double phi,
58 const G4StokesVector & p0,const G4StokesVector & p1,
59 G4int flag=0) override;
60
61 G4double XSection(const G4StokesVector & pol2,const G4StokesVector & pol3) override;
62 G4double TotalXSection(G4double xmin, G4double xmax, G4double y,
63 const G4StokesVector & pol0,
64 const G4StokesVector & pol1) override;
65 // return expected mean polarisation
66 G4StokesVector GetPol2() override;
67 G4StokesVector GetPol3() override;
68private:
69 G4double phi0;
70 // - part depending on the polarization of the final positron
71 G4ThreeVector phi2;
72 // - part depending on the polarization of the final electron
73 G4ThreeVector phi3;
74
75};
76#endif
