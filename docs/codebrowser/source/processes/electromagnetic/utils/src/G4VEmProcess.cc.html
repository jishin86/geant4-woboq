75G4VEmProcess::G4VEmProcess(const G4String& name, G4ProcessType type):
76 G4VDiscreteProcess(name, type),
77 secondaryParticle(nullptr),
78 buildLambdaTable(true),
79 numberOfModels(0),
80 theLambdaTable(nullptr),
81 theLambdaTablePrim(nullptr),
82 theDensityFactor(nullptr),
83 theDensityIdx(nullptr),
84 integral(false),
85 applyCuts(false),
86 startFromNull(false),
87 splineFlag(true),
88 isIon(false),
89 currentCouple(nullptr),
90 isTheMaster(true),
91 masterProc(nullptr),
92 theData(nullptr),
93 currentModel(nullptr),
94 particle(nullptr),
95 currentParticle(nullptr)
96{
97 theParameters = G4EmParameters::Instance();
98 SetVerboseLevel(1);
99
100 // Size of tables assuming spline
101 minKinEnergy = 0.1*keV;
102 maxKinEnergy = 100.0*TeV;
103 nLambdaBins = 84;
104 minKinEnergyPrim = DBL_MAX;
105 actBinning = actSpline = actMinKinEnergy = actMaxKinEnergy = false;
106
107 // default lambda factor
108 lambdaFactor = 0.8;
109
110 // default limit on polar angle
111 biasFactor = fFactor = 1.0;
112
113 // particle types
114 theGamma = G4Gamma::Gamma();
115 theElectron = G4Electron::Electron();
116 thePositron = G4Positron::Positron();
117
118 theCuts = theCutsGamma = theCutsElectron = theCutsPositron = nullptr;
119
120 pParticleChange = &fParticleChange;
121 fParticleChange.SetSecondaryWeightByProcess(true);
122 secParticles.reserve(5);
123
124 baseMaterial = currentMaterial = nullptr;
125
126 preStepLambda = preStepKinEnergy = 0.0;
127 mfpKinEnergy = DBL_MAX;
128 massRatio = 1.0;
129
130 idxLambda = idxLambdaPrim = currentCoupleIndex
131 = basedCoupleIndex = 0;
132
133 modelManager = new G4EmModelManager();
134 biasManager = nullptr;
135 biasFlag = false;
136 weightFlag = false;
137 lManager = G4LossTableManager::Instance();
138 lManager->Register(this);
139 secID = fluoID = augerID = biasID = -1;
140 mainSecondaries = 100;
141 if("phot" == GetProcessName() || "compt" == GetProcessName()
142 || "e-_G4DNAIonisation" == GetProcessName()
143 || "hydrogen_G4DNAIonisation" == GetProcessName()
144 || "helium_G4DNAIonisation" == GetProcessName()
145 || "alpha_G4DNAIonisation" == GetProcessName()
146 || "alpha+_G4DNAIonisation" == GetProcessName()
147 || "proton_G4DNAIonisation" == GetProcessName()
148 || "GenericIon_G4DNAIonisation" == GetProcessName() )
149 {
150 mainSecondaries = 1;
151 }
152}

254void G4VEmProcess::PreparePhysicsTable(const G4ParticleDefinition& part)
255{
256 isTheMaster = lManager->IsMaster();
257
258 if(!particle) { SetParticle(&part); }
259
260 if(part.GetParticleType() == "nucleus" &&
261 part.GetParticleSubType() == "generic") {
262
263 G4String pname = part.GetParticleName();
264 if(pname != "deuteron" && pname != "triton" &&
265 pname != "alpha" && pname != "He3" &&
266 pname != "alpha+" && pname != "helium" &&
267 pname != "hydrogen") {
268
269 particle = G4GenericIon::GenericIon();
270 isIon = true;
271 }
272 }
273
274 if(1 < verboseLevel) {
275 G4cout << "G4VEmProcess::PreparePhysicsTable() for "
276 << GetProcessName()
277 << " and particle " << part.GetParticleName()
278 << " local particle " << particle->GetParticleName()
279 << G4endl;
280 }
281
282 if(particle != &part) { return; }
283
284 G4LossTableBuilder* bld = lManager->GetTableBuilder();
285
286 lManager->PreparePhysicsTable(&part, this, isTheMaster);
287
288 Clear();
289 InitialiseProcess(particle);
290
291 const G4ProductionCutsTable* theCoupleTable=
292 G4ProductionCutsTable::GetProductionCutsTable();
293 size_t n = theCoupleTable->GetTableSize();
294
295 theEnergyOfCrossSectionMax.resize(n, 0.0);
296 theCrossSectionMax.resize(n, DBL_MAX);
297
298 // initialisation of the process
299 if(!actMinKinEnergy) { minKinEnergy = theParameters->MinKinEnergy(); }
300 if(!actMaxKinEnergy) { maxKinEnergy = theParameters->MaxKinEnergy(); }
301 if(!actSpline) { splineFlag = theParameters->Spline(); }
302
303 if(isTheMaster) {
304 SetVerboseLevel(theParameters->Verbose());
305 if(!theData) { theData = new G4EmDataHandler(2); }
306 } else {
307 SetVerboseLevel(theParameters->WorkerVerbose());
308 }
309 applyCuts = theParameters->ApplyCuts();
310 lambdaFactor = theParameters->LambdaFactor();
311 theParameters->DefineRegParamForEM(this);
312
313 // initialisation of models
314 numberOfModels = modelManager->NumberOfModels();
315 for(G4int i=0; i<numberOfModels; ++i) {
316 G4VEmModel* mod = modelManager->GetModel(i);
317 if(0 == i) { currentModel = mod; }
318 mod->SetPolarAngleLimit(theParameters->MscThetaLimit());
319 mod->SetMasterThread(isTheMaster);
320 if(mod->HighEnergyLimit() > maxKinEnergy) {
321 mod->SetHighEnergyLimit(maxKinEnergy);
322 }
323 }
324
325 if(lManager->AtomDeexcitation()) { modelManager->SetFluoFlag(true); }
326 theCuts = modelManager->Initialise(particle,secondaryParticle,
327 2.,verboseLevel);
328 theCutsGamma = theCoupleTable->GetEnergyCutsVector(idxG4GammaCut);
329 theCutsElectron = theCoupleTable->GetEnergyCutsVector(idxG4ElectronCut);
330 theCutsPositron = theCoupleTable->GetEnergyCutsVector(idxG4PositronCut);
331
332 // prepare tables
333 if(buildLambdaTable && isTheMaster){
334 theLambdaTable = theData->MakeTable(0);
335 bld->InitialiseBaseMaterials(theLambdaTable);
336 }
337 // high energy table
338 if(isTheMaster && minKinEnergyPrim < maxKinEnergy){
339 theLambdaTablePrim = theData->MakeTable(1);
340 bld->InitialiseBaseMaterials(theLambdaTablePrim);
341 }
342 // forced biasing
343 if(biasManager) {
344 biasManager->Initialise(part,GetProcessName(),verboseLevel);
345 biasFlag = false;
346 }
347 // defined ID of secondary particles
348 G4String nam1 = GetProcessName();
349 secID = G4PhysicsModelCatalog::Register(nam1);
350 if(100 > mainSecondaries) {
351 G4String nam2 = nam1 + "_fluo" ;
352 G4String nam3 = nam1 + "_auger";
353 G4String nam4 = nam1 + "_split";
354 fluoID = G4PhysicsModelCatalog::Register(nam2);
355 augerID = G4PhysicsModelCatalog::Register(nam3);
356 biasID = G4PhysicsModelCatalog::Register(nam4);
357 }
358}

361
362void G4VEmProcess::BuildPhysicsTable(const G4ParticleDefinition& part)
363{
364 if(!masterProc) {
365 if(isTheMaster) { masterProc = this; }
366 else { masterProc = static_cast<const G4VEmProcess*>(GetMasterProcess());}
367 }
368
369 G4String num = part.GetParticleName();
370 if(1 < verboseLevel) {
371 G4cout << "### G4VEmProcess::BuildPhysicsTable() for "
372 << GetProcessName()
373 << " and particle " << num
374 << " buildLambdaTable= " << buildLambdaTable
375 << " isTheMaster= " << isTheMaster
376 << " " << masterProc
377 << G4endl;
378 }
379
380 if(particle == &part) {
381
382 G4LossTableBuilder* bld = lManager->GetTableBuilder();
383
384 // worker initialisation
385 if(!isTheMaster) {
386 theLambdaTable = masterProc->LambdaTable();
387 theLambdaTablePrim = masterProc->LambdaTablePrim();
388
389 if(theLambdaTable) {
390 bld->InitialiseBaseMaterials(theLambdaTable);
391 } else if(theLambdaTablePrim) {
392 bld->InitialiseBaseMaterials(theLambdaTablePrim);
393 }
394 theDensityFactor = bld->GetDensityFactors();
395 theDensityIdx = bld->GetCoupleIndexes();
396 if(theLambdaTable) { FindLambdaMax(); }
397
398 // local initialisation of models
399 G4bool printing = true;
400 numberOfModels = modelManager->NumberOfModels();
401 for(G4int i=0; i<numberOfModels; ++i) {
402 G4VEmModel* mod = GetModelByIndex(i, printing);
403 G4VEmModel* mod0= masterProc->GetModelByIndex(i, printing);
404 //G4cout << i << ". " << mod << " " << mod0 << " "
405 // << particle->GetParticleName() << G4endl;
406 mod->InitialiseLocal(particle, mod0);
407 }
408 // master thread
409 } else {
410 theDensityFactor = bld->GetDensityFactors();
411 theDensityIdx = bld->GetCoupleIndexes();
412 if(buildLambdaTable || minKinEnergyPrim < maxKinEnergy) {
413 BuildLambdaTable();
414 }
415 }
416 }
417
418 // explicitly defined printout by particle name
419 if(1 < verboseLevel ||
420 (0 < verboseLevel && (num == "gamma" || num == "e-" ||
421 num == "e+" || num == "mu+" ||
422 num == "mu-" || num == "proton"||
423 num == "pi+" || num == "pi-" ||
424 num == "kaon+" || num == "kaon-" ||
425 num == "alpha" || num == "anti_proton" ||
426 num == "GenericIon"|| num == "alpha++" ||
427 num == "alpha+" || num == "helium" ||
428 num == "hydrogen")))
429 {
430 StreamInfo(G4cout, part);
431 }
432
433 if(1 < verboseLevel) {
434 G4cout << "### G4VEmProcess::BuildPhysicsTable() done for "
435 << GetProcessName()
436 << " and particle " << num
437 << G4endl;
438 }
439}

443void G4VEmProcess::BuildLambdaTable()
444{
445 if(1 < verboseLevel) {
446 G4cout << "G4EmProcess::BuildLambdaTable() for process "
447 << GetProcessName() << " and particle "
448 << particle->GetParticleName() << " " << this
449 << G4endl;
450 }
451
452 // Access to materials
453 const G4ProductionCutsTable* theCoupleTable=
454 G4ProductionCutsTable::GetProductionCutsTable();
455 size_t numOfCouples = theCoupleTable->GetTableSize();
456
457 G4LossTableBuilder* bld = lManager->GetTableBuilder();
458
459 G4PhysicsLogVector* aVector = nullptr;
460 G4PhysicsLogVector* aVectorPrim = nullptr;
461 G4PhysicsLogVector* bVectorPrim = nullptr;
462
463 G4double scale = theParameters->MaxKinEnergy()/theParameters->MinKinEnergy();
464 G4int nbin =
465 theParameters->NumberOfBinsPerDecade()*G4lrint(std::log10(scale));
466 scale = G4Log(scale);
467 if(actBinning) { nbin = std::max(nbin, nLambdaBins); }
468 G4double emax1 = std::min(maxKinEnergy, minKinEnergyPrim);
469
470 for(size_t i=0; i<numOfCouples; ++i) {
471
472 if (bld->GetFlag(i)) {
473
474 // create physics vector and fill it
475 const G4MaterialCutsCouple* couple =
476 theCoupleTable->GetMaterialCutsCouple(i);
477
478 // build main table
479 if(buildLambdaTable) {
480 delete (*theLambdaTable)[i];
481
482 // if start from zero then change the scale
483 G4double emin = minKinEnergy;
484 G4bool startNull = false;
485 if(startFromNull) {
486 G4double e = MinPrimaryEnergy(particle,couple->GetMaterial());
487 if(e >= emin) {
488 emin = e;
489 startNull = true;
490 }
491 }
492 G4double emax = emax1;
493 if(emax <= emin) { emax = 2*emin; }
494 G4int bin = G4lrint(nbin*G4Log(emax/emin)/scale);
495 if(bin < 3) { bin = 3; }
496 aVector = new G4PhysicsLogVector(emin, emax, bin);
497 aVector->SetSpline(splineFlag);
498 modelManager->FillLambdaVector(aVector, couple, startNull);
499 if(splineFlag) { aVector->FillSecondDerivatives(); }
500 G4PhysicsTableHelper::SetPhysicsVector(theLambdaTable, i, aVector);
501 }
502 // build high energy table
503 if(minKinEnergyPrim < maxKinEnergy) {
504 delete (*theLambdaTablePrim)[i];
505
506 // start not from zero
507 if(!bVectorPrim) {
508 G4int bin = G4lrint(nbin*G4Log(maxKinEnergy/minKinEnergyPrim)/scale);
509 if(bin < 3) { bin = 3; }
510 aVectorPrim =
511 new G4PhysicsLogVector(minKinEnergyPrim, maxKinEnergy, bin);
512 bVectorPrim = aVectorPrim;
513 } else {
514 aVectorPrim = new G4PhysicsLogVector(*bVectorPrim);
515 }
516 // always use spline
517 aVectorPrim->SetSpline(splineFlag);
518 modelManager->FillLambdaVector(aVectorPrim, couple, false,
519 fIsCrossSectionPrim);
520 aVectorPrim->FillSecondDerivatives();
521 G4PhysicsTableHelper::SetPhysicsVector(theLambdaTablePrim, i,
522 aVectorPrim);
523 }
524 }
525 }
526
527 if(buildLambdaTable) { FindLambdaMax(); }
528
529 if(1 < verboseLevel) {
530 G4cout << "Lambda table is built for "
531 << particle->GetParticleName()
532 << G4endl;
533 }
534}

538void G4VEmProcess::StreamInfo(std::ostream& out,
539 const G4ParticleDefinition& part, G4bool rst) const
540{
541 G4String indent = (rst ? " " : "");
542 out << std::setprecision(6);
543 out << G4endl << indent << GetProcessName() << ": ";
544 if (!rst) {
545 out << " for " << part.GetParticleName();
546 if (integral) { out << ","; }
547 }
548 if(integral) { out << " integral:1 "; }
549 if(applyCuts) { out << " applyCuts:1 "; }
550 out << " SubType=" << GetProcessSubType();
551 if(biasFactor != 1.0) { out << " BiasingFactor= " << biasFactor; }
552 out << " BuildTable=" << buildLambdaTable << G4endl;
553 if(buildLambdaTable) {
554 if(particle == &part) {
555 size_t length = theLambdaTable->length();
556 for(size_t i=0; i<length; ++i) {
557 G4PhysicsVector* v = (*theLambdaTable)[i];
558 if(v) {
559 out << " Lambda table from ";
560 G4double emin = v->Energy(0);
561 G4double emax = v->GetMaxEnergy();
562 G4int nbin = v->GetVectorLength() - 1;
563 if(emin > minKinEnergy) { out << "threshold "; }
564 else { out << G4BestUnit(emin,"Energy"); }
565 out << " to "
566 << G4BestUnit(emax,"Energy")
567 << ", " << G4lrint(nbin/std::log10(emax/emin))
568 << " bins/decade, spline: "
569 << splineFlag << G4endl;
570 break;
571 }
572 }
573 } else {
574 out << " Used Lambda table of "
575 << particle->GetParticleName() << G4endl;
576 }
577 }
578 if(minKinEnergyPrim < maxKinEnergy) {
579 if(particle == &part) {
580 size_t length = theLambdaTablePrim->length();
581 for(size_t i=0; i<length; ++i) {
582 G4PhysicsVector* v = (*theLambdaTablePrim)[i];
583 if(v) {
584 out << " LambdaPrime table from "
585 << G4BestUnit(v->Energy(0),"Energy")
586 << " to "
587 << G4BestUnit(v->GetMaxEnergy(),"Energy")
588 << " in " << v->GetVectorLength()-1
589 << " bins " << G4endl;
590 break;
591 }
592 }
593 } else {
594 out << " Used LambdaPrime table of "
595 << particle->GetParticleName() << G4endl;
596 }
597 }
598 StreamProcessInfo(out);
599 modelManager->DumpModelList(out, verboseLevel);
600
601 if(verboseLevel > 2 && buildLambdaTable) {
602 out << " LambdaTable address= " << theLambdaTable << G4endl;
603 if(theLambdaTable && particle == &part) {
604 out << (*theLambdaTable) << G4endl;
605 }
606 }
607}

611void G4VEmProcess::StartTracking(G4Track* track)
612{
613 // reset parameters for the new track
614 currentParticle = track->GetParticleDefinition();
615 theNumberOfInteractionLengthLeft = -1.0;
616 mfpKinEnergy = DBL_MAX;
617 if(isIon) { massRatio = proton_mass_c2/currentParticle->GetPDGMass(); }
618
619 // forced biasing only for primary particles
620 if(biasManager) {
621 if(0 == track->GetParentID()) {
622 // primary particle
623 biasFlag = true;
624 biasManager->ResetForcedInteraction();
625 }
626 }
627}

629//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
630
631G4double G4VEmProcess::PostStepGetPhysicalInteractionLength(
632 const G4Track& track,
633 G4double previousStepSize,
634 G4ForceCondition* condition)
635{
636 *condition = NotForced;
637 G4double x = DBL_MAX;
638
639 DefineMaterial(track.GetMaterialCutsCouple());
640 preStepKinEnergy = track.GetKineticEnergy();
641 G4double scaledEnergy = preStepKinEnergy*massRatio;
642 SelectModel(scaledEnergy, currentCoupleIndex);
643
644 if(!currentModel->IsActive(scaledEnergy)) {
645 theNumberOfInteractionLengthLeft = -1.0;
646 currentInteractionLength = DBL_MAX;
647 return x;
648 }
649
650 // forced biasing only for primary particles
651 if(biasManager) {
652 if(0 == track.GetParentID()) {
653 if(biasFlag &&
654 biasManager->ForcedInteractionRegion(currentCoupleIndex)) {
655 return biasManager->GetStepLimit(currentCoupleIndex, previousStepSize);
656 }
657 }
658 }
659
660 // compute mean free path
661 if(preStepKinEnergy < mfpKinEnergy) {
662 if (integral) { ComputeIntegralLambda(preStepKinEnergy); }
663 else { preStepLambda = GetCurrentLambda(preStepKinEnergy); }
664
665 // zero cross section
666 if(preStepLambda <= 0.0) {
667 theNumberOfInteractionLengthLeft = -1.0;
668 currentInteractionLength = DBL_MAX;
669 }
670 }
671
672 // non-zero cross section
673 if(preStepLambda > 0.0) {
674
675 if (theNumberOfInteractionLengthLeft < 0.0) {
676
677 // beggining of tracking (or just after DoIt of this process)
678 theNumberOfInteractionLengthLeft = -G4Log( G4UniformRand() );
679 theInitialNumberOfInteractionLength = theNumberOfInteractionLengthLeft;
680
681 } else if(currentInteractionLength < DBL_MAX) {
682
683 theNumberOfInteractionLengthLeft -=
684 previousStepSize/currentInteractionLength;
685 theNumberOfInteractionLengthLeft =
686 std::max(theNumberOfInteractionLengthLeft, 0.0);
687 }
688
689 // new mean free path and step limit for the next step
690 currentInteractionLength = 1.0/preStepLambda;
691 x = theNumberOfInteractionLengthLeft * currentInteractionLength;
692 }
693 return x;
694}

696//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
697
698void G4VEmProcess::ComputeIntegralLambda(G4double e)
699{
700 // condition to skip recomputation of cross section
701 G4double epeak = theEnergyOfCrossSectionMax[currentCoupleIndex];
702 if(e <= epeak && e/lambdaFactor >= mfpKinEnergy) { return; }
703
704 // recomputation is needed
705 if (e <= epeak) {
706 preStepLambda = GetCurrentLambda(e);
707 mfpKinEnergy = e;
708
709 } else {
710 G4double e1 = e*lambdaFactor;
711 if(e1 > epeak) {
712 preStepLambda = GetCurrentLambda(e);
713 mfpKinEnergy = e;
714 G4double preStepLambda1 = GetCurrentLambda(e1);
715 if(preStepLambda1 > preStepLambda) {
716 mfpKinEnergy = e1;
717 preStepLambda = preStepLambda1;
718 }
719 } else {
720 preStepLambda = fFactor*theCrossSectionMax[currentCoupleIndex];
721 mfpKinEnergy = epeak;
722 }
723 }
724}

726//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
727
728G4VParticleChange* G4VEmProcess::PostStepDoIt(const G4Track& track,
729 const G4Step& step)
730{
731 // In all cases clear number of interaction lengths
732 theNumberOfInteractionLengthLeft = -1.0;
733 mfpKinEnergy = DBL_MAX;
734
735 fParticleChange.InitializeForPostStep(track);
736
737 // Do not make anything if particle is stopped, the annihilation then
738 // should be performed by the AtRestDoIt!
739 if (track.GetTrackStatus() == fStopButAlive) { return &fParticleChange; }
740
741 G4double finalT = track.GetKineticEnergy();
742
743 // forced process - should happen only once per track
744 if(biasFlag) {
745 if(biasManager->ForcedInteractionRegion(currentCoupleIndex)) {
746 biasFlag = false;
747 }
748 }
749
750 // Integral approach
751 if (integral) {
752 G4double lx = GetLambda(finalT, currentCouple);
753 if(preStepLambda<lx && 1 < verboseLevel) {
754 G4cout << "WARNING: for " << currentParticle->GetParticleName()
755 << " and " << GetProcessName()
756 << " E(MeV)= " << finalT/MeV
757 << " preLambda= " << preStepLambda << " < "
758 << lx << " (postLambda) "
759 << G4endl;
760 }
761
762 if(preStepLambda*G4UniformRand() > lx) {
763 ClearNumberOfInteractionLengthLeft();
764 return &fParticleChange;
765 }
766 }
767
768 G4double scaledEnergy = finalT*massRatio;
769 SelectModel(scaledEnergy, currentCoupleIndex);
770 if(!currentModel->IsActive(scaledEnergy)) { return &fParticleChange; }
771
772 // define new weight for primary and secondaries
773 G4double weight = fParticleChange.GetParentWeight();
774 if(weightFlag) {
775 weight /= biasFactor;
776 fParticleChange.ProposeWeight(weight);
777 }
778
779 /*
780 if(0 < verboseLevel) {
781 G4cout << "G4VEmProcess::PostStepDoIt: Sample secondary; E= "
782 << finalT/MeV
783 << " MeV; model= (" << currentModel->LowEnergyLimit()
784 << ", " << currentModel->HighEnergyLimit() << ")"
785 << G4endl;
786 }
787 */
788
789 // sample secondaries
790 secParticles.clear();
791 currentModel->SampleSecondaries(&secParticles,
792 currentCouple,
793 track.GetDynamicParticle(),
794 (*theCuts)[currentCoupleIndex]);
795
796 G4int num0 = secParticles.size();
797
798 // splitting or Russian roulette
799 if(biasManager) {
800 if(biasManager->SecondaryBiasingRegion(currentCoupleIndex)) {
801 G4double eloss = 0.0;
802 weight *= biasManager->ApplySecondaryBiasing(
803 secParticles, track, currentModel, &fParticleChange, eloss,
804 currentCoupleIndex, (*theCuts)[currentCoupleIndex],
805 step.GetPostStepPoint()->GetSafety());
806 if(eloss > 0.0) {
807 eloss += fParticleChange.GetLocalEnergyDeposit();
808 fParticleChange.ProposeLocalEnergyDeposit(eloss);
809 }
810 }
811 }
812
813 // save secondaries
814 G4int num = secParticles.size();
815 if(num > 0) {
816
817 fParticleChange.SetNumberOfSecondaries(num);
818 G4double edep = fParticleChange.GetLocalEnergyDeposit();
819 G4double time = track.GetGlobalTime();
820
821 for (G4int i=0; i<num; ++i) {
822 if (secParticles[i]) {
823 G4DynamicParticle* dp = secParticles[i];
824 const G4ParticleDefinition* p = dp->GetParticleDefinition();
825 G4double e = dp->GetKineticEnergy();
826 G4bool good = true;
827 if(applyCuts) {
828 if (p == theGamma) {
829 if (e < (*theCutsGamma)[currentCoupleIndex]) { good = false; }
830
831 } else if (p == theElectron) {
832 if (e < (*theCutsElectron)[currentCoupleIndex]) { good = false; }
833
834 } else if (p == thePositron) {
835 if (electron_mass_c2 < (*theCutsGamma)[currentCoupleIndex] &&
836 e < (*theCutsPositron)[currentCoupleIndex]) {
837 good = false;
838 e += 2.0*electron_mass_c2;
839 }
840 }
841 // added secondary if it is good
842 }
843 if (good) {
844 G4Track* t = new G4Track(dp, time, track.GetPosition());
845 t->SetTouchableHandle(track.GetTouchableHandle());
846 if (biasManager) {
847 t->SetWeight(weight * biasManager->GetWeight(i));
848 } else {
849 t->SetWeight(weight);
850 }
851 pParticleChange->AddSecondary(t);
852
853 // define type of secondary
854 if(i < mainSecondaries) { t->SetCreatorModelIndex(secID); }
855 else if(i < num0) {
856 if(p == theGamma) {
857 t->SetCreatorModelIndex(fluoID);
858 } else {
859 t->SetCreatorModelIndex(augerID);
860 }
861 } else {
862 t->SetCreatorModelIndex(biasID);
863 }
864 /*
865 G4cout << "Secondary(post step) has weight " << t->GetWeight()
866 << ", Ekin= " << t->GetKineticEnergy()/MeV << " MeV "
867 << GetProcessName() << " fluoID= " << fluoID
868 << " augerID= " << augerID <<G4endl;
869 */
870 } else {
871 delete dp;
872 edep += e;
873 }
874 }
875 }
876 fParticleChange.ProposeLocalEnergyDeposit(edep);
877 }
878
879 if(0.0 == fParticleChange.GetProposedKineticEnergy() &&
880 fAlive == fParticleChange.GetTrackStatus()) {
881 if(particle->GetProcessManager()->GetAtRestProcessVector()->size() > 0)
882 { fParticleChange.ProposeTrackStatus(fStopButAlive); }
883 else { fParticleChange.ProposeTrackStatus(fStopAndKill); }
884 }
885
886 return &fParticleChange;
887}

1069//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
1070
1071void G4VEmProcess::FindLambdaMax()
1072{
1073 if(1 < verboseLevel) {
1074 G4cout << "### G4VEmProcess::FindLambdaMax: "
1075 << particle->GetParticleName()
1076 << " and process " << GetProcessName() << " " << G4endl;
1077 }
1078 size_t n = theLambdaTable->length();
1079 G4PhysicsVector* pv;
1080 G4double e, ss, emax, smax;
1081
1082 size_t i;
1083
1084 // first loop on existing vectors
1085 for (i=0; i<n; ++i) {
1086 pv = (*theLambdaTable)[i];
1087 if(pv) {
1088 size_t nb = pv->GetVectorLength();
1089 emax = DBL_MAX;
1090 smax = 0.0;
1091 if(nb > 0) {
1092 for (size_t j=0; j<nb; ++j) {
1093 e = pv->Energy(j);
1094 ss = (*pv)(j);
1095 if(ss > smax) {
1096 smax = ss;
1097 emax = e;
1098 }
1099 }
1100 }
1101 theEnergyOfCrossSectionMax[i] = emax;
1102 theCrossSectionMax[i] = smax;
1103 if(1 < verboseLevel) {
1104 G4cout << "For " << particle->GetParticleName()
1105 << " Max CS at i= " << i << " emax(MeV)= " << emax/MeV
1106 << " lambda= " << smax << G4endl;
1107 }
1108 }
1109 }
1110 // second loop using base materials
1111 for (i=0; i<n; ++i) {
1112 pv = (*theLambdaTable)[i];
1113 if(!pv){
1114 G4int j = (*theDensityIdx)[i];
1115 theEnergyOfCrossSectionMax[i] = theEnergyOfCrossSectionMax[j];
1116 theCrossSectionMax[i] = (*theDensityFactor)[i]*theCrossSectionMax[j];
1117 }
1118 }
1119}
