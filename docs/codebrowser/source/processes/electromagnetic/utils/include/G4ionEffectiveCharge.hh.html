102//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
103//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
104
105inline G4double G4ionEffectiveCharge::EffectiveChargeSquareRatio(
106 const G4ParticleDefinition* p,
107 const G4Material* material,
108 G4double kineticEnergy)
109{
110 G4double charge = effCharge;
111 if( kineticEnergy != lastKinEnergy || material != lastMat || p != lastPart) {
112 charge = EffectiveCharge(p,material,kineticEnergy);
113 }
114 charge *= chargeCorrection*inveplus;
115
116 return charge*charge;
117}
