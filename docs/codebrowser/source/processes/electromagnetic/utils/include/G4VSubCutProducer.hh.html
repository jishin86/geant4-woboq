49#ifndef G4VSubCutProducer_h
50#define G4VSubCutProducer_h 1
51
52#include "globals.hh"
53#include "G4Step.hh"
54#include "G4Track.hh"
55#include <vector>
56
57class G4VSubCutProducer
58{
59public:
60
61 explicit G4VSubCutProducer(const G4String& name) : fName(name) {};
62
63 virtual ~G4VSubCutProducer() {};
64
65 // Sample direction in global coordinate system,
66 // this means for zero scattering angle this direction is the same
67 // as the direction of primary
68 virtual void SampleSecondaries(const G4Step& step,
69 std::vector<G4Track*>& tracks,
70 G4double& eloss,
71 G4double cut) const = 0;
72
73 inline const G4String& GetName() const;
74
75private:
76
77 // hide assignment operator
78 G4VSubCutProducer & operator=(const G4VSubCutProducer &right) = delete;
79 G4VSubCutProducer(const G4VSubCutProducer&) = delete;
80
81 G4String fName;
82};
83
84inline const G4String& G4VSubCutProducer::GetName() const
85{
86 return fName;
87}
88
89#endif
