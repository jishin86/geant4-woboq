105inline const G4Element* G4EmElementSelector::SelectRandomAtom(G4double e) const
106{
107 const G4Element* element = (*theElementVector)[nElmMinusOne];
108 if (nElmMinusOne > 0) {
109 G4double x = G4UniformRand();
110 size_t idx(0);
111 for(G4int i=0; i<nElmMinusOne; ++i) {
112 if (x <= (xSections[i])->Value(e, idx)) {
113 element = (*theElementVector)[i];
114 break;
115 }
116 }
117 }
118 return element;
119}
120
121inline const G4Material* G4EmElementSelector::GetMaterial() const
122{
123 return material;
124}
