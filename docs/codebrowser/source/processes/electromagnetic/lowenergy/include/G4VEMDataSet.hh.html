49#ifndef G4VEMDATASET_HH
50#define G4VEMDATASET_HH 1
51
52#include "globals.hh"
53#include "G4DataVector.hh"
54
55class G4VEMDataSet
56{
57public:
58
59 G4VEMDataSet();
60 virtual ~G4VEMDataSet();
61
62 virtual G4double FindValue(G4double x, G4int componentId = 0) const = 0;
63
64 virtual void PrintData(void) const = 0;
65
66 virtual const G4VEMDataSet* GetComponent(G4int componentId) const = 0;
67 virtual void AddComponent(G4VEMDataSet* dataSet) = 0;
68 virtual size_t NumberOfComponents(void) const = 0;
69
70 virtual const G4DataVector& GetEnergies(G4int componentId) const = 0;
71 virtual const G4DataVector& GetData(G4int componentId) const = 0;
72 virtual const G4DataVector& GetLogEnergies(G4int componentId) const = 0;
73 virtual const G4DataVector& GetLogData(G4int componentId) const = 0;
74
75 virtual void SetEnergiesData(G4DataVector* x, G4DataVector* data, G4int component=0) = 0;
76 virtual void SetLogEnergiesData(G4DataVector* x,
77 G4DataVector* data,
78 G4DataVector* Log_x,
79 G4DataVector* Log_data,
80 G4int component=0) = 0;
81
82 virtual G4bool LoadData(const G4String& fileName) = 0;
83 virtual G4bool LoadNonLogData(const G4String& fileName) = 0;
84
85 virtual G4bool SaveData(const G4String& fileName) const = 0;
86
87 virtual G4double RandomSelect(G4int componentId = 0) const = 0;
88
89private:
90 // Hide copy constructor and assignment operator
91 G4VEMDataSet(const G4VEMDataSet& copy);
92 G4VEMDataSet& operator=(const G4VEMDataSet& right);
93};
