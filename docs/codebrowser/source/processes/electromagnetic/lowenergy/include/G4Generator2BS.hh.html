67public:

74 G4double out_energy,

80protected:

86 // hide assignment operator

88 G4Generator2BS(const G4Generator2BS&) = delete;
89
90 G4double fz;
91 G4double ratio;
92 G4double ratio1;
93 G4double ratio2;
94 G4double delta;
95
96 G4Pow* g4pow;
97 G4int nwarn;
98
99};
100
101inline G4double G4Generator2BS::RejectionFunction(G4double y) const
102{
103 G4double y2 = (1 + y)*(1 + y);
104 G4double x = 4*y*ratio/y2;
105 return 4*x - ratio1 - (ratio2 - x)*G4Log(delta + fz/y2);
106}
