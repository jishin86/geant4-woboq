50
51public:

58 void Clear();

60

67
68private:

71
72 //G4double xlow;
73 //G4double xhigh;
74
75 G4DataVector* x; //grid points, in increasing order
76 G4DataVector* pac; //value of the cumulative pdf at x_i
77 G4DataVector* a; // rational inverse cumulative inverse distribution parameters
78 G4DataVector* b;

80 std::vector<size_t> *ITTL; //largest j for which pac(j) < (i-1)/(np-1)

83 G4int np; //number of grid points

85
