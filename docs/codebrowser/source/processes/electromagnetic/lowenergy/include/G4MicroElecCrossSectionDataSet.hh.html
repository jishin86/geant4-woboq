40public:

57 virtual size_t NumberOfComponents(void) const

72 virtual void SetEnergiesData(G4DataVector* x, G4DataVector* values, G4int componentId);

74 virtual void SetLogEnergiesData(G4DataVector* x,

111
112 void CleanUpComponents(void);
