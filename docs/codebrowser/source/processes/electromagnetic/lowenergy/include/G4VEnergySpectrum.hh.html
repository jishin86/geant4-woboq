85 G4double kineticEnergy,
86 G4int shell = 0,
87 const G4ParticleDefinition* pd = 0) const = 0;
88
89 virtual G4double MaxEnergyOfSecondaries(G4double kineticEnergy,
90 G4int Z = 0,
91 const G4ParticleDefinition* pd = 0) const = 0;

95 virtual void PrintData() const = 0;

99private:
