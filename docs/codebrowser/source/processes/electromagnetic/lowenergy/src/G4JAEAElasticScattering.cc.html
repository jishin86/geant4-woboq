49
50G4JAEAElasticScattering::G4JAEAElasticScattering(const G4String& processName,
51 G4ProcessType type):G4VEmProcess (processName, type),
52 isInitialised(false)
53{
54 SetStartFromNullFlag(false);
55 SetBuildTableFlag(true);
56 SetProcessSubType(fG4JAEAElasticScattering);
57 SetMinKinEnergyPrim(100*keV);
58 SetSplineFlag(false);
59
60}
61

68//Check particle for the process
69G4bool G4JAEAElasticScattering::IsApplicable(const G4ParticleDefinition& p)
70{
71 return (&p == G4Gamma::Gamma());
72}
73
74//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
75//intialize the process
76void G4JAEAElasticScattering::InitialiseProcess(const G4ParticleDefinition*)
77{
78if(!isInitialised) {
79 isInitialised = true;
80 if(!EmModel()) { SetEmModel(new G4JAEAElasticScatteringModel); }
81 AddEmModel(1, EmModel());
82 }
83
84}
