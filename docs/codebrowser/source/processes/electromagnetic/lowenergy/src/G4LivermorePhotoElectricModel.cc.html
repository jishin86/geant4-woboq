65G4LivermorePhotoElectricModel::G4LivermorePhotoElectricModel(const G4String& nam)
66 : G4VEmModel(nam),fParticleChange(nullptr),maxZ(99),
67 nShellLimit(100),fDeexcitationActive(false),isInitialised(false),
68 fAtomDeexcitation(nullptr)
69{
70 verboseLevel= 0;
71 // Verbosity scale:
72 // 0 = nothing
73 // 1 = warning for energy non-conservation
74 // 2 = details of energy budget
75 // 3 = calculation of cross sections, file openings, sampling of atoms
76 // 4 = entering in methods
77
78 theGamma = G4Gamma::Gamma();
79 theElectron = G4Electron::Electron();
80
81 // default generator
82 SetAngularDistribution(new G4SauterGavrilaAngularDistribution());
83
84 if(verboseLevel>0) {
85 G4cout << "Livermore PhotoElectric is constructed "
86 << " nShellLimit= " << nShellLimit << G4endl;
87 }
88
89 //Mark this model as "applicable" for atomic deexcitation
90 SetDeexcitationFlag(true);
91 fSandiaCof.resize(4,0.0);
92 fCurrSection = 0.0;
93}

97G4LivermorePhotoElectricModel::~G4LivermorePhotoElectricModel()
98{
99 if(IsMaster()) {
100 delete fShellCrossSection;
101 fShellCrossSection = nullptr;
102 for(G4int i=0; i<maxZ; ++i) {
103 delete fParamHigh[i];
104 fParamHigh[i] = nullptr;
105 delete fParamLow[i];

111 }
112 }
113}

117void
118G4LivermorePhotoElectricModel::Initialise(const G4ParticleDefinition*,
119 const G4DataVector&)
120{
121 if (verboseLevel > 2) {
122 G4cout << "Calling G4LivermorePhotoElectricModel::Initialise() " << G4endl;
123 }
124
125 if(IsMaster()) {
126
127 if(!fWater) {
128 fWater = G4Material::GetMaterial("G4_WATER", false);
129 if(fWater) { fWaterEnergyLimit = 13.6*eV; }
130 }
131
132 if(!fShellCrossSection) { fShellCrossSection = new G4ElementData(); }
133
134 char* path = getenv("G4LEDATA");
135
136 G4ProductionCutsTable* theCoupleTable =
137 G4ProductionCutsTable::GetProductionCutsTable();
138 G4int numOfCouples = theCoupleTable->GetTableSize();
139
140 for(G4int i=0; i<numOfCouples; ++i) {
141 const G4MaterialCutsCouple* couple =
142 theCoupleTable->GetMaterialCutsCouple(i);
143 const G4Material* material = couple->GetMaterial();
144 const G4ElementVector* theElementVector = material->GetElementVector();
145 G4int nelm = material->GetNumberOfElements();
146
147 for (G4int j=0; j<nelm; ++j) {
148 G4int Z = std::min(maxZ, (*theElementVector)[j]->GetZasInt());
149 if((!fCrossSectionLE[Z] && 1>fNShells[Z]) ||
150 (!fCrossSection[Z] && (1<=fNShells[Z])) ) { ReadData(Z, path); }
151 }
152 }
153 }
154
155 if (verboseLevel > 2) {
156 G4cout << "Loaded cross section files for new LivermorePhotoElectric model"
157 << G4endl;
158 }
159 if(!isInitialised) {
160 isInitialised = true;
161 fParticleChange = GetParticleChangeForGamma();
162
163 fAtomDeexcitation = G4LossTableManager::Instance()->AtomDeexcitation();
164 }
165 fDeexcitationActive = false;
166 if(fAtomDeexcitation) {
167 fDeexcitationActive = fAtomDeexcitation->IsFluoActive();
168 }
169
170 if (verboseLevel > 0) {
171 G4cout << "LivermorePhotoElectric model is initialized " << G4endl
172 << G4endl;
173 }
174}

178G4double G4LivermorePhotoElectricModel::CrossSectionPerVolume(
179 const G4Material* material,
180 const G4ParticleDefinition* p,
181 G4double energy,
182 G4double, G4double)
183{
184 fCurrSection = 0.0;
185 if(fWater && (material == fWater ||
186 material->GetBaseMaterial() == fWater)) {
187 if(energy <= fWaterEnergyLimit) {
188 fWater->GetSandiaTable()->GetSandiaCofWater(energy, fSandiaCof);
189
190 G4double energy2 = energy*energy;
191 G4double energy3 = energy*energy2;
192 G4double energy4 = energy2*energy2;
193
194 fCurrSection = material->GetDensity()*
195 (fSandiaCof[0]/energy + fSandiaCof[1]/energy2 +
196 fSandiaCof[2]/energy3 + fSandiaCof[3]/energy4);
197 }
198 }
199 if(0.0 == fCurrSection) {
200 fCurrSection = G4VEmModel::CrossSectionPerVolume(material, p, energy);
201 }
202 return fCurrSection;
203}

207G4double G4LivermorePhotoElectricModel::ComputeCrossSectionPerAtom(
208 const G4ParticleDefinition*,
209 G4double energy,
210 G4double ZZ, G4double,
211 G4double, G4double)
212{
213 if (verboseLevel > 3) {
214 G4cout << "\n G4LivermorePhotoElectricModel::ComputeCrossSectionPerAtom():"
215 << " Z= " << ZZ << " R(keV)= " << energy/keV << G4endl;
216 }
217 G4double cs = 0.0;
218 G4int Z = G4lrint(ZZ);
219 if(Z < 1 || Z >= maxZ) { return cs; }
220 // if element was not initialised
221
222 // do initialisation safely for MT mode
223 if((!fCrossSectionLE[Z] && 1>fNShells[Z]) || (!fCrossSection[Z] && (1<=fNShells[Z])) ) {
224 InitialiseForElement(0, Z);
225 if(!fCrossSectionLE[Z] && !fCrossSection[Z]) { return cs; }
226 }
227 //7: rows in the parameterization file; 5: number of parameters
228 G4int idx = fNShells[Z]*7 - 5;
229
230 if (energy < (*(fParamHigh[Z]))[idx-1]) {
231 energy = (*(fParamHigh[Z]))[idx-1];
232 }
233
234 G4double x1 = 1.0/energy;
235 G4double x2 = x1*x1;
236 G4double x3 = x2*x1;
237
238 // high energy parameterisation
239 if(energy >= (*(fParamHigh[Z]))[0]) {
240
241 G4double x4 = x2*x2;
242 G4double x5 = x4*x1;
243
244 cs = x1*((*(fParamHigh[Z]))[idx] + x1*(*(fParamHigh[Z]))[idx+1]
245 + x2*(*(fParamHigh[Z]))[idx+2] + x3*(*(fParamHigh[Z]))[idx+3]
246 + x4*(*(fParamHigh[Z]))[idx+4]+ x5*(*(fParamHigh[Z]))[idx+5]);
247
248 }
249 // low energy parameterisation
250 else if(energy >= (*(fParamLow[Z]))[0]) {
251
252 G4double x4 = x2*x2;
253 G4double x5 = x4*x1; //this variable usage can probably be optimized
254 cs = x1*((*(fParamLow[Z]))[idx] + x1*(*(fParamLow[Z]))[idx+1]
255 + x2*(*(fParamLow[Z]))[idx+2] + x3*(*(fParamLow[Z]))[idx+3]
256 + x4*(*(fParamLow[Z]))[idx+4]+ x5*(*(fParamLow[Z]))[idx+5]);
257
258 }
259 // Tabulated values above k-shell ionization energy
260 else if(energy >= (*(fParamHigh[Z]))[1]) {
261 cs = x3*(fCrossSection[Z])->Value(energy);
262 }
263 // Tabulated values below k-shell ionization energy
264 else
265 {
266 cs = x3*(fCrossSectionLE[Z])->Value(energy);
267 }
268 if (verboseLevel > 1) {
269 G4cout << "G4LivermorePhotoElectricModel: E(keV)= " << energy/keV
270 << " Z= " << Z << " cross(barn)= " << cs/barn << G4endl;
271 }
272 return cs;
273}

277void
278G4LivermorePhotoElectricModel::SampleSecondaries(
279 std::vector<G4DynamicParticle*>* fvect,
280 const G4MaterialCutsCouple* couple,
281 const G4DynamicParticle* aDynamicGamma,
282 G4double, G4double)
283{
284 G4double gammaEnergy = aDynamicGamma->GetKineticEnergy();
285 if (verboseLevel > 3) {
286 G4cout << "G4LivermorePhotoElectricModel::SampleSecondaries() Egamma(keV)= "
287 << gammaEnergy/keV << G4endl;
288 }
289
290
291 // kill incident photon
292 fParticleChange->ProposeTrackStatus(fStopAndKill);
293 fParticleChange->SetProposedKineticEnergy(0.);
294
295 // low-energy photo-effect in water - full absorption
296 const G4Material* material = couple->GetMaterial();
297 if(fWater && (material == fWater ||
298 material->GetBaseMaterial() == fWater)) {
299 if(gammaEnergy <= fWaterEnergyLimit) {
300 fParticleChange->ProposeLocalEnergyDeposit(gammaEnergy);
301 return;
302 }
303 }
304
305 // Returns the normalized direction of the momentum
306 G4ThreeVector photonDirection = aDynamicGamma->GetMomentumDirection();
307
308 // Select randomly one element in the current material
309 //G4cout << "Select random atom Egamma(keV)= " << gammaEnergy/keV << G4endl;
310 const G4Element* elm = SelectRandomAtom(material, theGamma, gammaEnergy);
311 G4int Z = elm->GetZasInt();
312
313 // Select the ionised shell in the current atom according to shell
314 // cross sections
315 // G4cout << "Select random shell Z= " << Z << G4endl;
316
317 if(Z >= maxZ) { Z = maxZ-1; }
318
319 // element was not initialised gamma should be absorbed
320 if(!fCrossSectionLE[Z] && !fCrossSection[Z]) {
321 fParticleChange->ProposeLocalEnergyDeposit(gammaEnergy);
322 return;
323 }
324
325 // SAMPLING OF THE SHELL INDEX
326 size_t shellIdx = 0;
327 size_t nn = fNShellsUsed[Z];
328 if(nn > 1)
329 {
330 if(gammaEnergy >= (*(fParamHigh[Z]))[0])
331 {
332
333 G4double x1 = 1.0/gammaEnergy;
334 G4double x2 = x1*x1;
335 G4double x3 = x2*x1;
336 G4double x4 = x3*x1;
337 G4double x5 = x4*x1;
338 G4int idx = nn*7 - 5;
339 // when do sampling common factors are not taken into account
340 // so cross section is not real
341
342 G4double rand=G4UniformRand();
343 G4double cs0 = rand*( (*(fParamHigh[Z]))[idx]
344 + x1*(*(fParamHigh[Z]))[idx+1]
345 + x2*(*(fParamHigh[Z]))[idx+2]
346 + x3*(*(fParamHigh[Z]))[idx+3]
347 + x4*(*(fParamHigh[Z]))[idx+4]
348 + x5*(*(fParamHigh[Z]))[idx+5]);
349
350 for(shellIdx=0; shellIdx<nn; ++shellIdx)
351 {
352 idx = shellIdx*7 + 2;
353 if(gammaEnergy > (*(fParamHigh[Z]))[idx-1])
354 {
355 G4double cs =
356 (*(fParamHigh[Z]))[idx]
357 + x1*(*(fParamHigh[Z]))[idx+1]
358 + x2*(*(fParamHigh[Z]))[idx+2]
359 + x3*(*(fParamHigh[Z]))[idx+3]
360 + x4*(*(fParamHigh[Z]))[idx+4]
361 + x5*(*(fParamHigh[Z]))[idx+5];
362
363 if(cs >= cs0) { break; }
364 }
365 }
366 if(shellIdx >= nn) { shellIdx = nn-1; }
367
368 }
369 else if(gammaEnergy >= (*(fParamLow[Z]))[0])
370 {
371 G4double x1 = 1.0/gammaEnergy;
372 G4double x2 = x1*x1;
373 G4double x3 = x2*x1;
374 G4double x4 = x3*x1;
375 G4double x5 = x4*x1;
376 G4int idx = nn*7 - 5;
377 // when do sampling common factors are not taken into account
378 // so cross section is not real
379 G4double cs0 = G4UniformRand()*((*(fParamLow[Z]))[idx]
380 + x1*(*(fParamLow[Z]))[idx+1]
381 + x2*(*(fParamLow[Z]))[idx+2]
382 + x3*(*(fParamLow[Z]))[idx+3]
383 + x4*(*(fParamLow[Z]))[idx+4]
384 + x5*(*(fParamLow[Z]))[idx+5]);
385 for(shellIdx=0; shellIdx<nn; ++shellIdx)
386 {
387 idx = shellIdx*7 + 2;
388 if(gammaEnergy > (*(fParamLow[Z]))[idx-1])
389 {
390 G4double cs = (*(fParamLow[Z]))[idx] + x1*(*(fParamLow[Z]))[idx+1]
391 + x2*(*(fParamLow[Z]))[idx+2] + x3*(*(fParamLow[Z]))[idx+3]
392 + x4*(*(fParamLow[Z]))[idx+4]+ x5*(*(fParamLow[Z]))[idx+5];
393 if(cs >= cs0) { break; }
394 }
395 }
396 if(shellIdx >= nn) {shellIdx = nn-1;}
397 }
398 else
399 {
400 // when do sampling common factors are not taken into account
401 // so cross section is not real
402 G4double cs = G4UniformRand();
403
404 if(gammaEnergy >= (*(fParamHigh[Z]))[1]) {
405 //above K-shell binding energy
406 cs*= (fCrossSection[Z])->Value(gammaEnergy);
407 }
408 else
409 {
410 //below K-shell binding energy
411 cs *= (fCrossSectionLE[Z])->Value(gammaEnergy);
412 }
413
414 for(size_t j=0; j<nn; ++j) {
415
416 shellIdx = (size_t)fShellCrossSection->GetComponentID(Z, j);
417 if(gammaEnergy > (*(fParamLow[Z]))[7*shellIdx+1]) {
418 cs -= fShellCrossSection->GetValueForComponent(Z, j, gammaEnergy);
419 }
420 if(cs <= 0.0 || j+1 == nn) {break;}
421 }
422 }
423 }
424 // END: SAMPLING OF THE SHELL
425
426
427 G4double bindingEnergy = (*(fParamHigh[Z]))[shellIdx*7 + 1];
428 //G4cout << "Z= " << Z << " shellIdx= " << shellIdx
429 // << " nShells= " << fNShells[Z]
430 // << " Ebind(keV)= " << bindingEnergy/keV
431 // << " Egamma(keV)= " << gammaEnergy/keV << G4endl;
432
433 const G4AtomicShell* shell = 0;
434
435 // no de-excitation from the last shell
436 if(fDeexcitationActive && shellIdx + 1 < nn) {
437 G4AtomicShellEnumerator as = G4AtomicShellEnumerator(shellIdx);
438 shell = fAtomDeexcitation->GetAtomicShell(Z, as);
439 }
440
441 // If binding energy of the selected shell is larger than photon energy
442 // do not generate secondaries
443 if(gammaEnergy < bindingEnergy) {
444 fParticleChange->ProposeLocalEnergyDeposit(gammaEnergy);
445
446 return;
447 }
448
449 // Primary outcoming electron
450 G4double eKineticEnergy = gammaEnergy - bindingEnergy;
451 G4double edep = bindingEnergy;
452
453 // Calculate direction of the photoelectron
454 G4ThreeVector electronDirection =
455 GetAngularDistribution()->SampleDirection(aDynamicGamma,
456 eKineticEnergy,
457 shellIdx,
458 couple->GetMaterial());
459
460 // The electron is created
461 G4DynamicParticle* electron = new G4DynamicParticle (theElectron,
462 electronDirection,
463 eKineticEnergy);
464 fvect->push_back(electron);
465
466 // Sample deexcitation
467 if(shell) {
468 G4int index = couple->GetIndex();
469 if(fAtomDeexcitation->CheckDeexcitationActiveRegion(index)) {
470 G4int nbefore = fvect->size();
471
472 fAtomDeexcitation->GenerateParticles(fvect, shell, Z, index);
473 G4int nafter = fvect->size();
474 if(nafter > nbefore) {
475 G4double esec = 0.0;
476 for (G4int j=nbefore; j<nafter; ++j) {
477
478 G4double e = ((*fvect)[j])->GetKineticEnergy();
479 if(esec + e > edep) {
480 // correct energy in order to have energy balance
481 e = edep - esec;
482 ((*fvect)[j])->SetKineticEnergy(e);
483 esec += e;
484 // delete the rest of secondaries (should not happens)
485 for (G4int jj=nafter-1; jj>j; --jj) {
486 delete (*fvect)[jj];
487 fvect->pop_back();
488 }
489 break;
490 }
491 esec += e;
492 }
493 edep -= esec;
494 }
495 }
496 }
497 // energy balance - excitation energy left
498 if(edep > 0.0) {
499 fParticleChange->ProposeLocalEnergyDeposit(edep);
500 }
501}

505void
506G4LivermorePhotoElectricModel::ReadData(G4int Z, const char* path)
507{
508 if (verboseLevel > 1)
509 {
510 G4cout << "Calling ReadData() of G4LivermorePhotoElectricModel"
511 << G4endl;
512 }
513
514 if(fCrossSection[Z] || fCrossSectionLE[Z]) { return; }
515
516 const char* datadir = path;
517
518 if(!datadir)
519 {
520 datadir = getenv("G4LEDATA");
521 if(!datadir)
522 {
523 G4Exception("G4LivermorePhotoElectricModel::ReadData()",
524 "em0006",FatalException,
525 "Environment variable G4LEDATA not defined");
526 return;
527 }
528 }
529
530 // spline for photoeffect total x-section above K-shell
531 // but below the parameterized ones
532 fCrossSection[Z] = new G4LPhysicsFreeVector();
533 fCrossSection[Z]->SetSpline(true);
534 std::ostringstream ost;
535 ost << datadir << "/livermore/phot_epics2014/pe-cs-" << Z <<".dat";
536 std::ifstream fin(ost.str().c_str());
537 if( !fin.is_open()) {
538 G4ExceptionDescription ed;
539 ed << "G4LivermorePhotoElectricModel data file <" << ost.str().c_str()
540 << "> is not opened!" << G4endl;
541 G4Exception("G4LivermorePhotoElectricModel::ReadData()",
542 "em0003",FatalException,
543 ed,"G4LEDATA version should be G4EMLOW7.2 or later.");
544 return;
545 } else {
546 if(verboseLevel > 3) { G4cout << "File " << ost.str().c_str()
547 << " is opened by G4LivermorePhotoElectricModel" << G4endl;}
548 fCrossSection[Z]->Retrieve(fin, true);
549 fCrossSection[Z]->ScaleVector(MeV, barn);
550 fin.close();
551 }
552
553
554 // read high-energy fit parameters
555 fParamHigh[Z] = new std::vector<G4double>;
556 G4int n1 = 0;
557 G4int n2 = 0;
558 G4double x;
559 std::ostringstream ost1;
560 ost1 << datadir << "/livermore/phot_epics2014/pe-high-" << Z <<".dat";
561 std::ifstream fin1(ost1.str().c_str());
562 if( !fin1.is_open()) {
563 G4ExceptionDescription ed;
564 ed << "G4LivermorePhotoElectricModel data file <" << ost1.str().c_str()
565 << "> is not opened!" << G4endl;
566 G4Exception("G4LivermorePhotoElectricModel::ReadData()",
567 "em0003",FatalException,
568 ed,"G4LEDATA version should be G4EMLOW7.2 or later.");
569 return;
570 } else {
571 if(verboseLevel > 3) {
572 G4cout << "File " << ost1.str().c_str()
573 << " is opened by G4LivermorePhotoElectricModel" << G4endl;
574 }
575 fin1 >> n1;
576 if(fin1.fail()) { return; }
577 if(0 > n1 || n1 >= INT_MAX) { n1 = 0; }
578
579 fin1 >> n2;
580 if(fin1.fail()) { return; }
581 if(0 > n2 || n2 >= INT_MAX) { n2 = 0; }
582
583 fin1 >> x;
584 if(fin1.fail()) { return; }
585
586 fNShells[Z] = n1;
587 fParamHigh[Z]->reserve(7*n1+1);
588 fParamHigh[Z]->push_back(x*MeV);
589 for(G4int i=0; i<n1; ++i) {
590 for(G4int j=0; j<7; ++j) {
591 fin1 >> x;
592 if(0 == j) { x *= MeV; }
593 else { x *= barn; }
594 fParamHigh[Z]->push_back(x);
595 }
596 }
597 fin1.close();
598 }
599
600 // read low-energy fit parameters
601 fParamLow[Z] = new std::vector<G4double>;
602 G4int n1_low = 0;
603 G4int n2_low = 0;
604 G4double x_low;
605 std::ostringstream ost1_low;
606 ost1_low << datadir << "/livermore/phot_epics2014/pe-low-" << Z <<".dat";
607 std::ifstream fin1_low(ost1_low.str().c_str());
608 if( !fin1_low.is_open()) {
609 G4ExceptionDescription ed;
610 ed << "G4LivermorePhotoElectricModel data file <" << ost1_low.str().c_str()
611 << "> is not opened!" << G4endl;
612 G4Exception("G4LivermorePhotoElectricModel::ReadData()",
613 "em0003",FatalException,
614 ed,"G4LEDATA version should be G4EMLOW7.2 or later.");
615 return;
616 } else {
617 if(verboseLevel > 3) {
618 G4cout << "File " << ost1_low.str().c_str()
619 << " is opened by G4LivermorePhotoElectricModel" << G4endl;
620 }
621 fin1_low >> n1_low;
622 if(fin1_low.fail()) { return; }
623 if(0 > n1_low || n1_low >= INT_MAX) { n1_low = 0; }
624
625 fin1_low >> n2_low;
626 if(fin1_low.fail()) { return; }
627 if(0 > n2_low || n2_low >= INT_MAX) { n2_low = 0; }
628
629 fin1_low >> x_low;
630 if(fin1_low.fail()) { return; }
631
632 fNShells[Z] = n1_low;
633 fParamLow[Z]->reserve(7*n1_low+1);
634 fParamLow[Z]->push_back(x_low*MeV);
635 for(G4int i=0; i<n1_low; ++i) {
636 for(G4int j=0; j<7; ++j) {
637 fin1_low >> x_low;
638 if(0 == j) { x_low *= MeV; }
639 else { x_low *= barn; }
640 fParamLow[Z]->push_back(x_low);
641 }
642 }
643 fin1_low.close();
644 }
645
646
647 // there is a possibility to use only main shells
648 if(nShellLimit < n2) { n2 = nShellLimit; }
649 fShellCrossSection->InitialiseForComponent(Z, n2);//number of shells
650 fNShellsUsed[Z] = n2;
651
652
653
654 if(1 < n2) {
655 std::ostringstream ost2;
656 ost2 << datadir << "/livermore/phot_epics2014/pe-ss-cs-" << Z <<".dat";
657 std::ifstream fin2(ost2.str().c_str());
658 if( !fin2.is_open()) {
659 G4ExceptionDescription ed;
660 ed << "G4LivermorePhotoElectricModel data file <" << ost2.str().c_str()
661 << "> is not opened!" << G4endl;
662 G4Exception("G4LivermorePhotoElectricModel::ReadData()",
663 "em0003",FatalException,
664 ed,"G4LEDATA version should be G4EMLOW7.2 or later.");
665 return;
666 } else {
667 if(verboseLevel > 3) {
668 G4cout << "File " << ost2.str().c_str()
669 << " is opened by G4LivermorePhotoElectricModel" << G4endl;
670 }
671
672 G4int n3, n4;
673 G4double y;
674 for(G4int i=0; i<n2; ++i) {
675 fin2 >> x >> y >> n3 >> n4;
676 G4LPhysicsFreeVector* v = new G4LPhysicsFreeVector(n3, x, y);
677 for(G4int j=0; j<n3; ++j) {
678 fin2 >> x >> y;
679 v->PutValues(j, x*MeV, y*barn);
680 }
681 fShellCrossSection->AddComponent(Z, n4, v);
682 }
683 fin2.close();
684 }
685 }
686
687 // no spline for photoeffect total x-section below K-shell
688 if(1 < fNShells[Z]) {
689 fCrossSectionLE[Z] = new G4LPhysicsFreeVector();
690 std::ostringstream ost3;
691 ost3 << datadir << "/livermore/phot_epics2014/pe-le-cs-" << Z <<".dat";
692 std::ifstream fin3(ost3.str().c_str());
693 if( !fin3.is_open()) {
694 G4ExceptionDescription ed;
695 ed << "G4LivermorePhotoElectricModel data file <" << ost3.str().c_str()
696 << "> is not opened!" << G4endl;
697 G4Exception("G4LivermorePhotoElectricModel::ReadData()",
698 "em0003",FatalException,
699 ed,"G4LEDATA version should be G4EMLOW7.2 or later.");
700 return;
701 } else {
702 if(verboseLevel > 3) {
703 G4cout << "File " << ost3.str().c_str()
704 << " is opened by G4LivermorePhotoElectricModel" << G4endl;
705 }
706 fCrossSectionLE[Z]->Retrieve(fin3, true);
707 fCrossSectionLE[Z]->ScaleVector(MeV, barn);
708 fin3.close();
709 }
710 }
711}

715G4double G4LivermorePhotoElectricModel::GetBindingEnergy(G4double ZZ, G4int shell)
716{
717 G4int Z = G4lrint(ZZ);
718 //Control on Z
719 if(Z < 1 || Z >= maxZ) { return -1;} //If Z is out of the supported return 0
720 //If necessary load data for Z
721 InitialiseForElement(0, Z);
722 if(!fCrossSectionLE[Z] && !fCrossSection[Z]) { return -1; } //If there are not data for that Z return -1
723 if(shell < 0 || shell >= fNShellsUsed[Z]) { return -1;}
724 if(Z>2)
725 return fShellCrossSection->GetComponentDataByIndex(Z, shell)->Energy(0);
726 else
727 return fCrossSection[Z]->Energy(0);
728
729}
