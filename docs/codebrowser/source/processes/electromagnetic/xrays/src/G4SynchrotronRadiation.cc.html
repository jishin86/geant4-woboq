56G4SynchrotronRadiation::G4SynchrotronRadiation(const G4String& processName,
57 G4ProcessType type):G4VDiscreteProcess (processName, type),
58 theGamma (G4Gamma::Gamma() )
59{
60 G4TransportationManager* transportMgr =
61 G4TransportationManager::GetTransportationManager();
62
63 fFieldPropagator = transportMgr->GetPropagatorInField();
64
65 SetProcessSubType(fSynchrotronRadiation);
66 verboseLevel = 1;
67 FirstTime = true;
68 FirstTime1 = true;
69 genAngle = nullptr;
70 SetAngularGenerator(new G4DipBustGenerator());
71}

107G4double
108G4SynchrotronRadiation::GetMeanFreePath(const G4Track& trackData,
109 G4double,
110 G4ForceCondition* condition)
111{
112 // gives the MeanFreePath in GEANT4 internal units
113 G4double MeanFreePath = DBL_MAX;
114
115 const G4DynamicParticle* aDynamicParticle = trackData.GetDynamicParticle();
116
117 *condition = NotForced;
118
119 G4double gamma = aDynamicParticle->GetTotalEnergy()/
120 aDynamicParticle->GetMass();
121
122 G4double particleCharge = aDynamicParticle->GetDefinition()->GetPDGCharge();
123
124 if ( gamma < 1.0e3 || 0.0 == particleCharge) { MeanFreePath = DBL_MAX; }
125 else
126 {
127
128 G4ThreeVector FieldValue;
129 const G4Field* pField = nullptr;
130 G4bool fieldExertsForce = false;
131
132
133 G4FieldManager* fieldMgr =
134 fFieldPropagator->FindAndSetFieldManager(trackData.GetVolume());
135
136 if ( fieldMgr != nullptr )
137 {
138 // If the field manager has no field, there is no field !
139
140 fieldExertsForce = ( fieldMgr->GetDetectorField() != nullptr );
141 }
142
143 if ( fieldExertsForce )
144 {
145 pField = fieldMgr->GetDetectorField();
146 G4ThreeVector globPosition = trackData.GetPosition();
147
148 G4double globPosVec[4], FieldValueVec[6];
149
150 globPosVec[0] = globPosition.x();
151 globPosVec[1] = globPosition.y();
152 globPosVec[2] = globPosition.z();
153 globPosVec[3] = trackData.GetGlobalTime();
154
155 pField->GetFieldValue( globPosVec, FieldValueVec );
156
157 FieldValue = G4ThreeVector( FieldValueVec[0],
158 FieldValueVec[1],
159 FieldValueVec[2] );
160
161 G4ThreeVector unitMomentum = aDynamicParticle->GetMomentumDirection();
162 G4ThreeVector unitMcrossB = FieldValue.cross(unitMomentum);
163 G4double perpB = unitMcrossB.mag();
164
165 static const G4double fLambdaConst = std::sqrt(3.0)*eplus/
166 (2.5*fine_structure_const*c_light);
167
168 if( perpB > 0.0 )
169 {
170 MeanFreePath =
171 fLambdaConst*aDynamicParticle->GetDefinition()->GetPDGMass()
172 /(perpB*particleCharge*particleCharge);
173 }
174 if(verboseLevel > 0 && FirstTime)
175 {
176 G4cout << "G4SynchrotronRadiation::GetMeanFreePath "
177 << " for particle "
178 << aDynamicParticle->GetDefinition()->GetParticleName()
179 << ":" << '\n' //hbunew
180 << " MeanFreePath = " << G4BestUnit(MeanFreePath, "Length")
181 << G4endl;
182 if(verboseLevel > 1)
183 {
184 G4ThreeVector pvec = aDynamicParticle->GetMomentum();
185 G4double Btot = FieldValue.getR();
186 G4double ptot = pvec.getR();
187 G4double rho = ptot / (MeV * c_light * Btot );
188 // full bending radius
189 G4double Theta=unitMomentum.theta(FieldValue);
190 // angle between particle and field
191 G4cout << " B = " << Btot/tesla << " Tesla"
192 << " perpB = " << perpB/tesla << " Tesla"
193 << " Theta = " << Theta << " std::sin(Theta)="
194 << std::sin(Theta) << '\n'
195 << " ptot = " << G4BestUnit(ptot,"Energy")
196 << " rho = " << G4BestUnit(rho,"Length")
197 << G4endl;
198 }
199 FirstTime=false;
200 }
201 }
202 }
203 return MeanFreePath;
204}

210G4VParticleChange*
211G4SynchrotronRadiation::PostStepDoIt(const G4Track& trackData,
212 const G4Step& stepData )
213
214{
215 aParticleChange.Initialize(trackData);
216
217 const G4DynamicParticle* aDynamicParticle = trackData.GetDynamicParticle();
218
219 G4double gamma = aDynamicParticle->GetTotalEnergy()/
220 (aDynamicParticle->GetDefinition()->GetPDGMass());
221
222 G4double particleCharge = aDynamicParticle->GetDefinition()->GetPDGCharge();
223 if(gamma <= 1.0e3 || 0.0 == particleCharge)
224 {
225 return G4VDiscreteProcess::PostStepDoIt(trackData,stepData);
226 }
227
228 G4ThreeVector FieldValue;
229 const G4Field* pField = nullptr;
230
231 G4bool fieldExertsForce = false;
232 G4FieldManager* fieldMgr =
233 fFieldPropagator->FindAndSetFieldManager(trackData.GetVolume());
234
235 if ( fieldMgr != nullptr )
236 {
237 // If the field manager has no field, there is no field !
238 fieldExertsForce = ( fieldMgr->GetDetectorField() != nullptr );
239 }
240
241 if ( fieldExertsForce )
242 {
243 pField = fieldMgr->GetDetectorField();
244 G4ThreeVector globPosition = trackData.GetPosition();
245 G4double globPosVec[4], FieldValueVec[6];
246 globPosVec[0] = globPosition.x();
247 globPosVec[1] = globPosition.y();
248 globPosVec[2] = globPosition.z();
249 globPosVec[3] = trackData.GetGlobalTime();
250
251 pField->GetFieldValue( globPosVec, FieldValueVec );
252 FieldValue = G4ThreeVector( FieldValueVec[0],
253 FieldValueVec[1],
254 FieldValueVec[2] );
255
256 G4ThreeVector unitMomentum = aDynamicParticle->GetMomentumDirection();
257 G4ThreeVector unitMcrossB = FieldValue.cross(unitMomentum);
258 G4double perpB = unitMcrossB.mag();
259 if(perpB > 0.0)
260 {
261 // M-C of synchrotron photon energy
262
263 G4double energyOfSR =
264 GetRandomEnergySR(gamma,perpB,
265 aDynamicParticle->GetDefinition()->GetPDGMass());
266
267 // check against insufficient energy
268
269 if( energyOfSR <= 0.0 )
270 {
271 return G4VDiscreteProcess::PostStepDoIt(trackData,stepData);
272 }
273 G4double kineticEnergy = aDynamicParticle->GetKineticEnergy();
274 G4ThreeVector gammaDirection =
275 genAngle->SampleDirection(aDynamicParticle,
276 energyOfSR, 1, 0);
277
278 G4ThreeVector gammaPolarization = FieldValue.cross(gammaDirection);
279 gammaPolarization = gammaPolarization.unit();
280
281 // create G4DynamicParticle object for the SR photon
282
283 G4DynamicParticle* aGamma= new G4DynamicParticle ( theGamma,
284 gammaDirection,
285 energyOfSR );
286 aGamma->SetPolarization( gammaPolarization.x(),
287 gammaPolarization.y(),
288 gammaPolarization.z() );
289
290 aParticleChange.SetNumberOfSecondaries(1);
291 aParticleChange.AddSecondary(aGamma);
292
293 // Update the incident particle
294
295 G4double newKinEnergy = kineticEnergy - energyOfSR;
296
297 if (newKinEnergy > 0.)
298 {
299 aParticleChange.ProposeEnergy( newKinEnergy );
300 }
301 else
302 {
303 aParticleChange.ProposeEnergy( 0. );
304 }
305 }
306 }
307 return G4VDiscreteProcess::PostStepDoIt(trackData,stepData);
308}

373G4double G4SynchrotronRadiation::GetRandomEnergySR(
374 G4double gamma, G4double perpB, G4double mass_c2)
375{
376
377 static const G4double fEnergyConst = 1.5*c_light*c_light*eplus*hbar_Planck;
378 G4double Ecr=fEnergyConst*gamma*gamma*perpB/mass_c2;
379
380 if(verboseLevel > 0 && FirstTime1)
381 {
382 // mean and rms of photon energy
383 G4double Emean=8./(15.*std::sqrt(3.))*Ecr;
384 G4double E_rms=std::sqrt(211./675.)*Ecr;
385 G4int prec = G4cout.precision();
386 G4cout << "G4SynchrotronRadiation::GetRandomEnergySR :" << '\n'
387 << std::setprecision(4)
388 << " Ecr = " << G4BestUnit(Ecr,"Energy") << '\n'
389 << " Emean = " << G4BestUnit(Emean,"Energy") << '\n'
390 << " E_rms = " << G4BestUnit(E_rms,"Energy") << G4endl;
391 FirstTime1=false;
392 G4cout.precision(prec);
393 }
394
395 G4double energySR=Ecr*InvSynFracInt(G4UniformRand());
396 return energySR;
397}
