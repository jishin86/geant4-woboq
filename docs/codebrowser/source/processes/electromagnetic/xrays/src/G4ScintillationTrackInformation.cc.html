40const G4String G4ScintillationTrackInformation::BaseType = "G4ScintillationTrackInformation";
41
42G4ScintillationTrackInformation::G4ScintillationTrackInformation(const G4ScintillationType& aType)
43: G4VUserTrackInformation(BaseType),
44scintillationType(aType)
45{
46}

48G4ScintillationTrackInformation::~G4ScintillationTrackInformation()

50}

52G4ScintillationTrackInformation::G4ScintillationTrackInformation(const G4ScintillationTrackInformation& right)

56}

58G4ScintillationTrackInformation& G4ScintillationTrackInformation::operator=(const G4ScintillationTrackInformation& right)
59{
60 G4VUserTrackInformation::operator=(right);
61 this->scintillationType = right.scintillationType;
62 return *this;
63}

65void G4ScintillationTrackInformation::Print() const
66{
67 G4cout << "The user track information is a scintillation" << G4endl;
68}
69
70G4bool G4ScintillationTrackInformation::IsScintillationTrackInformation(const G4VUserTrackInformation* const aTI)
71{
72 G4bool isSTI = (aTI && aTI->GetType() == BaseType.c_str());
73 return isSTI;
74}
75
76G4ScintillationTrackInformation* G4ScintillationTrackInformation::Cast(const G4VUserTrackInformation* const aTI)
77{
78 G4ScintillationTrackInformation* STI = nullptr;
79 if(IsScintillationTrackInformation(aTI))
80 {
81 // No change will be done to the pointer and to the pointed data
82 G4VUserTrackInformation* temp = const_cast<G4VUserTrackInformation*>(aTI);
83 STI = dynamic_cast<G4ScintillationTrackInformation*>(temp);
84 }
85 return STI;
86}
