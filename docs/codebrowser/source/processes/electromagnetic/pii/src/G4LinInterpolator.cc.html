53G4double G4LinInterpolator::Calculate(G4double x, G4int bin,
54 const G4DataVector& points,
55 const G4DataVector& data) const
56{
57 G4int nBins = data.size() - 1;
58 G4double value = 0.;
59 if (x < points[0])
60 {
61 value = 0.;
62 }
63 else if (bin < nBins)
64 {
65 G4double e1 = points[bin];
66 G4double e2 = points[bin+1];
67 G4double d1 = data[bin];
68 G4double d2 = data[bin+1];
69 value = d1 + (d2 - d1)*(x - e1)/(e2 - e1);
70 }
71 else
72 {
73 value = data[nBins];
74 }
75 return value;
76}
