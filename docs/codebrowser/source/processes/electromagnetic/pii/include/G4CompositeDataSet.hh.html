38// Composite data set
39// Applies a Composite design pattern for data library management
40
41// -------------------------------------------------------------------
42
43#ifndef G4COMPOSITEDATASET_HH
44#define G4COMPOSITEDATASET_HH 1
45
46#include <vector>
47#include <CLHEP/Units/SystemOfUnits.h>
48
49#include "globals.hh"
50#include "G4IDataSet.hh"
51
52class G4IInterpolator;
53
54class G4CompositeDataSet : public G4IDataSet
55{
56public:
57 G4CompositeDataSet(G4IInterpolator* argAlgorithm,
58 G4double eUnit=CLHEP::MeV,
59 G4double dataUnit=CLHEP::barn,
60 G4int zMin=1,
61 G4int zMax=99);
62
63 virtual ~G4CompositeDataSet();
64
65 virtual G4double FindValue(G4double x, G4int componentId=0) const;
66
67 virtual void PrintData(void) const;
68
69 virtual const G4IDataSet* GetComponent(G4int componentId) const
70 { return components[componentId]; }
71
72 virtual void AddComponent(G4IDataSet* dataSet)
73 { components.push_back(dataSet); }
74
75 virtual size_t NumberOfComponents() const
76 { return components.size(); }
77
78 virtual const G4DataVector& GetEnergies(G4int componentId) const
79 { return GetComponent(componentId)->GetEnergies(0); }
80
81 virtual const G4DataVector& GetData(G4int componentId) const
82 { return GetComponent(componentId)->GetData(0); }
83
84 virtual void SetEnergiesData(G4DataVector* x, G4DataVector* data, G4int componentId);
85
86 virtual G4bool LoadData(const G4String& fileName);
87 virtual G4bool SaveData(const G4String& fileName) const;
88
89 virtual G4double RandomSelect(G4int componentId) const;
90
91private:
92
93 void CleanUpComponents(void);
94
95 // Hide copy constructor and assignment operator
96 G4CompositeDataSet();
97 G4CompositeDataSet(const G4CompositeDataSet& copy);
98 G4CompositeDataSet& operator=(const G4CompositeDataSet& right);
99
100 std::vector<G4IDataSet*> components; // Owned pointers
101
102 G4IInterpolator* algorithm; // Owned pointer
103
104 G4double unitEnergies;
105 G4double unitData;
106
107 G4int minZ;
108 G4int maxZ;
109};
110#endif /* G4COMPOSITEDATASET_HH */
