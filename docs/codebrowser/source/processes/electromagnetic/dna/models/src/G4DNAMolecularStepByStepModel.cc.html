39G4DNAMolecularStepByStepModel::G4DNAMolecularStepByStepModel(const G4String& name) :
40 G4VITStepModel(name),
41 fMolecularReactionTable(reference_cast<const G4DNAMolecularReactionTable*>(fpReactionTable))
42{
43 fpTimeStepper = new G4DNAMoleculeEncounterStepper();
44 fpReactionProcess = new G4DNAMolecularReaction();
45
46 fType1 = G4Molecule::ITType();
47 fType2 = G4Molecule::ITType();
48 fReactionModel = 0;
49}
50
51G4DNAMolecularStepByStepModel::~G4DNAMolecularStepByStepModel()
52{
53 if(fReactionModel) delete fReactionModel;
54}
55
56G4DNAMolecularStepByStepModel& G4DNAMolecularStepByStepModel::operator=(const G4DNAMolecularStepByStepModel& right)
57{
58 G4ExceptionDescription exceptionDescription("Use copy constructor rather than assignement operator.");
59 G4Exception("G4DNAMolecularStepByStepModel::operator=(const G4DNAMolecularStepByStepModel&)",
60 "G4DNAMolecularStepByStepModel001",
61 FatalErrorInArgument,
62 exceptionDescription);
63
64 if(&right == this) return *this;
65 return *this; // avoid warnings
66}
67
68G4DNAMolecularStepByStepModel::G4DNAMolecularStepByStepModel(const G4DNAMolecularStepByStepModel& right) :
69 G4VITStepModel(right),
70 fMolecularReactionTable(reference_cast<const G4DNAMolecularReactionTable*>(fpReactionTable))
71{
72 fpReactionTable = right.fpReactionTable;
73 if(right.fReactionModel)
74 {
75 fReactionModel = right.fReactionModel->Clone();
76 ((G4DNAMolecularReaction*) fpReactionProcess)->SetReactionModel(fReactionModel);
77 ((G4DNAMoleculeEncounterStepper*) fpTimeStepper)->SetReactionModel(fReactionModel);
78 }
79 else fReactionModel = 0;
80}
81
82void G4DNAMolecularStepByStepModel::Initialize()
83{
84 if(fpReactionTable == 0)
85 {
86 SetReactionTable(G4DNAMolecularReactionTable::GetReactionTable());
87 }
88
89 if(fReactionModel == 0)
90 {
91 fReactionModel = new G4DNASmoluchowskiReactionModel();
92 }
93
94 fReactionModel->SetReactionTable((const G4DNAMolecularReactionTable*) fpReactionTable);
95
96 ((G4DNAMolecularReaction*) fpReactionProcess)->SetReactionModel(fReactionModel);
97 ((G4DNAMoleculeEncounterStepper*) fpTimeStepper)->SetReactionModel(fReactionModel);
98
99 G4VITStepModel::Initialize();
100}
