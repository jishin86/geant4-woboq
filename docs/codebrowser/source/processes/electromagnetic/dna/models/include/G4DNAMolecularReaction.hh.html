47#ifndef G4MOLECULARREACTION_H
48#define G4MOLECULARREACTION_H
49
50#include <G4VITReactionProcess.hh>
51
52class G4DNAMolecularReactionTable;
53class G4VDNAReactionModel;
54
55/**
56 * G4DNAMolecularReaction is the reaction process
57 * used in G4DNAMolecularStepByStepModel between
58 * two molecules.
59 * After the global track steps, it test whether
60 * the molecules can react. If so, the reaction is made.
61 */
62
63class G4DNAMolecularReaction : public G4VITReactionProcess
64{
65 public:
66 /** Default constructor */
67 G4DNAMolecularReaction();
68 /** Default destructor */
69 virtual ~G4DNAMolecularReaction();
70 /** Copy constructor
71 * \param other Object to copy from
72 */
73 G4DNAMolecularReaction(const G4DNAMolecularReaction& other);
74 /** Assignment operator
75 * \param other Object to assign from
76 * \return A reference to this
77 */
78 G4DNAMolecularReaction& operator=(const G4DNAMolecularReaction& other);
79
80 G4IT_ADD_CLONE(G4VITReactionProcess, G4DNAMolecularReaction)
81
82 /** Given two tracks, it tells you whether they can react
83 */
84 virtual G4bool TestReactibility(const G4Track&,
85 const G4Track&,
86 const double currentStepTime,
87 const double previousStepTime,
88 bool userStepTimeLimit) /*const*/ ;
89
90 /** Will generate the products of the two given tracks
91 */
92 virtual G4ITReactionChange* MakeReaction(const G4Track&, const G4Track&) ;
93
94 inline void SetReactionModel(G4VDNAReactionModel*);
95 inline void SetReactionTable(const G4DNAMolecularReactionTable*);
96
97 inline void SetVerbose(int);
98 // 1 : only when make reaction is called
99 // 2 : both make reaction + test reactibility are called
100
101 protected:
102 const G4DNAMolecularReactionTable*& fMolReactionTable;
103 G4VDNAReactionModel* fReactionModel;
104 G4int fVerbose;
105 G4double fReactionRadius ;
106 G4double fDistance;
107 private:
108};
109
110inline void G4DNAMolecularReaction::SetReactionModel(G4VDNAReactionModel* model)
111{
112 fReactionModel = model;
113}
114
115inline void G4DNAMolecularReaction::SetVerbose(int verb)
116{
117 fVerbose = verb;
118}
119
120#endif // G4MOLECULARREACTION_H
