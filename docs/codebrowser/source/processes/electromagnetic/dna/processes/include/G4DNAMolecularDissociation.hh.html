47#ifndef G4MOLECULARDECAYPROCESS_HH
48#define G4MOLECULARDECAYPROCESS_HH
49
50#include "G4VITRestProcess.hh"
51#include "G4VITRestDiscreteProcess.hh"
52#include <map>
53
54class G4ParticleChange;
55class G4VMolecularDecayDisplacer;
56class G4MoleculeDefinition;
57
58/**
59 * G4DNAMolecularDissociation should be called only for molecules.
60 * It will dissociate the molecules using the decay associated to
61 * this molecule and if a displacement scheme has been registered,
62 * it will place the products to the expected position.
63 */
64
65class G4DNAMolecularDissociation: public G4VITRestDiscreteProcess //G4VITRestProcess
66{
67public:
68 G4DNAMolecularDissociation(const G4String& processName = "DNAMolecularDecay",
69 G4ProcessType type = fDecay);
70
71 virtual ~G4DNAMolecularDissociation();
72
73 G4IT_ADD_CLONE(G4VITProcess, G4DNAMolecularDissociation)
74
75 virtual G4bool IsApplicable(const G4ParticleDefinition&);
76
77
78 virtual G4double
79 PostStepGetPhysicalInteractionLength(const G4Track& track,
80 G4double previousStepSize,
81 G4ForceCondition* condition);
82
83 inline G4double AtRestGetPhysicalInteractionLength(
84 const G4Track& track,
85 G4ForceCondition* condition
86 );
87 inline G4VParticleChange* AtRestDoIt(
88 const G4Track& track,
89 const G4Step& step
90 );
91
92 inline G4VParticleChange* PostStepDoIt(
93 const G4Track& track,
94 const G4Step& step
95 )
96 {
97 return AtRestDoIt(track, step);
98 }
99
100 inline void SetVerbose(G4int);
101
102 //__________________________________________________________________
103 void SetDecayDisplacer(const G4ParticleDefinition*, G4VMolecularDecayDisplacer*);
104 G4VMolecularDecayDisplacer* GetDecayDisplacer(const G4ParticleDefinition*);
105
106 //__________________________________________________________________
107 void SetDisplacer(const G4ParticleDefinition*, G4VMolecularDecayDisplacer*);
108 G4VMolecularDecayDisplacer* GetDisplacer(const G4ParticleDefinition*);
109
110protected:
111 //__________________________________________________________________
112 // Make the decay
113 virtual G4VParticleChange* DecayIt(const G4Track& ,const G4Step&);
114 virtual G4double GetMeanLifeTime(const G4Track&,G4ForceCondition*);
115 virtual G4double GetMeanFreePath(const G4Track& ,
116 G4double ,
117 G4ForceCondition* )
118 {
119 return 0;
120 }
121
122private:
123 G4DNAMolecularDissociation();
124 G4DNAMolecularDissociation(const G4DNAMolecularDissociation &right);
125 G4DNAMolecularDissociation & operator=(const G4DNAMolecularDissociation &right);
126
127private:
128 G4bool fDecayAtFixedTime ;
129
130 typedef std::map<const G4ParticleDefinition*, G4VMolecularDecayDisplacer*> DisplacementMap;
131 DisplacementMap fDisplacementMap;
132
133 G4int fVerbose;
134};
135
136inline void G4DNAMolecularDissociation::SetVerbose(G4int verbose)
137{
138 fVerbose = verbose ;
139}
140
141inline G4double G4DNAMolecularDissociation::AtRestGetPhysicalInteractionLength(
142 const G4Track& track,
143 G4ForceCondition* condition)
144{
145 if(fDecayAtFixedTime)
146 {
147 return GetMeanLifeTime(track, condition);
148 }
149
150 return G4VITRestDiscreteProcess::AtRestGetPhysicalInteractionLength(track, condition);
151// return G4VITRestProcess::AtRestGetPhysicalInteractionLength(track, condition);
152}
153
154inline G4VParticleChange* G4DNAMolecularDissociation::AtRestDoIt(
155 const G4Track& track,
156 const G4Step& step
157 )
158{
159 ClearNumberOfInteractionLengthLeft();
160 ClearInteractionTimeLeft();
161 return DecayIt(track, step);
162}
