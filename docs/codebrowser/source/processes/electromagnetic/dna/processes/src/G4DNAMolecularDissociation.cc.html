50G4DNAMolecularDissociation::
51G4DNAMolecularDissociation(const G4String& processName,
52 G4ProcessType type) :
53 G4VITRestDiscreteProcess(processName, type)
54{
55 // set Process Sub Type
56 SetProcessSubType(59); // DNA sub-type
57 enableAlongStepDoIt = false;
58 enablePostStepDoIt = true;
59 enableAtRestDoIt = true;
60
61 fVerbose = 0;
62
63#ifdef G4VERBOSE
64 if(verboseLevel > 1)
65 {
66 G4cout << "G4MolecularDissociationProcess constructor " << " Name:"
67 << processName << G4endl;
68 }
69#endif
70
71 pParticleChange = &aParticleChange;
72
73 fDecayAtFixedTime = true;
74 fProposesTimeStep = true;
75}

79G4DNAMolecularDissociation::~G4DNAMolecularDissociation()
80{
81 DisplacementMap::iterator it = fDisplacementMap.begin();
82
83 for(; it != fDisplacementMap.end(); it++)
84 {
85 if(it->second)
86 {
87 delete it->second;
88 it->second = 0;
89 }
90 }
91 fDisplacementMap.clear();
92}

140G4VParticleChange* G4DNAMolecularDissociation::DecayIt(const G4Track& track,
141 const G4Step&)
142{
143 // DEBUG
144 // G4cout << "Is calling G4MolecularDecayProcess::DecayIt" << G4endl;
145
146 aParticleChange.Initialize(track);
147 const G4Molecule * theMotherMolecule = GetMolecule(track);
148 const G4MoleculeDefinition* moleculeDefinition = theMotherMolecule
149 ->GetDefinition();
150
151 // DEBUG
152 // G4cout <<"Calling G4MolecularDecayProcess::DecayIt"<<G4endl;
153 // G4cout << "The mother molecule state : " << G4endl;
154 // theMotherMolecule -> PrintState();
155
156 if(moleculeDefinition->GetDecayTable())
157 {
158 const vector<const G4MolecularDissociationChannel*>* DecayVector =
159 (theMotherMolecule->GetDecayChannel());
160
161 if(DecayVector == 0)
162 {
163 G4ExceptionDescription exceptionDescription;
164 theMotherMolecule->PrintState();
165 exceptionDescription << "No decay channel was found for the molecule : "
166 << theMotherMolecule->GetName() << G4endl;
167 G4Exception("G4DNAMolecularDissociation::DecayIt",
168 "G4DNAMolecularDissociation::NoDecayChannel",
169 FatalException,
170 exceptionDescription);
171 return &aParticleChange;
172 }
173
174 G4int DecayVectorSize = DecayVector->size();
175 // DEBUG
176 // G4cout<< "Number of decay channels : " << DecayVectorSize <<G4endl;
177 G4double RdmValue = G4UniformRand();
178
179 const G4MolecularDissociationChannel* decayChannel(0);
180 G4int i = 0;
181 do
182 {
183 decayChannel = (*DecayVector)[i];
184 if(RdmValue < decayChannel->GetProbability()) break;
185 RdmValue -= decayChannel->GetProbability();
186 i++;
187 }
188 while(i < DecayVectorSize);
189
190 // DEBUG
191 // G4cout << "Selected Decay channel : "
192 // << decayChannel->GetName() << G4endl;
193
194 G4double decayEnergy = decayChannel->GetEnergy();
195 G4int nbProducts = decayChannel->GetNbProducts();
196
197 if(decayEnergy)
198 {
199 // DEBUG
200 // G4cout << "Deposit energy :"
201 // << decayChannel->GetEnergy()/eV << " eV" << G4endl;
202
203 aParticleChange.ProposeLocalEnergyDeposit(decayChannel->GetEnergy());
204 }
205
206 if(nbProducts)
207 {
208
209 // DEBUG
210 // G4cout << "Number of products :" << nbProducts << G4endl;
211
212 vector<G4ThreeVector> ProductsDisplacement(nbProducts);
213 G4ThreeVector theMotherMoleculeDisplacement;
214
215 DisplacementMap::iterator it =
216 fDisplacementMap.find(moleculeDefinition);
217
218 if(it != fDisplacementMap.end())
219 {
220 G4VMolecularDecayDisplacer* displacer = it->second;
221 ProductsDisplacement = displacer->GetProductsDisplacement(decayChannel);
222 theMotherMoleculeDisplacement =
223 displacer->GetMotherMoleculeDisplacement(decayChannel);
224 }
225 else
226 {
227 G4ExceptionDescription errMsg;
228 errMsg << "No G4MolecularDecayProcess::theDecayDisplacementMap["
229 << theMotherMolecule->GetName() + "]";
230 G4Exception("G4MolecularDecayProcess::DecayIt",
231 "DNAMolecularDecay001",
232 FatalErrorInArgument,
233 errMsg);
234 }
235
236 aParticleChange.SetNumberOfSecondaries(nbProducts);
237
238// G4cout << " nbProducts = " << nbProducts << G4endl;
239// theMotherMolecule->PrintState();
240
241#ifdef G4VERBOSE
242 if(fVerbose)
243 {
244 G4cout << "Decay Process : " << theMotherMolecule->GetName()
245 << " (trackID :" << track.GetTrackID() << ") "
246 << decayChannel->GetName() << G4endl;
247 }
248#endif
249
250 G4ITNavigator* navigator =
251 G4ITTransportationManager::GetTransportationManager()
252 ->GetNavigatorForTracking();
253
254 for(G4int j = 0; j < nbProducts; j++)
255 {
256 G4Molecule* product = new G4Molecule(decayChannel->GetProduct(j));
257
258 G4ThreeVector displacement = theMotherMoleculeDisplacement
259 + ProductsDisplacement[j];
260 double mag_displacement = displacement.mag();
261 G4ThreeVector displacement_direction =
262 displacement/(mag_displacement+1e-30);
263
264 double prNewSafety = DBL_MAX;
265
266 //double step =
267 navigator->CheckNextStep(track.GetPosition(),
268 displacement_direction,
269 mag_displacement,
270 prNewSafety);
271
272 //if(prNewSafety < mag_displacement || step < mag_displacement)
273 mag_displacement = min(prNewSafety*0.8, mag_displacement);
274
275 G4ThreeVector product_pos = track.GetPosition()
276 + displacement_direction * mag_displacement;
277
278 const G4AffineTransform& transform = navigator
279 ->GetGlobalToLocalTransform();
280
281 G4ThreeVector localPoint =
282 transform.TransformPoint(product_pos); //track.GetPosition());
283
284 if(track.GetTouchable()->GetSolid()->Inside(localPoint) !=
285 EInside::kInside)
286 {
287 G4cout << "Mother volume: "
288 << track.GetTouchable()->GetVolume()->GetName()
289 << G4endl;
290 G4Exception("G4DNAMolecularDissociation::DecayIt",
291 "OUTSIDE_OF_MOTHER_VOLUME",
292 FatalException,
293 "Product has been placed outside of the volume "
294 "containing the mother molecule");
295 }
296
297 G4Track* secondary =
298 product->BuildTrack(track.GetGlobalTime(),
299 product_pos);
300
301 secondary->SetTrackStatus(fAlive);
302#ifdef G4VERBOSE
303 if(fVerbose)
304 {
305 G4cout << "Product : " << product->GetName() << G4endl;
306 }
307#endif
308 // add the secondary track in the List
309 aParticleChange.G4VParticleChange::AddSecondary(secondary);
310 }
311#ifdef G4VERBOSE
312 if(fVerbose) G4cout << "-------------" << G4endl;
313#endif
314 }
315 //DEBUG
316 else if(fVerbose && decayEnergy)
317 {
318 G4cout << "No products for this channel" << G4endl;
319 G4cout<<"-------------"<<G4endl;
320 }
321 /*
322 else if(!decayEnergy && !nbProducts)
323 {
324 G4ExceptionDescription errMsg;
325 errMsg << "There is no products and no energy specified in the molecular "
326 "dissociation channel";
327 G4Exception("G4MolecularDissociationProcess::DecayIt",
328 "DNAMolecularDissociation002",
329 FatalErrorInArgument,
330 errMsg);
331 }
332 */
333 }
334
335 aParticleChange.ProposeTrackStatus(fStopAndKill);
336
337 return &aParticleChange;
338}
