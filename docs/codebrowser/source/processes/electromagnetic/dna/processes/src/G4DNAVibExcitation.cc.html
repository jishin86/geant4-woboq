55//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
56
57void G4DNAVibExcitation::InitialiseProcess(const G4ParticleDefinition* p)
58{
59 if(!isInitialised)
60 {
61 isInitialised = true;
62 SetBuildTableFlag(false);
63
64 G4String name = p->GetParticleName();
65
66 if(name == "e-" )
67 {
68 if(!EmModel())
69 {
70 SetEmModel(new G4DNASancheExcitationModel);
71 EmModel()->SetLowEnergyLimit(2*eV);
72 EmModel()->SetHighEnergyLimit(100*eV);
73 }
74 AddEmModel(1, EmModel());
75 } else if(name == "e+")
76 {
77 if(!EmModel())
78 {
79 SetEmModel(new G4LEPTSVibExcitationModel);
80 EmModel()->SetLowEnergyLimit(2*eV);
81 EmModel()->SetHighEnergyLimit(100*eV);
82 }
83 AddEmModel(1, EmModel());
84 }
85 }
86}
