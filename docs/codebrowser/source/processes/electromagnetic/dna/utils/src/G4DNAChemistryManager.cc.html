296void G4DNAChemistryManager::SetNewValue(G4UIcommand* command, G4String value)
297{
298 if (command == fpActivateChem)
299 {
300 Activated(G4UIcmdWithABool::GetNewBoolValue(value));
301 }
302 else if (command == fpRunChem)
303 {
304 Run();
305 }
306 /*
307 else if(command == fpGridSize)
308 {
309 fDefaultGridResolution = fpGridSize->ConvertToDimensionedDouble(value);
310 }*/
311 else if (command == fpSkipReactionsFromChemList)
312 {
313 fSkipReactions = true;
314 }
315 else if(command == fpScaleForNewTemperature)
316 {
317 SetGlobalTemperature(fpScaleForNewTemperature->
318 ConvertToDimensionedDouble(value));
319 }
320 else if(command == fpInitChem)
321 {
322 Initialize();
323 InitializeThread();
324 }
325}

340
341void G4DNAChemistryManager::Run()
342{
343 if (fActiveChemistry)
344 {
345 InitializeThread();
346
347 if (fMasterInitialized == false)
348 {
349 G4ExceptionDescription description;
350 description << "Global components were not initialized.";
351 G4Exception("G4DNAChemistryManager::Run", "MASTER_INIT", FatalException,
352 description);
353 }
354
355 if (fpThreadData->fThreadInitialized_tl == 0)
356 {
357 G4ExceptionDescription description;
358 description << "Thread local components were not initialized.";
359 G4Exception("G4DNAChemistryManager::Run", "THREAD_INIT", FatalException,
360 description);
361 }
362
363 G4MoleculeTable::Instance()->Finalize();
364 G4Scheduler::Instance()->Process();
365 if(fResetCounterWhenRunEnds)
366 {
367 G4VMoleculeCounter::Instance()->ResetCounter();
368 }
369 CloseFile();
370 }
371}

381//------------------------------------------------------------------------------
382
383void G4DNAChemistryManager::Initialize()
384{
385 //===========================================================================
386 // MT MODE
387 //===========================================================================
388 if(G4Threading::IsMultithreadedApplication())
389 {
390 //==========================================================================
391 // ON WORKER THREAD
392 //==========================================================================
393 if(G4Threading::IsWorkerThread())
394 {
395 InitializeThread(); // Will create and initialize G4Scheduler
396 return;
397 }
398 //==========================================================================
399 // ON MASTER THREAD
400 //==========================================================================
401 else
402 {
403 InitializeMaster();
404 return;
405 }
406 }
407 //===========================================================================
408 // IS NOT IN MT MODE
409 //===========================================================================
410 else
411 {
412 InitializeMaster();
413 // In this case: InitializeThread is called when Run() is called
414 return;
415 }
416
417}

419//------------------------------------------------------------------------------
420
421void G4DNAChemistryManager::InitializeMaster()
422{
423 if (fMasterInitialized == false)
424 {
425 if(fVerbose)
426 {
427 G4cout << "G4DNAChemistryManager::InitializeMaster() is called" << G4endl;
428 }
429
430 G4Scheduler::Instance();
431 // creates a concrete object of the scheduler
432
433 if (fpUserChemistryList)
434 {
435 fpUserChemistryList->ConstructDissociationChannels();
436 if(fSkipReactions == false)
437 {
438 fpUserChemistryList->ConstructReactionTable(
439 G4DNAMolecularReactionTable::GetReactionTable());
440 }
441 else
442 {
443 G4DNAMolecularReactionTable::GetReactionTable(); // init pointer
444 }
445 fMasterInitialized = true;
446 }
447 else
448 {
449 if (fActiveChemistry)
450 {
451 G4ExceptionDescription description;
452 description << "No user chemistry list has been provided.";
453 G4Exception("G4DNAChemistryManager::InitializeMaster", "NO_CHEM_LIST",
454 FatalException, description);
455 }
456 }
457 }

460//------------------------------------------------------------------------------
461
462void G4DNAChemistryManager::InitializeThread()
463{
464 if (fpThreadData->fThreadInitialized_tl == false
465 || fForceThreadReinitialization == true)
466 {
467 if (fpUserChemistryList)
468 {
469 if(fVerbose)
470 {
471 G4cout << "G4DNAChemistryManager::InitializeThread() is called"
472 << G4endl;
473 }
474
475 if (fBuildPhysicsTable && fPhysicsTableBuilt == false)
476 {
477 if(fVerbose)
478 {
479 G4cout << "G4DNAChemistryManager: Build the physics tables for "
480 "molecules."
481 << G4endl;
482 }
483
484 fpUserChemistryList->BuildPhysicsTable();
485 if (fGeometryClosed == false)
486 {
487 if(fVerbose)
488 {
489 G4cout << "G4DNAChemistryManager: Close geometry"
490 << G4endl;
491 }
492
493 G4GeometryManager* geomManager = G4GeometryManager::GetInstance();
494 // G4cout << "Start closing geometry." << G4endl;
495 geomManager->OpenGeometry();
496 geomManager->CloseGeometry(true, true);
497 fGeometryClosed = true;
498 }
499
500 fPhysicsTableBuilt = true;
501 }
502 fpUserChemistryList->ConstructTimeStepModel(
503 G4DNAMolecularReactionTable::GetReactionTable());
504 G4Scheduler::Instance()->Initialize();
505
506 fpThreadData->fThreadInitialized_tl = true;
507 }
508 else
509 {
510 G4ExceptionDescription description;
511 description << "No user chemistry list has been provided.";
512 G4Exception("G4DNAChemistryManager::InitializeThread", "NO_CHEM_LIST",
513 FatalException, description);
514 }
515
516 G4VMoleculeCounter::InitializeInstance();
517 }
518
519 InitializeFile();
520}

627//------------------------------------------------------------------------------
628
629void
630G4DNAChemistryManager::CreateWaterMolecule(ElectronicModification modification,
631 G4int electronicLevel,
632 const G4Track* theIncomingTrack)
633{
634 if(fpThreadData->fpPhysChemIO){
635 G4double energy = -1.;
636
637 switch (modification)
638 {
639 case eDissociativeAttachment:
640 energy = 0;
641 break;
642 case eExcitedMolecule:
643 energy = GetExcitationLevel()->ExcitationEnergy(electronicLevel);
644 break;
645 case eIonizedMolecule:
646 energy = GetIonisationLevel()->IonisationEnergy(electronicLevel);
647 break;
648 }
649
650 fpThreadData->fpPhysChemIO->CreateWaterMolecule(modification,
651 4-electronicLevel,
652 energy,
653 theIncomingTrack);
654 }
655
656 if(fActiveChemistry)
657 {
658 G4Molecule * H2O = new G4Molecule (G4H2O::Definition());
659
660 switch (modification)
661 {
662 case eDissociativeAttachment:
663 H2O -> AddElectron(5,1);
664 break;
665 case eExcitedMolecule :
666 H2O -> ExciteMolecule(4-electronicLevel);
667 break;
668 case eIonizedMolecule :
669 H2O -> IonizeMolecule(4-electronicLevel);
670 break;
671 }
672
673 G4Track * H2OTrack = H2O->BuildTrack(1*picosecond,
674 theIncomingTrack->GetPosition());
675
676 H2OTrack -> SetParentID(theIncomingTrack->GetTrackID());
677 H2OTrack -> SetTrackStatus(fStopButAlive);
678 H2OTrack -> SetKineticEnergy(0.);
679 G4VITTrackHolder::Instance()->Push(H2OTrack);
680 }
681}

683//------------------------------------------------------------------------------
684
685void
686G4DNAChemistryManager::CreateSolvatedElectron(const G4Track* theIncomingTrack,
687 G4ThreeVector* finalPosition)
688// finalPosition is a pointer because this argument is optional
689{
690 if(fpThreadData->fpPhysChemIO){
691 fpThreadData->fpPhysChemIO->CreateSolvatedElectron(theIncomingTrack,
692 finalPosition);
693 }
694
695 if(fActiveChemistry)
696 {
697 G4Molecule* e_aq = new G4Molecule(G4Electron_aq::Definition());
698 G4Track * e_aqTrack(0);
699 if(finalPosition)
700 {
701 e_aqTrack = e_aq->BuildTrack(picosecond,*finalPosition);
702 }
703 else
704 {
705 e_aqTrack = e_aq->BuildTrack(picosecond,theIncomingTrack->GetPosition());
706 }
707 e_aqTrack -> SetTrackStatus(fAlive);
708 e_aqTrack -> SetParentID(theIncomingTrack->GetTrackID());
709 G4VITTrackHolder::Instance()->Push(e_aqTrack);
710 }
711}
