75/**
76 * WARNING: THIS CLASS IS A PROTOTYPE
77 * G4DNAChemistryManager is called from the physics models.
78 * It creates the water molecules and the solvated electrons and
79 * and send them to G4ITStepManager to be treated in the chemistry stage.
80 * For this, the fActiveChemistry flag needs to be on.
81 * It is also possible to give already molecule's pointers already built.
82 * G4DNAChemistryManager will then be in charge of creating the track and loading
83 * it to the IT system.
84 * The user can also ask to create a file containing a information about the
85 * creation of water molecules and solvated electrons.
86 */
87
88class G4DNAChemistryManager: public G4UImessenger,
89 public G4VStateDependent
90{
91protected:
92 virtual ~G4DNAChemistryManager();
93
94public:
95 //============================================================================
96 // STATIC METHODS
97 //============================================================================
98 static G4DNAChemistryManager* Instance();
99 static G4DNAChemistryManager* GetInstanceIfExists();
100
101 /**
102 * You should rather use DeleteInstance than the destructor of this class
103 */
104 static void DeleteInstance();
105 G4bool IsChemistryActivated();
106 void SetChemistryActivation(G4bool);
107 static G4bool IsActivated();
108 static void Activated(G4bool flag = true);
109
110 //============================================================================
111 // VIRTUAL METHODS
112 //============================================================================
113 virtual G4bool Notify(G4ApplicationState requestedState);
114 virtual void SetNewValue(G4UIcommand*, G4String);
115 virtual G4String GetCurrentValue(G4UIcommand * command);
116
117 //============================================================================
118 // INITIALIZATION AND FINALIZATION METHODS
119 //============================================================================
120 void Initialize();
121 inline void SetChemistryList(G4VUserChemistryList*);
122 inline void Deregister(G4VUserChemistryList*);
123 void SetGlobalTemperature(G4double temp_K);
124
125 inline void ForceMasterReinitialization();
126 inline void TagThreadForReinitialization();
127
128 void Run();
129 void Clear();
130 void Gun(G4ITGun*, bool physicsTableToBuild = true);
131 inline void ForceThreadReinitialization();
132 inline void ForceRebuildingPhysicsTable();
133
134 //============================================================================
135 // FILE OPERATIONS
136 //============================================================================
137 /**
138 * Tells the chemMan to write into a file
139 * the position and electronic state of the water molecule
140 * and the position thermalized or not of the solvated electron
141 */
142 void WriteInto(const G4String&, std::ios_base::openmode mode =
143 std::ios_base::out);
144 void AddEmptyLineInOuputFile();
145
146 /**
147 * Close the file specified with WriteInto
148 */
149 void CloseFile();
150
151 //============================================================================
152 // PUSH MOLECULES
153 //============================================================================
154 /**
155 * Method used by DNA physics model to create a water molecule.
156 * The ElectronicModification is a flag telling whether the molecule
157 * is ionized or excited, the electronic level is calculated by the
158 * model and the IncomingTrack is the track responsible for the creation
159 * of this molecule (electron, proton...).
160 */
161 void CreateWaterMolecule(ElectronicModification,
162 G4int /*electronicLevel*/,
163 const G4Track* /*theIncomingTrack*/);
164
165 /**
166 * On the same idea as the previous method but for solvated electron.
167 * This method should be used by the physics model of the ElectronSolvatation
168 * process.
169 */
170 void CreateSolvatedElectron(const G4Track* /*theIncomingTrack*/,
171 G4ThreeVector* finalPosition = 0);
172
173 /**
174 * WARNING : In case chemistry is not activated, PushMolecule will take care
175 * of deleting the transfered molecule.
176 * Before calling this method, it is also possible to check if the chemistry
177 * is activated through IsChemistryActived().
178 * This method will create the track corresponding to the transfered molecule
179 * and will be in charge of loading the new track to the system.
180 */
181
182 void PushMolecule(G4Molecule*& molecule,
183 G4double time,
184 const G4ThreeVector& position,
185 G4int parentID);
186
187 /**
188 * WARNING : In case chemistry is not activated, PushMoleculeAtParentTimeAndPlace
189 * will take care of deleting the transfered molecule.
190 * Before calling this method, it is also possible to check if the chemistry
191 * is activated through IsChemistryActived().
192 * This method will create the track corresponding to the transfered molecule
193 * and will be in charge of loading the new track to the system.
194 */
195 void PushMoleculeAtParentTimeAndPlace(G4Molecule*& molecule,
196 const G4Track* /*theIncomingTrack*/);

222 void InitializeFile();

226 G4DNAChemistryManager();

268inline void G4DNAChemistryManager::ForceRebuildingPhysicsTable()
269{
270 fPhysicsTableBuilt = false;
271}

273inline void G4DNAChemistryManager::SetChemistryList(G4VUserChemistryList* chemistryList)
274{
275 fpUserChemistryList = chemistryList;
276 Activated();
277}
278
279inline void G4DNAChemistryManager::Deregister(G4VUserChemistryList* chemistryList)
280{
281 if (fpUserChemistryList == chemistryList) fpUserChemistryList = 0;
282}
283
284inline void G4DNAChemistryManager::ForceMasterReinitialization()
285{
286 fMasterInitialized = false;
287 InitializeMaster();
288}

290inline void G4DNAChemistryManager::ForceThreadReinitialization()
291{
292 fForceThreadReinitialization = true;
293}
294
295inline void G4DNAChemistryManager::TagThreadForReinitialization()
296{
297 fpThreadData->fThreadInitialized_tl = false;
298}
