53class G4Material;
54class G4MolecularConfiguration;

56/**
57 * \struct CompareMaterial
58 * \brief Materials can be described as a derivation of existing "parent"
59 * materials in order to alter few of their features, such as density.
60 * \p CompareMaterial compare materials taking into account
61 * their possible "affiliation".
62 */
63struct CompareMaterial
64{
65 bool operator()(const G4Material* mat1, const G4Material* mat2) const;
66};
67
68typedef std::map<const G4Material*, double, CompareMaterial> ComponentMap;
69
70/**
71 * \class G4DNAMolecularMaterial
72 * \brief G4DNAMolecularMaterial builds tables of molecular densities for chosen
73 * molecular materials. The class handles homogeneous, composite and
74 * derived materials. A material of interest is labeled as molecular if built
75 * using the number of atoms rather than the mass fractions.
76 *
77 * \details
78 * - Initialization:
79 * G4DNAMolecularMaterial is initialized when
80 * G4ApplicationState == G4State_Idle.
81 * It should be initialized on the master thread and used in read-only mode
82 * during stepping. The singleton is thread-shared.
83 *
84 * - For Developers:
85 * Use GetNumMolPerVolTableFor(molecule) in the concrete implementation of
86 * G4VEmModel::Initialise or G4VProcess::PreparePhysicsTable
87 * at run initialization to retrieve a read-only, thread-safe, table.
88 * The table is then built on the master thread at initialization time and
89 * shared between all threads and models.
90 *
91 * \note A G4material is labeled as molecular if built using the number of atoms
92 *
93 */
94
95class G4DNAMolecularMaterial: public G4VStateDependent
96{
97public:
98 static G4DNAMolecularMaterial* Instance();
99 static void DeleteInstance();
100 void Initialize();
101 void Clear();
102
103 virtual G4bool Notify(G4ApplicationState requestedState);
104
105 //----------------------------------------------------------------------------
106
107 /**
108 * \fn const std::vector<double>* \
109 * GetDensityTableFor(const G4Material* searchedMaterial) const
110 * \brief Retrieve a table of volumetric mass densities (mass per unit volume)
111 * in the G4 unit system for chosen material.
112 *
113 * @param[in] searchedMaterial
114 * The material which you'd like to retrieve the volumic mass
115 * @pre The \p searchedMaterial used in parameter must be built as a
116 * molecular material, using the number of atoms rather than the density
117 * fractions.
118 * \return
119 * Pointer to a table of molecular densities for the \p searchedMaterial
120 * indexed on the (parent) material index.
121 *
122 */
123 const std::vector<double>* GetDensityTableFor(const G4Material*) const;
124
125 /**
126 * \fn const std::vector<double>* \
127 * GetNumMolPerVolTableFor(const G4Material* searchedMaterial) const
128 * \brief Retrieve a table of molecular densities (number of molecules per
129 * unit volume) in the G4 unit system for chosen material.
130 *
131 * @param[in] searchedMaterial
132 * The material which you'd like to retrieve the molecular density
133 * @pre The \p searchedMaterial used in parameter must be built as a
134 * molecular material, using the number of atoms rather than the density
135 * fractions.
136 * \return
137 * Pointer to a table of molecular densities for the \p searchedMaterial
138 * indexed on the (parent) material index.
139 */
140 const std::vector<double>* GetNumMolPerVolTableFor(const G4Material*) const;
141
142 inline const std::vector<ComponentMap>* GetMassFractionTable() const{
143 return fpCompFractionTable;
144 }
145 inline const std::vector<ComponentMap>* GetDensityTable() const{
146 return fpCompDensityTable;
147 }
148
149 //----------------------------------------------------------------------------
150
151 G4MolecularConfiguration* GetMolecularConfiguration(const G4Material*) const;
152
153 /**
154 * \fn void SetMolecularConfiguration(const G4Material* material, \
155 * G4MolecularConfiguration* molConf)
156 * \brief Associate a molecular configuration to a G4material.
157 *
158 * @param[in] material
159 * Pointer to a G4 material. The material
160 * does not need to be defined as a molecular material.
161 * @param[in] molConf
162 * The molecular configuration corresponding to
163 * the G4 \p material.
164 */
165 void SetMolecularConfiguration(const G4Material*,
166 G4MolecularConfiguration*);
167
168 /**
169 * \fn void SetMolecularConfiguration(const G4Material* material, \
170 * const G4String& molConf)
171 * \brief Associate a molecular configuration to a G4material.
172 *
173 * @param[in] material
174 * Pointer to a G4 material. The material
175 * does not need to be defined as a molecular material.
176 * @param[in] molConf
177 * User ID of the molecular configuration corresponding to
178 * the G4 \p material.
179 */
180 void SetMolecularConfiguration(const G4Material*,
181 const G4String&);
182
183 /**
184 * \fn void SetMolecularConfiguration(const G4Material* material, \
185 * const G4String& molConf)
186 * \brief Associate a molecular configuration to a G4material.
187 *
188 * @param[in] material
189 * Name of the G4 material. The material
190 * does not need to be defined as a molecular material.
191 * @param[in] molConf
192 * User ID of the molecular configuration corresponding to
193 * the G4 \p material.
194 */
195 void SetMolecularConfiguration(const G4String& materialName,
196 const G4String& molUserIF);
197
198 //----------------------------------------------------------------------------
199
200 /**
201 * \brief Deprecated
202 * \deprecated Will return a G4 fatal exception.
203 * Use instead GetNumMolPerVolTableFor(molecule) at run
204 * initialization to retrieve a read-only, thread-safe, table.
205 * \note A G4material is labeled as molecular if built using
206 * the number of atoms.
207 */
208 G4double GetNumMoleculePerVolumeUnitForMaterial(const G4Material *mat);
209
210 /**
211 * \brief Deprecated
212 * \deprecated Will return a G4 fatal exception.
213 * Use instead GetNumMolPerVolTableFor(molecule) at run
214 * initialization to retrieve a read-only, thread-safe, table.
215 * \note A G4material is labeled as molecular if built using
216 * the number of atoms.
217 */
218 G4double GetNumMolPerVolForComponentInComposite(const G4Material *composite,
219 const G4Material *component,
220 G4double massFraction);
221
222protected:
223 static G4DNAMolecularMaterial* fInstance;
224 G4DNAMolecularMaterial();
225 G4DNAMolecularMaterial(const G4DNAMolecularMaterial& right);
226 G4DNAMolecularMaterial& operator=(const G4DNAMolecularMaterial&);
227 virtual ~G4DNAMolecularMaterial();
228 void Create();
229 void InitializeNumMolPerVol();
230 void InitializeDensity();
231 void RecordMolecularMaterial(G4Material* parentMaterial,
232 G4Material* molecularMaterial,
233 G4double fraction);
234 void SearchMolecularMaterial(G4Material* parentMaterial,
235 G4Material* material,
236 double currentFraction);
237
238 void AddMaterial(const G4Material*, double fraction);
239
240 void PrintNotAMolecularMaterial(const char* methodName,
241 const G4Material* lookForMaterial) const;

244 std::vector<ComponentMap>* fpCompFractionTable;

248 mutable std::map<const G4Material*, std::vector<double>*, CompareMaterial>
249 fAskedDensityTable;

251 fAskedNumPerVolTable;
252 mutable std::map<const G4Material*, bool, CompareMaterial> fWarningPrinted;
253
254 std::map<int /*Material ID*/,
255 G4MolecularConfiguration*> fMaterialToMolecularConf;
256
257 G4bool fIsInitialized;
258 size_t fNMaterials;
259};
