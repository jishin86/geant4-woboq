33#include "G4VMoleculeCounter.hh"
34#include "G4MoleculeCounter.hh"
35
36using namespace std;
37
38G4bool G4VMoleculeCounter::fUse = false;
39G4ThreadLocal
40G4VMoleculeCounter* G4VMoleculeCounter::fpInstance = 0;
41
42//------------------------------------------------------------------------------
43
44void
45G4VMoleculeCounter::SetInstance(G4VMoleculeCounter* instance)
46{
47 if (fpInstance!=0){
48 G4ExceptionDescription errMsg;
49 errMsg<< "The G4MoleculeCounter was already initialized." << G4endl
50 << "The previous instance will be deleted in order to use yours." << G4endl
51 << "However this can generate conflicts. Make sure you call G4MoleculeCounter::SetInstance at the beginning of your application."
52 << "A good place would be ActionInitialization::Build & BuildForMaster"
53 << G4endl;
54
55 G4Exception("G4MoleculeCounter::SetInstance",
56 "SINGLETON_ALREADY_INITIALIZED",
57 JustWarning, errMsg);
58 delete fpInstance;
59 fpInstance = 0;
60 }
61
62 fpInstance = instance;
63}
64
65//------------------------------------------------------------------------------
66
67G4VMoleculeCounter* G4VMoleculeCounter::Instance()
68{
69 if (!fpInstance) fpInstance = new G4MoleculeCounter();
70 return fpInstance;
71}
72
73//------------------------------------------------------------------------------
74
75void G4VMoleculeCounter::DeleteInstance()
76{
77 if (fpInstance){
78 delete fpInstance;
79 fpInstance = 0;
80 }
81}
82
83//------------------------------------------------------------------------------
84
85void G4VMoleculeCounter::InitializeInstance()
86{
87 if(fpInstance) fpInstance->Initialize();
88}
89
90//------------------------------------------------------------------------------
91
92void G4VMoleculeCounter::Use(G4bool flag)
93{
94 fUse=flag;
95}
96
97//------------------------------------------------------------------------------
98
99G4bool G4VMoleculeCounter::InUse()
100{
101 return fUse;
102}
