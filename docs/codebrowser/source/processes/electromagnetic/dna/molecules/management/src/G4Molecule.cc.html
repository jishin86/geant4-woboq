203//______________________________________________________________________________
204/** Build a molecule at a specific excitation/ionisation state according
205 * to a ground state that can be obtained from G4GenericMoleculeManager.
206 * Put 0 in the second option if this is a ionisation.
207 */
208//////////////////////////
209G4Molecule::G4Molecule(G4MoleculeDefinition * moleculeDefinition,
210 G4int OrbitalToFree,
211 G4int OrbitalToFill) :
212 G4VUserTrackInformation("G4Molecule"), G4IT()
213//////////////////////////
214{
215 if (moleculeDefinition->GetGroundStateElectronOccupancy())
216 {
217 G4ElectronOccupancy dynElectronOccupancy(
218 *moleculeDefinition->GetGroundStateElectronOccupancy());
219
220 if (OrbitalToFill != 0)
221 {
222 dynElectronOccupancy.RemoveElectron(OrbitalToFree - 1, 1);
223 dynElectronOccupancy.AddElectron(OrbitalToFill - 1, 1);
224 // dynElectronOccupancy.DumpInfo(); // DEBUG
225 }
226
227 if (OrbitalToFill == 0)
228 {
229 dynElectronOccupancy.RemoveElectron(OrbitalToFree - 1, 1);
230 // dynElectronOccupancy.DumpInfo(); // DEBUG
231 }
232
233 fpMolecularConfiguration =
234 G4MolecularConfiguration::GetOrCreateMolecularConfiguration(
235 moleculeDefinition, dynElectronOccupancy);
236 }
237 else
238 {
239 fpMolecularConfiguration = 0;
240 G4Exception(
241 "G4Molecule::G4Molecule(G4MoleculeDefinition * moleculeDefinition, "
242 "G4int OrbitalToFree, G4int OrbitalToFill)",
243 "G4Molecule_wrong_usage_of_constructor",
244 FatalErrorInArgument,
245 "If you want to use this constructor, the molecule definition has to be "
246 "first defined with electron occupancies");
247 }
248}

250//______________________________________________________________________________
251/** Specific builder for water molecules to be used in Geant4-DNA,
252 * the last option Excitation is true if the molecule is excited, is
253 * false is the molecule is ionized.
254 */
255
256G4Molecule::G4Molecule(G4MoleculeDefinition * moleculeDefinition,
257 G4int Level,
258 G4bool Excitation) :
259 G4VUserTrackInformation("G4Molecule"), G4IT()
260{
261 if (moleculeDefinition->GetGroundStateElectronOccupancy())
262 {
263 G4ElectronOccupancy dynElectronOccupancy(
264 *moleculeDefinition->GetGroundStateElectronOccupancy());
265
266 if (Excitation == true)
267 {
268 dynElectronOccupancy.RemoveElectron(Level, 1);
269 dynElectronOccupancy.AddElectron(5, 1);
270 // dynElectronOccupancy.DumpInfo(); // DEBUG
271 }
272
273 if (Excitation == false)
274 {
275 dynElectronOccupancy.RemoveElectron(Level, 1);
276 // dynElectronOccupancy.DumpInfo(); // DEBUG
277 }
278
279 fpMolecularConfiguration =
280 G4MolecularConfiguration::GetOrCreateMolecularConfiguration(
281 moleculeDefinition, dynElectronOccupancy);
282 }
283 else
284 {
285 fpMolecularConfiguration = 0;
286 G4Exception(
287 "G4Molecule::G4Molecule(G4MoleculeDefinition * moleculeDefinition, "
288 "G4int OrbitalToFree, G4int OrbitalToFill)",
289 "G4Molecule_wrong_usage_of_constructor",
290 FatalErrorInArgument,
291 "If you want to use this constructor, the molecule definition has to be "
292 "first defined with electron occupancies");
293
294 }
295}

394G4Track * G4Molecule::BuildTrack(G4double globalTime,
395 const G4ThreeVector& position)
396{
397 if (fpTrack != 0){
398 G4Exception("G4Molecule::BuildTrack", "Molecule001", FatalErrorInArgument,
399 "A track was already assigned to this molecule");
400 }
401
402 // Kinetic Values
403 // Set a random direction to the molecule
404 G4double costheta = (2 * G4UniformRand()-1);
405 G4double theta = acos(costheta);
406 G4double phi = 2 * pi * G4UniformRand();
407
408 G4double xMomentum = cos(phi) * sin(theta);
409 G4double yMomentum = sin(theta) * sin(phi);
410 G4double zMomentum = costheta;
411
412 G4ThreeVector MomentumDirection(xMomentum, yMomentum, zMomentum);
413 G4double KineticEnergy = GetKineticEnergy();
414
415 G4DynamicParticle* dynamicParticle = new G4DynamicParticle(
416 fpMolecularConfiguration->GetDefinition(), MomentumDirection,
417 KineticEnergy);
418
419 if (G4VMoleculeCounter::InUse()){
420 G4VMoleculeCounter::Instance()->
421 AddAMoleculeAtTime(fpMolecularConfiguration,
422 globalTime,
423 &(fpTrack->GetPosition()));
424 }
425
426 //Set the Track
427 fpTrack = new G4Track(dynamicParticle, globalTime, position);
428 fpTrack->SetUserInformation(this);
429
430 return fpTrack;
431}

435G4double G4Molecule::GetKineticEnergy() const
436{
437 ////
438 // Ideal Gaz case
439 double v = GetDiffusionVelocity();
440 double E = (fpMolecularConfiguration->GetMass() / (c_squared)) * (v * v) / 2.;
441 ////
442 return E;
443}

445//______________________________________________________________________________
446
447G4double G4Molecule::GetDiffusionVelocity() const
448{
449 double moleculeMass = fpMolecularConfiguration->GetMass() / (c_squared);
450
451 ////
452 // Different possibilities
453 ////
454 // Ideal Gaz case : Maxwell Boltzmann Distribution
455 // double sigma = k_Boltzmann * fgTemperature / mass;
456 // return G4RandGauss::shoot( 0, sigma );
457 ////
458 // Ideal Gaz case : mean velocity from equipartition theorem
459 return sqrt(3 * k_Boltzmann *
460 G4MolecularConfiguration::GetGlobalTemperature()/ moleculeMass);
461 ////
462 // Using this approximation for liquid is wrong
463 // However the brownian process avoid taking
464 // care of energy consideration and plays only
465 // with positions
466}
