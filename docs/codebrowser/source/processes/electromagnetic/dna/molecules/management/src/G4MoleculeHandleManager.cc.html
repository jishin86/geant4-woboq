32G4ThreadLocal G4MoleculeHandleManager* G4MoleculeHandleManager::fInstance(0);
33
34G4MoleculeHandleManager::G4MoleculeHandleManager()
35{
36 // G4cout << "G4MoleculeHandleManager::G4MoleculeHandleManager()" << G4endl;
37}
38
39G4bool G4MoleculeHandleManager::CompMoleculePointer::operator()(const G4Molecule* mol1,
40 const G4Molecule* mol2) const
41{
42 return (*mol1) < (*mol2);
43}
44
45G4MoleculeHandleManager::~G4MoleculeHandleManager()
46{
47 if (fMoleculeHandle.empty() == false)
48 {
49 MoleculeHandleMap::iterator it = fMoleculeHandle.begin();
50 for (; it != fMoleculeHandle.end(); it++)
51 {
52 it->second.reset();
53 }
54 }
55}
56
57void G4MoleculeHandleManager::DeleteInstance()
58{
59 if (fInstance)
60 {
61 delete fInstance;
62 fInstance = 0;
63 }
64}
65
66G4MoleculeHandleManager* G4MoleculeHandleManager::Instance()
67{
68 if (!fInstance)
69 {
70 fInstance = new G4MoleculeHandleManager;
71 }
72 return fInstance;
73}
74
75G4MoleculeHandle G4MoleculeHandleManager::GetMoleculeHandle(const G4Molecule* molecule)
76{
77 MoleculeHandleMap::iterator it = fMoleculeHandle.find(molecule);
78 G4MoleculeHandle molHandle;
79
80 if (it != fMoleculeHandle.end())
81 {
82 molHandle = G4MoleculeHandle(it->second);
83 }
84 else
85 {
86 molHandle = G4MoleculeHandle(molecule);
87 fMoleculeHandle.insert(make_pair(molecule, G4MoleculeHandle(molHandle)));
88 }
89
90 return molHandle;
91}
