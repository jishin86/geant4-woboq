70/* Stored data for one dissociation channel :
71 * products, energy, time of decay and probability.
72 */
73class G4MolecularDissociationChannel
74{
75public:
76 G4MolecularDissociationChannel();
77 G4MolecularDissociationChannel(G4String);
78 ~G4MolecularDissociationChannel();
79 G4MolecularDissociationChannel(const G4MolecularDissociationChannel&);
80 G4MolecularDissociationChannel&
81 operator=(const G4MolecularDissociationChannel &right);
82
83public:
84 //____________________________________________________________________________
85 // Construct
86
87 void AddProduct(const G4Molecule*, G4double = 0);
88 void AddProduct(G4MolecularConfiguration*, G4double = 0);
89 void AddProduct(const G4String& molecule, G4double displacement = 0);
90
91 //____________________________________________________________________________
92
93 inline void SetName(const G4String&);
94 inline void SetEnergy(G4double);
95 inline void SetProbability(G4double);
96 inline void SetDecayTime(G4double);
97 inline void SetRMSMotherMoleculeDisplacement(G4double);
98 inline void SetDisplacementType(DisplacementType);
99
100 //____________________________________________________________________________
101
102 inline const G4String& GetName() const;
103 G4int GetNbProducts() const;
104 G4MolecularConfiguration* GetProduct(int) const;
105 inline const std::vector<G4double>& GetRMSProductsDisplacement() const;
106 inline G4double GetEnergy() const;
107 inline G4double GetProbability() const;
108 inline G4double GetDecayTime() const;
109 inline G4double GetRMSMotherMoleculeDisplacement() const;
110 inline DisplacementType GetDisplacementType() const;
111
112 //____________________________________________________________________________
113 //Root Mean Square radial distance thermalisation of a product
114 G4double GetRMSRadialDisplacementOfProduct(const G4Molecule*);
115
116private:
117
118 DisplacementType fDisplacementType;
119 G4String fName;
120// std::vector<const G4Molecule*>* fProductsVector;
121 std::vector<G4MolecularConfiguration*>* fProductsVector;
122 G4double fReleasedEnergy;
123 G4double fProbability;
124 G4double fDecayTime; // To be taken into account in the next releases
125
126 //Root Mean Square radial distance jump of the MotherMolecule molecule
127 G4double fRMSMotherMoleculeDisplacement;
128 std::vector<G4double> fRMSProductsDisplacementVector;
129};
