47#ifndef G4FakeParticleID_h
48#define G4FakeParticleID_h 1
49
50#include "G4Types.hh"
51
52class G4FakeParticleID
53{
54private :
55 friend G4FakeParticleID operator +(const G4FakeParticleID& left,const int& right);
56 friend G4FakeParticleID operator -(const G4FakeParticleID& left,const int& right);
57 int fValue;
58
59 static G4ThreadLocal int fLastValue;
60
61public :
62
63 static int Last()
64 {
65 return fLastValue ;
66 }
67
68 static G4FakeParticleID Create()
69 {
70 fLastValue ++;
71 return G4FakeParticleID(fLastValue);
72 }
73
74 static G4FakeParticleID Initialize(int i)
75 {
76 fLastValue = i;
77 return G4FakeParticleID(i);
78 }
79
80 G4FakeParticleID(const int d_) : fValue(d_){;}
81
82 G4FakeParticleID(){fValue=0;}
83 G4FakeParticleID(const G4FakeParticleID & d_) : fValue(d_.fValue){;}
84 inline G4FakeParticleID & operator=(const G4FakeParticleID & rhs) { this->fValue = rhs.fValue; return *this;}
85 G4FakeParticleID & operator=(const int & rhs) { this->fValue = rhs; return *this;}
86 inline operator int & () { return fValue; }
87 inline operator const int & () const { return fValue; }
88 inline bool operator==(const G4FakeParticleID & rhs) const { return fValue == rhs.fValue; }
89 inline bool operator==(const int & rhs) const { return fValue == rhs; }
90 inline bool operator<(const G4FakeParticleID & rhs) const { return fValue < rhs.fValue; }
91};
92
93extern G4FakeParticleID gStartCounter;
94
95#endif
