441inline G4double G4ITStepProcessor::CalculateSafety()
442{
443 return std::max(fpState->fEndpointSafety - (fpState->fEndpointSafOrigin
444 - fpPostStepPoint->GetPosition()).mag(),
445 kCarTolerance);
446}
