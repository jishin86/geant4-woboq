46#ifndef G4VITMODEL_HH
47#define G4VITMODEL_HH
48
49#include "AddClone_def.hh"
50#include "G4VITTimeStepComputer.hh"

54/**
55 * Define what to do before stepping and after stepping.
56 * The concrete implementation of G4VITModel defines the interaction
57 * between two G4IT types. The type might be just equal like :
58 * Molecule + Molecule, or different : Molecule + Atom.
59 */
60
61class G4VITStepModel
62{
63public:
64 /** Default constructor */
65 G4VITStepModel(const G4String& aName = "NoName");
66 /** Default destructor */
67 virtual ~G4VITStepModel();
68
69 /* Macro define in AddClone_def*/
70 G4IT_TO_BE_CLONED(G4VITStepModel)
71
72 void IsApplicable(G4ITType& type1, G4ITType& type2) ;
73 void virtual PrintInfo(){;}
74
75 virtual void Initialize();
76
77 inline void SetTimeStepper(G4VITTimeStepComputer* timeStepper);
78 inline void SetReactionProcess(G4VITReactionProcess* reactionProcess);
79
80 inline G4VITTimeStepComputer* GetTimeStepper();
81 inline const G4String& GetName();
82
83 inline G4VITReactionProcess* GetReactionProcess();
84 inline void SetReactionTable(G4ITReactionTable*);
85 inline const G4ITReactionTable* GetReactionTable();
86
87protected:
88
89 G4String fName;
90
91 G4VITTimeStepComputer* fpTimeStepper;
92 G4VITReactionProcess* fpReactionProcess;
93
94 const G4ITReactionTable* fpReactionTable ;
95
96 G4ITType fType1;
97 G4ITType fType2;
98
99protected :
100 /** Copy constructor
101 * \param other Object to copy from
102 */
103 G4VITStepModel(const G4VITStepModel& other);
104 /** Assignment operator
105 * \param other Object to assign from
106 * \return A reference to this
107 */
108 G4VITStepModel& operator=(const G4VITStepModel& other);
109};
110
111inline void G4VITStepModel::SetReactionTable(G4ITReactionTable* table)
112{
113 fpReactionTable = table;
114}
115
116inline const G4ITReactionTable* G4VITStepModel::GetReactionTable()
117{
118 return fpReactionTable ;
119}
120
121inline void G4VITStepModel::SetTimeStepper(G4VITTimeStepComputer* timeStepper)
122{
123 fpTimeStepper = timeStepper ;
124}
125
126inline void G4VITStepModel::SetReactionProcess(G4VITReactionProcess* reactionProcess)
127{
128 fpReactionProcess = reactionProcess ;
129}
130
131inline G4VITTimeStepComputer* G4VITStepModel::GetTimeStepper()
132{
133 return fpTimeStepper;
134}
135
136inline G4VITReactionProcess* G4VITStepModel::GetReactionProcess()
137{
138 return fpReactionProcess ;
139}
140
141inline const G4String& G4VITStepModel::GetName()
142{
143 return fName;
144}
145
146#endif // G4VITMODEL_HH
