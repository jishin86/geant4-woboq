62class G4VITRestDiscreteProcess : public G4VITProcess
63{
64 // Abstract class which defines the public behavior of
65 // rest + discrete physics interactions.
66public:
67
68 G4VITRestDiscreteProcess(const G4String&, G4ProcessType aType = fNotDefined);
69 G4VITRestDiscreteProcess(const G4VITRestDiscreteProcess &);
70
71 virtual ~G4VITRestDiscreteProcess();
72
73public:
74 // with description
75 virtual G4double
76 PostStepGetPhysicalInteractionLength(const G4Track& track,
77 G4double previousStepSize,
78 G4ForceCondition* condition);
79
80 virtual G4VParticleChange* PostStepDoIt(const G4Track&, const G4Step&);
81
82 virtual G4double AtRestGetPhysicalInteractionLength(const G4Track&,
83 G4ForceCondition*);
84
85 virtual G4VParticleChange* AtRestDoIt(const G4Track&, const G4Step&);
86
87 // no operation in AlongStepDoIt
88 virtual G4double AlongStepGetPhysicalInteractionLength(const G4Track&,
89 G4double,
90 G4double,
91 G4double&,
92 G4GPILSelection*)
93 {
94 return -1.0;
95 }
96
97 // no operation in AlongStepDoIt
98 virtual G4VParticleChange* AlongStepDoIt(const G4Track&, const G4Step&)
99 {
100 return 0;
101 }
102
103protected:
104 // with description
105 virtual G4double GetMeanFreePath(const G4Track& aTrack,
106 G4double previousStepSize,
107 G4ForceCondition* condition)=0;
108 // Calculates from the macroscopic cross section a mean
109 // free path, the value is returned in units of distance.
110
111 virtual G4double GetMeanLifeTime(const G4Track& aTrack,
112 G4ForceCondition* condition)=0;
113 // Calculates the mean life-time (i.e. for decays) of the
114 // particle at rest due to the occurence of the given process,
115 // or converts the probability of interaction (i.e. for
116 // annihilation) into the life-time of the particle for the
117 // occurence of the given process.
118
119protected:
120 // hide default constructor and assignment operator as private
121 G4VITRestDiscreteProcess();
122 G4VITRestDiscreteProcess & operator=(const G4VITRestDiscreteProcess &right);
123
124};

129inline G4double G4VITRestDiscreteProcess::PostStepGetPhysicalInteractionLength(const G4Track& track,
130 G4double previousStepSize,
131 G4ForceCondition* condition)
132{
133 if((previousStepSize < 0.0) || (fpState->theNumberOfInteractionLengthLeft
134 <= 0.0))
135 {
136 // beggining of tracking (or just after DoIt of this process)
137 ResetNumberOfInteractionLengthLeft();
138 }
139 else if(previousStepSize > 0.0)
140 {
141 // subtract NumberOfInteractionLengthLeft
142 SubtractNumberOfInteractionLengthLeft(previousStepSize);
143 }
144 else
145 {
146 // zero step
147 // DO NOTHING
148 }
149
150 // condition is set to "Not Forced"
151 *condition = NotForced;
152
153 // get mean free path
154 fpState->currentInteractionLength = GetMeanFreePath(track,
155 previousStepSize,
156 condition);
157
158 G4double value;
159 if(fpState->currentInteractionLength < DBL_MAX)
160 {
161 value = fpState->theNumberOfInteractionLengthLeft * (fpState->currentInteractionLength);
162 }
163 else
164 {
165 value = DBL_MAX;
166 }
167#ifdef G4VERBOSE
168 if(verboseLevel > 1)
169 {
170 G4cout << "G4VITRestDiscreteProcess::PostStepGetPhysicalInteractionLength ";
171 G4cout << "[ " << GetProcessName() << "]" << G4endl;
172 track.GetDynamicParticle()->DumpInfo();
173 G4cout << " in Material " << track.GetMaterial()->GetName() << G4endl;
174 G4cout << "InteractionLength= " << value / CLHEP::cm << "[cm] " << G4endl;
175 }
176#endif
177 return value;
178}

180inline G4VParticleChange* G4VITRestDiscreteProcess::PostStepDoIt(const G4Track&,
181 const G4Step&)
182{
183// reset NumberOfInteractionLengthLeft
184 ClearNumberOfInteractionLengthLeft();
185
186 return pParticleChange;
187}
188
189inline G4double G4VITRestDiscreteProcess::AtRestGetPhysicalInteractionLength(const G4Track& track,
190 G4ForceCondition* condition)
191{
192 // beggining of tracking
193 ResetNumberOfInteractionLengthLeft();
194
195 // condition is set to "Not Forced"
196 *condition = NotForced;
197
198 // get mean life time
199 fpState->currentInteractionLength = GetMeanLifeTime(track, condition);
200
201#ifdef G4VERBOSE
202 if((fpState->currentInteractionLength < 0.0) || (verboseLevel > 2))
203 {
204 G4cout << "G4VITRestDiscreteProcess::AtRestGetPhysicalInteractionLength ";
205 G4cout << "[ " << GetProcessName() << "]" << G4endl;
206 track.GetDynamicParticle()->DumpInfo();
207 G4cout << " in Material " << track.GetMaterial()->GetName() << G4endl;
208 G4cout << "MeanLifeTime = " << fpState->currentInteractionLength / CLHEP::ns
209 << "[ns]" << G4endl;
210 }
211#endif
212
213 return fpState->theNumberOfInteractionLengthLeft
214 * (fpState->currentInteractionLength);
215}

217inline G4VParticleChange* G4VITRestDiscreteProcess::AtRestDoIt(const G4Track&,
218 const G4Step&)
219{
220// clear NumberOfInteractionLengthLeft
221 ClearNumberOfInteractionLengthLeft();
222
223 return pParticleChange;
224}
