218template<class OBJECT>
219 class G4FastList
220 {
221 protected:
222 G4int fNbObjects;
223// G4FastListNode<OBJECT> * fpStart;
224// G4FastListNode<OBJECT> * fpFinish;
225 G4shared_ptr<_ListRef<G4FastList<OBJECT> > > fListRef;
226
227 G4FastListNode<OBJECT> fBoundary;
228 // Must be empty and link to the last non-empty node of the list
229 // and to the first non-empty node of the list (begin())
230 // The iterator returned by end() is linked to this empty node
231
232 public:
233 class Watcher
234 {
235 public:
236 enum Priority
237 {
238 eExtreme,
239 eHigh,
240 eNormal,
241 eLow,
242 eVeryLow
243 };
244
245 typedef G4FastList<OBJECT> list;
246
247 Watcher()
248 {
249 fPriority = Priority::eVeryLow;
250 }
251
252 virtual ~Watcher()
253 {
254 typename std::set<G4FastList<OBJECT>*>::iterator it = fWatching.begin();
255 typename std::set<G4FastList<OBJECT>*>::iterator end = fWatching.end();
256 for(;it!=end;it++)
257 {
258 (*it)->RemoveWatcher(this);
259 }
260 }
261
262 virtual G4String GetWatcherName(){
263 return "";
264 }
265
266 Priority GetPriority() const{
267 return fPriority;
268 }

281 void Watch(G4FastList<OBJECT>* fastList)
282 {
283 fWatching.insert(fastList);
284 fastList->AddWatcher(this);
285 }
286
287 void StopWatching(G4FastList<OBJECT>* fastList, bool removeWatcher = true)
288 {
289 typename std::set<G4FastList<OBJECT>*>::iterator it = fWatching.find(fastList);
290 if(it == fWatching.end()) return; //TODO: exception?
291 fWatching.erase(it);
292 if(removeWatcher) fastList->RemoveWatcher(this);
293 }
294
295 protected:
296 Priority fPriority;
297
298 private:
299 std::set<G4FastList<OBJECT>*> fWatching;
300 };

433template<class OBJECT>
434 struct sortWatcher
435 {
436 bool operator()(const typename G4FastList<OBJECT>::Watcher* left,
437 const typename G4FastList<OBJECT>::Watcher* right) const
438 {
439 if(left && right)
440 {
441 if(left->GetPriority() != right->GetPriority())
442 {
443 return left->GetPriority() < right->GetPriority();
444 }
445 return left < right;
446 }
447 return false;
448 }
449 };
