35#include "G4ITFinder.hh"
36
37using namespace std;
38
39G4ThreadLocal G4AllITFinder* G4AllITFinder::fpInstance = 0;
40
41G4AllITFinder::G4AllITFinder()
42{
43 fVerbose = 0;
44}
45
46G4AllITFinder* G4AllITFinder::Instance()
47{
48 if (!fpInstance) fpInstance = new G4AllITFinder();
49 return fpInstance;
50}
51
52void G4AllITFinder::DeleteInstance()
53{
54 if (fpInstance)
55 {
56 delete fpInstance;
57 fpInstance = 0;
58 }
59}
60
61G4AllITFinder::~G4AllITFinder()
62{
63 std::map<G4ITType, G4VITFinder*>::iterator it;
64 std::map<G4ITType, G4VITFinder*>::iterator it_tmp;
65
66 for (it = fITSubManager.begin(); it != fITSubManager.end();)
67 {
68 if (it->second) delete it->second;
69 it_tmp = it;
70 it++;
71 fITSubManager.erase(it_tmp);
72 }
73 fpInstance = 0;
74}
75
76void G4AllITFinder::UpdatePositionMap()
77{
78 std::map<G4ITType, G4VITFinder*>::iterator it = fITSubManager.begin();
79
80 for (; it != fITSubManager.end(); it++)
81 {
82 it->second->UpdatePositionMap();
83 }
84}
85
86G4VITFinder* G4AllITFinder::GetInstance(G4ITType type)
87{
88 map<G4ITType, G4VITFinder*>::iterator it = fITSubManager.find(type);
89
90 if (it == fITSubManager.end()) return 0;
91
92 return it->second;
93}
94
95void G4AllITFinder::RegisterManager(G4VITFinder* manager)
96{
97 fITSubManager[manager->GetITType()] = manager;
98}
99
100void G4AllITFinder::Push(G4Track* track)
101{
102 fITSubManager[GetIT(track)->GetITType()]->Push(track);
103}
