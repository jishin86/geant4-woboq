142// ********************************************************************
143// LocateGlobalPointAndSetup
144//
145// Locate the point in the hierarchy return 0 if outside
146// The direction is required
147// - if on an edge shared by more than two surfaces
148// (to resolve likely looping in tracking)
149// - at initial location of a particle
150// (to resolve potential ambiguity at boundary)
151//
152// Flags on exit: (comments to be completed)
153// fEntering - True if entering `daughter' volume (or replica)
154// whether daughter of last mother directly
155// or daughter of that volume's ancestor.
156// ********************************************************************
157//
158G4VPhysicalVolume*
159G4ITNavigator1::LocateGlobalPointAndSetup( const G4ThreeVector& globalPoint,
160 const G4ThreeVector* pGlobalDirection,
161 const G4bool relativeSearch,
162 const G4bool ignoreDirection )
163{
164 G4bool notKnownContained=true, noResult;
165 G4VPhysicalVolume *targetPhysical;
166 G4LogicalVolume *targetLogical;
167 G4VSolid *targetSolid=0;
168 G4ThreeVector localPoint, globalDirection;
169 EInside insideCode;
170
171 G4bool considerDirection = (!ignoreDirection) || fLocatedOnEdge;
172
173 fLastTriedStepComputation= false;
174 fChangedGrandMotherRefFrame= false; // For local exit normal
175
176 if( considerDirection && pGlobalDirection != 0 )
177 {
178 globalDirection=*pGlobalDirection;
179 }
180
181
182#ifdef G4VERBOSE
183 if( fVerbose > 2 )
184 {
185 G4int oldcoutPrec = G4cout.precision(8);
186 G4cout << "*** G4ITNavigator1::LocateGlobalPointAndSetup: ***" << G4endl;
187 G4cout << " Called with arguments: " << G4endl
188 << " Globalpoint = " << globalPoint << G4endl
189 << " RelativeSearch = " << relativeSearch << G4endl;
190 if( fVerbose == 4 )
191 {
192 G4cout << " ----- Upon entering:" << G4endl;
193 PrintState();
194 }
195 G4cout.precision(oldcoutPrec);
196 }
197#endif
198
199 if ( !relativeSearch )
200 {
201 ResetStackAndState();
202 }
203 else
204 {
205 if ( fWasLimitedByGeometry )
206 {
207 fWasLimitedByGeometry = false;
208 fEnteredDaughter = fEntering; // Remember
209 fExitedMother = fExiting; // Remember
210 if ( fExiting )
211 {
212 if ( fHistory.GetDepth() )
213 {
214 fBlockedPhysicalVolume = fHistory.GetTopVolume();
215 fBlockedReplicaNo = fHistory.GetTopReplicaNo();
216 fHistory.BackLevel();
217 }
218 else
219 {
220 fLastLocatedPointLocal = localPoint;
221 fLocatedOutsideWorld = true;
222 return 0; // Have exited world volume
223 }
224 // A fix for the case where a volume is "entered" at an edge
225 // and a coincident surface exists outside it.
226 // - This stops it from exiting further volumes and cycling
227 // - However ReplicaNavigator treats this case itself
228 //
229 if ( fLocatedOnEdge && (VolumeType(fBlockedPhysicalVolume)!=kReplica ))
230 {
231 fExiting= false;
232 }
233 }
234 else
235 if ( fEntering )
236 {
237 switch (VolumeType(fBlockedPhysicalVolume))
238 {
239 case kNormal:
240 fHistory.NewLevel(fBlockedPhysicalVolume, kNormal,
241 fBlockedPhysicalVolume->GetCopyNo());
242 break;
243 case kReplica:
244 freplicaNav.ComputeTransformation(fBlockedReplicaNo,
245 fBlockedPhysicalVolume);
246 fHistory.NewLevel(fBlockedPhysicalVolume, kReplica,
247 fBlockedReplicaNo);
248 fBlockedPhysicalVolume->SetCopyNo(fBlockedReplicaNo);
249 break;
250 case kParameterised:
251 if( fBlockedPhysicalVolume->GetRegularStructureId() == 0 )
252 {
253 G4VSolid *pSolid;
254 G4VPVParameterisation *pParam;
255 G4TouchableHistory parentTouchable( fHistory );
256 pParam = fBlockedPhysicalVolume->GetParameterisation();
257 pSolid = pParam->ComputeSolid(fBlockedReplicaNo,
258 fBlockedPhysicalVolume);
259 pSolid->ComputeDimensions(pParam, fBlockedReplicaNo,
260 fBlockedPhysicalVolume);
261 pParam->ComputeTransformation(fBlockedReplicaNo,
262 fBlockedPhysicalVolume);
263 fHistory.NewLevel(fBlockedPhysicalVolume, kParameterised,
264 fBlockedReplicaNo);
265 fBlockedPhysicalVolume->SetCopyNo(fBlockedReplicaNo);
266 //
267 // Set the correct solid and material in Logical Volume
268 //
269 G4LogicalVolume *pLogical;
270 pLogical = fBlockedPhysicalVolume->GetLogicalVolume();
271 pLogical->SetSolid( pSolid );
272 pLogical->UpdateMaterial(pParam ->
273 ComputeMaterial(fBlockedReplicaNo,
274 fBlockedPhysicalVolume,
275 &parentTouchable));
276 }
277 break;
278 }
279 fEntering = false;
280 fBlockedPhysicalVolume = 0;
281 localPoint = fHistory.GetTopTransform().TransformPoint(globalPoint);
282 notKnownContained = false;
283 }
284 }
285 else
286 {
287 fBlockedPhysicalVolume = 0;
288 fEntering = false;
289 fEnteredDaughter = false; // Full Step was not taken, did not enter
290 fExiting = false;
291 fExitedMother = false; // Full Step was not taken, did not exit
292 }
293 }
294 //
295 // Search from top of history up through geometry until
296 // containing volume found:
297 // If on
298 // o OUTSIDE - Back up level, not/no longer exiting volumes
299 // o SURFACE and EXITING - Back up level, setting new blocking no.s
300 // else
301 // o containing volume found
302 //
303 G4int noLevelsExited=0 ;
304
305 while (notKnownContained)
306 {
307 if ( fHistory.GetTopVolumeType()!=kReplica )
308 {
309 targetSolid = fHistory.GetTopVolume()->GetLogicalVolume()->GetSolid();
310 localPoint = fHistory.GetTopTransform().TransformPoint(globalPoint);
311 insideCode = targetSolid->Inside(localPoint);
312#ifdef G4VERBOSE
313 if(( fVerbose == 1 ) && ( fCheck ))
314 {
315 G4String solidResponse = "-kInside-";
316 if (insideCode == kOutside)
317 solidResponse = "-kOutside-";
318 else if (insideCode == kSurface)
319 solidResponse = "-kSurface-";
320 G4cout << "*** G4ITNavigator1::LocateGlobalPointAndSetup(): ***" << G4endl
321 << " Invoked Inside() for solid: " << targetSolid->GetName()
322 << ". Solid replied: " << solidResponse << G4endl
323 << " For local point p: " << localPoint << G4endl;
324 }
325#endif
326 }
327 else
328 {
329 insideCode = freplicaNav.BackLocate(fHistory, globalPoint, localPoint,
330 fExiting, notKnownContained);
331 // !CARE! if notKnownContained returns false then the point is within
332 // the containing placement volume of the replica(s). If insidecode
333 // will result in the history being backed up one level, then the
334 // local point returned is the point in the system of this new level
335 }
336
337
338 if ( insideCode==kOutside )
339 {
340 noLevelsExited++;
341 if ( fHistory.GetDepth() )
342 {
343 fBlockedPhysicalVolume = fHistory.GetTopVolume();
344 fBlockedReplicaNo = fHistory.GetTopReplicaNo();
345 fHistory.BackLevel();
346 fExiting = false;
347
348 if( noLevelsExited > 1 )
349 {
350 // The first transformation was done by the sub-navigator
351 //
352 const G4RotationMatrix* mRot = fBlockedPhysicalVolume->GetRotation();
353 if( mRot )
354 {
355 fGrandMotherExitNormal *= (*mRot).inverse();
356 fChangedGrandMotherRefFrame= true;
357 }
358 }
359 }
360 else
361 {
362 fLastLocatedPointLocal = localPoint;
363 fLocatedOutsideWorld = true;
364 // No extra transformation for ExitNormal - is in frame of Top Volume
365 return 0; // Have exited world volume
366 }
367 }
368 else
369 if ( insideCode==kSurface )
370 {
371 G4bool isExiting = fExiting;
372 if( (!fExiting)&&considerDirection )
373 {
374 // Figure out whether we are exiting this level's volume
375 // by using the direction
376 //
377 G4bool directionExiting = false;
378 G4ThreeVector localDirection =
379 fHistory.GetTopTransform().TransformAxis(globalDirection);
380
381 // Make sure localPoint in correct reference frame
382 // ( Was it already correct ? How ? )
383 //
384 localPoint= fHistory.GetTopTransform().TransformPoint(globalPoint);
385 if ( fHistory.GetTopVolumeType()!=kReplica )
386 {
387 G4ThreeVector normal = targetSolid->SurfaceNormal(localPoint);
388 directionExiting = normal.dot(localDirection) > 0.0;
389 isExiting = isExiting || directionExiting;
390 }
391 }
392 if( isExiting )
393 {
394 noLevelsExited++;
395 if ( fHistory.GetDepth() )
396 {
397 fBlockedPhysicalVolume = fHistory.GetTopVolume();
398 fBlockedReplicaNo = fHistory.GetTopReplicaNo();
399 fHistory.BackLevel();
400 //
401 // Still on surface but exited volume not necessarily convex
402 //
403 fValidExitNormal = false;
404
405 if( noLevelsExited > 1 )
406 {
407 // The first transformation was done by the sub-navigator
408 //
409 const G4RotationMatrix* mRot=fBlockedPhysicalVolume->GetRotation();
410 if( mRot )
411 {
412 fGrandMotherExitNormal *= (*mRot).inverse();
413 fChangedGrandMotherRefFrame= true;
414 }
415 }
416 }
417 else
418 {
419 fLastLocatedPointLocal = localPoint;
420 fLocatedOutsideWorld = true;
421 // No extra transformation for ExitNormal, is in frame of Top Vol
422 return 0; // Have exited world volume
423 }
424 }
425 else
426 {
427 notKnownContained=false;
428 }
429 }
430 else
431 {
432 notKnownContained=false;
433 }
434 } // END while (notKnownContained)
435 //
436 // Search downwards until deepest containing volume found,
437 // blocking fBlockedPhysicalVolume/BlockedReplicaNum
438 //
439 // 3 Cases:
440 //
441 // o Parameterised daughters
442 // =>Must be one G4PVParameterised daughter & voxels
443 // o Positioned daughters & voxels
444 // o Positioned daughters & no voxels
445
446 noResult = true; // noResult should be renamed to
447 // something like enteredLevel, as that is its meaning.
448 do
449 {
450 // Determine `type' of current mother volume
451 //
452 targetPhysical = fHistory.GetTopVolume();
453 if (!targetPhysical) { break; }
454 targetLogical = targetPhysical->GetLogicalVolume();
455 switch( CharacteriseDaughters(targetLogical) )
456 {
457 case kNormal:
458 if ( targetLogical->GetVoxelHeader() ) // use optimised navigation
459 {
460 noResult = fvoxelNav.LevelLocate(fHistory,
461 fBlockedPhysicalVolume,
462 fBlockedReplicaNo,
463 globalPoint,
464 pGlobalDirection,
465 considerDirection,
466 localPoint);
467 }
468 else // do not use optimised navigation
469 {
470 noResult = fnormalNav.LevelLocate(fHistory,
471 fBlockedPhysicalVolume,
472 fBlockedReplicaNo,
473 globalPoint,
474 pGlobalDirection,
475 considerDirection,
476 localPoint);
477 }
478 break;
479 case kReplica:
480 noResult = freplicaNav.LevelLocate(fHistory,
481 fBlockedPhysicalVolume,
482 fBlockedReplicaNo,
483 globalPoint,
484 pGlobalDirection,
485 considerDirection,
486 localPoint);
487 break;
488 case kParameterised:
489 if( GetDaughtersRegularStructureId(targetLogical) != 1 )
490 {
491 noResult = fparamNav.LevelLocate(fHistory,
492 fBlockedPhysicalVolume,
493 fBlockedReplicaNo,
494 globalPoint,
495 pGlobalDirection,
496 considerDirection,
497 localPoint);
498 }
499 else // Regular structure
500 {
501 noResult = fregularNav.LevelLocate(fHistory,
502 fBlockedPhysicalVolume,
503 fBlockedReplicaNo,
504 globalPoint,
505 pGlobalDirection,
506 considerDirection,
507 localPoint);
508 }
509 break;
510 }
511
512 // LevelLocate returns true if it finds a daughter volume
513 // in which globalPoint is inside (or on the surface).
514
515 if ( noResult )
516 {
517 // Entering a daughter after ascending
518 //
519 // The blocked volume is no longer valid - it was for another level
520 //
521 fBlockedPhysicalVolume = 0;
522 fBlockedReplicaNo = -1;
523
524 // fEntering should be false -- else blockedVolume is assumed good.
525 // fEnteredDaughter is used for ExitNormal
526 //
527 fEntering = false;
528 fEnteredDaughter = true;
529
530 if( fExitedMother )
531 {
532 G4VPhysicalVolume* enteredPhysical = fHistory.GetTopVolume();
533 const G4RotationMatrix* mRot = enteredPhysical->GetRotation();
534 if( mRot )
535 {
536 fGrandMotherExitNormal *= (*mRot).inverse();
537 }
538 }
539
540#ifdef G4DEBUG_NAVIGATION
541 if( fVerbose > 2 )
542 {
543 G4VPhysicalVolume* enteredPhysical = fHistory.GetTopVolume();
544 G4cout << "*** G4ITNavigator1::LocateGlobalPointAndSetup() ***" << G4endl;
545 G4cout << " Entering volume: " << enteredPhysical->GetName()
546 << G4endl;
547 }
548#endif
549 }
550 } while (noResult);
551
552 fLastLocatedPointLocal = localPoint;
553
554#ifdef G4VERBOSE
555 if( fVerbose >= 4 )
556 {
557 G4int oldcoutPrec = G4cout.precision(8);
558 G4String curPhysVol_Name("None");
559 if (targetPhysical) { curPhysVol_Name = targetPhysical->GetName(); }
560 G4cout << " Return value = new volume = " << curPhysVol_Name << G4endl;
561 G4cout << " ----- Upon exiting:" << G4endl;
562 PrintState();
563 if( fVerbose == 5 )
564 {
565 G4cout << "Upon exiting LocateGlobalPointAndSetup():" << G4endl;
566 G4cout << " History = " << G4endl << fHistory << G4endl << G4endl;
567 }
568 G4cout.precision(oldcoutPrec);
569 }
570#endif
571
572 fLocatedOutsideWorld= false;
573
574 return targetPhysical;
575}

577// ********************************************************************
578// LocateGlobalPointWithinVolume
579//
580// -> the state information of this Navigator and its subNavigators
581// is updated in order to start the next step at pGlobalpoint
582// -> no check is performed whether pGlobalpoint is inside the
583// original volume (this must be the case).
584//
585// Note: a direction could be added to the arguments, to aid in future
586// optional checking (via the old code below, flagged by OLD_LOCATE).
587// [ This would be done only in verbose mode ]
588// ********************************************************************
589//
590void
591G4ITNavigator1::LocateGlobalPointWithinVolume(const G4ThreeVector& pGlobalpoint)
592{
593 fLastLocatedPointLocal = ComputeLocalPoint(pGlobalpoint);
594 fLastTriedStepComputation= false;
595 fChangedGrandMotherRefFrame= false; // Frame for Exit Normal
596
597#ifdef G4DEBUG_NAVIGATION
598 if( fVerbose > 2 )
599 {
600 G4cout << "Entering LocateGlobalWithinVolume(): History = " << G4endl;
601 G4cout << fHistory << G4endl;
602 }
603#endif
604
605 // For the case of Voxel (or Parameterised) volume the respective
606 // Navigator must be messaged to update its voxel information etc
607
608 // Update the state of the Sub Navigators
609 // - in particular any voxel information they store/cache
610 //
611 G4VPhysicalVolume* motherPhysical = fHistory.GetTopVolume();
612 G4LogicalVolume* motherLogical = motherPhysical->GetLogicalVolume();
613 G4SmartVoxelHeader* pVoxelHeader = motherLogical->GetVoxelHeader();
614
615 if ( fHistory.GetTopVolumeType()!=kReplica )
616 {
617 switch( CharacteriseDaughters(motherLogical) )
618 {
619 case kNormal:
620 if ( pVoxelHeader )
621 {
622 fvoxelNav.VoxelLocate( pVoxelHeader, fLastLocatedPointLocal );
623 }
624 break;
625 case kParameterised:
626 if( GetDaughtersRegularStructureId(motherLogical) != 1 )
627 {
628 // Resets state & returns voxel node
629 //
630 fparamNav.ParamVoxelLocate( pVoxelHeader, fLastLocatedPointLocal );
631 }
632 break;
633 case kReplica:
634 G4Exception("G4ITNavigator1::LocateGlobalPointWithinVolume()",
635 "GeomNav0001", FatalException,
636 "Not applicable for replicated volumes.");
637 break;
638 }
639 }
640
641 // Reset the state variables
642 // - which would have been affected
643 // by the 'equivalent' call to LocateGlobalPointAndSetup
644 // - who's values have been invalidated by the 'move'.
645 //
646 fBlockedPhysicalVolume = 0;
647 fBlockedReplicaNo = -1;
648 fEntering = false;
649 fEnteredDaughter = false; // Boundary not encountered, did not enter
650 fExiting = false;
651 fExitedMother = false; // Boundary not encountered, did not exit
652}

753// <!

755// ********************************************************************
756// ComputeStep
757//
758// Computes the next geometric Step: intersections with current
759// mother and `daughter' volumes.
760//
761// NOTE:
762//
763// Flags on entry:
764// --------------
765// fValidExitNormal - Normal of exited volume is valid (convex, not a
766// coincident boundary)
767// fExitNormal - Surface normal of exited volume
768// fExiting - True if have exited solid
769//
770// fBlockedPhysicalVolume - Ptr to exited volume (or 0)
771// fBlockedReplicaNo - Replication no of exited volume
772// fLastStepWasZero - True if last Step size was zero.
773//
774// Flags on exit:
775// -------------
776// fValidExitNormal - True if surface normal of exited volume is valid
777// fExitNormal - Surface normal of exited volume rotated to mothers
778// reference system
779// fExiting - True if exiting mother
780// fEntering - True if entering `daughter' volume (or replica)
781// fBlockedPhysicalVolume - Ptr to candidate (entered) volume
782// fBlockedReplicaNo - Replication no of candidate (entered) volume
783// fLastStepWasZero - True if this Step size was zero.
784// ********************************************************************
785//
786G4double G4ITNavigator1::ComputeStep( const G4ThreeVector &pGlobalpoint,
787 const G4ThreeVector &pDirection,
788 const G4double pCurrentProposedStepLength,
789 G4double &pNewSafety)
790{
791 G4ThreeVector localDirection = ComputeLocalAxis(pDirection);
792 G4double Step = kInfinity;
793 G4VPhysicalVolume *motherPhysical = fHistory.GetTopVolume();
794 G4LogicalVolume *motherLogical = motherPhysical->GetLogicalVolume();
795
796 // All state relating to exiting normals must be reset
797 //
798 fExitNormalGlobalFrame= G4ThreeVector( 0., 0., 0.);
799 // Reset value - to erase its memory
800 fChangedGrandMotherRefFrame= false;
801 // Reset - used for local exit normal
802 fGrandMotherExitNormal= G4ThreeVector( 0., 0., 0.);
803 fCalculatedExitNormal = false;
804 // Reset for new step
805
806 static G4ThreadLocal G4int sNavCScalls=0;
807 sNavCScalls++;
808
809 fLastTriedStepComputation= true;
810
811#ifdef G4VERBOSE
812 if( fVerbose > 0 )
813 {
814 G4cout << "*** G4ITNavigator1::ComputeStep: ***" << G4endl;
815 G4cout << " Volume = " << motherPhysical->GetName()
816 << " - Proposed step length = " << pCurrentProposedStepLength
817 << G4endl;
818#ifdef G4DEBUG_NAVIGATION
819 if( fVerbose >= 2 )
820 {
821 G4cout << " Called with the arguments: " << G4endl
822 << " Globalpoint = " << std::setw(25) << pGlobalpoint << G4endl
823 << " Direction = " << std::setw(25) << pDirection << G4endl;
824 if( fVerbose >= 4 )
825 {
826 G4cout << " ---- Upon entering : State" << G4endl;
827 PrintState();
828 }
829 }
830#endif
831 }
832#endif
833
834 G4ThreeVector newLocalPoint = ComputeLocalPoint(pGlobalpoint);
835 if( newLocalPoint != fLastLocatedPointLocal )
836 {
837 // Check whether the relocation is within safety
838 //
839 G4ThreeVector oldLocalPoint = fLastLocatedPointLocal;
840 G4double moveLenSq = (newLocalPoint-oldLocalPoint).mag2();
841
842 if ( moveLenSq >= kCarTolerance*kCarTolerance )
843 {
844#ifdef G4VERBOSE
845 ComputeStepLog(pGlobalpoint, moveLenSq);
846#endif
847 // Relocate the point within the same volume
848 //
849 LocateGlobalPointWithinVolume( pGlobalpoint );
850 fLastTriedStepComputation= true; // Ensure that this is set again !!
851 }
852 }
853 if ( fHistory.GetTopVolumeType()!=kReplica )
854 {
855 switch( CharacteriseDaughters(motherLogical) )
856 {
857 case kNormal:
858 if ( motherLogical->GetVoxelHeader() )
859 {
860 Step = fvoxelNav.ComputeStep(fLastLocatedPointLocal,
861 localDirection,
862 pCurrentProposedStepLength,
863 pNewSafety,
864 fHistory,
865 fValidExitNormal,
866 fExitNormal,
867 fExiting,
868 fEntering,
869 &fBlockedPhysicalVolume,
870 fBlockedReplicaNo);
871
872 }
873 else
874 {
875 if( motherPhysical->GetRegularStructureId() == 0 )
876 {
877 Step = fnormalNav.ComputeStep(fLastLocatedPointLocal,
878 localDirection,
879 pCurrentProposedStepLength,
880 pNewSafety,
881 fHistory,
882 fValidExitNormal,
883 fExitNormal,
884 fExiting,
885 fEntering,
886 &fBlockedPhysicalVolume,
887 fBlockedReplicaNo);
888 }
889 else // Regular (non-voxelised) structure
890 {
891 LocateGlobalPointAndSetup( pGlobalpoint, &pDirection, true, true );
892 fLastTriedStepComputation= true; // Ensure that this is set again !!
893 //
894 // if physical process limits the step, the voxel will not be the
895 // one given by ComputeStepSkippingEqualMaterials() and the local
896 // point will be wrongly calculated.
897
898 // There is a problem: when msc limits the step and the point is
899 // assigned wrongly to phantom in previous step (while it is out
900 // of the container volume). Then LocateGlobalPointAndSetup() has
901 // reset the history topvolume to world.
902 //
903 if(fHistory.GetTopVolume()->GetRegularStructureId() == 0 )
904 {
905 G4Exception("G4ITNavigator1::ComputeStep()",
906 "GeomNav1001", JustWarning,
907 "Point is relocated in voxels, while it should be outside!");
908 Step = fnormalNav.ComputeStep(fLastLocatedPointLocal,
909 localDirection,
910 pCurrentProposedStepLength,
911 pNewSafety,
912 fHistory,
913 fValidExitNormal,
914 fExitNormal,
915 fExiting,
916 fEntering,
917 &fBlockedPhysicalVolume,
918 fBlockedReplicaNo);
919 }
920 else
921 {
922 Step = fregularNav.
923 ComputeStepSkippingEqualMaterials(fLastLocatedPointLocal,
924 localDirection,
925 pCurrentProposedStepLength,
926 pNewSafety,
927 fHistory,
928 fValidExitNormal,
929 fExitNormal,
930 fExiting,
931 fEntering,
932 &fBlockedPhysicalVolume,
933 fBlockedReplicaNo,
934 motherPhysical);
935 }
936 }
937 }
938 break;
939 case kParameterised:
940 if( GetDaughtersRegularStructureId(motherLogical) != 1 )
941 {
942 Step = fparamNav.ComputeStep(fLastLocatedPointLocal,
943 localDirection,
944 pCurrentProposedStepLength,
945 pNewSafety,
946 fHistory,
947 fValidExitNormal,
948 fExitNormal,
949 fExiting,
950 fEntering,
951 &fBlockedPhysicalVolume,
952 fBlockedReplicaNo);
953 }
954 else // Regular structure
955 {
956 Step = fregularNav.ComputeStep(fLastLocatedPointLocal,
957 localDirection,
958 pCurrentProposedStepLength,
959 pNewSafety,
960 fHistory,
961 fValidExitNormal,
962 fExitNormal,
963 fExiting,
964 fEntering,
965 &fBlockedPhysicalVolume,
966 fBlockedReplicaNo);
967 }
968 break;
969 case kReplica:
970 G4Exception("G4ITNavigator1::ComputeStep()", "GeomNav0001",
971 FatalException, "Not applicable for replicated volumes.");
972 break;
973 }
974 }
975 else
976 {
977 // In the case of a replica, it must handle the exiting
978 // edge/corner problem by itself
979 //
980 G4bool exitingReplica = fExitedMother;
981 G4bool calculatedExitNormal= false;
982
983 Step = freplicaNav.ComputeStep(pGlobalpoint,
984 pDirection,
985 fLastLocatedPointLocal,
986 localDirection,
987 pCurrentProposedStepLength,
988 pNewSafety,
989 fHistory,
990 fValidExitNormal,
991 calculatedExitNormal,
992 fExitNormal,
993 exitingReplica,
994 fEntering,
995 &fBlockedPhysicalVolume,
996 fBlockedReplicaNo);
997 fExiting= exitingReplica;
998 fCalculatedExitNormal= calculatedExitNormal;
999 }
1000
1001 // Remember last safety origin & value.
1002 //
1003 fPreviousSftOrigin = pGlobalpoint;
1004 fPreviousSafety = pNewSafety;
1005
1006 // Count zero steps - one can occur due to changing momentum at a boundary
1007 // - one, two (or a few) can occur at common edges between
1008 // volumes
1009 // - more than two is likely a problem in the geometry
1010 // description or the Navigation
1011
1012 // Rule of thumb: likely at an Edge if two consecutive steps are zero,
1013 // because at least two candidate volumes must have been
1014 // checked
1015 //
1016 fLocatedOnEdge = fLastStepWasZero && (Step==0.0);
1017 fLastStepWasZero = (Step==0.0);
1018 if (fPushed) { fPushed = fLastStepWasZero; }
1019
1020 // Handle large number of consecutive zero steps
1021 //
1022 if ( fLastStepWasZero )
1023 {
1024 fNumberZeroSteps++;
1025#ifdef G4DEBUG_NAVIGATION
1026 if( fNumberZeroSteps > 1 )
1027 {
1028 G4cout << "G4ITNavigator1::ComputeStep(): another zero step, # "
1029 << fNumberZeroSteps
1030 << " at " << pGlobalpoint
1031 << " in volume " << motherPhysical->GetName()
1032 << " nav-comp-step calls # " << sNavCScalls
1033 << G4endl;
1034 }
1035#endif
1036 if( fNumberZeroSteps > fActionThreshold_NoZeroSteps-1 )
1037 {
1038 // Act to recover this stuck track. Pushing it along direction
1039 //
1040 Step += 100*kCarTolerance;
1041#ifdef G4VERBOSE
1042 if ((!fPushed) && (fWarnPush))
1043 {
1044 std::ostringstream message;
1045 message << "Track stuck or not moving." << G4endl
1046 << " Track stuck, not moving for "
1047 << fNumberZeroSteps << " steps" << G4endl
1048 << " in volume -" << motherPhysical->GetName()
1049 << "- at point " << pGlobalpoint << G4endl
1050 << " direction: " << pDirection << "." << G4endl
1051 << " Potential geometry or navigation problem !"
1052 << G4endl
1053 << " Trying pushing it of " << Step << " mm ...";
1054 G4Exception("G4ITNavigator1::ComputeStep()", "GeomNav1002",
1055 JustWarning, message, "Potential overlap in geometry!");
1056 }
1057#endif
1058 fPushed = true;
1059 }
1060 if( fNumberZeroSteps > fAbandonThreshold_NoZeroSteps-1 )
1061 {
1062 // Must kill this stuck track
1063 //
1064 std::ostringstream message;
1065 message << "Stuck Track: potential geometry or navigation problem."
1066 << G4endl
1067 << " Track stuck, not moving for "
1068 << fNumberZeroSteps << " steps" << G4endl
1069 << " in volume -" << motherPhysical->GetName()
1070 << "- at point " << pGlobalpoint << G4endl
1071 << " direction: " << pDirection << ".";
1072 motherPhysical->CheckOverlaps(5000, false);
1073 G4Exception("G4ITNavigator1::ComputeStep()", "GeomNav0003",
1074 EventMustBeAborted, message);
1075 }
1076 }
1077 else
1078 {
1079 if (!fPushed) fNumberZeroSteps = 0;
1080 }
1081
1082 fEnteredDaughter = fEntering; // I expect to enter a volume in this Step
1083 fExitedMother = fExiting;
1084
1085 fStepEndPoint = pGlobalpoint + Step * pDirection;
1086 fLastStepEndPointLocal = fLastLocatedPointLocal + Step * localDirection;
1087
1088 if( fExiting )
1089 {
1090#ifdef G4DEBUG_NAVIGATION
1091 if( fVerbose > 2 )
1092 {
1093 G4cout << " At G4Nav CompStep End - if(exiting) - fExiting= " << fExiting
1094 << " fValidExitNormal = " << fValidExitNormal << G4endl;
1095 G4cout << " fExitNormal= " << fExitNormal << G4endl;
1096 }
1097#endif
1098
1099 if(fValidExitNormal || fCalculatedExitNormal)
1100 {
1101 if ( fHistory.GetTopVolumeType()!=kReplica )
1102 {
1103 // Convention: fExitNormal is in the 'grand-mother' coordinate system
1104 //
1105 fGrandMotherExitNormal= fExitNormal;
1106 fCalculatedExitNormal= true;
1107 }
1108 else
1109 {
1110 fGrandMotherExitNormal = fExitNormal;
1111 }
1112 }
1113 else
1114 {
1115 // We must calculate the normal anyway (in order to have it if requested)
1116 //
1117 G4ThreeVector finalLocalPoint =
1118 fLastLocatedPointLocal + localDirection*Step;
1119
1120 if ( fHistory.GetTopVolumeType()!=kReplica )
1121 {
1122 // Find normal in the 'mother' coordinate system
1123 //
1124 G4ThreeVector exitNormalMotherFrame=
1125 motherLogical->GetSolid()->SurfaceNormal(finalLocalPoint);
1126
1127 // Transform it to the 'grand-mother' coordinate system
1128 //
1129 const G4RotationMatrix* mRot = motherPhysical->GetRotation();
1130 if( mRot )
1131 {
1132 fChangedGrandMotherRefFrame= true;
1133 fGrandMotherExitNormal = (*mRot).inverse() * exitNormalMotherFrame;
1134 }
1135 else
1136 {
1137 fGrandMotherExitNormal = exitNormalMotherFrame;
1138 }
1139
1140 // Do not set fValidExitNormal -- this signifies
1141 // that the solid is convex!
1142 //
1143 fCalculatedExitNormal= true;
1144 }
1145 else
1146 {
1147 fCalculatedExitNormal = false;
1148 //
1149 // Nothing can be done at this stage currently - to solve this
1150 // Replica Navigation must have calculated the normal for this case
1151 // already.
1152 // Cases: mother is not convex, and exit is at previous replica level
1153
1154#ifdef G4DEBUG_NAVIGATION
1155 G4ExceptionDescription desc;
1156
1157 desc << "Problem in ComputeStep: Replica Navigation did not provide"
1158 << " valid exit Normal. " << G4endl;
1159 desc << " Do not know how calculate it in this case." << G4endl;
1160 desc << " Location = " << finalLocalPoint << G4endl;
1161 desc << " Volume name = " << motherPhysical->GetName()
1162 << " copy/replica No = " << motherPhysical->GetCopyNo() << G4endl;
1163 G4Exception("G4ITNavigator1::ComputeStep()", "GeomNav0003",
1164 JustWarning, desc, "Normal not available for exiting.");
1165#endif
1166 }
1167 }
1168
1169 // Now transform it to the global reference frame !!
1170 //
1171 if( fValidExitNormal || fCalculatedExitNormal )
1172 {
1173 G4int depth= fHistory.GetDepth();
1174 if( depth > 0 )
1175 {
1176 G4AffineTransform GrandMotherToGlobalTransf =
1177 fHistory.GetTransform(depth-1).Inverse();
1178 fExitNormalGlobalFrame =
1179 GrandMotherToGlobalTransf.TransformAxis( fGrandMotherExitNormal );
1180 }
1181 else
1182 {
1183 fExitNormalGlobalFrame= fGrandMotherExitNormal;
1184 }
1185 }
1186 else
1187 {
1188 fExitNormalGlobalFrame= G4ThreeVector( 0., 0., 0.);
1189 }
1190 }
1191 fStepEndPoint= pGlobalpoint+Step*pDirection;
1192
1193 if( (Step == pCurrentProposedStepLength) && (!fExiting) && (!fEntering) )
1194 {
1195 // This if Step is not really limited by the geometry.
1196 // The Navigator is obliged to return "infinity"
1197 //
1198 Step = kInfinity;
1199 }
1200
1201#ifdef G4VERBOSE
1202 if( fVerbose > 1 )
1203 {
1204 if( fVerbose >= 4 )
1205 {
1206 G4cout << " ----- Upon exiting :" << G4endl;
1207 PrintState();
1208 }
1209 G4cout << " Returned step= " << Step;
1210 if( fVerbose > 5 ) G4cout << G4endl;
1211 if( Step == kInfinity )
1212 {
1213 G4cout << " Requested step= " << pCurrentProposedStepLength ;
1214 if( fVerbose > 5) G4cout << G4endl;
1215 }
1216 G4cout << " Safety = " << pNewSafety << G4endl;
1217 }
1218#endif
1219
1220 return Step;
1221}

1337 }

1340// ********************************************************************
1341// GetLocalExitNormal
1342//
1343// Obtains the Normal vector to a surface (in local coordinates)
1344// pointing out of previous volume and into current volume
1345// ********************************************************************
1346//
1347G4ThreeVector G4ITNavigator1::GetLocalExitNormal( G4bool* valid )
1348{
1349 G4ThreeVector ExitNormal(0.,0.,0.);
1350 G4VSolid *currentSolid=0;
1351 G4LogicalVolume *candidateLogical;
1352 if ( fLastTriedStepComputation )
1353 {
1354 // use fLastLocatedPointLocal and next candidate volume
1355 //
1356 G4ThreeVector nextSolidExitNormal(0.,0.,0.);
1357
1358 if( fEntering && (fBlockedPhysicalVolume!=0) )
1359 {
1360 candidateLogical= fBlockedPhysicalVolume->GetLogicalVolume();
1361 if( candidateLogical )
1362 {
1363 // fLastStepEndPointLocal is in the coordinates of the mother
1364 // we need it in the daughter's coordinate system.
1365
1366 // The following code should also work in case of Replica
1367 {
1368 // First transform fLastLocatedPointLocal to the new daughter
1369 // coordinates
1370 //
1371 G4AffineTransform MotherToDaughterTransform=
1372 GetMotherToDaughterTransform( fBlockedPhysicalVolume,
1373 fBlockedReplicaNo,
1374 VolumeType(fBlockedPhysicalVolume) );
1375 G4ThreeVector daughterPointOwnLocal=
1376 MotherToDaughterTransform.TransformPoint( fLastStepEndPointLocal );
1377
1378 // OK if it is a parameterised volume
1379 //
1380 EInside inSideIt;
1381 G4bool onSurface;
1382 G4double safety= -1.0;
1383 currentSolid= candidateLogical->GetSolid();
1384 inSideIt = currentSolid->Inside(daughterPointOwnLocal);
1385 onSurface = (inSideIt == kSurface);
1386 if( ! onSurface )
1387 {
1388 if( inSideIt == kOutside )
1389 {
1390 safety = (currentSolid->DistanceToIn(daughterPointOwnLocal));
1391 onSurface = safety < 100.0 * kCarTolerance;
1392 }
1393 else if (inSideIt == kInside )
1394 {
1395 safety = (currentSolid->DistanceToOut(daughterPointOwnLocal));
1396 onSurface = safety < 100.0 * kCarTolerance;
1397 }
1398 }
1399
1400 if( onSurface )
1401 {
1402 nextSolidExitNormal =
1403 currentSolid->SurfaceNormal(daughterPointOwnLocal);
1404
1405 // Entering the solid ==> opposite
1406 //
1407 ExitNormal = -nextSolidExitNormal;
1408 fCalculatedExitNormal= true;
1409 }
1410 else
1411 {
1412#ifdef G4VERBOSE
1413 if(( fVerbose == 1 ) && ( fCheck ))
1414 {
1415 std::ostringstream message;
1416 message << "Point not on surface ! " << G4endl
1417 << " Point = "
1418 << daughterPointOwnLocal << G4endl
1419 << " Physical volume = "
1420 << fBlockedPhysicalVolume->GetName() << G4endl
1421 << " Logical volume = "
1422 << candidateLogical->GetName() << G4endl
1423 << " Solid = " << currentSolid->GetName()
1424 << " Type = "
1425 << currentSolid->GetEntityType() << G4endl
1426 << *currentSolid << G4endl;
1427 if( inSideIt == kOutside )
1428 {
1429 message << "Point is Outside. " << G4endl
1430 << " Safety (from outside) = " << safety << G4endl;
1431 }
1432 else // if( inSideIt == kInside )
1433 {
1434 message << "Point is Inside. " << G4endl
1435 << " Safety (from inside) = " << safety << G4endl;
1436 }
1437 G4Exception("G4ITNavigator1::GetLocalExitNormal()", "GeomNav1001",
1438 JustWarning, message);
1439 }
1440#endif
1441 }
1442 *valid = onSurface; // was =true;
1443 }
1444 }
1445 }
1446 else if ( fExiting )
1447 {
1448 ExitNormal = fGrandMotherExitNormal;
1449 *valid = true;
1450 fCalculatedExitNormal= true; // Should be true already
1451 }
1452 else // i.e. ( fBlockedPhysicalVolume == 0 )
1453 {
1454 *valid = false;
1455 G4Exception("G4ITNavigator1::GetLocalExitNormal()",
1456 "GeomNav0003", JustWarning,
1457 "Incorrect call to GetLocalSurfaceNormal." );
1458 }
1459 }
1460 else // ( ! fLastTriedStepComputation ) ie. last call was to Locate
1461 {
1462 if ( EnteredDaughterVolume() )
1463 {
1464 G4VSolid* daughterSolid =fHistory.GetTopVolume()->GetLogicalVolume()
1465 ->GetSolid();
1466 ExitNormal= -(daughterSolid->SurfaceNormal(fLastLocatedPointLocal));
1467 if( std::fabs(ExitNormal.mag2()-1.0 ) > CLHEP::perMillion )
1468 {
1469 G4ExceptionDescription desc;
1470 desc << " Parameters of solid: " << *daughterSolid
1471 << " Point for surface = " << fLastLocatedPointLocal << std::endl;
1472 G4Exception("G4ITNavigator1::GetLocalExitNormal()",
1473 "GeomNav0003", FatalException, desc,
1474 "Surface Normal returned by Solid is not a Unit Vector." );
1475 }
1476 fCalculatedExitNormal= true;
1477 *valid = true;
1478 }
1479 else
1480 {
1481 if( fExitedMother )
1482 {
1483 ExitNormal = fGrandMotherExitNormal;
1484 *valid = true;
1485 fCalculatedExitNormal= true;
1486 }
1487 else // We are not at a boundary. ExitNormal remains (0,0,0)
1488 {
1489 *valid = false;
1490 fCalculatedExitNormal= false;
1491 G4ExceptionDescription message;
1492 message << "Function called when *NOT* at a Boundary." << G4endl;
1493 G4Exception("G4ITNavigator1::GetLocalExitNormal()",
1494 "GeomNav0003", JustWarning, message);
1495 }
1496 }
1497 }
1498 return ExitNormal;
1499}

1501// ********************************************************************

1503//

1505// ********************************************************************

1507G4AffineTransform
1508G4ITNavigator1::GetMotherToDaughterTransform( G4VPhysicalVolume *pEnteringPhysVol, // not Const
1509 G4int enteringReplicaNo,
1510 EVolume enteringVolumeType )
1511{
