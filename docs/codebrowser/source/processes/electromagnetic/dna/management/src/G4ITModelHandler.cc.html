38G4ITModelHandler::G4ITModelHandler()
39{
40 //ctor
41 fIsInitialized = false;
42 fTimeStepComputerFlag = false;
43 fReactionProcessFlag = false;
44
45 size_t IT_size(G4ITType::size());
46
47 fModelManager.assign(IT_size, std::vector<G4ITModelManager*>());
48 for (G4int i = 0; i < (int) IT_size; i++)
49 {
50 fModelManager[i].assign(IT_size, 0);
51 }
52}
53
54G4ITModelHandler::~G4ITModelHandler()
55{
56 //dtor
57 G4int size = fModelManager.size();
58
59 for (G4int i = 0; i < size; i++)
60 {
61 for (G4int j = 0; j <= i; j++)
62 {
63 if (fModelManager[i][j])
64 {
65 delete fModelManager[i][j];
66 fModelManager[i][j] = 0;
67 fModelManager[j][i] = 0;
68 }
69 }
70 }
71 fModelManager.clear();
72}
73
74G4ITModelHandler::G4ITModelHandler(const G4ITModelHandler& other)
75{
76 //copy ctor
77 size_t IT_size(G4ITType::size());
78
79 fModelManager.assign(IT_size, std::vector<G4ITModelManager*>());
80 for (int i = 0; i < (int) IT_size; i++)
81 {
82 fModelManager[i].assign(IT_size, 0);
83 for (int j = 0; j < (int) IT_size; j++)
84 {
85 if (other.fModelManager[i][j] != 0)
86 {
87 fModelManager[i][j] = new G4ITModelManager(
88 *(other.fModelManager[i][j]));
89 }
90 }
91 }
92
93 fIsInitialized = other.fIsInitialized;
94 fTimeStepComputerFlag = other.fTimeStepComputerFlag;
95 fReactionProcessFlag = other.fReactionProcessFlag;
96}

100 if (this == &rhs) return *this; // handle self assignment

105void G4ITModelHandler::Initialize()
106{
107 fIsInitialized = true;
108
109 for (G4int i = 0; i < int(fModelManager.size()); i++)
110 {
111 for (G4int j = 0; j <= i; j++)
112 {
113 G4ITModelManager* modman = fModelManager[i][j];
114 if (modman)
115 {
116 modman->Initialize();
117 }
118 }
119 }
120}

122void G4ITModelHandler::RegisterModel(G4VITStepModel* aModel,
123 G4double startingTime)
124{
125 assert(aModel != 0);
126
127 //________________________________________________
128 // Prepare the correct model manager
129 if (fModelManager.empty())
130 {
131 size_t IT_size(G4ITType::size());
132 fModelManager.assign(IT_size, std::vector<G4ITModelManager*>());
133
134 for (int i = 0; i < (int) IT_size; i++)
135 {
136 fModelManager[i].assign((size_t) i, 0);
137 }
138 }
139
140 G4ITType type1;
141 G4ITType type2;
142
143 //________________________________________________
144 // Retrieve applicability
145 aModel->IsApplicable(type1, type2);
146
147 if (type1 > type2)
148 {
149 G4ITType buffer(-1);
150 buffer = type1;
151 type1 = type2;
152 type2 = buffer;
153 }
154
155 if (fModelManager[type1][type2] == 0)
156 {
157 fModelManager[type1][type2] = new G4ITModelManager();
158 }
159
160 fModelManager[type1][type2]->SetModel(aModel, startingTime);
161
162 //________________________________________________
163 // Setup ITStepManager
164 if (aModel->GetTimeStepper())
165 {
166 fTimeStepComputerFlag = true;
167 }
168 if (aModel->GetReactionProcess())
169 {
170 fReactionProcessFlag = true;
171 }
172}

174void G4ITModelHandler::SetModel(G4ITType type1,
175 G4ITType type2,
176 G4VITStepModel* aModel,
177 G4double startingTime)
178{
179 assert(aModel == 0);
180
181 if (type1 > type2)
182 {
183 G4ITType buffer(-1);
184 buffer = type1;
185 type1 = type2;
186 type2 = buffer;
187 }
188
189 if (type1 > (int) fModelManager.capacity())
190 {
191 fModelManager.reserve(type1);
192 }
193
194 if (type2 > (int) fModelManager[type1].capacity())
195 {
196 fModelManager[type1].reserve(type2);
197 }
198
199 fModelManager[type1][type2]->SetModel(aModel, startingTime);
200}

202G4VITStepModel* G4ITModelHandler::GetModel(G4ITType type1,
203 G4ITType type2,
204 const G4double globalTime)
205{
206 if (fModelManager.empty())
207 {
208 return 0;
209 }
210
211 if ((int) fModelManager.size() < type1) return 0;
212
213 std::vector<G4ITModelManager*>* v = &(fModelManager.at(type1));
214
215 if ((int) v->size() < type2) return 0;
216
217 if (v->at(type2))
218 {
219 return v->at(type2)->GetModel(globalTime);
220 }
221 return 0;
222}
