41G4ITModelManager::G4ITModelManager() :
42 fIsInitialized(FALSE)
43{
44 ;
45}
46
47G4ITModelManager::~G4ITModelManager()
48{
49 //dtor
50 mapModels::iterator it;
51
52 for (it = fModels.begin(); it != fModels.end(); it++)
53 {
54 delete it->second;
55 }
56 fModels.clear();
57}
58
59G4ITModelManager::G4ITModelManager(const G4ITModelManager& right)
60{
61 mapModels::const_iterator it = right.fModels.begin();
62
63 for (; it != right.fModels.end(); it++)
64 {
65 fModels[it->first] = it->second->Clone();
66 }
67
68 fIsInitialized = right.fIsInitialized;
69}
70
71G4ITModelManager& G4ITModelManager::operator=(const G4ITModelManager& rhs)
72{
73 if (this == &rhs) return *this; // handle self assignment
74 //assignment operator
75 return *this;
76}
77
78void G4ITModelManager::Initialize()
79{
80 mapModels::iterator it = fModels.begin();
81
82 for (; it != fModels.end(); it++)
83 {
84 G4VITStepModel* model = it->second;
85 if (model != 0)
86 {
87 model->Initialize();
88 }
89 }
90}
91
92void G4ITModelManager::SetModel(G4VITStepModel* aModel, G4double startingTime)
93{
94 assert(fIsInitialized == FALSE);
95 if (fIsInitialized == true)
96 {
97 G4ExceptionDescription exceptionDescription;
98 exceptionDescription
99 << "You are trying to insert a new model after initialization of th model manager.";
100 G4Exception("G4ITModelManager::SetModel", "ITModelManager001",
101 FatalErrorInArgument, exceptionDescription);
102 }
103 fModels[startingTime] = aModel;
104}
105
106G4VITStepModel* G4ITModelManager::GetModel(const G4double globalTime)
107{
108 if (!fModels.empty())
109 {
110 mapModels::reverse_iterator rit = fModels.rbegin();
111 if (rit != fModels.rend())
112 {
113 if (globalTime > rit->first)
114 {
115 return rit->second;
116 }
117 else
118 {
119 mapModels::iterator it = fModels.begin();
120
121 if (globalTime < it->first)
122 {
123 G4ExceptionDescription exceptionDescription;
124 exceptionDescription << "No model was found at time ";
125 exceptionDescription << G4BestUnit(globalTime, "Time");
126 exceptionDescription << ". The first model is registered at time : ";
127 exceptionDescription << G4BestUnit(it->first, "Time") << ". ";
128 G4Exception("G4ITModelManager::GetModel", "ITModelManager003",
129 FatalErrorInArgument, exceptionDescription);
130 }
131
132 it = fModels.lower_bound(globalTime);
133
134 if (it != fModels.end()) return it->second;
135 }
136 }
137 }
138
139 G4ExceptionDescription exceptionDescription;
140 exceptionDescription << "No model was found.";
141 G4Exception("G4ITModelManager::GetModel", "ITModelManager004",
142 FatalErrorInArgument, exceptionDescription);
143 return 0;
144}
