141// ********************************************************************
142// LocateGlobalPointAndSetup
143//
144// Locate the point in the hierarchy return 0 if outside
145// The direction is required
146// - if on an edge shared by more than two surfaces
147// (to resolve likely looping in tracking)
148// - at initial location of a particle
149// (to resolve potential ambiguity at boundary)
150//
151// Flags on exit: (comments to be completed)
152// fEntering - True if entering `daughter' volume (or replica)
153// whether daughter of last mother directly
154// or daughter of that volume's ancestor.
155// ********************************************************************
156//
157G4VPhysicalVolume*
158G4ITNavigator2::LocateGlobalPointAndSetup( const G4ThreeVector& globalPoint,
159 const G4ThreeVector* pGlobalDirection,
160 const G4bool relativeSearch,
161 const G4bool ignoreDirection )
162{
163 CheckNavigatorStateIsValid();
164 G4bool notKnownContained=true, noResult;
165 G4VPhysicalVolume *targetPhysical;
166 G4LogicalVolume *targetLogical;
167 G4VSolid *targetSolid=0;
168 G4ThreeVector localPoint, globalDirection;
169 EInside insideCode;
170
171 G4bool considerDirection = (!ignoreDirection) || fLocatedOnEdge;
172
173 fLastTriedStepComputation= false;
174 fChangedGrandMotherRefFrame= false; // For local exit normal
175
176 if( considerDirection && pGlobalDirection != 0 )
177 {
178 globalDirection=*pGlobalDirection;
179 }
180
181
182#ifdef G4VERBOSE
183 if( fVerbose > 2 )
184 {
185 static int nCalls = 0;
186 G4int oldcoutPrec = G4cout.precision(8);
187 G4cout << "*** G4ITNavigator2::LocateGlobalPointAndSetup: ***" << G4endl;
188 G4cout << " Called with arguments: " << G4endl
189 << " Globalpoint = " << globalPoint << G4endl
190 << " RelativeSearch = " << relativeSearch << G4endl;
191 if( fVerbose == 4 )
192 {
193 G4cout << " ----- Upon entering:" << G4endl;
194 PrintState();
195 }
196 G4cout.precision(oldcoutPrec);
197 nCalls++;
198// if(nCalls == 2) abort();
199 }
200#endif
201
202 G4int noLevelsExited=0 ;
203 G4int noLevelsEntered= 0;
204
205 if ( !relativeSearch )
206 {
207 ResetStackAndState();
208 }
209 else
210 {
211 if ( fWasLimitedByGeometry )
212 {
213 fWasLimitedByGeometry = false;
214 fEnteredDaughter = fEntering; // Remember
215 fExitedMother = fExiting; // Remember
216 if ( fExiting )
217 {
218 noLevelsExited++; // count this first level entered too
219
220 if ( fHistory.GetDepth() )
221 {
222 fBlockedPhysicalVolume = fHistory.GetTopVolume();
223 fBlockedReplicaNo = fHistory.GetTopReplicaNo();
224 fHistory.BackLevel();
225 }
226 else
227 {
228 fLastLocatedPointLocal = localPoint;
229 fLocatedOutsideWorld = true;
230 fBlockedPhysicalVolume = 0; // to be sure
231 fBlockedReplicaNo = -1;
232 fEntering = false; // No longer
233 fEnteredDaughter = false;
234 fExitedMother = true; // ??
235
236 return 0; // Have exited world volume
237 }
238 // A fix for the case where a volume is "entered" at an edge
239 // and a coincident surface exists outside it.
240 // - This stops it from exiting further volumes and cycling
241 // - However ReplicaNavigator treats this case itself
242 //
243 // assert( fBlockedPhysicalVolume!=0 );
244
245 // Expect to be on edge => on surface
246 //
247 if ( fLocatedOnEdge && (VolumeType(fBlockedPhysicalVolume)!=kReplica ))
248 {
249 fExiting= false;
250 // Consider effect on Exit Normal !?
251 }
252 }
253 else
254 if ( fEntering )
255 {
256 // assert( fBlockedPhysicalVolume!=0 );
257
258 noLevelsEntered++; // count the first level entered too
259
260 switch (VolumeType(fBlockedPhysicalVolume))
261 {
262 case kNormal:
263 fHistory.NewLevel(fBlockedPhysicalVolume, kNormal,
264 fBlockedPhysicalVolume->GetCopyNo());
265 break;
266 case kReplica:
267 freplicaNav.ComputeTransformation(fBlockedReplicaNo,
268 fBlockedPhysicalVolume);
269 fHistory.NewLevel(fBlockedPhysicalVolume, kReplica,
270 fBlockedReplicaNo);
271 fBlockedPhysicalVolume->SetCopyNo(fBlockedReplicaNo);
272 break;
273 case kParameterised:
274 if( fBlockedPhysicalVolume->GetRegularStructureId() == 0 )
275 {
276 G4VSolid *pSolid;
277 G4VPVParameterisation *pParam;
278 G4TouchableHistory parentTouchable( fHistory );
279 pParam = fBlockedPhysicalVolume->GetParameterisation();
280 pSolid = pParam->ComputeSolid(fBlockedReplicaNo,
281 fBlockedPhysicalVolume);
282 pSolid->ComputeDimensions(pParam, fBlockedReplicaNo,
283 fBlockedPhysicalVolume);
284 pParam->ComputeTransformation(fBlockedReplicaNo,
285 fBlockedPhysicalVolume);
286 fHistory.NewLevel(fBlockedPhysicalVolume, kParameterised,
287 fBlockedReplicaNo);
288 fBlockedPhysicalVolume->SetCopyNo(fBlockedReplicaNo);
289 //
290 // Set the correct solid and material in Logical Volume
291 //
292 G4LogicalVolume *pLogical;
293 pLogical = fBlockedPhysicalVolume->GetLogicalVolume();
294 pLogical->SetSolid( pSolid );
295 pLogical->UpdateMaterial(pParam ->
296 ComputeMaterial(fBlockedReplicaNo,
297 fBlockedPhysicalVolume,
298 &parentTouchable));
299 }
300 break;
301 }
302 fEntering = false;
303 fBlockedPhysicalVolume = 0;
304 localPoint = fHistory.GetTopTransform().TransformPoint(globalPoint);
305 notKnownContained = false;
306 }
307 }
308 else
309 {
310 fBlockedPhysicalVolume = 0;
311 fEntering = false;
312 fEnteredDaughter = false; // Full Step was not taken, did not enter
313 fExiting = false;
314 fExitedMother = false; // Full Step was not taken, did not exit
315 }
316 }
317 //
318 // Search from top of history up through geometry until
319 // containing volume found:
320 // If on
321 // o OUTSIDE - Back up level, not/no longer exiting volumes
322 // o SURFACE and EXITING - Back up level, setting new blocking no.s
323 // else
324 // o containing volume found
325 //
326
327 while (notKnownContained)
328 {
329 if ( fHistory.GetTopVolumeType()!=kReplica )
330 {
331 targetSolid = fHistory.GetTopVolume()->GetLogicalVolume()->GetSolid();
332 localPoint = fHistory.GetTopTransform().TransformPoint(globalPoint);
333 insideCode = targetSolid->Inside(localPoint);
334#ifdef G4VERBOSE
335 if(( fVerbose == 1 ) && ( fCheck ))
336 {
337 G4String solidResponse = "-kInside-";
338 if (insideCode == kOutside)
339 solidResponse = "-kOutside-";
340 else if (insideCode == kSurface)
341 solidResponse = "-kSurface-";
342 G4cout << "*** G4ITNavigator2::LocateGlobalPointAndSetup(): ***" << G4endl
343 << " Invoked Inside() for solid: " << targetSolid->GetName()
344 << ". Solid replied: " << solidResponse << G4endl
345 << " For local point p: " << localPoint << G4endl;
346 }
347#endif
348 }
349 else
350 {
351 insideCode = freplicaNav.BackLocate(fHistory, globalPoint, localPoint,
352 fExiting, notKnownContained);
353 // !CARE! if notKnownContained returns false then the point is within
354 // the containing placement volume of the replica(s). If insidecode
355 // will result in the history being backed up one level, then the
356 // local point returned is the point in the system of this new level
357 }
358
359
360 if ( insideCode==kOutside )
361 {
362 noLevelsExited++;
363 if ( fHistory.GetDepth() )
364 {
365 fBlockedPhysicalVolume = fHistory.GetTopVolume();
366 fBlockedReplicaNo = fHistory.GetTopReplicaNo();
367 fHistory.BackLevel();
368 fExiting = false;
369
370 if( noLevelsExited > 1 )
371 {
372 // The first transformation was done by the sub-navigator
373 //
374 const G4RotationMatrix* mRot = fBlockedPhysicalVolume->GetRotation();
375 if( mRot )
376 {
377 fGrandMotherExitNormal *= (*mRot).inverse();
378 fChangedGrandMotherRefFrame= true;
379 }
380 }
381 }
382 else
383 {
384 fLastLocatedPointLocal = localPoint;
385 fLocatedOutsideWorld = true;
386 // No extra transformation for ExitNormal - is in frame of Top Volume
387 return 0; // Have exited world volume
388 }
389 }
390 else
391 if ( insideCode==kSurface )
392 {
393 G4bool isExiting = fExiting;
394 if( (!fExiting)&&considerDirection )
395 {
396 // Figure out whether we are exiting this level's volume
397 // by using the direction
398 //
399 G4bool directionExiting = false;
400 G4ThreeVector localDirection =
401 fHistory.GetTopTransform().TransformAxis(globalDirection);
402
403 // Make sure localPoint in correct reference frame
404 // ( Was it already correct ? How ? )
405 //
406 localPoint= fHistory.GetTopTransform().TransformPoint(globalPoint);
407 if ( fHistory.GetTopVolumeType()!=kReplica )
408 {
409 G4ThreeVector normal = targetSolid->SurfaceNormal(localPoint);
410 directionExiting = normal.dot(localDirection) > 0.0;
411 isExiting = isExiting || directionExiting;
412 }
413 }
414 if( isExiting )
415 {
416 noLevelsExited++;
417 if ( fHistory.GetDepth() )
418 {
419 fBlockedPhysicalVolume = fHistory.GetTopVolume();
420 fBlockedReplicaNo = fHistory.GetTopReplicaNo();
421 fHistory.BackLevel();
422 //
423 // Still on surface but exited volume not necessarily convex
424 //
425 fValidExitNormal = false;
426
427 if( noLevelsExited > 1 )
428 {
429 // The first transformation was done by the sub-navigator
430 //
431 const G4RotationMatrix* mRot =
432 fBlockedPhysicalVolume->GetRotation();
433 if( mRot )
434 {
435 fGrandMotherExitNormal *= (*mRot).inverse();
436 fChangedGrandMotherRefFrame= true;
437 }
438 }
439 }
440 else
441 {
442 fLastLocatedPointLocal = localPoint;
443 fLocatedOutsideWorld = true;
444 // No extra transformation for ExitNormal, is in frame of Top Vol
445 return 0; // Have exited world volume
446 }
447 }
448 else
449 {
450 notKnownContained=false;
451 }
452 }
453 else
454 {
455 notKnownContained=false;
456 }
457 } // END while (notKnownContained)
458 //
459 // Search downwards until deepest containing volume found,
460 // blocking fBlockedPhysicalVolume/BlockedReplicaNum
461 //
462 // 3 Cases:
463 //
464 // o Parameterised daughters
465 // =>Must be one G4PVParameterised daughter & voxels
466 // o Positioned daughters & voxels
467 // o Positioned daughters & no voxels
468
469 noResult = true; // noResult should be renamed to
470 // something like enteredLevel, as that is its meaning.
471 do
472 {
473 // Determine `type' of current mother volume
474 //
475 targetPhysical = fHistory.GetTopVolume();
476 if (!targetPhysical) { break; }
477 targetLogical = targetPhysical->GetLogicalVolume();
478 switch( CharacteriseDaughters(targetLogical) )
479 {
480 case kNormal:
481 if ( targetLogical->GetVoxelHeader() ) // use optimised navigation
482 {
483 noResult = fvoxelNav.LevelLocate(fHistory,
484 fBlockedPhysicalVolume,
485 fBlockedReplicaNo,
486 globalPoint,
487 pGlobalDirection,
488 considerDirection,
489 localPoint);
490 }
491 else // do not use optimised navigation
492 {
493 noResult = fnormalNav.LevelLocate(fHistory,
494 fBlockedPhysicalVolume,
495 fBlockedReplicaNo,
496 globalPoint,
497 pGlobalDirection,
498 considerDirection,
499 localPoint);
500 }
501 break;
502 case kReplica:
503 noResult = freplicaNav.LevelLocate(fHistory,
504 fBlockedPhysicalVolume,
505 fBlockedReplicaNo,
506 globalPoint,
507 pGlobalDirection,
508 considerDirection,
509 localPoint);
510 break;
511 case kParameterised:
512 if( GetDaughtersRegularStructureId(targetLogical) != 1 )
513 {
514 noResult = fparamNav.LevelLocate(fHistory,
515 fBlockedPhysicalVolume,
516 fBlockedReplicaNo,
517 globalPoint,
518 pGlobalDirection,
519 considerDirection,
520 localPoint);
521 }
522 else // Regular structure
523 {
524 noResult = fregularNav.LevelLocate(fHistory,
525 fBlockedPhysicalVolume,
526 fBlockedReplicaNo,
527 globalPoint,
528 pGlobalDirection,
529 considerDirection,
530 localPoint);
531 }
532 break;
533 }
534
535 // LevelLocate returns true if it finds a daughter volume
536 // in which globalPoint is inside (or on the surface).
537
538 if ( noResult )
539 {
540 noLevelsEntered++;
541
542 // Entering a daughter after ascending
543 //
544 // The blocked volume is no longer valid - it was for another level
545 //
546 fBlockedPhysicalVolume = 0;
547 fBlockedReplicaNo = -1;
548
549 // fEntering should be false -- else blockedVolume is assumed good.
550 // fEnteredDaughter is used for ExitNormal
551 //
552 fEntering = false;
553 fEnteredDaughter = true;
554
555 if( fExitedMother )
556 {
557 G4VPhysicalVolume* enteredPhysical = fHistory.GetTopVolume();
558 const G4RotationMatrix* mRot = enteredPhysical->GetRotation();
559 if( mRot )
560 {
561 // Go deeper, i.e. move 'down' in the hierarchy
562 // Apply direct rotation, not inverse
563 //
564 fGrandMotherExitNormal *= (*mRot);
565 fChangedGrandMotherRefFrame= true;
566 }
567 }
568
569#ifdef G4DEBUG_NAVIGATION
570 if( fVerbose > 2 )
571 {
572 G4VPhysicalVolume* enteredPhysical = fHistory.GetTopVolume();
573 G4cout << "*** G4ITNavigator2::LocateGlobalPointAndSetup() ***" << G4endl;
574 G4cout << " Entering volume: " << enteredPhysical->GetName()
575 << G4endl;
576 }
577#endif
578 }
579 } while (noResult);
580
581 fLastLocatedPointLocal = localPoint;
582
583#ifdef G4VERBOSE
584 if( fVerbose >= 4 )
585 {
586 G4int oldcoutPrec = G4cout.precision(8);
587 G4String curPhysVol_Name("None");
588 if (targetPhysical) { curPhysVol_Name = targetPhysical->GetName(); }
589 G4cout << " Return value = new volume = " << curPhysVol_Name << G4endl;
590 G4cout << " ----- Upon exiting:" << G4endl;
591 PrintState();
592 if( fVerbose >= 5 )
593 {
594 G4cout << "Upon exiting LocateGlobalPointAndSetup():" << G4endl;
595 G4cout << " History = " << G4endl << fHistory << G4endl << G4endl;
596 }
597 G4cout.precision(oldcoutPrec);
598 }
599#endif
600
601 fLocatedOutsideWorld= false;
602
603 return targetPhysical;
604}

606// ********************************************************************
607// LocateGlobalPointWithinVolume
608//
609// -> the state information of this Navigator and its subNavigators
610// is updated in order to start the next step at pGlobalpoint
611// -> no check is performed whether pGlobalpoint is inside the
612// original volume (this must be the case).
613//
614// Note: a direction could be added to the arguments, to aid in future
615// optional checking (via the old code below, flagged by OLD_LOCATE).
616// [ This would be done only in verbose mode ]
617// ********************************************************************
618//
619void
620G4ITNavigator2::LocateGlobalPointWithinVolume(const G4ThreeVector& pGlobalpoint)
621{
622 CheckNavigatorStateIsValid();
623
624#ifdef G4DEBUG_NAVIGATION
625 // Check: Either step was not limited by a boundary
626 // or else the full step is no longer being taken
627 assert( !fWasLimitedByGeometry );
628#endif
629
630 fLastLocatedPointLocal = ComputeLocalPoint(pGlobalpoint);
631 fLastTriedStepComputation= false;
632 fChangedGrandMotherRefFrame= false; // Frame for Exit Normal
633
634#ifdef G4DEBUG_NAVIGATION
635 if( fVerbose > 2 )
636 {
637 G4cout << "Entering LocateGlobalWithinVolume(): History = " << G4endl;
638 G4cout << fHistory << G4endl;
639 }
640#endif
641
642 // For the case of Voxel (or Parameterised) volume the respective
643 // Navigator must be messaged to update its voxel information etc
644
645 // Update the state of the Sub Navigators
646 // - in particular any voxel information they store/cache
647 //
648 G4VPhysicalVolume* motherPhysical = fHistory.GetTopVolume();
649 G4LogicalVolume* motherLogical = motherPhysical->GetLogicalVolume();
650 G4SmartVoxelHeader* pVoxelHeader = motherLogical->GetVoxelHeader();
651
652 if ( fHistory.GetTopVolumeType()!=kReplica )
653 {
654 switch( CharacteriseDaughters(motherLogical) )
655 {
656 case kNormal:
657 if ( pVoxelHeader )
658 {
659 fvoxelNav.VoxelLocate( pVoxelHeader, fLastLocatedPointLocal );
660 }
661 break;
662 case kParameterised:
663 if( GetDaughtersRegularStructureId(motherLogical) != 1 )
664 {
665 // Resets state & returns voxel node
666 //
667 fparamNav.ParamVoxelLocate( pVoxelHeader, fLastLocatedPointLocal );
668 }
669 break;
670 case kReplica:
671 G4Exception("G4ITNavigator2::LocateGlobalPointWithinVolume()",
672 "GeomNav0001", FatalException,
673 "Not applicable for replicated volumes.");
674 break;
675 }
676 }
677
678 // Reset the state variables
679 // - which would have been affected
680 // by the 'equivalent' call to LocateGlobalPointAndSetup
681 // - who's values have been invalidated by the 'move'.
682 //
683 fBlockedPhysicalVolume = 0;
684 fBlockedReplicaNo = -1;
685 fEntering = false;
686 fEnteredDaughter = false; // Boundary not encountered, did not enter
687 fExiting = false;
688 fExitedMother = false; // Boundary not encountered, did not exit
689}

1293#endif
1294 }
1295 }
1296
1297 // Now transform it to the global reference frame !!
1298 //
1299 if( fValidExitNormal || fCalculatedExitNormal )
1300 {
1301 G4int depth= fHistory.GetDepth();
1302 if( depth > 0 )
1303 {

1312 }
1313 }
1314 else
1315 {
1316 fExitNormalGlobalFrame= G4ThreeVector( 0., 0., 0.);
1317 }
1318 }
1319

1323 // The Navigator is obliged to return "infinity"
1324 //
1325 Step = kInfinity;
1326 }
1327
1328#ifdef G4VERBOSE
1329 if( fVerbose > 1 )
1330 {
1331 if( fVerbose >= 4 )
1332 {
1333 G4cout << " ----- Upon exiting :" << G4endl;
1334 PrintState();
1335 }
1336 G4cout << " Returned step= " << Step;
1337 if( fVerbose > 5 ) G4cout << G4endl;

1339 {
1340 G4cout << " Requested step= " << pCurrentProposedStepLength ;

1342 }
1343 G4cout << " Safety = " << pNewSafety << G4endl;
1344 }
1345#endif
1346
1347 return Step;
1348}
