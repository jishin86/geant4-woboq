120void G4TrackingInformation::RecordCurrentPositionNTime(G4Track* track)
121{
122 if(track)
123 {
124 fRecordedTrackPosition = track->GetPosition();
125 fRecordedTrackLocalTime = track->GetLocalTime();
126 fRecordedTrackGlobalTime = track->GetGlobalTime();
127 }
128}
