188//
189// Bethe screening functions for the elastic (coherent) scattering:
190// Bethe's phi1, phi2 coherent screening functions were computed numerically
191// by using (the universal) atomic form factors computed based on the Thomas-
192// Fermi model of the atom (using numerical solution of the Thomas-Fermi
193// screening function instead of Moliere's analytical approximation). The
194// numerical results can be well approximated (better than Butcher & Messel
195// especially near the delta=1 limit) by:
196// ## if delta <= 1.4
197// phi1(delta) = 20.806 - delta*(3.190 - 0.5710*delta)
198// phi2(delta) = 20.234 - delta*(2.126 - 0.0903*delta)
199// ## if delta > 1.4
200// phi1(delta) = phi2(delta) = 21.0190 - 4.145*ln(delta + 0.958)
201// with delta = 136mc^2kZ^{-1/3}/[E(Eg-E)] = 136Z^{-1/3}eps0/[eps(1-eps)] where
202// Eg is the initial photon energy, E is the total energy transferred to one of
203// the e-/e+ pair, eps0 = mc^2/Eg and eps = E/Eg.
204
205inline
206void G4PairProductionRelModel::ComputePhi12(const G4double delta, G4double &phi1,
207 G4double &phi2)
208{
209 if (delta > 1.4) {
210 phi1 = 21.0190 - 4.145*G4Log(delta + 0.958);
211 phi2 = phi1;
212 } else {
213 phi1 = 20.806 - delta*(3.190 - 0.5710*delta);
214 phi2 = 20.234 - delta*(2.126 - 0.0903*delta);
215 }
216}
217
218
219// Compute the value of the screening function 3*PHI1(delta) - PHI2(delta):
220inline G4double G4PairProductionRelModel::ScreenFunction1(const G4double delta)
221{
222 return (delta > 1.4) ? 42.038 - 8.29*G4Log(delta + 0.958)
223 : 42.184 - delta*(7.444 - 1.623*delta);
224}
225
226
227// Compute the value of the screening function 1.5*PHI1(delta) +0.5*PHI2(delta):
228inline G4double G4PairProductionRelModel::ScreenFunction2(const G4double delta)
229{
230 return (delta > 1.4) ? 42.038 - 8.29*G4Log(delta + 0.958)
231 : 41.326 - delta*(5.848 - 0.902*delta);
232}
233
234
235// Same as ScreenFunction1 and ScreenFunction2 but computes them at once
236inline void G4PairProductionRelModel::ScreenFunction12(const G4double delta,
237 G4double &f1, G4double &f2)
238{
239 if (delta > 1.4) {
240 f1 = 42.038 - 8.29*G4Log(delta + 0.958);
241 f2 = f1;
242 } else {
243 f1 = 42.184 - delta*(7.444 - 1.623*delta);
244 f2 = 41.326 - delta*(5.848 - 0.902*delta);
245 }
