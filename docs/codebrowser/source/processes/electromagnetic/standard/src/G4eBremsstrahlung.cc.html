61G4eBremsstrahlung::G4eBremsstrahlung(const G4String& name):
62 G4VEnergyLossProcess(name),
63 isInitialised(false)
64{
65 SetProcessSubType(fBremsstrahlung);

77G4bool G4eBremsstrahlung::IsApplicable(const G4ParticleDefinition& p)

84void
85G4eBremsstrahlung::InitialiseEnergyLossProcess(const G4ParticleDefinition*,
86 const G4ParticleDefinition*)
87{
88 if(!isInitialised) {
89 G4EmParameters* param = G4EmParameters::Instance();
90
91 G4double emin = param->MinKinEnergy();
92 G4double emax = param->MaxKinEnergy();
93 G4VEmFluctuationModel* fm = nullptr;
94
95 if (!EmModel(0)) { SetEmModel(new G4SeltzerBergerModel()); }
96 EmModel(0)->SetLowEnergyLimit(emin);
97 G4double energyLimit = std::min(EmModel(0)->HighEnergyLimit(), GeV);
98 EmModel(0)->SetHighEnergyLimit(energyLimit);
99 EmModel(0)->SetSecondaryThreshold(param->BremsstrahlungTh());
100 EmModel(0)->SetLPMFlag(false);
101 AddEmModel(1, EmModel(0), fm);
102
103 if(emax > energyLimit) {
104 if (!EmModel(1)) { SetEmModel(new G4eBremsstrahlungRelModel()); }
105 EmModel(1)->SetLowEnergyLimit(energyLimit);
106 EmModel(1)->SetHighEnergyLimit(emax);
107 EmModel(1)->SetSecondaryThreshold(param->BremsstrahlungTh());
108 EmModel(1)->SetLPMFlag(param->LPM());
109 AddEmModel(1, EmModel(1), fm);
110 }
111 isInitialised = true;
112 }
113}

117void G4eBremsstrahlung::StreamProcessInfo(std::ostream& out) const
118{
119 if(EmModel(0)) {
120 G4EmParameters* param = G4EmParameters::Instance();
121 G4double eth = param->BremsstrahlungTh();
122 out << " LPM flag: " << param->LPM() << " for E > "
123 << EmModel(0)->HighEnergyLimit()/GeV << " GeV";
124 if(eth < DBL_MAX) {
125 out << ", VertexHighEnergyTh(GeV)= " << eth/GeV;
126 }
127 out << G4endl;
128 }
129}
