77G4SeltzerBergerModel::G4SeltzerBergerModel(const G4ParticleDefinition* p,
78 const G4String& nam)
79: G4eBremsstrahlungRelModel(p,nam), fIsUseBicubicInterpolation(false),
80 fIsUseSamplingTables(true), fNumWarnings(0), fIndx(0), fIndy(0)
81{

89G4SeltzerBergerModel::~G4SeltzerBergerModel()
90{
91 // delete SB-DCS data per Z
92 if (IsMaster()) {
93 for (size_t iz = 0; iz < gMaxZet; ++iz) {
94 if (gSBDCSData[iz]) {
95 delete gSBDCSData[iz];
96 gSBDCSData[iz] = nullptr;
97 }
98 }
99 if (gSBSamplingTable) {
100 delete gSBSamplingTable;
101 gSBSamplingTable = nullptr;
102 }
103 }

106void G4SeltzerBergerModel::Initialise(const G4ParticleDefinition* p,
107 const G4DataVector& cuts)
108{
109 if (p) {
110 SetParticle(p);
111 }
112 fIsUseSamplingTables = G4EmParameters::Instance()->EnableSamplingTable();
113 // Access to elements
114 if (IsMaster()) {
115 // check environment variable
116 // build the complete string identifying the file with the data set
117 char* path = getenv("G4LEDATA");
118 const G4ElementTable* theElemTable = G4Element::GetElementTable();
119 size_t numOfElem = G4Element::GetNumberOfElements();
120 for (size_t ie = 0; ie < numOfElem; ++ie) {
121 G4int izet = std::min(G4lrint(((*theElemTable)[ie])->GetZ()), gMaxZet-1);
122 izet = std::max(1, izet);
123 // load SB-DCS data for this atomic number if it has not been loaded yet
124 if (!gSBDCSData[izet]) {
125 ReadData(izet, path);
126 }
127 }
128 // elem.selectr. only for master: base class init-local will set for workers
129 if (LowEnergyLimit() < HighEnergyLimit()) {
130 InitialiseElementSelectors(p,cuts);
131 }
132 // init sampling tables if it was requested
133 if (fIsUseSamplingTables) {
134 if (!gSBSamplingTable) {
135 gSBSamplingTable = new G4SBBremTable();
136 }
137 gSBSamplingTable->Initialize(std::max(fLowestKinEnergy,LowEnergyLimit()),
138 HighEnergyLimit());
139 }
140 }
141 //
142 if (!fParticleChange) { fParticleChange = GetParticleChangeForLoss(); }
143 if (GetTripletModel()) {
144 GetTripletModel()->Initialise(p, cuts);
145 fIsScatOffElectron = true;
146 }
147}

153void G4SeltzerBergerModel::ReadData(G4int izet, const char* path) {
154 // return if it has been already loaded
155 if (gSBDCSData[izet]) {
156 return;
157 }
158 const char* datadir = path;
159 if (!datadir) {
160 datadir = getenv("G4LEDATA");
161 if (!datadir) {
162 G4Exception("G4SeltzerBergerModel::ReadData()","em0006",FatalException,
163 "Environment variable G4LEDATA not defined");
164 return;
165 }
166 }
167 std::ostringstream ost;
168 ost << datadir << DirectoryPath() << izet;
169 std::ifstream fin(ost.str().c_str());
170 if (!fin.is_open()) {
171 G4ExceptionDescription ed;
172 ed << "Bremsstrahlung data file <" << ost.str().c_str()
173 << "> is not opened!";
174 G4Exception("G4SeltzerBergerModel::ReadData()","em0003",FatalException,
175 ed,"G4LEDATA version should be G4EMLOW6.23 or later.");
176 return;
177 }
178 //G4cout << "G4SeltzerBergerModel read from <" << ost.str().c_str()
179 // << ">" << G4endl;
180 G4Physics2DVector* v = new G4Physics2DVector();
181 if (v->Retrieve(fin)) {
182 v->SetBicubicInterpolation(fIsUseBicubicInterpolation);
183 gSBDCSData[izet] = v;
184 static const G4double emaxlog = 4*G4Log(10.);
185 gYLimitData[izet] = v->Value(0.97, emaxlog, fIndx, fIndy);
186 } else {
187 G4ExceptionDescription ed;
188 ed << "Bremsstrahlung data file <" << ost.str().c_str()
189 << "> is not retrieved!";
190 G4Exception("G4SeltzerBergerModel::ReadData()","em0005",FatalException,
191 ed,"G4LEDATA version should be G4EMLOW6.23 or later.");
192 delete v;
193 }
194 // G4cout << dataSB[Z] << G4endl;
195}

197G4double G4SeltzerBergerModel::ComputeDXSectionPerAtom(G4double gammaEnergy)
198{
199 static const G4double kMC2 = CLHEP::electron_mass_c2;
200 static const G4double kAlpha = CLHEP::twopi*CLHEP::fine_structure_const;
201 G4double dxsec = 0.0;
202 if (gammaEnergy < 0.0 || fPrimaryKinEnergy <= 0.0) {
203 return dxsec;
204 }
205 // reduced photon energy
206 const G4double x = gammaEnergy/fPrimaryKinEnergy;
207 // l-kinetic energy of the e-/e+
208 const G4double y = G4Log(fPrimaryKinEnergy/CLHEP::MeV);
209 // make sure that the Z-related SB-DCS are loaded
210 // NOTE: fCurrentIZ should have been set before.
211 fCurrentIZ = std::max(std::min(fCurrentIZ, gMaxZet-1), 1);
212 if (!gSBDCSData[fCurrentIZ]) {
213 InitialiseForElement(nullptr, fCurrentIZ);
214 }
215 /*
216 G4ExceptionDescription ed;
217 ed << "Bremsstrahlung data for Z= " << Z
218 << " are not initialized!";
219 G4Exception("G4SeltzerBergerModel::ComputeDXSectionPerAtom()","em0005",
220 FatalException, ed,
221 "G4LEDATA version should be G4EMLOW6.23 or later.");
222 */
223 // NOTE: SetupForMaterial should have been called before!
224 const G4double pt2 = fPrimaryKinEnergy*(fPrimaryKinEnergy+2.*kMC2);
225 const G4double invb2 = fPrimaryTotalEnergy*fPrimaryTotalEnergy/pt2;
226 G4double val = gSBDCSData[fCurrentIZ]->Value(x,y,fIndx,fIndy);
227 dxsec = val*invb2*CLHEP::millibarn/gBremFactor;
228 // e+ correction
229 if (!fIsElectron) {
230 const G4double invbeta1 = std::sqrt(invb2);
231 const G4double e2 = fPrimaryKinEnergy-gammaEnergy;
232 if (e2 > 0.0) {
233 const G4double invbeta2 = (e2+kMC2)/std::sqrt(e2*(e2+2.0*kMC2));
234 const G4double dum0 = kAlpha*fCurrentIZ*(invbeta1-invbeta2);
235 if (dum0 < gExpNumLimit) {
236 dxsec = 0.0;
237 } else {
238 dxsec *= G4Exp(dum0);
239 }
240 } else {
241 dxsec = 0.0;
242 }
243 }
244 return dxsec;
245}

247void
248G4SeltzerBergerModel::SampleSecondaries(std::vector<G4DynamicParticle*>* vdp,
249 const G4MaterialCutsCouple* couple,
250 const G4DynamicParticle* dp,
251 G4double cutEnergy,
252 G4double maxEnergy)
253{
254 static const G4double kMC2 = CLHEP::electron_mass_c2;
255 const G4double kinEnergy = dp->GetKineticEnergy();
256// const G4double logKinEnergy = dp->GetLogKineticEnergy();
257 const G4double logKinEnergy = G4Log(kinEnergy); // WILL BE REMOVED
258 const G4double tmin = std::min(cutEnergy, kinEnergy);
259 const G4double tmax = std::min(maxEnergy, kinEnergy);
260 if (tmin >= tmax) {
261 return;
262 }
263 // set local variables and select target element
264 SetupForMaterial(fPrimaryParticle, couple->GetMaterial(), kinEnergy);
265 const G4Element* elm = SelectRandomAtom(couple, fPrimaryParticle, kinEnergy,
266 tmin, tmax);
267// const G4Element* elm = SelectTargetAtom(couple, fPrimaryParticle, kinEnergy,
268// logKinEnergy, fElemSelectorEkinIndx,
269// tmin, tmax);
270 fCurrentIZ = std::max(std::min(elm->GetZasInt(),gMaxZet-1), 1);
271 //
272 const G4double totMomentum = std::sqrt(kinEnergy*(fPrimaryTotalEnergy+kMC2));
273 /*
274 G4cout << "G4SeltzerBergerModel::SampleSecondaries E(MeV)= "
275 << kinEnergy/MeV
276 << " Z= " << fCurrentIZ << " cut(MeV)= " << tmin/MeV
277 << " emax(MeV)= " << tmax/MeV << " corr= " << fDensityCorr << G4endl;
278 */
279 // sample emitted photon energy either by rejection or from samplign tables
280 const G4double gammaEnergy = !fIsUseSamplingTables
281 ? SampleEnergyTransfer(kinEnergy, tmin, tmax)
282 : gSBSamplingTable->SampleEnergyTransfer(kinEnergy,
283 logKinEnergy, tmin, fDensityCorr, fCurrentIZ,
284 couple->GetIndex(), fIsElectron);
285 // should never happen under normal conditions but protect it
286 if (gammaEnergy <= 0.) {
287 return;
288 }
289 //
290 // angles of the emitted gamma. ( Z - axis along the parent particle) use
291 // general interface
292 G4ThreeVector gamDir = GetAngularDistribution()->SampleDirection(dp,
293 fPrimaryTotalEnergy-gammaEnergy, fCurrentIZ, couple->GetMaterial());
294 // create G4DynamicParticle object for the emitted Gamma
295 G4DynamicParticle* gamma = new G4DynamicParticle(fGammaParticle, gamDir,
296 gammaEnergy);
297 vdp->push_back(gamma);
298 //
299 // compute post-interaction kinematics of the primary e-/e+
300 G4ThreeVector dir = (totMomentum*dp->GetMomentumDirection()-gammaEnergy*gamDir).unit();
301 const G4double finalE = kinEnergy - gammaEnergy;
302 /*
303 G4cout << "### G4SBModel: v= "
304 << " Eg(MeV)= " << gammaEnergy
305 << " Ee(MeV)= " << kineticEnergy
306 << " DirE " << direction << " DirG " << gammaDirection
307 << G4endl;
308 */
309 // if secondary gamma energy is higher than threshold(very high by default)
310 // then stop tracking the primary particle and create new secondary e-/e+
311 // instead of the primary
312 if (gammaEnergy > SecondaryThreshold()) {
313 fParticleChange->ProposeTrackStatus(fStopAndKill);
314 fParticleChange->SetProposedKineticEnergy(0.0);
315 G4DynamicParticle* el = new G4DynamicParticle(
316 const_cast<G4ParticleDefinition*>(fPrimaryParticle), dir, finalE);
317 vdp->push_back(el);
318 } else { // continue tracking the primary e-/e+ otherwise
319 fParticleChange->SetProposedMomentumDirection(dir);
320 fParticleChange->SetProposedKineticEnergy(finalE);
321 }
322}

324// sample emitted photon energy by usign rejection
325G4double
326G4SeltzerBergerModel::SampleEnergyTransfer(const G4double kinEnergy,
327 const G4double tmin,
328 const G4double tmax)
329{
330 static const G4double kMC2 = CLHEP::electron_mass_c2;
331 static const G4double kAlpha = CLHEP::twopi*CLHEP::fine_structure_const;
332 // min max of the transformed variable: x(k) = ln(k^2+k_p^2) that is in
333 // [ln(k_c^2+k_p^2), ln(E_k^2+k_p^2)]
334 const G4double xmin = G4Log(tmin*tmin+fDensityCorr);
335 const G4double xrange = G4Log(tmax*tmax+fDensityCorr)-xmin;
336 const G4double y = G4Log(kinEnergy/CLHEP::MeV);
337 // majoranta
338 const G4double x0 = tmin/kinEnergy;
339 G4double vmax;
340 if (fCurrentIZ < 93) {
341 vmax = gSBDCSData[fCurrentIZ]->Value(x0, y, fIndx, fIndy)*1.02;
342 } else {
343 // reset cashed x and y indices
344 fIndx = 0;
345 fIndy = 0;
346 vmax = gSBDCSData[fCurrentIZ]->Value(x0, y, fIndx, fIndy)*1.2;
347 }
348 //
349 static const G4double kEPeakLim = 300.*CLHEP::MeV;
350 static const G4double kELowLim = 20.*CLHEP::keV;
351 // majoranta corrected for e-
352 if (fIsElectron && x0 < 0.97 && ((kinEnergy>kEPeakLim) || (kinEnergy<kELowLim))) {
353 const G4double ylim = std::min(gYLimitData[fCurrentIZ],
354 1.1*gSBDCSData[fCurrentIZ]->Value(0.97,y,fIndx,fIndy));
355 if (ylim > vmax) {
356 vmax = ylim;
357 }
358 }
359 if (x0 < 0.05) {
360 vmax *= 1.2;
361 }
362 //G4cout<<"y= "<<y<<" xmin= "<<xmin<<" xmax= "<<xmax
363 //<<" vmax= "<<vmax<<G4endl;
364 static const G4int kNCountMax = 100;
365 CLHEP::HepRandomEngine* rndmEngine = G4Random::getTheEngine();
366 G4double rndm[2];
367 G4double gammaEnergy, v;
368 for (G4int nn = 0; nn < kNCountMax; ++nn) {
369 rndmEngine->flatArray(2, rndm);
370 gammaEnergy = std::sqrt(std::max(G4Exp(xmin + rndm[0]*xrange)-fDensityCorr,0.));
371 v = gSBDCSData[fCurrentIZ]->Value(gammaEnergy/kinEnergy, y, fIndx, fIndy);
372 // e+ correction
373 if (!fIsElectron) {
374 const G4double e1 = kinEnergy - tmin;
375 const G4double invbeta1 = (e1+kMC2)/std::sqrt(e1*(e1+2.*kMC2));
376 const G4double e2 = kinEnergy-gammaEnergy;
377 const G4double invbeta2 = (e2+kMC2)/std::sqrt(e2*(e2+2.*kMC2));
378 const G4double dum0 = kAlpha*fCurrentIZ*(invbeta1-invbeta2);
379 if (dum0 < gExpNumLimit) {
380 v = 0.0;
381 } else {
382 v *= G4Exp(dum0);
383 }
384 }
385 if (v > 1.05*vmax && fNumWarnings < 11) {
386 ++fNumWarnings;
387 G4ExceptionDescription ed;
388 ed << "### G4SeltzerBergerModel Warning: Majoranta exceeded! "
389 << v << " > " << vmax << " by " << v/vmax
390 << " Niter= " << nn
391 << " Egamma(MeV)= " << gammaEnergy
392 << " Ee(MeV)= " << kinEnergy
393 << " Z= " << fCurrentIZ << " " << fPrimaryParticle->GetParticleName();
394 //
395 if (10 == fNumWarnings) {
396 ed << "\n ### G4SeltzerBergerModel Warnings stopped";
397 }
398 G4Exception("G4SeltzerBergerModel::SampleScattering","em0044",
399 JustWarning, ed,"");
400 }
401 if (v >= vmax*rndm[1]) {
402 break;
403 }
404 }
405 return gammaEnergy;
406}

420void G4SeltzerBergerModel::SetupForMaterial(const G4ParticleDefinition*,
421 const G4Material* mat,
422 G4double kineticEnergy)
423{
424 fDensityFactor = gMigdalConstant*mat->GetElectronDensity();
425 // calculate threshold for density effect: gamma*k_p = sqrt(fDensityCorr)
426 fPrimaryKinEnergy = kineticEnergy;
427 fPrimaryTotalEnergy = kineticEnergy+CLHEP::electron_mass_c2;
428 fDensityCorr = fDensityFactor*fPrimaryTotalEnergy*fPrimaryTotalEnergy;
429 fIsLPMActive = LPMFlag();
430}
