59G4ePairProduction::G4ePairProduction(const G4String& name)
60 : G4VEnergyLossProcess(name),
61 theParticle(nullptr),
62 lowestKinEnergy(100.*MeV),
63 isInitialised(false)
64{
65 SetProcessSubType(fPairProdByCharged);
66 SetSecondaryParticle(G4Positron::Positron());
67 SetIonisation(false);
68}

72G4bool G4ePairProduction::IsApplicable(const G4ParticleDefinition& p)
73{
74 return (G4Electron::Electron() == &p || G4Positron::Positron() == &p);
75}

88void G4ePairProduction::InitialiseEnergyLossProcess(
89 const G4ParticleDefinition* part,
90 const G4ParticleDefinition*)
91{
92 if (!isInitialised) {
93 isInitialised = true;
94
95 theParticle = part;
96
97 G4MuPairProductionModel* mod = new G4MuPairProductionModel(part, "ePairProd");
98 SetEmModel(mod);
99
100 lowestKinEnergy = std::max(lowestKinEnergy, part->GetPDGMass()*8.0);
101 mod->SetLowestKineticEnergy(lowestKinEnergy);
102
103 G4VEmFluctuationModel* fm = nullptr;
104 G4EmParameters* param = G4EmParameters::Instance();
105 mod->SetLowEnergyLimit(param->MinKinEnergy());
106 mod->SetHighEnergyLimit(param->MaxKinEnergy());
107 AddEmModel(1, mod, fm);
108 }
109}

113void G4ePairProduction::StreamProcessInfo(std::ostream& out) const
114{
115 G4ElementData* ed = EmModel(0)->GetElementData();
116 if(ed) {
117 for(G4int Z=1; Z<93; ++Z) {
118 G4Physics2DVector* pv = ed->GetElement2DData(Z);
119 if(pv) {
120 out << " Sampling table " << pv->GetLengthY()
121 << "x" << pv->GetLengthX() << "; from "
122 << G4Exp(pv->GetY(0))/GeV << " GeV to "
123 << G4Exp(pv->GetY(pv->GetLengthY()-1))/TeV
124 << " TeV " << G4endl;
125 break;
126 }
127 }
128 }
129}
