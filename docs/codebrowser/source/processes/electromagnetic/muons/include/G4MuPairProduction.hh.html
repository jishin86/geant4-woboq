74{

84 const G4Material*, G4double cut) override;
85
86 inline void SetLowestKineticEnergy(G4double e);
87
88 // print description in html
89 virtual void ProcessDescription(std::ostream&) const override;
90
91protected:
92
93 // Print out of the class parameters
94 virtual void StreamProcessInfo(std::ostream& outFile) const override;
95
96 virtual void
97 InitialiseEnergyLossProcess(const G4ParticleDefinition*,
98 const G4ParticleDefinition*) override;
99
100private:

102 G4MuPairProduction & operator=(const G4MuPairProduction &right) = delete;
103 G4MuPairProduction(const G4MuPairProduction&) = delete;
104
105protected:
106
107 const G4ParticleDefinition* theParticle;
108 G4double lowestKinEnergy;
109 G4bool isInitialised;
110
111};
