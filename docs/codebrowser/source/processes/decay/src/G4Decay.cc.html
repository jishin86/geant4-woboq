88G4bool G4Decay::IsApplicable(const G4ParticleDefinition& aParticleType)
89{
90 // check if the particle is stable?
91 if (aParticleType.GetPDGLifeTime() <0.0) {
92 return false;
93 } else if (aParticleType.GetPDGMass() <= 0.0*MeV) {
94 return false;
95 } else {
96 return true;
97 }

100G4double G4Decay::GetMeanLifeTime(const G4Track& aTrack ,
101 G4ForceCondition*)
102{
103 // returns the mean free path in GEANT4 internal units
104 G4double meanlife;
105
106 // get particle
107 const G4DynamicParticle* aParticle = aTrack.GetDynamicParticle();
108 const G4ParticleDefinition* aParticleDef = aParticle->GetDefinition();
109 G4double aLife = aParticleDef->GetPDGLifeTime();
110
111 // check if the particle is stable?
112 if (aParticleDef->GetPDGStable()) {
113 //1000000 times the life time of the universe
114 meanlife = 1e24 * s;
115
116 } else {
117 meanlife = aLife;
118 }
119
120#ifdef G4VERBOSE
121 if (GetVerboseLevel()>1) {
122 G4cout << "mean life time: "<< meanlife/ns << "[ns]" << G4endl;
123 }
124#endif
125
126 return meanlife;
127}

129G4double G4Decay::GetMeanFreePath(const G4Track& aTrack,G4double, G4ForceCondition*)
130{
131 // get particle
132 const G4DynamicParticle* aParticle = aTrack.GetDynamicParticle();
133 const G4ParticleDefinition* aParticleDef = aParticle->GetDefinition();
134 G4double aMass = aParticle->GetMass();
135 G4double aLife = aParticleDef->GetPDGLifeTime();
136
137
138 // returns the mean free path in GEANT4 internal units
139 G4double pathlength;
140 G4double aCtau = c_light * aLife;
141
142 // check if the particle is stable?
143 if (aParticleDef->GetPDGStable()) {
144 pathlength = DBL_MAX;
145
146 //check if the particle has very short life time ?
147 } else if (aCtau < DBL_MIN) {
148 pathlength = DBL_MIN;
149
150 } else {
151 //calculate the mean free path
152 // by using normalized kinetic energy (= Ekin/mass)
153 G4double rKineticEnergy = aParticle->GetKineticEnergy()/aMass;
154 if ( rKineticEnergy > HighestValue) {
155 // gamma >> 1
156 pathlength = ( rKineticEnergy + 1.0)* aCtau;
157 } else if ( rKineticEnergy < DBL_MIN ) {
158 // too slow particle
159#ifdef G4VERBOSE
160 if (GetVerboseLevel()>1) {
161 G4cout << "G4Decay::GetMeanFreePath() !!particle stops!!";
162 G4cout << aParticleDef->GetParticleName() << G4endl;
163 G4cout << "KineticEnergy:" << aParticle->GetKineticEnergy()/GeV <<"[GeV]";
164 }
165#endif
166 pathlength = DBL_MIN;
167 } else {
168 // beta <1
169 pathlength = (aParticle->GetTotalMomentum())/aMass*aCtau ;
170 }
171 }
172 return pathlength;
173}

180G4VParticleChange* G4Decay::DecayIt(const G4Track& aTrack, const G4Step& )
181{
182 // The DecayIt() method returns by pointer a particle-change object.
183 // Units are expressed in GEANT4 internal units.
184
185 // Initialize ParticleChange
186 // all members of G4VParticleChange are set to equal to
187 // corresponding member in G4Track
188 fParticleChangeForDecay.Initialize(aTrack);
189
190 // get particle
191 const G4DynamicParticle* aParticle = aTrack.GetDynamicParticle();
192 const G4ParticleDefinition* aParticleDef = aParticle->GetDefinition();
193
194 // check if the particle is stable
195 if (aParticleDef->GetPDGStable()) return &fParticleChangeForDecay ;
196
197
198 //check if thePreAssignedDecayProducts exists
199 const G4DecayProducts* o_products = (aParticle->GetPreAssignedDecayProducts());
200 G4bool isPreAssigned = (o_products != nullptr);
201 G4DecayProducts* products = nullptr;
202
203 // decay table
204 G4DecayTable *decaytable = aParticleDef->GetDecayTable();
205
206 // check if external decayer exists
207 G4bool isExtDecayer = (decaytable == nullptr) && (pExtDecayer != nullptr);
208
209 // Error due to NO Decay Table
210 if ( (decaytable == nullptr) && !isExtDecayer && !isPreAssigned ){
211 if (GetVerboseLevel()>0) {
212 G4cout << "G4Decay::DoIt : decay table not defined for ";
213 G4cout << aParticle->GetDefinition()->GetParticleName()<< G4endl;
214 }
215 G4Exception( "G4Decay::DecayIt ",
216 "DECAY101",JustWarning,
217 "Decay table is not defined");
218
219 fParticleChangeForDecay.SetNumberOfSecondaries(0);
220 // Kill the parent particle
221 fParticleChangeForDecay.ProposeTrackStatus( fStopAndKill ) ;
222 fParticleChangeForDecay.ProposeLocalEnergyDeposit(0.0);
223
224 ClearNumberOfInteractionLengthLeft();
225 return &fParticleChangeForDecay ;
226 }
227
228 if (isPreAssigned) {
229 // copy decay products
230 products = new G4DecayProducts(*o_products);
231 } else if ( isExtDecayer ) {
232 // decay according to external decayer
233 products = pExtDecayer->ImportDecayProducts(aTrack);
234 } else {
235 // Decay according to decay table.
236 // Keep trying to choose a candidate decay channel if the dynamic mass
237 // of the decaying particle is below the sum of the PDG masses of the
238 // candidate daughter particles.
239 // This is needed because the decay table used in Geant4 is based on
240 // the assumption of nominal PDG masses, but a wide resonance can have
241 // a dynamic masses well below its nominal PDG masses, and therefore
242 // some of its decay channels can be below the kinematical threshold.
243 // Note that, for simplicity, we ignore here the possibility that
244 // one or more of the candidate daughter particles can be, in turn,
245 // wide resonance. However, if this is the case, and the channel is
246 // accepted, then the masses of the resonance daughter particles will
247 // be sampled by taking into account their widths.
248 G4VDecayChannel* decaychannel = nullptr;
249 G4double massParent = aParticle->GetMass();
250 decaychannel = decaytable->SelectADecayChannel(massParent);
251 if ( decaychannel == nullptr) {
252 // decay channel not found
253 G4ExceptionDescription ed;
254 ed << "Can not determine decay channel for "
255 << aParticleDef->GetParticleName() << G4endl
256 << " mass of dynamic particle: "
257 << massParent/GeV << " (GEV)" << G4endl
258 << " dacay table has " << decaytable->entries()
259 << " entries" << G4endl;
260 G4double checkedmass=massParent;
261 if (massParent < 0.) {
262 checkedmass=aParticleDef->GetPDGMass();
263 ed << "Using PDG mass ("<<checkedmass/GeV
264 << "(GeV)) in IsOKWithParentMass" << G4endl;
265 }
266 for (G4int ic =0;ic <decaytable->entries();++ic) {
267 G4VDecayChannel * dc= decaytable->GetDecayChannel(ic);
268 ed << ic << ": BR " << dc->GetBR() << ", IsOK? "
269 << dc->IsOKWithParentMass(checkedmass)
270 << ", --> ";
271 G4int ndaughters=dc->GetNumberOfDaughters();
272 for (G4int id=0;id<ndaughters;++id) {
273 if (id>0) ed << " + "; // seperator, except for first
274 ed << dc->GetDaughterName(id);
275 }
276 ed << G4endl;
277 }
278 G4Exception("G4Decay::DoIt", "DECAY003", FatalException,ed);
279 } else {
280 // execute DecayIt()
281#ifdef G4VERBOSE
282 G4int temp = decaychannel->GetVerboseLevel();
283 if (GetVerboseLevel()>1) {
284 G4cout << "G4Decay::DoIt : selected decay channel addr:"
285 << decaychannel <<G4endl;
286 decaychannel->SetVerboseLevel(GetVerboseLevel());
287 }
288#endif
289 products = decaychannel->DecayIt(aParticle->GetMass());
290#ifdef G4VERBOSE
291 if (GetVerboseLevel()>1) {
292 decaychannel->SetVerboseLevel(temp);
293 }
294#endif
295#ifdef G4VERBOSE
296 if (GetVerboseLevel()>2) {
297 if (! products->IsChecked() ) products->DumpInfo();
298 }
299#endif
300 }
301 }
302
303 // get parent particle information ...................................
304 G4double ParentEnergy = aParticle->GetTotalEnergy();
305 G4double ParentMass = aParticle->GetMass();
306 if (ParentEnergy < ParentMass) {
307 if (GetVerboseLevel()>0) {
308 G4cout << "G4Decay::DoIt : Total Energy is less than its mass" << G4endl;
309 G4cout << " Particle: " << aParticle->GetDefinition()->GetParticleName();
310 G4cout << " Energy:" << ParentEnergy/MeV << "[MeV]";
311 G4cout << " Mass:" << ParentMass/MeV << "[MeV]";
312 G4cout << G4endl;
313 }
314 G4Exception( "G4Decay::DecayIt ",
315 "DECAY102",JustWarning,
316 "Total Energy is less than its mass");
317 ParentEnergy = ParentMass;
318 }
319
320 G4ThreeVector ParentDirection(aParticle->GetMomentumDirection());
321
322 //boost all decay products to laboratory frame
323 G4double energyDeposit = 0.0;
324 G4double finalGlobalTime = aTrack.GetGlobalTime();
325 G4double finalLocalTime = aTrack.GetLocalTime();
326 if (aTrack.GetTrackStatus() == fStopButAlive ){
327 // AtRest case
328 finalGlobalTime += fRemainderLifeTime;
329 finalLocalTime += fRemainderLifeTime;
330 energyDeposit += aParticle->GetKineticEnergy();
331 if (isPreAssigned) products->Boost( ParentEnergy, ParentDirection);
332 } else {
333 // PostStep case
334 if (!isExtDecayer) products->Boost( ParentEnergy, ParentDirection);
335 }
336 // set polarization for daughter particles
337 DaughterPolarization(aTrack, products);
338
339
340 //add products in fParticleChangeForDecay
341 G4int numberOfSecondaries = products->entries();
342 fParticleChangeForDecay.SetNumberOfSecondaries(numberOfSecondaries);
343#ifdef G4VERBOSE
344 if (GetVerboseLevel()>1) {
345 G4cout << "G4Decay::DoIt : Decay vertex :";
346 G4cout << " Time: " << finalGlobalTime/ns << "[ns]";
347 G4cout << " X:" << (aTrack.GetPosition()).x() /cm << "[cm]";
348 G4cout << " Y:" << (aTrack.GetPosition()).y() /cm << "[cm]";
349 G4cout << " Z:" << (aTrack.GetPosition()).z() /cm << "[cm]";
350 G4cout << G4endl;
351 G4cout << "G4Decay::DoIt : decay products in Lab. Frame" << G4endl;
352 products->DumpInfo();
353 }
354#endif
355 G4int index;
356 G4ThreeVector currentPosition;
357 const G4TouchableHandle thand = aTrack.GetTouchableHandle();
358 for (index=0; index < numberOfSecondaries; index++){
359 // get current position of the track
360 currentPosition = aTrack.GetPosition();
361 // create a new track object
362 G4Track* secondary = new G4Track( products->PopProducts(),
363 finalGlobalTime ,
364 currentPosition );
365 // switch on good for tracking flag
366 secondary->SetGoodForTrackingFlag();
367 secondary->SetTouchableHandle(thand);
368 // add the secondary track in the List
369 fParticleChangeForDecay.AddSecondary(secondary);
370 }
371 delete products;
372
373 // Kill the parent particle
374 fParticleChangeForDecay.ProposeTrackStatus( fStopAndKill ) ;
375 fParticleChangeForDecay.ProposeLocalEnergyDeposit(energyDeposit);
376 fParticleChangeForDecay.ProposeLocalTime( finalLocalTime );
377
378 // Clear NumberOfInteractionLengthLeft
379 ClearNumberOfInteractionLengthLeft();
380
381 return &fParticleChangeForDecay ;
382}

408G4double G4Decay::PostStepGetPhysicalInteractionLength(
409 const G4Track& track,
410 G4double previousStepSize,
411 G4ForceCondition* condition
412 )
413{
414 // condition is set to "Not Forced"
415 *condition = NotForced;
416
417 // pre-assigned Decay time
418 G4double pTime = track.GetDynamicParticle()->GetPreAssignedDecayProperTime();
419 G4double aLife = track.GetDynamicParticle()->GetDefinition()->GetPDGLifeTime();
420
421 if (pTime < 0.) {
422 // normal case
423 if ( previousStepSize > 0.0){
424 // subtract NumberOfInteractionLengthLeft
425 SubtractNumberOfInteractionLengthLeft(previousStepSize);
426 if(theNumberOfInteractionLengthLeft<0.){
427 theNumberOfInteractionLengthLeft=perMillion;
428 }
429 fRemainderLifeTime = theNumberOfInteractionLengthLeft*aLife;
430 }
431 // get mean free path
432 currentInteractionLength = GetMeanFreePath(track, previousStepSize, condition);
433
434#ifdef G4VERBOSE
435 if ((currentInteractionLength <=0.0) || (verboseLevel>2)){
436 G4cout << "G4Decay::PostStepGetPhysicalInteractionLength " << G4endl;
437 track.GetDynamicParticle()->DumpInfo();
438 G4cout << " in Material " << track.GetMaterial()->GetName() <<G4endl;
439 G4cout << "MeanFreePath = " << currentInteractionLength/cm << "[cm]" <<G4endl;
440 }
441#endif
442
443 G4double value;
444 if (currentInteractionLength <DBL_MAX) {
445 value = theNumberOfInteractionLengthLeft * currentInteractionLength;
446 //fRemainderLifeTime = theNumberOfInteractionLengthLeft*aLife;
447 } else {
448 value = DBL_MAX;
449 }
450
451 return value;
452
453 } else {
454 //pre-assigned Decay time case
455 // reminder proper time
456 fRemainderLifeTime = pTime - track.GetProperTime();
457 if (fRemainderLifeTime <= 0.0) fRemainderLifeTime = 0.0;
458
459 G4double rvalue=0.0;
460 // use pre-assigned Decay time to determine PIL
461 if (aLife>0.0) {
462 // ordinary particle
463 rvalue = (fRemainderLifeTime/aLife)*GetMeanFreePath(track, previousStepSize, condition);
464 } else {
465 // shortlived particle
466 rvalue = c_light * fRemainderLifeTime;
467 // by using normalized kinetic energy (= Ekin/mass)
468 G4double aMass = track.GetDynamicParticle()->GetMass();
469 rvalue *= track.GetDynamicParticle()->GetTotalMomentum()/aMass;
470 }
471 return rvalue;
472 }
473}

475G4double G4Decay::AtRestGetPhysicalInteractionLength(
476 const G4Track& track,
477 G4ForceCondition* condition
478 )
479{
480 // condition is set to "Not Forced"
481 *condition = NotForced;
482
483 G4double pTime = track.GetDynamicParticle()->GetPreAssignedDecayProperTime();
484 if (pTime >= 0.) {
485 fRemainderLifeTime = pTime - track.GetProperTime();
486 if (fRemainderLifeTime <= 0.0) fRemainderLifeTime = DBL_MIN;
487 } else {
488 fRemainderLifeTime =
489 theNumberOfInteractionLengthLeft * GetMeanLifeTime(track, condition);
490 }
491 return fRemainderLifeTime;
492}

505G4VParticleChange* G4Decay::PostStepDoIt(
506 const G4Track& aTrack,
507 const G4Step& aStep
508 )
509{
510 if ( (aTrack.GetTrackStatus() == fStopButAlive ) ||
511 (aTrack.GetTrackStatus() == fStopAndKill ) ){
512 fParticleChangeForDecay.Initialize(aTrack);
513 return &fParticleChangeForDecay;
514 } else {
515 return DecayIt(aTrack, aStep);
516 }
