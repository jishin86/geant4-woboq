33// Class description:
34//
35// G4Transportation is a process responsible for the transportation of
36// a particle, i.e. the geometrical propagation encountering the
37// geometrical sub-volumes of the detectors.
38// It is also tasked with part of updating the "safety".
39
40// =======================================================================
41// Created: 19 March 1997, J. Apostolakis
42// =======================================================================
43#ifndef G4Transportation_hh
44#define G4Transportation_hh 1
45
46#include "G4VProcess.hh"
47#include "G4FieldManager.hh"
48
49#include "G4Navigator.hh"
50#include "G4TransportationManager.hh"
51#include "G4PropagatorInField.hh"
52#include "G4Track.hh"
53#include "G4Step.hh"
54#include "G4ParticleChangeForTransport.hh"
55
56class G4SafetyHelper;
57class G4CoupledTransportation;
58class G4TransportationLogger;
59
60class G4Transportation : public G4VProcess
61{
62 // Concrete class that does the geometrical transport
63
64 public: // with description
65
66 G4Transportation( G4int verbosityLevel= 1);
67 ~G4Transportation();
68
69 G4double AlongStepGetPhysicalInteractionLength(
70 const G4Track& track,
71 G4double previousStepSize,
72 G4double currentMinimumStep,
73 G4double& currentSafety,
74 G4GPILSelection* selection
75 ); // override;
76
77 G4VParticleChange* AlongStepDoIt(
78 const G4Track& track,
79 const G4Step& stepData
80 ); // override;
81
82 G4VParticleChange* PostStepDoIt(
83 const G4Track& track,
84 const G4Step& stepData
85 ); // override;
86 // Responsible for the relocation
87
88 G4double PostStepGetPhysicalInteractionLength(
89 const G4Track& ,
90 G4double previousStepSize,
91 G4ForceCondition* pForceCond
92 ); // override;
93 // Forces the PostStepDoIt action to be called,
94 // but does not limit the step
95
96 inline G4bool FieldExertedForce() { return fFieldExertedForce; }
97
98 G4PropagatorInField* GetPropagatorInField();
99 void SetPropagatorInField( G4PropagatorInField* pFieldPropagator);
100 // Access/set the assistant class that Propagate in a Field
101
102 inline G4double GetThresholdWarningEnergy() const;
103 inline G4double GetThresholdImportantEnergy() const;
104 inline G4int GetThresholdTrials() const;
105
106 inline void SetThresholdWarningEnergy( G4double newEnWarn );
107 inline void SetThresholdImportantEnergy( G4double newEnImp );
108 inline void SetThresholdTrials(G4int newMaxTrials );
109 // Get/Set parameters for killing loopers:
110 // Above 'important' energy a 'looping' particle in field will
111 // *NOT* be abandoned, except after fThresholdTrials attempts.
112 // Below Warning energy, no verbosity for looping particles is issued
113
114 void SetHighLooperThresholds(); // Shortcut method - old values (meant for HEP)
115 void SetLowLooperThresholds(); // Set low thresholds - for low-E applications
116 void PushThresholdsToLogger(); // Inform logger of current thresholds
117 void ReportLooperThresholds(); // Print values of looper thresholds
118
119 inline G4double GetMaxEnergyKilled() const;
120 inline G4double GetSumEnergyKilled() const;
121 inline void ResetKilledStatistics( G4int report = 1);
122 // Statistics for tracks killed (currently due to looping in field)
123
124 inline void EnableShortStepOptimisation(G4bool optimise=true);
125 // Whether short steps < safety will avoid to call Navigator (if field=0)
126
127 static G4bool EnableUseMagneticMoment(G4bool useMoment=true);
128 // Whether to deflect particles with force due to magnetic moment
129
130 public: // without description
131
132 G4double AtRestGetPhysicalInteractionLength( const G4Track&,
133 G4ForceCondition*)
134 { return -1.0; } // No operation in AtRestGPIL
135
136 G4VParticleChange* AtRestDoIt( const G4Track&, const G4Step& )
137 { return 0; } // No operation in AtRestDoIt
138
139 void StartTracking(G4Track* aTrack);
140 // Reset state for new (potentially resumed) track
141
142 virtual void ProcessDescription(std::ostream& outFile) const; // override;
143 void PrintStatistics( std::ostream& outStr) const;
144
145 protected:
146
147 G4bool DoesGlobalFieldExist();
148 // Checks whether a field exists for the "global" field manager
149
150 void ReportMissingLogger(const char * methodName);
151
152 private:
153
154 G4Navigator* fLinearNavigator;
155 G4PropagatorInField* fFieldPropagator;
156 // The Propagators used to transport the particle
157
158 G4ThreeVector fTransportEndPosition= G4ThreeVector( 0.0, 0.0, 0.0 );
159 G4ThreeVector fTransportEndMomentumDir= G4ThreeVector( 0.0, 0.0, 0.0 );
160 G4double fTransportEndKineticEnergy= 0.0;
161 G4ThreeVector fTransportEndSpin= G4ThreeVector( 0.0, 0.0, 0.0 );
162 G4bool fMomentumChanged= true;
163 G4bool fEndGlobalTimeComputed= false;
164 G4double fCandidateEndGlobalTime= 0.0;
165 // The particle's state after this Step, Store for DoIt
166
167 G4bool fParticleIsLooping = false;
168 G4bool fNewTrack= true; // Flag from StartTracking
169 G4bool fFirstStepInVolume= true;
170 G4bool fLastStepInVolume= false; // Last step - almost same as next flag
171 // (temporary redundancy for checking)
172 G4bool fGeometryLimitedStep= true;
173 // Flag to determine whether a boundary was reached
174
175 G4bool fFieldExertedForce= false; // During current step
176
177 G4TouchableHandle fCurrentTouchableHandle;
178
179 G4ThreeVector fPreviousSftOrigin;
180 G4double fPreviousSafety;
181 // Remember last safety origin & value.
182
183 G4ParticleChangeForTransport fParticleChange;
184 // New ParticleChange
185
186 G4double fEndPointDistance;
187
188 // Thresholds for looping particles:
189 //
190 G4double fThreshold_Warning_Energy = 1.0 * CLHEP::keV; // Warn above this energy
191 G4double fThreshold_Important_Energy = 1.0 * CLHEP::MeV; // Give a few trial above this E
192 G4int fThresholdTrials = 10; // Number of trials an important looper survives
193 // Above 'important' energy a 'looping' particle in field will
194 // *NOT* be abandoned, except after fThresholdTrials attempts.
195 G4int fAbandonUnstableTrials = 0; // Number of trials after which to abandon
196 // unstable loopers ( 0 = never )
197
198 // Counter for steps in which particle reports 'looping',
199 // ( Used if it is above 'Important' Energy. )
200 G4int fNoLooperTrials= 0;
201
202 // Statistics for tracks abandoned due to looping - and 'saved' despite looping
203 //
204 G4double fSumEnergyKilled= 0.0;
205 G4double fSumEnerSqKilled= 0.0;
206 G4double fMaxEnergyKilled= -1.0;
207 G4int fMaxEnergyKilledPDG= 0;
208 unsigned long fNumLoopersKilled= 0;
209 G4double fSumEnergyKilled_NonElectron= 0.0;
210 G4double fSumEnerSqKilled_NonElectron= 0.0;
211 G4double fMaxEnergyKilled_NonElectron= -1.0;
212 G4int fMaxEnergyKilled_NonElecPDG= 0;
213 unsigned long fNumLoopersKilled_NonElectron= 0;
214 G4double fSumEnergySaved= 0.0;
215 G4double fMaxEnergySaved= -1.0;
216 G4double fSumEnergyUnstableSaved = 0.0;
217 // Whether to avoid calling G4Navigator for short step ( < safety)
218 // If using it, the safety estimate for endpoint will likely be smaller.
219 //
220 G4bool fShortStepOptimisation;
221
222 G4SafetyHelper* fpSafetyHelper; // To pass it the safety value obtained
223 G4TransportationLogger* fpLogger; // Reports issues / raises warnings
224
225 private:
226
227 friend class G4CoupledTransportation;
228 static G4bool fUseMagneticMoment;
229};
