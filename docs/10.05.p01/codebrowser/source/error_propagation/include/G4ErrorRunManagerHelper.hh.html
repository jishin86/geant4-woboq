29// Class Description:
30//
31// This class helps G4ErrorPropagatorManager with the initialization of
32// geometry and physics, definition of actions and run termination
33// (things done by G4RunManager in an standard GEANT4 run)
34// It holds a pointer to G4RunManagerKernel
35
36// History:
37// - Created: Pedro Arce, January 2005
38// --------------------------------------------------------------------
39
40#ifndef G4ErrorRunManagerHelper_hh
41#define G4ErrorRunManagerHelper_hh
42
43#include "tls.hh"
44
45class G4RunManagerKernel;
46class G4VUserDetectorConstruction;
47class G4VPhysicalVolume;
48class G4VUserPhysicsList;
49class G4UserTrackingAction;
50class G4UserSteppingAction;
51
52class G4ErrorRunManagerHelper
53{
54
55 public: // with description
56
57 G4ErrorRunManagerHelper();
58 virtual ~G4ErrorRunManagerHelper();
59
60 static G4ErrorRunManagerHelper* GetRunManagerKernel();
61 // Static method which returns the singleton pointer of
62 // G4ErrorRunManagerHelper
63
64 void SetUserInitialization(G4VUserDetectorConstruction* userInit);
65 // Initialize geometry by passing a G4VUserDetectorConstruction
66 void SetUserInitialization(G4VPhysicalVolume* userInit);
67 // Initialize geometry by passing a G4VPhysicalVolume
68
69 void SetUserInitialization(G4VUserPhysicsList* userInit);
70 // Initializes physics
71
72 void SetUserAction(G4UserTrackingAction* userAction);
73 // Set the user tracking action
74 void SetUserAction(G4UserSteppingAction* userAction);
75 // Set the user stepping action
76
77 void RunInitialization();
78 // Invokes G4RunManagerKernel RunInitialization();
79
80 void InitializeGeometry();
81 // Initializes GEANT4 geometry
82 void InitializePhysics();
83 // Initializes GEANT4 physics
84
85 void RunTermination();
86 // Invokes G4RunManagerKernel RunTermination();
87
88 G4VUserPhysicsList* GetUserPhysicsList() const
89 { return theUserPhysicsList; }
90
91 private:
92
93 static G4ThreadLocal G4ErrorRunManagerHelper* fRunManagerKernel;
94
95 G4VUserPhysicsList* theUserPhysicsList;
96 G4VPhysicalVolume* theUserWorld;
97
98 G4RunManagerKernel* theG4RunManagerKernel;
99};

101#endif
