29// Class Description:
30//
31// Holds the 5 independent variables of the trajectory for a
32// G4ErrorSurfaceTrajState object. It is not used for anything but for
33// printing, but anyhow it is updated everytime the position and momentum
34// are updated
35
36// History:
37// - Created: P. Arce
38// --------------------------------------------------------------------
39
40#ifndef G4ErrorSurfaceTrajParam_hh
41#define G4ErrorSurfaceTrajParam_hh
42
43#include "G4Point3D.hh"
44#include "G4Vector3D.hh"
45#include "G4Plane3D.hh"
46#include "G4ThreeVector.hh"
47
48#include "globals.hh"
49#include "G4Track.hh"
50
51class G4ErrorSurfaceTrajParam
52{
53 public: // with description
54
55 G4ErrorSurfaceTrajParam()
56 : fInvP(0.), fPV(0.), fPW(0.), fV(0.), fW(0.) {}
57 G4ErrorSurfaceTrajParam( const G4Point3D& pos, const G4Vector3D& mom,
58 const G4Vector3D& vecV, const G4Vector3D& vecW );
59 G4ErrorSurfaceTrajParam( const G4Point3D& pos, const G4Vector3D& mom,
60 const G4Plane3D& plane );
61 virtual ~G4ErrorSurfaceTrajParam(){}
62
63 friend
64 std::ostream& operator<<(std::ostream&, const G4ErrorSurfaceTrajParam& ts);
65
66 // Get and Set methods
67
68 void SetParameters( const G4Point3D& pos, const G4Vector3D& mom,
69 const G4Vector3D& vecV, const G4Vector3D& vecW );
70 void SetParameters( const G4Point3D& pos, const G4Vector3D& mom,
71 const G4Plane3D& plane );
72
73 G4Vector3D GetDirection() const { return fDir; }
74 G4Vector3D GetPlaneNormal() const { return fVectorV.cross(fVectorW); }
75 G4Vector3D GetVectorV() const { return fVectorV; }
76 G4Vector3D GetVectorW() const { return fVectorW; }
77 G4double GetPV() const{ return fPV; }
78 G4double GetPW() const{ return fPW; }
79 G4double GetV() const{ return fV; }
80 G4double GetW() const{ return fW; }
81 G4double GetInvP() const{ return fInvP; }
82
83 private:
84
85 G4ThreeVector fDir;
86 G4Vector3D fVectorV; //one of the vectors defining the plane
87 G4Vector3D fVectorW; //one of the vectors defining the plane
88 G4double fInvP; // inverse of momentum
89 G4double fPV; // projection of momentum in one direction
90 G4double fPW; // projection of momentum in one direction
91 G4double fV; // projection of position in one direction
92 G4double fW; // projection of position in one direction
93
94};
95
96#endif
