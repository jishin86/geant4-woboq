32inline G4ErrorSymMatrix::G4ErrorSymMatrix()
33 : m(0), nrow(0), size(0)

35}

42inline G4int G4ErrorSymMatrix::num_col() const

47inline G4int G4ErrorSymMatrix::num_size() const

52inline G4double & G4ErrorSymMatrix::fast(G4int row,G4int col)
53{
54 return *(m.begin()+(row*(row-1))/2+(col-1));
55}
56
57inline const G4double & G4ErrorSymMatrix::fast(G4int row,G4int col) const
58{
59 return *(m.begin()+(row*(row-1))/2+(col-1));
60}
61
62inline G4double &
63G4ErrorSymMatrix::operator()(G4int row, G4int col)
64{
65 return (row>=col? fast(row,col) : fast(col,row));
66}
67
68inline const G4double &
69G4ErrorSymMatrix::operator()(G4int row, G4int col) const
70{
71 return (row>=col? fast(row,col) : fast(col,row));
72}

98inline G4double &G4ErrorSymMatrix::G4ErrorSymMatrix_row::operator[](G4int c)
99{
100 if (_r >= c )
101 {
102 return *(_a.m.begin() + (_r+1)*_r/2 + c);
103 }
104 else
105 {
106 return *(_a.m.begin() + (c+1)*c/2 + _r);
107 }
108}

110inline const G4double &
111G4ErrorSymMatrix::G4ErrorSymMatrix_row_const::operator[](G4int c) const
112{
113 if (_r >= c )
114 {
115 return *(_a.m.begin() + (_r+1)*_r/2 + c);
116 }
117 else
118 {
119 return *(_a.m.begin() + (c+1)*c/2 + _r);
120 }
121}

134inline G4ErrorSymMatrix G4ErrorSymMatrix::inverse(G4int &ifail) const
135{
