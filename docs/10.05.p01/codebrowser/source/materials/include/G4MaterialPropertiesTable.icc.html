41inline
42void G4MaterialPropertiesTable::AddConstProperty(const char *key,
43 G4double PropertyValue)
44{
45 // Provides a way of adding a constant property to the Material Properties
46 // Table given a key
47 G4String k(key);
48 if (std::find(G4MaterialConstPropertyName.begin(),
49 G4MaterialConstPropertyName.end(), k) ==
50 G4MaterialConstPropertyName.end()) {
51 G4MaterialConstPropertyName.push_back(k);
52 }
53 G4int index = GetConstPropertyIndex(k);
54
55 MCP[index] = PropertyValue;
56}
57
58inline
59void G4MaterialPropertiesTable::RemoveConstProperty(const char *key)
60{
61 G4int index = GetConstPropertyIndex(G4String(key));
62
63 MCP.erase(index);
64}
65
66inline
67void G4MaterialPropertiesTable::RemoveProperty(const char *key)
68{
69 G4int index = GetPropertyIndex(G4String(key));
70 MP.erase(index);
71}
