26/// \file materials/include/G4LatticePhysical.hh
27/// \brief Definition of the G4LatticePhysical class
28//
29//
30// 20131114 Add verbosity for diagnostic output
31// 20131116 Replace G4Transform3D with G4RotationMatrix
32
33#ifndef G4LatticePhysical_h
34#define G4LatticePhysical_h 1
35
36#include "G4LatticeLogical.hh"
37#include "G4RotationMatrix.hh"
38#include "G4ThreeVector.hh"
39
40
41class G4LatticePhysical {
42public:
43 G4LatticePhysical(const G4LatticeLogical* Lat=0,
44 const G4RotationMatrix* Rot=0); // Use FRAME rotation
45 virtual ~G4LatticePhysical();
46
47 void SetVerboseLevel(G4int vb) { verboseLevel = vb; }
48
49 // NOTE: Pass by value to allow in-situ rotations
50 G4double MapKtoV(G4int, G4ThreeVector) const;
51 G4ThreeVector MapKtoVDir(G4int, G4ThreeVector) const;
52
53 void SetLatticeLogical(const G4LatticeLogical* Lat) { fLattice = Lat; }
54 void SetPhysicalOrientation(const G4RotationMatrix* Rot); // FRAME rotation
55 void SetLatticeOrientation(G4double, G4double);
56 void SetMillerOrientation(G4int, G4int, G4int);
57
58public:
59 const G4LatticeLogical* GetLattice() const { return fLattice; }
60
61 G4double GetScatteringConstant() const { return fLattice->GetScatteringConstant(); }
62 G4double GetAnhDecConstant() const { return fLattice->GetAnhDecConstant(); }
63 G4double GetLDOS() const { return fLattice->GetLDOS(); }
64 G4double GetSTDOS() const { return fLattice->GetSTDOS(); }
65 G4double GetFTDOS() const { return fLattice->GetFTDOS(); }
66 G4double GetBeta() const { return fLattice->GetBeta(); }
67 G4double GetGamma() const { return fLattice->GetGamma(); }
68 G4double GetLambda() const { return fLattice->GetLambda(); }
69 G4double GetMu() const { return fLattice->GetMu(); }
70
71 // Apply orientation transforms to specified vector
72 G4ThreeVector RotateToGlobal(const G4ThreeVector& dir) const;
73 G4ThreeVector RotateToLocal(const G4ThreeVector& dir) const;
74
75private:
76 G4int verboseLevel; // Enable diagnostic output
77
78 G4double fTheta, fPhi; // Lattice orientation within object
79 const G4LatticeLogical* fLattice; // Underlying lattice parameters
80
81 G4RotationMatrix fLocalToGlobal;
82 G4RotationMatrix fGlobalToLocal;
83};
84
85#endif
