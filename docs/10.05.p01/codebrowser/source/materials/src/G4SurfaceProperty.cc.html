49G4SurfacePropertyTable G4SurfaceProperty::theSurfacePropertyTable;
50
51//
52// Constructor and destructor
53//
54G4SurfaceProperty::G4SurfaceProperty( const G4String& name,
55 G4SurfaceType type )
56 : theName(name), theType(type)
57{
58 theSurfacePropertyTable.push_back(this);
59}
60
61//
62// Dummy constructor
63//
64G4SurfaceProperty::G4SurfaceProperty()
65 : theName("Dielectric"), theType(dielectric_metal)
66{
67 theSurfacePropertyTable.push_back(this);
68}
69
70G4SurfaceProperty::~G4SurfaceProperty()
71{
72}

74//
75// Methods
76//
77
78const G4SurfacePropertyTable* G4SurfaceProperty::GetSurfacePropertyTable()
79{
80 return &theSurfacePropertyTable;
81}
82
83size_t G4SurfaceProperty::GetNumberOfSurfaceProperties()
84{
85 return theSurfacePropertyTable.size();
86}
87
88// Dump info for known surface properties
89//
90void G4SurfaceProperty::DumpTableInfo()
91{
92 G4cout << "***** Surface Property Table : Nb of Surface Properties = "
93 << GetNumberOfSurfaceProperties() << " *****" << G4endl;
94
95 for (size_t i=0; i<theSurfacePropertyTable.size(); i++)
96 {
97 G4SurfaceProperty* pSurfaceProperty = theSurfacePropertyTable[i];
98 G4cout << pSurfaceProperty->GetName() << " : " << G4endl
99 << " Surface Property type = "
100 << pSurfaceProperty->GetType()
101 << G4endl;
102 }
103 G4cout << G4endl;
104}
105
106void G4SurfaceProperty::CleanSurfacePropertyTable()
107{
108 DumpTableInfo();
109 G4SurfacePropertyTable::iterator pos;
110 for(pos=theSurfacePropertyTable.begin();
111 pos!=theSurfacePropertyTable.end(); pos++)
112 {
113 if (*pos) delete *pos;
114 }
115 theSurfacePropertyTable.clear();
116 DumpTableInfo();
117}
