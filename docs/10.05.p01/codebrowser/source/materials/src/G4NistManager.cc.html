71G4NistManager* G4NistManager::Instance()
72{
73 if (instance == nullptr) {
74#ifdef G4MULTITHREADED
75 G4MUTEXLOCK(&nistManagerMutex);
76 if (instance == nullptr) {
77#endif
78 static G4NistManager manager;
79 instance = &manager;
80#ifdef G4MULTITHREADED
81 }
82 G4MUTEXUNLOCK(&nistManagerMutex);
83#endif
84 }
85 return instance;
86}

90G4NistManager::~G4NistManager()
91{
92 // G4cout << "NistManager: start material destruction" << G4endl;
93 const G4MaterialTable* theMaterialTable = G4Material::GetMaterialTable();
94 size_t nmat = theMaterialTable->size();
95 size_t i;
96 for(i=0; i<nmat; i++) {
97 if((*theMaterialTable)[i]) { delete (*theMaterialTable)[i]; }
98 }
99 // G4cout << "NistManager: start element destruction" << G4endl;
100 const G4ElementTable* theElementTable = G4Element::GetElementTable();
101 size_t nelm = theElementTable->size();
102 for(i=0; i<nelm; i++) {
103 if((*theElementTable)[i]) { delete (*theElementTable)[i]; }
104 }
105 // G4cout << "NistManager: start isotope destruction" << G4endl;
106 const G4IsotopeTable* theIsotopeTable = G4Isotope::GetIsotopeTable();
107 size_t niso = theIsotopeTable->size();
108 for(i=0; i<niso; i++) {
109 if((*theIsotopeTable)[i]) { delete (*theIsotopeTable)[i]; }
110 }
111 // G4cout << "NistManager: end isotope destruction" << G4endl;
112 delete messenger;
113 delete matBuilder;
114 delete elmBuilder;
115 delete fICRU90;
116 // G4cout << "NistManager: end destruction" << G4endl;
117}

121G4Material*
122G4NistManager::BuildMaterialWithNewDensity(const G4String& name,
123 const G4String& basename,
124 G4double density,
125 G4double temperature,
126 G4double pressure)
127{
128 G4Material* bmat = FindOrBuildMaterial(name);
129 if(bmat) {
130 G4cout << "G4NistManager::BuildMaterialWithNewDensity ERROR: " << G4endl;
131 G4cout << " New material <" << name << "> cannot be built because material"
132 << " with the same name already exist" << G4endl;
133 G4Exception("G4NistManager::BuildMaterialWithNewDensity()", "mat101",
134 FatalException, "Wrong material name");
135 return 0;
136 }
137 bmat = FindOrBuildMaterial(basename);
138 if(!bmat) {
139 G4cout << "G4NistManager::BuildMaterialWithNewDensity ERROR: " << G4endl;
140 G4cout << " New material <" << name << "> cannot be built because "
141 << G4endl;
142 G4cout << " base material <" << basename << "> does not exist" << G4endl;
143 G4Exception("G4NistManager::BuildMaterialWithNewDensity()", "mat102",
144 FatalException, "Wrong material name");
145 return 0;
146 }
147 G4double dens = density;
148 G4double temp = temperature;
149 G4double pres = pressure;
150 if(dens == 0.0) {
151 dens = bmat->GetDensity();
152 temp = bmat->GetTemperature();
153 pres = bmat->GetPressure();
154 }
155 G4Material* mat = new G4Material(name, dens, bmat, bmat->GetState(),
156 temp, pres);
157 return mat;
158}

162void G4NistManager::PrintElement(const G4String& symbol) const
163{

165 else { elmBuilder->PrintElement(elmBuilder->GetZ(symbol)); }
166}

170void G4NistManager::PrintG4Element(const G4String& name) const
171{
172 const G4ElementTable* theElementTable = G4Element::GetElementTable();
173 size_t nelm = theElementTable->size();
174 for(size_t i=0; i<nelm; i++) {
175 G4Element* elm = (*theElementTable)[i];
176 if ( name == elm->GetName() || "all" == name) {
177 G4cout << *elm << G4endl;
178 }
179 }
180}

207G4NistManager::G4NistManager()
208{
209 nElements = 0;
210 nMaterials = 0;
211 verbose = 0;
212
213 elmBuilder = new G4NistElementBuilder(verbose);
214 matBuilder = new G4NistMaterialBuilder(elmBuilder,verbose);
215
216 messenger = new G4NistMessenger(this);
217 g4pow = G4Pow::GetInstance();
218
219 // compute frequently used values for mean atomic numbers
220 for(G4int j=1; j<101; ++j) {
221 G4double A = elmBuilder->GetAtomicMassAmu(j);
222 POWERA27[j] = std::pow(A,0.27);
223 LOGAZ[j] = std::log(A);
224 }
225 POWERA27[0] = 1.0;
226 LOGAZ[0] = 0.0;
227 fICRU90 = nullptr;
228}

232G4ICRU90StoppingData* G4NistManager::GetICRU90StoppingData()
233{
234 if (!fICRU90) {
235#ifdef G4MULTITHREADED
236 G4MUTEXLOCK(&nistManagerMutex);
237 if (!fICRU90) {
238#endif
239 fICRU90 = new G4ICRU90StoppingData();
240#ifdef G4MULTITHREADED
241 }
242 G4MUTEXUNLOCK(&nistManagerMutex);
243#endif
244 }
245 return fICRU90;
246}
