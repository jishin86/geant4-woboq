50G4NuclideTableMessenger::G4NuclideTableMessenger(G4NuclideTable* nuclideTable)
51 :theNuclideTable(nuclideTable)
52{
53 //Commnad /particle/manage/nuclide
54 thisDirectory = new G4UIdirectory("/particle/nuclideTable/");
55 thisDirectory->SetGuidance("Nuclide table control commands.");
56
57 ///particle/manage/nuclide/min_halflife
58 lifetimeCmd = new G4UIcmdWithADoubleAndUnit("/particle/nuclideTable/min_halflife",this);
59 lifetimeCmd->SetGuidance("Set threshold of half-life.");
60 lifetimeCmd->SetGuidance("Unit of the time can be :");
61 lifetimeCmd->SetGuidance(" s, ms, ns (default)");
62 lifetimeCmd->SetParameterName("life",false);
63 lifetimeCmd->SetDefaultValue( 1000.0 );
64 lifetimeCmd->SetRange("life >0.0");
65 lifetimeCmd->SetDefaultUnit("ns");
66 lifetimeCmd->AvailableForStates(G4State_PreInit,G4State_Idle);
67
68 ///particle/manage/nuclide/level_tolerance
69 lToleranceCmd = new G4UIcmdWithADoubleAndUnit("/particle/nuclideTable/level_tolerance",this);
70 lToleranceCmd->SetGuidance("Set tolerance in level searching.");
71 lToleranceCmd->SetGuidance("Unit of the energy can be :");
72 lToleranceCmd->SetGuidance(" MeV, keV, eV (default)");
73 lToleranceCmd->SetParameterName("lTolerance",false);
74 lToleranceCmd->SetDefaultValue( 1.0 );
75 lToleranceCmd->SetRange("lTolerance >0.0");
76 lToleranceCmd->SetDefaultUnit("eV");
77 lToleranceCmd->AvailableForStates(G4State_PreInit,G4State_Idle);
78
79}
80
81G4NuclideTableMessenger::~G4NuclideTableMessenger()
82{
83 delete thisDirectory;
84 delete lifetimeCmd;
85 delete lToleranceCmd;
86}
87
88void G4NuclideTableMessenger::SetNewValue(G4UIcommand * command,G4String newValue)
89{
90 if (command == lifetimeCmd ) {
91 //Commnad /particle/manage/nuclideTable/min_halflife
92 theNuclideTable->SetThresholdOfHalfLife(lifetimeCmd->GetNewDoubleValue(newValue));
93 } else if (command == lToleranceCmd ) {
94 //Commnad /particle/manage/nuclideTable/level_tolerance
95 theNuclideTable->SetLevelTolerance(lToleranceCmd->GetNewDoubleValue(newValue));
96 }
97}
