124G4DynamicParticle::G4DynamicParticle(const G4ParticleDefinition * aParticleDefinition,
125 const G4ThreeVector& aMomentumDirection,
126 G4double aKineticEnergy,
127 const G4double dynamicalMass):
128 theMomentumDirection(aMomentumDirection),
129 theParticleDefinition(aParticleDefinition),
130 theKineticEnergy(aKineticEnergy),
131 theProperTime(0.0),
132 theDynamicalMass(aParticleDefinition->GetPDGMass()),
133 theDynamicalCharge(aParticleDefinition->GetPDGCharge()),
134 theDynamicalSpin(aParticleDefinition->GetPDGSpin()),
135 theDynamicalMagneticMoment(aParticleDefinition->GetPDGMagneticMoment()),
136 theElectronOccupancy(nullptr),
137 thePreAssignedDecayProducts(nullptr),
138 thePreAssignedDecayTime(-1.0),
139 verboseLevel(1),
140 primaryParticle(nullptr),
141 thePDGcode(0)
142{
143 if (std::abs(theDynamicalMass-dynamicalMass)> EnergyMomentumRelationAllowance) {
144 if (dynamicalMass>EnergyMomentumRelationAllowance) theDynamicalMass= dynamicalMass;
145 else theDynamicalMass= 0.0;
146 }
147}

190G4DynamicParticle::G4DynamicParticle(const G4ParticleDefinition * aParticleDefinition,
191 G4double totalEnergy,
192 const G4ThreeVector &aParticleMomentum):
193 theParticleDefinition(aParticleDefinition),
194 theKineticEnergy(0.0),
195 theProperTime(0.0),
196 theDynamicalMass(aParticleDefinition->GetPDGMass()),
197 theDynamicalCharge(aParticleDefinition->GetPDGCharge()),
198 theDynamicalSpin(aParticleDefinition->GetPDGSpin()),
199 theDynamicalMagneticMoment(aParticleDefinition->GetPDGMagneticMoment()),
200 theElectronOccupancy(nullptr),
201 thePreAssignedDecayProducts(nullptr),
202 thePreAssignedDecayTime(-1.0),
203 verboseLevel(1),
204 primaryParticle(nullptr),
205 thePDGcode(0)
206{
207 // total energy and 3-dim momentum are given
208 G4double pModule2 = aParticleMomentum.mag2();
209 if (pModule2>0.0) {
210 G4double mass2 = totalEnergy*totalEnergy - pModule2;
211 G4double PDGmass2 = (aParticleDefinition->GetPDGMass())*(aParticleDefinition->GetPDGMass());
212 SetMomentumDirection(aParticleMomentum.unit());
213 if (mass2 < EnergyMomentumRelationAllowance*EnergyMomentumRelationAllowance) {
214 theDynamicalMass = 0.;
215 SetKineticEnergy(totalEnergy);
216 } else {
217 if (std::abs(PDGmass2-mass2)>EnergyMomentumRelationAllowance*EnergyMomentumRelationAllowance){
218 theDynamicalMass = std::sqrt(mass2);
219 SetKineticEnergy(totalEnergy-theDynamicalMass);
220 } else {
221 SetKineticEnergy(totalEnergy-theDynamicalMass);
222 }
223 }
224 } else {
225 SetMomentumDirection(1.0,0.0,0.0);
226 SetKineticEnergy(0.0);
227 }
228}

268////////////////////
269// -- operators ----
270////////////////////
271G4DynamicParticle & G4DynamicParticle::operator=(const G4DynamicParticle &right)
272{
273 if (this != &right) {
274 theMomentumDirection = right.theMomentumDirection;
275 theParticleDefinition = right.theParticleDefinition;
276 thePolarization = right.thePolarization;
277 theKineticEnergy = right.theKineticEnergy;
278 theProperTime = right.theProperTime;
279
280 theDynamicalMass = right.theDynamicalMass;
281 theDynamicalCharge = right.theDynamicalCharge;
282 theDynamicalSpin = right.theDynamicalSpin;
283 theDynamicalMagneticMoment = right.theDynamicalMagneticMoment;
284
285 if (theElectronOccupancy != nullptr) delete theElectronOccupancy;
286 if (right.theElectronOccupancy != nullptr){
287 theElectronOccupancy =
288 new G4ElectronOccupancy(*right.theElectronOccupancy);
289 } else {
290 theElectronOccupancy = nullptr;
291 }
292
293 // thePreAssignedDecayProducts must not be copied.
294 thePreAssignedDecayProducts = 0;
295 thePreAssignedDecayTime = -1.0;
296
297 verboseLevel = right.verboseLevel;
298
299 // Primary particle information must be preserved
300 //*** primaryParticle = right.primaryParticle;
301
302 thePDGcode = right.thePDGcode;
303 }
304 return *this;
305}

307////////////////////
308void G4DynamicParticle::SetDefinition(const G4ParticleDefinition * aParticleDefinition)
309{
310 // remove preassigned decay
311 if (thePreAssignedDecayProducts != nullptr) {
312#ifdef G4VERBOSE
313 if (verboseLevel>0) {
314 G4cout << " G4DynamicParticle::SetDefinition()::"
315 << "!!! Pre-assigned decay products is attached !!!! " << G4endl;
316 G4cout << "!!! New Definition is " << aParticleDefinition->GetParticleName()
317 << " !!! " << G4endl;
318 G4cout << "!!! Pre-assigned decay products will be deleted !!!! " << G4endl;
319 }
320#endif
321 delete thePreAssignedDecayProducts;
322 }
323 thePreAssignedDecayProducts = nullptr;
324
325 theParticleDefinition = aParticleDefinition;
326
327 // set Dynamic mass/chrge
328 theDynamicalMass = theParticleDefinition->GetPDGMass();
329 theDynamicalCharge = theParticleDefinition->GetPDGCharge();
330 theDynamicalSpin = theParticleDefinition->GetPDGSpin();
331 theDynamicalMagneticMoment = theParticleDefinition->GetPDGMagneticMoment();
332
333 // Set electron orbits
334 if (theElectronOccupancy != nullptr) delete theElectronOccupancy;
335 theElectronOccupancy = nullptr;
336
337}

370////////////////////
371// -- methods for setting Energy/Momentum --
372////////////////////
373void G4DynamicParticle::SetMomentum(const G4ThreeVector &momentum)
374{
375 G4double pModule2 = momentum.mag2();
376 if (pModule2>0.0) {
377 G4double mass = theDynamicalMass;
378 SetMomentumDirection(momentum.unit());
379 SetKineticEnergy(std::sqrt(pModule2 + mass*mass)-mass);
380 } else {
381 SetMomentumDirection(1.0,0.0,0.0);
382 SetKineticEnergy(0.0);
383 }
384}
385
386////////////////////
387void G4DynamicParticle::Set4Momentum(const G4LorentzVector &momentum )
388{
389 G4double pModule2 = momentum.vect().mag2();
390 if (pModule2>0.0) {
391 SetMomentumDirection(momentum.vect().unit());
392 G4double totalenergy = momentum.t();
393 G4double mass2 = totalenergy*totalenergy - pModule2;
394 G4double PDGmass2 = (theParticleDefinition->GetPDGMass())*(theParticleDefinition->GetPDGMass());
395 if (mass2 < EnergyMomentumRelationAllowance*EnergyMomentumRelationAllowance) {
396 theDynamicalMass = 0.;
397 } else if (std::abs(PDGmass2-mass2)>EnergyMomentumRelationAllowance*EnergyMomentumRelationAllowance){
398 theDynamicalMass = std::sqrt(mass2);
399 }
400 SetKineticEnergy(totalenergy-theDynamicalMass);
401 } else {
402 SetMomentumDirection(1.0,0.0,0.0);
403 SetKineticEnergy(0.0);
404 }
405}

447void G4DynamicParticle::SetMass(G4double newMass)
448{
449 if(std::abs(newMass-theParticleDefinition->GetPDGMass())>EnergyMomentumRelationAllowance*0.1) {
450 theDynamicalMass = newMass;
451 }

453////////////////////////
454G4double G4DynamicParticle::GetElectronMass() const
455{
456 static G4ThreadLocal G4double electronMass = 0.0;
457
458 // check if electron exits and get the mass
459 if (electronMass<=0.0) {
460 G4ParticleDefinition* electron = G4ParticleTable::GetParticleTable()->FindParticle("e-");
461 if (electron == nullptr) {
462 G4Exception("G4DynamicParticle::GetElectronMass()","PART021",
463 FatalException,"G4DynamicParticle: G4Electron is not defined !!");
464 }
465 electronMass = electron->GetPDGMass();
466 }
467
468 return electronMass;
