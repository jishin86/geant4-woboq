41inline void * G4DynamicParticle::operator new(size_t)
42{
43 if (!pDynamicParticleAllocator()) pDynamicParticleAllocator() =
44 new G4Allocator<G4DynamicParticle>;
45 return pDynamicParticleAllocator()->MallocSingle();
46}

48inline void G4DynamicParticle::operator delete(void * aDynamicParticle)
49{
50 pDynamicParticleAllocator()->FreeSingle((G4DynamicParticle *) aDynamicParticle);
51}

80inline void G4DynamicParticle::AddElectron(G4int orbit, G4int number )
81{
82 if ( theElectronOccupancy == 0) AllocateElectronOccupancy();
83 if ( theElectronOccupancy != 0) {
84 G4int n = theElectronOccupancy->AddElectron(orbit, number );
85 theDynamicalCharge -= CLHEP::eplus * n;
86 theDynamicalMass += GetElectronMass() * n;
87 }
88}
89
90inline void G4DynamicParticle::RemoveElectron(G4int orbit, G4int number)
91{
92 if ( theElectronOccupancy == 0) AllocateElectronOccupancy();
93 if ( theElectronOccupancy != 0) {
94 G4int n = theElectronOccupancy->RemoveElectron(orbit, number );
95 theDynamicalCharge += CLHEP::eplus * n;
96 theDynamicalMass -= GetElectronMass() * n;
97 }
98}

151inline G4ThreeVector G4DynamicParticle::GetMomentum() const
152{
153 G4double pModule = std::sqrt(theKineticEnergy*theKineticEnergy +
154 2*theKineticEnergy*theDynamicalMass);
155 G4ThreeVector pMomentum(theMomentumDirection.x()*pModule,
156 theMomentumDirection.y()*pModule,
157 theMomentumDirection.z()*pModule);
158 return pMomentum;
159}
160
161inline G4LorentzVector G4DynamicParticle::Get4Momentum() const
162{
163 G4double mass = theDynamicalMass;
164 G4double energy = theKineticEnergy;
165 G4double momentum = std::sqrt(energy*energy+2.0*mass*energy);
166 G4LorentzVector p4( theMomentumDirection.x()*momentum,
167 theMomentumDirection.y()*momentum,
168 theMomentumDirection.z()*momentum,
169 energy+mass);
170 return p4;
171}
172
173inline G4double G4DynamicParticle::GetTotalMomentum() const
174{
175 // The momentum is returned in energy equivalent.
176 return std::sqrt((theKineticEnergy + 2.*theDynamicalMass)* theKineticEnergy);
177}
