38#ifndef G4DecayProducts_h
39#define G4DecayProducts_h 1
40
41#include "G4ios.hh"
42#include "globals.hh"
43#include "G4DynamicParticle.hh"
44#include <vector>
45
46class G4DecayProducts
47{
48 public: // With Description
49
50 // constructors
51 G4DecayProducts();
52 G4DecayProducts(const G4DynamicParticle &aParticle);
53
54 public:
55 // copy constructor and assignment operator
56 // Deep copy: for G4DynamicParticle pointer
57 G4DecayProducts(const G4DecayProducts &right);
58 G4DecayProducts & operator=(const G4DecayProducts &right);
59
60 //destructor
61 ~G4DecayProducts();
62
63 // (un)equal operator
64 G4bool operator==(const G4DecayProducts &right) const;
65 G4bool operator!=(const G4DecayProducts &right) const;
66
67 public: // With Description
68 // set-get methods for the parent particle
69 // theParentPaticle is used to get information of parent particle
70 // when decay products are filled
71 // new G4DynamicParticle object is created in set methods
72 const G4DynamicParticle* GetParentParticle() const {return theParentParticle;};
73 void SetParentParticle(const G4DynamicParticle &aParticle);
74
75 // boost all products
76 void Boost(G4double totalEnergy, const G4ThreeVector &momentumDirection);
77 void Boost(G4double betax, G4double betay, G4double betaz);
78
79 // push-pop methods for decay products pointer
80 G4DynamicParticle* PopProducts();
81 G4int PushProducts(G4DynamicParticle *aParticle);
82
83 G4DynamicParticle* operator[](G4int anIndex) const;
84
85 G4int entries() const {return numberOfProducts;};
86
87 // check energy/momentum of products
88 G4bool IsChecked() const;
89
90 //
91 void DumpInfo() const;
92
93 typedef std::vector<G4DynamicParticle*> G4DecayProductVector;
94
95 private:
96 G4int numberOfProducts;
97 G4DynamicParticle* theParentParticle;
98 G4DecayProductVector* theProductVector;
99
100};
101
102// ------------------------
103// Inlined operators
104// ------------------------
105
106inline
107 G4bool G4DecayProducts::operator==(const G4DecayProducts &right) const
108{
109 return (this == (G4DecayProducts *) &right);
110}
111
112inline
113 G4bool G4DecayProducts::operator!=(const G4DecayProducts &right) const
114{
115 return (this != (G4DecayProducts *) &right);
116}
