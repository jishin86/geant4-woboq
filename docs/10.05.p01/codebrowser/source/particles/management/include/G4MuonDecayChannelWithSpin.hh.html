50class G4MuonDecayChannelWithSpin : public G4MuonDecayChannel
51{
52 // Class Decription
53 // This class describes muon decay kinemtics.
54 // This version assumes V-A coupling with 1st order radiative correctons,
55 // the standard model Michel parameter values, but
56 // gives incorrect energy spectrum for neutrinos
57
58public: // With Description
59

62 G4double theBR);

71private:

78private:
79// Radiative Correction Factors
80
81 G4double F_c(G4double x, G4double x0 , G4double omega);
82 G4double F_theta(G4double x, G4double x0 , G4double omega);
83 G4double R_c(G4double x , G4double omega);
84
85};
86
87inline G4double G4MuonDecayChannelWithSpin::F_c(G4double x, G4double x0 , G4double omega)
88{
89
90 G4double f_c;
91
92 f_c = (5.+17.*x-34.*x*x)*(omega+std::log(x))-22.*x+34.*x*x;
93 f_c = (1.-x)/(3.*x*x)*f_c;
94 f_c = (6.-4.*x)*R_c(x,omega)+(6.-6.*x)*std::log(x) + f_c;
95 f_c = (CLHEP::fine_structure_const/CLHEP::twopi) * (x*x-x0*x0) * f_c;
96
97 return f_c;
98}
99
100inline G4double G4MuonDecayChannelWithSpin::F_theta(G4double x, G4double x0,G4double omega)
101{
102 G4double f_theta;
103
104 f_theta = (1.+x+34*x*x)*(omega+std::log(x))+3.-7.*x-32.*x*x;
105 f_theta = f_theta + ((4.*(1.-x)*(1.-x))/x)*std::log(1.-x);
106 f_theta = (1.-x)/(3.*x*x) * f_theta;
107 f_theta = (2.-4.*x)*R_c(x,omega)+(2.-6.*x)*std::log(x)-f_theta;
108 f_theta = (CLHEP::fine_structure_const/CLHEP::twopi) * (x*x-x0*x0) * f_theta;
109
110 return f_theta;
111}
