41inline
42 G4int G4ParticlePropertyData::GetQuarkContent(G4int flavor) const
43{
44 if ((flavor>0) && (flavor<=NumberOfQuarkFlavor)){
45 return theQuarkContent[flavor-1];
46 }else {
47 return 0;
48 }
49}
