35#include "G4GeometryCell.hh"
36
37G4GeometryCell::G4GeometryCell(const G4VPhysicalVolume &aVolume,
38 G4int RepNum)
39 : fVPhysicalVolume(&aVolume),
40 fRepNum(RepNum)
41{}
42
43G4GeometryCell::~G4GeometryCell()
44{}
45
46
47const G4VPhysicalVolume &G4GeometryCell::GetPhysicalVolume() const
48{
49 return *fVPhysicalVolume;
50}
51
52G4int G4GeometryCell::GetReplicaNumber() const
53{
54 return fRepNum;
55}
56
57
58G4GeometryCell::G4GeometryCell(const G4GeometryCell &rhs)
59 : fVPhysicalVolume(rhs.fVPhysicalVolume),
60 fRepNum(rhs.fRepNum)
61{
62}
63
64G4GeometryCell &G4GeometryCell::operator=(const G4GeometryCell &rhs){
65 if (this != &rhs) {
66 fVPhysicalVolume = rhs.fVPhysicalVolume; // this is treated
67 // as identifyer
68 fRepNum = rhs.fRepNum;
69 }
70 return *this;
71}
72
73
74G4bool operator==(const G4GeometryCell &k1, const G4GeometryCell &k2)
75{
76 G4bool equal=true;
77 if (&(k1.GetPhysicalVolume()) != &(k2.GetPhysicalVolume())) {
78 equal = false;
79 }
80 else if (k1.GetReplicaNumber() != k2.GetReplicaNumber()) {
81 equal = false;
82 }
83 return equal;
84}
85
86G4bool operator!=(const G4GeometryCell &k1, const G4GeometryCell &k2)
87{
88 G4bool unequal = false;
89 if (&(k1.GetPhysicalVolume()) != &(k2.GetPhysicalVolume())) {
90 unequal = true;
91 }
92 else if (k1.GetReplicaNumber() != k2.GetReplicaNumber()) {
93 unequal = true;
94 }
95 return unequal;
96}
