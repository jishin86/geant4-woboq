56///////////////////////////////////////////////////////////////////////////////
57//
58G4bool G4VFacet::operator== (const G4VFacet &right) const
59{
60 G4double tolerance = kCarTolerance*kCarTolerance/4.0;
61
62 if (GetNumberOfVertices() != right.GetNumberOfVertices())
63 return false;
64 else if ((GetCircumcentre()-right.GetCircumcentre()).mag2() > tolerance)
65 return false;
66 else if (std::fabs((right.GetSurfaceNormal()).dot(GetSurfaceNormal())) < 0.9999999999)
67 return false;
68
69 G4bool coincident = true;
70 G4int i = 0;
71 do // Loop checking, 13.08.2015, G.Cosmo
72 {
73 coincident = false;
74 G4int j = 0;
75 do // Loop checking, 13.08.2015, G.Cosmo
76 {
77 coincident = (GetVertex(i)-right.GetVertex(j)).mag2() < tolerance;
78 } while (!coincident && ++j < GetNumberOfVertices());
79 } while (coincident && ++i < GetNumberOfVertices());
80
81 return coincident;
82}

84///////////////////////////////////////////////////////////////////////////////
85//
86void G4VFacet::ApplyTranslation(const G4ThreeVector v)
87{
88 G4int n = GetNumberOfVertices();
89 for (G4int i = 0; i < n; ++i)
90 {
91 SetVertex(i, GetVertex(i) + v);
92 }
93}

97std::ostream &G4VFacet::StreamInfo(std::ostream &os) const
98{
99 os << G4endl;
100 os << "*********************************************************************"
101 << G4endl;
102 os << "FACET TYPE = " << GetEntityType() << G4endl;
103 os << "ABSOLUTE VECTORS = " << G4endl;
104 G4int n = GetNumberOfVertices();
105 for (G4int i = 0; i < n; ++i)
106 os << "P[" << i << "] = " << GetVertex(i) << G4endl;
107 os << "*********************************************************************"
108 << G4endl;
109
110 return os;
111}

113G4bool G4VFacet::IsInside (const G4ThreeVector &p) const
114{
115 G4ThreeVector d = p - GetVertex(0);
116 G4double displacement = d.dot(GetSurfaceNormal());
117 return displacement <= 0.0;
118}
