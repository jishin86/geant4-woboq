50G4TwistTubsHypeSide::G4TwistTubsHypeSide(const G4String &name,
51 const G4RotationMatrix &rot,
52 const G4ThreeVector &tlate,
53 const G4int handedness,
54 const G4double kappa,
55 const G4double tanstereo,
56 const G4double r0,
57 const EAxis axis0,
58 const EAxis axis1,
59 G4double axis0min,
60 G4double axis1min,
61 G4double axis0max,
62 G4double axis1max )
63 : G4VTwistSurface(name, rot, tlate, handedness, axis0, axis1,
64 axis0min, axis1min, axis0max, axis1max),
65 fKappa(kappa), fTanStereo(tanstereo),
66 fTan2Stereo(tanstereo*tanstereo), fR0(r0), fR02(r0*r0), fDPhi(twopi)
67{
68 if ( (axis0 == kZAxis) && (axis1 == kPhi) )
69 {
70 G4Exception("G4TwistTubsHypeSide::G4TwistTubsHypeSide()",
71 "GeomSolids0002", FatalErrorInArgument,
72 "Should swap axis0 and axis1!");
73 }
74
75 fInside.gp.set(kInfinity, kInfinity, kInfinity);
76 fInside.inside = kOutside;
77 fIsValidNorm = false;
78
79 SetCorners();
80 SetBoundaries();
81
82}

84G4TwistTubsHypeSide::G4TwistTubsHypeSide(const G4String &name,
85 G4double EndInnerRadius[2],
86 G4double EndOuterRadius[2],
87 G4double DPhi,
88 G4double EndPhi[2],
89 G4double EndZ[2],
90 G4double InnerRadius,
91 G4double OuterRadius,
92 G4double Kappa,
93 G4double TanInnerStereo,
94 G4double TanOuterStereo,
95 G4int handedness)
96 : G4VTwistSurface(name)
97{
98
99 fHandedness = handedness; // +z = +ve, -z = -ve
100 fAxis[0] = kPhi;
101 fAxis[1] = kZAxis;
102 fAxisMin[0] = kInfinity; // we cannot fix boundary min of Phi,
103 fAxisMax[0] = kInfinity; // because it depends on z.
104 fAxisMin[1] = EndZ[0];
105 fAxisMax[1] = EndZ[1];
106 fKappa = Kappa;
107 fDPhi = DPhi ;
108
109 if (handedness < 0) { // inner hyperbolic surface
110 fTanStereo = TanInnerStereo;
111 fR0 = InnerRadius;
112 } else { // outer hyperbolic surface
113 fTanStereo = TanOuterStereo;
114 fR0 = OuterRadius;
115 }
116 fTan2Stereo = fTanStereo * fTanStereo;
117 fR02 = fR0 * fR0;
118
119 fTrans.set(0, 0, 0);
120 fIsValidNorm = false;
121
122 fInside.gp.set(kInfinity, kInfinity, kInfinity);
123 fInside.inside = kOutside;
124
125 SetCorners(EndInnerRadius, EndOuterRadius, DPhi, EndPhi, EndZ) ;
126
127 SetBoundaries();
128}

148
149G4ThreeVector G4TwistTubsHypeSide::GetNormal(const G4ThreeVector &tmpxx,
150 G4bool isGlobal)
151{
152 // GetNormal returns a normal vector at a surface (or very close
153 // to surface) point at tmpxx.
154 // If isGlobal=true, it returns the normal in global coordinate.
155 //
156
157 G4ThreeVector xx;
158 if (isGlobal) {
159 xx = ComputeLocalPoint(tmpxx);
160 if ((xx - fCurrentNormal.p).mag() < 0.5 * kCarTolerance) {
161 return ComputeGlobalDirection(fCurrentNormal.normal);
162 }
163 } else {
164 xx = tmpxx;
165 if (xx == fCurrentNormal.p) {
166 return fCurrentNormal.normal;
167 }
168 }
169
170 fCurrentNormal.p = xx;
171
172 G4ThreeVector normal( xx.x(), xx.y(), -xx.z() * fTan2Stereo);
173 normal *= fHandedness;
174 normal = normal.unit();
175
176 if (isGlobal) {
177 fCurrentNormal.normal = ComputeLocalDirection(normal);
178 } else {
179 fCurrentNormal.normal = normal;
180 }
181 return fCurrentNormal.normal;
182}

185//* Inside() ----------------------------------------------------------
186
187EInside G4TwistTubsHypeSide::Inside(const G4ThreeVector &gp)
188{
189 // Inside returns
190 const G4double halftol
191 = 0.5 * G4GeometryTolerance::GetInstance()->GetRadialTolerance();
192
193 if (fInside.gp == gp) {
194 return fInside.inside;
195 }
196 fInside.gp = gp;
197
198 G4ThreeVector p = ComputeLocalPoint(gp);
199
200
201 if (p.mag() < DBL_MIN) {
202 fInside.inside = kOutside;
203 return fInside.inside;
204 }
205
206 G4double rhohype = GetRhoAtPZ(p);
207 G4double distanceToOut = fHandedness * (rhohype - p.getRho());
208 // +ve : inside
209
210 if (distanceToOut < -halftol) {
211
212 fInside.inside = kOutside;
213
214 } else {
215
216 G4int areacode = GetAreaCode(p);
217 if (IsOutside(areacode)) {
218 fInside.inside = kOutside;
219 } else if (IsBoundary(areacode)) {
220 fInside.inside = kSurface;
221 } else if (IsInside(areacode)) {
222 if (distanceToOut <= halftol) {
223 fInside.inside = kSurface;
224 } else {
225 fInside.inside = kInside;
226 }
227 } else {
228 G4cout << "WARNING - G4TwistTubsHypeSide::Inside()" << G4endl
229 << " Invalid option !" << G4endl
230 << " name, areacode, distanceToOut = "
231 << GetName() << ", " << std::hex << areacode << std::dec << ", "
232 << distanceToOut << G4endl;
233 }
234 }
235
236 return fInside.inside;
237}

240//* DistanceToSurface -------------------------------------------------
241
242G4int G4TwistTubsHypeSide::DistanceToSurface(const G4ThreeVector &gp,
243 const G4ThreeVector &gv,
244 G4ThreeVector gxx[],
245 G4double distance[],
246 G4int areacode[],
247 G4bool isvalid[],
248 EValidate validate)
249{
250 //
251 // Decide if and where a line intersects with a hyperbolic
252 // surface (of infinite extent)
253 //
254 // Arguments:
255 // p - (in) Point on trajectory
256 // v - (in) Vector along trajectory
257 // r2 - (in) Square of radius at z = 0
258 // tan2phi - (in) std::tan(stereo)**2
259 // s - (out) Up to two points of intersection, where the
260 // intersection point is p + s*v, and if there are
261 // two intersections, s[0] < s[1]. May be negative.
262 // Returns:
263 // The number of intersections. If 0, the trajectory misses.
264 //
265 //
266 // Equation of a line:
267 //
268 // x = x0 + s*tx y = y0 + s*ty z = z0 + s*tz
269 //
270 // Equation of a hyperbolic surface:
271 //
272 // x**2 + y**2 = r**2 + (z*tanPhi)**2
273 //
274 // Solution is quadratic:
275 //
276 // a*s**2 + b*s + c = 0
277 //
278 // where:
279 //
280 // a = tx**2 + ty**2 - (tz*tanPhi)**2
281 //
282 // b = 2*( x0*tx + y0*ty - z0*tz*tanPhi**2 )
283 //
284 // c = x0**2 + y0**2 - r**2 - (z0*tanPhi)**2
285 //
286
287 fCurStatWithV.ResetfDone(validate, &gp, &gv);
288
289 if (fCurStatWithV.IsDone()) {
290 G4int i;
291 for (i=0; i<fCurStatWithV.GetNXX(); i++) {
292 gxx[i] = fCurStatWithV.GetXX(i);
293 distance[i] = fCurStatWithV.GetDistance(i);
294 areacode[i] = fCurStatWithV.GetAreacode(i);
295 isvalid[i] = fCurStatWithV.IsValid(i);
296 }
297 return fCurStatWithV.GetNXX();
298 } else {
299 // initialize
300 G4int i;
301 for (i=0; i<2; i++) {
302 distance[i] = kInfinity;
303 areacode[i] = sOutside;
304 isvalid[i] = false;
305 gxx[i].set(kInfinity, kInfinity, kInfinity);
306 }
307 }
308
309 G4ThreeVector p = ComputeLocalPoint(gp);
310 G4ThreeVector v = ComputeLocalDirection(gv);
311 G4ThreeVector xx[2];
312
313 //
314 // special case! p is on origin.
315 //
316
317 if (p.mag() == 0) {
318 // p is origin.
319 // unique solution of 2-dimension question in r-z plane
320 // Equations:
321 // r^2 = fR02 + z^2*fTan2Stere0
322 // r = beta*z
323 // where
324 // beta = vrho / vz
325 // Solution (z value of intersection point):
326 // xxz = +- std::sqrt (fR02 / (beta^2 - fTan2Stereo))
327 //
328
329 G4double vz = v.z();
330 G4double absvz = std::fabs(vz);
331 G4double vrho = v.getRho();
332 G4double vslope = vrho/vz;
333 G4double vslope2 = vslope * vslope;
334 if (vrho == 0 || (vrho/absvz) <= (absvz*std::fabs(fTanStereo)/absvz)) {
335 // vz/vrho is bigger than slope of asymptonic line
336 distance[0] = kInfinity;
337 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
338 isvalid[0], 0, validate, &gp, &gv);
339 return 0;
340 }
341
342 if (vz) {
343 G4double xxz = std::sqrt(fR02 / (vslope2 - fTan2Stereo))
344 * (vz / std::fabs(vz)) ;
345 G4double t = xxz / vz;
346 xx[0].set(t*v.x(), t*v.y(), xxz);
347 } else {
348 // p.z = 0 && v.z =0
349 xx[0].set(v.x()*fR0, v.y()*fR0, 0); // v is a unit vector.
350 }
351 distance[0] = xx[0].mag();
352 gxx[0] = ComputeGlobalPoint(xx[0]);
353
354 if (validate == kValidateWithTol) {
355 areacode[0] = GetAreaCode(xx[0]);
356 if (!IsOutside(areacode[0])) {
357 if (distance[0] >= 0) isvalid[0] = true;
358 }
359 } else if (validate == kValidateWithoutTol) {
360 areacode[0] = GetAreaCode(xx[0], false);
361 if (IsInside(areacode[0])) {
362 if (distance[0] >= 0) isvalid[0] = true;
363 }
364 } else { // kDontValidate
365 areacode[0] = sInside;
366 if (distance[0] >= 0) isvalid[0] = true;
367 }
368
369 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
370 isvalid[0], 1, validate, &gp, &gv);
371 return 1;
372 }
373
374 //
375 // special case end.
376 //
377
378 G4double a = v.x()*v.x() + v.y()*v.y() - v.z()*v.z()*fTan2Stereo;
379 G4double b = 2.0 * ( p.x() * v.x() + p.y() * v.y() - p.z() * v.z() * fTan2Stereo );
380 G4double c = p.x()*p.x() + p.y()*p.y() - fR02 - p.z()*p.z()*fTan2Stereo;
381 G4double D = b*b - 4*a*c; //discriminant
382 G4int vout = 0;
383
384 if (std::fabs(a) < DBL_MIN) {
385 if (std::fabs(b) > DBL_MIN) { // single solution
386
387 distance[0] = -c/b;
388 xx[0] = p + distance[0]*v;
389 gxx[0] = ComputeGlobalPoint(xx[0]);
390
391 if (validate == kValidateWithTol) {
392 areacode[0] = GetAreaCode(xx[0]);
393 if (!IsOutside(areacode[0])) {
394 if (distance[0] >= 0) isvalid[0] = true;
395 }
396 } else if (validate == kValidateWithoutTol) {
397 areacode[0] = GetAreaCode(xx[0], false);
398 if (IsInside(areacode[0])) {
399 if (distance[0] >= 0) isvalid[0] = true;
400 }
401 } else { // kDontValidate
402 areacode[0] = sInside;
403 if (distance[0] >= 0) isvalid[0] = true;
404 }
405
406 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
407 isvalid[0], 1, validate, &gp, &gv);
408 vout = 1;
409
410 } else {
411 // if a=b=0 and c != 0, p is origin and v is parallel to asymptotic line.
412 // if a=b=c=0, p is on surface and v is paralell to stereo wire.
413 // return distance = infinity.
414
415 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
416 isvalid[0], 0, validate, &gp, &gv);
417
418 vout = 0;
419 }
420
421 } else if (D > DBL_MIN) { // double solutions
422
423 D = std::sqrt(D);
424 G4double factor = 0.5/a;
425 G4double tmpdist[2] = {kInfinity, kInfinity};
426 G4ThreeVector tmpxx[2] ;
427 G4int tmpareacode[2] = {sOutside, sOutside};
428 G4bool tmpisvalid[2] = {false, false};
429 G4int i;
430
431 for (i=0; i<2; i++) {
432 tmpdist[i] = factor*(-b - D);
433 D = -D;
434 tmpxx[i] = p + tmpdist[i]*v;
435
436 if (validate == kValidateWithTol) {
437 tmpareacode[i] = GetAreaCode(tmpxx[i]);
438 if (!IsOutside(tmpareacode[i])) {
439 if (tmpdist[i] >= 0) tmpisvalid[i] = true;
440 continue;
441 }
442 } else if (validate == kValidateWithoutTol) {
443 tmpareacode[i] = GetAreaCode(tmpxx[i], false);
444 if (IsInside(tmpareacode[i])) {
445 if (tmpdist[i] >= 0) tmpisvalid[i] = true;
446 continue;
447 }
448 } else { // kDontValidate
449 tmpareacode[i] = sInside;
450 if (tmpdist[i] >= 0) tmpisvalid[i] = true;
451 continue;
452 }
453 }
454
455 if (tmpdist[0] <= tmpdist[1]) {
456 distance[0] = tmpdist[0];
457 distance[1] = tmpdist[1];
458 xx[0] = tmpxx[0];
459 xx[1] = tmpxx[1];
460 gxx[0] = ComputeGlobalPoint(tmpxx[0]);
461 gxx[1] = ComputeGlobalPoint(tmpxx[1]);
462 areacode[0] = tmpareacode[0];
463 areacode[1] = tmpareacode[1];
464 isvalid[0] = tmpisvalid[0];
465 isvalid[1] = tmpisvalid[1];
466 } else {
467 distance[0] = tmpdist[1];
468 distance[1] = tmpdist[0];
469 xx[0] = tmpxx[1];
470 xx[1] = tmpxx[0];
471 gxx[0] = ComputeGlobalPoint(tmpxx[1]);
472 gxx[1] = ComputeGlobalPoint(tmpxx[0]);
473 areacode[0] = tmpareacode[1];
474 areacode[1] = tmpareacode[0];
475 isvalid[0] = tmpisvalid[1];
476 isvalid[1] = tmpisvalid[0];
477 }
478
479 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
480 isvalid[0], 2, validate, &gp, &gv);
481 fCurStatWithV.SetCurrentStatus(1, gxx[1], distance[1], areacode[1],
482 isvalid[1], 2, validate, &gp, &gv);
483 vout = 2;
484
485 } else {
486 // if D<0, no solution
487 // if D=0, just grazing the surfaces, return kInfinity
488
489 fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
490 isvalid[0], 0, validate, &gp, &gv);
491 vout = 0;
492 }
493 return vout;
494}

499
500G4int G4TwistTubsHypeSide::DistanceToSurface(const G4ThreeVector &gp,
501 G4ThreeVector gxx[],
502 G4double distance[],
503 G4int areacode[])
504{
505 // Find the approximate distance of a point of a hyperbolic surface.
506 // The distance must be an underestimate.
507 // It will also be nice (although not necessary) that the estimate is
508 // always finite no matter how close the point is.
509 //
510 // We arranged G4Hype::ApproxDistOutside and G4Hype::ApproxDistInside
511 // for this function. See these discriptions.
512
513 const G4double halftol
514 = 0.5 * G4GeometryTolerance::GetInstance()->GetRadialTolerance();
515
516 fCurStat.ResetfDone(kDontValidate, &gp);
517
518 if (fCurStat.IsDone()) {
519 for (G4int i=0; i<fCurStat.GetNXX(); i++) {
520 gxx[i] = fCurStat.GetXX(i);
521 distance[i] = fCurStat.GetDistance(i);
522 areacode[i] = fCurStat.GetAreacode(i);
523 }
524 return fCurStat.GetNXX();
525 } else {
526 // initialize
527 for (G4int i=0; i<2; i++) {
528 distance[i] = kInfinity;
529 areacode[i] = sOutside;
530 gxx[i].set(kInfinity, kInfinity, kInfinity);
531 }
532 }
533
534
535 G4ThreeVector p = ComputeLocalPoint(gp);
536 G4ThreeVector xx;
537
538 //
539 // special case!
540 // If p is on surface, return distance = 0 immediatery .
541 //
542 G4ThreeVector lastgxx[2];
543 for (G4int i=0; i<2; i++) {
544 lastgxx[i] = fCurStatWithV.GetXX(i);
545 }
546
547 if ((gp - lastgxx[0]).mag() < halftol || (gp - lastgxx[1]).mag() < halftol) {
548 // last winner, or last poststep point is on the surface.
549 xx = p;
550 gxx[0] = gp;
551 distance[0] = 0;
552
553 G4bool isvalid = true;
554 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
555 isvalid, 1, kDontValidate, &gp);
556
557 return 1;
558
559 }
560 //
561 // special case end
562 //
563
564 G4double prho = p.getRho();
565 G4double pz = std::fabs(p.z()); // use symmetry
566 G4double r1 = std::sqrt(fR02 + pz * pz * fTan2Stereo);
567
568 G4ThreeVector pabsz(p.x(), p.y(), pz);
569
570 if (prho > r1 + halftol) { // p is outside of Hyperbolic surface
571
572 // First point xx1
573 G4double t = r1 / prho;
574 G4ThreeVector xx1(t * pabsz.x(), t * pabsz.y() , pz);
575
576 // Second point xx2
577 G4double z2 = (prho * fTanStereo + pz) / (1 + fTan2Stereo);
578 G4double r2 = std::sqrt(fR02 + z2 * z2 * fTan2Stereo);
579 t = r2 / prho;
580 G4ThreeVector xx2(t * pabsz.x(), t * pabsz.y() , z2);
581
582 G4double len = (xx2 - xx1).mag();
583 if (len < DBL_MIN) {
584 // xx2 = xx1?? I guess we
585 // must have really bracketed the normal
586 distance[0] = (pabsz - xx1).mag();
587 xx = xx1;
588 } else {
589 distance[0] = DistanceToLine(pabsz, xx1, (xx2 - xx1) , xx);
590 }
591
592 } else if (prho < r1 - halftol) { // p is inside of Hyperbolic surface.
593
594 // First point xx1
595 G4double t;
596 G4ThreeVector xx1;
597 if (prho < DBL_MIN) {
598 xx1.set(r1, 0. , pz);
599 } else {
600 t = r1 / prho;
601 xx1.set(t * pabsz.x(), t * pabsz.y() , pz);
602 }
603
604 // dr, dz is tangential vector of Hyparbolic surface at xx1
605 // dr = r, dz = z*tan2stereo
606 G4double dr = pz * fTan2Stereo;
607 G4double dz = r1;
608 G4double tanbeta = dr / dz;
609 G4double pztanbeta = pz * tanbeta;
610
611 // Second point xx2
612 // xx2 is intersection between x-axis and tangential vector
613 G4double r2 = r1 - pztanbeta;
614 G4ThreeVector xx2;
615 if (prho < DBL_MIN) {
616 xx2.set(r2, 0. , 0.);
617 } else {
618 t = r2 / prho;
619 xx2.set(t * pabsz.x(), t * pabsz.y() , 0.);
620 }
621
622 G4ThreeVector d = xx2 - xx1;
623 distance[0] = DistanceToLine(pabsz, xx1, d, xx);
624
625 } else { // p is on Hyperbolic surface.
626
627 distance[0] = 0;
628 xx.set(p.x(), p.y(), pz);
629
630 }
631
632 if (p.z() < 0) {
633 G4ThreeVector tmpxx(xx.x(), xx.y(), -xx.z());
634 xx = tmpxx;
635 }
636
637 gxx[0] = ComputeGlobalPoint(xx);
638 areacode[0] = sInside;
639 G4bool isvalid = true;
640 fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
641 isvalid, 1, kDontValidate, &gp);
642 return 1;
643}

645//=====================================================================
646//* GetAreaCode -------------------------------------------------------
647
648G4int G4TwistTubsHypeSide::GetAreaCode(const G4ThreeVector &xx,
649 G4bool withTol)
650{
651 const G4double ctol = 0.5 * kCarTolerance;
652 G4int areacode = sInside;
653
654 if ((fAxis[0] == kPhi && fAxis[1] == kZAxis)) {
655 //G4int phiaxis = 0;
656 G4int zaxis = 1;
657
658 if (withTol) {
659
660 G4bool isoutside = false;
661 G4int phiareacode = GetAreaCodeInPhi(xx);
662 G4bool isoutsideinphi = IsOutside(phiareacode);
663
664 // test boundary of phiaxis
665
666 if ((phiareacode & sAxisMin) == sAxisMin) {
667
668 areacode |= (sAxis0 & (sAxisPhi | sAxisMin)) | sBoundary;
669 if (isoutsideinphi) isoutside = true;
670
671 } else if ((phiareacode & sAxisMax) == sAxisMax) {
672
673 areacode |= (sAxis0 & (sAxisPhi | sAxisMax)) | sBoundary;
674 if (isoutsideinphi) isoutside = true;
675
676 }
677
678 // test boundary of zaxis
679
680 if (xx.z() < fAxisMin[zaxis] + ctol) {
681
682 areacode |= (sAxis1 & (sAxisZ | sAxisMin));
683 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
684 else areacode |= sBoundary;
685
686 if (xx.z() <= fAxisMin[zaxis] - ctol) isoutside = true;
687
688 } else if (xx.z() > fAxisMax[zaxis] - ctol) {
689
690 areacode |= (sAxis1 & (sAxisZ | sAxisMax));
691 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
692 else areacode |= sBoundary;
693
694 if (xx.z() >= fAxisMax[zaxis] + ctol) isoutside = true;
695 }
696
697 // if isoutside = true, clear sInside bit.
698 // if not on boundary, add boundary information.
699
700 if (isoutside) {
701 G4int tmpareacode = areacode & (~sInside);
702 areacode = tmpareacode;
703 } else if ((areacode & sBoundary) != sBoundary) {
704 areacode |= (sAxis0 & sAxisPhi) | (sAxis1 & sAxisZ);
705 }
706
707 return areacode;
708
709 } else {
710
711 G4int phiareacode = GetAreaCodeInPhi(xx, false);
712
713 // test boundary of z-axis
714
715 if (xx.z() < fAxisMin[zaxis]) {
716
717 areacode |= (sAxis1 & (sAxisZ | sAxisMin)) | sBoundary;
718
719 } else if (xx.z() > fAxisMax[zaxis]) {
720
721 areacode |= (sAxis1 & (sAxisZ | sAxisMax)) | sBoundary;
722
723 }
724
725 // boundary of phi-axis
726
727 if (phiareacode == sAxisMin) {
728
729 areacode |= (sAxis0 & (sAxisPhi | sAxisMin));
730 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
731 else areacode |= sBoundary;
732
733 } else if (phiareacode == sAxisMax) {
734
735 areacode |= (sAxis0 & (sAxisPhi | sAxisMax));
736 if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
737 else areacode |= sBoundary;
738
739 }
740
741 // if not on boundary, add boundary information.
742
743 if ((areacode & sBoundary) != sBoundary) {
744 areacode |= (sAxis0 & sAxisPhi) | (sAxis1 & sAxisZ);
745 }
746 return areacode;
747 }
748 } else {
749 std::ostringstream message;
750 message << "Feature NOT implemented !" << G4endl
751 << " fAxis[0] = " << fAxis[0] << G4endl
752 << " fAxis[1] = " << fAxis[1];
753 G4Exception("G4TwistTubsHypeSide::GetAreaCode()",
754 "GeomSolids0001", FatalException, message);
755 }
756 return areacode;
757}

759//=====================================================================
760//* GetAreaCodeInPhi --------------------------------------------------
761
762G4int G4TwistTubsHypeSide::GetAreaCodeInPhi(const G4ThreeVector &xx,
763 G4bool withTol)
764{
765
766 G4ThreeVector lowerlimit; // lower phi-boundary limit at z = xx.z()
767 G4ThreeVector upperlimit; // upper phi-boundary limit at z = xx.z()
768 lowerlimit = GetBoundaryAtPZ(sAxis0 & sAxisMin, xx);
769 upperlimit = GetBoundaryAtPZ(sAxis0 & sAxisMax, xx);
770
771 G4int areacode = sInside;
772 G4bool isoutside = false;
773
774 if (withTol) {
775
776 if (AmIOnLeftSide(xx, lowerlimit) >= 0) { // xx is on lowerlimit
777 areacode |= (sAxisMin | sBoundary);
778 if (AmIOnLeftSide(xx, lowerlimit) > 0) isoutside = true;
779
780 } else if (AmIOnLeftSide(xx, upperlimit) <= 0) { // xx is on upperlimit
781 areacode |= (sAxisMax | sBoundary);
782 if (AmIOnLeftSide(xx, upperlimit) < 0) isoutside = true;
783 }
784
785 // if isoutside = true, clear inside bit.
786
787 if (isoutside) {
788 G4int tmpareacode = areacode & (~sInside);
789 areacode = tmpareacode;
790 }
791
792
793 } else {
794
795 if (AmIOnLeftSide(xx, lowerlimit, false) >= 0) {
796 areacode |= (sAxisMin | sBoundary);
797 } else if (AmIOnLeftSide(xx, upperlimit, false) <= 0) {
798 areacode |= (sAxisMax | sBoundary);
799 }
800 }
801
802 return areacode;
803
804}

926//* GetFacets() -------------------------------------------------------
927
928void G4TwistTubsHypeSide::GetFacets( G4int k, G4int n, G4double xyz[][3],
929 G4int faces[][4], G4int iside )
930{
931
932 G4double z ; // the two parameters for the surface equation
933 G4double x,xmin,xmax ;
934
935 G4ThreeVector p ; // a point on the surface, given by (z,u)
936
937 G4int nnode ;
938 G4int nface ;
939
940 // calculate the (n-1)*(k-1) vertices
941
942 G4int i,j ;
943
944 for ( i = 0 ; i<n ; i++ ) {
945
946 z = fAxisMin[1] + i*(fAxisMax[1]-fAxisMin[1])/(n-1) ;
947
948 for ( j = 0 ; j<k ; j++ )
949 {
950 nnode = GetNode(i,j,k,n,iside) ;
951
952 xmin = GetBoundaryMin(z) ;
953 xmax = GetBoundaryMax(z) ;
954
955 if (fHandedness < 0) { // inner hyperbolic surface
956 x = xmin + j*(xmax-xmin)/(k-1) ;
957 } else { // outer hyperbolic surface
958 x = xmax - j*(xmax-xmin)/(k-1) ;
959 }
960
961 p = SurfacePoint(x,z,true) ; // surface point in global coord.system
962
963 xyz[nnode][0] = p.x() ;
964 xyz[nnode][1] = p.y() ;
965 xyz[nnode][2] = p.z() ;
966
967 if ( i<n-1 && j<k-1 ) { // clock wise filling
968
969 nface = GetFace(i,j,k,n,iside) ;
970
971 faces[nface][0] = GetEdgeVisibility(i,j,k,n,0,1) * ( GetNode(i ,j ,k,n,iside)+1) ;
972 faces[nface][1] = GetEdgeVisibility(i,j,k,n,1,1) * ( GetNode(i+1,j ,k,n,iside)+1) ;
973 faces[nface][2] = GetEdgeVisibility(i,j,k,n,2,1) * ( GetNode(i+1,j+1,k,n,iside)+1) ;
974 faces[nface][3] = GetEdgeVisibility(i,j,k,n,3,1) * ( GetNode(i ,j+1,k,n,iside)+1) ;
975
976 }
977 }
978 }
