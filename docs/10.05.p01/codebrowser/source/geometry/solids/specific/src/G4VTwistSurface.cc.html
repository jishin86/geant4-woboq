156
157G4int G4VTwistSurface::AmIOnLeftSide(const G4ThreeVector &me,
158 const G4ThreeVector &vec,
159 G4bool withtol)
160{
161 // AmIOnLeftSide returns phi-location of "me"
162 // (phi relation between me and vec projected on z=0 plane).
163 // If "me" is on -ve-phi-side of "vec", it returns 1.
164 // On the other hand, if "me" is on +ve-phi-side of "vec",
165 // it returns -1.
166 // (The return value represents z-coordinate of normal vector
167 // of me.cross(vec).)
168 // If me is on boundary of vec, return 0.
169
170 const G4double kAngTolerance
171 = G4GeometryTolerance::GetInstance()->GetAngularTolerance();
172
173 G4RotationMatrix unitrot;
174 const G4RotationMatrix rottol = unitrot.rotateZ(0.5*kAngTolerance);
175 const G4RotationMatrix invrottol = unitrot.rotateZ(-1.*kAngTolerance);
176
177 if (fAmIOnLeftSide.me == me
178 && fAmIOnLeftSide.vec == vec
179 && fAmIOnLeftSide.withTol == withtol) {
180 return fAmIOnLeftSide.amIOnLeftSide;
181 }
182
183 fAmIOnLeftSide.me = me;
184 fAmIOnLeftSide.vec = vec;
185 fAmIOnLeftSide.withTol = withtol;
186
187 G4ThreeVector met = (G4ThreeVector(me.x(), me.y(), 0.)).unit();
188 G4ThreeVector vect = (G4ThreeVector(vec.x(), vec.y(), 0.)).unit();
189
190 G4ThreeVector ivect = invrottol * vect;
191 G4ThreeVector rvect = rottol * vect;
192
193 G4double metcrossvect = met.x() * vect.y() - met.y() * vect.x();
194
195 if (withtol) {
196 if (met.x() * ivect.y() - met.y() * ivect.x() > 0 &&
197 metcrossvect >= 0) {
198 fAmIOnLeftSide.amIOnLeftSide = 1;
199 } else if (met.x() * rvect.y() - met.y() * rvect.x() < 0 &&
200 metcrossvect <= 0) {
201 fAmIOnLeftSide.amIOnLeftSide = -1;
202 } else {
203 fAmIOnLeftSide.amIOnLeftSide = 0;
204 }
205 } else {
206 if (metcrossvect > 0) {
207 fAmIOnLeftSide.amIOnLeftSide = 1;
208 } else if (metcrossvect < 0 ) {
209 fAmIOnLeftSide.amIOnLeftSide = -1;
210 } else {
211 fAmIOnLeftSide.amIOnLeftSide = 0;
212 }
213 }
214
215#ifdef G4TWISTDEBUG
216 G4cout << " === G4VTwistSurface::AmIOnLeftSide() =============="
217 << G4endl;
218 G4cout << " Name , returncode : " << fName << " "
219 << fAmIOnLeftSide.amIOnLeftSide << G4endl;
220 G4cout << " me, vec : " << std::setprecision(14) << me
221 << " " << vec << G4endl;
222 G4cout << " met, vect : " << met << " " << vect << G4endl;
223 G4cout << " ivec, rvec : " << ivect << " " << rvect << G4endl;
224 G4cout << " met x vect : " << metcrossvect << G4endl;
225 G4cout << " met x ivec : " << met.cross(ivect) << G4endl;
226 G4cout << " met x rvec : " << met.cross(rvect) << G4endl;
227 G4cout << " =============================================="
228 << G4endl;
229#endif
230
231 return fAmIOnLeftSide.amIOnLeftSide;
232}

235//* DistanceToBoundary ------------------------------------------------
236
237G4double G4VTwistSurface::DistanceToBoundary(G4int areacode,
238 G4ThreeVector &xx,
239 const G4ThreeVector &p)
240{
241 // DistanceToBoundary
242 //
243 // return distance to nearest boundary from arbitrary point p
244 // in local coodinate.
245 // Argument areacode must be one of them:
246 // sAxis0 & sAxisMin, sAxis0 & sAxisMax,
247 // sAxis1 & sAxisMin, sAxis1 & sAxisMax.
248 //
249
250 G4ThreeVector d; // direction vector of the boundary
251 G4ThreeVector x0; // reference point of the boundary
252 G4double dist = kInfinity;
253 G4int boundarytype;
254
255 if (IsAxis0(areacode) && IsAxis1(areacode)) {
256 std::ostringstream message;
257 message << "Point is in the corner area." << G4endl
258 << " Point is in the corner area. This function returns"
259 << G4endl
260 << " a direction vector of a boundary line." << G4endl
261 << " areacode = " << areacode;
262 G4Exception("G4VTwistSurface::DistanceToBoundary()", "GeomSolids0003",
263 FatalException, message);
264 } else if (IsAxis0(areacode) || IsAxis1(areacode)) {
265 GetBoundaryParameters(areacode, d, x0, boundarytype);
266 if (boundarytype == sAxisPhi) {
267 G4double t = x0.getRho() / p.getRho();
268 xx.set(t*p.x(), t*p.y(), x0.z());
269 dist = (xx - p).mag();
270 } else {
271 // linear boundary
272 // sAxisX, sAxisY, sAxisZ, sAxisRho
273 dist = DistanceToLine(p, x0, d, xx);
274 }
275 } else {
276 std::ostringstream message;
277 message << "Bad areacode of boundary." << G4endl
278 << " areacode = " << areacode;
279 G4Exception("G4VTwistSurface::DistanceToBoundary()", "GeomSolids0003",
280 FatalException, message);
281 }
282 return dist;
283}

285//=====================================================================
286//* DistanceToIn ------------------------------------------------------
287
288G4double G4VTwistSurface::DistanceToIn(const G4ThreeVector &gp,
289 const G4ThreeVector &gv,
290 G4ThreeVector &gxxbest)
291{
292#ifdef G4TWISTDEBUG
293 G4cout << " ~~~~~ G4VTwistSurface::DistanceToIn(p,v) - Start ~~~~~" << G4endl;
294 G4cout << " Name : " << fName << G4endl;
295 G4cout << " gp : " << gp << G4endl;
296 G4cout << " gv : " << gv << G4endl;
297 G4cout << "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" << G4endl;
298#endif
299
300 G4ThreeVector gxx[G4VSURFACENXX];
301 G4double distance[G4VSURFACENXX] ;
302 G4int areacode[G4VSURFACENXX] ;
303 G4bool isvalid[G4VSURFACENXX] ;
304
305 for (G4int i = 0 ; i<G4VSURFACENXX ; i++ ) {
306 distance[i] = kInfinity ;
307 areacode[i] = sOutside ;
308 isvalid[i] = false ;
309 }
310
311 G4double bestdistance = kInfinity;
312#ifdef G4TWISTDEBUG
313 G4int besti = -1;
314#endif
315 G4ThreeVector bestgxx(kInfinity, kInfinity, kInfinity);
316
317 G4int nxx = DistanceToSurface(gp, gv, gxx, distance, areacode,
318 isvalid, kValidateWithTol);
319
320 for (G4int i=0; i< nxx; i++) {
321
322 // skip this intersection if:
323 // - invalid intersection
324 // - particle goes outword the surface
325
326 if (!isvalid[i]) {
327 // xx[i] is sOutside or distance[i] < 0
328 continue;
329 }
330
331 G4ThreeVector normal = GetNormal(gxx[i], true);
332
333 if ((normal * gv) >= 0) {
334
335#ifdef G4TWISTDEBUG
336 G4cout << " G4VTwistSurface::DistanceToIn(p,v): "
337 << "particle goes outword the surface." << G4endl;
338#endif
339 continue;
340 }
341
342 //
343 // accept this intersection if the intersection is inside.
344 //
345
346 if (IsInside(areacode[i])) {
347 if (distance[i] < bestdistance) {
348 bestdistance = distance[i];
349 bestgxx = gxx[i];
350#ifdef G4TWISTDEBUG
351 besti = i;
352 G4cout << " G4VTwistSurface::DistanceToIn(p,v): "
353 << " areacode sInside name, distance = "
354 << fName << " "<< bestdistance << G4endl;
355#endif
356 }
357
358 //
359 // else, the intersection is on boundary or corner.
360 //
361
362 } else {
363
364 G4VTwistSurface *neighbours[2];
365 G4bool isaccepted[2] = {false, false};
366 G4int nneighbours = GetNeighbours(areacode[i], neighbours);
367
368 for (G4int j=0; j< nneighbours; j++) {
369 // if on corner, nneighbours = 2.
370 // if on boundary, nneighbours = 1.
371
372 G4ThreeVector tmpgxx[G4VSURFACENXX];
373 G4double tmpdist[G4VSURFACENXX] ;
374 G4int tmpareacode[G4VSURFACENXX] ;
375 G4bool tmpisvalid[G4VSURFACENXX] ;
376
377 for (G4int l = 0 ; l<G4VSURFACENXX ; l++ ) {
378 tmpdist[l] = kInfinity ;
379 tmpareacode[l] = sOutside ;
380 tmpisvalid[l] = false ;
381 }
382
383 G4int tmpnxx = neighbours[j]->DistanceToSurface(
384 gp, gv, tmpgxx, tmpdist,
385 tmpareacode, tmpisvalid,
386 kValidateWithTol);
387 G4ThreeVector neighbournormal;
388
389 for (G4int k=0; k< tmpnxx; k++) {
390
391 //
392 // if tmpxx[k] is valid && sInside, the final winner must
393 // be neighbour surface. return kInfinity.
394 // else , choose tmpxx on same boundary of xx, then check normal
395 //
396
397 if (IsInside(tmpareacode[k])) {
398
399#ifdef G4TWISTDEBUG
400 G4cout << " G4VTwistSurface:DistanceToIn(p,v): "
401 << " intersection "<< tmpgxx[k] << G4endl
402 << " is inside of neighbour surface of " << fName
403 << " . returning kInfinity." << G4endl;
404 G4cout << "~~~~~ G4VTwistSurface::DistanceToIn(p,v) - return ~~~~"
405 << G4endl;
406 G4cout << " No intersections " << G4endl;
407 G4cout << " Name : " << fName << G4endl;
408 G4cout << "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~"
409 << G4endl;
410#endif
411 if (tmpisvalid[k]) return kInfinity;
412 continue;
413
414 //
415 // if tmpxx[k] is valid && sInside, the final winner must
416 // be neighbour surface. return .
417 //
418
419 } else if (IsSameBoundary(this,areacode[i],
420 neighbours[j], tmpareacode[k])) {
421 // tmpxx[k] is same boundary (or corner) of xx.
422
423 neighbournormal = neighbours[j]->GetNormal(tmpgxx[k], true);
424 if (neighbournormal * gv < 0) isaccepted[j] = true;
425 }
426 }
427
428 // if nneighbours = 1, chabge isaccepted[1] before
429 // exiting neighboursurface loop.
430
431 if (nneighbours == 1) isaccepted[1] = true;
432
433 } // neighboursurface loop end
434
435 // now, we can accept xx intersection
436
437 if (isaccepted[0] == true && isaccepted[1] == true) {
438 if (distance[i] < bestdistance) {
439 bestdistance = distance[i];
440 gxxbest = gxx[i];
441#ifdef G4TWISTDEBUG
442 besti = i;
443 G4cout << " G4VTwistSurface::DistanceToIn(p,v): "
444 << " areacode sBoundary & sBoundary distance = "
445 << fName << " " << distance[i] << G4endl;
446#endif
447 }
448 }
449
450 } // else end
451 } // intersection loop end
452
453 gxxbest = bestgxx;
454
455#ifdef G4TWISTDEBUG
456 if (besti < 0) {
457 G4cout << "~~~~~ G4VTwistSurface::DistanceToIn(p,v) - return ~~~~" << G4endl;
458 G4cout << " No intersections " << G4endl;
459 G4cout << " Name : " << fName << G4endl;
460 G4cout << "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" << G4endl;
461 } else {
462 G4cout << "~~~~~ G4VTwistSurface::DistanceToIn(p,v) : return ~~~~" << G4endl;
463 G4cout << " Name, i : " << fName << " , " << besti << G4endl;
464 G4cout << " gxx[i] : " << gxxbest << G4endl;
465 G4cout << " bestdist : " << bestdistance << G4endl;
466 G4cout << "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" << G4endl;
467 }
468
469#endif
470
471 return bestdistance;
472}

493 G4int i;
494
495 for ( i = 0 ; i<G4VSURFACENXX ; i++ )
496 {
497 distance[i] = kInfinity ;

500 }
501
502 G4int nxx;

504

518 << fName << " " << normal

520#endif
521 } else {
522 // gxx[i] is accepted.
523 if (distance[i] < bestdistance) {
524 bestdistance = distance[i];
525 gxxbest = gxx[i];
526 }
527 }
528 }
529
530#ifdef G4TWISTDEBUG
531 if (besti < 0) {
532 G4cout << "~~~~~ G4VTwistSurface::DistanceToOut(p,v) - return ~~~" << G4endl;
533 G4cout << " No intersections " << G4endl;
534 G4cout << " Name : " << fName << G4endl;

536 G4cout << "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" << G4endl;
537 } else {
538 G4cout << "~~~~~ G4VTwistSurface::DistanceToOut(p,v) : return ~~~" << G4endl;
539 G4cout << " Name, i : " << fName << " , " << i << G4endl;
540 G4cout << " gxx[i] : " << gxxbest << G4endl;
541 G4cout << " bestdist : " << bestdistance << G4endl;
542 G4cout << "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" << G4endl;
543 }
544#endif
545
546 return bestdistance;
547}

588//* IsSameBoundary ----------------------------------------------------
589
590G4bool G4VTwistSurface::IsSameBoundary(G4VTwistSurface *surface1, G4int areacode1,
591 G4VTwistSurface *surface2, G4int areacode2 ) const
592{
593 //
594 // IsSameBoundary
595 //
596 // checking tool whether two boundaries on different surfaces are same or not.
597 //
598
599 G4bool testbitmode = true;
600 G4bool iscorner[2] = {IsCorner(areacode1, testbitmode),
601 IsCorner(areacode2, testbitmode)};
602
603 if (iscorner[0] && iscorner[1]) {
604 // on corner
605 G4ThreeVector corner1 =
606 surface1->ComputeGlobalPoint(surface1->GetCorner(areacode1));
607 G4ThreeVector corner2 =
608 surface2->ComputeGlobalPoint(surface2->GetCorner(areacode2));
609
610 if ((corner1 - corner2).mag() < kCarTolerance) {
611 return true;
612 } else {
613 return false;
614 }
615
616 } else if ((IsBoundary(areacode1, testbitmode) && (!iscorner[0])) &&
617 (IsBoundary(areacode2, testbitmode) && (!iscorner[1]))) {
618 // on boundary
619 G4ThreeVector d1, d2, ld1, ld2;
620 G4ThreeVector x01, x02, lx01, lx02;
621 G4int type1, type2;
622 surface1->GetBoundaryParameters(areacode1, ld1, lx01, type1);
623 surface2->GetBoundaryParameters(areacode2, ld2, lx02, type2);
624
625 x01 = surface1->ComputeGlobalPoint(lx01);
626 x02 = surface2->ComputeGlobalPoint(lx02);
627 d1 = surface1->ComputeGlobalDirection(ld1);
628 d2 = surface2->ComputeGlobalDirection(ld2);
629
630 if ((x01 - x02).mag() < kCarTolerance &&
631 (d1 - d2).mag() < kCarTolerance) {
632 return true;
633 } else {
634 return false;
635 }
636
637 } else {
638 return false;
639 }
640}

644
645void G4VTwistSurface::GetBoundaryParameters(const G4int &areacode,
646 G4ThreeVector &d,
647 G4ThreeVector &x0,
648 G4int &boundarytype) const
649{
650 // areacode must be one of them:

672//* GetBoundaryAtPZ ---------------------------------------------------
673
674G4ThreeVector G4VTwistSurface::GetBoundaryAtPZ(G4int areacode,
675 const G4ThreeVector &p) const
676{
677 // areacode must be one of them:
678 // sAxis0 & sAxisMin, sAxis0 & sAxisMax,
679 // sAxis1 & sAxisMin, sAxis1 & sAxisMax.
680
681 if (areacode & sAxis0 && areacode & sAxis1) {
682 std::ostringstream message;
683 message << "Point is in the corner area." << G4endl
684 << " This function returns "
685 << "a direction vector of a boundary line." << G4endl
686 << " areacode = " << areacode;
687 G4Exception("G4VTwistSurface::GetBoundaryAtPZ()", "GeomSolids0003",
688 FatalException, message);
689 }
690
691 G4ThreeVector d;
692 G4ThreeVector x0;
693 G4int boundarytype;
694 G4bool found = false;
695
696 for (G4int i=0; i<4; i++) {
697 if (fBoundaries[i].GetBoundaryParameters(areacode, d, x0,
698 boundarytype)){
699 found = true;
700 continue;
701 }
702 }
703
704 if (!found) {
705 std::ostringstream message;
706 message << "Not registered boundary." << G4endl
707 << " Boundary at areacode " << areacode << G4endl
708 << " is not registered.";
709 G4Exception("G4VTwistSurface::GetBoundaryAtPZ()", "GeomSolids0002",
710 FatalException, message);
711 }
712
713 if (((boundarytype & sAxisPhi) == sAxisPhi) ||
714 ((boundarytype & sAxisRho) == sAxisRho)) {

719 G4Exception("G4VTwistSurface::GetBoundaryAtPZ()", "GeomSolids0002",

870//* GetFace -----------------------------------------------------------
871
872G4int G4VTwistSurface::GetFace( G4int i, G4int j, G4int k,
873 G4int n, G4int iside )
874{
875 // this is the face mapping function
876 // (i,j) -> face number
877
878 if ( iside == 0 ) {
879 return i * ( k - 1 ) + j ;
880 }
881
882 else if ( iside == 1 ) {
883 return (k-1)*(k-1) + i*(k-1) + j ;
884 }
885
886 else if ( iside == 2 ) {
887 return 2*(k-1)*(k-1) + i*(k-1) + j ;
888 }
889
890 else if ( iside == 3 ) {
891 return 2*(k-1)*(k-1) + (n-1)*(k-1) + i*(k-1) + j ;
892 }
893
894 else if ( iside == 4 ) {
895 return 2*(k-1)*(k-1) + 2*(n-1)*(k-1) + i*(k-1) + j ;
896 }
897
898 else if ( iside == 5 ) {
899 return 2*(k-1)*(k-1) + 3*(n-1)*(k-1) + i*(k-1) + j ;
900 }
901
902 else {
903
904 std::ostringstream message;
905 message << "Not correct side number: "
906 << GetName() << G4endl
907 << "iside is " << iside << " but should be "
908 << "0,1,2,3,4 or 5" << ".";
909 G4Exception("G4TwistSurface::G4GetFace()", "GeomSolids0002",
910 FatalException, message);
911
912
913 }
914
915 return -1 ; // wrong face
916}

918//=====================================================================
919//* GetNode -----------------------------------------------------------
920
921G4int G4VTwistSurface::GetNode( G4int i, G4int j, G4int k,
922 G4int n, G4int iside )
923{
924 // this is the node mapping function
925 // (i,j) -> node number
926 // Depends on the side iside and the used meshing of the surface
927
928 if ( iside == 0 ) {
929 // lower endcap is kxk squared.
930 // n = k
931 return i * k + j ;
932 }
933
934 if ( iside == 1 ) {
935 // upper endcap is kxk squared. Shift by k*k
936 // n = k
937 return k*k + i*k + j ;
938 }
939
940 else if ( iside == 2 ) {
941 // front side.
942 if ( i == 0 ) { return j ; }
943 else if ( i == n-1 ) { return k*k + j ; }
944 else { return 2*k*k + 4*(i-1)*(k-1) + j ; }
945 }
946
947 else if ( iside == 3 ) {
948 // right side
949 if ( i == 0 ) { return (j+1)*k - 1 ; }
950 else if ( i == n-1 ) { return k*k + (j+1)*k - 1 ; }
951 else { return 2*k*k + 4*(i-1)*(k-1) + (k-1) + j ; }
952 }
953 else if ( iside == 4 ) {
954 // back side.
955 if ( i == 0 ) { return k*k - 1 - j ; } // reversed order
956 else if ( i == n-1 ) { return 2*k*k - 1 - j ; } // reversed order
957 else { return 2*k*k + 4*(i-1)*(k-1) + 2*(k-1) + j ; // normal order
958 }
959 }
960 else if ( iside == 5 ) {
961 // left side
962 if ( i == 0 ) { return k*k - (j+1)*k ; } // reversed order
963 else if ( i == n-1) { return 2*k*k - (j+1)*k ; } // reverded order
964 else {
965 if ( j == k-1 ) { return 2*k*k + 4*(i-1)*(k-1) ; } // special case
966 else { return 2*k*k + 4*(i-1)*(k-1) + 3*(k-1) + j ; } // normal order
967 }
968 }
969
970 else {
971
972 std::ostringstream message;
973 message << "Not correct side number: "
974 << GetName() << G4endl
975 << "iside is " << iside << " but should be "
976 << "0,1,2,3,4 or 5" << ".";
977 G4Exception("G4TwistSurface::G4GetNode()", "GeomSolids0002",
978 FatalException, message);
979 }
980 return -1 ; // wrong node
981}

983//=====================================================================
984//* GetEdgeVisiblility ------------------------------------------------
985
986G4int G4VTwistSurface::GetEdgeVisibility( G4int i, G4int j, G4int k, G4int n, G4int number, G4int orientation)
987{
988
989 // clockwise filling -> positive orientation
990 // counter clockwise filling -> negative orientation
991
992 //
993 // d C c
994 // +------+
995 // | |
996 // | |
997 // | |
998 // D | |B
999 // | |
1000 // | |
1001 // | |
1002 // +------+
1003 // a A b
1004 //
1005 // a = +--+ A = ---+
1006 // b = --++ B = --+-
1007 // c = -++- C = -+--
1008 // d = ++-- D = +---
1009
1010
1011 // check first invisible faces
1012
1013 if ( ( i>0 && i<n-2 ) && ( j>0 && j<k-2 ) ) {
1014 return -1 ; // always invisible, signs: ----
1015 }
1016
1017 // change first the vertex number (depends on the orientation)
1018 // 0,1,2,3 -> 3,2,1,0
1019 if ( orientation < 0 ) { number = ( 3 - number ) ; }
1020
1021 // check true edges
1022 if ( ( j>=1 && j<=k-3 ) ) {
1023
1024 if ( i == 0 ) { // signs (A): ---+
1025 return ( number == 3 ) ? 1 : -1 ;
1026 }
1027
1028 else if ( i == n-2 ) { // signs (C): -+--
1029 return ( number == 1 ) ? 1 : -1 ;
1030 }
1031
1032 else {
1033 std::ostringstream message;
1034 message << "Not correct face number: " << GetName() << " !";
1035 G4Exception("G4TwistSurface::G4GetEdgeVisibility()",
1036 "GeomSolids0003", FatalException, message);
1037 }
1038 }
1039
1040 if ( ( i>=1 && i<=n-3 ) ) {
1041
1042 if ( j == 0 ) { // signs (D): +---
1043 return ( number == 0 ) ? 1 : -1 ;
1044 }
1045
1046 else if ( j == k-2 ) { // signs (B): --+-
1047 return ( number == 2 ) ? 1 : -1 ;
1048 }
1049
1050 else {
1051 std::ostringstream message;
1052 message << "Not correct face number: " << GetName() << " !";
1053 G4Exception("G4TwistSurface::G4GetEdgeVisibility()",
1054 "GeomSolids0003", FatalException, message);
1055 }
1056 }
1057
1058 // now the corners
1059 if ( i == 0 && j == 0 ) { // signs (a) : +--+
1060 return ( number == 0 || number == 3 ) ? 1 : -1 ;
1061 }
1062 else if ( i == 0 && j == k-2 ) { // signs (b) : --++
1063 return ( number == 2 || number == 3 ) ? 1 : -1 ;
1064 }
1065 else if ( i == n-2 && j == k-2 ) { // signs (c) : -++-
1066 return ( number == 1 || number == 2 ) ? 1 : -1 ;
1067 }
1068 else if ( i == n-2 && j == 0 ) { // signs (d) : ++--
1069 return ( number == 0 || number == 1 ) ? 1 : -1 ;
1070 }
1071 else {
1072 std::ostringstream message;
1073 message << "Not correct face number: " << GetName() << " !";
1074 G4Exception("G4TwistSurface::G4GetEdgeVisibility()",
1075 "GeomSolids0003", FatalException, message);
1076 }
1077
1078 std::ostringstream message;
1079 message << "Not correct face number: " << GetName() << " !";
1080 G4Exception("G4TwistSurface::G4GetEdgeVisibility()", "GeomSolids0003",
1081 FatalException, message);
1082
1083 return 0 ;
1084}
