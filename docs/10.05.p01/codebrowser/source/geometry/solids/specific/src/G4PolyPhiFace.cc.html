50//
51// Constructor
52//
53// Points r,z should be supplied in clockwise order in r,z. For example:
54//
55// [1]---------[2] ^ R
56// | | |
57// | | +--> z
58// [0]---------[3]
59//
60G4PolyPhiFace::G4PolyPhiFace( const G4ReduciblePolygon *rz,
61 G4double phi,
62 G4double deltaPhi,
63 G4double phiOther )
64 : fSurfaceArea(0.), triangles(0)
65{
66 kCarTolerance = G4GeometryTolerance::GetInstance()->GetSurfaceTolerance();
67
68 numEdges = rz->NumVertices();
69
70 rMin = rz->Amin();
71 rMax = rz->Amax();
72 zMin = rz->Bmin();
73 zMax = rz->Bmax();
74
75 //
76 // Is this the "starting" phi edge of the two?
77 //
78 G4bool start = (phiOther > phi);
79
80 //
81 // Build radial vector
82 //
83 radial = G4ThreeVector( std::cos(phi), std::sin(phi), 0.0 );
84
85 //
86 // Build normal
87 //
88 G4double zSign = start ? 1 : -1;
89 normal = G4ThreeVector( zSign*radial.y(), -zSign*radial.x(), 0 );
90
91 //
92 // Is allBehind?
93 //
94 allBehind = (zSign*(std::cos(phiOther)*radial.y() - std::sin(phiOther)*radial.x()) < 0);
95
96 //
97 // Adjacent edges
98 //
99 G4double midPhi = phi + (start ? +0.5 : -0.5)*deltaPhi;
100 G4double cosMid = std::cos(midPhi),
101 sinMid = std::sin(midPhi);
102
103 //
104 // Allocate corners
105 //
106 corners = new G4PolyPhiFaceVertex[numEdges];
107 //
108 // Fill them
109 //
110 G4ReduciblePolygonIterator iterRZ(rz);
111
112 G4PolyPhiFaceVertex *corn = corners;
113 G4PolyPhiFaceVertex *helper=corners;
114
115 iterRZ.Begin();
116 do // Loop checking, 13.08.2015, G.Cosmo
117 {
118 corn->r = iterRZ.GetA();
119 corn->z = iterRZ.GetB();
120 corn->x = corn->r*radial.x();
121 corn->y = corn->r*radial.y();
122
123 // Add pointer on prev corner
124 //
125 if( corn == corners )
126 { corn->prev = corners+numEdges-1;}
127 else
128 { corn->prev = helper; }
129
130 // Add pointer on next corner
131 //
132 if( corn < corners+numEdges-1 )
133 { corn->next = corn+1;}
134 else
135 { corn->next = corners; }
136
137 helper = corn;
138 } while( ++corn, iterRZ.Next() );
139
140 //
141 // Allocate edges
142 //
143 edges = new G4PolyPhiFaceEdge[numEdges];
144
145 //
146 // Fill them
147 //
148 G4double rFact = std::cos(0.5*deltaPhi);
149 G4double rFactNormalize = 1.0/std::sqrt(1.0+rFact*rFact);
150
151 G4PolyPhiFaceVertex *prev = corners+numEdges-1,
152 *here = corners;
153 G4PolyPhiFaceEdge *edge = edges;
154 do // Loop checking, 13.08.2015, G.Cosmo
155 {
156 G4ThreeVector sideNorm;
157
158 edge->v0 = prev;
159 edge->v1 = here;
160
161 G4double dr = here->r - prev->r,
162 dz = here->z - prev->z;
163
164 edge->length = std::sqrt( dr*dr + dz*dz );
165
166 edge->tr = dr/edge->length;
167 edge->tz = dz/edge->length;
168
169 if ((here->r < DBL_MIN) && (prev->r < DBL_MIN))
170 {
171 //
172 // Sigh! Always exceptions!
173 // This edge runs at r==0, so its adjoing surface is not a
174 // PolyconeSide or PolyhedraSide, but the opposite PolyPhiFace.
175 //
176 G4double zSignOther = start ? -1 : 1;
177 sideNorm = G4ThreeVector( zSignOther*std::sin(phiOther),
178 -zSignOther*std::cos(phiOther), 0 );
179 }
180 else
181 {
182 sideNorm = G4ThreeVector( edge->tz*cosMid,
183 edge->tz*sinMid,
184 -edge->tr*rFact );
185 sideNorm *= rFactNormalize;
186 }
187 sideNorm += normal;
188
189 edge->norm3D = sideNorm.unit();
190 } while( edge++, prev=here, ++here < corners+numEdges );
191
192 //
193 // Go back and fill in corner "normals"
194 //
195 G4PolyPhiFaceEdge *prevEdge = edges+numEdges-1;
196 edge = edges;
197 do // Loop checking, 13.08.2015, G.Cosmo
198 {
199 //
200 // Calculate vertex 2D normals (on the phi surface)
201 //
202 G4double rPart = prevEdge->tr + edge->tr;
203 G4double zPart = prevEdge->tz + edge->tz;
204 G4double norm = std::sqrt( rPart*rPart + zPart*zPart );
205 G4double rNorm = +zPart/norm;
206 G4double zNorm = -rPart/norm;
207
208 edge->v0->rNorm = rNorm;
209 edge->v0->zNorm = zNorm;
210
211 //
212 // Calculate the 3D normals.
213 //
214 // Find the vector perpendicular to the z axis
215 // that defines the plane that contains the vertex normal
216 //
217 G4ThreeVector xyVector;
218
219 if (edge->v0->r < DBL_MIN)
220 {
221 //
222 // This is a vertex at r==0, which is a special
223 // case. The normal we will construct lays in the
224 // plane at the center of the phi opening.
225 //
226 // We also know that rNorm < 0
227 //
228 G4double zSignOther = start ? -1 : 1;
229 G4ThreeVector normalOther( zSignOther*std::sin(phiOther),
230 -zSignOther*std::cos(phiOther), 0 );
231
232 xyVector = - normal - normalOther;
233 }
234 else
235 {
236 //
237 // This is a vertex at r > 0. The plane
238 // is the average of the normal and the
239 // normal of the adjacent phi face
240 //
241 xyVector = G4ThreeVector( cosMid, sinMid, 0 );
242 if (rNorm < 0)
243 xyVector -= normal;
244 else
245 xyVector += normal;
246 }
247
248 //
249 // Combine it with the r/z direction from the face
250 //
251 edge->v0->norm3D = rNorm*xyVector.unit() + G4ThreeVector( 0, 0, zNorm );
252 } while( prevEdge=edge, ++edge < edges+numEdges );
253
254 //
255 // Build point on surface
256 //
257 G4double rAve = 0.5*(rMax-rMin),
258 zAve = 0.5*(zMax-zMin);
259 surface = G4ThreeVector( rAve*radial.x(), rAve*radial.y(), zAve );
260}

263//
264// Diagnose
265//
266// Throw an exception if something is found inconsistent with
267// the solid.
268//
269// For debugging purposes only
270//
271void G4PolyPhiFace::Diagnose( G4VSolid *owner )
272{
273 G4PolyPhiFaceVertex *corner = corners;
274 do // Loop checking, 13.08.2015, G.Cosmo
275 {
276 G4ThreeVector test(corner->x, corner->y, corner->z);
277 test -= 1E-6*corner->norm3D;
278
279 if (owner->Inside(test) != kInside)
280 G4Exception( "G4PolyPhiFace::Diagnose()", "GeomSolids0002",
281 FatalException, "Bad vertex normal found." );
282 } while( ++corner < corners+numEdges );
283}

333//
334// CopyStuff (protected)
335//
336void G4PolyPhiFace::CopyStuff( const G4PolyPhiFace &source )
337{
338 //
339 // The simple stuff
340 //
341 numEdges = source.numEdges;
342 normal = source.normal;
343 radial = source.radial;
344 surface = source.surface;
345 rMin = source.rMin;
346 rMax = source.rMax;
347 zMin = source.zMin;
348 zMax = source.zMax;
349 allBehind = source.allBehind;
350 triangles = 0;
351
352 kCarTolerance = source.kCarTolerance;
353 fSurfaceArea = source.fSurfaceArea;
354
355 //
356 // Corner dynamic array
357 //
358 corners = new G4PolyPhiFaceVertex[numEdges];
359 G4PolyPhiFaceVertex *corn = corners,
360 *sourceCorn = source.corners;
361 do // Loop checking, 13.08.2015, G.Cosmo
362 {
363 *corn = *sourceCorn;
364 } while( ++sourceCorn, ++corn < corners+numEdges );
365
366 //
367 // Edge dynamic array
368 //
369 edges = new G4PolyPhiFaceEdge[numEdges];
370
371 G4PolyPhiFaceVertex *prev = corners+numEdges-1,
372 *here = corners;
373 G4PolyPhiFaceEdge *edge = edges,
374 *sourceEdge = source.edges;
375 do // Loop checking, 13.08.2015, G.Cosmo
376 {
377 *edge = *sourceEdge;
378 edge->v0 = prev;
379 edge->v1 = here;
380 } while( ++sourceEdge, ++edge, prev=here, ++here < corners+numEdges );
381}

384//
385// Intersect
386//
387G4bool G4PolyPhiFace::Intersect( const G4ThreeVector &p,
388 const G4ThreeVector &v,
389 G4bool outgoing,
390 G4double surfTolerance,
391 G4double &distance,
392 G4double &distFromSurface,
393 G4ThreeVector &aNormal,
394 G4bool &isAllBehind )
395{
396 G4double normSign = outgoing ? +1 : -1;
397
398 //
399 // These don't change
400 //
401 isAllBehind = allBehind;
402 aNormal = normal;
403
404 //
405 // Correct normal? Here we have straight sides, and can safely ignore
406 // intersections where the dot product with the normal is zero.
407 //
408 G4double dotProd = normSign*normal.dot(v);
409
410 if (dotProd <= 0) return false;
411
412 //
413 // Calculate distance to surface. If the side is too far
414 // behind the point, we must reject it.
415 //
416 G4ThreeVector ps = p - surface;
417 distFromSurface = -normSign*ps.dot(normal);
418
419 if (distFromSurface < -surfTolerance) return false;
420
421 //
422 // Calculate precise distance to intersection with the side
423 // (along the trajectory, not normal to the surface)
424 //
425 distance = distFromSurface/dotProd;
426
427 //
428 // Calculate intersection point in r,z
429 //
430 G4ThreeVector ip = p + distance*v;
431
432 G4double r = radial.dot(ip);
433
434 //
435 // And is it inside the r/z extent?
436 //
437 return InsideEdgesExact( r, ip.z(), normSign, p, v );
438}

441//
442// Distance
443//
444G4double G4PolyPhiFace::Distance( const G4ThreeVector &p, G4bool outgoing )
445{
446 G4double normSign = outgoing ? +1 : -1;
447 //
448 // Correct normal?
449 //
450 G4ThreeVector ps = p - surface;
451 G4double distPhi = -normSign*normal.dot(ps);
452
453 if (distPhi < -0.5*kCarTolerance)
454 return kInfinity;
455 else if (distPhi < 0)
456 distPhi = 0.0;
457
458 //
459 // Calculate projected point in r,z
460 //
461 G4double r = radial.dot(p);
462
463 //
464 // Are we inside the face?
465 //
466 G4double distRZ2;
467
468 if (InsideEdges( r, p.z(), &distRZ2, 0 ))
469 {
470 //
471 // Yup, answer is just distPhi
472 //
473 return distPhi;
474 }
475 else
476 {
477 //
478 // Nope. Penalize by distance out
479 //
480 return std::sqrt( distPhi*distPhi + distRZ2 );
481 }
482}

485//
486// Inside
487//
488EInside G4PolyPhiFace::Inside( const G4ThreeVector &p,
489 G4double tolerance,
490 G4double *bestDistance )
491{
492 //
493 // Get distance along phi, which if negative means the point
494 // is nominally inside the shape.
495 //
496 G4ThreeVector ps = p - surface;
497 G4double distPhi = normal.dot(ps);
498
499 //
500 // Calculate projected point in r,z
501 //
502 G4double r = radial.dot(p);
503
504 //
505 // Are we inside the face?
506 //
507 G4double distRZ2;
508 G4PolyPhiFaceVertex *base3Dnorm=0;
509 G4ThreeVector *head3Dnorm=0;
510
511 if (InsideEdges( r, p.z(), &distRZ2, &base3Dnorm, &head3Dnorm ))
512 {
513 //
514 // Looks like we're inside. Distance is distance in phi.
515 //
516 *bestDistance = std::fabs(distPhi);
517
518 //
519 // Use distPhi to decide fate
520 //
521 if (distPhi < -tolerance) return kInside;
522 if (distPhi < tolerance) return kSurface;
523 return kOutside;
524 }
525 else
526 {
527 //
528 // We're outside the extent of the face,
529 // so the distance is penalized by distance from edges in RZ
530 //
531 *bestDistance = std::sqrt( distPhi*distPhi + distRZ2 );
532
533 //
534 // Use edge normal to decide fate
535 //
536 G4ThreeVector cc( base3Dnorm->r*radial.x(),
537 base3Dnorm->r*radial.y(),
538 base3Dnorm->z );
539 cc = p - cc;
540 G4double normDist = head3Dnorm->dot(cc);
541 if ( distRZ2 > tolerance*tolerance )
542 {
543 //
544 // We're far enough away that kSurface is not possible
545 //
546 return normDist < 0 ? kInside : kOutside;
547 }
548
549 if (normDist < -tolerance) return kInside;
550 if (normDist < tolerance) return kSurface;
551 return kOutside;
552 }
553}

600//
601// Extent
602//
603// This actually isn't needed by polycone or polyhedra...
604//
605G4double G4PolyPhiFace::Extent( const G4ThreeVector axis )
606{
607 G4double max = -kInfinity;
608
609 G4PolyPhiFaceVertex *corner = corners;
610 do // Loop checking, 13.08.2015, G.Cosmo
611 {
612 G4double here = axis.x()*corner->r*radial.x()
613 + axis.y()*corner->r*radial.y()
614 + axis.z()*corner->z;
615 if (here > max) max = here;
616 } while( ++corner < corners + numEdges );
617
618 return max;
619}

622//
623// CalculateExtent
624//
625// See notes in G4VCSGface
626//
627void G4PolyPhiFace::CalculateExtent( const EAxis axis,
628 const G4VoxelLimits &voxelLimit,
629 const G4AffineTransform &transform,
630 G4SolidExtentList &extentList )
631{
632 //
633 // Construct a (sometimes big) clippable polygon,
634 //
635 // Perform the necessary transformations while doing so
636 //
637 G4ClippablePolygon polygon;
638
639 G4PolyPhiFaceVertex *corner = corners;
640 do // Loop checking, 13.08.2015, G.Cosmo
641 {
642 G4ThreeVector point( 0, 0, corner->z );
643 point += radial*corner->r;
644
645 polygon.AddVertexInOrder( transform.TransformPoint( point ) );
646 } while( ++corner < corners + numEdges );
647
648 //
649 // Clip away
650 //
651 if (polygon.PartialClip( voxelLimit, axis ))
652 {
653 //
654 // Add it to the list
655 //
656 polygon.SetNormal( transform.TransformAxis(normal) );
657 extentList.AddSurface( polygon );
658 }
659}

666//
667// InsideEdgesExact
668//
669// Decide if the point in r,z is inside the edges of our face,
670// **but** do so consistently with other faces.
671//
672// This routine has functionality similar to InsideEdges, but uses
673// an algorithm to decide if a trajectory falls inside or outside the
674// face that uses only the trajectory p,v values and the three dimensional
675// points representing the edges of the polygon. The objective is to plug up
676// any leaks between touching G4PolyPhiFaces (at r==0) and any other face
677// that uses the same convention.
678//
679// See: "Computational Geometry in C (Second Edition)"
680// http://cs.smith.edu/~orourke/
681//
682G4bool G4PolyPhiFace::InsideEdgesExact( G4double r, G4double z,
683 G4double normSign,
684 const G4ThreeVector &p,
685 const G4ThreeVector &v )
686{
687 //
688 // Quick check of extent
689 //
690 if ( (r < rMin-kCarTolerance)
691 || (r > rMax+kCarTolerance) ) return false;
692
693 if ( (z < zMin-kCarTolerance)
694 || (z > zMax+kCarTolerance) ) return false;
695
696 //
697 // Exact check: loop over all vertices
698 //
699 G4double qx = p.x() + v.x(),
700 qy = p.y() + v.y(),
701 qz = p.z() + v.z();
702
703 G4int answer = 0;
704 G4PolyPhiFaceVertex *corn = corners,
705 *prev = corners+numEdges-1;
706
707 G4double cornZ, prevZ;
708
709 prevZ = ExactZOrder( z, qx, qy, qz, v, normSign, prev );
710 do // Loop checking, 13.08.2015, G.Cosmo
711 {
712 //
713 // Get z order of this vertex, and compare to previous vertex
714 //
715 cornZ = ExactZOrder( z, qx, qy, qz, v, normSign, corn );
716
717 if (cornZ < 0)
718 {
719 if (prevZ < 0) continue;
720 }
721 else if (cornZ > 0)
722 {
723 if (prevZ > 0) continue;
724 }
725 else
726 {
727 //
728 // By chance, we overlap exactly (within precision) with
729 // the current vertex. Continue if the same happened previously
730 // (e.g. the previous vertex had the same z value)
731 //
732 if (prevZ == 0) continue;
733
734 //
735 // Otherwise, to decide what to do, we need to know what is
736 // coming up next. Specifically, we need to find the next vertex
737 // with a non-zero z order.
738 //
739 // One might worry about infinite loops, but the above conditional
740 // should prevent it
741 //
742 G4PolyPhiFaceVertex *next = corn;
743 G4double nextZ;
744 do // Loop checking, 13.08.2015, G.Cosmo
745 {
746 next++;
747 if (next == corners+numEdges) next = corners;
748
749 nextZ = ExactZOrder( z, qx, qy, qz, v, normSign, next );
750 } while( nextZ == 0 );
751
752 //
753 // If we won't be changing direction, go to the next vertex
754 //
755 if (nextZ*prevZ < 0) continue;
756 }
757
758
759 //
760 // We overlap in z with the side of the face that stretches from
761 // vertex "prev" to "corn". On which side (left or right) do
762 // we lay with respect to this segment?
763 //
764 G4ThreeVector qa( qx - prev->x, qy - prev->y, qz - prev->z ),
765 qb( qx - corn->x, qy - corn->y, qz - corn->z );
766
767 G4double aboveOrBelow = normSign*qa.cross(qb).dot(v);
768
769 if (aboveOrBelow > 0)
770 answer++;
771 else if (aboveOrBelow < 0)
772 answer--;
773 else
774 {
775 //
776 // A precisely zero answer here means we exactly
777 // intersect (within roundoff) the edge of the face.
778 // Return true in this case.
779 //
780 return true;
781 }
782 } while( prevZ = cornZ, prev=corn, ++corn < corners+numEdges );
783
784 return answer!=0;
785}

788//
789// InsideEdges (don't care aboud distance)
790//
791// Decide if the point in r,z is inside the edges of our face
792//
793// This routine can be made a zillion times quicker by implementing
794// better code, for example:
795//
796// int pnpoly(int npol, float *xp, float *yp, float x, float y)
797// {
798// int i, j, c = 0;
799// for (i = 0, j = npol-1; i < npol; j = i++) {
800// if ((((yp[i]<=y) && (y<yp[j])) ||
801// ((yp[j]<=y) && (y<yp[i]))) &&
802// (x < (xp[j] - xp[i]) * (y - yp[i]) / (yp[j] - yp[i]) + xp[i]))
803//
804// c = !c;
805// }
806// return c;
807// }
808//
809// See "Point in Polyon Strategies", Eric Haines [Graphic Gems IV] pp. 24-46
810//
811// My algorithm below is rather unique, but is based on code needed to
812// calculate the distance to the shape. I left it in here because ...
813// well ... to test it better.
814//
815G4bool G4PolyPhiFace::InsideEdges( G4double r, G4double z )
816{
817 //
818 // Quick check of extent
819 //
820 if ( r < rMin || r > rMax ) return false;
821 if ( z < zMin || z > zMax ) return false;
822
823 //
824 // More thorough check
825 //
826 G4double notUsed;
827
828 return InsideEdges( r, z, &notUsed, 0 );
829}

832//
833// InsideEdges (care about distance)
834//
835// Decide if the point in r,z is inside the edges of our face
836//
837G4bool G4PolyPhiFace::InsideEdges( G4double r, G4double z,
838 G4double *bestDist2,
839 G4PolyPhiFaceVertex **base3Dnorm,
840 G4ThreeVector **head3Dnorm )
841{
842 G4double bestDistance2 = kInfinity;
843 G4bool answer = 0;
844
845 G4PolyPhiFaceEdge *edge = edges;
846 do // Loop checking, 13.08.2015, G.Cosmo
847 {
848 G4PolyPhiFaceVertex *testMe;
849 //
850 // Get distance perpendicular to the edge
851 //
852 G4double dr = (r-edge->v0->r), dz = (z-edge->v0->z);
853
854 G4double distOut = dr*edge->tz - dz*edge->tr;
855 G4double distance2 = distOut*distOut;
856 if (distance2 > bestDistance2) continue; // No hope!
857
858 //
859 // Check to see if normal intersects edge within the edge's boundary
860 //
861 G4double q = dr*edge->tr + dz*edge->tz;
862
863 //
864 // If it doesn't, penalize distance2 appropriately
865 //
866 if (q < 0)
867 {
868 distance2 += q*q;
869 testMe = edge->v0;
870 }
871 else if (q > edge->length)
872 {
873 G4double s2 = q-edge->length;
874 distance2 += s2*s2;
875 testMe = edge->v1;
876 }
877 else
878 {
879 testMe = 0;
880 }
881
882 //
883 // Closest edge so far?
884 //
885 if (distance2 < bestDistance2)
886 {
887 bestDistance2 = distance2;
888 if (testMe)
889 {
890 G4double distNorm = dr*testMe->rNorm + dz*testMe->zNorm;
891 answer = (distNorm <= 0);
892 if (base3Dnorm)
893 {
894 *base3Dnorm = testMe;
895 *head3Dnorm = &testMe->norm3D;
896 }
897 }
898 else
899 {
900 answer = (distOut <= 0);
901 if (base3Dnorm)
902 {
903 *base3Dnorm = edge->v0;
904 *head3Dnorm = &edge->norm3D;
905 }
906 }
907 }
908 } while( ++edge < edges + numEdges );
909
910 *bestDist2 = bestDistance2;
911 return answer;
912}

1054// if diagonal s of segment (a,b) is convex or reflex
1055//
1056G4bool G4PolyPhiFace::Diagonalie( G4PolyPhiFaceVertex *a,
1057 G4PolyPhiFaceVertex *b )
1058{

1060 G4PolyPhiFaceVertex *corner_next=triangles;
1061
1062 // For each Edge (corner,corner_next)
1063 do // Loop checking, 13.08.2015, G.Cosmo
1064 {
1065 corner_next=corner->next;
1066
1067 // Skip edges incident to a of b
1068 //
1069 if( (corner!=a)&&(corner_next!=a)
1070 &&(corner!=b)&&(corner_next!=b) )
1071 {
1072 G4TwoVector rz1,rz2,rz3,rz4;
1073 rz1 = G4TwoVector(a->r,a->z);
1074 rz2 = G4TwoVector(b->r,b->z);

1076 rz4 = G4TwoVector(corner_next->r,corner_next->z);

1088// being a the center of the Cone

1090G4bool G4PolyPhiFace::InCone( G4PolyPhiFaceVertex *a, G4PolyPhiFaceVertex *b )

1093 //

1096 a0=a->prev;

1100 arz1=G4TwoVector(a1->r,a1->z);brz=G4TwoVector(b->r,b->z);
1101
1102
1103 if(LeftOn(arz,arz1,arz0)) // If a is convex vertex
1104 {
1105 return Left(arz,brz,arz0)&&Left(brz,arz,arz1);
1106 }

1146//
1147// Triangulisation by ear tips for Polycone or Polyhedra
1148// For details see "Computational Geometry in C" by Joseph O'Rourke
1149//
1150void G4PolyPhiFace::Triangulate()
1151{
1152 // The copy of Polycone is made and this copy is reordered in order to
1153 // have a list of triangles. This list is used for GetPointOnFace().
1154
1155 G4PolyPhiFaceVertex *tri_help = new G4PolyPhiFaceVertex[numEdges];
1156 triangles = tri_help;
1157 G4PolyPhiFaceVertex *triang = triangles;
1158
1159 std::vector<G4double> areas;
1160 std::vector<G4ThreeVector> points;
1161 G4double area=0.;
1162 G4PolyPhiFaceVertex *v0,*v1,*v2,*v3,*v4;
1163 v2=triangles;
1164
1165 // Make copy for prev/next for triang=corners
1166 //
1167 G4PolyPhiFaceVertex *helper = corners;
1168 G4PolyPhiFaceVertex *helper2 = corners;
1169 do // Loop checking, 13.08.2015, G.Cosmo
1170 {
1171 triang->r = helper->r;
1172 triang->z = helper->z;
1173 triang->x = helper->x;
1174 triang->y= helper->y;
1175
1176 // add pointer on prev corner
1177 //
1178 if( helper==corners )
1179 { triang->prev=triangles+numEdges-1; }
1180 else
1181 { triang->prev=helper2; }
1182
1183 // add pointer on next corner
1184 //
1185 if( helper<corners+numEdges-1 )
1186 { triang->next=triang+1; }
1187 else
1188 { triang->next=triangles; }
1189 helper2=triang;
1190 helper=helper->next;
1191 triang=triang->next;
1192
1193 } while( helper!=corners );
1194
1195 EarInit();
1196
1197 G4int n=numEdges;
1198 G4int i=0;
1199 G4ThreeVector p1,p2,p3,p4;
1200 const G4int max_n_loops=numEdges*10000; // protection against infinite loop
1201
1202 // Each step of outer loop removes one ear
1203 //
1204 while(n>3) // Loop checking, 13.08.2015, G.Cosmo
1205 { // Inner loop searches for one ear
1206 v2=triangles;
1207 do // Loop checking, 13.08.2015, G.Cosmo
1208 {
1209 if(v2->ear) // Ear found. Fill variables
1210 {
1211 // (v1,v3) is diagonal
1212 //
1213 v3=v2->next; v4=v3->next;
1214 v1=v2->prev; v0=v1->prev;
1215
1216 // Calculate areas and points
1217
1218 p1=G4ThreeVector((v2)->x,(v2)->y,(v2)->z);
1219 p2=G4ThreeVector((v1)->x,(v1)->y,(v1)->z);
1220 p3=G4ThreeVector((v3)->x,(v3)->y,(v3)->z);
1221
1222 G4double result1 = SurfaceTriangle(p1,p2,p3,&p4 );
1223 points.push_back(p4);
1224 areas.push_back(result1);
1225 area=area+result1;
1226
1227 // Update earity of diagonal endpoints
1228 //
1229 v1->ear=Diagonal(v0,v3);
1230 v3->ear=Diagonal(v1,v4);
1231
1232 // Cut off the ear v2
1233 // Has to be done for a copy and not for real PolyPhiFace
1234 //
1235 v1->next=v3;
1236 v3->prev=v1;
1237 triangles=v3; // In case the head was v2
1238 n--;
1239
1240 break; // out of inner loop
1241 } // end if ear found
1242
1243 v2=v2->next;
1244
1245 } while( v2!=triangles );
1246
1247 i++;
1248 if(i>=max_n_loops)
1249 {
1250 G4Exception( "G4PolyPhiFace::Triangulation()",
1251 "GeomSolids0003", FatalException,
1252 "Maximum number of steps is reached for triangulation!" );
1253 }
1254 } // end outer while loop
1255
1256 if(v2->next)
1257 {
1258 // add last triangle
1259 //
1260 v2=v2->next;
1261 p1=G4ThreeVector((v2)->x,(v2)->y,(v2)->z);
1262 p2=G4ThreeVector((v2->next)->x,(v2->next)->y,(v2->next)->z);
1263 p3=G4ThreeVector((v2->prev)->x,(v2->prev)->y,(v2->prev)->z);
1264 G4double result1 = SurfaceTriangle(p1,p2,p3,&p4 );
1265 points.push_back(p4);
1266 areas.push_back(result1);
1267 area=area+result1;
1268 }
1269
1270 // Surface Area is stored
1271 //
1272 fSurfaceArea = area;
1273
1274 // Second Step: choose randomly one surface
1275 //
1276 G4double chose = area*G4UniformRand();
1277
1278 // Third Step: Get a point on choosen surface
1279 //
1280 G4double Achose1, Achose2;
1281 Achose1=0; Achose2=0.;
1282 i=0;
1283 do // Loop checking, 13.08.2015, G.Cosmo
1284 {
1285 Achose2+=areas[i];
1286 if(chose>=Achose1 && chose<Achose2)
1287 {
1288 G4ThreeVector point;
1289 point=points[i] ;
1290 surface_point=point;
1291 break;
1292 }
1293 i++; Achose1=Achose2;
1294 } while( i<numEdges-2 );
1295
1296 delete [] tri_help;
1297 tri_help = 0;
1298}
