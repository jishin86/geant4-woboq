68/////////////////////////////////////////////////////////////////////////
69//
70// Constructor - check parameters
71
72G4EllipticalCone::G4EllipticalCone(const G4String& pName,
73 G4double pxSemiAxis,
74 G4double pySemiAxis,
75 G4double pzMax,
76 G4double pzTopCut)
77 : G4VSolid(pName), fRebuildPolyhedron(false), fpPolyhedron(0),
78 fCubicVolume(0.), fSurfaceArea(0.), zTopCut(0.)
79{
80 halfCarTol = 0.5*kCarTolerance;
81
82 // Check Semi-Axis & Z-cut
83 //
84 if ( (pxSemiAxis <= 0.) || (pySemiAxis <= 0.) || (pzMax <= 0.) )
85 {
86 std::ostringstream message;
87 message << "Invalid semi-axis or height for solid: " << GetName()
88 << "\n X semi-axis, Y semi-axis, height = "
89 << pxSemiAxis << ", " << pySemiAxis << ", " << pzMax;
90 G4Exception("G4EllipticalCone::G4EllipticalCone()", "GeomSolids0002",
91 FatalErrorInArgument, message);
92 }
93
94 if ( pzTopCut <= 0 )
95 {
96 std::ostringstream message;
97 message << "Invalid z-coordinate for cutting plane for solid: " << GetName()
98 << "\n Z top cut = " << pzTopCut;
99 G4Exception("G4EllipticalCone::G4EllipticalCone()", "GeomSolids0002",
100 FatalErrorInArgument, message);
101 }
102
103 SetSemiAxis( pxSemiAxis, pySemiAxis, pzMax );
104 SetZCut(pzTopCut);
105}

171/////////////////////////////////////////////////////////////////////////
172//
173// Get bounding box
174
175void G4EllipticalCone::BoundingLimits(G4ThreeVector& pMin,
176 G4ThreeVector& pMax) const
177{
178 G4double zcut = GetZTopCut();
179 G4double height = GetZMax();
180 G4double xmax = GetSemiAxisX()*(height+zcut);
181 G4double ymax = GetSemiAxisY()*(height+zcut);
182 pMin.set(-xmax,-ymax,-zcut);
183 pMax.set( xmax, ymax, zcut);
184
185 // Check correctness of the bounding box
186 //
187 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
188 {
189 std::ostringstream message;
190 message << "Bad bounding box (min >= max) for solid: "
191 << GetName() << " !"
192 << "\npMin = " << pMin
193 << "\npMax = " << pMax;
194 G4Exception("G4EllipticalCone::BoundingLimits()", "GeomMgt0001",
195 JustWarning, message);
196 DumpInfo();
197 }
198}

200/////////////////////////////////////////////////////////////////////////
201//
202// Calculate extent under transform and specified limit
203
204G4bool
205G4EllipticalCone::CalculateExtent(const EAxis pAxis,
206 const G4VoxelLimits& pVoxelLimit,
207 const G4AffineTransform& pTransform,
208 G4double& pMin, G4double& pMax) const
209{
210 G4ThreeVector bmin,bmax;
211 G4bool exist;
212
213 // Check bounding box (bbox)
214 //
215 BoundingLimits(bmin,bmax);
216 G4BoundingEnvelope bbox(bmin,bmax);
217#ifdef G4BBOX_EXTENT
218 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
219#endif
220 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
221 {
222 return exist = (pMin < pMax) ? true : false;
223 }
224
225 // Set bounding envelope (benv) and calculate extent
226 //
227 static const G4int NSTEPS = 48; // number of steps for whole circle
228 static const G4double ang = twopi/NSTEPS;
229 static const G4double sinHalf = std::sin(0.5*ang);
230 static const G4double cosHalf = std::cos(0.5*ang);
231 static const G4double sinStep = 2.*sinHalf*cosHalf;
232 static const G4double cosStep = 1. - 2.*sinHalf*sinHalf;
233 G4double zcut = bmax.z();
234 G4double height = GetZMax();
235 G4double sxmin = GetSemiAxisX()*(height-zcut)/cosHalf;
236 G4double symin = GetSemiAxisY()*(height-zcut)/cosHalf;
237 G4double sxmax = bmax.x()/cosHalf;
238 G4double symax = bmax.y()/cosHalf;
239
240 G4double sinCur = sinHalf;
241 G4double cosCur = cosHalf;
242 G4ThreeVectorList baseA(NSTEPS),baseB(NSTEPS);
243 for (G4int k=0; k<NSTEPS; ++k)
244 {
245 baseA[k].set(sxmax*cosCur,symax*sinCur,-zcut);
246 baseB[k].set(sxmin*cosCur,symin*sinCur, zcut);
247
248 G4double sinTmp = sinCur;
249 sinCur = sinCur*cosStep + cosCur*sinStep;
250 cosCur = cosCur*cosStep - sinTmp*sinStep;
251 }
252
253 std::vector<const G4ThreeVectorList *> polygons(2);
254 polygons[0] = &baseA;
255 polygons[1] = &baseB;
256 G4BoundingEnvelope benv(bmin,bmax,polygons);
257 exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
258 return exist;
259}

261/////////////////////////////////////////////////////////////////////////
262//
263// Determine where is point: inside, outside or on surface
264
265EInside G4EllipticalCone::Inside(const G4ThreeVector& p) const
266{
267 G4double hp = std::sqrt(p.x()*p.x()*invXX + p.y()*p.y()*invYY) + p.z();
268 G4double ds = (hp - zheight)*cosAxisMin;
269 G4double dz = std::abs(p.z()) - zTopCut;
270 G4double dist = std::max(ds,dz);
271
272 if (dist > halfCarTol) return kOutside;
273 return (dist > -halfCarTol) ? kSurface : kInside;
274}

276/////////////////////////////////////////////////////////////////////////
277//
278// Return unit normal at surface closest to p
279
280G4ThreeVector G4EllipticalCone::SurfaceNormal( const G4ThreeVector& p) const
281{
282 G4ThreeVector norm(0,0,0);
283 G4int nsurf = 0; // number of surfaces where p is placed
284
285 G4double hp = std::sqrt(p.x()*p.x()*invXX + p.y()*p.y()*invYY) + p.z();
286 G4double ds = (hp - zheight)*cosAxisMin;
287 if (std::abs(ds) <= halfCarTol)
288 {
289 norm = G4ThreeVector(p.x()*invXX, p.y()*invYY, hp - p.z());
290 G4double mag = norm.mag();
291 if (mag == 0) return G4ThreeVector(0,0,1); // apex
292 norm *= (1/mag);
293 ++nsurf;
294 }
295 G4double dz = std::abs(p.z()) - zTopCut;
296 if (std::abs(dz) <= halfCarTol)
297 {
298 norm += G4ThreeVector(0., 0.,(p.z() < 0) ? -1. : 1.);
299 ++nsurf;
300 }
301
302 if (nsurf == 1) return norm;
303 else if (nsurf > 1) return norm.unit(); // elliptic edge
304 else
305 {
306 // Point is not on the surface
307 //
308#ifdef G4CSGDEBUG
309 std::ostringstream message;
310 G4int oldprc = message.precision(16);
311 message << "Point p is not on surface (!?) of solid: "
312 << GetName() << G4endl;
313 message << "Position:\n";
314 message << " p.x() = " << p.x()/mm << " mm\n";
315 message << " p.y() = " << p.y()/mm << " mm\n";
316 message << " p.z() = " << p.z()/mm << " mm";
317 G4cout.precision(oldprc);
318 G4Exception("G4EllipticalCone::SurfaceNormal(p)", "GeomSolids1002",
319 JustWarning, message );
320 DumpInfo();
321#endif
322 return ApproxSurfaceNormal(p);
323 }
324}

326/////////////////////////////////////////////////////////////////////////
327//
328// Find surface nearest to point and return corresponding normal.
329// The algorithm is similar to the algorithm used in Inside().
330// This method normally should not be called.
331
332G4ThreeVector
333G4EllipticalCone::ApproxSurfaceNormal(const G4ThreeVector& p) const
334{
335 G4double hp = std::sqrt(p.x()*p.x()*invXX + p.y()*p.y()*invYY) + p.z();
336 G4double ds = (hp - zheight)*cosAxisMin;
337 G4double dz = std::abs(p.z()) - zTopCut;
338 if (ds > dz && std::abs(hp - p.z()) > halfCarTol)
339 return G4ThreeVector(p.x()*invXX, p.y()*invYY, hp - p.z()).unit();
340 else
341 return G4ThreeVector(0., 0.,(p.z() < 0) ? -1. : 1.);
342}

344////////////////////////////////////////////////////////////////////////
345//
346// Calculate distance to shape from outside, along normalised vector
347// return kInfinity if no intersection, or intersection distance <= tolerance
348
349G4double G4EllipticalCone::DistanceToIn( const G4ThreeVector& p,
350 const G4ThreeVector& v ) const
351{
352 G4double distMin = kInfinity;
353
354 // code from EllipticalTube
355
356 G4double sigz = p.z()+zTopCut;
357
358 //
359 // Check z = -dz planer surface
360 //
361
362 if (sigz < halfCarTol)
363 {
364 //
365 // We are "behind" the shape in z, and so can
366 // potentially hit the rear face. Correct direction?
367 //
368 if (v.z() <= 0)
369 {
370 //
371 // As long as we are far enough away, we know we
372 // can't intersect
373 //
374 if (sigz < 0) return kInfinity;
375
376 //
377 // Otherwise, we don't intersect unless we are
378 // on the surface of the ellipse
379 //
380
381 if ( sqr(p.x()/( xSemiAxis - halfCarTol ))
382 + sqr(p.y()/( ySemiAxis - halfCarTol )) <= sqr( zheight + zTopCut ) )
383 return kInfinity;
384
385 }
386 else
387 {
388 //
389 // How far?
390 //
391 G4double q = -sigz/v.z();
392
393 //
394 // Where does that place us?
395 //
396 G4double xi = p.x() + q*v.x(),
397 yi = p.y() + q*v.y();
398
399 //
400 // Is this on the surface (within ellipse)?
401 //
402 if ( sqr(xi/xSemiAxis) + sqr(yi/ySemiAxis) <= sqr( zheight + zTopCut ) )
403 {
404 //
405 // Yup. Return q, unless we are on the surface
406 //
407 return (sigz < -halfCarTol) ? q : 0;
408 }
409 else if (xi/(xSemiAxis*xSemiAxis)*v.x()
410 + yi/(ySemiAxis*ySemiAxis)*v.y() >= 0)
411 {
412 //
413 // Else, if we are traveling outwards, we know
414 // we must miss
415 //
416 // return kInfinity;
417 }
418 }
419 }
420
421 //
422 // Check z = +dz planer surface
423 //
424 sigz = p.z() - zTopCut;
425
426 if (sigz > -halfCarTol)
427 {
428 if (v.z() >= 0)
429 {
430
431 if (sigz > 0) return kInfinity;
432
433 if ( sqr(p.x()/( xSemiAxis - halfCarTol ))
434 + sqr(p.y()/( ySemiAxis - halfCarTol )) <= sqr( zheight-zTopCut ) )
435 return kInfinity;
436
437 }
438 else {
439 G4double q = -sigz/v.z();
440
441 G4double xi = p.x() + q*v.x(),
442 yi = p.y() + q*v.y();
443
444 if ( sqr(xi/xSemiAxis) + sqr(yi/ySemiAxis) <= sqr( zheight - zTopCut ) )
445 {
446 return (sigz > -halfCarTol) ? q : 0;
447 }
448 else if (xi/(xSemiAxis*xSemiAxis)*v.x()
449 + yi/(ySemiAxis*ySemiAxis)*v.y() >= 0)
450 {
451 // return kInfinity;
452 }
453 }
454 }
455
456
457#if 0
458
459 // check to see if Z plane is relevant
460 //
461 if (p.z() < -zTopCut - halfCarTol)
462 {
463 if (v.z() <= 0.0)
464 return distMin;
465
466 G4double lambda = (-zTopCut - p.z())/v.z();
467
468 if ( sqr((lambda*v.x()+p.x())/xSemiAxis) +
469 sqr((lambda*v.y()+p.y())/ySemiAxis) <=
470 sqr(zTopCut + zheight + halfCarTol) )
471 {
472 return distMin = std::fabs(lambda);
473 }
474 }
475
476 if (p.z() > zTopCut + halfCarTol)
477 {
478 if (v.z() >= 0.0)
479 { return distMin; }
480
481 G4double lambda = (zTopCut - p.z()) / v.z();
482
483 if ( sqr((lambda*v.x() + p.x())/xSemiAxis) +
484 sqr((lambda*v.y() + p.y())/ySemiAxis) <=
485 sqr(zheight - zTopCut + halfCarTol) )
486 {
487 return distMin = std::fabs(lambda);
488 }
489 }
490
491 if (p.z() > zTopCut - halfCarTol
492 && p.z() < zTopCut + halfCarTol )
493 {
494 if (v.z() > 0.)
495 { return kInfinity; }
496
497 return distMin = 0.;
498 }
499
500 if (p.z() < -zTopCut + halfCarTol
501 && p.z() > -zTopCut - halfCarTol)
502 {
503 if (v.z() < 0.)
504 { return distMin = kInfinity; }
505
506 return distMin = 0.;
507 }
508
509#endif
510
511 // if we are here then it either intersects or grazes the curved surface
512 // or it does not intersect at all
513 //
514 G4double A = sqr(v.x()/xSemiAxis) + sqr(v.y()/ySemiAxis) - sqr(v.z());
515 G4double B = 2*(v.x()*p.x()/sqr(xSemiAxis) +
516 v.y()*p.y()/sqr(ySemiAxis) + v.z()*(zheight-p.z()));
517 G4double C = sqr(p.x()/xSemiAxis) + sqr(p.y()/ySemiAxis) -
518 sqr(zheight - p.z());
519
520 G4double discr = B*B - 4.*A*C;
521
522 // if the discriminant is negative it never hits the curved object
523 //
524 if ( discr < -halfCarTol )
525 { return distMin; }
526
527 // case below is when it hits or grazes the surface
528 //
529 if ( (discr >= -halfCarTol ) && (discr < halfCarTol ) )
530 {
531 return distMin = std::fabs(-B/(2.*A));
532 }
533
534 G4double plus = (-B+std::sqrt(discr))/(2.*A);
535 G4double minus = (-B-std::sqrt(discr))/(2.*A);
536
537 // Special case::Point on Surface, Check norm.dot(v)
538
539 if ( ( std::fabs(plus) < halfCarTol )||( std::fabs(minus) < halfCarTol ) )
540 {
541 G4ThreeVector truenorm(p.x()/(xSemiAxis*xSemiAxis),
542 p.y()/(ySemiAxis*ySemiAxis),
543 -( p.z() - zheight ));
544 if ( truenorm*v >= 0) // going outside the solid from surface
545 {
546 return kInfinity;
547 }
548 else
549 {
550 return 0;
551 }
552 }
553
554 // G4double lambda = std::fabs(plus) < std::fabs(minus) ? plus : minus;
555 G4double lambda = 0;
556
557 if ( minus > halfCarTol && minus < distMin )
558 {
559 lambda = minus ;
560 // check normal vector n * v < 0
561 G4ThreeVector pin = p + lambda*v;
562 if(std::fabs(pin.z())< zTopCut + halfCarTol)
563 {
564 G4ThreeVector truenorm(pin.x()/(xSemiAxis*xSemiAxis),
565 pin.y()/(ySemiAxis*ySemiAxis),
566 - ( pin.z() - zheight ));
567 if ( truenorm*v < 0)
568 { // yes, going inside the solid
569 distMin = lambda;
570 }
571 }
572 }
573 if ( plus > halfCarTol && plus < distMin )
574 {
575 lambda = plus ;
576 // check normal vector n * v < 0
577 G4ThreeVector pin = p + lambda*v;
578 if(std::fabs(pin.z()) < zTopCut + halfCarTol)
579 {
580 G4ThreeVector truenorm(pin.x()/(xSemiAxis*xSemiAxis),
581 pin.y()/(ySemiAxis*ySemiAxis),
582 - ( pin.z() - zheight ) );
583 if ( truenorm*v < 0)
584 { // yes, going inside the solid
585 distMin = lambda;
586 }
587 }
588 }
589 if (distMin < halfCarTol) distMin=0.;
590 return distMin ;
591}

593/////////////////////////////////////////////////////////////////////////
594//
595// Calculate distance (<= actual) to closest surface of shape from outside
596// Return 0 if point inside
597
598G4double G4EllipticalCone::DistanceToIn(const G4ThreeVector& p) const
599{
600 G4double hp = std::sqrt(p.x()*p.x()*invXX + p.y()*p.y()*invYY) + p.z();
601 G4double ds = (hp - zheight)*cosAxisMin;
602 G4double dz = std::abs(p.z()) - zTopCut;
603 G4double dist = std::max(ds,dz);
604 return (dist > 0) ? dist : 0.;
605}

607////////////////////////////////////////////////////////////////////////
608//
609// Calculate distance to surface of shape from `inside',
610// allowing for tolerance
611
612G4double G4EllipticalCone::DistanceToOut(const G4ThreeVector& p,
613 const G4ThreeVector& v,
614 const G4bool calcNorm,
615 G4bool *validNorm,
616 G4ThreeVector *n ) const
617{
618 G4double distMin, lambda;
619 enum surface_e {kPlaneSurf, kCurvedSurf, kNoSurf} surface;
620
621 distMin = kInfinity;
622 surface = kNoSurf;
623
624 if (v.z() < 0.0)
625 {
626 lambda = (-p.z() - zTopCut)/v.z();
627
628 if ( (sqr((p.x() + lambda*v.x())/xSemiAxis) +
629 sqr((p.y() + lambda*v.y())/ySemiAxis)) <
630 sqr(zheight + zTopCut + halfCarTol) )
631 {
632 distMin = std::fabs(lambda);
633
634 if (!calcNorm) { return distMin; }
635 }
636 distMin = std::fabs(lambda);
637 surface = kPlaneSurf;
638 }
639
640 if (v.z() > 0.0)
641 {
642 lambda = (zTopCut - p.z()) / v.z();
643
644 if ( (sqr((p.x() + lambda*v.x())/xSemiAxis)
645 + sqr((p.y() + lambda*v.y())/ySemiAxis) )
646 < (sqr(zheight - zTopCut + halfCarTol)) )
647 {
648 distMin = std::fabs(lambda);
649 if (!calcNorm) { return distMin; }
650 }
651 distMin = std::fabs(lambda);
652 surface = kPlaneSurf;
653 }
654
655 // if we are here then it either intersects or grazes the
656 // curved surface...
657 //
658 G4double A = sqr(v.x()/xSemiAxis) + sqr(v.y()/ySemiAxis) - sqr(v.z());
659 G4double B = 2.*(v.x()*p.x()/sqr(xSemiAxis) +
660 v.y()*p.y()/sqr(ySemiAxis) + v.z()*(zheight-p.z()));
661 G4double C = sqr(p.x()/xSemiAxis) + sqr(p.y()/ySemiAxis)
662 - sqr(zheight - p.z());
663
664 G4double discr = B*B - 4.*A*C;
665
666 if ( discr >= - halfCarTol && discr < halfCarTol )
667 {
668 if(!calcNorm) { return distMin = std::fabs(-B/(2.*A)); }
669 }
670
671 else if ( discr > halfCarTol )
672 {
673 G4double plus = (-B+std::sqrt(discr))/(2.*A);
674 G4double minus = (-B-std::sqrt(discr))/(2.*A);
675
676 if ( plus > halfCarTol && minus > halfCarTol )
677 {
678 // take the shorter distance
679 //
680 lambda = std::fabs(plus) < std::fabs(minus) ? plus : minus;
681 }
682 else
683 {
684 // at least one solution is close to zero or negative
685 // so, take small positive solution or zero
686 //
687 lambda = plus > -halfCarTol ? plus : 0;
688 }
689
690 if ( std::fabs(lambda) < distMin )
691 {
692 if( std::fabs(lambda) > halfCarTol)
693 {
694 distMin = std::fabs(lambda);
695 surface = kCurvedSurf;
696 }
697 else // Point is On the Surface, Check Normal
698 {
699 G4ThreeVector truenorm(p.x()/(xSemiAxis*xSemiAxis),
700 p.y()/(ySemiAxis*ySemiAxis),
701 -( p.z() - zheight ));
702 if( truenorm.dot(v) > 0 )
703 {
704 distMin = 0.0;
705 surface = kCurvedSurf;
706 }
707 }
708 }
709 }
710
711 // set normal if requested
712 //
713 if (calcNorm)
714 {
715 if (surface == kNoSurf)
716 {
717 *validNorm = false;
718 }
719 else
720 {
721 *validNorm = true;
722 switch (surface)
723 {
724 case kPlaneSurf:
725 {
726 *n = G4ThreeVector(0.,0.,(v.z() > 0.0 ? 1. : -1.));
727 }
728 break;
729
730 case kCurvedSurf:
731 {
732 G4ThreeVector pexit = p + distMin*v;
733 G4ThreeVector truenorm( pexit.x()/(xSemiAxis*xSemiAxis),
734 pexit.y()/(ySemiAxis*ySemiAxis),
735 -( pexit.z() - zheight ) );
736 truenorm /= truenorm.mag();
737 *n= truenorm;
738 }
739 break;
740
741 default: // Should never reach this case ...
742 DumpInfo();
743 std::ostringstream message;
744 G4int oldprc = message.precision(16);
745 message << "Undefined side for valid surface normal to solid."
746 << G4endl
747 << "Position:" << G4endl
748 << " p.x() = " << p.x()/mm << " mm" << G4endl
749 << " p.y() = " << p.y()/mm << " mm" << G4endl
750 << " p.z() = " << p.z()/mm << " mm" << G4endl
751 << "Direction:" << G4endl
752 << " v.x() = " << v.x() << G4endl
753 << " v.y() = " << v.y() << G4endl
754 << " v.z() = " << v.z() << G4endl
755 << "Proposed distance :" << G4endl
756 << " distMin = " << distMin/mm << " mm";
757 message.precision(oldprc);
758 G4Exception("G4EllipticalCone::DistanceToOut(p,v,..)",
759 "GeomSolids1002", JustWarning, message);
760 break;
761 }
762 }
763 }
764
765 if (distMin < halfCarTol) { distMin=0; }
766
767 return distMin;
768}

770/////////////////////////////////////////////////////////////////////////
771//
772// Calculate distance (<=actual) to closest surface of shape from inside
773
774G4double G4EllipticalCone::DistanceToOut(const G4ThreeVector& p) const
775{
776#ifdef G4SPECSDEBUG
777 if( Inside(p) == kOutside )
778 {
779 std::ostringstream message;
780 G4int oldprc = message.precision(16);
781 message << "Point p is outside (!?) of solid: " << GetName() << "\n"
782 << "Position:\n"
783 << " p.x() = " << p.x()/mm << " mm\n"
784 << " p.y() = " << p.y()/mm << " mm\n"
785 << " p.z() = " << p.z()/mm << " mm";
786 message.precision(oldprc) ;
787 G4Exception("G4Ellipsoid::DistanceToOut(p)", "GeomSolids1002",
788 JustWarning, message);
789 DumpInfo();
790 }
791#endif
792 G4double hp = std::sqrt(p.x()*p.x()*invXX + p.y()*p.y()*invYY) + p.z();
793 G4double ds = (zheight - hp)*cosAxisMin;
794 G4double dz = zTopCut - std::abs(p.z());
795 G4double dist = std::min(ds,dz);
796 return (dist > 0) ? dist : 0.;
797}

840/////////////////////////////////////////////////////////////////////////
841//
842// Return random point on the surface of the solid
843
844G4ThreeVector G4EllipticalCone::GetPointOnSurface() const
845{
846 G4double x0 = xSemiAxis*zheight; // x semi axis at z=0
847 G4double y0 = ySemiAxis*zheight; // y semi axis at z=0
848 G4double s0 = G4GeomTools::EllipticConeLateralArea(x0,y0,zheight);
849 G4double kmin = (zTopCut >= zheight ) ? 0. : (zheight - zTopCut)/zheight;
850 G4double kmax = (zTopCut >= zheight ) ? 2. : (zheight + zTopCut)/zheight;
851
852 // Set areas (base at -Z, side surface, base at +Z)
853 //
854 G4double szmin = pi*x0*y0*kmax*kmax;
855 G4double szmax = pi*x0*y0*kmin*kmin;
856 G4double sside = s0*(kmax*kmax - kmin*kmin);
857 G4double ssurf[3] = { szmin, sside, szmax };
858 for (G4int i=1; i<3; ++i) { ssurf[i] += ssurf[i-1]; }
859
860 // Select surface
861 //
862 G4double select = ssurf[2]*G4UniformRand();
863 G4int k = 2;
864 if (select <= ssurf[1]) k = 1;
865 if (select <= ssurf[0]) k = 0;
866
867 // Pick random point on selected surface
868 //
869 G4ThreeVector p;
870 switch(k)
871 {
872 case 0: // base at -Z, uniform distribution, rejection sampling
873 {
874 G4double zh = zheight + zTopCut;
875 G4TwoVector rho = G4RandomPointInEllipse(zh*xSemiAxis,zh*ySemiAxis);
876 p.set(rho.x(),rho.y(),-zTopCut);
877 break;
878 }
879 case 1: // side surface, uniform distribution, rejection sampling
880 {
881 G4double zh = G4RandomRadiusInRing(zheight-zTopCut, zheight+zTopCut);
882 G4double a = x0;
883 G4double b = y0;
884
885 G4double hh = zheight*zheight;
886 G4double aa = a*a;
887 G4double bb = b*b;
888 G4double R = std::max(a,b);
889 G4double mu_max = R*std::sqrt(hh + R*R);
890
891 G4double x,y;
892 for (G4int i=0; i<1000; ++i)
893 {
894 G4double phi = CLHEP::twopi*G4UniformRand();
895 x = std::cos(phi);
896 y = std::sin(phi);
897 G4double xx = x*x;
898 G4double yy = y*y;
899 G4double E = hh + aa*xx + bb*yy;
900 G4double F = (aa-bb)*x*y;
901 G4double G = aa*yy + bb*xx;
902 G4double mu = std::sqrt(E*G - F*F);
903 if (mu_max*G4UniformRand() <= mu) break;
904 }
905 p.set(zh*xSemiAxis*x,zh*ySemiAxis*y,zheight-zh);
906 break;
907 }
908 case 2: // base at +Z, uniform distribution, rejection sampling
909 {
910 G4double zh = zheight - zTopCut;
911 G4TwoVector rho = G4RandomPointInEllipse(zh*xSemiAxis,zh*ySemiAxis);
912 p.set(rho.x(),rho.y(),zTopCut);
913 break;
914 }
915 }
916 return p;
917}

921// Get cubic volume
922
923G4double G4EllipticalCone::GetCubicVolume()
924{
925 if (fCubicVolume == 0)
926 {
927 G4double x0 = xSemiAxis*zheight; // x semi axis at z=0
928 G4double y0 = ySemiAxis*zheight; // y semi axis at z=0
929 G4double v0 = CLHEP::pi*x0*y0*zheight/3.;
930 G4double kmin = (zTopCut >= zheight ) ? 0. : (zheight - zTopCut)/zheight;
931 G4double kmax = (zTopCut >= zheight ) ? 2. : (zheight + zTopCut)/zheight;
932 fCubicVolume = (kmax - kmin)*(kmax*kmax + kmax*kmin + kmin*kmin)*v0;
933 }
934 return fCubicVolume;

941G4double G4EllipticalCone::GetSurfaceArea()

947 G4double s0 = G4GeomTools::EllipticConeLateralArea(x0,y0,zheight);
