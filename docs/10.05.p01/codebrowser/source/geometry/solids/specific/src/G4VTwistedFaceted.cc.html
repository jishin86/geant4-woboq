67//* constructors ------------------------------------------------------
68
69G4VTwistedFaceted::
70G4VTwistedFaceted( const G4String &pname, // Name of instance
71 G4double PhiTwist, // twist angle
72 G4double pDz, // half z length
73 G4double pTheta, // direction between end planes
74 G4double pPhi, // defined by polar and azim. angles
75 G4double pDy1, // half y length at -pDz
76 G4double pDx1, // half x length at -pDz,-pDy
77 G4double pDx2, // half x length at -pDz,+pDy
78 G4double pDy2, // half y length at +pDz
79 G4double pDx3, // half x length at +pDz,-pDy
80 G4double pDx4, // half x length at +pDz,+pDy
81 G4double pAlph // tilt angle
82 )
83 : G4VSolid(pname), fRebuildPolyhedron(false), fpPolyhedron(0),
84 fLowerEndcap(0), fUpperEndcap(0), fSide0(0),
85 fSide90(0), fSide180(0), fSide270(0),
86 fSurfaceArea(0.)
87{
88
89 G4double pDytmp ;
90 G4double fDxUp ;
91 G4double fDxDown ;
92
93 fDx1 = pDx1 ;
94 fDx2 = pDx2 ;
95 fDx3 = pDx3 ;
96 fDx4 = pDx4 ;
97 fDy1 = pDy1 ;
98 fDy2 = pDy2 ;
99 fDz = pDz ;
100
101 G4double kAngTolerance
102 = G4GeometryTolerance::GetInstance()->GetAngularTolerance();
103
104 // maximum values
105 //
106 fDxDown = ( fDx1 > fDx2 ? fDx1 : fDx2 ) ;
107 fDxUp = ( fDx3 > fDx4 ? fDx3 : fDx4 ) ;
108 fDx = ( fDxUp > fDxDown ? fDxUp : fDxDown ) ;
109 fDy = ( fDy1 > fDy2 ? fDy1 : fDy2 ) ;
110
111 // planarity check
112 //
113 if ( fDx1 != fDx2 && fDx3 != fDx4 )
114 {
115 pDytmp = fDy1 * ( fDx3 - fDx4 ) / ( fDx1 - fDx2 ) ;
116 if ( std::fabs(pDytmp - fDy2) > kCarTolerance )
117 {
118 std::ostringstream message;
119 message << "Not planar surface in untwisted Trapezoid: "
120 << GetName() << G4endl
121 << "fDy2 is " << fDy2 << " but should be "
122 << pDytmp << ".";
123 G4Exception("G4VTwistedFaceted::G4VTwistedFaceted()", "GeomSolids0002",
124 FatalErrorInArgument, message);
125 }
126 }
127
128#ifdef G4TWISTDEBUG
129 if ( fDx1 == fDx2 && fDx3 == fDx4 )
130 {
131 G4cout << "Trapezoid is a box" << G4endl ;
132 }
133
134#endif
135
136 if ( ( fDx1 == fDx2 && fDx3 != fDx4 ) || ( fDx1 != fDx2 && fDx3 == fDx4 ) )
137 {
138 std::ostringstream message;
139 message << "Not planar surface in untwisted Trapezoid: "
140 << GetName() << G4endl
141 << "One endcap is rectangular, the other is a trapezoid." << G4endl
142 << "For planarity reasons they have to be rectangles or trapezoids "
143 << "on both sides.";
144 G4Exception("G4VTwistedFaceted::G4VTwistedFaceted()", "GeomSolids0002",
145 FatalErrorInArgument, message);
146 }
147
148 // twist angle
149 //
150 fPhiTwist = PhiTwist ;
151
152 // tilt angle
153 //
154 fAlph = pAlph ;
155 fTAlph = std::tan(fAlph) ;
156
157 fTheta = pTheta ;
158 fPhi = pPhi ;
159
160 // dx in surface equation
161 //
162 fdeltaX = 2 * fDz * std::tan(fTheta) * std::cos(fPhi) ;
163
164 // dy in surface equation
165 //
166 fdeltaY = 2 * fDz * std::tan(fTheta) * std::sin(fPhi) ;
167
168 if ( ! ( ( fDx1 > 2*kCarTolerance)
169 && ( fDx2 > 2*kCarTolerance)
170 && ( fDx3 > 2*kCarTolerance)
171 && ( fDx4 > 2*kCarTolerance)
172 && ( fDy1 > 2*kCarTolerance)
173 && ( fDy2 > 2*kCarTolerance)
174 && ( fDz > 2*kCarTolerance)
175 && ( std::fabs(fPhiTwist) > 2*kAngTolerance )
176 && ( std::fabs(fPhiTwist) < pi/2 )
177 && ( std::fabs(fAlph) < pi/2 )
178 && ( fTheta < pi/2 && fTheta >= 0 ) )
179 )
180 {
181 std::ostringstream message;
182 message << "Invalid dimensions. Too small, or twist angle too big: "
183 << GetName() << G4endl
184 << "fDx 1-4 = " << fDx1/cm << ", " << fDx2/cm << ", "
185 << fDx3/cm << ", " << fDx4/cm << " cm" << G4endl
186 << "fDy 1-2 = " << fDy1/cm << ", " << fDy2/cm << ", "
187 << " cm" << G4endl
188 << "fDz = " << fDz/cm << " cm" << G4endl
189 << " twistangle " << fPhiTwist/deg << " deg" << G4endl
190 << " phi,theta = " << fPhi/deg << ", " << fTheta/deg << " deg";
191 G4Exception("G4TwistedTrap::G4VTwistedFaceted()",
192 "GeomSolids0002", FatalErrorInArgument, message);
193 }
194 CreateSurfaces();
195 fCubicVolume = 2 * fDz * ( ( fDx1 + fDx2 ) * fDy1 + ( fDx3 + fDx4 ) * fDy2 );
196}

254
255G4VTwistedFaceted& G4VTwistedFaceted::operator = (const G4VTwistedFaceted& rhs)
256{
257 // Check assignment to self
258 //
259 if (this == &rhs) { return *this; }
260
261 // Copy base class data
262 //
263 G4VSolid::operator=(rhs);
264
265 // Copy data
266 //
267 fTheta = rhs.fTheta; fPhi = rhs.fPhi;
268 fDy1= rhs.fDy1; fDx1= rhs.fDx1; fDx2= rhs.fDx2; fDy2= rhs.fDy2;
269 fDx3= rhs.fDx3; fDx4= rhs.fDx4; fDz= rhs.fDz; fDx= rhs.fDx; fDy= rhs.fDy;
270 fAlph= rhs.fAlph; fTAlph= rhs.fTAlph; fdeltaX= rhs.fdeltaX;
271 fdeltaY= rhs.fdeltaY; fPhiTwist= rhs.fPhiTwist; fLowerEndcap= 0;
272 fUpperEndcap= 0; fSide0= 0; fSide90= 0; fSide180= 0; fSide270= 0;
273 fCubicVolume= rhs.fCubicVolume; fSurfaceArea= rhs.fSurfaceArea;
274 fRebuildPolyhedron = false;
275 delete fpPolyhedron; fpPolyhedron= 0;
276 fLastInside= rhs.fLastInside; fLastNormal= rhs.fLastNormal;
277 fLastDistanceToIn= rhs.fLastDistanceToIn;
278 fLastDistanceToOut= rhs.fLastDistanceToOut;
279 fLastDistanceToInWithV= rhs.fLastDistanceToInWithV;
280 fLastDistanceToOutWithV= rhs.fLastDistanceToOutWithV;
281
282 CreateSurfaces();
283
284 return *this;
285}

334//=====================================================================
335//* Inside ------------------------------------------------------------
336
337EInside G4VTwistedFaceted::Inside(const G4ThreeVector& p) const
338{
339
340 G4ThreeVector *tmpp;
341 EInside *tmpin;
342 if (fLastInside.p == p) {
343 return fLastInside.inside;
344 } else {
345 tmpp = const_cast<G4ThreeVector*>(&(fLastInside.p));
346 tmpin = const_cast<EInside*>(&(fLastInside.inside));
347 tmpp->set(p.x(), p.y(), p.z());
348 }
349
350 *tmpin = kOutside ;
351
352 G4double phi = p.z()/(2*fDz) * fPhiTwist ; // rotate the point to z=0
353 G4double cphi = std::cos(-phi) ;
354 G4double sphi = std::sin(-phi) ;
355
356 G4double px = p.x() + fdeltaX * ( -phi/fPhiTwist) ; // shift
357 G4double py = p.y() + fdeltaY * ( -phi/fPhiTwist) ;
358 G4double pz = p.z() ;
359
360 G4double posx = px * cphi - py * sphi ; // rotation
361 G4double posy = px * sphi + py * cphi ;
362 G4double posz = pz ;
363
364 G4double xMin = Xcoef(posy,phi,fTAlph) - 2*Xcoef(posy,phi,0.) ;
365 G4double xMax = Xcoef(posy,phi,fTAlph) ;
366
367 G4double yMax = GetValueB(phi)/2. ; // b(phi)/2 is limit
368 G4double yMin = -yMax ;
369
370#ifdef G4TWISTDEBUG
371
372 G4cout << "inside called: p = " << p << G4endl ;
373 G4cout << "fDx1 = " << fDx1 << G4endl ;
374 G4cout << "fDx2 = " << fDx2 << G4endl ;
375 G4cout << "fDx3 = " << fDx3 << G4endl ;
376 G4cout << "fDx4 = " << fDx4 << G4endl ;
377
378 G4cout << "fDy1 = " << fDy1 << G4endl ;
379 G4cout << "fDy2 = " << fDy2 << G4endl ;
380
381 G4cout << "fDz = " << fDz << G4endl ;
382
383 G4cout << "Tilt angle alpha = " << fAlph << G4endl ;
384 G4cout << "phi,theta = " << fPhi << " , " << fTheta << G4endl ;
385
386 G4cout << "Twist angle = " << fPhiTwist << G4endl ;
387
388 G4cout << "posx = " << posx << G4endl ;
389 G4cout << "posy = " << posy << G4endl ;
390 G4cout << "xMin = " << xMin << G4endl ;
391 G4cout << "xMax = " << xMax << G4endl ;
392 G4cout << "yMin = " << yMin << G4endl ;
393 G4cout << "yMax = " << yMax << G4endl ;
394
395#endif
396
397
398 if ( posx <= xMax - kCarTolerance*0.5
399 && posx >= xMin + kCarTolerance*0.5 )
400 {
401 if ( posy <= yMax - kCarTolerance*0.5
402 && posy >= yMin + kCarTolerance*0.5 )
403 {
404 if (std::fabs(posz) <= fDz - kCarTolerance*0.5 ) *tmpin = kInside ;
405 else if (std::fabs(posz) <= fDz + kCarTolerance*0.5 ) *tmpin = kSurface ;
406 }
407 else if ( posy <= yMax + kCarTolerance*0.5
408 && posy >= yMin - kCarTolerance*0.5 )
409 {
410 if (std::fabs(posz) <= fDz + kCarTolerance*0.5 ) *tmpin = kSurface ;
411 }
412 }
413 else if ( posx <= xMax + kCarTolerance*0.5
414 && posx >= xMin - kCarTolerance*0.5 )
415 {
416 if ( posy <= yMax + kCarTolerance*0.5
417 && posy >= yMin - kCarTolerance*0.5 )
418 {
419 if (std::fabs(posz) <= fDz + kCarTolerance*0.5) *tmpin = kSurface ;
420 }
421 }
422
423#ifdef G4TWISTDEBUG
424 G4cout << "inside = " << fLastInside.inside << G4endl ;
425#endif
426
427 return fLastInside.inside;
428
429}

432//=====================================================================
433//* SurfaceNormal -----------------------------------------------------
434
435G4ThreeVector G4VTwistedFaceted::SurfaceNormal(const G4ThreeVector& p) const
436{
437 //
438 // return the normal unit vector to the Hyperbolical Surface at a point
439 // p on (or nearly on) the surface
440 //
441 // Which of the three or four surfaces are we closest to?
442 //
443
444 if (fLastNormal.p == p)
445 {
446 return fLastNormal.vec;
447 }
448
449 G4ThreeVector *tmpp = const_cast<G4ThreeVector*>(&(fLastNormal.p));
450 G4ThreeVector *tmpnormal = const_cast<G4ThreeVector*>(&(fLastNormal.vec));
451 G4VTwistSurface **tmpsurface = const_cast<G4VTwistSurface**>(fLastNormal.surface);
452 tmpp->set(p.x(), p.y(), p.z());
453
454 G4double distance = kInfinity;
455
456 G4VTwistSurface *surfaces[6];
457
458 surfaces[0] = fSide0 ;
459 surfaces[1] = fSide90 ;
460 surfaces[2] = fSide180 ;
461 surfaces[3] = fSide270 ;
462 surfaces[4] = fLowerEndcap;
463 surfaces[5] = fUpperEndcap;
464
465 G4ThreeVector xx;
466 G4ThreeVector bestxx;
467 G4int i;
468 G4int besti = -1;
469 for (i=0; i< 6; i++)
470 {
471 G4double tmpdistance = surfaces[i]->DistanceTo(p, xx);
472 if (tmpdistance < distance)
473 {
474 distance = tmpdistance;
475 bestxx = xx;
476 besti = i;
477 }
478 }
479
480 tmpsurface[0] = surfaces[besti];
481 *tmpnormal = tmpsurface[0]->GetNormal(bestxx, true);
482
483 return fLastNormal.vec;
484}

487//=====================================================================
488//* DistanceToIn (p, v) -----------------------------------------------
489
490G4double G4VTwistedFaceted::DistanceToIn (const G4ThreeVector& p,
491 const G4ThreeVector& v ) const
492{
493
494 // DistanceToIn (p, v):
495 // Calculate distance to surface of shape from `outside'
496 // along with the v, allowing for tolerance.
497 // The function returns kInfinity if no intersection or
498 // just grazing within tolerance.
499
500 //
501 // checking last value
502 //
503
504 G4ThreeVector *tmpp;
505 G4ThreeVector *tmpv;
506 G4double *tmpdist;
507 if (fLastDistanceToInWithV.p == p && fLastDistanceToInWithV.vec == v)
508 {
509 return fLastDistanceToIn.value;
510 }
511 else
512 {
513 tmpp = const_cast<G4ThreeVector*>(&(fLastDistanceToInWithV.p));
514 tmpv = const_cast<G4ThreeVector*>(&(fLastDistanceToInWithV.vec));
515 tmpdist = const_cast<G4double*>(&(fLastDistanceToInWithV.value));
516 tmpp->set(p.x(), p.y(), p.z());
517 tmpv->set(v.x(), v.y(), v.z());
518 }
519
520 //
521 // Calculate DistanceToIn(p,v)
522 //
523
524 EInside currentside = Inside(p);
525
526 if (currentside == kInside)
527 {
528 }
529 else if (currentside == kSurface)
530 {
531 // particle is just on a boundary.
532 // if the particle is entering to the volume, return 0
533 //
534 G4ThreeVector normal = SurfaceNormal(p);
535 if (normal*v < 0)
536 {
537 *tmpdist = 0;
538 return fLastDistanceToInWithV.value;
539 }
540 }
541
542 // now, we can take smallest positive distance.
543
544 // Initialize
545 //
546 G4double distance = kInfinity;
547
548 // Find intersections and choose nearest one
549 //
550 G4VTwistSurface *surfaces[6];
551
552 surfaces[0] = fSide0;
553 surfaces[1] = fSide90 ;
554 surfaces[2] = fSide180 ;
555 surfaces[3] = fSide270 ;
556 surfaces[4] = fLowerEndcap;
557 surfaces[5] = fUpperEndcap;
558
559 G4ThreeVector xx;
560 G4ThreeVector bestxx;
561 G4int i;
562 for (i=0; i < 6 ; i++)
563 {
564#ifdef G4TWISTDEBUG
565 G4cout << G4endl << "surface " << i << ": " << G4endl << G4endl ;
566#endif
567 G4double tmpdistance = surfaces[i]->DistanceToIn(p, v, xx);
568#ifdef G4TWISTDEBUG
569 G4cout << "Solid DistanceToIn : distance = " << tmpdistance << G4endl ;
570 G4cout << "intersection point = " << xx << G4endl ;
571#endif
572 if (tmpdistance < distance)
573 {
574 distance = tmpdistance;
575 bestxx = xx;
576 }
577 }
578
579#ifdef G4TWISTDEBUG
580 G4cout << "best distance = " << distance << G4endl ;
581#endif
582
583 *tmpdist = distance;
584 // timer.Stop();
585 return fLastDistanceToInWithV.value;
586}

589//=====================================================================
590//* DistanceToIn (p) --------------------------------------------------
591
592G4double G4VTwistedFaceted::DistanceToIn (const G4ThreeVector& p) const
593{
594 // DistanceToIn(p):
595 // Calculate distance to surface of shape from `outside',
596 // allowing for tolerance
597 //
598
599 //
600 // checking last value
601 //
602
603 G4ThreeVector *tmpp;
604 G4double *tmpdist;
605 if (fLastDistanceToIn.p == p)
606 {
607 return fLastDistanceToIn.value;
608 }
609 else
610 {
611 tmpp = const_cast<G4ThreeVector*>(&(fLastDistanceToIn.p));
612 tmpdist = const_cast<G4double*>(&(fLastDistanceToIn.value));
613 tmpp->set(p.x(), p.y(), p.z());
614 }
615
616 //
617 // Calculate DistanceToIn(p)
618 //
619
620 EInside currentside = Inside(p);
621
622 switch (currentside)
623 {
624 case (kInside) :
625 {
626 }
627
628 case (kSurface) :
629 {
630 *tmpdist = 0.;
631 return fLastDistanceToIn.value;
632 }
633
634 case (kOutside) :
635 {
636 // Initialize
637 //
638 G4double distance = kInfinity;
639
640 // Find intersections and choose nearest one
641 //
642 G4VTwistSurface *surfaces[6];
643
644 surfaces[0] = fSide0;
645 surfaces[1] = fSide90 ;
646 surfaces[2] = fSide180 ;
647 surfaces[3] = fSide270 ;
648 surfaces[4] = fLowerEndcap;
649 surfaces[5] = fUpperEndcap;
650
651 G4int i;
652 G4ThreeVector xx;
653 G4ThreeVector bestxx;
654 for (i=0; i< 6; i++)
655 {
656 G4double tmpdistance = surfaces[i]->DistanceTo(p, xx);
657 if (tmpdistance < distance)
658 {
659 distance = tmpdistance;
660 bestxx = xx;
661 }
662 }
663 *tmpdist = distance;
664 return fLastDistanceToIn.value;
665 }
666
667 default :
668 {
669 G4Exception("G4VTwistedFaceted::DistanceToIn(p)", "GeomSolids0003",
670 FatalException, "Unknown point location!");
671 }
672 } // switch end
673
674 return 0;
675}

680
681G4double
682G4VTwistedFaceted::DistanceToOut( const G4ThreeVector& p,
683 const G4ThreeVector& v,
684 const G4bool calcNorm,
685 G4bool *validNorm,
686 G4ThreeVector *norm ) const
687{
688 // DistanceToOut (p, v):
689 // Calculate distance to surface of shape from `inside'
690 // along with the v, allowing for tolerance.
691 // The function returns kInfinity if no intersection or
692 // just grazing within tolerance.
693
694 //
695 // checking last value
696 //
697
698 G4ThreeVector *tmpp;
699 G4ThreeVector *tmpv;
700 G4double *tmpdist;
701 if (fLastDistanceToOutWithV.p == p && fLastDistanceToOutWithV.vec == v )
702 {
703 return fLastDistanceToOutWithV.value;
704 }
705 else
706 {
707 tmpp = const_cast<G4ThreeVector*>(&(fLastDistanceToOutWithV.p));
708 tmpv = const_cast<G4ThreeVector*>(&(fLastDistanceToOutWithV.vec));
709 tmpdist = const_cast<G4double*>(&(fLastDistanceToOutWithV.value));
710 tmpp->set(p.x(), p.y(), p.z());
711 tmpv->set(v.x(), v.y(), v.z());
712 }
713
714 //
715 // Calculate DistanceToOut(p,v)
716 //
717
718 EInside currentside = Inside(p);
719
720 if (currentside == kOutside)
721 {
722 }
723 else if (currentside == kSurface)
724 {
725 // particle is just on a boundary.
726 // if the particle is exiting from the volume, return 0
727 //
728 G4ThreeVector normal = SurfaceNormal(p);
729 G4VTwistSurface *blockedsurface = fLastNormal.surface[0];
730 if (normal*v > 0)
731 {
732 if (calcNorm)
733 {
734 *norm = (blockedsurface->GetNormal(p, true));
735 *validNorm = blockedsurface->IsValidNorm();
736 }
737 *tmpdist = 0.;
738 // timer.Stop();
739 return fLastDistanceToOutWithV.value;
740 }
741 }
742
743 // now, we can take smallest positive distance.
744
745 // Initialize
746 G4double distance = kInfinity;
747
748 // find intersections and choose nearest one.
749 G4VTwistSurface *surfaces[6];
750
751 surfaces[0] = fSide0;
752 surfaces[1] = fSide90 ;
753 surfaces[2] = fSide180 ;
754 surfaces[3] = fSide270 ;
755 surfaces[4] = fLowerEndcap;
756 surfaces[5] = fUpperEndcap;
757
758 G4int i;
759 G4int besti = -1;
760 G4ThreeVector xx;
761 G4ThreeVector bestxx;
762 for (i=0; i< 6 ; i++) {
763 G4double tmpdistance = surfaces[i]->DistanceToOut(p, v, xx);
764 if (tmpdistance < distance)
765 {
766 distance = tmpdistance;
767 bestxx = xx;
768 besti = i;
769 }
770 }
771
772 if (calcNorm)
773 {
774 if (besti != -1)
775 {
776 *norm = (surfaces[besti]->GetNormal(p, true));
777 *validNorm = surfaces[besti]->IsValidNorm();
778 }
779 }
780
781 *tmpdist = distance;
782 // timer.Stop();
783 return fLastDistanceToOutWithV.value;
784}

790G4double G4VTwistedFaceted::DistanceToOut( const G4ThreeVector& p ) const
791{
792 // DistanceToOut(p):
793 // Calculate distance to surface of shape from `inside',
794 // allowing for tolerance
795
796 //
797 // checking last value
798 //
799
800 G4ThreeVector *tmpp;
801 G4double *tmpdist;
802
803 if (fLastDistanceToOut.p == p)
804 {
805 return fLastDistanceToOut.value;
806 }
807 else
808 {
809 tmpp = const_cast<G4ThreeVector*>(&(fLastDistanceToOut.p));
810 tmpdist = const_cast<G4double*>(&(fLastDistanceToOut.value));
811 tmpp->set(p.x(), p.y(), p.z());
812 }
813
814 //
815 // Calculate DistanceToOut(p)
816 //
817
818 EInside currentside = Inside(p);
819 G4double retval = kInfinity;
820
821 switch (currentside)
822 {
823 case (kOutside) :
824 {
825#ifdef G4SPECSDEBUG
826 G4int oldprc = G4cout.precision(16) ;
827 G4cout << G4endl ;
828 DumpInfo();
829 G4cout << "Position:" << G4endl << G4endl ;
830 G4cout << "p.x() = " << p.x()/mm << " mm" << G4endl ;
831 G4cout << "p.y() = " << p.y()/mm << " mm" << G4endl ;
832 G4cout << "p.z() = " << p.z()/mm << " mm" << G4endl << G4endl ;
833 G4cout.precision(oldprc) ;
834 G4Exception("G4VTwistedFaceted::DistanceToOut(p)", "GeomSolids1002",
835 JustWarning, "Point p is outside !?" );
836#endif
837 break;
838 }
839 case (kSurface) :
840 {
841 *tmpdist = 0.;
842 retval = fLastDistanceToOut.value;
843 break;
844 }
845
846 case (kInside) :
847 {
848 // Initialize
849 //
850 G4double distance = kInfinity;
851
852 // find intersections and choose nearest one
853 //
854 G4VTwistSurface *surfaces[6];
855
856 surfaces[0] = fSide0;
857 surfaces[1] = fSide90 ;
858 surfaces[2] = fSide180 ;
859 surfaces[3] = fSide270 ;
860 surfaces[4] = fLowerEndcap;
861 surfaces[5] = fUpperEndcap;
862
863 G4int i;
864 G4ThreeVector xx;
865 G4ThreeVector bestxx;
866 for (i=0; i< 6; i++)
867 {
868 G4double tmpdistance = surfaces[i]->DistanceTo(p, xx);
869 if (tmpdistance < distance)
870 {
871 distance = tmpdistance;
872 bestxx = xx;
873 }
874 }
875 *tmpdist = distance;
876
877 retval = fLastDistanceToOut.value;
878 break;
879 }
880
881 default :
882 {
883 G4Exception("G4VTwistedFaceted::DistanceToOut(p)", "GeomSolids0003",
884 FatalException, "Unknown point location!");
885 break;
886 }
887 } // switch end
888
889 return retval;
890}

952//=====================================================================
953//* CreateSurfaces ----------------------------------------------------
954
955void G4VTwistedFaceted::CreateSurfaces()
956{
957
958 // create 6 surfaces of TwistedTub.
959
960 if ( fDx1 == fDx2 && fDx3 == fDx4 ) // special case : Box
961 {
962 fSide0 = new G4TwistBoxSide("0deg", fPhiTwist, fDz, fTheta, fPhi,
963 fDy1, fDx1, fDx1, fDy2, fDx3, fDx3, fAlph, 0.*deg);
964 fSide180 = new G4TwistBoxSide("180deg", fPhiTwist, fDz, fTheta, fPhi+pi,
965 fDy1, fDx1, fDx1, fDy2, fDx3, fDx3, fAlph, 180.*deg);
966 }
967 else // default general case
968 {
969 fSide0 = new G4TwistTrapAlphaSide("0deg" ,fPhiTwist, fDz, fTheta,
970 fPhi, fDy1, fDx1, fDx2, fDy2, fDx3, fDx4, fAlph, 0.*deg);
971 fSide180 = new G4TwistTrapAlphaSide("180deg", fPhiTwist, fDz, fTheta,
972 fPhi+pi, fDy1, fDx2, fDx1, fDy2, fDx4, fDx3, fAlph, 180.*deg);
973 }
974
975 // create parallel sides
976 //
977 fSide90 = new G4TwistTrapParallelSide("90deg", fPhiTwist, fDz, fTheta,
978 fPhi, fDy1, fDx1, fDx2, fDy2, fDx3, fDx4, fAlph, 0.*deg);
979 fSide270 = new G4TwistTrapParallelSide("270deg", fPhiTwist, fDz, fTheta,
980 fPhi+pi, fDy1, fDx2, fDx1, fDy2, fDx4, fDx3, fAlph, 180.*deg);
981
982 // create endcaps
983 //
984 fUpperEndcap = new G4TwistTrapFlatSide("UpperCap",fPhiTwist, fDx3, fDx4, fDy2,
985 fDz, fAlph, fPhi, fTheta, 1 );
986 fLowerEndcap = new G4TwistTrapFlatSide("LowerCap",fPhiTwist, fDx1, fDx2, fDy1,
987 fDz, fAlph, fPhi, fTheta, -1 );
988
989 // Set neighbour surfaces
990
991 fSide0->SetNeighbours( fSide270 , fLowerEndcap , fSide90 , fUpperEndcap );
992 fSide90->SetNeighbours( fSide0 , fLowerEndcap , fSide180 , fUpperEndcap );
993 fSide180->SetNeighbours(fSide90 , fLowerEndcap , fSide270 , fUpperEndcap );
994 fSide270->SetNeighbours(fSide180 , fLowerEndcap , fSide0 , fUpperEndcap );
995 fUpperEndcap->SetNeighbours( fSide180, fSide270 , fSide0 , fSide90 );
996 fLowerEndcap->SetNeighbours( fSide180, fSide270 , fSide0 , fSide90 );
997
998}

1051//* GetPointOnSurface -------------------------------------------------
1052
1053G4ThreeVector G4VTwistedFaceted::GetPointOnSurface() const
1054{
1055
1056 G4double phi = G4RandFlat::shoot(-fPhiTwist/2.,fPhiTwist/2.);
1057 G4double u , umin, umax ; // variable for twisted surfaces
1058 G4double y ; // variable for flat surface (top and bottom)
1059
1060 // Compute the areas. Attention: Only correct for trapezoids
1061 // where the twisting is done along the z-axis. In the general case
1062 // the computed surface area is more difficult. However this simplification
1063 // does not affect the tracking through the solid.
1064
1065 G4double a1 = fSide0->GetSurfaceArea();
1066 G4double a2 = fSide90->GetSurfaceArea();
1067 G4double a3 = fSide180->GetSurfaceArea() ;
1068 G4double a4 = fSide270->GetSurfaceArea() ;
1069 G4double a5 = fLowerEndcap->GetSurfaceArea() ;
1070 G4double a6 = fUpperEndcap->GetSurfaceArea() ;
1071
1072#ifdef G4TWISTDEBUG
1073 G4cout << "Surface 0 deg = " << a1 << G4endl ;
1074 G4cout << "Surface 90 deg = " << a2 << G4endl ;
1075 G4cout << "Surface 180 deg = " << a3 << G4endl ;
1076 G4cout << "Surface 270 deg = " << a4 << G4endl ;
1077 G4cout << "Surface Lower = " << a5 << G4endl ;
1078 G4cout << "Surface Upper = " << a6 << G4endl ;
1079#endif
1080
1081 G4double chose = G4RandFlat::shoot(0.,a1 + a2 + a3 + a4 + a5 + a6) ;
1082
1083 if(chose < a1)
1084 {
1085
1086 umin = fSide0->GetBoundaryMin(phi) ;
1087 umax = fSide0->GetBoundaryMax(phi) ;
1088 u = G4RandFlat::shoot(umin,umax) ;
1089
1090 return fSide0->SurfacePoint(phi, u, true) ; // point on 0deg surface
1091 }
1092
1093 else if( (chose >= a1) && (chose < a1 + a2 ) )
1094 {
1095
1096 umin = fSide90->GetBoundaryMin(phi) ;
1097 umax = fSide90->GetBoundaryMax(phi) ;
1098
1099 u = G4RandFlat::shoot(umin,umax) ;
1100
1101 return fSide90->SurfacePoint(phi, u, true); // point on 90deg surface
1102 }
1103
1104 else if( (chose >= a1 + a2 ) && (chose < a1 + a2 + a3 ) )
1105 {
1106
1107 umin = fSide180->GetBoundaryMin(phi) ;
1108 umax = fSide180->GetBoundaryMax(phi) ;
1109 u = G4RandFlat::shoot(umin,umax) ;
1110
1111 return fSide180->SurfacePoint(phi, u, true); // point on 180 deg surface
1112 }
1113
1114 else if( (chose >= a1 + a2 + a3 ) && (chose < a1 + a2 + a3 + a4 ) )
1115 {
1116
1117 umin = fSide270->GetBoundaryMin(phi) ;
1118 umax = fSide270->GetBoundaryMax(phi) ;
1119 u = G4RandFlat::shoot(umin,umax) ;
1120
1121 return fSide270->SurfacePoint(phi, u, true); // point on 270 deg surface
1122 }
1123
1124 else if( (chose >= a1 + a2 + a3 + a4 ) && (chose < a1 + a2 + a3 + a4 + a5 ) )
1125 {
1126
1127 y = G4RandFlat::shoot(-fDy1,fDy1) ;
1128 umin = fLowerEndcap->GetBoundaryMin(y) ;
1129 umax = fLowerEndcap->GetBoundaryMax(y) ;
1130 u = G4RandFlat::shoot(umin,umax) ;
1131
1132 return fLowerEndcap->SurfacePoint(u,y,true); // point on lower endcap
1133 }
1134 else {
1135
1136 y = G4RandFlat::shoot(-fDy2,fDy2) ;
1137 umin = fUpperEndcap->GetBoundaryMin(y) ;
1138 umax = fUpperEndcap->GetBoundaryMax(y) ;
1139 u = G4RandFlat::shoot(umin,umax) ;
1140
1141 return fUpperEndcap->SurfacePoint(u,y,true) ; // point on upper endcap
1142
1143 }
1144}

1148//* CreatePolyhedron --------------------------------------------------
1149
1150G4Polyhedron* G4VTwistedFaceted::CreatePolyhedron () const
1151{
1152 // number of meshes
1153 const G4int k =
1154 G4int(G4Polyhedron::GetNumberOfRotationSteps() *
1155 std::abs(fPhiTwist) / twopi) + 2;
1156 const G4int n = k;
1157
1158 const G4int nnodes = 4*(k-1)*(n-2) + 2*k*k ;
1159 const G4int nfaces = 4*(k-1)*(n-1) + 2*(k-1)*(k-1) ;
1160
1161 G4Polyhedron *ph=new G4Polyhedron;
1162 typedef G4double G4double3[3];
1163 typedef G4int G4int4[4];
1164 G4double3* xyz = new G4double3[nnodes]; // number of nodes
1165 G4int4* faces = new G4int4[nfaces] ; // number of faces
1166
1167 fLowerEndcap->GetFacets(k,k,xyz,faces,0) ;
1168 fUpperEndcap->GetFacets(k,k,xyz,faces,1) ;
1169 fSide270->GetFacets(k,n,xyz,faces,2) ;
1170 fSide0->GetFacets(k,n,xyz,faces,3) ;
1171 fSide90->GetFacets(k,n,xyz,faces,4) ;
1172 fSide180->GetFacets(k,n,xyz,faces,5) ;
1173
1174 ph->createPolyhedron(nnodes,nfaces,xyz,faces);
1175
1176 return ph;
1177}
