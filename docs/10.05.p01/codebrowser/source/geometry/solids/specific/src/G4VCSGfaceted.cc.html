137//
138// CopyStuff (protected)
139//
140// Copy the contents of source
141//
142void G4VCSGfaceted::CopyStuff( const G4VCSGfaceted &source )
143{
144 numFace = source.numFace;
145 if (numFace == 0) { return; } // odd, but permissable?
146
147 faces = new G4VCSGface*[numFace];
148
149 G4VCSGface **face = faces,
150 **sourceFace = source.faces;
151 do // Loop checking, 13.08.2015, G.Cosmo
152 {
153 *face = (*sourceFace)->Clone();
154 } while( ++sourceFace, ++face < faces+numFace );
155 fCubicVolume = source.fCubicVolume;
156 fSurfaceArea = source.fSurfaceArea;
157 fRebuildPolyhedron = false;
158 fpPolyhedron = 0;
159}

162//
163// DeleteStuff (protected)
164//
165// Delete all allocated objects
166//
167void G4VCSGfaceted::DeleteStuff()
168{
169 if (numFace)
170 {
171 G4VCSGface **face = faces;
172 do // Loop checking, 13.08.2015, G.Cosmo
173 {
174 delete *face;
175 } while( ++face < faces + numFace );
176
177 delete [] faces;
178 }
179 delete fpPolyhedron; fpPolyhedron = 0;
180}

183//
184// CalculateExtent
185//
186G4bool G4VCSGfaceted::CalculateExtent( const EAxis axis,
187 const G4VoxelLimits &voxelLimit,
188 const G4AffineTransform &transform,
189 G4double &min,
190 G4double &max ) const
191{
192 G4SolidExtentList extentList( axis, voxelLimit );
193
194 //
195 // Loop over all faces, checking min/max extent as we go.
196 //
197 G4VCSGface **face = faces;
198 do // Loop checking, 13.08.2015, G.Cosmo
199 {
200 (*face)->CalculateExtent( axis, voxelLimit, transform, extentList );
201 } while( ++face < faces + numFace );
202
203 //
204 // Return min/max value
205 //
206 return extentList.GetExtent( min, max );
207}

210//
211// Inside
212//
213// It could be a good idea to override this virtual
214// member to add first a simple test (such as spherical
215// test or whatnot) and to call this version only if
216// the simplier test fails.
217//
218EInside G4VCSGfaceted::Inside( const G4ThreeVector &p ) const
219{
220 EInside answer=kOutside;
221 G4VCSGface **face = faces;
222 G4double best = kInfinity;
223 do // Loop checking, 13.08.2015, G.Cosmo
224 {
225 G4double distance;
226 EInside result = (*face)->Inside( p, kCarTolerance/2, &distance );
227 if (result == kSurface) { return kSurface; }
228 if (distance < best)
229 {
230 best = distance;
231 answer = result;
232 }
233 } while( ++face < faces + numFace );
234
235 return answer;
236}

262//
263// DistanceToIn(p,v)
264//
265G4double G4VCSGfaceted::DistanceToIn( const G4ThreeVector &p,
266 const G4ThreeVector &v ) const
267{
268 G4double distance = kInfinity;
269 G4double distFromSurface = kInfinity;
270 G4VCSGface **face = faces;
271 G4VCSGface *bestFace = *face;
272 do // Loop checking, 13.08.2015, G.Cosmo
273 {
274 G4double faceDistance,
275 faceDistFromSurface;
276 G4ThreeVector faceNormal;
277 G4bool faceAllBehind;
278 if ((*face)->Intersect( p, v, false, kCarTolerance/2,
279 faceDistance, faceDistFromSurface,
280 faceNormal, faceAllBehind ) )
281 {
282 //
283 // Intersecting face
284 //
285 if (faceDistance < distance)
286 {
287 distance = faceDistance;
288 distFromSurface = faceDistFromSurface;
289 bestFace = *face;
290 if (distFromSurface <= 0) { return 0; }
291 }
292 }
293 } while( ++face < faces + numFace );
294
295 if (distance < kInfinity && distFromSurface<kCarTolerance/2)
296 {
297 if (bestFace->Distance(p,false) < kCarTolerance/2) { distance = 0; }
298 }
299
300 return distance;
301}

313//
314// DistanceToOut(p,v)
315//
316G4double G4VCSGfaceted::DistanceToOut( const G4ThreeVector &p,
317 const G4ThreeVector &v,
318 const G4bool calcNorm,
319 G4bool *validNorm,
320 G4ThreeVector *n ) const
321{
322 G4bool allBehind = true;
323 G4double distance = kInfinity;
324 G4double distFromSurface = kInfinity;
325 G4ThreeVector normal;
326
327 G4VCSGface **face = faces;
328 G4VCSGface *bestFace = *face;
329 do // Loop checking, 13.08.2015, G.Cosmo
330 {
331 G4double faceDistance,
332 faceDistFromSurface;
333 G4ThreeVector faceNormal;
334 G4bool faceAllBehind;
335 if ((*face)->Intersect( p, v, true, kCarTolerance/2,
336 faceDistance, faceDistFromSurface,
337 faceNormal, faceAllBehind ) )
338 {
339 //
340 // Intersecting face
341 //
342 if ( (distance < kInfinity) || (!faceAllBehind) ) { allBehind = false; }
343 if (faceDistance < distance)
344 {
345 distance = faceDistance;
346 distFromSurface = faceDistFromSurface;
347 normal = faceNormal;
348 bestFace = *face;
349 if (distFromSurface <= 0) { break; }
350 }
351 }
352 } while( ++face < faces + numFace );
353
354 if (distance < kInfinity)
355 {
356 if (distFromSurface <= 0)
357 {
358 distance = 0;
359 }
360 else if (distFromSurface<kCarTolerance/2)
361 {
362 if (bestFace->Distance(p,true) < kCarTolerance/2) { distance = 0; }
363 }
364
365 if (calcNorm)
366 {
367 *validNorm = allBehind;
368 *n = normal;
369 }
370 }
371 else
372 {
373 if (Inside(p) == kSurface) { distance = 0; }
374 if (calcNorm) { *validNorm = false; }
375 }
376
377 return distance;
378}

390//
391// DistanceTo
392//
393// Protected routine called by DistanceToIn and DistanceToOut
394//
395G4double G4VCSGfaceted::DistanceTo( const G4ThreeVector &p,
396 const G4bool outgoing ) const
397{
398 G4VCSGface **face = faces;
399 G4double best = kInfinity;
400 do // Loop checking, 13.08.2015, G.Cosmo
401 {
402 G4double distance = (*face)->Distance( p, outgoing );
403 if (distance < best) { best = distance; }
404 } while( ++face < faces + numFace );
405
406 return (best < 0.5*kCarTolerance) ? 0 : best;
407}

419//
420// GetExtent
421//
422// Define the sides of the box into which our solid instance would fit.
423//
424G4VisExtent G4VCSGfaceted::GetExtent() const
425{
426 static const G4ThreeVector xMax(1,0,0), xMin(-1,0,0),
427 yMax(0,1,0), yMin(0,-1,0),
428 zMax(0,0,1), zMin(0,0,-1);
429 static const G4ThreeVector *axes[6] =
430 { &xMin, &xMax, &yMin, &yMax, &zMin, &zMax };
431
432 G4double answers[6] =
433 {-kInfinity, -kInfinity, -kInfinity, -kInfinity, -kInfinity, -kInfinity};
434
435 G4VCSGface **face = faces;
436 do // Loop checking, 13.08.2015, G.Cosmo
437 {
438 const G4ThreeVector **axis = axes+5 ;
439 G4double *answer = answers+5;
440 do // Loop checking, 13.08.2015, G.Cosmo
441 {
442 G4double testFace = (*face)->Extent( **axis );
443 if (testFace > *answer) { *answer = testFace; }
444 }
445 while( --axis, --answer >= answers );
446
447 } while( ++face < faces + numFace );
448
449 return G4VisExtent( -answers[0], answers[1],
450 -answers[2], answers[3],
451 -answers[4], answers[5] );
452}

599//
600// GetPointOnSurfaceGeneric proportional to Areas of faces
601// in case of GenericPolycone or GenericPolyhedra
602//
603G4ThreeVector G4VCSGfaceted::GetPointOnSurfaceGeneric( ) const
604{
605 // Preparing variables
606 //
607 G4ThreeVector answer=G4ThreeVector(0.,0.,0.);
608 G4VCSGface **face = faces;
609 G4double area = 0;
610 G4int i;
611 std::vector<G4double> areas;
612
613 // First step: calculate surface areas
614 //
615 do // Loop checking, 13.08.2015, G.Cosmo
616 {
617 G4double result = (*face)->SurfaceArea( );
618 areas.push_back(result);
619 area=area+result;
620 } while( ++face < faces + numFace );
621
622 // Second Step: choose randomly one surface
623 //
624 G4VCSGface **face1 = faces;
625 G4double chose = area*G4UniformRand();
626 G4double Achose1, Achose2;
627 Achose1=0; Achose2=0.;
628 i=0;
629
630 do
631 {
632 Achose2+=areas[i];
633 if(chose>=Achose1 && chose<Achose2)
634 {
635 G4ThreeVector point;
636 point= (*face1)->GetPointOnFace();
637 return point;
638 }
639 i++;
640 Achose1=Achose2;
641 } while( ++face1 < faces + numFace );
642
643 return answer;
644}
