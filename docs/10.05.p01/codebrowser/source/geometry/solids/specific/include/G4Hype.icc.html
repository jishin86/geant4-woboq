66inline
67void G4Hype::SetInnerRadius (G4double newIRad)
68 {
69 innerRadius= newIRad;
70 innerRadius2= newIRad*newIRad;
71 endInnerRadius2=HypeInnerRadius2(halfLenZ);
72 endInnerRadius=std::sqrt(endInnerRadius2);
73 fCubicVolume = 0.;
74 fRebuildPolyhedron = true;
75 }

96inline
97void G4Hype::SetInnerStereo (G4double newISte)
98 {
99 innerStereo= std::fabs(newISte);
100 tanInnerStereo=std::tan(innerStereo);
101 tanInnerStereo2=tanInnerStereo*tanInnerStereo;
102 endInnerRadius2=HypeInnerRadius2(halfLenZ);
103 endInnerRadius=std::sqrt(endInnerRadius2);
104 fCubicVolume = 0.;
105 fRebuildPolyhedron = true;
106 }
107
108inline
109void G4Hype::SetOuterStereo (G4double newOSte)
110 {
111 outerStereo= std::fabs(newOSte);
112 tanOuterStereo=std::tan(outerStereo);
113 tanOuterStereo2=tanOuterStereo*tanOuterStereo;
114 endOuterRadius2=HypeOuterRadius2(halfLenZ);
115 endOuterRadius=std::sqrt(endOuterRadius2);
116 fCubicVolume = 0.;
117 fRebuildPolyhedron = true;
118 }

120inline
121G4bool G4Hype::InnerSurfaceExists() const
122 {
123 return (innerRadius > DBL_MIN) || (innerStereo != 0);
124 }
125
126inline
127G4double G4Hype::HypeInnerRadius2(G4double zVal) const
128 {
129 return (tanInnerStereo2*zVal*zVal+innerRadius2);
130 }
131
132inline
133G4double G4Hype::HypeOuterRadius2(G4double zVal) const
134 {
135 return (tanOuterStereo2*zVal*zVal+outerRadius2);
136 }
