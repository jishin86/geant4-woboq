158inline
159G4double G4TwistTubsHypeSide::GetRhoAtPZ(const G4ThreeVector &p,
160 G4bool isglobal) const
161{
162 // Get Rho at p.z() on Hyperbolic Surface.
163 G4ThreeVector tmpp;
164 if (isglobal) {
165 tmpp = fRot.inverse()*p - fTrans;
166 } else {
167 tmpp = p;
168 }
169 return std::sqrt(fR02 + tmpp.z() * tmpp.z() * fTan2Stereo);
170}
171
172inline
173G4ThreeVector G4TwistTubsHypeSide::
174SurfacePoint(G4double phi , G4double z , G4bool isGlobal)
175{
176 G4double rho = std::sqrt(fR02 + z * z * fTan2Stereo) ;
177
178 G4ThreeVector SurfPoint (rho*std::cos(phi), rho*std::sin(phi), z) ;
179
180 if (isGlobal) { return (fRot * SurfPoint + fTrans); }
181 return SurfPoint;
182}
183
184inline
185G4double G4TwistTubsHypeSide::GetBoundaryMin(G4double z)
186{
187 G4ThreeVector ptmp(0,0,z) ; // temporary point with z Komponent only
188 G4ThreeVector lowerlimit; // lower phi-boundary limit at z = ptmp.z()
189 lowerlimit = GetBoundaryAtPZ(sAxis0 & sAxisMin, ptmp);
190 return std::atan2( lowerlimit.y(), lowerlimit.x() ) ;
191}
192
193inline
194G4double G4TwistTubsHypeSide::GetBoundaryMax(G4double z )
195{
196 G4ThreeVector ptmp(0,0,z) ; // temporary point with z Komponent only
197 G4ThreeVector upperlimit; // upper phi-boundary limit at z = ptmp.z()
198 upperlimit = GetBoundaryAtPZ(sAxis0 & sAxisMax, ptmp);
199 return std::atan2( upperlimit.y(), upperlimit.x() ) ;
200}
201
202inline
203G4double G4TwistTubsHypeSide::GetSurfaceArea()
204{
205 // approximation with tube surface
206
207 return ( fAxisMax[1] - fAxisMin[1] ) * fR0 * fDPhi ;
208}
