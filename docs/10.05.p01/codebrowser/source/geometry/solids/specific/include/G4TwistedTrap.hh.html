33// G4TwistedTrap
34//
35// Class description:
36//
37// A G4TwistedTrap is a general twisted trapezoid: The faces perpendicular to the
38// z planes are trapezia, and their centres are not necessarily on
39// a line parallel to the z axis.
40//
41// pDz Half-length along the z-axis
42// pTheta Polar angle of the line joining the centres of the faces
43// at -/+pDz
44// pPhi Azimuthal angle of the line joing the centre of the face at
45// -pDz to the centre of the face at +pDz
46// pDy1 Half-length along y of the face at -pDz
47// pDx1 Half-length along x of the side at y=-pDy1 of the face at -pDz
48// pDx2 Half-length along x of the side at y=+pDy1 of the face at -pDz
49//
50// pDy2 Half-length along y of the face at +pDz
51// pDx3 Half-length along x of the side at y=-pDy2 of the face at +pDz
52// pDx4 Half-length along x of the side at y=+pDy2 of the face at +pDz
53// pAlph Angle with respect to the y axis from the centre of the side
54//
55//
56// A special regular case of a trapezoid with equal endcaps is available,
57// with polar,azimuthal and tilt angles set to zero.
58//
59
60// Author:
61//
62// 27-Oct-2004 - O.Link (Oliver.Link@cern.ch)
63//
64// --------------------------------------------------------------------

68#include "G4VTwistedFaceted.hh"

70class G4TwistedTrap : public G4VTwistedFaceted
71{
72 public: // with description
73
74 G4TwistedTrap(const G4String &pName,
75 G4double pPhiTwist,
76 G4double pDx1, // half x length at -pDz,-pDy
77 G4double pDx2, // half x length at -pDz,+pDy
78 G4double pDy,
79 G4double pDz);
80
81
82 G4TwistedTrap(const G4String &pName, // Name of instance
83 G4double pPhiTwist, // twist angle
84 G4double pDz, // half z length
85 G4double pTheta, // direction between end planes
86 G4double pPhi, // defined by polar and azim. angles
87 G4double pDy1, // half y length at -pDz
88 G4double pDx1, // half x length at -pDz,-pDy
89 G4double pDx2, // half x length at -pDz,+pDy
90 G4double pDy2, // half y length at +pDz
91 G4double pDx3, // half x length at +pDz,-pDy
92 G4double pDx4, // half x length at +pDz,+pDy
93 G4double pAlph // tilt angle
94 );
95
96 virtual ~G4TwistedTrap();
97
98 // accessors
99
100 inline G4double GetY1HalfLength() const { return GetDy1() ; }
101 inline G4double GetX1HalfLength() const { return GetDx1() ; }
102 inline G4double GetX2HalfLength() const { return GetDx2() ; }
103 inline G4double GetY2HalfLength() const { return GetDy2() ; }
104 inline G4double GetX3HalfLength() const { return GetDx3() ; }
105 inline G4double GetX4HalfLength() const { return GetDx4() ; }
106 inline G4double GetZHalfLength() const { return GetDz() ; }
107 inline G4double GetPhiTwist() const { return GetTwistAngle() ; }
108 inline G4double GetTiltAngleAlpha() const { return GetAlpha() ; }
109 inline G4double GetPolarAngleTheta() const { return GetTheta() ; }
110 inline G4double GetAzimuthalAnglePhi() const { return GetPhi() ; }
111
112 G4GeometryType GetEntityType() const;
113
114 G4VSolid* Clone() const;
115
116 std::ostream &StreamInfo(std::ostream& os) const;
117
118 public: // without description
119
120 G4TwistedTrap(__void__&);
121 // Fake default constructor for usage restricted to direct object
122 // persistency for clients requiring preallocation of memory for
123 // persistifiable objects.
124
125 G4TwistedTrap(const G4TwistedTrap& rhs);
126 G4TwistedTrap& operator=(const G4TwistedTrap& rhs);
127 // Copy constructor and assignment operator.
128
129} ;

131#endif
