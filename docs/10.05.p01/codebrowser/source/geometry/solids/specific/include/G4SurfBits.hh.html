31// G4SurfBits
32//
33// Class description:
34//
35// This class provides a simple container of bits, to be used for
36// optimization of tessellated surfaces (G4TessellatedSolid).
37// Each bit can be set and tested via the functions SetBitNumber and
38// TestBitNumber.
39// The default value of all bits is false.
40// The size of the container is automatically extended when a bit
41// number is either set or tested. To reduce the memory size of the
42// container use the Compact function, this will discard the memory
43// occupied by the upper bits that are 0.
44//
45
46// History:
47// 19.10.12 Marek Gayer, created and adapted from original implementation
48// of Root's TBits class by P.Canal
49// --------------------------------------------------------------------
50

58class G4SurfBits
59{
60 public:
61
62 G4SurfBits(unsigned int nbits = 0);
63 G4SurfBits(const G4SurfBits&);
64 G4SurfBits& operator=(const G4SurfBits&);
65 ~G4SurfBits();
66
67 //----- Bit manipulation
68 void ResetAllBits(G4bool value=false); // if value=1 set all bits to 1
69 void ResetBitNumber(unsigned int bitnumber);
70 void SetBitNumber(unsigned int bitnumber, G4bool value = true);
71 G4bool TestBitNumber(unsigned int bitnumber) const;
72
73 //----- Accessors and operator
74 G4bool operator[](unsigned int bitnumber) const;
75
76 //----- Optimized setters
77 // Each of these will replace the contents of the receiver with the
78 // bitvector in the parameter array. The number of bits is changed
79 // to nbits. If nbits is smaller than fNBits, the receiver will NOT
80 // be compacted.
81 void set(unsigned int nbits, const char *array);
82 void set(unsigned int nbits, const G4int *array);
83
84 //----- Optimized getters
85 // Each of these will replace the contents of the parameter array with the
86 // bits in the receiver. The parameter array must be large enough to hold
87 // all of the bits in the receiver.
88 // Note on semantics: any bits in the parameter array that go beyond the
89 // number of the bits in the receiver will have an unspecified value. For
90 // example, if you call Get(Int*) with an array of one integer and the
91 // G4SurfBits object has less than 32 bits, then the remaining bits in the
92 // integer will have an unspecified value.
93 void Get(char *array) const;
94 void Get(G4int *array) const;
95
96 //----- Utilities
97 void Clear();
98 void Compact(); // Reduce the space used.
99
100 unsigned int GetNbits() const { return fNBits; }
101 unsigned int GetNbytes() const { return fNBytes; }
102
103 void Print() const; // to show the list of active bits
104 void Output(std::ostream &) const;
105
106 protected:
107
108 void ReserveBytes(unsigned int nbytes);
109
110 public:
111
112 unsigned char *fAllBits; // [fNBytes] array of UChars
113
114 protected:
115
116 unsigned int fNBits; // Highest bit set + 1
117 unsigned int fNBytes; // Number of UChars in fAllBits
118};

122inline void G4SurfBits::SetBitNumber(unsigned int bitnumber, G4bool value)
123{
124 // set bit number 'bitnumber' to be value
125 if (bitnumber >= fNBits) {
126 unsigned int new_size = (bitnumber/8) + 1;
127 if (new_size > fNBytes) {
128 if (new_size < 100 * 1024 * 1024)
129 new_size *= 2;
130 unsigned char *old_location = fAllBits;
131 fAllBits = new unsigned char[new_size];
132 std::memcpy(fAllBits,old_location,fNBytes);
133 std::memset(fAllBits+fNBytes ,0, new_size-fNBytes);
134 fNBytes = new_size;
135 delete [] old_location;
136 }
137 fNBits = bitnumber+1;
138 }
139 unsigned int loc = bitnumber/8;
140 unsigned char bit = bitnumber%8;
141 if (value)
142 fAllBits[loc] |= (1<<bit);
143 else
144 fAllBits[loc] &= (0xFF ^ (1<<bit));
145}
146
147inline G4bool G4SurfBits::TestBitNumber(unsigned int bitnumber) const
148{
149 // Return the current value of the bit
150
151 if (bitnumber >= fNBits) return false;
152 unsigned int loc = bitnumber/8;
153 unsigned char value = fAllBits[loc];
154 unsigned char bit = bitnumber%8;
155 G4bool result = (value & (1<<bit)) != 0;
156 return result;
157 // short: return 0 != (fAllBits[bitnumber/8] & (1<< (bitnumber%8)));
158}
159
160inline void G4SurfBits::ResetBitNumber(unsigned int bitnumber)
161{
162 SetBitNumber(bitnumber,false);
163}
164
165inline G4bool G4SurfBits::operator[](unsigned int bitnumber) const
166{
167 return TestBitNumber(bitnumber);
168}
