41//* DistanceToPlaneWithV ----------------------------------------------
42
43inline
44G4double G4VTwistSurface::DistanceToPlaneWithV(const G4ThreeVector &p,
45 const G4ThreeVector &v,
46 const G4ThreeVector &x0,
47 const G4ThreeVector &n0,
48 G4ThreeVector &xx)
49{
50 G4double q = n0 * v;
51 G4double t = kInfinity;
52 if (q) { t = (n0 * (x0 - p)) / q; }
53 xx = p + t * v;
54 return t;
55}

57//=====================================================================
58//* DistanceToPlane ---------------------------------------------------
59
60inline
61G4double G4VTwistSurface::DistanceToPlane(const G4ThreeVector &p,
62 const G4ThreeVector &x0,
63 const G4ThreeVector &n0,
64 G4ThreeVector &xx)
65{
66 // DistanceToPlane :
67 // Calculate distance to plane in local coordinate,
68 // then return distance and global intersection points.
69 //
70 // p - location of flying particle
71 // x0 - reference point of surface
72 // xx - a foot of perpendicular line from p to the plane
73 // t - distance from xx to p
74 // n - a unit normal of this plane from plane to p.
75 //
76 // equation of plane:
77 // n*(x - x0) = 0;
78 //
79 // vector to xx:
80 // xx = p - t*n
81 //
82 // where
83 // t = n * (p - x0) / std::fabs(n)
84 //
85 G4double t;
86 G4ThreeVector n = n0.unit();
87 t = n * (p - x0);
88 xx = p - t * n;
89 return t;
90}

92//=====================================================================
93//* DistanceToPlane ---------------------------------------------------
94
95inline
96G4double G4VTwistSurface::DistanceToPlane(const G4ThreeVector &p,
97 const G4ThreeVector &x0,
98 const G4ThreeVector &t1,
99 const G4ThreeVector &t2,
100 G4ThreeVector &xx,
101 G4ThreeVector &n)
102{
103 // DistanceToPlane :
104 // Calculate distance to plane in local coordinate,
105 // then return distance and global intersection points.
106 // t1 - 1st. vector lying on the plane
107 // t2 - 2nd. vector lying on the plane
108
109 n = (t1.cross(t2)).unit();
110 return DistanceToPlane(p, x0, n, xx);
111}

113//=====================================================================
114//* DistanceToLine ----------------------------------------------------
115
116inline
117G4double G4VTwistSurface::DistanceToLine(const G4ThreeVector &p,
118 const G4ThreeVector &x0,
119 const G4ThreeVector &d,
120 G4ThreeVector &xx)
121{
122 // DistanceToLine :
123 // Calculate distance to line,
124 // then return distance and global intersection points.
125 //
126 // p - location of flying particle
127 // x0 - reference point of line
128 // d - direction vector of line
129 // xx - a foot of perpendicular line from p to the plane
130 // t - distance from xx to p
131 //
132 // Equation
133 //
134 // distance^2 = |(xx - p)|^2
135 // with
136 // xx = x0 + t*d
137 //
138 // (d/dt)distance^2 = (d/dt)|((x0 + t*d) - p)|^2
139 // = 2*t*|d|^2 + 2*d*(x0 - p)
140 // = 0 // smallest distance
141 // then
142 // t = - d*(x0 - p) / |d|^2
143 //
144
145 G4double t;
146 G4ThreeVector dir = d.unit();
147 t = - dir * (x0 - p); // |dir|^2 = 1.
148 xx = x0 + t * dir;
149
150 G4ThreeVector dist = xx - p;
151 return dist.mag();
152}

186
187inline
188G4bool G4VTwistSurface::IsInside(G4int areacode, G4bool testbitmode) const
189{
190 if (areacode & sInside) {
191 if (testbitmode) {
192 return true;
193 } else {
194 if (!((areacode & sBoundary) || (areacode & sCorner))) return true;
195 }
196 }
197 return false;

232//=====================================================================
233//* GetAxisType -------------------------------------------------------
234
235inline
236G4int G4VTwistSurface::GetAxisType(G4int areacode, G4int whichaxis) const
237{
238 G4int axiscode = areacode & sAxisMask & whichaxis;
239
240 if (axiscode == (sAxisX & sAxis0) ||
241 axiscode == (sAxisX & sAxis1)) {
242 return sAxisX;
243 } else if (axiscode == (sAxisY & sAxis0) ||
244 axiscode == (sAxisY & sAxis1)) {
245 return sAxisY;
246 } else if (axiscode == (sAxisZ & sAxis0) ||
247 axiscode == (sAxisZ & sAxis1)) {
248 return sAxisZ;
249 } else if (axiscode == (sAxisRho & sAxis0) ||
250 axiscode == (sAxisRho & sAxis1)) {
251 return sAxisRho;
252 } else if (axiscode == (sAxisPhi & sAxis0) ||
253 axiscode == (sAxisPhi & sAxis1)) {
254 return sAxisPhi;
255 } else {
256 std::ostringstream message;
257 message << "Configuration not supported." << G4endl
258 << " areacode = " << areacode;
259 G4Exception("G4VTwistSurface::GetAxisType()","GeomSolids0001",
260 FatalException, message);
261 }
262 return 1;
263}

304inline
305void G4VTwistSurface::SetNeighbours(G4VTwistSurface* axis0min, G4VTwistSurface* axis1min,
306 G4VTwistSurface* axis0max, G4VTwistSurface* axis1max)
307{
308 fNeighbours[0] = axis0min;
309 fNeighbours[1] = axis1min;
310 fNeighbours[2] = axis0max;
311 fNeighbours[3] = axis1max;
312}
313
314//=====================================================================
315//* GetNeighbours -----------------------------------------------------
316
317inline
318G4int G4VTwistSurface::GetNeighbours(G4int areacode, G4VTwistSurface** surfaces)
319{
320
321 int sAxis0Min = sAxis0 & sAxisMin ;
322 int sAxis1Min = sAxis1 & sAxisMin ;
323 int sAxis0Max = sAxis0 & sAxisMax ;
324 int sAxis1Max = sAxis1 & sAxisMax ;
325
326 G4int i = 0;
327
328 if ( (areacode & sAxis0Min ) == sAxis0Min ) {
329 surfaces[i] = fNeighbours[0] ;
330 i++ ;
331 }
332
333 if ( ( areacode & sAxis1Min ) == sAxis1Min ) {
334 surfaces[i] = fNeighbours[1] ;
335 i++ ;
336 if ( i == 2 ) return i ;
337 }
338
339 if ( ( areacode & sAxis0Max ) == sAxis0Max ) {
340 surfaces[i] = fNeighbours[2] ;
341 i++ ;
342 if ( i == 2 ) return i ;
343 }
344
345 if ( ( areacode & sAxis1Max ) == sAxis1Max ) {
346 surfaces[i] = fNeighbours[3] ;
347 i++ ;
348 if ( i == 2 ) return i ;
349 }
350
351 return i ;
352}

354//=====================================================================
355//* GetCorner ---------------------------------------------------------
356
357inline
358G4ThreeVector G4VTwistSurface::GetCorner(G4int areacode) const
359{
360 if (!(areacode & sCorner)){
361 std::ostringstream message;
362 message << "Area code must represent corner." << G4endl
363 << " areacode = " << areacode;
364 G4Exception("G4VTwistSurface::GetCorner()","GeomSolids0002",
365 FatalException, message);
366 }
367
368 if ((areacode & sC0Min1Min) == sC0Min1Min) {
369 return fCorners[0];
370 } else if ((areacode & sC0Max1Min) == sC0Max1Min) {
371 return fCorners[1];
372 } else if ((areacode & sC0Max1Max) == sC0Max1Max) {
373 return fCorners[2];
374 } else if ((areacode & sC0Min1Max) == sC0Min1Max) {
375 return fCorners[3];
376 } else {
377 std::ostringstream message;
378 message << "Configuration not supported." << G4endl
379 << " areacode = " << areacode;
380 G4Exception("G4VTwistSurface::GetCorner()", "GeomSolids0001",
381 FatalException, message);
382 }
383 return fCorners[0];
384}
