33// G4PolyconeSide
34//
35// Class description:
36//
37// Class implmenting a face that represents one conical side
38// of a polycone:
39//
40// G4PolyconeSide( const G4PolyconeSideRZ *prevRZ,
41// const G4PolyconeSideRZ *tail,
42// const G4PolyconeSideRZ *head,
43// const G4PolyconeSideRZ *nextRZ,
44// G4double phiStart, G4double deltaPhi,
45// G4bool phiIsOpen, G4bool isAllBehind=false )
46//
47// Values for r1,z1 and r2,z2 should be specified in clockwise
48// order in (r,z).
49
50// Author:
51// David C. Williams (davidw@scipp.ucsc.edu)
52// --------------------------------------------------------------------
53
54#ifndef G4PolyconeSide_hh
55#define G4PolyconeSide_hh
56
57#include "G4VCSGface.hh"
58
59class G4IntersectingCone;
60
61struct G4PolyconeSideRZ
62{
63 G4double r, z; // start of vector
64};
65
66// ----------------------------------------------------------------------------
67// MT-specific utility code
68
69#include "G4GeomSplitter.hh"
70
71// The class G4PlSideData is introduced to encapsulate the
72// fields of the class G4PolyconeSide that may not be read-only.
73//
74class G4PlSideData
75{
76 public:
77
78 void initialize()
79 {
80 fPhi.first = G4ThreeVector(0,0,0);
81 fPhi.second= 0.0;
82 }
83
84 std::pair<G4ThreeVector, G4double> fPhi; // Cached value for phi
85};
86
87// The type G4PlSideManager is introduced to
88// encapsulate the methods used by both the master thread and
89// worker threads to allocate memory space for the fields encapsulated
90// by the class G4PlSideData.
91//
92typedef G4GeomSplitter<G4PlSideData> G4PlSideManager;

97class G4PolyconeSide : public G4VCSGface
98{
99 public:
100
101 G4PolyconeSide( const G4PolyconeSideRZ *prevRZ,
102 const G4PolyconeSideRZ *tail,
103 const G4PolyconeSideRZ *head,
104 const G4PolyconeSideRZ *nextRZ,
105 G4double phiStart, G4double deltaPhi,
106 G4bool phiIsOpen, G4bool isAllBehind=false );
107 virtual ~G4PolyconeSide();
108
109 G4PolyconeSide( const G4PolyconeSide &source );
110 G4PolyconeSide& operator=( const G4PolyconeSide &source );
111
112 G4bool Intersect( const G4ThreeVector &p, const G4ThreeVector &v,
113 G4bool outgoing, G4double surfTolerance,
114 G4double &distance, G4double &distFromSurface,
115 G4ThreeVector &normal, G4bool &isAllBehind );
116
117 G4double Distance( const G4ThreeVector &p, G4bool outgoing );
118
119 EInside Inside( const G4ThreeVector &p, G4double tolerance,
120 G4double *bestDistance );
121
122 G4ThreeVector Normal( const G4ThreeVector &p, G4double *bestDistance );
123
124 G4double Extent( const G4ThreeVector axis );
125
126 void CalculateExtent( const EAxis axis,
127 const G4VoxelLimits &voxelLimit,
128 const G4AffineTransform &tranform,
129 G4SolidExtentList &extentList );
130
131 G4VCSGface *Clone() { return new G4PolyconeSide( *this ); }
132
133 G4double SurfaceArea();
134 G4ThreeVector GetPointOnFace();
135
136 public: // without description
137
138 G4PolyconeSide(__void__&);
139 // Fake default constructor for usage restricted to direct object
140 // persistency for clients requiring preallocation of memory for
141 // persistifiable objects.
142
143 inline G4int GetInstanceID() const { return instanceID; }
144 // Returns the instance ID.
145
146 static const G4PlSideManager& GetSubInstanceManager();
147 // Returns the private data instance manager.
148
149 protected:
150
151 G4double DistanceAway( const G4ThreeVector &p, G4bool opposite,
152 G4double &distOutside2, G4double *rzNorm=0 );
153
154 G4double DistanceAway( const G4ThreeVector &p,
155 G4double &distOutside2, G4double *edgeRZnorm );
156
157 G4bool PointOnCone( const G4ThreeVector &hit, G4double normSign,
158 const G4ThreeVector &p,
159 const G4ThreeVector &v, G4ThreeVector &normal );
160
161 void CopyStuff( const G4PolyconeSide &source );
162
163 static void FindLineIntersect( G4double x1, G4double y1,
164 G4double tx1, G4double ty1,
165 G4double x2, G4double y2,
166 G4double tx2, G4double ty2,
167 G4double &x, G4double &y );
168
169 G4double GetPhi( const G4ThreeVector& p );
170
171 protected:
172
173 G4double r[2], z[2]; // r, z parameters, in specified order
174 G4double startPhi, // Start phi (0 to 2pi), if phiIsOpen
175 deltaPhi; // Delta phi (0 to 2pi), if phiIsOpen
176 G4bool phiIsOpen; // True if there is a phi slice
177 G4bool allBehind; // True if the entire solid is "behind" this face
178
179 G4IntersectingCone *cone; // Our intersecting utility class
180
181 G4double rNorm, zNorm; // Normal to surface in r,z space
182 G4double rS, zS; // Unit vector along surface in r,z space
183 G4double length; // Length of face in r,z space
184 G4double prevRS,
185 prevZS; // Unit vector along previous polyconeSide
186 G4double nextRS,
187 nextZS; // Unit vector along next polyconeSide
188
189 G4double rNormEdge[2],
190 zNormEdge[2]; // Normal to edges
191
192 G4int ncorners;
193 G4ThreeVector *corners; // The coordinates of the corners (if phiIsOpen)
194
195 private:
196
197 G4double kCarTolerance; // Geometrical surface thickness
198 G4double fSurfaceArea; // Used for surface calculation
199
200 G4int instanceID;
201 // This field is used as instance ID.
202 G4GEOM_DLL static G4PlSideManager subInstanceManager;
203 // This field helps to use the class G4PlSideManager introduced above.
204};
