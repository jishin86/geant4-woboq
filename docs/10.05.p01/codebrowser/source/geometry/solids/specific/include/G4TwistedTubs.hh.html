33// G4TwistedTubs
34//
35// Class description:
36//
37// G4TwistedTubs is a sort of twisted cylinder.
38// A twisted cylinder which is placed along with z-axis and is
39// separated into phi-segments should become a hyperboloid, and
40// its each segmented piece should be tilted with a stereo angle.
41// G4TwistedTubs is a G4VSolid.
42// It can have inner & outer surfaces as well as G4TwistedTubs,
43// but cannot has different stereo angles between the inner surface
44// and outer surface.
45
46// Author:
47// 01-Aug-2002 - Kotoyo Hoshina (hoshina@hepburn.s.chiba-u.ac.jp)
48//
49// History:
50// 13-Nov-2003 - O.Link (Oliver.Link@cern.ch), Integration in Geant4
51// from original version in Jupiter-2.5.02 application.
52// --------------------------------------------------------------------

352inline
353void G4TwistedTubs::SetFields(G4double phitwist, G4double innerrad,
354 G4double outerrad, G4double negativeEndz,
355 G4double positiveEndz)
356{
357 fCubicVolume = 0.;
358 fPhiTwist = phitwist;
359 fEndZ[0] = negativeEndz;
360 fEndZ[1] = positiveEndz;
361 fEndZ2[0] = fEndZ[0] * fEndZ[0];
362 fEndZ2[1] = fEndZ[1] * fEndZ[1];
363 fInnerRadius = innerrad;
364 fOuterRadius = outerrad;
365 fInnerRadius2 = fInnerRadius * fInnerRadius;
366 fOuterRadius2 = fOuterRadius * fOuterRadius;
367
368 if (std::fabs(fEndZ[0]) >= std::fabs(fEndZ[1])) {
369 fZHalfLength = std::fabs(fEndZ[0]);
370 } else {
371 fZHalfLength = std::fabs(fEndZ[1]);
372 }
373
374 G4double parity = (fPhiTwist > 0 ? 1 : -1);
375 G4double tanHalfTwist = std::tan(0.5 * fPhiTwist);
376 G4double innerNumerator = std::fabs(fInnerRadius * tanHalfTwist) * parity;
377 G4double outerNumerator = std::fabs(fOuterRadius * tanHalfTwist) * parity;
378
379 fTanInnerStereo = innerNumerator / fZHalfLength;
380 fTanOuterStereo = outerNumerator / fZHalfLength;
381 fTanInnerStereo2 = fTanInnerStereo * fTanInnerStereo;
382 fTanOuterStereo2 = fTanOuterStereo * fTanOuterStereo;
383 fInnerStereo = std::atan2(innerNumerator, fZHalfLength);
384 fOuterStereo = std::atan2(outerNumerator, fZHalfLength);
385 fEndInnerRadius[0] = std::sqrt(fInnerRadius2 + fEndZ2[0] * fTanInnerStereo2);
386 fEndInnerRadius[1] = std::sqrt(fInnerRadius2 + fEndZ2[1] * fTanInnerStereo2);
387 fEndOuterRadius[0] = std::sqrt(fOuterRadius2 + fEndZ2[0] * fTanOuterStereo2);
388 fEndOuterRadius[1] = std::sqrt(fOuterRadius2 + fEndZ2[1] * fTanOuterStereo2);
389
390 fKappa = tanHalfTwist / fZHalfLength;
391 fEndPhi[0] = std::atan2(fEndZ[0] * tanHalfTwist, fZHalfLength);
392 fEndPhi[1] = std::atan2(fEndZ[1] * tanHalfTwist, fZHalfLength);
393
394#ifdef G4TWISTDEBUG
395 G4cout << "/********* G4TwistedTubs::SetFields() Field Parameters ***************** " << G4endl;
396 G4cout << "/* fPhiTwist : " << fPhiTwist << G4endl;
397 G4cout << "/* fEndZ(0, 1) : " << fEndZ[0] << " , " << fEndZ[1] << G4endl;
398 G4cout << "/* fEndPhi(0, 1) : " << fEndPhi[0] << " , " << fEndPhi[1] << G4endl;
399 G4cout << "/* fInnerRadius, fOuterRadius : " << fInnerRadius << " , " << fOuterRadius << G4endl;
400 G4cout << "/* fEndInnerRadius(0, 1) : " << fEndInnerRadius[0] << " , "
401 << fEndInnerRadius[1] << G4endl;
402 G4cout << "/* fEndOuterRadius(0, 1) : " << fEndOuterRadius[0] << " , "
403 << fEndOuterRadius[1] << G4endl;
404 G4cout << "/* fInnerStereo, fOuterStereo : " << fInnerStereo << " , " << fOuterStereo << G4endl;
405 G4cout << "/* tanHalfTwist, fKappa : " << tanHalfTwist << " , " << fKappa << G4endl;
406 G4cout << "/*********************************************************************** " << G4endl;
407#endif
408}
