160inline
161G4double G4TwistTrapAlphaSide::GetValueA(G4double phi)
162{
163 return ( fDx4plus2 + fDx4minus2 * ( 2 * phi ) / fPhiTwist ) ;
164}
165
166inline
167G4double G4TwistTrapAlphaSide::GetValueD(G4double phi)
168{
169 return ( fDx3plus1 + fDx3minus1 * ( 2 * phi) / fPhiTwist ) ;
170}
171
172inline
173G4double G4TwistTrapAlphaSide::GetValueB(G4double phi)
174{
175 return ( fDy2plus1 + fDy2minus1 * ( 2 * phi ) / fPhiTwist ) ;
176}
177
178
179inline
180G4double G4TwistTrapAlphaSide::Xcoef(G4double u, G4double phi)
181{
182
183 return GetValueA(phi)/2. + (GetValueD(phi)-GetValueA(phi))/4.
184 - u*( ( GetValueD(phi)-GetValueA(phi) )/( 2 * GetValueB(phi) ) - fTAlph );
185
186}
187
188inline G4ThreeVector
189G4TwistTrapAlphaSide::SurfacePoint(G4double phi, G4double u , G4bool isGlobal)
190{
191 // function to calculate a point on the surface, given by parameters phi,u
192
193 G4ThreeVector SurfPoint ( Xcoef(u,phi) * std::cos(phi)
194 - u * std::sin(phi) + fdeltaX*phi/fPhiTwist,
195 Xcoef(u,phi) * std::sin(phi)
196 + u * std::cos(phi) + fdeltaY*phi/fPhiTwist,
197 2*fDz*phi/fPhiTwist );
198 if (isGlobal) { return (fRot * SurfPoint + fTrans); }
199 return SurfPoint;
200}

202inline
203G4double G4TwistTrapAlphaSide::GetBoundaryMin(G4double phi)
204{
205 return -0.5*GetValueB(phi) ;
206}
207
208inline
209G4double G4TwistTrapAlphaSide::GetBoundaryMax(G4double phi)
210{
211 return 0.5*GetValueB(phi) ;
212}
213
214inline
215G4double G4TwistTrapAlphaSide::GetSurfaceArea()
216{
217 return (fDz*(std::sqrt(16*fDy1*fDy1
218 + (fa1md1 + 4*fDy1*fTAlph)*(fa1md1 + 4*fDy1*fTAlph))
219 + std::sqrt(16*fDy2*fDy2 + (fa2md2 + 4*fDy2*fTAlph)
220 * (fa2md2 + 4*fDy2*fTAlph))))/2. ;
221}
222
223inline
224G4ThreeVector G4TwistTrapAlphaSide::NormAng( G4double phi, G4double u )
225{
226 // function to calculate the norm at a given point on the surface
227 // replace a1-d1
228
229 G4ThreeVector nvec ( fDy1* fDz*(4*fDy1*std::cos(phi)
230 + (fa1md1 + 4*fDy1*fTAlph)*std::sin(phi)),
231 -(fDy1* fDz*((fa1md1 + 4*fDy1*fTAlph)*std::cos(phi)
232 - 4*fDy1*std::sin(phi))),
233 (fDy1*(-8*(fDx3minus1 + fDx4minus2)*fDy1
234 + fa1md1*(fDx2 + fDx3plus1 + fDx4)*fPhiTwist
235 + 4*(fDx2 + fDx3plus1 + fDx4)*fDy1*fPhiTwist
236 *fTAlph + 2*(fDx3minus1 + fDx4minus2)
237 *(fa1md1 + 4*fDy1*fTAlph)*phi)
238 + fPhiTwist*(16*fDy1*fDy1
239 + (fa1md1 + 4*fDy1*fTAlph)
240 *(fa1md1 + 4*fDy1*fTAlph))*u
241 + 4*fDy1*(fa1md1*fdeltaY - 4*fdeltaX*fDy1
242 + 4*fdeltaY*fDy1*fTAlph)* std::cos(phi)
243 - 4*fDy1*(fa1md1*fdeltaX + 4*fDy1*(fdeltaY
244 + fdeltaX*fTAlph))*std::sin(phi))/ 8. ) ;
245 return nvec.unit();
246}
