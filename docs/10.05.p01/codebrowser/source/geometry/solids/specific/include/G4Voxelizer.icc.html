37template <typename T> inline
38G4int G4Voxelizer::BinarySearch(const std::vector<T> &vec, T value)
39{
40 typename std::vector<T>::const_iterator begin=vec.begin(), end=vec.end();
41 G4int res = std::upper_bound(begin, end, value) - begin - 1;
42 return res;
43}

57inline
58void G4Voxelizer::GetVoxel(std::vector<G4int> &curVoxel,
59 const G4ThreeVector &point) const
60{
61 for (G4int i=0; i<=2; ++i)
62 {
63 const std::vector<double> &boundary = GetBoundary(i);
64 G4int n = BinarySearch(boundary, point[i]);
65 if (n == -1)
66 { n = 0; }
67 else if (n == G4int(boundary.size()) - 1)
68 { n--; }
69 curVoxel[i] = n;
70 }
71}

79inline
80G4int G4Voxelizer::GetVoxelsIndex(G4int x, G4int y, G4int z) const
81{
82 if (x < 0 || y < 0 || z < 0) { return -1; }
83 G4int maxX = fBoundaries[0].size();
84 G4int maxY = fBoundaries[1].size();
85 G4int index = x + y*maxX + z*maxX*maxY;
86
87 return index;
88}

94}

96inline
97G4bool G4Voxelizer::GetPointVoxel(const G4ThreeVector& p,
98 std::vector<G4int>& voxels) const
99{
100 for (G4int i = 0; i <= 2; ++i)
101 {
102 if (p[i] < *fBoundaries[i].begin() || p[i] > *fBoundaries[i].end())
103 {
104 return false;
105 }
106 }
107 for (G4int i = 0; i <= 2; ++i)
108 {
109 voxels[i] = BinarySearch(fBoundaries[i], p[i]);
110 }
111
112 return true;
113}

115inline
116G4int G4Voxelizer::GetPointIndex(const G4ThreeVector &p) const
117{
118 G4int maxX = fBoundaries[0].size();
119 G4int maxY = fBoundaries[1].size();
120 G4int x = BinarySearch(fBoundaries[0], p[0]);
121 G4int y = BinarySearch(fBoundaries[1], p[1]);
122 G4int z = BinarySearch(fBoundaries[2], p[2]);
123 G4int index = x + y*maxX + z*maxX*maxY;
124
125 return index;
126}

153inline
154long long G4Voxelizer::CountVoxels(std::vector<G4double> boundaries[]) const
155{
156 long long sx = boundaries[0].size() - 1;
157 long long sy = boundaries[1].size() - 1;
158 long long sz = boundaries[2].size() - 1;
159
160 return sx * sy *sz;
161}

163inline
164const std::vector<G4int>&
165G4Voxelizer::GetCandidates(std::vector<G4int> &curVoxel) const
166{
167 G4int voxelsIndex = GetVoxelsIndex(curVoxel);
168
169 if (voxelsIndex >= 0 && !fEmpty[voxelsIndex])
170 {
171 return fCandidates[voxelsIndex];
172 }
173 return fNoCandidates;
174}

201inline
202G4ThreeVector
203G4Voxelizer::GetGlobalPoint(const G4Transform3D& trans,
204 const G4ThreeVector& local) const
205{
206 // Returns global point coordinates converted from the local frame defined
207 // by the transformation. This is defined by multiplying this transformation
208 // with the local vector.
209
210 return trans*G4Point3D(local);
211}
