33// G4PolyPhiFace
34//
35// Class description:
36//
37// Definition of a face that bounds a polycone or polyhedra when
38// it has a phi opening:
39//
40// G4PolyPhiFace( const G4ReduciblePolygon *rz,
41// G4double phi,
42// G4double deltaPhi,
43// G4double phiOther )
44//
45// Specifically: a face that lies on a plane that passes through
46// the z axis. It has boundaries that are straight lines of arbitrary
47// length and direction, but with corners aways on the same side of
48// the z axis.
49
50// Author:
51// David C. Williams (davidw@scipp.ucsc.edu)
52// --------------------------------------------------------------------
53
54#ifndef G4PolyPhiFace_hh
55#define G4PolyPhiFace_hh

60class G4ReduciblePolygon;
61
62struct G4PolyPhiFaceVertex
63{
64 G4double x, y, r, z; // position
65 G4double rNorm,
66 zNorm; // r/z normal
67 G4ThreeVector norm3D; // 3D normal
68
69 // Needed for Triangulation Algorithm
70 //
71 G4bool ear;
72 G4PolyPhiFaceVertex *next,*prev;
73};
74
75struct G4PolyPhiFaceEdge
76{
77 G4PolyPhiFaceEdge(): v0(0), v1(0), tr(.0), tz(0.), length(0.) {}
78 G4PolyPhiFaceVertex *v0, *v1; // Corners
79 G4double tr, tz, // Unit vector along edge
80 length; // Length of edge
81 G4ThreeVector norm3D; // 3D edge normal vector
82};
