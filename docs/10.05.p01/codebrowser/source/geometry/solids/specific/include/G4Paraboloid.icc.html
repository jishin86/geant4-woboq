55inline
56void G4Paraboloid::SetZHalfLength(G4double pDz)
57{
58 if(pDz <= 0)
59 {
60 G4Exception("G4Paraboloid::SetZHalfLength()", "GeomSolids0002",
61 FatalException, "Invalid dimensions.");
62 }
63 else
64 {
65 dz = pDz;
66 k1 = (sqr(r2) - sqr(r1)) / (2 * dz);
67 k2 = (sqr(r2) + sqr(r1)) / 2;
68
69 // This informs GetSurfaceArea() and GetCubicVolume() that it needs
70 // to recalculate buffered value.
71 //
72 fSurfaceArea = 0.;
73 fCubicVolume = 0.;
74 fRebuildPolyhedron = true;
75 }
76}

78inline
79void G4Paraboloid::SetRadiusPlusZ(G4double pR2)
80{
81 if(pR2 <= 0 || pR2 <= r1)
82 {
83 G4Exception("G4Paraboloid::SetRadiusPlusZ()", "GeomSolids0002",
84 FatalException, "Invalid dimensions.");
85 }
86 else
87 {
88 r2 = pR2;
89 k1 = (sqr(r2) - sqr(r1)) / (2 * dz);
90 k2 = (sqr(r2) + sqr(r1)) / 2;
91
92 // This informs GetSurfaceArea() and GetCubicVolume() that it needs
93 // to recalculate buffered value.
94 //
95 fSurfaceArea = 0.;
96 fCubicVolume = 0.;
97 fRebuildPolyhedron = true;
98 }
99}

102void G4Paraboloid::SetRadiusMinusZ(G4double pR1)
103{

106 G4Exception("G4Paraboloid::SetRadiusMinusZ()", "GeomSolids0002",
107 FatalException, "Invalid dimensions.");

109 else
110 {
111 r1 = pR1;

124inline
125G4double G4Paraboloid::GetCubicVolume()
126{
127 if(fCubicVolume != 0 ) {;}
128 else
129 {
130 fCubicVolume = CLHEP::twopi * k2 * dz;
131 }
132 return fCubicVolume;
133}

136inline
137G4double G4Paraboloid::CalculateSurfaceArea() const
138{
139 G4double h1, h2, A1, A2;
140
141 h1 = k2/k1 + dz;
142 h2 = k2/k1 - dz;
143
144 // Calculate surface area for the paraboloid full paraboloid
145 // cutoff at z = dz (not the cutoff area though).
146
147 A1 = sqr(r2) + 4 * sqr(h1);
148 A1 *= sqr(A1); // Sets A1 = A1^3
149 A1 = CLHEP::pi * r2 /6 / sqr(h1) * ( std::sqrt(A1) - r2 * r2 * r2);
150
151 // Calculate surface area for the paraboloid full paraboloid
152 // cutoff at z = -dz (not the cutoff area though).
153
154 A2 = sqr(r1) + 4 * sqr(h2);
155 A2 *= sqr(A2);// Sets A2 = A2^3
156
157 if(h2 != 0)
158 { A2 = CLHEP::pi * r1 /6 / sqr(h2) * ( std::sqrt(A2) - r1 * r1 * r1); }
159 else
160 { A2 = 0.; }
161
162 return fSurfaceArea = A1 - A2 + (sqr(r1) + sqr(r2))*CLHEP::pi;
163}
164
165inline
166G4double G4Paraboloid::GetSurfaceArea()
167{
168 if(fSurfaceArea == 0.) CalculateSurfaceArea();
169
170 return fSurfaceArea;
171}
