33// G4Polycone
34//
35// Class description:
36//
37// Class implementing a CSG-like type "PCON" Geant 3.21 volume,
38// inherited from class G4VCSGfaceted:
39//
40// G4Polycone( const G4String& name,
41// G4double phiStart, // initial phi starting angle
42// G4double phiTotal, // total phi angle
43// G4int numZPlanes, // number of z planes
44// const G4double zPlane[], // position of z planes
45// const G4double rInner[], // tangent distance to inner surface
46// const G4double rOuter[]) // tangent distance to outer surface
47//
48// Alternative constructor, but limited to increasing-only Z sections:
49//
50// G4Polycone( const G4String& name,
51// G4double phiStart, // initial phi starting angle
52// G4double phiTotal, // total phi angle
53// G4int numRZ, // number corners in r,z space
54// const G4double r[], // r coordinate of these corners
55// const G4double z[]) // z coordinate of these corners
56
57// Author:
58// David C. Williams (davidw@scipp.ucsc.edu)
59// --------------------------------------------------------------------
