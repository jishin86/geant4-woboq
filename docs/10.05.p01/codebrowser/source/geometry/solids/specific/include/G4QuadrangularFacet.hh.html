30// Class G4QuadrangularFacet
31//
32// Class description:
33//
34// The G4QuadrangularFacet class is used for the contruction of
35// G4TessellatedSolid.
36// It is defined by four fVertices, which shall be in the same plane and be
37// supplied in anti-clockwise order looking from the outsider of the solid
38// where it belongs. Its constructor
39//
40// G4QuadrangularFacet (const G4ThreeVector Pt0, const G4ThreeVector vt1,
41// const G4ThreeVector vt2, const G4ThreeVector vt3,
42// G4FacetVertexType);
43//
44// takes 5 parameters to define the four fVertices:
45// 1) G4FacetvertexType = "ABSOLUTE": in this case Pt0, vt1, vt2 and vt3
46// are the four fVertices required in anti-clockwise order when looking
47// from the outsider.
48// 2) G4FacetvertexType = "RELATIVE": in this case the first vertex is Pt0,
49// the second vertex is Pt0+vt, the third vertex is Pt0+vt2 and
50// the fourth vertex is Pt0+vt3, in anti-clockwise order when looking
51// from the outsider.
52

121///////////////////////////////////////////////////////////////////////////////
122// Inlined Methods
123///////////////////////////////////////////////////////////////////////////////
124
125inline G4int G4QuadrangularFacet::GetNumberOfVertices () const
126{
127 return 4;
128}
129
130inline G4ThreeVector G4QuadrangularFacet::GetVertex (G4int i) const
131{
132 return i == 3 ? fFacet2.GetVertex(2) : fFacet1.GetVertex(i);
133}
134
135
136inline G4double G4QuadrangularFacet::GetRadius () const
137{
138 return fRadius;
139}
140
141inline G4ThreeVector G4QuadrangularFacet::GetCircumcentre () const
142{
143 return fCircumcentre;
144}
145
146inline void G4QuadrangularFacet::SetVertex (G4int i, const G4ThreeVector &val)
147{
148 switch (i)
149 {
150 case 0:
151 fFacet1.SetVertex(0, val);
152 fFacet2.SetVertex(0, val);
153 break;
154 case 1:
155 fFacet1.SetVertex(1, val);
156 break;
157 case 2:
158 fFacet1.SetVertex(2, val);
159 fFacet2.SetVertex(1, val);
160 break;
161 case 3:
162 fFacet2.SetVertex(2, val);
163 break;
164 }
165}
166
167inline void G4QuadrangularFacet::SetVertices(std::vector<G4ThreeVector> *v)
168{
169 fFacet1.SetVertices(v);
170 fFacet2.SetVertices(v);
171}
172
173inline G4bool G4QuadrangularFacet::IsDefined () const
174{
175 return fFacet1.IsDefined();
176}
177
178inline G4int G4QuadrangularFacet::GetVertexIndex (G4int i) const
179{
180 return i == 3 ? fFacet2.GetVertexIndex(2) : fFacet1.GetVertexIndex(i);
181}
182
183
184inline void G4QuadrangularFacet::SetVertexIndex (G4int i, G4int val)
185{
186 switch (i)
187 {
188 case 0:
189 fFacet1.SetVertexIndex(0, val);
190 fFacet2.SetVertexIndex(0, val);
191 break;
192 case 1:
193 fFacet1.SetVertexIndex(1, val);
194 break;
195 case 2:
196 fFacet1.SetVertexIndex(2, val);
197 fFacet2.SetVertexIndex(1, val);
198 break;
199 case 3:
200 fFacet2.SetVertexIndex(2, val);
201 break;
202 }
203}
204
205inline G4int G4QuadrangularFacet::AllocatedMemory()
206{
207 return sizeof(*this) + fFacet1.AllocatedMemory() + fFacet2.AllocatedMemory();
208}
