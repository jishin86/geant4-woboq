37inline
38G4double G4Ellipsoid::GetSemiAxisMax (G4int i) const
39{
40 return (i==0) ? xSemiAxis
41 : (i==1) ? ySemiAxis
42 : zSemiAxis;
43}
44
45inline
46G4double G4Ellipsoid::GetZBottomCut() const
47{
48 return zBottomCut;
49}
50
51inline
52G4double G4Ellipsoid::GetZTopCut() const
53{
54 return zTopCut;
55}

57inline
58void G4Ellipsoid::SetSemiAxis (G4double newxSemiAxis,
59 G4double newySemiAxis,
60 G4double newzSemiAxis)
61{
62 xSemiAxis= newxSemiAxis; ySemiAxis= newySemiAxis; zSemiAxis= newzSemiAxis;
63 semiAxisMax = xSemiAxis > ySemiAxis ? xSemiAxis : ySemiAxis;
64 if (zSemiAxis > semiAxisMax) { semiAxisMax= zSemiAxis; }
65 if (zBottomCut < -zSemiAxis) { zBottomCut = -zSemiAxis; }
66 if (zTopCut > +zSemiAxis) { zTopCut = +zSemiAxis; }
67 fRebuildPolyhedron = true;
68}
69
70inline
71void G4Ellipsoid::SetZCuts (G4double newzBottomCut, G4double newzTopCut)
72{
73 if (newzBottomCut < -zSemiAxis)
74 { zBottomCut = -zSemiAxis; }
75 else
76 { zBottomCut = newzBottomCut; }
77
78 if (newzTopCut > +zSemiAxis)
79 { zTopCut = +zSemiAxis; }
80 else
81 { zTopCut = newzTopCut; }
82 fRebuildPolyhedron = true;
83}

85inline
86G4double G4Ellipsoid::GetCubicVolume()
87{
88 if(fCubicVolume != 0 ) {;}
89 else
90 {
91 if ((zTopCut > +zSemiAxis && zBottomCut < -zSemiAxis)
92 || (zTopCut == 0 && zBottomCut == 0) )
93 {
94 fCubicVolume = (4./3.)*CLHEP::pi*xSemiAxis*ySemiAxis*zSemiAxis;
95 }
96 else
97 {
98 fCubicVolume = CLHEP::pi*xSemiAxis*ySemiAxis
99 * ((zTopCut-std::pow(zTopCut,3.)/(3.*sqr(zSemiAxis)))
100 - (zBottomCut-std::pow(zBottomCut,3.)/(3.*sqr(zSemiAxis))));
101 }
102 }
103 return fCubicVolume;
104}
