42#ifndef G4UPOLYHEDRA_HH
43#define G4UPOLYHEDRA_HH
44
45#include "G4UAdapter.hh"
46
47#if ( defined(G4GEOM_USE_USOLIDS) || defined(G4GEOM_USE_PARTIAL_USOLIDS) )
48
49#include <volumes/UnplacedPolyhedron.h>
50
51#include "G4TwoVector.hh"
52#include "G4PolyhedraSide.hh"
53#include "G4PolyhedraHistorical.hh"
54#include "G4Polyhedron.hh"
55
56class G4EnclosingCylinder;
57class G4ReduciblePolygon;
58
59class G4UPolyhedra : public G4UAdapter<vecgeom::UnplacedPolyhedron>
60{
61 using Shape_t = vecgeom::UnplacedPolyhedron;
62 using Base_t = G4UAdapter<vecgeom::UnplacedPolyhedron>;
63
64 public: // with description
65
66 G4UPolyhedra( const G4String& name,
67 G4double phiStart, // initial phi starting angle
68 G4double phiTotal, // total phi angle
69 G4int numSide, // number sides
70 G4int numZPlanes, // number of z planes
71 const G4double zPlane[], // position of z planes
72 const G4double rInner[], // tangent distance to inner surface
73 const G4double rOuter[] ); // tangent distance to outer surface
74
75 G4UPolyhedra( const G4String& name,
76 G4double phiStart, // initial phi starting angle
77 G4double phiTotal, // total phi angle
78 G4int numSide, // number sides
79 G4int numRZ, // number corners in r,z space
80 const G4double r[], // r coordinate of these corners
81 const G4double z[] ); // z coordinate of these corners
82
83 ~G4UPolyhedra();
84
85 void ComputeDimensions( G4VPVParameterisation* p,
86 const G4int n,
87 const G4VPhysicalVolume* pRep);
88
89 G4VSolid* Clone() const;
90
91 G4int GetNumSide() const;
92 G4double GetStartPhi() const;
93 G4double GetEndPhi() const;
94 G4double GetSinStartPhi() const;
95 G4double GetCosStartPhi() const;
96 G4double GetSinEndPhi() const;
97 G4double GetCosEndPhi() const;
98 G4bool IsOpen() const;
99 G4bool IsGeneric() const;
100 G4int GetNumRZCorner() const;
101 G4PolyhedraSideRZ GetCorner( const G4int index ) const;
102 G4PolyhedraHistorical* GetOriginalParameters() const;
103 void SetOriginalParameters(G4PolyhedraHistorical* pars);
104
105 G4bool Reset();
106
107 inline G4GeometryType GetEntityType() const;
108
109 public: // without description
110
111 G4UPolyhedra(__void__&);
112 // Fake default constructor for usage restricted to direct object
113 // persistency for clients requiring preallocation of memory for
114 // persistifiable objects.
115
116 G4UPolyhedra( const G4UPolyhedra &source );
117 G4UPolyhedra &operator=( const G4UPolyhedra &source );
118 // Copy constructor and assignment operator.
119
120 void BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const;
121
122 G4bool CalculateExtent(const EAxis pAxis,
123 const G4VoxelLimits& pVoxelLimit,
124 const G4AffineTransform& pTransform,
125 G4double& pMin, G4double& pMax) const;
126
127 G4Polyhedron* CreatePolyhedron() const;
128
129 protected:
130
131 void SetOriginalParameters();
132
133 G4bool fGenericPgon; // true if created through the 2nd generic constructor
134 G4PolyhedraHistorical fOriginalParameters; // original input parameters
135
136 private:
137
138 G4double wrStart;
139 G4double wrDelta;
140 G4int wrNumSide;
141 std::vector<G4TwoVector> rzcorners;
142};

144// --------------------------------------------------------------------
145// Inline methods

149{
150 return "G4Polyhedra";
151}
