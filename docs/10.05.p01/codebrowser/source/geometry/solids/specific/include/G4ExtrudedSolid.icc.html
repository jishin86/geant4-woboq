42inline G4TwoVector G4ExtrudedSolid::GetVertex(G4int index) const

83inline
84G4bool G4ExtrudedSolid::PointInPolygon(const G4ThreeVector& p) const
85{
86 G4bool in = 0;
87 G4int icur = (fPolygon[fNv-1].y() > p.y()), iprev = 0;
88 for (G4int i = 0; i < fNv; i++)
89 {
90 iprev = icur;
91 if ((icur = (fPolygon[i].y() > p.y())) != iprev)
92 {
93 in ^= (p.y()*fLines[i].k + fLines[i].m < p.x());
94 }
95 }
96 return in;
97}

99inline
100G4double G4ExtrudedSolid::DistanceToPolygonSqr(const G4ThreeVector& p) const
101{
102 G4double dd = DBL_MAX;
103 for (G4int i=0, k=fNv-1; i<fNv; k=i++)
104 {
105 G4double ix = p.x() - fPolygon[i].x();
106 G4double iy = p.y() - fPolygon[i].y();
107 G4double u = fPlanes[i].a*iy - fPlanes[i].b*ix;
108 if (u < 0)
109 {
110 G4double tmp = ix*ix + iy*iy;
111 if (tmp < dd) dd = tmp;
112 }
113 else if (u > fLengths[i])
114 {
115 G4double kx = p.x() - fPolygon[k].x();
116 G4double ky = p.y() - fPolygon[k].y();
117 G4double tmp = kx*kx + ky*ky;
118 if (tmp < dd) dd = tmp;
119 }
120 else
121 {
122 G4double tmp = fPlanes[i].a*p.x() + fPlanes[i].b*p.y() + fPlanes[i].d;
123 tmp *= tmp;
124 if (tmp < dd) dd = tmp;
125 }
126 }
127 return dd;
128}
