41#ifndef G4UTET_HH
42#define G4UTET_HH
43
44#include "G4UAdapter.hh"
45
46#if ( defined(G4GEOM_USE_USOLIDS) || defined(G4GEOM_USE_PARTIAL_USOLIDS) )
47
48#include <volumes/UnplacedTet.h>
49
50#include "G4Polyhedron.hh"
51
52class G4UTet : public G4UAdapter<vecgeom::UnplacedTet>
53{
54
55 using Shape_t = vecgeom::UnplacedTet;
56 using Base_t = G4UAdapter<vecgeom::UnplacedTet>;
57
58 public: // with description
59
60 G4UTet(const G4String& pName,
61 G4ThreeVector anchor,
62 G4ThreeVector p2,
63 G4ThreeVector p3,
64 G4ThreeVector p4,
65 G4bool *degeneracyFlag=0);
66
67 ~G4UTet();
68
69 inline G4GeometryType GetEntityType() const;
70
71 public: // without description
72
73 G4UTet(__void__&);
74 // Fake default constructor for usage restricted to direct object
75 // persistency for clients requiring preallocation of memory for
76 // persistifiable objects.
77
78 G4UTet(const G4UTet& rhs);
79 G4UTet& operator=(const G4UTet& rhs);
80 // Copy constructor and assignment operator.
81
82 void BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const;
83
84 G4bool CalculateExtent(const EAxis pAxis,
85 const G4VoxelLimits& pVoxelLimit,
86 const G4AffineTransform& pTransform,
87 G4double& pMin, G4double& pMax) const;
88
89 G4Polyhedron* CreatePolyhedron() const;
90
91 std::vector<G4ThreeVector> GetVertices() const;
92 // Return the four vertices of the shape.
93};
94
95// --------------------------------------------------------------------
96// Inline methods

98
99inline G4GeometryType G4UTet::GetEntityType() const
100{
101 return "G4Tet";
102}
