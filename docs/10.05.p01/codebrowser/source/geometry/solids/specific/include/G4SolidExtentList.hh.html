33// G4SolidExtentList
34//
35// Class description:
36//
37// Defines a list of (voxel) extents along one axis.
38//
39// This utility class is designed for one specific purpose:
40// to calculate the extent of a CSG-like solid for a voxel
41// (G4VSolid::CalculateExtent).
42
43// Author:
44// David C. Williams (davidw@scipp.ucsc.edu)
45// --------------------------------------------------------------------
46#ifndef G4SolidExtentList_hh
47#define G4SolidExtentList_hh
48
49#include "G4Types.hh"
50
51#include "G4ClippablePolygon.hh"
52
53class G4SolidExtentList
54{
55 public:
56
57 G4SolidExtentList();
58 G4SolidExtentList( const EAxis targetAxis,
59 const G4VoxelLimits &voxelLimits );
60 ~G4SolidExtentList();
61
62
63 void AddSurface( const G4ClippablePolygon &surface );
64
65 G4bool GetExtent( G4double &min, G4double &max ) const;
66
67 protected:
68
69 EAxis axis; // Target axis
70 G4bool limited; // True if limited
71 G4double minLimit; // ... min limit
72 G4double maxLimit; // ... max limit
73
74 G4ClippablePolygon minSurface, // Minimum surface within limits
75 maxSurface, // Maximum
76 minAbove, // Minimum surface totally above max limit
77 maxBelow; // Maximum surface totally below min limit
78};
