94///////////////////////////////////////////////////////////////
95//
96// Assignment operator
97//
98G4ScaledSolid& G4ScaledSolid::operator = (const G4ScaledSolid& rhs)
99{
100 // Check assignment to self
101 //
102 if (this == &rhs) { return *this; }
103
104 // Copy base class data
105 //
106 G4VSolid::operator=(rhs);
107
108 // Copy data
109 //
110 fPtrSolid = rhs.fPtrSolid;
111 delete fScale;
112 fScale = new G4ScaleTransform(*(rhs.fScale));
113 fCubicVolume = rhs.fCubicVolume;
114 fSurfaceArea = rhs.fSurfaceArea;
115 fRebuildPolyhedron = false;
116 delete fpPolyhedron; fpPolyhedron= 0;
117
118 return *this;
119}

130//////////////////////////////////////////////////////////////////////////
131//
132// Get bounding box
133
134void G4ScaledSolid::BoundingLimits(G4ThreeVector& pMin,
135 G4ThreeVector& pMax) const
136{
137 G4ThreeVector bmin,bmax;
138 G4ThreeVector scale = fScale->GetScale();
139
140 fPtrSolid->BoundingLimits(bmin,bmax);
141 pMin.set(bmin.x()*scale.x(),bmin.y()*scale.y(),bmin.z()*scale.z());
142 pMax.set(bmax.x()*scale.x(),bmax.y()*scale.y(),bmax.z()*scale.z());
143
144 // Check correctness of the bounding box
145 //
146 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
147 {
148 std::ostringstream message;
149 message << "Bad bounding box (min >= max) for solid: "
150 << GetName() << " !"
151 << "\npMin = " << pMin
152 << "\npMax = " << pMax;
153 G4Exception("G4ScaledSolid::BoundingLimits()", "GeomMgt0001",
154 JustWarning, message);
155 DumpInfo();
156 }
157}

159//////////////////////////////////////////////////////////////////////////
160//
161// Calculate extent under transform and specified limit
162//
163G4bool
164G4ScaledSolid::CalculateExtent( const EAxis pAxis,
165 const G4VoxelLimits& pVoxelLimit,
166 const G4AffineTransform& pTransform,
167 G4double& pMin,
168 G4double& pMax ) const
169{
170 // Find bounding box of unscaled solid
171 G4ThreeVector bmin,bmax;
172 fPtrSolid->BoundingLimits(bmin,bmax);
173
174 // Set combined transformation
175 G4Transform3D transform3D =
176 G4Transform3D(pTransform.NetRotation().inverse(),
177 pTransform.NetTranslation())*GetScaleTransform();
178
179 // Find extent
180 G4BoundingEnvelope bbox(bmin,bmax);
181 return bbox.CalculateExtent(pAxis,pVoxelLimit,transform3D,pMin,pMax);
182}

193//////////////////////////////////////////////////////////////
194//
195// SurfaceNormal
196//
197G4ThreeVector
198G4ScaledSolid::SurfaceNormal( const G4ThreeVector& p ) const
199{
200 // Transform point to unscaled shape frame
201 G4ThreeVector newPoint;
202 fScale->Transform(p, newPoint);
203
204 // Compute normal in unscaled frame
205 G4ThreeVector newNormal = fPtrSolid->SurfaceNormal(newPoint);
206 G4ThreeVector normal;
207
208 // Convert normal to scaled frame
209 fScale->InverseTransformNormal(newNormal, normal);
210 return normal/normal.mag();
211}

213/////////////////////////////////////////////////////////////
214//
215// The same algorithm as in DistanceToIn(p)
216//
217G4double
218G4ScaledSolid::DistanceToIn( const G4ThreeVector& p,
219 const G4ThreeVector& v ) const
220{
221 // Transform point and direction to unscaled shape frame
222 G4ThreeVector newPoint;
223 fScale->Transform(p, newPoint);
224
225 // Direction is un-normalized after scale transformation
226 G4ThreeVector newDirection;
227 fScale->Transform(v, newDirection);
228 newDirection = newDirection/newDirection.mag();
229
230 // Compute distance in unscaled system
231 G4double dist = fPtrSolid->DistanceToIn(newPoint,newDirection);
232
233 // Return converted distance to global
234 return fScale->InverseTransformDistance(dist, newDirection);
235}

237////////////////////////////////////////////////////////
238//
239// Approximate nearest distance from the point p to the solid from outside
240//
241G4double
242G4ScaledSolid::DistanceToIn( const G4ThreeVector& p ) const
243{
244 // Transform point to unscaled shape frame
245 G4ThreeVector newPoint;
246 fScale->Transform(p, newPoint);
247
248 // Compute unscaled safety, then scale it
249 G4double dist = fPtrSolid->DistanceToIn(newPoint);
250 return fScale->InverseTransformDistance(dist);
251}

253//////////////////////////////////////////////////////////
254//
255// The same algorithm as DistanceToOut(p)
256//
257G4double
258G4ScaledSolid::DistanceToOut( const G4ThreeVector& p,
259 const G4ThreeVector& v,
260 const G4bool calcNorm,
261 G4bool *validNorm,
262 G4ThreeVector *n ) const
263{
264 // Transform point and direction to unscaled shape frame
265 G4ThreeVector newPoint;
266 fScale->Transform(p, newPoint);
267
268 // Direction is un-normalized after scale transformation
269 G4ThreeVector newDirection;
270 fScale->Transform(v, newDirection);
271 newDirection = newDirection/newDirection.mag();
272
273 // Compute distance in unscaled system
274 G4ThreeVector solNorm;
275 G4double dist = fPtrSolid->DistanceToOut(newPoint,newDirection,
276 calcNorm,validNorm,&solNorm);
277 if(calcNorm)
278 {
279 G4ThreeVector normal;
280 fScale->TransformNormal(solNorm, normal);
281 *n = normal/normal.mag();
282 }
283
284 // Return distance converted to global
285 return fScale->InverseTransformDistance(dist, newDirection);
286}

358//////////////////////////////////////////////////////////////////////////
359//
360// Setting the scaling transformation
361//
362void G4ScaledSolid::SetScaleTransform(const G4Scale3D& scale)
363{
364 if (fScale) { delete fScale; }
365 fScale = new G4ScaleTransform(scale);
366 fRebuildPolyhedron = true;
367}

369//////////////////////////////////////////////////////////////////////////
370//
371// Get volume of the scaled solid
372//
373G4double G4ScaledSolid::GetCubicVolume()
374{
375 if(fCubicVolume < 0.)
376 {
377 fCubicVolume = fPtrSolid->GetCubicVolume() *
378 fScale->GetScale().x() *
379 fScale->GetScale().y() *
380 fScale->GetScale().z();
381 }
382 return fCubicVolume;
383}

432//////////////////////////////////////////////////////////////////////////
433//
434// CreatePolyhedron
435//
436G4Polyhedron*
437G4ScaledSolid::CreatePolyhedron () const
438{
439 G4Polyhedron* polyhedron = fPtrSolid->CreatePolyhedron();
440 if (polyhedron)
441 {
442 polyhedron->Transform(GetScaleTransform());
443 }
444 else
445 {
446 DumpInfo();
447 G4Exception("G4ScaledSolid::CreatePolyhedron()",
448 "GeomSolids2003", JustWarning,
449 "No G4Polyhedron for scaled solid");
450 }
451 return polyhedron;
452}

454//////////////////////////////////////////////////////////////////////////
455//
456// GetPolyhedron
457//
458G4Polyhedron* G4ScaledSolid::GetPolyhedron () const
459{
460 if (!fpPolyhedron ||
461 fRebuildPolyhedron ||
462 fpPolyhedron->GetNumberOfRotationStepsAtTimeOfCreation() !=
463 fpPolyhedron->GetNumberOfRotationSteps())
464 {
465 fpPolyhedron = CreatePolyhedron();
466 fRebuildPolyhedron = false;
467 }
468 return fpPolyhedron;
469}
