124//______________________________________________________________________________
125G4double G4MultiUnion::GetCubicVolume()
126{
127 // Computes the cubic volume of the "G4MultiUnion" structure using
128 // random points
129
130 if (!fCubicVolume)
131 {
132 G4ThreeVector extentMin, extentMax, d, p, point;
133 G4int inside = 0, generated;
134 BoundingLimits(extentMin, extentMax);
135 d = (extentMax - extentMin) / 2.;
136 p = (extentMax + extentMin) / 2.;
137 G4ThreeVector left = p - d;
138 G4ThreeVector length = d * 2;
139 for (generated = 0; generated < 10000; generated++)
140 {
141 G4ThreeVector rvec(G4UniformRand(), G4UniformRand(), G4UniformRand());
142 point = left + G4ThreeVector(length.x()*rvec.x(),
143 length.y()*rvec.y(),
144 length.z()*rvec.z());
145 if (Inside(point) != EInside::kOutside) inside++;
146 }
147 G4double vbox = (2 * d.x()) * (2 * d.y()) * (2 * d.z());
148 fCubicVolume = inside * vbox / generated;
149 }
150 return fCubicVolume;
151}

154G4double
155G4MultiUnion::DistanceToInNoVoxels(const G4ThreeVector& aPoint,
156 const G4ThreeVector& aDirection) const
157{
158 G4ThreeVector direction = aDirection.unit();
159 G4ThreeVector localPoint, localDirection;
160 G4double minDistance = kInfinity;
161
162 G4int numNodes = fSolids.size();
163 for (G4int i = 0 ; i < numNodes ; ++i)
164 {
165 G4VSolid& solid = *fSolids[i];
166 const G4Transform3D& transform = fTransformObjs[i];
167
168 localPoint = GetLocalPoint(transform, aPoint);
169 localDirection = GetLocalVector(transform, direction);
170
171 G4double distance = solid.DistanceToIn(localPoint, localDirection);
172 if (minDistance > distance) minDistance = distance;
173 }
174 return minDistance;
175}

177//______________________________________________________________________________
178G4double G4MultiUnion::DistanceToInCandidates(const G4ThreeVector& aPoint,
179 const G4ThreeVector& direction,
180 std::vector<G4int>& candidates,
181 G4SurfBits& bits) const
182{
183 G4int candidatesCount = candidates.size();
184 G4ThreeVector localPoint, localDirection;
185
186 G4double minDistance = kInfinity;
187 for (G4int i = 0 ; i < candidatesCount; ++i)
188 {
189 G4int candidate = candidates[i];
190 G4VSolid& solid = *fSolids[candidate];
191 const G4Transform3D& transform = fTransformObjs[candidate];
192
193 localPoint = GetLocalPoint(transform, aPoint);
194 localDirection = GetLocalVector(transform, direction);
195 G4double distance = solid.DistanceToIn(localPoint, localDirection);
196 if (minDistance > distance) minDistance = distance;
197 bits.SetBitNumber(candidate);
198 if (minDistance == 0) break;
199 }
200 return minDistance;
201}
202
203// Algorithm note: we have to look also for all other objects in next voxels,
204// if the distance is not shorter ... we have to do it because,
205// for example for objects which starts in first voxel in which they
206// do not collide with direction line, but in second it collides...
207// The idea of crossing voxels would be still applicable,
208// because this way we could exclude from the testing such solids,
209// which were found that obviously are not good candidates, because
210// they would return infinity
211// But if distance is smaller than the shift to next voxel, we can return
212// it immediately
213//______________________________________________________________________________
214G4double G4MultiUnion::DistanceToIn(const G4ThreeVector& aPoint,
215 const G4ThreeVector& aDirection) const
216{
217 G4double minDistance = kInfinity;
218 G4ThreeVector direction = aDirection.unit();
219 G4double shift = fVoxels.DistanceToFirst(aPoint, direction);
220 if (shift == kInfinity) return shift;
221
222 G4ThreeVector currentPoint = aPoint;
223 if (shift) currentPoint += direction * shift;
224
225 G4SurfBits exclusion(fVoxels.GetBitsPerSlice());
226 std::vector<G4int> candidates, curVoxel(3);
227 fVoxels.GetVoxel(curVoxel, currentPoint);
228
229 do
230 {
231 {
232 if (fVoxels.GetCandidatesVoxelArray(curVoxel, candidates, &exclusion))
233 {
234 G4double distance = DistanceToInCandidates(aPoint, direction,
235 candidates, exclusion);
236 if (minDistance > distance) minDistance = distance;
237 if (distance < shift) break;
238 }
239 }
240 shift = fVoxels.DistanceToNext(aPoint, direction, curVoxel);
241 }
242 while (minDistance > shift);
243
244 return minDistance;
245}

247//______________________________________________________________________________
248G4double G4MultiUnion::DistanceToOutNoVoxels(const G4ThreeVector& aPoint,
249 const G4ThreeVector& aDirection,
250 G4ThreeVector* aNormal) const
251{
252 // Computes distance from a point presumably outside the solid to the solid
253 // surface. Ignores first surface if the point is actually inside.
254 // Early return infinity in case the safety to any surface is found greater
255 // than the proposed step aPstep.
256 // The normal vector to the crossed surface is filled only in case the box
257 // is crossed, otherwise aNormal->IsNull() is true.
258
259 // algorithm:
260 G4ThreeVector direction = aDirection.unit();
261 G4ThreeVector localPoint, localDirection;
262 G4int ignoredSolid = -1;
263 G4double resultDistToOut = 0;
264 G4ThreeVector currentPoint = aPoint;
265
266 G4int numNodes = fSolids.size();
267 for (G4int i = 0; i < numNodes; ++i)
268 {
269 if (i != ignoredSolid)
270 {
271 G4VSolid& solid = *fSolids[i];
272 const G4Transform3D& transform = fTransformObjs[i];
273 localPoint = GetLocalPoint(transform, currentPoint);
274 localDirection = GetLocalVector(transform, direction);
275 EInside location = solid.Inside(localPoint);
276 if (location != EInside::kOutside)
277 {
278 G4double distance = solid.DistanceToOut(localPoint, localDirection,
279 aNormal);
280 if (distance < kInfinity)
281 {
282 if (resultDistToOut == kInfinity) resultDistToOut = 0;
283 if (distance > 0)
284 {
285 currentPoint = GetGlobalPoint(transform, localPoint
286 + distance*localDirection);
287 resultDistToOut += distance;
288 ignoredSolid = i; // skip the solid which we have just left
289 i = -1; // force the loop to continue from 0
290 }
291 }
292 }
293 }
294 }
295 return resultDistToOut;
296}

308//______________________________________________________________________________
309G4double G4MultiUnion::DistanceToOutVoxels(const G4ThreeVector& aPoint,
310 const G4ThreeVector& aDirection,
311 G4ThreeVector* aNormal) const
312{
313 // Computes distance from a point presumably inside the solid to the solid
314 // surface. Ignores first surface along each axis systematically (for points
315 // inside or outside. Early returns zero in case the second surface is behind
316 // the starting point.
317 // o The proposed step is ignored.
318 // o The normal vector to the crossed surface is always filled.
319
320 // In the case the considered point is located inside the G4MultiUnion
321 // structure, the treatments are as follows:
322 // - investigation of the candidates for the passed point
323 // - progressive moving of the point towards the surface, along the
324 // passed direction
325 // - processing of the normal
326
327 G4ThreeVector direction = aDirection.unit();
328 std::vector<G4int> candidates;
329 G4double distance = 0;
330 G4int numNodes = 2*fSolids.size();
331 G4int count=0;
332
333 if (fVoxels.GetCandidatesVoxelArray(aPoint, candidates))
334 {
335 // For normal case for which we presume the point is inside
336 G4ThreeVector localPoint, localDirection, localNormal;
337 G4ThreeVector currentPoint = aPoint;
338 G4SurfBits exclusion(fVoxels.GetBitsPerSlice());
339 G4bool notOutside;
340 G4ThreeVector maxNormal;
341
342 do
343 {
344 notOutside = false;
345
346 G4double maxDistance = -kInfinity;
347 G4int maxCandidate = 0;
348 G4ThreeVector maxLocalPoint;
349
350 G4int limit = candidates.size();
351 for (G4int i = 0 ; i < limit ; ++i)
352 {
353 G4int candidate = candidates[i];
354 // ignore the current component (that you just got out of) since
355 // numerically the propagated point will be on its surface
356
357 G4VSolid& solid = *fSolids[candidate];
358 const G4Transform3D& transform = fTransformObjs[candidate];
359
360 // The coordinates of the point are modified so as to fit the
361 // intrinsic solid local frame:
362 localPoint = GetLocalPoint(transform, currentPoint);
363
364 // DistanceToOut at least for Trd sometimes return non-zero value
365 // even from points that are outside. Therefore, this condition
366 // must currently be here, otherwise it would not work.
367 // But it means it would be slower.
368
369 if (solid.Inside(localPoint) != EInside::kOutside)
370 {
371 notOutside = true;
372
373 localDirection = GetLocalVector(transform, direction);
374
375 // propagate with solid.DistanceToOut
376 G4double shift = solid.DistanceToOut(localPoint, localDirection,
377 false, 0, &localNormal);
378 if (maxDistance < shift)
379 {
380 maxDistance = shift;
381 maxCandidate = candidate;
382 maxNormal = localNormal;
383 }
384 }
385 }
386
387 if (notOutside)
388 {
389 const G4Transform3D& transform = fTransformObjs[maxCandidate];
390
391 // convert from local normal
392 if (aNormal) *aNormal = GetGlobalVector(transform, maxNormal);
393
394 distance += maxDistance;
395 currentPoint += maxDistance * direction;
396 if(maxDistance == 0.) count++;
397
398 // the current component will be ignored
399 exclusion.SetBitNumber(maxCandidate);
400 EInside location = InsideWithExclusion(currentPoint, &exclusion);
401
402 // perform a Inside
403 // it should be excluded current solid from checking
404 // we have to collect the maximum distance from all given candidates.
405 // such "maximum" candidate should be then used for finding next
406 // candidates
407 if (location == EInside::kOutside)
408 {
409 // else return cumulated distances to outside of the traversed
410 // components
411 break;
412 }
413 // if inside another component, redo 1 to 3 but add the next
414 // DistanceToOut on top of the previous.
415
416 // and fill the candidates for the corresponding voxel (just
417 // exiting current component along direction)
418 candidates.clear();
419
420 fVoxels.GetCandidatesVoxelArray(currentPoint, candidates, &exclusion);
421 exclusion.ResetBitNumber(maxCandidate);
422 }
423 }
424 while ((notOutside) && (count < numNodes));
425 }
426
427 return distance;
428}

430//______________________________________________________________________________
431EInside G4MultiUnion::InsideWithExclusion(const G4ThreeVector& aPoint,
432 G4SurfBits* exclusion) const
433{
434 // Classify point location with respect to solid:
435 // o eInside - inside the solid
436 // o eSurface - close to surface within tolerance
437 // o eOutside - outside the solid
438
439 // Hitherto, it is considered that only parallelepipedic nodes
440 // can be added to the container
441
442 // Implementation using voxelisation techniques:
443 // ---------------------------------------------
444
445 G4ThreeVector localPoint;
446 EInside location = EInside::kOutside;
447
448 std::vector<G4int> candidates;
449 std::vector<G4MultiUnionSurface> surfaces;
450
451 // TODO: test if it works well and if so measure performance
452 // TODO: getPointIndex should not be used, instead GetVoxel + GetVoxelsIndex
453 // should be used
454 // TODO: than pass result to GetVoxel further to GetCandidatesVoxelArray
455 // TODO: eventually GetVoxel should be inlined here, early exit if any
456 // binary search is -1
457
458 G4int limit = fVoxels.GetCandidatesVoxelArray(aPoint, candidates, exclusion);
459 for (G4int i = 0 ; i < limit ; ++i)
460 {
461 G4int candidate = candidates[i];
462 G4VSolid& solid = *fSolids[candidate];
463 const G4Transform3D& transform = fTransformObjs[candidate];
464
465 // The coordinates of the point are modified so as to fit the intrinsic
466 // solid local frame:
467 localPoint = GetLocalPoint(transform, aPoint);
468 location = solid.Inside(localPoint);
469 if (location == EInside::kInside) return EInside::kInside;
470 else if (location == EInside::kSurface)
471 {
472 G4MultiUnionSurface surface;
473 surface.point = localPoint;
474 surface.solid = &solid;
475 surfaces.push_back(surface);
476 }
477 }
478
479 ///////////////////////////////////////////////////////////////////////////
480 // Important comment: When two solids touch each other along a flat
481 // surface, the surface points will be considered as kSurface, while points
482 // located around will correspond to kInside (cf. G4UnionSolid)
483
484 G4int size = surfaces.size();
485 for (G4int i = 0; i < size - 1; ++i)
486 {
487 G4MultiUnionSurface& left = surfaces[i];
488 for (G4int j = i + 1; j < size; ++j)
489 {
490 G4MultiUnionSurface& right = surfaces[j];
491 G4ThreeVector n, n2;
492 n = left.solid->SurfaceNormal(left.point);
493 n2 = right.solid->SurfaceNormal(right.point);
494 if ((n + n2).mag2() < 1000 * kRadTolerance)
495 return EInside::kInside;
496 }
497 }
498
499 location = size ? EInside::kSurface : EInside::kOutside;
500
501 return location;
502}

505EInside G4MultiUnion::Inside(const G4ThreeVector& aPoint) const
506{
507 // Classify point location with respect to solid:
508 // o eInside - inside the solid
509 // o eSurface - close to surface within tolerance
510 // o eOutside - outside the solid
511
512 // Hitherto, it is considered that only parallelepipedic nodes can be
513 // added to the container
514
515 // Implementation using voxelisation techniques:
516 // ---------------------------------------------
517
518 // return InsideIterator(aPoint);
519
520 EInside location = InsideWithExclusion(aPoint);
521 return location;
522}

524//______________________________________________________________________________
525EInside G4MultiUnion::InsideNoVoxels(const G4ThreeVector& aPoint) const
526{
527 G4ThreeVector localPoint;
528 EInside location = EInside::kOutside;
529 G4int countSurface = 0;
530
531 G4int numNodes = fSolids.size();
532 for (G4int i = 0 ; i < numNodes ; ++i)
533 {
534 G4VSolid& solid = *fSolids[i];
535 G4Transform3D transform = GetTransformation(i);
536
537 // The coordinates of the point are modified so as to fit the
538 // intrinsic solid local frame:
539 localPoint = GetLocalPoint(transform, aPoint);
540
541 location = solid.Inside(localPoint);
542
543 if (location == EInside::kSurface)
544 countSurface++;
545
546 if (location == EInside::kInside) return EInside::kInside;
547 }
548 if (countSurface != 0) return EInside::kSurface;
549 return EInside::kOutside;
550}

552//______________________________________________________________________________
553void G4MultiUnion::Extent(EAxis aAxis, G4double& aMin, G4double& aMax) const
554{
555 // Determines the bounding box for the considered instance of "UMultipleUnion"
556 G4ThreeVector min, max;
557
558 G4int numNodes = fSolids.size();
559 for (G4int i = 0 ; i < numNodes ; ++i)
560 {
561 G4VSolid& solid = *fSolids[i];
562 G4Transform3D transform = GetTransformation(i);
563 solid.BoundingLimits(min, max);
564
565 TransformLimits(min, max, transform);
566
567 if (i == 0)
568 {
569 switch (aAxis)
570 {
571 case kXAxis:
572 aMin = min.x();
573 aMax = max.x();
574 break;
575 case kYAxis:
576 aMin = min.y();
577 aMax = max.y();
578 break;
579 case kZAxis:
580 aMin = min.z();
581 aMax = max.z();
582 break;
583 default:
584 break;
585 }
586 }
587 else
588 {
589 // Determine the min/max on the considered axis:
590 switch (aAxis)
591 {
592 case kXAxis:
593 if (min.x() < aMin)
594 aMin = min.x();
595 if (max.x() > aMax)
596 aMax = max.x();
597 break;
598 case kYAxis:
599 if (min.y() < aMin)
600 aMin = min.y();
601 if (max.y() > aMax)
602 aMax = max.y();
603 break;
604 case kZAxis:
605 if (min.z() < aMin)
606 aMin = min.z();
607 if (max.z() > aMax)
608 aMax = max.z();
609 break;
610 default:
611 break;
612 }
613 }
614 }
615}

643//______________________________________________________________________________
644G4ThreeVector G4MultiUnion::SurfaceNormal(const G4ThreeVector& aPoint) const
645{
646 // Computes the localNormal on a surface and returns it as a unit vector.
647 // Must return a valid vector. (even if the point is not on the surface).
648 //
649 // On an edge or corner, provide an average localNormal of all facets within
650 // tolerance
651 // NOTE: the tolerance value used in here is not yet the global surface
652 // tolerance - we will have to revise this value - TODO
653
654 std::vector<G4int> candidates;
655 G4ThreeVector localPoint, normal, localNormal;
656 G4double safety = kInfinity;
657 G4int node = 0;
658
659 ///////////////////////////////////////////////////////////////////////////
660 // Important comment: Cases for which the point is located on an edge or
661 // on a vertice remain to be treated
662
663 // determine weather we are in voxel area
664 if (fVoxels.GetCandidatesVoxelArray(aPoint, candidates))
665 {
666 G4int limit = candidates.size();
667 for (G4int i = 0 ; i < limit ; ++i)
668 {
669 G4int candidate = candidates[i];
670 const G4Transform3D& transform = fTransformObjs[candidate];
671
672 // The coordinates of the point are modified so as to fit the intrinsic
673 // solid local frame:
674 localPoint = GetLocalPoint(transform, aPoint);
675 G4VSolid& solid = *fSolids[candidate];
676 EInside location = solid.Inside(localPoint);
677
678 if (location == EInside::kSurface)
679 {
680 // normal case when point is on surface, we pick first solid
681 normal = GetGlobalVector(transform, solid.SurfaceNormal(localPoint));
682 return normal.unit();
683 }
684 else
685 {
686 // collect the smallest safety and remember solid node
687 G4double s = (location == EInside::kInside)
688 ? solid.DistanceToOut(localPoint)
689 : solid.DistanceToIn(localPoint);
690 if (s < safety)
691 {
692 safety = s;
693 node = candidate;
694 }
695 }
696 }
697 // on none of the solids, the point was not on the surface
698 G4VSolid& solid = *fSolids[node];
699 const G4Transform3D& transform = fTransformObjs[node];
700 localPoint = GetLocalPoint(transform, aPoint);
701
702 normal = GetGlobalVector(transform, solid.SurfaceNormal(localPoint));
703 return normal.unit();
704 }
705 else
706 {
707 // for the case when point is certainly outside:
708
709 // find a solid in union with the smallest safety
710 node = SafetyFromOutsideNumberNode(aPoint, safety);
711 G4VSolid& solid = *fSolids[node];
712
713 const G4Transform3D& transform = fTransformObjs[node];
714 localPoint = GetLocalPoint(transform, aPoint);
715
716 // evaluate normal for point at this found solid
717 // and transform multi-union coordinates
718 normal = GetGlobalVector(transform, solid.SurfaceNormal(localPoint));
719
720 return normal.unit();
721 }
722}

724//______________________________________________________________________________
725G4double G4MultiUnion::DistanceToOut(const G4ThreeVector& point) const
726{
727 // Estimates isotropic distance to the surface of the solid. This must
728 // be either accurate or an underestimate.
729 // Two modes: - default/fast mode, sacrificing accuracy for speed
730 // - "precise" mode, requests accurate value if available.
731
732 std::vector<G4int> candidates;
733 G4ThreeVector localPoint;
734 G4double safetyMin = kInfinity;
735
736 // In general, the value return by DistanceToIn(p) will not be the exact
737 // but only an undervalue (cf. overlaps)
738 fVoxels.GetCandidatesVoxelArray(point, candidates);
739
740 G4int limit = candidates.size();
741 for (G4int i = 0; i < limit; ++i)
742 {
743 G4int candidate = candidates[i];
744
745 // The coordinates of the point are modified so as to fit the intrinsic
746 // solid local frame:
747 const G4Transform3D& transform = fTransformObjs[candidate];
748 localPoint = GetLocalPoint(transform, point);
749 G4VSolid& solid = *fSolids[candidate];
750 if (solid.Inside(localPoint) == EInside::kInside)
751 {
752 G4double safety = solid.DistanceToOut(localPoint);
753 if (safetyMin > safety) safetyMin = safety;
754 }
755 }
756 if (safetyMin == kInfinity) safetyMin = 0; // we are not inside
757
758 return safetyMin;
759}

762G4double G4MultiUnion::DistanceToIn(const G4ThreeVector& point) const
763{
764 // Estimates the isotropic safety from a point outside the current solid to
765 // any of its surfaces. The algorithm may be accurate or should provide a fast
766 // underestimate.
767
768 if (!fAccurate) { return fVoxels.DistanceToBoundingBox(point); }
769
770 const std::vector<G4VoxelBox>& boxes = fVoxels.GetBoxes();
771 G4double safetyMin = kInfinity;
772 G4ThreeVector localPoint;
773
774 G4int numNodes = fSolids.size();
775 for (G4int j = 0; j < numNodes; ++j)
776 {
777 G4ThreeVector dxyz;
778 if (j > 0)
779 {
780 const G4ThreeVector& pos = boxes[j].pos;
781 const G4ThreeVector& hlen = boxes[j].hlen;
782 for (G4int i = 0; i <= 2; ++i)
783 // distance to middle point - hlength => distance from point to border
784 // of x,y,z
785 if ((dxyz[i] = std::abs(point[i] - pos[i]) - hlen[i]) > safetyMin)
786 continue;
787
788 G4double d2xyz = 0.;
789 for (G4int i = 0; i <= 2; ++i)
790 if (dxyz[i] > 0) d2xyz += dxyz[i] * dxyz[i];
791
792 // minimal distance is at least this, but could be even higher. therefore,
793 // we can stop if previous was already lower, let us check if it does any
794 // chance to be better tha previous values...
795 if (d2xyz >= safetyMin * safetyMin)
796 {
797 continue;
798 }
799 }
800 const G4Transform3D& transform = fTransformObjs[j];
801 localPoint = GetLocalPoint(transform, point);
802 G4VSolid& solid = *fSolids[j];
803
804 G4double safety = solid.DistanceToIn(localPoint);
805 if (safety <= 0) return safety;
806 // it was detected, that the point is not located outside
807 if (safetyMin > safety) safetyMin = safety;
808 }
809 return safetyMin;
810}

828//______________________________________________________________________________
829G4int G4MultiUnion::SafetyFromOutsideNumberNode(const G4ThreeVector& aPoint,
830 G4double& safetyMin) const
831{
832 // Method returning the closest node from a point located outside a
833 // G4MultiUnion.
834 // This is used to compute the normal in the case no candidate has been found.
835
836 const std::vector<G4VoxelBox>& boxes = fVoxels.GetBoxes();
837 safetyMin = kInfinity;
838 G4int safetyNode = 0;
839 G4ThreeVector localPoint;
840
841 G4int numNodes = fSolids.size();
842 for (G4int i = 0; i < numNodes; ++i)
843 {
844 G4double d2xyz = 0.;
845 G4double dxyz0 = std::abs(aPoint.x() - boxes[i].pos.x()) - boxes[i].hlen.x();
846 if (dxyz0 > safetyMin) continue;
847 G4double dxyz1 = std::abs(aPoint.y() - boxes[i].pos.y()) - boxes[i].hlen.y();
848 if (dxyz1 > safetyMin) continue;
849 G4double dxyz2 = std::abs(aPoint.z() - boxes[i].pos.z()) - boxes[i].hlen.z();
850 if (dxyz2 > safetyMin) continue;
851
852 if (dxyz0 > 0) d2xyz += dxyz0 * dxyz0;
853 if (dxyz1 > 0) d2xyz += dxyz1 * dxyz1;
854 if (dxyz2 > 0) d2xyz += dxyz2 * dxyz2;
855 if (d2xyz >= safetyMin * safetyMin) continue;
856
857 G4VSolid& solid = *fSolids[i];
858 const G4Transform3D& transform = fTransformObjs[i];
859 localPoint = GetLocalPoint(transform, aPoint);
860 fAccurate = true;
861 G4double safety = solid.DistanceToIn(localPoint);
862 fAccurate = false;
863 if (safetyMin > safety)
864 {
865 safetyMin = safety;
866 safetyNode = i;
867 }
868 }
869 return safetyNode;
870}

872//______________________________________________________________________________
873void G4MultiUnion::TransformLimits(G4ThreeVector& min, G4ThreeVector& max,
874 const G4Transform3D& transformation) const
875{
876 // The goal of this method is to convert the quantities min and max
877 // (representing the bounding box of a given solid in its local frame)
878 // to the main frame, using "transformation"
879
880 G4ThreeVector vertices[8] = // Detemination of the vertices thanks to
881 { // the extension of each solid:
882 G4ThreeVector(min.x(), min.y(), min.z()), // 1st vertice:
883 G4ThreeVector(min.x(), max.y(), min.z()), // 2nd vertice:
884 G4ThreeVector(max.x(), max.y(), min.z()),
885 G4ThreeVector(max.x(), min.y(), min.z()),
886 G4ThreeVector(min.x(), min.y(), max.z()),
887 G4ThreeVector(min.x(), max.y(), max.z()),
888 G4ThreeVector(max.x(), max.y(), max.z()),
889 G4ThreeVector(max.x(), min.y(), max.z())
890 };
891
892 min.set(kInfinity,kInfinity,kInfinity);
893 max.set(-kInfinity,-kInfinity,-kInfinity);
894
895 // Loop on th vertices
896 G4int limit = sizeof(vertices) / sizeof(G4ThreeVector);
897 for (G4int i = 0 ; i < limit; ++i)
898 {
899 // From local frame to the global one:
900 // Current positions on the three axis:
901 G4ThreeVector current = GetGlobalPoint(transformation, vertices[i]);
902
903 // If need be, replacement of the min & max values:
904 if (current.x() > max.x()) max.setX(current.x());
905 if (current.x() < min.x()) min.setX(current.x());
906
907 if (current.y() > max.y()) max.setY(current.y());
908 if (current.y() < min.y()) min.setY(current.y());
909
910 if (current.z() > max.z()) max.setZ(current.z());
911 if (current.z() < min.z()) min.setZ(current.z());
912 }
913}

943G4ThreeVector G4MultiUnion::GetPointOnSurface() const
944{
945 G4ThreeVector point;
946
947 G4long size = fSolids.size();
948
949 do

951 G4long rnd = G4RandFlat::shootInt(G4long(0), size);
952 G4VSolid& solid = *fSolids[rnd];
953 point = solid.GetPointOnSurface();
954 const G4Transform3D& transform = fTransformObjs[rnd];

970G4Polyhedron* G4MultiUnion::CreatePolyhedron() const
971{
972 HepPolyhedronProcessor processor;
973 HepPolyhedronProcessor::Operation operation = HepPolyhedronProcessor::UNION;
974
975 G4VSolid* solidA = GetSolid(0);
976 const G4Transform3D transform0=GetTransformation(0);
977 G4DisplacedSolid dispSolidA("placedA",solidA,transform0);
978
979 G4Polyhedron* top = new G4Polyhedron(*dispSolidA.GetPolyhedron());
980
981 for(G4int i=1; i<GetNumberOfSolids(); ++i)
982 {
983 G4VSolid* solidB = GetSolid(i);
984 const G4Transform3D transform=GetTransformation(i);
985 G4DisplacedSolid dispSolidB("placedB",solidB,transform);
986 G4Polyhedron* operand = dispSolidB.GetPolyhedron();
987 processor.push_back (operation, *operand);
988 }
989
990 if (processor.execute(*top)) { return top; }
991 else { return 0; }
992}
