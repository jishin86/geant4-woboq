61 G4VSolid* pSolidB )

63{

69}

155//////////////////////////////////////////////////////////////////////////
156//
157// Get bounding box
158
159void G4UnionSolid::BoundingLimits(G4ThreeVector& pMin,
160 G4ThreeVector& pMax) const
161{
162 G4ThreeVector minA,maxA, minB,maxB;
163 fPtrSolidA->BoundingLimits(minA,maxA);
164 fPtrSolidB->BoundingLimits(minB,maxB);
165
166 pMin.set(std::min(minA.x(),minB.x()),
167 std::min(minA.y(),minB.y()),
168 std::min(minA.z(),minB.z()));
169
170 pMax.set(std::max(maxA.x(),maxB.x()),
171 std::max(maxA.y(),maxB.y()),
172 std::max(maxA.z(),maxB.z()));
173
174 // Check correctness of the bounding box
175 //
176 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
177 {
178 std::ostringstream message;
179 message << "Bad bounding box (min >= max) for solid: "
180 << GetName() << " !"
181 << "\npMin = " << pMin
182 << "\npMax = " << pMax;
183 G4Exception("G4UnionSolid::BoundingLimits()", "GeomMgt0001",
184 JustWarning, message);
185 DumpInfo();
186 }
187}

189//////////////////////////////////////////////////////////////////////////
190//
191// Calculate extent under transform and specified limit
192
193G4bool
194G4UnionSolid::CalculateExtent( const EAxis pAxis,
195 const G4VoxelLimits& pVoxelLimit,
196 const G4AffineTransform& pTransform,
197 G4double& pMin,
198 G4double& pMax ) const
199{
200 G4bool touchesA, touchesB, out ;
201 G4double minA = kInfinity, minB = kInfinity,
202 maxA = -kInfinity, maxB = -kInfinity;
203
204 touchesA = fPtrSolidA->CalculateExtent( pAxis, pVoxelLimit,
205 pTransform, minA, maxA);
206 touchesB = fPtrSolidB->CalculateExtent( pAxis, pVoxelLimit,
207 pTransform, minB, maxB);
208 if( touchesA || touchesB )
209 {
210 pMin = std::min( minA, minB );
211 pMax = std::max( maxA, maxB );
212 out = true ;
213 }
214 else
215 {
216 out = false ;
217 }
218
219 return out ; // It exists in this slice if either one does.
220}

222/////////////////////////////////////////////////////
223//
224// Important comment: When solids A and B touch together along flat
225// surface the surface points will be considered as kSurface, while points
226// located around will correspond to kInside
227
228EInside G4UnionSolid::Inside( const G4ThreeVector& p ) const
229{
230 if (std::max(p.z()-fPMax.z(),fPMin.z()-p.z()) > 0) return kOutside;
231
232 EInside positionA = fPtrSolidA->Inside(p);
233 if (positionA == kInside) { return positionA; } // inside A
234 EInside positionB = fPtrSolidB->Inside(p);
235 if (positionA == kOutside) { return positionB; }
236
237 if (positionB == kInside) { return positionB; } // inside B
238 if (positionB == kOutside) { return positionA; } // surface A
239
240 // Both points are on surface
241 //
242 static const G4double rtol
243 = 1000*G4GeometryTolerance::GetInstance()->GetRadialTolerance();
244
245 return ((fPtrSolidA->SurfaceNormal(p) +
246 fPtrSolidB->SurfaceNormal(p)).mag2() < rtol) ? kInside : kSurface;
247}

249//////////////////////////////////////////////////////////////
250//
251// Get surface normal
252
253G4ThreeVector
254G4UnionSolid::SurfaceNormal( const G4ThreeVector& p ) const
255{
256 EInside positionA = fPtrSolidA->Inside(p);
257 EInside positionB = fPtrSolidB->Inside(p);
258
259 if (positionA == kSurface &&
260 positionB == kOutside) return fPtrSolidA->SurfaceNormal(p);
261
262 if (positionA == kOutside &&
263 positionB == kSurface) return fPtrSolidB->SurfaceNormal(p);
264
265 if (positionA == kSurface &&
266 positionB == kSurface)
267 {
268 if (Inside(p) == kSurface)
269 {
270 G4ThreeVector normalA = fPtrSolidA->SurfaceNormal(p);
271 G4ThreeVector normalB = fPtrSolidB->SurfaceNormal(p);
272 return (normalA + normalB).unit();
273 }
274 }
275#ifdef G4BOOLDEBUG
276 G4String surf[3] = { "OUTSIDE", "SURFACE", "INSIDE" };
277 std::ostringstream message;
278 G4int oldprc = message.precision(16);
279 message << "Invalid call of SurfaceNormal(p) for union solid: "
280 << GetName() << " !"
281 << "\nPoint p" << p << " is " << surf[Inside(p)] << " !!!";
282 message.precision(oldprc);
283 G4Exception("G4UnionSolid::SurfaceNormal()", "GeomMgt0001",
284 JustWarning, message);
285#endif
286 return fPtrSolidA->SurfaceNormal(p);
287}

289/////////////////////////////////////////////////////////////
290//
291// The same algorithm as in DistanceToIn(p)
292
293G4double
294G4UnionSolid::DistanceToIn( const G4ThreeVector& p,
295 const G4ThreeVector& v ) const
296{
297#ifdef G4BOOLDEBUG
298 if( Inside(p) == kInside )
299 {
300 G4cout << "WARNING - Invalid call in "
301 << "G4UnionSolid::DistanceToIn(p,v)" << G4endl
302 << " Point p is inside !" << G4endl;
303 G4cout << " p = " << p << G4endl;
304 G4cout << " v = " << v << G4endl;
305 G4cerr << "WARNING - Invalid call in "
306 << "G4UnionSolid::DistanceToIn(p,v)" << G4endl
307 << " Point p is inside !" << G4endl;
308 G4cerr << " p = " << p << G4endl;
309 G4cerr << " v = " << v << G4endl;
310 }
311#endif
312
313 return std::min(fPtrSolidA->DistanceToIn(p,v),
314 fPtrSolidB->DistanceToIn(p,v) ) ;
315}

317////////////////////////////////////////////////////////
318//
319// Approximate nearest distance from the point p to the union of
320// two solids
321
322G4double
323G4UnionSolid::DistanceToIn( const G4ThreeVector& p) const
324{
325#ifdef G4BOOLDEBUG
326 if( Inside(p) == kInside )
327 {
328 G4cout << "WARNING - Invalid call in "
329 << "G4UnionSolid::DistanceToIn(p)" << G4endl
330 << " Point p is inside !" << G4endl;
331 G4cout << " p = " << p << G4endl;
332 G4cerr << "WARNING - Invalid call in "
333 << "G4UnionSolid::DistanceToIn(p)" << G4endl
334 << " Point p is inside !" << G4endl;
335 G4cerr << " p = " << p << G4endl;
336 }
337#endif
338 G4double distA = fPtrSolidA->DistanceToIn(p) ;
339 G4double distB = fPtrSolidB->DistanceToIn(p) ;
340 G4double safety = std::min(distA,distB) ;
341 if(safety < 0.0) safety = 0.0 ;
342 return safety ;
343}

345//////////////////////////////////////////////////////////
346//
347// The same algorithm as DistanceToOut(p)
348
349G4double
350G4UnionSolid::DistanceToOut( const G4ThreeVector& p,
351 const G4ThreeVector& v,
352 const G4bool calcNorm,
353 G4bool *validNorm,
354 G4ThreeVector *n ) const
355{
356 G4double dist = 0.0, disTmp = 0.0 ;
357 G4ThreeVector normTmp;
358 G4ThreeVector* nTmp= &normTmp;
359
360 if( Inside(p) == kOutside )
361 {
362#ifdef G4BOOLDEBUG
363 G4cout << "Position:" << G4endl << G4endl;
364 G4cout << "p.x() = " << p.x()/mm << " mm" << G4endl;
365 G4cout << "p.y() = " << p.y()/mm << " mm" << G4endl;
366 G4cout << "p.z() = " << p.z()/mm << " mm" << G4endl << G4endl;
367 G4cout << "Direction:" << G4endl << G4endl;
368 G4cout << "v.x() = " << v.x() << G4endl;
369 G4cout << "v.y() = " << v.y() << G4endl;
370 G4cout << "v.z() = " << v.z() << G4endl << G4endl;
371 G4cout << "WARNING - Invalid call in "
372 << "G4UnionSolid::DistanceToOut(p,v)" << G4endl
373 << " Point p is outside !" << G4endl;
374 G4cout << " p = " << p << G4endl;
375 G4cout << " v = " << v << G4endl;
376 G4cerr << "WARNING - Invalid call in "
377 << "G4UnionSolid::DistanceToOut(p,v)" << G4endl
378 << " Point p is outside !" << G4endl;
379 G4cerr << " p = " << p << G4endl;
380 G4cerr << " v = " << v << G4endl;
381#endif
382 }
383 else
384 {
385 EInside positionA = fPtrSolidA->Inside(p) ;
386 // EInside positionB = fPtrSolidB->Inside(p) ;
387
388 if( positionA != kOutside )
389 {
390 do // Loop checking, 13.08.2015, G.Cosmo
391 {
392 disTmp = fPtrSolidA->DistanceToOut(p+dist*v,v,calcNorm,
393 validNorm,nTmp);
394 dist += disTmp ;
395
396 if(fPtrSolidB->Inside(p+dist*v) != kOutside)
397 {
398 disTmp = fPtrSolidB->DistanceToOut(p+dist*v,v,calcNorm,
399 validNorm,nTmp);
400 dist += disTmp ;
401 }
402 }
403 while( (fPtrSolidA->Inside(p+dist*v) != kOutside)
404 && (disTmp > 0.5*kCarTolerance) );
405 }
406 else // if( positionB != kOutside )
407 {
408 do // Loop checking, 13.08.2015, G.Cosmo
409 {
410 disTmp = fPtrSolidB->DistanceToOut(p+dist*v,v,calcNorm,
411 validNorm,nTmp);
412 dist += disTmp ;
413
414 if(fPtrSolidA->Inside(p+dist*v) != kOutside)
415 {
416 disTmp = fPtrSolidA->DistanceToOut(p+dist*v,v,calcNorm,
417 validNorm,nTmp);
418 dist += disTmp ;
419 }
420 }
421 while( (fPtrSolidB->Inside(p+dist*v) != kOutside)
422 && (disTmp > 0.5*kCarTolerance) );
423 }
424 }
425 if( calcNorm )
426 {
427 *validNorm = false ;
428 *n = *nTmp ;
429 }
430 return dist ;
431}

433//////////////////////////////////////////////////////////////
434//
435// Inverted algorithm of DistanceToIn(p)
436
437G4double
438G4UnionSolid::DistanceToOut( const G4ThreeVector& p ) const
439{
440 G4double distout = 0.0;
441 if( Inside(p) == kOutside )
442 {
443#ifdef G4BOOLDEBUG
444 G4cout << "WARNING - Invalid call in "
445 << "G4UnionSolid::DistanceToOut(p)" << G4endl
446 << " Point p is outside !" << G4endl;
447 G4cout << " p = " << p << G4endl;
448 G4cerr << "WARNING - Invalid call in "
449 << "G4UnionSolid::DistanceToOut(p)" << G4endl
450 << " Point p is outside !" << G4endl;
451 G4cerr << " p = " << p << G4endl;
452#endif
453 }
454 else
455 {
456 EInside positionA = fPtrSolidA->Inside(p) ;
457 EInside positionB = fPtrSolidB->Inside(p) ;
458
459 // Is this equivalent ??
460 // if( ! ( (positionA == kOutside)) &&
461 // (positionB == kOutside)) )
462 if((positionA == kInside && positionB == kInside ) ||
463 (positionA == kInside && positionB == kSurface ) ||
464 (positionA == kSurface && positionB == kInside ) )
465 {
466 distout= std::max(fPtrSolidA->DistanceToOut(p),
467 fPtrSolidB->DistanceToOut(p) ) ;
468 }
469 else
470 {
471 if(positionA == kOutside)
472 {
473 distout= fPtrSolidB->DistanceToOut(p) ;
474 }
475 else
476 {
477 distout= fPtrSolidA->DistanceToOut(p) ;
478 }
479 }
480 }
481 return distout;
482}

526
527G4Polyhedron*
528G4UnionSolid::CreatePolyhedron () const
529{
530 HepPolyhedronProcessor processor;
531 // Stack components and components of components recursively
532 // See G4BooleanSolid::StackPolyhedron
533 G4Polyhedron* top = StackPolyhedron(processor, this);
534 G4Polyhedron* result = new G4Polyhedron(*top);
535 if (processor.execute(*result)) { return result; }
536 else { return 0; }
537}
