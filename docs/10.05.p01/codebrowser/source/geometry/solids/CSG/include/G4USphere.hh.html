32// G4USphere
33//
34// Class description:
35//
36// Wrapper class for G4Sphere to make use of VecGeom Sphere.
37
38// History:
39// 13.09.13 G.Cosmo, CERN/PH
40// --------------------------------------------------------------------
41#ifndef G4USPHERE_HH
42#define G4USPHERE_HH
43
44#include "G4UAdapter.hh"
45
46#if ( defined(G4GEOM_USE_USOLIDS) || defined(G4GEOM_USE_PARTIAL_USOLIDS) )
47
48#include <volumes/UnplacedSphere.h>
49
50#include "G4Polyhedron.hh"
51
52class G4USphere : public G4UAdapter<vecgeom::UnplacedSphere>
53{
54 using Shape_t = vecgeom::UnplacedSphere;
55 using Base_t = G4UAdapter<vecgeom::UnplacedSphere>;
56
57 public: // with description
58
59 G4USphere(const G4String& pName,
60 G4double pRmin, G4double pRmax,
61 G4double pSPhi, G4double pDPhi,
62 G4double pSTheta, G4double pDTheta);
63 // Constructs a sphere or sphere shell section
64 // with the given name and dimensions
65
66 ~G4USphere();
67
68 void ComputeDimensions( G4VPVParameterisation* p,
69 const G4int n,
70 const G4VPhysicalVolume* pRep);
71
72 G4VSolid* Clone() const;
73
74 G4double GetInnerRadius () const;
75 G4double GetOuterRadius () const;
76 G4double GetStartPhiAngle () const;
77 G4double GetDeltaPhiAngle () const;
78 G4double GetStartThetaAngle() const;
79 G4double GetDeltaThetaAngle() const;
80 G4double GetSinStartPhi () const;
81 G4double GetCosStartPhi () const;
82 G4double GetSinEndPhi () const;
83 G4double GetCosEndPhi () const;
84 G4double GetSinStartTheta () const;
85 G4double GetCosStartTheta () const;
86 G4double GetSinEndTheta () const;
87 G4double GetCosEndTheta () const;
88
89 void SetInnerRadius (G4double newRMin);
90 void SetOuterRadius (G4double newRmax);
91 void SetStartPhiAngle (G4double newSphi, G4bool trig=true);
92 void SetDeltaPhiAngle (G4double newDphi);
93 void SetStartThetaAngle(G4double newSTheta);
94 void SetDeltaThetaAngle(G4double newDTheta);
95
96 inline G4GeometryType GetEntityType() const;
97
98 void BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const;
99
100 G4bool CalculateExtent(const EAxis pAxis,
101 const G4VoxelLimits& pVoxelLimit,
102 const G4AffineTransform& pTransform,
103 G4double& pMin, G4double& pMax) const;
104
105 G4Polyhedron* CreatePolyhedron() const;
106
107 public: // without description
108
109 G4USphere(__void__&);
110 //
111 // Fake default constructor for usage restricted to direct object
112 // persistency for clients requiring preallocation of memory for
113 // persistifiable objects.
114
115 G4USphere(const G4USphere& rhs);
116 G4USphere& operator=(const G4USphere& rhs);
117 // Copy constructor and assignment operator.
118};
119
120// --------------------------------------------------------------------
121// Inline methods
122// --------------------------------------------------------------------
123
124inline G4GeometryType G4USphere::GetEntityType() const
125{
126 return "G4Sphere";
127}

129#endif // G4GEOM_USE_USOLIDS

131#endif
