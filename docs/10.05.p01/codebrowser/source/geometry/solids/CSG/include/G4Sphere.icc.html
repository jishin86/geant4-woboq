128 fCubicVolume = 0.;
129 fSurfaceArea = 0.;
130 fRebuildPolyhedron = true;
131}

133inline
134void G4Sphere::InitializePhiTrigonometry()
135{
136 hDPhi = 0.5*fDPhi; // half delta phi
137 cPhi = fSPhi + hDPhi;
138 ePhi = fSPhi + fDPhi;
139
140 sinCPhi = std::sin(cPhi);
141 cosCPhi = std::cos(cPhi);
142 cosHDPhiIT = std::cos(hDPhi - 0.5*kAngTolerance); // inner/outer tol half dphi
143 cosHDPhiOT = std::cos(hDPhi + 0.5*kAngTolerance);
144 sinSPhi = std::sin(fSPhi);
145 cosSPhi = std::cos(fSPhi);
146 sinEPhi = std::sin(ePhi);
147 cosEPhi = std::cos(ePhi);
148}
149
150inline
151void G4Sphere::InitializeThetaTrigonometry()
152{
153 eTheta = fSTheta + fDTheta;
154
155 sinSTheta = std::sin(fSTheta);
156 cosSTheta = std::cos(fSTheta);
157 sinETheta = std::sin(eTheta);
158 cosETheta = std::cos(eTheta);
159
160 tanSTheta = std::tan(fSTheta);
161 tanSTheta2 = tanSTheta*tanSTheta;
162 tanETheta = std::tan(eTheta);
163 tanETheta2 = tanETheta*tanETheta;
164}

166inline
167void G4Sphere::CheckThetaAngles(G4double sTheta, G4double dTheta)
168{
169 if ( (sTheta<0) || (sTheta>CLHEP::pi) )
170 {
171 std::ostringstream message;
172 message << "sTheta outside 0-PI range." << G4endl
173 << "Invalid starting Theta angle for solid: " << GetName();
174 G4Exception("G4Sphere::CheckThetaAngles()", "GeomSolids0002",
175 FatalException, message);
176 }
177 else
178 {
179 fSTheta=sTheta;
180 }
181 if ( dTheta+sTheta >= CLHEP::pi )
182 {
183 fDTheta=CLHEP::pi-sTheta;
184 }
185 else if ( dTheta > 0 )
186 {
187 fDTheta=dTheta;
188 }
189 else
190 {
191 std::ostringstream message;
192 message << "Invalid dTheta." << G4endl
193 << "Negative delta-Theta (" << dTheta << "), for solid: "
194 << GetName();
195 G4Exception("G4Sphere::CheckThetaAngles()", "GeomSolids0002",
196 FatalException, message);
197 }
198 if ( fDTheta-fSTheta < CLHEP::pi ) { fFullThetaSphere = false; }
199 else { fFullThetaSphere = true ; }
200 fFullSphere = fFullPhiSphere && fFullThetaSphere;
201
202 InitializeThetaTrigonometry();
203}

205inline
206void G4Sphere::CheckSPhiAngle(G4double sPhi)
207{
208 // Ensure fSphi in 0-2PI or -2PI-0 range if shape crosses 0
209
210 if ( sPhi < 0 )
211 {
212 fSPhi = CLHEP::twopi - std::fmod(std::fabs(sPhi),CLHEP::twopi);
213 }
214 else
215 {
216 fSPhi = std::fmod(sPhi,CLHEP::twopi) ;
217 }
218 if ( fSPhi+fDPhi > CLHEP::twopi )
219 {
220 fSPhi -= CLHEP::twopi ;
221 }
222}
223
224inline
225void G4Sphere::CheckDPhiAngle(G4double dPhi)
226{
227 fFullPhiSphere = true;
228 if ( dPhi >= CLHEP::twopi-kAngTolerance*0.5 )
229 {
230 fDPhi=CLHEP::twopi;
231 }
232 else
233 {
234 fFullPhiSphere = false;
235 if ( dPhi > 0 )
236 {
237 fDPhi = dPhi;
238 }
239 else
240 {
241 std::ostringstream message;
242 message << "Invalid dphi." << G4endl
243 << "Negative delta-Phi (" << dPhi << "), for solid: "
244 << GetName();
245 G4Exception("G4Sphere::CheckDPhiAngle()", "GeomSolids0002",
246 FatalException, message);
247 }
248 }
249}

251inline
252void G4Sphere::CheckPhiAngles(G4double sPhi, G4double dPhi)
253{
254 CheckDPhiAngle(dPhi);
255 if (!fFullPhiSphere && sPhi) { CheckSPhiAngle(sPhi); }
256 fFullSphere = fFullPhiSphere && fFullThetaSphere;
257
258 InitializePhiTrigonometry();
259}

262void G4Sphere::SetInsideRadius(G4double newRmin)
263{
264 fRmin= newRmin;

266 Initialize();
267}

283inline
284void G4Sphere::SetStartPhiAngle(G4double newSPhi, G4bool compute)
285{
286 // Flag 'compute' can be used to explicitely avoid recomputation of
287 // trigonometry in case SetDeltaPhiAngle() is invoked afterwards
288
289 CheckSPhiAngle(newSPhi);
290 fFullPhiSphere = false;
291 if (compute) { InitializePhiTrigonometry(); }
292 Initialize();
293}

356{
357 if(fCubicVolume != 0.) {;}
358 else { fCubicVolume = fDPhi*(std::cos(fSTheta)-std::cos(fSTheta+fDTheta))*
359 (fRmax*fRmax*fRmax-fRmin*fRmin*fRmin)/3.; }
