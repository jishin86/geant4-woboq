103{
104 // Check assignment to self
105 //
106 if (this == &rhs) { return *this; }

113 //

115 halfRmaxTol = rhs.halfRmaxTol;

117 sqrRmaxMinusTol = rhs.sqrRmaxMinusTol;

120}

122//////////////////////////////////////////////////////////////////////////
123//
124// Check radius and initialize dada members
125
126void G4Orb::Initialize()
127{
128 const G4double fEpsilon = 2.e-11; // relative tolerance of fRmax
129
130 // Check radius
131 //
132 if ( fRmax < 10*kCarTolerance )
133 {
134 G4Exception("G4Orb::Initialize()", "GeomSolids0002", FatalException,
135 "Invalid radius < 10*kCarTolerance.");
136 }
137 halfRmaxTol = 0.5 * std::max(kCarTolerance, fEpsilon*fRmax);
138 G4double rmaxPlusTol = fRmax + halfRmaxTol;
139 G4double rmaxMinusTol = fRmax - halfRmaxTol;
140 sqrRmaxPlusTol = rmaxPlusTol*rmaxPlusTol;
141 sqrRmaxMinusTol = rmaxMinusTol*rmaxMinusTol;
142}

156//////////////////////////////////////////////////////////////////////////
157//
158// Get bounding box
159
160void G4Orb::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
161{
162 G4double radius = GetRadius();
163 pMin.set(-radius,-radius,-radius);
164 pMax.set( radius, radius, radius);
165
166 // Check correctness of the bounding box
167 //
168 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
169 {
170 std::ostringstream message;
171 message << "Bad bounding box (min >= max) for solid: "
172 << GetName() << " !"
173 << "\npMin = " << pMin
174 << "\npMax = " << pMax;
175 G4Exception("G4Orb::BoundingLimits()", "GeomMgt0001",
176 JustWarning, message);
177 DumpInfo();
178 }
179}

181//////////////////////////////////////////////////////////////////////////
182//
183// Calculate extent under transform and specified limit
184
185G4bool G4Orb::CalculateExtent(const EAxis pAxis,
186 const G4VoxelLimits& pVoxelLimit,
187 const G4AffineTransform& pTransform,
188 G4double& pMin, G4double& pMax) const
189{
190 G4ThreeVector bmin, bmax;
191 G4bool exist;
192
193 // Get bounding box
194 BoundingLimits(bmin,bmax);
195
196 // Check bounding box
197 G4BoundingEnvelope bbox(bmin,bmax);
198#ifdef G4BBOX_EXTENT
199 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
200#endif
201 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
202 {
203 return exist = (pMin < pMax) ? true : false;
204 }
205
206 // Find bounding envelope and calculate extent
207 //
208 static const G4int NTHETA = 8; // number of steps along Theta
209 static const G4int NPHI = 16; // number of steps along Phi
210 static const G4double sinHalfTheta = std::sin(halfpi/NTHETA);
211 static const G4double cosHalfTheta = std::cos(halfpi/NTHETA);
212 static const G4double sinHalfPhi = std::sin(pi/NPHI);
213 static const G4double cosHalfPhi = std::cos(pi/NPHI);
214 static const G4double sinStepTheta = 2.*sinHalfTheta*cosHalfTheta;
215 static const G4double cosStepTheta = 1. - 2.*sinHalfTheta*sinHalfTheta;
216 static const G4double sinStepPhi = 2.*sinHalfPhi*cosHalfPhi;
217 static const G4double cosStepPhi = 1. - 2.*sinHalfPhi*sinHalfPhi;
218
219 G4double radius = GetRadius();
220 G4double rtheta = radius/cosHalfTheta;
221 G4double rphi = rtheta/cosHalfPhi;
222
223 // set reference circle
224 G4TwoVector xy[NPHI];
225 G4double sinCurPhi = sinHalfPhi;
226 G4double cosCurPhi = cosHalfPhi;
227 for (G4int k=0; k<NPHI; ++k)
228 {
229 xy[k].set(cosCurPhi,sinCurPhi);
230 G4double sinTmpPhi = sinCurPhi;
231 sinCurPhi = sinCurPhi*cosStepPhi + cosCurPhi*sinStepPhi;
232 cosCurPhi = cosCurPhi*cosStepPhi - sinTmpPhi*sinStepPhi;
233 }
234
235 // set bounding circles
236 G4ThreeVectorList circles[NTHETA];
237 for (G4int i=0; i<NTHETA; ++i) circles[i].resize(NPHI);
238
239 G4double sinCurTheta = sinHalfTheta;
240 G4double cosCurTheta = cosHalfTheta;
241 for (G4int i=0; i<NTHETA; ++i)
242 {
243 G4double z = rtheta*cosCurTheta;
244 G4double rho = rphi*sinCurTheta;
245 for (G4int k=0; k<NPHI; ++k)
246 {
247 circles[i][k].set(rho*xy[k].x(),rho*xy[k].y(),z);
248 }
249 G4double sinTmpTheta = sinCurTheta;
250 sinCurTheta = sinCurTheta*cosStepTheta + cosCurTheta*sinStepTheta;
251 cosCurTheta = cosCurTheta*cosStepTheta - sinTmpTheta*sinStepTheta;
252 }
253
254 // set envelope and calculate extent
255 std::vector<const G4ThreeVectorList *> polygons;
256 polygons.resize(NTHETA);
257 for (G4int i=0; i<NTHETA; ++i) polygons[i] = &circles[i];
258
259 G4BoundingEnvelope benv(bmin,bmax,polygons);
260 exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
261 return exist;
262}

264//////////////////////////////////////////////////////////////////////////
265//
266// Return whether point is inside/outside/on surface
267
268EInside G4Orb::Inside( const G4ThreeVector& p ) const
269{
270 G4double rr = p.mag2();
271 if (rr > sqrRmaxPlusTol) return kOutside;
272 return (rr > sqrRmaxMinusTol) ? kSurface : kInside;
273}

284//////////////////////////////////////////////////////////////////////////
285//
286// Calculate distance to the surface of the orb from outside
287// - return kInfinity if no intersection or
288// intersection distance <= tolerance
289
290G4double G4Orb::DistanceToIn( const G4ThreeVector& p,
291 const G4ThreeVector& v ) const
292{
293 // Check if point is on the surface and traveling away
294 //
295 G4double rr = p.mag2();
296 G4double pv = p.dot(v);
297 if (rr >= sqrRmaxMinusTol && pv >= 0) return kInfinity;
298
299 // Find intersection
300 //
301 // Sphere eqn: x^2 + y^2 + z^2 = R^2
302 //
303 // => (px + t*vx)^2 + (py + t*vy)^2 + (pz + t*vz)^2 = R^2
304 // => r^2 + 2t(p.v) + t^2 = R^2
305 // => tmin = -(p.v) - Sqrt((p.v)^2 - (r^2 - R^2))
306 //
307 G4double D = pv*pv - rr + fRmax*fRmax;
308 if (D < 0) return kInfinity; // no intersection
309
310 G4double sqrtD = std::sqrt(D);
311 G4double dist = -pv - sqrtD;
312
313 // Avoid rounding errors due to precision issues seen on 64 bits systems.
314 // Split long distances and recompute
315 //
316 G4double Dmax = 32*fRmax;
317 if (dist > Dmax)
318 {
319 dist = dist - 1.e-8*dist - fRmax; // to stay outside after the move
320 dist += DistanceToIn(p + dist*v, v);
321 return (dist >= kInfinity) ? kInfinity : dist;
322 }
323
324 if (sqrtD*2 <= halfRmaxTol) return kInfinity; // touch
325 return (dist < halfRmaxTol) ? 0. : dist;
326}

328//////////////////////////////////////////////////////////////////////////
329//
330// Calculate shortest distance to the boundary from outside
331// - Return 0 if point is inside
332
333G4double G4Orb::DistanceToIn( const G4ThreeVector& p ) const
334{
335 G4double dist = p.mag() - fRmax;
336 return (dist > 0) ? dist : 0.;
337}

339//////////////////////////////////////////////////////////////////////////
340//
341// Calculate distance to the surface of the orb from inside and
342// find normal at exit point, if required
343// - when leaving the surface, return 0
344
345G4double G4Orb::DistanceToOut( const G4ThreeVector& p,
346 const G4ThreeVector& v,
347 const G4bool calcNorm,
348 G4bool *validNorm,
349 G4ThreeVector *n ) const
350{
351 // Check if point is on the surface and traveling away
352 //
353 G4double rr = p.mag2();
354 G4double pv = p.dot(v);
355 if (rr >= sqrRmaxMinusTol && pv > 0)
356 {
357 if (calcNorm)
358 {
359 *validNorm = true;
360 *n = p*(1./std::sqrt(rr));
361 }
362 return 0.;
363 }
364
365 // Find intersection
366 //
367 // Sphere eqn: x^2 + y^2 + z^2 = R^2
368 //
369 // => (px + t*vx)^2 + (py + t*vy)^2 + (pz + t*vz)^2 = R^2
370 // => r^2 + 2t(p.v) + t^2 = R^2
371 // => tmax = -(p.v) + Sqrt((p.v)^2 - (r^2 - R^2))
372 //
373 G4double D = pv*pv - rr + fRmax*fRmax;
374 G4double tmax = (D <= 0) ? 0. : std::sqrt(D) - pv;
375 if (tmax < halfRmaxTol) tmax = 0.;
376 if (calcNorm)
377 {
378 *validNorm = true;
379 G4ThreeVector ptmax = p + tmax*v;
380 *n = ptmax*(1./ptmax.mag());
381 }
382 return tmax;
383}

385//////////////////////////////////////////////////////////////////////////
386//
387// Calculate distance (<=actual) to closest surface of shape from inside
388
389G4double G4Orb::DistanceToOut( const G4ThreeVector& p ) const
390{
391#ifdef G4CSGDEBUG
392 if( Inside(p) == kOutside )
393 {
394 std::ostringstream message;
395 G4int oldprc = message.precision(16);
396 message << "Point p is outside (!?) of solid: " << GetName() << "\n";
397 message << "Position:\n";
398 message << " p.x() = " << p.x()/mm << " mm\n";
399 message << " p.y() = " << p.y()/mm << " mm\n";
400 message << " p.z() = " << p.z()/mm << " mm";
401 G4cout.precision(oldprc);
402 G4Exception("G4Trap::DistanceToOut(p)", "GeomSolids1002",
403 JustWarning, message );
404 DumpInfo();
405 }
406#endif
407 G4double dist = fRmax - p.mag();
408 return (dist > 0) ? dist : 0.;
409}
