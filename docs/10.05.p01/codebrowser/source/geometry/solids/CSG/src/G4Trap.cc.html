97//////////////////////////////////////////////////////////////////////////
98//
99// Constructor - Design of trapezoid based on 8 G4ThreeVector parameters,
100// which are its vertices. Checking of planarity with preparation of
101// fPlanes[] and than calculation of other members
102
103G4Trap::G4Trap( const G4String& pName,
104 const G4ThreeVector pt[8] )
105 : G4CSGSolid(pName), halfCarTolerance(0.5*kCarTolerance)
106{
107 // Start with check of centering - the center of gravity trap line
108 // should cross the origin of frame
109 //
110 if (!( pt[0].z() < 0
111 && pt[0].z() == pt[1].z()
112 && pt[0].z() == pt[2].z()
113 && pt[0].z() == pt[3].z()
114
115 && pt[4].z() > 0
116 && pt[4].z() == pt[5].z()
117 && pt[4].z() == pt[6].z()
118 && pt[4].z() == pt[7].z()
119
120 && std::fabs( pt[0].z() + pt[4].z() ) < kCarTolerance
121
122 && pt[0].y() == pt[1].y()
123 && pt[2].y() == pt[3].y()
124 && pt[4].y() == pt[5].y()
125 && pt[6].y() == pt[7].y()
126
127 && std::fabs(pt[0].y()+pt[2].y()+pt[4].y()+pt[6].y()) < kCarTolerance
128 && std::fabs(pt[0].x()+pt[1].x()+pt[4].x()+pt[5].x() +
129 pt[2].x()+pt[3].x()+pt[6].x()+pt[7].x()) < kCarTolerance ))
130 {
131 std::ostringstream message;
132 message << "Invalid vertice coordinates for Solid: " << GetName();
133 G4Exception("G4Trap::G4Trap()", "GeomSolids0002",
134 FatalException, message);
135 }
136
137 // Set parameters
138 //
139 fDz = (pt[7]).z();
140
141 fDy1 = ((pt[2]).y()-(pt[1]).y())*0.5;
142 fDx1 = ((pt[1]).x()-(pt[0]).x())*0.5;
143 fDx2 = ((pt[3]).x()-(pt[2]).x())*0.5;
144 fTalpha1 = ((pt[2]).x()+(pt[3]).x()-(pt[1]).x()-(pt[0]).x())*0.25/fDy1;
145
146 fDy2 = ((pt[6]).y()-(pt[5]).y())*0.5;
147 fDx3 = ((pt[5]).x()-(pt[4]).x())*0.5;
148 fDx4 = ((pt[7]).x()-(pt[6]).x())*0.5;
149 fTalpha2 = ((pt[6]).x()+(pt[7]).x()-(pt[5]).x()-(pt[4]).x())*0.25/fDy2;
150
151 fTthetaCphi = ((pt[4]).x()+fDy2*fTalpha2+fDx3)/fDz;
152 fTthetaSphi = ((pt[4]).y()+fDy2)/fDz;
153
154 CheckParameters();
155 MakePlanes(pt);
156}

269// Assignment operator

272{
273 // Check assignment to self
274 //
275 if (this == &rhs) { return *this; }

282 //

284 fDz = rhs.fDz; fTthetaCphi = rhs.fTthetaCphi; fTthetaSphi = rhs.fTthetaSphi;
285 fDy1 = rhs.fDy1; fDx1 = rhs.fDx1; fDx2 = rhs.fDx2; fTalpha1 = rhs.fTalpha1;

287 for (G4int i=0; i<4; ++i) { fPlanes[i] = rhs.fPlanes[i]; }
288 fTrapType = rhs.fTrapType;
289 return *this;
290}

346//////////////////////////////////////////////////////////////////////////
347//
348// Compute vertices and set side planes
349
350void G4Trap::MakePlanes()
351{
352 G4double DzTthetaCphi = fDz*fTthetaCphi;
353 G4double DzTthetaSphi = fDz*fTthetaSphi;
354 G4double Dy1Talpha1 = fDy1*fTalpha1;
355 G4double Dy2Talpha2 = fDy2*fTalpha2;
356
357 G4ThreeVector pt[8] =
358 {
359 G4ThreeVector(-DzTthetaCphi-Dy1Talpha1-fDx1,-DzTthetaSphi-fDy1,-fDz),
360 G4ThreeVector(-DzTthetaCphi-Dy1Talpha1+fDx1,-DzTthetaSphi-fDy1,-fDz),
361 G4ThreeVector(-DzTthetaCphi+Dy1Talpha1-fDx2,-DzTthetaSphi+fDy1,-fDz),
362 G4ThreeVector(-DzTthetaCphi+Dy1Talpha1+fDx2,-DzTthetaSphi+fDy1,-fDz),
363 G4ThreeVector( DzTthetaCphi-Dy2Talpha2-fDx3, DzTthetaSphi-fDy2, fDz),
364 G4ThreeVector( DzTthetaCphi-Dy2Talpha2+fDx3, DzTthetaSphi-fDy2, fDz),
365 G4ThreeVector( DzTthetaCphi+Dy2Talpha2-fDx4, DzTthetaSphi+fDy2, fDz),
366 G4ThreeVector( DzTthetaCphi+Dy2Talpha2+fDx4, DzTthetaSphi+fDy2, fDz)
367 };
368
369 MakePlanes(pt);
370}

372//////////////////////////////////////////////////////////////////////////
373//
374// Set side planes, check planarity
375
376void G4Trap::MakePlanes(const G4ThreeVector pt[8])
377{
378 G4int iface[4][4] = { {0,4,5,1}, {2,3,7,6}, {0,2,6,4}, {1,5,7,3} };
379 G4String side[4] = { "~-Y", "~+Y", "~-X", "~+X" };
380
381 for (G4int i=0; i<4; ++i)
382 {
383 if (MakePlane(pt[iface[i][0]],
384 pt[iface[i][1]],
385 pt[iface[i][2]],
386 pt[iface[i][3]],
387 fPlanes[i])) continue;
388
389 // Non planar side face
390 G4ThreeVector normal(fPlanes[i].a,fPlanes[i].b,fPlanes[i].c);
391 G4double dmax = 0;
392 for (G4int k=0; k<4; ++k)
393 {
394 G4double dist = normal.dot(pt[iface[i][k]]) + fPlanes[i].d;
395 if (std::abs(dist) > std::abs(dmax)) dmax = dist;
396 }
397 std::ostringstream message;
398 message << "Side face " << side[i] << " is not planar for solid: "
399 << GetName() << "\nDiscrepancy: " << dmax/mm << " mm\n";
400 StreamInfo(message);
401 G4Exception("G4Trap::MakePlanes()", "GeomSolids0002",
402 FatalException, message);
403 }
404
405 // Define type of trapezoid
406 fTrapType = 0;
407 if (fPlanes[0].b == -1 && fPlanes[1].b == 1 &&
408 std::abs(fPlanes[0].a) < DBL_EPSILON &&
409 std::abs(fPlanes[0].c) < DBL_EPSILON &&
410 std::abs(fPlanes[1].a) < DBL_EPSILON &&
411 std::abs(fPlanes[1].c) < DBL_EPSILON)
412 {
413 fTrapType = 1; // YZ section is a rectangle ...
414 if (std::abs(fPlanes[2].a + fPlanes[3].a) < DBL_EPSILON &&
415 std::abs(fPlanes[2].c - fPlanes[3].c) < DBL_EPSILON &&
416 fPlanes[2].b == 0 &&
417 fPlanes[3].b == 0)
418 {
419 fTrapType = 2; // ... and XZ section is a isosceles trapezoid
420 fPlanes[2].a = -fPlanes[3].a;
421 fPlanes[2].c = fPlanes[3].c;
422 }
423 if (std::abs(fPlanes[2].a + fPlanes[3].a) < DBL_EPSILON &&
424 std::abs(fPlanes[2].b - fPlanes[3].b) < DBL_EPSILON &&
425 fPlanes[2].c == 0 &&
426 fPlanes[3].c == 0)
427 {
428 fTrapType = 3; // ... and XY section is a isosceles trapezoid
429 fPlanes[2].a = -fPlanes[3].a;
430 fPlanes[2].b = fPlanes[3].b;
431 }
432 }
433}

435///////////////////////////////////////////////////////////////////////
436//
437// Calculate the coef's of the plane p1->p2->p3->p4->p1
438// where the ThreeVectors 1-4 are in anti-clockwise order when viewed
439// from infront of the plane (i.e. from normal direction).
440//
441// Return true if the points are coplanar, false otherwise
442
443G4bool G4Trap::MakePlane( const G4ThreeVector& p1,
444 const G4ThreeVector& p2,
445 const G4ThreeVector& p3,
446 const G4ThreeVector& p4,
447 TrapSidePlane& plane )
448{
449 G4ThreeVector normal = ((p4 - p2).cross(p3 - p1)).unit();
450 if (std::abs(normal.x()) < DBL_EPSILON) normal.setX(0);
451 if (std::abs(normal.y()) < DBL_EPSILON) normal.setY(0);
452 if (std::abs(normal.z()) < DBL_EPSILON) normal.setZ(0);
453 normal = normal.unit();
454
455 G4ThreeVector centre = (p1 + p2 + p3 + p4)*0.25;
456 plane.a = normal.x();
457 plane.b = normal.y();
458 plane.c = normal.z();
459 plane.d = -normal.dot(centre);
460
461 // compute distances and check planarity
462 G4double d1 = std::abs(normal.dot(p1) + plane.d);
463 G4double d2 = std::abs(normal.dot(p2) + plane.d);
464 G4double d3 = std::abs(normal.dot(p3) + plane.d);
465 G4double d4 = std::abs(normal.dot(p4) + plane.d);
466 G4double dmax = std::max(std::max(std::max(d1,d2),d3),d4);
467
468 return (dmax > 1000 * kCarTolerance) ? false : true;
469}

471///////////////////////////////////////////////////////////////////////
472//
473// Get volume
474
475G4double G4Trap::GetCubicVolume()
476{
477 if (fCubicVolume == 0)
478 {
479 G4ThreeVector pt[8];
480 GetVertices(pt);
481
482 G4double dz = pt[4].z() - pt[0].z();
483 G4double dy1 = pt[2].y() - pt[0].y();
484 G4double dx1 = pt[1].x() - pt[0].x();
485 G4double dx2 = pt[3].x() - pt[2].x();
486 G4double dy2 = pt[6].y() - pt[4].y();
487 G4double dx3 = pt[5].x() - pt[4].x();
488 G4double dx4 = pt[7].x() - pt[6].x();
489
490 fCubicVolume = ((dx1 + dx2 + dx3 + dx4)*(dy1 + dy2) +
491 (dx4 + dx3 - dx2 - dx1)*(dy2 - dy1)/3)*dz*0.125;
492 }
493 return fCubicVolume;
494}

496///////////////////////////////////////////////////////////////////////
497//
498// Get surface area
499
500G4double G4Trap::GetSurfaceArea()
501{
502 if (fSurfaceArea == 0)
503 {
504 G4ThreeVector pt[8];
505 G4int iface [6][4] =
506 { {0,1,3,2}, {0,4,5,1}, {2,3,7,6}, {0,2,6,4}, {1,5,7,3}, {4,6,7,5} };
507
508 GetVertices(pt);
509 for (G4int i=0; i<6; ++i)
510 {
511 fSurfaceArea += G4GeomTools::QuadAreaNormal(pt[iface[i][0]],
512 pt[iface[i][1]],
513 pt[iface[i][2]],
514 pt[iface[i][3]]).mag();
515 }
516 }
517 return fSurfaceArea;
518}

532///////////////////////////////////////////////////////////////////////
533//
534// Get bounding box
535
536void G4Trap::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
537{
538 G4ThreeVector pt[8];
539 GetVertices(pt);
540
541 G4double xmin = kInfinity, xmax = -kInfinity;
542 G4double ymin = kInfinity, ymax = -kInfinity;
543 for (G4int i=0; i<8; ++i)
544 {
545 G4double x = pt[i].x();
546 if (x < xmin) xmin = x;
547 if (x > xmax) xmax = x;
548 G4double y = pt[i].y();
549 if (y < ymin) ymin = y;
550 if (y > ymax) ymax = y;
551 }
552
553 G4double dz = GetZHalfLength();
554 pMin.set(xmin,ymin,-dz);
555 pMax.set(xmax,ymax, dz);
556
557 // Check correctness of the bounding box
558 //
559 if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
560 {
561 std::ostringstream message;
562 message << "Bad bounding box (min >= max) for solid: "
563 << GetName() << " !"
564 << "\npMin = " << pMin
565 << "\npMax = " << pMax;
566 G4Exception("G4Trap::BoundingLimits()", "GeomMgt0001",
567 JustWarning, message);
568 DumpInfo();
569 }
570}

572///////////////////////////////////////////////////////////////////////
573//
574// Calculate extent under transform and specified limit
575
576G4bool G4Trap::CalculateExtent( const EAxis pAxis,
577 const G4VoxelLimits& pVoxelLimit,
578 const G4AffineTransform& pTransform,
579 G4double& pMin, G4double& pMax) const
580{
581 G4ThreeVector bmin, bmax;
582 G4bool exist;
583
584 // Check bounding box (bbox)
585 //
586 BoundingLimits(bmin,bmax);
587 G4BoundingEnvelope bbox(bmin,bmax);
588#ifdef G4BBOX_EXTENT
589 if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
590#endif
591 if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
592 {
593 return exist = (pMin < pMax) ? true : false;
594 }
595
596 // Set bounding envelope (benv) and calculate extent
597 //
598 G4ThreeVector pt[8];
599 GetVertices(pt);
600
601 G4ThreeVectorList baseA(4), baseB(4);
602 baseA[0] = pt[0];
603 baseA[1] = pt[1];
604 baseA[2] = pt[3];
605 baseA[3] = pt[2];
606
607 baseB[0] = pt[4];
608 baseB[1] = pt[5];
609 baseB[2] = pt[7];
610 baseB[3] = pt[6];
611
612 std::vector<const G4ThreeVectorList *> polygons(2);
613 polygons[0] = &baseA;
614 polygons[1] = &baseB;
615
616 G4BoundingEnvelope benv(bmin,bmax,polygons);
617 exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
618 return exist;
619}

621///////////////////////////////////////////////////////////////////////
622//
623// Return whether point is inside/outside/on_surface
624
625EInside G4Trap::Inside( const G4ThreeVector& p ) const
626{
627 G4double dz = std::abs(p.z())-fDz;
628 if (dz > halfCarTolerance) return kOutside;
629
630 switch (fTrapType)
631 {
632 case 0: // General case
633 {
634 G4double dy1 = fPlanes[0].b*p.y()+fPlanes[0].c*p.z()+fPlanes[0].d;
635 G4double dy2 = fPlanes[1].b*p.y()+fPlanes[1].c*p.z()+fPlanes[1].d;
636 G4double dy = std::max(dz,std::max(dy1,dy2));
637
638 G4double dx1 = fPlanes[2].a*p.x()+fPlanes[2].b*p.y()
639 + fPlanes[2].c*p.z()+fPlanes[2].d;
640 G4double dx2 = fPlanes[3].a*p.x()+fPlanes[3].b*p.y()
641 + fPlanes[3].c*p.z()+fPlanes[3].d;
642 G4double dist = std::max(dy,std::max(dx1,dx2));
643
644 if (dist > halfCarTolerance) return kOutside;
645 return (dist > -halfCarTolerance) ? kSurface : kInside;
646 }
647 case 1: // YZ section is a rectangle
648 {
649 G4double dy = std::max(dz,std::abs(p.y())+fPlanes[1].d);
650 G4double dx1 = fPlanes[2].a*p.x()+fPlanes[2].b*p.y()
651 + fPlanes[2].c*p.z()+fPlanes[2].d;
652 G4double dx2 = fPlanes[3].a*p.x()+fPlanes[3].b*p.y()
653 + fPlanes[3].c*p.z()+fPlanes[3].d;
654 G4double dist = std::max(dy,std::max(dx1,dx2));
655
656 if (dist > halfCarTolerance) return kOutside;
657 return (dist > -halfCarTolerance) ? kSurface : kInside;
658 }
659 case 2: // YZ section is a rectangle and
660 { // XZ section is an isosceles trapezoid
661 G4double dy = std::max(dz,std::abs(p.y())+fPlanes[1].d);
662 G4double dx = fPlanes[3].a*std::abs(p.x())
663 + fPlanes[3].c*p.z()+fPlanes[3].d;
664 G4double dist = std::max(dy,dx);
665
666 if (dist > halfCarTolerance) return kOutside;
667 return (dist > -halfCarTolerance) ? kSurface : kInside;
668 }
669 case 3: // YZ section is a rectangle and
670 { // XY section is an isosceles trapezoid
671 G4double dy = std::max(dz,std::abs(p.y())+fPlanes[1].d);
672 G4double dx = fPlanes[3].a*std::abs(p.x())
673 + fPlanes[3].b*p.y()+fPlanes[3].d;
674 G4double dist = std::max(dy,dx);
675
676 if (dist > halfCarTolerance) return kOutside;
677 return (dist > -halfCarTolerance) ? kSurface : kInside;
678 }
679 }
680 return kOutside;
681}

683///////////////////////////////////////////////////////////////////////
684//
685// Determine side, and return corresponding normal
686
687G4ThreeVector G4Trap::SurfaceNormal( const G4ThreeVector& p ) const
688{
689 G4int nsurf = 0; // number of surfaces where p is placed
690 G4double nx = 0, ny = 0, nz = 0;
691 G4double dz = std::abs(p.z()) - fDz;
692 if (std::abs(dz) <= halfCarTolerance)
693 {
694 nz = (p.z() < 0) ? -1 : 1;
695 ++nsurf;
696 }
697
698 switch (fTrapType)
699 {
700 case 0: // General case
701 {
702 for (G4int i=0; i<2; ++i)
703 {
704 G4double dy = fPlanes[i].b*p.y() + fPlanes[i].c*p.z() + fPlanes[i].d;
705 if (std::abs(dy) > halfCarTolerance) continue;
706 ny = fPlanes[i].b;
707 nz += fPlanes[i].c;
708 ++nsurf;
709 break;
710 }
711 for (G4int i=2; i<4; ++i)
712 {
713 G4double dx = fPlanes[i].a*p.x() +
714 fPlanes[i].b*p.y() + fPlanes[i].c*p.z() + fPlanes[i].d;
715 if (std::abs(dx) > halfCarTolerance) continue;
716 nx = fPlanes[i].a;
717 ny += fPlanes[i].b;
718 nz += fPlanes[i].c;
719 ++nsurf;
720 break;
721 }
722 break;
723 }
724 case 1: // YZ section is a rectangle
725 {
726 G4double dy = std::abs(p.y()) + fPlanes[1].d;
727 if (std::abs(dy) <= halfCarTolerance) ny = (p.y() < 0) ? -1 : 1;
728 for (G4int i=2; i<4; ++i)
729 {
730 G4double dx = fPlanes[i].a*p.x() +
731 fPlanes[i].b*p.y() + fPlanes[i].c*p.z() + fPlanes[i].d;
732 if (std::abs(dx) > halfCarTolerance) continue;
733 nx = fPlanes[i].a;
734 ny += fPlanes[i].b;
735 nz += fPlanes[i].c;
736 ++nsurf;
737 break;
738 }
739 break;
740 }
741 case 2: // YZ section is a rectangle and
742 { // XZ section is an isosceles trapezoid
743 G4double dy = std::abs(p.y()) + fPlanes[1].d;
744 if (std::abs(dy) <= halfCarTolerance) ny = (p.y() < 0) ? -1 : 1;
745 G4double dx = fPlanes[3].a*std::abs(p.x()) +
746 fPlanes[3].c*p.z() + fPlanes[3].d;
747 if (std::abs(dx) <= halfCarTolerance)
748 {
749 nx = (p.x() < 0) ? -fPlanes[3].a : fPlanes[3].a;
750 nz += fPlanes[3].c;
751 ++nsurf;
752 }
753 break;
754 }
755 case 3: // YZ section is a rectangle and
756 { // XY section is an isosceles trapezoid
757 G4double dy = std::abs(p.y()) + fPlanes[1].d;
758 if (std::abs(dy) <= halfCarTolerance) ny = (p.y() < 0) ? -1 : 1;
759 G4double dx = fPlanes[3].a*std::abs(p.x()) +
760 fPlanes[3].b*p.y() + fPlanes[3].d;
761 if (std::abs(dx) <= halfCarTolerance)
762 {
763 nx = (p.x() < 0) ? -fPlanes[3].a : fPlanes[3].a;
764 ny += fPlanes[3].b;
765 ++nsurf;
766 }
767 break;
768 }
769 }
770
771 // Return normal
772 //
773 if (nsurf == 1) return G4ThreeVector(nx,ny,nz);
774 else if (nsurf != 0) return G4ThreeVector(nx,ny,nz).unit(); // edge or corner
775 else
776 {
777 // Point is not on the surface
778 //
779#ifdef G4CSGDEBUG
780 std::ostringstream message;
781 G4int oldprc = message.precision(16);
782 message << "Point p is not on surface (!?) of solid: "
783 << GetName() << G4endl;
784 message << "Position:\n";
785 message << " p.x() = " << p.x()/mm << " mm\n";
786 message << " p.y() = " << p.y()/mm << " mm\n";
787 message << " p.z() = " << p.z()/mm << " mm";
788 G4cout.precision(oldprc) ;
789 G4Exception("G4Trap::SurfaceNormal(p)", "GeomSolids1002",
790 JustWarning, message );
791 DumpInfo();
792#endif
793 return ApproxSurfaceNormal(p);
794 }
795}

797///////////////////////////////////////////////////////////////////////
798//
799// Algorithm for SurfaceNormal() following the original specification
800// for points not on the surface
801
802G4ThreeVector G4Trap::ApproxSurfaceNormal( const G4ThreeVector& p ) const
803{
804 G4double dist = -DBL_MAX;
805 G4int iside = 0;
806 for (G4int i=0; i<4; ++i)
807 {
808 G4double d = fPlanes[i].a*p.x() +
809 fPlanes[i].b*p.y() +
810 fPlanes[i].c*p.z() + fPlanes[i].d;
811 if (d > dist) { dist = d; iside = i; }
812 }
813
814 G4double distz = std::abs(p.z()) - fDz;
815 if (dist > distz)
816 return G4ThreeVector(fPlanes[iside].a, fPlanes[iside].b, fPlanes[iside].c);
817 else
818 return G4ThreeVector(0, 0, (p.z() < 0) ? -1 : 1);
819}

821///////////////////////////////////////////////////////////////////////
822//
823// Calculate distance to shape from outside
824// - return kInfinity if no intersection
825
826G4double G4Trap::DistanceToIn(const G4ThreeVector& p,
827 const G4ThreeVector& v ) const
828{
829 // Z intersections
830 //
831 if ((std::abs(p.z()) - fDz) >= -halfCarTolerance && p.z()*v.z() >= 0)
832 return kInfinity;
833 G4double invz = (-v.z() == 0) ? DBL_MAX : -1./v.z();
834 G4double dz = (invz < 0) ? fDz : -fDz;
835 G4double tzmin = (p.z() + dz)*invz;
836 G4double tzmax = (p.z() - dz)*invz;
837
838 // Y intersections
839 //
840 G4double tymin = 0, tymax = DBL_MAX;
841 G4int i = 0;
842 for ( ; i<2; ++i)
843 {
844 G4double cosa = fPlanes[i].b*v.y() + fPlanes[i].c*v.z();
845 G4double dist = fPlanes[i].b*p.y() + fPlanes[i].c*p.z() + fPlanes[i].d;
846 if (dist >= -halfCarTolerance)
847 {
848 if (cosa >= 0) return kInfinity;
849 G4double tmp = -dist/cosa;
850 if (tymin < tmp) tymin = tmp;
851 }
852 else if (cosa > 0)
853 {
854 G4double tmp = -dist/cosa;
855 if (tymax > tmp) tymax = tmp;
856 }
857 }
858
859 // Z intersections
860 //
861 G4double txmin = 0, txmax = DBL_MAX;
862 for ( ; i<4; ++i)
863 {
864 G4double cosa = fPlanes[i].a*v.x()+fPlanes[i].b*v.y()+fPlanes[i].c*v.z();
865 G4double dist = fPlanes[i].a*p.x()+fPlanes[i].b*p.y()+fPlanes[i].c*p.z() +
866 fPlanes[i].d;
867 if (dist >= -halfCarTolerance)
868 {
869 if (cosa >= 0) return kInfinity;
870 G4double tmp = -dist/cosa;
871 if (txmin < tmp) txmin = tmp;
872 }
873 else if (cosa > 0)
874 {
875 G4double tmp = -dist/cosa;
876 if (txmax > tmp) txmax = tmp;
877 }
878 }
879
880 // Find distance
881 //
882 G4double tmin = std::max(std::max(txmin,tymin),tzmin);
883 G4double tmax = std::min(std::min(txmax,tymax),tzmax);
884
885 if (tmax <= tmin + halfCarTolerance) return kInfinity; // touch or no hit
886 return (tmin < halfCarTolerance ) ? 0. : tmin;
887}

946////////////////////////////////////////////////////////////////////////////
947//
948// Calculate distance to surface of shape from inside and
949// find normal at exit point, if required
950// - when leaving the surface, return 0
951
952G4double G4Trap::DistanceToOut(const G4ThreeVector& p, const G4ThreeVector& v,
953 const G4bool calcNorm,
954 G4bool *validNorm, G4ThreeVector *n) const
955{
956 // Z intersections
957 //
958 if ((std::abs(p.z()) - fDz) >= -halfCarTolerance && p.z()*v.z() > 0)
959 {
960 if (calcNorm)
961 {
962 *validNorm = true;
963 n->set(0, 0, (p.z() < 0) ? -1 : 1);
964 }
965 return 0;
966 }
967 G4double vz = v.z();
968 G4double tmax = (vz == 0) ? DBL_MAX : (std::copysign(fDz,vz) - p.z())/vz;
969 G4int iside = (vz < 0) ? -4 : -2; // little trick: (-4+3)=-1, (-2+3)=+1
970
971 // Y intersections
972 //
973 G4int i = 0;
974 for ( ; i<2; ++i)
975 {
976 G4double cosa = fPlanes[i].b*v.y() + fPlanes[i].c*v.z();
977 if (cosa > 0)
978 {
979 G4double dist = fPlanes[i].b*p.y() + fPlanes[i].c*p.z() + fPlanes[i].d;
980 if (dist >= -halfCarTolerance)
981 {
982 if (calcNorm)
983 {
984 *validNorm = true;
985 n->set(0, fPlanes[i].b, fPlanes[i].c);
986 }
987 return 0;
988 }
989 G4double tmp = -dist/cosa;
990 if (tmax > tmp) { tmax = tmp; iside = i; }
991 }
992 }
993
994 // X intersections
995 //
996 for ( ; i<4; ++i)
997 {
998 G4double cosa = fPlanes[i].a*v.x()+fPlanes[i].b*v.y()+fPlanes[i].c*v.z();
999 if (cosa > 0)
1000 {
1001 G4double dist = fPlanes[i].a*p.x() +
1002 fPlanes[i].b*p.y() + fPlanes[i].c*p.z() + fPlanes[i].d;
1003 if (dist >= -halfCarTolerance)
1004 {
1005 if (calcNorm)
1006 {
1007 *validNorm = true;
1008 n->set(fPlanes[i].a, fPlanes[i].b, fPlanes[i].c);
1009 }
1010 return 0;
1011 }
1012 G4double tmp = -dist/cosa;
1013 if (tmax > tmp) { tmax = tmp; iside = i; }
1014 }
1015 }
1016
1017 // Set normal, if required, and return distance
1018 //
1019 if (calcNorm)
1020 {
1021 *validNorm = true;
1022 if (iside < 0)
1023 n->set(0, 0, iside + 3); // (-4+3)=-1, (-2+3)=+1
1024 else
1025 n->set(fPlanes[iside].a, fPlanes[iside].b, fPlanes[iside].c);
1026 }
1027 return tmax;
1028}

1174/////////////////////////////////////////////////////////////////////////
1175//
1176// Generate random point on the surface
1177
1178G4ThreeVector G4Trap::GetPointOnSurface() const
1179{
1180 G4ThreeVector pt[8];
1181 G4int iface [6][4] =
1182 { {0,1,3,2}, {0,4,5,1}, {2,3,7,6}, {0,2,6,4}, {1,5,7,3}, {4,6,7,5} };
1183 G4double sface[6];
1184
1185 GetVertices(pt);
1186 G4double stotal = 0;
1187 for (G4int i=0; i<6; ++i)
1188 {
1189 G4double ss = G4GeomTools::QuadAreaNormal(pt[iface[i][0]],
1190 pt[iface[i][1]],
1191 pt[iface[i][2]],
1192 pt[iface[i][3]]).mag();
1193 stotal += ss;
1194 sface[i] = stotal;
1195 }
1196
1197 // Select face
1198 //
1199 G4double select = stotal*G4UniformRand();
1200 G4int k = 5;
1201 if (select <= sface[4]) k = 4;
1202 if (select <= sface[3]) k = 3;
1203 if (select <= sface[2]) k = 2;
1204 if (select <= sface[1]) k = 1;
1205 if (select <= sface[0]) k = 0;
1206
1207 // Select sub-triangle
1208 //
1209 G4int i0 = iface[k][0];
1210 G4int i1 = iface[k][1];
1211 G4int i2 = iface[k][2];
1212 G4int i3 = iface[k][3];
1213 G4double s1 = G4GeomTools::TriangleAreaNormal(pt[i0],pt[i1],pt[i3]).mag();
1214 G4double s2 = G4GeomTools::TriangleAreaNormal(pt[i2],pt[i1],pt[i3]).mag();
1215 if ((s1+s2)*G4UniformRand() > s1) i0 = i2;
1216
1217 // Generate point
1218 //
1219 G4double u = G4UniformRand();
1220 G4double v = G4UniformRand();
1221 if (u + v > 1.) { u = 1. - u; v = 1. - v; }
1222 return (1.-u-v)*pt[i0] + u*pt[i1] + v*pt[i3];
1223}
