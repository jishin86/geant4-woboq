38#ifndef G4PVDIVISION_FACTORY_HH
39#define G4PVDIVISION_FACTORY_HH
40
41#include "geomdefs.hh"
42#include "G4VPVDivisionFactory.hh"
43
44class G4LogicalVolume;
45
46class G4PVDivisionFactory : public G4VPVDivisionFactory
47{
48 public: // with description
49
50 virtual ~G4PVDivisionFactory();
51
52 virtual G4VPhysicalVolume* CreatePVDivision(
53 const G4String& pName,
54 G4LogicalVolume* pLogical,
55 G4LogicalVolume* pMother,
56 const EAxis pAxis,
57 const G4int nReplicas,
58 const G4double width,
59 const G4double offset );
60 // Create division - with number of divisions and width.
61
62 virtual G4VPhysicalVolume* CreatePVDivision(
63 const G4String& pName,
64 G4LogicalVolume* pLogical,
65 G4LogicalVolume* pMotherLogical,
66 const EAxis pAxis,
67 const G4int nReplicas,
68 const G4double offset );
69 // Create division - with number of divisions.
70
71 virtual G4VPhysicalVolume* CreatePVDivision(
72 const G4String& pName,
73 G4LogicalVolume* pLogical,
74 G4LogicalVolume* pMotherLogical,
75 const EAxis pAxis,
76 const G4double width,
77 const G4double offset );
78 // Create division - with width.
79
80 virtual G4VPhysicalVolume* CreatePVDivision(
81 const G4String& pName,
82 G4LogicalVolume* pLogical,
83 G4LogicalVolume* pMotherLogical,
84 const G4VPVParameterisation* param);
85 // Create division - with parameterisation.
86
87 virtual G4bool IsPVDivision(const G4VPhysicalVolume* pv) const;
88 // Returns true if pv is division.
89
90 static G4PVDivisionFactory* GetInstance();
91 // Create the unique instance of the singleton.
92
93 protected:
94
95 G4PVDivisionFactory();
96
97};

99#endif
