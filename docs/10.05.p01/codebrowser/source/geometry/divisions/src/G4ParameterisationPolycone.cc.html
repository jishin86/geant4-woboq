43//-----------------------------------------------------------------------
44G4VParameterisationPolycone::
45G4VParameterisationPolycone( EAxis axis, G4int nDiv, G4double width,
46 G4double offset, G4VSolid* msolid,
47 DivisionType divType )
48 : G4VDivisionParameterisation( axis, nDiv, width, offset, divType, msolid )
49{
50#ifdef G4MULTITHREADED
51 std::ostringstream message;
52 message << "Divisions for G4Polycone currently NOT supported in MT-mode."
53 << G4endl
54 << "Sorry! Solid: " << msolid->GetName();
55 G4Exception("G4VParameterisationPolycone::G4VParameterisationPolycone()",
56 "GeomDiv0001", FatalException, message);
57#endif
58 G4Polycone* msol = (G4Polycone*)(msolid);
59 if (msolid->GetEntityType() == "G4ReflectedSolid")
60 {
61 // Get constituent solid
62 G4VSolid* mConstituentSolid
63 = ((G4ReflectedSolid*)msolid)->GetConstituentMovedSolid();
64 msol = (G4Polycone*)(mConstituentSolid);
65
66 // Get parameters
67 G4int nofZplanes = msol->GetOriginalParameters()->Num_z_planes;
68 G4double* zValues = msol->GetOriginalParameters()->Z_values;
69 G4double* rminValues = msol->GetOriginalParameters()->Rmin;
70 G4double* rmaxValues = msol->GetOriginalParameters()->Rmax;
71
72 // Invert z values
73 G4double* zValuesRefl = new double[nofZplanes];
74 for (G4int i=0; i<nofZplanes; i++) zValuesRefl[i] = - zValues[i];
75
76 G4Polycone* newSolid
77 = new G4Polycone(msol->GetName(),
78 msol->GetStartPhi(),
79 msol->GetEndPhi() - msol->GetStartPhi(),
80 nofZplanes, zValuesRefl, rminValues, rmaxValues);
81
82 delete [] zValuesRefl;
83
84 msol = newSolid;
85 fmotherSolid = newSolid;
86 fReflectedSolid = true;
87 fDeleteSolid = true;
88 }
89}

97G4ParameterisationPolyconeRho::
98G4ParameterisationPolyconeRho( EAxis axis, G4int nDiv,
99 G4double width, G4double offset,
100 G4VSolid* msolid, DivisionType divType )
101 : G4VParameterisationPolycone( axis, nDiv, width, offset, msolid, divType )
102{
103 CheckParametersValidity();
104 SetType( "DivisionPolyconeRho" );
105
106 G4Polycone* msol = (G4Polycone*)(fmotherSolid);
107 G4PolyconeHistorical* origparamMother = msol->GetOriginalParameters();
108
109 if( divType == DivWIDTH )
110 {
111 fnDiv = CalculateNDiv( origparamMother->Rmax[0]
112 - origparamMother->Rmin[0], width, offset );
113 }
114 else if( divType == DivNDIV )
115 {
116 fwidth = CalculateWidth( origparamMother->Rmax[0]
117 - origparamMother->Rmin[0], nDiv, offset );
118 }
119
120#ifdef G4DIVDEBUG
121 if( verbose >= -1 )
122 {
123 G4cout << " G4ParameterisationPolyconeRho - # divisions " << fnDiv
124 << " = " << nDiv << G4endl
125 << " Offset " << foffset << " = " << offset << G4endl
126 << " Width " << fwidth << " = " << width << G4endl;
127 }
128#endif
129}

136//---------------------------------------------------------------------
137void G4ParameterisationPolyconeRho::CheckParametersValidity()
138{
139 G4VDivisionParameterisation::CheckParametersValidity();
140
141 G4Polycone* msol = (G4Polycone*)(fmotherSolid);
142
143 if( fDivisionType == DivNDIVandWIDTH || fDivisionType == DivWIDTH )
144 {
145 std::ostringstream message;
146 message << "In solid " << msol->GetName() << G4endl
147 << "Division along R will be done with a width "
148 << "different for each solid section." << G4endl
149 << "WIDTH will not be used !";
150 G4Exception("G4VParameterisationPolycone::CheckParametersValidity()",
151 "GeomDiv1001", JustWarning, message);
152 }
153 if( foffset != 0. )
154 {
155 std::ostringstream message;
156 message << "In solid " << msol->GetName() << G4endl
157 << "Division along R will be done with a width "
158 << "different for each solid section." << G4endl
159 << "OFFSET will not be used !";
160 G4Exception("G4VParameterisationPolycone::CheckParametersValidity()",
161 "GeomDiv1001", JustWarning, message);
162 }
163}

209//---------------------------------------------------------------------
210void
211G4ParameterisationPolyconeRho::
212ComputeDimensions( G4Polycone& pcone, const G4int copyNo,
213 const G4VPhysicalVolume* ) const
214{
215 G4Polycone* msol = (G4Polycone*)(fmotherSolid);
216
217 G4PolyconeHistorical* origparamMother = msol->GetOriginalParameters();
218 G4PolyconeHistorical origparam( *origparamMother );
219 G4int nZplanes = origparamMother->Num_z_planes;
220
221 G4double width = 0.;
222 for( G4int ii = 0; ii < nZplanes; ii++ )
223 {
224 width = CalculateWidth( origparamMother->Rmax[ii]
225 - origparamMother->Rmin[ii], fnDiv, foffset );
226 origparam.Rmin[ii] = origparamMother->Rmin[ii]+foffset+width*copyNo;
227 origparam.Rmax[ii] = origparamMother->Rmin[ii]+foffset+width*(copyNo+1);
228 }
229
230 pcone.SetOriginalParameters(&origparam); // copy values & transfer pointers
231 pcone.Reset(); // reset to new solid parameters
232
233#ifdef G4DIVDEBUG
234 if( verbose >= -2 )
235 {
236 G4cout << "G4ParameterisationPolyconeRho::ComputeDimensions()" << G4endl
237 << "-- Parametrised pcone copy-number: " << copyNo << G4endl;
238 pcone.DumpInfo();
239 }
240#endif
241}

244G4ParameterisationPolyconePhi::
245G4ParameterisationPolyconePhi( EAxis axis, G4int nDiv,
246 G4double width, G4double offset,
247 G4VSolid* msolid, DivisionType divType )
248 : G4VParameterisationPolycone( axis, nDiv, width, offset, msolid, divType )
249{
250 CheckParametersValidity();
251 SetType( "DivisionPolyconePhi" );
252
253 G4Polycone* msol = (G4Polycone*)(fmotherSolid);
254 G4double deltaPhi = msol->GetEndPhi() - msol->GetStartPhi();
255
256 if( divType == DivWIDTH )
257 {
258 fnDiv = CalculateNDiv( deltaPhi, width, offset );
259 }
260 else if( divType == DivNDIV )
261 {
262 fwidth = CalculateWidth( deltaPhi, nDiv, offset );
263 }
264
265#ifdef G4DIVDEBUG
266 if( verbose >= 1 )
267 {
268 G4cout << " G4ParameterisationPolyconePhi - # divisions " << fnDiv
269 << " = " << nDiv << G4endl
270 << " Offset " << foffset/deg << " = " << offset/deg << G4endl
271 << " Width " << fwidth/deg << " = " << width/deg << G4endl;
272 }
273#endif
274}

313#ifdef G4DIVDEBUG

350G4ParameterisationPolyconeZ::
351G4ParameterisationPolyconeZ( EAxis axis, G4int nDiv,
352 G4double width, G4double offset,
353 G4VSolid* msolid, DivisionType divType)
354 : G4VParameterisationPolycone( axis, nDiv, width, offset, msolid, divType ),
355 fNSegment(0),
356 fOrigParamMother(((G4Polycone*)fmotherSolid)->GetOriginalParameters())
357{
358
359 CheckParametersValidity();
360 SetType( "DivisionPolyconeZ" );
361
362 if( divType == DivWIDTH )
363 {
364 fnDiv =
365 CalculateNDiv( fOrigParamMother->Z_values[fOrigParamMother->Num_z_planes-1]
366 - fOrigParamMother->Z_values[0] , width, offset );
367 }
368 else if( divType == DivNDIV )
369 {
370 fwidth =
371 CalculateNDiv( fOrigParamMother->Z_values[fOrigParamMother->Num_z_planes-1]
372 - fOrigParamMother->Z_values[0] , nDiv, offset );
373 }
374
375#ifdef G4DIVDEBUG
376 if( verbose >= 1 )

378 G4cout << " G4ParameterisationPolyconeZ - # divisions " << fnDiv << " = "
379 << nDiv << G4endl
380 << " Offset " << foffset << " = " << offset << G4endl
381 << " Width " << fwidth << " = " << width << G4endl;

391//------------------------------------------------------------------------
392G4double G4ParameterisationPolyconeZ::GetR(G4double z,
393 G4double z1, G4double r1,
394 G4double z2, G4double r2) const
395{
396 // Linear parameterisation:
397 // r = az + b
398 // a = (r1 - r2)/(z1-z2)
399 // b = r1 - a*z1
400
401 return (r1-r2)/(z1-z2)*z + ( r1 - (r1-r2)/(z1-z2)*z1 ) ;
402}

404//------------------------------------------------------------------------
405G4double G4ParameterisationPolyconeZ::GetRmin(G4double z, G4int nseg) const

407// Get Rmin in the given z position for the given polycone segment

435//---------------------------------------------------------------------
436void G4ParameterisationPolyconeZ::CheckParametersValidity()
437{
438 G4VDivisionParameterisation::CheckParametersValidity();
439
440 // Division will be following the mother polycone segments
441 if( fDivisionType == DivNDIV ) {
442 if( fnDiv > fOrigParamMother->Num_z_planes-1 ) {
443 std::ostringstream error;
444 error << "Configuration not supported." << G4endl
445 << "Division along Z will be done by splitting in the defined"
446 << G4endl
447 << "Z planes, i.e, the number of division would be: "
448 << fOrigParamMother->Num_z_planes-1
449 << ", instead of: " << fnDiv << " !";
450 G4Exception("G4ParameterisationPolyconeZ::CheckParametersValidity()",
451 "GeomDiv0001", FatalException, error);
452 }
453 }
454
455 // Division will be done within one polycone segment
456 // with applying given width and offset
457 if( fDivisionType == DivNDIVandWIDTH || fDivisionType == DivWIDTH ) {
458 // Check if divided region does not span over more
459 // than one z segment
460
461 G4int isegstart = -1; // number of the segment containing start position
462 G4int isegend = -1; // number of the segment containing end position
463
464 if ( ! fReflectedSolid ) {
465 // The start/end position of the divided region
466 G4double zstart
467 = fOrigParamMother->Z_values[0] + foffset;
468 G4double zend
469 = fOrigParamMother->Z_values[0] + foffset + fnDiv* fwidth;
470
471 G4int counter = 0;
472 while ( isegend < 0 && counter < fOrigParamMother->Num_z_planes-1 ) {
473 // first segment
474 if ( zstart >= fOrigParamMother->Z_values[counter] &&
475 zstart < fOrigParamMother->Z_values[counter+1] ) {
476 isegstart = counter;
477 }
478 // last segment
479 if ( zend > fOrigParamMother->Z_values[counter] &&
480 zend <= fOrigParamMother->Z_values[counter+1] ) {
481 isegend = counter;
482 }
483 ++counter;
484 } // Loop checking, 06.08.2015, G.Cosmo
485 }
486 else {
487 // The start/end position of the divided region
488 G4double zstart
489 = fOrigParamMother->Z_values[0] - foffset;
490 G4double zend
491 = fOrigParamMother->Z_values[0] - ( foffset + fnDiv* fwidth);
492
493 G4int counter = 0;
494 while ( isegend < 0 && counter < fOrigParamMother->Num_z_planes-1 ) {
495 // first segment
496 if ( zstart <= fOrigParamMother->Z_values[counter] &&
497 zstart > fOrigParamMother->Z_values[counter+1] ) {
498 isegstart = counter;
499 }
500 // last segment
501 if ( zend < fOrigParamMother->Z_values[counter] &&
502 zend >= fOrigParamMother->Z_values[counter+1] ) {
503 isegend = counter;
504 }
505 ++counter;
506 } // Loop checking, 06.08.2015, G.Cosmo
507 }
508
509
510 if ( isegstart != isegend ) {
511 std::ostringstream message;
512 message << "Condiguration not supported." << G4endl
513 << "Division with user defined width." << G4endl
514 << "Solid " << fmotherSolid->GetName() << G4endl
515 << "Divided region is not between two z planes.";
516 G4Exception("G4ParameterisationPolyconeZ::CheckParametersValidity()",
517 "GeomDiv0001", FatalException, message);
518 }
519
520 fNSegment = isegstart;
521 }
522}

524//---------------------------------------------------------------------
525void
526G4ParameterisationPolyconeZ::
527ComputeTransformation( const G4int copyNo, G4VPhysicalVolume* physVol) const
528{
529 if ( fDivisionType == DivNDIV ) {
530 // The position of the centre of copyNo-th mother polycone segment
531 G4double posi
532 = ( fOrigParamMother->Z_values[copyNo]
533 + fOrigParamMother->Z_values[copyNo+1])/2;
534 physVol->SetTranslation( G4ThreeVector(0, 0, posi) );
535 }
536
537 if ( fDivisionType == DivNDIVandWIDTH || fDivisionType == DivWIDTH ) {
538 // The position of the centre of copyNo-th division
539 G4double posi = fOrigParamMother->Z_values[0];
540
541 if ( ! fReflectedSolid )
542 posi += foffset + (2*copyNo + 1) * fwidth/2.;
543 else
544 posi -= foffset + (2*copyNo + 1) * fwidth/2.;
545
546 physVol->SetTranslation( G4ThreeVector(0, 0, posi) );
547 }
548
549 //----- calculate rotation matrix: unit
550
551#ifdef G4DIVDEBUG
552 if( verbose >= 2 )
553 {
554 G4cout << " G4ParameterisationPolyconeZ - position: " << posi << G4endl
555 << " copyNo: " << copyNo << " - foffset: " << foffset/deg
556 << " - fwidth: " << fwidth/deg << G4endl;
557 }
558#endif
559
560 ChangeRotMatrix( physVol );
561
562#ifdef G4DIVDEBUG
563 if( verbose >= 2 )
564 {
565 G4cout << std::setprecision(8) << " G4ParameterisationPolyconeZ "
566 << copyNo << G4endl
567 << " Position: " << origin << " - Width: " << fwidth
568 << " - Axis: " << faxis << G4endl;
569 }
570#endif
571}

573//---------------------------------------------------------------------
574void
575G4ParameterisationPolyconeZ::
576ComputeDimensions( G4Polycone& pcone, const G4int copyNo,
577 const G4VPhysicalVolume* ) const
578{
579
580 // Define division solid
581 G4PolyconeHistorical origparam;
582 G4int nz = 2;
583 origparam.Num_z_planes = nz;
584 origparam.Start_angle = fOrigParamMother->Start_angle;
585 origparam.Opening_angle = fOrigParamMother->Opening_angle;
586
587 // Define division solid z sections
588 origparam.Z_values = new G4double[nz];
589 origparam.Rmin = new G4double[nz];
590 origparam.Rmax = new G4double[nz];
591
592 if ( fDivisionType == DivNDIV ) {
593 // The position of the centre of copyNo-th mother polycone segment
594 G4double posi = (fOrigParamMother->Z_values[copyNo]
595 + fOrigParamMother->Z_values[copyNo+1])/2;
596
597 origparam.Z_values[0] = fOrigParamMother->Z_values[copyNo] - posi;
598 origparam.Z_values[1] = fOrigParamMother->Z_values[copyNo+1] - posi;
599 origparam.Rmin[0] = fOrigParamMother->Rmin[copyNo];
600 origparam.Rmin[1] = fOrigParamMother->Rmin[copyNo+1];
601 origparam.Rmax[0] = fOrigParamMother->Rmax[copyNo];
602 origparam.Rmax[1] = fOrigParamMother->Rmax[copyNo+1];
603 }
604
605 if ( fDivisionType == DivNDIVandWIDTH || fDivisionType == DivWIDTH ) {
606 if ( ! fReflectedSolid ) {
607 origparam.Z_values[0] = - fwidth/2.;
608 origparam.Z_values[1] = fwidth/2.;
609
610 // The position of the centre of copyNo-th division
611 G4double posi
612 = fOrigParamMother->Z_values[0] + foffset + (2*copyNo + 1) * fwidth/2.;
613
614 // The first and last z sides z values
615 G4double zstart = posi - fwidth/2.;
616 G4double zend = posi + fwidth/2.;
617 origparam.Rmin[0] = GetRmin(zstart, fNSegment);
618 origparam.Rmax[0] = GetRmax(zstart, fNSegment);
619 origparam.Rmin[1] = GetRmin(zend, fNSegment);
620 origparam.Rmax[1] = GetRmax(zend, fNSegment);
621 }
622 else {
623 origparam.Z_values[0] = fwidth/2.;
624 origparam.Z_values[1] = - fwidth/2.;
625
626 // The position of the centre of copyNo-th division
627 G4double posi
628 = fOrigParamMother->Z_values[0] - ( foffset + (2*copyNo + 1) * fwidth/2.);
629
630 // The first and last z sides z values
631 G4double zstart = posi + fwidth/2.;
632 G4double zend = posi - fwidth/2.;
633 origparam.Rmin[0] = GetRmin(zstart, fNSegment);
634 origparam.Rmax[0] = GetRmax(zstart, fNSegment);
635 origparam.Rmin[1] = GetRmin(zend, fNSegment);
636 origparam.Rmax[1] = GetRmax(zend, fNSegment);
637 }
638
639 // It can happen due to rounding errors
640 if ( origparam.Rmin[0] < 0.0 ) origparam.Rmin[0] = 0.0;
641 if ( origparam.Rmin[nz-1] < 0.0 ) origparam.Rmin[1] = 0.0;
642 }
643
644 pcone.SetOriginalParameters(&origparam); // copy values & transfer pointers
645 pcone.Reset(); // reset to new solid parameters
646
647#ifdef G4DIVDEBUG
648 if( verbose >= 2 )
649 {
650 G4cout << "G4ParameterisationPolyconeZ::ComputeDimensions()" << G4endl
651 << "-- Parametrised pcone copy-number: " << copyNo << G4endl;
652 pcone.DumpInfo();
653 }
654#endif
