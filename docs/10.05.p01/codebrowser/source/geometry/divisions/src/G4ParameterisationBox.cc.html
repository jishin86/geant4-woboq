45//--------------------------------------------------------------------------
46G4VParameterisationBox::
47G4VParameterisationBox( EAxis axis, G4int nDiv, G4double width,
48 G4double offset, G4VSolid* msolid,
49 DivisionType divType )
50 : G4VDivisionParameterisation( axis, nDiv, width, offset, divType, msolid )
51{
52 G4Box* msol = (G4Box*)(msolid);
53 if (msolid->GetEntityType() == "G4ReflectedSolid")
54 {
55 // Get constituent solid
56 G4VSolid* mConstituentSolid
57 = ((G4ReflectedSolid*)msolid)->GetConstituentMovedSolid();
58 msol = (G4Box*)(mConstituentSolid);
59 fmotherSolid = msol;
60 fReflectedSolid = true;
61 }
62}

69//--------------------------------------------------------------------------
70G4ParameterisationBoxX::
71G4ParameterisationBoxX( EAxis axis, G4int nDiv, G4double width,
72 G4double offset, G4VSolid* msolid,
73 DivisionType divType )
74 : G4VParameterisationBox( axis, nDiv, width, offset, msolid, divType )
75{
76 CheckParametersValidity();
77 SetType( "DivisionBoxX" );
78
79 G4Box* mbox = (G4Box*)(fmotherSolid);
80 if( divType == DivWIDTH )
81 {
82 fnDiv = CalculateNDiv( 2*mbox->GetXHalfLength(), width, offset );
83 }
84 else if( divType == DivNDIV )
85 {
86 fwidth = CalculateWidth( 2*mbox->GetXHalfLength(), nDiv, offset );
87 }
88#ifdef G4DIVDEBUG
89 if( verbose >= 1 )
90 {
91 G4cout << " G4ParameterisationBoxX - no divisions "
92 << fnDiv << " = " << nDiv << G4endl
93 << " Offset " << foffset << " = " << offset << G4endl
94 << " Width " << fwidth << " = " << width << G4endl;
95 }
96#endif
97}

111//------------------------------------------------------------------------
112void
113G4ParameterisationBoxX::
114ComputeTransformation( const G4int copyNo, G4VPhysicalVolume* physVol ) const
115{
116 G4Box* msol = (G4Box*)(fmotherSolid );
117 G4double mdx = msol->GetXHalfLength( );
118
119 //----- translation
120 G4ThreeVector origin(0.,0.,0.);
121 G4double posi = -mdx + foffset+(copyNo+0.5)*fwidth;
122
123 if( faxis == kXAxis )
124 {
125 origin.setX( posi );
126 }
127 else
128 {
129 std::ostringstream message;
130 message << "Only axes along X are allowed ! Axis: " << faxis;
131 G4Exception("G4ParameterisationBoxX::ComputeTransformation()",
132 "GeomDiv0002", FatalException, message);
133 }
134#ifdef G4DIVDEBUG
135 if( verbose >= 2 )
136 {
137 G4cout << std::setprecision(8) << " G4ParameterisationBoxX: "
138 << copyNo << G4endl
139 << " Position " << origin << " Axis " << faxis << G4endl;
140 }
141#endif
142 //----- set translation
143 physVol->SetTranslation( origin );
144}

146//------------------------------------------------------------------------
147void
148G4ParameterisationBoxX::
149ComputeDimensions( G4Box& box, const G4int,
150 const G4VPhysicalVolume* ) const
151{
152 G4Box* msol = (G4Box*)(fmotherSolid);
153
154 G4double pDx = fwidth/2. - fhgap;
155 G4double pDy = msol->GetYHalfLength();
156 G4double pDz = msol->GetZHalfLength();
157
158 box.SetXHalfLength( pDx );
159 box.SetYHalfLength( pDy );
160 box.SetZHalfLength( pDz );
161
162#ifdef G4DIVDEBUG
163 if( verbose >= 2 )
164 {
165 G4cout << " G4ParameterisationBoxX::ComputeDimensions()" << G4endl
166 << " pDx: " << pDz << G4endl;
167 box.DumpInfo();
168 }
169#endif
170}

214//------------------------------------------------------------------------
215void
216G4ParameterisationBoxY::
217ComputeTransformation( const G4int copyNo, G4VPhysicalVolume* physVol ) const
218{
219 G4Box* msol = (G4Box*)(fmotherSolid);
220 G4double mdy = msol->GetYHalfLength();
221
222 //----- translation
223 G4ThreeVector origin(0.,0.,0.);
224 G4double posi = -mdy + foffset + (copyNo+0.5)*fwidth;
225 if( faxis == kYAxis )
226 {
227 origin.setY( posi );
228 }
229 else
230 {
231 std::ostringstream message;
232 message << "Only axes along Y are allowed ! Axis: " << faxis;
233 G4Exception("G4ParameterisationBoxY::ComputeTransformation()",
234 "GeomDiv0002", FatalException, message);
235 }
236#ifdef G4DIVDEBUG
237 if( verbose >= 2 )
238 {
239 G4cout << std::setprecision(8) << " G4ParameterisationBoxY: "
240 << copyNo << G4endl
241 << " Position " << origin << " Axis " << faxis << G4endl;
242 }
243#endif
244 //----- set translation
245 physVol->SetTranslation( origin );
