45//--------------------------------------------------------------------------
46G4VParameterisationCons::
47G4VParameterisationCons( EAxis axis, G4int nDiv, G4double width,
48 G4double offset, G4VSolid* msolid,
49 DivisionType divType )
50 : G4VDivisionParameterisation( axis, nDiv, width, offset, divType, msolid )
51{
52 G4Cons* msol = (G4Cons*)(msolid);
53 if (msolid->GetEntityType() == "G4ReflectedSolid")
54 {
55 // Get constituent solid
56 G4VSolid* mConstituentSolid
57 = ((G4ReflectedSolid*)msolid)->GetConstituentMovedSolid();
58 msol = (G4Cons*)(mConstituentSolid);
59
60 // Create a new solid with inversed parameters
61 G4Cons* newSolid
62 = new G4Cons(msol->GetName(),
63 msol->GetInnerRadiusPlusZ(), msol->GetOuterRadiusPlusZ(),
64 msol->GetInnerRadiusMinusZ(), msol->GetOuterRadiusMinusZ(),
65 msol->GetZHalfLength(),
66 msol->GetStartPhiAngle(), msol->GetDeltaPhiAngle());
67 msol = newSolid;
68 fmotherSolid = newSolid;
69 fReflectedSolid = true;
70 fDeleteSolid = true;
71 }
72}

79//--------------------------------------------------------------------------
80G4ParameterisationConsRho::
81G4ParameterisationConsRho( EAxis axis, G4int nDiv,
82 G4double width, G4double offset,
83 G4VSolid* msolid, DivisionType divType )
84 : G4VParameterisationCons( axis, nDiv, width, offset, msolid, divType )
85{
86 CheckParametersValidity();
87 SetType( "DivisionConsRho" );
88
89 G4Cons* msol = (G4Cons*)(fmotherSolid);
90 if( msol->GetInnerRadiusPlusZ() == 0. )
91 {
92 std::ostringstream message;
93 message << "OuterRadiusMinusZ = 0" << G4endl
94 << "Width is calculated as that of OuterRadiusMinusZ !";
95 G4Exception("G4ParameterisationConsRho::G4ParameterisationConsRho()",
96 "GeomDiv1001", JustWarning, message);
97 }
98
99 if( divType == DivWIDTH )
100 {
101 fnDiv = CalculateNDiv( msol->GetOuterRadiusMinusZ()
102 - msol->GetInnerRadiusMinusZ(), width, offset );
103 }
104 else if( divType == DivNDIV )
105 {
106 G4Cons* mconsol = (G4Cons*)(msolid);
107 fwidth = CalculateWidth( mconsol->GetOuterRadiusMinusZ()
108 - mconsol->GetInnerRadiusMinusZ(), nDiv, offset );
109 }
110
111#ifdef G4DIVDEBUG
112 if( verbose >= 1 )
113 {
114 G4cout << " G4ParameterisationConsRho - no divisions " << fnDiv << " = "
115 << nDiv << G4endl
116 << " Offset " << foffset << " = " << offset
117 << " - Width " << fwidth << " = " << width << G4endl;
118 }
119#endif
120}

167//--------------------------------------------------------------------------
168void
169G4ParameterisationConsRho::
170ComputeDimensions( G4Cons& cons, const G4int copyNo,
171 const G4VPhysicalVolume* ) const
172{
173 G4Cons* msol = (G4Cons*)(fmotherSolid);
174
175 G4double pRMin1 = msol->GetInnerRadiusMinusZ() + foffset + fwidth * copyNo;
176 G4double pRMax1 = msol->GetInnerRadiusMinusZ() + foffset + fwidth * (copyNo+1);
177
178 //width at Z Plus
179 //- G4double fwidthPlus =
180 // fwidth * ( msol->GetOuterRadiusPlusZ()/ msol->GetInnerRadiusPlusZ())
181 //- / ( msol->GetOuterRadiusMinusZ() - msol->GetInnerRadiusMinusZ());
182 G4double fwidthPlus = CalculateWidth( msol->GetOuterRadiusPlusZ()
183 - msol->GetInnerRadiusPlusZ(), fnDiv, foffset );
184 G4double pRMin2 = msol->GetInnerRadiusPlusZ()
185 + foffset + fwidthPlus * copyNo;
186 G4double pRMax2 = msol->GetInnerRadiusPlusZ()
187 + foffset + fwidthPlus * (copyNo+1);
188 G4double pDz = msol->GetZHalfLength();
189
190 G4double d_half_gap = fhgap * pRMax2 / pRMax1;
191 //- already rotated double pSR = foffset + copyNo*fwidth;
192 G4double pSPhi = msol->GetStartPhiAngle();
193 G4double pDPhi = msol->GetDeltaPhiAngle();;
194
195 cons.SetInnerRadiusMinusZ( pRMin1 + fhgap );
196 cons.SetOuterRadiusMinusZ( pRMax1 - fhgap );
197 cons.SetInnerRadiusPlusZ( pRMin2 + d_half_gap );
198 cons.SetOuterRadiusPlusZ( pRMax2 - d_half_gap );
199 cons.SetZHalfLength( pDz );
200 cons.SetStartPhiAngle( pSPhi, false );
201 cons.SetDeltaPhiAngle( pDPhi );
202
203#ifdef G4DIVDEBUG
204 if( verbose >= 2 )
205 {
206 G4cout << " G4ParameterisationConsRho::ComputeDimensions()" << G4endl
207 << " pRMin: " << pRMin1 << " - pRMax: " << pRMax1 << G4endl;
208 if( verbose >= 4 ) cons.DumpInfo();
209 }
210#endif
211}

256//--------------------------------------------------------------------------
257void
258G4ParameterisationConsPhi::
259ComputeTransformation( const G4int copyNo, G4VPhysicalVolume *physVol ) const
260{
261 //----- translation
262 G4ThreeVector origin(0.,0.,0.);
263 //----- set translation
264 physVol->SetTranslation( origin );
265
266 //----- calculate rotation matrix (so that all volumes point to the centre)
267 G4double posi = foffset + copyNo*fwidth;
268
269#ifdef G4DIVDEBUG
270 if( verbose >= 2 )
271 {
272 G4cout << " G4ParameterisationConsPhi - position: " << posi/deg << G4endl
273 << " Origin: " << origin << " copyNo: " << copyNo
274 << " - foffset: " << foffset/deg
275 << " - fwidth: " << fwidth/deg << G4endl
276 << " - Axis: " << faxis << G4endl;
277 }
278#endif
279
280 ChangeRotMatrix( physVol, -posi );
281}

283//--------------------------------------------------------------------------
284void
285G4ParameterisationConsPhi::
286ComputeDimensions( G4Cons& cons, const G4int,
287 const G4VPhysicalVolume* ) const
288{
289 G4Cons* msol = (G4Cons*)(fmotherSolid);
290
291 G4double pRMin1 = msol->GetInnerRadiusMinusZ();
292 G4double pRMax1 = msol->GetOuterRadiusMinusZ();
293 G4double pRMin2 = msol->GetInnerRadiusPlusZ();
294 G4double pRMax2 = msol->GetOuterRadiusPlusZ();
295 G4double pDz = msol->GetZHalfLength();
296
297 //- already rotated double pSPhi = foffset + copyNo*fwidth;
298 G4double pSPhi = foffset + msol->GetStartPhiAngle() + fhgap;
299 G4double pDPhi = fwidth - 2.*fhgap;
300
301 cons.SetInnerRadiusMinusZ( pRMin1 );
302 cons.SetOuterRadiusMinusZ( pRMax1 );
303 cons.SetInnerRadiusPlusZ( pRMin2 );
304 cons.SetOuterRadiusPlusZ( pRMax2 );
305 cons.SetZHalfLength( pDz );
306 cons.SetStartPhiAngle( pSPhi, false );
307 cons.SetDeltaPhiAngle( pDPhi );
308
309#ifdef G4DIVDEBUG
310 if( verbose >= 2 )
311 {
312 G4cout << " G4ParameterisationConsPhi::ComputeDimensions" << G4endl
313 << " pSPhi: " << pSPhi << " - pDPhi: " << pDPhi << G4endl;
314 if( verbose >= 4 ) cons.DumpInfo();
315 }
316#endif
317}

365G4ParameterisationConsZ::
366ComputeTransformation( const G4int copyNo, G4VPhysicalVolume* physVol ) const
367{

369 G4Cons* motherCons = (G4Cons*)(GetMotherSolid());
370 G4double posi = - motherCons->GetZHalfLength() + OffsetZ()
371 + fwidth/2 + copyNo*fwidth;

373 physVol->SetTranslation( origin );
374
375 //----- calculate rotation matrix: unit

391//--------------------------------------------------------------------------
392void
393G4ParameterisationConsZ::
394ComputeDimensions( G4Cons& cons, const G4int copyNo,
395 const G4VPhysicalVolume* ) const
396{
397 G4Cons* msol = (G4Cons*)(fmotherSolid);
398
399 G4double mHalfLength = msol->GetZHalfLength() - fhgap;
400 G4double aRInner = (msol->GetInnerRadiusPlusZ()
401 - msol->GetInnerRadiusMinusZ()) / (2*mHalfLength);
402 G4double bRInner = (msol->GetInnerRadiusPlusZ()
403 + msol->GetInnerRadiusMinusZ()) / 2;
404 G4double aROuter = (msol->GetOuterRadiusPlusZ()
405 - msol->GetOuterRadiusMinusZ()) / (2*mHalfLength);
406 G4double bROuter = (msol->GetOuterRadiusPlusZ()
407 + msol->GetOuterRadiusMinusZ()) / 2;
408 G4double xMinusZ = -mHalfLength + OffsetZ() + fwidth*copyNo + fhgap;
409 G4double xPlusZ = -mHalfLength + OffsetZ() + fwidth*(copyNo+1) - fhgap;
410 cons.SetInnerRadiusMinusZ( aRInner * xMinusZ + bRInner );
411 cons.SetOuterRadiusMinusZ( aROuter * xMinusZ + bROuter );
412 cons.SetInnerRadiusPlusZ( aRInner * xPlusZ + bRInner );
413 cons.SetOuterRadiusPlusZ( aROuter * xPlusZ + bROuter );
414
415 G4double pDz = fwidth / 2. - fhgap;
416 G4double pSPhi = msol->GetStartPhiAngle();
417 G4double pDPhi = msol->GetDeltaPhiAngle();
418
419 cons.SetZHalfLength( pDz );
420 cons.SetStartPhiAngle( pSPhi, false );
421 cons.SetDeltaPhiAngle( pDPhi );
422
423#ifdef G4DIVDEBUG
424 if( verbose >= 2 )
425 {
426 G4cout << " G4ParameterisationConsZ::ComputeDimensions()" << G4endl
427 << " pDz: " << pDz << G4endl;
428 if( verbose >= 4 ) cons.DumpInfo();
429 }
430#endif
