38// This new field helps to use the class G4PVManager
39//
40G4PVManager G4VPhysicalVolume::subInstanceManager;
41
42// These macros change the references to fields that are now encapsulated
43// in the class G4PVData.
44//
45#define G4MT_rot ((subInstanceManager.offset[instanceID]).frot)
46#define G4MT_trans ((subInstanceManager.offset[instanceID]).ftrans)
47#define G4MT_pvdata (subInstanceManager.offset[instanceID])
48
49// Constructor: init parameters and register in Store
50//
51G4VPhysicalVolume::G4VPhysicalVolume( G4RotationMatrix *pRot,
52 const G4ThreeVector &tlate,
53 const G4String& pName,
54 G4LogicalVolume* pLogical,
55 G4VPhysicalVolume* )
56 : flogical(pLogical),
57 fname(pName), flmother(0)
58{
59 instanceID = subInstanceManager.CreateSubInstance();
60
61 this->SetRotation( pRot ); // G4MT_rot = pRot;
62 this->SetTranslation( tlate ); // G4MT_trans = tlate;
63
64 // Initialize 'Shadow' data structure - for use by object persistency
65 pvdata = new G4PVData();
66 pvdata->frot = pRot;
67 pvdata->ftrans = G4ThreeVector(tlate);
68
69 G4PhysicalVolumeStore::Register(this);
70}
71
72// Fake default constructor - sets only member data and allocates memory
73// for usage restricted to object persistency.
74//
75G4VPhysicalVolume::G4VPhysicalVolume( __void__& )
76 : flogical(0), fname(""), flmother(0), pvdata(0)
77{
78 // Register to store
79 //
80 instanceID = subInstanceManager.CreateSubInstance();
81
82 G4PhysicalVolumeStore::Register(this);
83}
84
85// Destructor - remove from Store
86//
87G4VPhysicalVolume::~G4VPhysicalVolume()
88{
89 delete pvdata;
90 G4PhysicalVolumeStore::DeRegister(this);
91}

93// This method is similar to the constructor. It is used by each worker
94// thread to achieve the same effect as that of the master thread exept
95// to register the new created instance. This method is invoked explicitly.
96// It does not create a new G4VPhysicalVolume instance.
97// It only assign the value for the fields encapsulated by the class G4PVData.
98//
99void G4VPhysicalVolume::
100InitialiseWorker( G4VPhysicalVolume* /*pMasterObject*/,
101 G4RotationMatrix *pRot,
102 const G4ThreeVector &tlate)
103{
104 subInstanceManager.SlaveCopySubInstanceArray();
105
106 this->SetRotation( pRot ); // G4MT_rot = pRot;
107 this->SetTranslation( tlate ); // G4MT_trans = tlate;
108 // G4PhysicalVolumeStore::Register(this);
109}

163G4RotationMatrix* G4VPhysicalVolume::GetObjectRotation() const
164{
165 static G4RotationMatrix aRotM;
166 static G4RotationMatrix IdentityRM;
167
168 G4RotationMatrix* retval = &IdentityRM;
169
170 // Insure against frot being a null pointer
171 if(this->GetRotation())
172 {
173 aRotM = GetRotation()->inverse();
174 retval= &aRotM;
175 }
176 return retval;
177}
178
179G4RotationMatrix G4VPhysicalVolume::GetObjectRotationValue() const
180{
181 G4RotationMatrix aRotM; // Initialised to identity
182
183 // Insure against G4MT_rot being a null pointer
184 if(G4MT_rot)
185 {
186 aRotM= G4MT_rot->inverse();
187 }
188 return aRotM;
189}
