58///////////////////////////////////////////////////////////////////////////
59//
60// Further restrict limits
61// No checks for illegal restrictions
62//
63
64void G4VoxelLimits::AddLimit( const EAxis pAxis,
65 const G4double pMin,
66 const G4double pMax )
67{
68 if ( pAxis == kXAxis )
69 {
70 if ( pMin > fxAxisMin ) fxAxisMin = pMin ;
71 if ( pMax < fxAxisMax ) fxAxisMax = pMax ;
72 }
73 else if ( pAxis == kYAxis )
74 {
75 if ( pMin > fyAxisMin ) fyAxisMin = pMin ;
76 if ( pMax < fyAxisMax ) fyAxisMax = pMax ;
77 }
78 else
79 {
80 assert( pAxis == kZAxis ) ;
81
82 if ( pMin > fzAxisMin ) fzAxisMin = pMin ;
83 if ( pMax < fzAxisMax ) fzAxisMax = pMax ;
84 }
85}

87///////////////////////////////////////////////////////////////////////////
88//
89// ClipToLimits
90//
91// Clip the line segment pStart->pEnd to the volume described by the
92// current limits. Return true if the line remains after clipping,
93// else false, and leave the vectors in an undefined state.
94//
95// Process:
96//
97// Use Cohen-Sutherland clipping in 3D
98// [Fundamentals of Interactive Computer Graphics,Foley & Van Dam]
99//
100
101G4bool G4VoxelLimits::ClipToLimits( G4ThreeVector& pStart,
102 G4ThreeVector& pEnd ) const
103{
104 G4int sCode, eCode ;
105 G4bool remainsAfterClip ;
106
107 // Determine if line is trivially inside (both outcodes==0) or outside
108 // (logical AND of outcodes !=0)
109
110 sCode = OutCode(pStart) ;
111 eCode = OutCode(pEnd) ;
112
113 if ( sCode & eCode )
114 {
115 // Trivially outside, no intersection with region
116
117 remainsAfterClip = false;
118 }
119 else if ( sCode == 0 && eCode == 0 )
120 {
121 // Trivially inside, no intersections
122
123 remainsAfterClip = true ;
124 }
125 else
126 {
127 // Line segment *may* cut volume boundaries
128 // At most, one end point is inside
129
130 G4double x1, y1, z1, x2, y2, z2 ;
131
132 x1 = pStart.x() ;
133 y1 = pStart.y() ;
134 z1 = pStart.z() ;
135
136 x2 = pEnd.x() ;
137 y2 = pEnd.y() ;
138 z2 = pEnd.z() ;
139 /*
140 if( std::abs(x1-x2) < kCarTolerance*kCarTolerance)
141 {
142 G4cout<<"x1 = "<<x1<<"\t"<<"x2 = "<<x2<<G4endl;
143 }
144 if( std::abs(y1-y2) < kCarTolerance*kCarTolerance)
145 {
146 G4cout<<"y1 = "<<y1<<"\t"<<"y2 = "<<y2<<G4endl;
147 }
148 if( std::abs(z1-z2) < kCarTolerance*kCarTolerance)
149 {
150 G4cout<<"z1 = "<<z1<<"\t"<<"z2 = "<<z2<<G4endl;
151 }
152 */
153 while ( sCode != eCode ) // Loop checking, 06.08.2015, G.Cosmo
154 {
155 // Copy vectors to work variables x1-z1,x2-z2
156 // Ensure x1-z1 lies outside volume, swapping vectors and outcodes
157 // if necessary
158
159 if ( sCode )
160 {
161 if ( sCode & 0x01 ) // Clip against fxAxisMin
162 {
163 z1 += (fxAxisMin-x1)*(z2-z1)/(x2-x1);
164 y1 += (fxAxisMin-x1)*(y2-y1)/(x2-x1);
165 x1 = fxAxisMin;
166 }
167 else if ( sCode & 0x02 ) // Clip against fxAxisMax
168 {
169 z1 += (fxAxisMax-x1)*(z2-z1)/(x2-x1);
170 y1 += (fxAxisMax-x1)*(y2-y1)/(x2-x1);
171 x1 = fxAxisMax ;
172 }
173 else if ( sCode & 0x04 ) // Clip against fyAxisMin
174 {
175 x1 += (fyAxisMin-y1)*(x2-x1)/(y2-y1);
176 z1 += (fyAxisMin-y1)*(z2-z1)/(y2-y1);
177 y1 = fyAxisMin;
178 }
179 else if ( sCode & 0x08 ) // Clip against fyAxisMax
180 {
181 x1 += (fyAxisMax-y1)*(x2-x1)/(y2-y1);
182 z1 += (fyAxisMax-y1)*(z2-z1)/(y2-y1);
183 y1 = fyAxisMax;
184 }
185 else if ( sCode & 0x10 ) // Clip against fzAxisMin
186 {
187 x1 += (fzAxisMin-z1)*(x2-x1)/(z2-z1);
188 y1 += (fzAxisMin-z1)*(y2-y1)/(z2-z1);
189 z1 = fzAxisMin;
190 }
191 else if ( sCode & 0x20 ) // Clip against fzAxisMax
192 {
193 x1 += (fzAxisMax-z1)*(x2-x1)/(z2-z1);
194 y1 += (fzAxisMax-z1)*(y2-y1)/(z2-z1);
195 z1 = fzAxisMax;
196 }
197 }
198 if ( eCode ) // Clip 2nd end: repeat of 1st, but 1<>2
199 {
200 if ( eCode & 0x01 ) // Clip against fxAxisMin
201 {
202 z2 += (fxAxisMin-x2)*(z1-z2)/(x1-x2);
203 y2 += (fxAxisMin-x2)*(y1-y2)/(x1-x2);
204 x2 = fxAxisMin;
205 }
206 else if ( eCode & 0x02 ) // Clip against fxAxisMax
207 {
208 z2 += (fxAxisMax-x2)*(z1-z2)/(x1-x2);
209 y2 += (fxAxisMax-x2)*(y1-y2)/(x1-x2);
210 x2 = fxAxisMax;
211 }
212 else if ( eCode & 0x04 ) // Clip against fyAxisMin
213 {
214 x2 += (fyAxisMin-y2)*(x1-x2)/(y1-y2);
215 z2 += (fyAxisMin-y2)*(z1-z2)/(y1-y2);
216 y2 = fyAxisMin;
217 }
218 else if (eCode&0x08) // Clip against fyAxisMax
219 {
220 x2 += (fyAxisMax-y2)*(x1-x2)/(y1-y2);
221 z2 += (fyAxisMax-y2)*(z1-z2)/(y1-y2);
222 y2 = fyAxisMax;
223 }
224 else if ( eCode & 0x10 ) // Clip against fzAxisMin
225 {
226 x2 += (fzAxisMin-z2)*(x1-x2)/(z1-z2);
227 y2 += (fzAxisMin-z2)*(y1-y2)/(z1-z2);
228 z2 = fzAxisMin;
229 }
230 else if ( eCode & 0x20 ) // Clip against fzAxisMax
231 {
232 x2 += (fzAxisMax-z2)*(x1-x2)/(z1-z2);
233 y2 += (fzAxisMax-z2)*(y1-y2)/(z1-z2);
234 z2 = fzAxisMax;
235 }
236 }
237 // G4endl; G4cout<<"x1 = "<<x1<<"\t"<<"x2 = "<<x2<<G4endl<<G4endl;
238 pStart = G4ThreeVector(x1,y1,z1);
239 pEnd = G4ThreeVector(x2,y2,z2);
240 sCode = OutCode(pStart);
241 eCode = OutCode(pEnd);
242 }
243 if ( sCode == 0 && eCode == 0 ) remainsAfterClip = true;
244 else remainsAfterClip = false;
245 }
246 return remainsAfterClip;
247}

249////////////////////////////////////////////////////////////////////////////
250//
251// Calculate the `outcode' for the specified vector:
252// The following bits are set:
253// 0 pVec.x()<fxAxisMin && IsXLimited()
254// 1 pVec.x()>fxAxisMax && IsXLimited()
255// 2 pVec.y()<fyAxisMin && IsYLimited()
256// 3 pVec.y()>fyAxisMax && IsYLimited()
257// 4 pVec.z()<fzAxisMin && IsZLimited()
258// 5 pVec.z()>fzAxisMax && IsZLimited()
259//
260
261G4int G4VoxelLimits::OutCode( const G4ThreeVector& pVec ) const
262{
263 G4int code = 0 ; // The outcode
264
265 if ( IsXLimited() )
266 {
267 if ( pVec.x() < fxAxisMin ) code |= 0x01 ;
268 if ( pVec.x() > fxAxisMax ) code |= 0x02 ;
269 }
270 if ( IsYLimited() )
271 {
272 if ( pVec.y() < fyAxisMin ) code |= 0x04 ;
273 if ( pVec.y() > fyAxisMax ) code |= 0x08 ;
274 }
275 if (IsZLimited())
276 {
277 if ( pVec.z() < fzAxisMin ) code |= 0x10 ;
278 if ( pVec.z() > fzAxisMax ) code |= 0x20 ;
279 }
280 return code;
281}

285std::ostream& operator << (std::ostream& os, const G4VoxelLimits& pLim)
286{
287 os << "{";
288 if (pLim.IsXLimited())
289 {
290 os << "(" << pLim.GetMinXExtent()
291 << "," << pLim.GetMaxXExtent() << ") ";
292 }
293 else
294 {
295 os << "(-,-) ";
296 }
297 if (pLim.IsYLimited())
298 {
299 os << "(" << pLim.GetMinYExtent()
300 << "," << pLim.GetMaxYExtent() << ") ";
301 }
302 else
303 {
304 os << "(-,-) ";
305 }
306 if (pLim.IsZLimited())
307 {
308 os << "(" << pLim.GetMinZExtent()
309 << "," << pLim.GetMaxZExtent() << ")";
310 }
311 else
312 {
313 os << "(-,-)";
314 }
315 os << "}";
316 return os;
