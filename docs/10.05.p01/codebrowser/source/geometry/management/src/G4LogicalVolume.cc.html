70// ********************************************************************
71// Constructor - sets member data and adds to logical Store,
72// voxel pointer for optimisation set to 0 by default.
73// Initialises daughter vector to 0 length.
74// ********************************************************************
75//
76G4LogicalVolume::G4LogicalVolume( G4VSolid* pSolid,
77 G4Material* pMaterial,
78 const G4String& name,
79 G4FieldManager* pFieldMgr,
80 G4VSensitiveDetector* pSDetector,
81 G4UserLimits* pULimits,
82 G4bool optimise )
83 : fDaughters(0,(G4VPhysicalVolume*)0),
84 fVoxel(0), fOptimise(optimise), fRootRegion(false), fLock(false),
85 fSmartless(2.), fVisAttributes(0), fRegion(0), fBiasWeight(1.)
86{
87 // Initialize 'Shadow'/master pointers - for use in copying to workers
88 //
89 fSolid = pSolid;
90 fSensitiveDetector = pSDetector;
91 fFieldManager = pFieldMgr;
92
93 instanceID = subInstanceManager.CreateSubInstance();
94 AssignFieldManager(pFieldMgr);
95
96 G4MT_mass = 0.;
97 G4MT_ccouple = 0;
98
99 SetSolid(pSolid);
100 SetMaterial(pMaterial);
101 SetName(name);
102 SetSensitiveDetector(pSDetector);
103 SetUserLimits(pULimits);
104
105 // Initialize 'Shadow' data structure - for use by object persistency
106 //
107 lvdata = new G4LVData();
108 lvdata->fSolid = pSolid;
109 lvdata->fMaterial = pMaterial;
110
111 //
112 // Add to store
113 //
114 G4LogicalVolumeStore::Register(this);
115}

157// ********************************************************************
158// InitialiseWorker
159//
160// This method is similar to the constructor. It is used by each worker
161// thread to achieve the same effect as that of the master thread exept
162// to register the new created instance. This method is invoked explicitly.
163// It does not create a new G4LogicalVolume instance. It only assign the value
164// for the fields encapsulated by the class G4LVData.
165// ********************************************************************
166//
167void G4LogicalVolume::
168InitialiseWorker( G4LogicalVolume* /*pMasterObject*/,
169 G4VSolid* pSolid,
170 G4VSensitiveDetector* pSDetector)
171{
172 subInstanceManager.SlaveCopySubInstanceArray();
173
174 SetSolid(pSolid);
175 SetSensitiveDetector(pSDetector); // How this object is available now ?
176 AssignFieldManager(fFieldManager);
177 // Should be set - but a per-thread copy is not available yet
178 // Must not call SetFieldManager(), which propagates FieldMgr
179
180#ifdef CLONE_FIELD_MGR
181 // Create a field FieldManager by cloning
182 //
183 G4FieldManager workerFldMgr= fFieldManager->GetWorkerClone(G4bool* created);
184 if( created || (GetFieldManager() != workerFldMgr) )
185 {
186 SetFieldManager(fFieldManager, false); // which propagates FieldMgr
187 }
188 else
189 {
190 // Field manager existed and is equal to current one
191 //
192 AssignFieldManager(workerFldMgr);
193 }
194#endif
195}

258// ********************************************************************
259// SetFieldManager
260// ********************************************************************
261//
262void
263G4LogicalVolume::SetFieldManager(G4FieldManager* pNewFieldMgr,
264 G4bool forceAllDaughters)
265{
266 AssignFieldManager(pNewFieldMgr);
267
268 G4int NoDaughters = GetNoDaughters();
269 while ( (NoDaughters--)>0 )
270 {
271 G4LogicalVolume* DaughterLogVol;
272 DaughterLogVol = GetDaughter(NoDaughters)->GetLogicalVolume();
273 if ( forceAllDaughters || (DaughterLogVol->GetFieldManager() == 0) )
274 {
275 DaughterLogVol->SetFieldManager(pNewFieldMgr, forceAllDaughters);
276 }
277 }
278}

280// ********************************************************************
281// AddDaughter
282// ********************************************************************
283//
284void G4LogicalVolume::AddDaughter(G4VPhysicalVolume* pNewDaughter)
285{
286 if( !fDaughters.empty() && fDaughters[0]->IsReplicated() )
287 {
288 std::ostringstream message;
289 message << "ERROR - Attempt to place a volume in a mother volume" << G4endl
290 << " already containing a replicated volume." << G4endl
291 << " A volume can either contain several placements" << G4endl
292 << " or a unique replica or parameterised volume !" << G4endl
293 << " Mother logical volume: " << GetName() << G4endl
294 << " Placing volume: " << pNewDaughter->GetName() << G4endl;
295 G4Exception("G4LogicalVolume::AddDaughter()", "GeomMgt0002",
296 FatalException, message,
297 "Replica or parameterised volume must be the only daughter !");
298 }
299
300 // Invalidate previous calculation of mass - if any - for all threads
301 //
302 G4MT_mass = 0.;
303 fDaughters.push_back(pNewDaughter);
304
305 G4LogicalVolume* pDaughterLogical = pNewDaughter->GetLogicalVolume();
306
307 // Propagate the Field Manager, if the daughter has no field Manager
308 //
309 G4FieldManager* pDaughterFieldManager = pDaughterLogical->GetFieldManager();
310
311 // Avoid propagating the fieldManager pointer if null
312 // and daughter's one is null as well...
313 //
314 if( (G4MT_fmanager !=0 ) && (pDaughterFieldManager == 0) )
315 {
316 pDaughterLogical->SetFieldManager(G4MT_fmanager, false);
317 }
318 if (fRegion)
319 {
320 PropagateRegion();
321 fRegion->RegionModified(true);
322 }
323}

325// ********************************************************************
326// RemoveDaughter
327// ********************************************************************
328//
329void G4LogicalVolume::RemoveDaughter(const G4VPhysicalVolume* p)
330{
331 G4PhysicalVolumeList::iterator i;
332 for ( i=fDaughters.begin(); i!=fDaughters.end(); ++i )
333 {
334 if (**i==*p)
335 {
336 fDaughters.erase(i);
337 break;
338 }
339 }
340 if (fRegion)
341 {
342 fRegion->RegionModified(true);
343 }
344 G4MT_mass = 0.;
345}

469// ********************************************************************
470// IsAncestor
471//
472// Finds out if the current logical volume is an ancestor of a given
473// physical volume
474// ********************************************************************
475//
476G4bool
477G4LogicalVolume::IsAncestor(const G4VPhysicalVolume* aVolume) const
478{
479 G4bool isDaughter = IsDaughter(aVolume);
480 if (!isDaughter)
481 {
482 for (G4PhysicalVolumeList::const_iterator itDau = fDaughters.begin();
483 itDau != fDaughters.end(); itDau++)
484 {
485 isDaughter = (*itDau)->GetLogicalVolume()->IsAncestor(aVolume);
486 if (isDaughter) break;
487 }
488 }
489 return isDaughter;
490}

492// ********************************************************************
493// TotalVolumeEntities
494//
495// Returns the total number of physical volumes (replicated or placed)
496// in the tree represented by the current logical volume.
497// ********************************************************************
498//
499G4int G4LogicalVolume::TotalVolumeEntities() const
500{
501 G4int vols = 1;
502 for (G4PhysicalVolumeList::const_iterator itDau = fDaughters.begin();
503 itDau != fDaughters.end(); itDau++)
504 {
505 G4VPhysicalVolume* physDaughter = (*itDau);
506 vols += physDaughter->GetMultiplicity()
507 *physDaughter->GetLogicalVolume()->TotalVolumeEntities();
508 }
509 return vols;
510}

512// ********************************************************************
513// GetMass
514//
515// Returns the mass of the logical volume tree computed from the
516// estimated geometrical volume of each solid and material associated
517// to the logical volume and its daughters.
518// NOTE: the computation may require considerable amount of time,
519// depending from the complexity of the geometry tree.
520// The returned value is cached and can be used for successive
521// calls (default), unless recomputation is forced by providing
522// 'true' for the boolean argument in input. Computation should
523// be forced if the geometry setup has changed after the previous
524// call. By setting the 'propagate' boolean flag to 'false' the
525// method returns the mass of the present logical volume only
526// (subtracted for the volume occupied by the daughter volumes).
527// The extra argument 'parMaterial' is internally used to
528// consider cases of geometrical parameterisations by material.
529// ********************************************************************
530//
531G4double G4LogicalVolume::GetMass(G4bool forced,
532 G4bool propagate,
533 G4Material* parMaterial)
534{
535 // Return the cached non-zero value, if not forced
536 //
537 if ( (G4MT_mass) && (!forced) ) { return G4MT_mass; }
538
539 // Global density and computed mass associated to the logical
540 // volume without considering its daughters
541 //
542 G4Material* logMaterial = parMaterial ? parMaterial : GetMaterial();
543 if (!logMaterial)
544 {
545 std::ostringstream message;
546 message << "No material associated to the logical volume: "
547 << fName << " !" << G4endl
548 << "Sorry, cannot compute the mass ...";
549 G4Exception("G4LogicalVolume::GetMass()", "GeomMgt0002",
550 FatalException, message);
551 return 0;
552 }
553 if ( !GetSolid() )
554 {
555 std::ostringstream message;
556 message << "No solid is associated to the logical volume: "
557 << fName << " !" << G4endl
558 << "Sorry, cannot compute the mass ...";
559 G4Exception("G4LogicalVolume::GetMass()", "GeomMgt0002",
560 FatalException, message);
561 return 0;
562 }
563 G4double globalDensity = logMaterial->GetDensity();
564 G4double motherMass= GetSolid()->GetCubicVolume() * globalDensity;
565 G4double massSum = motherMass;
566
567 // For each daughter in the tree, subtract the mass occupied
568 // and if required by the propagate flag, add the real daughter's
569 // one computed recursively
570
571 for (G4PhysicalVolumeList::const_iterator itDau = fDaughters.begin();
572 itDau != fDaughters.end(); itDau++)
573 {
574 G4VPhysicalVolume* physDaughter = (*itDau);
575 G4LogicalVolume* logDaughter = physDaughter->GetLogicalVolume();
576 G4double subMass=0.;
577 G4VSolid* daughterSolid = 0;
578 G4Material* daughterMaterial = 0;
579
580 // Compute the mass to subtract and to add for each daughter
581 // considering its multiplicity (i.e. replicated or not) and
582 // eventually its parameterisation (by solid and/or by material)
583 //
584 for (G4int i=0; i<physDaughter->GetMultiplicity(); i++)
585 {
586 G4VPVParameterisation* physParam = physDaughter->GetParameterisation();
587 if (physParam)
588 {
589 daughterSolid = physParam->ComputeSolid(i, physDaughter);
590 daughterSolid->ComputeDimensions(physParam, i, physDaughter);
591 daughterMaterial = physParam->ComputeMaterial(i, physDaughter);
592 }
593 else
594 {
595 daughterSolid = logDaughter->GetSolid();
596 daughterMaterial = logDaughter->GetMaterial();
597 }
598 subMass = daughterSolid->GetCubicVolume() * globalDensity;
599
600 // Subtract the daughter's portion for the mass and, if required,
601 // add the real daughter's mass computed recursively
602 //
603 massSum -= subMass;
604 if (propagate)
605 {
606 massSum += logDaughter->GetMass(true, true, daughterMaterial);
607 }
608 }
609 }
610 G4MT_mass= massSum;
611 return massSum;
612}
