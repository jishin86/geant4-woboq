40std::vector<G4LogicalVolume*> G4LogicalCrystalVolume::fLCVvec;

42// --------------------------------------------------------------------
43
44G4LogicalCrystalVolume::
45G4LogicalCrystalVolume(G4VSolid* pSolid, G4ExtendedMaterial* pMaterial,
46 const G4String& name, G4FieldManager* pFieldMgr,
47 G4VSensitiveDetector* pSDetector,
48 G4UserLimits* pULimits, G4bool optimise,
49 G4int h, G4int k, G4int l, G4double rot)
50: G4LogicalVolume(pSolid,pMaterial,name,pFieldMgr,pSDetector,pULimits,optimise),
51 hMiller(1), kMiller(0), lMiller(0), fRot(0), verboseLevel(0)
52{
53 SetMillerOrientation(h, k, l, rot);
54 fLCVvec.push_back(this);
55}
56
57// --------------------------------------------------------------------
58
59G4LogicalCrystalVolume::~G4LogicalCrystalVolume()
60{
61 fLCVvec.erase( std::remove(fLCVvec.begin(),fLCVvec.end(), this ),
62 fLCVvec.end() );
63}

65// --------------------------------------------------------------------
66
67G4bool G4LogicalCrystalVolume::IsLattice(G4LogicalVolume* aLV)
68{
69 return std::find(fLCVvec.begin(), fLCVvec.end(), aLV) != fLCVvec.end();
70}

89void G4LogicalCrystalVolume::SetMillerOrientation(G4int h,
90 G4int k,
91 G4int l,
92 G4double rot)
93{
94 // Align Miller normal vector (hkl) with +Z axis, and rotation about axis
95
96 if (verboseLevel)
97 {
98 G4cout << "G4LatticePhysical::SetMillerOrientation(" << h << " "
99 << k << " " << l << ", " << rot/CLHEP::deg << " deg)" << G4endl;
100 }
101
102 hMiller = h;
103 kMiller = k;
104 lMiller = l;
105 fRot = rot;
106
107 G4ThreeVector norm = (h*GetBasis(0)+k*GetBasis(1)+l*GetBasis(2)).unit();
108
109 if (verboseLevel>1) G4cout << " norm = " << norm << G4endl;
110
111 // Aligns geometry +Z axis with lattice (hkl) normal
112 fOrient = G4RotationMatrix::IDENTITY;
113 fOrient.rotateZ(rot).rotateY(norm.theta()).rotateZ(norm.phi());
114 fInverse = fOrient.inverse();
115
116 if (verboseLevel>1) G4cout << " fOrient = " << fOrient << G4endl;
117
118 // FIXME: Is this equivalent to (phi,theta,rot) Euler angles???
119}

121// --------------------------------------------------------------------
122
123// Rotate input vector between lattice and solid orientations
124
125const G4ThreeVector&
126G4LogicalCrystalVolume::RotateToLattice(G4ThreeVector& dir) const
127{
128 return dir.transform(fOrient);
129}
130
131const G4ThreeVector&
132G4LogicalCrystalVolume::RotateToSolid(G4ThreeVector& dir) const
133{
134 return dir.transform(fInverse);
135}
