68// ***************************************************************************
69// Delete all elements from the store
70// ***************************************************************************
71//
72void G4SolidStore::Clean()
73{
74 // Do nothing if geometry is closed
75 //
76 if (G4GeometryManager::IsGeometryClosed())
77 {
78 G4cout << "WARNING - Attempt to delete the solid store"
79 << " while geometry closed !" << G4endl;
80 return;
81 }
82
83 // Locks store for deletion of solids. De-registration will be
84 // performed at this stage. G4VSolids will not de-register themselves.
85 //
86 locked = true;
87
88 size_t i=0;
89 G4SolidStore* store = GetInstance();
90
91#ifdef G4GEOMETRY_VOXELDEBUG
92 G4cout << "Deleting Solids ... ";
93#endif
94
95 for(iterator pos=store->begin(); pos!=store->end(); pos++)
96 {
97 if (fgNotifier) { fgNotifier->NotifyDeRegistration(); }
98 if (*pos) { delete *pos; }
99 i++;
100 }
101
102#ifdef G4GEOMETRY_VOXELDEBUG
103 if (store->size() < i-1)
104 { G4cout << "No solids deleted. Already deleted by user ?" << G4endl; }
105 else
106 { G4cout << i-1 << " solids deleted !" << G4endl; }
107#endif
108
109 locked = false;
110 store->clear();
111}

123// ***************************************************************************
124// Add Solid to container
125// ***************************************************************************
126//
127void G4SolidStore::Register(G4VSolid* pSolid)
128{
129 GetInstance()->push_back(pSolid);
130 if (fgNotifier) { fgNotifier->NotifyRegistration(); }
131}
132
133// ***************************************************************************
134// Remove Solid from container
135// ***************************************************************************
136//
137void G4SolidStore::DeRegister(G4VSolid* pSolid)
138{
139 if (!locked) // Do not de-register if locked !
140 {
141 if (fgNotifier) { fgNotifier->NotifyDeRegistration(); }
142 for (reverse_iterator i=GetInstance()->rbegin();
143 i!=GetInstance()->rend(); i++)
144 {
145 if (**i==*pSolid)
146 {
147 GetInstance()->erase(std::next(i).base());
148 break;
149 }
150 }
151 }
152}

154// ***************************************************************************
155// Retrieve the first solid pointer in the container having that name
156// ***************************************************************************
157//
158G4VSolid* G4SolidStore::GetSolid(const G4String& name, G4bool verbose) const
159{
160 for (iterator i=GetInstance()->begin(); i!=GetInstance()->end(); i++)
161 {
162 if ((*i)->GetName() == name) { return *i; }
163 }
164 if (verbose)
165 {
166 std::ostringstream message;
167 message << "Solid " << name << " not found in store !" << G4endl
168 << "Returning NULL pointer.";
169 G4Exception("G4SolidStore::GetSolid()",
170 "GeomMgt1001", JustWarning, message);
171 }
172 return 0;
173}
174
175// ***************************************************************************
176// Return ptr to Store, setting if necessary
177// ***************************************************************************
178//
179G4SolidStore* G4SolidStore::GetInstance()
180{
181 static G4SolidStore worldStore;
182 if (!fgInstance)
183 {
184 fgInstance = &worldStore;
185 }
186 return fgInstance;
187}
