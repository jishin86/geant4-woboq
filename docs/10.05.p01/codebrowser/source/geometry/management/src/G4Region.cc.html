62// *******************************************************************
63// Constructor:
64// - Adds self to region Store
65// *******************************************************************
66//
67G4Region::G4Region(const G4String& pName)
68 : fName(pName), fRegionMod(true), fCut(0), fUserInfo(0), fUserLimits(0),
69 fFieldManager(0), fWorldPhys(0),
70 fInMassGeometry(false), fInParallelGeometry(false)
71{
72
73 instanceID = subInstanceManager.CreateSubInstance();
74 G4MT_fsmanager = 0;
75 G4MT_rsaction = 0;
76
77 G4RegionStore* rStore = G4RegionStore::GetInstance();
78 if (rStore->GetRegion(pName,false))
79 {
80 std::ostringstream message;
81 message << "The region has NOT been registered !" << G4endl
82 << " Region " << pName << " already existing in store !"
83 << G4endl;
84 G4Exception("G4Region::G4Region()", "GeomMgt1001",
85 JustWarning, message);
86 }
87 else
88 {
89 rStore->Register(this);
90 }
91}

159// *******************************************************************
160// ScanVolumeTree:
161// - Scans recursively the 'lv' logical volume tree, retrieves
162// and places all materials in the list.
163// - The boolean flag 'region' identifies if the volume tree must
164// have region reset (false) or if the current region must be
165// associated to the logical volume 'lv' and its tree (true).
166// *******************************************************************
167//
168void G4Region::ScanVolumeTree(G4LogicalVolume* lv, G4bool region)
169{
170 // If logical volume is going to become a region, add
171 // its material to the list if not already present
172 //
173 G4Region* currentRegion = 0;
174 size_t noDaughters = lv->GetNoDaughters();
175 G4Material* volMat = lv->GetMaterial();
176 if(!volMat && fInMassGeometry)
177 {
178 std::ostringstream message;
179 message << "Logical volume <" << lv->GetName() << ">" << G4endl
180 << "does not have a valid material pointer." << G4endl
181 << "A logical volume belonging to the (tracking) world volume "
182 << "must have a valid material.";
183 G4Exception("G4Region::ScanVolumeTree()", "GeomMgt0002",
184 FatalException, message, "Check your geometry construction.");
185 }
186 if (region)
187 {
188 currentRegion = this;
189 if (volMat)
190 {
191 AddMaterial(volMat);
192 G4Material* baseMat = const_cast<G4Material*>(volMat->GetBaseMaterial());
193 if (baseMat) { AddMaterial(baseMat); }
194 }
195 }
196
197 // Set the LV region to be either the current region or NULL,
198 // according to the boolean selector
199 //
200 lv->SetRegion(currentRegion);
201
202 // Stop recursion here if no further daughters are involved
203 //
204 if(noDaughters==0) return;
205
206 G4VPhysicalVolume* daughterPVol = lv->GetDaughter(0);
207 if (daughterPVol->IsParameterised())
208 {
209 // Adopt special treatment in case of parameterised volumes,
210 // where parameterisation involves a new material scan
211 //
212 G4VPVParameterisation* pParam = daughterPVol->GetParameterisation();
213
214 if (pParam->GetMaterialScanner())
215 {
216 size_t matNo = pParam->GetMaterialScanner()->GetNumberOfMaterials();
217 for (size_t mat=0; mat<matNo; mat++)
218 {
219 volMat = pParam->GetMaterialScanner()->GetMaterial(mat);
220 if(!volMat && fInMassGeometry)
221 {
222 std::ostringstream message;
223 message << "The parameterisation for the physical volume <"
224 << daughterPVol->GetName() << ">" << G4endl
225 << "does not return a valid material pointer." << G4endl
226 << "A volume belonging to the (tracking) world volume must "
227 << "have a valid material.";
228 G4Exception("G4Region::ScanVolumeTree()", "GeomMgt0002",
229 FatalException, message, "Check your parameterisation.");
230 }
231 if (volMat)
232 {
233 AddMaterial(volMat);
234 G4Material* baseMat = const_cast<G4Material*>(volMat->GetBaseMaterial());
235 if (baseMat) { AddMaterial(baseMat); }
236 }
237 }
238 }
239 else
240 {
241 size_t repNo = daughterPVol->GetMultiplicity();
242 for (size_t rep=0; rep<repNo; rep++)
243 {
244 volMat = pParam->ComputeMaterial(rep, daughterPVol);
245 if(!volMat && fInMassGeometry)
246 {
247 std::ostringstream message;
248 message << "The parameterisation for the physical volume <"
249 << daughterPVol->GetName() << ">" << G4endl
250 << "does not return a valid material pointer." << G4endl
251 << "A volume belonging to the (tracking) world volume must "
252 << "have a valid material.";
253 G4Exception("G4Region::ScanVolumeTree()", "GeomMgt0002",
254 FatalException, message, "Check your parameterisation.");
255 }
256 if(volMat)
257 {
258 AddMaterial(volMat);
259 G4Material* baseMat = const_cast<G4Material*>(volMat->GetBaseMaterial());
260 if (baseMat) { AddMaterial(baseMat); }
261 }
262 }
263 }
264 G4LogicalVolume* daughterLVol = daughterPVol->GetLogicalVolume();
265 ScanVolumeTree(daughterLVol, region);
266 }
267 else
268 {
269 for (size_t i=0; i<noDaughters; i++)
270 {
271 G4LogicalVolume* daughterLVol = lv->GetDaughter(i)->GetLogicalVolume();
272 if (!daughterLVol->IsRootRegion())
273 {
274 // Set daughter's LV to be a region and store materials in
275 // the materials list, if the LV is not already a root region
276 //
277 ScanVolumeTree(daughterLVol, region);
278 }
279 }
280 }
281}

283// *******************************************************************
284// AddRootLogicalVolume:
285// - Adds a root logical volume and sets its daughters flags as
286// regions. It also recomputes the materials list for the region.
287// *******************************************************************
288//
289void G4Region::AddRootLogicalVolume(G4LogicalVolume* lv, G4bool search)
290{
291 // Check the logical volume is not already in the list
292 //
293 if (search)
294 {
295 G4RootLVList::iterator pos;
296 pos = std::find(fRootVolumes.begin(),fRootVolumes.end(),lv);
297 if (pos == fRootVolumes.end())
298 {
299 // Insert the root volume in the list and set it as root region
300 //
301 fRootVolumes.push_back(lv);
302 lv->SetRegionRootFlag(true);
303 }
304 }
305 else // WARNING: user *MUST* guarantee lv is not already inserted.
306 { // Providing speedup for very complex flat geometries
307 fRootVolumes.push_back(lv);
308 lv->SetRegionRootFlag(true);
309 }
310 // Scan recursively the tree of daugther volumes and set regions
311 //
312 ScanVolumeTree(lv, true);
313
314 // Set region as modified
315 //
316 fRegionMod = true;
317}

319// *******************************************************************
320// RemoveRootLogicalVolume:
321// - Removes a root logical volume and resets its daughters flags as
322// regions. It also recomputes the materials list for the region.
323// *******************************************************************
324//
325void G4Region::RemoveRootLogicalVolume(G4LogicalVolume* lv, G4bool scan)
326{
327 // Find and remove logical volume from the list
328 //
329 G4RootLVList::iterator pos;
330 pos = std::find(fRootVolumes.begin(),fRootVolumes.end(),lv);
331 if (pos != fRootVolumes.end())
332 {
333 if (fRootVolumes.size() != 1) // Avoid resetting flag for world since
334 { // volume may be already deleted !
335 lv->SetRegionRootFlag(false);
336 }
337 fRootVolumes.erase(pos);
338 }
339
340 if (scan) // Update the materials list
341 {
342 UpdateMaterialList();
343 }
344
345 // Set region as modified
346 //
347 fRegionMod = true;
348}

369// *******************************************************************
370// UpdateMaterialList:
371// - computes material list looping through
372// each root logical volume in the region.
373// *******************************************************************
374//
375void G4Region::UpdateMaterialList()
376{
377 // Reset the materials list
378 //
379 ClearMaterialList();
380
381 // Loop over the root logical volumes and rebuild the list
382 // of materials from scratch
383 //
384 G4RootLVList::iterator pLV;
385 for (pLV=fRootVolumes.begin(); pLV!=fRootVolumes.end(); pLV++)
386 {
387 ScanVolumeTree(*pLV, true);
388 }
389}

391// *******************************************************************
392// SetWorld:
393// - Set the world physical volume if this region belongs to this
394// world. If the given pointer is null, reset the pointer.

397void G4Region::SetWorld(G4VPhysicalVolume* wp)
398{
399 if(!wp)
400 { fWorldPhys = 0; }

402 { if(BelongsTo(wp)) fWorldPhys = wp; }
403
404 return;

407// *******************************************************************
408// BelongsTo:
409// - Returns whether this region belongs to the given physical volume
410// (recursively scanned to the bottom of the hierarchy)
411// *******************************************************************
412//
413G4bool G4Region::BelongsTo(G4VPhysicalVolume* thePhys) const
414{
415 G4LogicalVolume* currLog = thePhys->GetLogicalVolume();
416 if (currLog->GetRegion()==this) {return true;}
417
418 G4int nDaughters = currLog->GetNoDaughters();
419 while (nDaughters--) // Loop checking, 06.08.2015, G.Cosmo
420 {
421 if (BelongsTo(currLog->GetDaughter(nDaughters))) {return true;}
422 }
423
424 return false;
425}

427// *******************************************************************
428// ClearFastSimulationManager:
429// - Set G4FastSimulationManager pointer to the one for the parent region
430// if it exists. Otherwise set to null.
431// *******************************************************************
432//
433void G4Region::ClearFastSimulationManager()
434{
435 G4bool isUnique;
436 G4Region* parent = GetParentRegion(isUnique);
437 if(parent)
438 {
439 if (isUnique)
440 {
441 G4MT_fsmanager = parent->GetFastSimulationManager();
442 }
443 else
444 {
445 std::ostringstream message;
446 message << "Region <" << fName << "> belongs to more than"
447 << " one parent region !" << G4endl
448 << "A region cannot belong to more than one direct parent region,"
449 << G4endl
450 << "to have fast-simulation assigned.";
451 G4Exception("G4Region::ClearFastSimulationManager()",
452 "GeomMgt1002", JustWarning, message);
453 G4MT_fsmanager = 0;
454 }
455 }
456 else
457 {
458 G4MT_fsmanager = 0;
459 }
460}

462// *******************************************************************
463// GetParentRegion:
464// - Returns a region that contains this region.
465// Otherwise null is returned.
466// *******************************************************************
467//
468G4Region* G4Region::GetParentRegion(G4bool& unique) const
469{
470 G4Region* parent = 0; unique = true;
471 G4LogicalVolumeStore* lvStore = G4LogicalVolumeStore::GetInstance();
472 G4LogicalVolumeStore::iterator lvItr;
473
474 // Loop over all logical volumes in the store
475 //
476 for(lvItr=lvStore->begin(); lvItr!=lvStore->end(); lvItr++)
477 {
478 G4int nD = (*lvItr)->GetNoDaughters();
479 G4Region* aR = (*lvItr)->GetRegion();
480
481 // Loop over all daughters of each logical volume
482 //
483 for(G4int iD=0; iD<nD; iD++)
484 {
485 if((*lvItr)->GetDaughter(iD)->GetLogicalVolume()->GetRegion()==this)
486 {
487 if(parent)
488 {
489 if(parent!=aR) { unique = false; }
490 }
491 else // Cache LV parent region which includes a daughter volume
492 // with the same associated region as the current one
493 {
494 parent = aR;
495 }
496 }
497 }
498 }
499 return parent;
