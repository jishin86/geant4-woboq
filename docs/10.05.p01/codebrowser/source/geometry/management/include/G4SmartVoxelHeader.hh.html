28// class G4SmartVoxelHeader
29//
30// Class description:
31//
32// Represents a set of voxels, created by a single axis of virtual division.
33// Contains the individual voxels, which are potentially further divided
34// along different axes.
35//
36// Member data:
37//
38// EAxis faxis
39// - The (cartesian) slicing/division axis
40// G4double fmaxExtent
41// G4double fminExtent
42// - Minimum and maximum coordiantes along the axis
43// std::vector<G4SmartVoxelProxy*> fslices
44// - The slices along the axis
45//
46// G4int fminEquivalent
47// G4int fmaxEquivalent
48// - Minimum and maximum equivalent slice nos.
49// [Applies to the level of the header, not its nodes]
50
51// History:
52// 18.04.01 G.Cosmo Migrated to STL vector
53// 13.07.95 P.Kent Initial version
54// --------------------------------------------------------------------
55#ifndef G4SMARTVOXELHEADER_HH
56#define G4SMARTVOXELHEADER_HH
57
58#include "G4Types.hh"
59#include "geomdefs.hh"
60
61#include "G4SmartVoxelProxy.hh"
62#include "G4SmartVoxelNode.hh"
63
64#include <vector>
65
66// Forward declarations
67class G4LogicalVolume;
68class G4VoxelLimits;
69class G4VPhysicalVolume;
70
71// Typedefs
72typedef std::vector<G4SmartVoxelProxy*> G4ProxyVector;
73typedef std::vector<G4SmartVoxelNode*> G4NodeVector;
74typedef std::vector<G4int> G4VolumeNosVector;
75typedef std::vector<G4double> G4VolumeExtentVector;
76
77class G4SmartVoxelHeader
78{
79 public: // with description
80
81 G4SmartVoxelHeader(G4LogicalVolume* pVolume, G4int pSlice=0);
82 // Constructor for topmost header, to begin voxel construction at a
83 // given logical volume. pSlice is used to set max and min equivalent
84 // slice nos for the header - they apply to the level of the header,
85 // not its nodes.
86
87 ~G4SmartVoxelHeader();
88 // Delete all referenced nodes [but *not* referenced physical volumes].
89
90 G4int GetMaxEquivalentSliceNo() const;
91 void SetMaxEquivalentSliceNo(G4int pMax);
92 G4int GetMinEquivalentSliceNo() const;
93 void SetMinEquivalentSliceNo(G4int pMin);
94 // Access functions for min/max equivalent slices (nodes & headers).
95
96 EAxis GetAxis() const;
97 // Return the current division axis.
98 EAxis GetParamAxis() const;
99 // Return suggested division axis for parameterised volume.
100
101 G4double GetMaxExtent() const;
102 // Return the maximum coordinate limit along the current axis.
103 G4double GetMinExtent() const;
104 // Return the minimum coordinate limit along the current axis.
105
106 G4int GetNoSlices() const;
107 // Return the no of slices along the current axis.
108
109 G4SmartVoxelProxy* GetSlice(G4int n) const;
110 // Return ptr to the proxy for the nth slice (numbering from 0,
111 // no bounds checking performed).
112
113 G4bool AllSlicesEqual() const;
114 // True if all slices equal (after collection).
115
116 public: // without description
117
118 G4bool operator == (const G4SmartVoxelHeader& pHead) const;
119
120 friend std::ostream&
121 operator << (std::ostream&s, const G4SmartVoxelHeader& h);
122
123 G4SmartVoxelHeader(G4LogicalVolume* pVolume,
124 const G4VoxelLimits& pLimits,
125 const G4VolumeNosVector* pCandidates,
126 G4int pSlice=0);
127 // Build and refine voxels between specified limits, considering only
128 // the physical volumes numbered `pCandidates'. pSlice is used to set max
129 // and min equivalent slice nos for the header - they apply to the level
130 // of the header, not its nodes.
131
132 protected:
133
134 // `Worker' / operation functions:
135
136 void BuildVoxels(G4LogicalVolume* pVolume);
137 // Build and refine voxels for daughters of specified volume which
138 // DOES NOT contain a REPLICATED daughter.
139
140 void BuildReplicaVoxels(G4LogicalVolume* pVolume);
141 // Build voxels for specified volume containing a single
142 // replicated volume.
143
144 void BuildConsumedNodes(G4int nReplicas);
145 // Construct nodes in simple consuming case.
146
147 void BuildVoxelsWithinLimits(G4LogicalVolume* pVolume,
148 G4VoxelLimits pLimits,
149 const G4VolumeNosVector* pCandidates);
150 // Build and refine voxels between specified limits, considering only
151 // the physical volumes `pCandidates'. Main entry point for "construction".
152 // Hardwired to stop at third level of refinement, using the xyz cartesian
153 // axes in any order.
154
155 void BuildEquivalentSliceNos();
156 // Calculate and Store the minimum and maximum equivalent neighbour
157 // values for all slices.
158
159 void CollectEquivalentNodes();
160 // Collect common nodes, deleting all but one to save memory,
161 // and adjusting stored slice ptrs appropriately.
162
163 void CollectEquivalentHeaders();
164 // Collect common headers, deleting all but one to save memory,
165 // and adjusting stored slice ptrs appropriately.
166
167
168 G4ProxyVector* BuildNodes(G4LogicalVolume* pVolume,
169 G4VoxelLimits pLimits,
170 const G4VolumeNosVector* pCandidates,
171 EAxis pAxis);
172 // Build the nodes corresponding to the specified axis, within
173 // the specified limits, considering the daughters numbered pCandidates
174 // of the logical volume.
175
176 G4double CalculateQuality(G4ProxyVector *pSlice);
177 // Calculate a "quality value" for the specified vector of voxels
178 // The value returned should be >0 and such that the smaller the
179 // number the higher the quality of the slice.
180 // pSlice must consist of smartvoxelnodeproxies only.
181
182 void RefineNodes(G4LogicalVolume* pVolume,G4VoxelLimits pLimits);
183 // Examined each contained node, refine (create a replacement additional
184 // dimension of voxels) when there is more than one voxel in the slice.
185
186 G4int fminEquivalent;
187 G4int fmaxEquivalent;
188 // Min and max equivalent slice nos for previous level.
189
190 EAxis faxis, fparamAxis;
191 // Axis for slices.
192
193 G4double fmaxExtent;
194 G4double fminExtent;
195 // Max and min coordinate along faxis.
196
197 G4ProxyVector fslices;
198 // Slices along axis.
199};

203#endif
