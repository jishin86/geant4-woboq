38#ifndef G4VPVDIVISION_FACTORY_HH
39#define G4VPVDIVISION_FACTORY_HH
40
41#include "geomdefs.hh"
42
43class G4LogicalVolume;
44class G4VPhysicalVolume;
45class G4VPVParameterisation;
46
47class G4VPVDivisionFactory
48{
49 public: // with description
50
51 virtual ~G4VPVDivisionFactory();
52
53 static G4VPVDivisionFactory* Instance();
54
55 virtual G4VPhysicalVolume* CreatePVDivision(
56 const G4String& pName,
57 G4LogicalVolume* pLogical,
58 G4LogicalVolume* pMother,
59 const EAxis pAxis,
60 const G4int nReplicas,
61 const G4double width,
62 const G4double offset ) = 0;
63 // Create division - with number of divisions and width.
64
65 virtual G4VPhysicalVolume* CreatePVDivision(
66 const G4String& pName,
67 G4LogicalVolume* pLogical,
68 G4LogicalVolume* pMotherLogical,
69 const EAxis pAxis,
70 const G4int nReplicas,
71 const G4double offset ) = 0;
72 // Create division - with number of divisions.
73
74 virtual G4VPhysicalVolume* CreatePVDivision(
75 const G4String& pName,
76 G4LogicalVolume* pLogical,
77 G4LogicalVolume* pMotherLogical,
78 const EAxis pAxis,
79 const G4double width,
80 const G4double offset ) = 0;
81 // Create division - with width.
82
83 virtual G4VPhysicalVolume* CreatePVDivision(
84 const G4String& pName,
85 G4LogicalVolume* pLogical,
86 G4LogicalVolume* pMotherLogical,
87 const G4VPVParameterisation* param) = 0;
88 // Create division - with parameterisation.
89
90 virtual G4bool IsPVDivision(const G4VPhysicalVolume* pv) const = 0;
91 // Returns true if pv is division.
92
93 protected:
94
95 G4VPVDivisionFactory();
96
97 protected:
98
99 static G4ThreadLocal G4VPVDivisionFactory* fgInstance;
100
101};

103#endif
