81inline
82EVolume G4VPhysicalVolume::VolumeType() const
83{
84 EVolume type;
85 EAxis axis;
86 G4int nReplicas;
87 G4double width,offset;
88 G4bool consuming;
89 if ( IsReplicated() )
90 {
91 GetReplicationData(axis,nReplicas,width,offset,consuming);
92 type = (consuming) ? kReplica : kParameterised;
93 }
94 else
95 {
96 type = kNormal;
97 }
98 return type;
99}
