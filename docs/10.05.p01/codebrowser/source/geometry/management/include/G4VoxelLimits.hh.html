28// class G4VoxelLimits
29//
30// Class description:
31//
32// Represents limitation/restrictions of space, where restrictions
33// are only made perpendicular to the cartesian axes.
34//
35//
36// Member data:
37//
38// G4double fxAxisMin,fxAxisMax
39// G4double fyAxisMin,fyAxisMax
40// G4double fzAxisMin,fzAxisMax
41// - The min and max values along each axis. +-kInfinity if not restricted.
42//
43//
44// Notes:
45//
46// Beware no break statements after returns in switch(pAxis)s.
47
48// History:
49// 13.07.95 P.Kent Initial version.
50// --------------------------------------------------------------------
51#ifndef G4VOXELLIMITS_HH
52#define G4VOXELLIMITS_HH
53
54#include "G4Types.hh"
55#include "geomdefs.hh"
56
57#include "G4ThreeVector.hh"
58
59#include <assert.h>
60
61class G4VoxelLimits
62{
63 public: // with description
64
65 G4VoxelLimits();
66 // Constructor - initialise to be unlimited. Volume unrestricted.
67
68 ~G4VoxelLimits();
69 // Destructor. No actions.
70
71 void AddLimit(const EAxis pAxis, const G4double pMin,const G4double pMax);
72 // Restrict the volume to between specified min and max along the
73 // given axis. Cartesian axes only, pMin<=pMax.
74
75 G4double GetMaxXExtent() const;
76 // Return maximum x extent.
77 G4double GetMaxYExtent() const;
78 // Return maximum y extent.
79 G4double GetMaxZExtent() const;
80 // Return maximum z extent.
81
82 G4double GetMinXExtent() const;
83 // Return minimum x extent.
84 G4double GetMinYExtent() const;
85 // Return minimum y extent.
86 G4double GetMinZExtent() const;
87 // Return minimum z extent.
88
89 G4double GetMaxExtent(const EAxis pAxis) const;
90 // Return maximum extent of volume along specified axis.
91 G4double GetMinExtent(const EAxis pAxis) const;
92 // Return minimum extent of volume along specified axis.
93
94 G4bool IsXLimited() const;
95 // Return true if the x axis is limited.
96 G4bool IsYLimited() const;
97 // Return true if the y axis is limited.
98 G4bool IsZLimited() const;
99 // Return true if the z axis is limited.
100
101 G4bool IsLimited() const;
102 // Return true if limited along any axis
103 G4bool IsLimited(const EAxis pAxis) const;
104 // Return true if the specified axis is restricted/limited.
105
106 G4bool ClipToLimits(G4ThreeVector& pStart,G4ThreeVector& pEnd) const;
107 // Clip the line segment pStart->pEnd to the volume described by the
108 // current limits. Return true if the line remains after clipping,
109 // else false, and leave the vectors in an undefined state.
110
111 G4bool Inside(const G4ThreeVector& pVec) const;
112 // Return true if the specified vector is inside/on boundaries of limits.
113
114 G4int OutCode(const G4ThreeVector& pVec) const;
115 // Calculate the `outcode' for the specified vector.
116 // Intended for use during clipping against the limits
117 // The bits are set given the following conditions:
118 // 0 pVec.x()<fxAxisMin && IsXLimited()
119 // 1 pVec.x()>fxAxisMax && IsXLimited()
120 // 2 pVec.y()<fyAxisMin && IsYLimited()
121 // 3 pVec.y()>fyAxisMax && IsYLimited()
122 // 4 pVec.z()<fzAxisMin && IsZLimited()
123 // 5 pVec.z()>fzAxisMax && IsZLimited()
124
125 private:
126
127 G4double fxAxisMin,fxAxisMax;
128 G4double fyAxisMin,fyAxisMax;
129 G4double fzAxisMin,fzAxisMax;
130};
131
132#include "G4VoxelLimits.icc"
133
134std::ostream& operator << (std::ostream& os, const G4VoxelLimits& pLim);
135 // Print the limits to the stream in the form:
136 // "{(xmin,xmax) (ymin,ymax) (zmin,zmax)}"
137 // Replace (xmin,xmax) by (-,-) when not limited.
138
139#endif
