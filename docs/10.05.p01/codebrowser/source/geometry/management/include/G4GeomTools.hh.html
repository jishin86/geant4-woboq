30// class G4GeomTools
31//
32// Class description:
33//

42#ifndef G4GEOMTOOLS_HH
43#define G4GEOMTOOLS_HH
44
45#include <vector>
46#include "G4TwoVector.hh"
47#include "G4ThreeVector.hh"
48
49typedef std::vector<G4TwoVector> G4TwoVectorList;
50typedef std::vector<G4ThreeVector> G4ThreeVectorList;
51
52class G4GeomTools
53{
54 public:
55
56 // ==================================================================
57 // 2D Utilities
58 // ------------------------------------------------------------------
59
60 static G4double TriangleArea(G4double Ax, G4double Ay,
61 G4double Bx, G4double By,
62 G4double Cx, G4double Cy);
63
64 static G4double TriangleArea(const G4TwoVector& A,
65 const G4TwoVector& B,
66 const G4TwoVector& C);
67 // Calcuate area of 2D triangle, return value is positive if
68 // vertices of the triangle are given in anticlockwise order,
69 // otherwise it is negative
70
71 static G4double QuadArea(const G4TwoVector& A,
72 const G4TwoVector& B,
73 const G4TwoVector& C,
74 const G4TwoVector& D);
75 // Calcuate area of 2D quadrilateral, return value is positive if
76 // vertices of the quadrilateral are given in anticlockwise order,
77 // otherwise it is negative
78
79 static G4double PolygonArea(const G4TwoVectorList& polygon);
80 // Calcuate area of 2D polygon, return value is positive if
81 // vertices of the polygon are defined in anticlockwise order,
82 // otherwise it is negative
83
84 static G4bool PointInTriangle(G4double Px, G4double Py,
85 G4double Ax, G4double Ay,
86 G4double Bx, G4double By,
87 G4double Cx, G4double Cy);
88 // Decide if point (Px,Py) is inside triangle (Ax,Ay)(Bx,By)(Cx,Cy)
89
90 static G4bool PointInTriangle(const G4TwoVector& P,
91 const G4TwoVector& A,
92 const G4TwoVector& B,
93 const G4TwoVector& C);
94 // Decide if point P is inside triangle ABC
95
96 static G4bool PointInPolygon(const G4TwoVector& P,
97 const G4TwoVectorList& Polygon);
98 // Decide if point P is inside Polygon
99
100 static G4bool IsConvex(const G4TwoVectorList& polygon);
101 // Decide if 2D polygon is convex, i.e. all internal angles are
102 // less than pi
103
104 static G4bool TriangulatePolygon(const G4TwoVectorList& polygon,
105 G4TwoVectorList& result);
106
107 static G4bool TriangulatePolygon(const G4TwoVectorList& polygon,
108 std::vector<G4int>& result);
109 // Simple implementation of "ear clipping" algorithm for
110 // triangulation of a simple contour/polygon, it places results
111 // in a std::vector as triplets of vertices. If triangulation
112 // is sucsessfull then the function returns true, otherwise false
113
114 static void RemoveRedundantVertices(G4TwoVectorList& polygon,
115 std::vector<G4int>& iout,
116 G4double tolerance = 0);
117 // Remove collinear and coincident points from 2D polygon.
118 // Indices of removed points are available in iout.
119
120 static G4bool DiskExtent(G4double rmin, G4double rmax,
121 G4double startPhi, G4double delPhi,
122 G4TwoVector& pmin, G4TwoVector& pmax);
123 // Calculate bounding rectangle of a disk sector,
124 // it returns false if input parameters do not meet the following:
125 // rmin >= 0
126 // rmax > rmin + kCarTolerance
127 // delPhi > 0 + kCarTolerance
128
129 static void DiskExtent(G4double rmin, G4double rmax,
130 G4double sinPhiStart, G4double cosPhiStart,
131 G4double sinPhiEnd, G4double cosPhiEnd,
132 G4TwoVector& pmin, G4TwoVector& pmax);
133 // Calculate bounding rectangle of a disk sector,
134 // faster version without check of parameters
135
136 static G4double EllipsePerimeter(G4double a,
137 G4double b);
138 // Compute the circumference (perimeter) of an ellipse
139
140 static G4double EllipticConeLateralArea(G4double a,
141 G4double b,
142 G4double h);
143 // Compute the lateral surface area of an elliptic cone
144
145 // ==================================================================
146 // 3D Utilities
147 // ------------------------------------------------------------------
148
149 static G4ThreeVector TriangleAreaNormal(const G4ThreeVector& A,
150 const G4ThreeVector& B,
151 const G4ThreeVector& C);
152 // Find the normal to the plane of 3D triangle ABC,
153 // length of the normal is equal to the area of the triangle
154
155 static G4ThreeVector QuadAreaNormal(const G4ThreeVector& A,
156 const G4ThreeVector& B,
157 const G4ThreeVector& C,
158 const G4ThreeVector& D);
159 // Find normal to the plane of 3D quadrilateral ABCD,
160 // length of the normal is equal to the area of the quadrilateral
161
162 static G4ThreeVector PolygonAreaNormal(const G4ThreeVectorList& polygon);
163 // Find normal to the plane of 3D polygon
164 // length of the normal is equal to the area of the polygon
165
166 /*
167 static G4bool IsPlanar(const G4ThreeVector& A,
168 const G4ThreeVector& B,
169 const G4ThreeVector& C,
170 const G4ThreeVector& D);
171 // Decide if 3D quadrilateral ABCD is planar
172
173 static G4bool IsPlanar(const G4ThreeVectorList& polygon
174 const G4ThreeVector& normal);
175 // Decide if 3D polygon is planar
176 */
177
178 static G4double DistancePointSegment(const G4ThreeVector& P,
179 const G4ThreeVector& A,
180 const G4ThreeVector& B);
181 // Calculate distance between point P and line segment AB in 3D
182
183 static G4ThreeVector ClosestPointOnSegment(const G4ThreeVector& P,
184 const G4ThreeVector& A,
185 const G4ThreeVector& B);
186 // Find point on 3D line segment AB closest to point P
187
188 static G4ThreeVector ClosestPointOnTriangle(const G4ThreeVector& P,
189 const G4ThreeVector& A,
190 const G4ThreeVector& B,
191 const G4ThreeVector& C);
192 // Find point on 3D triangle ABC closest to point P
193
194 static G4bool SphereExtent(G4double rmin, G4double rmax,
195 G4double startTheta, G4double delTheta,
196 G4double startPhi, G4double delPhi,
197 G4ThreeVector& pmin, G4ThreeVector& pmax);
198 // Calculate bounding box of a spherical sector,
199 // it returns false if input parameters do not meet the following:
200 // rmin >= 0
201 // rmax > rmin + kCarTolerance
202 // startTheta >= 0 && <= pi;
203 // delTheta > 0 + kCarTolerance
204 // delPhi > 0 + kCarTolerance
205
206 private:
207
208 static G4bool CheckSnip(const G4TwoVectorList& contour,
209 G4int a, G4int b, G4int c,
210 G4int n, const G4int* V);
211 // Helper function for TriangulatePolygon()
212
213 static G4double comp_ellint_2(G4double e);
214 // Complete Elliptic Integral of the Second Kind
215};
216
217#endif // G4GEOMTOOLS_HH
