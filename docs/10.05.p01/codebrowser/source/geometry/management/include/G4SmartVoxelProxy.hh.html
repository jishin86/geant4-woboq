28// class G4SmartVoxelProxy
29//
30// Class description:
31//
32// Class for proxying smart voxels. The class represents either a header
33// (in turn refering to more VoxelProxies) or a node. If created as a node,
34// calls to GetHeader cause an exception, and likewise GetNode when a header.
35//
36// Note that the proxy does NOT gain deletion responsibility for proxied
37// objects.
38
39// History:
40// 12.07.95 P.Kent Initial version
41// 03.08.95 P.Kent Updated to become non abstract class, removing
42// HeaderProxy and NodeProxy derived classes
43// --------------------------------------------------------------------
44#ifndef G4SMARTVOXELPROXY_HH
45#define G4SMARTVOXELPROXY_HH
46
47#include "G4Types.hh"
48#include <assert.h>
49
50class G4SmartVoxelNode;
51class G4SmartVoxelHeader;
52
53class G4SmartVoxelProxy
54{
55
56 public: // with description
57
58 G4SmartVoxelProxy(G4SmartVoxelHeader *pHeader)
59 : fHeader(pHeader), fNode(0) {}
60 // Proxy for the specified header.
61
62 G4SmartVoxelProxy(G4SmartVoxelNode *pNode)
63 : fHeader(0), fNode(pNode) {}
64 // Proxy for the specified node.
65
66 ~G4SmartVoxelProxy();
67 // Destructor - do nothing. Not responsible for proxied objects.
68
69 G4bool IsHeader() const;
70 // Return true if proxying for a header, else false.
71
72 G4bool IsNode() const;
73 // Return true if proxying for a node, else false.
74
75 G4SmartVoxelNode* GetNode() const;
76 // Return ptr to proxied node, else call G4Exception.
77
78 G4SmartVoxelHeader* GetHeader() const;
79 // Return ptr to proxied header, else call G4Exception
80
81 G4bool operator == (const G4SmartVoxelProxy& v) const;
82 // Equality operator.
83 // True when objects share same address.
84
85 private:
86
87 G4SmartVoxelHeader* fHeader;
88 G4SmartVoxelNode* fNode;
89};

93#endif
