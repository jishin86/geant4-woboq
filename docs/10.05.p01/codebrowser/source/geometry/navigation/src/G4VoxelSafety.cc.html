68// ********************************************************************
69// ComputeSafety
70//
71// Calculates the isotropic distance to the nearest boundary from the
72// specified point in the local coordinate system.
73// The localpoint utilised must be within the current volume.
74// ********************************************************************
75//
76G4double
77G4VoxelSafety::ComputeSafety(const G4ThreeVector& localPoint,
78 const G4VPhysicalVolume& currentPhysical,
79 G4double maxLength)
80{
81 G4LogicalVolume *motherLogical;
82 G4VSolid *motherSolid;
83 G4SmartVoxelHeader *motherVoxelHeader;
84 G4double motherSafety, ourSafety;
85 G4int localNoDaughters;
86 G4double daughterSafety;
87
88 motherLogical = currentPhysical.GetLogicalVolume();
89 fpMotherLogical= motherLogical; // For use by the other methods
90 motherSolid = motherLogical->GetSolid();
91 motherVoxelHeader= motherLogical->GetVoxelHeader();
92
93#ifdef G4VERBOSE
94 if( fVerbose > 0 )
95 {
96 G4cout << "*** G4VoxelSafety::ComputeSafety(): ***" << G4endl;
97 }
98#endif
99
100 // Check that point is inside mother volume
101 //
102 EInside insideMother= motherSolid->Inside(localPoint);
103 if( insideMother != kInside )
104 {
105#ifdef G4DEBUG_NAVIGATION
106 if( insideMother == kOutside )
107 {
108 std::ostringstream message;
109 message << "Safety method called for location outside current Volume."
110 << G4endl
111 << "Location for safety is Outside this volume. " << G4endl
112 << "The approximate distance to the solid "
113 << "(safety from outside) is: "
114 << motherSolid->DistanceToIn( localPoint ) << G4endl;
115 message << " Problem occurred with physical volume: "
116 << " Name: " << currentPhysical.GetName()
117 << " Copy No: " << currentPhysical.GetCopyNo() << G4endl
118 << " Local Point = " << localPoint << G4endl;
119 message << " Description of solid: " << G4endl
120 << *motherSolid << G4endl;
121 G4Exception("G4VoxelSafety::ComputeSafety()", "GeomNav0003",
122 FatalException, message);
123 }
124#endif
125 return 0.0;
126 }
127
128 // First limit: mother safety - distance to outer boundaries
129 //
130 motherSafety = motherSolid->DistanceToOut(localPoint);
131 ourSafety = motherSafety; // Working isotropic safety
132
133#ifdef G4VERBOSE
134 if(( fCheck ) ) // && ( fVerbose == 1 ))
135 {
136 G4cout << " Invoked DistanceToOut(p) for mother solid: "
137 << motherSolid->GetName()
138 << ". Solid replied: " << motherSafety << G4endl
139 << " For local point p: " << localPoint
140 << ", to be considered as 'mother safety'." << G4endl;
141 }
142#endif
143 localNoDaughters = motherLogical->GetNoDaughters();
144
145 fBlockList.Enlarge(localNoDaughters);
146 fBlockList.Reset();
147
148 fVoxelDepth = -1; // Resets the depth -- must be done for next method
149 daughterSafety= SafetyForVoxelHeader(motherVoxelHeader, localPoint, maxLength,
150 currentPhysical, 0, ourSafety);
151 ourSafety= std::min( motherSafety, daughterSafety );
152
153 return ourSafety;
154}

156// ********************************************************************
157// SafetyForVoxelNode
158//
159// Calculate the safety for volumes included in current Voxel Node
160// ********************************************************************
161//
162G4double
163G4VoxelSafety::SafetyForVoxelNode( const G4SmartVoxelNode *curVoxelNode,
164 const G4ThreeVector& localPoint )
165{
166 G4double ourSafety= DBL_MAX;
167
168 G4int curNoVolumes, contentNo, sampleNo;
169 G4VPhysicalVolume *samplePhysical;
170
171 G4double sampleSafety=0.0;
172 G4ThreeVector samplePoint;
173 G4VSolid* ptrSolid=0;
174
175 curNoVolumes = curVoxelNode->GetNoContained();
176
177 for ( contentNo=curNoVolumes-1; contentNo>=0; contentNo-- )
178 {
179 sampleNo = curVoxelNode->GetVolume(contentNo);
180 if ( !fBlockList.IsBlocked(sampleNo) )
181 {
182 fBlockList.BlockVolume(sampleNo);
183
184 samplePhysical = fpMotherLogical->GetDaughter(sampleNo);
185 G4AffineTransform sampleTf(samplePhysical->GetRotation(),
186 samplePhysical->GetTranslation());
187 sampleTf.Invert();
188 samplePoint = sampleTf.TransformPoint(localPoint);
189 ptrSolid = samplePhysical->GetLogicalVolume()->GetSolid();
190
191 sampleSafety = ptrSolid->DistanceToIn(samplePoint);
192 ourSafety = std::min( sampleSafety, ourSafety );
193#ifdef G4VERBOSE
194 if(( fCheck ) && ( fVerbose == 1 ))
195 {
196 // ReportSolidSafetyToIn( MethodName, solid, value, point );
197 G4cout << "*** G4VoxelSafety::SafetyForVoxelNode(): ***" << G4endl
198 << " Invoked DistanceToIn(p) for daughter solid: "
199 << ptrSolid->GetName()
200 << ". Solid replied: " << sampleSafety << G4endl
201 << " For local point p: " << samplePoint
202 << ", to be considered as 'daughter safety'." << G4endl;
203 }
204#endif
205 }
206 } // end for contentNo
207
208 return ourSafety;
209}

211// ********************************************************************
212// SafetyForVoxelHeader
213//
214// Cycles through levels of headers to process each node level
215// Obtained by modifying VoxelLocate (to cycle through Node Headers)
216// *********************************************************************
217//
218G4double
219G4VoxelSafety::SafetyForVoxelHeader( const G4SmartVoxelHeader* pHeader,
220 const G4ThreeVector& localPoint,
221 G4double maxLength,
222 const G4VPhysicalVolume& currentPhysical, //Debug
223 G4double distUpperDepth_Sq,
224 G4double previousMinSafety
225 )
226{
227 const G4SmartVoxelHeader * const targetVoxelHeader=pHeader;
228 G4SmartVoxelNode *targetVoxelNode=0;
229
230 const G4SmartVoxelProxy *sampleProxy;
231 EAxis targetHeaderAxis;
232 G4double targetHeaderMin, targetHeaderMax, targetHeaderNodeWidth;
233 G4int targetHeaderNoSlices;
234 G4int targetNodeNo;
235
236 G4double minSafety= previousMinSafety;
237 G4double ourSafety= DBL_MAX;
238 unsigned int checkedNum= 0;
239
240 fVoxelDepth++;
241 // fVoxelDepth set by ComputeSafety or previous level call
242
243 targetHeaderAxis = targetVoxelHeader->GetAxis();
244 targetHeaderNoSlices = targetVoxelHeader->GetNoSlices();
245 targetHeaderMin = targetVoxelHeader->GetMinExtent();
246 targetHeaderMax = targetVoxelHeader->GetMaxExtent();
247
248 targetHeaderNodeWidth = (targetHeaderMax-targetHeaderMin)
249 / targetHeaderNoSlices;
250
251 G4double localCrd= localPoint(targetHeaderAxis);
252
253 const G4int candNodeNo= G4int( (localCrd-targetHeaderMin)
254 / targetHeaderNodeWidth );
255 // Ensure that it is between 0 and targetHeader->GetMaxExtent() - 1
256
257#ifdef G4DEBUG_VOXELISATION
258 if( candNodeNo < 0 || candNodeNo > targetHeaderNoSlices-1 )
259 {
260 G4ExceptionDescription ed;
261 ed << " Potential ERROR."
262 << " Point is outside range of Voxel in current coordinate" << G4endl;
263 ed << " Node number of point " << localPoint
264 << "is outside the range. " << G4endl;
265 ed << " Voxel node Num= " << candNodeNo << " versus minimum= " << 0
266 << " and maximum= " << targetHeaderNoSlices-1 << G4endl;
267 ed << " Axis = " << targetHeaderAxis
268 << " No of slices = " << targetHeaderNoSlices << G4endl;
269 ed << " Local coord = " << localCrd
270 << " Voxel Min = " << targetHeaderMin
271 << " Max = " << targetHeaderMax << G4endl;
272 G4LogicalVolume *pLogical= currentPhysical.GetLogicalVolume();
273 ed << " Current volume (physical) = " << currentPhysical.GetName()
274 << " (logical) = " << pLogical->GetName() << G4endl;
275 G4VSolid* pSolid= pLogical->GetSolid();
276 ed << " Solid type = " << pSolid->GetEntityType() << G4endl;
277 ed << *pSolid << G4endl;
278 G4Exception("G4VoxelSafety::SafetyForVoxelHeader()", "GeomNav1003",
279 JustWarning, ed,
280 "Point is outside range of Voxel in current coordinate");
281 }
282#endif
283
284 const G4int pointNodeNo =
285 std::max( 0, std::min( candNodeNo, targetHeaderNoSlices-1 ) );
286
287#ifdef G4VERBOSE
288 if( fVerbose > 2 )
289 {
290 G4cout << G4endl;
291 G4cout << "**** G4VoxelSafety::SafetyForVoxelHeader " << G4endl;
292 G4cout << " Called at Depth = " << fVoxelDepth;
293 G4cout << " Calculated pointNodeNo= " << pointNodeNo
294 << " from position= " << localPoint(targetHeaderAxis)
295 << " min= " << targetHeaderMin
296 << " max= " << targetHeaderMax
297 << " width= " << targetHeaderNodeWidth
298 << " no-slices= " << targetHeaderNoSlices
299 << " axis= " << targetHeaderAxis << G4endl;
300 }
301 else if (fVerbose == 1)
302 {
303 G4cout << " VoxelSafety: Depth = " << fVoxelDepth
304 << " Number of Slices = " << targetHeaderNoSlices
305 << " Header (address) = " << targetVoxelHeader << G4endl;
306 }
307#endif
308
309 // Stack info for stepping
310 //
311 fVoxelAxisStack[fVoxelDepth] = targetHeaderAxis;
312 fVoxelNoSlicesStack[fVoxelDepth] = targetHeaderNoSlices;
313 fVoxelSliceWidthStack[fVoxelDepth] = targetHeaderNodeWidth;
314
315 fVoxelHeaderStack[fVoxelDepth] = pHeader;
316
317 G4int trialUp= -1, trialDown= -1;
318 G4double distUp= DBL_MAX, distDown= DBL_MAX;
319
320 // Using Equivalent voxels - this is pre-initialisation only
321 //
322 G4int nextUp= pointNodeNo+1;
323 G4int nextDown= pointNodeNo-1;
324
325 G4int nextNodeNo= pointNodeNo;
326 G4double distAxis; // Distance in current Axis
327 distAxis= 0.0; // Starting in node containing local Coordinate
328
329 G4bool nextIsInside= false;
330
331 G4double distMaxInterest= std::min( previousMinSafety, maxLength);
332 // We will not look beyond this distance.
333 // This distance will be updated to reflect the
334 // max ( minSafety, maxLength ) at each step
335
336 targetNodeNo= pointNodeNo;
337 do
338 {
339 G4double nodeSafety= DBL_MAX, headerSafety= DBL_MAX;
340 fVoxelNodeNoStack[fVoxelDepth] = targetNodeNo;
341
342 checkedNum++;
343
344 sampleProxy = targetVoxelHeader->GetSlice(targetNodeNo);
345
346#ifdef G4DEBUG_NAVIGATION
347 assert( sampleProxy != 0);
348 if( fVerbose > 2 )
349 {
350 G4cout << " -Checking node " << targetNodeNo
351 << " is proxy with address " << sampleProxy << G4endl;
352 }
353#endif
354
355 if ( sampleProxy == 0 )
356 {
357 G4ExceptionDescription ed;
358 ed << " Problem for node number= " << targetNodeNo
359 << " Number of slides = " << targetHeaderNoSlices
360 << G4endl;
361 G4Exception( "G4VoxelSafety::SafetyForVoxelHeader()", "GeomNav0003",
362 FatalException, ed,
363 "Problem sampleProxy is Zero. Failure in loop.");
364 }
365 else if ( sampleProxy->IsNode() )
366 {
367 targetVoxelNode = sampleProxy->GetNode();
368
369 // Deal with the node here [ i.e. the last level ]
370 //
371 nodeSafety= SafetyForVoxelNode( targetVoxelNode, localPoint);
372#ifdef G4DEBUG_NAVIGATION
373 if( fVerbose > 2 )
374 {
375 G4cout << " -- It is a Node ";
376 G4cout << " its safety= " << nodeSafety
377 << " our level Saf = " << ourSafety
378 << " MinSaf= " << minSafety << G4endl;
379 }
380#endif
381 ourSafety= std::min( ourSafety, nodeSafety );
382
383 trialUp = targetVoxelNode->GetMaxEquivalentSliceNo()+1;
384 trialDown = targetVoxelNode->GetMinEquivalentSliceNo()-1;
385 }
386 else
387 {
388 const G4SmartVoxelHeader *pNewVoxelHeader = sampleProxy->GetHeader();
389
390 G4double distCombined_Sq;
391 distCombined_Sq = distUpperDepth_Sq + distAxis*distAxis;
392
393#ifdef G4DEBUG_NAVIGATION
394 if( fVerbose > 2 )
395 {
396 G4double distCombined= std::sqrt( distCombined_Sq );
397 G4double distUpperDepth= std::sqrt ( distUpperDepth_Sq );
398 G4cout << " -- It is a Header " << G4endl;
399 G4cout << " Recurse to deal with next level, fVoxelDepth= "
400 << fVoxelDepth+1 << G4endl;
401 G4cout << " Distances: Upper= " << distUpperDepth
402 << " Axis= " << distAxis
403 << " Combined= " << distCombined << G4endl;
404 }
405#endif
406
407 // Recurse to deal with lower levels
408 //
409 headerSafety= SafetyForVoxelHeader( pNewVoxelHeader, localPoint,
410 maxLength, currentPhysical,
411 distCombined_Sq, minSafety);
412 ourSafety= std::min( ourSafety, headerSafety );
413
414#ifdef G4DEBUG_NAVIGATION
415 if( fVerbose > 2 )
416 {
417 G4cout << " >> Header safety = " << headerSafety
418 << " our level Saf = " << ourSafety << G4endl;
419 }
420#endif
421 trialUp = pNewVoxelHeader->GetMaxEquivalentSliceNo()+1;
422 trialDown = pNewVoxelHeader->GetMinEquivalentSliceNo()-1;
423 }
424 minSafety= std::min( minSafety, ourSafety );
425
426 // Find next closest Voxel
427 // - first try: by simple subtraction
428 // - later: using distance (TODO - tbc)
429 //
430 if( targetNodeNo >= pointNodeNo )
431 {
432 nextUp = trialUp;
433 // distUp = std::max( targetHeaderMax-localCrd, 0.0 );
434 G4double lowerEdgeOfNext = targetHeaderMin
435 + nextUp * targetHeaderNodeWidth;
436 distUp = lowerEdgeOfNext-localCrd ;
437 if( distUp < 0.0 )
438 {
439 distUp = DBL_MAX; // On the wrong side - must not be considered
440 }
441#ifdef G4DEBUG_NAVIGATION
442 if( fVerbose > 2 )
443 {
444 G4cout << " > Updated nextUp= " << nextUp << G4endl;
445 }
446#endif
447 }
448
449 if( targetNodeNo <= pointNodeNo )
450 {
451 nextDown = trialDown;
452 // distDown = std::max( localCrd-targetHeaderMin, 0.0);
453 G4double upperEdgeOfNext = targetHeaderMin
454 + (nextDown+1) * targetHeaderNodeWidth;
455 distDown = localCrd-upperEdgeOfNext;
456 if( distDown < 0.0 )
457 {
458 distDown= DBL_MAX; // On the wrong side - must not be considered
459 }
460#ifdef G4DEBUG_NAVIGATION
461 if( fVerbose > 2 )
462 {
463 G4cout << " > Updated nextDown= " << nextDown << G4endl;
464 }
465#endif
466 }
467
468#ifdef G4DEBUG_NAVIGATION
469 if( fVerbose > 2 )
470 {
471 G4cout << " Node= " << pointNodeNo
472 << " Up: next= " << nextUp << " d# "
473 << nextUp - pointNodeNo
474 << " trialUp= " << trialUp << " d# "
475 << trialUp - pointNodeNo
476 << " Down: next= " << nextDown << " d# "
477 << targetNodeNo - nextDown
478 << " trialDwn= " << trialDown << " d# "
479 << targetNodeNo - trialDown
480 << " condition (next is Inside)= " << nextIsInside
481 << G4endl;
482 }
483#endif
484
485 G4bool UpIsClosest;
486 UpIsClosest= distUp < distDown;
487
488 if( (nextUp < targetHeaderNoSlices)
489 && (UpIsClosest || (nextDown < 0)) )
490 {
491 nextNodeNo=nextUp;
492 distAxis = distUp;
493 ++nextUp; // Default
494#ifdef G4VERBOSE
495 if( fVerbose > 2 )
496 {
497 G4cout << " > Chose Up. Depth= " << fVoxelDepth
498 << " Nodes: next= " << nextNodeNo
499 << " new nextUp= " << nextUp
500 << " Dist = " << distAxis << G4endl;
501 }
502#endif
503 }
504 else
505 {
506 nextNodeNo=nextDown;
507 distAxis = distDown;
508 --nextDown; // A default value
509#ifdef G4VERBOSE
510 if( fVerbose > 2 )
511 {
512 G4cout << " > Chose Down. Depth= " << fVoxelDepth
513 << " Nodes: next= " << nextNodeNo
514 << " new nextDown= " << nextDown
515 << " Dist = " << distAxis << G4endl;
516 }
517#endif
518 }
519
520 nextIsInside = (nextNodeNo >= 0) && (nextNodeNo < targetHeaderNoSlices);
521 if( nextIsInside )
522 {
523 targetNodeNo= nextNodeNo;
524
525#ifdef G4DEBUG_NAVIGATION
526 assert( targetVoxelHeader->GetSlice(nextNodeNo) != 0 );
527 G4bool bContinue= (distAxis<minSafety);
528 if( !bContinue )
529 {
530 if( fVerbose > 2 )
531 {
532 G4cout << " Can skip remaining at depth " << targetHeaderAxis
533 << " >> distAxis= " << distAxis
534 << " minSafety= " << minSafety << G4endl;
535 }
536 }
537#endif
538 }
539 else
540 {
541#ifdef G4DEBUG_NAVIGATION
542 if( fVerbose > 2)
543 {
544 G4cout << " VoxSaf> depth= " << fVoxelDepth << G4endl;
545 G4cout << " VoxSaf> No more candidates: nodeDown= " << nextDown
546 << " nodeUp= " << nextUp
547 << " lastSlice= " << targetHeaderNoSlices << G4endl;
548 }
549#endif
550 }
551
552 // This calculation can be 'hauled'-up to where minSafety is calculated
553 //
554 distMaxInterest = std::min( minSafety, maxLength );
555
556 } while ( nextIsInside && ( distAxis*distAxis + distUpperDepth_Sq
557 < distMaxInterest*distMaxInterest ) );
558
559#ifdef G4VERBOSE
560 if( fVerbose > 0 )
561 {
562 G4cout << " Ended for targetNodeNo -- checked " << checkedNum << " out of "
563 << targetHeaderNoSlices << " slices." << G4endl;
564 G4cout << " ===== Returning from SafetyForVoxelHeader "
565 << " Depth= " << fVoxelDepth << G4endl
566 << G4endl;
567 }
568#endif
569
570 // Go back one level
571 fVoxelDepth--;
572
573 return ourSafety;
574}
