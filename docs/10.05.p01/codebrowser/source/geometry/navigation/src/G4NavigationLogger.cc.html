53// ********************************************************************
54// PreComputeStepLog
55// ********************************************************************
56//
57void
58G4NavigationLogger::PreComputeStepLog(const G4VPhysicalVolume* motherPhysical,
59 G4double motherSafety,
60 const G4ThreeVector& localPoint) const
61{
62 G4VSolid* motherSolid = motherPhysical->GetLogicalVolume()->GetSolid();
63 G4String fType = fId + "::ComputeStep()";
64 // const double millimeter = CLHEP::millimeter;
65
66 if ( fVerbose == 1 || fVerbose > 4 )
67 {
68 G4cout << "*************** " << fType << " *****************" << G4endl
69 << " VolType "
70 << std::setw(15) << "Safety/mm" << " "
71 << std::setw(15) << "Distance/mm" << " "
72 << std::setw(52) << "Position (local coordinates)"
73 << " - Solid" << G4endl;
74 G4cout << " Mother "
75 << std::setw(15) << motherSafety / millimeter << " "
76 << std::setw(15) << "N/C" << " " << localPoint << " - "
77 << motherSolid->GetEntityType() << ": " << motherSolid->GetName()
78 << G4endl;
79 }
80 if ( motherSafety < 0.0 )
81 {
82 std::ostringstream message;
83 message << "Negative Safety In Voxel Navigation !" << G4endl
84 << " Current solid " << motherSolid->GetName()
85 << " gave negative safety: " << motherSafety / millimeter << G4endl
86 << " for the current (local) point " << localPoint;
87 message << " Solid info: " << *motherSolid << G4endl;
88 G4Exception(fType, "GeomNav0003", FatalException, message);
89 }
90 if( motherSolid->Inside(localPoint)==kOutside )
91 {
92 std::ostringstream message;
93 message << "Point is outside Current Volume - " << G4endl
94 << " Point " << localPoint / millimeter
95 << " is outside current volume '" << motherPhysical->GetName()
96 << "'" << G4endl;
97 G4double estDistToSolid= motherSolid->DistanceToIn(localPoint);
98 message << " Estimated isotropic distance to solid (distToIn)= "
99 << estDistToSolid << G4endl;
100 if( estDistToSolid > 100.0 * motherSolid->GetTolerance() )
101 {
102 message << " Solid info: " << *motherSolid << G4endl;
103 G4Exception(fType, "GeomNav0003", JustWarning, message,
104 "Point is far outside Current Volume !" );
105 }
106 else
107 {
108 G4Exception(fType, "GeomNav1001", JustWarning, message,
109 "Point is a little outside Current Volume.");
110 }
111 }
112
113 // Verification / verbosity
114 //
115 if ( fVerbose > 1 )
116 {
117 static const G4int precVerf= 16; // Precision
118 G4int oldprec = G4cout.precision(precVerf);
119 G4cout << " - Information on mother / key daughters ..." << G4endl;
120 G4cout << " Type " << std::setw(12) << "Solid-Name" << " "
121 << std::setw(3*(6+precVerf)) << " local point" << " "
122 << std::setw(4+precVerf) << "solid-Safety" << " "
123 << std::setw(4+precVerf) << "solid-Step" << " "
124 << std::setw(17) << "distance Method "
125 << std::setw(3*(6+precVerf)) << " local direction" << " "
126 << G4endl;
127 G4cout << " Mother " << std::setw(12) << motherSolid->GetName() << " "
128 << std::setw(4+precVerf) << localPoint << " "
129 << std::setw(4+precVerf) << motherSafety << " "
130 << G4endl;
131 G4cout.precision(oldprec);
132 }
133}

135// ********************************************************************
136// AlongComputeStepLog
137// ********************************************************************
138//
139void
140G4NavigationLogger::AlongComputeStepLog(const G4VSolid* sampleSolid,
141 const G4ThreeVector& samplePoint,
142 const G4ThreeVector& sampleDirection,
143 const G4ThreeVector& localDirection,
144 G4double sampleSafety,
145 G4double sampleStep) const
146{
147 // Check to see that the resulting point is indeed in/on volume.
148 // This check could eventually be made only for successful candidate.
149
150 if ( sampleStep < kInfinity )
151 {
152 G4ThreeVector intersectionPoint;
153 intersectionPoint= samplePoint + sampleStep * sampleDirection;
154 EInside insideIntPt= sampleSolid->Inside(intersectionPoint);
155 G4String fType = fId + "::ComputeStep()";
156
157 G4String solidResponse = "-kInside-";
158 if (insideIntPt == kOutside)
159 { solidResponse = "-kOutside-"; }
160 else if (insideIntPt == kSurface)
161 { solidResponse = "-kSurface-"; }
162
163 if ( fVerbose == 1 || fVerbose > 4 )
164 {
165 G4cout << " Invoked Inside() for solid: "
166 << sampleSolid->GetName()
167 << ". Solid replied: " << solidResponse << G4endl
168 << " For point p: " << intersectionPoint
169 << ", considered as 'intersection' point." << G4endl;
170 }
171
172 G4double safetyIn= -1, safetyOut= -1; // Set to invalid values
173 G4double newDistIn= -1, newDistOut= -1;
174 if( insideIntPt != kInside )
175 {
176 safetyIn= sampleSolid->DistanceToIn(intersectionPoint);
177 newDistIn= sampleSolid->DistanceToIn(intersectionPoint,
178 sampleDirection);
179 }
180 if( insideIntPt != kOutside )
181 {
182 safetyOut= sampleSolid->DistanceToOut(intersectionPoint);
183 newDistOut= sampleSolid->DistanceToOut(intersectionPoint,
184 sampleDirection);
185 }
186 if( insideIntPt != kSurface )
187 {
188 std::ostringstream message;
189 message.precision(16);
190 message << "Conflicting response from Solid." << G4endl
191 << " Inaccurate solid DistanceToIn"
192 << " for solid " << sampleSolid->GetName() << G4endl
193 << " Solid gave DistanceToIn = "
194 << sampleStep << " yet returns " << solidResponse
195 << " for this point !" << G4endl
196 << " Original Point = " << samplePoint << G4endl
197 << " Original Direction = " << sampleDirection << G4endl
198 << " Intersection Point = " << intersectionPoint << G4endl
199 << " Safety values: " << G4endl;
200 if ( insideIntPt != kInside )
201 {
202 message << " DistanceToIn(p) = " << safetyIn;
203 }
204 if ( insideIntPt != kOutside )
205 {
206 message << " DistanceToOut(p) = " << safetyOut;
207 }
208 message << G4endl;
209 message << " Solid Parameters: " << *sampleSolid;
210 G4Exception(fType, "GeomNav1001", JustWarning, message);
211 }
212 else
213 {
214 // If it is on the surface, *ensure* that either DistanceToIn
215 // or DistanceToOut returns a finite value ( >= Tolerance).
216 //
217 if( std::max( newDistIn, newDistOut ) <=
218 G4GeometryTolerance::GetInstance()->GetSurfaceTolerance() )
219 {
220 std::ostringstream message;
221 message << "Zero from both Solid DistanceIn and Out(p,v)." << G4endl
222 << " Identified point for which the solid "
223 << sampleSolid->GetName() << G4endl
224 << " has MAJOR problem: " << G4endl
225 << " --> Both DistanceToIn(p,v) and DistanceToOut(p,v) "
226 << "return Zero, an equivalent value or negative value."
227 << G4endl
228 << " Solid: " << sampleSolid << G4endl
229 << " Point p= " << intersectionPoint << G4endl
230 << " Direction v= " << sampleDirection << G4endl
231 << " DistanceToIn(p,v) = " << newDistIn << G4endl
232 << " DistanceToOut(p,v,..) = " << newDistOut << G4endl
233 << " Safety values: " << G4endl
234 << " DistanceToIn(p) = " << safetyIn << G4endl
235 << " DistanceToOut(p) = " << safetyOut;
236 G4Exception(fType, "GeomNav0003", FatalException, message);
237 }
238 }
239
240 // Verification / verbosity
241 //
242 if ( fVerbose > 1 )
243 {
244 static const G4int precVerf= 20; // Precision
245 G4int oldprec = G4cout.precision(precVerf);
246 G4cout << "Daughter "
247 << std::setw(12) << sampleSolid->GetName() << " "
248 << std::setw(4+precVerf) << samplePoint << " "
249 << std::setw(4+precVerf) << sampleSafety << " "
250 << std::setw(4+precVerf) << sampleStep << " "
251 << std::setw(16) << "distanceToIn" << " "
252 << std::setw(4+precVerf) << localDirection << " "
253 << G4endl;
254 G4cout.precision(oldprec);
255 }
256 }
257}

259// ********************************************************************
260// CheckDaughterEntryPoint
261// ********************************************************************
262//
263void
264G4NavigationLogger::CheckDaughterEntryPoint(const G4VSolid* sampleSolid,
265 const G4ThreeVector& samplePoint,
266 const G4ThreeVector& sampleDirection,
267 const G4VSolid* motherSolid,
268 const G4ThreeVector& localPoint,
269 const G4ThreeVector& localDirection,
270 G4double motherStep,
271 G4double sampleStep) const
272{
273 const G4double kCarTolerance = motherSolid->GetTolerance();
274
275 // Double check the expected condition of being called
276 //
277 G4bool SuspiciousDaughterDist = ( sampleStep >= motherStep )
278 && ( sampleStep < kInfinity );
279
280 if( sampleStep >= kInfinity )
281 {
282 G4ExceptionDescription msg;
283 msg.precision(12);
284 msg << " WARNING - Called with 'infinite' step. " << G4endl;
285 msg << " Checks have no meaning if daughter step is infinite." << G4endl;
286 msg << " kInfinity = " << kInfinity / millimeter << G4endl;
287 msg << " sampleStep = " << sampleStep / millimeter << G4endl;
288 msg << " sampleStep < kInfinity " << (sampleStep<kInfinity) << G4endl;
289 msg << " kInfinity - sampleStep " << (kInfinity-sampleStep) / millimeter << G4endl;
290 msg << " Returning immediately.";
291 G4Exception("G4NavigationLogger::CheckDaughterEntryPoint()",
292 "GeomNav0003", JustWarning, msg);
293 return;
294 }
295
296 // The intersection point with the daughter is after the exit point
297 // from the mother volume !!
298 // This is legal / allowed to occur only if the mother is concave
299 // ****************************************************************
300 // If mother is convex the daughter volume must be encountered
301 // before the exit from the current volume!
302
303 // Check #1) whether the track will re-enter the current mother
304 // in the extension past its current exit point
305 G4ThreeVector localExitMotherPos= localPoint+motherStep*localDirection;
306 G4double distExitToReEntry= motherSolid->DistanceToIn(localExitMotherPos,
307 localDirection);
308
309 // Check #2) whether the 'entry' point in the daughter is inside the mother
310 //
311 G4ThreeVector localEntryInDaughter = localPoint+sampleStep*localDirection;
312 EInside insideMother= motherSolid->Inside( localEntryInDaughter );
313
314 G4String solidResponse = "-kInside-";
315 if (insideMother == kOutside) { solidResponse = "-kOutside-"; }
316 else if (insideMother == kSurface) { solidResponse = "-kSurface-"; }
317
318 G4double distToReEntry = distExitToReEntry + motherStep;
319 G4ThreeVector localReEntryPoint = localPoint+distToReEntry*localDirection;
320
321 // Clear error -- Daughter entry point is bad
322 G4bool DaughterEntryIsOutside = SuspiciousDaughterDist
323 && ( (sampleStep < distToReEntry) || (insideMother == kOutside ) );
324 G4bool EntryIsMotherExit = std::fabs(sampleStep-motherStep) < kCarTolerance;
325
326 // Check for more subtle error - is exit point of daughter correct ?
327 G4ThreeVector sampleEntryPoint= samplePoint+sampleStep*sampleDirection;
328 G4double sampleCrossingDist= sampleSolid->DistanceToOut( sampleEntryPoint,
329 sampleDirection );
330 G4double sampleExitDist = sampleStep+sampleCrossingDist;
331 G4ThreeVector sampleExitPoint= samplePoint+sampleExitDist*sampleDirection;
332
333 G4bool TransitProblem = ( (sampleStep < motherStep)
334 && (sampleExitDist > motherStep + kCarTolerance) )
335 || ( EntryIsMotherExit && (sampleCrossingDist > kCarTolerance) );
336
337 if( DaughterEntryIsOutside
338 || TransitProblem
339 || (SuspiciousDaughterDist && (fVerbose > 3) ) )
340 {
341 G4ExceptionDescription msg;
342 msg.precision(16);
343
344 if( DaughterEntryIsOutside )
345 {
346 msg << "WARNING> Intersection distance to Daughter volume is further"
347 << " than the distance to boundary." << G4endl
348 << " It appears that part of the daughter volume is *outside*"
349 << " this mother. " << G4endl;
350 msg << " One of the following checks signaled a problem:" << G4endl
351 << " -sampleStep (dist to daugh) < mother-exit dist + distance "
352 << "to ReEntry point for mother " << G4endl
353 << " -position of daughter intersection is outside mother volume."
354 << G4endl;
355 }
356 else if( TransitProblem )
357 {
358 G4double protrusion = sampleExitDist - motherStep;
359
360 msg << "WARNING> Daughter volume extends beyond mother boundary. "
361 << G4endl;
362 if ( ( sampleStep < motherStep )
363 && (sampleExitDist > motherStep + kCarTolerance ) )
364 {
365 // 1st Issue with Daughter
366 msg << " Crossing distance in the daughter causes is to extend"
367 << " beyond the mother exit. " << G4endl;
368 msg << " Length protruding = " << protrusion << G4endl;
369 }
370 if( EntryIsMotherExit )
371 {
372 // 1st Issue with Daughter
373 msg << " Intersection distance to Daughter is within "
374 << " tolerance of the distance" << G4endl;
375 msg << " to the mother boundary * and * " << G4endl;
376 msg << " the crossing distance in the daughter is > tolerance."
377 << G4endl;
378 }
379 }
380 else
381 {
382 msg << "NearMiss> Intersection to Daughter volume is in extension past the"
383 << " current exit point of the mother volume." << G4endl;
384 msg << " This is not an error - just an unusual occurrence,"
385 << " possible in the case of concave volume. " << G4endl;
386 }
387 msg << "---- Information about intersection with daughter, mother: "
388 << G4endl;
389 msg << " sampleStep (daughter) = " << sampleStep << G4endl
390 << " motherStep = " << motherStep << G4endl
391 << " distToRentry(mother) = " << distToReEntry << G4endl
392 << " Inside(entry pnt daug): " << solidResponse << G4endl
393 << " dist across daughter = " << sampleCrossingDist << G4endl;
394 msg << " Mother Name (Solid) : " << motherSolid->GetName() << G4endl
395 << " In local (mother) coordinates: " << G4endl
396 << " Starting Point = " << localPoint << G4endl
397 << " Direction = " << localDirection << G4endl
398 << " Exit Point (mother)= " << localExitMotherPos << G4endl
399 << " Entry Point (daughter)= " << localPoint+sampleStep*localDirection
400 << G4endl;
401 if( distToReEntry < kInfinity )
402 {
403 msg << " ReEntry Point (mother)= " << localReEntryPoint << G4endl;
404 }
405 else
406 {
407 msg << " No ReEntry - track does not encounter mother volume again! "
408 << G4endl;
409 }
410 msg << " Daughter Name (Solid): " << sampleSolid->GetName() << G4endl
411 << " In daughter coordinates: " << G4endl
412 << " Starting Point = " << samplePoint << G4endl
413 << " Direction = " << sampleDirection << G4endl
414 << " Entry Point (daughter)= " << sampleEntryPoint
415 << G4endl;
416 msg << " Description of mother solid: " << G4endl
417 << *motherSolid << G4endl
418 << " Description of daughter solid: " << G4endl
419 << *sampleSolid << G4endl;
420 G4String fType = fId + "::ComputeStep()";
421
422 if( DaughterEntryIsOutside || TransitProblem )
423 {
424 G4Exception(fType, "GeomNav0003", JustWarning, msg);
425 }
426 else
427 {
428 G4cout << fType
429 << " -- Checked distance of Entry to daughter vs exit of mother"
430 << G4endl;
431 G4cout << msg.str();
432 G4cout << G4endl;
433 }
434 }
435}

556// ********************************************************************
557// CheckAndReportBadNormal
558// ********************************************************************
559//
560G4bool
561G4NavigationLogger::
562CheckAndReportBadNormal(const G4ThreeVector& unitNormal,
563 const G4ThreeVector& localPoint,
564 const G4ThreeVector& localDirection,
565 G4double step,
566 const G4VSolid* solid,
567 const char* msg ) const
568{
569 G4double normMag2 = unitNormal.mag2();
570 G4bool badLength = ( std::fabs ( normMag2 - 1.0 ) > CLHEP::perMillion );
571
572 if( badLength )
573 {
574 G4double normMag= std::sqrt(normMag2);
575 G4ExceptionDescription message;
576 message.precision(10);
577 message << "============================================================"
578 << G4endl;
579 message << " WARNING> Normal is not a unit vector. "
580 << " - but |normal| = " << normMag
581 << " - and |normal|^2 = " << normMag2 << G4endl
582 << " which differ from 1.0 by: " << G4endl
583 << " |normal|-1 = " << normMag - 1.0
584 << " and |normal|^2 - 1 = " << normMag2 - 1.0 << G4endl
585 << " n = " << unitNormal << G4endl;
586 message << " Info string: " << msg << G4endl;
587 message << "============================================================"
588 << G4endl;
589
590 message.precision(16);
591
592 message << " Information on call to DistanceToOut: " << G4endl;
593 message << " Position = " << localPoint << G4endl
594 << " Direction = " << localDirection << G4endl;
595 message << " Obtained> distance = " << step << G4endl;
596 message << " > Exit position = " << localPoint+step*localDirection
597 << G4endl;
598 message << " Parameters of solid: " << G4endl;
599 message << *solid;
600 message << "============================================================";
601
602 G4String fMethod = fId + "::ComputeStep()";
603 G4Exception( fMethod, "GeomNav0003", JustWarning, message);
604 }
605 return badLength;
606}

608// ********************************************************************

612G4bool
613G4NavigationLogger::
614CheckAndReportBadNormal(const G4ThreeVector& rotatedNormal,

618{

621
622 if( badLength )
623 {
624 G4double normMag= std::sqrt(normMag2);
625 G4ExceptionDescription message;
626 message.precision(10);
627 message << "============================================================"
628 << G4endl;
629 message << " WARNING> Rotated n(ormal) is not a unit vector. " << G4endl
630 << " |normal| = " << normMag
631 << " and |normal|^2 = " << normMag2 << G4endl
632 << " Diff from 1.0: " << G4endl
633 << " |normal|-1 = " << normMag - 1.0
634 << " and |normal|^2 - 1 = " << normMag2 - 1.0 << G4endl;
635 message << " Rotated n = (" << rotatedNormal.x() << "," << rotatedNormal.y() << ","
636 << rotatedNormal.z() << ")" << G4endl;
637 message << " Original n = (" << originalNormal.x() << "," << originalNormal.y() << ","
638 << originalNormal.z() << ")" << G4endl;
639 message << " Info string: " << msg << G4endl;
640 message << "============================================================"
641 << G4endl;
642
643 message.precision(16);
644
645 message << " Information on RotationMatrix : " << G4endl;
646 message << " Original: " << G4endl;
647 message << rotationM << G4endl;
648 message << " Inverse (used in transformation): " << G4endl;
649 message << rotationM.inverse() << G4endl;
650 message << "============================================================";
651
652 G4String fMethod = fId + "::ComputeStep()";
653 G4Exception( fMethod, "GeomNav0003", JustWarning, message);
654 }
655 return badLength;
656}

658// ********************************************************************
659// ReportOutsideMother
660// ********************************************************************
661//
662// Report Exception if point is outside mother.
663// Fatal exception will be used if either 'checkMode error is > triggerDist
664//
665void
666G4NavigationLogger::ReportOutsideMother(const G4ThreeVector& localPoint,
667 const G4ThreeVector& localDirection,
668 const G4VPhysicalVolume* physical,
669 G4double triggerDist) const
670{
671 const G4LogicalVolume* logicalVol = physical
672 ? physical->GetLogicalVolume() : 0;
673 const G4VSolid* solid = logicalVol
674 ? logicalVol->GetSolid() : 0;
675
676 G4String fMethod = fId + "::ComputeStep()";
677
678 if( solid == 0 )
679 {
680 G4Exception(fMethod, "GeomNav0003", FatalException,
681 "Erroneous call to ReportOutsideMother: no Solid is available");
682 return;
683 }
684 const G4double kCarTolerance = solid->GetTolerance();
685
686 // Double check reply - it should be kInfinity
687 const G4double distToOut = solid->DistanceToOut(localPoint, localDirection);
688 // const G4double distToOutNeg = solid->DistanceToOut(localPoint, -localDirection);
689
690 const EInside inSolid = solid->Inside(localPoint);
691 const G4double safetyToIn = solid->DistanceToIn(localPoint);
692 const G4double safetyToOut = solid->DistanceToOut(localPoint);
693
694 const G4double distToInPos = solid->DistanceToIn(localPoint, localDirection);
695 // const G4double distToInNeg = solid->DistanceToIn(localPoint, -localDirection);
696
697 // 1. Check consistency between Safety obtained and report from distance
698 // We must ensure that (mother)Safety <= 0.0
699 // in the case that the point is outside!
700 // [ If this is not the case, this problem can easily go undetected,
701 // except in Check mode ! ]
702 if( safetyToOut > kCarTolerance
703 && ( distToOut < 0.0 || distToOut >= kInfinity ) )
704 {
705 G4ExceptionDescription msg1;
706 // fNavClerk->ReportBadSafety(localPoint, localDirection,
707 // motherPhysical, motherSafety, motherStep);
708 msg1 << " Dangerous inconsistency in response of solid." << G4endl
709 << " Solid type: " << solid->GetEntityType()
710 << " Name= " << solid->GetName() << G4endl;
711 msg1 << " Mother volume gives safety > 0 despite being called for *Outside* point "
712 << G4endl
713 << " Location = " << localPoint << G4endl
714 << " Direction= " << localDirection << G4endl
715 << " - Safety (Isotropic d) = " << safetyToOut << G4endl
716 << " - Intersection Distance= " << distToOut << G4endl
717 << G4endl;
718 G4Exception( fMethod, "GeomNav0123", JustWarning, msg1);
719 }
720
721 // 2. Inconsistency - Too many distances are zero (or will be rounded to zero)
722
723 G4ExceptionDescription msg2;
724 if( std::fabs(distToOut) < kCarTolerance && std::fabs(distToInPos) < kCarTolerance )
725 {
726 // If both distanceToIn and distanceToOut (p,v) are zero for one direction,
727 // the particle could get stuck!
728 }
729
730 G4ExceptionDescription msg;
731 msg.precision(10);
732 // G4bool reportIssue= true;
733
734 if( std::fabs(distToOut) < kCarTolerance )
735 {
736 // 3. Soft error - safety is not rounded to zero
737 // Report nothing - except in 'loud' mode
738 if( fReportSoftWarnings )
739 {
740 // reportIssue= true;
741 msg << " Warning> DistanceToOut(p,v): Distance from surface is not rounded to zero" << G4endl;
742 } else {
743 // reportIssue= false;
744 return;
745 }
746 }
747 else
748 {
749 // 4. General message - complain that the point is outside!
750 // and provide all information about the current location,
751 // direction and the answers of the solid
752 msg << "============================================================" << G4endl;
753 msg << " WARNING> Current Point appears to be Outside mother volume !! " << G4endl;
754 msg << " Response of DistanceToOut was negative or kInfinity when called in "
755 << fMethod << G4endl;
756 }
757
758 // Generate and 'print'/stream all the information needed
759 this->ReportVolumeAndIntersection( msg, localPoint, localDirection, physical );
760
761 // Default for distance of 'major' error
762 if( triggerDist <= 0.0 ) {
763 // triggerDist = 1.e+6 * kCarTolerance; // Well beyond tolerance
764 triggerDist = std::max ( 1.0e+6 * kCarTolerance, // Well beyond tolerance
765 fMinTriggerDistance );
766 }
767
768 G4bool majorError = inSolid==kOutside
769 ? ( safetyToIn > triggerDist )
770 : ( safetyToOut > triggerDist );
771
772 G4ExceptionSeverity exceptionType = JustWarning;
773 if ( majorError )
774 {
775 exceptionType = FatalException;
776 }
777
778 G4Exception( fMethod, "GeomNav0003", exceptionType, msg);
779}

785void
786G4NavigationLogger::ReportVolumeAndIntersection( std::ostream& os,
787 const G4ThreeVector& localPoint,
788 const G4ThreeVector& localDirection,
789 const G4VPhysicalVolume* physical ) const
790{
791 G4String fMethod = fId + "::ComputeStep()";
792 const G4LogicalVolume* logicalVol = physical
793 ? physical->GetLogicalVolume() : 0;
794 const G4VSolid* solid = logicalVol
795 ? logicalVol->GetSolid() : 0;
796 if( solid == 0 )
797 {
798 os << " ERROR> Solid is not available. Logical Volume = " << logicalVol << std::endl;
799 return;
800 }
801 const G4double kCarTolerance = solid->GetTolerance();
802
803 // Double check reply - it should be kInfinity
804 const G4double distToOut = solid->DistanceToOut(localPoint, localDirection);
805 const G4double distToOutNeg = solid->DistanceToOut(localPoint, -localDirection);
806
807 const EInside inSolid = solid->Inside(localPoint);
808 const G4double safetyToIn = solid->DistanceToIn(localPoint);
809 const G4double safetyToOut = solid->DistanceToOut(localPoint);
810
811 const G4double distToInPos = solid->DistanceToIn(localPoint, localDirection);
812 const G4double distToInNeg = solid->DistanceToIn(localPoint, -localDirection);
813
814 const G4ThreeVector exitNormal = solid->SurfaceNormal(localPoint);
815
816 // Double check whether points nearby are in/surface/out
817 const G4double epsilonDist = 1000.0 * kCarTolerance;
818 const G4ThreeVector PointPlusDir = localPoint + epsilonDist * localDirection;
819 const G4ThreeVector PointMinusDir = localPoint - epsilonDist * localDirection;
820 const G4ThreeVector PointPlusNorm = localPoint + epsilonDist * exitNormal;
821 const G4ThreeVector PointMinusNorm = localPoint - epsilonDist * exitNormal;
822
823 const EInside inPlusDir= solid->Inside(PointPlusDir);
824 const EInside inMinusDir= solid->Inside(PointMinusDir);
825 const EInside inPlusNorm= solid->Inside(PointPlusNorm);
826 const EInside inMinusNorm= solid->Inside(PointMinusNorm);
827
828 // Basic information
829 os << " Current physical volume = " << physical->GetName() << G4endl;
830 os << " Position (loc) = " << localPoint << G4endl
831 << " Direction (dir) = " << localDirection << G4endl;
832 os << " For confirmation:" << G4endl;
833 os << " Response of DistanceToOut (loc, +dir)= " << distToOut << G4endl;
834 os << " Response of DistanceToOut (loc, -dir)= " << distToOutNeg << G4endl;
835
836 os << " Inside responds = " << inSolid << " , ie: ";
837 if( inSolid == kOutside ) {
838 os << " Outside -- a problem, as observed in " << fMethod << G4endl;
839 } else if( inSolid == kSurface ) {
840 os << " Surface -- unexpected / inconsistent response ! " << G4endl;
841 } else {
842 os << " Inside -- unexpected / inconsistent response ! " << G4endl;
843 }
844 os << " Obtain safety(ToIn) = " << safetyToIn << G4endl;
845 os << " Obtain safety(ToOut) = " << safetyToOut << G4endl;
846 os << " Response of DistanceToIn (loc, +dir)= " << distToInPos << G4endl;
847 os << " Response of DistanceToIn (loc, -dir)= " << distToInNeg << G4endl;
848
849 os << " Exit Normal at loc = " << exitNormal << G4endl;
850 os << " Dir . Normal = " << exitNormal.dot( localDirection );
851 os << G4endl;
852
853 os << " Checking points moved from position by distance/dir. Solid responses: " << G4endl
854 << " +eps in direction : " << G4NavigationLogger_Namespace::EInsideNames[inPlusDir]
855 << " +eps in Normal : " << G4NavigationLogger_Namespace::EInsideNames[inPlusNorm] << G4endl
856 << " -eps in direction : " << G4NavigationLogger_Namespace::EInsideNames[inMinusDir]
857 << " -eps in Normal : " << G4NavigationLogger_Namespace::EInsideNames[inMinusNorm] << G4endl;
858
859 os << " Parameters of solid: " << G4endl;
860 os << *solid;
861 os << "============================================================";
862}
