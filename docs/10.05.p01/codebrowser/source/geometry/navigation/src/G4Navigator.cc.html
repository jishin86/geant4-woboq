111// ********************************************************************
112// LocateGlobalPointAndSetup
113//
114// Locate the point in the hierarchy return 0 if outside
115// The direction is required
116// - if on an edge shared by more than two surfaces
117// (to resolve likely looping in tracking)
118// - at initial location of a particle
119// (to resolve potential ambiguity at boundary)
120//
121// Flags on exit: (comments to be completed)
122// fEntering - True if entering `daughter' volume (or replica)
123// whether daughter of last mother directly
124// or daughter of that volume's ancestor.
125// fExiting - True if exited 'mother' volume
126// (always ? - how about if going back down ? - tbc)
127// ********************************************************************
128//
129G4VPhysicalVolume*
130G4Navigator::LocateGlobalPointAndSetup( const G4ThreeVector& globalPoint,
131 const G4ThreeVector* pGlobalDirection,
132 const G4bool relativeSearch,
133 const G4bool ignoreDirection )
134{
135 G4bool notKnownContained=true, noResult;
136 G4VPhysicalVolume *targetPhysical;
137 G4LogicalVolume *targetLogical;
138 G4VSolid *targetSolid=0;
139 G4ThreeVector localPoint, globalDirection;
140 EInside insideCode;
141
142 G4bool considerDirection = (!ignoreDirection) || fLocatedOnEdge;
143
144 fLastTriedStepComputation= false;
145 fChangedGrandMotherRefFrame= false; // For local exit normal
146
147 if( considerDirection && pGlobalDirection != 0 )
148 {
149 globalDirection=*pGlobalDirection;
150 }
151
152#ifdef G4VERBOSE
153 if( fVerbose > 2 )
154 {
155 G4int oldcoutPrec = G4cout.precision(8);
156 G4cout << "*** G4Navigator::LocateGlobalPointAndSetup: ***" << G4endl;
157 G4cout << " Called with arguments: " << G4endl
158 << " Globalpoint = " << globalPoint << G4endl
159 << " RelativeSearch = " << relativeSearch << G4endl;
160 if( fVerbose >= 4 )
161 {
162 G4cout << " ----- Upon entering:" << G4endl;
163 PrintState();
164 }
165 G4cout.precision(oldcoutPrec);
166 }
167#endif
168
169 G4int noLevelsExited=0 ;
170 G4int noLevelsEntered= 0;
171
172 if ( !relativeSearch )
173 {
174 ResetStackAndState();
175 }
176 else
177 {
178 if ( fWasLimitedByGeometry )
179 {
180 fWasLimitedByGeometry = false;
181 fEnteredDaughter = fEntering; // Remember
182 fExitedMother = fExiting; // Remember
183 if ( fExiting )
184 {
185 noLevelsExited++; // count this first level entered too
186
187 if ( fHistory.GetDepth() )
188 {
189 fBlockedPhysicalVolume = fHistory.GetTopVolume();
190 fBlockedReplicaNo = fHistory.GetTopReplicaNo();
191 fHistory.BackLevel();
192 }
193 else
194 {
195 fLastLocatedPointLocal = localPoint;
196 fLocatedOutsideWorld = true;
197 fBlockedPhysicalVolume = 0; // to be sure
198 fBlockedReplicaNo = -1;
199 fEntering = false; // No longer
200 fEnteredDaughter = false;
201 fExitedMother = true; // ??
202
203 return 0; // Have exited world volume
204 }
205 // A fix for the case where a volume is "entered" at an edge
206 // and a coincident surface exists outside it.
207 // - This stops it from exiting further volumes and cycling
208 // - However ReplicaNavigator treats this case itself
209 //
210 // assert( fBlockedPhysicalVolume!=0 );
211
212 // Expect to be on edge => on surface
213 //
214 if ( fLocatedOnEdge && (VolumeType(fBlockedPhysicalVolume)!=kReplica ))
215 {
216 fExiting= false;
217 // Consider effect on Exit Normal !?
218 }
219 }
220 else
221 if ( fEntering )
222 {
223 // assert( fBlockedPhysicalVolume!=0 );
224
225 noLevelsEntered++; // count the first level entered too
226
227 switch (VolumeType(fBlockedPhysicalVolume))
228 {
229 case kNormal:
230 fHistory.NewLevel(fBlockedPhysicalVolume, kNormal,
231 fBlockedPhysicalVolume->GetCopyNo());
232 break;
233 case kReplica:
234 freplicaNav.ComputeTransformation(fBlockedReplicaNo,
235 fBlockedPhysicalVolume);
236 fHistory.NewLevel(fBlockedPhysicalVolume, kReplica,
237 fBlockedReplicaNo);
238 fBlockedPhysicalVolume->SetCopyNo(fBlockedReplicaNo);
239 break;
240 case kParameterised:
241 if( fBlockedPhysicalVolume->GetRegularStructureId() == 0 )
242 {
243 G4VSolid *pSolid;
244 G4VPVParameterisation *pParam;
245 G4TouchableHistory parentTouchable( fHistory );
246 pParam = fBlockedPhysicalVolume->GetParameterisation();
247 pSolid = pParam->ComputeSolid(fBlockedReplicaNo,
248 fBlockedPhysicalVolume);
249 pSolid->ComputeDimensions(pParam, fBlockedReplicaNo,
250 fBlockedPhysicalVolume);
251 pParam->ComputeTransformation(fBlockedReplicaNo,
252 fBlockedPhysicalVolume);
253 fHistory.NewLevel(fBlockedPhysicalVolume, kParameterised,
254 fBlockedReplicaNo);
255 fBlockedPhysicalVolume->SetCopyNo(fBlockedReplicaNo);
256 //
257 // Set the correct solid and material in Logical Volume
258 //
259 G4LogicalVolume *pLogical;
260 pLogical = fBlockedPhysicalVolume->GetLogicalVolume();
261 pLogical->SetSolid( pSolid );
262 pLogical->UpdateMaterial(pParam ->
263 ComputeMaterial(fBlockedReplicaNo,
264 fBlockedPhysicalVolume,
265 &parentTouchable));
266 }
267 break;
268 }
269 fEntering = false;
270 fBlockedPhysicalVolume = 0;
271 localPoint = fHistory.GetTopTransform().TransformPoint(globalPoint);
272 notKnownContained = false;
273 }
274 }
275 else
276 {
277 fBlockedPhysicalVolume = 0;
278 fEntering = false;
279 fEnteredDaughter = false; // Full Step was not taken, did not enter
280 fExiting = false;
281 fExitedMother = false; // Full Step was not taken, did not exit
282 }
283 }
284 //
285 // Search from top of history up through geometry until
286 // containing volume found:
287 // If on
288 // o OUTSIDE - Back up level, not/no longer exiting volumes
289 // o SURFACE and EXITING - Back up level, setting new blocking no.s
290 // else
291 // o containing volume found
292 //
293
294 while (notKnownContained) // Loop checking, 07.10.2016, J.Apostolakis
295 {
296 if ( fHistory.GetTopVolumeType()!=kReplica )
297 {
298 targetSolid = fHistory.GetTopVolume()->GetLogicalVolume()->GetSolid();
299 localPoint = fHistory.GetTopTransform().TransformPoint(globalPoint);
300 insideCode = targetSolid->Inside(localPoint);
301#ifdef G4VERBOSE
302 if(( fVerbose == 1 ) && ( fCheck ))
303 {
304 G4String solidResponse = "-kInside-";
305 if (insideCode == kOutside)
306 solidResponse = "-kOutside-";
307 else if (insideCode == kSurface)
308 solidResponse = "-kSurface-";
309 G4cout << "*** G4Navigator::LocateGlobalPointAndSetup(): ***" << G4endl
310 << " Invoked Inside() for solid: " << targetSolid->GetName()
311 << ". Solid replied: " << solidResponse << G4endl
312 << " For local point p: " << localPoint << G4endl;
313 }
314#endif
315 }
316 else
317 {
318 insideCode = freplicaNav.BackLocate(fHistory, globalPoint, localPoint,
319 fExiting, notKnownContained);
320 // !CARE! if notKnownContained returns false then the point is within
321 // the containing placement volume of the replica(s). If insidecode
322 // will result in the history being backed up one level, then the
323 // local point returned is the point in the system of this new level
324 }
325
326
327 if ( insideCode==kOutside )
328 {
329 noLevelsExited++;
330 if ( fHistory.GetDepth() )
331 {
332 fBlockedPhysicalVolume = fHistory.GetTopVolume();
333 fBlockedReplicaNo = fHistory.GetTopReplicaNo();
334 fHistory.BackLevel();
335 fExiting = false;
336
337 if( noLevelsExited > 1 )
338 {
339 // The first transformation was done by the sub-navigator
340 //
341 const G4RotationMatrix* mRot = fBlockedPhysicalVolume->GetRotation();
342 if( mRot )
343 {
344 fGrandMotherExitNormal *= (*mRot).inverse();
345 fChangedGrandMotherRefFrame= true;
346 }
347 }
348 }
349 else
350 {
351 fLastLocatedPointLocal = localPoint;
352 fLocatedOutsideWorld = true;
353 // No extra transformation for ExitNormal - is in frame of Top Volume
354 return 0; // Have exited world volume
355 }
356 }
357 else
358 if ( insideCode==kSurface )
359 {
360 G4bool isExiting = fExiting;
361 if( (!fExiting)&&considerDirection )
362 {
363 // Figure out whether we are exiting this level's volume
364 // by using the direction
365 //
366 G4bool directionExiting = false;
367 G4ThreeVector localDirection =
368 fHistory.GetTopTransform().TransformAxis(globalDirection);
369
370 // Make sure localPoint in correct reference frame
371 // ( Was it already correct ? How ? )
372 //
373 localPoint= fHistory.GetTopTransform().TransformPoint(globalPoint);
374 if ( fHistory.GetTopVolumeType()!=kReplica )
375 {
376 G4ThreeVector normal = targetSolid->SurfaceNormal(localPoint);
377 directionExiting = normal.dot(localDirection) > 0.0;
378 isExiting = isExiting || directionExiting;
379 }
380 }
381 if( isExiting )
382 {
383 noLevelsExited++;
384 if ( fHistory.GetDepth() )
385 {
386 fBlockedPhysicalVolume = fHistory.GetTopVolume();
387 fBlockedReplicaNo = fHistory.GetTopReplicaNo();
388 fHistory.BackLevel();
389 //
390 // Still on surface but exited volume not necessarily convex
391 //
392 fValidExitNormal = false;
393
394 if( noLevelsExited > 1 )
395 {
396 // The first transformation was done by the sub-navigator
397 //
398 const G4RotationMatrix* mRot =
399 fBlockedPhysicalVolume->GetRotation();
400 if( mRot )
401 {
402 fGrandMotherExitNormal *= (*mRot).inverse();
403 fChangedGrandMotherRefFrame= true;
404 }
405 }
406 }
407 else
408 {
409 fLastLocatedPointLocal = localPoint;
410 fLocatedOutsideWorld = true;
411 // No extra transformation for ExitNormal, is in frame of Top Vol
412 return 0; // Have exited world volume
413 }
414 }
415 else
416 {
417 notKnownContained=false;
418 }
419 }
420 else
421 {
422 notKnownContained=false;
423 }
424 } // END while (notKnownContained)
425 //
426 // Search downwards until deepest containing volume found,
427 // blocking fBlockedPhysicalVolume/BlockedReplicaNum
428 //
429 // 3 Cases:
430 //
431 // o Parameterised daughters
432 // =>Must be one G4PVParameterised daughter & voxels
433 // o Positioned daughters & voxels
434 // o Positioned daughters & no voxels
435
436 noResult = true; // noResult should be renamed to
437 // something like enteredLevel, as that is its meaning.
438 do
439 {
440 // Determine `type' of current mother volume
441 //
442 targetPhysical = fHistory.GetTopVolume();
443 if (!targetPhysical) { break; }
444 targetLogical = targetPhysical->GetLogicalVolume();
445 switch( CharacteriseDaughters(targetLogical) )
446 {
447 case kNormal:
448 if ( targetLogical->GetVoxelHeader() ) // use optimised navigation
449 {
450 noResult = fvoxelNav.LevelLocate(fHistory,
451 fBlockedPhysicalVolume,
452 fBlockedReplicaNo,
453 globalPoint,
454 pGlobalDirection,
455 considerDirection,
456 localPoint);
457 }
458 else // do not use optimised navigation
459 {
460 noResult = fnormalNav.LevelLocate(fHistory,
461 fBlockedPhysicalVolume,
462 fBlockedReplicaNo,
463 globalPoint,
464 pGlobalDirection,
465 considerDirection,
466 localPoint);
467 }
468 break;
469 case kReplica:
470 noResult = freplicaNav.LevelLocate(fHistory,
471 fBlockedPhysicalVolume,
472 fBlockedReplicaNo,
473 globalPoint,
474 pGlobalDirection,
475 considerDirection,
476 localPoint);
477 break;
478 case kParameterised:
479 if( GetDaughtersRegularStructureId(targetLogical) != 1 )
480 {
481 noResult = fparamNav.LevelLocate(fHistory,
482 fBlockedPhysicalVolume,
483 fBlockedReplicaNo,
484 globalPoint,
485 pGlobalDirection,
486 considerDirection,
487 localPoint);
488 }
489 else // Regular structure
490 {
491 noResult = fregularNav.LevelLocate(fHistory,
492 fBlockedPhysicalVolume,
493 fBlockedReplicaNo,
494 globalPoint,
495 pGlobalDirection,
496 considerDirection,
497 localPoint);
498 }
499 break;
500 }
501
502 // LevelLocate returns true if it finds a daughter volume
503 // in which globalPoint is inside (or on the surface).
504
505 if ( noResult )
506 {
507 noLevelsEntered++;
508
509 // Entering a daughter after ascending
510 //
511 // The blocked volume is no longer valid - it was for another level
512 //
513 fBlockedPhysicalVolume = 0;
514 fBlockedReplicaNo = -1;
515
516 // fEntering should be false -- else blockedVolume is assumed good.
517 // fEnteredDaughter is used for ExitNormal
518 //
519 fEntering = false;
520 fEnteredDaughter = true;
521
522 if( fExitedMother )
523 {
524 G4VPhysicalVolume* enteredPhysical = fHistory.GetTopVolume();
525 const G4RotationMatrix* mRot = enteredPhysical->GetRotation();
526 if( mRot )
527 {
528 // Go deeper, i.e. move 'down' in the hierarchy
529 // Apply direct rotation, not inverse
530 //
531 fGrandMotherExitNormal *= (*mRot);
532 fChangedGrandMotherRefFrame= true;
533 }
534 }
535
536#ifdef G4DEBUG_NAVIGATION
537 if( fVerbose > 2 )
538 {
539 G4VPhysicalVolume* enteredPhysical = fHistory.GetTopVolume();
540 G4cout << "*** G4Navigator::LocateGlobalPointAndSetup() ***" << G4endl;
541 G4cout << " Entering volume: " << enteredPhysical->GetName()
542 << G4endl;
543 }
544#endif
545 }
546 } while (noResult); // Loop checking, 07.10.2016, J.Apostolakis
547
548 fLastLocatedPointLocal = localPoint;
549
550#ifdef G4VERBOSE
551 if( fVerbose >= 4 )
552 {
553 G4int oldcoutPrec = G4cout.precision(8);
554 G4String curPhysVol_Name("None");
555 if (targetPhysical) { curPhysVol_Name = targetPhysical->GetName(); }
556 G4cout << " Return value = new volume = " << curPhysVol_Name << G4endl;
557 G4cout << " ----- Upon exiting:" << G4endl;
558 PrintState();
559 if( fVerbose >= 5 )
560 {
561 G4cout << "Upon exiting LocateGlobalPointAndSetup():" << G4endl;
562 G4cout << " History = " << G4endl << fHistory << G4endl << G4endl;
563 }
564 G4cout.precision(oldcoutPrec);
565 }
566#endif
567
568 fLocatedOutsideWorld= false;
569
570 return targetPhysical;
571}

573// ********************************************************************
574// LocateGlobalPointWithinVolume
575//
576// -> the state information of this Navigator and its subNavigators
577// is updated in order to start the next step at pGlobalpoint
578// -> no check is performed whether pGlobalpoint is inside the
579// original volume (this must be the case).
580//
581// Note: a direction could be added to the arguments, to aid in future
582// optional checking (via the old code below, flagged by OLD_LOCATE).
583// [ This would be done only in verbose mode ]
584// ********************************************************************
585//
586void
587G4Navigator::LocateGlobalPointWithinVolume(const G4ThreeVector& pGlobalpoint)
588{
589#ifdef G4DEBUG_NAVIGATION
590 // Check: Either step was not limited by a boundary
591 // or else the full step is no longer being taken
592 assert( !fWasLimitedByGeometry );
593#endif
594
595 fLastLocatedPointLocal = ComputeLocalPoint(pGlobalpoint);
596 fLastTriedStepComputation= false;
597 fChangedGrandMotherRefFrame= false; // Frame for Exit Normal
598
599#ifdef G4DEBUG_NAVIGATION
600 if( fVerbose > 2 )
601 {
602 G4cout << "Entering LocateGlobalWithinVolume(): History = " << G4endl;
603 G4cout << fHistory << G4endl;
604 }
605#endif
606
607 // For the case of Voxel (or Parameterised) volume the respective
608 // Navigator must be messaged to update its voxel information etc
609
610 // Update the state of the Sub Navigators
611 // - in particular any voxel information they store/cache
612 //
613 G4VPhysicalVolume* motherPhysical = fHistory.GetTopVolume();
614 G4LogicalVolume* motherLogical = motherPhysical->GetLogicalVolume();
615 G4SmartVoxelHeader* pVoxelHeader = motherLogical->GetVoxelHeader();
616
617 if ( fHistory.GetTopVolumeType()!=kReplica )
618 {
619 switch( CharacteriseDaughters(motherLogical) )
620 {
621 case kNormal:
622 if ( pVoxelHeader )
623 {
624 fvoxelNav.VoxelLocate( pVoxelHeader, fLastLocatedPointLocal );
625 }
626 break;
627 case kParameterised:
628 if( GetDaughtersRegularStructureId(motherLogical) != 1 )
629 {
630 // Resets state & returns voxel node
631 //
632 fparamNav.ParamVoxelLocate( pVoxelHeader, fLastLocatedPointLocal );
633 }
634 break;
635 case kReplica:
636 G4Exception("G4Navigator::LocateGlobalPointWithinVolume()",
637 "GeomNav0001", FatalException,
638 "Not applicable for replicated volumes.");
639 break;
640 }
641 }
642
643 // Reset the state variables
644 // - which would have been affected
645 // by the 'equivalent' call to LocateGlobalPointAndSetup
646 // - who's values have been invalidated by the 'move'.
647 //
648 fBlockedPhysicalVolume = 0;
649 fBlockedReplicaNo = -1;
650 fEntering = false;
651 fEnteredDaughter = false; // Boundary not encountered, did not enter
652 fExiting = false;
653 fExitedMother = false; // Boundary not encountered, did not exit
654}

721// ********************************************************************
722// ComputeStep
723//
724// Computes the next geometric Step: intersections with current
725// mother and `daughter' volumes.
726//
727// NOTE:
728//
729// Flags on entry:
730// --------------
731// fValidExitNormal - Normal of exited volume is valid (convex, not a
732// coincident boundary)
733// fExitNormal - Surface normal of exited volume
734// fExiting - True if have exited solid
735//
736// fBlockedPhysicalVolume - Ptr to exited volume (or 0)
737// fBlockedReplicaNo - Replication no of exited volume
738// fLastStepWasZero - True if last Step size was almost zero.
739//
740// Flags on exit:
741// -------------
742// fValidExitNormal - True if surface normal of exited volume is valid
743// fExitNormal - Surface normal of exited volume rotated to mothers
744// reference system
745// fExiting - True if exiting mother
746// fEntering - True if entering `daughter' volume (or replica)
747// fBlockedPhysicalVolume - Ptr to candidate (entered) volume
748// fBlockedReplicaNo - Replication no of candidate (entered) volume
749// fLastStepWasZero - True if this Step size was almost zero.
750// ********************************************************************
751//
752G4double G4Navigator::ComputeStep( const G4ThreeVector &pGlobalpoint,
753 const G4ThreeVector &pDirection,
754 const G4double pCurrentProposedStepLength,
755 G4double &pNewSafety)
756{
757 G4ThreeVector localDirection = ComputeLocalAxis(pDirection);
758 G4double Step = kInfinity;
759 G4VPhysicalVolume *motherPhysical = fHistory.GetTopVolume();
760 G4LogicalVolume *motherLogical = motherPhysical->GetLogicalVolume();
761
762 // All state relating to exiting normals must be reset
763 //
764 fExitNormalGlobalFrame= G4ThreeVector( 0., 0., 0.);
765 // Reset value - to erase its memory
766 fChangedGrandMotherRefFrame= false;
767 // Reset - used for local exit normal
768 fGrandMotherExitNormal= G4ThreeVector( 0., 0., 0.);
769 fCalculatedExitNormal = false;
770 // Reset for new step
771
772 static G4ThreadLocal G4int sNavCScalls=0;
773 sNavCScalls++;
774
775 fLastTriedStepComputation= true;
776
777#ifdef G4VERBOSE
778 if( fVerbose > 0 )
779 {
780 G4cout << "*** G4Navigator::ComputeStep: ***" << G4endl;
781 G4cout << " Volume = " << motherPhysical->GetName()
782 << " - Proposed step length = " << pCurrentProposedStepLength
783 << G4endl;
784#ifdef G4DEBUG_NAVIGATION

788 << " Globalpoint = " << std::setw(25) << pGlobalpoint << G4endl

795 }
796#endif

798#endif

804 //

806 G4double moveLenSq = (newLocalPoint-oldLocalPoint).mag2();
807

809 {
810#ifdef G4VERBOSE
811 ComputeStepLog(pGlobalpoint, moveLenSq);
812#endif
813 // Relocate the point within the same volume
814 //
815 LocateGlobalPointWithinVolume( pGlobalpoint );
816 fLastTriedStepComputation= true; // Ensure that this is set again !!
817 }
818 }
819 if ( fHistory.GetTopVolumeType()!=kReplica )
820 {
821 switch( CharacteriseDaughters(motherLogical) )
822 {
823 case kNormal:
824 if ( motherLogical->GetVoxelHeader() )
825 {
826 Step = fvoxelNav.ComputeStep(fLastLocatedPointLocal,
827 localDirection,
828 pCurrentProposedStepLength,
829 pNewSafety,
830 fHistory,
831 fValidExitNormal,
832 fExitNormal,
833 fExiting,
834 fEntering,
835 &fBlockedPhysicalVolume,
836 fBlockedReplicaNo);
837
838 }
839 else
840 {

842 {
843 Step = fnormalNav.ComputeStep(fLastLocatedPointLocal,
844 localDirection,
845 pCurrentProposedStepLength,
846 pNewSafety,
847 fHistory,
848 fValidExitNormal,
849 fExitNormal,
850 fExiting,
851 fEntering,
852 &fBlockedPhysicalVolume,
853 fBlockedReplicaNo);
854 }
855 else // Regular (non-voxelised) structure
856 {
857 LocateGlobalPointAndSetup( pGlobalpoint, &pDirection, true, true );
858 fLastTriedStepComputation= true; // Ensure that this is set again !!
859 //
860 // if physical process limits the step, the voxel will not be the
861 // one given by ComputeStepSkippingEqualMaterials() and the local
862 // point will be wrongly calculated.
863
864 // There is a problem: when msc limits the step and the point is
865 // assigned wrongly to phantom in previous step (while it is out
866 // of the container volume). Then LocateGlobalPointAndSetup() has
867 // reset the history topvolume to world.
868 //
869 if(fHistory.GetTopVolume()->GetRegularStructureId() == 0 )
870 {
871 G4Exception("G4Navigator::ComputeStep()",
872 "GeomNav1001", JustWarning,
873 "Point is relocated in voxels, while it should be outside!");

875 localDirection,
876 pCurrentProposedStepLength,
877 pNewSafety,
878 fHistory,
879 fValidExitNormal,
880 fExitNormal,
881 fExiting,
882 fEntering,
883 &fBlockedPhysicalVolume,
884 fBlockedReplicaNo);
885 }
886 else
887 {
888 Step = fregularNav.
889 ComputeStepSkippingEqualMaterials(fLastLocatedPointLocal,
890 localDirection,
891 pCurrentProposedStepLength,
892 pNewSafety,
893 fHistory,
894 fValidExitNormal,
895 fExitNormal,
896 fExiting,
897 fEntering,
898 &fBlockedPhysicalVolume,
899 fBlockedReplicaNo,
900 motherPhysical);
901 }
902 }
903 }
904 break;
905 case kParameterised:
906 if( GetDaughtersRegularStructureId(motherLogical) != 1 )
907 {
908 Step = fparamNav.ComputeStep(fLastLocatedPointLocal,
909 localDirection,
910 pCurrentProposedStepLength,
911 pNewSafety,
912 fHistory,
913 fValidExitNormal,
914 fExitNormal,
915 fExiting,
916 fEntering,
917 &fBlockedPhysicalVolume,
918 fBlockedReplicaNo);
919 }

921 {
922 Step = fregularNav.ComputeStep(fLastLocatedPointLocal,
923 localDirection,
924 pCurrentProposedStepLength,
925 pNewSafety,
926 fHistory,
927 fValidExitNormal,
928 fExitNormal,
929 fExiting,
930 fEntering,
931 &fBlockedPhysicalVolume,
932 fBlockedReplicaNo);

937 FatalException, "Not applicable for replicated volumes.");
938 break;
939 }
940 }
941 else
942 {
943 // In the case of a replica, it must handle the exiting
944 // edge/corner problem by itself
945 //
946 G4bool exitingReplica = fExitedMother;
947 G4bool calculatedExitNormal;
948 Step = freplicaNav.ComputeStep(pGlobalpoint,
949 pDirection,
950 fLastLocatedPointLocal,
951 localDirection,
952 pCurrentProposedStepLength,
953 pNewSafety,
954 fHistory,
955 fValidExitNormal,
956 calculatedExitNormal,
957 fExitNormal,
958 exitingReplica,
959 fEntering,
960 &fBlockedPhysicalVolume,
961 fBlockedReplicaNo);
962 fExiting= exitingReplica;
963 fCalculatedExitNormal= calculatedExitNormal;
964 }

966 // Remember last safety origin & value.
967 //
968 fPreviousSftOrigin = pGlobalpoint;
969 fPreviousSafety = pNewSafety;
970
971 // Count zero steps - one can occur due to changing momentum at a boundary
972 // - one, two (or a few) can occur at common edges between
973 // volumes
974 // - more than two is likely a problem in the geometry
975 // description or the Navigation
976
977 // Rule of thumb: likely at an Edge if two consecutive steps are zero,
978 // because at least two candidate volumes must have been
979 // checked
980 //
981 fLocatedOnEdge = fLastStepWasZero && (Step==0.0);
982 fLastStepWasZero = (Step<fMinStep);
983 if (fPushed) { fPushed = fLastStepWasZero; }
984
985 // Handle large number of consecutive zero steps
986 //
987 if ( fLastStepWasZero )
988 {
989 fNumberZeroSteps++;
990#ifdef G4DEBUG_NAVIGATION
991 if( fNumberZeroSteps > 1 )
992 {
993 G4cout << "G4Navigator::ComputeStep(): another 'zero' step, # "
994 << fNumberZeroSteps

996 << ", in volume " << motherPhysical->GetName()
997 << ", nav-comp-step calls # " << sNavCScalls
998 << ", Step= " << Step
999 << G4endl;
1000 }
1001#endif
1002 if( fNumberZeroSteps > fActionThreshold_NoZeroSteps-1 )
1003 {
1004 // Act to recover this stuck track. Pushing it along direction
1005 //
1006 Step += 100*kCarTolerance;
1007#ifdef G4VERBOSE
1008 if ((!fPushed) && (fWarnPush))
1009 {
1010 std::ostringstream message;
1011 message.precision(16);
1012 message << "Track stuck or not moving." << G4endl
1013 << " Track stuck, not moving for "
1014 << fNumberZeroSteps << " steps" << G4endl
1015 << " in volume -" << motherPhysical->GetName()
1016 << "- at point " << pGlobalpoint
1017 << " (local point " << newLocalPoint << ")" << G4endl
1018 << " direction: " << pDirection
1019 << " (local direction: " << localDirection << ")." << G4endl
1020 << " Potential geometry or navigation problem !"
1021 << G4endl
1022 << " Trying pushing it of " << Step << " mm ...";
1023 G4Exception("G4Navigator::ComputeStep()", "GeomNav1002",
1024 JustWarning, message, "Potential overlap in geometry!");
1025 }
1026#endif
1027 fPushed = true;
1028 }
1029 if( fNumberZeroSteps > fAbandonThreshold_NoZeroSteps-1 )
1030 {
1031 // Must kill this stuck track
1032 //
1033 std::ostringstream message;
1034 message << "Stuck Track: potential geometry or navigation problem."
1035 << G4endl
1036 << " Track stuck, not moving for "
1037 << fNumberZeroSteps << " steps" << G4endl
1038 << " in volume -" << motherPhysical->GetName()
1039 << "- at point " << pGlobalpoint << G4endl
1040 << " direction: " << pDirection << ".";
1041#ifdef G4VERBOSE
1042 if ( fWarnPush )
1043 {
1044 motherPhysical->CheckOverlaps(5000, false);
1045 }
1046#endif
1047 G4Exception("G4Navigator::ComputeStep()", "GeomNav0003",
1048 EventMustBeAborted, message);
1049 }
1050 }
1051 else
1052 {
1053 if (!fPushed) fNumberZeroSteps = 0;
1054 }
1055
1056 fEnteredDaughter = fEntering; // I expect to enter a volume in this Step
1057 fExitedMother = fExiting;
1058
1059 fStepEndPoint = pGlobalpoint
1060 + std::min(Step,pCurrentProposedStepLength) * pDirection;
1061 fLastStepEndPointLocal = fLastLocatedPointLocal + Step * localDirection;
1062
1063 if( fExiting )
1064 {
1065#ifdef G4DEBUG_NAVIGATION
1066 if( fVerbose > 2 )
1067 {
1068 G4cout << " At G4Nav CompStep End - if(exiting) - fExiting= " << fExiting
1069 << " fValidExitNormal = " << fValidExitNormal << G4endl;
1070 G4cout << " fExitNormal= " << fExitNormal << G4endl;
1071 }
1072#endif
1073
1074 if(fValidExitNormal || fCalculatedExitNormal)
1075 {
1076 if ( fHistory.GetTopVolumeType()!=kReplica )
1077 {
1078 // Convention: fExitNormal is in the 'grand-mother' coordinate system
1079 //
1080 fGrandMotherExitNormal= fExitNormal;
1081 fCalculatedExitNormal= true;
1082 }
1083 else
1084 {
1085 fGrandMotherExitNormal = fExitNormal;
1086 }
1087 }
1088 else
1089 {
1090 // We must calculate the normal anyway (in order to have it if requested)
1091 //
1092 G4ThreeVector finalLocalPoint =
1093 fLastLocatedPointLocal + localDirection*Step;
1094
1095 if ( fHistory.GetTopVolumeType()!=kReplica )
1096 {
1097 // Find normal in the 'mother' coordinate system
1098 //
1099 G4ThreeVector exitNormalMotherFrame=
1100 motherLogical->GetSolid()->SurfaceNormal(finalLocalPoint);
1101
1102 // Transform it to the 'grand-mother' coordinate system
1103 //
1104 const G4RotationMatrix* mRot = motherPhysical->GetRotation();
1105 if( mRot )
1106 {
1107 fChangedGrandMotherRefFrame= true;
1108 fGrandMotherExitNormal = (*mRot).inverse() * exitNormalMotherFrame;
1109 }
1110 else
1111 {
1112 fGrandMotherExitNormal = exitNormalMotherFrame;
1113 }
1114
1115 // Do not set fValidExitNormal -- this signifies
1116 // that the solid is convex!
1117 //
1118 fCalculatedExitNormal= true;
1119 }
1120 else
1121 {
1122 fCalculatedExitNormal = false;
1123 //
1124 // Nothing can be done at this stage currently - to solve this
1125 // Replica Navigation must have calculated the normal for this case
1126 // already.
1127 // Cases: mother is not convex, and exit is at previous replica level
1128
1129#ifdef G4DEBUG_NAVIGATION
1130 G4ExceptionDescription desc;
1131
1132 desc << "Problem in ComputeStep: Replica Navigation did not provide"
1133 << " valid exit Normal. " << G4endl;
1134 desc << " Do not know how calculate it in this case." << G4endl;
1135 desc << " Location = " << finalLocalPoint << G4endl;
1136 desc << " Volume name = " << motherPhysical->GetName()
1137 << " copy/replica No = " << motherPhysical->GetCopyNo() << G4endl;
1138 G4Exception("G4Navigator::ComputeStep()", "GeomNav0003",
1139 JustWarning, desc, "Normal not available for exiting.");
1140#endif
1141 }
1142 }
1143
1144 // Now transform it to the global reference frame !!
1145 //
1146 if( fValidExitNormal || fCalculatedExitNormal )
1147 {
1148 G4int depth= fHistory.GetDepth();
1149 if( depth > 0 )
1150 {
1151 fExitNormalGlobalFrame =
1152 fHistory.GetTransform(depth-1).InverseTransformAxis( fGrandMotherExitNormal );
1153 }
1154 else
1155 {
1156 fExitNormalGlobalFrame= fGrandMotherExitNormal;
1157 }
1158 }
1159 else
1160 {
1161 fExitNormalGlobalFrame= G4ThreeVector( 0., 0., 0.);
1162 }
1163 }
1164
1165 if( (Step == pCurrentProposedStepLength) && (!fExiting) && (!fEntering) )
1166 {
1167 // This if Step is not really limited by the geometry.
1168 // The Navigator is obliged to return "infinity"
1169 //
1170 Step = kInfinity;
1171 }
1172
1173#ifdef G4VERBOSE
1174 if( fVerbose > 1 )
1175 {
1176 if( fVerbose >= 4 )
1177 {
1178 G4cout << " ----- Upon exiting :" << G4endl;
1179 PrintState();
1180 }
1181 G4cout << " Returned step= " << Step;
1182 if( fVerbose > 5 ) G4cout << G4endl;
1183 if( Step == kInfinity )
1184 {
1185 G4cout << " Requested step= " << pCurrentProposedStepLength ;
1186 if( fVerbose > 5) G4cout << G4endl;
1187 }
1188 G4cout << " Safety = " << pNewSafety << G4endl;
1189 }
1190#endif
1191
1192 return Step;
1193}

1425 }
1426 G4Exception("G4Navigator::GetLocalExitNormal()", "GeomNav1001",
1427 JustWarning, message);
1428 }
1429#endif
1430 }
1431 *valid = onSurface; // was =true;
1432 }
1433 }
1434 }
1435 else if ( fExiting )
1436 {
1437 ExitNormal = fGrandMotherExitNormal;
1438 *valid = true;
1439 fCalculatedExitNormal= true; // Should be true already
1440 }
1441 else // i.e. ( fBlockedPhysicalVolume == 0 )

1443 *valid = false;
1444 G4Exception("G4Navigator::GetLocalExitNormal()",
1445 "GeomNav0003", JustWarning,
1446 "Incorrect call to GetLocalSurfaceNormal." );
1447 }
1448 }
1449 else // ( ! fLastTriedStepComputation ) ie. last call was to Locate

1455 ExitNormal= -(daughterSolid->SurfaceNormal(fLastLocatedPointLocal));

1483 G4Exception("G4Navigator::GetLocalExitNormal()",
1484 "GeomNav0003", JustWarning, message);
1485 }
1486 }
1487 }
1488 return ExitNormal;
1489}

1895// ********************************************************************
1896// RecheckDistanceToCurrentBoundary
1897//
1898// Trial method for checking potential displacement for MS
1899// Check position aDisplacedGlobalpoint, to see whether it is in the
1900// current volume (mother).
1901// If in mother, check distance to boundary along aNewDirection.
1902// If in entering daughter, check distance back to boundary.
1903// NOTE:
1904// Can be called only after ComputeStep() is called - before ReLocation
1905// Deals only with current volume (and potentially entered)
1906// Caveats
1907// First VERSION: Does not consider daughter volumes if it remained in mother
1908// neither for checking whether it has exited current (mother) volume,
1909// nor for determining distance to exit this (mother) volume.
1910// ********************************************************************
1911//
1912G4bool G4Navigator::RecheckDistanceToCurrentBoundary(
1913 const G4ThreeVector &aDisplacedGlobalPoint,
1914 const G4ThreeVector &aNewDirection,
1915 const G4double ProposedMove,
1916 G4double *prDistance,
1917 G4double *prNewSafety) const
1918{
1919 G4ThreeVector localPosition = ComputeLocalPoint(aDisplacedGlobalPoint);
1920 G4ThreeVector localDirection = ComputeLocalAxis(aNewDirection);
1921 // G4double Step = kInfinity;
1922
1923 G4bool validExitNormal;
1924 G4ThreeVector exitNormal;
1925 // Check against mother solid
1926 G4VPhysicalVolume *motherPhysical = fHistory.GetTopVolume();
1927 G4LogicalVolume *motherLogical = motherPhysical->GetLogicalVolume();
1928
1929#ifdef CHECK_ORDER_OF_METHODS
1930 if( ! fLastTriedStepComputation )
1931 {
1932 G4Exception("G4Navigator::RecheckDistanceToCurrentBoundary()",
1933 "GeomNav0001", FatalException,
1934 "Method must be called after ComputeStep(), before call to LocateMethod.");
1935 }
1936#endif
1937
1938 EInside locatedDaug; // = kUndefined;
1939 G4double daughterStep= DBL_MAX;
1940 G4double daughterSafety= DBL_MAX;
1941
1942 if( fEnteredDaughter )
1943 {
1944 if( motherLogical->CharacteriseDaughters() == kReplica ) { return false; }
1945
1946 // Track arrived at boundary of a daughter volume at
1947 // the last call of ComputeStep().
1948 // In case the proposed displaced point is inside this daughter,
1949 // it must backtrack at least to the entry point.
1950 // NOTE: No check is made against other daughter volumes. It is
1951 // assumed that the proposed displacement is small enough that
1952 // this is not needed.
1953
1954 // Must check boundary of current daughter
1955 //
1956 G4VPhysicalVolume *candPhysical= fBlockedPhysicalVolume;
1957 G4LogicalVolume *candLogical= candPhysical->GetLogicalVolume();
1958 G4VSolid *candSolid= candLogical->GetSolid();
1959
1960 G4AffineTransform nextLevelTrf(candPhysical->GetRotation(),
1961 candPhysical->GetTranslation());
1962
1963 G4ThreeVector dgPosition= nextLevelTrf.TransformPoint(localPosition);
1964 G4ThreeVector dgDirection= nextLevelTrf.TransformAxis(localDirection);
1965 locatedDaug = candSolid->Inside(dgPosition);
1966
1967 if( locatedDaug == kInside )
1968 {
1969 // Reverse direction - and find first exit. ( Is it valid?)
1970 // Must backtrack
1971 G4double distanceBackOut =
1972 candSolid->DistanceToOut(dgPosition,
1973 - dgDirection, // Reverse direction
1974 true, &validExitNormal, &exitNormal);
1975 daughterStep= - distanceBackOut;
1976 // No check is made whether the particle could have arrived at
1977 // at this location without encountering another volume or
1978 // a different psurface of the current volume
1979 if( prNewSafety )
1980 {
1981 daughterSafety= candSolid->DistanceToOut(dgPosition);
1982 }
1983 }
1984 else
1985 {
1986 if( locatedDaug == kOutside )
1987 {
1988 // See whether it still intersects it
1989 //
1990 daughterStep= candSolid->DistanceToIn(dgPosition,
1991 dgDirection);
1992 if( prNewSafety )
1993 {
1994 daughterSafety= candSolid->DistanceToIn(dgPosition);
1995 }
1996 }
1997 else
1998 {
1999 // The point remains on the surface of candidate solid
2000 //
2001 daughterStep= 0.0;
2002 daughterSafety= 0.0;
2003 }
2004 }
2005
2006 // If trial point is in daughter (or on its surface) we have the
2007 // answer, the rest is not relevant
2008 //
2009 if( locatedDaug != kOutside )
2010 {
2011 *prDistance= daughterStep;
2012 if( prNewSafety ) { *prNewSafety= daughterSafety; }
2013 return true;
2014 }
2015 // If ever extended, so that some type of mother cut daughter,
2016 // this would change
2017 }
2018
2019 G4VSolid *motherSolid= motherLogical->GetSolid();
2020
2021 G4double motherStep= DBL_MAX, motherSafety= DBL_MAX;
2022
2023 // Check distance to boundary of mother
2024 //
2025 if ( fHistory.GetTopVolumeType()!=kReplica )
2026 {
2027 EInside locatedMoth = motherSolid->Inside(localPosition);
2028
2029 if( locatedMoth == kInside )
2030 {
2031 motherSafety= motherSolid->DistanceToOut(localPosition);
2032 if( ProposedMove >= motherSafety )
2033 {
2034 motherStep= motherSolid->DistanceToOut(localPosition,
2035 localDirection,
2036 true, &validExitNormal, &exitNormal);
2037 }
2038 else
2039 {
2040 motherStep= ProposedMove;
2041 }
2042 }
2043 else if( locatedMoth == kOutside)
2044 {
2045 motherSafety= motherSolid->DistanceToIn(localPosition);
2046 if( ProposedMove >= motherSafety )
2047 {
2048 motherStep= - motherSolid->DistanceToIn(localPosition,
2049 -localDirection);
2050 }
2051 }
2052 else
2053 {
2054 motherSafety= 0.0;
2055 *prDistance= 0.0; // On surface - no move // motherStep;
2056 if( prNewSafety ) { *prNewSafety= motherSafety; }
2057 return false;
2058 }
2059 }
2060 else
2061 {
2062 return false;
2063 }
2064
2065 *prDistance= std::min( motherStep, daughterStep );
2066 if( prNewSafety )
2067 {
2068 *prNewSafety= std::min( motherSafety, daughterSafety );
2069 }
2070 return true;
2071}
