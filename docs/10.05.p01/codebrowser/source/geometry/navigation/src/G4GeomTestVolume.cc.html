125//
126// TestRecursiveOverlap
127//
128void G4GeomTestVolume::TestRecursiveOverlap( G4int slevel, G4int depth )
129{
130 // If reached requested level of depth (i.e. set to 0), exit.
131 // If not depth specified (i.e. set to -1), visit the whole tree.
132 // If requested initial level of depth is not zero, visit from beginning
133 //
134 if (depth == 0) return;
135 if (depth != -1) depth--;
136 if (slevel != 0) slevel--;
137
138 //
139 // As long as we reached the requested
140 // initial level of depth, test ourselves
141 //
142 if ( slevel==0 )
143 {
144 target->CheckOverlaps(resolution, tolerance, verbosity, maxErr);
145 }
146
147 //
148 // Loop over unique daughters
149 //
150 std::set<const G4LogicalVolume *> tested;
151
152 const G4LogicalVolume *logical = target->GetLogicalVolume();
153 G4int nDaughter = logical->GetNoDaughters();
154 G4int iDaughter;
155 for( iDaughter=0; iDaughter<nDaughter; ++iDaughter )
156 {
157 G4VPhysicalVolume *daughter = logical->GetDaughter(iDaughter);
158
159 // Tested already?
160 //
161 // const G4LogicalVolume *daughterLogical =
162 // daughter->GetLogicalVolume();
163 // std::pair<std::set<const G4LogicalVolume *>::iterator, G4bool>
164 // there = tested.insert(daughterLogical);
165 // if (!there.second) continue;
166
167 //
168 // Recurse
169 //
170 G4GeomTestVolume vTest( daughter, tolerance, resolution, verbosity );
171 vTest.SetErrorsThreshold(maxErr);
172 vTest.TestRecursiveOverlap( slevel,depth );
173 }
174}
