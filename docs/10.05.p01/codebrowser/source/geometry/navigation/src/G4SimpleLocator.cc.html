47// --------------------------------------------------------------------------
48// G4bool G4PropagatorInField::LocateIntersectionPoint(
49// const G4FieldTrack& CurveStartPointVelocity, // A
50// const G4FieldTrack& CurveEndPointVelocity, // B
51// const G4ThreeVector& TrialPoint, // E
52// G4FieldTrack& IntersectedOrRecalculated // Output
53// G4bool& recalculated ) // Out
54// --------------------------------------------------------------------------
55//
56// Function that returns the intersection of the true path with the surface
57// of the current volume (either the external one or the inner one with one
58// of the daughters:
59//
60// A = Initial point
61// B = another point
62//
63// Both A and B are assumed to be on the true path:
64//
65// E is the first point of intersection of the chord AB with
66// a volume other than A (on the surface of A or of a daughter)
67//
68// Convention of Use :
69// i) If it returns "true", then IntersectionPointVelocity is set
70// to the approximate intersection point.
71// ii) If it returns "false", no intersection was found.
72// The validity of IntersectedOrRecalculated depends on 'recalculated'
73// a) if latter is false, then IntersectedOrRecalculated is invalid.
74// b) if latter is true, then IntersectedOrRecalculated is
75// the new endpoint, due to a re-integration.
76// --------------------------------------------------------------------------
77// NOTE: implementation taken from G4PropagatorInField
78//
79G4bool G4SimpleLocator::EstimateIntersectionPoint(
80 const G4FieldTrack& CurveStartPointVelocity, // A
81 const G4FieldTrack& CurveEndPointVelocity, // B
82 const G4ThreeVector& TrialPoint, // E
83 G4FieldTrack& IntersectedOrRecalculatedFT, // Output
84 G4bool& recalculatedEndPoint, // Out
85 G4double &fPreviousSafety, //In/Out
86 G4ThreeVector &fPreviousSftOrigin) //In/Out
87{
88 // Find Intersection Point ( A, B, E ) of true path AB - start at E.
89
90 G4bool found_approximate_intersection = false;
91 G4bool there_is_no_intersection = false;
92
93 G4FieldTrack CurrentA_PointVelocity = CurveStartPointVelocity;
94 G4FieldTrack CurrentB_PointVelocity = CurveEndPointVelocity;
95 G4ThreeVector CurrentE_Point = TrialPoint;
96 G4bool validNormalAtE = false;
97 G4ThreeVector NormalAtEntry;
98
99 G4FieldTrack ApproxIntersecPointV(CurveEndPointVelocity); // FT-Def-Construct
100 G4double NewSafety = 0.0;
101 G4bool last_AF_intersection = false;
102 G4bool final_section = true; // Shows whether current section is last
103 // (i.e. B=full end)
104 recalculatedEndPoint = false;
105
106 G4bool restoredFullEndpoint = false;
107
108 G4int substep_no = 0;
109
110 // Limits for substep number
111 //
112 const G4int max_substeps = 100000000; // Test 120 (old value 100 )
113 const G4int warn_substeps = 1000; // 100
114
115 // Statistics for substeps
116 //
117 static G4ThreadLocal G4int max_no_seen= -1;
118
119 NormalAtEntry = GetSurfaceNormal( CurrentE_Point, validNormalAtE);
120
121#ifdef G4DEBUG_FIELD
122 const G4double tolerance = 1.0e-8;
123 G4ThreeVector StartPosition= CurveStartPointVelocity.GetPosition();
124 if( (TrialPoint - StartPosition).mag() < tolerance * CLHEP::mm )
125 {
126 G4Exception("G4SimpleLocator::EstimateIntersectionPoint()",
127 "GeomNav1002", JustWarning,
128 "Intersection point F is exactly at start point A." );
129 }
130#endif
131
132 do
133 {
134 G4ThreeVector Point_A = CurrentA_PointVelocity.GetPosition();
135 G4ThreeVector Point_B = CurrentB_PointVelocity.GetPosition();
136
137 // F = a point on true AB path close to point E
138 // (the closest if possible)
139 //
140 ApproxIntersecPointV = GetChordFinderFor()
141 ->ApproxCurvePointV( CurrentA_PointVelocity,
142 CurrentB_PointVelocity,
143 CurrentE_Point,
144 GetEpsilonStepFor());
145 // The above method is the key & most intuitive part ...
146
147#ifdef G4DEBUG_FIELD
148 if( ApproxIntersecPointV.GetCurveLength() >
149 CurrentB_PointVelocity.GetCurveLength() * (1.0 + tolerance) )
150 {
151 G4Exception("G4SimpleLocator::EstimateIntersectionPoint()",
152 "GeomNav0003", FatalException,
153 "Intermediate F point is past end B point!" );
154 }
155#endif
156
157 G4ThreeVector CurrentF_Point= ApproxIntersecPointV.GetPosition();
158
159 // First check whether EF is small - then F is a good approx. point
160 // Calculate the length and direction of the chord AF
161 //
162 G4ThreeVector ChordEF_Vector = CurrentF_Point - CurrentE_Point;
163
164 G4ThreeVector NewMomentumDir= ApproxIntersecPointV.GetMomentumDir();
165 G4double MomDir_dot_Norm= NewMomentumDir.dot( NormalAtEntry ) ;
166
167 G4ThreeVector ChordAB = Point_B - Point_A;
168
169#ifdef G4DEBUG_FIELD
170 G4VIntersectionLocator::
171 ReportTrialStep( substep_no, ChordAB, ChordEF_Vector,
172 NewMomentumDir, NormalAtEntry, validNormalAtE );
173#endif
174 // Check Sign is always exiting !! TODO
175 // Could ( > -epsilon) be used instead?
176 //
177 G4bool adequate_angle = ( MomDir_dot_Norm >= 0.0 )
178 || (! validNormalAtE ); // Invalid
179 G4double EF_dist2= ChordEF_Vector.mag2();
180 if ( ( EF_dist2 <= sqr(fiDeltaIntersection) && ( adequate_angle ) )
181 || ( EF_dist2 <= kCarTolerance*kCarTolerance ) )
182 {
183 found_approximate_intersection = true;
184
185 // Create the "point" return value
186 //
187 IntersectedOrRecalculatedFT = ApproxIntersecPointV;
188 IntersectedOrRecalculatedFT.SetPosition( CurrentE_Point );
189
190 if ( GetAdjustementOfFoundIntersection() )
191 {
192 // Try to Get Correction of IntersectionPoint using SurfaceNormal()
193 //
194 G4ThreeVector IP;
195 G4ThreeVector MomentumDir= ApproxIntersecPointV.GetMomentumDirection();
196 G4bool goodCorrection = AdjustmentOfFoundIntersection( Point_A,
197 CurrentE_Point, CurrentF_Point, MomentumDir,
198 last_AF_intersection, IP, NewSafety,
199 fPreviousSafety, fPreviousSftOrigin );
200
201 if(goodCorrection)
202 {
203 IntersectedOrRecalculatedFT = ApproxIntersecPointV;
204 IntersectedOrRecalculatedFT.SetPosition(IP);
205 }
206 }
207
208 // Note: in order to return a point on the boundary,
209 // we must return E. But it is F on the curve.
210 // So we must "cheat": we are using the position at point E
211 // and the velocity at point F !!!
212 //
213 // This must limit the length we can allow for displacement!
214 }
215 else // E is NOT close enough to the curve (ie point F)
216 {
217 // Check whether any volumes are encountered by the chord AF
218 // ---------------------------------------------------------
219 // First relocate to restore any Voxel etc information
220 // in the Navigator before calling ComputeStep()
221 //
222 GetNavigatorFor()->LocateGlobalPointWithinVolume( Point_A );
223
224 G4ThreeVector PointG; // Candidate intersection point
225 G4double stepLengthAF;
226 G4bool usedNavigatorAF = false;
227 G4bool Intersects_AF = IntersectChord( Point_A,
228 CurrentF_Point,
229 NewSafety,
230 fPreviousSafety,
231 fPreviousSftOrigin,
232 stepLengthAF,
233 PointG,
234 &usedNavigatorAF );
235 last_AF_intersection = Intersects_AF;
236 if( Intersects_AF )
237 {
238 // G is our new Candidate for the intersection point.
239 // It replaces "E" and we will repeat the test to see if
240 // it is a good enough approximate point for us.
241 // B <- F
242 // E <- G
243
244 CurrentB_PointVelocity = ApproxIntersecPointV;
245 CurrentE_Point = PointG;
246
247 // Need to recalculate the Exit Normal at the new PointG
248 // Relies on a call to Navigator::ComputeStep in IntersectChord above
249 // If the safety was adequate (for the step) this would NOT be called!
250 // But this must not occur, no intersection can be found in that case,
251 // so this branch, ie if( Intersects_AF ) would not be reached!
252 //
253 G4bool validNormalLast;
254 NormalAtEntry = GetSurfaceNormal( PointG, validNormalLast );
255 validNormalAtE = validNormalLast;
256
257 // By moving point B, must take care if current
258 // AF has no intersection to try current FB!!
259 //
260 final_section= false;
261
262#ifdef G4VERBOSE
263 if( fVerboseLevel > 3 )
264 {
265 G4cout << "G4PiF::LI> Investigating intermediate point"
266 << " at s=" << ApproxIntersecPointV.GetCurveLength()
267 << " on way to full s="
268 << CurveEndPointVelocity.GetCurveLength() << G4endl;
269 }
270#endif
271 }
272 else // not Intersects_AF
273 {
274 // In this case:
275 // There is NO intersection of AF with a volume boundary.
276 // We must continue the search in the segment FB!
277 //
278 GetNavigatorFor()->LocateGlobalPointWithinVolume( CurrentF_Point );
279
280 G4double stepLengthFB;
281 G4ThreeVector PointH;
282 G4bool usedNavigatorFB=false;
283
284 // Check whether any volumes are encountered by the chord FB
285 // ---------------------------------------------------------
286
287 G4bool Intersects_FB = IntersectChord( CurrentF_Point, Point_B,
288 NewSafety,fPreviousSafety,
289 fPreviousSftOrigin,
290 stepLengthFB,
291 PointH, &usedNavigatorFB );
292 if( Intersects_FB )
293 {
294 // There is an intersection of FB with a volume boundary
295 // H <- First Intersection of Chord FB
296
297 // H is our new Candidate for the intersection point.
298 // It replaces "E" and we will repeat the test to see if
299 // it is a good enough approximate point for us.
300
301 // Note that F must be in volume volA (the same as A)
302 // (otherwise AF would meet a volume boundary!)
303 // A <- F
304 // E <- H
305 //
306 CurrentA_PointVelocity = ApproxIntersecPointV;
307 CurrentE_Point = PointH;
308
309 // Need to recalculate the Exit Normal at the new PointG
310 // Relies on call to Navigator::ComputeStep in IntersectChord above
311 // If safety was adequate (for the step) this would NOT be called!
312 // But this must not occur, no intersection found in that case,
313 // so this branch, i.e. if( Intersects_AF ) would not be reached!
314 //
315 G4bool validNormalLast;
316 NormalAtEntry = GetSurfaceNormal( PointH, validNormalLast );
317 validNormalAtE = validNormalLast;
318 }
319 else // not Intersects_FB
320 {
321 // There is NO intersection of FB with a volume boundary
322
323 if( final_section )
324 {
325 // If B is the original endpoint, this means that whatever
326 // volume(s) intersected the original chord, none touch the
327 // smaller chords we have used.
328 // The value of 'IntersectedOrRecalculatedFT' returned is
329 // likely not valid
330
331 there_is_no_intersection = true; // real final_section
332 }
333 else
334 {
335 // We must restore the original endpoint
336
337 CurrentA_PointVelocity = CurrentB_PointVelocity; // Got to B
338 CurrentB_PointVelocity = CurveEndPointVelocity;
339 restoredFullEndpoint = true;
340 }
341 } // Endif (Intersects_FB)
342 } // Endif (Intersects_AF)
343
344 // Ensure that the new endpoints are not further apart in space
345 // than on the curve due to different errors in the integration
346 //
347 G4double linDistSq, curveDist;
348 linDistSq = ( CurrentB_PointVelocity.GetPosition()
349 - CurrentA_PointVelocity.GetPosition() ).mag2();
350 curveDist = CurrentB_PointVelocity.GetCurveLength()
351 - CurrentA_PointVelocity.GetCurveLength();
352
353 // Change this condition for very strict parameters of propagation
354 //
355 if( curveDist*curveDist*(1+2* GetEpsilonStepFor()) < linDistSq )
356 {
357 // Re-integrate to obtain a new B
358 //
359 G4FieldTrack newEndPointFT =
360 ReEstimateEndpoint( CurrentA_PointVelocity,
361 CurrentB_PointVelocity,
362 linDistSq, // to avoid recalculation
363 curveDist );
364 G4FieldTrack oldPointVelB = CurrentB_PointVelocity;
365 CurrentB_PointVelocity = newEndPointFT;
366
367 if( (final_section)) // real final section
368 {
369 recalculatedEndPoint = true;
370 IntersectedOrRecalculatedFT = newEndPointFT;
371 // So that we can return it, if it is the endpoint!
372 }
373 }
374 if( curveDist < 0.0 )
375 {
376 fVerboseLevel = 5; // Print out a maximum of information
377 printStatus( CurrentA_PointVelocity, CurrentB_PointVelocity,
378 -1.0, NewSafety, substep_no );
379 std::ostringstream message;
380 message << "Error in advancing propagation." << G4endl
381 << " Point A (start) is " << CurrentA_PointVelocity
382 << G4endl
383 << " Point B (end) is " << CurrentB_PointVelocity
384 << G4endl
385 << " Curve distance is " << curveDist << G4endl
386 << G4endl
387 << "The final curve point is not further along"
388 << " than the original!" << G4endl;
389
390 if( recalculatedEndPoint )
391 {
392 message << "Recalculation of EndPoint was called with fEpsStep= "
393 << GetEpsilonStepFor() << G4endl;
394 }
395 message.precision(20);
396 message << " Point A (Curve start) is " << CurveStartPointVelocity
397 << G4endl
398 << " Point B (Curve end) is " << CurveEndPointVelocity
399 << G4endl
400 << " Point A (Current start) is " << CurrentA_PointVelocity
401 << G4endl
402 << " Point B (Current end) is " << CurrentB_PointVelocity
403 << G4endl
404 << " Point E (Trial Point) is " << CurrentE_Point
405 << G4endl
406 << " Point F (Intersection) is " << ApproxIntersecPointV
407 << G4endl
408 << " LocateIntersection parameters are : Substep no= "
409 << substep_no;
410
411 G4Exception("G4SimpleLocator::EstimateIntersectionPoint()",
412 "GeomNav0003", FatalException, message);
413 }
414
415 if(restoredFullEndpoint)
416 {
417 final_section = restoredFullEndpoint;
418 restoredFullEndpoint = false;
419 }
420 } // EndIf ( E is close enough to the curve, ie point F. )
421 // tests ChordAF_Vector.mag() <= maximum_lateral_displacement
422
423#ifdef G4DEBUG_LOCATE_INTERSECTION
424 G4int trigger_substepno_print= warn_substeps - 20;
425
426 if( substep_no >= trigger_substepno_print )
427 {
428 G4cout << "Difficulty in converging in "
429 << "G4SimpleLocator::EstimateIntersectionPoint():"
430 << G4endl
431 << " Substep no = " << substep_no << G4endl;
432 if( substep_no == trigger_substepno_print )
433 {
434 printStatus( CurveStartPointVelocity, CurveEndPointVelocity,
435 -1.0, NewSafety, 0);
436 }
437 G4cout << " State of point A: ";
438 printStatus( CurrentA_PointVelocity, CurrentA_PointVelocity,
439 -1.0, NewSafety, substep_no-1, 0);
440 G4cout << " State of point B: ";
441 printStatus( CurrentA_PointVelocity, CurrentB_PointVelocity,
442 -1.0, NewSafety, substep_no);
443 }
444#endif
445 substep_no++;
446
447 } while ( ( ! found_approximate_intersection )
448 && ( ! there_is_no_intersection )
449 && ( substep_no <= max_substeps) ); // UNTIL found or failed
450
451 if( substep_no > max_no_seen )
452 {
453 max_no_seen = substep_no;
454#ifdef G4DEBUG_LOCATE_INTERSECTION
455 if( max_no_seen > warn_substeps )
456 {
457 trigger_substepno_print = max_no_seen-20; // Want to see last 20 steps
458 }
459#endif
460 }
461
462 if( ( substep_no >= max_substeps)
463 && !there_is_no_intersection
464 && !found_approximate_intersection )
465 {
466 G4cout << "ERROR - G4SimpleLocator::EstimateIntersectionPoint()" << G4endl
467 << " Start and Endpoint of Requested Step:" << G4endl;
468 printStatus( CurveStartPointVelocity, CurveEndPointVelocity,
469 -1.0, NewSafety, 0);
470 G4cout << G4endl
471 << " Start and end-point of current Sub-Step:" << G4endl;
472 printStatus( CurrentA_PointVelocity, CurrentA_PointVelocity,
473 -1.0, NewSafety, substep_no-1);
474 printStatus( CurrentA_PointVelocity, CurrentB_PointVelocity,
475 -1.0, NewSafety, substep_no);
476
477 std::ostringstream message;
478 message << "Convergence is requiring too many substeps: "
479 << substep_no << G4endl
480 << " Abandoning effort to intersect." << G4endl
481 << " Found intersection = "
482 << found_approximate_intersection << G4endl
483 << " Intersection exists = "
484 << !there_is_no_intersection << G4endl;
485 message.precision(10);
486 G4double done_len = CurrentA_PointVelocity.GetCurveLength();
487 G4double full_len = CurveEndPointVelocity.GetCurveLength();
488 message << " Undertaken only length: " << done_len
489 << " out of " << full_len << " required." << G4endl
490 << " Remaining length = " << full_len-done_len;
491
492 G4Exception("G4SimpleLocator::EstimateIntersectionPoint()",
493 "GeomNav0003", FatalException, message);
494 }
495 else if( substep_no >= warn_substeps )
496 {
497 std::ostringstream message;
498 message.precision(10);
499
500 message << "Many substeps while trying to locate intersection." << G4endl
501 << " Undertaken length: "
502 << CurrentB_PointVelocity.GetCurveLength()
503 << " - Needed: " << substep_no << " substeps." << G4endl
504 << " Warning level = " << warn_substeps
505 << " and maximum substeps = " << max_substeps;
506 G4Exception("G4SimpleLocator::EstimateIntersectionPoint()",
507 "GeomNav1002", JustWarning, message);
508 }
509 return !there_is_no_intersection; // Success or failure
510}
