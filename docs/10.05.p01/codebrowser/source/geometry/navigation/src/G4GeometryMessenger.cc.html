55//
56// Constructor
57//
58G4GeometryMessenger::G4GeometryMessenger(G4TransportationManager* tman)
59 : tol(0.0), recLevel(0), recDepth(-1), tmanager(tman), tvolume(0)
60{
61 geodir = new G4UIdirectory( "/geometry/" );
62 geodir->SetGuidance( "Geometry control commands." );
63
64 //
65 // Geometry navigator commands
66 //
67 navdir = new G4UIdirectory( "/geometry/navigator/" );
68 navdir->SetGuidance( "Geometry navigator control setup." );
69
70 resCmd = new G4UIcmdWithoutParameter( "/geometry/navigator/reset", this );
71 resCmd->SetGuidance( "Reset navigator and navigation history." );
72 resCmd->SetGuidance( "NOTE: must be called only after kernel has been" );
73 resCmd->SetGuidance( " initialized once through the run manager!" );
74 resCmd->AvailableForStates(G4State_Idle);
75
76 verbCmd = new G4UIcmdWithAnInteger( "/geometry/navigator/verbose", this );
77 verbCmd->SetGuidance( "Set run-time verbosity for the navigator." );
78 verbCmd->SetGuidance(" 0 : Silent (default)");
79 verbCmd->SetGuidance(" 1 : Display volume positioning and step lengths");
80 verbCmd->SetGuidance(" 2 : Display step/safety info on point location");
81 verbCmd->SetGuidance(" 3 : Display minimal state at -every- step");
82 verbCmd->SetGuidance(" 4 : Maximum verbosity (very detailed!)");
83 verbCmd->SetGuidance( "NOTE: this command has effect -only- if Geant4 has" );
84 verbCmd->SetGuidance( " been installed with the G4VERBOSE flag set!" );
85 verbCmd->SetParameterName("level",true);
86 verbCmd->SetDefaultValue(0);
87 verbCmd->SetRange("level >=0 && level <=4");
88
89 chkCmd = new G4UIcmdWithABool( "/geometry/navigator/check_mode", this );
90 chkCmd->SetGuidance( "Set navigator in -check_mode- state." );
91 chkCmd->SetGuidance( "This will cause extra checks to be applied during" );
92 chkCmd->SetGuidance( "navigation. More strict and less tolerant conditions" );
93 chkCmd->SetGuidance( "are applied. A run-time performance penalty may be" );
94 chkCmd->SetGuidance( "observed when the -check_mode- state is activated." );
95 chkCmd->SetGuidance( "NOTE: this command has effect -only- if Geant4 has" );
96 chkCmd->SetGuidance( " been installed with the G4VERBOSE flag set!" );
97 chkCmd->SetParameterName("checkFlag",true);
98 chkCmd->SetDefaultValue(false);
99 chkCmd->AvailableForStates(G4State_Idle);
100
101 pchkCmd = new G4UIcmdWithABool( "/geometry/navigator/push_notify", this );
102 pchkCmd->SetGuidance( "Set navigator verbosity push notifications." );
103 pchkCmd->SetGuidance( "This allows to disable/re-enable verbosity in" );
104 pchkCmd->SetGuidance( "navigation, when tracks may get stuck and require" );
105 pchkCmd->SetGuidance( "one artificial push along the direction by the" );
106 pchkCmd->SetGuidance( "navigator. Notification is active by default." );
107 pchkCmd->SetGuidance( "NOTE: this command has effect -only- if Geant4 has" );
108 pchkCmd->SetGuidance( " been installed with the G4VERBOSE flag set!" );
109 pchkCmd->SetParameterName("pushFlag",true);
110 pchkCmd->SetDefaultValue(true);
111 pchkCmd->AvailableForStates(G4State_Idle);
112
113 //
114 // Geometry verification test commands
115 //
116 testdir = new G4UIdirectory( "/geometry/test/" );
117 testdir->SetGuidance( "Geometry verification control setup." );
118 testdir->SetGuidance( "Helps in detecting possible overlapping regions." );
119
120 tolCmd = new G4UIcmdWithADoubleAndUnit( "/geometry/test/tolerance",this );
121 tolCmd->SetGuidance( "Define tolerance (in mm) by which overlaps reports" );
122 tolCmd->SetGuidance( "should be reported. By default, all overlaps are" );
123 tolCmd->SetGuidance( "reported, i.e. tolerance is set to: 0*mm." );
124 tolCmd->SetParameterName( "Tolerance", true, true );
125 tolCmd->SetDefaultValue( 0 );
126 tolCmd->SetDefaultUnit( "mm" );
127 tolCmd->SetUnitCategory( "Length" );
128
129 verCmd = new G4UIcmdWithABool( "/geometry/test/verbosity", this );
130 verCmd->SetGuidance( "Specify if running in verbosity mode or not." );
131 verCmd->SetGuidance( "By default verbosity is set to ON (TRUE)." );
132 verCmd->SetParameterName("verbosity",true);
133 verCmd->SetDefaultValue(true);
134 verCmd->AvailableForStates(G4State_Idle);
135
136 rslCmd = new G4UIcmdWithAnInteger( "/geometry/test/resolution", this );
137 rslCmd->SetGuidance( "Set the number of points on surface to be generated for" );
138 rslCmd->SetGuidance( "checking overlaps." );
139 rslCmd->SetParameterName("resolution",true);
140 rslCmd->SetDefaultValue(10000);
141
142 rcsCmd = new G4UIcmdWithAnInteger( "/geometry/test/recursion_start", this );
143 rcsCmd->SetGuidance( "Set the initial level in the geometry tree for recursion." );
144 rcsCmd->SetGuidance( "recursive_test will then start from the specified level." );
145 rcsCmd->SetParameterName("initial_level",true);
146 rcsCmd->SetDefaultValue(0);
147
148 rcdCmd = new G4UIcmdWithAnInteger( "/geometry/test/recursion_depth", this );
149 rcdCmd->SetGuidance( "Set the depth in the geometry tree for recursion." );
150 rcdCmd->SetGuidance( "recursive_test will then stop after reached the specified depth." );
151 rcdCmd->SetGuidance( "By default, recursion will proceed for the whole depth." );
152 rcdCmd->SetParameterName("recursion_depth",true);
153 rcdCmd->SetDefaultValue(-1);
154
155 errCmd = new G4UIcmdWithAnInteger( "/geometry/test/maximum_errors", this );
156 errCmd->SetGuidance( "Set the maximum number of overlap errors to report" );
157 errCmd->SetGuidance( "for each single volume being checked." );
158 errCmd->SetGuidance( "Once reached the maximum number specified, overlaps" );
159 errCmd->SetGuidance( "affecting that volume further than that are simply ignored." );
160 errCmd->SetParameterName("maximum_errors",true);
161 errCmd->SetDefaultValue(1);
162
163 recCmd = new G4UIcmdWithoutParameter( "/geometry/test/run", this );
164 recCmd->SetGuidance( "Start running the recursive overlap check." );
165 recCmd->SetGuidance( "Volumes are recursively asked to verify for overlaps" );
166 recCmd->SetGuidance( "for points generated on the surface against their" );
167 recCmd->SetGuidance( "respective mother volume and sisters at the same" );
168 recCmd->SetGuidance( "level, performing for all daughters and daughters of" );
169 recCmd->SetGuidance( "daughters, etc." );
170 recCmd->SetGuidance( "NOTE: it may take a very long time," );
171 recCmd->SetGuidance( " depending on the geometry complexity !");
172 recCmd->AvailableForStates(G4State_Idle);
173}

188//
189// Init
190//
191void
192G4GeometryMessenger::Init()
193{
194 // Create checker...
195 //
196 if (!tvolume)
197 {
198 // Get the world volume
199 //
200 G4VPhysicalVolume* world =
201 tmanager->GetNavigatorForTracking()->GetWorldVolume();
202
203 // Test the actual detector...
204 //
205 tvolume = new G4GeomTestVolume(world);
206 }
207}

209//
210// SetNewValue
211//
212void
213G4GeometryMessenger::SetNewValue( G4UIcommand* command, G4String newValues )
214{
215 if (command == resCmd) {
216 ResetNavigator();
217 }
218 else if (command == verbCmd) {
219 SetVerbosity( newValues );
220 }
221 else if (command == chkCmd) {
222 SetCheckMode( newValues );
223 }
224 else if (command == tolCmd) {
225 Init();
226 tol = tolCmd->GetNewDoubleValue( newValues )
227 * tolCmd->GetNewUnitValue( newValues );
228 tvolume->SetTolerance(tol);
229 }
230 else if (command == verCmd) {
231 Init();
232 tvolume->SetVerbosity(verCmd->GetNewBoolValue( newValues ));
233 }
234 else if (command == rslCmd) {
235 Init();
236 tvolume->SetResolution(rslCmd->GetNewIntValue( newValues ));
237 }
238 else if (command == rcsCmd) {
239 recLevel = rcsCmd->GetNewIntValue( newValues );
240 }
241 else if (command == rcdCmd) {
242 recDepth = rcdCmd->GetNewIntValue( newValues );
243 }
244 else if (command == errCmd) {
245 Init();
246 tvolume->SetErrorsThreshold(errCmd->GetNewIntValue( newValues ));
247 }
248 else if (command == recCmd) {
249 Init();
250 G4cout << "Running geometry overlaps check..." << G4endl;
251 RecursiveOverlapTest();
252 G4cout << "Geometry overlaps check completed !" << G4endl;
253 }
254}

256//
257// GetCurrentValue
258//
259G4String
260G4GeometryMessenger::GetCurrentValue(G4UIcommand* command )
261{
262 G4String cv = "";
263 if (command == tolCmd) {
264 cv = tolCmd->ConvertToString( tol, "mm" );
265 }
266 return cv;
267}

269//
270// CheckGeometry
271//
272void
273G4GeometryMessenger::CheckGeometry()
274{
275 // Verify that the geometry is closed
276 //
277 G4GeometryManager* geomManager = G4GeometryManager::GetInstance();
278 if (!geomManager->IsGeometryClosed()) {
279 geomManager->OpenGeometry();
280 geomManager->CloseGeometry(true);
281 }
282}
283
284//
285// ResetNavigator
286//
287void
288G4GeometryMessenger::ResetNavigator()
289{
290 // Close geometry and reset optimisation if necessary
291 //
292 CheckGeometry();
293
294 // Reset navigator's state
295 //
296 G4ThreeVector pt(0,0,0);
297 G4Navigator* navigator = tmanager->GetNavigatorForTracking();
298 navigator->LocateGlobalPointAndSetup(pt,0,false);
299}

334//
335// Recursive Overlap Test
336//
337void
338G4GeometryMessenger::RecursiveOverlapTest()
339{
340 // Close geometry if necessary
341 //
342 CheckGeometry();
343
344 // Make test on single line supplied by user recursively
345 //
346 tvolume->TestRecursiveOverlap( recLevel, recDepth );
347}
