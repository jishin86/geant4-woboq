29// class G4PartialPhantomParameterisation
30//
31// Class description:
32//
33// Describes partial regular parameterisations, i.e. the voxels do not
34// completely fill the container in the three dimensions
35
36// History:
37// - Created. P. Arce, September 2010
38// *********************************************************************
39
40#ifndef G4PartialPhantomParameterisation_HH
41#define G4PartialPhantomParameterisation_HH
42
43#include <vector>
44#include <set>
45#include <map>
46
47#include "G4Types.hh"
48#include "G4PhantomParameterisation.hh"
49#include "G4AffineTransform.hh"
50
51class G4VPhysicalVolume;
52class G4VTouchable;
53class G4VSolid;
54class G4Material;
55
56class G4PartialPhantomParameterisation : public G4PhantomParameterisation
57{
58 public: // with description
59
60 G4PartialPhantomParameterisation();
61 ~G4PartialPhantomParameterisation();
62
63 void ComputeTransformation(const G4int, G4VPhysicalVolume *) const;
64
65 G4Material* ComputeMaterial(const G4int repNo,
66 G4VPhysicalVolume *currentVol,
67 const G4VTouchable *parentTouch=0);
68
69 G4int GetReplicaNo( const G4ThreeVector& localPoint,
70 const G4ThreeVector& localDir );
71 // Get the voxel number corresponding to the point in the container
72 // frame. Use 'localDir' to avoid precision problems at the surfaces.
73
74 G4ThreeVector GetTranslation(const G4int copyNo ) const;
75
76 size_t GetMaterialIndex( size_t nx, size_t ny, size_t nz) const;
77 size_t GetMaterialIndex( size_t copyNo) const;
78
79 G4Material* GetMaterial( size_t nx, size_t ny, size_t nz) const;
80 G4Material* GetMaterial( size_t copyNo ) const;
81
82 void SetFilledIDs( std::multimap<G4int,G4int> fid ){
83 fFilledIDs = fid;
84 }
85
86 void SetFilledMins( std::map< G4int, std::map<G4int,G4int> > fmins ) {
87 fFilledMins = fmins;
88 }
89
90 void BuildContainerWalls();
91
92 private:
93
94 void ComputeVoxelIndices(const G4int copyNo, size_t& nx,
95 size_t& ny, size_t& nz ) const;
96 // Convert the copyNo to voxel numbers in x, y and z.
97
98 void CheckCopyNo( const G4int copyNo ) const;
99 // Check that the copy number is within limits.
100
101 private:
102
103 std::multimap<G4int,G4int> fFilledIDs;
104 std::map< G4int, std::map<G4int,G4int> > fFilledMins;
105};

107#endif
