33// ********************************************************************
34// VoxelLocate
35// ********************************************************************
36//
37inline
38G4int
39G4ReplicaNavigation::VoxelLocate( const G4SmartVoxelHeader* pHead,
40 const G4ThreeVector& localPoint,
41 const G4int blocked ) const
42{
43 EAxis targetHeaderAxis;
44 G4double coord=0.;
45 G4double targetHeaderMin, targetHeaderMax;
46 G4double targetHeaderNodeWidth, targetNodePos;
47 G4int targetHeaderNoSlices, targetNodeNo;
48
49 targetHeaderAxis = pHead->GetAxis();
50 targetHeaderNoSlices = pHead->GetNoSlices();
51 targetHeaderMin = pHead->GetMinExtent();
52 targetHeaderMax = pHead->GetMaxExtent();
53 targetHeaderNodeWidth = ( targetHeaderMax-targetHeaderMin )
54 / targetHeaderNoSlices;
55
56 switch (targetHeaderAxis)
57 {
58 case kXAxis:
59 coord = localPoint.x();
60 break;
61 case kYAxis:
62 coord = localPoint.y();
63 break;
64 case kZAxis:
65 coord = localPoint.z();
66 break;
67 case kRho:
68 coord = localPoint.perp();
69 break;
70 case kPhi:
71 coord = localPoint.phi();
72 if ( (coord<0) && (coord<targetHeaderMin) ) coord += CLHEP::twopi;
73 break;
74 case kRadial3D:
75 default:
76 break;
77 }
78 targetNodePos = (coord-targetHeaderMin)/targetHeaderNodeWidth;
79 targetNodeNo = (G4int) targetNodePos;
80
81 if ( targetNodeNo==blocked )
82 {
83 targetNodeNo = (targetNodePos-targetNodeNo<0.5)
84 ? targetNodeNo-1 : targetNodeNo+1;
85
86 // Do not need to check range: If on outer edge of zeroth
87 // voxel & it is blocked => should have exited mother
88 // (or similar) P.Kent
89 // assert(targetNodeNo>=0&&targetNodeNo<targetHeaderNoSlices);
90
91 if( (targetNodeNo<0) || (targetNodeNo>=targetHeaderNoSlices) )
92 {
93
94#ifdef G4DEBUG_NAVIGATION
95 std::ostringstream message;
96 message << "Voxel location failed:" << G4endl
97 << " (targetNodeNo>=0&&targetNodeNo<targetHeaderNoSlices) "
98 << G4endl
99 << " - targetNodeNo= " << targetNodeNo << G4endl
100 << " - Number of Slices = " << targetHeaderNoSlices;
101 G4Exception("G4ReplicaNavigation::VoxelLocate()",
102 "GeomNav1002", JustWarning, message);
103#endif
104 // In the case of rotational symmetry and an extent over the
105 // whole 360 degrees, the above is not true and you can go from
106 // the last voxel to the zeroth and vice versa
107 // H.Boie, April 30, 2001
108 if ( (targetHeaderAxis==kPhi)
109 && (targetHeaderMin==0) && (targetHeaderMax==CLHEP::twopi) )
110 {
111 if ( targetNodeNo<0 )
112 targetNodeNo = targetHeaderNoSlices-1;
113 else if ( targetNodeNo>=targetHeaderNoSlices )
114 targetNodeNo = 0;
115 }
116 else
117 {
118 if( targetNodeNo<0 )
119 targetNodeNo = 0;
120 else if ( targetNodeNo>=targetHeaderNoSlices )
121 targetNodeNo = targetHeaderNoSlices-1;
122 }
123 }
124 }
125 else
126 {
127 // Rounding protection
128 //
129 if ( targetNodeNo<0 )
130 {
131 targetNodeNo = 0;
132 }
133 else if ( targetNodeNo>=targetHeaderNoSlices )
134 {
135 targetNodeNo = targetHeaderNoSlices-1;
136 }
137 }
138 return targetNodeNo;
139}

141// ********************************************************************
142// LevelLocate
143// ********************************************************************
144//
145inline
146G4bool
147G4ReplicaNavigation::LevelLocate( G4NavigationHistory& history,
148 const G4VPhysicalVolume* blockedVol,
149 const G4int blockedNum,
150 const G4ThreeVector&, // globalPoint
151 const G4ThreeVector*, // globalDirection
152 const G4bool, // pLocatedOnEdge
153 G4ThreeVector& localPoint )
154{
155 G4VPhysicalVolume *motherPhysical, *pPhysical;
156 G4LogicalVolume *motherLogical;
157 G4SmartVoxelHeader *motherVoxelHeader;
158 G4int nodeNo;
159
160 motherPhysical = history.GetTopVolume();
161 motherLogical = motherPhysical->GetLogicalVolume();
162 motherVoxelHeader = motherLogical->GetVoxelHeader();
163 pPhysical = motherLogical->GetDaughter(0);
164
165 if ( blockedVol==pPhysical )
166 {
167 nodeNo = VoxelLocate(motherVoxelHeader, localPoint, blockedNum);
168 }
169 else
170 {
171 nodeNo = VoxelLocate(motherVoxelHeader, localPoint);
172 }
173
174 ComputeTransformation(nodeNo, pPhysical, localPoint);
175 history.NewLevel(pPhysical, kReplica, nodeNo);
176 pPhysical->SetCopyNo(nodeNo);
177
178 return true;
179}

181// ********************************************************************
182// SetPhiTransformation
183// ********************************************************************
184//
185inline
186void
187G4ReplicaNavigation::SetPhiTransformation( const G4double ang,
188 G4VPhysicalVolume* pVol ) const
189{
190 G4RotationMatrix rm;
191 rm.rotateZ(ang);
192 if ( pVol )
193 *pVol->GetRotation() = rm;
194}
