58struct G4ExitNormal
59{
60 // Bucket to hold value of Normal (3-vector),
61 // bools for calculated and leave-behind or 'validConvex',
62 // and exiting side.
63
64 enum ESide {kNull,kRMin,kRMax,kSPhi,kEPhi,kPX,kMX,kPY,kMY,kPZ,kMZ,kMother};
65 // Identity of 'Side' of Replicas. Used by DistanceToOut methods.
66
67 G4ThreeVector exitNormal;
68 G4bool calculated; // Normal
69 G4bool validConvex; // Solid locally convex
70 ESide exitSide;
71
72 public:
73
74 G4ExitNormal(G4ThreeVector norm = G4ThreeVector(0.,0.,0.),
75 G4bool calc = false,
76 G4bool valid= false,
77 ESide side = kNull )
78 { exitNormal= norm; calculated= calc; validConvex=valid; exitSide=side;}
79};
