29// class G4RegularNavigation
30//
31// Class description:
32//
33// Utility for fast navigation in volumes containing a regular
34// parameterisation. If two contiguous voxels have the same material,
35// navigation does not stop at the surface
36
37// History:
38// - Created. P. Arce, May 2007
39// --------------------------------------------------------------------
40#ifndef G4RegularNavigation_HH
41#define G4RegularNavigation_HH
42
43#include <vector>
44
45#include "G4Types.hh"
46#include "G4ThreeVector.hh"
47
48class G4NormalNavigation;
49class G4VPhysicalVolume;
50class G4Navigator;
51class G4NavigationHistory;
52
53class G4RegularNavigation
54{
55 public: // with description
56
57 G4RegularNavigation();
58 ~G4RegularNavigation();
59
60 G4bool LevelLocate( G4NavigationHistory& history,
61 const G4VPhysicalVolume* blockedVol,
62 const G4int blockedNum,
63 const G4ThreeVector& globalPoint,
64 const G4ThreeVector* globalDirection,
65 const G4bool pLocatedOnEdge,
66 G4ThreeVector& localPoint );
67 // Locate point using its position with respect to regular
68 // parameterisation container volume.
69
70 G4double ComputeStep( const G4ThreeVector& globalPoint,
71 const G4ThreeVector& globalDirection,
72 const G4double currentProposedStepLength,
73 G4double& newSafety,
74 G4NavigationHistory& history,
75 G4bool& validExitNormal,
76 G4ThreeVector& exitNormal,
77 G4bool& exiting,
78 G4bool& entering,
79 G4VPhysicalVolume *(*pBlockedPhysical),
80 G4int& blockedReplicaNo );
81 // Method never called because to be called the daughter has to be a
82 // 'regular' volume. This would only happen if the track is in the
83 // mother of voxels volume. But the voxels fill completely their mother,
84 // so when a track enters the mother it automatically enters a voxel.
85
86 G4double ComputeStepSkippingEqualMaterials(
87 G4ThreeVector& localPoint,
88 const G4ThreeVector& globalDirection,
89 const G4double currentProposedStepLength,
90 G4double& newSafety,
91 G4NavigationHistory& history,
92 G4bool& validExitNormal,
93 G4ThreeVector& exitNormal,
94 G4bool& exiting,
95 G4bool& entering,
96 G4VPhysicalVolume *(*pBlockedPhysical),
97 G4int& blockedReplicaNo,
98 G4VPhysicalVolume* pCurrentPhysical);
99 // Compute the step skipping surfaces when they separate voxels with
100 // equal materials. Loop to voxels until a different material is found:
101 // invokes G4NormalNavigation::ComputeStep() in each voxel and move the
102 // point to the next voxel.
103
104 G4double ComputeSafety( const G4ThreeVector& localPoint,
105 const G4NavigationHistory& history,
106 const G4double pProposedMaxLength=DBL_MAX );
107 // Method never called because to be called the daughter has to be a
108 // 'regular' volume. This would only happen if the track is in the
109 // mother of voxels volume. But the voxels fill completely their mother,
110 // so when a track enters the mother it automatically enters a voxel.
111
112 public: // without description
113
114 // Set and Get methods
115
116 void SetVerboseLevel(G4int level) { fverbose = level; }
117 void CheckMode(G4bool mode) { fcheck = mode; }
118 void SetNormalNavigation( G4NormalNavigation* fnormnav )
119 { fnormalNav = fnormnav; }
120
121 private:
122
123 G4int fverbose;
124 G4bool fcheck;
125
126 G4NormalNavigation* fnormalNav;
127 G4double kCarTolerance;
128};
