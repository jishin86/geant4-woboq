174inline
175void G4PropagatorInField::SetThresholdNoZeroStep( G4int noAct,
176 G4int noHarsh,
177 G4int noAbandon )
178{
179 if( noAct>0 )
180 fActionThreshold_NoZeroSteps = noAct;
181
182 if( noHarsh > fActionThreshold_NoZeroSteps )
183 fSevereActionThreshold_NoZeroSteps = noHarsh;
184 else
185 fSevereActionThreshold_NoZeroSteps = 2*(fActionThreshold_NoZeroSteps+1);
186
187 if( noAbandon > fSevereActionThreshold_NoZeroSteps+5 )
188 fAbandonThreshold_NoZeroSteps = noAbandon;
189 else
190 fAbandonThreshold_NoZeroSteps = 2*(fSevereActionThreshold_NoZeroSteps+3);
191}
192
193inline
194G4int G4PropagatorInField::GetThresholdNoZeroSteps( G4int i )
195{
196 G4int t=0;
197 if( i==0 ) { t = 3; } // No of parameters
198 else if (i==1) { t = fActionThreshold_NoZeroSteps; }
199 else if (i==2) { t = fSevereActionThreshold_NoZeroSteps; }
200 else if (i==3) { t = fAbandonThreshold_NoZeroSteps; }
201
202 return t;
203}

270inline
271G4bool G4PropagatorInField::IntersectChord( const G4ThreeVector& StartPointA,
272 const G4ThreeVector& EndPointB,
273 G4double &NewSafety,
274 G4double &LinearStepLength,
275 G4ThreeVector &IntersectionPoint )
276{
277 // Calculate the direction and length of the chord AB
278 //
279#ifdef G4DEBUG_PROPAGATION
280 if( fVerbTracePiF )
281 G4cout << "**** G4PropagatorInField::IntersectChord called."
282 << " InPut: StartPointA: " << StartPointA
283 << " EndPointB= " << EndPointB
284 << " StepLength= " << LinearStepLength
285 << " IntersecLen= " << IntersectionPoint
286 << G4endl;
287#endif
288
289 G4bool retVal=
290 fIntersectionLocator
291 ->IntersectChord(StartPointA,EndPointB,NewSafety,
292 fPreviousSafety,fPreviousSftOrigin,
293 LinearStepLength,IntersectionPoint);
294
295#ifdef G4DEBUG_PROPAGATION
296 if( fVerbTracePiF )
297 G4cout << "**** G4PropagatorInField::IntersectChord ended."
298 << " OutPut: Safety= " << NewSafety
299 << " StepLength= " << LinearStepLength
300 << " IntersecPt= " << IntersectionPoint
301 << G4endl;
302#endif
303
304 return retVal;
305}

307inline G4EquationOfMotion* G4PropagatorInField::GetCurrentEquationOfMotion()
308{
309 if(auto pChordFinder = GetChordFinder()) {
310 if(auto pIntDriver = pChordFinder->GetIntegrationDriver()) {
311 return pIntDriver->GetEquationOfMotion();
312 }
313 }
314 return nullptr;
