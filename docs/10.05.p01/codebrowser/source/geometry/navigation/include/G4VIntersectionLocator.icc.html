101inline G4bool
102G4VIntersectionLocator::IntersectChord( const G4ThreeVector& StartPointA,
103 const G4ThreeVector& EndPointB,
104 G4double &NewSafety,
105 G4double &PreviousSafety,
106 G4ThreeVector &PreviousSftOrigin,
107 G4double &LinearStepLength,
108 G4ThreeVector &IntersectionPoint,
109 G4bool *ptrCalledNavigator )
110{
111 G4bool CalledNavigator = false;
112
113 // Calculate the direction and length of the chord AB
114
115 G4ThreeVector ChordAB_Vector = EndPointB - StartPointA;
116 G4double ChordAB_Length = ChordAB_Vector.mag(); // Magnitude (norm)
117 G4ThreeVector ChordAB_Dir = ChordAB_Vector.unit();
118 G4bool intersects;
119 G4ThreeVector OriginShift = StartPointA - PreviousSftOrigin ;
120 G4double MagSqShift = OriginShift.mag2() ;
121 G4double currentSafety;
122
123 if( MagSqShift >= sqr(PreviousSafety) )
124 {
125 currentSafety = 0.0 ;
126 }
127 else
128 {
129 currentSafety = PreviousSafety - std::sqrt(MagSqShift) ;
130 }
131
132 if( fiUseSafety && (ChordAB_Length <= currentSafety) )
133 {
134 // The Step is guaranteed to be taken
135
136 LinearStepLength = ChordAB_Length;
137 intersects = false;
138 NewSafety= currentSafety;
139 CalledNavigator= false;
140 // G4cout << " IntersectChord> Step 'guaranteed' taken: safety= " << currentSafety << " chordAB-len= " << ChordAB_Length << G4endl;
141 }
142 else
143 {
144 // G4cout << " IntersectChord> Step asking Navigator: safety= " << currentSafety << " chordAB-len= " << ChordAB_Length << G4endl;
145
146 // Check whether any volumes are encountered by the chord AB
147
148 LinearStepLength = GetNavigatorFor()->ComputeStep( StartPointA,
149 ChordAB_Dir, ChordAB_Length, NewSafety );
150 intersects = (LinearStepLength <= ChordAB_Length);
151 // G4Navigator contracts to return k_infinity if len==asked
152 // and it did not find a surface boundary at that length
153
154 // G4cout << "G4VIntersectionLocator: intersect= " << intersects
155 // << " step= " << LinearStepLength << " Chord length= " << ChordAB_Length;
156
157 LinearStepLength = std::min( LinearStepLength, ChordAB_Length);
158 CalledNavigator = true;
159
160 // Save the last calculated safety!
161
162 PreviousSftOrigin = StartPointA;
163 PreviousSafety = NewSafety;
164
165 if( intersects )
166 {
167 // Intersection Point of chord AB and either volume A's surface
168 // or a daughter volume's surface ..
169 IntersectionPoint = StartPointA + LinearStepLength * ChordAB_Dir;
170 }
171 }
172 if( ptrCalledNavigator )
173 {
174 *ptrCalledNavigator = CalledNavigator;
175 }
176
177 return intersects;
178}
