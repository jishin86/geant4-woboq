29// class G4DrawVoxels
30//
31// Class description:
32//
33// Utility class for the visualization of voxels in the detector geometry.
34// Define G4DrawVoxelsDebug in the environment at compilation for debugging
35// information printed to G4cout.
36
37// History:
38// 03/08/1999 The G4VisAttributes have been made member data for lifetime
39// reasons / visualisation - L.G (ask John Allison for further
40// explanation).
41// 29/07/1999 First comitted version - L.G.
42// --------------------------------------------------------------------
43#ifndef G4DrawVoxels_HH
44#define G4DrawVoxels_HH
45
46#include "G4VisAttributes.hh"
47#include "G4VoxelLimits.hh"
48#include "G4PlacedPolyhedron.hh"
49
50class G4SmartVoxelHeader;
51class G4LogicalVolume;
52
53// ***********************************************************************
54
55class G4DrawVoxels
56{
57 public: // with description
58
59 G4DrawVoxels();
60 // Constructor. It initialises the members data to default colors
61 // Copy constructor and assignment operator not supported (array
62 // fvoxelcolours ...).
63
64 ~G4DrawVoxels();
65 // Destructor NOT virtual. Not a base class.
66
67 void DrawVoxels(const G4LogicalVolume* lv) const;
68 G4PlacedPolyhedronList* CreatePlacedPolyhedra(const G4LogicalVolume*) const;
69
70 void SetVoxelsVisAttributes(G4VisAttributes&,
71 G4VisAttributes&,
72 G4VisAttributes&);
73 void SetBoundingBoxVisAttributes(G4VisAttributes&);
74
75 private:
76
77 // Member data
78 G4VisAttributes fVoxelsVisAttributes[3];
79 G4VisAttributes fBoundingBoxVisAttributes;
80
81 void ComputeVoxelPolyhedra(const G4LogicalVolume*,
82 const G4SmartVoxelHeader*,
83 G4VoxelLimits&,
84 G4PlacedPolyhedronList*) const;
85
86 // Copy constructor Assignment operator not supported
87 // (array fvoxelcolours ...)
88 G4DrawVoxels(const G4DrawVoxels&);
89 G4DrawVoxels operator=(const G4DrawVoxels&);
90};

92#endif
