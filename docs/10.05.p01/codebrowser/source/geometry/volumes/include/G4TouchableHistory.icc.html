32extern G4GEOM_DLL G4Allocator<G4TouchableHistory>*& aTouchableHistoryAllocator();
33
34inline
35void G4TouchableHistory::UpdateYourself( G4VPhysicalVolume* pPhysVol,
36 const G4NavigationHistory* pHistory )
37{
38 fhistory = *pHistory;
39 const G4AffineTransform& tf = fhistory.GetTopTransform();
40 if( pPhysVol == 0 )
41 {
42 // This means that the track has left the World Volume.
43 // Since the Navigation History does not already reflect this,
44 // we must correct this problem here.
45 //
46 fhistory.SetFirstEntry(pPhysVol);
47 }
48 ftlate = tf.InverseNetTranslation();
49 frot = tf.InverseNetRotation();
50}
51
52inline
53G4int G4TouchableHistory::CalculateHistoryIndex( G4int stackDepth ) const
54{
55 return (fhistory.GetDepth()-stackDepth); // was -1
56}
57
58inline
59G4VPhysicalVolume* G4TouchableHistory::GetVolume( G4int depth ) const
60{
61 return fhistory.GetVolume(CalculateHistoryIndex(depth));
62}
63
64inline
65G4VSolid* G4TouchableHistory::GetSolid( G4int depth ) const
66{
67 return fhistory.GetVolume(CalculateHistoryIndex(depth))
68 ->GetLogicalVolume()->GetSolid();
69}
70
71inline
72G4int G4TouchableHistory::GetReplicaNumber( G4int depth ) const
73{
74 return fhistory.GetReplicaNo(CalculateHistoryIndex(depth));
75}
76
77inline
78G4int G4TouchableHistory::GetHistoryDepth() const
79{
80 return fhistory.GetDepth();
81}
82
83inline
84G4int G4TouchableHistory::MoveUpHistory( G4int num_levels )
85{
86 G4int maxLevelsMove = fhistory.GetDepth();
87 G4int minLevelsMove = 0; // Cannot redescend today!
88 // Soon it will be possible
89 // by adding a data member here
90 // fCurrentDepth;
91 if( num_levels > maxLevelsMove )
92 {
93 num_levels = maxLevelsMove;
94 }
95 else if( num_levels < minLevelsMove )
96 {
97 num_levels = minLevelsMove;
98 }
99 fhistory.BackLevel( num_levels );
100
101 return num_levels;
102}
103
104inline
105const G4NavigationHistory* G4TouchableHistory::GetHistory() const
106{
107 return &fhistory;
108}

110// There is no provision in case this class is subclassed.
111// If it is subclassed, this will fail and may not give errors!
112//
113inline
114void* G4TouchableHistory::operator new(size_t)
115{
116 // Once the Navigator calls InitializeAllocator,
117 // the if block below can be removed.
118 //
119 if (!aTouchableHistoryAllocator())
120 {
121 aTouchableHistoryAllocator() = new G4Allocator<G4TouchableHistory>;
122 }
123 return (void *) aTouchableHistoryAllocator()->MallocSingle();
124}
125
126inline
127void G4TouchableHistory::operator delete(void *aTH)
128{
129 aTouchableHistoryAllocator()->FreeSingle((G4TouchableHistory *) aTH);
130}
