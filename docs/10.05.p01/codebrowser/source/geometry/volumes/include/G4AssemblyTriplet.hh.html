29// Class G4AssemblyTriplet
30//
31// Class description:
32//
33// A class to help place logical or assembly volumes inside a generic
34// containers (like STL vector ) together with information about its rotation,
35// placement and eventually reflection.
36// How to interpret the rotation and translation depends on the class which
37// uses a container of these triplets. The first class using G4AssemblyTriplet
38// is G4AssemblyVolume class.
39// The pointer to the logical or assembly volume is copied so this class
40// does not take its ownership and does not delete the object behind.
41
42// Author: Radovan Chytracek
43// Version: 1.0
44// Date: November 2000
45//
46// History:
47// March 2006, I.Hrivnacova - Extended to support assembly of assemblies
48// of volumes and reflections
49// ----------------------------------------------------------------------
50#ifndef G4_ASSEMBLYTRIPLET_H
51#define G4_ASSEMBLYTRIPLET_H
52
53#include "G4ThreeVector.hh"
54#include "G4RotationMatrix.hh"
55
56class G4LogicalVolume;
57class G4AssemblyVolume;
58
59class G4AssemblyTriplet
60{
61 public: // with description
62
63 G4AssemblyTriplet();
64 // Default constructor
65
66 G4AssemblyTriplet( G4LogicalVolume* pVolume,
67 G4ThreeVector& translation,
68 G4RotationMatrix* pRotation,
69 G4bool isReflection = false);
70 // An explicit constructor for a logical volume
71
72 G4AssemblyTriplet( G4AssemblyVolume* pAssembly,
73 G4ThreeVector& translation,
74 G4RotationMatrix* pRotation,
75 G4bool isReflection = false);
76 // An explicit constructor for an assembly volume
77
78 G4AssemblyTriplet( const G4AssemblyTriplet& second );
79 // Copy constructor
80
81 ~G4AssemblyTriplet();
82 // Destructor
83
84 G4AssemblyTriplet& operator=( const G4AssemblyTriplet& second );
85 // Assignment operator
86
87 inline G4LogicalVolume* GetVolume() const;
88 // Retrieve the logical volume reference
89
90 inline void SetVolume( G4LogicalVolume* pVolume );
91 // Update the logical volume reference
92
93 inline G4AssemblyVolume* GetAssembly() const;
94 // Retrieve the assembly volume reference
95
96 inline void SetAssembly( G4AssemblyVolume* pAssembly );
97 // Update the assembly volume reference
98
99 inline G4ThreeVector GetTranslation() const;
100 // Retrieve the logical volume translation
101
102 inline void SetTranslation( G4ThreeVector& pVolume );
103 // Update the logical volume translation
104
105 inline G4RotationMatrix* GetRotation() const;
106 // Retrieve the logical volume rotation
107
108 inline void SetRotation( G4RotationMatrix* pVolume );
109 // Update the logical volume rotation
110
111 inline G4bool IsReflection() const;
112 // Return true if the logical or assembly volume has reflection
113
114 private:
115
116 G4LogicalVolume* fVolume;
117 // A logical volume
118
119 G4ThreeVector fTranslation;
120 // A logical volume translation
121
122 G4RotationMatrix* fRotation;
123 // A logical volume rotation
124
125 private:
126
127 // Member data for handling assemblies of assemblies and reflections
128
129 G4AssemblyVolume* fAssembly;
130 // An assembly volume
131
132 G4bool fIsReflection;
133 // True if the logical or assembly volume has reflection
134};

136#include "G4AssemblyTriplet.icc"
137
138#endif // G4_ASSEMBLYTRIPLET_H
