33extern G4GEOM_DLL G4Allocator<G4NavigationHistory>*& aNavigHistoryAllocator();
34
35// There is no provision that this class is subclassed.
36// If it is subclassed & new data members are added then the
37// following "new" & "delete" will fail and give errors.
38//
39inline
40void* G4NavigationHistory::operator new(size_t)
41{
42 if (!aNavigHistoryAllocator())
43 {
44 aNavigHistoryAllocator() = new G4Allocator<G4NavigationHistory>;
45 }
46 return (void *) aNavigHistoryAllocator()->MallocSingle();
47}
48
49inline
50void G4NavigationHistory::operator delete(void *aHistory)
51{
52 aNavigHistoryAllocator()->FreeSingle((G4NavigationHistory *) aHistory);
53}
54
55inline
56G4NavigationHistory&
57G4NavigationHistory::operator=(const G4NavigationHistory &h)
58{
59 if (&h == this) { return *this; }
60
61 // *fNavHistory=*(h.fNavHistory); // This works, but is very slow.
62
63 if( GetMaxDepth() != h.GetMaxDepth() )
64 {
65 fNavHistory->resize( h.GetMaxDepth() );
66 }
67
68 for ( G4int ilev=h.fStackDepth; ilev>=0; --ilev )
69 {
70 (*fNavHistory)[ilev] = (*h.fNavHistory)[ilev];
71 }
72 fStackDepth=h.fStackDepth;
73
74 return *this;
75}
76
77inline
78void G4NavigationHistory::Reset()
79{
80 fStackDepth=0;
81}
82
83inline
84void G4NavigationHistory::Clear()
85{
86 G4AffineTransform origin(G4ThreeVector(0.,0.,0.));
87 G4NavigationLevel tmpNavLevel = G4NavigationLevel(0, origin, kNormal, -1) ;
88
89 Reset();
90 for (G4int ilev=fNavHistory->size()-1; ilev>=0; ilev--)
91 {
92 (*fNavHistory)[ilev] = tmpNavLevel;
93 }
94}

96inline
97void G4NavigationHistory::SetFirstEntry(G4VPhysicalVolume* pVol)
98{
99 G4ThreeVector translation(0.,0.,0.);
100 G4int copyNo = -1;
101
102 // Protection needed in case pVol=null
103 // so that a touchable-history can signal OutOfWorld
104 //
105 if( pVol!=0 )
106 {
107 translation = pVol->GetTranslation();
108 copyNo = pVol->GetCopyNo();
109 }
110 (*fNavHistory)[0] =
111 G4NavigationLevel( pVol, G4AffineTransform(translation), kNormal, copyNo );
112}

181inline
182void G4NavigationHistory::BackLevel()
183{
184 assert( fStackDepth>0 );
185
186 // Tell the level that I am forgetting it
187 // delete (*fNavHistory)[fStackDepth];
188 //
189 fStackDepth--;
190}
191
192inline
193void G4NavigationHistory::BackLevel(G4int n)
194{
195 assert( n<=fStackDepth );
196 fStackDepth-=n;
197}

199inline
200void G4NavigationHistory::EnlargeHistory()
201{
202 G4int len = fNavHistory->size();
203 if ( len==fStackDepth )
204 {
205 // Note: Resize operation clears additional entries
206 //
207 G4int nlen = len+kHistoryStride;
208 fNavHistory->resize(nlen);
209 }
210}

213inline
214void G4NavigationHistory::NewLevel( G4VPhysicalVolume *pNewMother,
215 EVolume vType,
216 G4int nReplica )
217{
218 fStackDepth++;
219 EnlargeHistory(); // Enlarge if required
220 (*fNavHistory)[fStackDepth] =
221 G4NavigationLevel( pNewMother,
222 (*fNavHistory)[fStackDepth-1].GetTransform(),
223 G4AffineTransform(pNewMother->GetRotation(),
224 pNewMother->GetTranslation()),
225 vType,
226 nReplica );
227 // The constructor computes the new global->local transform
228}
