29// class G4PVParameterised
30//
31// Class description:
32//
33// Represents many touchable detector elements differing in their
34// positioning and dimensions. Both are calculated by means
35// of a G4VParameterisation object. The positioning is assumed to
36// be dominant along a cartesian axis (specified).
37
38// History:
39// 29.07.95 P.Kent First non-stub version
40// ----------------------------------------------------------------------
41#ifndef G4PVPARAMETERISED_HH
42#define G4PVPARAMETERISED_HH
43
44#include "G4PVReplica.hh"
45
46class G4PVParameterised : public G4PVReplica
47{
48 public: // with description
49
50 G4PVParameterised(const G4String& pName,
51 G4LogicalVolume* pLogical,
52 G4LogicalVolume* pMotherLogical,
53 const EAxis pAxis,
54 const G4int nReplicas,
55 G4VPVParameterisation *pParam,
56 G4bool pSurfChk=false);
57 // Replicate the volume nReplicas Times using the paramaterisation pParam,
58 // within the mother volume pMotherLogical.
59 // The positioning of the replicas is dominant along the specified axis.
60 // pSurfChk if true activates check for overlaps with existing volumes.
61
62 public: // without description
63
64 G4PVParameterised(const G4String& pName,
65 G4LogicalVolume* pLogical,
66 G4VPhysicalVolume* pMother,
67 const EAxis pAxis,
68 const G4int nReplicas,
69 G4VPVParameterisation *pParam,
70 G4bool pSurfChk=false);
71 // Almost exactly similar to first constructor, changing only mother
72 // pointer's type to PhysicalVolume.
73
74 G4PVParameterised(__void__&);
75 // Fake default constructor for usage restricted to direct object
76 // persistency for clients requiring preallocation of memory for
77 // persistifiable objects.
78
79 public: // with description
80
81 virtual ~G4PVParameterised();
82 // Virtual empty destructor.
83
84 G4bool IsParameterised() const;
85 // Returns true to identify it is a parameterised physical volume.
86
87 G4VPVParameterisation* GetParameterisation() const;
88 // Returns the current pointer to the parameterisation.
89
90 void GetReplicationData(EAxis& axis,
91 G4int& nReplicas,
92 G4double& width,
93 G4double& offset,
94 G4bool& consuming) const;
95 // Fills arguments with the attributes from the base replica.
96 virtual void SetRegularStructureId( G4int Code );
97 // Method sets code and can prepare for special type of regular volumes.
98
99 G4bool CheckOverlaps(G4int res=1000, G4double tol=0.,
100 G4bool verbose=true, G4int maxErr=1);
101 // Verifies if each instance of the parameterised volume is overlapping
102 // with other instances or with the mother volume. Provides default
103 // resolution for the number of points to be generated and verified.
104 // A tolerance for the precision of the overlap check can be specified,
105 // by default it is set to maximum precision.
106 // Reports a maximum of overlaps errors according to parameter in input.
107 // Returns true if an overlap occurs.
108
109 private:
110
111 G4VPVParameterisation *fparam;
112};
113
114#endif
