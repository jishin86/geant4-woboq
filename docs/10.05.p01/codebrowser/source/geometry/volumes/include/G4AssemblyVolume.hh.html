29// Class G4AssemblyVolume
30//
31// Class description:
32//
33// G4AssemblyVolume is a helper class to make the build process of geometry
34// easier. It allows to combine several volumes together in an arbitrary way
35// in 3D space and then work with the result as with a single logical volume
36// for placement.
37// The resulting objects are independent copies of each of the assembled
38// logical volumes. The placements are not, however, bound one to each other
39// when placement is done. They are seen as independent physical volumes in
40// space.
41
42// Author: Radovan Chytracek, John Apostolakis, Gabriele Cosmo
43// Date: November 2000
44//
45// History:
46// March 2006, I.Hrivnacova - Extended to support assembly of assemblies
47// of volumes and reflections
48// ----------------------------------------------------------------------
49#ifndef G4_ASSEMBLYVOLUME_H
50#define G4_ASSEMBLYVOLUME_H
51
52#include <vector>
53
54#include "G4Transform3D.hh"
55#include "G4AssemblyTriplet.hh"
56
57class G4VPhysicalVolume;
58
59class G4AssemblyVolume
60{
61 public: // with description
62
63 G4AssemblyVolume();
64 G4AssemblyVolume( G4LogicalVolume* volume,
65 G4ThreeVector& translation,
66 G4RotationMatrix* rotation);
67 ~G4AssemblyVolume();
68 //
69 // Constructors & destructor.
70 // At destruction all the generated physical volumes and associated
71 // rotation matrices of the imprints will be destroyed.
72 //
73 // The rotation matrix passed as argument can be 0 (identity) or an address
74 // even of an object on the upper stack frame. During assembly imprint, a
75 // new matrix is created anyway and it is kept track of it so it can be
76 // automatically deleted later at the end of the application.
77 // This policy is adopted since user has no control on the way the
78 // rotations are combined.
79
80 void AddPlacedVolume( G4LogicalVolume* pPlacedVolume,
81 G4ThreeVector& translation,
82 G4RotationMatrix* rotation);
83 //
84 // Place the given volume 'pPlacedVolume' inside the assembly.
85 //
86 // The adopted approach:
87 //
88 // - Place it w.r.t. the assembly coordinate system.
89 // This step is applied to each of the participating volumes.
90 //
91 // The other possible approaches:
92 //
93 // - Place w.r.t. the firstly added volume.
94 // When placed the first, the virtual coordinate system becomes
95 // the coordinate system of the first one.
96 // Every next volume being added into the assembly will be placed
97 // w.r.t to the first one.
98 //
99 // - Place w.r.t the last placed volume.
100 // When placed the first, the virtual coordinate system becomes
101 // the coordinate system of the first one.
102 // Every next volume being added into the assembly will be placed
103 // w.r.t to the previous one.
104 //
105 // The rotation matrix passed as argument can be 0 (identity) or an address
106 // even of an object on the upper stack frame. During assembly imprint, a
107 // new matrix is created anyway and it is kept track of it so it can be
108 // automatically deleted later at the end of the application.
109 // This policy is adopted since user has no control on the way the
110 // rotations are combined.
111
112 void AddPlacedVolume( G4LogicalVolume* pPlacedVolume,
113 G4Transform3D& transformation);
114 //
115 // The same as previous, but takes complete 3D transformation in space
116 // as its argument.
117
118 void AddPlacedAssembly( G4AssemblyVolume* pAssembly,
119 G4Transform3D& transformation);
120 //
121 // The same as previous AddPlacedVolume(), but takes an assembly volume
122 // as its argument.
123
124 void AddPlacedAssembly( G4AssemblyVolume* pAssembly,
125 G4ThreeVector& translation,
126 G4RotationMatrix* rotation);
127 //
128 // The same as above AddPlacedVolume(), but takes an assembly volume
129 // as its argument with translation and rotation.
130
131 void MakeImprint( G4LogicalVolume* pMotherLV,
132 G4ThreeVector& translationInMother,
133 G4RotationMatrix* pRotationInMother,
134 G4int copyNumBase = 0,
135 G4bool surfCheck = false );
136 //
137 // Creates instance of an assembly volume inside the given mother volume.
138
139 void MakeImprint( G4LogicalVolume* pMotherLV,
140 G4Transform3D& transformation,
141 G4int copyNumBase = 0,
142 G4bool surfCheck = false );
143 //
144 // The same as previous Imprint() method, but takes complete 3D
145 // transformation in space as its argument.
146
147 inline std::vector<G4VPhysicalVolume*>::iterator GetVolumesIterator();
148 inline unsigned int TotalImprintedVolumes() const;
149 //
150 // Methods to access the physical volumes imprinted with the assembly.
151
152 unsigned int GetImprintsCount() const;
153 //
154 // Return the number of made imprints.
155
156 unsigned int GetInstanceCount() const;
157 //
158 // Return the number of existing instance of G4AssemblyVolume class.
159
160 unsigned int GetAssemblyID() const;
161 //
162 // Return instance number of this concrete object.
163
164 protected:
165
166 void SetInstanceCount( unsigned int value );
167 void SetAssemblyID( unsigned int value );
168
169 void InstanceCountPlus();
170 void InstanceCountMinus();
171
172 void SetImprintsCount( unsigned int value );
173 void ImprintsCountPlus();
174 void ImprintsCountMinus();
175 //
176 // Internal counting mechanism, used to compute unique the names of
177 // physical volumes created by MakeImprint() methods.
178
179 private:
180
181 void MakeImprint( G4AssemblyVolume* pAssembly,
182 G4LogicalVolume* pMotherLV,
183 G4Transform3D& transformation,
184 G4int copyNumBase = 0,
185 G4bool surfCheck = false );
186 //
187 // Function for placement of the given assembly in the given mother
188 // (called recursively if the assembly contains an assembly).
189
190 private:
191
192 std::vector<G4AssemblyTriplet> fTriplets;
193 //
194 // Participating volumes represented as a vector of
195 // <logical volume, translation, rotation>.
196
197 std::vector<G4VPhysicalVolume*> fPVStore;
198 //
199 // We need to keep list of physical volumes created by MakeImprint() method
200 // in order to be able to cleanup the objects when not needed anymore.
201 // This requires the user to keep assembly objects in memory during the
202 // whole job or during the life-time of G4Navigator, logical volume store
203 // and physical volume store keep pointers to physical volumes generated by
204 // the assembly volume.
205 // When an assembly object is about to die it will destroy all its
206 // generated physical volumes and rotation matrices as well !
207
208 unsigned int fImprintsCounter;
209 //
210 // Number of imprints of the given assembly volume.
211
212 static G4ThreadLocal unsigned int fsInstanceCounter;
213 //
214 // Class instance counter.
215
216 unsigned int fAssemblyID;
217 //
218 // Assembly object ID derived from instance counter at construction time.
219
220};

224#endif // G4_ASSEMBLYVOLUME_H
