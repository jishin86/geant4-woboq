29// class G4PVPlacement
30//
31// Class description:
32//
33// Class representing a single volume positioned within and relative
34// to a mother volume.
35
36// History:
37// 24.07.95 P.Kent First non-stub version
38// 25.07.96 P.Kent Modified interface for new `Replica' capable geometry
39// 28.08.96 P.Kent Tidied + transform replaced by rotmat+vector
40// 28.02.97 J.Apostolakis Added 2nd constructor with G4Transform3D of solid.
41// 11.07.97 J.Apostolakis Added 3rd constructor with pMotherLogical
42// 11.05.98 J.Apostolakis Added 4th constructor with G4Transform3D & pMotherLV
43// ----------------------------------------------------------------------
44#ifndef G4PVPLACEMENT_HH
45#define G4PVPLACEMENT_HH
46
47#include "G4VPhysicalVolume.hh"
48#include "G4Transform3D.hh"
49
50class G4PVPlacement : public G4VPhysicalVolume
51{
52 public: // with description
53
54 G4PVPlacement(G4RotationMatrix *pRot,
55 const G4ThreeVector &tlate,
56 G4LogicalVolume *pCurrentLogical,
57 const G4String& pName,
58 G4LogicalVolume *pMotherLogical,
59 G4bool pMany,
60 G4int pCopyNo,
61 G4bool pSurfChk=false);
62 // Initialise a single volume, positioned in a frame which is rotated by
63 // *pRot and traslated by tlate, relative to the coordinate system of the
64 // mother volume pMotherLogical.
65 // If pRot=0 the volume is unrotated with respect to its mother.
66 // The physical volume is added to the mother's logical volume.
67 // Arguments particular to G4PVPlacement:
68 // pMany Currently NOT used. For future use to identify if the volume
69 // is meant to be considered an overlapping structure, or not.
70 // pCopyNo should be set to 0 for the first volume of a given type.
71 // pSurfChk if true activates check for overlaps with existing volumes.
72 // This is a very natural way of defining a physical volume, and is
73 // especially useful when creating subdetectors: the mother volumes are
74 // not placed until a later stage of the assembly program.
75
76 G4PVPlacement(const G4Transform3D &Transform3D,
77 G4LogicalVolume *pCurrentLogical,
78 const G4String& pName,
79 G4LogicalVolume *pMotherLogical,
80 G4bool pMany,
81 G4int pCopyNo,
82 G4bool pSurfChk=false);
83 // Additional constructor, which expects a G4Transform3D that represents
84 // the direct rotation and translation of the solid (NOT of the frame).
85 // The G4Transform3D argument should be constructed by:
86 // i) First rotating it to align the solid to the system of
87 // reference of its mother volume *pMotherLogical, and
88 // ii) Then placing the solid at the location Transform3D.getTranslation(),
89 // with respect to the origin of the system of coordinates of the
90 // mother volume.
91 // [ This is useful for the people who prefer to think in terms
92 // of moving objects in a given reference frame. ]
93 // All other arguments are the same as for the previous constructor.
94
95 public: // without description
96
97 G4PVPlacement(G4RotationMatrix *pRot,
98 const G4ThreeVector &tlate,
99 const G4String &pName,
100 G4LogicalVolume *pLogical,
101 G4VPhysicalVolume *pMother,
102 G4bool pMany,
103 G4int pCopyNo,
104 G4bool pSurfChk=false);
105 // A simple variation of the 1st constructor, only specifying the
106 // mother volume as a pointer to its physical volume instead of its
107 // logical volume. The effect is exactly the same.
108
109 G4PVPlacement(const G4Transform3D &Transform3D,
110 const G4String &pName,
111 G4LogicalVolume *pLogical,
112 G4VPhysicalVolume *pMother,
113 G4bool pMany,
114 G4int pCopyNo,
115 G4bool pSurfChk=false);
116 // Utilises both variations above (from 2nd and 3rd constructor).
117 // The effect is the same as for the 2nd constructor.
118
119 public: // with description
120
121 virtual ~G4PVPlacement();
122 // Default destructor.
123
124 inline G4int GetCopyNo() const { return fcopyNo; }
125
126 void SetCopyNo(G4int CopyNo);
127 // Gets and sets the copy number of the volume.
128
129 G4bool CheckOverlaps(G4int res=1000, G4double tol=0.,
130 G4bool verbose=true, G4int maxErr=1);
131 // Verifies if the placed volume is overlapping with existing
132 // daughters or with the mother volume. Provides default resolution
133 // for the number of points to be generated and verified.
134 // A tolerance for the precision of the overlap check can be specified,
135 // by default it is set to maximum precision.
136 // Reports a maximum of overlaps errors according to parameter in input.
137 // Returns true if the volume is overlapping.
138
139 public: // without description
140
141 G4PVPlacement(__void__&);
142 // Fake default constructor for usage restricted to direct object
143 // persistency for clients requiring preallocation of memory for
144 // persistifiable objects.
145
146 G4bool IsMany() const;
147 G4bool IsReplicated() const;
148 G4bool IsParameterised() const;
149 G4VPVParameterisation* GetParameterisation() const;
150 void GetReplicationData(EAxis& axis,
151 G4int& nReplicas,
152 G4double& width,
153 G4double& offset,
154 G4bool& consuming) const;
155 G4bool IsRegularStructure() const;
156 G4int GetRegularStructureId() const;
157
158 private:
159
160 static G4RotationMatrix* NewPtrRotMatrix(const G4RotationMatrix &RotMat);
161 // Auxiliary function for 2nd constructor (one with G4Transform3D).
162 // Creates a new RotMatrix on the heap (using "new") and copies
163 // its argument into it.
164
165 G4PVPlacement(const G4PVPlacement&);
166 G4PVPlacement& operator=(const G4PVPlacement&);
167 // Private copy constructor and assignment operator.
168
169 private:
170
171 G4bool fmany; // flag for overlapping structure - not used
172 G4bool fallocatedRotM; // flag for allocation of Rotation Matrix
173 G4int fcopyNo; // for identification
174
175};
176
177#endif
