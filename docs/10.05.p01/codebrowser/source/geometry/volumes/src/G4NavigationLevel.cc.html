32#include "G4NavigationLevel.hh"
33
34G4Allocator<G4NavigationLevel>*& aNavigationLevelAllocator()
35{
36 G4ThreadLocalStatic G4Allocator<G4NavigationLevel>* _instance = nullptr;
37 return _instance;
38}
39
40G4NavigationLevel::G4NavigationLevel( G4VPhysicalVolume* pPhysVol,
41 const G4AffineTransform& afTransform,
42 EVolume volTp,
43 G4int repNo )
44{
45 fLevelRep = new G4NavigationLevelRep( pPhysVol, afTransform, volTp, repNo );
46}
47
48G4NavigationLevel::G4NavigationLevel( G4VPhysicalVolume* pPhysVol,
49 const G4AffineTransform& levelAbove,
50 const G4AffineTransform& relativeCurrent,
51 EVolume volTp,
52 G4int repNo )
53{
54 fLevelRep = new G4NavigationLevelRep( pPhysVol,
55 levelAbove,
56 relativeCurrent,
57 volTp,
58 repNo );
59}
60
61G4NavigationLevel::G4NavigationLevel()
62{
63 fLevelRep = new G4NavigationLevelRep();
64}
65
66G4NavigationLevel::G4NavigationLevel(const G4NavigationLevel& right)
67 : fLevelRep( right.fLevelRep )
68{
69 fLevelRep->AddAReference();
70}
71
72G4NavigationLevel::~G4NavigationLevel()
73{
74 if( fLevelRep->RemoveAReference() ) { delete fLevelRep; }
75}
76
77G4NavigationLevel& G4NavigationLevel::operator=(const G4NavigationLevel &right)
78{
79 if ( &right != this )
80 {
81 right.fLevelRep->AddAReference();
82 if( fLevelRep->RemoveAReference() ) { delete fLevelRep; }
83 fLevelRep = right.fLevelRep;
84 }
85 return *this;
86}
