35G4Allocator<G4TouchableHistory>*& aTouchableHistoryAllocator()
36{
37 G4ThreadLocalStatic G4Allocator<G4TouchableHistory>* _instance = nullptr;
38 return _instance;
39}
40
41G4TouchableHistory::G4TouchableHistory()
42 : frot(G4RotationMatrix()),
43 ftlate(G4ThreeVector(0.,0.,0.)),
44 fhistory()
45{
46 G4VPhysicalVolume* pPhysVol=0;
47 fhistory.SetFirstEntry(pPhysVol);
48}
49
50G4TouchableHistory::G4TouchableHistory( const G4NavigationHistory &history )
51 : fhistory(history)
52{
53 const G4AffineTransform& tf = fhistory.GetTopTransform();
54 ftlate = tf.InverseNetTranslation();
55 frot = tf.InverseNetRotation();
56}
57
58G4TouchableHistory::~G4TouchableHistory()
59{
60}
61
62const G4ThreeVector&
63G4TouchableHistory::GetTranslation(G4int depth) const
64{
65 // The value returned will change at the next call
66 // Copy it if you want to use it!
67 //
68 static G4ThreadLocal G4ThreeVector* ctrans = 0;
69 if ( !ctrans ) { ctrans = new G4ThreeVector; }
70 if(depth==0.0)
71 {
72 return ftlate;
73 }
74 else
75 {
76 *ctrans =
77 fhistory.GetTransform(CalculateHistoryIndex(depth)).NetTranslation();
78 return *ctrans;
79 }
80}
81
82const G4RotationMatrix*
83G4TouchableHistory::GetRotation(G4int depth) const
84{
85 // The value returned will change at the next call
86 // Copy it if you want to use it!
87 //
88 static G4ThreadLocal G4RotationMatrix* rotM = 0;
89 if (!rotM ) { rotM = new G4RotationMatrix(); }
90
91 if(depth==0)
92 {
93 return &frot;
94 }
95 else
96 {
97 *rotM = fhistory.GetTransform(CalculateHistoryIndex(depth)).NetRotation();
98 return rotM;
99 }
100}
