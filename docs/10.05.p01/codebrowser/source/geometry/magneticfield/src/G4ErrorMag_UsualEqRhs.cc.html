47//---------------------------------------------------------------------
48void
49G4ErrorMag_UsualEqRhs::EvaluateRhsGivenB( const G4double y[],
50 const G4double B[3],
51 G4double dydx[] ) const
52{
53
54 G4Mag_UsualEqRhs::EvaluateRhsGivenB(y, B, dydx );
55
56 if(G4ErrorPropagatorData::GetErrorPropagatorData()->GetMode()
57 == G4ErrorMode_PropBackwards)
58 {
59 G4double momentum_mag_square = sqr(y[3]) + sqr(y[4]) + sqr(y[5]);
60 G4double inv_momentum_magnitude = 1.0 / std::sqrt( momentum_mag_square );
61
62 G4double cof = FCof()*inv_momentum_magnitude;
63
64 dydx[3] = cof*(y[4]*(-B[2]) - y[5]*(-B[1])) ;
65 dydx[4] = cof*(y[5]*(-B[0]) - y[3]*(-B[2])) ;
66 dydx[5] = cof*(y[3]*(-B[1]) - y[4]*(-B[0])) ;
67 }
68 return;
69}
