38G4UniformMagField::G4UniformMagField(const G4ThreeVector& FieldVector )
39{
40 fFieldComponents[0] = FieldVector.x();
41 fFieldComponents[1] = FieldVector.y();
42 fFieldComponents[2] = FieldVector.z();
43}
44
45G4Field* G4UniformMagField::Clone() const
46{
47 return new G4UniformMagField( G4ThreeVector(fFieldComponents[0],
48 fFieldComponents[1],
49 fFieldComponents[2]) );
50}
51
52void
53G4UniformMagField::SetFieldValue(const G4ThreeVector& newFieldVector )
54{
55 fFieldComponents[0] = newFieldVector.x();
56 fFieldComponents[1] = newFieldVector.y();
57 fFieldComponents[2] = newFieldVector.z();
58}
59
60G4UniformMagField::G4UniformMagField(G4double vField,
61 G4double vTheta,
62 G4double vPhi )
63{
64 if ( (vField<0) || (vTheta<0) || (vTheta>pi) || (vPhi<0) || (vPhi>twopi) )
65 {
66 std::ostringstream msg;
67 msg << "ERROR in G4UniformMagField::G4UniformMagField(double, double, double) : "
68 << "Invalid parameter(s). " << std::endl;
69 msg << " Expected " << std::endl;
70
71 msg << " - Magnitude vField: Value = " << vField
72 << " Expected vField > 0 " ;
73 if ( vField<0) { msg << " <------ Erroneous "; }
74 msg << std::endl;
75
76 msg << " - Theta angle: Value = " << vTheta
77 << " Expected between 0 <= theta <= pi = " << pi << " ";
78 if ( (vTheta<0) || (vTheta>pi) ) { msg << " <------ Erroneous "; }
79
80 msg << std::endl;
81 msg << " - Phi angle: Value = " << vPhi
82 << " Expected between 0 <= phi <= 2*pi = " << twopi << std::endl;
83 if ( (vPhi<0) || (vPhi>twopi) ) { msg << " <------ Erroneous "; }
84
85 G4Exception("G4UniformMagField::G4UniformMagField()",
86 "GeomField0002", FatalException, msg ); // "Invalid parameters.") ;
87 }
88 fFieldComponents[0] = vField*std::sin(vTheta)*std::cos(vPhi) ;
89 fFieldComponents[1] = vField*std::sin(vTheta)*std::sin(vPhi) ;
90 fFieldComponents[2] = vField*std::cos(vTheta) ;
91}
92
93G4UniformMagField::~G4UniformMagField()
94{
95}
96
97G4UniformMagField::G4UniformMagField (const G4UniformMagField &p)
98 : G4MagneticField(p)
99{
100 for (G4int i=0; i<3; i++)
101 {
102 fFieldComponents[i] = p.fFieldComponents[i];
103 }
104}
105
106G4UniformMagField& G4UniformMagField::operator = (const G4UniformMagField &p)
107{
108 if (&p == this) return *this;
109 G4MagneticField::operator=(p);
110 for (G4int i=0; i<3; i++)
111 {
112 fFieldComponents[i] = p.fFieldComponents[i];
113 }
114 return *this;
115}

119void G4UniformMagField::GetFieldValue (const G4double [4],
120 G4double *B ) const
121{
122 B[0]= fFieldComponents[0] ;
123 B[1]= fFieldComponents[1] ;
124 B[2]= fFieldComponents[2] ;
125}
126
127G4ThreeVector G4UniformMagField::GetConstantFieldValue() const
128{
129 G4ThreeVector B(fFieldComponents[0],
130 fFieldComponents[1],
131 fFieldComponents[2]);
132 return B;
133}
