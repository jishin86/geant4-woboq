38// Constructor sets the number of *State* variables (default = 8)
39// The number of variables integrated is always 6
40
41G4ConstRK4::G4ConstRK4(G4Mag_EqRhs* EqRhs, G4int numStateVariables)
42 : G4MagErrorStepper(EqRhs, 6, numStateVariables)
43{
44 // const G4int numberOfVariables= 6;
45 if( numStateVariables < 8 )
46 {
47 std::ostringstream message;
48 message << "The number of State variables at least 8 " << G4endl
49 << "Instead it is - numStateVariables= " << numStateVariables;
50 G4Exception("G4ConstRK4::G4ConstRK4()", "GeomField0002",
51 FatalException, message, "Use another Stepper!");
52 }
53
54 fEq = EqRhs;
55 yMiddle = new G4double[8];
56 dydxMid = new G4double[8];
57 yInitial = new G4double[8];
58 yOneStep = new G4double[8];
59
60 dydxm = new G4double[8];
61 dydxt = new G4double[8];
62 yt = new G4double[8];
63 Field[0]=0.; Field[1]=0.; Field[2]=0.;
64}

81//////////////////////////////////////////////////////////////////////
82//
83// Given values for the variables y[0,..,n-1] and their derivatives
84// dydx[0,...,n-1] known at x, use the classical 4th Runge-Kutta
85// method to advance the solution over an interval h and return the
86// incremented variables as yout[0,...,n-1], which is not a distinct
87// array from y. The user supplies the routine RightHandSide(x,y,dydx),
88// which returns derivatives dydx at x. The source is routine rk4 from
89// NRC p. 712-713 .
90
91void G4ConstRK4::DumbStepper( const G4double yIn[],
92 const G4double dydx[],
93 G4double h,
94 G4double yOut[])
95{
96 G4double hh = h*0.5 , h6 = h/6.0 ;
97
98 // 1st Step K1=h*dydx
99 yt[5] = yIn[5] + hh*dydx[5] ;
100 yt[4] = yIn[4] + hh*dydx[4] ;
101 yt[3] = yIn[3] + hh*dydx[3] ;
102 yt[2] = yIn[2] + hh*dydx[2] ;
103 yt[1] = yIn[1] + hh*dydx[1] ;
104 yt[0] = yIn[0] + hh*dydx[0] ;
105 RightHandSideConst(yt,dydxt) ;
106
107 // 2nd Step K2=h*dydxt
108 yt[5] = yIn[5] + hh*dydxt[5] ;
109 yt[4] = yIn[4] + hh*dydxt[4] ;
110 yt[3] = yIn[3] + hh*dydxt[3] ;
111 yt[2] = yIn[2] + hh*dydxt[2] ;
112 yt[1] = yIn[1] + hh*dydxt[1] ;
113 yt[0] = yIn[0] + hh*dydxt[0] ;
114 RightHandSideConst(yt,dydxm) ;
115
116 // 3rd Step K3=h*dydxm
117 // now dydxm=(K2+K3)/h
118 yt[5] = yIn[5] + h*dydxm[5] ;
119 dydxm[5] += dydxt[5] ;
120 yt[4] = yIn[4] + h*dydxm[4] ;
121 dydxm[4] += dydxt[4] ;
122 yt[3] = yIn[3] + h*dydxm[3] ;
123 dydxm[3] += dydxt[3] ;
124 yt[2] = yIn[2] + h*dydxm[2] ;
125 dydxm[2] += dydxt[2] ;
126 yt[1] = yIn[1] + h*dydxm[1] ;
127 dydxm[1] += dydxt[1] ;
128 yt[0] = yIn[0] + h*dydxm[0] ;
129 dydxm[0] += dydxt[0] ;
130 RightHandSideConst(yt,dydxt) ;
131
132 // 4th Step K4=h*dydxt
133 yOut[5] = yIn[5]+h6*(dydx[5]+dydxt[5]+2.0*dydxm[5]);
134 yOut[4] = yIn[4]+h6*(dydx[4]+dydxt[4]+2.0*dydxm[4]);
135 yOut[3] = yIn[3]+h6*(dydx[3]+dydxt[3]+2.0*dydxm[3]);
136 yOut[2] = yIn[2]+h6*(dydx[2]+dydxt[2]+2.0*dydxm[2]);
137 yOut[1] = yIn[1]+h6*(dydx[1]+dydxt[1]+2.0*dydxm[1]);
138 yOut[0] = yIn[0]+h6*(dydx[0]+dydxt[0]+2.0*dydxm[0]);
139
140} // end of DumbStepper ....................................................

142////////////////////////////////////////////////////////////////
143//
144// Stepper
145
146void
147G4ConstRK4::Stepper( const G4double yInput[],
148 const G4double dydx[],
149 G4double hstep,
150 G4double yOutput[],
151 G4double yError [] )
152{
153 const G4int nvar = 6; // number of variables integrated
154 const G4int maxvar= GetNumberOfStateVariables();
155
156 // Correction for Richardson extrapolation
157 G4double correction = 1. / ( (1 << IntegratorOrder()) -1 );
158
159 G4int i;
160
161 // Saving yInput because yInput and yOutput can be aliases for same array
162 for (i=0; i<maxvar; i++) { yInitial[i]= yInput[i]; }
163
164 // Must copy the part of the state *not* integrated to the output
165 for (i=nvar; i<maxvar; i++) { yOutput[i]= yInput[i]; }
166
167 // yInitial[7]= yInput[7]; // The time is typically needed
168 yMiddle[7] = yInput[7]; // Copy the time from initial value
169 yOneStep[7] = yInput[7]; // As it contributes to final value of yOutput ?
170 // yOutput[7] = yInput[7]; // -> dumb stepper does it too for RK4
171 yError[7] = 0.0;
172
173 G4double halfStep = hstep * 0.5;
174
175 // Do two half steps
176 //
177 GetConstField(yInitial,Field);
178 DumbStepper (yInitial, dydx, halfStep, yMiddle);
179 RightHandSideConst(yMiddle, dydxMid);
180 DumbStepper (yMiddle, dydxMid, halfStep, yOutput);
181
182 // Store midpoint, chord calculation
183 //
184 fMidPoint = G4ThreeVector( yMiddle[0], yMiddle[1], yMiddle[2]);
185
186 // Do a full Step
187 //
188 DumbStepper(yInitial, dydx, hstep, yOneStep);
189 for(i=0;i<nvar;i++)
190 {
191 yError [i] = yOutput[i] - yOneStep[i] ;
192 yOutput[i] += yError[i]*correction ;
193 // Provides accuracy increased by 1 order via the
194 // Richardson extrapolation
195 }
196
197 fInitialPoint = G4ThreeVector( yInitial[0], yInitial[1], yInitial[2]);
198 fFinalPoint = G4ThreeVector( yOutput[0], yOutput[1], yOutput[2]);
199
200 return;
201}

203////////////////////////////////////////////////////////////////
204//
205// Estimate the maximum distance from the curve to the chord
206//
207// We estimate this using the distance of the midpoint to chord.
208// The method below is good only for angle deviations < 2 pi;
209// this restriction should not be a problem for the Runge Kutta methods,
210// which generally cannot integrate accurately for large angle deviations
211
212G4double G4ConstRK4::DistChord() const
213{
214 G4double distLine, distChord;
215
216 if (fInitialPoint != fFinalPoint)
217 {
218 distLine= G4LineSection::Distline( fMidPoint, fInitialPoint, fFinalPoint );
219 // This is a class method that gives distance of Mid
220 // from the Chord between the Initial and Final points
221 distChord = distLine;
222 }
223 else
224 {
225 distChord = (fMidPoint-fInitialPoint).mag();
226 }
227 return distChord;
228}
