43void G4RKG3_Stepper::Stepper( const G4double yInput[8],
44 const G4double dydx[6],
45 G4double Step,
46 G4double yOut[8],
47 G4double yErr[])
48{
49 G4double B[3];
50 G4int nvar = 6 ;
51 G4int i;
52 G4double by15 = 1. / 15. ; // was 0.066666666 ;
53
54 G4double yTemp[8], dydxTemp[6], yIn[8] ;
55 // Saving yInput because yInput and yOut can be aliases for same array
56 for(i=0;i<nvar;i++) yIn[i]=yInput[i];
57 yIn[6] = yInput[6];
58 yIn[7] = yInput[7];
59 G4double h = Step * 0.5;
60 hStep=Step;
61 // Do two half steps
62
63 StepNoErr(yIn, dydx,h, yTemp,B) ;
64
65 //Store Bfld for DistChord Calculation
66 for(i=0;i<3;i++)BfldIn[i]=B[i];
67
68 // RightHandSide(yTemp,dydxTemp) ;
69
70 GetEquationOfMotion()->EvaluateRhsGivenB(yTemp,B,dydxTemp) ;
71 StepNoErr(yTemp,dydxTemp,h,yOut,B);
72
73 // Store midpoint, chord calculation
74
75 fyMidPoint = G4ThreeVector( yTemp[0], yTemp[1], yTemp[2]);
76
77 // Do a full Step
78
79 h *= 2 ;
80 StepNoErr(yIn,dydx,h,yTemp,B);
81 for(i=0;i<nvar;i++)
82 {
83 yErr[i] = yOut[i] - yTemp[i] ;
84 yOut[i] += yErr[i]*by15 ; // Provides 5th order of accuracy
85 }
86
87 //Store values for DistChord method
88
89 fyInitial = G4ThreeVector( yIn[0], yIn[1], yIn[2]);
90 fpInitial = G4ThreeVector( yIn[3], yIn[4], yIn[5]);
91 fyFinal = G4ThreeVector( yOut[0], yOut[1], yOut[2]);
92
93 // NormaliseTangentVector( yOut ); // Deleted
94}

120// Integrator RK Stepper from G3 with only two field evaluation per Step.
121// It is used in propagation initial Step by small substeps after solution
122// error and delta geometry considerations. B[3] is magnetic field which
123// is passed from substep to substep.
124
125void G4RKG3_Stepper::StepNoErr(const G4double tIn[8],
126 const G4double dydx[6],
127 G4double Step,
128 G4double tOut[8],
129 G4double B[3] ) // const
130
131{
132
133 // Copy and edit the routine above, to delete alpha2, beta2, ...
134 G4double K1[7],K2[7],K3[7],K4[7] ;
135 G4double tTemp[8], yderiv[6] ;
136
137 // Need Momentum value to give correct values to the coefficients in equation
138 // Integration on unit velocity, but tIn[3,4,5] is momentum
139 G4double mom,inverse_mom;
140 G4int i ;
141 const G4double c1=0.5,c2=0.125,c3=1./6.;
142
143 // GetEquationOfMotion()->EvaluateRhsReturnB(tIn,dydx,B1) ;
144 // Correction for momentum not a velocity
145 // Need the protection !!! must be not zero
146 mom=std::sqrt(tIn[3]*tIn[3]+tIn[4]*tIn[4]+tIn[5]*tIn[5]);
147 inverse_mom=1./mom;
148 for(i=0;i<3;i++)
149 {
150 K1[i] = Step * dydx[i+3]*inverse_mom;
151 tTemp[i] = tIn[i] + Step*(c1*tIn[i+3]*inverse_mom + c2*K1[i]) ;
152 tTemp[i+3] = tIn[i+3] + c1*K1[i]*mom ;
153
154 }
155
156 GetEquationOfMotion()->EvaluateRhsReturnB(tTemp,yderiv,B) ;
157
158
159 for(i=0;i<3;i++)
160 {
161 K2[i] = Step * yderiv[i+3]*inverse_mom;
162 tTemp[i+3] = tIn[i+3] + c1*K2[i]*mom ;
163 }
164
165 // Given B, calculate yderiv !
166 GetEquationOfMotion()->EvaluateRhsGivenB(tTemp,B,yderiv) ;
167
168 for(i=0;i<3;i++)
169 {
170 K3[i] = Step * yderiv[i+3]*inverse_mom;
171 tTemp[i] = tIn[i] + Step*(tIn[i+3]*inverse_mom + c1*K3[i]) ;
172 tTemp[i+3] = tIn[i+3] + K3[i]*mom ;
173 }
174
175
176 // Calculates y-deriv(atives) & returns B too!
177 GetEquationOfMotion()->EvaluateRhsReturnB(tTemp,yderiv,B) ;
178
179
180 for(i=0;i<3;i++) // Output trajectory vector
181 {
182 K4[i] = Step * yderiv[i+3]*inverse_mom;
183 tOut[i] = tIn[i] + Step*(tIn[i+3]*inverse_mom+ (K1[i] + K2[i] + K3[i])*c3) ;
184 tOut[i+3] = tIn[i+3] + mom*(K1[i] + 2*K2[i] + 2*K3[i] +K4[i])*c3 ;
185 }
186 tOut[6] = tIn[6];
187 tOut[7] = tIn[7];
188 // NormaliseTangentVector( tOut );
189
190
191}

196 G4double G4RKG3_Stepper::DistChord() const
197 {
198 // Soon: must check whether h/R > 2 pi !!
199 // Method below is good only for < 2 pi
200 G4double distChord,distLine;
201
202 if (fyInitial != fyFinal) {
203 distLine= G4LineSection::Distline(fyMidPoint,fyInitial,fyFinal );
204
205 distChord = distLine;
206 }else{
207 distChord = (fyMidPoint-fyInitial).mag();
208 }
209
210
211 return distChord;
212
213 }
