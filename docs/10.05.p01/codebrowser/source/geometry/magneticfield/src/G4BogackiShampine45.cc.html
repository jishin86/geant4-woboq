40// This is the source file of G4BogackiShampine45 class containing the
41// definition of the stepper() method that evaluates one step in
42// field propagation.
43//
44// The Butcher table of the Bogacki-Shampine-8-4-5 method is:
45//
46// 0 |
47// 1/6 | 1/6
48// 2/9 | 2/27 4/27
49// 3/7 | 183/1372 -162/343 1053/1372
50// 2/3 | 68/297 -4/11 42/143 1960/3861
51// 3/4 | 597/22528 81/352 63099/585728 58653/366080 4617/20480
52// 1 | 174197/959244 -30942/79937 8152137/19744439 666106/1039181 -29421/29068 482048/414219
53// 1 | 587/8064 0 4440339/15491840 24353/124800 387/44800 2152/5985 7267/94080
54//-------------------------------------------------------------------------------------------------------------------
55// 587/8064 0 4440339/15491840 24353/124800 387/44800 2152/5985 7267/94080 0
56// 2479/34992 0 123/416 612941/3411720 43/1440 2272/6561 79937/1113912 3293/556956
57//
58// Do NOT re-indent the lines above - their meaning becomes lost
59// ********************************
60// Coefficients have been obtained from rksuite.f : http://www.netlib.org/ode/rksuite/
61
62// Note on meaning of label "non-FSAL version":
63// This method calculates the deriviative dy/dx at the endpoint of the integration interval at each step.
64// as part of its evaluation of the endpoint and its error.
65// So this value is available to be returned, for re-use in case of a successful step.
66// ( This is done in a 'later' version using a refined interface. )

76//Constructor
77G4BogackiShampine45::G4BogackiShampine45(G4EquationOfMotion *EqRhs,
78 G4int noIntegrationVariables,
79 G4bool primary)
80 : G4MagIntegratorStepper(EqRhs, noIntegrationVariables),
81 fLastStepLength(-1.0),
82 fAuxStepper(nullptr),
83 fPreparedInterpolation(false)
84{
85
86 const G4int numberOfVariables = noIntegrationVariables;
87
88 //New Chunk of memory being created for use by the stepper
89
90 //aki - for storing intermediate RHS
91 ak2 = new G4double[numberOfVariables];
92 ak3 = new G4double[numberOfVariables];
93 ak4 = new G4double[numberOfVariables];
94 ak5 = new G4double[numberOfVariables];
95 ak6 = new G4double[numberOfVariables];
96 ak7 = new G4double[numberOfVariables];
97 ak8 = new G4double[numberOfVariables];
98 ak9 = new G4double[numberOfVariables];
99 ak10 = new G4double[numberOfVariables];
100 ak11 = new G4double[numberOfVariables];
101
102 for (int i = 0; i < 6; i++) {
103 p[i]= new G4double[numberOfVariables];
104 }
105
106 assert ( GetNumberOfStateVariables() >= 8 );
107 const G4int numStateVars = std::max(noIntegrationVariables,
108 GetNumberOfStateVariables() );
109
110 // Must ensure space extra 'state' variables exists - i.e. yIn[7]
111 yTemp = new G4double[numStateVars];
112 yIn = new G4double[numStateVars] ;
113
114 fLastInitialVector = new G4double[numStateVars] ;
115 fLastFinalVector = new G4double[numStateVars] ;
116 fLastDyDx = new G4double[numberOfVariables]; // Only derivatives
117
118 fMidVector = new G4double[numberOfVariables]; // new G4double[numStateVars];
119 fMidError = new G4double[numberOfVariables]; // new G4double[numStateVars];
120
121 if( ! fPreparedConstants )
122 PrepareConstants();
123
124 if( primary )
125 {
126 fAuxStepper = new G4BogackiShampine45(EqRhs, numberOfVariables, false);
127 }
128}

175//Stepper :
176
177// Passing in the value of yInput[],the first time dydx[] and Step length
178// Giving back yOut and yErr arrays for output and error respectively
179
180void G4BogackiShampine45::Stepper( const G4double yInput[],
181 const G4double DyDx[],
182 G4double Step,
183 G4double yOut[],
184 G4double yErr[] )
185{
186 G4int i;
187
188 // Constants from the Butcher tableu
189 const G4double
190 b21 = 1.0/6.0 ,
191 b31 = 2.0/27.0 , b32 = 4.0/27.0,
192
193 b41 = 183.0/1372.0 , b42 = -162.0/343.0, b43 = 1053.0/1372.0,
194
195 b51 = 68.0/297.0, b52 = -4.0/11.0,
196 b53 = 42.0/143.0, b54 = 1960.0/3861.0,
197
198 b61 = 597.0/22528.0, b62 = 81.0/352.0,
199 b63 = 63099.0/585728.0, b64 = 58653.0/366080.0,
200 b65 = 4617.0/20480.0,
201
202 b71 = 174197.0/959244.0, b72 = -30942.0/79937.0,
203 b73 = 8152137.0/19744439.0, b74 = 666106.0/1039181.0,
204 b75 = -29421.0/29068.0, b76 = 482048.0/414219.0,
205
206 b81 = 587.0/8064.0, b82 = 0.0,
207 b83 = 4440339.0/15491840.0, b84 = 24353.0/124800.0,
208 b85 = 387.0/44800.0, b86 = 2152.0/5985.0,
209 b87 = 7267.0/94080.0;
210
211// c1 = 2479.0/34992.0,
212// c2 = 0.0,
213// c3 = 123.0/416.0,
214// c4 = 612941.0/3411720.0,
215// c5 = 43.0/1440.0,
216// c6 = 2272.0/6561.0,
217// c7 = 79937.0/1113912.0,
218// c8 = 3293.0/556956.0,
219
220 // For the embedded higher order method only the difference of values
221 // taken and is used directly later (instead of defining the last row
222 // of Butcher table in separate constants and taking the
223 // difference)
224 const G4double
225 dc1 = b81 - 2479.0 / 34992.0 ,
226 dc2 = 0.0,
227 dc3 = b83 - 123.0 / 416.0 ,
228 dc4 = b84 - 612941.0 / 3411720.0,
229 dc5 = b85 - 43.0 / 1440.0,
230 dc6 = b86 - 2272.0 / 6561.0,
231 dc7 = b87 - 79937.0 / 1113912.0,
232 dc8 = - 3293.0 / 556956.0;
233
234 const G4int numberOfVariables= this->GetNumberOfVariables();
235
236 // The number of variables to be integrated over
237 yOut[7] = yTemp[7] = yIn[7] = yInput[7];
238 // Saving yInput because yInput and yOut can be aliases for same array
239
240 for(i=0;i<numberOfVariables;i++)
241 {
242 yIn[i]=yInput[i];
243 }
244
245 // RightHandSide(yIn, dydx) ;
246 // 1st Step - Not doing, getting passed
247
248 for(i=0;i<numberOfVariables;i++)
249 {
250 yTemp[i] = yIn[i] + b21*Step*DyDx[i] ;
251 }
252 RightHandSide(yTemp, ak2) ; // 2nd Step
253
254 for(i=0;i<numberOfVariables;i++)
255 {
256 yTemp[i] = yIn[i] + Step*(b31*DyDx[i] + b32*ak2[i]) ;
257 }
258 RightHandSide(yTemp, ak3) ; // 3rd Step
259
260 for(i=0;i<numberOfVariables;i++)
261 {
262 yTemp[i] = yIn[i] + Step*(b41*DyDx[i] + b42*ak2[i] + b43*ak3[i]) ;
263 }
264 RightHandSide(yTemp, ak4) ; // 4th Step
265
266 for(i=0;i<numberOfVariables;i++)
267 {
268 yTemp[i] = yIn[i] + Step*(b51*DyDx[i] + b52*ak2[i] + b53*ak3[i] +
269 b54*ak4[i]) ;
270 }
271 RightHandSide(yTemp, ak5) ; // 5th Step
272
273 for(i=0;i<numberOfVariables;i++)
274 {
275 yTemp[i] = yIn[i] + Step*(b61*DyDx[i] + b62*ak2[i] + b63*ak3[i] +
276 b64*ak4[i] + b65*ak5[i]) ;
277 }
278 RightHandSide(yTemp, ak6) ; // 6th Step
279
280 for(i=0;i<numberOfVariables;i++)
281 {
282 yTemp[i] = yIn[i] + Step*(b71*DyDx[i] + b72*ak2[i] + b73*ak3[i] +
283 b74*ak4[i] + b75*ak5[i] + b76*ak6[i]);
284 }
285 RightHandSide(yTemp, ak7); //7th Step
286
287 for(i=0;i<numberOfVariables;i++)
288 {
289 yOut[i] = yIn[i] + Step*(b81*DyDx[i] + b82*ak2[i] + b83*ak3[i] +
290 b84*ak4[i] + b85*ak5[i] + b86*ak6[i] +
291 b87*ak7[i]);
292 }
293 RightHandSide(yOut, ak8); //8th Step - Final one Using FSAL
294
295 for(i=0;i<numberOfVariables;i++)
296 {
297 yErr[i] = Step*(dc1*DyDx[i] + dc2*ak2[i] + dc3*ak3[i] + dc4*ak4[i] +
298 dc5*ak5[i] + dc6*ak6[i] + dc7*ak7[i] + dc8*ak8[i]) ;
299
300 // Store Input and Final values, for possible use in calculating chord
301 fLastInitialVector[i] = yIn[i] ;
302 fLastFinalVector[i] = yOut[i];
303 fLastDyDx[i] = DyDx[i];
304 }
305
306 fLastStepLength = Step;
307 fPreparedInterpolation= false;
308
309 return ;
310}

316G4double G4BogackiShampine45::DistChord() const
317{
318 G4double distLine, distChord;
319 G4ThreeVector initialPoint, finalPoint, midPoint;
320
321 // Store last initial and final points (they will be overwritten in self-Stepper call!)
322 initialPoint = G4ThreeVector( fLastInitialVector[0],
323 fLastInitialVector[1], fLastInitialVector[2]);
324 finalPoint = G4ThreeVector( fLastFinalVector[0],
325 fLastFinalVector[1], fLastFinalVector[2]);
326
327#if 1
328 // Old method -- Do half a step using StepNoErr
329 fAuxStepper->Stepper( fLastInitialVector, fLastDyDx, 0.5 * fLastStepLength,
330 fMidVector, fMidError);
331#else
332 // New method -- Using interpolation, requires only 3 extra stages (ie 3 extra field evaluations )
333
334 // Use Interpolation, instead of auxiliary stepper to evaluate midpoint
335 if( ! fPreparedInterpolation ) {
336 G4BogackiShampine45 *cThis= const_cast<G4BogackiShampine45 *>(this);
337 cThis-> SetupInterpolationHigh(); // ( fLastInitialVector, fLastDyDx, fLastStepLength );
338 }
339 //For calculating the output at the tau fraction of Step
340 G4double tau = 0.5;
341 // cThis->InterpolateHigh( /* fLastInitialVector, fLastDyDx, fLastStepLength, */, fMidVector, tau);
342 // Old arguments: ( /*yInput, dydx, step,*/ yOut, tau );
343 this->InterpolateHigh( tau, fMidVector );
344#endif
345
346 midPoint = G4ThreeVector( fMidVector[0], fMidVector[1], fMidVector[2]);
347
348 // Use stored values of Initial and Endpoint + new Midpoint to evaluate
349 // distance of Chord
350
351 if (initialPoint != finalPoint)
352 {
353 distLine = G4LineSection::Distline( midPoint, initialPoint, finalPoint );
354 distChord = distLine;
355 }
356 else
357 {
358 distChord = (midPoint-initialPoint).mag();
359 }
360 return distChord;
361}

363void G4BogackiShampine45::SetupInterpolationHigh()
364 // ( const G4double *yInput, const G4double *dydx, const G4double Step)
365{
366 //Coefficients for the additional stages :
367 const G4double
368 a91 = 455.0/6144.0 ,
369 a92 = 0.0 ,
370 a93 = 10256301.0/35409920.0 ,
371 a94 = 2307361.0/17971200.0 ,
372 a95 = -387.0/102400.0 ,
373 a96 = 73.0/5130.0 ,
374 a97 = -7267.0/215040.0 ,
375 a98 = 1.0/32.0 ,
376
377 a101 = -837888343715.0/13176988637184.0 ,
378 a102 = 30409415.0/52955362.0 ,
379 a103 = -48321525963.0/759168069632.0 ,
380 a104 = 8530738453321.0/197654829557760.0 ,
381 a105 = 1361640523001.0/1626788720640.0 ,
382 a106 = -13143060689.0/38604458898.0 ,
383 a107 = 18700221969.0/379584034816.0 ,
384 a108 = -5831595.0/847285792.0 ,
385 a109 = -5183640.0/26477681.0 ,
386
387 a111 = 98719073263.0/1551965184000.0 ,
388 a112 = 1307.0/123552.0 ,
389 a113 = 4632066559387.0/70181753241600.0 ,
390 a114 = 7828594302389.0/382182512025600.0 ,
391 a115 = 40763687.0/11070259200.0 ,
392 a116 = 34872732407.0/224610586200.0 ,
393 a117 = -2561897.0/30105600.0 ,
394 a118 = 1.0/10.0 ,
395 a119 = -1.0/10.0 ,
396 a1110 = -1403317093.0/11371610250.0 ;
397
398 const G4int numberOfVariables= this->GetNumberOfVariables();
399
400 // const G4double *yIn= fLastInitialVector;
401 const G4double *dydx= fLastDyDx;
402 const G4double Step = fLastStepLength;
403
404 yTemp[7] = yIn[7];
405
406 //Evaluate the extra stages :
407 for(int i=0; i<numberOfVariables; i++){
408 yTemp[i] = yIn[i] + Step*(a91*dydx[i] + a92*ak2[i] + a93*ak3[i] +
409 a94*ak4[i] + a95*ak5[i] + a96*ak6[i] +
410 a97*ak7[i] + a98*ak8[i] );
411 }
412
413 RightHandSide(yTemp, ak9); //9th stage
414
415 for(int i=0; i<numberOfVariables; i++){
416 yTemp[i] = yIn[i] + Step*(a101*dydx[i] + a102*ak2[i] + a103*ak3[i] +
417 a104*ak4[i] + a105*ak5[i] + a106*ak6[i] +
418 a107*ak7[i] + a108*ak8[i] + a109*ak9[i] );
419 }
420
421 RightHandSide(yTemp, ak10); //10th stage
422
423 for(int i=0; i<numberOfVariables; i++){
424 yTemp[i] = yIn[i] + Step*(a111*dydx[i] + a112*ak2[i] + a113*ak3[i] +
425 a114*ak4[i] + a115*ak5[i] + a116*ak6[i] +
426 a117*ak7[i] + a118*ak8[i] + a119*ak9[i] +
427 a1110*ak10[i] );
428 }
429 RightHandSide(yTemp, ak11); //11th stage
430
431 // In future we can restrict the number of variables interpolated
432 int nwant = numberOfVariables;
433
434// Form the coefficients of the interpolating polynomial in its shifted
435// and scaled form. The terms are grouped to minimize the errors
436// of the transformation, to cope with ill-conditioning. ( From RKSUITE )
437//
438 for (int l = 0; l < nwant; l++) {
439 // Coefficient of tau^6
440 p[5][l] = bi[5][6]*ak5[l] +
441 ((bi[10][6]*ak10[l] + bi[8][6]*ak8[l]) +
442 (bi[7][6]*ak7[l] + bi[6][6]*ak6[l])) +
443 ((bi[4][6]*ak4[l] + bi[9][6]*ak9[l]) +
444 (bi[3][6]*ak3[l] + bi[11][6]*ak11[l]) +
445 bi[1][6]*dydx[l]);
446 // Coefficient of tau^5
447 p[4][l] = (bi[10][5]*ak10[l] + bi[9][5]*ak9[l]) +
448 ((bi[7][5]*ak7[l] + bi[6][5]*ak6[l]) +
449 bi[5][5]*ak5[l]) + ((bi[4][5]*ak4[l] +
450 bi[8][5]*ak8[l]) + (bi[3][5]*ak3[l] +
451 bi[11][5]*ak11[l]) + bi[1][5]*dydx[l]);
452 // Coefficient of tau^4
453 p[3][l] = ((bi[4][4]*ak4[l] + bi[8][4]*ak8[l]) +
454 (bi[7][4]*ak7[l] + bi[6][4]*ak6[l]) +
455 bi[5][4]*ak5[l]) + ((bi[10][4]*ak10[l] +
456 bi[9][4]*ak9[l]) + (bi[3][4]*ak3[l] +
457 bi[11][4]*ak11[l]) + bi[1][4]*dydx[l]);
458 // Coefficient of tau^3
459 p[2][l] = bi[5][3]*ak5[l] + bi[6][3]*ak6[l] +
460 ((bi[3][3]*ak3[l] + bi[9][3]*ak9[l]) +
461 (bi[10][3]*ak10[l]+ bi[8][3]*ak8[l]) + bi[1][3]*dydx[l]) +
462 ((bi[4][3]*ak4[l] + bi[11][3]*ak11[l]) + bi[7][3]*ak7[l]);
463 // Coefficient of tau^2
464 p[1][l] = bi[5][2]*ak5[l] + ((bi[6][2]*ak6[l] +
465 bi[8][2]*ak8[l]) + bi[1][2]*dydx[l]) +
466 ((bi[3][2]*ak3[l] + bi[9][2]*ak9[l]) +
467 bi[10][2]*ak10[l])+ ((bi[4][2]*ak4[l] +
468 bi[11][2]*ak2[l]) + bi[7][2]*ak7[l]);
469 }
470//
471// Scale all the coefficients by the step size.
472//
473 for (int i = 0; i < 6; i++) {
474 for (int l = 0; l < nwant; l++) {
475 p[i][l] *= Step;
476 }
477 }
478
479 fPreparedInterpolation= true;
480}
