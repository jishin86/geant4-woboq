40G4double absoluteError(const G4double y[],
41 const G4double yError[],
42 G4double hstep)
43{
44 const G4double momentum2 = getValue2(y, Value3D::Momentum);
45 const G4double invMomentum2 = 1.0 / momentum2;
46 const G4double positionError2 = getValue2(yError, Value3D::Position);
47 const G4double momentumError2 = getValue2(yError, Value3D::Momentum);
48 const G4double relativeMomentumError2 = momentumError2 * invMomentum2;
49
50 return std::max(std::sqrt(positionError2), std::sqrt(relativeMomentumError2) * hstep);
51}
52
53G4double relativeError2(const G4double y[],
54 const G4double yerr[],
55 G4double h,
56 G4double eps_rel_max)
57{
58 G4double errmax_sq;
59
60 G4double inv_eps_vel_sq = 1.0 / (eps_rel_max * eps_rel_max);
61 G4double errvel_sq = 0.0; // square of momentum vector difference
62 G4double errspin_sq = 0.0; // square of spin vector difference
63
64 G4double spin_mag2 = getValue2(y, Value3D::Spin);
65 G4bool hasSpin = (spin_mag2 > 0.0);
66
67 G4double eps_pos = eps_rel_max * h;
68 G4double inv_eps_pos_sq = 1.0 / (eps_pos * eps_pos);
69
70 // Evaluate accuracy
71 G4double errpos_sq = getValue2(yerr, Value3D::Position);
72 errpos_sq *= inv_eps_pos_sq; // Scale relative to required tolerance
73
74 // Accuracy for momentum
75 G4double magvel_sq = getValue2(y, Value3D::Momentum);
76 G4double sumerr_sq = getValue2(yerr, Value3D::Momentum);
77 if (magvel_sq > 0.0)
78 {
79 errvel_sq = sumerr_sq / magvel_sq;
80 }
81 else
82 {
83 G4Exception("field_utils::relativeError","Field001",
84 JustWarning, "found case of zero momentum");
85 errvel_sq = sumerr_sq;
86 }
87 errvel_sq *= inv_eps_vel_sq;
88 errmax_sq = std::max(errpos_sq, errvel_sq);
89
90 if (hasSpin)
91 {
92 // Accuracy for spin
93 errspin_sq = getValue2(yerr, Value3D::Spin) / spin_mag2;
94 errspin_sq *= inv_eps_vel_sq;
95 errmax_sq = std::max( errmax_sq, errspin_sq );
96 }
97
98 return errmax_sq;
99}
100
101G4double relativeError(
102 const G4double y[],
103 const G4double yError[],
104 const G4double h,
105 const G4double errorTolerance)
106{
107 return std::sqrt(relativeError2(y, yError, h, errorTolerance));
108}

110void copy(G4double dst[], const G4double src[], size_t size)
111{
112 memcpy(dst, src, sizeof(G4double) * size);
113}

116G4double inverseCurvatureRadius(G4double particleCharge, G4double momentum, G4double BField)
117{
118 return -c_light * particleCharge * BField / momentum;
119}
