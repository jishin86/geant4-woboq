26// class G4HelixMixedStepper

28// Class description:
29//
30// G4HelixMixedStepper split the Method used for Integration in two:
31//
32// If Stepping Angle ( h / R_curve) < pi/3
33// use Stepper for small step(ClassicalRK4 by default)
34// Else use HelixExplicitEuler Stepper
35//
36// History:
37// Derived from ExactHelicalStepper 18/05/07

64G4HelixMixedStepper::
65G4HelixMixedStepper(G4Mag_EqRhs *EqRhs,
66 G4int stepperNumber,
67 G4double angleThreshold)
68 : G4MagHelicalStepper(EqRhs), fNumCallsRK4(0), fNumCallsHelix(0)
69{
70 SetVerbose(1);
71 if( angleThreshold < 0.0 ){
72 fAngle_threshold= (1.0/3.0)*pi;
73 }else{
74 fAngle_threshold= angleThreshold;
75 }
76
77 if(stepperNumber<0)
78 stepperNumber=4; // Default is RK4 (original)
79 // stepperNumber=745; // Default is DormandPrince745 (ie DoPri5)
80 // stepperNumber=8; // Default is CashKarp
81
82 fStepperNumber = stepperNumber; // Store the choice
83 fRK4Stepper = SetupStepper(EqRhs, fStepperNumber);
84}

90}

92void G4HelixMixedStepper::Stepper( const G4double yInput[7],
93 const G4double dydx[7],
94 G4double Step,
95 G4double yOut[7],
96 G4double yErr[])
97{
98 //Estimation of the Stepping Angle
99 G4ThreeVector Bfld;
100 MagFieldEvaluate(yInput, Bfld);
101
102 G4double Bmag = Bfld.mag();
103 const G4double *pIn = yInput+3;
104 G4ThreeVector initVelocity= G4ThreeVector( pIn[0], pIn[1], pIn[2]);
105 G4double velocityVal = initVelocity.mag();
106
107 const G4double R_1=std::abs(GetInverseCurve(velocityVal,Bmag)); // curv= inverse Radius
108 G4double Ang_curve= R_1 * Step;
109 // SetAngCurve(Ang_curve);
110 // SetCurve(std::abs(1/R_1)); // Move below, to avoid un-needed division if RK used
111
112 if(Ang_curve< fAngle_threshold)
113 {
114 fNumCallsRK4++;
115 fRK4Stepper->Stepper(yInput,dydx,Step,yOut,yErr);
116 }
117 else
118 {
119 constexpr G4int nvar = 6 ;
120 constexpr G4int nvarMax = 8 ;
121 G4double yTemp[nvarMax], yIn[nvarMax], yTemp2[nvarMax];
122 G4ThreeVector Bfld_midpoint;
123
124 SetAngCurve(Ang_curve);
125 SetCurve(std::abs(1.0/R_1));
126 fNumCallsHelix++;
127
128 // Saving yInput because yInput and yOut can be aliases for same array
129 for(G4int i=0;i<nvar;i++) yIn[i]=yInput[i];
130
131 G4double halfS = Step * 0.5;
132 // 1. Do first half step and full step
133 AdvanceHelix(yIn, Bfld, halfS, yTemp, yTemp2); // yTemp2 for s=2*h (halfS)
134 //**********
135
136 MagFieldEvaluate(yTemp, Bfld_midpoint) ;
137
138 // 2. Do second half step - with revised field
139 // NOTE: Could avoid this call if 'Bfld_midpoint == Bfld'
140 // or diff 'almost' zero
141 AdvanceHelix(yTemp, Bfld_midpoint, halfS, yOut);
142 // Not requesting y at s=2*h (halfS)
143 //**********
144
145 // 3. Estimate the integration error
146 // should be (nearly) zero if Bfield= constant
147 for(G4int i=0;i<nvar;i++) {
148 yErr[i] = yOut[i] - yTemp2[i] ;
149 }
150 }
151}

162G4double G4HelixMixedStepper::DistChord() const
163{
164 // Implementation : must check whether h/R > 2 pi !!
165 // If( h/R < pi) use G4LineSection::DistLine
166 // Else DistChord=R_helix
167 //
168 G4double distChord;
169 G4double Ang_curve=GetAngCurve();
170
171 if(Ang_curve<=pi){
172 distChord=GetRadHelix()*(1-std::cos(0.5*Ang_curve));
173 }
174 else
175 {
176 if(Ang_curve<twopi){
177 distChord=GetRadHelix()*(1+std::cos(0.5*(twopi-Ang_curve)));
178 }
179 else{
180 distChord=2.*GetRadHelix();
181 }
182 }
183
184 return distChord;
185}

195G4MagIntegratorStepper*
196G4HelixMixedStepper::SetupStepper(G4Mag_EqRhs* pE, G4int StepperNumber)
197{
198 G4MagIntegratorStepper* pStepper;
199 if (fVerbose>0) G4cout << " G4HelixMixedStepper: ";
200 switch ( StepperNumber )
201 {
202 // Robust, classic method
203 case 4:
204 pStepper = new G4ClassicalRK4( pE );
205 if (fVerbose>0) G4cout << "G4ClassicalRK4";
206 break;
207
208 // Steppers with embedded estimation of error
209 case 8:
210 pStepper = new G4CashKarpRKF45( pE );
211 if (fVerbose>0) G4cout << "G4CashKarpRKF45";
212 break;
213 case 13:
214 pStepper = new G4NystromRK4( pE );
215 if (fVerbose>0) G4cout << "G4NystromRK4";
216 break;
217
218 // Lowest order RK Stepper - experimental
219 case 1:
220 pStepper = new G4ImplicitEuler( pE );
221 if (fVerbose>0) G4cout << "G4ImplicitEuler";
222 break;
223
224 // Lower order RK Steppers - ok overall, good for uneven fields
225 case 2:
226 pStepper = new G4SimpleRunge( pE );
227 if (fVerbose>0) G4cout << "G4SimpleRunge";
228 break;
229 case 3:
230 pStepper = new G4SimpleHeum( pE );
231 if (fVerbose>0) G4cout << "G4SimpleHeum";
232 break;
233 case 23:
234 pStepper = new G4BogackiShampine23( pE );
235 if (fVerbose>0) G4cout << "G4BogackiShampine23";
236 break;
237
238 // Higher order RK Steppers
239 // for smoother fields and high accuracy requirements
240 case 45:
241 pStepper = new G4BogackiShampine45( pE );
242 if (fVerbose>0) G4cout << "G4BogackiShampine45";
243 break;
244 case 145:
245 pStepper = new G4TsitourasRK45( pE );
246 if (fVerbose>0) G4cout << "G4TsitourasRK45";
247 break;
248 case 745:
249 pStepper = new G4DormandPrince745( pE );
250 if (fVerbose>0) G4cout << "G4DormandPrince745";
251 break;
252
253 // Helical Steppers
254 case 6:
255 pStepper = new G4HelixImplicitEuler( pE );
256 if (fVerbose>0) G4cout << "G4HelixImplicitEuler";
257 break;
258 case 7:
259 pStepper = new G4HelixSimpleRunge( pE );
260 if (fVerbose>0) G4cout << "G4HelixSimpleRunge";
261 break;
262 case 5:
263 pStepper = new G4HelixExplicitEuler( pE );
264 if (fVerbose>0) G4cout << "G4HelixExplicitEuler";
265 break; // Since Helix Explicit is used for long steps,
266 // this is useful only to measure overhead.
267 // Exact Helix - likely good only for cases of
268 // i) uniform field (potentially over small distances)
269 // ii) segmented uniform field (maybe)
270 case 9:
271 pStepper = new G4ExactHelixStepper( pE );
272 if (fVerbose>0) G4cout << "G4ExactHelixStepper";
273 break;
274 case 10:
275 pStepper = new G4RKG3_Stepper( pE );
276 if (fVerbose>0) G4cout << "G4RKG3_Stepper";
277 break;
278
279 // Low Order Steppers - not good except for very weak fields
280 case 11:
281 pStepper = new G4ExplicitEuler( pE );
282 if (fVerbose>0) G4cout << "G4ExplicitEuler";
283 break;
284 case 12:
285 pStepper = new G4ImplicitEuler( pE );
286 if (fVerbose>0) G4cout << "G4ImplicitEuler";
287 break;
288
289 case 0:
290 case -1:
291 default:
292 pStepper = new G4DormandPrince745( pE ); // Was G4ClassicalRK4( pE );
293 if (fVerbose>0) G4cout << "G4DormandPrince745 (Default)";
294 break;
295 }
296 if(fVerbose>0)
297 G4cout << " chosen as stepper for small steps in G4HelixMixedStepper."
298 << G4endl;
299
300 return pStepper;
301}
