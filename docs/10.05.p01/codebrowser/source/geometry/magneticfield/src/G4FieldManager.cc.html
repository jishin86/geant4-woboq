40G4FieldManager::G4FieldManager(G4Field *detectorField,
41 G4ChordFinder *pChordFinder,
42 G4bool fieldChangesEnergy
43 )
44 : fDetectorField(detectorField),
45 fChordFinder(pChordFinder),
46 fAllocatedChordFinder(false),
47 fDelta_One_Step_Value( fDefault_Delta_One_Step_Value ),
48 fDelta_Intersection_Val( fDefault_Delta_Intersection_Val ),
49 fEpsilonMin( fEpsilonMinDefault ),
50 fEpsilonMax( fEpsilonMaxDefault)
51{
52 if ( detectorField )
53 fFieldChangesEnergy= detectorField->DoesFieldChangeEnergy();
54 else
55 fFieldChangesEnergy= fieldChangesEnergy;
56
57 // Add to store
58 G4FieldManagerStore::Register(this);
59}

74G4FieldManager* G4FieldManager::Clone() const
75{
76 G4Field* aField = 0;
77 G4FieldManager* aFM = 0;
78 G4ChordFinder* aCF = 0;
79 try {
80 if ( this->fDetectorField )
81 aField = this->fDetectorField->Clone();
82
83 //Create a new field manager, note that we do not set any chordfinder now.
84 aFM = new G4FieldManager( aField , 0 , this->fFieldChangesEnergy );
85
86 //Check if orignally we have the fAllocatedChordFinder variable set, in case, call chord
87 //constructor
88 if ( this->fAllocatedChordFinder )
89 {
90 aFM->CreateChordFinder( dynamic_cast<G4MagneticField*>(aField) );
91 }
92 else
93 {
94 //Chord was specified by user, should we clone?
95 //TODO: For the moment copy pointer, to be understood if cloning of ChordFinder is needed
96 aCF = this->fChordFinder;/*->Clone*/
97 aFM->fChordFinder = aCF;
98 }
99 //Copy values of other variables
100 aFM->fEpsilonMax = this->fEpsilonMax;
101 aFM->fEpsilonMin = this->fEpsilonMin;
102 // aFM->fDefault_Delta_Intersection_Val = this->fDefault_Delta_Intersection_Val; // now static
103 // aFM->fDefault_Delta_One_Step_Value = this->fDefault_Delta_One_Step_Value; // now static
104 aFM->fDelta_Intersection_Val = this->fDelta_Intersection_Val;
105 aFM->fDelta_One_Step_Value = this->fDelta_One_Step_Value;
106 //TODO: Should we really add to the store the cloned FM? Who will use this?
107 }
108 catch ( ... )
109 {
110 //Failed creating clone: probably user did not implement Clone method
111 //in derived classes?
112 //Perform clean-up after ourselves...
113 delete aField;
114 delete aFM;
115 delete aCF;
116 throw;
117 }
118 return aFM;
119}

135void
136G4FieldManager::CreateChordFinder(G4MagneticField *detectorMagField)
137{
138 if ( fAllocatedChordFinder )
139 delete fChordFinder;
140 fAllocatedChordFinder= false;
141
142 if( detectorMagField ) {
143 fChordFinder= new G4ChordFinder( detectorMagField );
144 fAllocatedChordFinder= true;
145 } else {
146 fChordFinder = nullptr;
147 }
148}

150void G4FieldManager::InitialiseFieldChangesEnergy()
151{
152 if ( fDetectorField )
153 fFieldChangesEnergy= fDetectorField->DoesFieldChangeEnergy();
154 else
155 fFieldChangesEnergy= false; // No field , no change!
156}

158G4bool G4FieldManager::SetDetectorField(G4Field *pDetectorField, int failMode )
159{
160 G4VIntegrationDriver* driver = nullptr;
161 G4EquationOfMotion* equation = nullptr;
162 // G4bool compatibleField= false;
163 G4bool ableToSet= false;
164
165 fDetectorField= pDetectorField;
166 InitialiseFieldChangesEnergy();
167
168 // Must 'propagate' the field to the dependent classes
169 //
170 if( fChordFinder )
171 {
172 failMode= std::max( failMode, 1) ; // If a chord finder exists, warn in case of error!
173
174 driver = fChordFinder->GetIntegrationDriver();
175 if( driver ){
176 equation = driver->GetEquationOfMotion();
177 // Should check the compatibility between the field and the equation HERE
178 if( equation ) {
179 equation->SetFieldObj(pDetectorField);
180 ableToSet = true;
181 }
182 }
183 }
184
185 if( !ableToSet && (failMode > 0) )
186 {
187 // If this fails, report the issue !
188 G4ExceptionDescription msg;
189 msg << "Unable to set the field in the dependent objects of G4FieldManager" << G4endl;
190 msg << "All the dependent classes must be fully initialised, before it is possible to call this method." << G4endl;
191 msg << "The problem encountered was the following: " << G4endl;
192 if( !fChordFinder ) { msg << " No ChordFinder. " ; }
193 else if( !driver) { msg << " No Integration Driver set. ";}
194 else if( !equation) { msg << " No Equation found. " ; }
195 // else if( !compatibleField ) { msg << " Field not compatible. ";}
196 else { msg << " Can NOT find reason for failure. ";}
197 msg << G4endl;
198 G4ExceptionSeverity severity= (failMode != 1) ? FatalException : JustWarning ;
199 G4Exception("G4FieldManager::SetDetectorField", "Geometry001",
200 severity, msg);
201 }
202 return ableToSet;
203}
