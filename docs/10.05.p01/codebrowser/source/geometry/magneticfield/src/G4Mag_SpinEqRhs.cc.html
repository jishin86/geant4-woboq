44G4Mag_SpinEqRhs::G4Mag_SpinEqRhs( G4MagneticField* MagField )
45 : G4Mag_EqRhs( MagField ), charge(0.), mass(0.), magMoment(0.),

47{
48}

54void
55G4Mag_SpinEqRhs::SetChargeMomentumMass(G4ChargeState particleCharge,
56 G4double MomentumXc,
57 G4double particleMass)
58{
59 G4Mag_EqRhs::SetChargeMomentumMass( particleCharge, MomentumXc, mass);
60
61 charge = particleCharge.GetCharge();
62 mass = particleMass;
63 magMoment = particleCharge.GetMagneticDipoleMoment();
64 spin = particleCharge.GetSpin();
65
66 omegac = (eplus/mass)*c_light;
67
68 G4double muB = 0.5*eplus*hbar_Planck/(mass/c_squared);
69
70 G4double g_BMT;
71 if ( spin != 0. ) g_BMT = (std::abs(magMoment)/muB)/spin;
72 else g_BMT = 2.;
73
74 anomaly = (g_BMT - 2.)/2.;
75
76 G4double E = std::sqrt(sqr(MomentumXc)+sqr(mass));
77 beta = MomentumXc/E;
78 gamma = E/mass;
79}

81void
82G4Mag_SpinEqRhs::EvaluateRhsGivenB( const G4double y[],
83 const G4double B[3],
84 G4double dydx[] ) const
85{
86 G4double momentum_mag_square = sqr(y[3]) + sqr(y[4]) + sqr(y[5]);
87 G4double inv_momentum_magnitude = 1.0 / std::sqrt( momentum_mag_square );
88 G4double cof = FCof()*inv_momentum_magnitude;
89
90 dydx[0] = y[3] * inv_momentum_magnitude; // (d/ds)x = Vx/V
91 dydx[1] = y[4] * inv_momentum_magnitude; // (d/ds)y = Vy/V
92 dydx[2] = y[5] * inv_momentum_magnitude; // (d/ds)z = Vz/V
93
94 if (charge == 0.) {
95 dydx[3] = 0.;
96 dydx[4] = 0.;
97 dydx[5] = 0.;
98 } else {
99 dydx[3] = cof*(y[4]*B[2] - y[5]*B[1]) ; // Ax = a*(Vy*Bz - Vz*By)
100 dydx[4] = cof*(y[5]*B[0] - y[3]*B[2]) ; // Ay = a*(Vz*Bx - Vx*Bz)
101 dydx[5] = cof*(y[3]*B[1] - y[4]*B[0]) ; // Az = a*(Vx*By - Vy*Bx)
102 }
103
104 G4ThreeVector u(y[3], y[4], y[5]);
105 u *= inv_momentum_magnitude;
106
107 G4ThreeVector BField(B[0],B[1],B[2]);
108
109 G4double udb = anomaly*beta*gamma/(1.+gamma) * (BField * u);
110 G4double ucb = (anomaly+1./gamma)/beta;
111
112 // Initialise the values of dydx that we do not update.
113 dydx[6] = dydx[7] = dydx[8] = 0.0;
114
115 G4ThreeVector Spin(y[9],y[10],y[11]);
116
117 G4double pcharge;
118 if (charge == 0.) pcharge = 1.;
119 else pcharge = charge;
120
121 G4ThreeVector dSpin(0.,0.,0.);
122 if (Spin.mag2() != 0.) {
123 dSpin = pcharge*omegac*(ucb*(Spin.cross(BField))-udb*(Spin.cross(u)));
124 }
125
126 dydx[ 9] = dSpin.x();
127 dydx[10] = dSpin.y();
128 dydx[11] = dSpin.z();
129
130 return ;
131}
