103// Minimum for Relative accuracy of any Step
104inline
105G4double G4FieldManager::GetMinimumEpsilonStep() const
106{
107 return fEpsilonMin;
108}
109
110inline
111void G4FieldManager::SetMinimumEpsilonStep( G4double newEpsMin )
112{
113 if( (newEpsMin > 0.0) && (std::fabs(1.0+newEpsMin) > 1.0) )
114 {
115 fEpsilonMin = newEpsMin;
116 }
117}
118
119// Maximum for Relative accuracy of any Step
120inline
121G4double G4FieldManager::GetMaximumEpsilonStep() const
122{
123 return fEpsilonMax;
124}
125
126inline
127void G4FieldManager::SetMaximumEpsilonStep( G4double newEpsMax )
128{
129 if( (newEpsMax > 0.0)
130 && (newEpsMax >= fEpsilonMin )
131 && (std::fabs(1.0+newEpsMax)>1.0) )
132 {
133 fEpsilonMax = newEpsMax;
134 }
135}

137inline
138void G4FieldManager::ChangeDetectorField(G4Field *detectorField)
139{
140 int errorType = 2; // Fatal !
141 this->SetDetectorField( detectorField, errorType );
142}

144inline
145void G4FieldManager::ProposeDetectorField(G4Field *detectorField)
146{
147 fDetectorField= detectorField;
148 this->InitialiseFieldChangesEnergy();
149}
