29// class G4EquationOfMotion
30//
31// Class description:
32//
33// Abstract Base Class for the right hand size of the equation of
34// motion of a particle in a field.
35
36// History:
37// - Created. J.Apostolakis
38// -------------------------------------------------------------------
39
40#ifndef G4_EquationOfMotion_DEF
41#define G4_EquationOfMotion_DEF
42
43#include "G4Types.hh" // "globals.hh"
44#include "G4Field.hh" // required in inline method implementations
45
46#include "G4ChargeState.hh"
47
48class G4EquationOfMotion
49{
50 public: // with description
51
52 G4EquationOfMotion( G4Field *Field );
53 virtual ~G4EquationOfMotion();
54 // Constructor and virtual destructor. No operations.
55
56 virtual void EvaluateRhsGivenB( const G4double y[],
57 const G4double B[3],
58 G4double dydx[] ) const = 0;
59 // Given the value of the field "B", this function
60 // calculates the value of the derivative dydx.
61 // --------------------------------------------------------
62 // This is the _only_ function a subclass must define.
63 // The other two functions use Rhs_givenB.
64
65 virtual void SetChargeMomentumMass(G4ChargeState particleCharge,
66 G4double MomentumXc,
67 G4double MassXc2) = 0;
68 // Set the charge, momentum and mass of the current particle
69 // --> used to set the equation's coefficients ...
70
71 inline
72 void RightHandSide( const G4double y[],
73 G4double dydx[] ) const;
74 // This calculates the value of the derivative dydx at y.
75 // It is the usual enquiry function.
76 // ---------------------------
77 // (It is not virtual, but calls the virtual function above.)
78
79 void EvaluateRhsReturnB( const G4double y[],
80 G4double dydx[],
81 G4double Field[] ) const;
82 // Same as RHS above, but also returns the value of B.
83 // Should be made the new default ? after putting dydx & B in a class.
84
85 void GetFieldValue( const G4double Point[4],
86 G4double Field[] ) const;
87 // Obtain only the field - the stepper assumes it is pure Magnetic.
88 // Not protected, because G4RKG3_Stepper uses it directly.
89
90 const G4Field* GetFieldObj() const;
91 G4Field* GetFieldObj();
92 void SetFieldObj(G4Field* pField);
93
94 private:
95 G4Field *itsField;
96
97};
98
99#include "G4EquationOfMotion.icc"
100
101#endif /* G4_EquationOfMotion_DEF */
