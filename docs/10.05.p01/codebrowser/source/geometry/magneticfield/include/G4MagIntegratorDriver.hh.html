29// class G4MagInt_Driver
30//
31// Class description:
32//
33// Provides a driver that talks to the Integrator Stepper, and insures that
34// the error is within acceptable bounds.
35
36// History:
37// - Created. J.Apostolakis.
38// --------------------------------------------------------------------
39
40#ifndef G4MagInt_Driver_Def
41#define G4MagInt_Driver_Def
42
43#include "G4VIntegrationDriver.hh"
44#include "G4MagIntegratorStepper.hh"
45#include "G4ChordFinderDelegate.hh"
46
47class G4MagInt_Driver : public G4VIntegrationDriver,
48 public G4ChordFinderDelegate<G4MagInt_Driver>
49{
50public: // with description
51
52 G4MagInt_Driver(G4double hminimum,
53 G4MagIntegratorStepper* pItsStepper,
54 G4int numberOfComponents = 6,
55 G4int statisticsVerbosity = 1);
56 virtual ~G4MagInt_Driver() override;
57 // Constructor, destructor.
58
59 G4MagInt_Driver(const G4MagInt_Driver&) = delete;
60 G4MagInt_Driver& operator=(const G4MagInt_Driver&) = delete;
61 // deleted Private copy constructor and assignment operator.
62
63 virtual G4double AdvanceChordLimited(G4FieldTrack& track,
64 G4double stepMax,
65 G4double epsStep,
66 G4double chordDistance) override
67 {
68 return ChordFinderDelegate::AdvanceChordLimitedImpl(track, stepMax, epsStep, chordDistance);
69 }
70
71 virtual void OnComputeStep() override
72 {
73 ChordFinderDelegate::ResetStepEstimate();
74 }
75
76 virtual void OnStartTracking() override {};
77
78
79 virtual G4bool AccurateAdvance(G4FieldTrack& y_current,
80 G4double hstep,
81 G4double eps, // Requested y_err/hstep
82 G4double hinitial = 0.0) override; // Suggested 1st interval
83 // Above drivers for integrator (Runge-Kutta) with stepsize control.
84 // Integrates ODE starting values y_current
85 // from current s (s=s0) to s=s0+h with accuracy eps.
86 // On output ystart is replaced by value at end of interval.
87 // The concept is similar to the odeint routine from NRC p.721-722.
88
89 virtual G4bool QuickAdvance(G4FieldTrack& y_val, // INOUT
90 const G4double dydx[],
91 G4double hstep,
92 G4double inverseCurvatureRadius,
93 G4double& dchord_step,
94 G4double& dyerr) override;
95 // QuickAdvance just tries one Step - it does not ensure accuracy.
96
97 G4bool QuickAdvance(G4FieldTrack& y_posvel, // INOUT
98 const G4double dydx[],
99 G4double hstep, // IN
100 G4double& dchord_step,
101 G4double& dyerr_pos_sq,
102 G4double& dyerr_mom_rel_sq ) ;
103 // New QuickAdvance that also just tries one Step
104 // (so also does not ensure accuracy)
105 // but does return the errors in position and
106 // momentum (normalised: Delta_Integration(p^2)/(p^2) )
107
108 inline G4double GetHmin() const;
109 inline G4double Hmin() const; // Obsolete
110 inline G4double GetSafety() const;
111 inline G4double GetPshrnk() const;
112 inline G4double GetPgrow() const;
113 inline G4double GetErrcon() const;
114 virtual void GetDerivatives(const G4FieldTrack& y_curr, // const, INput
115 G4double dydx[]) const override; // OUTput
116
117 virtual void GetDerivatives(const G4FieldTrack& track,
118 G4double dydx[],
119 G4double field[]) const override;
120 // Accessors.
121
122 virtual G4EquationOfMotion* GetEquationOfMotion() override;
123 virtual void SetEquationOfMotion(G4EquationOfMotion* equation) override;
124
125 virtual void RenewStepperAndAdjust(G4MagIntegratorStepper *pItsStepper) override;
126 // Sets a new stepper pItsStepper for this driver. Then it calls
127 // ReSetParameters to reset its parameters accordingly.
128
129 inline void ReSetParameters(G4double new_safety= 0.9 );
130 // i) sets the exponents (pgrow & pshrnk),
131 // using the current Stepper's order,
132 // ii) sets the safety
133 // ii) calculates "errcon" according to the above values.
134
135 inline void SetSafety(G4double valS);
136 inline void SetPshrnk(G4double valPs);
137 inline void SetPgrow (G4double valPg);
138 inline void SetErrcon(G4double valEc);
139 // When setting safety or pgrow, errcon will be set to a
140 // compatible value.
141
142 inline G4double ComputeAndSetErrcon();
143
144 virtual const G4MagIntegratorStepper* GetStepper() const override;
145 virtual G4MagIntegratorStepper* GetStepper() override;
146
147 void OneGoodStep(G4double ystart[], // Like old RKF45step()
148 const G4double dydx[],
149 G4double& x,
150 G4double htry,
151 G4double eps, // memb variables ?
152 G4double& hdid,
153 G4double& hnext ) ;
154 // This takes one Step that is as large as possible while
155 // satisfying the accuracy criterion of:
156 // yerr < eps * |y_end-y_start|
157
158 virtual G4double ComputeNewStepSize(G4double errMaxNorm, // normalised error
159 G4double hstepCurrent) override; // current step size
160 // Taking the last step's normalised error, calculate
161 // a step size for the next step.
162 // Do not limit the next step's size within a factor of the
163 // current one.
164
165 G4double ComputeNewStepSize_WithinLimits(G4double errMaxNorm, // normalised error
166 G4double hstepCurrent); // current step size
167 // Taking the last step's normalised error, calculate
168 // a step size for the next step.
169 // Limit the next step's size within a range around the current one.
170
171 inline G4int GetMaxNoSteps() const;
172 inline void SetMaxNoSteps( G4int val);
173 // Modify and Get the Maximum number of Steps that can be
174 // taken for the integration of a single segment -
175 // (ie a single call to AccurateAdvance).
176
177public: // without description
178
179 inline void SetHmin(G4double newval);
180 virtual void SetVerboseLevel(G4int newLevel) override;
181 virtual G4int GetVerboseLevel() const override;
182
183 inline G4double GetSmallestFraction() const;
184 void SetSmallestFraction( G4double val );
185
186protected: // without description
187 void WarnSmallStepSize(G4double hnext, G4double hstep,
188 G4double h, G4double xDone,
189 G4int noSteps);
190
191 void WarnTooManyStep(G4double x1start, G4double x2end, G4double xCurrent);
192 void WarnEndPointTooFar(G4double endPointDist,
193 G4double hStepSize ,
194 G4double epsilonRelative,
195 G4int debugFlag);
196 // Issue warnings for undesirable situations
197
198 void PrintStatus(const G4double* StartArr,
199 G4double xstart,
200 const G4double* CurrentArr,
201 G4double xcurrent,
202 G4double requestStep,
203 G4int subStepNo);
204 void PrintStatus(const G4FieldTrack& StartFT,
205 const G4FieldTrack& CurrentFT,
206 G4double requestStep,
207 G4int subStepNo);
208 void PrintStat_Aux(const G4FieldTrack& aFieldTrack,
209 G4double requestStep,
210 G4double actualStep,
211 G4int subStepNo,
212 G4double subStepSize,
213 G4double dotVelocities);
214 // Verbose output for debugging
215
216 void PrintStatisticsReport();
217 // Report on the number of steps, maximum errors etc.
218
219#ifdef QUICK_ADV_TWO
220 G4bool QuickAdvance( G4double yarrin[], // In
221 const G4double dydx[],
222 G4double hstep,
223 G4double yarrout[], // Out
224 G4double& dchord_step, // Out
225 G4double& dyerr ); // in length
226#endif
227
228private:
229 // ---------------------------------------------------------------
230 // INVARIANTS
231
232 G4double fMinimumStep;
233 // Minimum Step allowed in a Step (in absolute units)
234 G4double fSmallestFraction; // Expected range 1e-12 to 5e-15;
235 // Smallest fraction of (existing) curve length - in relative units
236 // below this fraction the current step will be the last
237
238 const G4int fNoIntegrationVariables; // Number of Variables in integration
239 const G4int fMinNoVars; // Minimum number for FieldTrack
240 const G4int fNoVars; // Full number of variable
241
242 G4int fMaxNoSteps;
243 G4int fMaxStepBase;
244
245 G4double safety;
246 G4double pshrnk; // exponent for shrinking
247 G4double pgrow; // exponent for growth
248 G4double errcon;
249 // Parameters used to grow and shrink trial stepsize.
250
251 G4int fStatisticsVerboseLevel;
252
253 // ---------------------------------------------------------------
254 // DEPENDENT Objects
255 G4MagIntegratorStepper *pIntStepper;
256
257 // ---------------------------------------------------------------
258 // STATE
259
260 unsigned long fNoTotalSteps, fNoBadSteps, fNoSmallSteps, fNoInitialSmallSteps;
261 unsigned long fNoCalls;
262 G4double fDyerr_max, fDyerr_mx2;
263 G4double fDyerrPos_smTot, fDyerrPos_lgTot, fDyerrVel_lgTot;
264 G4double fSumH_sm, fSumH_lg;
265 // Step Statistics
266
267 G4int fVerboseLevel; // Verbosity level for printing (debug, ..)
268 // Could be varied during tracking - to help identify issues
269
270 using ChordFinderDelegate = G4ChordFinderDelegate<G4MagInt_Driver>;
271
272};
