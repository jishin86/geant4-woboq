29// class G4TrialsCounter
30//
31// Class description:
32// Keep statistics of the number of trials,
33// including the Maximum and how many times it was reached.
34//
35
36// Author: Dec 8, 2006 John Apostolakis
37// -------------------------------------------------------------------
38#ifndef G4TRIALS_COUNTER_HH
39#define G4TRIALS_COUNTER_HH
40
41#include "G4Types.hh"
42#include "G4String.hh"
43
44class G4TrialsCounter
45{
46 public: // with description
47
48 G4TrialsCounter( const G4String& nameStats,
49 const G4String& description, G4bool printOnExit=false );
50 ~G4TrialsCounter();
51
52 inline void AccumulateCounts( G4int noTrials );
53 // Add this number to stats
54 void ClearCounts();
55 // Reset all counts
56 G4int ReturnTotals( G4int& calls, G4int& maxTrials, G4int& numMaxT ) ;
57 // Return number of count/trials, calls, max & no-max
58
59 void PrintStatistics();
60
61 private:
62
63 G4int fTotalNoTrials; // Counts sum of trials
64 G4int fNumberCalls; // Total # of calls to accumulate
65 G4int fmaxTrials; // Max value of trials
66 G4int fNoTimesMaxTrials; // How many times maximum is reached
67
68 G4String fName; // Identifies stats, and is printed
69 G4String fDescription; // Explanation of stats
70 G4bool fStatsVerbose; // If verbose and not printed, print on destruction
71 G4bool fPrinted; // Flag, to avoid reprinting on destruction
72};
