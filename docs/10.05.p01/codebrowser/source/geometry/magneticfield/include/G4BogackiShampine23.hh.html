26// Bogacki-Shampine - 4 - 3(2) non-FSAL implementation
27//
28// An implementation of the embedded RK method from the paper
29// [1] P. Bogacki and L. F. Shampine, “A 3(2) pair of Runge - Kutta formulas,”
30// Appl. Math. Lett., vol. 2, no. 4, pp. 321–325, Jan. 1989.
31//
32// This version does not utilise the FSAL property of the method,
33// which would allow the reuse of the last derivative in the next step.
34// (Alternative FSAL implementation created with revised interface)
35//
36// Implemented by Somnath Banerjee
37// Work supported by the Google Summer of Code 2015.
38// Supervision / code review: John Apostolakis
39//
40// First version: 20 May 2015
41//
42///////////////////////////////////////////////////////////////////////////////
43
44
45#ifndef G4BOGACKI_SHAMPINE23_H
46#define G4BOGACKI_SHAMPINE23_H
47
48#include "G4MagIntegratorStepper.hh"
49#include "G4FieldTrack.hh"
50
51class G4BogackiShampine23 : public G4MagIntegratorStepper{
52 public:
53 G4BogackiShampine23(G4EquationOfMotion* EqRhs,
54 G4int numberOfVariables = 6);
55
56 virtual void Stepper(const G4double yInput[],
57 const G4double dydx[],
58 G4double hstep,
59 G4double yOutput[],
60 G4double yError[]) override;
61
62 void Stepper(const G4double yInput[],
63 const G4double dydx[],
64 G4double hstep,
65 G4double yOutput[],
66 G4double yError[],
67 G4double dydxOutput[]);
68
69 G4BogackiShampine23(const G4BogackiShampine23&) = delete;
70 G4BogackiShampine23& operator = (const G4BogackiShampine23&) = delete;
71
72 virtual G4double DistChord() const override;
73 virtual G4int IntegratorOrder() const override { return 3; }
74
75 private:
76 void makeStep(const G4double yInput[],
77 const G4double dydx[],
78 const G4double hstep,
79 G4double yOutput[],
80 G4double* dydxOutput = nullptr,
81 G4double* yError = nullptr) const;
82
83 G4double fyIn[G4FieldTrack::ncompSVEC],
84 fdydx[G4FieldTrack::ncompSVEC],
85 fyOut[G4FieldTrack::ncompSVEC],
86 fdydxOut[G4FieldTrack::ncompSVEC];
87 G4double fhstep = -1.0;
88
89};
