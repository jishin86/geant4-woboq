29// class G4FSALIntegrationDriver
30//
31// Class description:
32//
33// Driver class which controls the integration error of a
34// Runge-Kutta stepper
35
36// History:
37// - Created. D.Sorokin
38// --------------------------------------------------------------------
39
40#ifndef G4FSALIntegrationDriver_HH
41#define G4FSALIntegrationDriver_HH
42
43#include "G4RKIntegrationDriver.hh"
44#include "G4ChordFinderDelegate.hh"
45
46template <class T>
47class G4FSALIntegrationDriver : public G4RKIntegrationDriver<T> ,
48 public G4ChordFinderDelegate<G4FSALIntegrationDriver<T>> {
49public:
50 G4FSALIntegrationDriver( G4double hminimum,
51 T* stepper,
52 G4int numberOfComponents = 6,
53 G4int statisticsVerbosity = 1);
54
55 virtual ~G4FSALIntegrationDriver() override;
56
57 G4FSALIntegrationDriver(const G4FSALIntegrationDriver &) = delete;
58 const G4FSALIntegrationDriver& operator =(const G4FSALIntegrationDriver &) = delete;
59
60 virtual G4double AdvanceChordLimited(G4FieldTrack& track,
61 G4double hstep,
62 G4double eps,
63 G4double chordDistance) override
64 {
65 return ChordFinderDelegate::AdvanceChordLimitedImpl(track, hstep, eps, chordDistance);
66 }
67
68 virtual void OnStartTracking() override
69 {
70 ChordFinderDelegate::ResetStepEstimate();
71 }
72
73 virtual void OnComputeStep() override {};
74
75 // Integrates ODE from current s (s=s0) to s=s0+h with accuracy eps.
76 // On output track is replaced by value at end of interval.
77 // The concept is similar to the odeint routine from NRC p.721-722.
78 virtual G4bool AccurateAdvance(
79 G4FieldTrack& track,
80 G4double hstep,
81 G4double eps, // Requested y_err/hstep
82 G4double hinitial = 0) override; // Suggested 1st interval
83
84 // QuickAdvance just tries one Step - it does not ensure accuracy.
85 virtual G4bool QuickAdvance(
86 G4FieldTrack& fieldTrack,
87 const G4double dydx[],
88 G4double hstep,
89 G4double inverseCurvatureRadius,
90 G4double& dchord_step,
91 G4double& dyerr) override;
92
93 virtual void SetVerboseLevel(G4int newLevel) override;
94 virtual G4int GetVerboseLevel() const override;
95
96 // Accessors.
97 G4double GetMinimumStep() const;
98 void SetMinimumStep(G4double newval);
99
100 // This takes one Step that is of size htry, or as large
101 // as possible while satisfying the accuracy criterion of:
102 // yerr < eps * |y_end-y_start|
103 void OneGoodStep(G4double y[], // InOut
104 G4double dydx[],
105 G4double& curveLength,
106 G4double htry,
107 G4double eps,
108 G4double& hdid,
109 G4double& hnext);
110
111 G4double GetSmallestFraction() const;
112 void SetSmallestFraction(G4double val);
113
114protected:
115 void IncrementQuickAdvanceCalls();
116
117private:
118 void CheckStep(const G4ThreeVector& posIn,
119 const G4ThreeVector& posOut,
120 G4double hdid);
121
122 // Minimum Step allowed in a Step (in absolute units)
123 G4double fMinimumStep;
124
125 // Smallest fraction of (existing) curve length - in relative units
126 // below this fraction the current step will be the last
127 G4double fSmallestFraction;
128 // Expected range: smaller than 0.1 * epsilon and bigger than 5e-13
129 // ( Note: this range is not enforced. )
130
131 // Verbosity level for printing (debug, ..)
132 // Could be varied during tracking - to help identify issues
133 G4int fVerboseLevel;
134
135 G4int fNoQuickAvanceCalls;
136 G4int fNoAccurateAdvanceCalls;
137 G4int fNoAccurateAdvanceBadSteps;
138 G4int fNoAccurateAdvanceGoodSteps;
139
140 using Base = G4RKIntegrationDriver<T>;
141 using ChordFinderDelegate = G4ChordFinderDelegate<G4FSALIntegrationDriver<T>>;
142};
143
144#include "G4FSALIntegrationDriver.icc"
145
146#endif
