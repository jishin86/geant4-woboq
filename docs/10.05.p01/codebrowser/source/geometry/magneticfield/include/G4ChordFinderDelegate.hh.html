41#ifndef G4CHORD_FINDER_DELEGATE_HH
42#define G4CHORD_FINDER_DELEGATE_HH
43
44#include "G4VIntegrationDriver.hh"
45
46template <class Driver>
47class G4ChordFinderDelegate {
48public:
49 virtual ~G4ChordFinderDelegate();
50
51 G4double AdvanceChordLimitedImpl(G4FieldTrack& track,
52 G4double hstep,
53 G4double eps,
54 G4double chordDistance);
55 void ResetStepEstimate();
56
57 void TestChordPrint(G4int noTrials,
58 G4int lastStepTrial,
59 G4double dChordStep,
60 G4double fDeltaChord,
61 G4double nextStepTrial);
62
63 // Get statistics about number of calls & trials in FindNextChord
64 G4int GetNoCalls();
65 G4int GetNoTrials(); // Total number of trials
66 G4int GetNoMaxTrials(); // Maximum # of trials for one call
67
68 // Parameters for performance ... change with great care
69 void SetFractions_Last_Next(G4double fractLast = 0.90,
70 G4double fractNext = 0.95);
71 void SetFirstFraction(G4double fractFirst);
72
73 // Printing for monitoring ...
74 G4double GetFirstFraction(); // Originally 0.999
75 G4double GetFractionLast(); // Originally 1.000
76 G4double GetFractionNextEstimate(); // Originally 0.980
77
78 G4double GetLastStepEstimateUnc();
79 void SetLastStepEstimateUnc(G4double stepEst);
80
81private:
82 Driver& GetDriver();
83
84 G4double FindNextChord(const G4FieldTrack& yStart,
85 G4double stepMax,
86 G4double epsStep,
87 G4double chordDistance,
88 G4FieldTrack& yEnd, // Endpoint
89 G4double& dyErrPos, // Error of endpoint
90 G4double& pStepForAccuracy);
91
92 G4double NewStep(G4double stepTrialOld,
93 G4double dChordStep, // Curr. dchord achieved
94 G4double fDeltaChord,
95 G4double& stepEstimate_Unconstrained);
96
97 void AccumulateStatistics(G4int noTrials);
98
99 void PrintStatistics();
100
101 G4double fFirstFraction = 0.999;
102 G4double fFractionLast = 1.0;
103 G4double fFractionNextEstimate = 0.98;
104 G4double fLastStepEstimate_Unconstrained = DBL_MAX;
105
106 G4int fTotalNoTrials = 0;
107 G4int fNoCalls = 0;
108 G4int fmaxTrials = 0;
109};
