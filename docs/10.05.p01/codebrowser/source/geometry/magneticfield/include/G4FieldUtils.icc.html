30namespace field_utils {
31
32namespace internal {
33
34template<class T>
35size_t getFirstIndex(const T& value) {
36 return static_cast<size_t>(value);
37}
38
39} // internal
40
41template <typename ArrayType>
42inline G4double getValue(const ArrayType& array, Value1D value)
43{
44 const auto begin = internal::getFirstIndex(value);
45 return array[begin];
46}
47
48template <typename ArrayType>
49G4double getValue2(const ArrayType& array, Value1D value)
50{
51 return sqr(getValue(array, value));
52}
53
54template <typename ArrayType>
55G4double getValue(const ArrayType& array, Value3D value)
56{
57 return std::sqrt(getValue2(array, value));
58}
59
60template <typename ArrayType>
61G4double getValue2(const ArrayType& array, const Value3D value)
62{
63 const auto begin = internal::getFirstIndex(value);
64 return sqr(array[begin]) + sqr(array[begin+1]) + sqr(array[begin+2]);
65}
66
67template <typename ArrayType>
68G4ThreeVector makeVector(const ArrayType& array, Value3D value)
69{
70 const auto begin = internal::getFirstIndex(value);
71 return G4ThreeVector(array[begin], array[begin + 1], array[begin + 2]);
72}
73
74template <typename SourceArray, typename TargetArray>
75void setValue(const SourceArray& src, Value1D value, TargetArray& trg)
76{
77 const auto begin = internal::getFirstIndex(value);
78 trg[begin] = src[begin];
79}
80
81template <typename SourceArray, typename TargetArray, typename ...TargetArrays>
82void setValue(const SourceArray& src, Value1D value, TargetArray& trg, TargetArrays&... trgs)
83{
84 const auto begin = internal::getFirstIndex(value);
85 trg[begin] = src[begin];
86 setValue(src, value, trgs...);
87}
88
89} // field_utils
