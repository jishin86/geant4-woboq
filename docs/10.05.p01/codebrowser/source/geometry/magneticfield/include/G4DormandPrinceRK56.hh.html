34#ifndef DORMAND_PRINCE_RK56_H
35#define DORMAND_PRINCE_RK56_H
36
37#include "G4MagIntegratorStepper.hh"
38
39class G4DormandPrinceRK56 : public G4MagIntegratorStepper
40{
41
42public:
43 //constructor
44 G4DormandPrinceRK56( G4EquationOfMotion *EqRhs,
45 G4int numberOfVariables = 6,
46 G4bool primary= true ) ;
47
48 //destructor
49 ~G4DormandPrinceRK56() ;
50
51 //Stepper
52 void Stepper( const G4double y[],
53 const G4double dydx[],
54 G4double h,
55 G4double yout[],
56 G4double yerr[] ) ;
57
58 G4double DistChord() const;
59 G4int IntegratorOrder() const { return 5; }
60
61 G4DormandPrinceRK56(const G4DormandPrinceRK56&);
62 G4DormandPrinceRK56& operator=(const G4DormandPrinceRK56&);
63
64 //For Preparing the Interpolant and calculating the extra stages
65 void SetupInterpolate_low( const G4double yInput[],
66 const G4double dydx[],
67 const G4double Step );
68
69 //For calculating the output at the tau fraction of Step
70 void Interpolate_low( const G4double yInput[],
71 const G4double dydx[],
72 const G4double Step,
73 G4double yOut[],
74 G4double tau );
75
76 void SetupInterpolation()
77 { SetupInterpolate( fLastInitialVector, fLastDyDx, fLastStepLength); }
78
79 inline void SetupInterpolate( const G4double yInput[],
80 const G4double dydx[],
81 const G4double Step ){
82 SetupInterpolate_low( yInput, dydx, Step);
83 }
84
85 //For calculating the output at the tau fraction of Step
86 inline void Interpolate( const G4double yInput[],
87 const G4double dydx[],
88 const G4double Step,
89 G4double yOut[],
90 G4double tau ){
91 Interpolate_low( yInput, dydx, Step, yOut, tau);
92 }
93
94 void Interpolate( G4double tau, G4double yOut[])
95 { Interpolate( fLastInitialVector, fLastDyDx, fLastStepLength, yOut, tau ); }
96
97 void SetupInterpolate_high( const G4double yInput[],
98 const G4double dydx[],
99 const G4double Step );
100
101 //For calculating the output at the tau fraction of Step
102 void Interpolate_high( const G4double yInput[],
103 const G4double dydx[],
104 const G4double Step,
105 G4double yOut[],
106 G4double tau );
107
108private:
109 G4double *ak2, *ak3, *ak4, *ak5, *ak6, *ak7, *ak8, *ak9; // for storing intermediate 'k' values in stepper
110 G4double *ak10_low, *ak10, *ak11, * ak12; //For the additional stages of Interpolant
111 G4double *yTemp, *yIn;
112
113 G4double fLastStepLength;
114 G4double *fLastInitialVector, *fLastFinalVector,
115 *fLastDyDx, *fMidVector, *fMidError;
116 // for DistChord calculations
117
118 G4DormandPrinceRK56* fAuxStepper;
119};
120
121#endif /* G4DormandPrinceRK56 */
