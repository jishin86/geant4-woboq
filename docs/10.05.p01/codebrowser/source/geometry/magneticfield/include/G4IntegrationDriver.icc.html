43template <class T>
44G4IntegrationDriver<T>::
45G4IntegrationDriver ( G4double hminimum, T* pStepper,
46 G4int numComponents, G4int statisticsVerbose )
47 : G4RKIntegrationDriver<T>(pStepper),
48 fMinimumStep(hminimum),
49 fSmallestFraction(1e-12),
50 fVerboseLevel(statisticsVerbose),
51 fNoQuickAvanceCalls(0),
52 fNoAccurateAdvanceCalls(0),
53 fNoAccurateAdvanceBadSteps(0),
54 fNoAccurateAdvanceGoodSteps(0)
55{
56 if (numComponents != Base::GetStepper()->GetNumberOfVariables())
57 {
58 std::ostringstream message;
59 message << "Driver's number of integrated components "
60 << numComponents
61 << " != Stepper's number of components "
62 << pStepper->GetNumberOfVariables();
63 G4Exception("G4IntegrationDriver","GeomField0002",
64 FatalException, message);
65 }
66}

81// Runge-Kutta driver with adaptive stepsize control. Integrate starting
82// values at y_current over hstep x2 with accuracy eps.
83// On output ystart is replaced by values at the end of the integration
84// interval. RightHandSide is the right-hand side of ODE system.
85// The source is similar to odeint routine from NRC p.721-722 .
86template <class T>
87G4bool G4IntegrationDriver<T>::
88AccurateAdvance(G4FieldTrack& track, G4double hstep,
89 G4double eps, G4double hinitial)
90{
91 ++fNoAccurateAdvanceCalls;
92
93 if (hstep == 0.0)
94 {
95 std::ostringstream message;
96 message << "Proposed step is zero; hstep = " << hstep << " !";
97 G4Exception("G4IntegrationDriver::AccurateAdvance()",
98 "GeomField1001", JustWarning, message);
99 return true;
100 }
101
102 if (hstep < 0)
103 {
104 std::ostringstream message;
105 message << "Invalid run condition." << G4endl
106 << "Proposed step is negative; hstep = " << hstep << "." << G4endl
107 << "Requested step cannot be negative! Aborting event.";
108 G4Exception("G4IntegrationDriver::AccurateAdvance()",
109 "GeomField0003", EventMustBeAborted, message);
110 return false;
111 }
112
113 G4double hnext, hdid;
114
115 G4double dydx[G4FieldTrack::ncompSVEC];
116 G4bool succeeded = true, lastStepSucceeded;
117
118 G4int noFullIntegr = 0, noSmallIntegr = 0;
119
120 G4double y[G4FieldTrack::ncompSVEC];
121 track.DumpToArray(y);
122
123 const G4double startCurveLength = track.GetCurveLength();
124 const G4double endCurveLength = startCurveLength + hstep;
125 const G4double hThreshold =
126 std::min(eps * hstep, fSmallestFraction * startCurveLength);
127
128 G4double h = hstep;
129 if (hinitial > perMillion * hstep && hinitial < hstep)
130 {
131 h = hinitial;
132 }
133
134 G4double curveLength = startCurveLength;
135
136 for (G4int nstp = 0; nstp < Base::GetMaxNoSteps(); ++nstp)
137 {
138 const G4ThreeVector StartPos =
139 field_utils::makeVector(y, field_utils::Value3D::Position);
140
141 Base::GetStepper()->RightHandSide(y, dydx);
142
143 if (h > GetMinimumStep())
144 {
145 OneGoodStep(y, dydx, curveLength, h, eps, hdid, hnext);
146 lastStepSucceeded = (hdid == h);
147 }
148 else
149 {
150 G4FieldTrack yFldTrk('0');
151 G4double dchord_step, dyerr, dyerr_len;
152 yFldTrk.LoadFromArray(y, Base::GetStepper()->GetNumberOfVariables());
153 yFldTrk.SetCurveLength(curveLength);
154
155 QuickAdvance(yFldTrk, dydx, h, Base::UNKNOWN_CURVATURE_RADIUS, dchord_step, dyerr_len);
156
157 yFldTrk.DumpToArray(y);
158
159 if (h == 0.0)
160 {
161 G4Exception("G4IntegrationDriver::AccurateAdvance()",
162 "GeomField0003", FatalException,
163 "Integration Step became Zero!");
164 }
165 dyerr = dyerr_len / h;
166 hdid = h;
167 curveLength += hdid;
168 hnext = Base::ComputeNewStepSize(dyerr / eps, h);
169 lastStepSucceeded = (dyerr <= eps);
170 }
171
172 if (lastStepSucceeded) { noFullIntegr++; }
173 else { noSmallIntegr++; }
174
175 const G4ThreeVector EndPos =
176 field_utils::makeVector(y, field_utils::Value3D::Position);
177
178 CheckStep(EndPos, StartPos, hdid);
179
180 // Avoid numerous small last steps
181 if (h < hThreshold || curveLength >= endCurveLength)
182 {
183 break;
184 }
185
186 h = std::max(hnext, GetMinimumStep());
187 if (curveLength + h > endCurveLength)
188 {
189 h = endCurveLength - curveLength;
190 }
191 }
192 // Have we reached the end ?
193 // --> a better test might be x-endCurveLength > an_epsilon
194 succeeded = (curveLength >= endCurveLength); // If it was a "forced" last step
195
196 track.LoadFromArray(y, Base::GetStepper()->GetNumberOfVariables());
197 track.SetCurveLength(curveLength);
198
199 return succeeded;
200}

202// Driver for one Runge-Kutta Step with monitoring of local truncation error
203// to ensure accuracy and adjust stepsize. Input are dependent variable
204// array y[0,...,5] and its derivative dydx[0,...,5] at the
205// starting value of the independent variable x . Also input are stepsize
206// to be attempted htry, and the required accuracy eps. On output y and x
207// are replaced by their new values, hdid is the stepsize that was actually
208// accomplished, and hnext is the estimated next stepsize.
209// This is similar to the function rkqs from the book:
210// Numerical Recipes in C: The Art of Scientific Computing (NRC), Second
211// Edition, by William H. Press, Saul A. Teukolsky, William T.
212// Vetterling, and Brian P. Flannery (Cambridge University Press 1992),
213// 16.2 Adaptive StepSize Control for Runge-Kutta, p. 719
214template <class T>
215void G4IntegrationDriver<T>::OneGoodStep(G4double y[], // InOut
216 const G4double dydx[],
217 G4double& curveLength, // InOut
218 G4double htry,
219 G4double eps_rel_max,
220 G4double& hdid, // Out
221 G4double& hnext) // Out
222
223{
224 G4double error2 = DBL_MAX;
225
226 G4double yerr[G4FieldTrack::ncompSVEC], ytemp[G4FieldTrack::ncompSVEC];
227
228 G4double h = htry;
229
230 static G4ThreadLocal G4int tot_no_trials = 0;
231 const G4int max_trials = 100;
232
233 for (G4int iter = 0; iter < max_trials; ++iter)
234 {
235 tot_no_trials++;
236
237 Base::GetStepper()->Stepper(y, dydx, h, ytemp, yerr);
238 error2 = field_utils::relativeError2(y, yerr, std::max(h, fMinimumStep), eps_rel_max);
239
240 if (error2 <= 1.0)
241 {
242 break;
243 }
244
245 h = Base::ShrinkStepSize2(h, error2);
246
247 G4double xnew = curveLength + h;
248 if(xnew == curveLength)
249 {
250 std::ostringstream message;
251 message << "Stepsize underflow in Stepper !" << G4endl
252 << "- Step's start x=" << curveLength << " and end x= " << xnew
253 << " are equal !! " << G4endl
254 << " Due to step-size= " << h
255 << ". Note that input step was " << htry;
256 G4Exception("G4IntegrationDriver::OneGoodStep()",
257 "GeomField1001", JustWarning, message);
258 break;
259 }
260 }
261
262 hnext = Base::GrowStepSize2(h, error2);
263 curveLength += (hdid = h);
264
265 field_utils::copy(y, ytemp, Base::GetStepper()->GetNumberOfVariables());
266}

268template <class T>
269G4bool G4IntegrationDriver<T>::
270QuickAdvance(G4FieldTrack& track, // INOUT
271 const G4double dydx[],
272 G4double hstep,
273 G4double /*inverseCurvatureRadius*/,
274 G4double& dchord_step,
275 G4double& dyerr)
276{
277 ++fNoQuickAvanceCalls;
278
279 G4double yIn[G4FieldTrack::ncompSVEC],
280 yOut[G4FieldTrack::ncompSVEC],
281 yError[G4FieldTrack::ncompSVEC];
282
283 track.DumpToArray(yIn);
284
285 Base::GetStepper()->Stepper(yIn, dydx, hstep, yOut, yError);
286
287 dchord_step = Base::GetStepper()->DistChord();
288 dyerr = field_utils::absoluteError(yOut, yError, hstep);
289 track.LoadFromArray(yOut, Base::GetStepper()->GetNumberOfVariables());
290 track.SetCurveLength(track.GetCurveLength() + hstep);
291
292 return true;
293}

295template <class T>
296void G4IntegrationDriver<T>::SetSmallestFraction(G4double newFraction)
297{
298 if (newFraction > 1.e-16 && newFraction < 1e-8)
299 {
300 fSmallestFraction = newFraction;
301 }
302 else
303 {
304 std::ostringstream message;
305 message << "Smallest Fraction not changed. " << G4endl
306 << " Proposed value was " << newFraction << G4endl
307 << " Value must be between 1.e-8 and 1.e-16";
308 G4Exception("G4IntegrationDriver::SetSmallestFraction()",
309 "GeomField1001", JustWarning, message);
310 }
311}

313template <class T>
314void G4IntegrationDriver<T>::
315CheckStep( const G4ThreeVector& posIn,
316 const G4ThreeVector& posOut, G4double hdid)
317{
318 const G4double endPointDist = (posOut - posIn).mag();
319 if (endPointDist >= hdid * (1. + perMillion))
320 {
321 ++fNoAccurateAdvanceBadSteps;
322#ifdef G4DEBUG_FIELD
323 // Issue a warning only for gross differences -
324 // we understand how small difference occur.
325 if (endPointDist >= hdid * (1. + perThousand))
326 {
327 G4Exception("G4IntegrationDriver::CheckStep()",
328 "GeomField1002", JustWarning,
329 "endPointDist >= hdid!");
330 }
331#endif
332 }
333 else
334 {
335 ++fNoAccurateAdvanceGoodSteps;
336 }
337}
