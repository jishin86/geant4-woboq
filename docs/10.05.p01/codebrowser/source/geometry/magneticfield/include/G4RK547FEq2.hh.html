29// Description:
30// An implementation of the 7 stage embedded Runge-Kutta 4,5 pair (RK547FEq2)
31// from the paper D. J. Higham and G. Hall,
32// “Embedded Runge-Kutta formulae with stable equilibrium states”,
33// J. Comput. Appl. Math., vol. 29, no. 1, pp. 25–33, Jan. 1990.
34//
35// Implementation by Dmitry Sorokin - GSoC 2017
36// Work supported by Google as part of Google Summer of Code 2017.
37// Supervision / code review: John Apostolakis
38
39#ifndef G4RK547FEq2_HH
40#define G4RK547FEq2_HH
41
42#include "G4MagIntegratorStepper.hh"
43#include "G4FieldTrack.hh"
44
45class G4RK547FEq2 : public G4MagIntegratorStepper {
46public:
47 G4RK547FEq2(G4EquationOfMotion* EqRhs, G4int integrationVariables = 6);
48
49 virtual void Stepper(
50 const G4double yInput[],
51 const G4double dydx[],
52 G4double hstep,
53 G4double yOutput[],
54 G4double yError[]) override;
55
56 void Stepper(
57 const G4double yInput[],
58 const G4double dydx[],
59 G4double hstep,
60 G4double yOutput[],
61 G4double yError[],
62 G4double dydxOutput[]);
63
64 G4RK547FEq2 (const G4RK547FEq2&) = delete;
65 G4RK547FEq2& operator = (const G4RK547FEq2&) = delete;
66
67 virtual G4double DistChord() const override;
68 virtual G4int IntegratorOrder() const override { return 4; }
69
70private:
71 void makeStep(
72 const G4double yInput[],
73 const G4double dydx[],
74 const G4double hstep,
75 G4double yOutput[],
76 G4double* dydxOutput = nullptr,
77 G4double* yError = nullptr) const;
78
79 G4double fyIn[G4FieldTrack::ncompSVEC],
80 fdydx[G4FieldTrack::ncompSVEC],
81 fyOut[G4FieldTrack::ncompSVEC],
82 fdydxOut[G4FieldTrack::ncompSVEC];
83 G4double fhstep= -1.0;
84};
85
86#endif
