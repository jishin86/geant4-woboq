65template <class Driver>
66G4double G4ChordFinderDelegate<Driver>::
67AdvanceChordLimitedImpl(G4FieldTrack& yCurrent, G4double stepMax,
68 G4double epsStep, G4double chordDistance)
69{
70 G4double dyErr;
71 G4FieldTrack yEnd = yCurrent;
72 G4double nextStep;
73
74 const G4double stepPossible = FindNextChord(yCurrent, stepMax, epsStep, chordDistance, yEnd, dyErr, nextStep);
75
76 if (dyErr < epsStep * stepPossible)
77 {
78 // Accept this accuracy.
79 yCurrent = yEnd;
80 return stepPossible;
81 }
82
83 // Advance more accurately to "end of chord"
84 const G4double startCurveLen = yCurrent.GetCurveLength();
85 const G4bool goodAdvance = GetDriver().AccurateAdvance(yCurrent, stepPossible, epsStep, nextStep);
86
87 return goodAdvance ? stepPossible : yCurrent.GetCurveLength() - startCurveLen;
88}

90// Returns Length of Step taken
91template <class T>
92G4double G4ChordFinderDelegate<T>::
93FindNextChord(const G4FieldTrack& yStart,
94 G4double stepMax,
95 G4double epsStep,
96 G4double chordDistance,
97 G4FieldTrack& yEnd, // Endpoint
98 G4double& dyErrPos, // Error of endpoint
99 G4double& stepForAccuracy)
100{
101 // 1.) Try to "leap" to end of interval
102 // 2.) Evaluate if resulting chord gives d_chord that is good enough.
103 // 2a.) If d_chord is not good enough, find one that is.
104
105 G4double dydx[G4FieldTrack::ncompSVEC];
106 G4double field[G4Field::MAX_NUMBER_OF_COMPONENTS];
107
108 G4bool validEndPoint = false;
109 G4double dChordStep, lastStepLength;
110
111 GetDriver().GetDerivatives(yStart, dydx, field);
112 const G4double inverseRadius = GetDriver().GetInverseCurvatureRadius(yStart, field);
113
114 const G4double safetyFactor = fFirstFraction; // 0.975 or 0.99 ? was 0.999
115
116 G4double stepTrial = std::min(stepMax, safetyFactor * fLastStepEstimate_Unconstrained);
117
118 G4double newStepEst_Uncons = 0.0;
119 G4double stepForChord;
120
121 G4int noTrials = 1;
122 constexpr G4int maxTrials = 75; // Avoid endless loop for bad convergence
123 for (; noTrials < maxTrials; ++noTrials)
124 {
125 yEnd = yStart; // Always start from initial point
126 GetDriver().QuickAdvance(yEnd, dydx, stepTrial, inverseRadius, dChordStep, dyErrPos);
127 lastStepLength = stepTrial;
128
129 validEndPoint = dChordStep < chordDistance;
130 stepForChord = NewStep(stepTrial, dChordStep, chordDistance, newStepEst_Uncons);
131
132 if (validEndPoint)
133 {
134 break;
135 }
136
137 if (stepTrial <= 0.0)
138 {
139 stepTrial = stepForChord;
140 }
141 else if (stepForChord <= stepTrial)
142 {
143 // Reduce by a fraction, possibly up to 20%
144 stepTrial = std::min( stepForChord, fFractionLast * stepTrial);
145 }
146 else
147 {
148 stepTrial *= 0.1;
149 }
150 }
151
152 if (noTrials >= maxTrials)
153 {
154 std::ostringstream message;
155 message << "Exceeded maximum number of trials= " << maxTrials << G4endl
156 << "Current sagita dist= " << dChordStep << G4endl
157 << "Max sagita dist= " << chordDistance << G4endl
158 << "Step sizes (actual and proposed): " << G4endl
159 << "Last trial = " << lastStepLength << G4endl
160 << "Next trial = " << stepTrial << G4endl
161 << "Proposed for chord = " << stepForChord << G4endl;
162 G4Exception("G4ChordFinder::FindNextChord()", "GeomField0003",
163 JustWarning, message);
164 }
165
166 if (newStepEst_Uncons > 0.0)
167 {
168 fLastStepEstimate_Unconstrained = newStepEst_Uncons;
169 }
170
171 AccumulateStatistics(noTrials);
172
173
174 // Calculate the step size required for accuracy, if it is needed
175 G4double dyErr_relative = dyErrPos / (epsStep * lastStepLength);
176 stepForAccuracy = dyErr_relative > 1 ?
177 GetDriver().ComputeNewStepSize(dyErr_relative, lastStepLength) : 0;
178
179 return stepTrial;
180}

182// Is called to estimate the next step size, even for successful steps,
183// in order to predict an accurate 'chord-sensitive' first step
184// which is likely to assist in more performant 'stepping'.
185template <class T>
186G4double G4ChordFinderDelegate<T>::
187NewStep(G4double stepTrialOld,
188 G4double dChordStep, // Curr. dchord achieved
189 G4double fDeltaChord,
190 G4double& stepEstimate_Unconstrained)
191{
192 G4double stepTrial;
193
194 if (dChordStep > 0.0)
195 {
196 stepEstimate_Unconstrained =
197 stepTrialOld * std::sqrt(fDeltaChord / dChordStep);
198 stepTrial = fFractionNextEstimate * stepEstimate_Unconstrained;
199 }
200 else
201 {
202 // Should not update the Unconstrained Step estimate: incorrect!
203 stepTrial = stepTrialOld * 2.;
204 }
205
206 if (stepTrial <= 0.001 * stepTrialOld)
207 {
208 if (dChordStep > 1000.0 * fDeltaChord)
209 {
210 stepTrial = stepTrialOld * 0.03;
211 }
212 else
213 {
214 if (dChordStep > 100. * fDeltaChord)
215 {
216 stepTrial = stepTrialOld * 0.1;
217 }
218 else // Try halving the length until dChordStep OK
219 {
220 stepTrial = stepTrialOld * 0.5;
221 }
222 }
223 }
224 else if (stepTrial > 1000.0 * stepTrialOld)
225 {
226 stepTrial = 1000.0 * stepTrialOld;
227 }
228
229 if (stepTrial == 0.0)
230 {
231 stepTrial= 0.000001;
232 }
233
234 // A more sophisticated chord-finder could figure out a better
235 // stepTrial, from dChordStep and the required d_geometry
236 // e.g.
237 // Calculate R, r_helix (eg at orig point)
238 // if( stepTrial < 2 pi R )
239 // stepTrial = R arc_cos( 1 - fDeltaChord / r_helix )
240 // else
241 // ??
242
243 return stepTrial;
244}

246template <class T>
247void G4ChordFinderDelegate<T>::AccumulateStatistics(G4int noTrials)
248{
249 fTotalNoTrials += noTrials;
250 ++fNoCalls;
251
252 if (noTrials > fmaxTrials)
253 {
254 fmaxTrials = noTrials;
255 }

291template <class T>
292void G4ChordFinderDelegate<T>::SetFractions_Last_Next(G4double fractLast,
293 G4double fractNext)
294{
295 // Use -1.0 as request for Default.
296 if (fractLast == -1.0) fractLast = 1.0; // 0.9;
297 if (fractNext == -1.0) fractNext = 0.98; // 0.9;
298
299 // fFirstFraction = 0.999; // Orig 0.999 A safe value, range: ~ 0.95 - 0.999
300 if (GetDriver().GetVerboseLevel() > 0)
301 {
302 G4cout << " ChordFnd> Trying to set fractions: "
303 << " first " << fFirstFraction
304 << " last " << fractLast
305 << " next " << fractNext
306 << G4endl;
307 }
308
309 if (fractLast > 0 && fractLast <= 1)
310 {
311 fFractionLast = fractLast;
312 } else
313 {
314 std::ostringstream message;
315 message << "Invalid fraction Last = " << fractLast
316 << "; must be 0 < fractionLast <= 1 ";
317 G4Exception("G4ChordFinderDelegate::SetFractions_Last_Next()",
318 "GeomField1001", JustWarning, message);
319 }
320 if (fractNext > 0. && fractNext < 1)
321 {
322 fFractionNextEstimate = fractNext;
323 } else
324 {
325 std::ostringstream message;
326 message << "Invalid fraction Next = " << fractNext
327 << "; must be 0 < fractionNext < 1 ";
328 G4Exception("G4ChordFinderDelegate::SetFractions_Last_Next()",
329 "GeomField1001", JustWarning, message);
330 }
331}

369template <class T>
370void G4ChordFinderDelegate<T>::TestChordPrint(G4int noTrials,
371 G4int lastStepTrial,
372 G4double dChordStep,
373 G4double fDeltaChord,
374 G4double nextStepTrial)
375{
376 G4int oldprec= G4cout.precision(5);
377 G4cout << " ChF/fnc: notrial " << std::setw( 3) << noTrials
378 << " this_step= " << std::setw(10) << lastStepTrial;
379 if( std::fabs( (dChordStep / fDeltaChord) - 1.0 ) < 0.001 )
380 {
381 G4cout.precision(8);
382 }
383 else
384 {
385 G4cout.precision(6);
386 }
387 G4cout << " dChordStep= " << std::setw(12) << dChordStep;
388 if( dChordStep > fDeltaChord ) { G4cout << " d+"; }
389 else { G4cout << " d-"; }
390 G4cout.precision(5);
391 G4cout << " new_step= " << std::setw(10)
392 << fLastStepEstimate_Unconstrained
393 << " new_step_constr= " << std::setw(10)
394 << lastStepTrial << G4endl;
395 G4cout << " nextStepTrial = " << std::setw(10) << nextStepTrial << G4endl;
396 G4cout.precision(oldprec);
397}
