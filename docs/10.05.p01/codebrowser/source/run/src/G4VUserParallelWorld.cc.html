56void G4VUserParallelWorld::SetSensitiveDetector
57(const G4String& logVolName, G4VSensitiveDetector* aSD, G4bool multi)
58{
59 G4bool found = false;
60 G4LogicalVolumeStore* store = G4LogicalVolumeStore::GetInstance();
61 for(G4LogicalVolumeStore::iterator pos=store->begin(); pos!=store->end(); pos++)
62 {
63 if((*pos)->GetName()==logVolName)
64 {
65 if(found && !multi)
66 {
67 G4String eM = "More than one logical volumes of the name <";
68 eM += (*pos)->GetName();
69 eM += "> are found and thus the sensitive detector <";
70 eM += aSD->GetName();
71 eM += "> cannot be uniquely assigned.";
72 G4Exception("G4VUserParallelWorld::SetSensitiveDetector",
73 "Run5052",FatalErrorInArgument,eM);
74 }
75 found = true;
76 SetSensitiveDetector(*pos,aSD);
77 }
78 }
79 if(!found)
80 {
81 G4String eM2 = "No logical volume of the name <";
82 eM2 += logVolName;
83 eM2 += "> is found. The specified sensitive detector <";
84 eM2 += aSD->GetName();
85 eM2 += "> couldn't be assigned to any volume.";
86 G4Exception("G4VUserParallelWorld::SetSensitiveDetector",
87 "Run5053",FatalErrorInArgument,eM2);
88 }
89}

91void G4VUserParallelWorld::SetSensitiveDetector
92(G4LogicalVolume* logVol, G4VSensitiveDetector* aSD)
93{
94 G4SDManager::GetSDMpointer()->AddNewDetector(aSD);
95 logVol->SetSensitiveDetector(aSD);
96}
