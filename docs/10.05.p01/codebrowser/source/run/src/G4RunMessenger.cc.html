373void G4RunMessenger::SetNewValue(G4UIcommand * command,G4String newValue)
374{
375 if( command==beamOnCmd )
376 {
377 G4int nev;
378 G4int nst;
379 const char* nv = (const char*)newValue;
380 std::istringstream is(nv);
381 is >> nev >> macroFileName >> nst;
382 if(macroFileName=="***NULL***")
383 { runManager->BeamOn(nev); }
384 else
385 { runManager->BeamOn(nev,macroFileName,nst); }
386 }
387 else if( command==verboseCmd )
388 { runManager->SetVerboseLevel(verboseCmd->GetNewIntValue(newValue)); }
389 else if( command == printProgCmd )
390 { runManager->SetPrintProgress(printProgCmd->GetNewIntValue(newValue)); }
391 else if( command==nThreadsCmd )
392 {
393 G4RunManager::RMType rmType = runManager->GetRunManagerType();
394 if( rmType==G4RunManager::masterRM )
395 {
396 static_cast<G4MTRunManager*>(runManager)->SetNumberOfThreads(
397 nThreadsCmd->GetNewIntValue(newValue));
398 }
399 else if ( rmType==G4RunManager::sequentialRM )
400 {
401 G4cout<<"*** /run/numberOfThreads command is issued in sequential mode."
402 <<"\nCommand is ignored."<<G4endl;
403 }
404 else
405 {
406 G4Exception("G4RunMessenger::ApplyNewCommand","Run0901",FatalException,
407 "/run/numberOfThreads command is issued to local thread.");
408 }
409 }
410 else if( command==maxThreadsCmd)
411 {
412 G4RunManager::RMType rmType = runManager->GetRunManagerType();
413 if( rmType==G4RunManager::masterRM )
414 {
415 static_cast<G4MTRunManager*>(runManager)->SetNumberOfThreads(
416 G4Threading::G4GetNumberOfCores() );
417 }
418 else if ( rmType==G4RunManager::sequentialRM )
419 {
420 G4cout<<"*** /run/useMaximumLogicalCores command is issued in sequential mode."
421 <<"\nCommand is ignored."<<G4endl;
422 }
423 else
424 {
425 G4Exception("G4RunMessenger::ApplyNewCommand","Run0901",FatalException,
426 "/run/useMaximumLogicalCores command is issued to local thread.");
427 }
428 }
429 else if ( command == pinAffinityCmd )
430 {
431 G4RunManager::RMType rmType = runManager->GetRunManagerType();
432 if( rmType==G4RunManager::masterRM )
433 {
434 static_cast<G4MTRunManager*>(runManager)->SetPinAffinity(
435 pinAffinityCmd->GetNewIntValue(newValue) );
436 }
437 else if ( rmType==G4RunManager::sequentialRM )
438 {
439 G4cout<<"*** /run/pinAffinity command is issued in sequential mode."
440 <<"\nCommand is ignored."<<G4endl;
441 }
442 else
443 {
444 G4Exception("G4RunMessenger::ApplyNewCommand","Run0901",FatalException,
445 "/run/pinAffinity command is issued to local thread.");
446 }
447
448 }
449 else if( command==evModCmd)
450 {
451 G4RunManager::RMType rmType = runManager->GetRunManagerType();
452 if( rmType==G4RunManager::masterRM )
453 {
454 G4int nevMod = 0;
455 G4int sOnce = 0;
456 const char* nv = (const char*)newValue;
457 std::istringstream is(nv);
458 is >> nevMod >> sOnce;
459 static_cast<G4MTRunManager*>(runManager)->SetEventModulo(nevMod);
460 G4MTRunManager::SetSeedOncePerCommunication(sOnce);
461 }
462 else if ( rmType==G4RunManager::sequentialRM )
463 {
464 G4cout<<"*** /run/eventModulo command is issued in sequential mode."
465 <<"\nCommand is ignored."<<G4endl;
466 }
467 else
468 {
469 G4Exception("G4RunMessenger::ApplyNewCommand","Run0902",FatalException,
470 "/run/eventModulo command is issued to local thread.");
471 }
472 }
473 else if( command==dumpRegCmd )
474 {
475 if(newValue=="**ALL**")
476 { runManager->DumpRegion(); }
477 else
478 { runManager->DumpRegion(newValue); }
479 }
480 else if( command==dumpCoupleCmd)
481 {
482 G4ProductionCutsTable::GetProductionCutsTable()->DumpCouples();
483 }
484 else if( command==optCmd )
485 { runManager->SetGeometryToBeOptimized(optCmd->GetNewBoolValue(newValue)); }
486 else if( command==brkBoECmd )
487 { G4UImanager::GetUIpointer()->SetPauseAtBeginOfEvent(brkBoECmd->GetNewBoolValue(newValue)); }
488 else if( command==brkEoECmd )
489 { G4UImanager::GetUIpointer()->SetPauseAtEndOfEvent(brkEoECmd->GetNewBoolValue(newValue)); }
490 else if( command==abortCmd )
491 { runManager->AbortRun(abortCmd->GetNewBoolValue(newValue)); }
492 else if( command==abortEventCmd )
493 { runManager->AbortEvent(); }
494 else if( command==initCmd )
495 { runManager->Initialize(); }
496 else if( command==geomCmd )
497 { runManager->GeometryHasBeenModified(false); }
498 else if( command==geomRebCmd )
499 { runManager->ReinitializeGeometry(geomRebCmd->GetNewBoolValue(newValue),false); }
500 else if( command==physCmd )
501 { runManager->PhysicsHasBeenModified(); }
502 else if( command==seedCmd )
503 {
504 G4Tokenizer next(newValue);
505 G4int idx=0;
506 long seeds[100];
507 G4String vl;
508 while(!(vl=next()).isNull())
509 { seeds[idx] = (long)(StoI(vl)); idx++; }
510 if(idx<2)
511 { G4cerr << "/random/setSeeds should have at least two integers. Command ignored." << G4endl; }
512 else
513 {
514 seeds[idx] = 0;
515 G4Random::setTheSeeds(seeds);
516 }
517 }
518 else if( command==randDirCmd )
519 { runManager->SetRandomNumberStoreDir(newValue); }
520 else if( command==savingFlagCmd )
521 { runManager->SetRandomNumberStore(savingFlagCmd->GetNewBoolValue(newValue)); }
522 else if( command==saveThisRunCmd )
523 { runManager->rndmSaveThisRun(); }
524 else if( command==saveThisEventCmd )
525 { runManager->rndmSaveThisEvent(); }
526 else if( command==restoreRandCmd )
527 { runManager->RestoreRandomNumberStatus(newValue); }
528 else if( command==randEvtCmd )
529 { runManager->StoreRandomNumberStatusToG4Event(randEvtCmd->GetNewIntValue(newValue)); }
530 else if( command==saveEachEventCmd)
531 { runManager->SetRandomNumberStorePerEvent(saveEachEventCmd->GetNewBoolValue(newValue)); }
532 else if( command==constScoreCmd )
533 { runManager->ConstructScoringWorlds(); }
534 else if( command==restoreRandCmdMT)
535 { runManager->RestoreRndmEachEvent(restoreRandCmdMT->GetNewBoolValue(newValue)); }
536 else if ( command==procUICmds)
537 {
538 G4RunManager::RMType rmType = runManager->GetRunManagerType();
539 if( rmType==G4RunManager::masterRM ) {
540 auto rm = dynamic_cast<G4MTRunManager*>(runManager);
541 if ( rm != nullptr ) {
542 rm->RequestWorkersProcessCommandsStack();
543 }
544 else { G4Exception("G4RunManager::ApplyNewCommand","Run0128",FatalException,"/run/workersProcessCmds command issued on a non-G4MTRunManager class instance."); }
545 } else if ( rmType==G4RunManager::sequentialRM ) {
546 G4cout<<"*** /run/workersProcessCmds command is issued in sequential mode."
547 <<"\nCommand is ignored."<<G4endl;
548 } else {
549 G4Exception("G4RunMessenger::ApplyNewCommand","Run0129",FatalException,
550 "/run/workersProcessCmds command is issued to local thread.");
551 }
552 }

556G4String G4RunMessenger::GetCurrentValue(G4UIcommand * command)
557{
558 G4String cv;
559
560 if( command==verboseCmd )
561 { cv = verboseCmd->ConvertToString(runManager->GetVerboseLevel()); }
562 else if( command == printProgCmd )
563 { cv = printProgCmd->ConvertToString(runManager->GetPrintProgress()); }
564 else if( command==randDirCmd )
565 { cv = runManager->GetRandomNumberStoreDir(); }
566 else if( command==randEvtCmd )
567 { cv = randEvtCmd->ConvertToString(runManager->GetFlagRandomNumberStatusToG4Event()); }
568 else if( command==nThreadsCmd )
569 {
570 G4RunManager::RMType rmType = runManager->GetRunManagerType();
571 if( rmType==G4RunManager::masterRM )
572 {
573 cv = nThreadsCmd->ConvertToString(
574 static_cast<G4MTRunManager*>(runManager)->GetNumberOfThreads());
575 }
576 else if ( rmType==G4RunManager::sequentialRM )
577 {
578 cv = "0";
579 }
580 }
581 else if( command==evModCmd)
582 {
583 G4RunManager::RMType rmType = runManager->GetRunManagerType();
584 if( rmType==G4RunManager::masterRM )
585 {
586 cv = evModCmd->ConvertToString(
587 static_cast<G4MTRunManager*>(runManager)->GetEventModulo() )
588 + " " + evModCmd->ConvertToString(G4MTRunManager::SeedOncePerCommunication());
589 }
590 else if ( rmType==G4RunManager::sequentialRM )
591 { G4cout<<"*** /run/eventModulo command is valid only in MT mode."<<G4endl; }
592 }
593
594 return cv;
595}
