173 eventManager = kernel->GetEventManager();

180 randomNumberStatusDir = "./";

182 G4Random::saveFullState(oss);

186}

188G4RunManager::~G4RunManager()
189{
190 G4StateManager* pStateManager = G4StateManager::GetStateManager();
191 // set the application state to the quite state
192 if(pStateManager->GetCurrentState()!=G4State_Quit)
193 {
194 if(verboseLevel>0) G4cout << "G4 kernel has come to Quit state." << G4endl;
195 pStateManager->SetNewState(G4State_Quit);
196 }
197
198 CleanUpPreviousEvents();
199 if(currentRun) delete currentRun;
200 delete timer;
201 delete runMessenger;
202 G4ParticleTable::GetParticleTable()->DeleteMessenger();
203 G4ProcessTable::GetProcessTable()->DeleteMessenger();
204 delete previousEvents;
205
206 //The following will work for all RunManager types
207 //if derived class does the correct thing in derived
208 //destructor that is set to zero pointers of
209 //user initialization objects for which does not have
210 //ownership
211 DeleteUserInitializations();
212 if(userRunAction)
213 {
214 delete userRunAction;
215 userRunAction = 0;
216 if(verboseLevel>1) G4cout << "UserRunAction deleted." << G4endl;
217 }
218 if(userPrimaryGeneratorAction)
219 {
220 delete userPrimaryGeneratorAction;
221 userPrimaryGeneratorAction = 0;
222 if(verboseLevel>1) G4cout << "UserPrimaryGenerator deleted." << G4endl;
223 }
224
225 if(verboseLevel>1) G4cout << "RunManager is deleting RunManagerKernel." << G4endl;
226
227 delete kernel;
228
229 fRunManager = 0;
230}

267void G4RunManager::BeamOn(G4int n_event,const char* macroFile,G4int n_select)
268{
269 TIMEMORY_AUTO_TIMER("");
270 if(n_event<=0) { fakeRun = true; }
271 else { fakeRun = false; }
272 G4bool cond = ConfirmBeamOnCondition();
273 if(cond)
274 {
275 numberOfEventToBeProcessed = n_event;
276 numberOfEventProcessed = 0;
277 ConstructScoringWorlds();
278 RunInitialization();
279 DoEventLoop(n_event,macroFile,n_select);
280 RunTermination();
281 }
282 fakeRun = false;
283}

285G4bool G4RunManager::ConfirmBeamOnCondition()
286{
287 G4StateManager* stateManager = G4StateManager::GetStateManager();
288
289 G4ApplicationState currentState = stateManager->GetCurrentState();
290 if(currentState!=G4State_PreInit && currentState!=G4State_Idle)
291 {
292 G4cerr << "Illegal application state - BeamOn() ignored." << G4endl;
293 return false;
294 }
295
296 if(!initializedAtLeastOnce)
297 {
298 G4cerr << " Geant4 kernel should be initialized" << G4endl;
299 G4cerr << "before the first BeamOn(). - BeamOn ignored." << G4endl;
300 return false;
301 }
302
303 if(!geometryInitialized || !physicsInitialized)
304 {
305 if(verboseLevel>0)
306 {
307 G4cout << "Start re-initialization because " << G4endl;
308 if(!geometryInitialized) G4cout << " Geometry" << G4endl;
309 if(!physicsInitialized) G4cout << " Physics processes" << G4endl;
310 G4cout << "has been modified since last Run." << G4endl;
311 }
312 Initialize();
313 }
314 return true;
315}

317void G4RunManager::RunInitialization()
318{
319 if(!(kernel->RunInitialization(fakeRun))) return;
320
321 TIMEMORY_AUTO_TIMER("");
322 runAborted = false;
323 numberOfEventProcessed = 0;
324
325 CleanUpPreviousEvents();
326 if(currentRun) delete currentRun;
327 currentRun = 0;
328
329 if(fakeRun) return;
330
331 if(fGeometryHasBeenDestroyed) G4ParallelWorldProcessStore::GetInstance()->UpdateWorlds();
332
333 if(userRunAction) currentRun = userRunAction->GenerateRun();
334 if(!currentRun) currentRun = new G4Run();
335
336 currentRun->SetRunID(runIDCounter);
337 currentRun->SetNumberOfEventToBeProcessed(numberOfEventToBeProcessed);
338
339 currentRun->SetDCtable(DCtable);
340 G4SDManager* fSDM = G4SDManager::GetSDMpointerIfExist();
341 if(fSDM)
342 { currentRun->SetHCtable(fSDM->GetHCtable()); }
343
344 if ( G4VScoreNtupleWriter::Instance() ) {
345 auto hce = fSDM->PrepareNewEvent();
346 isScoreNtupleWriter = G4VScoreNtupleWriter::Instance()->Book(hce);
347 delete hce;
348 }
349
350 std::ostringstream oss;
351 G4Random::saveFullState(oss);
352 randomNumberStatusForThisRun = oss.str();
353 currentRun->SetRandomNumberStatus(randomNumberStatusForThisRun);
354
355 for(G4int i_prev=0;i_prev<n_perviousEventsToBeStored;i_prev++)
356 { previousEvents->push_back((G4Event*)0); }
357
358 if(printModulo>=0 || verboseLevel>0)
359 { G4cout << "### Run " << currentRun->GetRunID() << " starts." << G4endl; }
360 if(userRunAction) userRunAction->BeginOfRunAction(currentRun);
361
362 if (isScoreNtupleWriter)
363 { G4VScoreNtupleWriter::Instance()->OpenFile(); }
364
365 if(storeRandomNumberStatus) {
366 G4String fileN = "currentRun";
367 if ( rngStatusEventsFlag ) {
368 std::ostringstream os;
369 os << "run" << currentRun->GetRunID();
370 fileN = os.str();
371 }
372 StoreRNGStatus(fileN);
373 }
374}

376void G4RunManager::DoEventLoop(G4int n_event,const char* macroFile,G4int n_select)
377{
378 TIMEMORY_AUTO_TIMER("");
379 InitializeEventLoop(n_event,macroFile,n_select);
380
381// Event loop
382 for(G4int i_event=0; i_event<n_event; i_event++ )
383 {
384 ProcessOneEvent(i_event);
385 TerminateOneEvent();
386 if(runAborted) break;
387 }
388
389 // For G4MTRunManager, TerminateEventLoop() is invoked after all threads are finished.
390 if(runManagerType==sequentialRM) TerminateEventLoop();
391}

393void G4RunManager::InitializeEventLoop(G4int n_event,const char* macroFile,G4int n_select)
394{
395 if(verboseLevel>0)
396 { timer->Start(); }
397
398 n_select_msg = n_select;
399 if(macroFile!=0)
400 {
401 if(n_select_msg<0) n_select_msg = n_event;
402 msgText = "/control/execute ";
403 msgText += macroFile;
404 selectMacro = macroFile;
405 }
406 else
407 {
408 n_select_msg = -1;
409 selectMacro = "";
410 }

413void G4RunManager::ProcessOneEvent(G4int i_event)
414{
415 TIMEMORY_AUTO_TIMER("");
416 currentEvent = GenerateEvent(i_event);
417 eventManager->ProcessOneEvent(currentEvent);
418 AnalyzeEvent(currentEvent);
419 UpdateScoring();
420 if(i_event<n_select_msg) G4UImanager::GetUIpointer()->ApplyCommand(msgText);
421}

423void G4RunManager::TerminateOneEvent()
424{
425 TIMEMORY_AUTO_TIMER("");
426 StackPreviousEvent(currentEvent);
427 currentEvent = 0;
428 numberOfEventProcessed++;
429}

431void G4RunManager::TerminateEventLoop()
432{
433 if(verboseLevel>0 && !fakeRun)
434 {
435 timer->Stop();
436 G4cout << " Run terminated." << G4endl;
437 G4cout << "Run Summary" << G4endl;
438 if(runAborted)
439 { G4cout << " Run Aborted after " << numberOfEventProcessed << " events processed." << G4endl; }
440 else
441 { G4cout << " Number of events processed : " << numberOfEventProcessed << G4endl; }
442 G4cout << " " << *timer << G4endl;
443 }
444 //////////////// G4ProductionCutsTable::GetProductionCutsTable()->PhysicsTableUpdated();
445 fGeometryHasBeenDestroyed = false;
446}

448G4Event* G4RunManager::GenerateEvent(G4int i_event)
449{
450 TIMEMORY_AUTO_TIMER("");
451 if(!userPrimaryGeneratorAction)
452 {
453 G4Exception("G4RunManager::GenerateEvent()", "Run0032", FatalException,
454 "G4VUserPrimaryGeneratorAction is not defined!");
455 return 0;
456 }
457
458 G4Event* anEvent = new G4Event(i_event);
459
460 if(storeRandomNumberStatusToG4Event==1 || storeRandomNumberStatusToG4Event==3)
461 {
462 std::ostringstream oss;
463 G4Random::saveFullState(oss);
464 randomNumberStatusForThisEvent = oss.str();
465 anEvent->SetRandomNumberStatus(randomNumberStatusForThisEvent);
466 }
467
468 if(storeRandomNumberStatus) {
469 G4String fileN = "currentEvent";
470 if ( rngStatusEventsFlag ) {
471 std::ostringstream os;
472 os << "run" << currentRun->GetRunID() << "evt" << anEvent->GetEventID();
473 fileN = os.str();
474 }
475 StoreRNGStatus(fileN);
476 }
477
478 if(printModulo > 0 && anEvent->GetEventID()%printModulo == 0 )
479 { G4cout << "--> Event " << anEvent->GetEventID() << " starts." << G4endl; }
480 userPrimaryGeneratorAction->GeneratePrimaries(anEvent);
481 return anEvent;
482}

498void G4RunManager::RunTermination()
499{
500 if(!fakeRun)
501 {
502 CleanUpUnnecessaryEvents(0);
503 if(userRunAction) userRunAction->EndOfRunAction(currentRun);
504 G4VPersistencyManager* fPersM = G4VPersistencyManager::GetPersistencyManager();
505 if(fPersM) fPersM->Store(currentRun);
506 // write & close analysis output
507 if (isScoreNtupleWriter)
508 { G4VScoreNtupleWriter::Instance()->Write(); }
509 runIDCounter++;
510 }
511
512 kernel->RunTermination();
513}

515void G4RunManager::CleanUpPreviousEvents()
516{
517 // Delete all events carried over from previous run.
518 // This method is invoked at the beginning of the next run
519 // or from the destructor of G4RunManager at the very end of
520 // the program.
521 // N.B. If ToBeKept() is true, the pointer of this event is
522 // kept in G4Run of the previous run, and deleted along with
523 // the deletion of G4Run.
524
525 std::list<G4Event*>::iterator evItr = previousEvents->begin();
526 while(evItr!=previousEvents->end())
527 {
528 G4Event* evt = *evItr;
529 if(evt && !(evt->ToBeKept())) delete evt;
530 evItr = previousEvents->erase(evItr);
531 }
532}
533
534void G4RunManager::CleanUpUnnecessaryEvents(G4int keepNEvents)
535{
536 // Delete events that are no longer necessary for post
537 // processing such as visualization.
538 // N.B. If ToBeKept() is true, the pointer of this event is
539 // kept in G4Run of the previous run, and deleted along with
540 // the deletion of G4Run.
541
542 std::list<G4Event*>::iterator evItr = previousEvents->begin();
543 while(evItr!=previousEvents->end())
544 {
545 if(G4int(previousEvents->size()) <= keepNEvents) return;
546
547 G4Event* evt = *evItr;
548 if(evt)
549 {
550 if(evt->GetNumberOfGrips()==0)
551 {
552 if(!(evt->ToBeKept())) delete evt;
553 evItr = previousEvents->erase(evItr);
554 }
555 else
556 { evItr++; }
557 }
558 else
559 { evItr = previousEvents->erase(evItr); }
560 }
561}
562
563void G4RunManager::StackPreviousEvent(G4Event* anEvent)
564{
565 if(anEvent->ToBeKept()) currentRun->StoreEvent(anEvent);
566
567 if(n_perviousEventsToBeStored==0)
568 {
569 if(anEvent->GetNumberOfGrips()==0)
570 { if(!(anEvent->ToBeKept())) delete anEvent; }
571 else
572 { previousEvents->push_back(anEvent); }
573 }
574
575 CleanUpUnnecessaryEvents(n_perviousEventsToBeStored);
576}

578void G4RunManager::Initialize()
579{
580 G4StateManager* stateManager = G4StateManager::GetStateManager();
581 G4ApplicationState currentState = stateManager->GetCurrentState();
582 if(currentState!=G4State_PreInit && currentState!=G4State_Idle)
583 {
584 G4cerr << "Illegal application state - "
585 << "G4RunManager::Initialize() ignored." << G4endl;
586 return;
587 }
588
589 stateManager->SetNewState(G4State_Init);
590 if(!geometryInitialized) InitializeGeometry();
591 if(!physicsInitialized) InitializePhysics();
592 initializedAtLeastOnce = true;
593 if(stateManager->GetCurrentState()!=G4State_Idle)
594 { stateManager->SetNewState(G4State_Idle); }
595}

597void G4RunManager::InitializeGeometry()
598{
599 if(!userDetector)
600 {
601 G4Exception("G4RunManager::InitializeGeometry", "Run0033",
602 FatalException, "G4VUserDetectorConstruction is not defined!");
603 return;
604 }
605
606 if(verboseLevel>1) G4cout << "userDetector->Construct() start." << G4endl;
607
608 G4StateManager* stateManager = G4StateManager::GetStateManager();
609 G4ApplicationState currentState = stateManager->GetCurrentState();
610 if(currentState==G4State_PreInit || currentState==G4State_Idle)
611 { stateManager->SetNewState(G4State_Init); }
612 kernel->DefineWorldVolume(userDetector->Construct(),false);
613 userDetector->ConstructSDandField();
614 nParallelWorlds = userDetector->ConstructParallelGeometries();
615 userDetector->ConstructParallelSD();
616 kernel->SetNumberOfParallelWorld(nParallelWorlds);
617 geometryInitialized = true;
618 stateManager->SetNewState(currentState);
619}

641void G4RunManager::AbortRun(G4bool softAbort)
642{
643 // This method is valid only for GeomClosed or EventProc state
644 G4ApplicationState currentState =
645 G4StateManager::GetStateManager()->GetCurrentState();
646 if(currentState==G4State_GeomClosed || currentState==G4State_EventProc)
647 {
648 runAborted = true;
649 if(currentState==G4State_EventProc && !softAbort)
650 {
651 currentEvent->SetEventAborted();
652 eventManager->AbortCurrentEvent();
653 }
654 }
655 else
656 {
657 G4cerr << "Run is not in progress. AbortRun() ignored." << G4endl;
658 }
659}

661void G4RunManager::AbortEvent()
662{
663 // This method is valid only for EventProc state

665 G4StateManager::GetStateManager()->GetCurrentState();
666 if(currentState==G4State_EventProc)
667 {
668 currentEvent->SetEventAborted();
669 eventManager->AbortCurrentEvent();
670 }
671 else
672 {
673 G4cerr << "Event is not in progress. AbortEevnt() ignored." << G4endl;
674 }

683void G4RunManager::rndmSaveThisRun()
684{
685 G4int runNumber = 0;
686 if(currentRun) runNumber = currentRun->GetRunID();
687 if(!storeRandomNumberStatus) {
688 G4cerr << "Warning from G4RunManager::rndmSaveThisRun():"
689 << " Random number status was not stored prior to this run."
690 << G4endl << "Command ignored." << G4endl;
691 return;
692 }
693
694 G4String fileIn = randomNumberStatusDir + "currentRun.rndm";
695
696 std::ostringstream os;
697 os << "run" << runNumber << ".rndm" << '\0';
698 G4String fileOut = randomNumberStatusDir + os.str();
699
700 G4String copCmd = "/control/shell cp "+fileIn+" "+fileOut;
701 G4UImanager::GetUIpointer()->ApplyCommand(copCmd);
702 if(verboseLevel>0) G4cout << "currentRun.rndm is copied to file: " << fileOut << G4endl;
703}

705void G4RunManager::rndmSaveThisEvent()
706{
707 if(!storeRandomNumberStatus || currentEvent == 0) {
708 G4cerr << "Warning from G4RunManager::rndmSaveThisEvent():"
709 << " there is no currentEvent or its RandomEngineStatus is not available."
710 << G4endl << "Command ignored." << G4endl;
711 return;
712 }
713
714 G4String fileIn = randomNumberStatusDir + "currentEvent.rndm";
715
716 std::ostringstream os;
717 os << "run" << currentRun->GetRunID() << "evt" << currentEvent->GetEventID()
718 << ".rndm" << '\0';
719 G4String fileOut = randomNumberStatusDir + os.str();
720
721 G4String copCmd = "/control/shell cp "+fileIn+" "+fileOut;
722 G4UImanager::GetUIpointer()->ApplyCommand(copCmd);
723 if(verboseLevel>0) G4cout << "currentEvent.rndm is copied to file: " << fileOut << G4endl;
724}

763void G4RunManager::ConstructScoringWorlds()
764{
765 G4ScoringManager* ScM = G4ScoringManager::GetScoringManagerIfExist();
766 if(!ScM) return;
767
768 G4int nPar = ScM->GetNumberOfMesh();
769 if(nPar<1) return;
770
771 TIMEMORY_AUTO_TIMER("");
772 G4ParticleTable::G4PTblDicIterator* theParticleIterator
773 = G4ParticleTable::GetParticleTable()->GetIterator();
774 for(G4int iw=0;iw<nPar;iw++)
775 {
776 G4VScoringMesh* mesh = ScM->GetMesh(iw);
777 if(fGeometryHasBeenDestroyed) mesh->GeometryHasBeenDestroyed();
778
779 G4VPhysicalVolume* pWorld
780 = G4TransportationManager::GetTransportationManager()
781 ->IsWorldExisting(ScM->GetWorldName(iw));
782 if(!pWorld)
783 {
784 pWorld = G4TransportationManager::GetTransportationManager()
785 ->GetParallelWorld(ScM->GetWorldName(iw));
786 pWorld->SetName(ScM->GetWorldName(iw));
787
788 G4ParallelWorldProcess* theParallelWorldProcess
789 = mesh->GetParallelWorldProcess();
790 if(theParallelWorldProcess)
791 { theParallelWorldProcess->SetParallelWorld(ScM->GetWorldName(iw)); }
792 else
793 {
794 theParallelWorldProcess = new G4ParallelWorldProcess(ScM->GetWorldName(iw));
795 mesh->SetParallelWorldProcess(theParallelWorldProcess);
796 theParallelWorldProcess->SetParallelWorld(ScM->GetWorldName(iw));
797
798 theParticleIterator->reset();
799 while( (*theParticleIterator)() ){
800 G4ParticleDefinition* particle = theParticleIterator->value();
801 G4ProcessManager* pmanager = particle->GetProcessManager();
802 if(pmanager)
803 {
804 pmanager->AddProcess(theParallelWorldProcess);
805 if(theParallelWorldProcess->IsAtRestRequired(particle))
806 { pmanager->SetProcessOrdering(theParallelWorldProcess, idxAtRest, 9900); }
807 pmanager->SetProcessOrderingToSecond(theParallelWorldProcess, idxAlongStep);
808 pmanager->SetProcessOrdering(theParallelWorldProcess, idxPostStep, 9900);
809 }
810 }
811 }
812 }
813
814 mesh->Construct(pWorld);
815 }
816
817 GeometryHasBeenModified();
818}

956void G4RunManager::ReinitializeGeometry(G4bool destroyFirst, G4bool prop)
957{
958 if(destroyFirst && G4Threading::IsMasterThread())
959 {
960 if(verboseLevel>0)
961 {
962 G4cout << "#### Assemblies, Volumes and Solids Stores are wiped out."
963 << G4endl;
964 }
965 G4GeometryManager::GetInstance()->OpenGeometry();
966 G4AssemblyStore::GetInstance()->Clean();
967 G4PhysicalVolumeStore::GetInstance()->Clean();
968 G4LogicalVolumeStore::GetInstance()->Clean();
969 G4SolidStore::GetInstance()->Clean();
970
971 // remove all logical volume pointers from regions
972 // exception: world logical volume pointer must be kept
973 G4RegionStore* regionStore = G4RegionStore::GetInstance();
974 std::vector<G4Region*>::iterator rItr;
975 for(rItr = regionStore->begin();rItr != regionStore->end(); rItr++)
976 {
977 if((*rItr)->GetName()=="DefaultRegionForTheWorld") continue;
978 //if((*rItr)->GetName()=="DefaultRegionForParallelWorld") continue;
979 std::vector<G4LogicalVolume*>::iterator lvItr
980 = (*rItr)->GetRootLogicalVolumeIterator();
981 for(size_t iRLV = 0;iRLV < (*rItr)->GetNumberOfRootVolumes(); iRLV++)
982 {
983 (*rItr)->RemoveRootLogicalVolume(*lvItr,false);
984 lvItr++;
985 }
986 if(verboseLevel>0)
987 { G4cout<<"#### Region <"<<(*rItr)->GetName()<<"> is cleared."<<G4endl; }
988 }
989
990 // clear transportation manager
991 fGeometryHasBeenDestroyed = true;
992 G4TransportationManager::GetTransportationManager()->ClearParallelWorlds();
993 }
994 if(prop)
995 { G4UImanager::GetUIpointer()->ApplyCommand("/run/reinitializeGeometry"); }
996 else
997 {
998 kernel->GeometryHasBeenModified();
999 geometryInitialized = false;
1000 // Notify the VisManager as well
1001 if(G4Threading::IsMasterThread())
1002 {
1003 G4VVisManager* pVVisManager = G4VVisManager::GetConcreteInstance();
1004 if(pVVisManager) pVVisManager->GeometryHasChanged();
1005 }
1006 }
