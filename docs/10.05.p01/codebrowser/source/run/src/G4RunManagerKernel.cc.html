152: physicsList(0),currentWorld(0),
153geometryInitialized(false),physicsInitialized(false),
154geometryToBeOptimized(true),
155physicsNeedsToBeReBuilt(true),verboseLevel(0),
156numberOfParallelWorld(0),geometryNeedsToBeClosed(true),
157 numberOfStaticAllocators(0)
158{
159//This version of the constructor should never be called in sequential mode!
160#ifndef G4MULTITHREADED
161 G4ExceptionDescription msg;
162 msg<<"Geant4 code is compiled without multi-threading support (-DG4MULTITHREADED is set to off).";
163 msg<<" This type of RunManagerKernel can only be used in mult-threaded applications.";
164 G4Exception("G4RunManagerKernel::G4RunManagerKernel(G4bool)","Run0105",FatalException,msg);
165#endif
166
167#ifdef G4FPE_DEBUG
168 if ( G4Threading::IsMasterThread() ) {
169 InvalidOperationDetection();
170 }
171#endif
172
173 defaultExceptionHandler = new G4ExceptionHandler();
174 if(fRunManagerKernel)
175 {
176 G4Exception("G4RunManagerKernel::G4RunManagerKernel()","Run0001",
177 FatalException,"More than one G4RunManagerKernel is constructed.");
178 }
179 fRunManagerKernel = this;
180 // construction of Geant4 kernel classes
181 eventManager = new G4EventManager();
182
183 switch(rmkType)
184 {
185 case masterRMK:
186 //Master thread behvior
187 defaultRegion = new G4Region("DefaultRegionForTheWorld"); // deleted by store
188 defaultRegionForParallelWorld = new G4Region("DefaultRegionForParallelWorld"); // deleted by store
189 defaultRegion->SetProductionCuts(
190 G4ProductionCutsTable::GetProductionCutsTable()->GetDefaultProductionCuts());
191 defaultRegionForParallelWorld->SetProductionCuts(
192 G4ProductionCutsTable::GetProductionCutsTable()->GetDefaultProductionCuts());
193 break;
194 case workerRMK:
195 //Worker thread behavior
196 defaultRegion = G4RegionStore::GetInstance()->GetRegion("DefaultRegionForTheWorld", true);
197 defaultRegionForParallelWorld
198 = G4RegionStore::GetInstance()->GetRegion("DefaultRegionForParallelWorld", true);
199 break;
200 default:
201 defaultRegion = 0;
202 defaultRegionForParallelWorld = 0;
203 G4ExceptionDescription msgx;
204 msgx<<" This type of RunManagerKernel can only be used in mult-threaded applications.";
205 G4Exception("G4RunManagerKernel::G4RunManagerKernel(G4bool)","Run0106",FatalException,msgx);
206 }
207 runManagerKernelType = rmkType;
208
209 // set the initial application state

211
212 // version banner
213 G4String vs = G4Version;
214 vs = vs.substr(1,vs.size()-2);
215 switch(rmkType)
216 {
217 case masterRMK:
218 versionString = " Geant4 version ";
219 versionString += vs;
220 versionString += " ";
221 versionString += G4Date;
222 G4cout << G4endl
223 << "**************************************************************" << G4endl

251void G4RunManagerKernel::SetupDefaultRegion()
252{
253 if(runManagerKernelType==workerRMK) return;
254
255 // Remove old world logical volume from the default region, if exist
256 if(defaultRegion->GetNumberOfRootVolumes())
257 {
258 if(defaultRegion->GetNumberOfRootVolumes()>size_t(1))
259 {
260 G4Exception("G4RunManager::SetupDefaultRegion",
261 "Run0005",
262 FatalException,
263 "Default world region should have a unique logical volume.");
264 }
265 std::vector<G4LogicalVolume*>::iterator lvItr
266 = defaultRegion->GetRootLogicalVolumeIterator();
267 defaultRegion->RemoveRootLogicalVolume(*lvItr,false);
268 if(verboseLevel>1) G4cout
269 << "Obsolete world logical volume is removed from the default region." << G4endl;
270 }

274G4RunManagerKernel::~G4RunManagerKernel()
275{
276 G4StateManager* pStateManager = G4StateManager::GetStateManager();
277 // set the application state to the quite state
278 if(pStateManager->GetCurrentState()!=G4State_Quit)
279 {
280 if(verboseLevel>0) G4cout << "G4 kernel has come to Quit state." << G4endl;
281 pStateManager->SetNewState(G4State_Quit);
282 }
283
284 // open geometry for deletion
285 G4GeometryManager::GetInstance()->OpenGeometry();
286
287 // deletion of Geant4 kernel classes
288 G4ParallelWorldProcessStore* pwps = G4ParallelWorldProcessStore::GetInstanceIfExist();
289 if(pwps) delete pwps;
290 G4SDManager* fSDM = G4SDManager::GetSDMpointerIfExist();
291 if(fSDM)
292 {
293 delete fSDM;
294 if(verboseLevel>1) G4cout << "G4SDManager deleted." << G4endl;
295 }
296 delete eventManager;
297 if(verboseLevel>1) G4cout << "EventManager deleted." << G4endl;
298
299 G4UnitDefinition::ClearUnitsTable();
300 if(verboseLevel>1) G4cout << "Units table cleared." << G4endl;
301
302 // deletion of path-finder field-manager store, geometry and transportation manager
303 G4PathFinder* pFinder = G4PathFinder::GetInstanceIfExist();
304 if (pFinder) delete pFinder;
305 G4FieldManagerStore* fmStore = G4FieldManagerStore::GetInstanceIfExist();
306 if (fmStore) delete fmStore;
307 G4GeometryManager* gManager = G4GeometryManager::GetInstanceIfExist();
308 if (gManager) delete gManager;
309 G4TransportationManager* tManager = G4TransportationManager::GetInstanceIfExist();
310 if (tManager)
311 {
312 delete tManager;
313 if(verboseLevel>1) G4cout << "TransportationManager deleted." << G4endl;
314 }
315
316 // deletion of navigation levels
317 if(verboseLevel>1) G4NavigationHistoryPool::GetInstance()->Print();
318 delete G4NavigationHistoryPool::GetInstance();
319
320 // deletion of G4RNGHelper singleton
321 if(runManagerKernelType!=workerRMK)
322 {
323 G4RNGHelper * rngHelper = G4RNGHelper::GetInstanceIfExist();
324 if(rngHelper)
325 {
326 delete rngHelper;
327 if(verboseLevel>1) G4cout << "G4RNGHelper object is deleted." << G4endl;
328 }
329 }
330
331 // deletion of allocators
332 G4AllocatorList* allocList = G4AllocatorList::GetAllocatorListIfExist();
333 if(allocList)
334 {
335 allocList->Destroy(numberOfStaticAllocators,verboseLevel);
336 delete allocList;
337 if(verboseLevel>1) G4cout << "G4Allocator objects are deleted." << G4endl;
338 }
339
340 G4UImanager* pUImanager = G4UImanager::GetUIpointer();
341 if((runManagerKernelType==workerRMK) && (verboseLevel>0))
342 {
343 G4cout << "Thread-local UImanager is to be deleted." << G4endl
344 << "There should not be any thread-local G4cout/G4cerr hereafter."
345 << G4endl;
346 verboseLevel = 0;
347 }
348 if(pUImanager) delete pUImanager;
349 if(verboseLevel>1) G4cout << "UImanager deleted." << G4endl;
350
351 delete pStateManager;
352 if(verboseLevel>1) G4cout << "StateManager deleted." << G4endl;
353 delete defaultExceptionHandler;
354 if(verboseLevel>0) G4cout << "RunManagerKernel is deleted. Good bye :)" << G4endl;
355 fRunManagerKernel = 0;
356}

358void G4RunManagerKernel::WorkerUpdateWorldVolume()
359{
360 G4MTRunManager* masterRM = G4MTRunManager::GetMasterRunManager();
361 G4TransportationManager* transM = G4TransportationManager::GetTransportationManager();
362 G4MTRunManager::masterWorlds_t masterWorlds= masterRM->GetMasterWorlds();
363 G4MTRunManager::masterWorlds_t::iterator itrMW = masterWorlds.begin();
364 for(;itrMW!=masterWorlds.end();itrMW++)
365 {
366 G4VPhysicalVolume* wv = (*itrMW).second;
367 G4VPhysicalVolume* pWorld
368 = G4TransportationManager::GetTransportationManager()
369 ->IsWorldExisting(wv->GetName());
370 if(!pWorld)
371 { transM->RegisterWorld(wv); }
372 }

375void G4RunManagerKernel::WorkerDefineWorldVolume(G4VPhysicalVolume* worldVol,
376 G4bool topologyIsChanged)
377{
378 G4StateManager* stateManager = G4StateManager::GetStateManager();
379 G4ApplicationState currentState = stateManager->GetCurrentState();
380 if(currentState!=G4State_Init)
381 {
382 if(!(currentState==G4State_Idle||currentState==G4State_PreInit))
383 {
384 G4cout << "Current application state is "
385 << stateManager->GetStateString(currentState) << G4endl;
386 G4Exception("G4RunManagerKernel::DefineWorldVolume",
387 "DefineWorldVolumeAtIncorrectState",
388 FatalException,
389 "Geant4 kernel is not Init state : Method ignored.");
390 return;
391 } else {
392 //G4Exception("G4RunManagerKernel::DefineWorldVolume",
393 // "DefineWorldVolumeAtIncorrectState",
394 // JustWarning,
395 // "Geant4 kernel is not Init state : Assuming Init state.");
396 //G4cout<<"Warning : Geant4 kernel is not Init state : Assuming Init state."
397 // <<G4endl;
398 stateManager->SetNewState(G4State_Init);
399 }
400 }
401
402 currentWorld = worldVol;
403 G4MTRunManager* masterRM = G4MTRunManager::GetMasterRunManager();
404 G4TransportationManager* transM = G4TransportationManager::GetTransportationManager();
405 G4MTRunManager::masterWorlds_t masterWorlds= masterRM->GetMasterWorlds();
406 G4MTRunManager::masterWorlds_t::iterator itrMW = masterWorlds.begin();
407 for(;itrMW!=masterWorlds.end();itrMW++)
408 {
409 if((*itrMW).first == 0)
410 {
411 if((*itrMW).second != currentWorld)
412 {
413 G4Exception("G4RunManagerKernel::WorkerDefineWorldVolume","RUN3091",
414 FatalException,"Mass world is inconsistent");
415 }
416 transM->SetWorldForTracking((*itrMW).second);
417 }
418 else
419 {
420 transM->RegisterWorld((*itrMW).second);
421 }
422 }
423
424 if(topologyIsChanged) geometryNeedsToBeClosed = true;
425
426 // Notify the VisManager as well
427 if(G4Threading::IsMasterThread())
428 {
429 G4VVisManager* pVVisManager = G4VVisManager::GetConcreteInstance();
430 if(pVVisManager) pVVisManager->GeometryHasChanged();
431 }
432
433 geometryInitialized = true;
434 stateManager->SetNewState(currentState);
435 if(physicsInitialized && currentState!=G4State_Idle)
436 { stateManager->SetNewState(G4State_Idle); }
437}

439void G4RunManagerKernel::DefineWorldVolume(G4VPhysicalVolume* worldVol,
440 G4bool topologyIsChanged)
441{
442 G4StateManager* stateManager = G4StateManager::GetStateManager();
443 G4ApplicationState currentState = stateManager->GetCurrentState();
444 if(currentState!=G4State_Init)
445 {
446 if(!(currentState==G4State_Idle||currentState==G4State_PreInit))
447 {
448 G4cout << "Current application state is "
449 << stateManager->GetStateString(currentState) << G4endl;
450 G4Exception("G4RunManagerKernel::DefineWorldVolume",
451 "DefineWorldVolumeAtIncorrectState",
452 FatalException,
453 "Geant4 kernel is not Init state : Method ignored.");
454 return;
455 } else {
456 //G4Exception("G4RunManagerKernel::DefineWorldVolume",
457 // "DefineWorldVolumeAtIncorrectState",
458 // JustWarning,
459 // "Geant4 kernel is not Init state : Assuming Init state.");
460 //G4cout<<"Warning : Geant4 kernel is not Init state : Assuming Init state."
461 // <<G4endl;
462 stateManager->SetNewState(G4State_Init);
463 }
464 }
465
466 // The world volume MUST NOT have a region defined by the user
467 if(worldVol->GetLogicalVolume()->GetRegion())
468 {
469 if(worldVol->GetLogicalVolume()->GetRegion()!=defaultRegion)
470 {
471 G4ExceptionDescription ED;
472 ED << "The world volume has a user-defined region <"
473 << worldVol->GetLogicalVolume()->GetRegion()->GetName()
474 << ">." << G4endl;
475 ED << "World would have a default region assigned by RunManagerKernel."
476 << G4endl;
477 G4Exception("G4RunManager::DefineWorldVolume",
478 "Run0004", FatalException, ED);
479 }
480 }
481
482 SetupDefaultRegion();
483
484 // Accept the world volume
485 currentWorld = worldVol;
486
487 // Set the default region to the world
488
489 G4LogicalVolume* worldLog = currentWorld->GetLogicalVolume();
490 worldLog->SetRegion(defaultRegion);
491 defaultRegion->AddRootLogicalVolume(worldLog);
492 if(verboseLevel>1) G4cout << worldLog->GetName()
493 << " is registered to the default region." << G4endl;
494
495 // Set the world volume, notify the Navigator and reset its state
496 G4TransportationManager::GetTransportationManager()
497 ->SetWorldForTracking(currentWorld);
498 if(topologyIsChanged) geometryNeedsToBeClosed = true;
499
500 // Notify the VisManager as well
501 if(G4Threading::IsMasterThread())
502 {
503 G4VVisManager* pVVisManager = G4VVisManager::GetConcreteInstance();
504 if(pVVisManager) pVVisManager->GeometryHasChanged();
505 }
506
507 geometryInitialized = true;
508 stateManager->SetNewState(currentState);
509 if(physicsInitialized && currentState!=G4State_Idle)
510 { stateManager->SetNewState(G4State_Idle); }
511}

540void G4RunManagerKernel::SetupPhysics()
541{
542 G4ParticleTable::GetParticleTable()->SetReadiness();
543
544 physicsList->ConstructParticle();
545
546 // For sanity reason
547 G4Geantino::GeantinoDefinition();
548 G4ParticleDefinition* gion = G4ParticleTable::GetParticleTable()->GetGenericIon();
549 if(gion)
550 { G4IonConstructor::ConstructParticle(); }
551 G4ParticleTable::GetParticleTable()->GetIonTable()->InitializeLightIons();
552
553 G4ParticleTable::G4PTblDicIterator* pItr = G4ParticleTable::GetParticleTable()->GetIterator();
554 pItr->reset();
555 while( (*pItr)() )
556 {
557 G4ParticleDefinition* particle = pItr->value();
558 if(!(particle->IsGeneralIon())) particle->SetParticleDefinitionID();
559 }
560
561 if(gion)
562 {
563 G4int gionId = gion->GetParticleDefinitionID();
564 pItr->reset(false);
565 while( (*pItr)() )
566 {
567 G4ParticleDefinition* particle = pItr->value();
568 if(particle->IsGeneralIon()) particle->SetParticleDefinitionID(gionId);
569 }
570 }
571#ifdef G4MULTITHREADED
572 G4UnitDefinition::GetUnitsTable().Synchronize();
573#endif
574}

590 G4Exception("G4RunManagerKernel::InitializePhysics",
591 "InitializePhysicsIncorrectState",
592 FatalException,

596 //G4Exception("G4RunManagerKernel::DefineWorldVolume",

608 G4Exception("G4RunManagerKernel::InitializePhysics",
609 "Run0012", FatalException,
610 "G4VUserPhysicsList is not defined");
611 return;
612 }
613
614 if(verboseLevel>1) G4cout << "physicsList->Construct() start." << G4endl;
615 if(numberOfParallelWorld>0) physicsList->UseCoupledTransportation();
616 physicsList->Construct();

625 physicsList->SetCuts();
626
627 }
628 CheckRegions();
629 l.unlock();
630

632// static G4bool createIsomerOnlyOnce = false;
633// if(G4Threading::IsMultithreadedApplication() && G4Threading::IsMasterThread())
634// {
635// if(!createIsomerOnlyOnce)
636// {
637// createIsomerOnlyOnce = true;

639// if(gion)

641// G4ParticleTable::GetParticleTable()->GetIonTable()->CreateAllIsomer();
642// PropagateGenericIonID();
643// }
644// }
645// }
646*********************/
647
648 physicsInitialized = true;
649#ifdef G4MULTITHREADED
650 G4UnitDefinition::GetUnitsTable().Synchronize();
651#endif
652 stateManager->SetNewState(currentState);
653 if(geometryInitialized && currentState!=G4State_Idle)
654 { stateManager->SetNewState(G4State_Idle); }

657G4bool G4RunManagerKernel::RunInitialization(G4bool fakeRun)
658{
659 TIMEMORY_AUTO_TIMER("");
660 G4StateManager* stateManager = G4StateManager::GetStateManager();
661 G4ApplicationState currentState = stateManager->GetCurrentState();
662
663 if(!geometryInitialized)
664 {
665 G4Exception("G4RunManagerKernel::RunInitialization",
666 "Run0021",
667 JustWarning,
668 "Geometry has not yet initialized : method ignored.");
669 return false;
670 }
671
672 if(!physicsInitialized)
673 {
674 G4Exception("G4RunManagerKernel::RunInitialization",
675 "Run0022",
676 JustWarning,
677 "Physics has not yet initialized : method ignored.");
678 return false;
679 }
680
681 if( currentState != G4State_Idle )
682 {
683 G4Exception("G4RunManagerKernel::RunInitialization",
684 "Run0023",
685 JustWarning,
686 "Geant4 kernel not in Idle state : method ignored.");
687 return false;
688 }
689
690 if(geometryNeedsToBeClosed) CheckRegularGeometry();
691
692 stateManager->SetNewState(G4State_Init);
693 PropagateGenericIonID();
694 SetupShadowProcess();
695 UpdateRegion();
696 BuildPhysicsTables(fakeRun);
697
698 if(geometryNeedsToBeClosed)
699 {
700 ResetNavigator();
701 // CheckRegularGeometry();
702 // Notify the VisManager as well
703 if(G4Threading::IsMasterThread())
704 {
705 G4VVisManager* pVVisManager = G4VVisManager::GetConcreteInstance();
706 if(pVVisManager) pVVisManager->GeometryHasChanged();
707 }
708 }
709
710 GetPrimaryTransformer()->CheckUnknown();
711
712#ifdef G4MULTITHREADED
713 G4UnitDefinition::GetUnitsTable().Synchronize();
714#endif
715 stateManager->SetNewState(G4State_Idle);
716 stateManager->SetNewState(G4State_GeomClosed);
717 return true;
718}

809void G4RunManagerKernel::CheckRegions()
810{
811 G4TransportationManager* transM = G4TransportationManager::GetTransportationManager();
812 size_t nWorlds = transM->GetNoWorlds();
813 std::vector<G4VPhysicalVolume*>::iterator wItr;
814 for(size_t i=0;i<G4RegionStore::GetInstance()->size();i++)
815 {
816 G4Region* region = (*(G4RegionStore::GetInstance()))[i];
817
818 //Let each region have a pointer to the world volume where it belongs to.
819 //G4Region::SetWorld() checks if the region belongs to the given world and set it
820 //only if it does. Thus, here we go through all the registered world volumes.
821 region->SetWorld(0); // reset
822 region->UsedInMassGeometry(false);
823 region->UsedInParallelGeometry(false);
824 wItr = transM->GetWorldsIterator();
825 for(size_t iw=0;iw<nWorlds;iw++)
826 {
827 if(region->BelongsTo(*wItr))
828 {
829 if(*wItr==currentWorld)
830 { region->UsedInMassGeometry(true); }
831 else
832 { region->UsedInParallelGeometry(true); }
833 }
834 region->SetWorld(*wItr);
835 wItr++;
836 }
837
838 G4ProductionCuts* cuts = region->GetProductionCuts();
839 if(!cuts)
840 {
841 if(region->IsInMassGeometry())
842 {
843 G4cout << "Warning : Region <" << region->GetName()
844 << "> does not have specific production cuts," << G4endl
845 << "even though it appears in the current tracking world." << G4endl;
846 G4cout << "Default cuts are used for this region." << G4endl;
847 }
848
849 if(region->IsInMassGeometry()||region->IsInParallelGeometry())
850 {
851 region->SetProductionCuts(
852 G4ProductionCutsTable::GetProductionCutsTable()
853 ->GetDefaultProductionCuts());
854 }
855 }
856 }
857
858 //
859 // If a parallel world has no region, set default region for parallel world
860 //
861
862 wItr = transM->GetWorldsIterator();
863 for(size_t iw=0;iw<nWorlds;iw++)
864 {
865 //G4cout << "+++ " << (*wItr)->GetName() << G4endl;
866 if(*wItr!=currentWorld)
867 {
868 G4LogicalVolume* pwLogical = (*wItr)->GetLogicalVolume();
869 if(!(pwLogical->GetRegion()))
870 {
871 pwLogical->SetRegion(defaultRegionForParallelWorld);
872 defaultRegionForParallelWorld->AddRootLogicalVolume(pwLogical);
873 //G4cout << "+++++ defaultRegionForParallelWorld is set to "
874 // << (*wItr)->GetName() << " +++++" << G4endl;
875 }
876 }
877 wItr++;
878 }
879
880}

1028void G4RunManagerKernel::SetupShadowProcess() const
1029{
1030 G4ParticleTable* theParticleTable = G4ParticleTable::GetParticleTable();
1031 G4ParticleTable::G4PTblDicIterator* theParticleIterator = theParticleTable->GetIterator();
1032 theParticleIterator->reset();
1033 //loop on particles and get process manager from there list of processes
1034 while((*theParticleIterator)())
1035 {
1036 G4ParticleDefinition* pd = theParticleIterator->value();
1037 G4ProcessManager* pm = pd->GetProcessManager();
1038 if(pm)
1039 {
1040 G4ProcessVector& procs = *(pm->GetProcessList());
1041 for ( G4int idx = 0 ; idx<procs.size() ; ++idx)
1042 {
1043 const G4VProcess* masterP = procs[idx]->GetMasterProcess();
1044 if ( ! masterP )
1045 {
1046 //Process does not have an associated shadow master process
1047 //We are in master mode or sequential
1048 procs[idx]->SetMasterProcess(const_cast<G4VProcess*>(procs[idx]));
1049 }
1050 }
1051 }
