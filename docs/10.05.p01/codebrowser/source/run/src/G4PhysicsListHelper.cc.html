97////////////////////////////////////////////////////////
98void G4PhysicsListHelper::CheckParticleList() const
99{
100 bool isElectron = false;
101 bool isPositron = false;
102 bool isGamma = false;
103 bool isProton = false;
104 bool isGenericIon = false;
105 bool isAnyIon = false;
106 bool isAnyChargedBaryon = false;
107 bool isEmProc = false;
108
109 // loop over all particles in G4ParticleTable
110 aParticleIterator->reset();
111 while( (*aParticleIterator)() ){
112 G4ParticleDefinition* particle = aParticleIterator->value();
113 G4String name = particle->GetParticleName();
114 // check if any EM process exists
115 if (!isEmProc) {
116 G4ProcessVector* list = particle->GetProcessManager()->GetProcessList();
117 for (int idx=0; idx<list->size(); idx++){
118 isEmProc = ((*list)[idx])->GetProcessType() == fElectromagnetic;
119 if (isEmProc) break;
120 }
121 }
122
123 if ( name == "e-") isElectron = true;
124 else if ( name == "e+") isPositron = true;
125 else if ( name == "gamma") isGamma = true;
126 else if ( name == "GenericIon") isGenericIon = true;
127 else if ( name == "proton") isProton = true;
128 else if ( particle->GetParticleType() == "nucleus") isAnyIon = true;
129 else if ( particle->GetParticleType() == "baryon") {
130 if ( particle->GetPDGCharge() != 0.0 ) isAnyChargedBaryon = true;
131 }
132 }
133
134 if (!isEmProc) return;
135
136 // RULE 1
137 // e+, e- and gamma should exist
138 // if one of them exist
139 bool isEmBasic = isElectron || isPositron || isGamma;
140 bool isMissingEmBasic = !isElectron || !isPositron || !isGamma;
141 if (isEmBasic && isMissingEmBasic) {
142 G4String missingName="";
143 if (!isElectron) missingName += "e- ";
144 if (!isPositron) missingName += "e+ ";
145 if (!isGamma) missingName += "gamma ";
146
147#ifdef G4VERBOSE
148 if (verboseLevel >0){
149 G4cout << "G4PhysicsListHelper::CheckParticleList: "
150 << missingName << " do not exist " << G4endl;
151 G4cout << " These particle are necessary for basic EM processes"
152 << G4endl;
153 }
154#endif
155 G4Exception("G4PhysicsListHelper::CheckParticleList",
156 "Run0101", FatalException,
157 "Missing EM basic particle");
158 }
159
160 // RULE 2
161 // proton should exist
162 // if any other charged baryon exist
163 if (!isProton && isAnyChargedBaryon) {
164 G4String missingName="proton ";
165
166#ifdef G4VERBOSE
167 if (verboseLevel >0){
168 G4cout << "G4PhysicsListHelper::CheckParticleList: "
169 << missingName << " does not exist "<< G4endl;
170 G4cout << " Proton is necessary for EM baryon processes" << G4endl;
171 }
172#endif
173 missingName += " should be created ";
174 G4Exception("G4PhysicsListHelper::CheckParticleList",
175 "Run0102", FatalException,
176 "Missing Proton");
177 }
178
179 // RULE 3
180 // GenericIonn should exist
181 // if any other ion exist
182 if (!isGenericIon && isAnyIon) {
183 G4String missingName="GenericIon ";
184
185#ifdef G4VERBOSE
186 if (verboseLevel >0){
187 G4cout << "G4PhysicsListHelper::CheckParticleList: "
188 << missingName << " does not exist "<< G4endl;
189 G4cout << " GenericIon should be created if any ion is necessary" << G4endl;
190 }
191#endif
192 G4Exception("G4PhysicsListHelper::CheckParticleList",
193 "Run0103", FatalException,
194 "Missing GenericIon");
195 }
196
197}

206void G4PhysicsListHelper::AddTransportation()
207{
208 G4int verboseLevelTransport = 0;
209
210#ifdef G4VERBOSE
211 if (verboseLevel >2){
212 G4cout << "G4PhysicsListHelper::AddTransportation() "<< G4endl;
213 }
214#endif
215
216 G4int nParaWorld =
217 G4RunManagerKernel::GetRunManagerKernel()->GetNumberOfParallelWorld();
218
219 if ( nParaWorld>0 ||
220 useCoupledTransportation ||
221 G4ScoringManager::GetScoringManagerIfExist())
222 {
223 auto coupledTransport= new G4CoupledTransportation(verboseLevelTransport);
224 if( theLooperThresholds == 0 ) coupledTransport->SetLowLooperThresholds();
225 if( theLooperThresholds == 2 ) coupledTransport->SetHighLooperThresholds();
226 theTransportationProcess = coupledTransport;
227
228 if (verboseLevel >0) {
229 G4cout << "--- G4CoupledTransportation is used " << G4endl;
230 }
231 } else {
232 auto simpleTransport = new G4Transportation(verboseLevelTransport);
233 if( theLooperThresholds == 0 ) simpleTransport->SetLowLooperThresholds();
234 if( theLooperThresholds == 2 ) simpleTransport->SetHighLooperThresholds();
235 theTransportationProcess = simpleTransport;
236 }
237
238 // loop over all particles in G4ParticleTable
239 aParticleIterator->reset();
240 while( (*aParticleIterator)() ){
241 G4ParticleDefinition* particle = aParticleIterator->value();
242 G4ProcessManager* pmanager = particle->GetProcessManager();
243 // Add transportation process for all particles
244 if ( pmanager == 0) {
245 // Error !! no process manager
246#ifdef G4VERBOSE
247 if (verboseLevel>0){
248 G4cout << "G4PhysicsListHelper::AddTransportation "
249 <<" : No Process Manager for "
250 << particle->GetParticleName() << G4endl;
251 }
252#endif
253 G4Exception("G4PhysicsListHelper::AddTransportation",
254 "Run0104", FatalException,
255 "No process manager");
256 continue;
257 }
258 // Molecule use different type transportation
259 if(particle->GetParticleType() == "Molecule") continue;
260
261 // add transportation with ordering = ( -1, "first", "first" )
262 pmanager ->AddProcess(theTransportationProcess);
263 pmanager ->SetProcessOrderingToFirst(theTransportationProcess, idxAlongStep);
264 pmanager ->SetProcessOrderingToFirst(theTransportationProcess, idxPostStep);
265 }
266}

271void G4PhysicsListHelper::ReadOrdingParameterTable()
272{
273 G4bool readInFile = false;
274 std::ifstream fIn;
275
276 if( getenv("G4ORDPARAMTABLE") ){
277 ordParamFileName = getenv("G4ORDPARAMTABLE");
278#ifdef G4VERBOSE
279 if (verboseLevel >1){
280 G4cout << "G4PhysicsListHelper::ReadOrdingParameterTable :"
281 << ordParamFileName << " is assigned to Ordering Parameter Table "
282 << G4endl;
283 }
284#endif
285 // open input file //
286 fIn.open(ordParamFileName, std::ios::in);
287 // check if the file has been opened successfully
288 if (!fIn) {
289#ifdef G4VERBOSE
290 if (verboseLevel >0) {
291 G4cout << "G4PhysicsListHelper::ReadOrdingParameterTable "
292 << " Can not open file " << ordParamFileName << G4endl;
293 }
294#endif
295 G4Exception("G4PhysicsListHelper::ReadOrdingParameterTable",
296 "Run0105", JustWarning,
297 "Fail to open ordering paramter table ");
298 } else {
299 readInFile = true;
300 }
301 }
302
303
304 // create OrdParamTable
305 if (theTable !=0) {
306 theTable->clear();
307 delete theTable;
308 theTable=0;
309 sizeOfTable=0;
310 }
311 theTable = new G4OrdParamTable();
312 sizeOfTable=0;
313
314 if (readInFile){
315 // read in the file and fill the table
316 while(!fIn.eof()) {
317 G4PhysicsListOrderingParameter tmp;
318 G4int flag;
319 fIn >> tmp.processTypeName >> tmp.processType >> tmp.processSubType
320 >> tmp.ordering[0] >> tmp.ordering[1] >> tmp.ordering[2] >> flag;
321 tmp.isDuplicable = (flag!=0);
322 theTable->push_back(tmp);
323 sizeOfTable +=1;
324 }
325 fIn.close();
326 } else {
327 ReadInDefaultOrderingParameter();
328 }
329
330 if (sizeOfTable==0){
331#ifdef G4VERBOSE
332 if (verboseLevel >0) {
333 G4cout << "G4PhysicsListHelper::ReadOrdingParameterTable "
334 << " Empty file " << ordParamFileName << G4endl;
335 }
336#endif
337 G4Exception("G4PhysicsListHelper::ReadOrdingParameterTable",
338 "Run0106", JustWarning,
339 "The ordering parameter table is empty ");
340 delete theTable;
341 theTable=0;
342 sizeOfTable=0;
343 }
344 return;
345}

347////////////////////////////////////////////////////////
348void G4PhysicsListHelper::DumpOrdingParameterTable(G4int subType) const
349{
350 if (theTable==0) {
351#ifdef G4VERBOSE
352 if (verboseLevel >0) {
353 G4cout << "G4PhysicsListHelper::DumpOrdingParameterTable "
354 << " No ordering parameter table : " << ordParamFileName
355 << G4endl;
356 }
357#endif
358 return;
359 }
360 G4cout << "G4PhysicsListHelper::DumpOrdingParameterTable : "
361 << ordParamFileName << G4endl;
362 G4cout << " TypeName "
363 << " ProcessType" << " SubType"
364 << " AtRest" << " AlongStep" << " PostStep"
365 << " Duplicable" << G4endl;
366 for (int i=0; i<sizeOfTable ; i++){
367 G4PhysicsListOrderingParameter* tmp=&(theTable->at(i));
368 if ((subType>=0) && (subType!=tmp->processSubType)) continue;
369 G4cout << std::setw(18) << tmp->processTypeName
370 << std::setw(15) << tmp->processType
371 << std::setw(15) << tmp->processSubType
372 << std::setw(15) << tmp->ordering[0]
373 << std::setw(15) << tmp->ordering[1]
374 << std::setw(15) << tmp->ordering[2];
375 if (tmp->isDuplicable) {
376 G4cout << " true";
377 } else {
378 G4cout << " false";
379 }
380 G4cout <<G4endl;
381 }
382}

384////////////////////////////////////////////////////////
385G4PhysicsListOrderingParameter G4PhysicsListHelper::GetOrdingParameter(G4int subType) const
386{
387 G4PhysicsListOrderingParameter value;
388
389 if (theTable==0) {
390#ifdef G4VERBOSE
391 if (verboseLevel >0) {
392 G4cout << "G4PhysicsListHelper::GetOrderingParameter : "
393 << " No ordering parameter table : " << ordParamFileName
394 << G4endl;
395 }
396#endif
397 return value;
398 }
399
400 for (int i=0; i<sizeOfTable ; i++){
401 G4PhysicsListOrderingParameter* tmp=&(theTable->at(i));
402 if (subType == tmp->processSubType){
403 value.processTypeName = tmp->processTypeName;
404 value.processType = tmp->processType;
405 value.processSubType = tmp->processSubType;
406 value.ordering[0] = tmp->ordering[0];
407 value.ordering[1] = tmp->ordering[1];
408 value.ordering[2] = tmp->ordering[2];
409 value.isDuplicable = tmp->isDuplicable;
410 }
411 }
412 return value;
413}

415////////////////////////////////////////////////////////
416G4bool G4PhysicsListHelper::RegisterProcess(G4VProcess* process,
417 G4ParticleDefinition* particle)
418{
419 if (theTable==0) {
420#ifdef G4VERBOSE
421 if (verboseLevel >0) {
422 G4cout << "G4PhysicsListHelper::RegisterProcess :"
423 << " No ordering parameter table : " << ordParamFileName
424 << G4endl;
425 }
426#endif
427 G4Exception("G4PhysicsListHelper::RegisterPorcess",
428 "Run0107", FatalException,
429 "No Ordering Parameter Table");
430 return false;
431 }
432
433 const G4String pName = process->GetProcessName();
434 const G4int pType = process->GetProcessType();
435 const G4int pSubType = process->GetProcessSubType();
436
437#ifdef G4VERBOSE
438 if (verboseLevel >2) {
439 G4cout << "G4PhysicsListHelper::RegisterProcess :"
440 << pName << " Process Type = " << pType
441 << " SubType = "<< pSubType
442 << " to " << particle->GetParticleName()
443 << G4endl;
444 }
445#endif
446
447 // Check Process Type/SubType
448 if ((pType <1)||(pSubType<1)) {
449#ifdef G4VERBOSE
450 if (verboseLevel >0) {
451 G4cout << "G4PhysicsListHelper::RegisterProcess :"
452 << pName << " for " << particle->GetParticleName()
453 << " has illegal Process Type = " << pType
454 << " SubType = "<< pSubType << G4endl;
455 }
456#endif
457 G4Exception("G4PhysicsListHelper::RegisterPorcess",
458 "Run0108", FatalException,
459 "No Matching process Type/SubType");
460 return false;
461 }
462
463 G4bool isFound = false;
464 G4int ord[3];
465 G4bool duplicable = false;
466 for (int i=0; i<sizeOfTable ; i++){
467 G4PhysicsListOrderingParameter* tmp=&(theTable->at(i));
468 if ((tmp->processType==pType)&&(tmp->processSubType==pSubType)){
469 ord[0] = tmp->ordering[0];
470 ord[1] = tmp->ordering[1];
471 ord[2] = tmp->ordering[2];
472 duplicable = tmp->isDuplicable;
473 isFound = true;
474 break;
475 }
476 }
477 if (!isFound) {
478#ifdef G4VERBOSE
479 if (verboseLevel >0) {
480 G4cout << "G4PhysicsListHelper::RegisterProcess :"
481 << pName << " for " << particle->GetParticleName()
482 << " with type/subtype ="
483 << pType << "/" << pSubType
484 << " is not reigstered in OrdingParameterTable "
485 << G4endl;
486 }
487#endif
488 G4Exception("G4PhysicsListHelper::RegisterPorcess",
489 "Run0109", FatalException,
490 "No Matching process Type/SubType");
491 return false;
492 }
493
494 // Check Process Manager
495 G4ProcessManager* pManager = particle->GetProcessManager();
496 if ( pManager == 0) {
497 // Error !! no process manager
498#ifdef G4VERBOSE
499 if (verboseLevel>0){
500 G4cout << "G4PhysicsListHelper::RegisterProcess "
501 <<" : No Process Manager for "
502 << particle->GetParticleName() << G4endl;
503 }
504#endif
505 G4Exception("G4PhysicsListHelper::RegisterProcess ",
506 "Riun0110", FatalException,
507 "No process manager");
508 return false;
509 }
510
511 // Check Duplication
512 if (!duplicable){
513 G4bool duplicated = false;
514 G4ProcessVector* pList = pManager->GetProcessList();
515 for (G4int idx=0; idx<pList->size(); idx++) {
516 const G4VProcess* p = (*pList)[idx];
517 if ((p->GetProcessType()== pType) &&
518 (p->GetProcessSubType()== pSubType)){
519 duplicated = true;
520#ifdef G4VERBOSE
521 if (verboseLevel >0) {
522 G4cout << "G4PhysicsListHelper::RegisterProcess :"
523 << pName << " for " << particle->GetParticleName()
524 << " with type/subtype ="
525 << pType << "/" << pSubType
526 << " is has same subType as "
527 << p->GetProcessName()
528 << " for " << particle->GetParticleName()
529 << G4endl;
530 G4cout << "It will not be added !!" << G4endl;
531 }
532#endif
533 G4Exception("G4PhysicsListHelper::RegisterPorcess",
534 "Run0111", JustWarning,
535 "Duplication of processes");
536 }
537 }
538 if (duplicated) return false;
539 }
540
541 // Add Process
542 G4int code = pManager ->AddProcess(process);
543 if (code <0) return false;
544
545 // Set Ordering Parameter
546 for(G4int idx=0; idx<3; idx++){
547 G4ProcessVectorDoItIndex idxOrd = static_cast<G4ProcessVectorDoItIndex>(idx);
548 if (ord[idx]<0) {
549 // Do Nothing because NO DOIT
550 } else if (ord[idx]==0) {
551 pManager->SetProcessOrderingToFirst( process, idxOrd );
552 } else if (ord[idx]<9999) {
553 pManager->SetProcessOrdering( process, idxOrd , ord[idx]);
554 } else {
555 pManager->SetProcessOrderingToLast( process, idxOrd );
556 }
557 }
558#ifdef G4VERBOSE
559 if (verboseLevel >1) {
560 G4cout << "G4PhysicsListHelper::RegisterProcess :"
561 << pName << " for " << particle->GetParticleName()
562 << " with type/subtype ="
563 << pType << "/" << pSubType
564 << " is sucessfully registered with ordering parameters "
565 << ord[0] << ":" << ord[1] << ":" << ord[2]
566 << G4endl;
567 }
568#endif
569 return true;
570}
