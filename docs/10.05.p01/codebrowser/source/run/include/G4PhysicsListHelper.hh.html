29// Class Description:
30// This class is a helper class for physics lists to register processes
31// according to the ordering parameter table
32// This class is a singleton
33// -------------------------------------------
34// History
35// first version 29 Apr. 2011 by H.Kurashige
36// ------------------------------------------------------------
37
38#ifndef G4PhysicsListHelper_h
39#define G4PhysicsListHelper_h 1
40#include "globals.hh"
41#include "G4ios.hh"
42#include <vector>
43
44#include "G4ParticleTable.hh"
45#include "G4ParticleDefinition.hh"
46#include "G4PhysicsListOrderingParameter.hh"
47
48class G4VProcess;
49
50class G4PhysicsListHelper
51{
52 private:
53 // Hide constructor and destructor
54 G4PhysicsListHelper();
55 virtual ~G4PhysicsListHelper();
56
57 public: // with description
58 // This method gives the ponter to the physics list helper
59 static G4PhysicsListHelper* GetPhysicsListHelper();
60
61 //Register a process to the particle type
62 // according to the ordering parameter table
63 // 'true' is returned if the process is registerd successfully
64 G4bool RegisterProcess(G4VProcess* process,
65 G4ParticleDefinition* particle);
66
67 // User must invoke this method in his ConstructProcess()
68 // implementation in order to insures particle transportation.
69 void AddTransportation();
70
71 // Set flag for using CoupledTransportation
72 void UseCoupledTransportation(G4bool vl=true);
73
74 // Change the thresholds for killing looping tracks of the
75 // transportation (simple or coupled.)
76 void UseHighLooperThresholds() { theLooperThresholds= 2;}
77 void UseLowLooperThresholds() { theLooperThresholds= 0;}
78
79 /////////////////////////////////////////////////////////////////
80 public:
81 // check consistencies of list of particles
82 void CheckParticleList() const;
83
84 ///////////////////////////////////////////////////////////////////////
85 public:
86 // Dump OrdingParameterTable
87 void DumpOrdingParameterTable(G4int subType = -1) const;
88 G4PhysicsListOrderingParameter GetOrdingParameter(G4int subType) const;
89
90 private:
91 void ReadOrdingParameterTable();
92 void ReadInDefaultOrderingParameter();
93
94 ///////////////////////////////////////////////////////////////////////
95 public: // with description
96 void SetVerboseLevel(G4int value);
97 G4int GetVerboseLevel() const;
98 // set/get controle flag for output message
99 // 0: Silent
100 // 1: Warning message
101 // 2: More
102
103 ////////////////////////////////////////////////////////////////////////
104 private:
105 static G4ThreadLocal G4PhysicsListHelper* pPLHelper;
106
107 // the particle table has the complete List of existing particle types
108 G4ParticleTable* theParticleTable;
109 G4ParticleTable::G4PTblDicIterator* aParticleIterator;
110
111 G4bool useCoupledTransportation;
112 G4int theLooperThresholds= 1; // 0 = Low, 1 = default, 2 = high
113 G4VProcess* theTransportationProcess;
114
115 G4int verboseLevel;
116
117 private:
118 typedef std::vector<G4PhysicsListOrderingParameter> G4OrdParamTable;
119 G4OrdParamTable* theTable;
120 G4int sizeOfTable;
121 G4String ordParamFileName;
122};
123
124
125inline
126void G4PhysicsListHelper::UseCoupledTransportation(G4bool vl)
127{
128 useCoupledTransportation = vl;
129}
130
131inline
132 void G4PhysicsListHelper::SetVerboseLevel(G4int value)
133{
134 verboseLevel = value;
135}
136
137inline
138 G4int G4PhysicsListHelper::GetVerboseLevel() const
139{
140 return verboseLevel;
141}
142
143
144#endif
