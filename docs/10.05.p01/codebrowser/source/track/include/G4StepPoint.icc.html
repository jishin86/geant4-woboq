90inline
91 G4ThreeVector G4StepPoint::GetMomentum() const
92 {
93 G4double tMomentum = std::sqrt(fKineticEnergy*fKineticEnergy +
94 2*fKineticEnergy*fMass);
95 return G4ThreeVector(fMomentumDirection.x()*tMomentum,
96 fMomentumDirection.y()*tMomentum,
97 fMomentumDirection.z()*tMomentum);
98 }
99 // Total momentum of the track
