67template<class T> TQGS_BIC<T>::TQGS_BIC(G4int ver): T()
68{
69 // default cut value (1.0mm)
70 // defaultCutValue = 1.0*CLHEP::mm;
71
72 G4DataQuestionaire it(photon);
73 G4cout << "<<< Geant4 Physics List simulation engine: QGS_BIC 2.0"<<G4endl;
74 G4cout <<G4endl;
75
76 this->defaultCutValue = 0.7*CLHEP::mm;
77 this->SetVerboseLevel(ver);
78
79 G4WarnPLStatus exp;
80 exp.Experimental("QGS_BIC");
81
82 // EM Physics
83 this->RegisterPhysics( new G4EmStandardPhysics(ver) );
84
85 // Synchroton Radiation & GN Physics
86 this->RegisterPhysics( new G4EmExtraPhysics(ver) );
87
88 // Decays
89 this->RegisterPhysics( new G4DecayPhysics(ver) );
90
91 // Hadron Elastic scattering
92 this->RegisterPhysics( new G4HadronElasticPhysics(ver) );
93
94 // Hadron Physics
95 this->RegisterPhysics( new G4HadronPhysicsQGS_BIC(ver) );
96
97 // Stopping Physics
98 this->RegisterPhysics( new G4StoppingPhysics(ver) );
99
100 // Ion Physics
101 this->RegisterPhysics( new G4IonPhysics(ver) );
102
103 // Neutron tracking cut
104 this->RegisterPhysics( new G4NeutronTrackingCut(ver) );
105
106 // Neutron cross sections
107 //AR-31Oct2012 this->RegisterPhysics( new G4NeutronCrossSectionXS(ver) );
108
109}

123 this->SetCutsWithDefault();
124

127
128}
