42#include "G4ParticleTable.hh"

44
45G4PiKBuilder::
46G4PiKBuilder(): wasActivated(false)
47{
48 thePionPlusInelastic=new G4PionPlusInelasticProcess;
49 thePionMinusInelastic=new G4PionMinusInelasticProcess;
50 theKaonPlusInelastic=new G4KaonPlusInelasticProcess;
51 theKaonMinusInelastic=new G4KaonMinusInelasticProcess;
52 theKaonZeroLInelastic=new G4KaonZeroLInelasticProcess;
53 theKaonZeroSInelastic=new G4KaonZeroSInelasticProcess;
54}
55
56void G4PiKBuilder::
57Build()
58{
59 wasActivated = true;
60
61 std::vector<G4VPiKBuilder *>::iterator i;
62 for(i=theModelCollections.begin(); i!=theModelCollections.end(); i++)
63 {
64 (*i)->Build(thePionPlusInelastic);
65 (*i)->Build(thePionMinusInelastic);
66 (*i)->Build(theKaonPlusInelastic);
67 (*i)->Build(theKaonMinusInelastic);
68 (*i)->Build(theKaonZeroLInelastic);
69 (*i)->Build(theKaonZeroSInelastic);
70 }
71 G4ProcessManager * theProcMan;
72 theProcMan = G4PionPlus::PionPlus()->GetProcessManager();
73 theProcMan->AddDiscreteProcess(thePionPlusInelastic);
74
75 theProcMan = G4PionMinus::PionMinus()->GetProcessManager();
76 theProcMan->AddDiscreteProcess(thePionMinusInelastic);
77
78 theProcMan = G4KaonPlus::KaonPlus()->GetProcessManager();
79 theProcMan->AddDiscreteProcess(theKaonPlusInelastic);
80
81 theProcMan = G4KaonMinus::KaonMinus()->GetProcessManager();
82 theProcMan->AddDiscreteProcess(theKaonMinusInelastic);
83
84 theProcMan = G4KaonZeroLong::KaonZeroLong()->GetProcessManager();
85 theProcMan->AddDiscreteProcess(theKaonZeroLInelastic);
86
87 theProcMan = G4KaonZeroShort::KaonZeroShort()->GetProcessManager();
88 theProcMan->AddDiscreteProcess(theKaonZeroSInelastic);
89}
90
91void G4PiKBuilder::RegisterMe(G4PhysicsBuilderInterface* aB) {
92 auto bld = dynamic_cast<G4VPiKBuilder*>(aB);
93 if ( bld != nullptr ) {
94 theModelCollections.push_back(bld);
95 } else {
96 G4PhysicsBuilderInterface::RegisterMe(aB);
97 }
98}
