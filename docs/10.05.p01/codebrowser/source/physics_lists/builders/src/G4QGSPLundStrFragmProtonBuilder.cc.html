40#include "G4ParticleDefinition.hh"
41#include "G4ParticleTable.hh"

44#include "G4HadronicParameters.hh"
45
46G4QGSPLundStrFragmProtonBuilder::
47G4QGSPLundStrFragmProtonBuilder(G4bool quasiElastic)
48 {
49 theMin = 100*GeV;
50 theModel = new G4TheoFSGenerator("QGSP");
51
52 theStringModel = new G4QGSModel< G4QGSParticipants >;
53 theStringDecay = new G4ExcitedStringDecay(theStrFragm = new G4LundStringFragmentation);
54 theStringModel->SetFragmentationModel(theStringDecay);
55
56 theCascade = new G4GeneratorPrecompoundInterface;
57
58 theModel->SetTransport(theCascade);
59 theModel->SetHighEnergyGenerator(theStringModel);
60 if (quasiElastic)
61 {
62 theQuasiElastic=new G4QuasiElasticChannel;
63 theModel->SetQuasiElasticChannel(theQuasiElastic);
64 } else
65 { theQuasiElastic=0;}
66
67 }
68
69void G4QGSPLundStrFragmProtonBuilder::
70Build(G4ProtonInelasticProcess * aP)
71 {
72 aP->AddDataSet(new G4BGGNucleonInelasticXS(G4Proton::Proton()));
73 theModel->SetMinEnergy(theMin);
74 theModel->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
75 aP->RegisterMe(theModel);
76 }
77
78G4QGSPLundStrFragmProtonBuilder::~G4QGSPLundStrFragmProtonBuilder()
79 {
80 if ( theQuasiElastic ) delete theQuasiElastic;
81 delete theStringDecay;
82 delete theStringModel;
83 delete theStrFragm;
84 }
