47G4HyperonFTFPBuilder::G4HyperonFTFPBuilder():
48 theLambdaInelastic(0),
49 theAntiLambdaInelastic(0),
50 theSigmaMinusInelastic(0),
51 theAntiSigmaMinusInelastic(0),
52 theSigmaPlusInelastic(0),
53 theAntiSigmaPlusInelastic(0),
54 theXiZeroInelastic(0),
55 theAntiXiZeroInelastic(0),
56 theXiMinusInelastic(0),
57 theAntiXiMinusInelastic(0),
58 theOmegaMinusInelastic(0),
59 theAntiOmegaMinusInelastic(0),
60 wasActivated(false)
61{
62
63// Hyperon : Bertini at low energies, then FTFP
64
65 HyperonFTFP = new G4TheoFSGenerator("FTFP");
66
67 HyperonFTFP->SetMinEnergy( 2.*GeV );
68 HyperonFTFP->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
69
70 theStringModel = new G4FTFModel;
71 theStringDecay = new G4ExcitedStringDecay(theLund = new G4LundStringFragmentation);
72 theStringModel->SetFragmentationModel(theStringDecay);
73
74 theCascade = new G4GeneratorPrecompoundInterface;
75
76 HyperonFTFP->SetTransport(theCascade);
77 HyperonFTFP->SetHighEnergyGenerator(theStringModel);
78
79 theBertini = new G4CascadeInterface;
80 theBertini->SetMinEnergy( 0.*GeV );
81 theBertini->SetMaxEnergy( 6.*GeV );
82
83// AntiHyperons: Use FTFP for full energy range, starting at 0.
84
85 AntiHyperonFTFP = new G4TheoFSGenerator("FTFP");
86 AntiHyperonFTFP->SetMinEnergy( 0.*GeV );
87 AntiHyperonFTFP->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
88 AntiHyperonFTFP->SetTransport(theCascade);
89 AntiHyperonFTFP->SetHighEnergyGenerator(theStringModel);
90
91// use CHIPS cross sections
92 theCHIPSInelastic = G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4ChipsHyperonInelasticXS::Default_Name());
93}
94
95G4HyperonFTFPBuilder::~G4HyperonFTFPBuilder()
96{
97 delete theStringDecay;
98 delete theLund;
99}

101void G4HyperonFTFPBuilder::Build()
102{
103 G4ProcessManager * aProcMan = 0;
104 wasActivated = true;
105
106 // Lambda
107 theLambdaInelastic = new G4LambdaInelasticProcess();
108 theLambdaInelastic->RegisterMe(theBertini);
109 theLambdaInelastic->RegisterMe(HyperonFTFP);
110 theLambdaInelastic->AddDataSet(theCHIPSInelastic);
111 aProcMan = G4Lambda::Lambda()->GetProcessManager();
112 aProcMan->AddDiscreteProcess(theLambdaInelastic);
113
114 // AntiLambda
115 theAntiLambdaInelastic = new G4AntiLambdaInelasticProcess();
116 theAntiLambdaInelastic->RegisterMe(AntiHyperonFTFP);
117 theAntiLambdaInelastic->AddDataSet(theCHIPSInelastic);
118
119 aProcMan = G4AntiLambda::AntiLambda()->GetProcessManager();
120 aProcMan->AddDiscreteProcess(theAntiLambdaInelastic);
121
122 // SigmaMinus
123 theSigmaMinusInelastic = new G4SigmaMinusInelasticProcess();
124 theSigmaMinusInelastic->RegisterMe(theBertini);
125 theSigmaMinusInelastic->RegisterMe(HyperonFTFP);
126 theSigmaMinusInelastic->AddDataSet(theCHIPSInelastic);
127
128 aProcMan = G4SigmaMinus::SigmaMinus()->GetProcessManager();
129 aProcMan->AddDiscreteProcess(theSigmaMinusInelastic);
130
131 // anti-SigmaMinus
132 theAntiSigmaMinusInelastic = new G4AntiSigmaMinusInelasticProcess();
133 theAntiSigmaMinusInelastic->RegisterMe(AntiHyperonFTFP);
134 theAntiSigmaMinusInelastic->AddDataSet(theCHIPSInelastic);
135
136 aProcMan = G4AntiSigmaMinus::AntiSigmaMinus()->GetProcessManager();
137 aProcMan->AddDiscreteProcess(theAntiSigmaMinusInelastic);
138
139 // SigmaPlus
140 theSigmaPlusInelastic = new G4SigmaPlusInelasticProcess();
141 theSigmaPlusInelastic->RegisterMe(theBertini);
142 theSigmaPlusInelastic->RegisterMe(HyperonFTFP);
143 theSigmaPlusInelastic->AddDataSet(theCHIPSInelastic);
144
145 aProcMan = G4SigmaPlus::SigmaPlus()->GetProcessManager();
146 aProcMan->AddDiscreteProcess(theSigmaPlusInelastic);
147
148 // anti-SigmaPlus
149 theAntiSigmaPlusInelastic = new G4AntiSigmaPlusInelasticProcess();
150 theAntiSigmaPlusInelastic->RegisterMe(AntiHyperonFTFP);
151 theAntiSigmaPlusInelastic->AddDataSet(theCHIPSInelastic);
152
153 aProcMan = G4AntiSigmaPlus::AntiSigmaPlus()->GetProcessManager();
154 aProcMan->AddDiscreteProcess(theAntiSigmaPlusInelastic);
155
156 // XiMinus
157 theXiMinusInelastic = new G4XiMinusInelasticProcess();
158 theXiMinusInelastic->RegisterMe(theBertini);
159 theXiMinusInelastic->RegisterMe(HyperonFTFP);
160 theXiMinusInelastic->AddDataSet(theCHIPSInelastic);
161
162 aProcMan = G4XiMinus::XiMinus()->GetProcessManager();
163 aProcMan->AddDiscreteProcess(theXiMinusInelastic);
164
165 // anti-XiMinus
166 theAntiXiMinusInelastic = new G4AntiXiMinusInelasticProcess();
167 theAntiXiMinusInelastic->RegisterMe(AntiHyperonFTFP);
168 theAntiXiMinusInelastic->AddDataSet(theCHIPSInelastic);
169
170 aProcMan = G4AntiXiMinus::AntiXiMinus()->GetProcessManager();
171 aProcMan->AddDiscreteProcess(theAntiXiMinusInelastic);
172
173 // XiZero
174 theXiZeroInelastic = new G4XiZeroInelasticProcess();
175 theXiZeroInelastic->RegisterMe(theBertini);
176 theXiZeroInelastic->RegisterMe(HyperonFTFP);
177 theXiZeroInelastic->AddDataSet(theCHIPSInelastic);
178
179 aProcMan = G4XiZero::XiZero()->GetProcessManager();
180 aProcMan->AddDiscreteProcess(theXiZeroInelastic);
181
182 // anti-XiZero
183 theAntiXiZeroInelastic = new G4AntiXiZeroInelasticProcess();
184 theAntiXiZeroInelastic->RegisterMe(AntiHyperonFTFP);
185 theAntiXiZeroInelastic->AddDataSet(theCHIPSInelastic);
186
187 aProcMan = G4AntiXiZero::AntiXiZero()->GetProcessManager();
188 aProcMan->AddDiscreteProcess(theAntiXiZeroInelastic);

190 // OmegaMinus
191 theOmegaMinusInelastic = new G4OmegaMinusInelasticProcess();
192 theOmegaMinusInelastic->RegisterMe(theBertini);
193 theOmegaMinusInelastic->RegisterMe(HyperonFTFP);
