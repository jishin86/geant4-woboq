53G4QGSPPiKBuilder::
54G4QGSPPiKBuilder(G4bool quasiElastic)
55{
56 theMin = 12*GeV;
57 theModel = new G4TheoFSGenerator("QGSP");
58
59 theStringModel = new G4QGSModel< G4QGSParticipants >;
60 theStringDecay = new G4ExcitedStringDecay(theQGSM = new G4QGSMFragmentation);
61 theStringModel->SetFragmentationModel(theStringDecay);
62
63 theCascade = new G4GeneratorPrecompoundInterface();
64
65 theModel->SetHighEnergyGenerator(theStringModel);
66 if (quasiElastic)
67 {
68 theQuasiElastic=new G4QuasiElasticChannel;
69 theModel->SetQuasiElasticChannel(theQuasiElastic);
70 } else
71 { theQuasiElastic=0;}
72
73 theModel->SetTransport(theCascade);
74}
75
76G4QGSPPiKBuilder::
77~G4QGSPPiKBuilder()
78{
79 if ( theQuasiElastic ) delete theQuasiElastic;
80 delete theStringDecay;
81 delete theStringModel;
82 delete theQGSM;
83}

85void G4QGSPPiKBuilder::
86Build(G4PionPlusInelasticProcess * aP)
87{
88 theModel->SetMinEnergy(theMin);
89 theModel->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
90 aP->AddDataSet( new G4BGGPionInelasticXS( G4PionPlus::Definition() ) );
91 aP->RegisterMe(theModel);
92}
