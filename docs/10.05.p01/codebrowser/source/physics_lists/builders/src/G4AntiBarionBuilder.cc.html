43G4AntiBarionBuilder::
44G4AntiBarionBuilder(): wasActivated(false)
45{
46 theAntiProtonInelastic=new G4AntiProtonInelasticProcess;
47 theAntiNeutronInelastic=new G4AntiNeutronInelasticProcess;
48 theAntiDeuteronInelastic=new G4AntiDeuteronInelasticProcess;
49 theAntiTritonInelastic=new G4AntiTritonInelasticProcess;
50 theAntiHe3Inelastic=new G4AntiHe3InelasticProcess;
51 theAntiAlphaInelastic=new G4AntiAlphaInelasticProcess;
52}
53
54
55void G4AntiBarionBuilder::
56Build()
57{
58 wasActivated = true;
59
60 std::vector<G4VAntiBarionBuilder *>::iterator i;
61 for(i=theModelCollections.begin(); i!=theModelCollections.end(); i++)
62 {
63 (*i)->Build(theAntiProtonInelastic);
64 (*i)->Build(theAntiNeutronInelastic);
65 (*i)->Build(theAntiDeuteronInelastic);
66 (*i)->Build(theAntiTritonInelastic);
67 (*i)->Build(theAntiHe3Inelastic);
68 (*i)->Build(theAntiAlphaInelastic);
69 }
70 G4ProcessManager * theProcMan;
71 theProcMan = G4AntiProton::AntiProton()->GetProcessManager();
72 theProcMan->AddDiscreteProcess(theAntiProtonInelastic);
73
74 theProcMan = G4AntiNeutron::AntiNeutron()->GetProcessManager();
75 theProcMan->AddDiscreteProcess(theAntiNeutronInelastic);
76
77 theProcMan = G4AntiDeuteron::AntiDeuteron()->GetProcessManager();
78 theProcMan->AddDiscreteProcess(theAntiDeuteronInelastic);
79
80 theProcMan = G4AntiTriton::AntiTriton()->GetProcessManager();
81 theProcMan->AddDiscreteProcess(theAntiTritonInelastic);
82
83 theProcMan = G4AntiHe3::AntiHe3()->GetProcessManager();
84 theProcMan->AddDiscreteProcess(theAntiHe3Inelastic);
85
86 theProcMan = G4AntiAlpha::AntiAlpha()->GetProcessManager();
87 theProcMan->AddDiscreteProcess(theAntiAlphaInelastic);
88}

90void G4AntiBarionBuilder::RegisterMe(G4PhysicsBuilderInterface* aB ) {
91 auto bld = dynamic_cast<G4VAntiBarionBuilder*>(aB);
92 if ( bld != nullptr ) {
93 theModelCollections.push_back(bld);
94 } else {
95 G4PhysicsBuilderInterface::RegisterMe(aB);
96 }
