53G4FTFBinaryKaonBuilder::
54G4FTFBinaryKaonBuilder(G4bool quasiElastic)
55{
56 theMin = 4*GeV;
57 theMax = G4HadronicParameters::Instance()->GetMaxEnergy();
58 theModel = new G4TheoFSGenerator("FTFB");
59
60 theStringModel = new G4FTFModel;
61 theStringDecay = new G4ExcitedStringDecay(new G4LundStringFragmentation);
62 theStringModel->SetFragmentationModel(theStringDecay);
63
64 theCascade = new G4BinaryCascade;
65
66 theModel->SetHighEnergyGenerator(theStringModel);
67 if (quasiElastic)
68 {
69 theQuasiElastic=new G4QuasiElasticChannel;
70 theModel->SetQuasiElasticChannel(theQuasiElastic);
71 } else
72 { theQuasiElastic=0;}
73
74 theModel->SetTransport(theCascade);
75 theModel->SetMinEnergy(theMin);
76 theModel->SetMaxEnergy(theMax);
77}
78
79G4FTFBinaryKaonBuilder:: ~G4FTFBinaryKaonBuilder()
80{
81 delete theStringDecay;
82 delete theStringModel;
83 //delete theModel;
84 if ( theQuasiElastic ) delete theQuasiElastic;
85}

90void G4FTFBinaryKaonBuilder::
91Build(G4KaonPlusInelasticProcess * aP)
92{
93 theModel->SetMinEnergy(theMin);
94 theModel->SetMaxEnergy(theMax);
95 aP->AddDataSet(G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4ChipsKaonPlusInelasticXS::Default_Name()));
96 aP->RegisterMe(theModel);
97}
