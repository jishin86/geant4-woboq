27#define G4INCLXXProtonBuilder_h

31#include "G4HadronElasticProcess.hh"
32#include "G4HadronFissionProcess.hh"

40/**
41 * Builder for proton processes using the INCL++ intra-nuclear
42 * cascade model.
43 *
44 * By default the INCL++ is used for projectile energies 0 - 3 GeV.
45 *
46 * The builder uses INCL++ cascade model with G4ExcitationHandler
47 * de-excitation. This is implemented in interface
48 * G4INCLXXInterface.
49 *
50 * @see G4INCLXXInterface
51 * @see G4INCLXXNeutronBuilder
52 * @see G4INCLXXPiKBuilder
53 */
54class G4INCLXXProtonBuilder : public G4VProtonBuilder
55{
56 public:
57 G4INCLXXProtonBuilder();
58 virtual ~G4INCLXXProtonBuilder() {}
59
60 /**
61 * Build elastic process.
62 *
63 * No elastic process is built.
64 */
65 virtual void Build(G4HadronElasticProcess *) final override {};
66
67 /**
68 * Build inelastic process.
69 */
70 virtual void Build(G4ProtonInelasticProcess * aP) final override;
71
72 /**
73 * Set the minimum energy limit for the model.
74 */
75 virtual void SetMinEnergy(G4double aM) final override {theMin = aM;}
76
77 /**
78 * Set the maximum energy limit for the model.
79 */
80 virtual void SetMaxEnergy(G4double aM) final override {theMax = aM;}
81
82 using G4VProtonBuilder::Build; //Prevent compiler warning
83
84 private:
85 G4INCLXXInterface * theModel;
86 G4VPreCompoundModel * thePreCompoundModel;
87 G4double thePreCompoundMin;
88 G4double thePreCompoundMax;
89 G4double theMin;
90 G4double theMax;
91
92};
