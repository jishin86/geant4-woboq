35#include "G4VNeutronBuilder.hh"
36
37#include "G4LENDElasticCrossSection.hh"
38#include "G4LENDElastic.hh"
39#include "G4LENDInelastic.hh"
40#include "G4LENDInelasticCrossSection.hh"

46class G4NeutronLENDBuilder : public G4VNeutronBuilder
47{
48 public:
49 G4NeutronLENDBuilder(G4String eva="");
50 virtual ~G4NeutronLENDBuilder() {}
51
52 virtual void Build(G4HadronElasticProcess * aP) final override;
53 virtual void Build(G4HadronFissionProcess * aP) final override;
54 virtual void Build(G4HadronCaptureProcess * aP) final override;
55 virtual void Build(G4NeutronInelasticProcess * aP) final override;
56
57 virtual void SetMinEnergy(G4double aM) final override
58 {
59 theMin=aM;
60 theIMin = theMin;
61 }
62 void SetMinInelasticEnergy(G4double aM)
63 {
64 theIMin=aM;
65 }
66 virtual void SetMaxEnergy(G4double aM) final override
67 {
68 theIMax = aM;
69 theMax=aM;
70 }
71 void SetMaxInelasticEnergy(G4double aM)
72 {
73 theIMax = aM;
74 }
75
76 using G4VNeutronBuilder::Build; //Prevent compiler warning
77
78 private:
79
80 G4double theMin;
81 G4double theIMin;
82 G4double theMax;
83 G4double theIMax;
84
85 G4LENDElastic * theLENDElastic;
86 G4LENDElasticCrossSection * theLENDElasticCrossSection;
87 G4LENDInelastic * theLENDInelastic;
88 G4LENDInelasticCrossSection * theLENDInelasticCrossSection;
89 G4LENDFission * theLENDFission;
90 G4LENDFissionCrossSection * theLENDFissionCrossSection;
91 G4LENDCapture * theLENDCapture;
92 G4LENDCaptureCrossSection * theLENDCaptureCrossSection;
93
94 G4String evaluation;
95};

97#endif
