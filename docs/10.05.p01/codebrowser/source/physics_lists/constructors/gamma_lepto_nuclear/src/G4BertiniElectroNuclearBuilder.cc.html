56G4BertiniElectroNuclearBuilder::G4BertiniElectroNuclearBuilder(G4bool eNucl) :
57 thePhotoNuclearProcess(nullptr), theElectronNuclearProcess(nullptr),
58 thePositronNuclearProcess(nullptr), theElectroReaction(nullptr),
59 theGammaReaction(nullptr), theModel(nullptr), theCascade(nullptr),
60 theStringModel(nullptr), theFragmentation(nullptr), theStringDecay(nullptr),
61 wasActivated(false), eActivated(eNucl)
62{}
63
64G4BertiniElectroNuclearBuilder::~G4BertiniElectroNuclearBuilder()
65{
66 if(wasActivated) {
67 delete theFragmentation;
68 delete theStringDecay;
69 }
70}
71
72void G4BertiniElectroNuclearBuilder::Build()
73{
74 if(wasActivated) return;
75 wasActivated=true;
76
77 thePhotoNuclearProcess = new G4PhotoNuclearProcess;
78 if(eActivated) {
79 theElectronNuclearProcess = new G4ElectronNuclearProcess;
80 thePositronNuclearProcess = new G4PositronNuclearProcess;
81 theElectroReaction = new G4ElectroVDNuclearModel;
82 }
83 theGammaReaction = new G4CascadeInterface;
84
85 theModel = new G4TheoFSGenerator;
86
87 theStringModel = new G4QGSModel< G4GammaParticipants >;
88 theStringDecay = new G4ExcitedStringDecay(theFragmentation=new G4QGSMFragmentation);
89 theStringModel->SetFragmentationModel(theStringDecay);
90
91 theCascade = new G4GeneratorPrecompoundInterface;
92
93 theModel->SetTransport(theCascade);
94 theModel->SetHighEnergyGenerator(theStringModel);
95
96 G4ProcessManager * aProcMan = nullptr;
97
98 theGammaReaction->SetMaxEnergy(3.5*GeV);
99 thePhotoNuclearProcess->RegisterMe(theGammaReaction);
100 theModel->SetMinEnergy(3.*GeV);
101 theModel->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
102 thePhotoNuclearProcess->RegisterMe(theModel);
103
104 G4GammaGeneralProcess* sp =
105 (G4GammaGeneralProcess*)G4LossTableManager::Instance()->GetGammaGeneralProcess();
106 if(sp) {
107 sp->AddHadProcess(thePhotoNuclearProcess);
108 } else {
109 aProcMan = G4Gamma::Gamma()->GetProcessManager();
110 aProcMan->AddDiscreteProcess(thePhotoNuclearProcess);
111 }
112
113 if(eActivated) {
114 aProcMan = G4Electron::Electron()->GetProcessManager();
115 theElectronNuclearProcess->RegisterMe(theElectroReaction);
116 aProcMan->AddDiscreteProcess(theElectronNuclearProcess);
117
118 aProcMan = G4Positron::Positron()->GetProcessManager();
119 thePositronNuclearProcess->RegisterMe(theElectroReaction);
120 aProcMan->AddDiscreteProcess(thePositronNuclearProcess);
121 }
122}
