411void G4OpticalPhysicsMessenger::SetNewValue(G4UIcommand* command,
412 G4String newValue)
413{
414/// Apply command to the associated object.
415 if (command == fActivateProcessCmd) {
416 std::istringstream is(newValue.data());
417 G4String pn;
418 G4String flag;
419 is >> pn >> flag;
420 if ( pn == "Cerenkov" ) {
421 fSelectedProcessIndex = kCerenkov;
422 } else if ( pn == "Scintillation" ) {
423 fSelectedProcessIndex = kScintillation;
424 } else if ( pn == "OpAbsorption" ) {
425 fSelectedProcessIndex = kAbsorption;
426 } else if ( pn == "OpRayleigh" ) {
427 fSelectedProcessIndex = kRayleigh;
428 } else if ( pn == "OpMieHG" ) {
429 fSelectedProcessIndex = kMieHG;
430 } else if ( pn == "OpBoundary" ) {
431 fSelectedProcessIndex = kBoundary;
432 } else if ( pn == "OpWLS" ) {
433 fSelectedProcessIndex = kWLS;
434 } else {
435 G4ExceptionDescription msg;
436 msg << "Not allowed process name: "<<pn<<" (UI: "<<newValue<<")";
437 G4Exception("G4OpticalPhysicsMessenger::SetNewValue(...)","Optical001",FatalException,msg);
438 }
