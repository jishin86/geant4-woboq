143void G4EmDNAPhysicsActivator::ConstructProcess()
144{
145 const std::vector<G4String>& regnamesDNA = theParameters->RegionsDNA();
146 G4int nreg = regnamesDNA.size();
147 if(0 == nreg)
148 {
149 return;
150 }
151 const std::vector<G4String>& typesDNA = theParameters->TypesDNA();
152
153 if(IsVerbose()) {
154 G4cout << "### G4EmDNAPhysicsActivator::ConstructProcess for " << nreg
155 << " regions; DNA physics type " << typesDNA[0] << G4endl;
156 }
157
158 // list of particles
159 const G4ParticleDefinition* elec = G4Electron::Electron();
160 const G4ParticleDefinition* prot = G4Proton::Proton();
161 const G4ParticleDefinition* gion = G4GenericIon::GenericIon();
162
163 G4DNAGenericIonsManager * genericIonsManager =
164 G4DNAGenericIonsManager::Instance();
165 const G4ParticleDefinition* alpha2 = G4Alpha::Alpha();
166 const G4ParticleDefinition* alpha1 = genericIonsManager->GetIon("alpha+");
167 const G4ParticleDefinition* alpha0 = genericIonsManager->GetIon("helium");
168 const G4ParticleDefinition* h0 = genericIonsManager->GetIon("hydrogen");
169
170 G4ProcessManager* eman = elec->GetProcessManager();
171 G4ProcessManager* pman = prot->GetProcessManager();
172 G4ProcessManager* iman = gion->GetProcessManager();
173 G4ProcessManager* a2man = alpha2->GetProcessManager();
174 G4ProcessManager* a1man = alpha1->GetProcessManager();
175 G4ProcessManager* a0man = alpha0->GetProcessManager();
176 G4ProcessManager* h0man = h0->GetProcessManager();
177
178 // alpha+ standard processes
179 G4PhysicsListHelper* ph = G4PhysicsListHelper::GetPhysicsListHelper();
180 G4ParticleDefinition* alpha11 = const_cast<G4ParticleDefinition*>(alpha1);
181 ph->RegisterProcess(new G4hMultipleScattering(), alpha11);
182 ph->RegisterProcess(new G4hIonisation(), alpha11);
183
184 G4bool emsc = HasMsc(eman);
185 G4bool pmsc = HasMsc(pman);
186 G4bool a2msc = HasMsc(a2man);
187 G4bool a1msc = HasMsc(a1man);
188 // G4bool imsc = HasMsc(iman);
189
190 // processes are defined with dummy models for the world
191 // elastic scatetring
192 G4DNAElastic* theDNAeElasticProcess = new G4DNAElastic("e-_G4DNAElastic");
193 theDNAeElasticProcess->SetEmModel(new G4DummyModel());
194 eman->AddDiscreteProcess(theDNAeElasticProcess);
195
196 G4DNAElastic* theDNApElasticProcess =
197 new G4DNAElastic("proton_G4DNAElastic");
198 theDNApElasticProcess->SetEmModel(new G4DummyModel());
199 pman->AddDiscreteProcess(theDNApElasticProcess);
200
201 G4DNAElastic* theDNAa2ElasticProcess =
202 new G4DNAElastic("alpha_G4DNAElastic");
203 theDNAa2ElasticProcess->SetEmModel(new G4DummyModel());
204 a2man->AddDiscreteProcess(theDNAa2ElasticProcess);
205
206 G4DNAElastic* theDNAa1ElasticProcess =
207 new G4DNAElastic("alpha+_G4DNAElastic");
208 theDNAa1ElasticProcess->SetEmModel(new G4DummyModel());
209 a1man->AddDiscreteProcess(theDNAa1ElasticProcess);
210
211 G4DNAElastic* theDNAa0ElasticProcess =
212 new G4DNAElastic("helium_G4DNAElastic");
213 theDNAa0ElasticProcess->SetEmModel(new G4DummyModel());
214 a0man->AddDiscreteProcess(theDNAa0ElasticProcess);
215
216 G4DNAElastic* theDNAh0ElasticProcess =
217 new G4DNAElastic("hydrogen_G4DNAElastic");
218 theDNAh0ElasticProcess->SetEmModel(new G4DummyModel());
219 h0man->AddDiscreteProcess(theDNAh0ElasticProcess);
220
221 // excitation
222 G4DNAExcitation* theDNAeExcProcess =
223 new G4DNAExcitation("e-_G4DNAExcitation");
224 theDNAeExcProcess->SetEmModel(new G4DummyModel());
225 eman->AddDiscreteProcess(theDNAeExcProcess);
226
227 G4DNAExcitation* theDNApExcProcess =
228 new G4DNAExcitation("proton_G4DNAExcitation");
229 theDNApExcProcess->SetEmModel(new G4DummyModel());
230 pman->AddDiscreteProcess(theDNApExcProcess);
231
232 G4DNAExcitation* theDNAa2ExcProcess =
233 new G4DNAExcitation("alpha_G4DNAExcitation");
234 theDNAa2ExcProcess->SetEmModel(new G4DummyModel());
235 a2man->AddDiscreteProcess(theDNAa2ExcProcess);
236
237 G4DNAExcitation* theDNAa1ExcProcess =
238 new G4DNAExcitation("alpha+_G4DNAExcitation");
239 theDNAa1ExcProcess->SetEmModel(new G4DummyModel());
240 a1man->AddDiscreteProcess(theDNAa1ExcProcess);
241
242 G4DNAExcitation* theDNAa0ExcProcess =
243 new G4DNAExcitation("helium_G4DNAExcitation");
244 theDNAa0ExcProcess->SetEmModel(new G4DummyModel());
245 a0man->AddDiscreteProcess(theDNAa0ExcProcess);
246
247 G4DNAExcitation* theDNAh0ExcProcess =
248 new G4DNAExcitation("hydrogen_G4DNAExcitation");
249 theDNAh0ExcProcess->SetEmModel(new G4DummyModel());
250 h0man->AddDiscreteProcess(theDNAh0ExcProcess);
251
252 // vibration excitation
253 G4DNAVibExcitation* theDNAeVibExcProcess =
254 new G4DNAVibExcitation("e-_G4DNAVibExcitation");
255 theDNAeVibExcProcess->SetEmModel(new G4DummyModel());
256 eman->AddDiscreteProcess(theDNAeVibExcProcess);
257
258 // ionisation
259 G4DNAIonisation* theDNAeIoniProcess =
260 new G4DNAIonisation("e-_G4DNAIonisation");
261 theDNAeIoniProcess->SetEmModel(new G4DummyModel());
262 eman->AddDiscreteProcess(theDNAeIoniProcess);
263
264 G4DNAIonisation* theDNApIoniProcess =
265 new G4DNAIonisation("proton_G4DNAIonisation");
266 theDNApIoniProcess->SetEmModel(new G4DummyModel());
267 pman->AddDiscreteProcess(theDNApIoniProcess);
268
269 G4DNAIonisation* theDNAa2IoniProcess =
270 new G4DNAIonisation("alpha_G4DNAIonisation");
271 theDNAa2IoniProcess->SetEmModel(new G4DummyModel());
272 a2man->AddDiscreteProcess(theDNAa2IoniProcess);
273
274 G4DNAIonisation* theDNAa1IoniProcess =
275 new G4DNAIonisation("alpha+_G4DNAIonisation");
276 theDNAa1IoniProcess->SetEmModel(new G4DummyModel());
277 a1man->AddDiscreteProcess(theDNAa1IoniProcess);
278
279 G4DNAIonisation* theDNAa0IoniProcess =
280 new G4DNAIonisation("helium_G4DNAIonisation");
281 theDNAa0IoniProcess->SetEmModel(new G4DummyModel());
282 a0man->AddDiscreteProcess(theDNAa0IoniProcess);
283
284 G4DNAIonisation* theDNAh0IoniProcess =
285 new G4DNAIonisation("hydrogen_G4DNAIonisation");
286 theDNAh0IoniProcess->SetEmModel(new G4DummyModel());
287 h0man->AddDiscreteProcess(theDNAh0IoniProcess);
288
289 G4DNAIonisation* theDNAiIoniProcess =
290 new G4DNAIonisation("GenericIon_G4DNAIonisation");
291 theDNAiIoniProcess->SetEmModel(new G4DummyModel());
292 iman->AddDiscreteProcess(theDNAiIoniProcess);
293
294 // attachment
295 G4DNAAttachment* theDNAAttachProcess =
296 new G4DNAAttachment("e-_G4DNAAttachment");
297 theDNAAttachProcess->SetEmModel(new G4DummyModel());
298 eman->AddDiscreteProcess(theDNAAttachProcess);
299
300 // charge exchange
301 G4DNAChargeDecrease* theDNApChargeDecreaseProcess =
302 new G4DNAChargeDecrease("proton_G4DNAChargeDecrease");
303 theDNApChargeDecreaseProcess->SetEmModel(new G4DummyModel());
304 pman->AddDiscreteProcess(theDNApChargeDecreaseProcess);
305
306 G4DNAChargeDecrease* theDNAa2ChargeDecreaseProcess =
307 new G4DNAChargeDecrease("alpha_G4DNAChargeDecrease");
308 theDNAa2ChargeDecreaseProcess->SetEmModel(new G4DummyModel());
309 a2man->AddDiscreteProcess(theDNAa2ChargeDecreaseProcess);
310
311 G4DNAChargeDecrease* theDNAa1ChargeDecreaseProcess =
312 new G4DNAChargeDecrease("alpha+_G4DNAChargeDecrease");
313 theDNAa1ChargeDecreaseProcess->SetEmModel(new G4DummyModel());
314 a1man->AddDiscreteProcess(theDNAa1ChargeDecreaseProcess);
315
316 G4DNAChargeIncrease* theDNAa1ChargeIncreaseProcess =
317 new G4DNAChargeIncrease("alpha+_G4DNAChargeIncrease");
318 theDNAa1ChargeIncreaseProcess->SetEmModel(new G4DummyModel());
319 a1man->AddDiscreteProcess(theDNAa1ChargeIncreaseProcess);
320
321 G4DNAChargeIncrease* theDNAa0ChargeIncreaseProcess =
322 new G4DNAChargeIncrease("helium_G4DNAChargeIncrease");
323 theDNAa0ChargeIncreaseProcess->SetEmModel(new G4DummyModel());
324 a0man->AddDiscreteProcess(theDNAa0ChargeIncreaseProcess);
325
326 G4DNAChargeIncrease* theDNAh0ChargeIncreaseProcess =
327 new G4DNAChargeIncrease("hydrogen_G4DNAChargeIncrease");
328 theDNAh0ChargeIncreaseProcess->SetEmModel(new G4DummyModel());
329 h0man->AddDiscreteProcess(theDNAh0ChargeIncreaseProcess);
330
331 // limits for DNA model applicability
332 static const G4double elowest= 7.4 * eV;
333 static const G4double elimel = 1 * MeV;
334 static const G4double pminbb = 2 * MeV;
335 static const G4double pmin = 0.1 * keV;
336 static const G4double pmax = 100 * MeV;
337 static const G4double hemin = 1 * keV;
338 static const G4double ionmin = 0.5 * MeV;
339
340 // low-energy capture
341 G4LowECapture* ecap = nullptr;
342 if(G4DNAChemistryManager::IsActivated() == false)
343 {
344 // Note: G4DNAElectronSolvation could also be used instead of G4LowECapture
345 ecap = new G4LowECapture(elowest);
346 // ecap->SetVerboseLevel(1);
347 eman->AddDiscreteProcess(ecap);
348 }
349 else
350 {
351 // When chemistry is activated: G4DNAElectronSolvation turns the electron
352 // to a solvated electron, otherwise it kills the electron at the
353 // corresponding high energy limit of the model
354 G4DNAElectronSolvation* pSolvatation =
355 new G4DNAElectronSolvation("e-_G4DNAElectronSolvation");
356 pSolvatation->SetEmModel(G4DNASolvationModelFactory::GetMacroDefinedModel());
357 eman->AddDiscreteProcess(pSolvatation);
358 }
359
360 G4LowECapture* pcap = new G4LowECapture(pmin);
361 pman->AddDiscreteProcess(pcap);
362 G4LowECapture* icap = new G4LowECapture(ionmin);
363 iman->AddDiscreteProcess(icap);
364 G4LowECapture* a2cap = new G4LowECapture(hemin);
365 a2man->AddDiscreteProcess(a2cap);
366 G4LowECapture* a1cap = new G4LowECapture(hemin);
367 a1man->AddDiscreteProcess(a1cap);
368 G4LowECapture* a0cap = new G4LowECapture(hemin);
369 a0man->AddDiscreteProcess(a0cap);
370 G4LowECapture* h0cap = new G4LowECapture(ionmin);
371 h0man->AddDiscreteProcess(h0cap);
372
373 // loop over regions
374 for(G4int i = 0; i < nreg; ++i)
375 {
376 G4String reg = regnamesDNA[i];
377 if(IsVerbose())
378 {
379 G4cout << "### DNA models type " << typesDNA[i]
380 << " are activated for G4Region " << reg << G4endl;
381 }
382
383 // type of DNA physics
384 G4int itype = 0;
385
386 if(0 == itype) {
387 AddElectronModels0(reg, ecap, emsc, elowest, elimel);
388 AddProtonModels0(reg, pmsc, elimel, pminbb, pmax);
389 AddHeliumModels0(reg, a1msc, a2msc, elimel, pminbb, pmax);
390 AddGenericIonModels0(reg, pminbb);
391 DeactivateNuclearStopping(pman, elimel);
392 DeactivateNuclearStopping(a1man, elimel);
393 DeactivateNuclearStopping(a2man, elimel);
394 }
395 }
396 G4LossTableManager::Instance()->EmConfigurator()->AddModels();
397}

401void G4EmDNAPhysicsActivator::AddElectronModels0(const G4String& reg,
402 G4LowECapture* ecap,
403 G4bool emsc,
404 G4double elowest,
405 G4double elimel)
406{
407 G4EmConfigurator* em_config =
408 G4LossTableManager::Instance()->EmConfigurator();
409 G4VEmModel* mod;
410
411 static const G4double elimin = 1 * MeV;
412 static const G4double elimvb = 100 * eV;
413 static const G4double elimat = 13 * eV;
414 static const G4double elim1 = 10 * keV;
415
416 // for e- 100 MeV is a limit between different msc models
417 G4double emax = theParameters->MaxKinEnergy();
418
419 if(emsc) {
420 G4UrbanMscModel* msc = new G4UrbanMscModel();
421 msc->SetActivationLowEnergyLimit(elimel);
422 G4double emaxmsc = std::min(100*MeV, emax);
423 em_config->SetExtraEmModel("e-", "msc", msc, reg, 0.0, emaxmsc);
424 } else {
425 mod = new G4eCoulombScatteringModel();
426 mod->SetActivationLowEnergyLimit(elimel);
427 em_config->SetExtraEmModel("e-", "CoulombScat", mod, reg, 0.0, emax);
428 }
429
430 // cuts and solvation
431 if(ecap) {
432 ecap->AddRegion(reg);
433 } else {
434 // For this condition to work, the chemistry list has to be called
435 // before any standard EM physics list
436 mod = new G4DNAOneStepThermalizationModel();
437 em_config->SetExtraEmModel("e-", "e-_G4DNAElectronSolvation",
438 mod, reg, 0., elowest);
439 }
440
441 // elastic
442 mod = new G4DNAChampionElasticModel();
443 em_config->SetExtraEmModel("e-", "e-_G4DNAElastic",
444 mod, reg, 0.0, elimel);
445 // ionisation
446 mod = new G4MollerBhabhaModel();
447 mod->SetActivationLowEnergyLimit(elimin);
448 em_config->SetExtraEmModel("e-", "eIoni",
449 mod, reg, 0.0, emax,
450 new G4UniversalFluctuation());
451
452 mod = new G4DNABornIonisationModel();
453 em_config->SetExtraEmModel("e-", "e-_G4DNAIonisation",
454 mod, reg, elim1, elimin);
455
456 mod = new G4DNAEmfietzoglouIonisationModel();
457 em_config->SetExtraEmModel("e-", "e-_G4DNAIonisation",
458 mod, reg, 0.0, elim1);
459
460 // exc
461 mod = new G4DNAEmfietzoglouExcitationModel();
462 em_config->SetExtraEmModel("e-", "e-_G4DNAExcitation",
463 mod, reg, 0.0, elim1);
464
465 mod = new G4DNABornExcitationModel();
466 em_config->SetExtraEmModel("e-", "e-_G4DNAExcitation",
467 mod, reg, elim1, elimin);
468
469 mod = new G4DNASancheExcitationModel();
470 em_config->SetExtraEmModel("e-", "e-_G4DNAVibExcitation",
471 mod, reg, 0.0, elimvb);
472
473 mod = new G4DNAMeltonAttachmentModel();
474 em_config->SetExtraEmModel("e-", "e-_G4DNAAttachment",
475 mod, reg, 0.0, elimat);
476
477}

481void G4EmDNAPhysicsActivator::AddProtonModels0(const G4String& reg,
482 G4bool pmsc, G4double elimel,
483 G4double pminbb, G4double pmax)
484{
485 G4EmConfigurator* em_config =
486 G4LossTableManager::Instance()->EmConfigurator();
487 G4VEmModel* mod;
488
489 static const G4double gmmax = 500 * keV;
490
491 G4double emax = theParameters->MaxKinEnergy();
492
493 // proton
494
495 // if SS physics list msc process does not exist
496 if(pmsc) {
497 G4WentzelVIModel* msc = new G4WentzelVIModel();
498 msc->SetActivationLowEnergyLimit(elimel);
499 em_config->SetExtraEmModel("proton", "msc", msc, reg, 0.0, emax);
500 }
501 // single scattering always applied
502 mod = new G4eCoulombScatteringModel();
503 mod->SetActivationLowEnergyLimit(elimel);
504 em_config->SetExtraEmModel("proton", "CoulombScat", mod, reg, 0.0, emax);
505
506 mod = new G4BraggModel();
507 mod->SetActivationLowEnergyLimit(std::min(pminbb, pmax));
508 em_config->SetExtraEmModel("proton", "hIoni",
509 mod, reg, 0.0, pminbb,
510 new G4UniversalFluctuation());
511
512 mod = new G4BetheBlochModel();
513 mod->SetActivationLowEnergyLimit(pmax);
514 em_config->SetExtraEmModel("proton", "hIoni",
515 mod, reg, pminbb, emax,
516 new G4UniversalFluctuation());
517
518 mod = new G4DNARuddIonisationModel();
519 em_config->SetExtraEmModel("proton", "proton_G4DNAIonisation",
520 mod, reg, 0.0, gmmax);
521
522 mod = new G4DNABornIonisationModel();
523 em_config->SetExtraEmModel("proton", "proton_G4DNAIonisation",
524 mod, reg, gmmax, pmax);
525
526 mod = new G4DNAMillerGreenExcitationModel();
527 em_config->SetExtraEmModel("proton", "proton_G4DNAExcitation",
528 mod, reg, 0.0, gmmax);
529
530 mod = new G4DNABornExcitationModel();
531 em_config->SetExtraEmModel("proton", "proton_G4DNAExcitation",
532 mod, reg, gmmax, pmax);
533
534 mod = new G4DNADingfelderChargeDecreaseModel();
535 em_config->SetExtraEmModel("proton", "proton_G4DNAChargeDecrease",
536 mod, reg, 0.0, pmax);
537
538 mod = new G4DNAIonElasticModel();
539 em_config->SetExtraEmModel("proton", "proton_G4DNAElastic",
540 mod, reg, 0.0, elimel);
541
542 // hydrogen
543 mod = new G4DNARuddIonisationModel();
544 em_config->SetExtraEmModel("hydrogen", "hydrogen_G4DNAIonisation",
545 mod, reg, 0.0, pmax);
546
547 mod = new G4DNAMillerGreenExcitationModel();
548 em_config->SetExtraEmModel("hydrogen", "hydrogen_G4DNAExcitation",
549 mod, reg, 0.0, gmmax);
550
551 mod = new G4DNADingfelderChargeIncreaseModel();
552 em_config->SetExtraEmModel("hydrogen", "hydrogen_G4DNAChargeIncrease",
553 mod, reg, 0.0, pmax);
554
555 mod = new G4DNAIonElasticModel();
556 em_config->SetExtraEmModel("hydrogen", "hydrogen_G4DNAElastic",
557 mod, reg, 0.0, elimel);
558
559}

563void G4EmDNAPhysicsActivator::AddGenericIonModels0(const G4String& reg,
564 G4double pminbb)
565{
566 G4EmConfigurator* em_config =
567 G4LossTableManager::Instance()->EmConfigurator();
568 G4VEmModel* mod;
569
570 G4double emax = theParameters->MaxKinEnergy();
571 G4double iemax = std::min(10*MeV, emax);
572 //G4double iemin = 100*eV;
573
574 mod = new G4BraggIonModel();
575 mod->SetActivationLowEnergyLimit(iemax);

720void G4EmDNAPhysicsActivator::DeactivateNuclearStopping(G4ProcessManager* pman,
721 G4double elimel)
722{
723 G4ProcessVector* pv = pman->GetProcessList();
724 G4int nproc = pman->GetProcessListLength();
725 for(G4int i = 0; i < nproc; ++i) {
726 if(((*pv)[i])->GetProcessSubType() == fNuclearStopping) {
727 G4VEmProcess* proc = static_cast<G4VEmProcess*>((*pv)[i]);
728 if(proc) {
729 G4VEmModel* mod = new G4ICRU49NuclearStoppingModel();
730 mod->SetActivationLowEnergyLimit(elimel);
731 proc->SetEmModel(mod);
732 }
733 break;
734 }
735 }
736}
