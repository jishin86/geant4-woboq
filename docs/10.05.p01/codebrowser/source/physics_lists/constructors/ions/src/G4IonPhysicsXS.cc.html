82
83void G4IonPhysicsXS::AddProcess(const G4String& name,
84 G4ParticleDefinition* part,
85 G4HadronicInteraction* theIonBC,
86 G4HadronicInteraction* theFTFP,
87 G4VCrossSectionDataSet* xs)
88{
89 G4HadronInelasticProcess* hadi = new G4HadronInelasticProcess(name, part);
90 G4ProcessManager* pManager = part->GetProcessManager();
91 pManager->AddDiscreteProcess(hadi);
92
93 if(part == G4GenericIon::GenericIon()) {
94 hadi->AddDataSet(xs);
95 } else {
96 hadi->AddDataSet(new G4ParticleInelasticXS(part));
97 }
98 hadi->RegisterMe(theIonBC);
99 hadi->RegisterMe(theFTFP);
100}
