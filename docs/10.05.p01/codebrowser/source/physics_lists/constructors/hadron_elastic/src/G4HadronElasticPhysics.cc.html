104void G4HadronElasticPhysics::ConstructProcess()
105{
106 const G4double elimitPi = 1.0*GeV;
107 const G4double elimitAntiNuc = 100.*MeV;
108 const G4double delta = 0.1*MeV;
109 if(verbose > 1) {
110 G4cout << "### HadronElasticPhysics::ConstructProcess: Elimit for pi "
111 << elimitPi/GeV << " GeV" << G4endl;
112 G4cout << " for anti-neuclei "
113 << elimitAntiNuc/GeV << " GeV" << G4endl;
114 }
115
116 G4AntiNuclElastic* anuc = new G4AntiNuclElastic();
117 anuc->SetMinEnergy(elimitAntiNuc);
118 G4CrossSectionElastic* anucxs =
119 new G4CrossSectionElastic(anuc->GetComponentCrossSection());
120
121 G4HadronElastic* lhep0 = new G4HadronElastic();
122 G4HadronElastic* lhep1 = new G4HadronElastic();
123 G4HadronElastic* lhep2 = new G4HadronElastic();
124 lhep1->SetMaxEnergy(elimitPi+delta);
125 lhep2->SetMaxEnergy(elimitAntiNuc+delta);
126
127 G4ElasticHadrNucleusHE* he = new G4ElasticHadrNucleusHE();
128 he->SetMinEnergy(elimitPi);
129
130 G4VCrossSectionDataSet* theComponentGGHadronNucleusData =
131 new G4CrossSectionElastic( new G4ComponentGGHadronNucleusXsc );
132
133 G4HadronElasticProcess* hel = nullptr;
134
135 auto myParticleIterator=GetParticleIterator();
136 myParticleIterator->reset();
137 while( (*myParticleIterator)() )
138 {
139 G4ParticleDefinition* particle = myParticleIterator->value();
140 G4ProcessManager* pmanager = particle->GetProcessManager();
141 G4String pname = particle->GetParticleName();
142 if(pname == "anti_lambda" ||
143 pname == "anti_neutron" ||
144 pname == "anti_omega-" ||
145 pname == "anti_sigma-" ||
146 pname == "anti_sigma+" ||
147 pname == "anti_xi-" ||
148 pname == "anti_xi0" ||
149 pname == "lambda" ||
150 pname == "omega-" ||
151 pname == "sigma-" ||
152 pname == "sigma+" ||
153 pname == "xi-"
154 ) {
155
156 hel = new G4HadronElasticProcess();
157 hel->RegisterMe(lhep0);
158 pmanager->AddDiscreteProcess(hel);
159 if(verbose > 1) {
160 G4cout << "### HadronElasticPhysics: " << hel->GetProcessName()
161 << " added for " << particle->GetParticleName() << G4endl;
162 }
163
164 } else if(pname == "alpha" ||
165 pname == "deuteron" ||
166 pname == "triton" ||
167 pname == "He3"
168 ) {
169 hel = new G4HadronElasticProcess();
170 G4VCrossSectionDataSet* theComponentGGNuclNuclData =
171 new G4CrossSectionElastic(new G4ComponentGGNuclNuclXsc());
172 hel->AddDataSet(theComponentGGNuclNuclData);
173 hel->RegisterMe(lhep0);
174 pmanager->AddDiscreteProcess(hel);
175 if(verbose > 1) {
176 G4cout << "### HadronElasticPhysics: " << hel->GetProcessName()
177 << " added for " << particle->GetParticleName() << G4endl;
178 }
179
180 } else if(pname == "proton") {
181
182 hel = new G4HadronElasticProcess();
183 hel->AddDataSet(new G4BGGNucleonElasticXS(particle));
184
185 //hel->AddDataSet(G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4ChipsProtonElasticXS::Default_Name()));
186
187 hel->RegisterMe(new G4ChipsElasticModel());
188 pmanager->AddDiscreteProcess(hel);
189 if(verbose > 1) {
190 G4cout << "### HadronElasticPhysics: " << hel->GetProcessName()
191 << " added for " << particle->GetParticleName() << G4endl;
192 }
193
194 } else if(pname == "neutron") {
195
196 hel = new G4HadronElasticProcess();
197 hel->AddDataSet(G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4NeutronElasticXS::Default_Name()));
198 hel->RegisterMe(new G4ChipsElasticModel());
199 pmanager->AddDiscreteProcess(hel);
200 if(verbose > 1) {
201 G4cout << "### HadronElasticPhysics: " << hel->GetProcessName()
202 << " added for " << particle->GetParticleName() << G4endl;
203 }
204
205 } else if (pname == "pi+" || pname == "pi-") {
206
207 hel = new G4HadronElasticProcess();
208 hel->AddDataSet(new G4BGGPionElasticXS(particle));
209 hel->RegisterMe(lhep1);
210 hel->RegisterMe(he);
211 pmanager->AddDiscreteProcess(hel);
212 if(verbose > 1) {
213 G4cout << "### HadronElasticPhysics: " << hel->GetProcessName()
214 << " added for " << particle->GetParticleName() << G4endl;
215 }
216
217 } else if(pname == "kaon-" ||
218 pname == "kaon+" ||
219 pname == "kaon0S" ||
220 pname == "kaon0L"
221 ) {
222
223 hel = new G4HadronElasticProcess();
224 //AR-14Aug2017 : Replaced Gheisha elastic kaon cross sections with
225 // Grichine's Glauber-Gribov ones. In this way, the
226 // total (elastic + inelastic) kaon cross sections
227 // are consistent with the PDG ones.
228 // For the time being, kept Gheisha elastic as
229 // final-state model.
230 hel->AddDataSet( theComponentGGHadronNucleusData );
231 hel->RegisterMe(lhep0);
232 pmanager->AddDiscreteProcess(hel);
233 if(verbose > 1) {
234 G4cout << "### HadronElasticPhysics: " << hel->GetProcessName()
235 << " added for " << particle->GetParticleName() << G4endl;
236 }
237
238 } else if(
239 pname == "anti_proton" ||
240 pname == "anti_alpha" ||
241 pname == "anti_deuteron" ||
242 pname == "anti_triton" ||
243 pname == "anti_He3" ) {
244
245 hel = new G4HadronElasticProcess();
246 hel->AddDataSet(anucxs);
247 hel->RegisterMe(lhep2);
248 hel->RegisterMe(anuc);
249 pmanager->AddDiscreteProcess(hel);
250 }
251 }
252}

254G4HadronicProcess*
255G4HadronElasticPhysics::GetElasticProcess(const G4ParticleDefinition* part) const
256{
257 G4HadronicProcess* hp = nullptr;
258 G4ProcessVector* pv = part->GetProcessManager()->GetPostStepProcessVector();
259 size_t n = pv->size();
260 for(size_t i=0; i<n; ++i) {
261 if((*pv)[i]->GetProcessSubType() == fHadronElastic) {
262 hp = static_cast<G4HadronicProcess*>((*pv)[i]);
263 break;
264 }
265 }
266 return hp;
267}

269G4HadronElastic*
270G4HadronElasticPhysics::GetElasticModel(const G4ParticleDefinition* part) const
271{
272 G4HadronElastic* mod = nullptr;
273 G4HadronicProcess* hel = GetElasticProcess(part);
274 if(hel) {
275 std::vector<G4HadronicInteraction*>& hi = hel->GetHadronicInteractionList();
276 if(hi.size() > 0) { mod = static_cast<G4HadronElastic*>(hi[0]); }
277 }
278 return mod;
279}

281G4HadronicProcess* G4HadronElasticPhysics::GetNeutronProcess() const
282{
283 return GetElasticProcess(G4Neutron::Neutron());
284}

286G4HadronElastic* G4HadronElasticPhysics::GetNeutronModel() const

291void G4HadronElasticPhysics::AddXSection(const G4ParticleDefinition* part,
292 G4VCrossSectionDataSet* cross) const
293{
294 G4HadronicProcess* hel = GetElasticProcess(part);
295 if(hel) { hel->AddDataSet(cross); }
