72void G4HadronPhysicsQGSP_BIC_AllHP::CreateModels()
73{
74 G4bool quasiElasticFTF= false; // Use built-in quasi-elastic (not add-on)
75 G4bool quasiElasticQGS= true; // For QGS, it must use it.
76
77 tpdata->theNeutrons=new G4NeutronBuilder( true ); // Fission on
78 tpdata->theNeutrons->RegisterMe(tpdata->theQGSPNeutron=new G4QGSPNeutronBuilder(quasiElasticQGS));
79 tpdata->theNeutrons->RegisterMe(tpdata->theFTFPNeutron=new G4FTFPNeutronBuilder(quasiElasticFTF));
80 tpdata->theFTFPNeutron->SetMinEnergy(minFTFP);
81 tpdata->theFTFPNeutron->SetMaxEnergy(maxFTFP);
82 tpdata->theNeutrons->RegisterMe(tpdata->theBinaryNeutron=new G4BinaryNeutronBuilder);
83 tpdata->theBinaryNeutron->SetMinEnergy(minBIC_neutron);
84 tpdata->theBinaryNeutron->SetMaxEnergy(maxBIC);
85 tpdata->theNeutrons->RegisterMe(tpdata->thePHPNeutron=new G4NeutronPHPBuilder);
86 tpdata->thePHPNeutron->SetMaxEnergy(maxHP_neutron);
87
88 tpdata->thePro=new G4ProtonBuilder;
89 tpdata->thePro->RegisterMe(tpdata->theQGSPPro=new G4QGSPProtonBuilder(quasiElasticQGS));
90 tpdata->thePro->RegisterMe(tpdata->theFTFPPro=new G4FTFPProtonBuilder(quasiElasticFTF));
91 tpdata->theFTFPPro->SetMinEnergy(minFTFP);
92 tpdata->theFTFPPro->SetMaxEnergy(maxFTFP);
93 tpdata->thePro->RegisterMe(tpdata->theBinaryPro=new G4BinaryProtonBuilder);
94 tpdata->theBinaryPro->SetMinEnergy(minBIC_proton);
95 tpdata->theBinaryPro->SetMaxEnergy(maxBIC);
96 tpdata->thePro->RegisterMe(tpdata->thePHPProton=new G4ProtonPHPBuilder);
97 tpdata->thePHPProton->SetMaxEnergy(maxHP_proton);
98
99 tpdata->thePiK=new G4PiKBuilder;
100 tpdata->thePiK->RegisterMe(tpdata->theQGSPPiK=new G4QGSPPiKBuilder(quasiElasticQGS));
101 tpdata->thePiK->RegisterMe(tpdata->theFTFPPiK=new G4FTFPPiKBuilder(quasiElasticFTF));
102 tpdata->theFTFPPiK->SetMaxEnergy(maxFTFP);
103 tpdata->thePiK->RegisterMe(tpdata->theBertiniPiK=new G4BertiniPiKBuilder);
104 tpdata->theBertiniPiK->SetMaxEnergy(maxBERT);
105
106 tpdata->theHyperon=new G4HyperonFTFPBuilder;
107
108 tpdata->theAntiBaryon=new G4AntiBarionBuilder;
109 tpdata->theAntiBaryon->RegisterMe(tpdata->theFTFPAntiBaryon=new G4FTFPAntiBarionBuilder(quasiElasticFTF));
110}

151#include "G4ProcessManager.hh"
152void G4HadronPhysicsQGSP_BIC_AllHP::ConstructProcess()
153{
154 if ( tpdata == 0 ) tpdata = new ThreadPrivate;
155 CreateModels();
156 tpdata->theNeutrons->Build();
157 tpdata->thePro->Build();
158 tpdata->thePiK->Build();
159
160 // --- Kaons ---
161 G4VCrossSectionDataSet * kaonxs =
162 new G4CrossSectionInelastic(new G4ComponentGGHadronNucleusXsc());
163 G4PhysListUtil::FindInelasticProcess(G4KaonMinus::KaonMinus())->AddDataSet(kaonxs);
164 G4PhysListUtil::FindInelasticProcess(G4KaonPlus::KaonPlus())->AddDataSet(kaonxs);
165 G4PhysListUtil::FindInelasticProcess(G4KaonZeroShort::KaonZeroShort())->AddDataSet(kaonxs);
166 G4PhysListUtil::FindInelasticProcess(G4KaonZeroLong::KaonZeroLong())->AddDataSet(kaonxs);
167
168 tpdata->theHyperon->Build();
169 tpdata->theAntiBaryon->Build();
170
171 // --- Neutrons ---
172 const G4ParticleDefinition* neutron = G4Neutron::Neutron();
173 G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
174 if (capture) {
175 G4NeutronRadCapture* theNeutronRadCapture = new G4NeutronRadCapture();
176 theNeutronRadCapture->SetMinEnergy( minBIC_neutron );
177 capture->RegisterMe( theNeutronRadCapture );
178 }
179 G4HadronicProcess* fission = G4PhysListUtil::FindFissionProcess(neutron);
180 if (fission) {
181 G4LFission* theNeutronLEPFission = new G4LFission();
182 theNeutronLEPFission->SetMinEnergy( minBIC_neutron );
183 theNeutronLEPFission->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
184 fission->RegisterMe( theNeutronLEPFission );
185 }
