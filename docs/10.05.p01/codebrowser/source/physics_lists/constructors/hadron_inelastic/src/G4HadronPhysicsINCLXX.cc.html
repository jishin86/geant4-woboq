134void G4HadronPhysicsINCLXX::Neutron()
135{
136 //General schema:
137 // 1) Create a builder
138 // 2) Call AddBuilder
139 // 3) Configure the builder, possibly with sub-builders
140 // 4) Call builder->Build()
141 auto neu = new G4NeutronBuilder( withNeutronHP );
142 AddBuilder(neu);
143 G4PhysicsBuilderInterface* string = nullptr;
144 if(withFTFP) {
145 string = new G4FTFPNeutronBuilder(quasiElasticFTF);
146 } else {
147 string = new G4QGSPNeutronBuilder(quasiElasticQGS);
148 }
149 string->SetMinEnergy(15.*GeV);
150 AddBuilder(string);
151 neu->RegisterMe(string);
152
153 auto inclxxn = new G4INCLXXNeutronBuilder;
154 inclxxn->SetMaxEnergy(20.*GeV);
155 AddBuilder(inclxxn);
156 neu->RegisterMe(inclxxn);
157
158 if(withNeutronHP) {
159 inclxxn->UsePreCompound(false);
160 inclxxn->SetMinEnergy(19.9*MeV);
161 auto hpn = new G4NeutronPHPBuilder;
162 AddBuilder(hpn);
163 neu->RegisterMe(hpn);
164 } else {
165 inclxxn->UsePreCompound(true);
166 inclxxn->SetMinPreCompoundEnergy(0.0*MeV);
167 inclxxn->SetMaxPreCompoundEnergy(2.0*MeV);
168 inclxxn->SetMinEnergy(1.0*MeV);
169 }
170
171 neu->Build();
172}

174void G4HadronPhysicsINCLXX::Proton()
175{
176 auto pro =new G4ProtonBuilder;
177 AddBuilder(pro);
178 G4PhysicsBuilderInterface* string = nullptr;
179 if(withFTFP) {
180 string = new G4FTFPProtonBuilder(quasiElasticFTF);
181 } else {
182 string = new G4QGSPProtonBuilder(quasiElasticQGS);
183 }
184 string->SetMinEnergy(15.*GeV);
185 AddBuilder(string);
186 pro->RegisterMe(string);
187
188 auto inclxxp = new G4INCLXXProtonBuilder;
189 AddBuilder(inclxxp);
190 inclxxp->SetMinEnergy(1.0*MeV);
191 inclxxp->SetMaxEnergy(20.0*GeV);
192 pro->RegisterMe(inclxxp);
193 pro->Build();
194}

219void G4HadronPhysicsINCLXX::Kaon()
220{
221 auto k = new G4KaonBuilder;
222 AddBuilder(k);
223 G4PhysicsBuilderInterface* string = nullptr;
224 if(withFTFP) {
225 string = new G4FTFPKaonBuilder(quasiElasticFTF);
226 } else {
227 string = new G4QGSPKaonBuilder(quasiElasticQGS);
228 }
229 string->SetMinEnergy(14.*GeV);
230 AddBuilder(string);
231 k->RegisterMe(string);
232
233 auto bert = new G4BertiniKaonBuilder;
234 bert->SetMinEnergy(0.0*GeV);
235 bert->SetMaxEnergy(15.0*GeV);
236 AddBuilder(bert);
237 k->RegisterMe(bert);
238
239 k->Build();
240}

280void G4HadronPhysicsINCLXX::ExtraConfiguration()
281{
282 // --- Kaons ---
283 auto xsk = new G4ComponentGGHadronNucleusXsc();
284 G4VCrossSectionDataSet * kaonxs = new G4CrossSectionInelastic(xsk);
285 G4PhysListUtil::FindInelasticProcess(G4KaonMinus::KaonMinus())->AddDataSet(kaonxs);
286 G4PhysListUtil::FindInelasticProcess(G4KaonPlus::KaonPlus())->AddDataSet(kaonxs);
287 G4PhysListUtil::FindInelasticProcess(G4KaonZeroShort::KaonZeroShort())->AddDataSet(kaonxs);
288 G4PhysListUtil::FindInelasticProcess(G4KaonZeroLong::KaonZeroLong())->AddDataSet(kaonxs);
289
290 // --- Neutrons ---
291 const G4ParticleDefinition* neutron = G4Neutron::Neutron();
292 G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
293 if (capture) {
294 G4NeutronRadCapture* theNeutronRadCapture = new G4NeutronRadCapture();
295 capture->RegisterMe(theNeutronRadCapture);
296 if ( withNeutronHP ) {
297 capture->AddDataSet( new G4ParticleHPCaptureData );
298 theNeutronRadCapture->SetMinEnergy( 19.9*MeV );
299 }
300 }
301 G4HadronicProcess* fission = G4PhysListUtil::FindFissionProcess(neutron);
302 if (fission && withNeutronHP) {
303 G4LFission* theNeutronLEPFission = new G4LFission();
304 theNeutronLEPFission->SetMinEnergy( 19.9*MeV );
305 theNeutronLEPFission->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
306 fission->RegisterMe( theNeutronLEPFission );
307 }
