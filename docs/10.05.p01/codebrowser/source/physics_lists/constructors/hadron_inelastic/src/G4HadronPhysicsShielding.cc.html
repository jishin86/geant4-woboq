174void G4HadronPhysicsShielding::Neutron()
175{
176 auto neu = new G4NeutronBuilder( true );
177 AddBuilder( neu );
178 auto ftfpn = new G4FTFPNeutronBuilder( false );
179 AddBuilder( ftfpn );
180 neu->RegisterMe( ftfpn );
181 ftfpn->SetMinEnergy( minFTFPEnergy_ );
182 auto bertn = new G4BertiniNeutronBuilder;
183 AddBuilder( bertn );
184 neu->RegisterMe( bertn );
185 bertn->SetMinEnergy( minNonHPNeutronEnergy_ );
186 bertn->SetMaxEnergy( maxBertiniEnergy_ );
187 if ( useLEND_ ) {
188 auto lendn = new G4NeutronLENDBuilder( evaluation_ );
189 AddBuilder( lendn );
190 neu->RegisterMe( lendn );
191 } else {
192 auto hpn = new G4NeutronPHPBuilder;
193 AddBuilder( hpn );
194 neu->RegisterMe( hpn );
195 }
196 neu->Build();
197}

275void G4HadronPhysicsShielding::ExtraConfiguration()
276{
277 // Modify cross sections for kaons
278 auto xsk = new G4ComponentGGHadronNucleusXsc();
279 G4VCrossSectionDataSet* kaonxs = new G4CrossSectionInelastic( xsk );
280 G4PhysListUtil::FindInelasticProcess( G4KaonMinus::KaonMinus() )->AddDataSet( kaonxs );
281 G4PhysListUtil::FindInelasticProcess( G4KaonPlus::KaonPlus() )->AddDataSet( kaonxs );
282 G4PhysListUtil::FindInelasticProcess( G4KaonZeroShort::KaonZeroShort() )->AddDataSet( kaonxs );
283 G4PhysListUtil::FindInelasticProcess( G4KaonZeroLong::KaonZeroLong() )->AddDataSet( kaonxs );
284
285 // Modify neutrons
286 const G4ParticleDefinition* neutron = G4Neutron::Neutron();
287 G4HadronicProcess* inel = G4PhysListUtil::FindInelasticProcess( neutron );
288 if ( inel ) {
289 // Register the G4ParticleHPJENDLHEInelasticData as the 2nd priority.
290 inel->GetCrossSectionDataStore()->AddDataSet( new G4ParticleHPJENDLHEInelasticData, 1 );
291 }
292 G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess( neutron );
293 if ( capture ) {
294 G4NeutronRadCapture* theNeutronRadCapture = new G4NeutronRadCapture;
295 theNeutronRadCapture->SetMinEnergy( minNonHPNeutronEnergy_ );
296 capture->RegisterMe( theNeutronRadCapture );
297 }
298 G4HadronicProcess* fission = G4PhysListUtil::FindFissionProcess( neutron );
299 if ( fission ) {
300 G4LFission* theNeutronLEPFission = new G4LFission;
301 theNeutronLEPFission->SetMinEnergy( minNonHPNeutronEnergy_ );
302 theNeutronLEPFission->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
303 fission->RegisterMe( theNeutronLEPFission );
304 }
305}
