94 : G4VPhysicsConstructor(name)

96{

98 maxBERT_pion = 12.0 * GeV;

101 minFTFP_proton = 3.0 * GeV;
102 maxBERT_proton = 12.0 * GeV;
103 minFTFP_neutron = 3.0 * GeV;
104 maxBERT_neutron = 12.0 * GeV;
105}

122void G4HadronPhysicsFTFP_BERT::DumpBanner()
123{
124 G4cout << G4endl
125 << " FTFP_BERT : new threshold between BERT and FTFP is over the interval " << G4endl
126 << " for pions : " << minFTFP_pion/GeV << " to " << maxBERT_pion/GeV << " GeV" << G4endl
127 << " for kaons : " << minFTFP_kaon/GeV << " to " << maxBERT_kaon/GeV << " GeV" << G4endl
128 << " for proton : " << minFTFP_proton/GeV << " to " << maxBERT_proton/GeV << " GeV" << G4endl
129 << " for neutron : " << minFTFP_neutron/GeV << " to " << maxBERT_neutron/GeV << " GeV" << G4endl
130 << G4endl;
131}

142void G4HadronPhysicsFTFP_BERT::Neutron()
143{
144 //General schema:
145 // 1) Create a builder
146 // 2) Call AddBuilder
147 // 3) Configure the builder, possibly with sub-builders
148 // 4) Call builder->Build()
149 auto neu = new G4NeutronBuilder;
150 AddBuilder(neu);
151 auto ftfpn = new G4FTFPNeutronBuilder(QuasiElastic);
152 AddBuilder( ftfpn );
153 neu->RegisterMe(ftfpn);
154 ftfpn->SetMinEnergy(minFTFP_neutron);
155 auto bertn = new G4BertiniNeutronBuilder;
156 AddBuilder(bertn);
157 neu->RegisterMe(bertn);
158 bertn->SetMinEnergy(0.*GeV);
159 bertn->SetMaxEnergy(maxBERT_neutron);
160 neu->Build();
161}

233#include "G4ProcessManager.hh"
234void G4HadronPhysicsFTFP_BERT::ExtraConfiguration()
235{
236 //Modify XS for kaons
237 auto xsk = new G4ComponentGGHadronNucleusXsc();
238 G4VCrossSectionDataSet * kaonxs = new G4CrossSectionInelastic(xsk);
239 G4PhysListUtil::FindInelasticProcess(G4KaonMinus::KaonMinus())->AddDataSet(kaonxs);
240 G4PhysListUtil::FindInelasticProcess(G4KaonPlus::KaonPlus())->AddDataSet(kaonxs);
241 G4PhysListUtil::FindInelasticProcess(G4KaonZeroShort::KaonZeroShort())->AddDataSet(kaonxs);
242 G4PhysListUtil::FindInelasticProcess(G4KaonZeroLong::KaonZeroLong())->AddDataSet(kaonxs);
243
244 //Modify Neutrons
245 const G4ParticleDefinition* neutron = G4Neutron::Neutron();
246 G4HadronicProcess* inel = G4PhysListUtil::FindInelasticProcess(neutron);
247 if(inel) { inel->AddDataSet(new G4NeutronInelasticXS()); }
248 G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
249 if (capture) {
250 capture->RegisterMe(new G4NeutronRadCapture());
251 }
252}
