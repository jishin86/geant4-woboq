33//
34// Sympson integration method
35//
36/////////////////////////////////////////////////////////////////////
37//
38// Integration of class member functions T::f by Simpson method.
39
40template <class T, class F>
41G4double G4Integrator<T,F>::Simpson( T& typeT,
42 F f,
43 G4double xInitial,
44 G4double xFinal,
45 G4int iterationNumber )
46{
47 G4int i ;
48 G4double step = (xFinal - xInitial)/iterationNumber ;
49 G4double x = xInitial ;
50 G4double xPlus = xInitial + 0.5*step ;
51 G4double mean = ( (typeT.*f)(xInitial) + (typeT.*f)(xFinal) )*0.5 ;
52 G4double sum = (typeT.*f)(xPlus) ;
53
54 for(i=1;i<iterationNumber;i++)
55 {
56 x += step ;
57 xPlus += step ;
58 mean += (typeT.*f)(x) ;
59 sum += (typeT.*f)(xPlus) ;
60 }
61 mean += 2.0*sum ;
62
63 return mean*step/3.0 ;
64}

136template <class T, class F>
137G4double G4Integrator<T,F>::Gauss( T& typeT, F f,
138 G4double xInitial, G4double xFinal )
139{
140 static const G4double root = 1.0/std::sqrt(3.0) ;
141
142 G4double xMean = (xInitial + xFinal)/2.0 ;
143 G4double Step = (xFinal - xInitial)/2.0 ;
144 G4double delta = Step*root ;
145 G4double sum = ((typeT.*f)(xMean + delta) +
146 (typeT.*f)(xMean - delta)) ;
147
148 return sum*Step ;
149}

165template <class T, class F>
166G4double G4Integrator<T,F>::Gauss( G4double (*f)(G4double),
167 G4double xInitial, G4double xFinal)
168{
169 static const G4double root = 1.0/std::sqrt(3.0) ;
170
171 G4double xMean = (xInitial + xFinal)/2.0 ;
172 G4double Step = (xFinal - xInitial)/2.0 ;
173 G4double delta = Step*root ;
174 G4double sum = ( (*f)(xMean + delta) + (*f)(xMean - delta) ) ;
175
176 return sum*Step ;
177}

183template <class T, class F>
184void G4Integrator<T,F>::AdaptGauss( T& typeT, F f, G4double xInitial,
185 G4double xFinal, G4double fTolerance,
186 G4double& sum,
187 G4int& depth )
188{
189 if(depth > 100)
190 {
191 G4cout<<"G4Integrator<T,F>::AdaptGauss: WARNING !!!"<<G4endl ;
192 G4cout<<"Function varies too rapidly to get stated accuracy in 100 steps "
193 <<G4endl ;
194
195 return ;
196 }
197 G4double xMean = (xInitial + xFinal)/2.0 ;
198 G4double leftHalf = Gauss(typeT,f,xInitial,xMean) ;
199 G4double rightHalf = Gauss(typeT,f,xMean,xFinal) ;
200 G4double full = Gauss(typeT,f,xInitial,xFinal) ;
201 if(std::fabs(leftHalf+rightHalf-full) < fTolerance)
202 {
203 sum += full ;
204 }
205 else
206 {
207 depth++ ;
208 AdaptGauss(typeT,f,xInitial,xMean,fTolerance,sum,depth) ;
209 AdaptGauss(typeT,f,xMean,xFinal,fTolerance,sum,depth) ;
210 }
211}

226void G4Integrator<T,F>::AdaptGauss( G4double (*f)(G4double),
227 G4double xInitial, G4double xFinal,
228 G4double fTolerance, G4double& sum,
229 G4int& depth )
230{
231 if(depth > 100)

236
237 return ;
238 }

241 G4double rightHalf = Gauss(f,xMean,xFinal) ;
242 G4double full = Gauss(f,xInitial,xFinal) ;
243 if(std::fabs(leftHalf+rightHalf-full) < fTolerance)
244 {
245 sum += full ;
246 }
247 else
248 {
249 depth++ ;
250 AdaptGauss(f,xInitial,xMean,fTolerance,sum,depth) ;
251 AdaptGauss(f,xMean,xFinal,fTolerance,sum,depth) ;
252 }
253}

297////////////////////////////////////////////////////////////////////////////
298// Gauss integration methods involving ortogonal polynomials
299////////////////////////////////////////////////////////////////////////////
300//
301// Methods involving Legendre polynomials
302//
303/////////////////////////////////////////////////////////////////////////
304//
305// The value nLegendre set the accuracy required, i.e the number of points
306// where the function pFunction will be evaluated during integration.
307// The function creates the arrays for abscissas and weights that used
308// in Gauss-Legendre quadrature method.
309// The values a and b are the limits of integration of the function f .
310// nLegendre MUST BE EVEN !!!
311// Returns the integral of the function f between a and b, by 2*fNumber point
312// Gauss-Legendre integration: the function is evaluated exactly
313// 2*fNumber times at interior points in the range of integration.
314// Since the weights and abscissas are, in this case, symmetric around
315// the midpoint of the range of integration, there are actually only
316// fNumber distinct values of each.
317// Convenient for using with some class object dataT
318
319template <class T, class F>
320G4double G4Integrator<T,F>::Legendre( T& typeT, F f, G4double a, G4double b,
321 G4int nLegendre )
322{
323 G4double nwt, nwt1, temp1, temp2, temp3, temp ;
324 G4double xDiff, xMean, dx, integral ;
325
326 const G4double tolerance = 1.6e-10 ;
327 G4int i, j, k = nLegendre ;
328 G4int fNumber = (nLegendre + 1)/2 ;
329
330 if(2*fNumber != k)
331 {
332 G4Exception("G4Integrator<T,F>::Legendre(T&,F, ...)", "InvalidCall",
333 FatalException, "Invalid (odd) nLegendre in constructor.");
334 }
335
336 G4double* fAbscissa = new G4double[fNumber] ;
337 G4double* fWeight = new G4double[fNumber] ;
338
339 for(i=1;i<=fNumber;i++) // Loop over the desired roots
340 {
341 nwt = std::cos(CLHEP::pi*(i - 0.25)/(k + 0.5)) ; // Initial root approximation
342
343 do // loop of Newton's method
344 {
345 temp1 = 1.0 ;
346 temp2 = 0.0 ;
347 for(j=1;j<=k;j++)
348 {
349 temp3 = temp2 ;
350 temp2 = temp1 ;
351 temp1 = ((2.0*j - 1.0)*nwt*temp2 - (j - 1.0)*temp3)/j ;
352 }
353 temp = k*(nwt*temp1 - temp2)/(nwt*nwt - 1.0) ;
354 nwt1 = nwt ;
355 nwt = nwt1 - temp1/temp ; // Newton's method
356 }
357 while(std::fabs(nwt - nwt1) > tolerance) ;
358
359 fAbscissa[fNumber-i] = nwt ;
360 fWeight[fNumber-i] = 2.0/((1.0 - nwt*nwt)*temp*temp) ;
361 }
362
363 //
364 // Now we ready to get integral
365 //
366
367 xMean = 0.5*(a + b) ;
368 xDiff = 0.5*(b - a) ;
369 integral = 0.0 ;
370 for(i=0;i<fNumber;i++)
371 {
372 dx = xDiff*fAbscissa[i] ;
373 integral += fWeight[i]*( (typeT.*f)(xMean + dx) +
374 (typeT.*f)(xMean - dx) ) ;
375 }
376 delete[] fAbscissa;
377 delete[] fWeight;
378 return integral *= xDiff ;
379}

393//
394// Convenient for using with global scope function f
395
396template <class T, class F>
397G4double G4Integrator<T,F>::Legendre( G4double (*f)(G4double),
398 G4double a, G4double b, G4int nLegendre)
399{
400 G4double nwt, nwt1, temp1, temp2, temp3, temp ;
401 G4double xDiff, xMean, dx, integral ;
402
403 const G4double tolerance = 1.6e-10 ;
404 G4int i, j, k = nLegendre ;
405 G4int fNumber = (nLegendre + 1)/2 ;
406
407 if(2*fNumber != k)
408 {
409 G4Exception("G4Integrator<T,F>::Legendre(...)", "InvalidCall",
410 FatalException, "Invalid (odd) nLegendre in constructor.");
411 }
412
413 G4double* fAbscissa = new G4double[fNumber] ;
414 G4double* fWeight = new G4double[fNumber] ;
415
416 for(i=1;i<=fNumber;i++) // Loop over the desired roots
417 {
418 nwt = std::cos(CLHEP::pi*(i - 0.25)/(k + 0.5)) ; // Initial root approximation
419
420 do // loop of Newton's method
421 {
422 temp1 = 1.0 ;
423 temp2 = 0.0 ;
424 for(j=1;j<=k;j++)
425 {
426 temp3 = temp2 ;
427 temp2 = temp1 ;
428 temp1 = ((2.0*j - 1.0)*nwt*temp2 - (j - 1.0)*temp3)/j ;
429 }
430 temp = k*(nwt*temp1 - temp2)/(nwt*nwt - 1.0) ;
431 nwt1 = nwt ;
432 nwt = nwt1 - temp1/temp ; // Newton's method
433 }
434 while(std::fabs(nwt - nwt1) > tolerance) ;
435
436 fAbscissa[fNumber-i] = nwt ;
437 fWeight[fNumber-i] = 2.0/((1.0 - nwt*nwt)*temp*temp) ;
438 }
439
440 //
441 // Now we ready to get integral
442 //
443
444 xMean = 0.5*(a + b) ;
445 xDiff = 0.5*(b - a) ;
446 integral = 0.0 ;
447 for(i=0;i<fNumber;i++)
448 {
449 dx = xDiff*fAbscissa[i] ;
450 integral += fWeight[i]*( (*f)(xMean + dx) + (*f)(xMean - dx) ) ;
451 }
452 delete[] fAbscissa;
453 delete[] fWeight;
454
455 return integral *= xDiff ;
456}

458////////////////////////////////////////////////////////////////////////////
459//
460// Returns the integral of the function to be pointed by T::f between a and b,
461// by ten point Gauss-Legendre integration: the function is evaluated exactly
462// ten times at interior points in the range of integration. Since the weights
463// and abscissas are, in this case, symmetric around the midpoint of the
464// range of integration, there are actually only five distinct values of each
465// Convenient for using with class object typeT
466
467template <class T, class F>
468G4double G4Integrator<T,F>::Legendre10( T& typeT, F f,G4double a, G4double b)
469{
470 G4int i ;
471 G4double xDiff, xMean, dx, integral ;
472
473 // From Abramowitz M., Stegan I.A. 1964 , Handbook of Math... , p. 916
474
475 static const G4double abscissa[] = { 0.148874338981631, 0.433395394129247,
476 0.679409568299024, 0.865063366688985,
477 0.973906528517172 } ;
478
479 static const G4double weight[] = { 0.295524224714753, 0.269266719309996,
480 0.219086362515982, 0.149451349150581,
481 0.066671344308688 } ;
482 xMean = 0.5*(a + b) ;
483 xDiff = 0.5*(b - a) ;
484 integral = 0.0 ;
485 for(i=0;i<5;i++)
486 {
487 dx = xDiff*abscissa[i] ;
488 integral += weight[i]*( (typeT.*f)(xMean + dx) + (typeT.*f)(xMean - dx)) ;
489 }
490 return integral *= xDiff ;
491}

741// quadrature method.

825//////////////////////////////////////////////////////////////////////
826//
827// Method involving Laguerre polynomials
828//
829//////////////////////////////////////////////////////////////////////
830//
831// Integral from zero to infinity of std::pow(x,alpha)*std::exp(-x)*f(x).
832// The value of nLaguerre sets the accuracy.
833// The function creates arrays fAbscissa[0,..,nLaguerre-1] and
834// fWeight[0,..,nLaguerre-1] .
835// Convenient for using with class object 'typeT' and (typeT.*f) function
836// (T::f)
837
838template <class T, class F>
839G4double G4Integrator<T,F>::Laguerre( T& typeT, F f, G4double alpha,
840 G4int nLaguerre )
841{
842 const G4double tolerance = 1.0e-10 ;
843 const G4int maxNumber = 12 ;
844 G4int i, j, k ;
845 G4double nwt=0., nwt1, temp1, temp2, temp3, temp, cofi ;
846 G4double integral = 0.0 ;
847
848 G4int fNumber = nLaguerre ;
849 G4double* fAbscissa = new G4double[fNumber] ;
850 G4double* fWeight = new G4double[fNumber] ;
851
852 for(i=1;i<=fNumber;i++) // Loop over the desired roots
853 {
854 if(i == 1)
855 {
856 nwt = (1.0 + alpha)*(3.0 + 0.92*alpha)
857 / (1.0 + 2.4*fNumber + 1.8*alpha) ;
858 }
859 else if(i == 2)
860 {
861 nwt += (15.0 + 6.25*alpha)/(1.0 + 0.9*alpha + 2.5*fNumber) ;
862 }
863 else
864 {
865 cofi = i - 2 ;
866 nwt += ((1.0+2.55*cofi)/(1.9*cofi)
867 + 1.26*cofi*alpha/(1.0+3.5*cofi))
868 * (nwt - fAbscissa[i-3])/(1.0 + 0.3*alpha) ;
869 }
870 for(k=1;k<=maxNumber;k++)
871 {
872 temp1 = 1.0 ;
873 temp2 = 0.0 ;
874
875 for(j=1;j<=fNumber;j++)
876 {
877 temp3 = temp2 ;
878 temp2 = temp1 ;
879 temp1 = ((2*j - 1 + alpha - nwt)*temp2 - (j - 1 + alpha)*temp3)/j ;
880 }
881 temp = (fNumber*temp1 - (fNumber +alpha)*temp2)/nwt ;
882 nwt1 = nwt ;
883 nwt = nwt1 - temp1/temp ;
884
885 if(std::fabs(nwt - nwt1) <= tolerance)
886 {
887 break ;
888 }
889 }
890 if(k > maxNumber)
891 {
892 G4Exception("G4Integrator<T,F>::Laguerre(T,F, ...)", "Error",
893 FatalException, "Too many (>12) iterations.");
894 }
895
896 fAbscissa[i-1] = nwt ;
897 fWeight[i-1] = -std::exp(GammaLogarithm(alpha + fNumber) -
898 GammaLogarithm((G4double)fNumber))/(temp*fNumber*temp2) ;
899 }
900
901 //
902 // Integral evaluation
903 //
904
905 for(i=0;i<fNumber;i++)
906 {
907 integral += fWeight[i]*(typeT.*f)(fAbscissa[i]) ;
908 }
909 delete[] fAbscissa;
910 delete[] fWeight;
911 return integral ;
912}

1006///////////////////////////////////////////////////////////////////////
1007//
1008// Auxiliary function which returns the value of std::log(gamma-function(x))
1009// Returns the value ln(Gamma(xx) for xx > 0. Full accuracy is obtained for
1010// xx > 1. For 0 < xx < 1. the reflection formula (6.1.4) can be used first.
1011// (Adapted from Numerical Recipes in C)
1012//

1014template <class T, class F>
1015G4double G4Integrator<T,F>::GammaLogarithm(G4double xx)
1016{
1017 static const G4double cof[6] = { 76.18009172947146, -86.50532032941677,
1018 24.01409824083091, -1.231739572450155,
1019 0.1208650973866179e-2, -0.5395239384953e-5 };
1020 G4int j;
1021 G4double x = xx - 1.0 ;
1022 G4double tmp = x + 5.5 ;
1023 tmp -= (x + 0.5) * std::log(tmp) ;
1024 G4double ser = 1.000000000190015 ;
1025
1026 for ( j = 0; j <= 5; j++ )
1027 {
1028 x += 1.0 ;
1029 ser += cof[j]/x ;
1030 }
1031 return -tmp + std::log(2.5066282746310005*ser) ;
1032}

1034///////////////////////////////////////////////////////////////////////
1035//
1036// Method involving Hermite polynomials
1037//
1038///////////////////////////////////////////////////////////////////////
1039//
1040//
1041// Gauss-Hermite method for integration of std::exp(-x*x)*f(x)
1042// from minus infinity to plus infinity .
1043//
1044
1045template <class T, class F>
1046G4double G4Integrator<T,F>::Hermite( T& typeT, F f, G4int nHermite )
1047{
1048 const G4double tolerance = 1.0e-12 ;
1049 const G4int maxNumber = 12 ;
1050
1051 G4int i, j, k ;
1052 G4double integral = 0.0 ;
1053 G4double nwt=0., nwt1, temp1, temp2, temp3, temp ;
1054
1055 G4double piInMinusQ = std::pow(CLHEP::pi,-0.25) ; // 1.0/std::sqrt(std::sqrt(pi)) ??
1056
1057 G4int fNumber = (nHermite +1)/2 ;
1058 G4double* fAbscissa = new G4double[fNumber] ;
1059 G4double* fWeight = new G4double[fNumber] ;
1060
1061 for(i=1;i<=fNumber;i++)
1062 {
1063 if(i == 1)
1064 {
1065 nwt = std::sqrt((G4double)(2*nHermite + 1)) -
1066 1.85575001*std::pow((G4double)(2*nHermite + 1),-0.16666999) ;
1067 }
1068 else if(i == 2)
1069 {
1070 nwt -= 1.14001*std::pow((G4double)nHermite,0.425999)/nwt ;
1071 }
1072 else if(i == 3)
1073 {
1074 nwt = 1.86002*nwt - 0.86002*fAbscissa[0] ;
1075 }
1076 else if(i == 4)
1077 {
1078 nwt = 1.91001*nwt - 0.91001*fAbscissa[1] ;
1079 }
1080 else
1081 {
1082 nwt = 2.0*nwt - fAbscissa[i - 3] ;
1083 }
1084 for(k=1;k<=maxNumber;k++)
1085 {
1086 temp1 = piInMinusQ ;
1087 temp2 = 0.0 ;
1088
1089 for(j=1;j<=nHermite;j++)
1090 {
1091 temp3 = temp2 ;
1092 temp2 = temp1 ;
1093 temp1 = nwt*std::sqrt(2.0/j)*temp2 -
1094 std::sqrt(((G4double)(j - 1))/j)*temp3 ;
1095 }
1096 temp = std::sqrt((G4double)2*nHermite)*temp2 ;
1097 nwt1 = nwt ;
1098 nwt = nwt1 - temp1/temp ;
1099
1100 if(std::fabs(nwt - nwt1) <= tolerance)
1101 {
1102 break ;
1103 }
1104 }
1105 if(k > maxNumber)
1106 {
1107 G4Exception("G4Integrator<T,F>::Hermite(T,F, ...)", "Error",
1108 FatalException, "Too many (>12) iterations.");
1109 }
1110 fAbscissa[i-1] = nwt ;
1111 fWeight[i-1] = 2.0/(temp*temp) ;
1112 }
1113
1114 //
1115 // Integral calculation
1116 //
1117
1118 for(i=0;i<fNumber;i++)
1119 {
1120 integral += fWeight[i]*( (typeT.*f)(fAbscissa[i]) +
1121 (typeT.*f)(-fAbscissa[i]) ) ;
1122 }
1123 delete[] fAbscissa;
1124 delete[] fWeight;
1125 return integral ;
1126}

1228//

1232// from minus unit to plus unit .
1233//

1235template <class T, class F>
1236G4double G4Integrator<T,F>::Jacobi( T& typeT, F f, G4double alpha,
1237 G4double beta, G4int nJacobi)
1238{

1240 const G4double maxNumber = 12 ;
1241 G4int i, k, j ;
1242 G4double alphaBeta, alphaReduced, betaReduced, root1=0., root2=0., root3=0. ;
1243 G4double a, b, c, nwt1, nwt2, nwt3, nwt, temp, root=0., rootTemp ;
1244
1245 G4int fNumber = nJacobi ;
1246 G4double* fAbscissa = new G4double[fNumber] ;
1247 G4double* fWeight = new G4double[fNumber] ;
1248
1249 for (i=1;i<=nJacobi;i++)
1250 {
1251 if (i == 1)
1252 {
1253 alphaReduced = alpha/nJacobi ;
1254 betaReduced = beta/nJacobi ;
1255 root1 = (1.0+alpha)*(2.78002/(4.0+nJacobi*nJacobi)+
1256 0.767999*alphaReduced/nJacobi) ;
1257 root2 = 1.0+1.48*alphaReduced+0.96002*betaReduced +
1258 0.451998*alphaReduced*alphaReduced +
1259 0.83001*alphaReduced*betaReduced ;
1260 root = 1.0-root1/root2 ;
1261 }
1262 else if (i == 2)
1263 {
1264 root1=(4.1002+alpha)/((1.0+alpha)*(1.0+0.155998*alpha)) ;
1265 root2=1.0+0.06*(nJacobi-8.0)*(1.0+0.12*alpha)/nJacobi ;
1266 root3=1.0+0.012002*beta*(1.0+0.24997*std::fabs(alpha))/nJacobi ;
1267 root -= (1.0-root)*root1*root2*root3 ;
1268 }
1269 else if (i == 3)
1270 {
1271 root1=(1.67001+0.27998*alpha)/(1.0+0.37002*alpha) ;
1272 root2=1.0+0.22*(nJacobi-8.0)/nJacobi ;
1273 root3=1.0+8.0*beta/((6.28001+beta)*nJacobi*nJacobi) ;
1274 root -= (fAbscissa[0]-root)*root1*root2*root3 ;
1275 }
1276 else if (i == nJacobi-1)
1277 {
1278 root1=(1.0+0.235002*beta)/(0.766001+0.118998*beta) ;
1279 root2=1.0/(1.0+0.639002*(nJacobi-4.0)/(1.0+0.71001*(nJacobi-4.0))) ;
1280 root3=1.0/(1.0+20.0*alpha/((7.5+alpha)*nJacobi*nJacobi)) ;
1281 root += (root-fAbscissa[nJacobi-4])*root1*root2*root3 ;
1282 }
1283 else if (i == nJacobi)
1284 {
1285 root1 = (1.0+0.37002*beta)/(1.67001+0.27998*beta) ;
1286 root2 = 1.0/(1.0+0.22*(nJacobi-8.0)/nJacobi) ;
1287 root3 = 1.0/(1.0+8.0*alpha/((6.28002+alpha)*nJacobi*nJacobi)) ;
1288 root += (root-fAbscissa[nJacobi-3])*root1*root2*root3 ;
1289 }
1290 else
1291 {
1292 root = 3.0*fAbscissa[i-2]-3.0*fAbscissa[i-3]+fAbscissa[i-4] ;
1293 }
1294 alphaBeta = alpha + beta ;
1295 for (k=1;k<=maxNumber;k++)
1296 {
1297 temp = 2.0 + alphaBeta ;
1298 nwt1 = (alpha-beta+temp*root)/2.0 ;
1299 nwt2 = 1.0 ;
1300 for (j=2;j<=nJacobi;j++)
1301 {
1302 nwt3 = nwt2 ;
1303 nwt2 = nwt1 ;
1304 temp = 2*j+alphaBeta ;
1305 a = 2*j*(j+alphaBeta)*(temp-2.0) ;
1306 b = (temp-1.0)*(alpha*alpha-beta*beta+temp*(temp-2.0)*root) ;
1307 c = 2.0*(j-1+alpha)*(j-1+beta)*temp ;
1308 nwt1 = (b*nwt2-c*nwt3)/a ;
1309 }
1310 nwt = (nJacobi*(alpha - beta - temp*root)*nwt1 +
1311 2.0*(nJacobi + alpha)*(nJacobi + beta)*nwt2)/
1312 (temp*(1.0 - root*root)) ;
1313 rootTemp = root ;
1314 root = rootTemp - nwt1/nwt ;
1315 if (std::fabs(root-rootTemp) <= tolerance)
1316 {
1317 break ;
1318 }
1319 }
1320 if (k > maxNumber)
1321 {
1322 G4Exception("G4Integrator<T,F>::Jacobi(T,F, ...)", "Error",
1323 FatalException, "Too many (>12) iterations.");
1324 }
1325 fAbscissa[i-1] = root ;
1326 fWeight[i-1] = std::exp(GammaLogarithm((G4double)(alpha+nJacobi)) +
1327 GammaLogarithm((G4double)(beta+nJacobi)) -
1328 GammaLogarithm((G4double)(nJacobi+1.0)) -
1329 GammaLogarithm((G4double)(nJacobi + alphaBeta + 1.0)))
1330 *temp*std::pow(2.0,alphaBeta)/(nwt*nwt2) ;
1331 }
1332
1333 //
1334 // Calculation of the integral
1335 //
1336
1337 G4double integral = 0.0 ;
1338 for(i=0;i<fNumber;i++)
1339 {
1340 integral += fWeight[i]*(typeT.*f)(fAbscissa[i]) ;
1341 }
1342 delete[] fAbscissa;
1343 delete[] fWeight;
1344 return integral ;
1345}
