35template<class T> void G4SimplexDownhill<T>::init()
36{
37 alpha = 2.0; // refrection coefficient: 0 < alpha
38 beta = 0.5; // contraction coefficient: 0 < beta < 1
39 gamma = 2.0; // expantion coefficient: 1 < gamma
40
41 maximum_no_trial = 10000;
42 max_se = FLT_MIN;
43 //max_ratio = FLT_EPSILON/1;
44 max_ratio = DBL_EPSILON/1;
45 minimized = false;
46}

62template<class T>
63G4double G4SimplexDownhill<T>::GetMinimum()
64{
65
66 initialize();
67
68// First Tryal;
69
70 //G4cout << "Begin First Trials" << G4endl;
71 doDownhill();
72 //G4cout << "End First Trials" << G4endl;
73
74 std::vector< G4double >::iterator it_minh =
75 std::min_element( currentHeights.begin() , currentHeights.end() );
76 G4int imin = -1;
77 G4int i = 0;
78 for ( std::vector< G4double >::iterator it = currentHeights.begin();
79 it != currentHeights.end(); it++ )
80 {
81 if ( it == it_minh )
82 {
83 imin = i;
84 }
85 i++;
86 }
87 minimumPoint = currentSimplex[ imin ];
88
89// Second Trial
90
91 //std::vector< G4double > minimumPoint = currentSimplex[ 0 ];
92 initialize();
93
94 currentSimplex[ numberOfVariable ] = minimumPoint;
95
96 //G4cout << "Begin Second Trials" << G4endl;
97 doDownhill();
98 //G4cout << "End Second Trials" << G4endl;
99
100 G4double sum = std::accumulate( currentHeights.begin() ,
101 currentHeights.end() , 0.0 );
102 G4double average = sum/(numberOfVariable+1);
103 G4double minimum = average;
104
105 minimized = true;
106
107 return minimum;
108
109}

115{

118 currentHeights.resize( numberOfVariable+1 );
119

126

129 currentSimplex[ numberOfVariable ] = avec;

148template<class T>
149std::vector< G4double > G4SimplexDownhill<T>::calCentroid( G4int ih )
150{
151
152 std::vector< G4double > centroid ( numberOfVariable , 0.0 );
153
154 G4int i = 0;
155 for ( std::vector< std::vector< G4double > >::iterator
156 it = currentSimplex.begin(); it != currentSimplex.end() ; it++ )
157 {
158 if ( i != ih )
159 {
160 for ( G4int j = 0 ; j < numberOfVariable ; j++ )
161 {
162 centroid[j] += (*it)[j]/numberOfVariable;
163 }
164 }
165 i++;
166 }
167
168 return centroid;
169}

173template<class T>
174std::vector< G4double > G4SimplexDownhill<T>::
175getReflectionPoint( std::vector< G4double > p ,
176 std::vector< G4double > centroid )
177{
178 //G4cout << "Reflection" << G4endl;
179
180 std::vector< G4double > reflectionP ( numberOfVariable , 0.0 );
181
182 for ( G4int i = 0 ; i < numberOfVariable ; i++ )
183 {
184 reflectionP[ i ] = ( 1 + alpha ) * centroid[ i ] - alpha * p[ i ];
185 }
186
187 return reflectionP;
188}

192template<class T>
193std::vector< G4double > G4SimplexDownhill<T>::
194getExpansionPoint( std::vector< G4double > p ,
195 std::vector< G4double > centroid )
196{
197 //G4cout << "Expantion" << G4endl;
198
199 std::vector< G4double > expansionP ( numberOfVariable , 0.0 );
200
201 for ( G4int i = 0 ; i < numberOfVariable ; i++ )
202 {
203 expansionP[i] = ( 1 - gamma ) * centroid[i] + gamma * p[i];
204 }
205
206 return expansionP;
207}

209template<class T>
210std::vector< G4double > G4SimplexDownhill<T>::
211getContractionPoint( std::vector< G4double > p ,
212 std::vector< G4double > centroid )
213{
214 //G4cout << "Contraction" << G4endl;
215
216 std::vector< G4double > contractionP ( numberOfVariable , 0.0 );
217
218 for ( G4int i = 0 ; i < numberOfVariable ; i++ )
219 {
220 contractionP[i] = ( 1 - beta ) * centroid[i] + beta * p[i];
221 }
222
223 return contractionP;
224}

228template<class T>
229G4bool G4SimplexDownhill<T>::isItGoodEnough()
230{
231 G4bool result = false;
232
233 G4double sum = std::accumulate( currentHeights.begin() ,
234 currentHeights.end() , 0.0 );
235 G4double average = sum/(numberOfVariable+1);
236 //G4cout << "average " << average << G4endl;
237
238 G4double delta = 0.0;
239 for ( G4int i = 0 ; i <= numberOfVariable ; i++ )
240 {
241 delta += std::abs ( currentHeights[ i ] - average );
242 }
243 //G4cout << "ratio of delta to average is "
244 // << delta / (numberOfVariable+1) / average << G4endl;
245
246 if ( delta/(numberOfVariable+1)/average < max_ratio )
247 {
248 result = true;
249 }
250
251/*
252 G4double sigma = 0.0;
253 G4cout << "average " << average << G4endl;
254 for ( G4int i = 0 ; i <= numberOfVariable ; i++ )
255 {
256 sigma += ( currentHeights[ i ] - average )
257 *( currentHeights[ i ] - average );
258 }
259
260 G4cout << "standard error of hs "
261 << std::sqrt ( sigma ) / (numberOfVariable+1) << G4endl;
262 if ( std::sqrt ( sigma ) / (numberOfVariable+1) < max_se )
263 {
264 result = true;
265 }
266*/
267
268 return result;
269}

273template<class T>
274void G4SimplexDownhill<T>::doDownhill()
275{
276
277 G4int nth_trial = 0;
278
279 while ( nth_trial < maximum_no_trial )
280 {
281
282/*
283 G4cout << "Begining " << nth_trial << "th trial " << G4endl;
284 for ( G4int j = 0 ; j <= numberOfVariable ; j++ )
285 {
286 G4cout << "SimplexPoint " << j << ": ";
287 for ( G4int i = 0 ; i < numberOfVariable ; i++ )
288 {
289 G4cout << currentSimplex[j][i]
290 << " ";
291 }
292 G4cout << G4endl;
293 }
294*/
295
296 calHeights();
297
298 if ( isItGoodEnough() )
299 {
300 break;
301 }
302
303 std::vector< G4double >::iterator it_maxh =
304 std::max_element( currentHeights.begin() , currentHeights.end() );
305 std::vector< G4double >::iterator it_minh =
306 std::min_element( currentHeights.begin() , currentHeights.end() );;
307
308 G4double h_H = *it_maxh;
309 G4double h_L = *it_minh;
310
311 G4int ih = 0;;
312 G4int il = 0;
313 G4double h_H2 =0.0;
314 G4int i = 0;
315 for ( std::vector< G4double >::iterator
316 it = currentHeights.begin(); it != currentHeights.end(); it++ )
317 {
318 if ( it == it_maxh )
319 {
320 ih = i;
321 }
322 else
323 {
324 h_H2 = std::max( h_H2 , *it );
325 }
326
327 if ( it == it_minh )
328 {
329 il = i;
330 }
331 i++;
332 }
333
334 //G4cout << "max " << h_H << " " << ih << G4endl;
335 //G4cout << "max-dash " << h_H2 << G4endl;
336 //G4cout << "min " << h_L << " " << il << G4endl;
337
338 std::vector< G4double > centroidPoint = calCentroid ( ih );
339
340 // REFLECTION
341 std::vector< G4double > reflectionPoint =
342 getReflectionPoint( currentSimplex[ ih ] , centroidPoint );
343
344 G4double h = getValue( reflectionPoint );
345
346 if ( h <= h_L )
347 {
348 // EXPANSION
349 std::vector< G4double > expansionPoint =
350 getExpansionPoint( reflectionPoint , centroidPoint );
351 G4double hh = getValue( expansionPoint );
352
353 if ( hh <= h_L )
354 {
355 // Replace
356 currentSimplex[ ih ] = expansionPoint;
357 //G4cout << "A" << G4endl;
358 }
359 else
360 {
361 // Replace
362 currentSimplex[ ih ] = reflectionPoint;
363 //G4cout << "B1" << G4endl;
364 }
365 }
366 else
367 {
368 if ( h <= h_H2 )
369 {
370 // Replace
371 currentSimplex[ ih ] = reflectionPoint;
372 //G4cout << "B2" << G4endl;
373 }
374 else
375 {
376 if ( h <= h_H )
377 {
378 // Replace
379 currentSimplex[ ih ] = reflectionPoint;
380 //G4cout << "BC" << G4endl;
381 }
382 // CONTRACTION
383 std::vector< G4double > contractionPoint =
384 getContractionPoint( currentSimplex[ ih ] , centroidPoint );
385 G4double hh = getValue( contractionPoint );
386 if ( hh <= h_H )
387 {
388 // Replace
389 currentSimplex[ ih ] = contractionPoint;
390 //G4cout << "C" << G4endl;
391 }
392 else
393 {
394 // Replace
395 for ( G4int j = 0 ; j <= numberOfVariable ; j++ )
396 {
397 std::vector< G4double > vec ( numberOfVariable , 0.0 );
398 for ( G4int k = 0 ; k < numberOfVariable ; k++ )
399 {
400 vec[ k ] = ( currentSimplex[ j ][ k ]
401 + currentSimplex[ il ][ k ] ) / 2.0;
402 }
403 currentSimplex[ j ] = vec;
404 }
405 //G4cout << "D" << G4endl;
406 }
407 }
408
409 }
410
411 nth_trial++;
412 }
413}

417template<class T>
418std::vector< G4double > G4SimplexDownhill<T>::GetMinimumPoint()
419{
420 if ( minimized != true )
421 {
422 GetMinimum();
423 }
424
425 std::vector< G4double >::iterator it_minh =
426 std::min_element( currentHeights.begin() , currentHeights.end() );;
427 G4int imin = -1;
428 G4int i = 0;
429 for ( std::vector< G4double >::iterator
430 it = currentHeights.begin(); it != currentHeights.end(); it++ )
431 {
432 if ( it == it_minh )
433 {
434 imin = i;
435 }
436 i++;
437 }
438 minimumPoint = currentSimplex[ imin ];
439
440 return minimumPoint;
