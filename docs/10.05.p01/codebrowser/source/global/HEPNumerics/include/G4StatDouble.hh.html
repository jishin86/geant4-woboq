30// Class G4StatDouble
31//
32// Class description:
33//
34// Class providing simple "one variable statistics" capability.
35
36// Original Author: Giovanni Santin (ESA) - October 2005 in GRAS tool
37// Adaptation and comments by: John Apostolakis (CERN) - November 2011
38
39#ifndef G4StatDouble_h
40#define G4StatDouble_h 1
41
42#include "globals.hh"
43
44class G4StatDouble
45{
46
47 public:
48 G4StatDouble();
49 G4StatDouble(G4double);
50 virtual ~G4StatDouble();
51
52 public:
53 G4StatDouble& operator=(const G4double &rhs)
54 {
55 reset();
56 fill(rhs);
57 return *this;
58 }
59 G4StatDouble& operator=(const G4StatDouble &rhs)
60 {
61 m_sum_wx = rhs.m_sum_wx;
62 m_sum_wx2 = rhs.m_sum_wx2;
63 m_n = rhs.m_n;
64 m_sum_w = rhs.m_sum_w;
65 m_sum_w2 = rhs.m_sum_w2;
66 m_scale = rhs.m_scale;
67 return *this;
68 }
69 G4StatDouble& operator+=(const G4double &rhs)
70 { fill(rhs); return *this; }
71 G4StatDouble& operator+=(const G4StatDouble &rhs)
72 { add(&rhs); return *this; }
73
74 public:
75
76 void reset();
77 void fill(G4double x, G4double weight=1.);
78 // Add new data point: value "x" with weight
79 void scale(G4double);
80 // Reset scale
81
82 G4double mean() const;
83 G4double rms();
84 // The moments
85
86 G4double mean(G4double ext_sum_w) const;
87 // Mean scaled to sum of weights
88 G4double rms(G4double ext_sum_w, G4int ext_n);
89 // RMS scaled to sum of weights
90
91 void add(const G4StatDouble*);
92 // merge 2 statistics
93
94 inline G4int n() const { return m_n; }
95 inline G4double sum_w() const { return m_sum_w; }
96 inline G4double sum_w2() const { return m_sum_w2; }
97 inline G4double sum_wx() const { return m_sum_wx; }
98 inline G4double sum_wx2() const { return m_sum_wx2; }
99
100 protected:
101
102 G4double rms(G4double sum_wx, G4double sum_wx2, G4double sum_w, G4int n);
103
104 protected:
105
106 G4double m_sum_wx;
107 G4double m_sum_wx2;
108 G4int m_n;
109 G4double m_sum_w;
110 G4double m_sum_w2;
111 G4double m_scale;
112};
113
114#endif
