59G4int G4coutDestination::ReceiveG4cout_(const G4String& msg)
60{
61 // Avoid copy of string if not necessary
62 if( transformersCout.size() > 0 )
63 {
64 G4String m = msg;
65 G4bool result = true;
66 for ( const auto& el : transformersCout )
67 {
68 result &= el(m);
69 if ( ! result ) break;
70 }
71 return ( result ? ReceiveG4cout(m) : 0 );
72 }
73 else
74 {
75 return ReceiveG4cout(msg);
76 }
77}
78
79G4int G4coutDestination::ReceiveG4cerr_(const G4String& msg)
80{
81 if( transformersCout.size() > 0 )
82 {
83 G4String m = msg;
84 std::for_each( transformersCerr.begin() , transformersCerr.end() ,
85 [&m](const Transformer& t) { t(m); }
86 // Call transforming function on message
87 );
88 return ReceiveG4cerr(m);
89 }
90 else
91 {
92 return ReceiveG4cerr(msg);
93 }
94}
