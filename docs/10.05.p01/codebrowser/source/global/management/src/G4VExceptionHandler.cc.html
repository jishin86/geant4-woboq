39G4VExceptionHandler::G4VExceptionHandler()

43}

45G4VExceptionHandler::~G4VExceptionHandler()
46{
47}

49G4VExceptionHandler::G4VExceptionHandler(const G4VExceptionHandler &right)
50{
51 *this = right;
52}

54G4VExceptionHandler& G4VExceptionHandler::operator=(const G4VExceptionHandler &right)
55{
56 if (&right == this) { return *this; }
57 *this = right;
58 return *this;
59}

62{
63 return (this == &right);
64}

66G4bool G4VExceptionHandler::operator!=(const G4VExceptionHandler &right) const
67{
68 return (this != &right);
69}
