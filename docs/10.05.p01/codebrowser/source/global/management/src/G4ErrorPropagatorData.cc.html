37G4ThreadLocal G4ErrorPropagatorData* G4ErrorPropagatorData::fpInstance = 0;
38G4ThreadLocal G4int G4ErrorPropagatorData::theVerbosity = 0;

40//-------------------------------------------------------------------
41
42G4ErrorPropagatorData::G4ErrorPropagatorData()
43 : theMode(G4ErrorMode_PropTest), theState(G4ErrorState_PreInit),
44 theStage(G4ErrorStage_Inflation), theTarget(0)
45{
46}
47
48G4ErrorPropagatorData::~G4ErrorPropagatorData()
49{
50 delete fpInstance; fpInstance = 0;
51}
52
53G4ErrorPropagatorData* G4ErrorPropagatorData::GetErrorPropagatorData()
54{
55 if (fpInstance == 0)
56 {
57 fpInstance = new G4ErrorPropagatorData;
58 }
59 return fpInstance;
60}
61
62G4int G4ErrorPropagatorData::verbose()
63{
64 return theVerbosity;
65}
66
67void G4ErrorPropagatorData::SetVerbose( G4int ver )
68{
69 theVerbosity = ver;
70}
