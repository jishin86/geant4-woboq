37inline
38 void G4PhysicsTable::clearAndDestroy()
39{
40 G4PhysicsVector* a=nullptr;
41 while (size()>0)
42 {
43 a = G4PhysCollection::back();
44 G4PhysCollection::pop_back();
45 if ( a ) { delete a; }
46 }
47 G4PhysCollection::clear();
48 vecFlag.clear();
49}
50
51inline
52 G4PhysicsVector*& G4PhysicsTable::operator()(size_t i)
53{
54 return (*this)[i];
55}
56
57inline
58 G4PhysicsVector* const& G4PhysicsTable::operator()(size_t i) const
59{
60 return (*this)[i];
61}
62
63inline
64 void G4PhysicsTable::push_back(G4PhysicsVector* pvec)
65{
66 G4PhysCollection::push_back(pvec);
67 vecFlag.push_back(true);
68}
69
70inline
71 void G4PhysicsTable::insert(G4PhysicsVector* pvec)
72{
73 G4PhysCollection::push_back(pvec);
74 vecFlag.push_back(true);
75}
76
77inline
78 void G4PhysicsTable::insertAt (size_t idx, G4PhysicsVector* pvec)
79{
80 G4PhysicsTableIterator itr=begin();
81 for (size_t i=0; i<idx; ++i) { itr++; }
82 G4PhysCollection::insert(itr, pvec);
83
84 G4FlagCollection::iterator itrF=vecFlag.begin();
85 for (size_t j=0; j<idx; ++j) { itrF++; }
86 vecFlag.insert(itrF, true);
87}
