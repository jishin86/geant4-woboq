32///////////////////////////////////////
33inline G4strstreambuf::G4strstreambuf()
34 : std::basic_streambuf<char>(),
35 count(0), destination(0)
36///////////////////////////////////////
37{
38 size= 4095;
39 buffer= new char[size+1];
40}
41
42
43////////////////////////////////////////
44inline G4strstreambuf::~G4strstreambuf()
45////////////////////////////////////////
46{
47 // flushing buffer...
48 // std::cout is used because destination object may not be alive.
49 if(count !=0) std::cout << buffer;
50
51 delete [] buffer;
52}

60 destination(right.destination)

63}

66/////////////////////////////////////////////////////////////////////////////
67inline G4strstreambuf& G4strstreambuf::operator=(const G4strstreambuf& right)
68/////////////////////////////////////////////////////////////////////////////
69{
70 if(&right==this) return *this;
71
72 destination= right.destination;
73 buffer= right.buffer;
74 count= right.count;
75 size= right.size;
76
77 return *this;
78}

81//////////////////////////////////////////////
82inline G4int G4strstreambuf::overflow(G4int c)
83//////////////////////////////////////////////
84{
85 G4int result= 0;
86 if(count>=size) result= sync();
87
88 buffer[count]= c;
89 count++;
90
91 return result;
92}
93
94///////////////////////////////////
95inline G4int G4strstreambuf::sync()
96///////////////////////////////////
97{
98 buffer[count] = '\0';
99 count= 0;
100 return ReceiveString();
101}

129/////////////////////////////////////////////
130inline G4int G4strstreambuf::ReceiveString ()
131/////////////////////////////////////////////
132{
133 G4String stringToSend(buffer);
134 G4int result= 0;
135
136 if(this == &G4coutbuf && destination != 0)
137 {
138 result= destination-> ReceiveG4cout_(stringToSend);
139 }
140 else if(this == &G4cerrbuf && destination != 0)
141 {
142 result= destination-> ReceiveG4cerr_(stringToSend);
143 }
144 else if(this == &G4coutbuf && destination == 0)
145 {
146 std::cout << stringToSend << std::flush;
147 result= 0;
148 }
149 else if(this == &G4cerrbuf && destination == 0)
150 {
151 std::cerr << stringToSend << std::flush;
152 result= 0;
153 }
154
155 return result;
156}
