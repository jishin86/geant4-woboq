32// G4PhysicsVector.hh
33//
34// Class description:
35//
36// A physics vector which has values of energy-loss, cross-section,
37// and other physics values of a particle in matter in a given
38// range of the energy, momentum, etc.
39// This class serves as the base class for a vector having various
40// energy scale, for example like 'log', 'linear', 'free', etc.
41
42// History:
43// 02 Dec. 1995, G.Cosmo : Structure created based on object model
44// 03 Mar. 1996, K.Amako : Implemented the 1st version
45// 27 Apr. 1996, K.Amako : Cache mechanism added
46// 01 Jul. 1996, K.Amako : Now GetValue not virtual
47// 21 Sep. 1996, K.Amako : Added [] and () operators
48// 11 Nov. 2000, H.Kurashige : Use STL vector for dataVector and binVector
49// 09 Mar. 2001, H.Kurashige : Added G4PhysicsVectorType & Store/Retrieve()
50// 02 Apr. 2008, A.Bagulya : Added SplineInterpolation() and SetSpline()
51// 11 May 2009, V.Ivanchenko : Added ComputeSecondDerivatives
52// 19 Jun. 2009, V.Ivanchenko : Removed hidden bin
53// 22 Dec. 2009 H.Kurashige : Use pointers to G4PVDataVector
54// 04 May. 2010 H.Kurashige : Use G4PhysicsVectorCache
55// 28 May 2010 H.Kurashige : Stop using pointers to G4PVDataVector
56// 16 Aug. 2011 H.Kurashige : Add dBin, baseBin and verboseLevel
57// 02 Oct. 2013 V.Ivanchenko : FindBinLocation method become inlined;
58// instead of G4Pow G4Log is used
59//---------------------------------------------------------------

75class G4PhysicsVector
76{
77 public:// with description
78
79 explicit G4PhysicsVector(G4bool spline = false);
80 // default constructor - vector will be filled via Retrieve() method
81
82 G4PhysicsVector(const G4PhysicsVector&);
83 G4PhysicsVector& operator=(const G4PhysicsVector&);
84 // Copy constructor and assignment operator.
85
86 virtual ~G4PhysicsVector();
87
88 G4double Value(G4double theEnergy, size_t& lastidx) const;
89 // Get the cross-section/energy-loss value corresponding to the
90 // given energy. An appropriate interpolation is used to calculate
91 // the value. Consumer code got changed index and may reuse it
92 // for the next call to save CPU for bin location.
93
94 inline G4double Value(G4double theEnergy) const;
95 // Get the cross-section/energy-loss value corresponding to the
96 // given energy. An appropriate interpolation is used to calculate
97 // the value. This method is kept for backward compatibility reason,
98 // it should be used instead of the previous method if bin location
99 // cannot be kept thread safe
100
101 inline G4double GetValue(G4double theEnergy, G4bool& isOutRange) const;
102 // Obsolete method to get value, isOutRange is not used anymore.
103 // This method is kept for the compatibility reason.
104
105 G4bool operator==(const G4PhysicsVector &right) const ;
106 G4bool operator!=(const G4PhysicsVector &right) const ;
107
108 inline G4double operator[](const size_t index) const ;
109 // Returns the value for the specified index of the dataVector
110 // The boundary check will not be done.
111
112 inline G4double operator()(const size_t index) const ;
113 // Returns the value for the specified index of the dataVector
114 // The boundary check will not be done.
115
116 inline void PutValue(size_t index, G4double theValue);
117 // Put 'theValue' into the dataVector specified by 'index'.
118 // Take note that the 'index' starts from '0'.
119 // To fill the vector, you have beforehand to construct a vector
120 // by the constructor with Emin, Emax, Nbin. 'theValue' should
121 // be the crosssection/energyloss value corresponding to the
122 // energy of the index.
123
124 virtual void ScaleVector(G4double factorE, G4double factorV);
125 // Scale all values of the vector and second derivatives
126 // by factorV, energies by vectorE. This method may be applied
127 // for example after Retrieve a vector from an external file to
128 // convert values into Geant4 units
129
130 inline G4double Energy(size_t index) const;
131 // Returns the value in the energy specified by 'index'
132 // of the energy vector. The boundary check will not be done.
133 // Use this function when compute cross section or dEdx
134 // before filling the vector by PutValue(..).
135
136 inline G4double GetMaxEnergy() const;
137 // Returns the energy of the last point of the vector
138
139 G4double GetLowEdgeEnergy(size_t binNumber) const;
140 // Obsolete method
141 // Get the energy value at the low edge of the specified bin.
142 // Take note that the 'binNumber' starts from '0'.
143 // The boundary check will not be done.
144
145 inline size_t GetVectorLength() const;
146 // Get the total length of the vector.
147
148 inline size_t FindBin(G4double energy, size_t idx) const;
149 // find low edge index of a bin for given energy
150 // min value 0, max value VectorLength-1
151 // idx is suggested bin number from user code
152
153 void FillSecondDerivatives();
154 // Initialise second derivatives for spline keeping
155 // 3d derivative continues - default algorithm
156 // Warning: this method should be called when the vector
157 // is already filled
158
159 void ComputeSecDerivatives();
160 // Initialise second derivatives for spline using algorithm
161 // which garantee only 1st derivative continues
162 // Warning: this method should be called when the vector
163 // is already filled
164
165 void ComputeSecondDerivatives(G4double firstPointDerivative,
166 G4double endPointDerivative);
167 // Initialise second derivatives for spline using
168 // user defined 1st derivatives at edge points
169 // Warning: this method should be called when the vector
170 // is already filled
171
172 G4double FindLinearEnergy(G4double rand) const;
173 // Find energy using linear interpolation for vector
174 // filled by cumulative probability function
175 // value of rand should be between 0 and 1
176
177 inline G4bool IsFilledVectorExist() const;
178 // Is non-empty physics vector already exist?
179
180 inline G4PhysicsVectorType GetType() const;
181 // Get physics vector type
182
183 inline void SetSpline(G4bool);
184 // Activate/deactivate Spline interpolation.
185
186 G4bool Store(std::ofstream& fOut, G4bool ascii=false) const;
187 virtual G4bool Retrieve(std::ifstream& fIn, G4bool ascii=false);
188 // To store/retrieve persistent data to/from file streams.
189
190 friend std::ostream& operator<<(std::ostream&, const G4PhysicsVector&);
191 void DumpValues(G4double unitE=1.0, G4double unitV=1.0) const;
192 // print vector
193
194 inline void SetVerboseLevel(G4int value);
195
196 protected:
197
198 void DeleteData();
199 void CopyData(const G4PhysicsVector& vec);
200 // Internal methods for allowing copy of objects
201
202 void PrintPutValueError(size_t index);
203
204 protected:
205
206 G4PhysicsVectorType type; // The type of PhysicsVector (enumerator)
207
208 G4double edgeMin; // Energy of first point
209 G4double edgeMax; // Energy of the last point
210
211 size_t numberOfNodes;
212
213 G4PVDataVector dataVector; // Vector to keep the crossection/energyloss
214 G4PVDataVector binVector; // Vector to keep energy
215 G4PVDataVector secDerivative; // Vector to keep second derivatives
216
217 private:
218
219 G4bool SplinePossible();
220
221 inline G4double LinearInterpolation(size_t idx, G4double energy) const;
222 // Linear interpolation function
223 inline G4double SplineInterpolation(size_t idx, G4double energy) const;
224 // Spline interpolation function
225
226 inline G4double Interpolation(size_t idx, G4double energy) const;
227
228 inline size_t FindBinLocation(G4double theEnergy) const;
229 // find low edge index of a bin for given energy
230 // min value 0, max value VectorLength-1
231
232 G4bool useSpline;
233
234 protected:
235
236 G4double dBin; // Bin width - useful only for fixed binning
237 G4double baseBin; // Set this in constructor for performance
238
239 G4int verboseLevel;
240};
