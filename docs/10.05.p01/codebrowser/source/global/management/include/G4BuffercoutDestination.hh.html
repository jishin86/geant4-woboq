32// Class Description:
33//
34// A cout destination that buffers received stream in a buffer and
35// dumps the buffer only when full.
36// Optionally allows for an infinite buffer.
37
38// ---------------- G4BuffercoutDestination ----------------
39//
40// Author: A.Dotti (SLAC), April 2017
41// --------------------------------------------------------------------
42#ifndef G4BUFFERCOUTDESTINATION_HH_
43#define G4BUFFERCOUTDESTINATION_HH_
44
45#include <sstream>
46
47#include "G4coutDestination.hh"
48
49class G4BuffercoutDestination : public G4coutDestination
50{
51 public:
52
53 explicit G4BuffercoutDestination(size_t maxSize = 0);
54 virtual ~G4BuffercoutDestination();
55
56 virtual G4int ReceiveG4cout(const G4String& msg) override;
57 virtual G4int ReceiveG4cerr(const G4String& msg) override;
58 // Flush buffer to std output
59 virtual G4int FlushG4cout();
60 // Flush buffer to std error
61 virtual G4int FlushG4cerr();
62 // Flsuh both buffers
63
64 virtual void Finalize();
65
66 // Set maximum size of buffer, when buffer grows to specified size,
67 // it will trigger flush. Dimension in char
68 void SetMaxSize(size_t max) { m_maxSize = max; }
69 size_t GetMaxSize() const { return m_maxSize; }
70 size_t GetCurrentSizeOut() const { return m_currentSize_out; }
71 size_t GetCurrentSizeErr() const { return m_currentSize_err; }
72
73 protected:
74
75 void ResetCout();
76 void ResetCerr();
77
78 std::ostringstream m_buffer_out;
79 std::ostringstream m_buffer_err;
80 size_t m_currentSize_out;
81 size_t m_currentSize_err;
82 size_t m_maxSize;
83};
84
85#endif /* G4BUFFERCOUTDESTINATION_HH_ */
