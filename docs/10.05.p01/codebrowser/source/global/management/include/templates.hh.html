55//
56// If HIGH_PRECISION is defined to TRUE (ie. != 0) then the type "Float"
57// is typedefed to "double". If it is FALSE (ie. 0) it is typedefed
58// to "float".
59//
60#ifndef HIGH_PRECISION
61#define HIGH_PRECISION 1
62#endif
63
64#if HIGH_PRECISION
65typedef double Float;
66#else
67typedef float Float;
68#endif

119template <class T>
120inline void G4SwapPtr(T*& a, T*& b)
121{
122 T* tmp= a;
123 a = b;
124 b = tmp;
125}
126
127template <class T>
128inline void G4SwapObj(T* a, T* b)
129{
130 T tmp= *a;
131 *a = *b;
132 *b = tmp;
133}

150inline int G4lrint(double ad)
151{
152 return (ad>0) ? static_cast<int>(ad+.5) : static_cast<int>(ad-.5);
153}
154
155inline int G4lint(double ad)
156{
157 return (ad>0) ? static_cast<int>(ad) : static_cast<int>(ad-1.);
158}
159
160inline int G4rint(double ad)
161{
162 return (ad>0) ? static_cast<int>(ad+1) : static_cast<int>(ad);
163}

167// Use the following function to get rid of "unused parameter" warnings
168// Example:
169//
170// #ifdef SOME_CONDITION
171// void doSomething(int val)
172// {
173// something = val;
174// }
175// #else
176// void doSomething(int)
177// { }
178// #endif
179//
180// can be simplified to:
181//
182// void doSomething(int val)
183// {
184// #ifdef SOME_CONDITION
185// something = val;
186// #else
187// G4ConsumeParameters(val);
188// #endif
189// }
190//
191// or:
192//
193// void doSomething(int val)
194// {
195// #ifdef SOME_CONDITION
196// something = val;
197// #endif
198// // function call does nothing -- will be "optimized" out
199// G4ConsumeParameters(val);
200// }
201//
202template <typename _Tp, typename... _Args>
203inline void G4ConsumeParameters(_Tp, _Args...)
204{ }
