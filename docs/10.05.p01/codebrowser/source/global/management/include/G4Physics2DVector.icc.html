103inline size_t G4Physics2DVector::FindBin(G4double z,
104 const G4PV2DDataVector& v,
105 size_t idx,
106 size_t idxmax) const
107{
108 size_t id = idx;
109 if(z < v[1]) {
110 id = 0;
111 } else if(z >= v[idxmax-2]) {
112 id = idxmax - 2;
113 } else if(idx > idxmax-2 || z < v[idx] || z >= v[idx+1]) {
114 id = FindBinLocation(z, v);
115 }
116 return id;
117}

136inline G4double
137G4Physics2DVector::DerivativeX(size_t idx, size_t idy, G4double fac) const
138{
139 size_t i1 = idx;
140 if(i1 > 0) { --i1; }
141 size_t i2 = idx;
142 if(i2+1 < numberOfXNodes) { ++i2; }
143 return fac*(GetValue(i2, idy) - GetValue(i1, idy))/(GetX(i2) - GetX(i1));
144}
145
146inline G4double
147G4Physics2DVector::DerivativeY(size_t idx, size_t idy, G4double fac) const
148{
149 size_t i1 = idy;
150 if(i1 > 0) { --i1; }
151 size_t i2 = idy;
152 if(i2+1 < numberOfYNodes) { ++i2; }
153 return fac*(GetValue(idx, i2) - GetValue(idx, i1))/(GetY(i2) - GetY(i1));
154}
155
156inline G4double
157G4Physics2DVector::DerivativeXY(size_t idx, size_t idy, G4double fac) const
158{
159 size_t i1 = idx;
160 if(i1 > 0) { --i1; }
161 size_t i2 = idx;
162 if(i2+1 < numberOfXNodes) { ++i2; }
163 size_t j1 = idy;
164 if(j1 > 0) { --j1; }
165 size_t j2 = idy;
166 if(j2+1 < numberOfYNodes) { ++j2; }
167 return fac*(GetValue(i2, j2) - GetValue(i1, j2) - GetValue(i2, j1)
168 + GetValue(i1, j1))/((GetX(i2) - GetX(i1))*(GetY(j2) - GetY(j1)));
169}
