83inline
84 G4double G4PhysicsVector::LinearInterpolation(size_t idx, G4double e) const
85{
86 // Linear interpolation is used to get the value. Before this method
87 // is called it is ensured that the energy is inside the bin
88 // 0 < idx < numberOfNodes-1
89
90 return dataVector[idx] +
91 ( dataVector[idx + 1]-dataVector[idx] ) * (e - binVector[idx])
92 /( binVector[idx + 1]-binVector[idx] );
93}

96
97inline
98 G4double G4PhysicsVector::SplineInterpolation(size_t idx, G4double e) const
99{
100 // Spline interpolation is used to get the value. Before this method
101 // is called it is ensured that the energy is inside the bin
102 // 0 < idx < numberOfNodes-1
103
104 static const G4double onesixth = 1.0/6.0;
105
106 // check bin value
107 G4double x1 = binVector[idx];
108 G4double x2 = binVector[idx + 1];
109 G4double delta = x2 - x1;
110
111 G4double a = (x2 - e)/delta;
112 G4double b = (e - x1)/delta;
113
114 // Final evaluation of cubic spline polynomial for return
115 G4double y1 = dataVector[idx];
116 G4double y2 = dataVector[idx + 1];
117
118 G4double res = a*y1 + b*y2 +
119 ( (a*a*a - a)*secDerivative[idx] +
120 (b*b*b - b)*secDerivative[idx + 1] )*delta*delta*onesixth;
121
122 return res;
123}

160// Flag useSpline is "true" only if second derivatives are filled
161inline
162 void G4PhysicsVector::SetSpline(G4bool val)
163{
164 if(val) {
165 if(0 == secDerivative.size() && 0 < dataVector.size()) {
166 FillSecondDerivatives();
167 }
168 } else {
169 useSpline = false;
170 secDerivative.clear();
171 }
172}

192inline
193size_t G4PhysicsVector::FindBinLocation(G4double theEnergy) const
194{
195 size_t bin;
196 if(type == T_G4PhysicsLogVector) {
197 bin = size_t(G4Log(theEnergy)/dBin - baseBin);
198 if(bin > 0 && theEnergy < binVector[bin]) { --bin; }
199 else if(theEnergy > binVector[bin+1]) { ++bin; }
200 } else if(type == T_G4PhysicsLinearVector) {
201 bin = size_t( theEnergy/dBin - baseBin );
202 if(bin > 0 && theEnergy < binVector[bin]) { --bin; }
203 else if(theEnergy > binVector[bin+1]) { ++bin; }
204 } else {
205 // Bin location proposed by K.Genser (FNAL)
206 bin = std::lower_bound(binVector.begin(), binVector.end(), theEnergy)
207 - binVector.begin() - 1;
208 }
209 return std::min(bin, numberOfNodes-2);
210}
211
212//---------------------------------------------------------------
213
214inline size_t G4PhysicsVector::FindBin(G4double e, size_t idx) const
215{
216 size_t id = idx;
217 if(e < binVector[1]) {
218 id = 0;
219 } else if(e >= binVector[numberOfNodes-2]) {
220 id = numberOfNodes - 2;
221 } else if(idx >= numberOfNodes-2 || e < binVector[idx]
222 || e > binVector[idx+1]) {
223 id = FindBinLocation(e);
224 }
225 return id;
226}
