29#ifndef G4VDigitizerModule_H
30#define G4VDigitizerModule_H 1
31
32class G4DigiManager;
33class G4VDigiCollection;
34#include "globals.hh"
35//#include "g4rw/tvordvec.h"
36#include <vector>
37
38// class description:
39//
40// This is the abstract base class of the digitizer module. The user's
41// digitizer module which generates digits must be derived from this
42// class.
43// In the derived class constructor, name(s) of digi collection(s) which
44// are made by the digitizer module must be set to "collectionName" string
45// vector.
46
47class G4VDigitizerModule
48{
49 public: // with description
50 G4VDigitizerModule(G4String modName);
51 // Constructor. The user's concrete class must use this constructor
52 // by the constructor initializer of the derived class. The name of
53 // the detector module must be unique.
54 public:
55 virtual ~G4VDigitizerModule();
56 G4bool operator==(const G4VDigitizerModule &right) const;
57 G4bool operator!=(const G4VDigitizerModule &right) const;
58
59 public: // with description
60 virtual void Digitize() = 0;
61 // The pure virtual method that the derived class must implement.
62 // In the concrete implementation of this method, necessary digi
63 // collection object must be constructed and set to G4DCofThisEvent
64 // by StoreDigiCollection protected method.
65
66 protected:
67 void StoreDigiCollection(G4VDigiCollection* aDC);
68 void StoreDigiCollection(G4int DCID,G4VDigiCollection* aDC);
69
70 protected:
71 G4DigiManager* DigiManager;
72 G4String moduleName;
73 std::vector<G4String> collectionName;
74 G4int verboseLevel;
75
76 public:
77 inline G4int GetNumberOfCollections() const
78 { return collectionName.size(); }
79 inline G4String GetCollectionName(G4int i) const
80 { return collectionName[i]; }
81 inline G4String GetName() const
82 { return moduleName; }
83 inline void SetVerboseLevel(G4int val)
84 { verboseLevel = val; }
85};
86
87#endif
