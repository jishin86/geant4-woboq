29#ifndef G4DigiManager_h
30#define G4DigiManager_h 1
31
32#include "globals.hh"
33class G4Event;
34#include "G4VDigitizerModule.hh"
35class G4VHitsCollection;
36class G4VDigiCollection;
37class G4DMmessenger;
38#include "G4DCtable.hh"
39class G4RunManager;
40class G4SDManager;
41//#include "g4rw/tpordvec.h"
42#include <vector>
43
44// class description:
45//
46// This is a singleton class which manages the digitizer modules.
47// The user cannot access to the constructor. The pointer of the
48// only existing object can be got via G4DigiManager::GetDMpointer()
49// static method. The first invokation in the program makes the
50// singleton object.
51//
52
53class G4DigiManager
54{
55 public: // with description
56 static G4DigiManager* GetDMpointer();
57 // Returns the pointer to the singleton object
58 public:
59 static G4DigiManager* GetDMpointerIfExist();
60
61 protected:
62 G4DigiManager();
63
64 public:
65 ~G4DigiManager();
66
67 public: // with description
68 void AddNewModule(G4VDigitizerModule* DM);
69 // Registers the user's digitizer mudule. This method must be invoked when
70 // the user construct his/her digitizer module(s).
71 void Digitize(G4String mName);
72 // Invokes Digitize() method of specified digitizer module. This is a kind
73 // of service method. The user can invoke Digitize() method of a particular
74 // module without knowing the pointer of the module object. The argument
75 // "mName" is the name of the module, which is defined at the constructor
76 // of the concrete digitizer module.
77 G4VDigitizerModule* FindDigitizerModule(G4String mName);
78 // Returns the pointer to the digitizer module object with the given name.
79 // Null will be returned if the name is not defined.
80 const G4VHitsCollection* GetHitsCollection(G4int HCID, G4int eventID = 0);
81 const G4VDigiCollection* GetDigiCollection(G4int DCID, G4int eventID = 0);
82 // These two methods return the pointer to the hits and digi collection
83 // object, respectively. "HCID" and "DCID" are the ID numbers of hits and
84 // digi collections, which can be obtained vir the next two methods.
85 // If "eventID" is greater than zero, corresponding hits or digi collection
86 // of "eventID" prevuois event is returned so that event overlap can be
87 // handled. To do this, necessary number of events must be set to G4RunManager
88 // by G4RunManager::SetNumberOfEventsToBeStored() method previously to the
89 // event loop.
90 G4int GetHitsCollectionID(G4String HCname);
91 G4int GetDigiCollectionID(G4String DCname);
92 // Returns the ID number of hits and digi collections, respectively. "HCname"
93 // and "DCname" can be the collection name if it is unique, or can be detector
94 // or module name and the collection name connected by "/".
95 void SetDigiCollection(G4int DCID, G4VDigiCollection* aDC);
96 // This method must exclusively used by the base class of G4VDigitizerModule.
97 // To set digi collection, the user must use SetDigiCollection of G4VDigitizerModule.
98
99 public:
100 void SetVerboseLevel(G4int vl);
101 void List() const;
102
103 private:
104 static G4ThreadLocal G4DigiManager * fDManager;
105 G4int verboseLevel;
106 std::vector<G4VDigitizerModule*> DMtable;
107 G4DCtable* DCtable;
108 G4DMmessenger* theMessenger;
109 G4RunManager* runManager;
110 G4SDManager* SDManager;
111
112 public:
113 inline G4int GetVerboseLevel() const
114 { return verboseLevel; }
115 inline G4int GetCollectionCapacity() const
116 { return DCtable->entries(); }
117 inline G4int GetModuleCapacity() const
118 { return DMtable.size(); }
119 inline G4DCtable* GetDCtable() const
120 { return DCtable; }
121 inline void RestoreDCtable(G4DCtable* dc)
122 {
123 if(DCtable) delete DCtable;
124 DCtable = dc;
125 }
126 private:
127 //Disable copy constructor and assignement operator
128 G4DigiManager(const G4DigiManager&);
129 G4DigiManager& operator=(const G4DigiManager&);
130};
