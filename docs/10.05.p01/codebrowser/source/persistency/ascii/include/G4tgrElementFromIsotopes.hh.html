39#ifndef G4tgrElementFromIsotopes_h
40#define G4tgrElementFromIsotopes_h
41
42#include "globals.hh"
43
44#include <vector>
45
46#include "G4tgrElement.hh"
47
48class G4tgrElementFromIsotopes : public G4tgrElement
49{
50 public: // with description
51
52 G4tgrElementFromIsotopes();
53 ~G4tgrElementFromIsotopes();
54
55 G4tgrElementFromIsotopes( const std::vector<G4String>& wl );
56 // Construct the G4tgrElementFromIsotopes (fill its data members)
57 // interpreting the data in the list of words 'wl'
58
59 G4int GetNumberOfIsotopes() const { return theNoIsotopes; }
60 const G4String& GetComponent( G4int n ) const { return theComponents[n]; }
61 G4double GetAbundance( G4int n ) const { return theAbundances[n]; }
62
63 friend std::ostream& operator<<(std::ostream&,
64 const G4tgrElementFromIsotopes& );
65 private:
66
67 G4int theNoIsotopes;
68 std::vector<G4String> theComponents;
69 std::vector<G4double> theAbundances;
70
71};
