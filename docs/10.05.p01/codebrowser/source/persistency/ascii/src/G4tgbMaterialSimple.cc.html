41// -------------------------------------------------------------------------
42G4tgbMaterialSimple::G4tgbMaterialSimple()
43 : theZ(0.), theA(0.)
44{
45}
46
47
48// -------------------------------------------------------------------------
49G4tgbMaterialSimple::~G4tgbMaterialSimple()
50{
51}
52
53
54// -------------------------------------------------------------------------
55G4tgbMaterialSimple::G4tgbMaterialSimple( G4tgrMaterial* hgmate)
56{
57 theTgrMate = hgmate;
58 G4tgrMaterialSimple* matesimp = static_cast<G4tgrMaterialSimple*>(hgmate);
59 theZ = matesimp->GetZ();
60 theA = matesimp->GetA();
61}
62
63
64// -------------------------------------------------------------------------
65G4Material* G4tgbMaterialSimple::BuildG4Material()
66{
67 //----- construct new G4Material with no components (only itself)
68
69 G4Material* mate = new G4Material( GetName(), GetZ(), GetA(),
70 theTgrMate->GetDensity(),
71 kStateUndefined, STP_Temperature );
72
73#ifdef G4VERBOSE
74 if( G4tgrMessenger::GetVerboseLevel() >= 2 )
75 {
76 G4cout << " Constructing new G4Material simple: " << *mate << G4endl;
77 }
78#endif
79
80 return mate;
81}
82
83
84// -------------------------------------------------------------------------
85std::ostream& operator<<(std::ostream& os, const G4tgbMaterialSimple& mate)
86{
87 os << "Simple Material: " << mate.GetName() << G4endl
88 << " Z = " << mate.GetZ()
89 << " A = " << mate.GetA()
90 << " density = " << mate.GetDensity() << G4endl;
91 return os;
92}
