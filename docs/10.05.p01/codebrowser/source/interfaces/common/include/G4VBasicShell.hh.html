29#ifndef G4VBasicShell_H
30#define G4VBasicShell_H 1
31
32class G4UIcommandTree;
33class G4UIcommand;
34
35#include "G4UIsession.hh"
36#include "globals.hh"
37
38// Class description :
39//
40// G4VBasicShell : a base class to extract common things to various
41// sessions.
42//
43// It handles "seek" completion logic, help logic.
44// VBasicShell handles also commands like "cd, ls, pwd"
45// without passing by a Geant4 "intercom" command. This feature,
46// which is similar to a UNIX shell one, had given.
47// its name to the class.
48//
49// Class description - end :
50
51class G4VBasicShell : public G4UIsession
52{
53 public:
54 G4VBasicShell();
55 virtual ~G4VBasicShell();
56
57 virtual G4UIsession* SessionStart() = 0;
58 // null should be returned for interactive session
59
60 virtual void PauseSessionStart(const G4String& Prompt) = 0;
61 // Prompt string can be ignored
62
63 protected:
64 G4String ModifyToFullPathCommand(const char* aCommandLine) const;
65 // convert "BeamOn 10" to "/run/BeamOn 10" if the
66 // current working directory is "/run/"
67
68 G4String GetCurrentWorkingDirectory() const;
69 // directory string starts with '/' and ends with '/'
70
71 G4bool ChangeDirectory(const char* newDir);
72 // change directory to newDir
73 // false will be returned if the target directory doesn't exist
74
75 G4UIcommandTree* FindDirectory(const char* dirName) const;
76 // find G4UIcommandTree object
77 // null returned if the taregt does not exist
78
79 G4UIcommand* FindCommand(const char* commandName) const;
80 // find G4UIcommand object
81 // null returned if the target does not exist
82
83 G4String Complete(const G4String&);
84 // command completion
85
86 G4String FindMatchingPath(G4UIcommandTree*, const G4String&);
87
88 /////////////////////////////////////////////
89 // Methods involving an interactive G4cout //
90 /////////////////////////////////////////////
91 virtual void ExecuteCommand(const G4String&);
92 virtual G4bool GetHelpChoice(G4int&) = 0;
93 virtual void ExitHelp() const = 0;
94 void ApplyShellCommand(const G4String&, G4bool&, G4bool&);
95 void ShowCurrent(const G4String&) const;
96 void ChangeDirectoryCommand(const G4String&);
97 void ListDirectory(const G4String&) const;
98 void TerminalHelp(const G4String&);
99
100 private:
101 G4String currentDirectory;
102
103 G4String ModifyPath(const G4String& tempPath) const;
104};
105
106#endif
