88// --------------------------------------------------------------------------
89G4UIExecutive::G4UIExecutive(G4int argc, char** argv, const G4String& type)
90 : selected(kNone), session(NULL), shell(NULL), isGUI(false)
91{
92 G4cout << "Available UI session types: [ ";
93 if ( qt_build ) G4cout << "Qt, ";
94 if ( xm_build ) G4cout << "Xm, ";
95 if ( win32_build) G4cout << "Win32, ";
96 if ( wt_build ) G4cout << "Wt, ";
97 G4cout << "GAG, ";
98 if (tcsh_build ) G4cout << "tcsh, ";
99 G4cout << "csh ]" << G4endl;
100
101 // selecting session type...
102 // 1st priority : in case argumant specified
103 G4String stype = type;
104 stype.toLower(); // session type is case-insensitive.
105 if (type != "") SelectSessionByArg(stype);
106
107 // 2nd priority : refer environment variables (as backword compatibility)
108 if ( selected == kNone ) SelectSessionByEnv();
109
110 // 3rd priority : refer $HOME/.g4session
111 if ( selected == kNone ) {
112 G4String appinput = argv[0];
113 G4String appname = "";
114 size_t islash = appinput.find_last_of("/\\");
115 if (islash == G4String::npos)
116 appname = appinput;
117 else
118 appname = appinput(islash+1, appinput.size()-islash-1);
119
120 SelectSessionByFile(appname);
121 }
122
123 // 4th, best guess of session type
124 if ( selected == kNone) SelectSessionByBestGuess();
125
126 // instantiate a session...
127 switch ( selected ) {
128 case kQt:
129#if defined(G4UI_BUILD_QT_SESSION)
130 session = new G4UIQt(argc, argv);
131 isGUI = true;
132#endif
133 break;
134 case kXm:
135#if defined(G4UI_BUILD_XM_SESSION)
136 session = new G4UIXm(argc, argv);
137 isGUI = true;
138#endif
139 break;
140 case kWin32:
141#if defined(G4UI_BUILD_WIN32_SESSION)
142 DISCARD_PARAMETER(argc);
143 DISCARD_PARAMETER(argv);
144 session = new G4UIWin32();
145#endif
146 break;
147 case kWt:
148#if defined(G4UI_BUILD_WT_SESSION)
149 session = new G4UIWt(argc, argv);
150 isGUI = true;
151#endif
152 break;
153 case kGag:
154 DISCARD_PARAMETER(argc);
155 DISCARD_PARAMETER(argv);
156 session = new G4UIGAG();
157 isGUI = true;
158 break;
159 case kTcsh:
160#ifndef WIN32
161 DISCARD_PARAMETER(argc);
162 DISCARD_PARAMETER(argv);
163 shell = new G4UItcsh;
164 session = new G4UIterminal(shell);
165#endif
166 break;
167 case kCsh:
168 DISCARD_PARAMETER(argc);
169 DISCARD_PARAMETER(argv);
170 shell = new G4UIcsh;
171 session = new G4UIterminal(shell);
172 default:
173 break;
174 }
175
176 // fallback (csh)
177 if ( session == NULL ) {
178 G4Exception("G4UIExecutive::G4UIExecutive()",
179 "UI0002",
180 JustWarning,
181 "Specified session type is not build in your system,\n"
182 "or no session type is specified.\n"
183 "A fallback session type is used.");
184
185 selected = kCsh;
186 DISCARD_PARAMETER(argc);
187 DISCARD_PARAMETER(argv);
188 shell = new G4UIcsh;
189 session = new G4UIterminal(shell);
190 }
191}

199// --------------------------------------------------------------------------
200void G4UIExecutive::SelectSessionByArg(const G4String& stype)
201{
202 if ( qt_build && stype == "qt" ) selected = kQt;
203 else if ( xm_build && stype == "xm" ) selected = kXm;
204 else if ( win32_build && stype == "win32" ) selected = kWin32;
205 else if ( wt_build && stype == "wt" ) selected = kWt;
206 else if ( stype == "gag" ) selected = kGag;
207 else if ( tcsh_build && stype == "tcsh" ) selected = kTcsh;
208 else if ( stype == "csh" ) selected = kCsh;
209}

211// --------------------------------------------------------------------------
212void G4UIExecutive::SelectSessionByEnv()
213{
214 if ( qt_build && getenv("G4UI_USE_QT") ) selected = kQt;
215 else if ( xm_build && getenv("G4UI_USE_XM") ) selected = kXm;
216 else if ( win32_build && getenv("G4UI_USE_WIN32") ) selected = kWin32;
217 else if ( wt_build && getenv("G4UI_USE_WT") ) selected = kWt;
218 else if ( getenv("G4UI_USE_GAG") ) selected = kGag;
219 else if ( tcsh_build && getenv("G4UI_USE_TCSH") ) selected = kTcsh;
220}

222// --------------------------------------------------------------------------
223void G4UIExecutive::SelectSessionByFile(const G4String& appname)
224{
225 const char* path = getenv("HOME");
226 if( path == NULL ) return;
227 G4String homedir = path;
228
229#ifndef WIN32
230 G4String fname= homedir + "/.g4session";
231#else
232 G4String fname= homedir + "\\.g4session";
233#endif
234
235 std::ifstream fsession;
236 enum { BUFSIZE= 1024 }; char linebuf[BUFSIZE];
237
238 fsession.open(fname, std::ios::in);
239
240 G4String default_session = "";
241 G4int iline = 1;
242 sessionMap.clear();
243 while( fsession.good() ) {
244 if( fsession.eof()) break;
245 fsession.getline(linebuf, BUFSIZE);
246 G4String aline = linebuf;
247 aline.strip(G4String::both);
248 if ( aline(0) == '#' ) continue;
249 if ( aline == "" ) continue;
250 if ( iline == 1 )
251 default_session = aline;
252 else {
253 size_t idx = aline.find_first_of(" ");
254 if ( idx == G4String::npos ) break;
255 G4String aname = aline.substr(0, idx);
256 idx = aline.find_first_not_of(" ", idx);
257 if (idx == G4String::npos ) break;
258 G4String sname = aline.substr(idx, aline.size()-idx);
259 sessionMap[aname] = sname;
260 }
261 iline++;
262 }
263 fsession.close();
264
265 G4String stype = "";
266 std::map<G4String, G4String>::iterator it = sessionMap.find(appname);
267 if ( it != sessionMap.end() ) stype = sessionMap[appname];
268 else stype = default_session;
269 stype.toLower();
270
271 // select session...
272 if ( qt_build && stype == "qt" ) selected = kQt;
273 else if ( xm_build && stype == "xm" ) selected = kXm;
274 else if ( win32_build && stype == "win32" ) selected = kWin32;
275 //else if ( wt_build && stype == "wt" ) selected = kWt;
276 else if ( stype == "gag" ) selected = kGag;
277 else if ( tcsh_build && stype == "tcsh" ) selected = kTcsh;
278 else if ( stype == "csh" ) selected = kCsh;
279}
