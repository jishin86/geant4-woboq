56G4bool G4PSEnergyDeposit::ProcessHits(G4Step* aStep,G4TouchableHistory*)
57{
58 G4double edep = aStep->GetTotalEnergyDeposit();
59 if ( edep == 0. ) return FALSE;
60 edep *= aStep->GetPreStepPoint()->GetWeight(); // (Particle Weight)
61 G4int index = GetIndex(aStep);
62 EvtMap->add(index,edep);
63 return TRUE;
64}
65
66void G4PSEnergyDeposit::Initialize(G4HCofThisEvent* HCE)
67{
68 EvtMap = new G4THitsMap<G4double>(GetMultiFunctionalDetector()->GetName(),
69 GetName());
70 if(HCID < 0) {HCID = GetCollectionID(0);}
71 HCE->AddHitsCollection(HCID, (G4VHitsCollection*)EvtMap);
72}

85void G4PSEnergyDeposit::PrintAll()
86{
87 G4cout << " MultiFunctionalDet " << detector->GetName() << G4endl;
88 G4cout << " PrimitiveScorer " << GetName() << G4endl;
89 G4cout << " Number of entries " << EvtMap->entries() << G4endl;
90 std::map<G4int,G4double*>::iterator itr = EvtMap->GetMap()->begin();
91 for(; itr != EvtMap->GetMap()->end(); itr++) {
92 G4cout << " copy no.: " << itr->first
93 << " energy deposit: "
94 << *(itr->second)/GetUnitValue()
95 << " [" << GetUnit()<<"]"
96 << G4endl;
97 }
98}
