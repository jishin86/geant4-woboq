32// (Description)
33// This is a primitive scorer class for 3D scoring cell charge.
34// The Cell Charge is defined by a sum of charge inside the cell
35// which calculates the deposited charge in the cell.
36//
37//
38//
39// Created: 2007-08-14 Tsukasa ASO
40// 2010-07-22 Introduce Unit specification.

44G4PSCellCharge3D::G4PSCellCharge3D(G4String name,
45 G4int ni, G4int nj, G4int nk,
46 G4int depi, G4int depj, G4int depk)
47 :G4PSCellCharge(name),
48 fDepthi(depi),fDepthj(depj),fDepthk(depk)
49{
50 fNi=ni;
51 fNj=nj;
52 fNk=nk;
53}
54
55G4PSCellCharge3D::G4PSCellCharge3D(G4String name,const G4String& unit,
56 G4int ni, G4int nj, G4int nk,
57 G4int depi, G4int depj, G4int depk)
58 :G4PSCellCharge(name),
59 fDepthi(depi),fDepthj(depj),fDepthk(depk)
60{
61 fNi=ni;
62 fNj=nj;
63 fNk=nk;
64 SetUnit(unit);
65}
66
67G4PSCellCharge3D::~G4PSCellCharge3D()
68{;}
69
70G4int G4PSCellCharge3D::GetIndex(G4Step* aStep)
71{
72 const G4VTouchable* touchable = aStep->GetPreStepPoint()->GetTouchable();
73 G4int i = touchable->GetReplicaNumber(fDepthi);
74 G4int j = touchable->GetReplicaNumber(fDepthj);
75 G4int k = touchable->GetReplicaNumber(fDepthk);
76
77 return i*fNj*fNk+j*fNk+k;
78}
