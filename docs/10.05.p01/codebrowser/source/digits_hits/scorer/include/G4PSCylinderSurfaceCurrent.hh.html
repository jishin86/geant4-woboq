29#ifndef G4PSCylinderSurfaceCurrent_h
30#define G4PSCylinderSurfaceCurrent_h 1
31
32#include "G4VPrimitiveScorer.hh"
33#include "G4THitsMap.hh"
34
35#include "G4Tubs.hh"
36#include "G4PSDirectionFlag.hh"
37////////////////////////////////////////////////////////////////////////////////
38// (Description)
39// This is a primitive scorer class for scoring Surface Current.
40// Current version assumes only for G4Tubs shape, and the surface
41// is defined at the inner plane of the tubs.
42// The current is given in the unit of area.
43// e.g. (Number of tracks)/mm2.
44//
45// Surface is defined at the inner surface of the tube.
46// Direction R R+dR
47// 0 IN || OUT ->|<- | fCurrent_InOut
48// 1 IN ->| | fCurrent_In
49// 2 OUT |<- | fCurrent_Out
50//
51//
52// Created: 2007-03-21 Tsukasa ASO
53// 2010-07-22 Introduce Unit specification.
54//
55///////////////////////////////////////////////////////////////////////////////
56
57class G4PSCylinderSurfaceCurrent : public G4VPrimitiveScorer
58{
59
60 public: // with description
61 G4PSCylinderSurfaceCurrent(G4String name ,G4int direction,
62 G4int depth=0);
63 G4PSCylinderSurfaceCurrent(G4String name ,G4int direction,
64 const G4String& unit, G4int depth=0);
65 virtual ~G4PSCylinderSurfaceCurrent();
66
67 inline void Weighted(G4bool flg=true) { weighted = flg; }
68 // Multiply track weight
69
70 inline void DivideByArea(G4bool flg=true) { divideByArea = flg; }
71 // Divided by Area.
72
73
74 protected: // with description
75 virtual G4bool ProcessHits(G4Step*,G4TouchableHistory*);
76 G4int IsSelectedSurface(G4Step*,G4Tubs*);
77
78 public:
79 virtual void Initialize(G4HCofThisEvent*);
80 virtual void EndOfEvent(G4HCofThisEvent*);
81 virtual void clear();
82 virtual void DrawAll();
83 virtual void PrintAll();
84
85 virtual void SetUnit(const G4String& unit);
86
87 protected:
88 virtual void DefineUnitAndCategory();
89
90 private:
91 G4int HCID;
92 G4int fDirection;
93 G4THitsMap<G4double>* EvtMap;
94 G4bool weighted;
95 G4bool divideByArea;
96};
97
98#endif
