30#define G4SDStructure_h 1

32// Globals

34// G4VSensitiveDetector
35#include "G4VSensitiveDetector.hh"

38class G4HCofThisEvent;
39
40// class description:
41//
42// This class is exclusively used by G4SDManager for handling the tree
43// structure of the user's sensitive detector names.
44//
45
46class G4SDStructure
47{
48 public:
49 G4SDStructure(G4String aPath);
50 ~G4SDStructure();
51
52 G4bool operator==(const G4SDStructure &right) const;
53
54 void AddNewDetector(G4VSensitiveDetector*aSD, G4String treeStructure);
55 void Activate(G4String aName, G4bool sensitiveFlag);
56 void Initialize(G4HCofThisEvent*HCE);
57 void Terminate(G4HCofThisEvent*HCE);
58 G4VSensitiveDetector* FindSensitiveDetector(G4String aName, G4bool warning = true);
59 G4VSensitiveDetector* GetSD(G4String aName);
60 void ListTree();
61
62 private:
63 G4SDStructure* FindSubDirectory(G4String subD);
64 G4String ExtractDirName(G4String aPath);
65 void RemoveSD(G4VSensitiveDetector*);
66
67 private:
68 std::vector<G4SDStructure*> structure;
69 std::vector<G4VSensitiveDetector*> detector;
70 G4String pathName;
71 G4String dirName;
72 G4int verboseLevel;
73
74 public:
75 inline void SetVerboseLevel(G4int vl)
76 {
77 verboseLevel = vl;
78 for(size_t i=0; i<structure.size(); i++)
79 { structure[i]->SetVerboseLevel(vl); }
80 for(size_t j=0; j<detector.size(); j++)
81 { detector[j]->SetVerboseLevel(vl); }
82 };
83
84};
