30#ifndef G4VReadOutGeometry_h
31#define G4VReadOutGeometry_h
32
33#include "G4SensitiveVolumeList.hh"
34#include "G4Step.hh"
35#include "G4TouchableHistory.hh"
36
37class G4Navigator;
38
39class G4VReadOutGeometry
40{
41 protected:
42 virtual G4VPhysicalVolume* Build() = 0; // must return the world of the ROGeometry;
43
44 public:
45 G4VReadOutGeometry();
46 G4VReadOutGeometry(G4String);
47 virtual ~G4VReadOutGeometry();
48
49 G4bool operator==(const G4VReadOutGeometry &right) const;
50 G4bool operator!=(const G4VReadOutGeometry &right) const;
51
52 // buildROGeomety must be invoked to Build (ie Build() method)
53 // the ROGeometry. It sets up in addition the needed
54 // the G4Navigator used to navigate inside this ROGeometry.
55 void BuildROGeometry();
56 virtual G4bool CheckROVolume(G4Step*,G4TouchableHistory*&);
57
58 protected:
59 G4VReadOutGeometry(const G4VReadOutGeometry &right);
60 G4VReadOutGeometry & operator=(const G4VReadOutGeometry &right);
61
62 virtual G4bool FindROTouchable(G4Step*);
63
64 protected:
65 G4VPhysicalVolume* ROworld;
66 G4SensitiveVolumeList* fincludeList;
67 G4SensitiveVolumeList* fexcludeList;
68 G4String name;
69
70 G4Navigator* ROnavigator;
71 G4TouchableHistory* touchableHistory;
72
73 public:
74 inline const G4SensitiveVolumeList* GetIncludeList() const
75 { return fincludeList; }
76 inline void SetIncludeList(G4SensitiveVolumeList* value)
77 { fincludeList = value; }
78 inline const G4SensitiveVolumeList* GetExcludeList() const
79 { return fexcludeList; }
80 inline void SetExcludeList(G4SensitiveVolumeList* value)
81 { fexcludeList = value; }
82 inline G4String GetName() const
83 { return name; }
84 inline void SetName(G4String value)
85 { name = value; }
86 // ADDED:
87 inline G4VPhysicalVolume* GetROWorld() const
88 { return ROworld;}
89};
