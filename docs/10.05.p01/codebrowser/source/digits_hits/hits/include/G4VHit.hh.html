29#ifndef G4VHit_h
30#define G4VHit_h 1
31
32#include "globals.hh"
33#include <vector>
34#include <map>
35
36class G4AttDef;
37class G4AttValue;
38
39// class description:
40//
41// This is the base class of hit object. The user should derive this
42// base class to make his/her own hit class. Two virtual method Draw()
43// and Print() can be implemented if the user wants these functionarities.
44// If a concrete hit class is used as a transient class, G4Allocator
45// must be used.
46
47class G4VHit
48{
49
50 public:
51 G4VHit();
52 virtual ~G4VHit();
53
54 G4bool operator==(const G4VHit &right) const;
55
56 virtual void Draw();
57 virtual void Print();
58
59 virtual const std::map<G4String,G4AttDef>* GetAttDefs() const
60 { return 0; }
61 // If implemented by a derived class, returns a pointer to a map
62 // of attribute definitions for the attribute values below. The
63 // user must test the validity of this pointer. See
64 // G4Trajectory for an example of a concrete implementation of
65 // this method.
66 virtual std::vector<G4AttValue>* CreateAttValues() const
67 { return 0; }
68 // If implemented by a derived class, returns a pointer to a
69 // list of attribute values suitable, e.g., for picking. Each
70 // must refer to an attribute definition in the above map; its
71 // name is the key. The user must test the validity of this
72 // pointer (it must be non-zero and conform to the G4AttDefs,
73 // which may be checked with G4AttCheck) and delete the list
74 // after use. See G4Trajectory for an example of a concrete
75 // implementation of this method and
76 // G4VTrajectory::ShowTrajectory for an example of its use.
77
78};
79
80#endif
