45GFlashHitMaker::GFlashHitMaker()
46{
47 fTouchableHandle = new G4TouchableHistory(); // talk to ?@@@
48 fpNavigator = new G4Navigator();
49 fNaviSetup = false;
50}
51
52GFlashHitMaker::~GFlashHitMaker()
53{
54 delete fpNavigator;
55}
56
57void GFlashHitMaker::make(GFlashEnergySpot * aSpot, const G4FastTrack * aT)
58{
59 // Locate the spot
60 if (!fNaviSetup)
61 {
62 fpNavigator->
63 SetWorldVolume(G4TransportationManager::GetTransportationManager()->
64 GetNavigatorForTracking()->GetWorldVolume() );
65 fpNavigator->
66 LocateGlobalPointAndUpdateTouchable(aSpot->GetPosition(),
67 fTouchableHandle(), false);
68 fNaviSetup = true;
69 }
70 else
71 {
72 fpNavigator->
73 LocateGlobalPointAndUpdateTouchable(aSpot->GetPosition(),
74 fTouchableHandle());
75 }
76
77 //--------------------------------------
78 // Fills attribute of the G4Step needed
79 // by our sensitive detector:
80 //-------------------------------------
81 // set spot information:
82 G4GFlashSpot theSpot(aSpot, aT, fTouchableHandle);
83 ///Navigator
84 //--------------------------------------
85 // Produce Hits
86 // call sensitive part: taken/adapted from the stepping:
87 // Send G4Step information to Hit/Dig if the volume is sensitive
88 //--------------G4TouchableHistory----------------------------------------
89
90 G4VPhysicalVolume* pCurrentVolume = fTouchableHandle()->GetVolume();
91 G4VSensitiveDetector* pSensitive;
92 if( pCurrentVolume != 0 )
93 {
94 pSensitive = pCurrentVolume->GetLogicalVolume()->GetSensitiveDetector();
95 G4VGFlashSensitiveDetector * gflashSensitive =
96 dynamic_cast<G4VGFlashSensitiveDetector * > (pSensitive);
97 if( gflashSensitive )
98 {
99 gflashSensitive->Hit(&theSpot);
100 }
101 else if (( pSensitive ) &&
102 ( pCurrentVolume->GetLogicalVolume()->GetFastSimulationManager() )
103 ) // Using gflash without implementing the
104 // gflashSensitive detector interface -> not allowed!
105
106 {
107 G4cerr << "ERROR - GFlashHitMaker::make()" << G4endl
108 << " It is required to implement the "<< G4endl
109 << " G4VGFlashSensitiveDetector interface in "<< G4endl
110 << " addition to the usual SensitiveDetector class."
111 << G4endl;
112 G4Exception("GFlashHitMaker::make()", "InvalidSetup", FatalException,
113 "G4VGFlashSensitiveDetector interface not implemented.");
114 }
115 }
116 else
117 {
118 #ifdef GFLASH_DEBUG
119 G4cout << "GFlashHitMaker::Out of volume "<< G4endl;
120 #endif
121 }
122}
