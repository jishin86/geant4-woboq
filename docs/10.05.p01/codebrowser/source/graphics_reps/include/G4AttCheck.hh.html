28#ifndef G4ATTCHECK_HH
29#define G4ATTCHECK_HH
30
31// Class Description:
32//
33// @class G4AttCheck
34//
35// @brief Checks G4AttValue's and their corresponding G4AttDef map.
36//
37// Usage (a): G4AttCheck(values,definitions);
38// or (b): G4cout << G4AttCheck(values,definitions) << G4endl;
39//
40// For further details, see the HepRep home page at http://heprep.freehep.org
41//
42// @author J.Allison
43// @author J.Perl
44// Class Description - End:
45
46#include "globals.hh"
47
48#include <set>
49#include <vector>
50#include <map>
51#include <iostream>
52
53class G4AttValue;
54class G4AttDef;
55
56class G4AttCheck {
57
58public: // With description
59
60 G4AttCheck
61 (const std::vector<G4AttValue>* values,
62 const std::map<G4String,G4AttDef>* definitions);
63
64 ~G4AttCheck();
65
66 const std::vector<G4AttValue>* GetAttValues() const {
67 return fpValues;
68 }
69
70 const std::map<G4String,G4AttDef>* GetAttDefs() const {
71 return fpDefinitions;
72 }
73
74 G4bool Check(const G4String& leader = "") const;
75 // Check only. Silent unless error - then G4cerr. Returns error.
76 // Provide leader, e.g., name of calling function.
77
78 G4bool Standard
79 (std::vector<G4AttValue>* standardValues,
80 std::map<G4String,G4AttDef>* standardDefinitions)
81 const;
82 // Places standard versions in provided containers and returns error.
83
84 friend std::ostream& operator<< (std::ostream&, const G4AttCheck&);
85
86private:
87
88 void AddValuesAndDefs
89 (std::vector<G4AttValue>* newValues,
90 std::map<G4String,G4AttDef>* newDefinitions,
91 const G4String& oldName,
92 const G4String& name,
93 const G4String& value,
94 const G4String& extra,
95 const G4String& description = "") const; // Utility function for Standard.
96
97 void Init(); // Initialises maps and sets
98
99 const std::vector<G4AttValue>* fpValues;
100 const std::map<G4String,G4AttDef>* fpDefinitions;
101
102 static G4ThreadLocal G4bool fFirst; // Flag for initialising the following containers.
103 static G4ThreadLocal std::set<G4String> *fUnitCategories; // Set of legal unit categories.
104 static G4ThreadLocal std::map<G4String,G4String> *fStandardUnits; // Standard units.
105 static G4ThreadLocal std::set<G4String> *fCategories; // Set of legal categories.
106 static G4ThreadLocal std::set<G4String> *fUnits; // Set of legal units.
107 static G4ThreadLocal std::set<G4String> *fValueTypes; // Set of legal value types.
108};
109
110#endif //G4ATTCHECK_HH
