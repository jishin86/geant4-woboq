30// Base class for all things visible, i.e., which have Vis Attributes.
31
32#include "G4Visible.hh"
33#include "G4VisAttributes.hh"
34#include "G4ios.hh"
35
36G4Visible::G4Visible ():
37 fpVisAttributes (0),
38 fAllocatedVisAttributes (false)
39{}
40
41G4Visible::G4Visible (const G4Visible& visible){
42 fAllocatedVisAttributes = visible.fAllocatedVisAttributes;
43 if (fAllocatedVisAttributes)
44 fpVisAttributes = new G4VisAttributes(*visible.fpVisAttributes);
45 else fpVisAttributes = visible.fpVisAttributes;
46}
47
48G4Visible::G4Visible (const G4VisAttributes* pVA):
49 fpVisAttributes (pVA),
50 fAllocatedVisAttributes (false)
51{}
52
53G4Visible::~G4Visible () {
54 if (fAllocatedVisAttributes) delete fpVisAttributes;
55}
56
57G4Visible& G4Visible::operator= (const G4Visible& rhs) {
58 if (&rhs == this) return *this;
59 fAllocatedVisAttributes = rhs.fAllocatedVisAttributes;
60 if (fAllocatedVisAttributes) {
61 delete fpVisAttributes;
62 fpVisAttributes = new G4VisAttributes(*rhs.fpVisAttributes);
63 }
64 else fpVisAttributes = rhs.fpVisAttributes;
65 return *this;
66}
67
68void G4Visible::SetVisAttributes (const G4VisAttributes& VA) {
69 // Allocate G4VisAttributes on the heap in case the user specifies a
70 // short-lived VA for a long-lived G4Visible. Flag so that it can
71 // be deleted in the destructor.
72 // First delete any G4VisAttributes already on the heap...
73 if (fAllocatedVisAttributes) delete fpVisAttributes;
74 fpVisAttributes = new G4VisAttributes(VA);
75 fAllocatedVisAttributes = true;
76}
77
78
79void G4Visible::SetVisAttributes (const G4VisAttributes* pVA) {
80 // First delete any G4VisAttributes already on the heap...
81 if (fAllocatedVisAttributes) delete fpVisAttributes;
82 fpVisAttributes = pVA;
83 fAllocatedVisAttributes = false;
84}
85
86G4bool G4Visible::operator != (const G4Visible& right) const {
87 if (fpVisAttributes && right.fpVisAttributes)
88 return *fpVisAttributes != *right.fpVisAttributes;
89 else if (!fpVisAttributes && !right.fpVisAttributes) return false;
90 else return true;
91}
92
93std::ostream& operator << (std::ostream& os, const G4Visible& v) {
94 if (v.fpVisAttributes) return os << *(v.fpVisAttributes);
95 else return os << "No Visualization Attributes";
96}
