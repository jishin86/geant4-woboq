30#include "G4UIcmdWith3Vector.hh"
31
32G4UIcmdWith3Vector::G4UIcmdWith3Vector
33(const char * theCommandPath,G4UImessenger * theMessenger)
34:G4UIcommand(theCommandPath,theMessenger)
35{
36 G4UIparameter * dblParamX = new G4UIparameter('d');
37 SetParameter(dblParamX);
38 G4UIparameter * dblParamY = new G4UIparameter('d');
39 SetParameter(dblParamY);
40 G4UIparameter * dblParamZ = new G4UIparameter('d');
41 SetParameter(dblParamZ);
42}
43
44G4ThreeVector G4UIcmdWith3Vector::GetNew3VectorValue(const char* paramString)
45{
46 return ConvertTo3Vector(paramString);
47}
48
49void G4UIcmdWith3Vector::SetParameterName
50(const char * theNameX,const char * theNameY,const char * theNameZ,
51G4bool omittable,G4bool currentAsDefault)
52{
53 G4UIparameter * theParamX = GetParameter(0);
54 theParamX->SetParameterName(theNameX);
55 theParamX->SetOmittable(omittable);
56 theParamX->SetCurrentAsDefault(currentAsDefault);
57 G4UIparameter * theParamY = GetParameter(1);
58 theParamY->SetParameterName(theNameY);
59 theParamY->SetOmittable(omittable);
60 theParamY->SetCurrentAsDefault(currentAsDefault);
61 G4UIparameter * theParamZ = GetParameter(2);
62 theParamZ->SetParameterName(theNameZ);
63 theParamZ->SetOmittable(omittable);
64 theParamZ->SetCurrentAsDefault(currentAsDefault);
65}
66
67void G4UIcmdWith3Vector::SetDefaultValue(G4ThreeVector vec)
68{
69 G4UIparameter * theParamX = GetParameter(0);
70 theParamX->SetDefaultValue(vec.x());
71 G4UIparameter * theParamY = GetParameter(1);
72 theParamY->SetDefaultValue(vec.y());
73 G4UIparameter * theParamZ = GetParameter(2);
74 theParamZ->SetDefaultValue(vec.z());
75}
