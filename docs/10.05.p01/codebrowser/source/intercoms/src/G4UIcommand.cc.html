124G4int G4UIcommand::DoIt(G4String parameterList)
125{
126 G4String correctParameters;
127 G4int n_parameterEntry = parameter.size();
128 if( n_parameterEntry != 0 )
129 {
130 G4String aToken;
131 G4String correctToken;
132 G4Tokenizer parameterToken( parameterList );
133 for( G4int i_thParameter=0; i_thParameter<n_parameterEntry; i_thParameter++ )
134 {
135 if(i_thParameter > 0)
136 {
137 correctParameters.append(" ");
138 }
139 aToken = parameterToken();
140 if( aToken.length()>0 && aToken(0)=='"' )
141 {
142 while( aToken(aToken.length()-1) != '"'
143 || ( aToken.length()==1 && aToken(0)=='"' ))
144 {
145 G4String additionalToken = parameterToken();
146 if( additionalToken.isNull() )
147 { return fParameterUnreadable+i_thParameter; }
148 aToken += " ";
149 aToken += additionalToken;
150 }
151 }
152 else if(i_thParameter==n_parameterEntry-1 && parameter[i_thParameter]->GetParameterType()=='s')
153 {
154 G4String anotherToken;
155 while(!((anotherToken=parameterToken()).isNull()))
156 {
157 G4int idxs = anotherToken.index("#");
158 if(idxs==G4int(std::string::npos))
159 {
160 aToken += " ";
161 aToken += anotherToken;
162 }
163 else if(idxs>0)
164 {
165 aToken += " ";
166 aToken += anotherToken(0,idxs);
167 break;
168 }
169 else
170 { break; }
171 }
172 }
173
174 if( aToken.isNull() || aToken == "!" )
175 {
176 if(parameter[i_thParameter]->IsOmittable())
177 {
178 if(parameter[i_thParameter]->GetCurrentAsDefault())
179 {
180 G4Tokenizer cvSt(messenger->GetCurrentValue(this));
181 G4String parVal;
182 for(G4int ii=0;ii<i_thParameter;ii++)
183 {
184 parVal = cvSt();
185 if (parVal(0)=='"')
186 {
187 while( parVal(parVal.length()-1) != '"' )
188 {
189 G4String additionalToken = cvSt();
190 if( additionalToken.isNull() )
191 { return fParameterUnreadable+i_thParameter; }
192 parVal += " ";
193 parVal += additionalToken;
194 }
195 }
196 }
197 G4String aCVToken = cvSt();
198 if (aCVToken(0)=='"')
199 {
200 while( aCVToken(aCVToken.length()-1) != '"' )
201 {
202 G4String additionalToken = cvSt();
203 if( additionalToken.isNull() )
204 { return fParameterUnreadable+i_thParameter; }
205 aCVToken += " ";
206 aCVToken += additionalToken;
207 }
208 // aCVToken.strip(G4String::both,'"');
209 }
210 correctParameters.append(aCVToken);
211 }
212 else
213 { correctParameters.append(parameter[i_thParameter]->GetDefaultValue()); }
214 }
215 else
216 { return fParameterUnreadable+i_thParameter; }
217 }
218 else
219 {
220 G4int stat = parameter[i_thParameter]->CheckNewValue( aToken );
221 if(stat) return stat+i_thParameter;
222 correctParameters.append(aToken);
223 }
224 }
225 }
226
227 if(CheckNewValue( correctParameters ))
228 { return fParameterOutOfRange+99; }
229
230 if(workerThreadOnly && G4Threading::IsMasterThread()) return 0;
231
232 messenger->SetNewValue( this, correctParameters );
233 return 0;
234}

322G4String G4UIcommand::UnitsList(const char* unitCategory)
323{
324 G4String retStr;
325 G4UnitsTable& UTbl = G4UnitDefinition::GetUnitsTable();
326 size_t i;
327 for(i=0;i<UTbl.size();i++)
328 { if(UTbl[i]->GetName()==unitCategory) break; }
329 if(i==UTbl.size())
330 {
331 G4cerr << "Unit category <" << unitCategory << "> is not defined." << G4endl;
332 return retStr;
333 }
334 G4UnitsContainer& UCnt = UTbl[i]->GetUnitsList();
335 retStr = UCnt[0]->GetSymbol();
336 G4int je = UCnt.size();
337 for(G4int j=1;j<je;j++)
338 {
339 retStr += " ";
340 retStr += UCnt[j]->GetSymbol();
341 }
342 for(G4int k=0;k<je;k++)
343 {
344 retStr += " ";
345 retStr += UCnt[k]->GetName();
346 }
347 return retStr;
348}

518// ------------------ type check routines -------------------
519
520G4int G4UIcommand::
521TypeCheck(const char * t)
522{
523 G4String aNewValue;
524 char type;
525 std::istringstream is(t);
526 for (unsigned i=0; i< parameter.size(); i++) {
527 is >> aNewValue;
528 type = toupper(parameter[i]->GetParameterType());
529 switch ( type ) {
530 case 'D':
531 if( IsDouble(aNewValue)==0 ){
532 G4cerr << aNewValue << ": double value expected."
533 << G4endl;
534 return 0;
535 } break;
536 case 'I':
537 if( IsInt(aNewValue,20)==0 ){
538 G4cerr <<aNewValue<<": integer expected."
539 <<G4endl;
540 return 0;
541 } break;
542 case 'S':
543 break;
544 case 'B':
545 aNewValue.toUpper();
546 if (aNewValue == "Y" || aNewValue == "N"
547 ||aNewValue == "YES" || aNewValue == "NO"
548 ||aNewValue == "1" || aNewValue == "0"
549 ||aNewValue == "T" || aNewValue == "F"
550 ||aNewValue == "TRUE" || aNewValue == "FALSE")
551 return 1;
552 else return 0;
553 break;
554 default: ;
555 }
556 }
557 return 1;
558}

594G4int G4UIcommand::
595IsDouble(const char* buf)
596{
597 const char* p= buf;
598 switch( *p) {
599 case '+': case '-': ++p;
600 if( isdigit(*p) ) {
601 while( isdigit( (G4int)(*p) )) { ++p; }
602 switch ( *p ) {
603 case '\0': return 1;
604 // break;
605 case 'E': case 'e':
606 return ExpectExponent(++p );
607 // break;
608 case '.': ++p;
609 if( *p == '\0' ) return 1;
610 if( *p == 'e' || *p =='E' ) return ExpectExponent(++p );
611 if( isdigit(*p) ) {
612 while( isdigit( (G4int)(*p) )) { ++p; }
613 if( *p == '\0' ) return 1;
614 if( *p == 'e' || *p =='E') return ExpectExponent(++p);
615 } else return 0; break;
616 default: return 0;
617 }
618 }
619 if( *p == '.' ) { ++p;
620 if( isdigit(*p) ) {
621 while( isdigit( (G4int)(*p) )) { ++p; }
622 if( *p == '\0' ) return 1;
623 if( *p == 'e' || *p =='E') return ExpectExponent(++p);
624 }
625 }
626 break;
627 case '.': ++p;
628 if( isdigit(*p) ) {
629 while( isdigit( (G4int)(*p) )) { ++p; }
630 if( *p == '\0' ) return 1;
631 if( *p == 'e' || *p =='E' ) return ExpectExponent(++p);
632 } break;
633 default: // digit is expected
634 if( isdigit(*p) ) {
635 while( isdigit( (G4int)(*p) )) { ++p; }
636 if( *p == '\0' ) return 1;
637 if( *p == 'e' || *p =='E') return ExpectExponent(++p);
638 if( *p == '.' ) { ++p;
639 if( *p == '\0' ) return 1;
640 if( *p == 'e' || *p =='E') return ExpectExponent(++p);
641 if( isdigit(*p) ) {
642 while( isdigit( (G4int)(*p) )) { ++p; }
643 if( *p == '\0' ) return 1;
644 if( *p == 'e' || *p =='E') return ExpectExponent(++p);
645 }
646 }
647 }
648 }
649 return 0;
650}

653// ------------------ range Check routines -------------------
654G4int G4UIcommand::
655RangeCheck(const char* t) {
656 yystype result;
657 char type;
658 bp = 0; // reset buffer pointer for G4UIpGetc()
659 std::istringstream is(t);
660 for (unsigned i=0; i< parameter.size(); i++) {
661 type= toupper(parameter[i]->GetParameterType());
662 switch ( type ) {
663 case 'D': is >> newVal[i].D; break;
664 case 'I': is >> newVal[i].I; break;
665 case 'S': is >> newVal[i].S; break;
666 case 'B': is >> newVal[i].C; break;
667 default: ;
668 }
669 }
670 // PrintToken(); // Print tokens (consumes all tokens)
671 token= Yylex();
672 result = Expression();
673
674 if( paramERR == 1 ) return 0;
675 if( result.type != CONSTINT) {
676 G4cerr << "Illegal Expression in parameter range." << G4endl;
677 return 0;
678 }
679 if ( result.I ) return 1;
680 G4cerr << "parameter out of range: "<< rangeString << G4endl;
681 return 0;
682}

695
696yystype G4UIcommand::
697LogicalORExpression(void)
698{
699 yystype result;
700 yystype p;
701 p = LogicalANDExpression();
702 if( token != LOGICALOR) return p;
703 if( p.type == CONSTSTRING || p.type == IDENTIFIER ) {
704 G4cerr << "Parameter range: illegal type at '||'" << G4endl;
705 paramERR = 1;
706 }
707 result.I = p.I;
708 while (token == LOGICALOR)
709 {
710 token = Yylex();
711 p = LogicalANDExpression();
712 if( p.type == CONSTSTRING || p.type == IDENTIFIER ) {
713 G4cerr << "Parameter range: illegal type at '||'" <<G4endl;
714 paramERR = 1;
715 }
716 switch (p.type) {
717 case CONSTINT:
718 result.I += p.I;
719 result.type = CONSTINT; break;
720 case CONSTDOUBLE:
721 result.I += (p.D != 0.0);
722 result.type = CONSTINT; break;
723 default:
724 G4cerr << "Parameter range: unknown type"<<G4endl;
725 paramERR = 1;
726 }
727 }
728 return result;
729}
730
731yystype G4UIcommand::
732LogicalANDExpression(void)
733{
734 yystype result;
735 yystype p;
736 p = EqualityExpression();
737 if( token != LOGICALAND) return p;
738 if( p.type == CONSTSTRING || p.type == IDENTIFIER ) {
739 G4cerr << "Parameter range: illegal type at '&&'" << G4endl;
740 paramERR = 1;
741 }
742 result.I = p.I;
743 while (token == LOGICALAND)
744 {
745 token = Yylex();
746 p = EqualityExpression();
747 if( p.type == CONSTSTRING || p.type == IDENTIFIER ) {
748 G4cerr << "Parameter range: illegal type at '&&'" << G4endl;
749 paramERR = 1;
750 }
751 switch (p.type) {
752 case CONSTINT:
753 result.I *= p.I;
754 result.type = CONSTINT; break;
755 case CONSTDOUBLE:
756 result.I *= (p.D != 0.0);

924
925G4int G4UIcommand::
926Eval2(yystype arg1, G4int op, yystype arg2)
927{
928 char newValtype;
929 if( (arg1.type != IDENTIFIER) && (arg2.type != IDENTIFIER)) {
930 G4cerr << commandName
931 << ": meaningless comparison"
932 << G4endl;
933 paramERR = 1;
934 }
935
936 if( arg1.type == IDENTIFIER) {
937 unsigned i = IndexOf( arg1.S );
938 newValtype = toupper(parameter[i]->GetParameterType());
939 switch ( newValtype ) {
940 case 'I':
941 if( arg2.type == CONSTINT ) {
942 return CompareInt( newVal[i].I, op, arg2.I );
943//===================================================================
944// MA - 2018.07.23
945 } else if( arg2.type == IDENTIFIER ) {
946 unsigned iii = IndexOf( arg2.S );
947 char newValtype2 = toupper(parameter[iii]->GetParameterType());
948 if( newValtype2 == 'I' ) {
949 return CompareInt( newVal[i].I, op, newVal[iii].I );
950 } else if( newValtype2 == 'D' ) {
951 G4cerr << "Warning : Integer is compared with double : "
952 << rangeString << G4endl;
953 return CompareDouble( newVal[i].I, op, newVal[iii].D );
954 }
955//===================================================================
956 } else {
957 G4cerr << "integer operand expected for "
958 << rangeString
959 << '.' << G4endl;
960 } break;
961 case 'D':
962 if( arg2.type == CONSTDOUBLE ) {
963 return CompareDouble( newVal[i].D, op, arg2.D );
964 } else if ( arg2.type == CONSTINT ) { // integral promotion
965 return CompareDouble( newVal[i].D, op, arg2.I );
966//===================================================================
967// MA - 2018.07.23
968 } else if( arg2.type == IDENTIFIER ) {
969 unsigned iii = IndexOf( arg2.S );
970 char newValtype2 = toupper(parameter[iii]->GetParameterType());
971 if( newValtype2 == 'I' ) {
972 return CompareDouble( newVal[i].D, op, newVal[iii].I );
973 } else if( newValtype2 == 'D' ) {
974 return CompareDouble( newVal[i].D, op, newVal[iii].D );
975 }
976//===================================================================
977 } break;
978 default: ;
979 }
980 }
981 if( arg2.type == IDENTIFIER) {
982 unsigned i = IndexOf( arg2.S );
983 newValtype = toupper(parameter[i]->GetParameterType());
984 switch ( newValtype ) {
985 case 'I':
986 if( arg1.type == CONSTINT ) {
987 return CompareInt( arg1.I, op, newVal[i].I );
988 } else {
989 G4cerr << "integer operand expected for "
990 << rangeString
991 << '.' << G4endl;
992 } break;
993 case 'D':
994 if( arg1.type == CONSTDOUBLE ) {
995 return CompareDouble( arg1.D, op, newVal[i].D );
996 } else
997 if ( arg1.type == CONSTINT ) { // integral promotion
998 return CompareDouble( arg1.I, op, newVal[i].D );
999 } break;
1000 default: ;
1001 }
1002 }
1003 return 0;
1004}

1090tokenNum G4UIcommand::
1091Yylex() // reads input and returns token number, KR486
1092{ // (returns EOF)
1093 G4int c;
1094 G4String buf;
1095
1096 while(( c= G4UIpGetc())==' '|| c=='\t' || c== '\n' )
1097 ;
1098 if (c== EOF)
1099 return (tokenNum)EOF; // KR488
1100 buf= "";
1101 if (isdigit(c) || c== '.') { // I or D
1102 do {
1103 buf += G4String((unsigned char)c);
1104 c=G4UIpGetc();
1105 } while (c=='.' || isdigit(c) ||
1106 c=='e' || c=='E' || c=='+' || c=='-');
1107 G4UIpUngetc(c);
1108 const char* t = buf;
1109 std::istringstream is(t);
1110 if ( IsInt(buf.data(),20) ) {
1111 is >> yylval.I;
1112 return CONSTINT;
1113 } else
1114 if ( IsDouble(buf.data()) ) {
1115 is >> yylval.D;
1116 return CONSTDOUBLE;
1117 } else {
1118 G4cerr << buf<<": numeric format error."<<G4endl;
1119 }
1120 }
1121 buf="";
1122 if (isalpha(c)|| c=='_') { // IDENTIFIER
1123 do {
1124 buf += G4String((unsigned char)c);
1125 } while ((c=G4UIpGetc()) != EOF && (isalnum(c) || c=='_'));
1126 G4UIpUngetc(c);
1127 if( IsParameter(buf) ) {
1128 yylval.S =buf;
1129 return IDENTIFIER;
1130 } else {
1131 G4cerr << buf << " is not a parameter name."<< G4endl;
1132 paramERR = 1;
1133 }
1134 }
1135 switch (c) {
1136 case '>': return (tokenNum) Follow('=', GE, GT);
1137 case '<': return (tokenNum) Follow('=', LE, LT);
1138 case '=': return (tokenNum) Follow('=', EQ, '=');
1139 case '!': return (tokenNum) Follow('=', NE, '!');
1140 case '|': return (tokenNum) Follow('|', LOGICALOR, '|');
1141 case '&': return (tokenNum) Follow('&', LOGICALAND, '&');
1142 default:
1143 return (tokenNum) c;
1144 }
