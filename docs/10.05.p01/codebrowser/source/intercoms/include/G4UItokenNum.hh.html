35enum tokenNum
36{
37 NONE = 0,
38 IDENTIFIER = 257,
39 CONSTINT = 258,
40 CONSTDOUBLE= 259,
41 CONSTCHAR = 260,
42 CONSTSTRING= 261,
43 GT = 262,
44 GE = 263,
45 LT = 264,
46 LE = 265,
47 EQ = 266,
48 NE = 267,
49 //LOGICALNOT = 268,
50 LOGICALOR = 269,
51 LOGICALAND = 270,
52 SCAREAMER = 33,
53 LPAREN = 40,
54 PLUS = 43,
55 MINUS = 45
56};
57
58
59typedef struct yystype
60{
61 tokenNum type;
62 G4double D;
63 G4int I;
64 char C;
65 G4String S;
66
67 yystype() : type(NONE), D(0.0), I(0), C(' '), S("")
68 {
69 }
70 G4bool operator==(const yystype& right) const
71 {
72 return (this == &right)?1:0;
73 }
74 yystype& operator=(const yystype& right)
75 {
76 if (&right==this) return *this;
77 type = right.type;
78 D = right.D;
79 I = right.I;
80 C = right.C;
81 S = right.S;
82 return *this;
83 }
84 yystype(const yystype& right)
85 {
86 *this=right;
87 }
88} yystype;
