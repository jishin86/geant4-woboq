30#ifndef G4UIcmdWithAString_H
31#define G4UIcmdWithAString_H 1
32
33#include "G4UIcommand.hh"
34
35// class description:
36// A concrete class of G4UIcommand. The command defined by this class
37// takes a string. Incase the parameter string contains space(s), it
38// must be enclosed by double-quotations (").
39// General information of G4UIcommand is given in G4UIcommand.hh.
40
41class G4UIcmdWithAString : public G4UIcommand
42{
43 public: // with description
44 G4UIcmdWithAString
45 (const char * theCommandPath,G4UImessenger * theMessenger);
46 // Constructor. The command string with full path directory
47 // and the pointer to the messenger must be given.
48 void SetParameterName(const char * theName,G4bool omittable,
49 G4bool currentAsDefault=false);
50 // Set the parameter name.
51 // If "omittable" is set as true, the user of this command can ommit
52 // the value when he/she applies the command. If "omittable" is false,
53 // the user must supply the parameter string.
54 // "currentAsDefault" flag is valid only if "omittable" is true. If this
55 // flag is true, the current value is used as the default value when the
56 // user ommit the parameter. If this flag is false, the value given by the
57 // next SetDefaultValue() method is used.
58 void SetCandidates(const char * candidateList);
59 // Defines the candidates of the parameter string. Candidates listed in
60 // the argument must be separated by space(s).
61 void SetDefaultValue(const char * defVal);
62 // Set the default value of the parameter. This default value is used
63 // when the user of this command ommits the parameter value, and
64 // "ommitable" is true and "curreutAsDefault" is false.
65};
66
67#endif
