37#ifndef G4SmoothTrajectoryPoint_h
38#define G4SmoothTrajectoryPoint_h 1
39
40#include "trkgdefs.hh"
41#include "G4VTrajectoryPoint.hh"
42#include "globals.hh" // Include from 'global'
43#include "G4ThreeVector.hh" // Include from 'geometry'
44#include "G4Allocator.hh" // Include from 'particle+matter'
45
46
47class G4SmoothTrajectoryPoint : public G4VTrajectoryPoint
48{
49
50//--------
51public: // without description
52//--------
53
54// Constructor/Destructor
55 G4SmoothTrajectoryPoint();
56 // No auxiliary points setter, so must set the points in the
57 // constructor already
58 G4SmoothTrajectoryPoint(G4ThreeVector pos,
59 std::vector<G4ThreeVector>* auxiliaryPoints);
60 G4SmoothTrajectoryPoint(G4ThreeVector pos);
61 G4SmoothTrajectoryPoint(const G4SmoothTrajectoryPoint &right);
62 virtual ~G4SmoothTrajectoryPoint();
63
64private:
65 G4SmoothTrajectoryPoint& operator= (const G4SmoothTrajectoryPoint&);
66
67public:
68
69// Operators
70 inline void *operator new(size_t);
71 inline void operator delete(void *aTrajectoryPoint);
72 inline G4bool operator==(const G4SmoothTrajectoryPoint& right) const
73 { return (this==&right); };
74
75// Get/Set functions
76 inline const G4ThreeVector GetPosition() const
77 { return fPosition; }
78 inline const std::vector<G4ThreeVector>* GetAuxiliaryPoints() const
79 { return fAuxiliaryPointVector; }
80
81// Get method for HEPRep style attributes
82 virtual const std::map<G4String,G4AttDef>* GetAttDefs() const;
83 virtual std::vector<G4AttValue>* CreateAttValues() const;
84
85//---------
86 private:
87//---------
88
89// Member data
90 G4ThreeVector fPosition;
91 std::vector<G4ThreeVector>* fAuxiliaryPointVector;
92};
93
94extern G4TRACKING_DLL G4Allocator<G4SmoothTrajectoryPoint>*& aSmoothTrajectoryPointAllocator();
95
96inline void* G4SmoothTrajectoryPoint::operator new(size_t)
97{
98 if (!aSmoothTrajectoryPointAllocator())
99 { aSmoothTrajectoryPointAllocator()= new G4Allocator<G4SmoothTrajectoryPoint>; }
100 return (void *) aSmoothTrajectoryPointAllocator()->MallocSingle();
101}
102
103inline void G4SmoothTrajectoryPoint::operator delete(void *aTrajectoryPoint)
104{
105 aSmoothTrajectoryPointAllocator()->FreeSingle((G4SmoothTrajectoryPoint *) aTrajectoryPoint);
106}

108#endif
