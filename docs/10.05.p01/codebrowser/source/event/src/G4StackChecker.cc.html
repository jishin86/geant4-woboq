35G4StackChecker::G4StackChecker():
36 nullDirection(G4ThreeVector(0.0,0.0,0.0))
37{}
38
39G4StackChecker::~G4StackChecker()
40{}
41
42G4ClassificationOfNewTrack G4StackChecker::ClassifyNewTrack
43(const G4Track* track)
44{
45 G4ClassificationOfNewTrack result = fUrgent;
46 G4double e = track->GetKineticEnergy();
47 if ( (!(e < 0.0) && !(e > 0.0) && !(e == 0.0)) ||
48 track->GetMomentumDirection() == nullDirection)
49 {
50 result = fKill;
51 G4String nam = track->GetDefinition()->GetParticleName();
52 G4cout << "### G4StackChecker: event# "
53 << (G4EventManager::GetEventManager())->GetConstCurrentEvent()->GetEventID()
54 << " unacceptable " << nam << " is killed in the stack" << G4endl;
55 G4cout << "### " << nam << " have been produced by the process "
56 << track->GetCreatorProcess()->GetProcessName()
57 << " trackID= " << track->GetTrackID()
58 << " parentID= " << track->GetParentID()
59 << G4endl;
60 G4cout << "### E= " << track->GetKineticEnergy()
61 << " position= " << track->GetPosition()
62 << " direction= " << track->GetMomentumDirection()
63 << " time= " << track->GetGlobalTime()
64 << G4endl;
65 }
66 return result;
67}
