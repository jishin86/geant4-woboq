33void G4SmartTrackStack::dumpStatistics()
34{
35 // Print to stderr so that we can split stats output from normal
36 // output of Geant4 which is typically being printed to stdout
37 for (int i = 0; i < nTurn; i++) {
38 G4cerr << stacks[i]->GetNTrack() << " ";
39 G4cerr << stacks[i]->getTotalEnergy() << " ";
40 }
41 G4cerr << G4endl;
42}

44G4SmartTrackStack::G4SmartTrackStack()
45 :fTurn(0), nTurn(5), maxNTracks(0), nTracks(0)
46{
47 for(int i=0;i<nTurn;i++)
48 {
49 stacks[i] = new G4TrackStack(5000);
50 energies[i] = 0.;
51 }
52}

54G4SmartTrackStack::~G4SmartTrackStack()
55{
56 for (int i = 0; i < nTurn; i++) {
57 delete stacks[i];
58 }
59}

82G4StackedTrack G4SmartTrackStack::PopFromStack()
83{
84 G4StackedTrack aStackedTrack;
85
86 if (nTracks) {
87 while (true) {
88 if (stacks[fTurn]->GetNTrack()) {
89 aStackedTrack = stacks[fTurn]->PopFromStack();
90 energies[fTurn] -= aStackedTrack.GetTrack()->GetDynamicParticle()->GetTotalEnergy();
91 nTracks--;
92 break;
93 } else {
94 fTurn = (fTurn+1) % nTurn;
95 }
96 }
97 }
98
99 // dumpStatistics();
100 return aStackedTrack;
101}

103enum {
104 electronCode = 11, positronCode = -11, gammaCode = 22, neutronCode = 2112
105};
106
107void G4SmartTrackStack::PushToStack( const G4StackedTrack& aStackedTrack )
108{
109
110 G4int iDest = 0;
111 if (aStackedTrack.GetTrack()->GetParentID()) {
112 G4int code = aStackedTrack.GetTrack()->GetDynamicParticle()->GetPDGcode();
113 if (code == electronCode)
114 iDest = 2;
115 else if (code == gammaCode)
116 iDest = 3;
117 else if (code == positronCode)
118 iDest = 4;
119 else if (code == neutronCode)
120 iDest = 1;
121 } else {
122 // We have a primary track, which should go first.
123 fTurn = 0; // reseting the turn
124 }
125 stacks[iDest]->PushToStack(aStackedTrack);
126 energies[iDest] += aStackedTrack.GetTrack()->GetDynamicParticle()->GetTotalEnergy();
127 nTracks++;
128
129 G4int dy1 = stacks[iDest]->GetNTrack() - stacks[iDest]->GetSafetyValve1();
130 G4int dy2 = stacks[fTurn]->GetNTrack() - stacks[fTurn]->GetSafetyValve2();
131
132 if (dy1 > 0 || dy1 > dy2 ||
133 (iDest == 2 &&
134 stacks[iDest]->GetNTrack() < 50 && energies[iDest] < energies[fTurn])) {
135 fTurn = iDest;
136 }
137
138 if (nTracks > maxNTracks) maxNTracks = nTracks;
139}

141void G4SmartTrackStack::clear()
142{
143 for (int i = 0; i < nTurn; i++) {
144 stacks[i]->clear();
145 energies[i] = 0.0;
146 fTurn = 0;
147 }
148 nTracks = 0;
149}
150
151void G4SmartTrackStack::clearAndDestroy()
152{
153 for (int i = 0; i < nTurn; i++) {
154 stacks[i]->clearAndDestroy();
155 energies[i] = 0.0;
156 fTurn = 0;
157 }
158 nTracks = 0;
159}
