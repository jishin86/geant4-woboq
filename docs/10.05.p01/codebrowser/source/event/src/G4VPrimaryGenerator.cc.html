43G4bool G4VPrimaryGenerator::CheckVertexInsideWorld
44 (const G4ThreeVector& pos)
45{
46 G4Navigator* navigator= G4TransportationManager::GetTransportationManager()
47 -> GetNavigatorForTracking();
48
49 G4VPhysicalVolume* world= navigator-> GetWorldVolume();
50 G4VSolid* solid= world-> GetLogicalVolume()-> GetSolid();
51 EInside qinside= solid-> Inside(pos);
52
53 if( qinside != kInside) return false;
54 else return true;
55}
