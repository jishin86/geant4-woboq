38G4RayShooter::G4RayShooter()
39{
40 SetInitialValues();
41}

43void G4RayShooter::SetInitialValues()
44{
45 particle_definition = 0;
46 G4ThreeVector zero;
47 particle_momentum_direction = (G4ParticleMomentum)zero;
48 particle_energy = 1.0*GeV;
49 particle_position = zero;
50 particle_time = 0.0;
51 particle_polarization = zero;
52}
53
54G4RayShooter::~G4RayShooter()
55{
56}
57
58void G4RayShooter::Shoot(G4Event* evt,G4ThreeVector vtx,G4ThreeVector direc)
59{
60 if(!particle_definition)
61 {
62 G4ParticleTable* particleTable = G4ParticleTable::GetParticleTable();
63 G4String particleName;
64 particle_definition = particleTable->FindParticle(particleName="geantino");
65 if(!particle_definition)
66 {
67 G4String msg;
68 msg = " G4RayTracer uses geantino to trace the ray, but your physics list does not\n";
69 msg += "define G4Geantino. Please add G4Geantino in your physics list.";
70 G4Exception("G4RayShooter::Shoot","RayTracer001",FatalException,msg);
71 }
72 }
73
74 // create a new vertex
75 G4PrimaryVertex* vertex = new G4PrimaryVertex(vtx,particle_time);
76
77 // create new primaries and set them to the vertex
78 G4double mass = particle_definition->GetPDGMass();
79 G4PrimaryParticle* particle =
80 new G4PrimaryParticle(particle_definition);
81 particle->SetKineticEnergy( particle_energy );
82 particle->SetMass( mass );
83 particle->SetMomentumDirection( direc );
84 particle->SetPolarization(particle_polarization.x(),
85 particle_polarization.y(),
86 particle_polarization.z());
87 vertex->SetPrimary( particle );
88
89 evt->AddPrimaryVertex( vertex );
90}
