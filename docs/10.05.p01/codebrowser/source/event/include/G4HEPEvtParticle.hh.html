29// class desccription:
30//
31// This class is exclusively used by G4HEPEvtInterface.
32// It represents one particle in /HEPEVT/ list.
33
34#ifndef G4HEPEvtParticle_h
35#define G4HEPEvtParticle_h 1
36
37#include "globals.hh"
38#include "evtdefs.hh"
39#include "G4Allocator.hh"
40#include "G4PrimaryParticle.hh"
41
42class G4HEPEvtParticle
43{
44 public:
45 inline void *operator new(size_t);
46 inline void operator delete(void *aStackedTrack);
47
48 G4HEPEvtParticle();
49 G4HEPEvtParticle(G4PrimaryParticle* pp,
50 G4int isthep, G4int jdahep1, G4int jdahep2);
51 ~G4HEPEvtParticle();
52
53 G4HEPEvtParticle & operator=(const G4HEPEvtParticle &right);
54 G4bool operator==(const G4HEPEvtParticle &right) const;
55 G4bool operator!=(const G4HEPEvtParticle &right) const;
56
57 private:
58 G4PrimaryParticle * theParticle;
59 G4int ISTHEP; // Status code of the entry
60 // Set to be 0 after generating links of
61 // G4PrimaryParticle object
62 G4int JDAHEP1;
63 G4int JDAHEP2;
64
65 public:
66 inline G4PrimaryParticle * GetTheParticle()
67 { return theParticle; }
68 inline void Done()
69 { ISTHEP *= -1; }
70 inline G4int GetISTHEP()
71 { return ISTHEP; }
72 inline G4int GetJDAHEP1()
73 { return JDAHEP1; }
74 inline G4int GetJDAHEP2()
75 { return JDAHEP2; }
76};
77
78extern G4EVENT_DLL G4Allocator<G4HEPEvtParticle>*& aHEPEvtParticleAllocator();
79
80inline void * G4HEPEvtParticle::operator new(size_t)
81{
82 if (!aHEPEvtParticleAllocator())
83 aHEPEvtParticleAllocator() = new G4Allocator<G4HEPEvtParticle>;
84 return (void *) aHEPEvtParticleAllocator()->MallocSingle();
85}
86
87inline void G4HEPEvtParticle::operator delete(void * aHEPEvtParticle)
88{
89 aHEPEvtParticleAllocator()->FreeSingle((G4HEPEvtParticle *) aHEPEvtParticle);
90}
91
92#endif
