43 //----- G4GMocrenFile, constructor
44G4GMocrenFile::G4GMocrenFile ()
45 : G4VGraphicsSystem ("gMocrenFile", "gMocrenFile",
46 "A gMocren file driver (ver.4)",
47 G4VGraphicsSystem::fileWriter),
48 //GMOCRENFILE_FEATURES, G4VGraphicsSystem::threeD),
49 kViewer(NULL), kSceneHandler(NULL), kMessenger(new G4GMocrenMessenger()) {
50 ;
51}
52
53 //----- G4GMocrenFile, destructor
54G4GMocrenFile::~G4GMocrenFile () {
55 delete kMessenger;
56}
57
58
59 //----- G4GMocrenFile::CreateSceneHandler (const G4String& name)
60G4VSceneHandler* G4GMocrenFile::CreateSceneHandler (const G4String& name)
61{
62 kSceneHandler = new G4GMocrenFileSceneHandler (*this, *kMessenger, name);
63 return kSceneHandler;
64}
65
66 //----- G4GMocrenFile::CreateViewer ()
67G4VViewer* G4GMocrenFile::CreateViewer (G4VSceneHandler& scene,
68 const G4String& name)
69{
70 kViewer = new G4GMocrenFileViewer((G4GMocrenFileSceneHandler&) scene,
71 *kMessenger, name);
72 return kViewer;
73}
