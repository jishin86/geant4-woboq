31// G4GMocrenTouchable class
32// is used to get densities of each box cell of patient geometry
33// defined by using the nest parameterisation.
34//
35#ifndef G4GMOCRENTOUCHABLE_HH
36#define G4GMOCRENTOUCHABLE_HH
37
38#include "G4VTouchable.hh"
39
40class G4GMocrenTouchable : public G4VTouchable {
41
42 public: // with description
43
44 G4GMocrenTouchable() {;}
45 G4GMocrenTouchable(G4int & _depth0, G4int & _depth1);
46 virtual ~G4GMocrenTouchable() {;}
47 // Constructor and destructor.
48
49 virtual const G4ThreeVector& GetTranslation(G4int depth=0) const;
50 virtual const G4RotationMatrix* GetRotation(G4int depth=0) const;
51 // Accessors for translation and rotation.
52
53 virtual G4int GetReplicaNumber(G4int depth=0) const;
54 // Methods for touchables with history.
55
56 void SetReplicaNumber(G4int _depth0, G4int _depth1);
57
58private:
59 G4int repno[2];
60};
61
62inline
63G4GMocrenTouchable::G4GMocrenTouchable(G4int & _depth0, G4int & _depth1) {
64 repno[0] = _depth0;
65 repno[1] = _depth1;
66}
67
68const G4ThreeVector& G4GMocrenTouchable::GetTranslation(G4int depth) const {
69 // never used
70 // in the purpose to avoid a warning in the compile process
71 G4ThreeVector * vec = new G4ThreeVector();
72 *vec *= static_cast<G4double>(depth);
73 return *vec;
74}
75const G4RotationMatrix* G4GMocrenTouchable::GetRotation(G4int depth) const {
76 // never used
77 // in the puspose to avoid a warning in the compile process
78 G4RotationMatrix * rot = new G4RotationMatrix();
79 rot->setPhi(static_cast<G4double>(depth));
80 return rot;
81}
82inline
83G4int G4GMocrenTouchable::GetReplicaNumber(G4int depth) const {
84 if(depth > 1) {
85 G4Exception("G4GMocrenTouchable::GetReplicaNumber(G4int)", "gMocren0001",
86 FatalException, "depth number is less than 2.");
87 }
88 G4int rvalue;
89 if(depth < 2) rvalue = depth;
90 else rvalue = 0;
91 return rvalue;
92}

95#endif
