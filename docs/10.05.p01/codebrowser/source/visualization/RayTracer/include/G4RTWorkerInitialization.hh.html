27#ifndef G4RTWorkerInitialization_hh
28#define G4RTWorkerInitialization_hh
29
30class G4UserRunAction;
31class G4VUserPrimaryGeneratorAction;
32class G4UserEventAction;
33class G4UserStackingAction;
34class G4UserTrackingAction;
35class G4UserSteppingAction;
36class G4RTRunAction;
37class G4RTPrimaryGeneratorAction;
38class G4RTTrackingAction;
39class G4RTSteppingAction;
40
41#include "G4Threading.hh"
42#include "G4UserWorkerInitialization.hh"
43
44class G4RTWorkerInitialization : public G4UserWorkerInitialization
45{
46public: // with description
47 G4RTWorkerInitialization();
48 virtual ~G4RTWorkerInitialization();
49
50 virtual void WorkerRunStart() const;
51 // This method is called before an event loop. Geometry and physics have
52 // already been set up for the thread. All threads are synchronized and
53 // ready to start the local event loop. This situation is identical to
54 // "Idle" state in the sequential mode.
55
56 virtual void WorkerRunEnd() const;
57 // This method is called for each thread, when the local event loop has
58 // finished but before the synchronization over threads.
59
60private:
61 static G4ThreadLocal const G4UserRunAction * theUserRunAction;
62 static G4ThreadLocal const G4VUserPrimaryGeneratorAction * theUserPrimaryGeneratorAction;
63 static G4ThreadLocal const G4UserEventAction * theUserEventAction;
64 static G4ThreadLocal const G4UserStackingAction * theUserStackingAction;
65 static G4ThreadLocal const G4UserTrackingAction * theUserTrackingAction;
66 static G4ThreadLocal const G4UserSteppingAction * theUserSteppingAction;
67
68 static G4ThreadLocal G4RTRunAction * theRTRunAction;
69 static G4ThreadLocal G4RTPrimaryGeneratorAction * theRTPrimaryGeneratorAction;
70 static G4ThreadLocal G4RTTrackingAction * theRTTrackingAction;
71 static G4ThreadLocal G4RTSteppingAction * theRTSteppingAction;
72
73};
74
75#endif //G4RTWorkerInitialization_hh
