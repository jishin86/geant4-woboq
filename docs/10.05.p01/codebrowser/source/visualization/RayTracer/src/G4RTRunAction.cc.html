35#include "G4RTRunAction.hh"
36#include "G4RTRun.hh"
37#include "G4SDManager.hh"
38
39G4RTRunAction::G4RTRunAction(){;}
40G4RTRunAction::~G4RTRunAction(){;}
41
42G4Run* G4RTRunAction::GenerateRun()
43{ return new G4RTRun; }
44
45void G4RTRunAction::BeginOfRunAction(const G4Run*)
46{
47 G4SDManager* theSDMan = G4SDManager::GetSDMpointerIfExist();
48 if(theSDMan) theSDMan->Activate("/",false);
49}
50
51void G4RTRunAction::EndOfRunAction(const G4Run*)
52{
53 G4SDManager* theSDMan = G4SDManager::GetSDMpointerIfExist();
54 if(theSDMan) theSDMan->Activate("/",true);
55}
