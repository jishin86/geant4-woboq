150void G4HepRepFileSceneHandler::AddSolid(const G4Box& box) {
151#ifdef G4HEPREPFILEDEBUG
152 G4cout <<
153 "G4HepRepFileSceneHandler::AddSolid(const G4Box& box) called for "
154 << box.GetName()
155 << G4endl;
156 PrintThings();
157#endif
158
159 if (drawingTraj)
160 return;
161
162 if (drawingHit)
163 InitHit();
164
165 haveVisible = false;
166 AddHepRepInstance("Prism", NULL);
167
168 G4HepRepMessenger* messenger = G4HepRepMessenger::GetInstance();
169
170 // Get and check applicable vis attributes.
171 fpVisAttribs = fpViewer->GetApplicableVisAttributes(fpVisAttribs);
172 if (fpVisAttribs && (fpVisAttribs->IsVisible()==0) && messenger->getCullInvisibles())
173 return;
174
175 hepRepXMLWriter->addPrimitive();
176
177 G4double dx = box.GetXHalfLength();
178 G4double dy = box.GetYHalfLength();
179 G4double dz = box.GetZHalfLength();
180
181 G4Point3D vertex1(G4Point3D( dx, dy,-dz));
182 G4Point3D vertex2(G4Point3D( dx,-dy,-dz));
183 G4Point3D vertex3(G4Point3D(-dx,-dy,-dz));
184 G4Point3D vertex4(G4Point3D(-dx, dy,-dz));
185 G4Point3D vertex5(G4Point3D( dx, dy, dz));
186 G4Point3D vertex6(G4Point3D( dx,-dy, dz));
187 G4Point3D vertex7(G4Point3D(-dx,-dy, dz));
188 G4Point3D vertex8(G4Point3D(-dx, dy, dz));
189
190 vertex1 = (fObjectTransformation) * vertex1;
191 vertex2 = (fObjectTransformation) * vertex2;
192 vertex3 = (fObjectTransformation) * vertex3;
193 vertex4 = (fObjectTransformation) * vertex4;
194 vertex5 = (fObjectTransformation) * vertex5;
195 vertex6 = (fObjectTransformation) * vertex6;
196 vertex7 = (fObjectTransformation) * vertex7;
197 vertex8 = (fObjectTransformation) * vertex8;
198
199 hepRepXMLWriter->addPoint(vertex1.x(), vertex1.y(), vertex1.z());
200 hepRepXMLWriter->addPoint(vertex2.x(), vertex2.y(), vertex2.z());
201 hepRepXMLWriter->addPoint(vertex3.x(), vertex3.y(), vertex3.z());
202 hepRepXMLWriter->addPoint(vertex4.x(), vertex4.y(), vertex4.z());
203 hepRepXMLWriter->addPoint(vertex5.x(), vertex5.y(), vertex5.z());
204 hepRepXMLWriter->addPoint(vertex6.x(), vertex6.y(), vertex6.z());
205 hepRepXMLWriter->addPoint(vertex7.x(), vertex7.y(), vertex7.z());
206 hepRepXMLWriter->addPoint(vertex8.x(), vertex8.y(), vertex8.z());
207}

210void G4HepRepFileSceneHandler::AddSolid(const G4Cons& cons) {
211#ifdef G4HEPREPFILEDEBUG
212 G4cout <<
213 "G4HepRepFileSceneHandler::AddSolid(const G4Cons& cons) called for "
214 << cons.GetName()
215 << G4endl;
216 PrintThings();
217#endif
218
219 // HepRApp does not correctly represent the end faces of cones at
220 // non-standard angles, let the base class convert these solids to polygons.
221 G4RotationMatrix r = fObjectTransformation.getRotation();
222 G4bool linedUpWithAnAxis = (std::fabs(r.phiX())<=.001 ||
223 std::fabs(r.phiY())<=.001 ||
224 std::fabs(r.phiZ())<=.001 ||
225 std::fabs(r.phiX()-pi)<=.001 ||
226 std::fabs(r.phiY()-pi)<=.001 ||
227 std::fabs(r.phiZ()-pi)<=.001);
228 //G4cout << "Angle X:" << r.phiX() << ", Angle Y:" << r.phiY() << ", Angle Z:" << r.phiZ() << G4endl;
229 //G4cout << "linedUpWithAnAxis:" << linedUpWithAnAxis << G4endl;
230
231 // HepRep does not have a primitive for a cut cone,
232 // so if this cone is cut, let the base class convert this
233 // solid to polygons.
234 G4HepRepMessenger* messenger = G4HepRepMessenger::GetInstance();
235 if (cons.GetDeltaPhiAngle() < twopi || !linedUpWithAnAxis || messenger->renderCylAsPolygons())
236 {
237 G4VSceneHandler::AddSolid(cons); // Invoke default action.
238 } else {
239
240 if (drawingTraj)
241 return;
242
243 if (drawingHit)
244 InitHit();
245
246 haveVisible = false;
247 AddHepRepInstance("Cylinder", NULL);
248
249 // Get and check applicable vis attributes.
250 fpVisAttribs = fpViewer->GetApplicableVisAttributes(fpVisAttribs);
251 if (fpVisAttribs && (fpVisAttribs->IsVisible()==0) && messenger->getCullInvisibles())
252 return;
253
254 G4Point3D vertex1(G4Point3D( 0., 0., -cons.GetZHalfLength()));
255 G4Point3D vertex2(G4Point3D( 0., 0., cons.GetZHalfLength()));
256
257 vertex1 = (fObjectTransformation) * vertex1;
258 vertex2 = (fObjectTransformation) * vertex2;
259
260 // Outer cylinder.
261 hepRepXMLWriter->addPrimitive();
262 hepRepXMLWriter->addAttValue("Radius1",messenger->getScale() * cons.GetOuterRadiusMinusZ());
263 hepRepXMLWriter->addAttValue("Radius2",messenger->getScale() * cons.GetOuterRadiusPlusZ());
264 hepRepXMLWriter->addPoint(vertex1.x(), vertex1.y(), vertex1.z());
265 hepRepXMLWriter->addPoint(vertex2.x(), vertex2.y(), vertex2.z());
266
267 // Inner cylinder.
268 hepRepXMLWriter->addPrimitive();
269 hepRepXMLWriter->addAttValue("Radius1",messenger->getScale() * cons.GetInnerRadiusMinusZ());
270 hepRepXMLWriter->addAttValue("Radius2",messenger->getScale() * cons.GetInnerRadiusPlusZ());
271 hepRepXMLWriter->addPoint(vertex1.x(), vertex1.y(), vertex1.z());
272 hepRepXMLWriter->addPoint(vertex2.x(), vertex2.y(), vertex2.z());
273 }
274}

516void G4HepRepFileSceneHandler::AddCompound (const G4VTrajectory& traj) {
517#ifdef G4HEPREPFILEDEBUG
518 G4cout << "G4HepRepFileSceneHandler::AddCompound(const G4VTrajectory&) " << G4endl;
519#endif
520
521 G4TrajectoriesModel* pTrModel =
522 dynamic_cast<G4TrajectoriesModel*>(fpModel);
523 if (!pTrModel) G4Exception
524 ("G4HepRepFileSceneHandler::AddCompound(const G4VTrajectory&)",
525 "vis-HepRep0001", FatalException, "Not a G4TrajectoriesModel.");
526
527 // Pointers to hold trajectory attribute values and definitions.
528 std::vector<G4AttValue>* rawTrajAttValues = traj.CreateAttValues();
529 trajAttValues =
530 new std::vector<G4AttValue>;
531 trajAttDefs =
532 new std::map<G4String,G4AttDef>;
533
534 // Iterators to use with attribute values and definitions.
535 std::vector<G4AttValue>::iterator iAttVal;
536 std::map<G4String,G4AttDef>::const_iterator iAttDef;
537 G4int i;
538
539 // Get trajectory attributes and definitions in standard HepRep style
540 // (uniform units, 3Vectors decomposed).
541 if (rawTrajAttValues) {
542 G4bool error = G4AttCheck(rawTrajAttValues,
543 traj.GetAttDefs()).Standard(trajAttValues,trajAttDefs);
544 if (error) {
545 G4cout << "G4HepRepFileSceneHandler::AddCompound(traj):"
546 "\nERROR found during conversion to standard trajectory attributes."
547 << G4endl;
548 }
549#ifdef G4HEPREPFILEDEBUG
550 G4cout <<
551 "G4HepRepFileSceneHandler::AddCompound(traj): standardised attributes:\n"
552 << G4AttCheck(trajAttValues,trajAttDefs) << G4endl;
553#endif
554 delete rawTrajAttValues;
555 }
556
557 // Open the HepRep output file if it is not already open.
558 CheckFileOpen();
559
560 // Add the Event Data Type if it hasn't already been added.
561 if (strcmp("Event Data",hepRepXMLWriter->prevTypeName[0])!=0) {
562 hepRepXMLWriter->addType("Event Data",0);
563 hepRepXMLWriter->addInstance();
564 }
565
566 // Add the Trajectories Type.
567 G4String previousName = hepRepXMLWriter->prevTypeName[1];
568 hepRepXMLWriter->addType("Trajectories",1);
569
570 // If this is the first trajectory of this event,
571 // specify attribute values common to all trajectories.
572 if (strcmp("Trajectories",previousName)!=0) {
573 hepRepXMLWriter->addAttValue("Layer",100);
574
575 // Take all Trajectory attDefs from first trajectory.
576 // Would rather be able to get these attDefs without needing a reference from any
577 // particular trajectory, but don't know how to do that.
578 // Write out trajectory attribute definitions.
579 if (trajAttValues && trajAttDefs) {
580 for (iAttVal = trajAttValues->begin();
581 iAttVal != trajAttValues->end(); ++iAttVal) {
582 iAttDef = trajAttDefs->find(iAttVal->GetName());
583 if (iAttDef != trajAttDefs->end()) {
584 // Protect against incorrect use of Category. Anything value other than the
585 // standard ones will be considered to be in the physics category.
586 G4String category = iAttDef->second.GetCategory();
587 if (strcmp(category,"Draw")!=0 &&
588 strcmp(category,"Physics")!=0 &&
589 strcmp(category,"Association")!=0 &&
590 strcmp(category,"PickAction")!=0)
591 category = "Physics";
592 hepRepXMLWriter->addAttDef(iAttVal->GetName(), iAttDef->second.GetDesc(),
593 category, iAttDef->second.GetExtra());
594 }
595 }
596 }
597
598 // Take all TrajectoryPoint attDefs from first point of first trajectory.
599 // Would rather be able to get these attDefs without needing a reference from any
600 // particular point, but don't know how to do that.
601 if ((trajContext->GetDrawStepPts() || trajContext->GetDrawAuxPts())
602 && traj.GetPointEntries()>0) {
603 G4VTrajectoryPoint* aTrajectoryPoint = traj.GetPoint(0);
604
605 // Pointers to hold trajectory point attribute values and definitions.
606 std::vector<G4AttValue>* rawPointAttValues = aTrajectoryPoint->CreateAttValues();
607 std::vector<G4AttValue>* pointAttValues =
608 new std::vector<G4AttValue>;
609 std::map<G4String,G4AttDef>* pointAttDefs =
610 new std::map<G4String,G4AttDef>;
611
612 // Get first trajectory point's attributes and definitions in standard HepRep style
613 // (uniform units, 3Vectors decomposed).
614 if (rawPointAttValues) {

627 pointAttDefs->find(iAttVal->GetName());
628 if (iAttDef != pointAttDefs->end()) {

1309void G4HepRepFileSceneHandler::AddHepRepInstance(const char* primName,
1310 const G4Visible visible) {
1311#ifdef G4HEPREPFILEDEBUG
1312 G4cout <<
1313 "G4HepRepFileSceneHandler::AddHepRepInstance called."
1314 << G4endl;
1315#endif
1316
1317 // Open the HepRep output file if it is not already open.
1318 CheckFileOpen();
1319
1320 G4VPhysicalVolume* pCurrentPV = 0;
1321 G4LogicalVolume* pCurrentLV = 0;
1322 G4int currentDepth = 0;
1323 G4PhysicalVolumeModel* pPVModel = dynamic_cast<G4PhysicalVolumeModel*>(fpModel);
1324 if (pPVModel) {
1325 pCurrentPV = pPVModel->GetCurrentPV();
1326 pCurrentLV = pPVModel->GetCurrentLV();
1327 currentDepth = pPVModel->GetCurrentDepth();
1328 }
1329
1330#ifdef G4HEPREPFILEDEBUG
1331 G4cout <<
1332 "pCurrentPV:" << pCurrentPV << ", readyForTransients:" << fReadyForTransients
1333 << G4endl;
1334#endif
1335
1336 if (drawingTraj || drawingHit) {
1337 // In this case, HepRep type, layer and instance were already created
1338 // in the AddCompound method.
1339 }
1340 else if (fReadyForTransients) {
1341 if (strcmp("Event Data",hepRepXMLWriter->prevTypeName[0])!=0) {
1342 hepRepXMLWriter->addType("Event Data",0);
1343 hepRepXMLWriter->addInstance();
1344 }
1345
1346 // Applications have the option of either calling AddSolid(G4VTrajectory&) and
1347 // AddSolid(G4VHits&), or of just decomposing these into simpler primitives.
1348 // In the former case, drawing will be handled above and will include setting of
1349 // physics attributes.
1350 // In the latter case, which is an older style of working, we end up drawing the
1351 // trajectories and hits here, where we have no access to physics attributes.
1352 // We receive primitives here. We can figure out that these are transients, but we
1353 // have to guess exactly what these transients represent.
1354 // We assume the primitives are being used as in G4VTrajectory, hence we assume:
1355 // Lines are Trajectories
1356 // Squares that come after we've seen trajectories are Auxiliary Points
1357 // Circles that come after we've seen trajectories are Step Points
1358 // Other primitives are Hits
1359
1360 int layer;
1361
1362 if (strcmp("Text",primName)==0) {
1363 hepRepXMLWriter->addType("EventID",1);
1364 } else {
1365 if (strcmp("Line",primName)==0) {
1366 hepRepXMLWriter->addType("TransientPolylines",1);
1367 layer = 100;
1368 } else {
1369 if (strcmp(hepRepXMLWriter->prevTypeName[1],"TransientPolylines")==0 &&
1370 strcmp("Square",primName)==0)
1371 {
1372 hepRepXMLWriter->addType("AuxiliaryPoints",2);
1373 layer = 110;
1374 } else {
1375 if (strcmp(hepRepXMLWriter->prevTypeName[1],"TransientPolylines")==0 &&
1376 strcmp("Circle",primName)==0)
1377 {
1378 hepRepXMLWriter->addType("StepPoints",2);
1379 layer = 120;
1380 } else {
1381 hepRepXMLWriter->addType("Hits",1);
1382 layer = 130;
1383 }
1384 }
1385 }
1386 hepRepXMLWriter->addAttValue("Layer",layer);
1387 }
1388
1389 hepRepXMLWriter->addInstance();
1390
1391 // Handle Type declaration for Axes, Ruler, etc.
1392 } else if (pCurrentPV==0) {
1393 if (strcmp("AxesEtc",hepRepXMLWriter->prevTypeName[0])!=0) {
1394 hepRepXMLWriter->addType("AxesEtc",0);
1395 hepRepXMLWriter->addInstance();
1396 }
1397
1398 int layer;
1399
1400 if (strcmp("Text",primName)==0) {
1401 hepRepXMLWriter->addType("Text",1);
1402 } else {
1403 if (strcmp("Line",primName)==0) {
1404 hepRepXMLWriter->addType("Polylines",1);
1405 layer = 100;
1406 } else {
1407 hepRepXMLWriter->addType("Points",1);
1408 layer = 130;
1409 }
1410 hepRepXMLWriter->addAttValue("Layer",layer);
1411 }
1412
1413 hepRepXMLWriter->addInstance();
1414
1415 // Handle Type declaration for Detector Geometry,
1416 // replacing G4's top geometry level name "worldPhysical" with the
1417 // name "Detector Geometry".
1418 } else {
1419 //G4cout << "CurrentDepth" << currentDepth << G4endl;
1420 //G4cout << "currentName" << pCurrentPV->GetName() << G4endl;
1421 if (strcmp("Detector Geometry",hepRepXMLWriter->prevTypeName[0])!=0) {
1422 //G4cout << "Adding Det Geom type" << G4endl;
1423 hepRepXMLWriter->addType("Detector Geometry",0);
1424 hepRepXMLWriter->addInstance();
1425 }
1426
1427 // Re-insert any layers of the hierarchy that were removed by G4's culling process.
1428 // Don't bother checking if same type name as last instance.
1429 if(strcmp(hepRepXMLWriter->prevTypeName[currentDepth+1],pCurrentPV->GetName())!=0) {
1430 //G4cout << "Looking for mother of:" << pCurrentLV->GetName() << G4endl;
1431 typedef G4PhysicalVolumeModel::G4PhysicalVolumeNodeID PVNodeID;
1432 typedef std::vector<PVNodeID> PVPath;
1433 const PVPath& drawnPVPath = pPVModel->GetDrawnPVPath();
1434 PVPath::const_reverse_iterator ri = ++drawnPVPath.rbegin();
1435 G4int drawnMotherDepth;
1436 if (ri != drawnPVPath.rend()) {
1437 // This volume has a mother.
1438 drawnMotherDepth = ri->GetNonCulledDepth();
1439 //G4cout << "drawnMotherDepth" << drawnMotherDepth << G4endl;
1440 } else {
1441 // This volume has no mother. Must be a top level volume.
1442 drawnMotherDepth = -1;
1443 //G4cout << "Mother must be very top" << G4endl;
1444 }
1445
1446 while (drawnMotherDepth < (currentDepth-1)) {
1447 G4String culledParentName = "Culled parent of " + pCurrentPV->GetName();
1448 //G4cout << "Inserting culled layer " << culledParentName << " at depth:" << drawnMotherDepth+2 << G4endl;
1449 hepRepXMLWriter->addType(culledParentName, drawnMotherDepth+2);
1450 hepRepXMLWriter->addInstance();
1451 drawnMotherDepth ++;
1452 }
1453 }
1454
1455 // Add the HepRepType for the current volume.
1456 hepRepXMLWriter->addType(pCurrentPV->GetName(),currentDepth+1);
1457 hepRepXMLWriter->addInstance();
1458
1459 G4HepRepMessenger* messenger = G4HepRepMessenger::GetInstance();
1460
1461 if (fpVisAttribs && (fpVisAttribs->IsVisible()==0) && messenger->getCullInvisibles())
1462 return;
1463
1464 // Additional attributes.
1465 hepRepXMLWriter->addAttValue("Layer",hepRepXMLWriter->typeDepth);
1466 hepRepXMLWriter->addAttValue("LVol", pCurrentLV->GetName());
1467 G4Region* region = pCurrentLV->GetRegion();
1468 G4String regionName = region? region->GetName(): G4String("No region");
1469 hepRepXMLWriter->addAttValue("Region", regionName);
1470 hepRepXMLWriter->addAttValue("RootRegion", pCurrentLV->IsRootRegion());
1471 hepRepXMLWriter->addAttValue("Solid", pCurrentLV->GetSolid()->GetName());
1472 hepRepXMLWriter->addAttValue("EType", pCurrentLV->GetSolid()->GetEntityType());
1473 G4Material * material = pPVModel->GetCurrentMaterial();
1474 G4String matName = material? material->GetName(): G4String("No material");
1475 hepRepXMLWriter->addAttValue("Material", matName);
1476 G4double matDensity = material? material->GetDensity(): 0.;
1477 hepRepXMLWriter->addAttValue("Density", matDensity*m3/kg);
1478 G4State matState = material? material->GetState(): kStateUndefined;
1479 hepRepXMLWriter->addAttValue("State", matState);
1480 G4double matRadlen = material? material->GetRadlen(): 0.;
1481 hepRepXMLWriter->addAttValue("Radlen", matRadlen/m);
1482 }
1483
1484 hepRepXMLWriter->addAttValue("DrawAs",primName);
1485
1486 // Handle color and visibility attributes.
1487 float redness;
1488 float greenness;
1489 float blueness;
1490 G4bool isVisible;
1491
1492 if (fpVisAttribs || haveVisible) {
1493 G4Colour colour;
1494
1495 if (fpVisAttribs) {
1496 colour = fpVisAttribs->GetColour();
1497 isVisible = fpVisAttribs->IsVisible();
1498 } else {
1499 colour = visible.GetVisAttributes()->GetColour();
1500 isVisible = fpViewer->
1501 GetApplicableVisAttributes(visible.GetVisAttributes())->IsVisible();
1502 }
1503
1504 redness = colour.GetRed();
1505 greenness = colour.GetGreen();
1506 blueness = colour.GetBlue();
1507
1508 // Avoiding drawing anything black on black.
1509 if (redness==0. && greenness==0. && blueness==0.) {
1510 redness = 1.;
1511 greenness = 1.;
1512 blueness = 1.;
1513 }
1514 } else {
1515#ifdef G4HEPREPFILEDEBUG
1516 G4cout <<
1517 "G4HepRepFileSceneHandler::AddHepRepInstance using default colour."
1518 << G4endl;
1519#endif
1520 redness = 1.;
1521 greenness = 1.;
1522 blueness = 1.;
1523 isVisible = true;
1524 }
1525
1526 if (strcmp(primName,"Point")==0)
1527 hepRepXMLWriter->addAttValue("MarkColor",redness,greenness,blueness);
1528 else
1529 hepRepXMLWriter->addAttValue("LineColor",redness,greenness,blueness);
1530
1531 hepRepXMLWriter->addAttValue("Visibility",isVisible);
1532}
