131void G4VRML2SCENEHANDLER::AddPrimitive(const G4Polyline& polyline)
132{
133#if defined DEBUG_SCENE_FUNC
134 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
135 G4cout << "***** AddPrimitive polyline" << "\n" ;
136#endif
137
138 if (fProcessing2D) {
139 static G4bool warned = false;
140 if (!warned) {
141 warned = true;
142 G4Exception
143 ("G4VRML2SCENEHANDLER::AddPrimitive (const G4Polyline&)",
144 "VRML-2001", JustWarning,
145 "2D polylines not implemented. Ignored.");
146 }
147 return;
148 }
149
150 VRMLBeginModeling () ;
151
152 // VRML codes are generated below
153
154 fDest << "#---------- POLYLINE" << "\n";
155 fDest << "Shape {" << "\n";
156
157 SendLineColor ( polyline.GetVisAttributes() );
158
159 fDest << "\t" << "geometry IndexedLineSet {" << "\n";
160
161 fDest << "\t\t" << "coord Coordinate {" << "\n";
162 fDest << "\t\t\t" << "point [" << "\n";
163 G4int e, i;
164 for (i = 0, e = polyline.size(); e; i++, e--) {
165
166 G4Point3D point = polyline[i];
167 point.transform( fObjectTransformation );
168
169 fDest << "\t\t\t\t";
170 fDest << point.x() << " ";
171 fDest << point.y() << " ";
172 fDest << point.z() << "," << "\n";
173
174 } // for
175
176 fDest << "\t\t\t" << "]" << "\n"; // point
177 fDest << "\t\t" << "}" << "\n"; // coord
178
179 fDest << "\t\t" << "coordIndex [";
180 for (i = 0, e = polyline.size(); e; i++, e--) {
181 if (i % 10 == 0)
182 fDest << "\n" << "\t\t\t";
183 fDest << i << ", ";
184 }
185 fDest << "-1" << "\n";
186 fDest << "\t\t" << "]" << "\n"; // coordIndex
187 fDest << "\t" << "}" << "\n"; // geometry IndexedLineSet
188
189 fDest << "}" << "\n"; // Shape
190}

193void G4VRML2SCENEHANDLER::AddPrimitive(const G4Polyhedron& polyhedron)
194{
195#if defined DEBUG_SCENE_FUNC
196 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
197 G4cout << "***** AddPrimitive(G4Polyhedron)" << "\n";
198#endif
199
200 if (polyhedron.GetNoFacets() == 0) return;
201
202 if (fProcessing2D) {
203 static G4bool warned = false;
204 if (!warned) {
205 warned = true;
206 G4Exception
207 ("G4VRML2SCENEHANDLER::AddPrimitive (const G4Polyhedron&)",
208 "VRML-2002", JustWarning,
209 "2D polyhedra not implemented. Ignored.");
210 }
211 return;
212 }
213
214 VRMLBeginModeling () ;
215
216 // Transparency checking: If completely transparent, skip drawing
217 if ( GetPVTransparency() > 0.99 ) { return ; }
218
219 // Current Model
220 const G4VModel* pv_model = GetModel();
221 G4String pv_name = "No model";
222 if (pv_model) pv_name = pv_model->GetCurrentTag() ;
223
224 // VRML codes are generated below
225
226 fDest << "#---------- SOLID: " << pv_name << "\n";
227
228 if ( IsPVPickable() ) {
229
230 fDest << "Anchor {" << "\n";
231 fDest << " description " << "\"" << pv_name << "\"" << "\n";
232 fDest << " url \"\" " << "\n";
233 fDest << " children [" << "\n";
234 }
235
236 fDest << "\t"; fDest << "Shape {" << "\n";
237
238 SendMaterialNode();
239
240 fDest << "\t\t" << "geometry IndexedFaceSet {" << "\n";
241
242 fDest << "\t\t\t" << "coord Coordinate {" << "\n";
243 fDest << "\t\t\t\t" << "point [" << "\n";
244 G4int i, j;
245 for (i = 1, j = polyhedron.GetNoVertices(); j; j--, i++) {
246 G4Point3D point = polyhedron.GetVertex(i);
247
248 point.transform( fObjectTransformation );
249
250 fDest << "\t\t\t\t\t";
251 fDest << point.x() << " ";
252 fDest << point.y() << " ";
253 fDest << point.z() << "," << "\n";
254 }
255 fDest << "\t\t\t\t" << "]" << "\n"; // point
256 fDest << "\t\t\t" << "}" << "\n"; // coord
257
258 fDest << "\t\t\t" << "coordIndex [" << "\n";
259
260 // facet loop
261 G4int f;
262 for (f = polyhedron.GetNoFacets(); f; f--) {
263
264 // edge loop
265 G4bool notLastEdge;
266 G4int index = -1, edgeFlag = 1;
267 fDest << "\t\t\t\t";
268 do {
269 notLastEdge = polyhedron.GetNextVertexIndex(index, edgeFlag);
270 fDest << index - 1 << ", ";
271 } while (notLastEdge);
272 fDest << "-1," << "\n";
273 }
274 fDest << "\t\t\t" << "]" << "\n"; // coordIndex
275
276 fDest << "\t\t\t" << "solid FALSE" << "\n"; // draw backfaces
277
278 fDest << "\t\t" << "}" << "\n"; // IndexFaceSet
279 fDest << "\t" << "}" << "\n"; // Shape
280
281 if ( IsPVPickable() ) {
282 fDest << " ]" << "\n"; // children
283 fDest << "}" << "\n"; // Anchor
284 }
285
286}

319{
320#if defined DEBUG_SCENE_FUNC
321 if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
322 G4cout << "***** AddPrimitive circle" << "\n" ;
323#endif
324
325 if (fProcessing2D) {
326 static G4bool warned = false;
327 if (!warned) {
328 warned = true;
329 G4Exception
330 ("G4VRML2SCENEHANDLER::AddPrimitive (const G4Circle&)",
331 "VRML-2004", JustWarning,
332 "2D circles not implemented. Ignored.");
333 }
334 return;
335 }
336
337 VRMLBeginModeling () ;
338
339 // Information
340// const char* info = NULL ;
341 const char* info = circle.GetInfo() ;
342
343 // Position and half size of circle
344 G4double X, Y, Z ; GetMarkerWorldPosition( circle, &X, &Y, &Z ) ;
345 G4double radius = GetMarkerHalfSize ( circle );
346
347 // VRML codes are generated below
348 fDest << "#---------- 3D MARKER (Circle)" << "\n";
349 fDest << "Anchor {" << "\n";
350
351 if( circle.GetInfo().isNull() ) {
352 fDest << " description " << "\"(" << X << " " ;
353 fDest << Y << " " ;
354 fDest << Z << ")\"" << "\n";
355 } else {
356 fDest << " description " << "\"" << info << "\"" << "\n";
357 }
358
359
360 fDest << " url \"\" " << "\n";
361
362 fDest << " children [" << "\n";
363
364 fDest << " Transform {" << "\n";
365
366 fDest << " translation ";
367 fDest << X << " " << Y << " " << Z << "\n";
368
369 fDest << " children [" << "\n";
370
371 fDest << "\t" << "Shape {" << "\n";
372
373 SendMarkerColor( circle );
374
375 // send shape with size
376 fDest << "\t\t" << "geometry Sphere {" << "\n";
377 fDest << "\t\t\t" << "radius " << radius << "\n";
378 fDest << "\t\t" << "}" << "\n";
379
380 fDest << "\t" << "}" << "\n"; // Shape
381
382 fDest << " ]" << "\n"; // children
383 fDest << " }" << "\n"; // Transform
384
385 fDest << " ]" << "\n"; // children

502void G4VRML2SCENEHANDLER::SendMaterialNode( const G4VisAttributes* pVA )
503{
504 // Initialization of drawing parameters
505 G4bool flagWF = false ;
506 G4double TRANSPARENCY = GetPVTransparency() ;
507
508 // Wireframe or not
509 const G4ViewParameters& cur_view_param = fpViewer -> GetViewParameters ();
510 if (pVA) {
511 if (pVA->IsForceDrawingStyle() &&
512 (pVA->GetForcedDrawingStyle() == G4VisAttributes::wireframe))
513 {
514 // wireframe style is assigned to this object
515 flagWF = true ;
516
517 } else if ( cur_view_param.GetDrawingStyle() == G4ViewParameters::wireframe )
518 {
519 // current viewing style is wireframe
520 flagWF = true ;
521 }
522 }
523
524 // VRML codes are generated below
525 if (!pVA) {
526
527 // Error recovery: Use default values of VRML browser
528 fDest << "\t\t" ;
529 fDest << "appearance Appearance { material Material {} }";
530 fDest << "\n";
531
532 } else {
533
534 const G4Color& color = pVA->GetColor();
535
536 fDest << "\t\t" << "appearance Appearance {" << "\n";
537 fDest << "\t\t\t" << "material Material {" << "\n";
538
539 // diffuse color
540 fDest << "\t\t\t\t" << "diffuseColor ";
541 fDest << color.GetRed() << " " ;
542 fDest << color.GetGreen() << " " ;
543 fDest << color.GetBlue() << "\n";
544
545 // emmisive color
546 // fDest << "\t\t\t\t" << "emissiveColor ";
547 // fDest << color.GetRed() << " " ;
548 // fDest << color.GetGreen() << " " ;
549 //fDest << color.GetBlue() << "\n";
550 //
551
552 // wireframe is expressed as transparency
553 if ( flagWF )
554 {
555 fDest << "\t\t\t\t" << "transparency " << TRANSPARENCY << "\n";
556 }
557
558 fDest << "\t\t\t" << "}" << "\n"; // material
559 fDest << "\t\t" << "}" << "\n"; // appearance
560 }
561
562}

643G4double G4VRML2SCENEHANDLER::GetMarkerHalfSize ( const G4VMarker& mark )
644{
645 //----- return value ( marker radius in 3d units)
646 G4double size = 1.0 ; // initialization
647
648 //----- parameters to calculate 3d size from 2d size
649 const double HALF_SCREEN_SIZE_2D = 300.0 ; // pixels
650 double zoom_factor = fpViewer->GetViewParameters().GetZoomFactor() ;
651 if ( zoom_factor <= 0.0 ) { zoom_factor = 1.0 ; }
652 double extent_radius_3d = GetScene()->GetExtent().GetExtentRadius() ;
653 if ( extent_radius_3d <= 0.0 ) { extent_radius_3d = 1.0 ; }
654
655 //----- get marker radius in 3D units
656 if ( mark.GetWorldSize() > 0.0 ) {
657
658 // get mark radius in 3D units
659 size = 0.5 * mark.GetWorldSize() ;
660
661 } else if ( mark.GetScreenSize() > 0.0 ) {
662
663 // local
664 double mark_radius_2d = 0.5 * mark.GetScreenSize() ;
665
666 // get mark radius in 3D units
667 size \
668 = extent_radius_3d * ( mark_radius_2d / HALF_SCREEN_SIZE_2D );
669 size *= zoom_factor ;
670
671 } else {
672 // local
673 double mark_radius_2d \
674 = fpViewer->GetViewParameters().GetDefaultMarker().GetScreenSize();
675 mark_radius_2d *= 0.1 ; // Magic number?
676
677 // get mark radius in 3D units
678 size \
679 = extent_radius_3d * ( mark_radius_2d / HALF_SCREEN_SIZE_2D );
680 size *= zoom_factor ;
681 }
682
683 //----- global rescaling
684 size *= fpViewer->GetViewParameters().GetGlobalMarkerScale();
685
686 //----- return size
687 return size ;
688
689}

710G4double G4VRML2SCENEHANDLER::SetPVTransparency ()
711{
712
713 G4double TRANSPARENCY = 0.7 ; // initialization
714
715 if( getenv( "G4VRML_TRANSPARENCY" ) != NULL ) {
716
717 // get a value from the env value
718// sscanf( getenv("G4VRML_TRANSPARENCY"), "%lg", &TRANSPARENCY ) ;
719 std::istringstream iss( getenv("G4VRML_TRANSPARENCY") );
720 iss >> TRANSPARENCY;
721
722 // truncation
