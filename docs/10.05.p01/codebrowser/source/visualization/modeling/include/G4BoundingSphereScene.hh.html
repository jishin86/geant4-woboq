30// An artificial scene to reuse G4VScene code to calculate a bounding sphere.
31
32#ifndef G4BOUNDINGSPHERESCENE_HH
33#define G4BOUNDINGSPHERESCENE_HH
34
35#include "G4PseudoScene.hh"
36#include "G4VisExtent.hh"
37
38class G4VModel;
39
40class G4BoundingSphereScene: public G4PseudoScene {
41
42public:
43
44 G4BoundingSphereScene (G4VModel* pModel = 0);
45
46 virtual ~G4BoundingSphereScene ();
47
48 G4VisExtent GetBoundingSphereExtent ();
49
50 const G4Point3D& GetCentre() const {return fCentre;}
51
52 G4double GetRadius() const {return fRadius;}
53
54 void SetCentre(const G4Point3D& centre) {fCentre = centre;}
55
56 ////////////////////////////////////////////////////////////////
57 // The following 2 functions can be used by any code which wishes to
58 // accrue a bounding sphere. Just instantiate a
59 // G4BoundingSphereScene and use AccrueBoundingSphere.
60
61 void ResetBoundingSphere ();
62 void AccrueBoundingSphere (const G4Point3D& centre,
63 G4double radius);
64
65private:
66
67 void ProcessVolume (const G4VSolid& solid);
68
69 G4VModel* fpModel; // Instantiating code may optionally set this.
70 G4Point3D fCentre;
71 G4double fRadius;
72};
73
74#endif
