29// Class Description:
30// Abstract base class for trajectory drawing models. Trajectory drawing
31// models are responsible for drawing individual trajectories according
32// to a particular style.
33// Class Description - End:
34
35#ifndef G4VTRAJECTORYMODEL_HH
36#define G4VTRAJECTORYMODEL_HH
37
38#include "G4String.hh"
39#include "G4VTrajectory.hh"
40
41class G4VisTrajContext;
42
43class G4VTrajectoryModel {
44
45public:
46
47 // Construct with context object
48 G4VTrajectoryModel(const G4String& name, G4VisTrajContext* fpContext=0);
49
50 // Destructor
51 virtual ~G4VTrajectoryModel();
52
53 // Draw method
54 virtual void Draw(const G4VTrajectory& trajectory,
55 const G4bool& visible = true) const = 0;
56
57 // Print configuration
58 virtual void Print(std::ostream& ostr) const = 0;
59
60 // Accessors
61 G4String Name() const ;
62 const G4VisTrajContext& GetContext() const;
63
64 // Set verbosity
65 void SetVerbose(const G4bool&);
66 G4bool GetVerbose() const;
67
68private:
69
70 // Private copy constructor and assigment operator - copying and
71 // assignment not allowed. Keeps Coverity happy.
72 G4VTrajectoryModel (const G4VTrajectoryModel&);
73 G4VTrajectoryModel& operator = (const G4VTrajectoryModel&);
74
75 G4String fName;
76 G4bool fVerbose;
77 G4VisTrajContext* fpContext;
78
79};
80
81#endif
