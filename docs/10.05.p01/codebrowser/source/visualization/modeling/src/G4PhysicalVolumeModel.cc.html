57G4PhysicalVolumeModel::G4PhysicalVolumeModel
58(G4VPhysicalVolume* pVPV
59 , G4int requestedDepth
60 , const G4Transform3D& modelTransformation
61 , const G4ModelingParameters* pMP
62 , G4bool useFullExtent
63 , const std::vector<G4PhysicalVolumeNodeID>& baseFullPVPath)
64: G4VModel (modelTransformation,pMP)
65, fpTopPV (pVPV)
66, fTopPVCopyNo (0)
67, fRequestedDepth (requestedDepth)
68, fUseFullExtent (useFullExtent)
69, fCurrentDepth (0)
70, fpCurrentPV (fpTopPV)
71, fCurrentPVCopyNo (fpTopPV? fpTopPV->GetCopyNo(): 0)
72, fpCurrentLV (fpTopPV? fpTopPV->GetLogicalVolume(): 0)
73, fpCurrentMaterial (fpCurrentLV? fpCurrentLV->GetMaterial(): 0)
74, fpCurrentTransform (const_cast<G4Transform3D*>(&modelTransformation))
75, fBaseFullPVPath (baseFullPVPath)
76, fAbort (false)
77, fCurtailDescent (false)
78, fpClippingSolid (0)
79, fClippingMode (subtraction)
80{
81 fType = "G4PhysicalVolumeModel";
82
83 if (!fpTopPV) {
84
85 // In some circumstances creating an "empty" G4PhysicalVolumeModel is
86 // allowed, so I have supressed the G4Exception below. If it proves to
87 // be a problem we might have to re-instate it, but it is unlikley to
88 // be used except by visualisation experts. See, for example, /vis/list,
89 // where it is used simply to get a list of G4AttDefs.
90 // G4Exception
91 // ("G4PhysicalVolumeModel::G4PhysicalVolumeModel",
92 // "modeling0010", FatalException, "Null G4PhysicalVolumeModel pointer.");
93
94 fTopPVName = "NULL";
95 fGlobalTag = "Empty";
96 fGlobalDescription = "G4PhysicalVolumeModel " + fGlobalTag;
97
98 } else {
99
100 fTopPVName = fpTopPV -> GetName ();
101 std::ostringstream oss;
102 oss << fpTopPV->GetName() << ':' << fpTopPV->GetCopyNo()
103 << " BasePath:" << fBaseFullPVPath;
104 fGlobalTag = oss.str();
105 fGlobalDescription = "G4PhysicalVolumeModel " + fGlobalTag;
106 CalculateExtent ();
107 }
108}

127void G4PhysicalVolumeModel::CalculateExtent ()
128{
129 if (fUseFullExtent) {
130 fExtent = fpTopPV -> GetLogicalVolume () -> GetSolid () -> GetExtent ();
131 }
132 else {
133 G4BoundingSphereScene bsScene(this);
134 const G4int tempRequestedDepth = fRequestedDepth;
135 fRequestedDepth = -1; // Always search to all depths to define extent.
136 const G4ModelingParameters* tempMP = fpMP;
137 G4ModelingParameters mParams
138 (0, // No default vis attributes needed.
139 G4ModelingParameters::wf, // wireframe (not relevant for this).
140 true, // Global culling.
141 true, // Cull invisible volumes.
142 false, // Density culling.
143 0., // Density (not relevant if density culling false).
144 true, // Cull daughters of opaque mothers.
145 24); // No of sides (not relevant for this operation).
146 fpMP = &mParams;
147 DescribeYourselfTo (bsScene);
148 G4double radius = bsScene.GetRadius();
149 if (radius < 0.) { // Nothing in the scene.
150 fExtent = fpTopPV -> GetLogicalVolume () -> GetSolid () -> GetExtent ();
151 } else {
152 // Transform back to coordinates relative to the top
153 // transformation, which is in G4VModel::fTransform. This makes
154 // it conform to all models, which are defined by a
155 // transformation and an extent relative to that
156 // transformation...
157 G4Point3D centre = bsScene.GetCentre();
158 centre.transform(fTransform.inverse());
159 fExtent = G4VisExtent(centre, radius);
160 }
161 fpMP = tempMP;
162 fRequestedDepth = tempRequestedDepth;
163 }
164}

166void G4PhysicalVolumeModel::DescribeYourselfTo
167(G4VGraphicsScene& sceneHandler)
168{
169 if (!fpTopPV) G4Exception
170 ("G4PhysicalVolumeModel::DescribeYourselfTo",
171 "modeling0012", FatalException, "No model.");
172
173 if (!fpMP) G4Exception
174 ("G4PhysicalVolumeModel::DescribeYourselfTo",
175 "modeling0003", FatalException, "No modeling parameters.");
176
177 G4Transform3D startingTransformation = fTransform;
178
179 VisitGeometryAndGetVisReps
180 (fpTopPV,
181 fRequestedDepth,
182 startingTransformation,
183 sceneHandler);
184
185 // Reset or clear data...
186 fCurrentDepth = 0;
187 fpCurrentPV = fpTopPV;
188 fCurrentPVCopyNo = fpTopPV->GetCopyNo();
189 fpCurrentLV = fpTopPV->GetLogicalVolume();
190 fpCurrentMaterial = fpCurrentLV? fpCurrentLV->GetMaterial(): 0;
191 fFullPVPath = fBaseFullPVPath;
192 fDrawnPVPath.clear();
193 fAbort = false;
194 fCurtailDescent = false;
195}

214void G4PhysicalVolumeModel::VisitGeometryAndGetVisReps
215(G4VPhysicalVolume* pVPV,
216 G4int requestedDepth,
217 const G4Transform3D& theAT,
218 G4VGraphicsScene& sceneHandler)
219{
220 // Visits geometry structure to a given depth (requestedDepth), starting
221 // at given physical volume with given starting transformation and
222 // describes volumes to the scene handler.
223 // requestedDepth < 0 (default) implies full visit.
224 // theAT is the Accumulated Transformation.
225
226 // Find corresponding logical volume and (later) solid, storing in
227 // local variables to preserve re-entrancy.
228 G4LogicalVolume* pLV = pVPV -> GetLogicalVolume ();
229
230 G4VSolid* pSol = nullptr;
231 G4Material* pMaterial = nullptr;
232
233 if (!(pVPV -> IsReplicated ())) {
234 // Non-replicated physical volume.
235 pSol = pLV -> GetSolid ();
236 pMaterial = pLV -> GetMaterial ();
237 DescribeAndDescend (pVPV, requestedDepth, pLV, pSol, pMaterial,
238 theAT, sceneHandler);
239 }
240 else {
241 // Replicated or parametrised physical volume.
242 EAxis axis;
243 G4int nReplicas;
244 G4double width;
245 G4double offset;
246 G4bool consuming;
247 pVPV -> GetReplicationData (axis, nReplicas, width, offset, consuming);
248 if (fCurrentDepth == 0) nReplicas = 1; // Just draw first
249 G4VPVParameterisation* pP = pVPV -> GetParameterisation ();
250 if (pP) { // Parametrised volume.
251 for (int n = 0; n < nReplicas; n++) {
252 pSol = pP -> ComputeSolid (n, pVPV);
253 pP -> ComputeTransformation (n, pVPV);
254 pSol -> ComputeDimensions (pP, n, pVPV);
255 pVPV -> SetCopyNo (n);
256 fCurrentPVCopyNo = n;
257 // Create a touchable of current parent for ComputeMaterial.
258 // fFullPVPath has not been updated yet so at this point it
259 // corresponds to the parent.
260 G4PhysicalVolumeModelTouchable parentTouchable(fFullPVPath);
261 pMaterial = pP -> ComputeMaterial (n, pVPV, &parentTouchable);
262 DescribeAndDescend (pVPV, requestedDepth, pLV, pSol, pMaterial,
263 theAT, sceneHandler);
264 }
265 }
266 else { // Plain replicated volume. From geometry_guide.txt...
267 // The replica's positions are claculated by means of a linear formula.
268 // Replication may occur along:
269 //
270 // o Cartesian axes (kXAxis,kYAxis,kZAxis)
271 //
272 // The replications, of specified width have coordinates of
273 // form (-width*(nReplicas-1)*0.5+n*width,0,0) where n=0.. nReplicas-1
274 // for the case of kXAxis, and are unrotated.
275 //
276 // o Radial axis (cylindrical polar) (kRho)
277 //
278 // The replications are cons/tubs sections, centred on the origin
279 // and are unrotated.
280 // They have radii of width*n+offset to width*(n+1)+offset
281 // where n=0..nReplicas-1
282 //
283 // o Phi axis (cylindrical polar) (kPhi)
284 // The replications are `phi sections' or wedges, and of cons/tubs form
285 // They have phi of offset+n*width to offset+(n+1)*width where
286 // n=0..nReplicas-1
287 //
288 pSol = pLV -> GetSolid ();
289 pMaterial = pLV -> GetMaterial ();
290 G4ThreeVector originalTranslation = pVPV -> GetTranslation ();
291 G4RotationMatrix* pOriginalRotation = pVPV -> GetRotation ();
292 G4double originalRMin = 0., originalRMax = 0.;
293 if (axis == kRho && pSol->GetEntityType() == "G4Tubs") {
294 originalRMin = ((G4Tubs*)pSol)->GetInnerRadius();
295 originalRMax = ((G4Tubs*)pSol)->GetOuterRadius();
296 }
297 G4bool visualisable = true;
298 for (int n = 0; n < nReplicas; n++) {
299 G4ThreeVector translation; // Identity.
300 G4RotationMatrix rotation; // Identity - life enough for visualizing.
301 G4RotationMatrix* pRotation = 0;
302 switch (axis) {
303 default:
304 case kXAxis:
305 translation = G4ThreeVector (-width*(nReplicas-1)*0.5+n*width,0,0);
306 break;
307 case kYAxis:
308 translation = G4ThreeVector (0,-width*(nReplicas-1)*0.5+n*width,0);
309 break;
310 case kZAxis:
311 translation = G4ThreeVector (0,0,-width*(nReplicas-1)*0.5+n*width);
312 break;
313 case kRho:
314 if (pSol->GetEntityType() == "G4Tubs") {
315 ((G4Tubs*)pSol)->SetInnerRadius(width*n+offset);
316 ((G4Tubs*)pSol)->SetOuterRadius(width*(n+1)+offset);
317 } else {
318 if (fpMP->IsWarning())
319 G4cout <<
320 "G4PhysicalVolumeModel::VisitGeometryAndGetVisReps: WARNING:"
321 "\n built-in replicated volumes replicated in radius for "
322 << pSol->GetEntityType() <<
323 "-type\n solids (your solid \""
324 << pSol->GetName() <<
325 "\") are not visualisable."
326 << G4endl;
327 visualisable = false;
328 }
329 break;
330 case kPhi:
331 rotation.rotateZ (-(offset+(n+0.5)*width));
332 // Minus Sign because for the physical volume we need the
333 // coordinate system rotation.
334 pRotation = &rotation;
335 break;
336 }
337 pVPV -> SetTranslation (translation);
338 pVPV -> SetRotation (pRotation);
339 pVPV -> SetCopyNo (n);
340 fCurrentPVCopyNo = n;
341 if (visualisable) {
342 DescribeAndDescend (pVPV, requestedDepth, pLV, pSol, pMaterial,
343 theAT, sceneHandler);
344 }
345 }
346 // Restore originals...
347 pVPV -> SetTranslation (originalTranslation);
348 pVPV -> SetRotation (pOriginalRotation);
349 if (axis == kRho && pSol->GetEntityType() == "G4Tubs") {
350 ((G4Tubs*)pSol)->SetInnerRadius(originalRMin);
351 ((G4Tubs*)pSol)->SetOuterRadius(originalRMax);
352 }
353 }
354 }
355}

357void G4PhysicalVolumeModel::DescribeAndDescend
358(G4VPhysicalVolume* pVPV,
359 G4int requestedDepth,
360 G4LogicalVolume* pLV,
361 G4VSolid* pSol,
362 G4Material* pMaterial,
363 const G4Transform3D& theAT,
364 G4VGraphicsScene& sceneHandler)
365{
366 // Maintain useful data members...
367 fpCurrentPV = pVPV;
368 fCurrentPVCopyNo = pVPV->GetCopyNo();
369 fpCurrentLV = pLV;
370 fpCurrentMaterial = pMaterial;
371
372 const G4RotationMatrix objectRotation = pVPV -> GetObjectRotationValue ();
373 const G4ThreeVector& translation = pVPV -> GetTranslation ();
374 G4Transform3D theLT (G4Transform3D (objectRotation, translation));
375
376 // Compute the accumulated transformation...
377 // Note that top volume's transformation relative to the world
378 // coordinate system is specified in theAT == startingTransformation
379 // = fTransform (see DescribeYourselfTo), so first time through the
380 // volume's own transformation, which is only relative to its
381 // mother, i.e., not relative to the world coordinate system, should
382 // not be accumulated.
383 G4Transform3D theNewAT (theAT);
384 if (fCurrentDepth != 0) theNewAT = theAT * theLT;
385 fpCurrentTransform = &theNewAT;
386
387 const G4VisAttributes* pVisAttribs = pLV->GetVisAttributes();
388 // If the volume does not have any vis attributes, create it.
389 G4VisAttributes* tempVisAtts = nullptr;
390 if (!pVisAttribs) {
391 tempVisAtts = new G4VisAttributes; // Default value.
392 // The user may request /vis/viewer/set/colourByDensity.
393 if (fpMP->GetCBDAlgorithmNumber() == 1) {
394 // Algorithm 1: 3 parameters: Simple rainbow mapping.
395 if (fpMP->GetCBDParameters().size() != 3) {
396 G4Exception("G4PhysicalVolumeModelTouchable::DescribeAndDescend",
397 "modeling0014",
398 FatalErrorInArgument,
399 "Algorithm-parameter mismatch for Colour By Density");
400 } else {
401 const G4double d = pMaterial? pMaterial->GetDensity(): 0.;
402 const G4double d0 = fpMP->GetCBDParameters()[0]; // Invisible d < d0.
403 const G4double d1 = fpMP->GetCBDParameters()[1]; // Rainbow d0->d1->d2.
404 const G4double d2 = fpMP->GetCBDParameters()[2]; // Blue d > d2.
405 if (d < d0) { // Density < d0 is invisible.
406 tempVisAtts->SetVisibility(false);
407 } else { // Intermediate densities are on a spectrum.
408 G4double red, green, blue;
409 if (d < d1) {
410 red = (d1-d)/(d1-d0); green = (d-d0)/(d1-d0); blue = 0.;
411 } else if (d < d2) {
412 red = 0.; green = (d2-d)/(d2-d1); blue = (d-d1)/(d2-d1);
413 } else { // Density >= d2 is blue.
414 red = 0.; green = 0.; blue = 1.;
415 }
416 tempVisAtts->SetColour(G4Colour(red,green,blue));
417 }
418 }
419 } else if (fpMP->GetCBDAlgorithmNumber() == 2) {
420 // Algorithm 2
421 // ...etc.
422 }
423 pVisAttribs = tempVisAtts;
424 }
425 // From here, can assume pVisAttribs is a valid pointer. This is necessary
426 // because PreAddSolid needs a vis attributes object.
427
428 // Make decision to draw...
429 G4bool thisToBeDrawn = true;
430
431 // Update full path of physical volumes...
432 G4int copyNo = fpCurrentPV->GetCopyNo();
433 fFullPVPath.push_back
434 (G4PhysicalVolumeNodeID
435 (fpCurrentPV,copyNo,fCurrentDepth,*fpCurrentTransform));
436
437 // Check if vis attributes are to be modified by a /vis/touchable/set/ command.
438 const auto& vams = fpMP->GetVisAttributesModifiers();
439 if (vams.size()) {
440 // OK, we have some VAMs (Vis Attributes Modifiers).
441 for (const auto& vam: vams) {
442 const auto& vamPath = vam.GetPVNameCopyNoPath();
443 if (vamPath.size() == fFullPVPath.size()) {
444 // OK, we have a size match.
445 // Check the volume name/copy number path.
446 auto iVAMNameCopyNo = vamPath.begin();
447 auto iPVNodeId = fFullPVPath.begin();
448 for (; iVAMNameCopyNo != vamPath.end(); ++iVAMNameCopyNo, ++iPVNodeId) {
449 if (!(
450 iVAMNameCopyNo->GetName() ==
451 iPVNodeId->GetPhysicalVolume()->GetName() &&
452 iVAMNameCopyNo->GetCopyNo() ==
453 iPVNodeId->GetPhysicalVolume()->GetCopyNo()
454 )) {
455 // This path element does NOT match.
456 break;
457 }
458 }
459 if (iVAMNameCopyNo == vamPath.end()) {
460 // OK, the paths match (the above loop terminated normally).
461 // Create a vis atts object for the modified vis atts.
462 // It is static so that we may return a reliable pointer to it.
463 static G4VisAttributes modifiedVisAtts;
464 // Initialise it with the current vis atts and reset the pointer.
465 modifiedVisAtts = *pVisAttribs;
466 pVisAttribs = &modifiedVisAtts;
467 const G4VisAttributes& transVisAtts = vam.GetVisAttributes();
468 switch (vam.GetVisAttributesSignifier()) {
469 case G4ModelingParameters::VASVisibility:
470 modifiedVisAtts.SetVisibility(transVisAtts.IsVisible());
471 break;
472 case G4ModelingParameters::VASDaughtersInvisible:
473 modifiedVisAtts.SetDaughtersInvisible
474 (transVisAtts.IsDaughtersInvisible());
475 break;
476 case G4ModelingParameters::VASColour:
477 modifiedVisAtts.SetColour(transVisAtts.GetColour());
478 break;
479 case G4ModelingParameters::VASLineStyle:
480 modifiedVisAtts.SetLineStyle(transVisAtts.GetLineStyle());
481 break;
482 case G4ModelingParameters::VASLineWidth:
483 modifiedVisAtts.SetLineWidth(transVisAtts.GetLineWidth());
484 break;
485 case G4ModelingParameters::VASForceWireframe:
486 if (transVisAtts.IsForceDrawingStyle()) {
487 if (transVisAtts.GetForcedDrawingStyle() ==
488 G4VisAttributes::wireframe) {
489 modifiedVisAtts.SetForceWireframe(true);
490 }
491 }
492 break;
493 case G4ModelingParameters::VASForceSolid:
494 if (transVisAtts.IsForceDrawingStyle()) {
495 if (transVisAtts.GetForcedDrawingStyle() ==
496 G4VisAttributes::solid) {
497 modifiedVisAtts.SetForceSolid(true);
498 }
499 }
500 break;
501 case G4ModelingParameters::VASForceAuxEdgeVisible:
502 if (transVisAtts.IsForceAuxEdgeVisible()) {
503 modifiedVisAtts.SetForceAuxEdgeVisible
504 (transVisAtts.IsForcedAuxEdgeVisible());
505 }
506 break;
507 case G4ModelingParameters::VASForceLineSegmentsPerCircle:
508 modifiedVisAtts.SetForceLineSegmentsPerCircle
509 (transVisAtts.GetForcedLineSegmentsPerCircle());
510 break;
511 }
512 }
513 }
514 }
515 }
516
517 // There are various reasons why this volume
518 // might not be drawn...
519 G4bool culling = fpMP->IsCulling();
520 G4bool cullingInvisible = fpMP->IsCullingInvisible();
521 G4bool markedVisible = pVisAttribs->IsVisible();
522 G4bool cullingLowDensity = fpMP->IsDensityCulling();
523 G4double density = pMaterial? pMaterial->GetDensity(): 0;
524 G4double densityCut = fpMP -> GetVisibleDensity ();
525
526 // 1) Global culling is on....
527 if (culling) {
528 // 2) Culling of invisible volumes is on...
529 if (cullingInvisible) {
530 // 3) ...and the volume is marked not visible...
531 if (!markedVisible) thisToBeDrawn = false;
532 }
533 // 4) Or culling of low density volumes is on...
534 if (cullingLowDensity) {
535 // 5) ...and density is less than cut value...
536 if (density < densityCut) thisToBeDrawn = false;
537 }
538 }
539 // 6) The user has asked for all further traversing to be aborted...
540 if (fAbort) thisToBeDrawn = false;
541
542 // Record thisToBeDrawn in path...
543 fFullPVPath.back().SetDrawn(thisToBeDrawn);
544
545 if (thisToBeDrawn) {
546
547 // Update path of drawn physical volumes...
548 fDrawnPVPath.push_back
549 (G4PhysicalVolumeNodeID
550 (fpCurrentPV,copyNo,fCurrentDepth,*fpCurrentTransform,thisToBeDrawn));
551
552 if (fpMP->IsExplode() && fDrawnPVPath.size() == 1) {
553 // For top-level drawn volumes, explode along radius...
554 G4Transform3D centering = G4Translate3D(fpMP->GetExplodeCentre());
555 G4Transform3D centred = centering.inverse() * theNewAT;
556 G4Scale3D oldScale;
557 G4Rotate3D oldRotation;
558 G4Translate3D oldTranslation;
559 centred.getDecomposition(oldScale, oldRotation, oldTranslation);
560 G4double explodeFactor = fpMP->GetExplodeFactor();
561 G4Translate3D newTranslation =
562 G4Translate3D(explodeFactor * oldTranslation.dx(),
563 explodeFactor * oldTranslation.dy(),
564 explodeFactor * oldTranslation.dz());
565 theNewAT = centering * newTranslation * oldRotation * oldScale;
566 }
567
568 DescribeSolid (theNewAT, pSol, pVisAttribs, sceneHandler);
569
570 }
571
572 // Make decision to draw daughters, if any. There are various
573 // reasons why daughters might not be drawn...
574
575 // First, reasons that do not depend on culling policy...
576 G4int nDaughters = pLV->GetNoDaughters();
577 G4bool daughtersToBeDrawn = true;
578 // 1) There are no daughters...
579 if (!nDaughters) daughtersToBeDrawn = false;
580 // 2) We are at the limit if requested depth...
581 else if (requestedDepth == 0) daughtersToBeDrawn = false;
582 // 3) The user has asked for all further traversing to be aborted...
583 else if (fAbort) daughtersToBeDrawn = false;
584 // 4) The user has asked that the descent be curtailed...
585 else if (fCurtailDescent) daughtersToBeDrawn = false;
586
587 // Now, reasons that depend on culling policy...
588 else {
589 G4bool daughtersInvisible = pVisAttribs->IsDaughtersInvisible();
590 // Culling of covered daughters request. This is computed in
591 // G4VSceneHandler::CreateModelingParameters() depending on view
592 // parameters...
593 G4bool cullingCovered = fpMP->IsCullingCovered();
594 G4bool surfaceDrawing =
595 fpMP->GetDrawingStyle() == G4ModelingParameters::hsr ||
596 fpMP->GetDrawingStyle() == G4ModelingParameters::hlhsr;
597 if (pVisAttribs->IsForceDrawingStyle()) {
598 switch (pVisAttribs->GetForcedDrawingStyle()) {
599 default:
600 case G4VisAttributes::wireframe: surfaceDrawing = false; break;
601 case G4VisAttributes::solid: surfaceDrawing = true; break;
602 }
603 }
604 G4bool opaque = pVisAttribs->GetColour().GetAlpha() >= 1.;
605 // 5) Global culling is on....
606 if (culling) {
607 // 6) ..and culling of invisible volumes is on...
608 if (cullingInvisible) {
609 // 7) ...and the mother requests daughters invisible
610 if (daughtersInvisible) daughtersToBeDrawn = false;
611 }
612 // 8) Or culling of covered daughters is requested...
613 if (cullingCovered) {
614 // 9) ...and surface drawing is operating...
615 if (surfaceDrawing) {
616 // 10) ...but only if mother is visible...
617 if (thisToBeDrawn) {
618 // 11) ...and opaque...
619 if (opaque) daughtersToBeDrawn = false;
620 }
621 }
622 }
623 }
624 }
625
626 if (daughtersToBeDrawn) {
627 for (G4int iDaughter = 0; iDaughter < nDaughters; iDaughter++) {
628 // Store daughter pVPV in local variable ready for recursion...
629 G4VPhysicalVolume* pDaughterVPV = pLV -> GetDaughter (iDaughter);
630 // Descend the geometry structure recursively...
631 fCurrentDepth++;
632 VisitGeometryAndGetVisReps
633 (pDaughterVPV, requestedDepth - 1, theNewAT, sceneHandler);
634 fCurrentDepth--;
635 }
636 }
637
638 delete tempVisAtts;
639
640 // Reset for normal descending of next volume at this level...
641 fCurtailDescent = false;
642
643 // Pop item from paths physical volumes...
644 fFullPVPath.pop_back();
645 if (thisToBeDrawn) {
646 fDrawnPVPath.pop_back();
647 }

650void G4PhysicalVolumeModel::DescribeSolid
651(const G4Transform3D& theAT,
652 G4VSolid* pSol,
653 const G4VisAttributes* pVisAttribs,
654 G4VGraphicsScene& sceneHandler)
655{
656 sceneHandler.PreAddSolid (theAT, *pVisAttribs);
657
658 G4VSolid* pSectionSolid = fpMP->GetSectionSolid();
659 G4VSolid* pCutawaySolid = fpMP->GetCutawaySolid();
660
661 if (!fpClippingSolid && !pSectionSolid && !pCutawaySolid) {
662
663 pSol -> DescribeYourselfTo (sceneHandler); // Standard treatment.
664
665 } else {
666
667 // Clipping, etc., performed by Boolean operations.
668
669 // First, get polyhedron for current solid...
670 if (pVisAttribs->IsForceLineSegmentsPerCircle())
671 G4Polyhedron::SetNumberOfRotationSteps
672 (pVisAttribs->GetForcedLineSegmentsPerCircle());
673 else
674 G4Polyhedron::SetNumberOfRotationSteps(fpMP->GetNoOfSides());
675 const G4Polyhedron* pOriginal = pSol->GetPolyhedron();
676 G4Polyhedron::ResetNumberOfRotationSteps();
677
678 if (!pOriginal) {
679
680 if (fpMP->IsWarning())
681 G4cout <<
682 "WARNING: G4PhysicalVolumeModel::DescribeSolid: solid\n \""
683 << pSol->GetName() <<
684 "\" has no polyhedron. Cannot by clipped."
685 << G4endl;
686 pSol -> DescribeYourselfTo (sceneHandler); // Standard treatment.
687
688 } else {
689
690 G4Polyhedron resultant(*pOriginal);
691 G4VisAttributes resultantVisAttribs(*pVisAttribs);
692 G4VSolid* resultantSolid = 0;
693
694 if (fpClippingSolid) {
695 switch (fClippingMode) {
696 default:
697 case subtraction:
698 resultantSolid = new G4SubtractionSolid
699 ("resultant_solid", pSol, fpClippingSolid, theAT.inverse());
700 break;
701 case intersection:
702 resultantSolid = new G4IntersectionSolid
703 ("resultant_solid", pSol, fpClippingSolid, theAT.inverse());
704 break;
705 }
706 }
707
708 if (pSectionSolid) {
709 resultantSolid = new G4IntersectionSolid
710 ("sectioned_solid", pSol, pSectionSolid, theAT.inverse());
711 }
712
713 if (pCutawaySolid) {
714 resultantSolid = new G4SubtractionSolid
715 ("cutaway_solid", pSol, pCutawaySolid, theAT.inverse());
716 }
717
718 G4Polyhedron* tmpResultant = resultantSolid->GetPolyhedron();
719 if (tmpResultant) resultant = *tmpResultant;
720 else {
721 if (fpMP->IsWarning())
722 G4cout <<
723 "WARNING: G4PhysicalVolumeModel::DescribeSolid: resultant polyhedron for"
724 "\n solid \"" << pSol->GetName() <<
725 "\" not defined due to error during Boolean processing."
726 "\n Original will be drawn in red."
727 << G4endl;
728 resultantVisAttribs.SetColour(G4Colour::Red());
729 }
730
731 delete resultantSolid;
732
733 // Finally, force polyhedron drawing...
734 resultant.SetVisAttributes(resultantVisAttribs);
735 sceneHandler.BeginPrimitives(theAT);
736 sceneHandler.AddPrimitive(resultant);
737 sceneHandler.EndPrimitives();
738 }
739 }
740 sceneHandler.PostAddSolid ();
741}

743G4bool G4PhysicalVolumeModel::Validate (G4bool warn)
744{
745 G4TransportationManager* transportationManager =
746 G4TransportationManager::GetTransportationManager ();
747
748 size_t nWorlds = transportationManager->GetNoWorlds();
749
750 G4bool found = false;
751
752 std::vector<G4VPhysicalVolume*>::iterator iterWorld =
753 transportationManager->GetWorldsIterator();
754 for (size_t i = 0; i < nWorlds; ++i, ++iterWorld) {
755 G4VPhysicalVolume* world = (*iterWorld);
756 if (!world) break; // This can happen if geometry has been cleared/destroyed.
757 // The idea now is to seek a PV with the same name and copy no
758 // in the hope it's the same one!!
759 G4PhysicalVolumeModel searchModel (world);
760 G4int verbosity = 0; // Suppress messages from G4PhysicalVolumeSearchScene.
761 G4PhysicalVolumeSearchScene searchScene
762 (&searchModel, fTopPVName, fTopPVCopyNo, verbosity);
763 G4ModelingParameters mp; // Default modeling parameters for this search.
764 mp.SetDefaultVisAttributes(fpMP? fpMP->GetDefaultVisAttributes(): 0);
765 searchModel.SetModelingParameters (&mp);
766 searchModel.DescribeYourselfTo (searchScene);
767 G4VPhysicalVolume* foundVolume = searchScene.GetFoundVolume ();
768 if (foundVolume) {
769 if (foundVolume != fpTopPV && warn) {
770 G4cout <<
771 "G4PhysicalVolumeModel::Validate(): A volume of the same name and"
772 "\n copy number (\""
773 << fTopPVName << "\", copy " << fTopPVCopyNo
774 << ") still exists and is being used."
775 "\n But it is not the same volume you originally specified"
776 "\n in /vis/scene/add/."
777 << G4endl;
778 }
779 fpTopPV = foundVolume;
780 CalculateExtent ();
781 found = true;
782 }
783 }
784 if (found) return true;
785 else {
786 if (warn) {
787 G4cout <<
788 "G4PhysicalVolumeModel::Validate(): No volume of name and"
789 "\n copy number (\""
790 << fTopPVName << "\", copy " << fTopPVCopyNo
791 << ") exists."
792 << G4endl;
793 }
794 return false;
795 }
