44G4ArrowModel::~G4ArrowModel ()
45{
46 delete fpHeadPolyhedron;
47 delete fpShaftPolyhedron;
48}
49
50G4ArrowModel::G4ArrowModel
51(G4double x1, G4double y1, G4double z1,
52 G4double x2, G4double y2, G4double z2,
53 G4double width, const G4Colour& colour,
54 const G4String& description,
55 G4int lineSegmentsPerCircle)
56{
57 fType = "G4ArrowModel";
58 fGlobalTag = fType;
59 fGlobalDescription = fType + ": " + description;
60 fExtent = G4VisExtent
61 (std::min(x1,x2),
62 std::max(x1,x2),
63 std::min(y1,y2),
64 std::max(y1,y2),
65 std::min(z1,z2),
66 std::max(z1,z2));
67
68 // Force number of line segments per circle (aka number of rotation steps)
69 G4int tempN = G4Polyhedron::GetNumberOfRotationSteps();
70 G4Polyhedron::SetNumberOfRotationSteps(lineSegmentsPerCircle);
71
72 // Make a cylinder slightly shorter than the arrow length so that it
73 // doesn't stick out of the head.
74 const G4double shaftLength = std::sqrt
75 (std::pow(x2-x1,2)+std::pow(y2-y1,2)+std::pow(z2-z1,2));
76 G4double shaftRadius = width/2.;
77 // Limit the radius
78 if (shaftRadius > shaftLength/100.) shaftRadius = shaftLength/100.;
79 const G4double halfShaftLength = shaftLength/2.;
80 const G4double halfReduction = 4.*shaftRadius;
81 const G4double halfLength = halfShaftLength-halfReduction;
82 G4Tubs shaft("shaft",0.,shaftRadius,halfLength,0.,twopi);
83 fpShaftPolyhedron = shaft.CreatePolyhedron();
84 // Move it a little so that the tail is at z = -halfShaftLength.
85 fpShaftPolyhedron->Transform(G4Translate3D(0,0,-halfReduction));
86
87 // Locate the head at +halfShaftLength.
88 const G4int numRZ = 3;
89 G4double r[] = {0,4,0};
90 G4double z[] = {0,-6,-4};
91 for (G4int i = 0; i < numRZ; i++) {
92 r[i] *= 2.*shaftRadius;
93 z[i] = halfShaftLength + z[i] * 2.*shaftRadius;
94 }
95 G4GenericPolycone head("head",0,twopi,numRZ,r,z);
96 fpHeadPolyhedron = head.CreatePolyhedron();
97
98 // Transform to position
99 const G4Vector3D arrowDirection = G4Vector3D(x2-x1,y2-y1,z2-z1).unit();
100 const G4double theta = arrowDirection.theta();
101 const G4double phi = arrowDirection.phi();
102 const G4Point3D arrowCentre(0.5*(x1+x2),0.5*(y1+y2),0.5*(z1+z2));
103 const G4Transform3D tr =
104 G4Translate3D(arrowCentre) * G4RotateZ3D(phi) * G4RotateY3D(theta);
105 fpShaftPolyhedron->Transform(tr);
106 fpHeadPolyhedron->Transform(tr);
107
108 G4VisAttributes va;
109 va.SetColour(colour);
110 va.SetForceSolid(true);
111 fpShaftPolyhedron->SetVisAttributes(va);
112 fpHeadPolyhedron->SetVisAttributes(va);
113
114 // Restore number of line segments per circle
115 G4Polyhedron::SetNumberOfRotationSteps(tempN);
116}
117
118void G4ArrowModel::DescribeYourselfTo (G4VGraphicsScene& sceneHandler)
119{
120 sceneHandler.BeginPrimitives();
121 sceneHandler.AddPrimitive(*fpShaftPolyhedron);
122 sceneHandler.AddPrimitive(*fpHeadPolyhedron);
123 sceneHandler.EndPrimitives();
124}
