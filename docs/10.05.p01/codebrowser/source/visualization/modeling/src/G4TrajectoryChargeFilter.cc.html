27// Filter trajectories according to charge. Only registered
28// charges will pass the filter.

32#include "G4TrajectoryChargeFilter.hh"
33#include <sstream>
34
35G4TrajectoryChargeFilter::G4TrajectoryChargeFilter(const G4String& name)
36 :G4SmartFilter<G4VTrajectory>(name)
37{}
38
39G4TrajectoryChargeFilter::~G4TrajectoryChargeFilter() {}
40
41bool
42G4TrajectoryChargeFilter::Evaluate(const G4VTrajectory& traj) const
43{
44 G4double charge = traj.GetCharge();
45
46 if (GetVerbose()) G4cout<<"G4TrajectoryChargeFilter processing trajectory with charge: "<<charge<<G4endl;
47
48 MyCharge myCharge;
49
50 if(charge>0.) myCharge = Positive;
51 else if(charge<0.) myCharge = Negative;
52 else myCharge = Neutral;
53
54 std::vector<MyCharge>::const_iterator iter = std::find(fCharges.begin(), fCharges.end(), myCharge);
55
56 // Fail if charge not registered
57 if (iter == fCharges.end()) return false;
58
59 return true;
60}
61
62void
63G4TrajectoryChargeFilter::Add(const G4String& charge)
64{
65 MyCharge myCharge;
66
67 if (!ConvertToCharge(charge, myCharge)) {
68 G4ExceptionDescription ed;
69 ed << "Invalid charge "<<charge;
70 G4Exception
71 ("G4TrajectoryChargeFilter::Add(const G4String& charge)",
72 "modeling0115", JustWarning, ed);
73 return;
74 }
75
76 return Add(myCharge);
77}
78
79void
80G4TrajectoryChargeFilter::Add(const MyCharge& charge)
81{
82 fCharges.push_back(charge);
83}
84
85void
86G4TrajectoryChargeFilter::Print(std::ostream& ostr) const
87{
88 ostr<<"Charges registered: "<<G4endl;
89 std::vector<MyCharge>::const_iterator iter = fCharges.begin();
90
91 while (iter != fCharges.end()) {
92 ostr<<*iter<<G4endl;
93 iter++;
94 }
95}
96
97void
98G4TrajectoryChargeFilter::Clear()
99{
100 // Clear registered charge vector
101 fCharges.clear();
102}
103
104G4bool
105G4TrajectoryChargeFilter::ConvertToCharge(const G4String& string, MyCharge& myCharge)
106{
107 bool result(true);
108
109 G4int charge;
110 std::istringstream is(string.c_str());
111 is >> charge;
112
113 switch (charge) {
114 case 1:
115 myCharge = G4TrajectoryChargeFilter::Positive;
116 break;
117 case 0:
118 myCharge = G4TrajectoryChargeFilter::Neutral;
119 break;
120 case -1:
121 myCharge = G4TrajectoryChargeFilter::Negative;
122 break;
123 default:
124 result = false;
125 }
126
127 return result;
128}
