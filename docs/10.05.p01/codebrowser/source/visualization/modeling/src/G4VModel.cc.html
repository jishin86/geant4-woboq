37G4VModel::G4VModel (const G4Transform3D& modelTransformation,
38 const G4ModelingParameters* pMP):
39 fType ("Other"),
40 fGlobalTag ("Empty"),
41 fGlobalDescription ("Empty"),
42 fTransform (modelTransformation),
43 fpMP (pMP)
44{}

46G4VModel::~G4VModel () {}

48G4String G4VModel::GetCurrentTag () const {
49 // Override in concrete class if concept of "current" is meaningful.
50 return fGlobalTag;

58G4bool G4VModel::Validate (G4bool) {
59 return true;
60}

62std::ostream& operator << (std::ostream& os, const G4VModel& model) {
63 os << model.fGlobalDescription;
64 os << "\n Modeling parameters:";
65 const G4ModelingParameters* mp = model.fpMP;
66 if (mp) os << "\n " << *mp;
67 else os << " none.";
68 os << "\n Extent: " << model.fExtent;
69 os << "\n Transformation: ";
70 os << "\n Rotation: ";
71 G4RotationMatrix rotation = model.fTransform.getRotation ();
72 os << rotation.thetaX() << ", "
73 << rotation.phiX() << ", "
74 << rotation.thetaY() << ", "
75 << rotation.phiY() << ", "
76 << rotation.thetaZ() << ", "
77 << rotation.phiZ();
78 os << "\n Translation: " << model.fTransform.getTranslation ();
79 return os;
80}
