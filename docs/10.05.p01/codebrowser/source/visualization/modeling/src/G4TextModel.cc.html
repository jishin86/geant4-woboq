39
40G4TextModel::~G4TextModel () {}
41
42G4TextModel::G4TextModel (const G4Text& g4Text): fG4Text(g4Text) {
43 fType = "G4TextModel";
44 std::ostringstream oss;
45 oss << "G4TextModel: \"" << fG4Text.GetText()
46 << "\" at " << G4BestUnit(g4Text.GetPosition(),"Length")
47 << "with size " << g4Text.GetScreenSize()
48 << " with offsets " << g4Text.GetXOffset() << ',' << g4Text.GetYOffset();
49 fGlobalTag = oss.str();
50 fGlobalDescription = fGlobalTag;
51}
52
53void G4TextModel::DescribeYourselfTo (G4VGraphicsScene& sceneHandler) {
54 sceneHandler.BeginPrimitives ();
55 sceneHandler.AddPrimitive (fG4Text);
56 sceneHandler.EndPrimitives ();
57}
