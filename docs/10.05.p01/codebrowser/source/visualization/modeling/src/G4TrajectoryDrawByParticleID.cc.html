34G4TrajectoryDrawByParticleID::G4TrajectoryDrawByParticleID(const G4String& name, G4VisTrajContext* context)
35 :G4VTrajectoryModel(name, context)
36 ,fDefault(G4Colour::Grey())
37{
38 Set("gamma", "green");
39 Set("e-", "red");
40 Set("e+", "blue");
41 Set("pi+", "magenta");
42 Set("pi-", "magenta");
43 Set("proton", "cyan");
44 Set("neutron", "yellow");
45}
46
47G4TrajectoryDrawByParticleID::~G4TrajectoryDrawByParticleID() {}
48
49void
50G4TrajectoryDrawByParticleID::Draw(const G4VTrajectory& traj, const G4bool& visible) const
51{
52 G4Colour colour(fDefault);
53 G4String particle = traj.GetParticleName();
54
55 fMap.GetColour(particle, colour);
56
57 G4VisTrajContext myContext(GetContext());
58
59 myContext.SetLineColour(colour);
60 myContext.SetVisible(visible);
61
62 if (GetVerbose()) {
63 G4cout<<"G4TrajectoryDrawByParticleID drawer named "<<Name();
64 G4cout<<", drawing trajectory with particle type, "<<particle<<G4endl;
65 G4cout<<", with configuration:"<<G4endl;
66 myContext.Print(G4cout);
67 }
68
69 G4TrajectoryDrawerUtils::DrawLineAndPoints(traj, myContext);
70}
71
72void
73G4TrajectoryDrawByParticleID::SetDefault(const G4String& colour)
74{
75 G4Colour myColour(G4Colour::White());
76
77 // Will not modify myColour if colour key does not exist
78 if (!G4Colour::GetColour(colour, myColour)) {
79 G4ExceptionDescription ed;
80 ed << "G4Colour with key "<<colour<<" does not exist ";
81 G4Exception
82 ("G4TrajectoryDrawByParticleID::SetDefault(const G4String& colour)",
83 "modeling0124", JustWarning, ed);
84 }
85
86 SetDefault(myColour);
87}
88
89void
90G4TrajectoryDrawByParticleID::SetDefault(const G4Colour& colour)
91{
92 fDefault = colour;
93}
94
95void
96G4TrajectoryDrawByParticleID::Set(const G4String& particle, const G4String& colour)
97{
98 fMap.Set(particle, colour);
99}
100
101void
102G4TrajectoryDrawByParticleID::Set(const G4String& particle, const G4Colour& colour)
103{
104 fMap[particle] = colour;
105}

107void
108G4TrajectoryDrawByParticleID::Print(std::ostream& ostr) const
109{
110 ostr<<"G4TrajectoryDrawByParticleID model "<< Name() <<" colour scheme: "<<std::endl;
111
112 ostr<<"Default colour: "<<fDefault<<G4endl;
113
114 fMap.Print(ostr);
115
116 ostr<<"Default configuration:"<<G4endl;
117 GetContext().Print(G4cout);
118}
