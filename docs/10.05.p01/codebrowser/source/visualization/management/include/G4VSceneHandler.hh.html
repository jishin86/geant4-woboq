35// a minimal abstract interface for the GEANT4 kernel.

75 ///////////////////////////////////////////////////////////////////
76 // Methods for adding raw GEANT4 objects to the scene handler. They
77 // must always be called in the triplet PreAddSolid, AddSolid and
78 // PostAddSolid. The transformation and visualization attributes
79 // must be set by the call to PreAddSolid. If your graphics system
80 // is sophisticated enough to handle a particular solid shape as a
81 // primitive, in your derived class write a function to override one
82 // or more of the following. See the implementation of
83 // G4VSceneHandler::AddSolid (const G4Box& box) for more
84 // suggestions. If not, please implement the base class invocation.
85
86 virtual void PreAddSolid (const G4Transform3D& objectTransformation,
87 const G4VisAttributes&);
88 // objectTransformation is the transformation in the world
89 // coordinate system of the object about to be added, and visAttribs
90 // is its visualization attributes.
91 // IMPORTANT: invoke this from your polymorphic versions, e.g.:
92 // void MyXXXSceneHandler::PreAddSolid
93 // (const G4Transform3D& objectTransformation,
94 // const G4VisAttributes& visAttribs) {
95 // G4VSceneHandler::PreAddSolid (objectTransformation, visAttribs);
96 // ...
97 // }
98
99 virtual void PostAddSolid ();
100 // IMPORTANT: invoke this from your polymorphic versions, e.g.:
101 // void MyXXXSceneHandler::PostAddSolid () {
102 // ...
103 // G4VSceneHandler::PostAddSolid ();
104 // }
105
106 // From geometry/solids/CSG
107 virtual void AddSolid (const G4Box&);
108 virtual void AddSolid (const G4Cons&);
109 virtual void AddSolid (const G4Orb&);
110 virtual void AddSolid (const G4Para&);
111 virtual void AddSolid (const G4Sphere&);
112 virtual void AddSolid (const G4Torus&);
113 virtual void AddSolid (const G4Trap&);
114 virtual void AddSolid (const G4Trd&);
115 virtual void AddSolid (const G4Tubs&);
116
117 // From geometry/solids/specific
118 virtual void AddSolid (const G4Ellipsoid&);
119 virtual void AddSolid (const G4Polycone&);
120 virtual void AddSolid (const G4Polyhedra&);
121
122 // For solids not above.
123 virtual void AddSolid (const G4VSolid&);

125 ///////////////////////////////////////////////////////////////////
126 // Methods for adding "compound" GEANT4 objects to the scene
127 // handler. These methods may either (a) invoke "user code" that
128 // uses the "user interface", G4VVisManager (see, for example,
129 // G4VSceneHandler, which for trajectories uses
130 // G4VTrajectory::DrawTrajectory, via G4TrajectoriesModel in the
131 // Modeling Category) or (b) invoke AddPrimitives below (between
132 // calls to Begin/EndPrimitives) or (c) use graphics-system-specific
133 // code or (d) any combination of the above.
134
135 virtual void AddCompound (const G4VTrajectory&);
136 virtual void AddCompound (const G4VHit&);
137 virtual void AddCompound (const G4VDigi&);
138 virtual void AddCompound (const G4THitsMap<G4double>&);
139 virtual void AddCompound (const G4THitsMap<G4StatDouble>&);

141 //////////////////////////////////////////////////////////////
142 // Functions for adding primitives.
143
144 virtual void BeginModeling ();
145 // IMPORTANT: invoke this from your polymorphic versions, e.g.:
146 // void MyXXXSceneHandler::BeginModeling () {
147 // G4VSceneHandler::BeginModeling ();
148 // ...
149 // }
150
151 virtual void EndModeling ();
152 // IMPORTANT: invoke this from your polymorphic versions, e.g.:
153 // void MyXXXSceneHandler::EndModeling () {
154 // ...
155 // G4VSceneHandler::EndModeling ();
156 // }
157
158 virtual void BeginPrimitives
159 (const G4Transform3D& objectTransformation);
160 // IMPORTANT: invoke this from your polymorphic versions, e.g.:
161 // void MyXXXSceneHandler::BeginPrimitives
162 // (const G4Transform3D& objectTransformation) {
163 // G4VSceneHandler::BeginPrimitives (objectTransformation);
164 // ...
165 // }
166
167 virtual void EndPrimitives ();
168 // IMPORTANT: invoke this from your polymorphic versions, e.g.:
169 // void MyXXXSceneHandler::EndPrimitives () {
170 // ...
171 // G4VSceneHandler::EndPrimitives ();
172 // }
173
174 virtual void BeginPrimitives2D
175 (const G4Transform3D& objectTransformation);
176 // The x,y coordinates of the primitives passed to AddPrimitive are
177 // intrepreted as screen coordinates, -1 < x,y < 1. The
178 // z-coordinate is ignored.
179 // IMPORTANT: invoke this from your polymorphic versions, e.g.:
180 // void MyXXXSceneHandler::BeginPrimitives2D
181 // (const G4Transform3D& objectTransformation) {
182 // G4VSceneHandler::BeginPrimitives2D ();
183 // ...
184 // }
185
186 virtual void EndPrimitives2D ();
187 // IMPORTANT: invoke this from your polymorphic versions, e.g.:
188 // void MyXXXSceneHandler::EndPrimitives2D () {
189 // ...
190 // G4VSceneHandler::EndPrimitives2D ();
191 // }
192
193 virtual void AddPrimitive (const G4Polyline&) = 0;
194 virtual void AddPrimitive (const G4Scale&);
195 // Default implementation in this class but can be over-ridden.
196 virtual void AddPrimitive (const G4Text&) = 0;
197 virtual void AddPrimitive (const G4Circle&) = 0;
198 virtual void AddPrimitive (const G4Square&) = 0;
199 virtual void AddPrimitive (const G4Polymarker&);
200 // Default implementation in this class but can be over-ridden.
201 virtual void AddPrimitive (const G4Polyhedron&) = 0;

320 // Generic clipping using the BooleanProcessor in graphics_reps is
321 // implemented in this class. Subclasses that implement their own
322 // clipping should provide an override that returns zero.
323
324 void LoadAtts(const G4Visible&, G4AttHolder*);
325 // Load G4AttValues and G4AttDefs associated with the G4Visible
326 // object onto the G4AttHolder object. It checks fpModel, and also
327 // loads the G4AttValues and G4AttDefs from G4PhysicalVolumeModel,
328 // G4VTrajectory, G4VTrajectoryPoint, G4VHit or G4VDigi, as
329 // appropriate. The G4AttHolder object is an object of a class that
330 // publicly inherits G4AttHolder - see, e.g., SoG4Polyhedron in the
331 // Open Inventor driver. G4AttHolder deletes G4AttValues in its
332 // destructor to ensure proper clean-up of G4AttValues.
