93G4VSceneHandler::G4VSceneHandler (G4VGraphicsSystem& system, G4int id, const G4String& name):
94 fSystem (system),
95 fSceneHandlerId (id),
96 fViewCount (0),
97 fpViewer (0),
98 fpScene (0),
99 fMarkForClearingTransientStore (true), // Ready for first
100 // ClearTransientStoreIfMarked(),
101 // e.g., at end of run (see
102 // G4VisManager.cc).
103 fReadyForTransients (true), // Only false while processing scene.
104 fProcessingSolid (false),
105 fProcessing2D (false),
106 fpModel (0),
107 fNestingDepth (0),
108 fpVisAttribs (0)
109{
110 G4VisManager* pVMan = G4VisManager::GetInstance ();
111 fpScene = pVMan -> GetCurrentScene ();
112 if (name == "") {
113 std::ostringstream ost;
114 ost << fSystem.GetName () << '-' << fSceneHandlerId;
115 fName = ost.str();
116 }
117 else {
118 fName = name;
119 }
120 fTransientsDrawnThisEvent = pVMan->GetTransientsDrawnThisEvent();
121 fTransientsDrawnThisRun = pVMan->GetTransientsDrawnThisRun();
122}

159void G4VSceneHandler::BeginPrimitives
160(const G4Transform3D& objectTransformation) {
161 //static G4int count = 0;
162 //G4cout << "G4VSceneHandler::BeginPrimitives: " << count++ << G4endl;
163 fNestingDepth++;

227template <class T> void G4VSceneHandler::AddSolidWithAuxiliaryEdges
228(const T& solid)
229{
230 // Get and check applicable vis attributes.
231 fpVisAttribs = fpViewer->GetApplicableVisAttributes(fpVisAttribs);
232 // Draw with auxiliary edges unless otherwise specified.
233 if (!fpVisAttribs->IsForceAuxEdgeVisible()) {
234 // Create a vis atts object for the modified vis atts.
235 // It is static so that we may return a reliable pointer to it.
236 static G4VisAttributes visAttsWithAuxEdges;
237 // Initialise it with the current vis atts and reset the pointer.
238 visAttsWithAuxEdges = *fpVisAttribs;
239 // Force auxiliary edges visible.
240 visAttsWithAuxEdges.SetForceAuxEdgeVisible();
241 fpVisAttribs = &visAttsWithAuxEdges;
242 }
243 RequestPrimitives (solid);
244}

335void G4VSceneHandler::AddCompound (const G4THitsMap<G4double>& hits) {
336 //G4cout << "AddCompound: hits: " << &hits << G4endl;
337 G4bool scoreMapHits = false;
338 G4ScoringManager* scoringManager = G4ScoringManager::GetScoringManagerIfExist();
339 if (scoringManager) {
340 size_t nMeshes = scoringManager->GetNumberOfMesh();
341 for (size_t iMesh = 0; iMesh < nMeshes; ++iMesh) {
342 G4VScoringMesh* mesh = scoringManager->GetMesh(iMesh);
343 if (mesh && mesh->IsActive()) {
344 MeshScoreMap scoreMap = mesh->GetScoreMap();
345 const G4String& mapNam = const_cast<G4THitsMap<G4double>&>(hits).GetName();
346 for(MeshScoreMap::const_iterator i = scoreMap.begin();
347 i != scoreMap.end(); ++i) {
348 const G4String& scoreMapName = i->first;
349 if (scoreMapName == mapNam) {
350 G4DefaultLinearColorMap colorMap("G4VSceneHandlerColorMap");
351 scoreMapHits = true;
352 mesh->DrawMesh(scoreMapName, &colorMap);
353 }
354 }
355 }
356 }
357 }
358 if (scoreMapHits) {
359 static G4bool first = true;
360 if (first) {
361 first = false;
362 G4cout <<
363 "Scoring map drawn with default parameters."
364 "\n To get gMocren file for gMocren browser:"
365 "\n /vis/open gMocrenFile"
366 "\n /vis/viewer/flush"
367 "\n Many other options available with /score/draw... commands."
368 "\n You might want to \"/vis/viewer/set/autoRefresh false\"."
369 << G4endl;
370 }
371 } else { // Not score map hits. Just call DrawAllHits.
372 // Cast away const because DrawAllHits is non-const!!!!
373 const_cast<G4THitsMap<G4double>&>(hits).DrawAllHits();
374 }
375}

423void G4VSceneHandler::AddPrimitive (const G4Scale& scale) {
424
425 const G4double margin(0.01);
426 // Fractional margin - ensures scale is comfortably inside viewing
427 // volume.
428 const G4double oneMinusMargin (1. - margin);
429
430 const G4VisExtent& sceneExtent = fpScene->GetExtent();
431
432 // Useful constants...
433 const G4double length(scale.GetLength());
434 const G4double halfLength(length / 2.);
435 const G4double tickLength(length / 20.);
436 const G4double piBy2(halfpi);
437
438 // Get size of scene...
439 const G4double xmin = sceneExtent.GetXmin();
440 const G4double xmax = sceneExtent.GetXmax();
441 const G4double ymin = sceneExtent.GetYmin();
442 const G4double ymax = sceneExtent.GetYmax();
443 const G4double zmin = sceneExtent.GetZmin();
444 const G4double zmax = sceneExtent.GetZmax();
445
446 // Create (empty) polylines having the same vis attributes...
447 G4Polyline scaleLine, tick11, tick12, tick21, tick22;
448 G4VisAttributes visAtts(*scale.GetVisAttributes()); // Long enough life.
449 scaleLine.SetVisAttributes(&visAtts);
450 tick11.SetVisAttributes(&visAtts);
451 tick12.SetVisAttributes(&visAtts);
452 tick21.SetVisAttributes(&visAtts);
453 tick22.SetVisAttributes(&visAtts);
454
455 // Add points to the polylines to represent an scale parallel to the
456 // x-axis centred on the origin...
457 G4Point3D r1(G4Point3D(-halfLength, 0., 0.));
458 G4Point3D r2(G4Point3D( halfLength, 0., 0.));
459 scaleLine.push_back(r1);
460 scaleLine.push_back(r2);
461 G4Point3D ticky(0., tickLength, 0.);
462 G4Point3D tickz(0., 0., tickLength);
463 tick11.push_back(r1 + ticky);
464 tick11.push_back(r1 - ticky);
465 tick12.push_back(r1 + tickz);
466 tick12.push_back(r1 - tickz);
467 tick21.push_back(r2 + ticky);
468 tick21.push_back(r2 - ticky);
469 tick22.push_back(r2 + tickz);
470 tick22.push_back(r2 - tickz);
471 G4Point3D textPosition(0., tickLength, 0.);
472
473 // Transform appropriately...
474
475 G4Transform3D transformation;
476 if (scale.GetAutoPlacing()) {
477 G4Transform3D rotation;
478 switch (scale.GetDirection()) {
479 case G4Scale::x:
480 break;
481 case G4Scale::y:
482 rotation = G4RotateZ3D(piBy2);
483 break;
484 case G4Scale::z:
485 rotation = G4RotateY3D(piBy2);
486 break;
487 }
488 G4double sxmid;
489 G4double symid;
490 G4double szmid;
491 sxmid = xmin + oneMinusMargin * (xmax - xmin);
492 symid = ymin + margin * (ymax - ymin);
493 szmid = zmin + oneMinusMargin * (zmax - zmin);
494 switch (scale.GetDirection()) {
495 case G4Scale::x:
496 sxmid -= halfLength;
497 break;
498 case G4Scale::y:
499 symid += halfLength;
500 break;
501 case G4Scale::z:
502 szmid -= halfLength;
503 break;
504 }
505 G4Translate3D translation(sxmid, symid, szmid);
506 transformation = translation * rotation;
507 } else {
508 if (fpModel) transformation = fpModel->GetTransformation();
509 }
510
511 // Draw...
512 // We would like to call BeginPrimitives(transformation) here but
513 // calling BeginPrimitives from within an AddPrimitive is not
514 // allowed! So we have to do our own transformation...
515 AddPrimitive(scaleLine.transform(transformation));
516 AddPrimitive(tick11.transform(transformation));
517 AddPrimitive(tick12.transform(transformation));
518 AddPrimitive(tick21.transform(transformation));
519 AddPrimitive(tick22.transform(transformation));
520 G4Text text(scale.GetAnnotation(),textPosition.transform(transformation));
521 G4VisAttributes va(G4VVisCommand::GetCurrentTextColour());
522 text.SetVisAttributes(va);
523 text.SetScreenSize(scale.GetAnnotationSize());
524 AddPrimitive(text);
525}

527void G4VSceneHandler::AddPrimitive (const G4Polymarker& polymarker) {
528 switch (polymarker.GetMarkerType()) {
529 default:
530 case G4Polymarker::dots:
531 {
532 for (size_t iPoint = 0; iPoint < polymarker.size (); iPoint++) {
533 G4Circle dot (polymarker);
534 dot.SetPosition (polymarker[iPoint]);
535 dot.SetWorldSize (0.);
536 dot.SetScreenSize (0.1); // Very small circle.
537 AddPrimitive (dot);
538 }
539 }
540 break;
541 case G4Polymarker::circles:
542 {
543 for (size_t iPoint = 0; iPoint < polymarker.size (); iPoint++) {
544 G4Circle circle (polymarker);
545 circle.SetPosition (polymarker[iPoint]);
546 AddPrimitive (circle);
547 }
548 }
549 break;
550 case G4Polymarker::squares:
551 {
552 for (size_t iPoint = 0; iPoint < polymarker.size (); iPoint++) {
553 G4Square square (polymarker);
554 square.SetPosition (polymarker[iPoint]);
555 AddPrimitive (square);
556 }
557 }
558 break;
559 }
560}

575void G4VSceneHandler::RequestPrimitives (const G4VSolid& solid) {
576 G4Polyhedron::SetNumberOfRotationSteps (GetNoOfSides (fpVisAttribs));
577 G4Polyhedron* pPolyhedron = solid.GetPolyhedron ();
578 G4Polyhedron::ResetNumberOfRotationSteps ();
579 if (pPolyhedron) {
580 pPolyhedron -> SetVisAttributes (fpVisAttribs);
581 BeginPrimitives (fObjectTransformation);
582 AddPrimitive (*pPolyhedron);
583 EndPrimitives ();
584 }
585 else {
586 G4VisManager::Verbosity verbosity = G4VisManager::GetVerbosity();
587 if (verbosity >= G4VisManager::errors) {
588 G4cerr <<
589 "ERROR: G4VSceneHandler::RequestPrimitives"
590 "\n Polyhedron not available for " << solid.GetName () <<
591 "\n Touchable path: ";
592 G4PhysicalVolumeModel* pPVModel = dynamic_cast<G4PhysicalVolumeModel*>(fpModel);
593 if (pPVModel) {
594 G4cerr << pPVModel->GetFullPVPath();
595 }
596 static G4bool explanation = false;
597 if (!explanation) {
598 explanation = true;
599 G4cerr <<
600 "\n This means it cannot be visualized on most systems (try RayTracer)."
601 "\n 1) The solid may not have implemented the CreatePolyhedron method."
602 "\n 2) For Boolean solids, the BooleanProcessor, which attempts to create"
603 "\n the resultant polyhedron, may have failed.";
604 }
605 G4cerr << G4endl;
606 }
607 }

610void G4VSceneHandler::ProcessScene () {
611
612 // Assumes graphics database store has already been cleared if
613 // relevant for the particular scene handler.
614
615 if (!fpScene) return;
616
617 if (fpScene->GetExtent() == G4VisExtent::GetNullExtent()) {
618 G4Exception
619 ("G4VSceneHandler::ProcessScene",
620 "visman0106", JustWarning,
621 "The scene has no extent.");
622 }
623
624 G4VisManager* visManager = G4VisManager::GetInstance();
625
626 if (!visManager->GetConcreteInstance()) return;
627
628 G4VisManager::Verbosity verbosity = visManager->GetVerbosity();
629
630 fReadyForTransients = false;
631
632 // Reset fMarkForClearingTransientStore. (Leaving
633 // fMarkForClearingTransientStore true causes problems with
634 // recomputing transients below.) Restore it again at end...
635 G4bool tmpMarkForClearingTransientStore = fMarkForClearingTransientStore;
636 fMarkForClearingTransientStore = false;
637
638 // Traverse geometry tree and send drawing primitives to window(s).
639
640 const std::vector<G4Scene::Model>& runDurationModelList =
641 fpScene -> GetRunDurationModelList ();
642
643 if (runDurationModelList.size ()) {
644 if (verbosity >= G4VisManager::confirmations) {
645 G4cout << "Traversing scene data..." << G4endl;
646 }
647
648 BeginModeling ();
649
650 // Create modeling parameters from view parameters...
651 G4ModelingParameters* pMP = CreateModelingParameters ();
652
653 for (size_t i = 0; i < runDurationModelList.size (); i++) {
654 if (runDurationModelList[i].fActive) {
655 fpModel = runDurationModelList[i].fpModel;
656 // Note: this is not the place to take action on
657 // pModel->GetTransformation(). The model must take care of
658 // this in pModel->DescribeYourselfTo(*this). See, for example,
659 // G4PhysicalVolumeModel and /vis/scene/add/logo.
660 fpModel -> SetModelingParameters (pMP);
661 fpModel -> DescribeYourselfTo (*this);
662 fpModel -> SetModelingParameters (0);
663 }
664 }
665
666 fpModel = 0;
667 delete pMP;
668
669 EndModeling ();
670 }
671
672 fReadyForTransients = true;
673
674 // Refresh event from end-of-event model list.
675 // Allow only in Idle or GeomClosed state...
676 G4StateManager* stateManager = G4StateManager::GetStateManager();
677 G4ApplicationState state = stateManager->GetCurrentState();
678 if (state == G4State_Idle || state == G4State_GeomClosed) {
679
680 visManager->SetEventRefreshing(true);
681
682 if (visManager->GetRequestedEvent()) {
683 DrawEvent(visManager->GetRequestedEvent());
684
685 } else {
686
687 G4RunManager* runManager = G4RunManager::GetRunManager();
688#ifdef G4MULTITHREADED
689 if(G4Threading::IsMultithreadedApplication())
690 { runManager = G4MTRunManager::GetMasterRunManager(); }
691#endif
692 if (runManager) {
693 const G4Run* run = runManager->GetCurrentRun();
694 const std::vector<const G4Event*>* events =
695 run? run->GetEventVector(): 0;
696 size_t nKeptEvents = 0;
697 if (events) nKeptEvents = events->size();
698 if (nKeptEvents) {
699
700 if (fpScene->GetRefreshAtEndOfEvent()) {
701
702 if (verbosity >= G4VisManager::confirmations) {
703 G4cout << "Refreshing event..." << G4endl;
704 }
705 const G4Event* event = 0;
706 if (events && events->size()) event = events->back();
707 if (event) DrawEvent(event);
708
709 } else { // Accumulating events.
710
711 if (verbosity >= G4VisManager::confirmations) {
712 G4cout << "Refreshing events in run..." << G4endl;
713 }
714 for (const auto& event: *events) {
715 if (event) DrawEvent(event);
716 }
717
718 if (!fpScene->GetRefreshAtEndOfRun()) {
719 if (verbosity >= G4VisManager::warnings) {
720 G4cout <<
721 "WARNING: Cannot refresh events accumulated over more"
722 "\n than one runs. Refreshed just the last run."
723 << G4endl;
724 }
725 }
726 }
727 }
728 }
729 }
730 visManager->SetEventRefreshing(false);
731 }
732
733 // Refresh end-of-run model list.
734 // Allow only in Idle or GeomClosed state...
735 if (state == G4State_Idle || state == G4State_GeomClosed) {
736 DrawEndOfRunModels();
737 }
738
739 fMarkForClearingTransientStore = tmpMarkForClearingTransientStore;
740}

784G4ModelingParameters* G4VSceneHandler::CreateModelingParameters ()
785{
786 // Create modeling parameters from View Parameters...
787 if (!fpViewer) return NULL;
788
789 const G4ViewParameters& vp = fpViewer -> GetViewParameters ();
790
791 // Convert drawing styles...
792 G4ModelingParameters::DrawingStyle modelDrawingStyle =
793 G4ModelingParameters::wf;
794 switch (vp.GetDrawingStyle ()) {
795 default:
796 case G4ViewParameters::wireframe:
797 modelDrawingStyle = G4ModelingParameters::wf;
798 break;
799 case G4ViewParameters::hlr:
800 modelDrawingStyle = G4ModelingParameters::hlr;
801 break;
802 case G4ViewParameters::hsr:
803 modelDrawingStyle = G4ModelingParameters::hsr;
804 break;
805 case G4ViewParameters::hlhsr:
806 modelDrawingStyle = G4ModelingParameters::hlhsr;
807 break;
808 }
809
810 // Decide if covered daughters are really to be culled...
811 G4bool reallyCullCovered =
812 vp.IsCullingCovered() // Culling daughters depends also on...
813 && !vp.IsSection () // Sections (DCUT) not requested.
814 && !vp.IsCutaway () // Cutaways not requested.
815 ;
816
817 G4ModelingParameters* pModelingParams = new G4ModelingParameters
818 (vp.GetDefaultVisAttributes (),
819 modelDrawingStyle,
820 vp.IsCulling (),
821 vp.IsCullingInvisible (),
822 vp.IsDensityCulling (),
823 vp.GetVisibleDensity (),
824 reallyCullCovered,
825 vp.GetNoOfSides ()
826 );
827
828 pModelingParams->SetWarning
829 (G4VisManager::GetVerbosity() >= G4VisManager::warnings);
830
831 pModelingParams->SetCBDAlgorithmNumber(vp.GetCBDAlgorithmNumber());
832 pModelingParams->SetCBDParameters(vp.GetCBDParameters());
833
834 pModelingParams->SetExplodeFactor(vp.GetExplodeFactor());
835 pModelingParams->SetExplodeCentre(vp.GetExplodeCentre());
836
837 pModelingParams->SetSectionSolid(CreateSectionSolid());
838 pModelingParams->SetCutawaySolid(CreateCutawaySolid());
839 // The polyhedron objects are deleted in the modeling parameters destructor.
840
841 pModelingParams->SetVisAttributesModifiers(vp.GetVisAttributesModifiers());
842
843 return pModelingParams;
844}

846G4VSolid* G4VSceneHandler::CreateSectionSolid()
847{
848 G4VSolid* sectioner = 0;
849 const G4ViewParameters& vp = fpViewer->GetViewParameters();
850 if (vp.IsSection () ) {
851 G4double radius = fpScene->GetExtent().GetExtentRadius();
852 G4double safe = radius + fpScene->GetExtent().GetExtentCentre().mag();
853 G4VSolid* sectionBox =
854 new G4Box("_sectioner", safe, safe, 1.e-5 * radius); // Thin in z-plane.
855 const G4Plane3D& sp = vp.GetSectionPlane ();
856 G4double a = sp.a();
857 G4double b = sp.b();
858 G4double c = sp.c();
859 G4double d = sp.d();
860 G4Transform3D transform = G4TranslateZ3D(-d);
861 const G4Normal3D normal(a,b,c);
862 if (normal != G4Normal3D(0,0,1)) {
863 const G4double angle = std::acos(normal.dot(G4Normal3D(0,0,1)));
864 const G4Vector3D axis = G4Normal3D(0,0,1).cross(normal);
865 transform = G4Rotate3D(angle, axis) * transform;
866 }
867 sectioner = new G4DisplacedSolid
868 ("_displaced_sectioning_box", sectionBox, transform);
869 }
870 return sectioner;
871}

903void G4VSceneHandler::LoadAtts(const G4Visible& visible, G4AttHolder* holder)
904{
905 // Load G4Atts from G4VisAttributes, if any...
906 const G4VisAttributes* va = visible.GetVisAttributes();
907 if (va) {
908 const std::map<G4String,G4AttDef>* vaDefs =
909 va->GetAttDefs();
910 if (vaDefs) {
911 holder->AddAtts(visible.GetVisAttributes()->CreateAttValues(), vaDefs);
912 }
913 }
914
915 G4PhysicalVolumeModel* pPVModel =
916 dynamic_cast<G4PhysicalVolumeModel*>(fpModel);
917 if (pPVModel) {
918 // Load G4Atts from G4PhysicalVolumeModel...
919 const std::map<G4String,G4AttDef>* pvDefs = pPVModel->GetAttDefs();
920 if (pvDefs) {
921 holder->AddAtts(pPVModel->CreateCurrentAttValues(), pvDefs);
922 }
923 }
924
925 G4TrajectoriesModel* trajModel = dynamic_cast<G4TrajectoriesModel*>(fpModel);
926 if (trajModel) {
927 // Load G4Atts from trajectory model...
928 const std::map<G4String,G4AttDef>* trajModelDefs = trajModel->GetAttDefs();
929 if (trajModelDefs) {
930 holder->AddAtts(trajModel->CreateCurrentAttValues(), trajModelDefs);
931 }
932 // Load G4Atts from trajectory...
933 const G4VTrajectory* traj = trajModel->GetCurrentTrajectory();
934 if (traj) {
935 const std::map<G4String,G4AttDef>* trajDefs = traj->GetAttDefs();
936 if (trajDefs) {
937 holder->AddAtts(traj->CreateAttValues(), trajDefs);
938 }
939 G4int nPoints = traj->GetPointEntries();
940 for (G4int i = 0; i < nPoints; ++i) {
941 G4VTrajectoryPoint* trajPoint = traj->GetPoint(i);
942 if (trajPoint) {
943 const std::map<G4String,G4AttDef>* pointDefs = trajPoint->GetAttDefs();
944 if (pointDefs) {
945 holder->AddAtts(trajPoint->CreateAttValues(), pointDefs);
946 }
947 }
948 }
949 }
950 }
951
952 G4HitsModel* hitsModel = dynamic_cast<G4HitsModel*>(fpModel);
953 if (hitsModel) {
954 // Load G4Atts from hit...
955 const G4VHit* hit = hitsModel->GetCurrentHit();
956 const std::map<G4String,G4AttDef>* hitsDefs = hit->GetAttDefs();
957 if (hitsDefs) {
958 holder->AddAtts(hit->CreateAttValues(), hitsDefs);
959 }
960 }

981G4ViewParameters::DrawingStyle G4VSceneHandler::GetDrawingStyle
982(const G4VisAttributes* pVisAttribs) {
983 // Drawing style is normally determined by the view parameters, but
984 // it can be overriddden by the ForceDrawingStyle flag in the vis
985 // attributes.
986 G4ViewParameters::DrawingStyle style =
987 fpViewer->GetViewParameters().GetDrawingStyle();
988 if (pVisAttribs -> IsForceDrawingStyle ()) {
989 G4VisAttributes::ForcedDrawingStyle forcedStyle =
990 pVisAttribs -> GetForcedDrawingStyle ();
991 // This is complicated because if hidden line and surface removal
992 // has been requested we wish to preserve this sometimes.
993 switch (forcedStyle) {
994 case (G4VisAttributes::solid):
995 switch (style) {
996 case (G4ViewParameters::hlr):
997 style = G4ViewParameters::hlhsr;
998 break;
999 case (G4ViewParameters::wireframe):
1000 style = G4ViewParameters::hsr;
1001 break;
1002 case (G4ViewParameters::hlhsr):
1003 case (G4ViewParameters::hsr):
1004 default:
1005 break;
1006 }
1007 break;
1008 case (G4VisAttributes::wireframe):
1009 default:
1010 // But if forced style is wireframe, do it, because one of its
1011 // main uses is in displaying the consituent solids of a Boolean
1012 // solid and their surfaces overlap with the resulting Booean
1013 // solid, making a mess if hlr is specified.
1014 style = G4ViewParameters::wireframe;
1015 break;
1016 }
1017 }
1018 return style;
1019}
