138G4double G4ViewParameters::GetCameraDistance (G4double radius) const {
139 G4double cameraDistance;
140 if (fFieldHalfAngle == 0.) {
141 cameraDistance = radius;
142 }
143 else {
144 cameraDistance = radius / std::sin (fFieldHalfAngle) - fDolly;
145 }
146 return cameraDistance;
147}

149G4double G4ViewParameters::GetNearDistance (G4double cameraDistance,
150 G4double radius) const {
151 const G4double small = 1.e-6 * radius;
152 G4double nearDistance = cameraDistance - radius;
153 if (nearDistance < small) nearDistance = small;
154 return nearDistance;
155}

157G4double G4ViewParameters::GetFarDistance (G4double cameraDistance,
158 G4double nearDistance,
159 G4double radius) const {
160 G4double farDistance = cameraDistance + radius;
161 if (farDistance < nearDistance) farDistance = nearDistance;
162 return farDistance;
163}

165G4double G4ViewParameters::GetFrontHalfHeight (G4double nearDistance,
166 G4double radius) const {
167 G4double frontHalfHeight;
168 if (fFieldHalfAngle == 0.) {
169 frontHalfHeight = radius / fZoomFactor;
170 }
171 else {
172 frontHalfHeight = nearDistance * std::tan (fFieldHalfAngle) / fZoomFactor;
173 }
174 return frontHalfHeight;
175}

200void G4ViewParameters::SetVisibleDensity (G4double visibleDensity) {
201 const G4double reasonableMaximum = 10.0 * g / cm3;
202 if (visibleDensity < 0) {
203 G4cout << "G4ViewParameters::SetVisibleDensity: attempt to set negative "
204 "density - ignored." << G4endl;
205 }
206 else {
207 if (visibleDensity > reasonableMaximum) {
208 G4cout << "G4ViewParameters::SetVisibleDensity: density > "
209 << G4BestUnit (reasonableMaximum, "Volumic Mass")
210 << " - did you mean this?"
211 << G4endl;
212 }
213 fVisibleDensity = visibleDensity;
214 }

229void G4ViewParameters::SetViewAndLights
230(const G4Vector3D& viewpointDirection) {
231
232 fViewpointDirection = viewpointDirection;
233
234 // If the requested viewpoint direction is parallel to the up
235 // vector, the orientation of the view is undefined...
236 if (fViewpointDirection.unit() * fUpVector.unit() > .9999) {
237 static G4bool firstTime = true;
238 if (firstTime) {
239 firstTime = false;
240 G4cout <<
241 "WARNING: Viewpoint direction is very close to the up vector direction."
242 "\n Consider setting the up vector to obtain definable behaviour."
243 << G4endl;
244 }
245 }
246
247 // Move the lights too if requested...
248 if (fLightsMoveWithCamera) {
249 G4Vector3D zprime = fViewpointDirection.unit ();
250 G4Vector3D xprime = (fUpVector.cross (zprime)).unit ();
251 G4Vector3D yprime = zprime.cross (xprime);
252 fActualLightpointDirection =
253 fRelativeLightpointDirection.x () * xprime +
254 fRelativeLightpointDirection.y () * yprime +
255 fRelativeLightpointDirection.x () * zprime;
256 } else {
257 fActualLightpointDirection = fRelativeLightpointDirection;
258 }
259}

267void G4ViewParameters::SetPan (G4double right, G4double up) {
268 G4Vector3D unitRight = (fUpVector.cross (fViewpointDirection)).unit();
269 G4Vector3D unitUp = (fViewpointDirection.cross (unitRight)).unit();
270 fCurrentTargetPoint = right * unitRight + up * unitUp;
271}
272
273void G4ViewParameters::IncrementPan (G4double right, G4double up) {
274 IncrementPan (right,up, 0);
275}
276
277void G4ViewParameters::IncrementPan (G4double right, G4double up, G4double distance) {
278 G4Vector3D unitRight = (fUpVector.cross (fViewpointDirection)).unit();
279 G4Vector3D unitUp = (fViewpointDirection.cross (unitRight)).unit();
280 fCurrentTargetPoint += right * unitRight + up * unitUp + distance * fViewpointDirection;
281}

283void G4ViewParameters::AddVisAttributesModifier
284(const G4ModelingParameters::VisAttributesModifier& vam) {
285 // If target exists with same signifier just change vis attributes.
286 G4bool duplicateTarget = false;
287 auto i = fVisAttributesModifiers.begin();
288 for (; i < fVisAttributesModifiers.end(); ++i) {
289 if (vam.GetPVNameCopyNoPath() == (*i).GetPVNameCopyNoPath() &&
290 vam.GetVisAttributesSignifier() == (*i).GetVisAttributesSignifier()) {
291 duplicateTarget = true;
292 break;
293 }
294 }
295 if (duplicateTarget) (*i).SetVisAttributes(vam.GetVisAttributes());
296 else fVisAttributesModifiers.push_back(vam);
297}

517G4String G4ViewParameters::TouchableCommands() const
518{
519 std::ostringstream oss;
520
521 oss << "#\n# Touchable commands";
522
523 const std::vector<G4ModelingParameters::VisAttributesModifier>& vams =
524 fVisAttributesModifiers;
525
526 if (vams.empty()) {
527 oss
528 << "\n# None"
529 << "\n/vis/viewer/clearVisAttributesModifiers";
530 oss << std::endl;
531 return oss.str();
532 }
533
534 oss
535 << "\n/vis/viewer/clearVisAttributesModifiers";
536
537 G4ModelingParameters::PVNameCopyNoPath lastPath;
538 std::vector<G4ModelingParameters::VisAttributesModifier>::const_iterator
539 iModifier;
540 for (iModifier = vams.begin();
541 iModifier != vams.end();
542 ++iModifier) {
543 const G4ModelingParameters::PVNameCopyNoPath& vamPath =
544 iModifier->GetPVNameCopyNoPath();
545 if (vamPath != lastPath) {
546 lastPath = vamPath;
547 oss << "\n/vis/set/touchable";
548 G4ModelingParameters::PVNameCopyNoPathConstIterator iVAM;

550 iVAM != vamPath.end();
551 ++iVAM) {
552 oss << ' ' << iVAM->GetName() << ' ' << iVAM->GetCopyNo();
553 }
554 }
555 const G4VisAttributes& vamVisAtts = iModifier->GetVisAttributes();
556 const G4Colour& c = vamVisAtts.GetColour();
557 switch (iModifier->GetVisAttributesSignifier()) {
558 case G4ModelingParameters::VASVisibility:

701void G4ViewParameters::PrintDifferences (const G4ViewParameters& v) const {
702
703 // Put performance-sensitive parameters first.
704 if (
705 // This first to optimise spin, etc.
706 (fViewpointDirection != v.fViewpointDirection) ||
707
708 // No particular order from here on.
709 (fDrawingStyle != v.fDrawingStyle) ||
710 (fAuxEdgeVisible != v.fAuxEdgeVisible) ||
711 (fCulling != v.fCulling) ||
712 (fCullInvisible != v.fCullInvisible) ||
713 (fDensityCulling != v.fDensityCulling) ||
714 (fVisibleDensity != v.fVisibleDensity) ||
715 (fCullCovered != v.fCullCovered) ||
716 (fCBDAlgorithmNumber != v.fCBDAlgorithmNumber) ||
717 (fSection != v.fSection) ||
718 (fNoOfSides != v.fNoOfSides) ||
719 (fUpVector != v.fUpVector) ||
720 (fFieldHalfAngle != v.fFieldHalfAngle) ||
721 (fZoomFactor != v.fZoomFactor) ||
722 (fScaleFactor != v.fScaleFactor) ||
723 (fCurrentTargetPoint != v.fCurrentTargetPoint) ||
724 (fDolly != v.fDolly) ||
725 (fRelativeLightpointDirection != v.fRelativeLightpointDirection) ||

1223/* Keep from :
1224 * ftp://ftp.trolltech.com/qt/source/qt-embedded-free-3.0.6.tar.gz/qt-embedded-free-3.0.6/src/kernel/qapplication_qws.cpp
1225 *
1226 * ParseGeometry parses strings of the form
1227 * "=<width>x<height>{+-}<xoffset>{+-}<yoffset>", where
1228 * width, height, xoffset, and yoffset are unsigned integers.
1229 * Example: "=80x24+300-49"
1230 * The equal sign is optional.
1231 * It returns a bitmask that indicates which of the four values
1232 * were actually found in the string. For each value found,
1233 * the corresponding argument is updated; for each value
1234 * not found, the corresponding argument is left unchanged.
1235 */
1236
1237int G4ViewParameters::ParseGeometry (
1238 const char *string,
1239 G4int *x,
1240 G4int *y,
1241 unsigned int *width,
1242 unsigned int *height)
1243{
1244
1245 G4int mask = fNoValue;
1246 char *strind;
1247 unsigned int tempWidth = 0;
1248 unsigned int tempHeight = 0;
1249 G4int tempX = 0;
1250 G4int tempY = 0;
1251 char *nextCharacter;
1252 if ( (string == NULL) || (*string == '\0')) {
1253 return(mask);
1254 }
1255 if (*string == '=')
1256 string++; /* ignore possible '=' at beg of geometry spec */
1257 strind = (char *)string;
1258 if (*strind != '+' && *strind != '-' && *strind != 'x') {
1259 tempWidth = ReadInteger(strind, &nextCharacter);
1260 if (strind == nextCharacter)
1261 return (0);
1262 strind = nextCharacter;
1263 mask |= fWidthValue;
1264 }
1265 if (*strind == 'x' || *strind == 'X') {
1266 strind++;
1267 tempHeight = ReadInteger(strind, &nextCharacter);
1268 if (strind == nextCharacter)
1269 return (0);
1270 strind = nextCharacter;
1271 mask |= fHeightValue;
1272 }
1273
1274 if ((*strind == '+') || (*strind == '-')) {
1275 if (*strind == '-') {
1276 strind++;
1277 tempX = -ReadInteger(strind, &nextCharacter);
1278 if (strind == nextCharacter)
1279 return (0);
1280 strind = nextCharacter;
1281 mask |= fXNegative;
1282
1283 }
1284 else
1285 { strind++;
1286 tempX = ReadInteger(strind, &nextCharacter);
1287 if (strind == nextCharacter)
1288 return(0);
1289 strind = nextCharacter;
1290 }
1291 mask |= fXValue;
1292 if ((*strind == '+') || (*strind == '-')) {
1293 if (*strind == '-') {
1294 strind++;
1295 tempY = -ReadInteger(strind, &nextCharacter);
1296 if (strind == nextCharacter)
1297 return(0);
1298 strind = nextCharacter;
1299 mask |= fYNegative;
1300 }
1301 else
1302 {
1303 strind++;
1304 tempY = ReadInteger(strind, &nextCharacter);
1305 if (strind == nextCharacter)
1306 return(0);
1307 strind = nextCharacter;
1308 }
1309 mask |= fYValue;
1310 }
1311 }
1312 /* If strind isn't at the end of the string the it's an invalid
1313 geometry specification. */
1314 if (*strind != '\0') return (0);
1315 if (mask & fXValue)
1316 *x = tempX;
1317 if (mask & fYValue)
1318 *y = tempY;
1319 if (mask & fWidthValue)
1320 *width = tempWidth;
1321 if (mask & fHeightValue)
1322 *height = tempHeight;
1323 return (mask);
1324}

1328 *
1329 */
1330G4int G4ViewParameters::ReadInteger(char *string, char **NextString)
1331{
1332 G4int Result = 0;
1333 G4int Sign = 1;
1334
1335 if (*string == '+')

1345 }
1346 *NextString = string;
1347 if (Sign >= 0)
1348 return (Result);
1349 else
1350 return (-Result);

1396 const G4double h00 = 2.*t*t*t - 3.*t*t +1;
1397 const G4double h10 = t*t*t -2.*t*t + t;
1398 const G4double h01 = -2.*t*t*t + 3.*t*t;
1399 const G4double h11 = t*t*t - t*t;

1406 // The Catmull-Rom cubic spline prescription is as follows:
1407 // Slope at first way point is v[1] - v[0].
1408 // Slope at last way point is v[n] - v[n-1].
1409 // Otherwise slope at way point i is 0.5*(v[i+1] - v[i-1]).
1410 // Result = h00*v[i] + h10*m[i] + h01*v[i+1] + h11*m[i+1],
1411 // where m[i] amd m[i+1] are the slopes at the start and end
1412 // of the interval for the particular value.
1413 // If (n == 1), linear interpolation results.
1414 // If (n == 2), quadratic interpolation results.
1415
1416 // Working variables
1417 G4double mi, mi1, real, x, y, z;

1419 // First, a crude interpolation of all parameters. Then, below, a
1420 // smooth interpolation of those for which it makes sense.
1421 holdingValues = t < 0.5? v[i]: v[i+1];

1423 // Catmull-Rom cubic spline interpolation
1424#define INTERPOLATE(param) \
1425 /* This works out the interpolated param in i'th interval */ \
1426 /* Assumes n >= 1 */ \
1427 if (i == 0) { \
1428 /* First interval */ \
1429 mi = v[1].param - v[0].param; \
1430 /* If there is only one interval, make start and end slopes equal */ \
1431 /* (This results in a linear interpolation) */ \
1432 if (n == 1) mi1 = mi; \
1433 /* else the end slope of the interval takes account of the next waypoint along */ \
1434 else mi1 = 0.5 * (v[2].param - v[0].param); \
1435 } else if (i >= n - 1) { \
1436 /* Similarly for last interval */ \
1437 mi1 = v[i+1].param - v[i].param; \
1438 /* If there is only one interval, make start and end slopes equal */ \
1439 if (n == 1) mi = mi1; \
1440 /* else the start slope of the interval takes account of the previous waypoint */ \
1441 else mi = 0.5 * (v[i+1].param - v[i-1].param); \
1442 } else { \
1443 /* Full Catmull-Rom slopes use previous AND next waypoints */ \
1444 mi = 0.5 * (v[i+1].param - v[i-1].param); \
1445 mi1 = 0.5 * (v[i+2].param - v[i ].param); \
1446 } \
1447 real = h00 * v[i].param + h10 * mi + h01 * v[i+1].param + h11 * mi1;
