50G4bool G4Scene::AddRunDurationModel (G4VModel* pModel, G4bool warn)
51{
52 std::vector<Model>::const_iterator i;
53 for (i = fRunDurationModelList.begin ();
54 i != fRunDurationModelList.end (); ++i) {
55 if (pModel -> GetGlobalDescription () ==
56 i->fpModel->GetGlobalDescription ()) break;
57 }
58 if (i != fRunDurationModelList.end ()) {
59 if (warn) {
60 G4cout << "G4Scene::AddRunDurationModel: model \""
61 << pModel -> GetGlobalDescription ()
62 << "\"\n is already in the run-duration list of scene \""
63 << fName
64 << "\"."
65 << G4endl;
66 }
67 return false;
68 }
69
70 for (i = fRunDurationModelList.begin ();
71 i != fRunDurationModelList.end (); ++i) {
72 if (pModel -> GetGlobalTag () ==
73 i->fpModel->GetGlobalTag ()) break;
74 }
75 if (i != fRunDurationModelList.end ()) {
76 if (warn) {
77 G4cout
78 << "G4Scene::AddRunDurationModel: The tag \""
79 << pModel->GetGlobalTag()
80 << "\"\n duplicates one already in scene \""
81 << fName
82 <<
83 "\".\n This may be intended but if not, you may inspect the scene with"
84 "\n \"/vis/scene/list\" and deactivate unwanted models with"
85 "\n \"/vis/scene/activateModel\". Or create a new scene."
86 << G4endl;
87 }
88 }
89
90 fRunDurationModelList.push_back (Model(pModel));
91
92 CalculateExtent ();
93
94 return true;
95}

97void G4Scene::CalculateExtent ()
98{
99 G4BoundingSphereScene boundingSphereScene;
100
101 for (size_t i = 0; i < fRunDurationModelList.size(); i++) {
102 if (fRunDurationModelList[i].fActive) {
103 G4VModel* model = fRunDurationModelList[i].fpModel;
104 if (model -> Validate()) { // Validates and also recomputes extent.
105 const G4VisExtent& thisExtent = model -> GetExtent ();
106 G4double thisRadius = thisExtent.GetExtentRadius ();
107 if (thisRadius > 0.) {
108 G4Point3D thisCentre = thisExtent.GetExtentCentre ();
109 thisCentre.transform (model -> GetTransformation ());
110 boundingSphereScene.AccrueBoundingSphere (thisCentre, thisRadius);
111 }
112 } else {
113 G4ExceptionDescription ed;
114 ed << "Invalid model \"" << model->GetGlobalDescription()
115 << "\".\n Not included in extent calculation.";
116 G4Exception
117 ("G4Scene::CalculateExtent",
118 "visman0201", JustWarning, ed);
119 }
120 }
121 }
122
123 for (size_t i = 0; i < fEndOfEventModelList.size(); i++) {
124 if (fEndOfEventModelList[i].fActive) {
125 G4VModel* model = fEndOfEventModelList[i].fpModel;
126 if (model -> Validate()) { // Validates and also recomputes extent.
127 const G4VisExtent& thisExtent = model -> GetExtent ();
128 G4double thisRadius = thisExtent.GetExtentRadius ();
129 if (thisRadius > 0.) {
130 G4Point3D thisCentre = thisExtent.GetExtentCentre ();
131 thisCentre.transform (model -> GetTransformation ());
132 boundingSphereScene.AccrueBoundingSphere (thisCentre, thisRadius);
133 }
134 } else {
135 G4ExceptionDescription ed;
136 ed << "Invalid model \"" << model->GetGlobalDescription()
137 << "\".\n Not included in extent calculation.";
138 G4Exception
139 ("G4Scene::CalculateExtent",
140 "visman0201", JustWarning, ed);
141 }
142 }
143 }
144
145 for (size_t i = 0; i < fEndOfRunModelList.size(); i++) {
146 if (fEndOfRunModelList[i].fActive) {
147 G4VModel* model = fEndOfRunModelList[i].fpModel;
148 if (model -> Validate()) { // Validates and also recomputes extent.
149 const G4VisExtent& thisExtent = model -> GetExtent ();
150 G4double thisRadius = thisExtent.GetExtentRadius ();
151 if (thisRadius > 0.) {
152 G4Point3D thisCentre = thisExtent.GetExtentCentre ();
153 thisCentre.transform (model -> GetTransformation ());
154 boundingSphereScene.AccrueBoundingSphere (thisCentre, thisRadius);
155 }
156 } else {
157 G4ExceptionDescription ed;
158 ed << "Invalid model \"" << model->GetGlobalDescription()
159 << "\".\n Not included in extent calculation.";
160 G4Exception
161 ("G4Scene::CalculateExtent",
162 "visman0201", JustWarning, ed);
163 }
164 }
165 }
166
167 fExtent = boundingSphereScene.GetBoundingSphereExtent ();
168 fStandardTargetPoint = fExtent.GetExtentCentre ();
169 if (fExtent.GetExtentRadius() <= 0.) {
170 G4Exception
171 ("G4Scene::CalculateExtent",
172 "visman0202", JustWarning,
173 "Scene has no extent. Please activate or add something."
174 "\nThe camera needs to have something to point at!"
175 "\nAdd a volume. (You may need \"/run/initialize\".)"
176 "\nOr use \"/vis/scene/add/extent\"."
177 "\n\"/vis/scene/list\" to see list of models.");
178 }
179}

181G4bool G4Scene::AddWorldIfEmpty (G4bool warn) {
182 G4bool successful = true;
183 if (IsEmpty ()) {
184 successful = false;
185 G4VPhysicalVolume* pWorld =
186 G4TransportationManager::GetTransportationManager ()
187 -> GetNavigatorForTracking () -> GetWorldVolume ();
188 if (pWorld) {
189 const G4VisAttributes* pVisAttribs =
190 pWorld -> GetLogicalVolume () -> GetVisAttributes ();
191 if (!pVisAttribs || pVisAttribs -> IsVisible ()) {
192 if (warn) {
193 G4cout <<
194 "Your \"world\" has no vis attributes or is marked as visible."
195 "\n For a better view of the contents, mark the world as"
196 " invisible, e.g.,"
197 "\n myWorldLogicalVol ->"
198 " SetVisAttributes (G4VisAttributes::GetInvisible());"
199 << G4endl;
200 }
201 }
202 successful = AddRunDurationModel (new G4PhysicalVolumeModel (pWorld));
203 // Note: default depth and no modeling parameters.
204 if (successful) {
205 if (warn) {
206 G4cout <<
207 "G4Scene::AddWorldIfEmpty: The scene was empty of run-duration models."
208 "\n \"world\" has been added.";
209 G4cout << G4endl;
210 }
211 }
212 }
213 }
214 return successful;
215}

217G4bool G4Scene::AddEndOfEventModel (G4VModel* pModel, G4bool warn) {
218 G4int i, nModels = fEndOfEventModelList.size ();
219 for (i = 0; i < nModels; i++) {
220 if (pModel -> GetGlobalDescription () ==
221 fEndOfEventModelList[i].fpModel -> GetGlobalDescription ()) break;
222 }
223 if (i < nModels) {
224 if (warn) {
225 G4cout << "G4Scene::AddEndOfEventModel: a model \""
226 << pModel -> GetGlobalDescription ()
227 << "\"\n is already in the end-of-event list of scene \""
228 << fName << "\"."
229 << G4endl;
230 }
231 return false;
232 }
233 fEndOfEventModelList.push_back (Model(pModel));
234 return true;
235}

257std::ostream& operator << (std::ostream& os, const G4Scene& scene) {
258
259 size_t i;
260
261 os << "Scene data:";
262
263 os << "\n Run-duration model list:";
264 for (i = 0; i < scene.fRunDurationModelList.size (); i++) {
265 if (scene.fRunDurationModelList[i].fActive) os << "\n Active: ";
266 else os << "\n Inactive: ";
267 os << *(scene.fRunDurationModelList[i].fpModel);
268 }
269
270 os << "\n End-of-event model list:";
271 for (i = 0; i < scene.fEndOfEventModelList.size (); i++) {
272 if (scene.fEndOfEventModelList[i].fActive) os << "\n Active: ";
273 else os << "\n Inactive: ";
274 os << *(scene.fEndOfEventModelList[i].fpModel);
275 }
276
277 os << "\n End-of-run model list:";
278 for (i = 0; i < scene.fEndOfRunModelList.size (); i++) {
279 if (scene.fEndOfRunModelList[i].fActive) os << "\n Active: ";
280 else os << "\n Inactive: ";
281 os << *(scene.fEndOfRunModelList[i].fpModel);
282 }
283
284 os << "\n Extent or bounding box: " << scene.fExtent;
285
286 os << "\n Standard target point: " << scene.fStandardTargetPoint;
287
288 os << "\n End of event action set to \"";
289 if (scene.fRefreshAtEndOfEvent) os << "refresh\"";
290 else {
291 os << "accumulate (maximum number of kept events: ";
292 if (scene.fMaxNumberOfKeptEvents >= 0) os << scene.fMaxNumberOfKeptEvents;
293 else os << "unlimited";
294 os << ")";
295 }
296
297 os << "\n End of run action set to \"";
298 if (scene.fRefreshAtEndOfRun) os << "refresh";
299 else os << "accumulate";
300 os << "\"";
301
302 return os;
303}

305G4bool G4Scene::operator != (const G4Scene& scene) const {
306 if (
307 (fRunDurationModelList.size () !=
308 scene.fRunDurationModelList.size ()) ||
309 (fEndOfEventModelList.size () !=
310 scene.fEndOfEventModelList.size ()) ||
311 (fEndOfRunModelList.size () !=
312 scene.fEndOfRunModelList.size ()) ||
313 (fExtent != scene.fExtent) ||
314 !(fStandardTargetPoint == scene.fStandardTargetPoint) ||
315 fRefreshAtEndOfEvent != scene.fRefreshAtEndOfEvent ||
316 fRefreshAtEndOfRun != scene.fRefreshAtEndOfRun ||
317 fMaxNumberOfKeptEvents != scene.fMaxNumberOfKeptEvents
318 ) return true;
319
320 /* A complete comparison should, perhaps, include a comparison of
321 individual models, but it is not easy to implement operator!= for
322 all models. Also, it would be unfeasible to ask users to
323 implement opeerator!= if we ever get round to allowing
324 user-defined models. Moreover, there is no editing of G4Scene
325 objects, apart from changing fRefreshAtEndOfEvent, etc; as far as
326 models are concerned, all you can ever do is add them, so a test
327 on size (above) is enough.
328
329 for (size_t i = 0; i < fRunDurationModelList.size (); i++) {
330 if (fRunDurationModelList[i] != scene.fRunDurationModelList[i])
331 return true;
332 }
333
334 for (size_t i = 0; i < fEndOfEventModelList.size (); i++) {
335 if (fEndOfEventModelList[i] != scene.fEndOfEventModelList[i])
336 return true;
337 }
338
339 for (size_t i = 0; i < fEndOfRunModelList.size (); i++) {
340 if (fEndOfRunModelList[i] != scene.fEndOfRunModelList[i])
341 return true;
342 }
343 */
344
345 return false;
346}
