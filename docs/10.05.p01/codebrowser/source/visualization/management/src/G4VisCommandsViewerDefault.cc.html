38G4VisCommandViewerDefaultHiddenEdge::G4VisCommandViewerDefaultHiddenEdge()
39{
40 G4bool omitable;
41 fpCommand = new G4UIcmdWithABool("/vis/viewer/default/hiddenEdge", this);
42 fpCommand->SetGuidance("Default hiddenEdge drawing for future viewers.");
43 fpCommand->SetGuidance
44 ("Edges become hidden/seen in wireframe or surface mode.");
45 fpCommand->SetParameterName("hidden-edge",omitable = true);
46 fpCommand->SetDefaultValue(true);
47}
48
49G4VisCommandViewerDefaultHiddenEdge::~G4VisCommandViewerDefaultHiddenEdge()
50{
51 delete fpCommand;
52}
53
54G4String G4VisCommandViewerDefaultHiddenEdge::GetCurrentValue(G4UIcommand*)
55{
56 return "";
57}
58
59void G4VisCommandViewerDefaultHiddenEdge::SetNewValue(G4UIcommand*, G4String newValue)
60{
61 // Follows /vis/viewer/set/hiddenEdge
62
63 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
64 G4ViewParameters vp = fpVisManager->GetDefaultViewParameters();
65 G4ViewParameters::DrawingStyle existingStyle = vp.GetDrawingStyle();
66
67 if (G4UIcommand::ConvertToBool(newValue)) {
68 switch (existingStyle) {
69 case G4ViewParameters::wireframe:
70 vp.SetDrawingStyle(G4ViewParameters::hlr);
71 break;
72 case G4ViewParameters::hlr:
73 break;
74 case G4ViewParameters::hsr:
75 vp.SetDrawingStyle(G4ViewParameters::hlhsr);
76 break;
77 case G4ViewParameters::hlhsr:
78 break;
79 }
80 }
81 else {
82 switch (existingStyle) {
83 case G4ViewParameters::wireframe:
84 break;
85 case G4ViewParameters::hlr:
86 vp.SetDrawingStyle(G4ViewParameters::wireframe);
87 break;
88 case G4ViewParameters::hsr:
89 break;
90 case G4ViewParameters::hlhsr:
91 vp.SetDrawingStyle(G4ViewParameters::hsr);
92 break;
93 }
94 }
95
96 fpVisManager->SetDefaultViewParameters(vp);
97
98 if (verbosity >= G4VisManager::confirmations) {
99 G4cout << "Default drawing style set to " << vp.GetDrawingStyle()
100 << G4endl;
101 }
102}

106G4VisCommandViewerDefaultStyle::G4VisCommandViewerDefaultStyle()
107{
108 G4bool omitable;
109 fpCommand = new G4UIcmdWithAString("/vis/viewer/default/style", this);
110 fpCommand->SetGuidance("Default drawing style for future viewers.");
111 fpCommand->SetGuidance
112 ("Set style of drawing - w[ireframe] or s[urface].");
113 fpCommand->SetGuidance
114 ("(Default hidden line drawing is controlled by \"/vis/viewer/default/hiddenEdge\".)");
115 fpCommand->SetParameterName ("style",omitable = false);
116 fpCommand->SetCandidates("w wireframe s surface");
117}

129void G4VisCommandViewerDefaultStyle::SetNewValue(G4UIcommand*, G4String newValue)
130{
131 // Follows /vis/viewer/set/style
132
133 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
134 G4ViewParameters vp = fpVisManager->GetDefaultViewParameters();
135 G4ViewParameters::DrawingStyle existingStyle = vp.GetDrawingStyle();
136
137 const size_t iPos0 = 0;
138 if (newValue[iPos0] == 'w') { // "wireframe"
139 switch (existingStyle) {
140 case G4ViewParameters::wireframe:
141 break;
142 case G4ViewParameters::hlr:
143 break;
144 case G4ViewParameters::hsr:
145 vp.SetDrawingStyle(G4ViewParameters::wireframe);
146 break;
147 case G4ViewParameters::hlhsr:
148 vp.SetDrawingStyle(G4ViewParameters::hlr);
149 break;
150 }
151 }
152 else if (newValue[iPos0] == 's') { // "surface"
153 switch (existingStyle) {
154 case G4ViewParameters::wireframe:
155 vp.SetDrawingStyle(G4ViewParameters::hsr);
156 break;
157 case G4ViewParameters::hlr:
158 vp.SetDrawingStyle(G4ViewParameters::hlhsr);
159 break;
160 case G4ViewParameters::hsr:
161 break;
162 case G4ViewParameters::hlhsr:
163 break;
164 }
165 }
166 else {
167 if (verbosity >= G4VisManager::errors) {
168 G4cerr << "ERROR: \"" << newValue << "\" not recognised."
169 " Looking for 'w' or 's' first character." << G4endl;
170 }
171 return;
172 }
173
174 fpVisManager->SetDefaultViewParameters(vp);
175
176 if (verbosity >= G4VisManager::confirmations) {
177 G4cout << "Default drawing style set to " << vp.GetDrawingStyle()
178 << G4endl;
179 }
180}
