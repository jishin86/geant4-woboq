73 parameter -> SetDefaultValue ("all");

88void G4VisCommandSceneActivateModel::SetNewValue (G4UIcommand*,
89 G4String newValue) {
90
91 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
92
93 G4String searchString, activateString;
94 std::istringstream is (newValue);
95 is >> searchString >> activateString;
96 G4bool activate = G4UIcommand::ConvertToBool(activateString);
97
98 G4Scene* pScene = fpVisManager->GetCurrentScene();
99 if (!pScene) {
100 if (verbosity >= G4VisManager::errors) {
101 G4cerr << "ERROR: No current scene. Please create one." << G4endl;
102 }
103 return;
104 }
105
106 G4VSceneHandler* pSceneHandler = fpVisManager->GetCurrentSceneHandler();
107 if (!pSceneHandler) {
108 if (verbosity >= G4VisManager::errors) {
109 G4cerr << "ERROR: No current sceneHandler. Please create one." << G4endl;
110 }
111 return;
112 }
113
114 if (searchString == "all" && !activate) {
115 if (verbosity >= G4VisManager::warnings) {
116 G4cout <<
117 "WARNING: You are not allowed to de-activate all models."
118 "\n Command ignored."
119 << G4endl;
120 }
121 return;
122 }
123
124 G4bool any = false;
125
126 std::vector<G4Scene::Model>& runDurationModelList =
127 pScene->SetRunDurationModelList();
128 for (size_t i = 0; i < runDurationModelList.size(); i++) {
129 const G4String& modelName =
130 runDurationModelList[i].fpModel->GetGlobalDescription();
131 if (searchString == "all" || modelName.find(searchString)
132 != std::string::npos) {
133 any = true;
134 runDurationModelList[i].fActive = activate;
135 if (verbosity >= G4VisManager::warnings) {
136 G4cout << "Model \"" << modelName;
137 if (activate) G4cout << "\" activated.";
138 else G4cout << "\" de-activated.";
139 G4cout << G4endl;
140 }
141 }
142 }
143
144 std::vector<G4Scene::Model>& endOfEventModelList =
145 pScene->SetEndOfEventModelList();
146 for (size_t i = 0; i < endOfEventModelList.size(); i++) {
147 const G4String& modelName =
148 endOfEventModelList[i].fpModel->GetGlobalDescription();
149 if (searchString == "all" || modelName.find(searchString)
150 != std::string::npos) {
151 any = true;
152 endOfEventModelList[i].fActive = activate;
153 if (verbosity >= G4VisManager::warnings) {
154 G4cout << "Model \"" << modelName;
155 if (activate) G4cout << "\" activated.";
156 else G4cout << "\" de-activated.";
157 G4cout << G4endl;
158 }
159 }
160 }
161
162 std::vector<G4Scene::Model>& endOfRunModelList =
163 pScene->SetEndOfRunModelList();
164 for (size_t i = 0; i < endOfRunModelList.size(); i++) {
165 const G4String& modelName =
166 endOfRunModelList[i].fpModel->GetGlobalDescription();
167 if (searchString == "all" || modelName.find(searchString)
168 != std::string::npos) {
169 any = true;
170 endOfRunModelList[i].fActive = activate;
171 if (verbosity >= G4VisManager::warnings) {
172 G4cout << "Model \"" << modelName;
173 if (activate) G4cout << "\" activated.";
174 else G4cout << "\" de-activated.";
175 G4cout << G4endl;
176 }
177 }
178 }
179
180 if (!any) {
181 if (verbosity >= G4VisManager::warnings) {
182 G4cout << "WARNING: No match found." << G4endl;
183 }
184 return;
185 }
186
187 CheckSceneAndNotifyHandlers (pScene);
188}

216void G4VisCommandSceneCreate::SetNewValue (G4UIcommand*, G4String newValue) {
217
218 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
219
220 G4String& newName = newValue;
221 G4String nextName = NextName ();
222
223 if (newName == "") {
224 newName = nextName;
225 }
226 if (newName == nextName) fId++;
227
228 G4SceneList& sceneList = fpVisManager -> SetSceneList ();
229 G4int iScene, nScenes = sceneList.size ();
230 for (iScene = 0; iScene < nScenes; iScene++) {
231 if (sceneList [iScene] -> GetName () == newName) break;
232 }
233 if (iScene < nScenes) {
234 if (verbosity >= G4VisManager::warnings) {
235 G4cout << "WARNING: Scene \"" << newName << "\" already exists."
236 << "\n New scene not created."
237 << G4endl;
238 }
239 } else {
240
241 // Add empty scene data object to list...
242 G4Scene* pScene = new G4Scene (newName);
243 sceneList.push_back (pScene);
244 fpVisManager -> SetCurrentScene (pScene);
245
246 if (verbosity >= G4VisManager::confirmations) {
247 G4cout << "New empty scene \"" << newName << "\" created." << G4endl;
248 }
249 }

284void G4VisCommandSceneEndOfEventAction::SetNewValue (G4UIcommand*,
285 G4String newValue) {
286
287 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
288
289 G4String action;
290 G4int maxNumberOfKeptEvents;
291 std::istringstream is (newValue);
292 is >> action >> maxNumberOfKeptEvents;
293
294 G4Scene* pScene = fpVisManager->GetCurrentScene();
295 if (!pScene) {
296 if (verbosity >= G4VisManager::errors) {
297 G4cerr << "ERROR: No current scene. Please create one." << G4endl;
298 }
299 return;
300 }
301
302 G4VSceneHandler* pSceneHandler = fpVisManager->GetCurrentSceneHandler();
303 if (!pSceneHandler) {
304 if (verbosity >= G4VisManager::errors) {
305 G4cerr << "ERROR: No current sceneHandler. Please create one." << G4endl;
306 }
307 return;
308 }
309
310 if (action == "accumulate") {
311 pScene->SetRefreshAtEndOfEvent(false);
312 pScene->SetMaxNumberOfKeptEvents(maxNumberOfKeptEvents);
313 }
314 else if (action == "refresh") {
315 if (!pScene->GetRefreshAtEndOfRun()) {
316 if (verbosity >= G4VisManager::errors) {
317 G4cerr <<
318 "ERROR: Cannot refresh events unless runs refresh too."
319 "\n Use \"/vis/scene/endOfRun refresh\"."
320 << G4endl;
321 }
322 } else {
323 pScene->SetRefreshAtEndOfEvent(true);
324 pScene->SetMaxNumberOfKeptEvents(maxNumberOfKeptEvents);
325 pSceneHandler->SetMarkForClearingTransientStore(true);
326 }
327 }
328 else {
329 if (verbosity >= G4VisManager::errors) {
330 G4cerr <<
331 "ERROR: unrecognised parameter \"" << action << "\"."
332 << G4endl;
333 }
334 return;
335 }
336
337 // Change of transients behaviour, so...
338 fpVisManager->ResetTransientsDrawnFlags();
339
340 // Are there any events currently kept...
341 size_t nCurrentlyKept = 0;
342 G4RunManager* runManager = G4RunManager::GetRunManager();
343#ifdef G4MULTITHREADED
344 if(G4Threading::IsMultithreadedApplication())
345 { runManager = G4MTRunManager::GetMasterRunManager(); }
346#endif
347 if (runManager) {
348 const G4Run* currentRun = runManager->GetCurrentRun();
349 if (currentRun) {
350 const std::vector<const G4Event*>* events =
351 currentRun->GetEventVector();
352 if (events) nCurrentlyKept = events->size();
353 }
354 }
355
356 if (verbosity >= G4VisManager::confirmations) {
357 G4cout << "End of event action set to ";
358 if (pScene->GetRefreshAtEndOfEvent()) G4cout << "\"refresh\".";
359 else {
360 G4cout << "\"accumulate\"."
361 "\n Maximum number of events to be kept: "
362 << maxNumberOfKeptEvents
363 << " (unlimited if negative)."
364 "\n This may be changed with, e.g., "
365 "\"/vis/scene/endOfEventAction accumulate 1000\".";
366 }
367 G4cout << G4endl;
368 }
369
370 if (!pScene->GetRefreshAtEndOfEvent() &&
371 maxNumberOfKeptEvents != 0 &&
372 verbosity >= G4VisManager::warnings) {
373 G4cout << "WARNING: ";
374 if (nCurrentlyKept) {
375 G4cout <<
376 "\n There are currently " << nCurrentlyKept
377 << " events kept for refreshing and/or reviewing.";
378 } else {
379 G4cout << "The vis manager will keep ";
380 if (maxNumberOfKeptEvents < 0) G4cout << "an unlimited number of";
381 else G4cout << "up to " << maxNumberOfKeptEvents;
382 G4cout << " events.";
383 if (maxNumberOfKeptEvents > 1 || maxNumberOfKeptEvents < 0)
384 G4cout <<
385 "\n This may use a lot of memory."
386 "\n It may be changed with, e.g., "
387 "\"/vis/scene/endOfEventAction accumulate 10\".";
388 }
389 G4cout << G4endl;
390 }
391}

507void G4VisCommandSceneList::SetNewValue (G4UIcommand*, G4String newValue) {
508 G4String name, verbosityString;
509 std::istringstream is (newValue);
510 is >> name >> verbosityString;
511 G4VisManager::Verbosity verbosity =
512 fpVisManager->GetVerbosityValue(verbosityString);
513 const G4Scene* currentScene = fpVisManager -> GetCurrentScene ();
514 G4String currentName;
515 if (currentScene) currentName = currentScene->GetName();
516
517 G4SceneList& sceneList = fpVisManager -> SetSceneList ();
518 G4int iScene, nScenes = sceneList.size ();
519 G4bool found = false;
520 for (iScene = 0; iScene < nScenes; iScene++) {
521 G4Scene* pScene = sceneList [iScene];
522 const G4String& iName = pScene -> GetName ();
523 if (name != "all") {
524 if (name != iName) continue;
525 }
526 found = true;
527 if (iName == currentName) {
528 G4cout << " (current)";
529 }
530 else {
531 G4cout << " ";
532 }
533 G4cout << " scene \"" << iName << "\"";
534 if (verbosity >= G4VisManager::warnings) {
535 G4int i;
536 G4cout << "\n Run-duration models:";
537 G4int nRunModels = pScene -> GetRunDurationModelList ().size ();
538 if (nRunModels == 0) {
539 G4cout << " none.";
540 }
541 for (i = 0; i < nRunModels; i++) {
542 if (pScene -> GetRunDurationModelList()[i].fActive)
543 G4cout << "\n Active: ";
544 else G4cout << "\n Inactive: ";
545 G4VModel* pModel = pScene -> GetRunDurationModelList()[i].fpModel;
546 G4cout << pModel -> GetGlobalDescription ();
547 }
548 G4cout << "\n End-of-event models:";
549 G4int nEOEModels = pScene -> GetEndOfEventModelList ().size ();
550 if (nEOEModels == 0) {
551 G4cout << " none.";
552 }
553 for (i = 0; i < nEOEModels; i++) {
554 if (pScene -> GetEndOfEventModelList()[i].fActive)
555 G4cout << "\n Active: ";
556 else G4cout << "\n Inactive: ";
557 G4VModel* pModel = pScene -> GetEndOfEventModelList()[i].fpModel;
558 G4cout << pModel -> GetGlobalDescription ();
559 }
560 G4cout << "\n End-of-run models:";
561 G4int nEORModels = pScene -> GetEndOfRunModelList ().size ();
562 if (nEORModels == 0) {
563 G4cout << " none.";
564 }
565 for (i = 0; i < nEORModels; i++) {
566 if (pScene -> GetEndOfRunModelList()[i].fActive)
567 G4cout << "\n Active: ";
568 else G4cout << "\n Inactive: ";
569 G4VModel* pModel = pScene -> GetEndOfRunModelList()[i].fpModel;
570 G4cout << pModel -> GetGlobalDescription ();
571 }
572 }
573 if (verbosity >= G4VisManager::parameters) {
574 G4cout << "\n " << *sceneList [iScene];
575 }
576 G4cout << G4endl;
577 }
578 if (!found) {
579 G4cout << "No scenes found";
580 if (name != "all") {
581 G4cout << " of name \"" << name << "\"";
582 }
583 G4cout << "." << G4endl;
584 }
585}

605 fpCommand -> SetGuidance

609 omitable = true);

627void G4VisCommandSceneNotifyHandlers::SetNewValue (G4UIcommand*,
628 G4String newValue) {
629
630 G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
631
632 G4String sceneName, refresh_flush;
633 std::istringstream is (newValue);
634 is >> sceneName >> refresh_flush;
635 G4bool flush = false;
636 if (refresh_flush(0) == 'f') flush = true;
637
638 const G4SceneList& sceneList = fpVisManager -> GetSceneList ();
639 G4SceneHandlerList& sceneHandlerList =
640 fpVisManager -> SetAvailableSceneHandlers ();
641
642 // Check scene name.
643 const G4int nScenes = sceneList.size ();
644 G4int iScene;
645 for (iScene = 0; iScene < nScenes; iScene++) {
646 G4Scene* scene = sceneList [iScene];
647 if (sceneName == scene -> GetName ()) break;
648 }
649 if (iScene >= nScenes ) {
650 if (verbosity >= G4VisManager::warnings) {
651 G4cout << "WARNING: Scene \"" << sceneName << "\" not found."
652 "\n /vis/scene/list to see scenes."
653 << G4endl;
654 }
655 return;
656 }
657
658 // Store current context...
659 G4VSceneHandler* pCurrentSceneHandler =
660 fpVisManager -> GetCurrentSceneHandler();
661 if (!pCurrentSceneHandler) {
662 if (verbosity >= G4VisManager::warnings) {
663 G4cout << "WARNING: No current scene handler."
664 << G4endl;
665 }
666 return;
667 }
668 G4VViewer* pCurrentViewer = fpVisManager -> GetCurrentViewer();
669 if (!pCurrentViewer) {
670 if (verbosity >= G4VisManager::warnings) {
671 G4cout << "WARNING: No current viewer."
672 << G4endl;
673 }
674 return;
675 }
676 G4Scene* pCurrentScene = fpVisManager -> GetCurrentScene();
677 if (!pCurrentScene) {
678 if (verbosity >= G4VisManager::warnings) {
679 G4cout << "WARNING: No current scene."
680 << G4endl;
681 }
682 return;
683 }
684
685 G4VisManager::Verbosity currentVerbosity = fpVisManager -> GetVerbosity();
686
687 // Suppress messages during this process (only print errors)...
688 //fpVisManager -> SetVerboseLevel(G4VisManager::errors);
689
690 // For each scene handler, if it contains the scene, clear and
691 // rebuild the graphical database, then for each viewer set (make
692 // current), clear, (re)draw, and show.
693 const G4int nSceneHandlers = sceneHandlerList.size ();
694 for (G4int iSH = 0; iSH < nSceneHandlers; iSH++) {
695 G4VSceneHandler* aSceneHandler = sceneHandlerList [iSH];
696 G4Scene* aScene = aSceneHandler -> GetScene ();
697 if (aScene) {
698 const G4String& aSceneName = aScene -> GetName ();
699 if (sceneName == aSceneName) {
700 aScene->CalculateExtent(); // Check and recalculate extent
701 G4ViewerList& viewerList = aSceneHandler -> SetViewerList ();
702 const G4int nViewers = viewerList.size ();
703 for (G4int iV = 0; iV < nViewers; iV++) {
704 G4VViewer* aViewer = viewerList [iV];
705 // Force rebuild of graphical database, if any.
706 aViewer -> NeedKernelVisit();
707 if (aViewer->GetViewParameters().IsAutoRefresh()) {
708 aSceneHandler -> SetCurrentViewer (aViewer);
709 // Ensure consistency of vis manager...
710 fpVisManager -> SetCurrentViewer(aViewer);
711 fpVisManager -> SetCurrentSceneHandler(aSceneHandler);
712 fpVisManager -> SetCurrentScene(aScene);
713 aViewer -> SetView ();
714 aViewer -> ClearView ();
715 aViewer -> DrawView ();
716 if (flush) aViewer -> ShowView ();
717 if (verbosity >= G4VisManager::confirmations) {
718 G4cout << "Viewer \"" << aViewer -> GetName ()
719 << "\" of scene handler \"" << aSceneHandler -> GetName ()
720 << "\"\n ";
721 if (flush) G4cout << "flushed";
722 else G4cout << "refreshed";
723 G4cout << " at request of scene \"" << sceneName
724 << "\"." << G4endl;
725 }
726 } else {
727 if (verbosity >= G4VisManager::confirmations) {
728 G4cout << "NOTE: The scene, \""
729 << sceneName
730 << "\", of viewer \""
731 << aViewer -> GetName ()
732 << "\"\n of scene handler \""
733 << aSceneHandler -> GetName ()
734 << "\" has changed. To see effect,"
735 << "\n \"/vis/viewer/select "
736 << aViewer -> GetShortName ()
737 << "\" and \"/vis/viewer/rebuild\"."
738 << G4endl;
739 }
740 }
741 }
742 }
743 }
744 else {
745 if (verbosity >= G4VisManager::warnings) {
746 G4cout << "WARNING: G4VisCommandSceneNotifyHandlers: scene handler \""
747 << aSceneHandler->GetName()
748 << "\" has a null scene."
749 << G4endl;
750 }
751 }
752 }
753
754 // Reclaim original context - but set viewer first, then scene
755 // handler, because the latter might have been created very recently
756 // and, not yet having a viewer, the current viewer will,
757 // temporarily, refer to another scene handler. SetCurrentViewer
758 // actually resets the scene handler, which is what we don't want,
759 // so we set it again on the next line...
760 fpVisManager -> SetCurrentViewer(pCurrentViewer);
761 fpVisManager -> SetCurrentSceneHandler(pCurrentSceneHandler);
762 fpVisManager -> SetCurrentScene(pCurrentScene);
763 fpVisManager -> SetVerboseLevel(currentVerbosity);
764 // Take care of special case of scene handler with no viewer yet.
765 if (pCurrentSceneHandler) {
766 G4ViewerList& viewerList = pCurrentSceneHandler -> SetViewerList ();
767 const G4int nViewers = viewerList.size ();
768 if (nViewers) {
769 pCurrentSceneHandler -> SetCurrentViewer (pCurrentViewer);
770 if (pCurrentViewer && pCurrentSceneHandler->GetScene()) {
771 pCurrentViewer -> SetView ();
772 }
773 }
774 }
775}
