124{
125 fpTrajDrawModelMgr = new G4VisModelManager<G4VTrajectoryModel>("/vis/modeling/trajectories");

131 ("Simple graded message scheme - digit or string (1st character defines):");
132 VerbosityGuidanceStrings.push_back
133 (" 0) quiet, // Nothing is printed.");

135 (" 1) startup, // Startup and endup messages are printed...");
136 VerbosityGuidanceStrings.push_back

138 VerbosityGuidanceStrings.push_back

140 VerbosityGuidanceStrings.push_back
141 (" 4) confirmations, // ...and confirming messages...");
142 VerbosityGuidanceStrings.push_back

164 << VerbosityString(fVerbosity)
165 << "\"..." << G4endl;
166 }
167
168 // Note: The specific graphics systems must be instantiated in a
169 // higher level library to avoid circular dependencies. Also,
170 // some specifically need additional external libararies that the
171 // user must supply. Therefore we ask the user to implement
172 // RegisterGraphicsSystems() and RegisterModelFactories()
173 // in a subclass. We have to wait for the subclass to instantiate
174 // so RegisterGraphicsSystems() cannot be called from this
175 // constructor; it is called from Initialise(). So we ask the
176 // user:
177 // (a) to write a subclass and implement RegisterGraphicsSystems()
178 // and RegisterModelFactories(). See
179 // visualization/include/G4VisExecutive.hh/icc as an example.
180 // (b) instantiate the subclass.
181 // (c) invoke the Initialise() method of the subclass.
182 // For example:
183 // ...
184 // #ifdef G4VIS_USE
185 // // Instantiate and initialise Visualization Manager.
186 // G4VisManager* visManager = new G4VisExecutive;
187 // visManager -> SetVerboseLevel (Verbose);
188 // visManager -> Initialise ();
189 // #endif
190 // // (Don't forget to delete visManager;)
191 // ...
192
193 // Make top level command directory...
194 // Vis commands should *not* be broadcast to threads (2nd argument).
195 G4UIcommand* directory = new G4UIdirectory ("/vis/",false);
196 directory -> SetGuidance ("Visualization commands.");
197 fDirectoryList.push_back (directory);
198
199 // Instantiate *basic* top level commands so that they can be used
200 // immediately after instantiation of the vis manager. Other top
201 // level and lower level commands are instantiated later in
202 // RegisterMessengers.
203 G4VVisCommand::SetVisManager (this); // Sets shared pointer
204 RegisterMessenger(new G4VisCommandVerbose);
205 RegisterMessenger(new G4VisCommandInitialize);
206}

208G4VisManager::~G4VisManager () {
209 fpInstance = 0;
210 size_t i;
211 for (i = 0; i < fSceneList.size (); ++i) {
212 delete fSceneList[i];
213 }
214 for (i = 0; i < fAvailableSceneHandlers.size (); ++i) {
215 if (fAvailableSceneHandlers[i] != NULL) {
216 delete fAvailableSceneHandlers[i];
217 }
218 }
219 for (i = 0; i < fAvailableGraphicsSystems.size (); ++i) {
220 if (fAvailableGraphicsSystems[i]) {
221 delete fAvailableGraphicsSystems[i];
222 }
223 }
224 if (fVerbosity >= startup) {
225 G4cout << "Graphics systems deleted." << G4endl;
226 G4cout << "Visualization Manager deleting..." << G4endl;
227 }
228 for (i = 0; i < fMessengerList.size (); ++i) {
229 delete fMessengerList[i];
230 }
231 for (i = 0; i < fDirectoryList.size (); ++i) {
232 delete fDirectoryList[i];
233 }
234
235 delete fpDigiFilterMgr;
236 delete fpHitFilterMgr;
237 delete fpTrajFilterMgr;
238 delete fpTrajDrawModelMgr;
239}

250void G4VisManager::Initialise () {
251
252 if (fInitialised && fVerbosity >= warnings) {
253 G4cout << "WARNING: G4VisManager::Initialise: already initialised."
254 << G4endl;
255 return;
256 }
257
258 if (fVerbosity >= startup) {
259 G4cout << "Visualization Manager initialising..." << G4endl;
260 }
261
262 if (fVerbosity >= parameters) {
263 G4cout <<
264 "\nYou have instantiated your own Visualization Manager, inheriting"
265 "\n G4VisManager and implementing RegisterGraphicsSystems(), in which"
266 "\n you should, normally, instantiate drivers which do not need"
267 "\n external packages or libraries, and, optionally, drivers under"
268 "\n control of environment variables."
269 "\n Also you should implement RegisterModelFactories()."
270 "\n See visualization/management/include/G4VisExecutive.hh/icc, for example."
271 "\n In your main() you will have something like:"
272 "\n #ifdef G4VIS_USE"
273 "\n G4VisManager* visManager = new G4VisExecutive;"
274 "\n visManager -> SetVerboseLevel (Verbose);"
275 "\n visManager -> Initialize ();"
276 "\n #endif"
277 "\n (Don't forget to delete visManager;)"
278 "\n"
279 << G4endl;
280 }
281
282 if (fVerbosity >= startup) {
283 G4cout << "Registering graphics systems..." << G4endl;
284 }
285
286 RegisterGraphicsSystems ();
287
288 if (fVerbosity >= startup) {
289 G4cout <<
290 "\nYou have successfully registered the following graphics systems."
291 << G4endl;
292 PrintAvailableGraphicsSystems (fVerbosity);
293 G4cout << G4endl;
294 }
295
296 // Make command directories for commands instantiated in the
297 // modeling subcategory...
298 G4UIcommand* directory;
299 directory = new G4UIdirectory ("/vis/modeling/");
300 directory -> SetGuidance ("Modeling commands.");
301 fDirectoryList.push_back (directory);
302 directory = new G4UIdirectory ("/vis/modeling/trajectories/");
303 directory -> SetGuidance ("Trajectory model commands.");
304 fDirectoryList.push_back (directory);
305 directory = new G4UIdirectory ("/vis/modeling/trajectories/create/");
306 directory -> SetGuidance ("Create trajectory models and messengers.");
307 fDirectoryList.push_back (directory);
308
309 // Filtering command directory
310 directory = new G4UIdirectory ("/vis/filtering/");
311 directory -> SetGuidance ("Filtering commands.");
312 fDirectoryList.push_back (directory);
313 directory = new G4UIdirectory ("/vis/filtering/trajectories/");
314 directory -> SetGuidance ("Trajectory filtering commands.");
315 fDirectoryList.push_back (directory);
316 directory = new G4UIdirectory ("/vis/filtering/trajectories/create/");
317 directory -> SetGuidance ("Create trajectory filters and messengers.");
318 fDirectoryList.push_back (directory);
319 directory = new G4UIdirectory ("/vis/filtering/hits/");
320 directory -> SetGuidance ("Hit filtering commands.");
321 fDirectoryList.push_back (directory);
322 directory = new G4UIdirectory ("/vis/filtering/hits/create/");
323 directory -> SetGuidance ("Create hit filters and messengers.");
324 fDirectoryList.push_back (directory);
325 directory = new G4UIdirectory ("/vis/filtering/digi/");
326 directory -> SetGuidance ("Digi filtering commands.");
327 fDirectoryList.push_back (directory);
328 directory = new G4UIdirectory ("/vis/filtering/digi/create/");
329 directory -> SetGuidance ("Create digi filters and messengers.");
330 fDirectoryList.push_back (directory);
331
332 RegisterMessengers ();
333
334 if (fVerbosity >= startup) {
335 G4cout << "Registering model factories..." << G4endl;
336 }
337
338 RegisterModelFactories();
339
340 if (fVerbosity >= startup) {
341 G4cout <<
342 "\nYou have successfully registered the following model factories."
343 << G4endl;
344 PrintAvailableModels (fVerbosity);
345 G4cout << G4endl;
346 }
347
348 if (fVerbosity >= startup) {
349 PrintAvailableUserVisActions (fVerbosity);
350 G4cout << G4endl;
351 }
352
353 InitialiseG4ColourMap();
354
355 if (fVerbosity >= startup) {
356 G4cout <<
357 "Some /vis commands (optionally) take a string to specify colour."
358 "\n\"/vis/list\" to see available colours."
359 << G4endl;
360 }
361
362 fInitialised = true;
363}

1145void G4VisManager::CreateViewer
1146(const G4String& name, const G4String& XGeometry)
1147{
1148
1149 if (!fInitialised) Initialise ();
1150
1151 if (!fpSceneHandler) {
1152 PrintInvalidPointers ();
1153 return;
1154 }
1155
1156 G4VViewer* p = fpGraphicsSystem -> CreateViewer (*fpSceneHandler, name);
1157
1158 if (!p) {
1159 if (fVerbosity >= errors) {
1160 G4cerr << "ERROR in G4VisManager::CreateViewer: null pointer during "
1161 << fpGraphicsSystem -> GetName ()
1162 << " viewer creation.\n No action taken."
1163 << G4endl;
1164 }
1165 return;
1166 }
1167
1168 if (p -> GetViewId() < 0) {
1169 if (fVerbosity >= errors) {
1170 G4cerr << "ERROR in G4VisManager::CreateViewer during "
1171 << fpGraphicsSystem -> GetName ()
1172 << " viewer instantiation.\n No action taken."
1173 << G4endl;
1174 }
1175 return;
1176 }
1177
1178 // Viewer is created, now we can set geometry parameters
1179 // Before 12/2008, it was done in G4VViewer.cc but it did not have to be there!
1180
1181 G4ViewParameters initialvp = p -> GetViewParameters();
1182 initialvp.SetXGeometryString(XGeometry); //parse string and store parameters
1183 p -> SetViewParameters(initialvp);
1184 p -> Initialise (); // (Viewer itself may change view parameters further.)
1185 if (p -> GetViewId() < 0) {
1186 if (fVerbosity >= errors) {
1187 G4cerr << "ERROR in G4VisManager::CreateViewer during "
1188 << fpGraphicsSystem -> GetName ()
1189 << " viewer initialisation.\n No action taken."
1190 << G4endl;
1191 }
1192 return;
1193 }
1194
1195 fpViewer = p; // Make current.
1196 fpSceneHandler -> AddViewerToList (fpViewer);
1197 fpSceneHandler -> SetCurrentViewer (fpViewer);
1198 if (fVerbosity >= confirmations) {
1199 G4cout << "G4VisManager::CreateViewer: new viewer created."
1200 << G4endl;
1201 }
1202
1203 const G4ViewParameters& vp = fpViewer->GetViewParameters();
1204 if (fVerbosity >= parameters) {
1205 G4cout << " view parameters are:\n " << vp << G4endl;
1206 }
1207
1208 if (vp.IsCulling () && vp.IsCullingInvisible ()) {
1209 static G4bool warned = false;
1210 if (fVerbosity >= confirmations) {
1211 if (!warned) {
1212 G4cout <<
1213 "NOTE: objects with visibility flag set to \"false\""
1214 " will not be drawn!"
1215 "\n \"/vis/viewer/set/culling global false\" to Draw such objects."
1216 "\n Also see other \"/vis/viewer/set\" commands."
1217 << G4endl;
1218 warned = true;
1219 }
1220 }
1221 }
1222 if (vp.IsCullingCovered ()) {
1223 static G4bool warned = false;
1224 if (fVerbosity >= warnings) {
1225 if (!warned) {
1226 G4cout <<
1227 "WARNING: covered objects in solid mode will not be rendered!"

1744G4ThreadFunReturnType G4VisManager::G4VisSubThread(G4ThreadFunArgType p)
1745{
1746 G4VisManager* pVisManager = (G4VisManager*)p;
1747 G4VSceneHandler* pSceneHandler = pVisManager->GetCurrentSceneHandler();
1748 if (!pSceneHandler) return 0;
1749 G4Scene* pScene = pSceneHandler->GetScene();
1750 if (!pScene) return 0;
1751 G4VViewer* pViewer = pVisManager->GetCurrentViewer();
1752 if (!pViewer) return 0;
1753
1754 G4UImanager::GetUIpointer()->SetUpForSpecialThread("G4VIS");
1755
1756// G4cout << "G4VisManager::G4VisSubThread: thread: "
1757// << G4Threading::G4GetThreadId() << std::endl;
1758
1759 // Set up geometry and navigation for a thread
1760 G4GeometryWorkspace::GetPool()->CreateAndUseWorkspace();
1761 G4SolidsWorkspace::GetPool()->CreateAndUseWorkspace();
1762 G4Navigator* navigator =
1763 G4TransportationManager::GetTransportationManager()->GetNavigatorForTracking();
1764 navigator->SetWorldVolume
1765 (G4MTRunManager::GetMasterRunManagerKernel()->GetCurrentWorld());
1766
1767 pViewer->SwitchToVisSubThread();
1768
1769 while (true) {
1770
1771 G4MUTEXLOCK(&mtVisSubThreadMutex);
1772 G4int eventQueueSize = mtVisEventQueue.size();
1773 G4MUTEXUNLOCK(&mtVisSubThreadMutex);
1774// G4cout << "Event queue size (A): " << eventQueueSize << G4endl;
1775
1776 while (eventQueueSize) {
1777
1778 G4MUTEXLOCK(&mtVisSubThreadMutex);
1779 const G4Event* event = mtVisEventQueue.front();
1780 G4MUTEXUNLOCK(&mtVisSubThreadMutex);
1781// G4int eventID = event->GetEventID();
1782// G4cout <<
1783// "G4VisManager::G4VisSubThread: Vis sub-thread: Dealing with event: "
1784// << eventID << G4endl;
1785
1786 // Here comes the event drawing
1787 pVisManager->SetTransientsDrawnThisEvent(false);
1788 pSceneHandler->SetTransientsDrawnThisEvent(false);
1789
1790 // We are about to draw the event (trajectories, etc.), but first we
1791 // have to clear the previous event(s) if necessary. If this event
1792 // needs to be drawn afresh, e.g., the first event or any event when
1793 // "accumulate" is not requested, the old event has to be cleared.
1794 // We have postponed this so that, for normal viewers like OGL, the
1795 // previous event(s) stay on screen until this new event comes
1796 // along. For a file-writing viewer the geometry has to be drawn.
1797 // See, for example, G4HepRepFileSceneHandler::ClearTransientStore.
1798 pVisManager->ClearTransientStoreIfMarked();
1799
1800 // Now draw the event...
1801 pSceneHandler->DrawEvent(event);
1802 ++pVisManager->fNoOfEventsDrawnThisRun;
1803
1804 if (pScene->GetRefreshAtEndOfEvent()) {
1805
1806 // ShowView guarantees the view is flushed to the screen. It also
1807 // triggers other features such picking (if enabled) and allows
1808 // file-writing viewers to close the file.
1809 pViewer->ShowView();
1810 pSceneHandler->SetMarkForClearingTransientStore(true);
1811
1812 }
1813
1814 // Testing.
1815// std::this_thread::sleep_for(std::chrono::seconds(5));
1816
1817 // Then pop and release event
1818 G4MUTEXLOCK(&mtVisSubThreadMutex);
1819 mtVisEventQueue.pop_front();
1820 event->PostProcessingFinished();
1821 eventQueueSize = mtVisEventQueue.size();
1822 G4MUTEXUNLOCK(&mtVisSubThreadMutex);
1823// G4cout << "Event queue size (B): " << eventQueueSize << G4endl;
1824 }
1825
1826 G4MUTEXLOCK(&mtVisSubThreadMutex);
1827 G4int runInProgress = mtRunInProgress;
1828 G4MUTEXUNLOCK(&mtVisSubThreadMutex);
1829 if (!runInProgress) {
1830 // EndOfRun on master thread has signalled end of run. There is
1831 // nothing to draw so...
1832 break;
1833 }
1834
1835 // Run still in progress but nothing to draw, so wait a while.
1836#ifdef G4VIS_USE_STD11
1837 std::this_thread::sleep_for(std::chrono::milliseconds(100));
1838#else
1839 G4THREADSLEEP(1);
1840#endif
1841 }
1842
1843 // Inform viewer that we have finished all sub-thread drawing
1844 pViewer->DoneWithVisSubThread();
1845 pViewer->MovingToMasterThread();
1846// G4cout << "G4VisManager::G4VisSubThread: Vis sub-thread: ending" << G4endl;
1847 return /*(G4ThreadFunReturnType)*/0;
1848}
