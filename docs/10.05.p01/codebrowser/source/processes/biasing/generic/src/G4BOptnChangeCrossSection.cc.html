26#include "G4BOptnChangeCrossSection.hh"
27#include "G4InteractionLawPhysical.hh"
28
29
30
31G4BOptnChangeCrossSection::G4BOptnChangeCrossSection(G4String name)
32 : G4VBiasingOperation( name ),
33 fInteractionOccured( false )
34{
35 fBiasedExponentialLaw = new G4InteractionLawPhysical("LawForOperation"+name);
36}
37
38G4BOptnChangeCrossSection::~G4BOptnChangeCrossSection()
39{
40 if ( fBiasedExponentialLaw ) delete fBiasedExponentialLaw;
41}
42
43const G4VBiasingInteractionLaw* G4BOptnChangeCrossSection::ProvideOccurenceBiasingInteractionLaw( const G4BiasingProcessInterface*, G4ForceCondition& )
44{
45 return fBiasedExponentialLaw;
46}
47
48void G4BOptnChangeCrossSection::SetBiasedCrossSection(G4double xst)
49{
50 fBiasedExponentialLaw->SetPhysicalCrossSection( xst );
51}
52
53G4double G4BOptnChangeCrossSection::GetBiasedCrossSection() const
54{
55 return fBiasedExponentialLaw->GetPhysicalCrossSection();
56}
57
58void G4BOptnChangeCrossSection::Sample()
59{
60 fInteractionOccured = false;
61 fBiasedExponentialLaw->Sample();
62}
63
64void G4BOptnChangeCrossSection::UpdateForStep( G4double truePathLength )
65{
66 fBiasedExponentialLaw->UpdateForStep( truePathLength );
67}
