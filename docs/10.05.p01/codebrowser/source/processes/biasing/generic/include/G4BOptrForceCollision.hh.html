30// G4BOptrForceCollision
31//
32// Class Description:
33// A G4VBiasingOperator that implements a "force collision" a la
34// MCNP. This is meant for neutral particles.
35// When the track enters the volume, it is cloned. One copy makes
36// a forced free flight up to the volume exit. The other copy makes
37// a forced collision inside the volume.
38//
39// ---------------------------------------------------------------
40// Initial version Nov. 2013 M. Verderi
41
42

44#define G4BOptrForceCollision_hh 1
45

48class G4BOptnForceCommonTruncatedExp;
49class G4BOptnCloning;
50class G4VProcess;
51class G4BiasingProcessInterface;
52class G4ParticleDefinition;

63
64private:

75 virtual void ExitBiasing( const G4Track*, const G4BiasingProcessInterface* ) final {};

86private:

95 const G4ParticleDefinition* fParticleToBias;
96};
