41#ifndef G4BOptnCloning_hh
42#define G4BOptnCloning_hh 1
43
44#include "G4VBiasingOperation.hh"
45#include "G4ParticleChange.hh"
46
47class G4BOptnCloning : public G4VBiasingOperation {
48public:
49 // -- Constructor :
50 G4BOptnCloning(G4String name);
51 // -- destructor:
52 virtual ~G4BOptnCloning();
53
54public:
55 // -- Methods from G4VBiasingOperation interface:
56 // -------------------------------------------
57 // -- Unsed:
58 virtual const G4VBiasingInteractionLaw* ProvideOccurenceBiasingInteractionLaw( const G4BiasingProcessInterface*, G4ForceCondition& ) {return 0;}
59 virtual G4VParticleChange* ApplyFinalStateBiasing( const G4BiasingProcessInterface*,
60 const G4Track*,
61 const G4Step*,
62 G4bool& ) {return 0;}
63 // -- Used:
64 virtual G4double DistanceToApplyOperation( const G4Track*,
65 G4double,
66 G4ForceCondition* condition)
67 {
68 *condition = NotForced; return 0; // -- acts immediately
69 }
70 virtual G4VParticleChange* GenerateBiasingFinalState( const G4Track*,
71 const G4Step* );
72
73public:
74 // -- Additional methods, specific to this class:
75 // ----------------------------------------------
76 void SetCloneWeights(G4double clone1Weight, G4double clone2Weight) {fClone1W = clone1Weight ; fClone2W = clone2Weight;}
77
78 G4Track* GetCloneTrack() const { return fCloneTrack; }
79
80private:
81 G4double fClone1W,
82 fClone2W;
83 G4ParticleChange fParticleChange;
84 G4Track* fCloneTrack;
85};
