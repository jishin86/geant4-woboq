30// G4ParticleChangeForOccurenceBiasing
31//
32// Class Description:
33// A G4VParticleChange dedicated to occurrence biasing : it
34// applies weights for non-interaction over a step, and for
35// interaction at the end of the step (if interaction occurs) on
36// top of a given particle change, that is the one produced by a
37// physics process (biased in its final state or not).
38//
39// ---------------------------------------------------------------
40// Initial version Nov. 2013 M. Verderi
41//
42#ifndef G4ParticleChangeForOccurenceBiasing_hh
43#define G4ParticleChangeForOccurenceBiasing_hh 1
44
45#include "G4VParticleChange.hh"
46
47class G4ParticleChangeForOccurenceBiasing : public G4VParticleChange {
48public:
49 G4ParticleChangeForOccurenceBiasing(G4String name);
50 ~G4ParticleChangeForOccurenceBiasing();
51
52public:
53 void SetOccurenceWeightForNonInteraction(G4double w) {fOccurenceWeightForNonInteraction = w;}
54 G4double GetOccurenceWeightForNonInteraction() const {return fOccurenceWeightForNonInteraction;}
55 void SetOccurenceWeightForInteraction(G4double w) {fOccurenceWeightForInteraction = w;}
56 G4double GetOccurenceWeightForInteraction() const {return fOccurenceWeightForInteraction;}
57
58public:
59 // -- set a wrapped particle change AND USE IT TO UPDATE this occurrence particle change state:
60 void SetWrappedParticleChange(G4VParticleChange* wpc);
61 G4VParticleChange* GetWrappedParticleChange() const {return fWrappedParticleChange;}
62public:
63 // -- collect the secondaries from the wrapped particle change, apply weight correction, and clear wrapped particle change:
64 void StealSecondaries();
65
66public:
67 // -- from base class G4VParticleChange:
68 virtual G4Step* UpdateStepForAtRest (G4Step* step);
69 virtual G4Step* UpdateStepForAlongStep(G4Step* step);
70 virtual G4Step* UpdateStepForPostStep (G4Step* step);
71
72public:
73 const G4String& GetName() const {return fName;}
74
75private:
76 G4String fName;
77 G4VParticleChange* fWrappedParticleChange;
78 G4double fOccurenceWeightForNonInteraction;
79 G4double fOccurenceWeightForInteraction;
80
81
82};
83
84#endif
