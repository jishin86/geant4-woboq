145inline
146G4bool G4OpWLS::IsApplicable(const G4ParticleDefinition& aParticleType)
147{
148 return ( &aParticleType == G4OpticalPhoton::OpticalPhoton() );
149}
150
151inline
152G4PhysicsTable* G4OpWLS::GetIntegralTable() const
153{
154 return theIntegralTable;
155}
156
157inline
158void G4OpWLS::DumpPhysicsTable() const
159{
160 G4int PhysicsTableSize = theIntegralTable->entries();
161 G4PhysicsOrderedFreeVector *v;
162
163 for (G4int i = 0 ; i < PhysicsTableSize ; i++ )
164 {
165 v = (G4PhysicsOrderedFreeVector*)(*theIntegralTable)[i];
166 v->DumpValues();
167 }
168}
