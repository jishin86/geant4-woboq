159inline
160G4bool G4OpRayleigh::IsApplicable(const G4ParticleDefinition& aParticleType)
161{
162 return ( &aParticleType == G4OpticalPhoton::OpticalPhoton() );
163}
164
165inline
166void G4OpRayleigh::DumpPhysicsTable() const
167
168{
169 G4int PhysicsTableSize = thePhysicsTable->entries();
170 G4PhysicsOrderedFreeVector *v;
171
172 for (G4int i = 0 ; i < PhysicsTableSize ; i++ )
173 {
174 v = (G4PhysicsOrderedFreeVector*)(*thePhysicsTable)[i];
175 v->DumpValues();
176 }
177}
178
179inline G4PhysicsTable* G4OpRayleigh::GetPhysicsTable() const
180{
181 return thePhysicsTable;
182}
