114void G4HadronicProcess::InitialiseLocal() {
115 theTotalResult = new G4ParticleChange();
116 theTotalResult->SetSecondaryWeightByProcess(true);
117 theInteraction = nullptr;
118 theCrossSectionDataStore = new G4CrossSectionDataStore();
119 theProcessStore = G4HadronicProcessStore::Instance();
120 theProcessStore->Register(this);
121 theInitialNumberOfInteractionLength = 0.0;
122 aScaleFactor = 1.0;
123 fWeight = 1.0;
124 xBiasOn = false;
125 nMatWarn = 0;

127 theLastCrossSection = 0.0;
128 nICelectrons = 0;
129 idxIC = -1;
130 G4HadronicProcess_debug_flag = false;
131 GetEnergyMomentumCheckEnvvars();
132}

147void G4HadronicProcess::RegisterMe( G4HadronicInteraction *a )
148{
149 if(!a) { return; }
150 try{ theEnergyRangeManager.RegisterMe( a ); }
151 catch(G4HadronicException & aE)
152 {
153 G4ExceptionDescription ed;
154 aE.Report(ed);
155 ed << "Unrecoverable error in " << GetProcessName()
156 << " to register " << a->GetModelName() << G4endl;
157 G4Exception("G4HadronicProcess::RegisterMe", "had001", FatalException,
158 ed);
159 }
160 G4HadronicProcessStore::Instance()->RegisterInteraction(this, a);
161}

163G4double
164G4HadronicProcess::GetElementCrossSection(const G4DynamicParticle * part,
165 const G4Element * elm,
166 const G4Material* mat)
167{
168 if(!mat)
169 {
170 ++nMatWarn;
171 static const G4int nmax = 5;
172 if(nMatWarn < nmax) {
173 G4ExceptionDescription ed;
174 ed << "Cannot compute Element x-section for " << GetProcessName()
175 << " because no material defined \n"
176 << " Please, specify material pointer or define simple material"
177 << " for Z= " << elm->GetZasInt();
178 G4Exception("G4HadronicProcess::GetElementCrossSection", "had066",
179 JustWarning, ed);
180 }
181 }
182 return
183 std::max(theCrossSectionDataStore->GetCrossSection(part, elm, mat),0.0);
184}

212G4double G4HadronicProcess::
213GetMeanFreePath(const G4Track &aTrack, G4double, G4ForceCondition *)
214{
215 //G4cout << "GetMeanFreePath " << aTrack.GetDefinition()->GetParticleName()
216 // << " Ekin= " << aTrack.GetKineticEnergy() << G4endl;
217 try
218 {
219 theLastCrossSection = aScaleFactor*
220 theCrossSectionDataStore->ComputeCrossSection(aTrack.GetDynamicParticle(),
221 aTrack.GetMaterial());
222 }
223 catch(G4HadronicException & aR)
224 {
225 G4ExceptionDescription ed;
226 aR.Report(ed);
227 DumpState(aTrack,"GetMeanFreePath",ed);
228 ed << " Cross section is not available" << G4endl;
229 G4Exception("G4HadronicProcess::GetMeanFreePath", "had002", FatalException,
230 ed);
231 }
232 G4double res = (theLastCrossSection>0.0) ? 1.0/theLastCrossSection : DBL_MAX;
233 //G4cout << " xsection= " << theLastCrossSection << G4endl;
234 return res;
235}

237G4VParticleChange*
238G4HadronicProcess::PostStepDoIt(const G4Track& aTrack, const G4Step&)
239{
240 //G4cout << "PostStepDoIt " << aTrack.GetDefinition()->GetParticleName()
241 // << " Ekin= " << aTrack.GetKineticEnergy() << G4endl;
242 // if primary is not Alive then do nothing
243 theTotalResult->Clear();
244 theTotalResult->Initialize(aTrack);
245 fWeight = aTrack.GetWeight();
246 theTotalResult->ProposeWeight(fWeight);
247 if(aTrack.GetTrackStatus() != fAlive) { return theTotalResult; }
248
249 // Find cross section at end of step and check if <= 0
250 //
251 const G4DynamicParticle* aParticle = aTrack.GetDynamicParticle();
252 const G4Material* aMaterial = aTrack.GetMaterial();
253
254 // check only for charged particles
255 if(aParticle->GetDefinition()->GetPDGCharge() != 0.0) {
256 G4double xs = 0.0;
257 try
258 {
259 xs = aScaleFactor*
260 theCrossSectionDataStore->ComputeCrossSection(aParticle,aMaterial);
261 }
262 catch(G4HadronicException & aR)
263 {
264 G4ExceptionDescription ed;
265 aR.Report(ed);
266 DumpState(aTrack,"PostStepDoIt",ed);
267 ed << " Cross section is not available" << G4endl;
268 G4Exception("G4HadronicProcess::PostStepDoIt","had002",FatalException,ed);
269 }
270 if(xs <= 0.0 || xs < theLastCrossSection*G4UniformRand()) {
271 // No interaction
272 return theTotalResult;
273 }
274 }
275
276 const G4Element* anElement = nullptr;
277 try
278 {
279 anElement = theCrossSectionDataStore->SampleZandA(aParticle, aMaterial,
280 targetNucleus);
281 }
282 catch(G4HadronicException & aR)
283 {
284 G4ExceptionDescription ed;
285 aR.Report(ed);
286 DumpState(aTrack,"SampleZandA",ed);
287 ed << " PostStepDoIt failed on element selection" << G4endl;
288 G4Exception("G4HadronicProcess::PostStepDoIt", "had003", FatalException,
289 ed);
290 }
291
292 // Next check for illegal track status
293 //
294 if (aTrack.GetTrackStatus() != fAlive &&
295 aTrack.GetTrackStatus() != fSuspend) {
296 if (aTrack.GetTrackStatus() == fStopAndKill ||
297 aTrack.GetTrackStatus() == fKillTrackAndSecondaries ||
298 aTrack.GetTrackStatus() == fPostponeToNextEvent) {
299 G4ExceptionDescription ed;
300 ed << "G4HadronicProcess: track in unusable state - "
301 << aTrack.GetTrackStatus() << G4endl;
302 ed << "G4HadronicProcess: returning unchanged track " << G4endl;
303 DumpState(aTrack,"PostStepDoIt",ed);
304 G4Exception("G4HadronicProcess::PostStepDoIt", "had004", JustWarning, ed);
305 }
306 // No warning for fStopButAlive which is a legal status here
307 return theTotalResult;
308 }
309
310 // Initialize the hadronic projectile from the track
311 thePro.Initialise(aTrack);
312
313 try
314 {
315 theInteraction =
316 ChooseHadronicInteraction( thePro, targetNucleus, aMaterial, anElement );
317 }
318 catch(G4HadronicException & aE)
319 {
320 G4ExceptionDescription ed;
321 aE.Report(ed);
322 ed << "Target element "<<anElement->GetName()<<" Z= "
323 << targetNucleus.GetZ_asInt() << " A= "
324 << targetNucleus.GetA_asInt() << G4endl;
325 DumpState(aTrack,"ChooseHadronicInteraction",ed);
326 ed << " No HadronicInteraction found out" << G4endl;
327 G4Exception("G4HadronicProcess::PostStepDoIt", "had005", FatalException,
328 ed);
329 }
330
331 G4HadFinalState* result = nullptr;
332 G4int reentryCount = 0;
333
334 do
335 {
336 try
337 {
338 // Save random engine if requested for debugging
339 if (G4Hadronic_Random_File) {
340 CLHEP::HepRandom::saveEngineStatus(G4Hadronic_Random_File);
341 }
342 // Call the interaction
343 result = theInteraction->ApplyYourself( thePro, targetNucleus);
344 ++reentryCount;
345 }
346 catch(G4HadronicException & aR)
347 {
348 G4ExceptionDescription ed;
349 aR.Report(ed);
350 ed << "Call for " << theInteraction->GetModelName() << G4endl;
351 ed << "Target element "<<anElement->GetName()<<" Z= "
352 << targetNucleus.GetZ_asInt()
353 << " A= " << targetNucleus.GetA_asInt() << G4endl;
354 DumpState(aTrack,"ApplyYourself",ed);
355 ed << " ApplyYourself failed" << G4endl;
356 G4Exception("G4HadronicProcess::PostStepDoIt", "had006", FatalException,
357 ed);
358 }
359
360 // Check the result for catastrophic energy non-conservation
361 result = CheckResult(thePro, targetNucleus, result);
362
363 if(reentryCount>100) {
364 G4ExceptionDescription ed;
365 ed << "Call for " << theInteraction->GetModelName() << G4endl;
366 ed << "Target element "<<anElement->GetName()<<" Z= "
367 << targetNucleus.GetZ_asInt()
368 << " A= " << targetNucleus.GetA_asInt() << G4endl;
369 DumpState(aTrack,"ApplyYourself",ed);
370 ed << " ApplyYourself does not completed after 100 attempts" << G4endl;
371 G4Exception("G4HadronicProcess::PostStepDoIt", "had006", FatalException,
372 ed);
373 }
374 }
375 while(!result); /* Loop checking, 30-Oct-2015, G.Folger */
376
377 // Check whether kaon0 or anti_kaon0 are present between the secondaries:
378 // if this is the case, transform them into either kaon0S or kaon0L,
379 // with equal, 50% probability, keeping their dynamical masses (and
380 // the other kinematical properties).
381 // When this happens - very rarely - a "JustWarning" exception is thrown.
382 G4int nSec = result->GetNumberOfSecondaries();
383 if ( nSec > 0 ) {
384 for ( G4int i = 0; i < nSec; ++i ) {
385 G4DynamicParticle* dynamicParticle = result->GetSecondary(i)->GetParticle();
386 const G4ParticleDefinition* particleDefinition = dynamicParticle->GetParticleDefinition();
387 if ( particleDefinition == G4KaonZero::Definition() ||
388 particleDefinition == G4AntiKaonZero::Definition() ) {
389 G4ParticleDefinition* newPart;
390 if( G4UniformRand() > 0.5 ) { newPart = G4KaonZeroShort::Definition(); }
391 else { newPart = G4KaonZeroLong::Definition(); }
392 dynamicParticle->SetDefinition( newPart );
393 G4ExceptionDescription ed;
394 ed << " Hadronic model " << theInteraction->GetModelName() << G4endl;
395 ed << " created " << particleDefinition->GetParticleName() << G4endl;
396 ed << " -> forced to be " << newPart->GetParticleName() << G4endl;
397 G4Exception( "G4HadronicProcess::PostStepDoIt", "had007", JustWarning, ed );
398 }
399 }
400 }
401
402 result->SetTrafoToLab(thePro.GetTrafoToLab());
403
404 ClearNumberOfInteractionLengthLeft();
405
406 FillResult(result, aTrack);
407
408 if (epReportLevel != 0) {
409 CheckEnergyMomentumConservation(aTrack, targetNucleus);
410 }
411 //G4cout << "PostStepDoIt done " << G4endl;
412 return theTotalResult;
413}

420
421G4double G4HadronicProcess::XBiasSurvivalProbability()
422{
423 G4double nLTraversed = GetTotalNumberOfInteractionLengthTraversed();
424 G4double biasedProbability = 1.-G4Exp(-nLTraversed);
425 G4double realProbability = 1-G4Exp(-nLTraversed/aScaleFactor);
426 G4double result = (biasedProbability-realProbability)/biasedProbability;
427 return result;
428}
429
430G4double G4HadronicProcess::XBiasSecondaryWeight()
431{
432 G4double nLTraversed = GetTotalNumberOfInteractionLengthTraversed();
433 G4double result =
434 1./aScaleFactor*G4Exp(-nLTraversed/aScaleFactor*(1-1./aScaleFactor));
435 return result;
436}

438void
439G4HadronicProcess::FillResult(G4HadFinalState * aR, const G4Track & aT)
440{
441 theTotalResult->ProposeLocalEnergyDeposit(aR->GetLocalEnergyDeposit());
442
443 G4double rotation = CLHEP::twopi*G4UniformRand();
444 G4ThreeVector it(0., 0., 1.);
445
446 G4double efinal = aR->GetEnergyChange();
447 if(efinal < 0.0) { efinal = 0.0; }
448
449 // check status of primary
450 if(aR->GetStatusChange() == stopAndKill) {
451 theTotalResult->ProposeTrackStatus(fStopAndKill);
452 theTotalResult->ProposeEnergy( 0.0 );
453
454 // check its final energy
455 } else if(0.0 == efinal) {
456 theTotalResult->ProposeEnergy( 0.0 );
457 if(aT.GetParticleDefinition()->GetProcessManager()
458 ->GetAtRestProcessVector()->size() > 0)
459 { theTotalResult->ProposeTrackStatus(fStopButAlive); }
460 else { theTotalResult->ProposeTrackStatus(fStopAndKill); }
461
462 // primary is not killed apply rotation and Lorentz transformation
463 } else {
464 theTotalResult->ProposeTrackStatus(fAlive);
465 G4double mass = aT.GetParticleDefinition()->GetPDGMass();
466 G4double newE = efinal + mass;
467 G4double newP = std::sqrt(efinal*(efinal + 2*mass));
468 G4ThreeVector newPV = newP*aR->GetMomentumChange();
469 G4LorentzVector newP4(newE, newPV);
470 newP4.rotate(rotation, it);
471 newP4 *= aR->GetTrafoToLab();
472 theTotalResult->ProposeMomentumDirection(newP4.vect().unit());
473 newE = newP4.e() - mass;
474 if(G4HadronicProcess_debug_flag && newE <= 0.0) {
475 G4ExceptionDescription ed;
476 DumpState(aT,"Primary has zero energy after interaction",ed);
477 G4Exception("G4HadronicProcess::FillResults", "had011", JustWarning, ed);
478 }
479 if(newE < 0.0) { newE = 0.0; }
480 theTotalResult->ProposeEnergy( newE );
481 }
482 //G4cout << "FillResult: Efinal= " << efinal << " status= "
483 // << theTotalResult->GetTrackStatus()
484 // << " fKill= " << fStopAndKill << G4endl;
485
486 // check secondaries: apply rotation and Lorentz transformation
487 nICelectrons = 0;
488 if(idxIC == -1) {
489 G4int idx = G4PhysicsModelCatalog::GetIndex("e-InternalConvertion");
490 idxIC = -1 == idx ? -2 : idx;
491 }
492 G4int nSec = aR->GetNumberOfSecondaries();
493 theTotalResult->SetNumberOfSecondaries(nSec);
494
495 if (nSec > 0) {
496 G4double time0 = aT.GetGlobalTime();
497 for (G4int i = 0; i < nSec; ++i) {
498 G4DynamicParticle* dynamicParticle = aR->GetSecondary(i)->GetParticle();
499 G4LorentzVector theM = dynamicParticle->Get4Momentum();
500 theM.rotate(rotation, it);
501 theM *= aR->GetTrafoToLab();
502 const G4ParticleDefinition* part = dynamicParticle->GetDefinition();
503 G4double mass = part->GetPDGMass();
504 G4double dmass= theM.mag();
505 // check if secondary is on the mass shell
506 if(std::abs(dmass - mass) > 1.5*CLHEP::MeV || theM.e() < mass) {
507 if(G4HadronicProcess_debug_flag) {
508 G4ExceptionDescription ed;
509 ed << "TrackID= "<< aT.GetTrackID()
510 << " " << aT.GetParticleDefinition()->GetParticleName()
511 << " Target Z= " << targetNucleus.GetZ_asInt() << " A= "
512 << targetNucleus.GetA_asInt()
513 << " Ekin(GeV)= " << aT.GetKineticEnergy()/CLHEP::GeV
514 << "\n Secondary is out of mass shell: " << part->GetParticleName()
515 << " Ekin(MeV)= " << theM.e() - mass
516 << " DeltaMass(MeV)= " << dmass - mass << G4endl;
517 G4Exception("G4HadronicProcess::FillResults", "had012", JustWarning, ed);
518 }
519 G4double e = std::max(theM.e(), mass);
520 G4double mom = std::sqrt((e - mass)*(e + mass));
521 G4ThreeVector v = theM.vect().unit();
522 theM.set(v.x()*mom,v.y()*mom,v.z()*mom,e);
523 }
524 dynamicParticle->Set4Momentum(theM);
525 dynamicParticle->SetMass(mass);
526
527 G4int idxModel = aR->GetSecondary(i)->GetCreatorModelType();
528 if(idxIC == idxModel) { ++nICelectrons; }
529
530 // time of interaction starts from zero
531 G4double time = aR->GetSecondary(i)->GetTime();
532 if (time < 0.0) { time = 0.0; }
533
534 // take into account global time
535 time += time0;
536
537 G4Track* track = new G4Track(dynamicParticle, time, aT.GetPosition());
538 track->SetCreatorModelIndex(idxModel);
539 G4double newWeight = fWeight*aR->GetSecondary(i)->GetWeight();
540 track->SetWeight(newWeight);
541 track->SetTouchableHandle(aT.GetTouchableHandle());
542 theTotalResult->AddSecondary(track);
543 if (G4HadronicProcess_debug_flag) {
544 G4double e = track->GetKineticEnergy();
545 if (e <= 0.0) {
546 G4ExceptionDescription ed;
547 DumpState(aT,"Secondary has zero energy",ed);
548 ed << "Secondary " << part->GetParticleName()
549 << G4endl;
550 G4Exception("G4HadronicProcess::FillResults", "had011",
551 JustWarning,ed);
552 }
553 }
554 }
555 }
556 aR->Clear();
557}

577G4HadFinalState* G4HadronicProcess::CheckResult(const G4HadProjectile & aPro,
578 const G4Nucleus &aNucleus,
579 G4HadFinalState * result)
580{
581 // check for catastrophic energy non-conservation
582 // to re-sample the interaction
583
584 G4HadronicInteraction * theModel = GetHadronicInteraction();
585 G4double nuclearMass(0);
586 if (theModel) {
587
588 // Compute final-state total energy
589 G4double finalE(0.);
590 G4int nSec = result->GetNumberOfSecondaries();
591
592 nuclearMass = G4NucleiProperties::GetNuclearMass(aNucleus.GetA_asInt(),
593 aNucleus.GetZ_asInt());
594 if (result->GetStatusChange() != stopAndKill) {
595 // Interaction didn't complete, returned "do nothing" state
596 // and reset nucleus or the primary survived the interaction
597 // (e.g. electro-nuclear ) => keep nucleus
598 finalE=result->GetLocalEnergyDeposit() +
599 aPro.GetDefinition()->GetPDGMass() + result->GetEnergyChange();
600 if( nSec == 0 ){
601 // Since there are no secondaries, there is no recoil nucleus.
602 // To check energy balance we must neglect the initial nucleus too.
603 nuclearMass=0.0;
604 }
605 }
606 for (G4int i = 0; i < nSec; i++) {
607 G4DynamicParticle *pdyn=result->GetSecondary(i)->GetParticle();
608 finalE += pdyn->GetTotalEnergy();
609 G4double mass_pdg=pdyn->GetDefinition()->GetPDGMass();
610 G4double mass_dyn=pdyn->GetMass();
611 if ( std::abs(mass_pdg - mass_dyn) > 0.1*mass_pdg + 1.*MeV){
612 result->Clear();
613 result = nullptr;
614 G4ExceptionDescription desc;
615 desc << "Warning: Secondary with off-shell dynamic mass detected: "
616 << G4endl
617 << " " << pdyn->GetDefinition()->GetParticleName()
618 << ", PDG mass: " << mass_pdg << ", dynamic mass: "
619 << mass_dyn << G4endl
620 << (epReportLevel<0 ? "abort the event"
621 : "re-sample the interaction") << G4endl
622 << " Process / Model: " << GetProcessName()<< " / "
623 << theModel->GetModelName() << G4endl
624 << " Primary: " << aPro.GetDefinition()->GetParticleName()
625 << " (" << aPro.GetDefinition()->GetPDGEncoding() << "), "
626 << " E= " << aPro.Get4Momentum().e()
627 << ", target nucleus (" << aNucleus.GetZ_asInt() << ", "
628 << aNucleus.GetA_asInt() << ")" << G4endl;
629 G4Exception("G4HadronicProcess:CheckResult()", "had012",
630 epReportLevel<0 ? EventMustBeAborted : JustWarning,desc);
631 // must return here.....
632 return result;
633 }
634 }
635 G4double deltaE= nuclearMass + aPro.GetTotalEnergy() - finalE;
636
637 std::pair<G4double, G4double> checkLevels =
638 theModel->GetFatalEnergyCheckLevels(); // (relative, absolute)
639 if (std::abs(deltaE) > checkLevels.second &&
640 std::abs(deltaE) > checkLevels.first*aPro.GetKineticEnergy()){
641 // do not delete result, this is a pointer to a data member;
642 result->Clear();
643 result = nullptr;
644 G4ExceptionDescription desc;
645 desc << "Warning: Bad energy non-conservation detected, will "
646 << (epReportLevel<0 ? "abort the event"
647 : "re-sample the interaction") << G4endl
648 << " Process / Model: " << GetProcessName()<< " / "
649 << theModel->GetModelName() << G4endl
650 << " Primary: " << aPro.GetDefinition()->GetParticleName()
651 << " (" << aPro.GetDefinition()->GetPDGEncoding() << "), "
652 << " E= " << aPro.Get4Momentum().e()
653 << ", target nucleus (" << aNucleus.GetZ_asInt() << ", "
654 << aNucleus.GetA_asInt() << ")" << G4endl
655 << " E(initial - final) = " << deltaE << " MeV." << G4endl;
656 G4Exception("G4HadronicProcess:CheckResult()", "had012",
657 epReportLevel<0 ? EventMustBeAborted : JustWarning,desc);
658 }
659 }
660 return result;
661}

663void
664G4HadronicProcess::CheckEnergyMomentumConservation(const G4Track& aTrack,
665 const G4Nucleus& aNucleus)
666{
667 G4int target_A=aNucleus.GetA_asInt();
668 G4int target_Z=aNucleus.GetZ_asInt();
669 G4double targetMass = G4NucleiProperties::GetNuclearMass(target_A,target_Z);
670 G4LorentzVector target4mom(0, 0, 0, targetMass
671 + nICelectrons*CLHEP::electron_mass_c2);
672
673 G4LorentzVector projectile4mom = aTrack.GetDynamicParticle()->Get4Momentum();
674 G4int track_A = aTrack.GetDefinition()->GetBaryonNumber();
675 G4int track_Z = G4lrint(aTrack.GetDefinition()->GetPDGCharge());
676
677 G4int initial_A = target_A + track_A;
678 G4int initial_Z = target_Z + track_Z - nICelectrons;
679
680 G4LorentzVector initial4mom = projectile4mom + target4mom;
681
682 // Compute final-state momentum for scattering and "do nothing" results
683 G4LorentzVector final4mom;
684 G4int final_A(0), final_Z(0);
685
686 G4int nSec = theTotalResult->GetNumberOfSecondaries();
687 if (theTotalResult->GetTrackStatus() != fStopAndKill) { // If it is Alive
688 // Either interaction didn't complete, returned "do nothing" state
689 // or the primary survived the interaction (e.g. electro-nucleus )
690 G4Track temp(aTrack);
691
692 // Use the final energy / momentum
693 temp.SetMomentumDirection(*theTotalResult->GetMomentumDirection());
694 temp.SetKineticEnergy(theTotalResult->GetEnergy());
695
696 if( nSec == 0 ){
697 // Interaction didn't complete, returned "do nothing" state
698 // - or suppressed recoil (e.g. Neutron elastic )
699 final4mom = temp.GetDynamicParticle()->Get4Momentum() + target4mom;
700 final_A = initial_A;
701 final_Z = initial_Z;
702 }else{
703 // The primary remains in final state (e.g. electro-nucleus )
704 final4mom = temp.GetDynamicParticle()->Get4Momentum();
705 final_A = track_A;
706 final_Z = track_Z;
707 // Expect that the target nucleus will have interacted,
708 // and its products, including recoil, will be included in secondaries.
709 }
710 }
711 if( nSec > 0 ) {
712 G4Track* sec;
713
714 for (G4int i = 0; i < nSec; i++) {
715 sec = theTotalResult->GetSecondary(i);
716 final4mom += sec->GetDynamicParticle()->Get4Momentum();
717 final_A += sec->GetDefinition()->GetBaryonNumber();
718 final_Z += G4lrint(sec->GetDefinition()->GetPDGCharge());
719 }
720 }
721
722 // Get level-checking information (used to cut-off relative checks)
723 G4String processName = GetProcessName();
724 G4HadronicInteraction* theModel = GetHadronicInteraction();
725 G4String modelName("none");
726 if (theModel) modelName = theModel->GetModelName();
727 std::pair<G4double, G4double> checkLevels = epCheckLevels;
728 if (!levelsSetByProcess) {
729 if (theModel) checkLevels = theModel->GetEnergyMomentumCheckLevels();
730 checkLevels.first= std::min(checkLevels.first, epCheckLevels.first);
731 checkLevels.second=std::min(checkLevels.second, epCheckLevels.second);
732 }
733
734 // Compute absolute total-energy difference, and relative kinetic-energy
735 G4bool checkRelative = (aTrack.GetKineticEnergy() > checkLevels.second);
736
737 G4LorentzVector diff = initial4mom - final4mom;
738 G4double absolute = diff.e();
739 G4double relative = checkRelative ? absolute/aTrack.GetKineticEnergy() : 0.;
740
741 G4double absolute_mom = diff.vect().mag();
742 G4double relative_mom = checkRelative ? absolute_mom/aTrack.GetMomentum().mag() : 0.;
743
744 // Evaluate relative and absolute conservation
745 G4bool relPass = true;
746 G4String relResult = "pass";
747 if ( std::abs(relative) > checkLevels.first
748 || std::abs(relative_mom) > checkLevels.first) {
749 relPass = false;
750 relResult = checkRelative ? "fail" : "N/A";
751 }
752
753 G4bool absPass = true;
754 G4String absResult = "pass";
755 if ( std::abs(absolute) > checkLevels.second
756 || std::abs(absolute_mom) > checkLevels.second ) {
757 absPass = false ;
758 absResult = "fail";
759 }
760
761 G4bool chargePass = true;
762 G4String chargeResult = "pass";
763 if ( (initial_A-final_A)!=0
764 || (initial_Z-final_Z)!=0 ) {
765 chargePass = checkLevels.second < DBL_MAX ? false : true;
766 chargeResult = "fail";
767 }
768
769 G4bool conservationPass = (relPass || absPass) && chargePass;
770
771 std::stringstream Myout;
772 G4bool Myout_notempty(false);
773 // Options for level of reporting detail:
774 // 0. off
775 // 1. report only when E/p not conserved
776 // 2. report regardless of E/p conservation
777 // 3. report only when E/p not conserved, with model names, process names, and limits
778 // 4. report regardless of E/p conservation, with model names, process names, and limits
779 // negative -1.., as above, but send output to stderr
780
781 if( std::abs(epReportLevel) == 4
782 || ( std::abs(epReportLevel) == 3 && ! conservationPass ) ){
783 Myout << " Process: " << processName << " , Model: " << modelName << G4endl;
784 Myout << " Primary: " << aTrack.GetParticleDefinition()->GetParticleName()
785 << " (" << aTrack.GetParticleDefinition()->GetPDGEncoding() << "),"
786 << " E= " << aTrack.GetDynamicParticle()->Get4Momentum().e()
787 << ", target nucleus (" << aNucleus.GetZ_asInt() << ","
788 << aNucleus.GetA_asInt() << ")" << G4endl;
789 Myout_notempty=true;
790 }
791 if ( std::abs(epReportLevel) == 4
792 || std::abs(epReportLevel) == 2
793 || ! conservationPass ){
794
795 Myout << " "<< relResult <<" relative, limit " << checkLevels.first << ", values E/T(0) = "
796 << relative << " p/p(0)= " << relative_mom << G4endl;
797 Myout << " "<< absResult << " absolute, limit (MeV) " << checkLevels.second/MeV << ", values E / p (MeV) = "
798 << absolute/MeV << " / " << absolute_mom/MeV << " 3mom: " << (diff.vect())*1./MeV << G4endl;
799 Myout << " "<< chargeResult << " charge/baryon number balance " << (initial_Z-final_Z) << " / " << (initial_A-final_A) << " "<< G4endl;
800 Myout_notempty=true;
801
802 }
803 Myout.flush();
804 if ( Myout_notempty ) {
805 if (epReportLevel > 0) G4cout << Myout.str()<< G4endl;
806 else if (epReportLevel < 0) G4cerr << Myout.str()<< G4endl;
807 }
808}
