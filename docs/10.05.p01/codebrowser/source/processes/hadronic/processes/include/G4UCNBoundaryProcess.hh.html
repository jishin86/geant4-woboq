70enum G4UCNBoundaryProcessStatus { Undefined,
71 NotAtBoundary,
72 SameMaterial,
73 StepTooSmall,
74 NoMPT, NoMRT,
75 NoMRCondition,
76 Absorption, Ezero, Flip,
77 SpecularReflection,
78 LambertianReflection, MRDiffuseReflection,
79 SnellTransmit, MRDiffuseTransmit
80 };

216////////////////////

219
220inline G4bool
221G4UCNBoundaryProcess::IsApplicable(const G4ParticleDefinition& aParticleType)
222{
223 return ( &aParticleType == G4Neutron::NeutronDefinition() );
224}
225
226inline
227G4UCNBoundaryProcessStatus G4UCNBoundaryProcess::GetStatus() const
228{
229 return theStatus;
230}
231
232inline
233G4bool G4UCNBoundaryProcess::High(G4double Energy, G4double FermiPotDiff)
234{
235 // Returns true for Energy > Fermi Potential Difference
236
237 return (Energy > FermiPotDiff);
238}
239
240inline
241void G4UCNBoundaryProcess::SetMicroRoughness(G4bool active)
242{
243 UseMicroRoughnessReflection = active;
244}
245
246inline
247G4bool G4UCNBoundaryProcess::GetMicroRoughness()
248{
249 return UseMicroRoughnessReflection;
250}
