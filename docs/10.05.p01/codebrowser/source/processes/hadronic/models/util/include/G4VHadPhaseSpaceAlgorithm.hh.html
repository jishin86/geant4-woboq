27// Abstract base class for multibody uniform phase space generators.
28// Subclasses implement a specific algorithm, such as Kopylov, GENBOD,
29// or Makoto's NBody. Subclasses are used by G4HadDecayGenerator.
30//
31// Author: Michael Kelsey (SLAC) <kelsey@slac.stanford.edu>
32
33#ifndef G4VHadPhaseSpaceAlgorithm_HH
34#define G4VHadPhaseSpaceAlgorithm_HH 1
35
36#include "globals.hh"
37#include "G4VHadDecayAlgorithm.hh"
38#include "G4LorentzVector.hh"
39#include "G4ThreeVector.hh"
40#include <vector>
41#include <iosfwd>
42
43class G4VHadPhaseSpaceAlgorithm : public G4VHadDecayAlgorithm {
44public:
45 G4VHadPhaseSpaceAlgorithm(const G4String& algName, G4int verbose=0)
46 : G4VHadDecayAlgorithm(algName, verbose) {;}
47 virtual ~G4VHadPhaseSpaceAlgorithm() {;}
48
49protected:
50 // Multi-body function remains pure-virtual from base class
51
52 // Two-body uniform distribution is common to all algorithms
53 virtual void GenerateTwoBody(G4double initialMass,
54 const std::vector<G4double>& masses,
55 std::vector<G4LorentzVector>& finalState);
56
57 // Sample spherical distribution
58 G4ThreeVector UniformVector(G4double mag=1.) const;
59};
60
61#endif /* G4VHadPhaseSpaceAlgorithm_HH */
