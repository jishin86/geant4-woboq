48G4HadronElastic::G4HadronElastic(const G4String& name)
49 : G4HadronicInteraction(name)
50{
51 SetMinEnergy( 0.0*GeV );
52 SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
53 lowestEnergyLimit= 1.e-6*eV;
54
55 theProton = G4Proton::Proton();
56 theNeutron = G4Neutron::Neutron();
57 theDeuteron = G4Deuteron::Deuteron();
58 theAlpha = G4Alpha::Alpha();
59}

61G4HadronElastic::~G4HadronElastic()
62{}

66{
67
68 outFile << "G4HadronElastic is a hadron-nucleus elastic scattering\n"
69 << "model which uses the Gheisha two-exponential momentum\n"
70 << "transfer parameterization. The model is fully relativistic\n"
71 << "as opposed to the original Gheisha model which was not.\n"
72 << "This model may be used for all long-lived hadrons at all\n"
73 << "incident energies.\n";

77G4HadFinalState* G4HadronElastic::ApplyYourself(
78 const G4HadProjectile& aTrack, G4Nucleus& targetNucleus)
79{
80 theParticleChange.Clear();
81
82 const G4HadProjectile* aParticle = &aTrack;
83 G4double ekin = aParticle->GetKineticEnergy();
84 if(ekin <= lowestEnergyLimit) {
85 theParticleChange.SetEnergyChange(ekin);
86 theParticleChange.SetMomentumChange(aTrack.Get4Momentum().vect().unit());
87 return &theParticleChange;
88 }
89
90 G4int A = targetNucleus.GetA_asInt();
91 G4int Z = targetNucleus.GetZ_asInt();
92
93 G4double plab = aParticle->GetTotalMomentum();
94
95 // Scattered particle referred to axis of incident particle
96 const G4ParticleDefinition* theParticle = aParticle->GetDefinition();
97 G4double m1 = theParticle->GetPDGMass();
98
99 if (verboseLevel>1) {
100 G4cout << "G4HadronElastic: "
101 << aParticle->GetDefinition()->GetParticleName()
102 << " Plab(GeV/c)= " << plab/GeV
103 << " Ekin(MeV) = " << ekin/MeV
104 << " scattered off Z= " << Z
105 << " A= " << A
106 << G4endl;
107 }
108
109 G4double mass2 = G4NucleiProperties::GetNuclearMass(A, Z);
110 G4LorentzVector lv1 = aParticle->Get4Momentum();
111 G4LorentzVector lv(0.0,0.0,0.0,mass2);
112 lv += lv1;
113
114 G4ThreeVector bst = lv.boostVector();
115 lv1.boost(-bst);
116
117 G4ThreeVector p1 = lv1.vect();
118 G4double momentumCMS = p1.mag();
119 G4double tmax = 4.0*momentumCMS*momentumCMS;
120
121 // Sampling in CM system
122 G4double t = SampleInvariantT(theParticle, plab, Z, A);
123 G4double phi = G4UniformRand()*CLHEP::twopi;
124 G4double cost = 1. - 2.0*t/tmax;
125 if(cost > 1.0) { cost = 1.0; }
126 else if(cost < -1.0) { cost = -1.0; }
127 G4double sint = std::sqrt((1.0-cost)*(1.0+cost));
128
129 if (verboseLevel>1) {
130 G4cout << " t= " << t << " tmax(GeV^2)= " << tmax/(GeV*GeV)
131 << " Pcms(GeV)= " << momentumCMS/GeV << " cos(t)=" << cost
132 << " sin(t)=" << sint << G4endl;
133 }
134 G4ThreeVector v1(sint*std::cos(phi),sint*std::sin(phi),cost);
135 v1 *= momentumCMS;
136 G4LorentzVector nlv1(v1.x(),v1.y(),v1.z(),
137 std::sqrt(momentumCMS*momentumCMS + m1*m1));
138
139 nlv1.boost(bst);
140
141 G4double eFinal = nlv1.e() - m1;
142 if (verboseLevel > 1) {
143 G4cout <<"G4HadronElastic: m= " << m1 << " Efin(MeV)= " << eFinal
144 << " Proj: 4-mom " << lv1 << " Final: " << nlv1
145 << G4endl;
146 }
147
148 // precision lost in kinematics, only energy is changed
149 if (eFinal <= 0.0) {
150 G4double mom = nlv1.mag();
151 if(mom == 0.0) {
152 nlv1.set(0.0,0.0,0.0,m1);
153 theParticleChange.SetEnergyChange(0.0);
154 } else {
155 eFinal = mom*mom/(std::sqrt(m1*m1 + mom*mom) + m1);
156 theParticleChange.SetEnergyChange(eFinal);
157 theParticleChange.SetMomentumChange(nlv1.vect().unit());
158 }
159 } else {
160 theParticleChange.SetMomentumChange(nlv1.vect().unit());
161 theParticleChange.SetEnergyChange(eFinal);
162 }
163 lv -= nlv1;
164 G4double erec = lv.e() - mass2;
165 if (verboseLevel > 1) {
166 G4cout << "Recoil: " <<" m= " << mass2 << " Erec(MeV)= " << erec
167 << " 4-mom: " << lv
168 << G4endl;
169 }
170
171 if(erec > GetRecoilEnergyThreshold()) {
172 G4ParticleDefinition * theDef = nullptr;
173 if(Z == 1 && A == 1) { theDef = theProton; }
174 else if (Z == 1 && A == 2) { theDef = theDeuteron; }
175 else if (Z == 1 && A == 3) { theDef = G4Triton::Triton(); }
176 else if (Z == 2 && A == 3) { theDef = G4He3::He3(); }
177 else if (Z == 2 && A == 4) { theDef = theAlpha; }
178 else {
179 theDef =
180 G4ParticleTable::GetParticleTable()->GetIonTable()->GetIon(Z,A,0.0);
181 }
182 G4DynamicParticle * aSec = new G4DynamicParticle(theDef, lv);
183 theParticleChange.AddSecondary(aSec);
184 } else if(erec > 0.0) {
185 theParticleChange.SetLocalEnergyDeposit(erec);
186 }
187
188 return &theParticleChange;
189}

191// sample momentum transfer in the CMS system
192G4double
193G4HadronElastic::SampleInvariantT(const G4ParticleDefinition* p,
194 G4double plab,
195 G4int Z, G4int A)
196{
197 static const G4double GeV2 = GeV*GeV;
198 G4double momentumCMS = ComputeMomentumCMS(p,plab,Z,A);
199 G4double tmax = 4.0*momentumCMS*momentumCMS/GeV2;
200 G4double aa, bb, cc;
201 G4double dd = 10.;
202 G4Pow* g4pow = G4Pow::GetInstance();
203 if (A <= 62) {
204 bb = 14.5*g4pow->Z23(A);
205 aa = g4pow->powZ(A, 1.63)/bb;
206 cc = 1.4*g4pow->Z13(A)/dd;
207 } else {
208 bb = 60.*g4pow->Z13(A);
209 aa = g4pow->powZ(A, 1.33)/bb;
210 cc = 0.4*g4pow->powZ(A, 0.4)/dd;
211 }
212 G4double q1 = 1.0 - G4Exp(-bb*tmax);
213 G4double q2 = 1.0 - G4Exp(-dd*tmax);
214 G4double s1 = q1*aa;
215 G4double s2 = q2*cc;
216 if((s1 + s2)*G4UniformRand() < s2) {
217 q1 = q2;
218 bb = dd;
219 }
220 return -GeV2*G4Log(1.0 - G4UniformRand()*q1)/bb;
221}
