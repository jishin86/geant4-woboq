51// //
52// This class contains the decay times and coefficients for calculating //
53// all the descendants in the decay chains of the named isotope. These //
54// data can be used to calculate their radioactivity at any given time. //
55// //
56////////////////////////////////////////////////////////////////////////////////
57
58#include "G4ios.hh"
59#include "globals.hh"
60#include "G4RadioactiveDecayRatesToDaughter.hh"
61#include <vector>
62
63typedef std::vector<G4RadioactiveDecayRatesToDaughter> G4RadioactiveDecayRates;
64
65class G4RadioactiveDecayChainsFromParent
66{
67 public:
68 G4RadioactiveDecayChainsFromParent();
69 virtual ~G4RadioactiveDecayChainsFromParent();
70
71 G4RadioactiveDecayChainsFromParent(const G4RadioactiveDecayChainsFromParent&);
72 G4RadioactiveDecayChainsFromParent& operator=(const G4RadioactiveDecayChainsFromParent&);
73
74 // equality operators
75 G4int operator==(const G4RadioactiveDecayChainsFromParent& right) const
76 {return (this == &right);}
77 G4int operator!=(const G4RadioactiveDecayChainsFromParent& right) const
78 {return (this != &right);}
79
80 public:
81
82 inline G4String GetIonName() const {return ionName;}
83 inline void SetIonName(G4String name) {ionName = name;}
84
85 // Retrieve the coefficients and decays of all descendants along the
86 // decay chains
87 inline G4RadioactiveDecayRates GetItsRates() const {return itsRates;}
88
89 // Fill in the coefficients and decay times in the chains
90 inline void SetItsRates(G4RadioactiveDecayRates arate) {itsRates = arate;}
91
92protected:
93 G4String ionName;
94 G4RadioactiveDecayRates itsRates;
95
96};
97#endif
