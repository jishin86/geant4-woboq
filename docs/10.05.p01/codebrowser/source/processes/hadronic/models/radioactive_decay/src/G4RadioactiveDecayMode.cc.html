27#include "G4RadioactiveDecayMode.hh"
28
29std::istream &operator >> (std::istream& strm, G4RadioactiveDecayMode& q)
30{
31 G4String a;
32 strm >> a;
33 if (a == "IT")
34 {q = IT;}
35 else if (a == "BetaMinus")
36 {q = BetaMinus;}
37 else if (a == "BetaPlus")
38 {q = BetaPlus;}
39 else if (a == "KshellEC")
40 {q = KshellEC;}
41 else if (a == "LshellEC")
42 {q = LshellEC;}
43 else if (a == "MshellEC")
44 {q = MshellEC;}
45 else if (a == "NshellEC")
46 {q = NshellEC;}
47 else if (a == "Alpha")
48 {q = Alpha;}
49 else if (a == "Proton")
50 {q = Proton;}
51 else if (a == "Neutron")
52 {q = Neutron;}
53 else if (a == "BDProton")
54 {q = BDProton;}
55 else if (a == "BDNeutron")
56 {q = BDNeutron;}
57 else if (a == "Beta2Minus")
58 {q = Beta2Minus;}
59 else if (a == "Beta2Plus")
60 {q = Beta2Plus;}
61 else if (a == "Proton2")
62 {q = Proton2;}
63 else if (a == "Neutron2")
64 {q = Neutron2;}
65 else if (a == "SpFission")
66 {q = SpFission;}
67 else
68 {q = RDM_ERROR;}
69 return strm;
70}
