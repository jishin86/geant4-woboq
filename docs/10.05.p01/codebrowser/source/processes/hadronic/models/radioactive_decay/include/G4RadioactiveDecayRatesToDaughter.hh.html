51// //
52// This class contains the Z, A, E and generation number in a decay chain //
53// of a given daughter nucleus. Decay rate coefficients and decay times of //
54// all ancestor nuclides in decay chains which lead to this daughter are //
55// also kept in this class. //
56// //
57////////////////////////////////////////////////////////////////////////////////
58
59#include "G4ios.hh"
60#include "globals.hh"
61#include <vector>
62
63
64class G4RadioactiveDecayRatesToDaughter
65{
66 public:
67 G4RadioactiveDecayRatesToDaughter();
68 virtual ~G4RadioactiveDecayRatesToDaughter();
69
70 G4RadioactiveDecayRatesToDaughter(const G4RadioactiveDecayRatesToDaughter&);
71 G4RadioactiveDecayRatesToDaughter& operator=(const G4RadioactiveDecayRatesToDaughter&);
72
73 G4int operator==(const G4RadioactiveDecayRatesToDaughter& right) const
74 {return (this == &right);};
75 G4int operator!=(const G4RadioactiveDecayRatesToDaughter& right) const
76 {return (this != &right);};
77
78 public:
79
80 inline G4int GetZ() const {return Z;}
81 inline G4int GetA() const {return A;}
82 inline G4double GetE() const {return E;}
83 inline G4int GetGeneration() const {return generation;}
84 inline std::vector<G4double> GetDecayRateC() const
85 {return decayRateC;}
86 inline std::vector<G4double> GetTaos() const {return taos;}
87
88 inline void SetZ(G4int value) {Z = value;}
89 inline void SetA(G4int value) {A = value;}
90 inline void SetE(G4double value) {E = value;}
91 inline void SetGeneration(G4int value) {generation = value;}
92 inline void SetDecayRateC(std::vector<G4double> value)
93 {decayRateC = value;}
94 inline void SetTaos(std::vector<G4double> value) {taos = value;}
95
96 protected:
97 G4int Z;
98 G4int A;
99 G4double E;
100 G4int generation;
101 std::vector<G4double> decayRateC;
102 std::vector<G4double> taos;
103
104 public:
105
106 inline void SetVerboseLevel(G4int value)
107 {verboseLevel = value;}
108 inline G4int GetVerboseLevel() const
109 {return verboseLevel;}
110 void DumpInfo();
111
112 private:
113 G4int verboseLevel;
114 // control flag for output message
115 // G4int verboseLevel;
116 // 0: Silent
117 // 1: Warning message
118 // 2: More
119
120};
