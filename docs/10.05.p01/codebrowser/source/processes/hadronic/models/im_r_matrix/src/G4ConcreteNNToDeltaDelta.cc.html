42G4ConcreteNNToDeltaDelta::G4ConcreteNNToDeltaDelta(const G4ParticleDefinition* aPrimary,
43 const G4ParticleDefinition* bPrimary,
44 const G4ParticleDefinition* aSecondary,
45 const G4ParticleDefinition* bSecondary) :
46 G4ConcreteNNTwoBodyResonance(NULL, NULL, NULL, NULL, NULL, NULL, NULL)
47{
48 if (!theSigmaTable_G4MT_TLS_) theSigmaTable_G4MT_TLS_ = new G4XDeltaDeltaTable;
49 G4XDeltaDeltaTable &theSigmaTable = *theSigmaTable_G4MT_TLS_;
50 establish_G4MT_TLS_G4ConcreteNNTwoBodyResonance(aPrimary,bPrimary,aSecondary,bSecondary,theSigmaTable);
51}
