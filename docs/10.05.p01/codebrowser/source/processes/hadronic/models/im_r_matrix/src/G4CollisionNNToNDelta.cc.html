28#include "globals.hh"
29#include "G4CollisionNNToNDelta.hh"
30#include "G4ConcreteNNToNDelta.hh"
31#include "G4ShortLivedConstructor.hh"
32
33G4CollisionNNToNDelta::G4CollisionNNToNDelta()
34{
35 components=0;
36 G4ShortLivedConstructor ShortLived;
37 ShortLived.ConstructParticle();
38 MakeNNToNDelta<DeltamPC, Delta0PC, DeltapPC, DeltappPC, G4ConcreteNNToNDelta>::Make(this);
39}
40
41G4CollisionNNToNDelta::~G4CollisionNNToNDelta()
42{
43 if (components) delete components;
44 components=0;
45}
