46typedef G4ConcreteNNToDeltaDelta channelType;
47typedef INT4<channelType, NeutronPC, NeutronPC, Delta0PC, Delta0PC> theC1;
48typedef INT4<channelType, NeutronPC, NeutronPC, DeltamPC, DeltapPC> theC2;
49typedef INT4<channelType, NeutronPC, ProtonPC, Delta0PC, DeltapPC> theC3;
50typedef INT4<channelType, NeutronPC, ProtonPC, DeltamPC, DeltappPC> theC4;
51typedef INT4<channelType, ProtonPC, ProtonPC, DeltapPC, DeltapPC> theC5;
52typedef INT4<channelType, ProtonPC, ProtonPC, Delta0PC, DeltappPC> theC6;
53
54typedef GROUP6(theC1, theC2, theC3, theC4, theC5, theC6) theChannels;
55
56G4CollisionNNToDeltaDelta::G4CollisionNNToDeltaDelta()
57{
58 Resolve aR;
59 G4ForEach<theChannels>::Apply(&aR, this);
60}
