38/** \file G4INCLCascadeAction.hh
39 * \brief Class containing default actions to be performed at intermediate cascade steps
40 *
41 * \date 22nd October 2013
42 * \author Davide Mancusi
43 */
44
45#ifndef G4INCLCASCADEACTION_HH
46#define G4INCLCASCADEACTION_HH 1
47
48#include "G4INCLIAvatar.hh"
49#include "G4INCLNucleus.hh"
50#include "G4INCLFinalState.hh"
51#include "G4INCLIPropagationModel.hh"
52#include "G4INCLIAvatar.hh"
53#include "G4INCLConfig.hh"
54
55namespace G4INCL {
56
57 class CascadeAction {
58 // class INCL must be a friend because it needs to call private methods
59 friend class INCL;
60
61 public:
62 CascadeAction();
63 virtual ~CascadeAction();
64
65 virtual void beforeRunUserAction(Config const *) {}
66 virtual void beforeCascadeUserAction(IPropagationModel *) {}
67 virtual void beforePropagationUserAction(IPropagationModel *) {}
68 virtual void beforeAvatarUserAction(IAvatar *, Nucleus *) {}
69 virtual void afterAvatarUserAction(IAvatar *, Nucleus *, FinalState *) {}
70 virtual void afterPropagationUserAction(IPropagationModel *, IAvatar *) {}
71 virtual void afterCascadeUserAction(Nucleus *) {}
72 virtual void afterRunUserAction() {}
73
74 private:
75 // These four methods should be private because the user must not be
76 // allowed to override them
77 void beforeRunAction(Config const *config);
78 void beforeCascadeAction(IPropagationModel *);
79 void beforePropagationAction(IPropagationModel *pm);
80 void beforeAvatarAction(IAvatar *a, Nucleus *n);
81 void afterAvatarAction(IAvatar *a, Nucleus *n, FinalState *fs);
82 void afterPropagationAction(IPropagationModel *pm, IAvatar *avatar);
83 void afterCascadeAction(Nucleus *);
84 void afterRunAction();
85
86 void beforeRunDefaultAction(Config const *config);
87 void beforeCascadeDefaultAction(IPropagationModel *pm);
88 void beforePropagationDefaultAction(IPropagationModel *pm);
89 void beforeAvatarDefaultAction(IAvatar *a, Nucleus *n);
90 void afterAvatarDefaultAction(IAvatar *a, Nucleus *n, FinalState *fs);
91 void afterPropagationDefaultAction(IPropagationModel *pm, IAvatar *avatar);
92 void afterCascadeDefaultAction(Nucleus *);
93 void afterRunDefaultAction();
94
95 private: // data members
96 long stepCounter;
97 };
98
99}
100#endif // G4INCLCASCADEACTION_HH
