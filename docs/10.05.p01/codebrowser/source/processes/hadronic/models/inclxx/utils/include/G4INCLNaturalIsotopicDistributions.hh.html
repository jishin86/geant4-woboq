38/** \file G4INCLNaturalIsotopicDistributions.hh
39 * \brief Classes that stores isotopic abundances
40 *
41 * \date 21st October 2012
42 * \author Davide Mancusi
43 */
44
45#ifndef G4INCLISOTOPICDISTRIBUTION_HH_
46#define G4INCLISOTOPICDISTRIBUTION_HH_
47
48#include <vector>
49#include <map>
50
51namespace G4INCL {
52
53 /// \brief Holds an isotope and an abundance
54 struct Isotope {
55 Isotope(const G4int A, const G4double abundance);
56 G4int theA;
57 G4double theAbundance;
58 };
59
60 typedef std::vector<Isotope> IsotopeVector;
61 typedef IsotopeVector::iterator IsotopeIter;
62
63 /// \brief Class that stores isotopic abundances for a given element
64 class IsotopicDistribution {
65 public:
66
67 /// \brief Constructor
68 IsotopicDistribution(IsotopeVector const &aVector);
69
70 /// \brief Draw a random isotope based on the abundance vector
71 G4int drawRandomIsotope() const;
72
73 /// \brief Get the isotope vector
74 IsotopeVector const &getIsotopes() const;
75
76 private:
77 IsotopeVector theIsotopes;
78 };
79
80 /// \brief Class that stores isotopic abundances for a given element
81 class NaturalIsotopicDistributions {
82 public:
83
84 /// \brief Constructor
85 NaturalIsotopicDistributions();
86
87 /** \brief Draw a random isotope
88 *
89 * \param Z the element number
90 */
91 G4int drawRandomIsotope(G4int const Z) const;
92
93 /** \brief Get an isotopic distribution
94 *
95 * \param Z the element number
96 */
97 IsotopicDistribution const &getIsotopicDistribution(G4int const Z) const;
98
99 private:
100 std::map<G4int, IsotopicDistribution> theDistributions;
101 };
102
103}
104#endif // G4INCLISOTOPICDISTRIBUTION_HH_
