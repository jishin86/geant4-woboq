64namespace G4INCL {
65
66 enum AvatarType {SurfaceAvatarType,
67 CollisionAvatarType,
68 DecayAvatarType,
69 ParticleEntryAvatarType,
70 UnknownAvatarType};
71
72 class IAvatar {
73 public:
74 IAvatar();
75 IAvatar(G4double time);
76 virtual ~IAvatar();
77
78 virtual G4INCL::IChannel* getChannel() = 0;
79 FinalState *getFinalState();
80 void fillFinalState(FinalState *fs);
81 virtual void preInteraction() = 0;
82 virtual void postInteraction(FinalState *) = 0;
83
84 G4double getTime() const { return theTime; };
85
86 virtual ParticleList getParticles() const = 0;
87
88 virtual std::string dump() const = 0;
89
90 AvatarType getType() const { return type; };
91 G4bool isACollision() const { return (type==CollisionAvatarType); };
92 G4bool isADecay() const { return (type==DecayAvatarType); };
93 void setType(AvatarType t) { type = t; };
94 long getID() const { return ID; };
95
96 std::string toString();
97 private:
98 long ID;
99 AvatarType type;
100 static G4ThreadLocal long nextID;
101 protected:
102 G4double theTime;
103
104 INCL_DECLARE_ALLOCATION_POOL(IAvatar)
105 };
106
107 typedef UnorderedVector<IAvatar*> IAvatarList;
108 typedef UnorderedVector<IAvatar*>::const_iterator IAvatarIter;
109 typedef UnorderedVector<IAvatar*>::iterator IAvatarMutableIter;
110
111}
