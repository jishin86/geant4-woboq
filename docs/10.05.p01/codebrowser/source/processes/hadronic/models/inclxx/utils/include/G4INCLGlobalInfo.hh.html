36#include "globals.hh"

38/** \file G4INCLGlobalInfo.hh
39 * \brief Simple container for output of calculation-wide results.
40 *
41 * Contains the final results of an INCL calculation.
42 *
43 * \date 21 January 2011
44 * \author Davide Mancusi
45 */
46
47#ifndef G4INCLGLOBALINFO_HH
48#define G4INCLGLOBALINFO_HH 1
49
50#ifdef INCL_ROOT_USE
51#include <Rtypes.h>
52#endif
53
54#include <string>
55#include <vector>
56
57namespace G4INCL {
58#ifndef INCL_ROOT_USE
59 typedef G4int Int_t;
60 typedef short Short_t;
61 typedef G4float Float_t;
62#endif
63
64 struct GlobalInfo {
65 GlobalInfo() :
66#ifdef INCL_ROOT_USE
67
68#endif
69 Ap(0),
70 Zp(0),
71 Sp(0),
72 At(0),
73 Zt(0),
74 St(0),
75 Ep((Float_t)0.0),
76 nShots(0),
77 geometricCrossSection((Float_t)0.0),
78 nTransparents(0),
79 reactionCrossSection((Float_t)0.0),
80 errorReactionCrossSection((Float_t)0.0),
81 nNucleonAbsorptions(0),
82 nucleonAbsorptionCrossSection((Float_t)0.0),
83 nPionAbsorptions(0),
84 pionAbsorptionCrossSection((Float_t)0.0),
85 nForcedTransparents(0),
86 nForcedCompoundNucleus(0),
87 forcedCNCrossSection((Float_t)0.0),
88 errorForcedCNCrossSection((Float_t)0.0),
89 nCompleteFusion(0),
90 completeFusionCrossSection((Float_t)0.0),
91 errorCompleteFusionCrossSection((Float_t)0.0),
92 nEnergyViolationInteraction(0),
93 energyViolationInteractionCrossSection((Float_t)0.0)
94 {
95#ifdef INCL_ROOT_USE
96
97#endif
98
99 }
100#ifdef INCL_ROOT_USE
101 /** \brief Selection string for an abridged version of the ROOT tree */
102 std::string rootSelection;
103#endif
104 /** \brief Name of the cascade model */
105 std::string cascadeModel;
106 /** \brief Name of the de-excitation model */
107 std::string deexcitationModel;
108 /** \brief Projectile mass number given as input */
109 Short_t Ap;
110 /** \brief Projectile charge number given as input */
111 Short_t Zp;
112 /** \brief Projectile strangeness number given as input */
113 Short_t Sp;
114 /** \brief Target mass number given as input */
115 Short_t At;
116 /** \brief Target charge number given as input */
117 Short_t Zt;
118 /** \brief Target strangeness number given as input */
119 Short_t St;
120 /** \brief Projectile kinetic energy given as input */
121 Float_t Ep;
122 /** \brief Number of shots */
123 Int_t nShots;
124 /** \brief Geometric cross section */
125 Float_t geometricCrossSection;
126 /** \brief Number of transparent shots */
127 Int_t nTransparents;
128 /** \brief Calculated reaction cross section */
129 Float_t reactionCrossSection;
130 /** \brief Error on the calculated reaction cross section */
131 Float_t errorReactionCrossSection;
132 /** \brief Number of nucleon absorptions (no outcoming particles) */
133 Int_t nNucleonAbsorptions;
134 /** \brief Nucleon absorption cross section */
135 Float_t nucleonAbsorptionCrossSection;
136 /** \brief Number of nucleon absorptions (no outcoming pions) */
137 Int_t nPionAbsorptions;
138 /** \brief Pion absorption cross section */
139 Float_t pionAbsorptionCrossSection;
140 /** \brief Number of forced transparents */
141 Int_t nForcedTransparents;
142 /** \brief Number of forced compound-nucleus events */
143 Int_t nForcedCompoundNucleus;
144 /** \brief Calculated forced-compound-nucleus cross section */
145 Float_t forcedCNCrossSection;
146 /** \brief Error on the calculated forced-compound-nucleus cross section */
147 Float_t errorForcedCNCrossSection;
148 /** \brief Number of complete-fusion events (nParticles==0) */
149 Int_t nCompleteFusion;
150 /** \brief Calculated complete-fusion cross section (nParticles==0) */
151 Float_t completeFusionCrossSection;
152 /** \brief Error on the calculated complete-fusion cross section (nParticles==0) */
153 Float_t errorCompleteFusionCrossSection;
154 /** \brief Number of attempted collisions/decays for which the energy-conservation algorithm failed to find a solution. */
155 Int_t nEnergyViolationInteraction;
156 /** \brief Cross section for attempted collisions/decays for which the energy-conservation algorithm failed to find a solution. */
157 Float_t energyViolationInteractionCrossSection;
158 /** \brief Initial seeds for the pseudo-random-number generator */
159 std::vector<Int_t> initialRandomSeeds;
160 /** \brief Final seeds for the pseudo-random-number generator */
161 std::vector<Int_t> finalRandomSeeds;
162 };
163}
164
165#endif /* G4INCLGLOBALINFO_HH */
