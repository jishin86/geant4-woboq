38/** \file G4INCLNDFParis.hh
39 * \brief NDF* class for the deuteron density according to the Paris
40 * potential.
41 *
42 * \date 16 July 2012
43 * \author Davide Mancusi
44 */
45
46#ifndef G4INCLNDFPARIS_HH_
47#define G4INCLNDFPARIS_HH_
48
49#include "G4INCLIFunction1D.hh"
50#include "G4INCLParticleTable.hh"
51#include "G4INCLDeuteronDensity.hh"
52
53namespace G4INCL {
54
55 namespace NuclearDensityFunctions {
56
57 class ParisR : public IFunction1D {
58 public:
59 ParisR() : // there are no free parameters in the Paris potential
60 // We let the function go up to 17.42 fm
61 IFunction1D(0., 17.42)
62 {};
63
64 inline G4double operator()(const G4double r) const {
65 return DeuteronDensity::densityR(r);
66 }
67 };
68
69 class ParisP : public IFunction1D {
70 public:
71 ParisP() : // there are no free parameters in the Paris potential
72 // We let the function go up to 630 MeV/c
73 IFunction1D(0., 630.)
74 {};
75
76 inline G4double operator()(const G4double p) const {
77 return DeuteronDensity::densityP(p);
78 }
79 };
80
81 }
82
83}
84
85#endif // G4INCLNDFPARIS_HH_
