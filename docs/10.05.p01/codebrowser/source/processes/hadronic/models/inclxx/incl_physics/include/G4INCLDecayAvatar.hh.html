49 /**
50 * Decay avatar
51 *
52 * The reflection avatar is created when a particle reaches the boundary of the nucleus.
53 * At this point it can either be reflected from the boundary or exit the nucleus.
54 */
55 class DecayAvatar: public InteractionAvatar {
56 public:
57 DecayAvatar(G4INCL::Particle *aParticle, G4double time, G4INCL::Nucleus *aNucleus, G4bool force=false);
58 DecayAvatar(G4INCL::Particle *aParticle, G4INCL::Particle *bParticle, G4double time, G4INCL::Nucleus *aNucleus, G4bool force=false);
59 virtual ~DecayAvatar();
60
61 IChannel* getChannel();
62 void fillFinalState(FinalState *fs);
63
64 virtual void preInteraction();
65 virtual void postInteraction(FinalState *fs);
66
67 ParticleList getParticles() const {

70 return theParticleList;
71 }
72
73 std::string dump() const;
74 private:
75 G4bool forced;
76 ThreeVector const incidentDirection;
77
78 INCL_DECLARE_ALLOCATION_POOL(DecayAvatar)
79 };
