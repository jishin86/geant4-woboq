31#ifndef G4INTERACTION_CASE_HH
32#define G4INTERACTION_CASE_HH
33
34#include "globals.hh"
35
36class G4InuclParticle;
37
38
39class G4InteractionCase {
40public:
41 G4InteractionCase() : bullet(0), target(0), inter_case(0) {}
42
43 G4InteractionCase(G4InuclParticle* part1, G4InuclParticle* part2) {
44 set(part1, part2);
45 }
46
47 void set(G4InuclParticle* part1, G4InuclParticle* part2);
48
49 void clear() {
50 bullet = target = 0;
51 inter_case = 0;
52 }
53
54 G4InuclParticle* getBullet() const { return bullet; }
55 G4InuclParticle* getTarget() const { return target; }
56
57 G4bool valid() const { return inter_case != 0; }
58
59 G4bool twoNuclei() const { return inter_case == -2; }
60 G4bool hadNucleus() const { return inter_case == -1; }
61 G4int hadrons() const { return inter_case; } // "rtype" or "is" code
62
63 // For compatibility with G4IntraNucleiCascader code
64 G4int code() const { return ((inter_case<0) ? -inter_case : 0); }
65
66private:
67 G4InuclParticle* bullet;
68 G4InuclParticle* target;
69
70 G4int inter_case;
71};
72
73#endif // G4INTERACTION_CASE_HH
