29// Description: Templated base class for exponential parametrization of
30// two-body angular distributions in Bertini-style cascade
31//
32// 20130227 Renamed from "ParamExp" to "ParamExp" to fix misattribution.
33
34#ifndef G4ParamExpTwoBodyAngDst_h
35#define G4ParamExpTwoBodyAngDst_h 1
36
37#include "G4VTwoBodyAngDst.hh"
38#include "G4CascadeInterpolator.hh"
39
40
41template <G4int NKEBINS>
42class G4ParamExpTwoBodyAngDst : public G4VTwoBodyAngDst {
43public:
44 enum { nKEbins=NKEBINS }; // For use in function arguments
45
46 G4ParamExpTwoBodyAngDst(const G4String& name,
47 const G4double (&kebins)[nKEbins],
48 const G4double (&pFrac)[nKEbins],
49 const G4double (&pA)[nKEbins],
50 const G4double (&pC)[nKEbins],
51 const G4double (&pCos)[nKEbins],
52 G4int verbose = 0)
53 : G4VTwoBodyAngDst(name, verbose), labKE(kebins), angleCut(pFrac),
54 smallScale(pA), largeScale(pC), cosScale(pCos), interpolator(kebins) {;}
55
56 virtual ~G4ParamExpTwoBodyAngDst() {;}
57
58 virtual G4double GetCosTheta(const G4double& ekin, const G4double& pcm) const;
59
60protected:
61 // Kinetic energies at which angular distributions are taken
62 const G4double (&labKE)[nKEbins];
63
64 // Cut-off for small- vs. large-angle functions at each energy
65 const G4double (&angleCut)[nKEbins];
66
67 // Scale factors for small- vs. large-angle regions
68 const G4double (&smallScale)[nKEbins];
69 const G4double (&largeScale)[nKEbins];
70
71 // Scale factor for exponential mapping to cos(theta)
72 const G4double (&cosScale)[nKEbins];
73
74 G4CascadeInterpolator<NKEBINS> interpolator;
75};
76
77// Implementations must be included for templated classes
78#include "G4ParamExpTwoBodyAngDst.icc"
79
80#endif /* G4ParamExpTwoBodyAngDst_h */
