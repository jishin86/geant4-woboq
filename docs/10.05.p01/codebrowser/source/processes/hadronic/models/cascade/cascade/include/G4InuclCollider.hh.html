45#ifndef G4INUCL_COLLIDER_HH
46#define G4INUCL_COLLIDER_HH
47
48#include "G4CascadeColliderBase.hh"
49#include "G4CollisionOutput.hh"
50
51class G4CascadParticle;
52class G4ElementaryParticleCollider;
53class G4IntraNucleiCascader;
54class G4InuclParticle;
55class G4KineticTrackVector;
56class G4V3DNucleus;
57class G4VCascadeDeexcitation;
58
59
60class G4InuclCollider : public G4CascadeColliderBase {
61public:
62 G4InuclCollider();
63 virtual ~G4InuclCollider();
64
65 void collide(G4InuclParticle* bullet, G4InuclParticle* target,
66 G4CollisionOutput& globalOutput);
67
68 // For use with top-level Propagate to preload a set of secondaries
69 void rescatter(G4InuclParticle* bullet, G4KineticTrackVector* theSecondaries,
70 G4V3DNucleus* theNucleus, G4CollisionOutput& globalOutput);
71
72 void setVerboseLevel(G4int verbose=0);
73
74 // Select betweeen different post-cascade de-excitation models
75 void useCascadeDeexcitation();
76 void usePreCompoundDeexcitation();
77
78protected:
79 void deexcite(const G4Fragment& fragment, G4CollisionOutput& globalOutput);
80
81 // Looks for non-gamma final state in photonuclear or leptonuclear
82 G4bool photonuclearOkay(G4CollisionOutput& checkOutput) const;
83
84private:
85 G4ElementaryParticleCollider* theElementaryParticleCollider;
86 G4IntraNucleiCascader* theIntraNucleiCascader;
87
88 G4VCascadeDeexcitation* theDeexcitation; // User switchable!
89
90 G4CollisionOutput output; // Secondaries from main cascade
91 G4CollisionOutput DEXoutput; // Secondaries from de-excitation
92
93private:
94 // Copying of modules is forbidden
95 G4InuclCollider(const G4InuclCollider&);
96 G4InuclCollider& operator=(const G4InuclCollider&);
97};
98
99#endif /* G4INUCL_COLLIDER_HH */
