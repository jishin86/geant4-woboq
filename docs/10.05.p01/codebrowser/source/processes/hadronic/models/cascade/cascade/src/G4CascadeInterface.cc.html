143// Constructor and destrutor
144
145G4CascadeInterface::G4CascadeInterface(const G4String& name)
146 : G4VIntraNuclearTransportModel(name),
147 randomFile(G4CascadeParameters::randomFile()),
148 maximumTries(20), numberOfTries(0),
149 collider(new G4InuclCollider), balance(new G4CascadeCheckBalance(name)),
150 bullet(0), target(0), output(new G4CollisionOutput) {
151 // Set up global objects for master thread or sequential build
152 if (G4Threading::IsMasterThread()) Initialize();
153
154 SetEnergyMomentumCheckLevels(5*perCent, 10*MeV);
155 balance->setLimits(5*perCent, 10*MeV/GeV); // Bertini internal units
156 this->SetVerboseLevel(G4CascadeParameters::verbose());
157
158 if (G4CascadeParameters::usePreCompound())
159 usePreCompoundDeexcitation();
160 else
161 useCascadeDeexcitation();
162}
163
164G4CascadeInterface::~G4CascadeInterface() {
165 clear();
166 delete collider; collider=0;
167 delete balance; balance=0;
168 delete output; output=0;
169}

171void G4CascadeInterface::ModelDescription(std::ostream& outFile) const
172{
173 outFile << "The Bertini-style cascade implements the inelastic scattering\n"
174 << "of hadrons by nuclei. Nucleons, pions, kaons and hyperons\n"
175 << "from 0 to 15 GeV may be used as projectiles in this model.\n"
176 << "Final state hadrons are produced by a classical cascade\n"
177 << "consisting of individual hadron-nucleon scatterings which use\n"
178 << "free-space partial cross sections, corrected for various\n"
179 << "nuclear medium effects. The target nucleus is modeled as a\n"
180 << "set of 1, 3 or 6 spherical shells, in which scattered hadrons\n"
181 << "travel in straight lines until they are reflected from or\n"
182 << "transmitted through shell boundaries.\n";
183}

244// Main Actions
245
246G4HadFinalState*
247G4CascadeInterface::ApplyYourself(const G4HadProjectile& aTrack,
248 G4Nucleus& theNucleus) {
249 if (verboseLevel)
250 G4cout << " >>> G4CascadeInterface::ApplyYourself" << G4endl;
251
252 if (aTrack.GetKineticEnergy() < 0.) {
253 G4cerr << " >>> G4CascadeInterface got negative-energy track: "
254 << aTrack.GetDefinition()->GetParticleName() << " Ekin = "
255 << aTrack.GetKineticEnergy() << G4endl;
256 }
257
258#ifdef G4CASCADE_DEBUG_INTERFACE
259 static G4int counter(0);
260 counter++;
261 G4cerr << "Reaction number "<< counter << " "
262 << aTrack.GetDefinition()->GetParticleName() << " "
263 << aTrack.GetKineticEnergy() << " MeV" << G4endl;
264#endif
265
266 if (!randomFile.empty()) { // User requested random-seed capture
267 if (verboseLevel>1)
268 G4cout << " Saving random engine state to " << randomFile << G4endl;
269 CLHEP::HepRandom::saveEngineStatus(randomFile);
270 }
271
272 theParticleChange.Clear();
273 clear();
274
275 // Abort processing if no interaction is possible
276 if (!IsApplicable(aTrack, theNucleus)) {
277 if (verboseLevel) G4cerr << " No interaction possible " << G4endl;
278 return NoInteraction(aTrack, theNucleus);
279 }
280
281 // Make conversion between native Geant4 and Bertini cascade classes.
282 if (!createBullet(aTrack)) {
283 if (verboseLevel) G4cerr << " Unable to create usable bullet" << G4endl;
284 return NoInteraction(aTrack, theNucleus);
285 }
286
287 if (!createTarget(theNucleus)) {
288 if (verboseLevel) G4cerr << " Unable to create usable target" << G4endl;
289 return NoInteraction(aTrack, theNucleus);
290 }
291
292 // Different retry conditions for proton target vs. nucleus
293 const G4bool isHydrogen = (theNucleus.GetA_asInt() == 1);
294
295 numberOfTries = 0;
296 do { // we try to create inelastic interaction
297 if (verboseLevel > 1)
298 G4cout << " Generating cascade attempt " << numberOfTries << G4endl;
299
300 output->reset();
301 collider->collide(bullet, target, *output);
302 balance->collide(bullet, target, *output);
303
304 numberOfTries++;
305 /* Loop checking 08.06.2015 MHK */
306 } while ( isHydrogen ? retryInelasticProton() : retryInelasticNucleus() );
307
308 // Null event if unsuccessful
309 if (numberOfTries >= maximumTries) {
310 if (verboseLevel)
311 G4cout << " Cascade aborted after trials " << numberOfTries << G4endl;
312 return NoInteraction(aTrack, theNucleus);
313 }
314
315 // Abort job if energy or momentum are not conserved
316 if (!balance->okay()) {
317 throwNonConservationFailure();
318 return NoInteraction(aTrack, theNucleus);
319 }
320
321 // Successful cascade -- clean up and return
322 if (verboseLevel) {
323 G4cout << " Cascade output after trials " << numberOfTries << G4endl;
324 if (verboseLevel > 1) output->printCollisionOutput();
325 }
326
327 // Rotate event to put Z axis along original projectile direction
328 // Removed by DHW to fix bug #1990
329 // output->rotateEvent(bulletInLabFrame);
330
331 copyOutputToHadronicResult();
332
333 // Report violations of conservation laws in original frame
334 checkFinalResult();
335
336 // Clean up and return final result;
337 clear();
338/*
339 G4int nSec = theParticleChange.GetNumberOfSecondaries();
340 for (G4int i = 0; i < nSec; i++) {
341 G4HadSecondary* sec = theParticleChange.GetSecondary(i);
342 G4DynamicParticle* dp = sec->GetParticle();
343 if (dp->GetDefinition()->GetParticleName() == "neutron")
344 G4cout << dp->GetDefinition()->GetParticleName() << " has "
345 << dp->GetKineticEnergy()/MeV << " MeV " << G4endl;
346 }
347*/
348 return &theParticleChange;
349}

351G4ReactionProductVector*
352G4CascadeInterface::Propagate(G4KineticTrackVector* theSecondaries,
353 G4V3DNucleus* theNucleus) {
354 if (verboseLevel) G4cout << " >>> G4CascadeInterface::Propagate" << G4endl;
355
356#ifdef G4CASCADE_DEBUG_INTERFACE
357 if (verboseLevel>1) {
358 G4cout << " G4V3DNucleus A " << theNucleus->GetMassNumber()
359 << " Z " << theNucleus->GetCharge()
360 << "\n " << theSecondaries->size() << " secondaries:" << G4endl;
361 for (size_t i=0; i<theSecondaries->size(); i++) {
362 G4KineticTrack* kt = (*theSecondaries)[i];
363 G4cout << " " << i << ": " << kt->GetDefinition()->GetParticleName()
364 << " p " << kt->Get4Momentum() << " @ " << kt->GetPosition()
365 << " t " << kt->GetFormationTime() << G4endl;
366 }
367 }
368#endif
369
370 if (!randomFile.empty()) { // User requested random-seed capture
371 if (verboseLevel>1)
372 G4cout << " Saving random engine state to " << randomFile << G4endl;
373 CLHEP::HepRandom::saveEngineStatus(randomFile);
374 }
375
376 theParticleChange.Clear();
377 clear();
378
379 // Process input secondaries list to eliminate resonances
380 G4DecayKineticTracks decay(theSecondaries);
381
382 // NOTE: Requires 9.4-ref-03 mods to base class and G4TheoFSGenerator
383 const G4HadProjectile* projectile = GetPrimaryProjectile();
384 if (projectile) createBullet(*projectile);
385
386 if (!createTarget(theNucleus)) {
387 if (verboseLevel) G4cerr << " Unable to create usable target" << G4endl;
388 return 0; // FIXME: This will cause a segfault later
389 }
390
391 numberOfTries = 0;
392 do {
393 if (verboseLevel > 1)
394 G4cout << " Generating rescatter attempt " << numberOfTries << G4endl;
395
396 output->reset();
397 collider->rescatter(bullet, theSecondaries, theNucleus, *output);
398 balance->collide(bullet, target, *output);
399
400 numberOfTries++;
401 // FIXME: retry checks will SEGFAULT until we can define the bullet!
402 } while (retryInelasticNucleus()); /* Loop checking 08.06.2015 MHK */
403
404 // Check whether repeated attempts have all failed; report and exit
405 if (numberOfTries >= maximumTries && !balance->okay()) {
406 throwNonConservationFailure(); // This terminates the job
407 }
408
409 // Successful cascade -- clean up and return
410 if (verboseLevel) {
411 G4cout << " Cascade rescatter after trials " << numberOfTries << G4endl;
412 if (verboseLevel > 1) output->printCollisionOutput();
413 }
414
415 // Does calling code take ownership? I hope so!
416 G4ReactionProductVector* propResult = copyOutputToReactionProducts();
417
418 // Clean up and and return final result
419 clear();
420 return propResult;
421}

426G4HadFinalState*
427G4CascadeInterface::NoInteraction(const G4HadProjectile& aTrack,
428 G4Nucleus& /*theNucleus*/) {
429 if (verboseLevel)
430 G4cout << " >>> G4CascadeInterface::NoInteraction" << G4endl;
431
432 theParticleChange.Clear();
433 theParticleChange.SetStatusChange(isAlive);
434
435 G4double ekin = aTrack.GetKineticEnergy()>0. ? aTrack.GetKineticEnergy() : 0.;
436 theParticleChange.SetEnergyChange(ekin); // Protect against rounding
437
438 return &theParticleChange;
439}

442// Convert input projectile to Bertini internal object
443
444G4bool G4CascadeInterface::createBullet(const G4HadProjectile& aTrack) {
445 const G4ParticleDefinition* trkDef = aTrack.GetDefinition();
446 G4int bulletType = 0; // For elementary particles
447 G4int bulletA = 0, bulletZ = 0; // For nucleus projectile
448
449 if (trkDef->GetAtomicMass() <= 1) {
450 bulletType = G4InuclElementaryParticle::type(trkDef);
451 } else {
452 bulletA = trkDef->GetAtomicMass();
453 bulletZ = trkDef->GetAtomicNumber();
454 }
455
456 if (0 == bulletType && 0 == bulletA*bulletZ) {
457 if (verboseLevel) {
458 G4cerr << " G4CascadeInterface: " << trkDef->GetParticleName()
459 << " not usable as bullet." << G4endl;
460 }
461 bullet = 0;
462 return false;
463 }
464
465 // Code momentum and energy -- Bertini wants z-axis and GeV units
466 G4LorentzVector projectileMomentum = aTrack.Get4Momentum()/GeV;
467
468 // Rotation/boost to get from z-axis back to original frame
469 // According to bug report #1990 this rotation is unnecessary and causes
470 // irreproducibility. Verifed and fixed by DHW 27 Nov 2017
471 // bulletInLabFrame = G4LorentzRotation::IDENTITY; // Initialize
472 // bulletInLabFrame.rotateZ(-projectileMomentum.phi());
473 // bulletInLabFrame.rotateY(-projectileMomentum.theta());
474 // bulletInLabFrame.invert();
475
476 G4LorentzVector momentumBullet(0., 0., projectileMomentum.rho(),
477 projectileMomentum.e());
478
479 if (G4InuclElementaryParticle::valid(bulletType)) {
480 hadronBullet.fill(momentumBullet, bulletType);
481 bullet = &hadronBullet;
482 } else {
483 nucleusBullet.fill(momentumBullet, bulletA, bulletZ);
484 bullet = &nucleusBullet;
485 }
486
487 if (verboseLevel > 2) G4cout << "Bullet: \n" << *bullet << G4endl;
488
489 return true;
490}

518// Convert Bertini particle to output (G4DynamicParticle)
519
520G4DynamicParticle* G4CascadeInterface::
521makeDynamicParticle(const G4InuclElementaryParticle& iep) const {
522 G4int outgoingType = iep.type();
523
524 if (iep.quasi_deutron()) {
525 G4cerr << " ERROR: G4CascadeInterface incompatible particle type "
526 << outgoingType << G4endl;
527 return 0;
528 }
529
530 // Copy local G4DynPart to public output (handle kaon mixing specially)
531 if (outgoingType == kaonZero || outgoingType == kaonZeroBar) {
532 G4ThreeVector momDir = iep.getMomentum().vect().unit();
533 G4double ekin = iep.getKineticEnergy()*GeV; // Bertini -> G4 units
534
535 G4ParticleDefinition* pd = G4KaonZeroShort::Definition();
536 if (G4UniformRand() > 0.5) pd = G4KaonZeroLong::Definition();
537
538 return new G4DynamicParticle(pd, momDir, ekin);
539 } else {
540 return new G4DynamicParticle(iep.getDynamicParticle());
541 }
542
543 return 0; // Should never get here!
544}

553 return new G4DynamicParticle(inuc.getDynamicParticle());
554}

557// Transfer Bertini internal final state to hadronics interface
558
559void G4CascadeInterface::copyOutputToHadronicResult() {
560 if (verboseLevel > 1)
561 G4cout << " >>> G4CascadeInterface::copyOutputToHadronicResult" << G4endl;
562
563 const std::vector<G4InuclNuclei>& outgoingNuclei = output->getOutgoingNuclei();
564 const std::vector<G4InuclElementaryParticle>& particles = output->getOutgoingParticles();
565
566 theParticleChange.SetStatusChange(stopAndKill);
567 theParticleChange.SetEnergyChange(0.);
568
569 // Get outcoming particles
570 if (!particles.empty()) {
571 particleIterator ipart = particles.begin();
572 for (; ipart != particles.end(); ipart++) {
573 theParticleChange.AddSecondary(makeDynamicParticle(*ipart));
574 }
575 }
576
577 // get nuclei fragments
578 if (!outgoingNuclei.empty()) {
579 nucleiIterator ifrag = outgoingNuclei.begin();
580 for (; ifrag != outgoingNuclei.end(); ifrag++) {
581 theParticleChange.AddSecondary(makeDynamicParticle(*ifrag));
582 }
583 }
584}
585
586G4ReactionProductVector* G4CascadeInterface::copyOutputToReactionProducts() {
587 if (verboseLevel > 1)
588 G4cout << " >>> G4CascadeInterface::copyOutputToReactionProducts" << G4endl;
589
590 const std::vector<G4InuclElementaryParticle>& particles = output->getOutgoingParticles();
591 const std::vector<G4InuclNuclei>& fragments = output->getOutgoingNuclei();
592
593 G4ReactionProductVector* propResult = new G4ReactionProductVector;
594
595 G4ReactionProduct* rp = 0; // Buffers to create outgoing tracks
596 G4DynamicParticle* dp = 0;
597
598 // Get outcoming particles
599 if (!particles.empty()) {
600 particleIterator ipart = particles.begin();
601 for (; ipart != particles.end(); ipart++) {
602 rp = new G4ReactionProduct;
603 dp = makeDynamicParticle(*ipart);
604 (*rp) = (*dp); // This does all the necessary copying
605 propResult->push_back(rp);
606 delete dp;
607 }
608 }
609
610 // get nuclei fragments
611 if (!fragments.empty()) {
612 nucleiIterator ifrag = fragments.begin();
613 for (; ifrag != fragments.end(); ifrag++) {
614 rp = new G4ReactionProduct;
615 dp = makeDynamicParticle(*ifrag);
616 (*rp) = (*dp); // This does all the necessary copying
617 propResult->push_back(rp);
618 delete dp;
619 }
620 }
621
622 return propResult;
623}

626// Report violations of conservation laws in original frame
627
628void G4CascadeInterface::checkFinalResult() {
629 balance->collide(bullet, target, *output);
630
631 if (verboseLevel > 2) {
632 if (!balance->baryonOkay()) {
633 G4cerr << "ERROR: no baryon number conservation, sum of baryons = "
634 << balance->deltaB() << G4endl;
635 }
636
637 if (!balance->chargeOkay()) {
638 G4cerr << "ERROR: no charge conservation, sum of charges = "
639 << balance->deltaQ() << G4endl;
640 }
641
642 if (std::abs(balance->deltaKE()) > 0.01 ) { // GeV
643 G4cerr << "Kinetic energy conservation violated by "
644 << balance->deltaKE() << " GeV" << G4endl;
645 }
646
647 G4double eInit = bullet->getEnergy() + target->getEnergy();
648 G4double eFinal = eInit + balance->deltaE();
649
650 G4cout << "Initial energy " << eInit << " final energy " << eFinal
651 << "\nTotal energy conservation at level "
652 << balance->deltaE() * GeV << " MeV" << G4endl;
653
654 if (balance->deltaKE() > 5.0e-5 ) { // 0.05 MeV
655 G4cerr << "FATAL ERROR: kinetic energy created "
656 << balance->deltaKE() * GeV << " MeV" << G4endl;
657 }
658 }

662// Evaluate whether any outgoing particles penetrated Coulomb barrier
663
664G4bool G4CascadeInterface::coulombBarrierViolation() const {
665 G4bool violated = false; // by default coulomb analysis is OK
666
667 const G4double coulumbBarrier = 8.7 * MeV/GeV; // Bertini uses GeV
668
669 const std::vector<G4InuclElementaryParticle>& p =
670 output->getOutgoingParticles();
671
672 for (particleIterator ipart=p.begin(); ipart != p.end(); ipart++) {
673 if (ipart->type() == proton) {
674 violated |= (ipart->getKineticEnergy() < coulumbBarrier);
675 }
676 }
677
678 return violated;
679}

764void G4CascadeInterface::throwNonConservationFailure() {

769 errInfo << " >>> G4CascadeInterface has non-conserving"
