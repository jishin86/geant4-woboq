40// Class Description
41// Trivial implementation of an intra-nuclear transport. It pworvides coupling
42// of high energy generators with pre equilibrium decay models.
43// To be used in your physics list in case you need this physics.
44// Class Description - End
45
46#ifndef G4GeneratorPrecompoundInterface_h
47#define G4GeneratorPrecompoundInterface_h 1
48
49#include "G4VIntraNuclearTransportModel.hh"
50#include "G4ReactionProductVector.hh"
51#include "G4HadProjectile.hh"
52#include "G4Nucleus.hh"
53#include "globals.hh"
54
55class G4KineticTrackVector;
56class G4V3DNucleus;
57class G4ParticleDefinition;
58
59class G4GeneratorPrecompoundInterface : public G4VIntraNuclearTransportModel
60{
61public:
62
63 G4GeneratorPrecompoundInterface(G4VPreCompoundModel* p = 0);
64 virtual ~G4GeneratorPrecompoundInterface();
65
66 virtual G4HadFinalState*
67 ApplyYourself(const G4HadProjectile &aTrack, G4Nucleus &targetNucleus );
68
69 virtual G4ReactionProductVector*
70 Propagate(G4KineticTrackVector* theSecondaries, G4V3DNucleus* theNucleus);
71
72 virtual G4ReactionProductVector* // Uzhi Nov. 2012
73 PropagateNuclNucl(G4KineticTrackVector* theSecondaries, G4V3DNucleus* theNucleus,
74 G4V3DNucleus* theProjectileNucleus);
75
76 inline void SetCaptureThreshold(G4double);
77
78 virtual void PropagateModelDescription(std::ostream&) const;
79
80private:
81
82 G4GeneratorPrecompoundInterface(const G4GeneratorPrecompoundInterface& right);
83 const G4GeneratorPrecompoundInterface& operator=(const G4GeneratorPrecompoundInterface &right);
84 G4int operator==(G4GeneratorPrecompoundInterface& right) {return (this == &right);}
85 G4int operator!=(G4GeneratorPrecompoundInterface& right) {return (this != &right);}
86
87 G4double CaptureThreshold;
88 const G4ParticleDefinition* proton;
89 const G4ParticleDefinition* neutron;
90
91 const G4ParticleDefinition* deuteron;
92 const G4ParticleDefinition* triton;
93 const G4ParticleDefinition* He3;
94 const G4ParticleDefinition* He4;
95
96 const G4ParticleDefinition* ANTIproton;
97 const G4ParticleDefinition* ANTIneutron;
98
99 const G4ParticleDefinition* ANTIdeuteron;
100 const G4ParticleDefinition* ANTItriton;
101 const G4ParticleDefinition* ANTIHe3;
102 const G4ParticleDefinition* ANTIHe4;
103};
104
105inline
106void G4GeneratorPrecompoundInterface::SetCaptureThreshold(G4double value)
107{
108 CaptureThreshold=value;
109}
110
111#endif // G4GeneratorPrecompoundInterface_h
