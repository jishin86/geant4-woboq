33#ifndef G4CameronTruranHilfShellCorrections_h
34#define G4CameronTruranHilfShellCorrections_h 1
35
36#include "globals.hh"
37
38class G4CameronTruranHilfShellCorrections
39{
40public:
41
42 explicit G4CameronTruranHilfShellCorrections();
43
44 inline G4bool GetShellCorrection(G4int N, G4int Z, G4double& result) const
45 {
46 G4bool res = false;
47 if(Z >= ZTableMin && Z <= ZTableMax && N >= NTableMin && N <= NTableMax) {
48 result = ShellZTable[Z-ZTableMin] + ShellNTable[N-NTableMin];
49 res = true;
50 }
51 return res;
52 }
53
54 enum { ZTableSize = 93, NTableSize = 146, ZTableMin = 10, ZTableMax = 102,
55 NTableMin = 10, NTableMax = 155 };
56
57private:
58
59 G4CameronTruranHilfShellCorrections(const G4CameronTruranHilfShellCorrections & right) = delete;
60 const G4CameronTruranHilfShellCorrections & operator=(const G4CameronTruranHilfShellCorrections & right) = delete;
61
62 static G4double ShellZTable[ZTableSize];
63 static G4double ShellNTable[NTableSize];
64
65};
66#endif
