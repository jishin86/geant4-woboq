33#ifndef G4CookShellCorrections_h
34#define G4CookShellCorrections_h 1
35
36#include "globals.hh"
37
38class G4CookShellCorrections
39{
40public:
41
42 explicit G4CookShellCorrections();
43
44 inline G4bool GetShellCorrection(G4int N, G4int Z, G4double& result) const
45 {
46 G4bool res = false;
47 if(Z >= ZTableMin && Z <= ZTableMax && N >= NTableMin && N <= NTableMax) {
48 result = ShellZTable[Z-ZTableMin] + ShellNTable[N-NTableMin];
49 res = true;
50 }
51 return res;
52 }
53
54 enum { ZTableSize = 68, NTableSize = 118, ZTableMin = 28, ZTableMax = 95,
55 NTableMin = 33, NTableMax = 150 };
56private:
57
58 G4CookShellCorrections(const G4CookShellCorrections & right) = delete;
59 const G4CookShellCorrections & operator=(const G4CookShellCorrections & right) = delete;
60
61 static G4double ShellZTable[ZTableSize];
62 static G4double ShellNTable[NTableSize];
63
64};
65#endif
