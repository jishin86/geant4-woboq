31#ifndef G4VCoulombBarrier_h
32#define G4VCoulombBarrier_h 1
33
34#include "globals.hh"
35
36class G4VCoulombBarrier
37{
38public:
39
40 explicit G4VCoulombBarrier(G4int anA, G4int aZ);
41 virtual ~G4VCoulombBarrier();
42
43 virtual G4double GetCoulombBarrier(G4int ARes, G4int ZRes,
44 G4double U) const = 0;
45
46 virtual G4double BarrierPenetrationFactor(G4int Eexc) const = 0;
47
48 void SetParameters(G4double rho, G4double r0);
49
50 inline G4int GetA(void) const { return theA; }
51 inline G4int GetZ(void) const { return theZ; }
52 inline G4double GetRho(void) const { return theRho; }
53 inline G4double GetR0(void) const { return theR0; }
54
55private:
56
57 G4VCoulombBarrier(const G4VCoulombBarrier & right) = delete;
58 const G4VCoulombBarrier & operator=(const G4VCoulombBarrier & right) = delete;
59 G4bool operator==(const G4VCoulombBarrier & right) const = delete;
60 G4bool operator!=(const G4VCoulombBarrier & right) const = delete;
61
62 G4int theA;
63 G4int theZ;
64 G4double theRho;
65 G4double theR0;
66};
