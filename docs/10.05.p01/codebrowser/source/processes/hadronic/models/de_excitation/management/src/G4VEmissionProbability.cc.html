40 :OPTxs(3),fVerbose(0),theZ(Z),theA(A),elimit(CLHEP::MeV),accuracy(0.02)

57void G4VEmissionProbability::ResetIntegrator(size_t nbin, G4double de, G4double eps)
58{
59 if(nbin > 0) {
60 fProb.clear();
61 fEner.clear();
62 fEner.resize(nbin+1, 0.0);
63 fProb.resize(nbin+1, 0.0);
64 length = nbin;
65 }
66 if(de > 0.0) { elimit = de; }
67 if(accuracy > 0.0) { accuracy = eps; }
68}

75G4double
76G4VEmissionProbability::IntegrateProbability(G4double elow, G4double ehigh, G4double cb)
77{
78 totProbability = 0.0;
79 if(elow >= ehigh) { return totProbability; }
80
81 emin = elow;
82 emax = ehigh;
83 eCoulomb = cb;
84
85 size_t nbin = (size_t)((emax - emin)/elimit) + 1;
86 G4double edelta = elimit;
87 if(nbin < 3) {
88 nbin = 3;
89 edelta = (emax - emin)/(G4double)nbin;
90 }
91 if(nbin > length) {
92 fEner.resize(nbin + 1);
93 fProb.resize(nbin + 1);
94 length = nbin;
95 }
96
97 G4double x(emin), del, y;
98 eprobmax = x + edelta*0.1;
99 G4double problast = ComputeProbability(eprobmax, eCoulomb);
100 probmax = problast;
101 fEner[0] = emin;
102 fProb[0] = probmax;
103 size_t i(0);
104 if(fVerbose > 1) {
105 G4cout << "### G4VEmissionProbability::IntegrateProbability: "
106 << " Emax= " << emax << " QB= " << cb << " nbin= " << nbin
107 << G4endl;
108 G4cout << " 0. E= " << emin << " prob= " << problast << G4endl;
109 }
110 static const G4double edeltamin = 0.2*CLHEP::MeV;
111 static const G4double edeltamax = 2*CLHEP::MeV;
112 G4bool peak = true;
113 do {
114 ++i;
115 x += edelta;
116 if(x > emax) {
117 x = emax;
118 edelta = emax - fEner[i-1];
119 }
120 y = ComputeProbability(x, eCoulomb);
121 if(fVerbose > 1) {
122 G4cout << " " << i << ". E= " << x << " prob= " << y
123 << " Edel= " << edelta << G4endl;
124 }
125 fEner[i] = x;
126 fProb[i] = y;
127 del = (y + problast)*edelta*0.5;
128 totProbability += del;
129
130 // smart step definition
131 if(del < accuracy*totProbability) { break; }
132 else if(del != totProbability && del > 0.8*totProbability
133 && edelta > edeltamin)
134 { edelta *= 0.5; }
135 else if(del < 0.1*totProbability && edelta < edeltamax)
136 { edelta *= 2.0; }
137 if(y > probmax) {
138 probmax = y;
139 eprobmax = x;
140 } else if(peak && y < probmax) {
141 edelta *= 2.0;
142 peak = false;
143 }
144 problast = y;
145 // Loop checking, 10-Mar-2017, Vladimir Ivanchenko
146 } while(i < nbin && x < emax);
147
148 nfilled = i;
149 if(fVerbose > 1) { G4cout << " Probability= " << totProbability << G4endl; }
150 return totProbability;
151}

153G4double G4VEmissionProbability::SampleEnergy()
154{
155 static const G4double fact = 1.1;
156 probmax *= fact;
157 if(0.0 == fProb[nfilled] && nfilled > 2) { --nfilled; }
158 for(size_t i=0; i<=nfilled; ++i) {
159 fProb[i] *= fact;
160 }
161 G4double ekin(0.0), s1(1.0), s2(0.0), ksi(0.0), psi(1.0), z(0.0);
162
163 if(fVerbose > 1) {
164 G4cout << "### G4VEmissionProbability::SampleEnergy: "
165 << " Emin= " << emin << " Emax= " << emax
166 << " Nf= " << nfilled << G4endl;
167 }
168 G4double x0 = emax;
169 G4double x1 = fEner[nfilled-1];
170 G4double x2 = fEner[nfilled];
171 G4double y0 = probmax;
172 G4double y1 = fProb[nfilled-1];
173 G4double y2 = fProb[nfilled];
174 G4bool islog(false);
175 // a condition if a special treatment of falling down
176 // distribution is needed
177 //
178 G4double ee = 0.5*(emax - emin);
179 if(nfilled > 5 && y2 > 0.0 && y2 < 0.1*y0 && y1 > 0.0 && x1 - emin < ee) {
180 ksi = G4Log(y1/y2)/G4Log(x2/x1);
181 x0 = x2*G4Exp(-G4Log(y0/y2)/ksi);
182 // general condition to have two sampling area
183 if(x0 < x1 && x0 > eprobmax && x0 - emin < ee) {
184 // condition when the first area does not exist
185 if(x0 <= emin) {
186 s1 = 0.0;
187 s2 = 1.0;
188 y0 *= G4Exp(G4Log(x0/emin)*ksi);
189 x0 = emin;
190 } else {
191 s1 = (x0 - emin)*y0;
192 }
193 // parameters of the second area
194 if(std::abs(1.0 - ksi) < 0.1) {
195 z = G4Log(emax/x0);
196 if(s1 > 0.0) { s2 = x0*y0*z; }
197 islog = true;
198 } else {
199 psi = 1.0/(1.0 - ksi);
200 z = G4Exp(G4Log(emax/x0)*(1. - ksi)) - 1.;
201 if(s1 > 0.0) { s2 = std::max(y0*x0*z*psi, 0.0); }
202 }
203 }
204 }
205 G4double sum = s1 + s2;
206 if(fVerbose > 1) {
207 G4cout << " Epeak= " << eprobmax << " e0= " << x0 << " e1= " << x1 << " e2= " << x2
208 << " psi= " << psi << G4endl;
209 G4cout << " s1= " << s1 << " s2= " << s2
210 << " y0= " << y0 << " y1= " << y1 << " y2= " << y2 << " z= " << z
211 << G4endl;
212 for(size_t i=0; i<=nfilled; ++i) {
213 G4cout << i << ". E= " << fEner[i] << " P= " << fProb[i] << G4endl;
214 }
215 }
216
217 CLHEP::HepRandomEngine* rndm = G4Random::getTheEngine();
218 static const G4int nmax = 100;
219 G4double gr, g;
220 for(size_t i=0; i<nmax; ++i) {
221 G4double q = sum*rndm->flat();
222 if(s2 <= 0.0 || q <= s1) {
223 gr = y0;
224 ekin = emin + (x0 - emin)*q/s1;
225 } else if(islog) {
226 ekin = x0*G4Exp((q - s1)*z/s2);
227 gr = y0*x0/ekin;
228 } else {
229 ekin = x0*G4Exp(G4Log(1.0 + (q - s1)*z/s2)*psi);
230 gr = y0*G4Exp(G4Log(x0/ekin)*ksi);
231 }
232 g = ComputeProbability(ekin, eCoulomb);
233 if(fVerbose > 1) {
234 G4cout << " " << i
235 << ". prob= " << g << " probmax= " << gr
236 << " Ekin= " << ekin << G4endl;
237 }
238 if(g > gr && fVerbose > 0) {
239 G4cout << "### G4VEmissionProbability::SampleEnergy Warning i= " << i
240 << " prob/probmax= " << g/gr << " rndm= " << q
241 << "\n prob= " << g << " probmax= " << gr
242 << " z= " << z << " ksi= " << ksi
243 << "\n Ekin= " << ekin << " Emin= " << emin
244 << " Emax= " << emax << " E0= " << x0 << G4endl;
245 G4cout << " s1= " << s1 << " s2= " << s2
246 << " y0= " << y0 << " y1= " << y1 << " y2= " << y2 << G4endl;
247 for(size_t j=0; j<=nfilled; ++j) {
248 G4cout << j << ". E= " << fEner[j] << " P= " << fProb[j] << G4endl;
249 }
250 }
251 if(gr*rndm->flat() <= g) { break; }
252 }
253 return ekin;
254}
