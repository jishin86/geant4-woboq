45G4CompetitiveFission::G4CompetitiveFission() : G4VEvaporationChannel("fission")
46{
47 theFissionBarrierPtr = new G4FissionBarrier;
48 MyOwnFissionBarrier = true;
49
50 theFissionProbabilityPtr = new G4FissionProbability;
51 MyOwnFissionProbability = true;
52
53 theLevelDensityPtr = new G4FissionLevelDensityParameter;
54 MyOwnLevelDensity = true;
55
56 MaximalKineticEnergy = 0.0;
57 FissionBarrier = 0.0;

69G4double G4CompetitiveFission::GetEmissionProbability(G4Fragment* fragment)
70{
71 G4int anA = fragment->GetA_asInt();
72 FissionProbability = 0.0;
73 if (anA >= 65) {
74 G4int aZ = fragment->GetZ_asInt();
75 G4double ExEnergy = fragment->GetExcitationEnergy() -
76 pairingCorrection->GetFissionPairingCorrection(anA,aZ);
77
78 // Saddle point excitation energy ---> A = 65
79 // Fission is excluded for A < 65
80 if (ExEnergy > 0.0) {
81 FissionBarrier = theFissionBarrierPtr->FissionBarrier(anA,aZ,ExEnergy);
82 MaximalKineticEnergy = ExEnergy - FissionBarrier;
83 FissionProbability =
84 theFissionProbabilityPtr->EmissionProbability(*fragment,
85 MaximalKineticEnergy);
86 }
87 }
88 return FissionProbability;
89}

91G4Fragment* G4CompetitiveFission::EmittedFragment(G4Fragment* theNucleus)
92{
93 G4Fragment * Fragment1 = nullptr;
94 // Nucleus data
95 // Atomic number of nucleus
96 G4int A = theNucleus->GetA_asInt();
97 // Charge of nucleus
98 G4int Z = theNucleus->GetZ_asInt();
99 // Excitation energy (in MeV)
100 G4double U = theNucleus->GetExcitationEnergy();
101 G4double pcorr = pairingCorrection->GetFissionPairingCorrection(A,Z);
102 if (U <= pcorr) { return Fragment1; }
103
104 // Atomic Mass of Nucleus (in MeV)
105 G4double M = theNucleus->GetGroundStateMass();
106
107 // Nucleus Momentum
108 G4LorentzVector theNucleusMomentum = theNucleus->GetMomentum();
109
110 // Calculate fission parameters
111 theParam.DefineParameters(A, Z, U-pcorr, FissionBarrier);
112
113 // First fragment
114 G4int A1 = 0;
115 G4int Z1 = 0;
116 G4double M1 = 0.0;
117
118 // Second fragment
119 G4int A2 = 0;
120 G4int Z2 = 0;
121 G4double M2 = 0.0;
122
123 G4double FragmentsExcitationEnergy = 0.0;
124 G4double FragmentsKineticEnergy = 0.0;
125
126 G4int Trials = 0;
127 do {
128
129 // First fragment
130 A1 = FissionAtomicNumber(A);
131 Z1 = FissionCharge(A, Z, A1);
132 M1 = G4NucleiProperties::GetNuclearMass(A1, Z1);
133
134 // Second Fragment
135 A2 = A - A1;
136 Z2 = Z - Z1;
137 if (A2 < 1 || Z2 < 0 || Z2 > A2) {
138 FragmentsExcitationEnergy = -1.0;
139 continue;
140 }
141 M2 = G4NucleiProperties::GetNuclearMass(A2, Z2);
142 // Maximal Kinetic Energy (available energy for fragments)
143 G4double Tmax = M + U - M1 - M2 - pcorr;
144
145 // Check that fragment masses are less or equal than total energy
146 if (Tmax < 0.0) {
147 FragmentsExcitationEnergy = -1.0;
148 continue;
149 }
150
151 FragmentsKineticEnergy = FissionKineticEnergy( A , Z,
152 A1, Z1,
153 A2, Z2,
154 U , Tmax);
155
156 // Excitation Energy
157 // FragmentsExcitationEnergy = Tmax - FragmentsKineticEnergy;
158 // JMQ 04/03/09 BUG FIXED: in order to fulfill energy conservation the
159 // fragments carry the fission pairing energy in form of
160 // excitation energy
161
162 FragmentsExcitationEnergy =
163 Tmax - FragmentsKineticEnergy + pcorr;
164
165 // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
166 } while (FragmentsExcitationEnergy < 0.0 && ++Trials < 100);
167
168 if (FragmentsExcitationEnergy <= 0.0) {
169 throw G4HadronicException(__FILE__, __LINE__,
170 "G4CompetitiveFission::BreakItUp: Excitation energy for fragments < 0.0!");
171 }
172
173 // Fragment 1
174 M1 += FragmentsExcitationEnergy * A1/static_cast<G4double>(A);
175 // Fragment 2
176 M2 += FragmentsExcitationEnergy * A2/static_cast<G4double>(A);
177 // primary
178 M += U;
179
180 G4double etot1 = ((M - M2)*(M + M2) + M1*M1)/(2*M);
181 G4ParticleMomentum Momentum1 =
182 std::sqrt((etot1 - M1)*(etot1+M1))*G4RandomDirection();
183 G4LorentzVector FourMomentum1(Momentum1, etot1);
184 FourMomentum1.boost(theNucleusMomentum.boostVector());
185
186 // Create Fragments
187 Fragment1 = new G4Fragment( A1, Z1, FourMomentum1);
188 theNucleusMomentum -= FourMomentum1;
189 theNucleus->SetZandA_asInt(Z2, A2);
190 theNucleus->SetMomentum(theNucleusMomentum);
191 return Fragment1;
192}

194G4int
195G4CompetitiveFission::FissionAtomicNumber(G4int A)
196 // Calculates the atomic number of a fission product
197{
198
199 // For Simplicity reading code
200 G4int A1 = theParam.GetA1();
201 G4int A2 = theParam.GetA2();
202 G4double As = theParam.GetAs();
203 G4double Sigma2 = theParam.GetSigma2();
204 G4double SigmaS = theParam.GetSigmaS();
205 G4double w = theParam.GetW();
206
207 G4double C2A = A2 + 3.72*Sigma2;
208 G4double C2S = As + 3.72*SigmaS;
209
210 G4double C2 = 0.0;
211 if (w > 1000.0 ) { C2 = C2S; }
212 else if (w < 0.001) { C2 = C2A; }
213 else { C2 = std::max(C2A,C2S); }
214
215 G4double C1 = A-C2;
216 if (C1 < 30.0) {
217 C2 = A-30.0;
218 C1 = 30.0;
219 }
220
221 G4double Am1 = (As + A1)*0.5;
222 G4double Am2 = (A1 + A2)*0.5;
223
224 // Get Mass distributions as sum of symmetric and asymmetric Gasussians
225 G4double Mass1 = MassDistribution(As,A);
226 G4double Mass2 = MassDistribution(Am1,A);
227 G4double Mass3 = MassDistribution(G4double(A1),A);
228 G4double Mass4 = MassDistribution(Am2,A);
229 G4double Mass5 = MassDistribution(G4double(A2),A);
230 // get maximal value among Mass1,...,Mass5
231 G4double MassMax = Mass1;
232 if (Mass2 > MassMax) { MassMax = Mass2; }
233 if (Mass3 > MassMax) { MassMax = Mass3; }
234 if (Mass4 > MassMax) { MassMax = Mass4; }
235 if (Mass5 > MassMax) { MassMax = Mass5; }
236
237 // Sample a fragment mass number, which lies between C1 and C2
238 G4double xm;
239 G4double Pm;
240 do {
241 xm = C1+G4UniformRand()*(C2-C1);
242 Pm = MassDistribution(xm,A);
243 // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
244 } while (MassMax*G4UniformRand() > Pm);
245 G4int ires = G4lrint(xm);
246
247 return ires;
248}

250G4double
251G4CompetitiveFission::MassDistribution(G4double x, G4int A)
252 // This method gives mass distribution F(x) = F_{asym}(x)+w*F_{sym}(x)
253 // which consist of symmetric and asymmetric sum of gaussians components.
254{
255 G4double y0 = (x-theParam.GetAs())/theParam.GetSigmaS();
256 G4double Xsym = G4Exp(-0.5*y0*y0);
257
258 G4double y1 = (x - theParam.GetA1())/theParam.GetSigma1();
259 G4double y2 = (x - theParam.GetA2())/theParam.GetSigma2();
260 G4double z1 = (x - A + theParam.GetA1())/theParam.GetSigma1();
261 G4double z2 = (x - A + theParam.GetA2())/theParam.GetSigma2();
262 G4double Xasym = G4Exp(-0.5*y1*y1) + G4Exp(-0.5*y2*y2)
263 + 0.5*( G4Exp(-0.5*z1*z1) + G4Exp(-0.5*z2*z2));
264
265 G4double res;
266 G4double w = theParam.GetW();
267 if (w > 1000) { res = Xsym; }
268 else if (w < 0.001) { res = Xasym; }
269 else { res = w*Xsym+Xasym; }
270 return res;
271}

273G4int G4CompetitiveFission::FissionCharge(G4int A, G4int Z, G4double Af)
274 // Calculates the charge of a fission product for a given atomic number Af
275{
276 static const G4double sigma = 0.6;
277 G4double DeltaZ = 0.0;
278 if (Af >= 134.0) { DeltaZ = -0.45; }
279 else if (Af <= (A-134.0)) { DeltaZ = 0.45; }
280 else { DeltaZ = -0.45*(Af-A*0.5)/(134.0-A*0.5); }
281
282 G4double Zmean = (Af/A)*Z + DeltaZ;
283
284 G4double theZ;
285 do {
286 theZ = G4RandGauss::shoot(Zmean,sigma);
287 // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
288 } while (theZ < 1.0 || theZ > (Z-1.0) || theZ > Af);
289
290 return G4lrint(theZ);
291}

293G4double
294G4CompetitiveFission::FissionKineticEnergy(G4int A, G4int Z,
295 G4int Af1, G4int /*Zf1*/,
296 G4int Af2, G4int /*Zf2*/,
297 G4double /*U*/, G4double Tmax)
298 // Gives the kinetic energy of fission products
299{
300 // Find maximal value of A for fragments
301 G4int AfMax = std::max(Af1,Af2);
302
303 // Weights for symmetric and asymmetric components
304 G4double Pas = 0.0;
305 if (theParam.GetW() <= 1000) {
306 G4double x1 = (AfMax-theParam.GetA1())/theParam.GetSigma1();
307 G4double x2 = (AfMax-theParam.GetA2())/theParam.GetSigma2();
308 Pas = 0.5*G4Exp(-0.5*x1*x1) + G4Exp(-0.5*x2*x2);
309 }
310
311 G4double Ps = 0.0;
312 if (theParam.GetW() >= 0.001) {
313 G4double xs = (AfMax-theParam.GetAs())/theParam.GetSigmaS();
314 Ps = theParam.GetW()*G4Exp(-0.5*xs*xs);
315 }
316 G4double Psy = Ps/(Pas+Ps);
317
318 // Fission fractions Xsy and Xas formed in symmetric and asymmetric modes
319 G4double PPas = theParam.GetSigma1() + 2.0 * theParam.GetSigma2();
320 G4double PPsy = theParam.GetW() * theParam.GetSigmaS();
321 G4double Xas = PPas / (PPas+PPsy);
322 G4double Xsy = PPsy / (PPas+PPsy);
323
324 // Average kinetic energy for symmetric and asymmetric components
325 G4double Eaverage = (0.1071*(Z*Z)/G4Pow::GetInstance()->Z13(A) + 22.2)*CLHEP::MeV;
326
327 // Compute maximal average kinetic energy of fragments and Energy Dispersion
328 G4double TaverageAfMax;
329 G4double ESigma = 10*CLHEP::MeV;
330 // Select randomly fission mode (symmetric or asymmetric)
331 if (G4UniformRand() > Psy) { // Asymmetric Mode
332 G4double A11 = theParam.GetA1()-0.7979*theParam.GetSigma1();
333 G4double A12 = theParam.GetA1()+0.7979*theParam.GetSigma1();
334 G4double A21 = theParam.GetA2()-0.7979*theParam.GetSigma2();
335 G4double A22 = theParam.GetA2()+0.7979*theParam.GetSigma2();
336 // scale factor
337 G4double ScaleFactor = 0.5*theParam.GetSigma1()*
338 (AsymmetricRatio(A,A11)+AsymmetricRatio(A,A12))+
339 theParam.GetSigma2()*(AsymmetricRatio(A,A21)+AsymmetricRatio(A,A22));
340 // Compute average kinetic energy for fragment with AfMax
341 TaverageAfMax = (Eaverage + 12.5 * Xsy) * (PPas/ScaleFactor) *
342 AsymmetricRatio(A,G4double(AfMax));
343
344 } else { // Symmetric Mode
345 G4double As0 = theParam.GetAs() + 0.7979*theParam.GetSigmaS();
346 // Compute average kinetic energy for fragment with AfMax
347 TaverageAfMax = (Eaverage - 12.5*CLHEP::MeV*Xas)
348 *SymmetricRatio(A, G4double(AfMax))/SymmetricRatio(A, As0);
349 ESigma = 8.0*CLHEP::MeV;
350 }
351
352 // Select randomly, in accordance with Gaussian distribution,
353 // fragment kinetic energy
354 G4double KineticEnergy;
355 G4int i = 0;
356 do {
357 KineticEnergy = G4RandGauss::shoot(TaverageAfMax, ESigma);
358 if (++i > 100) return Eaverage;
359 // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
360 } while (KineticEnergy < Eaverage-3.72*ESigma ||
361 KineticEnergy > Eaverage+3.72*ESigma ||
362 KineticEnergy > Tmax);
363
364 return KineticEnergy;
365}
