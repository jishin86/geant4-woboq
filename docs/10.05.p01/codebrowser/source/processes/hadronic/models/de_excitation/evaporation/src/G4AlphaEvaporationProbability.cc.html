46G4double G4AlphaEvaporationProbability::CalcAlphaParam(const G4Fragment& fragment)
47{
48 // Data comes from
49 // Dostrovsky, Fraenkel and Friedlander
50 // Physical Review, vol 116, num. 3 1959
51 //
52 // const G4int size = 5;
53 // G4double Zlist[5] = { 10.0, 20.0, 30.0, 50.0, 70.0};
54 // G4double Calpha[5] = { 0.10, 0.10, 0.10, 0.08, 0.06};
55
56 G4int aZ = fragment.GetZ_asInt()-GetZ();
57 G4double C;
58
59 if (aZ <= 30)
60 {
61 C = 0.10;
62 }
63 else if (aZ <= 50)
64 {
65 C = 0.1 - (aZ-30)*0.001;
66 }
67 else if (aZ < 70)
68 {
69 C = 0.08 - (aZ-50)*0.001;
70 }
71 else
72 {
73 C = 0.06;
74 }
75 return 1.0 + C;
76}
