58const G4int MAXDEPOINT = 10;
59const G4int MAXGRDATA = 300;
60
61class G4GammaTransition;
62class G4NuclearPolarizationStore;
63
64class G4PhotonEvaporation : public G4VEvaporationChannel {
65
66public:
67
68 explicit G4PhotonEvaporation(G4GammaTransition* ptr=nullptr);
69
70 virtual ~G4PhotonEvaporation();
71
72 virtual void Initialise() final;
73
74 // one photon or e- emission
75 virtual G4Fragment* EmittedFragment(G4Fragment* theNucleus) final;
76
77 // returns "false", emitted gamma and e- are added to the results
78 virtual G4bool
79 BreakUpChain(G4FragmentVector* theResult, G4Fragment* theNucleus) final;
80
81 // emitted gamma, e-, and residual fragment are added to the results
82 G4FragmentVector* BreakItUp(const G4Fragment& theNucleus);
83
84 // compute emission probability for both continum and discrete cases
85 // must be called before any method above
86 virtual G4double GetEmissionProbability(G4Fragment* theNucleus) final;
87
88 virtual G4double GetFinalLevelEnergy(G4int Z, G4int A, G4double energy) final;
89
90 virtual G4double GetUpperLevelEnergy(G4int Z, G4int A) final;
91
92 void SetGammaTransition(G4GammaTransition*);
93
94 virtual void SetICM(G4bool);
95
96 virtual void RDMForced (G4bool);
97
98 inline void SetVerboseLevel(G4int verbose);
99
100 inline G4int GetVacantShellNumber() const;
101
102private:
103
104 void InitialiseGRData();
105
106 G4Fragment* GenerateGamma(G4Fragment* nucleus);
107
108 inline void InitialiseLevelManager(G4int Z, G4int A);
109
110 G4PhotonEvaporation(const G4PhotonEvaporation & right) = delete;
111 const G4PhotonEvaporation & operator = (const G4PhotonEvaporation & right) = delete;
112
113 G4NuclearLevelData* fNuclearLevelData;
114 const G4LevelManager* fLevelManager;
115 G4GammaTransition* fTransition;
116 G4NuclearPolarizationStore* fNucPStore;
117
118 // fPolarization stores polarization tensor for consecutive
119 // decays of a nucleus
120 G4NuclearPolarization* fPolarization;
121
122 G4int fVerbose;
123 G4int theZ;
124 G4int theA;
125 G4int fPoints;
126 G4int fCode;
127 G4int vShellNumber;
128 size_t fIndex;
129
130 static G4float GREnergy[MAXGRDATA];
131 static G4float GRWidth[MAXGRDATA];
132
133 G4double fCummProbability[MAXDEPOINT];
134
135 G4double fLevelEnergyMax;
136 G4double fExcEnergy;
137 G4double fProbability;
138 G4double fStep;
139 G4double fMaxLifeTime;
140
141 G4double Tolerance;
142
143 G4bool fICM;
144 G4bool fRDM;
145 G4bool fSampleTime;
146 G4bool fCorrelatedGamma;
147 G4bool isInitialised;
148
149#ifdef G4MULTITHREADED
150 static G4Mutex PhotonEvaporationMutex;
151#endif
152};
153
154inline void G4PhotonEvaporation::SetVerboseLevel(G4int verbose)
155{
156 fVerbose = verbose;
157}
158
159inline void
160G4PhotonEvaporation::InitialiseLevelManager(G4int Z, G4int A)
161{
162 if(Z != theZ || A != theA) {
163 theZ = Z;
164 theA = A;
165 fIndex = 0;
166 fLevelManager = fNuclearLevelData->GetLevelManager(theZ, theA);
167 fLevelEnergyMax = fLevelManager ? fLevelManager->MaxLevelEnergy() : 0.0;
168 }
169}
170
171inline G4int G4PhotonEvaporation::GetVacantShellNumber() const
172{
173 return vShellNumber;
174}
