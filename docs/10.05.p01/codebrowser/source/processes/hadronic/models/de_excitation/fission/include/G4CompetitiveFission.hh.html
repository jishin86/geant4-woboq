118inline void G4CompetitiveFission::SetFissionBarrier(G4VFissionBarrier * aBarrier)
119{

151inline
152G4double G4CompetitiveFission::Ratio(G4double A, G4double A11,
153 G4double B1, G4double A00)
154{
155 G4double res;
156 if (A11 >= A*0.5 && A11 <= (A00+10.0)) {
157 G4double x = (A11-A00)/A;
158 res = 1.0 - B1*x*x;
159 } else {
160 G4double x = 10.0/A;
161 res = 1.0 - B1*x*x - 2.0*x*B1*(A11-A00-10.0)/A;
162 }
163 return res;
164}
165
166inline
167G4double G4CompetitiveFission::AsymmetricRatio(G4int A, G4double A11)
168{
169 return Ratio(G4double(A),A11,23.5,134.0);
170}
171
172inline
173G4double G4CompetitiveFission::SymmetricRatio(G4int A, G4double A11)
174{
175 G4double A0 = G4double(A);
176 return Ratio(A0,A11,5.32,A0*0.5);
177}
