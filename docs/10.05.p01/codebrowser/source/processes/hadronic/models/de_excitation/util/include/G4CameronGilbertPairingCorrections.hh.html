34#ifndef G4CameronGilbertPairingCorrections_h
35#define G4CameronGilbertPairingCorrections_h 1
36
37#include "globals.hh"
38
39class G4CameronGilbertPairingCorrections
40{
41public:
42
43 explicit G4CameronGilbertPairingCorrections();
44
45 inline G4bool GetPairingCorrection(G4int N, G4int Z, G4double& result) const
46 {
47 G4bool res = false;
48 if(Z >= ZTableMin && Z <= ZTableMax && N >= NTableMin && N <= NTableMax) {
49 result = PairingZTable[Z-ZTableMin] + PairingNTable[N-NTableMin];
50 res = true;
51 }
52 return res;
53 }
54
55 enum { ZTableSize = 88, NTableSize = 140, ZTableMin = 11, ZTableMax = 98,
56 NTableMin = 11, NTableMax = 150 };
57
58private:
59
60 G4CameronGilbertPairingCorrections(const G4CameronGilbertPairingCorrections & right) = delete;
61 const G4CameronGilbertPairingCorrections & operator=(const G4CameronGilbertPairingCorrections & right) = delete;
62
63 static G4double PairingZTable[ZTableSize];
64 static G4double PairingNTable[NTableSize];
65
66};
67#endif
