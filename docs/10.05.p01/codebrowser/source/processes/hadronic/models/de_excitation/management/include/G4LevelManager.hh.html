45#ifndef G4LEVELMANAGER_HH
46#define G4LEVELMANAGER_HH 1
47
48#include "globals.hh"
49#include "G4NucLevel.hh"
50#include <vector>
51#include <iostream>
52
53class G4LevelManager
54{
55
56public:
57 // levels - vector of nuclear level objects, ground state
58 // level has NULL pointer
59 // energies - list of excitation energies of nuclear levels starting
60 // from the ground state with energy zero
61 // spin - 2J, where J is the full angular momentum of the state
62 explicit G4LevelManager(G4int Z, G4int A, size_t nlev,
63 const std::vector<G4double>& energies,
64 const std::vector<G4int>& spin,
65 const std::vector<const G4NucLevel*>& levels);
66
67 ~G4LevelManager();
68
69 //===================================================================
70 // run time inlined const functions
71 //===================================================================
72
73 inline size_t NumberOfTransitions() const;
74
75 inline const G4NucLevel* GetLevel(size_t i) const;
76
77 inline G4double LevelEnergy(size_t i) const;
78
79 inline G4double MaxLevelEnergy() const;
80
81 size_t NearestLevelIndex(G4double energy, size_t index=0) const;
82
83 inline size_t NearestLowEdgeLevelIndex(G4double energy) const;
84
85 inline const G4NucLevel* NearestLevel(G4double energy, size_t index=0) const;
86
87 inline G4double NearestLevelEnergy(G4double energy, size_t index=0) const;
88
89 inline G4double NearestLowEdgeLevelEnergy(G4double energy) const;
90
91 // for stable isotopes life time is -1
92 inline G4double LifeTime(size_t i) const;
93
94 inline G4int SpinTwo(size_t i) const;
95
96 inline G4int Parity(size_t i) const;
97
98 inline G4int FloatingLevel(size_t i) const;
99
100 inline G4double PairingCorrection() const;
101
102 inline G4double ShellCorrection() const;
103
104 inline G4double LevelDensity(G4double U) const;
105
106 const G4String& FloatingType(size_t i) const;
107
108 void StreamInfo(std::ostream& os) const;
109
110private:
111
112#ifdef G4VERBOSE
113 void PrintError(size_t idx, const G4String&) const;
114#endif
115
116 G4LevelManager(const G4LevelManager & right) = delete;
117 const G4LevelManager& operator=(const G4LevelManager &right) = delete;
118 G4bool operator==(const G4LevelManager &right) const = delete;
119 G4bool operator!=(const G4LevelManager &right) const = delete;
120
121 std::vector<G4double> fLevelEnergy;
122 std::vector<G4int> fSpin;
123 std::vector<const G4NucLevel*> fLevels;
124
125 G4double fPairingCorrection;
126 G4double fShellCorrection;
127 G4double fLevelDensity;
128
129 size_t nTransitions;
130
131 static const G4int nfloting = 13;
132 static G4String fFloatingLevels[nfloting];
133
134};
135
136inline size_t G4LevelManager::NumberOfTransitions() const
137{
138 return nTransitions;
139}
140
141inline const G4NucLevel* G4LevelManager::GetLevel(size_t i) const
142{
143#ifdef G4VERBOSE
144 if(i > nTransitions) { PrintError(i, "GetLevel(idx)"); }
145#endif
146 return fLevels[i];
147}
148
149inline G4double G4LevelManager::LevelEnergy(size_t i) const
150{
151#ifdef G4VERBOSE
152 if(i > nTransitions) { PrintError(i, "LevelEnergy(idx)"); }
153#endif
154 return fLevelEnergy[i];
155}
156
157inline G4double G4LevelManager::MaxLevelEnergy() const
158{
159 return fLevelEnergy[nTransitions];
160}
161
162inline size_t G4LevelManager::NearestLowEdgeLevelIndex(G4double energy) const
163{
164 size_t idx = nTransitions;
165 if(energy < fLevelEnergy[nTransitions]) {
166 idx = std::lower_bound(fLevelEnergy.begin(), fLevelEnergy.end(), energy)
167 - fLevelEnergy.begin() - 1;
168 }
169 return idx;
170}
171
172inline const G4NucLevel*
173G4LevelManager::NearestLevel(G4double energy, size_t index) const
174{
175 return GetLevel(NearestLevelIndex(energy, index));
176}
177
178inline G4double
179G4LevelManager::NearestLevelEnergy(G4double energy, size_t index) const
180{
181 return LevelEnergy(NearestLevelIndex(energy, index));
182}
183
184inline G4double G4LevelManager::NearestLowEdgeLevelEnergy(G4double energy) const
185{
186 return LevelEnergy(NearestLowEdgeLevelIndex(energy));
187}
188
189inline G4double G4LevelManager::LifeTime(size_t i) const
190{
191#ifdef G4VERBOSE
192 if(i > nTransitions) { PrintError(i, "LifeTime"); }
193#endif
194 return (fLevels[i]) ? fLevels[i]->GetTimeGamma() : 0.0;
195}
196
197inline G4int G4LevelManager::SpinTwo(size_t i) const
198{
199#ifdef G4VERBOSE
200 if(i > nTransitions) { PrintError(i, "SpinTwo"); }
201#endif
202 return std::abs(fSpin[i]%100000 - 100);
203}
204
205inline G4int G4LevelManager::Parity(size_t i) const
206{
207#ifdef G4VERBOSE
208 if(i > nTransitions) { PrintError(i, "SpinTwo"); }
209#endif
210 return (fSpin[i]%100000 - 100 > 0) ? 1 : -1;
211}
212
213inline G4int G4LevelManager::FloatingLevel(size_t i) const
214{
215#ifdef G4VERBOSE
216 if(i > nTransitions) { PrintError(i, "Floating"); }
217#endif
218 return fSpin[i]/100000;
219}
220
221inline G4double G4LevelManager::PairingCorrection() const
222{
223 return fPairingCorrection;
224}
225
226inline G4double G4LevelManager::ShellCorrection() const
227{
228 return fShellCorrection;
229}
230
231inline G4double G4LevelManager::LevelDensity(G4double) const
232{
233 return fLevelDensity;
234}
235
236#endif
