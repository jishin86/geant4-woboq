43#ifndef G4VEvaporationChannel_h
44#define G4VEvaporationChannel_h 1
45
46#include "globals.hh"
47#include "G4Fragment.hh"
48
49class G4VEvaporationChannel
50{
51public:
52
53 explicit G4VEvaporationChannel(const G4String & aName = "");
54 virtual ~G4VEvaporationChannel();
55
56 virtual G4double GetEmissionProbability(G4Fragment* theNucleus) = 0;
57
58 // option definition
59 virtual void Initialise();
60
61 // return level life time, by default zero
62 virtual G4double GetLifeTime(G4Fragment* theNucleus);
63
64 // return emitted fragment, initial fragment is modified
65 // and not deleted
66 virtual G4Fragment* EmittedFragment(G4Fragment* theNucleus);
67
68 // returns "true" if primary fragment is decayed and deleted
69 // returns "false" if primary fragment is modified but stay alive
70 // emitted fragments are added to the vector of results
71 virtual G4bool
72 BreakUpChain(G4FragmentVector* theResult, G4Fragment* theNucleus);
73
74 // return vector of emitted fragments, initial fragment is modified
75 // but not included in this vector
76 inline G4FragmentVector* BreakUpFragment(G4Fragment* theNucleus);
77
78 virtual void Dump() const;
79
80 // enable internal conversion
81 virtual void SetICM(G4bool);
82
83 // flag of the radioactive decay module
84 virtual void RDMForced(G4bool);
85
86 // get energy of final level if this channel is photon evaporation
87 virtual G4double GetFinalLevelEnergy(G4int Z, G4int A, G4double energy);
88
89 // get energy of the upper level if this channel is photon evaporation
90 virtual G4double GetUpperLevelEnergy(G4int Z, G4int A);
91
92 // get energy of the upper level in level DB for any decay channel
93 inline G4double GetMaxLevelEnergy(G4int Z, G4int A);
94
95 // get energy of nearest level for any decay channel
96 inline G4double GetNearestLevelEnergy(G4int Z, G4int A, G4double energy);
97
98 // set pointer to photon evaporation in order to access level data
99 inline void SetPhotonEvaporation(G4VEvaporationChannel* p);
100
101 // for cross section selection
102 inline void SetOPTxs(G4int opt);
103 // for superimposed Coulomb Barrier for inverse cross sections
104 inline void UseSICB(G4bool use);
105
106protected:
107
108 G4int OPTxs;
109 G4bool useSICB;
110
111private:
112
113 G4VEvaporationChannel(const G4VEvaporationChannel & right) = delete;
114 const G4VEvaporationChannel & operator=
115 (const G4VEvaporationChannel & right) = delete;
116 G4bool operator==(const G4VEvaporationChannel & right) const = delete;
117 G4bool operator!=(const G4VEvaporationChannel & right) const = delete;
118
119 G4VEvaporationChannel* photonEvaporation;
120};
121
122inline G4FragmentVector*
123G4VEvaporationChannel::BreakUpFragment(G4Fragment* theNucleus)
124{
125 G4FragmentVector* results = new G4FragmentVector();
126 BreakUpChain(results, theNucleus);
127 return results;
128}
129
130
131inline G4double
132G4VEvaporationChannel::GetNearestLevelEnergy(G4int Z, G4int A, G4double energy)
133{
134 return (photonEvaporation) ?
135 photonEvaporation->GetFinalLevelEnergy(Z, A, energy) : energy;
136}
137
138inline G4double
139G4VEvaporationChannel::GetMaxLevelEnergy(G4int Z, G4int A)
140{
141 return (photonEvaporation) ?
142 photonEvaporation->GetUpperLevelEnergy(Z, A) : 0.0;
143}
144
145inline void
146G4VEvaporationChannel::SetPhotonEvaporation(G4VEvaporationChannel* p)
147{
148 photonEvaporation = p;
149}
150
151inline void G4VEvaporationChannel::SetOPTxs(G4int)
152{}
153
154inline void G4VEvaporationChannel::UseSICB(G4bool)
155{}
156
157#endif
