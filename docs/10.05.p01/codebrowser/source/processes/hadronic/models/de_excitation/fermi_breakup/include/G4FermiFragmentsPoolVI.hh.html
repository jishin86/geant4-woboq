31#ifndef G4FermiFragmentsPoolVI_hh
32#define G4FermiFragmentsPoolVI_hh 1
33
34#include "globals.hh"
35#include "G4FermiFragment.hh"
36#include "G4FermiPair.hh"
37#include "G4FermiChannels.hh"
38#include "G4FermiDecayProbability.hh"
39
40#include <vector>
41
42static const G4int maxZ = 9;
43static const G4int maxA = 17;
44
45class G4FermiFragmentsPoolVI
46{
47public:
48
49 explicit G4FermiFragmentsPoolVI();
50
51 ~G4FermiFragmentsPoolVI();
52
53 const G4FermiChannels* ClosestChannels(G4int Z, G4int A, G4double mass) const;
54
55 void DumpFragment(const G4FermiFragment*) const;
56
57 void Dump() const;
58
59 G4bool IsApplicable(G4int Z, G4int A, G4double etot) const;
60
61 G4bool IsPhysical(G4int Z, G4int A) const;
62
63 inline const G4FermiDecayProbability* FermiDecayProbability() const;
64
65 inline G4int GetMaxZ() const;
66
67 inline G4int GetMaxA() const;
68
69 inline G4double GetEnergyLimit() const;
70
71 inline G4double GetTolerance() const;
72
73private:
74
75 void Initialise();
76
77 G4bool IsInThePool(G4int Z, G4int A, G4double exc) const;
78
79 G4bool IsInPhysPairs(const G4FermiFragment* f1,
80 const G4FermiFragment* f2) const;
81
82 G4bool IsInUnphysPairs(const G4FermiFragment* f1,
83 const G4FermiFragment* f2) const;
84
85 G4double tolerance;
86 G4double elim;
87
88 G4float timelim;
89 G4float elimf;
90
91 G4FermiDecayProbability theDecay;
92
93 // pool
94 std::vector<const G4FermiFragment*> fragment_pool;
95 std::vector<const G4FermiFragment*> funstable;
96
97 // lists of configurations sorted by A
98
99 // "stable" fragments
100 std::vector<const G4FermiFragment*> list_f[maxA];
101 // list of channels for "stable" fragments
102 std::vector<G4FermiChannels*> list_c[maxA];
103 // pairs of stable fragments
104 std::vector<const G4FermiPair*> list_p[maxA];
105
106 // "unstable" fragments
107 std::vector<const G4FermiFragment*> list_g[maxA];
108 // list of channels of stable and unstable fragments
109 std::vector<G4FermiChannels*> list_d[maxA];
110 // pairs of stable and unstable fragments
111 std::vector<const G4FermiPair*> list_u[maxA];
112
113};
114
115inline G4int G4FermiFragmentsPoolVI::GetMaxZ() const
116{
117 return maxZ;
118}
119
120inline G4int G4FermiFragmentsPoolVI::GetMaxA() const
121{
122 return maxA;
123}
124
125inline const G4FermiDecayProbability*
126G4FermiFragmentsPoolVI::FermiDecayProbability() const
127{
128 return &theDecay;
129}
130
131inline G4double G4FermiFragmentsPoolVI::GetEnergyLimit() const
132{
133 return elim;
134}
135
136inline G4double G4FermiFragmentsPoolVI::GetTolerance() const
137{
138 return tolerance;
139}
140
141#endif
