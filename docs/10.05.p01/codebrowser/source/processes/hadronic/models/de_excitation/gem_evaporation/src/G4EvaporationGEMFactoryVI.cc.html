43std::vector<G4VEvaporationChannel*>* G4EvaporationGEMFactoryVI::GetChannel()
44{
45 std::vector<G4VEvaporationChannel*> * theChannel =
46 new std::vector<G4VEvaporationChannel*>;
47 theChannel->reserve(81);
48
49 theChannel->push_back( thePhotonEvaporation ); // Photon Channel
50 theChannel->push_back( new G4CompetitiveFission() ); // Fission Channel
51
52 theChannel->push_back( new G4GEMChannelVI( 1, 0) );// n
53 theChannel->push_back( new G4GEMChannelVI( 1, 1) );// p
54 theChannel->push_back( new G4GEMChannelVI( 2, 1) );// Deuteron
55 theChannel->push_back( new G4GEMChannelVI( 3, 1) );// Triton
56 theChannel->push_back( new G4GEMChannelVI( 3, 2) );// He3
57 theChannel->push_back( new G4GEMChannelVI( 4, 2) );// Alpha
58 theChannel->push_back( new G4GEMChannelVI( 5, 2) );// He5
59 theChannel->push_back( new G4GEMChannelVI( 5, 3) );// Li5
60 theChannel->push_back( new G4GEMChannelVI( 6, 3) );// Li6
61 theChannel->push_back( new G4GEMChannelVI( 7, 3) );// Li7
62 theChannel->push_back( new G4GEMChannelVI( 8, 3) );// Li8
63 theChannel->push_back( new G4GEMChannelVI( 9, 3) );// Li9
64 theChannel->push_back( new G4GEMChannelVI( 7, 4) );// Be7
65 theChannel->push_back( new G4GEMChannelVI( 8, 4) );// Be8
66 theChannel->push_back( new G4GEMChannelVI( 9, 4) );// Be9
67 theChannel->push_back( new G4GEMChannelVI(10, 4) );// Be10
68 theChannel->push_back( new G4GEMChannelVI(11, 4) );// Be11
69 theChannel->push_back( new G4GEMChannelVI( 8, 5) );// B8
70 theChannel->push_back( new G4GEMChannelVI( 9, 5) );// B9
71 theChannel->push_back( new G4GEMChannelVI(10, 5) );// B10
72 theChannel->push_back( new G4GEMChannelVI(11, 5) );// B11
73 theChannel->push_back( new G4GEMChannelVI(12, 5) );// B12
74 theChannel->push_back( new G4GEMChannelVI(13, 5) );// B13
75 theChannel->push_back( new G4GEMChannelVI(10, 6) );// C10
76 theChannel->push_back( new G4GEMChannelVI(11, 6) );// C11
77 theChannel->push_back( new G4GEMChannelVI(12, 6) );// C12
78 theChannel->push_back( new G4GEMChannelVI(13, 6) );// C13
79 theChannel->push_back( new G4GEMChannelVI(14, 6) );// C14
80 theChannel->push_back( new G4GEMChannelVI(15, 6) );// C15
81 theChannel->push_back( new G4GEMChannelVI(16, 6) );// C16
82 theChannel->push_back( new G4GEMChannelVI(13, 7) );// N13
83 theChannel->push_back( new G4GEMChannelVI(14, 7) );// N14
84 theChannel->push_back( new G4GEMChannelVI(15, 7) );// N15
85 theChannel->push_back( new G4GEMChannelVI(16, 7) );// N16
86 theChannel->push_back( new G4GEMChannelVI(17, 7) );// N17
87 theChannel->push_back( new G4GEMChannelVI(18, 7) );// N17
88 theChannel->push_back( new G4GEMChannelVI(15, 8) );// O15
89 theChannel->push_back( new G4GEMChannelVI(16, 8) );// O16
90 theChannel->push_back( new G4GEMChannelVI(17, 8) );// O17
91 theChannel->push_back( new G4GEMChannelVI(18, 8) );// O18
92 theChannel->push_back( new G4GEMChannelVI(19, 8) );// O19
93 theChannel->push_back( new G4GEMChannelVI(20, 8) );// O20
94 theChannel->push_back( new G4GEMChannelVI(21, 8) );// O21
95 theChannel->push_back( new G4GEMChannelVI(22, 8) );// O22
96 theChannel->push_back( new G4GEMChannelVI(17, 9) );// F17
97 theChannel->push_back( new G4GEMChannelVI(18, 9) );// F18
98 theChannel->push_back( new G4GEMChannelVI(19, 9) );// F19
99 theChannel->push_back( new G4GEMChannelVI(20, 9) );// F20
100 theChannel->push_back( new G4GEMChannelVI(21, 9) );// F21
101 theChannel->push_back( new G4GEMChannelVI(22, 9) );// F22
102 theChannel->push_back( new G4GEMChannelVI(23, 9) );// F23
103 theChannel->push_back( new G4GEMChannelVI(24, 9) );// F24
104 theChannel->push_back( new G4GEMChannelVI(25, 9) );// F25
105 theChannel->push_back( new G4GEMChannelVI(26, 9) );// F26
106 theChannel->push_back( new G4GEMChannelVI(27, 9) );// F27
107 theChannel->push_back( new G4GEMChannelVI(18,10) );// Ne18
108 theChannel->push_back( new G4GEMChannelVI(19,10) );// Ne19
109 theChannel->push_back( new G4GEMChannelVI(20,10) );// Ne20
110 theChannel->push_back( new G4GEMChannelVI(21,10) );// Ne21
111 theChannel->push_back( new G4GEMChannelVI(22,10) );// Ne22
112 theChannel->push_back( new G4GEMChannelVI(23,10) );// Ne23
113 theChannel->push_back( new G4GEMChannelVI(24,10) );// Ne24
114 theChannel->push_back( new G4GEMChannelVI(25,10) );// Ne25
115 theChannel->push_back( new G4GEMChannelVI(26,10) );// Ne26
116 theChannel->push_back( new G4GEMChannelVI(27,10) );// Ne27
117 theChannel->push_back( new G4GEMChannelVI(28,10) );// Ne28
118 theChannel->push_back( new G4GEMChannelVI(21,11) );// Na21
119 theChannel->push_back( new G4GEMChannelVI(22,11) );// Na22
120 theChannel->push_back( new G4GEMChannelVI(23,11) );// Na23
121 theChannel->push_back( new G4GEMChannelVI(24,11) );// Na24
122 theChannel->push_back( new G4GEMChannelVI(25,11) );// Na25
123 theChannel->push_back( new G4GEMChannelVI(26,11) );// Na26
124 theChannel->push_back( new G4GEMChannelVI(27,11) );// Na27
125 theChannel->push_back( new G4GEMChannelVI(28,11) );// Na28
126 theChannel->push_back( new G4GEMChannelVI(22,12) );// Mg22
127 theChannel->push_back( new G4GEMChannelVI(23,12) );// Mg23
128 theChannel->push_back( new G4GEMChannelVI(24,12) );// Mg24
129 theChannel->push_back( new G4GEMChannelVI(25,12) );// Mg25
130 theChannel->push_back( new G4GEMChannelVI(26,12) );// Mg26
131 theChannel->push_back( new G4GEMChannelVI(27,12) );// Mg27
132 theChannel->push_back( new G4GEMChannelVI(28,12) );// Mg28
133
134 size_t nn = theChannel->size();
135 for(size_t i=2; i<nn; ++i) {
136 (*theChannel)[i]->SetPhotonEvaporation(thePhotonEvaporation);
137 }
138
139 return theChannel;
140}
