39G4FermiFragmentsPoolVI::G4FermiFragmentsPoolVI()
40{
41 // G4cout << "### G4FermiFragmentsPoolVI is constructed" << G4endl;
42 G4DeexPrecoParameters* param =
43 G4NuclearLevelData::GetInstance()->GetParameters();
44 tolerance = (G4float)param->GetMinExcitation();
45 timelim = (G4float)param->GetMaxLifeTime();
46
47 elim = param->GetFBUEnergyLimit();
48 elimf= (G4float)elim;
49
50 fragment_pool.reserve(991);
51 funstable.reserve(80);
52
53 Initialise();
54}

56G4FermiFragmentsPoolVI::~G4FermiFragmentsPoolVI()
57{
58 size_t nn;
59 for(G4int i=0; i<maxA; ++i) {
60 nn = list_p[i].size();
61 for(size_t j=0; j<nn; ++j) { delete (list_p[i])[j]; }
62 nn = list_c[i].size();
63 for(size_t j=0; j<nn; ++j) { delete (list_c[i])[j]; }
64 nn = list_d[i].size();
65 for(size_t j=0; j<nn; ++j) { delete (list_d[i])[j]; }
66 nn = list_u[i].size();
67 for(size_t j=0; j<nn; ++j) { delete (list_u[i])[j]; }
68 }
69 nn = fragment_pool.size();
70 for(size_t j=0; j<nn; ++j) { delete fragment_pool[j]; }
71 nn = funstable.size();
72 for(size_t j=0; j<nn; ++j) { delete funstable[j]; }
73}

75G4bool
76G4FermiFragmentsPoolVI::IsApplicable(G4int Z, G4int A, G4double etot) const
77{
78 G4bool isInList = false;
79 size_t nn = list_f[A].size();
80 for(size_t i=0; i<nn; ++i) {
81 if(Z == (list_f[A])[i]->GetZ()) {
82 isInList = true;
83 if(etot <= (list_f[A])[i]->GetFragmentMass() + elim) { return true; }
84 }
85 }
86 if(isInList) { return false; }
87 nn = list_g[A].size();
88 for(size_t i=0; i<nn; ++i) {
89 if(Z == (list_g[A])[i]->GetZ() &&
90 etot <= (list_g[A])[i]->GetFragmentMass() + elim) { return true; }
91 }
92 return false;
93}

95const G4FermiChannels*
96G4FermiFragmentsPoolVI::ClosestChannels(G4int Z, G4int A, G4double e) const
97{
98 const G4FermiChannels* res = nullptr;
99 G4double demax = e;
100
101 // stable channels;
102 size_t nn = list_c[A].size();
103 for(size_t j=0; j<nn; ++j) {
104 const G4FermiFragment* frag = (list_f[A])[j];
105 if(frag->GetZ() != Z) { continue; }
106 G4double de = e - frag->GetTotalEnergy();
107 //G4cout << " Stab check " << j << " channel de= " << de << " tol= " << tolerance << G4endl;
108 if(std::abs(de) <= tolerance) {
109 res = (list_c[A])[j];
110 //G4cout << " Stab chan: " << j << " N= " << res->GetNumberOfChannels() << G4endl;
111 break;
112 } else if(de + tolerance > 0.0) {
113 if(de < demax) {
114 res = (list_c[A])[j];
115 demax = de;
116 //G4cout << " Stab chan: " << j << " N= " << res->GetNumberOfChannels() << G4endl;
117 }
118 }
119 }
120 // unstable channels
121 if(!res) {
122 nn = list_d[A].size();
123 for(size_t j=0; j<nn; ++j) {
124 const G4FermiFragment* frag = (list_g[A])[j];
125 if(frag->GetZ() != Z) { continue; }
126 G4double de = e - frag->GetTotalEnergy();
127 //G4cout << " Unst check " << j << " channel de= " << de << " tol= " << tolerance << G4endl;
128 if(std::abs(de) <= tolerance || de > 0.0) {
129 res = (list_d[A])[j];
130 //G4cout << " Unst chan No: " << j << " N= " << res->GetNumberOfChannels() << G4endl;
131 break;
132 }
133 }
134 }
135 return res;
136}

138G4bool G4FermiFragmentsPoolVI::IsPhysical(G4int Z, G4int A) const
139{
140 G4bool res = false;
141 G4int nn = list_f[A].size();
142 for(G4int i=0; i<nn; ++i) {
143 if((list_f[A])[i]->GetZ() == Z) {
144 res = true;
145 break;
146 }
147 }
148 return res;
149}

151G4bool G4FermiFragmentsPoolVI::IsInThePool(G4int Z, G4int A,
152 G4double exc) const
153{
154 G4bool res = false;
155 G4int nfrag = fragment_pool.size();
156 for(G4int i=0; i<nfrag; ++i) {
157 const G4FermiFragment* fr = fragment_pool[i];
158 if(fr->GetZ() == Z && fr->GetA() == A &&
159 std::abs(exc - fr->GetExcitationEnergy()) < tolerance) {
160 res = true;
161 break;
162 }
163 }
164 return res;
165}

167G4bool G4FermiFragmentsPoolVI::IsInPhysPairs(
168 const G4FermiFragment* f1, const G4FermiFragment* f2) const
169{
170 G4bool res = false;
171 G4int A1 = f1->GetA();
172 G4int A2 = f2->GetA();
173 G4int A = A1 + A2;
174 G4int nn = list_p[A].size();
175 for(G4int i=0; i<nn; ++i) {
176 if(f1 == (list_p[A])[i]->GetFragment1() &&
177 f2 == (list_p[A])[i]->GetFragment2()) {
178 res = true;
179 break;
180 }
181 }
182 return res;
183}

187{
188 G4bool res = false;
189 G4int A1 = f1->GetA();
190 G4int A2 = f2->GetA();
191 G4int A = A1 + A2;
192 G4int nn = list_u[A].size();
193 for(G4int i=0; i<nn; ++i) {
194 if(f1 == (list_u[A])[i]->GetFragment1() &&
195 f2 == (list_u[A])[i]->GetFragment2()) {
196 res = true;
197 break;
198 }
199 }
200 return res;

203void G4FermiFragmentsPoolVI::Initialise()
204{
205 static const G4int nmin = 8;
206
207 // stable particles
208 fragment_pool.push_back(new G4FermiFragment(1, 0, 1, 0.0, true, true));
209 fragment_pool.push_back(new G4FermiFragment(1, 1, 1, 0.0, true, true));
210 fragment_pool.push_back(new G4FermiFragment(2, 1, 2, 0.0, true, true));
211 fragment_pool.push_back(new G4FermiFragment(3, 1, 1, 0.0, true, true));
212 fragment_pool.push_back(new G4FermiFragment(3, 2, 1, 0.0, true, true));
213 fragment_pool.push_back(new G4FermiFragment(4, 2, 0, 0.0, true, true));
214 fragment_pool.push_back(new G4FermiFragment(5, 2, 3, 0.0, true, true));
215 fragment_pool.push_back(new G4FermiFragment(5, 3, 3, 0.0, true, true));
216
217 // use level data and construct the pool
218 G4NuclearLevelData* ndata = G4NuclearLevelData::GetInstance();
219 for(G4int Z=1; Z<maxZ; ++Z) {
220 G4int Amin = ndata->GetMinA(Z);
221 G4int Amax = std::min(maxA, ndata->GetMaxA(Z)+1);
222 for(G4int A=Amin; A<Amax; ++A) {
223 const G4LevelManager* man = ndata->GetLevelManager(Z, A);
224 if(man) {
225 size_t nn = man->NumberOfTransitions();
226 // very unstable state
227 if(ndata->MaxLevelEnergy(Z, A) == 0.0f && man->LifeTime(0) == 0.0f) {
228 continue;
229 }
230 for(size_t i=0; i<=nn; ++i) {
231 G4float exc = man->LevelEnergy(i);
232 // only levels below limit are consided
233 if(exc >= elimf) { continue; }
234 G4double excd = (G4double)exc;
235 // only new are considered
236 if(IsInThePool(Z, A, excd)) { continue; }
237 G4float ltime = man->LifeTime(i);
238 G4bool stable = (ltime < 0.0f || ltime > timelim) ? true : false;
239 //G4cout << "Z= " << Z << " A= " << A << " Eex= " << exc
240 // << " t= " << ltime << " tlim= " << timelim
241 // << " stable: "<< stable << G4endl;
242 fragment_pool.push_back(new G4FermiFragment(A,Z,man->SpinTwo(i),excd,stable,true));
243 }
244 }
245 }
246 }
247 // prepare structures per A for normal fragments
248 static const size_t lfmax[maxA] = {
249 0, 2, 1, 2, 1, 2, 8, 19, 28, 56, 70, 104, 74, 109, 143, 212, 160};
250 // 0, 2, 1, 2, 1, 2, 6, 14, 16, 22, 45, 53, 37, 44, 33, 58, 63};
251 for(G4int A=1; A<maxA; ++A) {
252 list_f[A].reserve(lfmax[A]);
253 list_c[A].reserve(lfmax[A]);
254 }
255 static const size_t lfch[maxA] = {
256 0, 0, 0, 0, 0, 1, 4, 8, 8, 15, 33, 48, 36, 27, 66, 69, 39};
257 // 0, 0, 0, 0, 0, 1, 2, 5, 6, 3, 12, 8, 4, 10, 1, 8, 6};
258 G4int nfrag = fragment_pool.size();
259 for(G4int i=0; i<nfrag; ++i) {
260 const G4FermiFragment* f = fragment_pool[i];
261 G4int A = f->GetA();
262 G4double exc = f->GetExcitationEnergy();
263 list_f[A].push_back(f);
264 list_c[A].push_back(new G4FermiChannels(lfch[A], exc, f->GetTotalEnergy()));
265 }
266
267 // list of unphysical fragments
268 static const size_t lfun[maxA] = {
269
270 0, 0, 2, 2, 4, 4, 6, 6, 6, 6, 6, 6, 7, 6, 7, 6, 6};
271 for(G4int A=1; A<maxA; ++A) {
272 list_g[A].reserve(lfun[A]);
273 list_d[A].reserve(lfun[A]);
274 }
275
276 static const size_t luch[maxA] = {
277 0, 0, 1, 1, 3, 3, 5, 6, 6, 8, 8, 8, 8, 8, 8, 8, 8};
278 for(G4int Z=0; Z<maxZ; ++Z) {
279 G4int A0 = std::max(Z, 1);
280 for(G4int A=A0; A<maxA; ++A) {
281 if(IsInThePool(Z, A, 0.0)) { continue; }
282 const G4FermiFragment* f = new G4FermiFragment(A, Z, -1, 0.0, false, false);
283 funstable.push_back(f);
284 list_g[A].push_back(f);
285 list_d[A].push_back(new G4FermiChannels(luch[A],0.0,f->GetTotalEnergy()));
286 }
287 }
288 static const size_t pphm[maxA] = {
289 0, 0, 2, 2, 4, 4, 6, 6, 6, 6, 6, 6, 7, 6, 7, 6, 6};
290 static const size_t punm[maxA] = {
291 0, 0, 2, 2, 8, 10, 22, 24, 24, 31, 35, 36, 44, 36, 44, 36, 36};
292 for(G4int A=1; A<maxA; ++A) {
293 list_p[A].reserve(pphm[A]);
294 list_u[A].reserve(punm[A]);
295 }
296 /*
297 G4cout << "G4FermiFragmentsPoolVI::Initialise main loop @@@@@@"
298 << " Nfrag= " << nfrag << " Nuns= " << funstable.size() << G4endl;
299 */
300 // list of fragment pairs ordered by A
301 for(G4int i=0; i<nfrag; ++i) {
302 const G4FermiFragment* f1 = fragment_pool[i];
303 G4int Z1 = f1->GetZ();
304 G4int A1 = f1->GetA();
305 G4double e1 = f1->GetTotalEnergy();
306 for(G4int j=0; j<nfrag; ++j) {
307 const G4FermiFragment* f2 = fragment_pool[j];
308 G4int Z2 = f2->GetZ();
309 G4int A2 = f2->GetA();
310 if(A2 < A1 || (A2 == A1 && Z2 < Z1)) { continue; }
311 G4int Z = Z1 + Z2;
312 G4int A = A1 + A2;
313
314 if(Z >= maxZ || A >= maxA ||
315 IsInPhysPairs(f1, f2) || IsInUnphysPairs(f1, f2)) { continue; }
316
317 G4double e2 = f2->GetTotalEnergy();
318 G4double minE = e1 + e2;
319 G4double exc = 0.0;
320 if(IsPhysical(Z, A)) {
321 minE += f1->GetCoulombBarrier(A2, Z2, 0.0);
322 exc = minE - G4NucleiProperties::GetNuclearMass(A, Z);
323 }
324 /*
325 G4cout << "Z= " << Z << " A= " << A
326 << " Z1= " << Z1 << " A1= " << A1
327 << " Z2= " << Z2 << " A2= " << A2 << " Eex= " << exc
328 << " Qb= " << f1->GetCoulombBarrier(A2, Z2, 0.0)
329 << " " << e1
330 << " " << e2
331 << " " << G4NucleiProperties::GetNuclearMass(A, Z)
332 << G4endl;
333 */
334 // ignore very excited case
335 if(exc >= elim) { continue; }
336 G4FermiPair* fpair = nullptr;
337 G4int kmax = list_f[A].size();
338 for(G4int k=0; k<kmax; ++k) {
339 const G4FermiFragment* f3 = (list_f[A])[k];
340 if(Z == f3->GetZ() &&
341 f3->GetTotalEnergy() - minE + tolerance >= 0.0) {
342 if(!fpair) {
343 fpair = new G4FermiPair(f1, f2);
344 list_p[A].push_back(fpair);
345 }
346 (list_c[A])[k]->AddChannel(fpair);
347 }
348 }
349 if(fpair) { continue; }
350
351 kmax = list_g[A].size();
352 for(G4int k=0; k<kmax; ++k) {
353 if((list_d[A])[k]->GetNumberOfChannels() >= nmin) { continue; }
354 const G4FermiFragment* f3 = (list_g[A])[k];
355 /*
356 if(Z==0) {
357 G4cout << "%%% A= " << A << " isStable: " << f3->IsStable()
358 << " de= " << f3->GetTotalEnergy() - minE << G4endl;
359 }
360 */
361 if(Z == f3->GetZ() &&
362 f3->GetTotalEnergy() - minE + tolerance >= 0.0) {
363 if(!fpair) {
364 fpair = new G4FermiPair(f1, f2);
365 list_u[A].push_back(fpair);
366 }
367 (list_d[A])[k]->AddChannel(fpair);
368 /*
369 if(Z==0) {
370 G4cout << " isAdded Unstable" << G4endl;
371 }
372 */
373 }
374 }
375 }
376 }
377
378 // G4cout << "@@@@@@ sec loop @@@@@@" << G4endl;
379 // list of fragment pairs (stable+unstable) ordered by A
380 G4int unphys = funstable.size();
381 for(G4int i=0; i<nfrag; ++i) {
382 const G4FermiFragment* f1 = fragment_pool[i];
383 G4int Z1 = f1->GetZ();
384 G4int A1 = f1->GetA();
385 G4double e1 = f1->GetTotalEnergy();
386 for(G4int j=0; j<unphys; ++j) {
387 const G4FermiFragment* f2 = funstable[j];
388 G4int Z2 = f2->GetZ();
389 G4int A2 = f2->GetA();
390 G4int Z = Z1 + Z2;
391 G4int A = A1 + A2;
392 if(Z >= maxZ || A >= maxA || IsInUnphysPairs(f1, f2) || IsPhysical(Z, A))
393 { continue; }
394 G4double e2 = f2->GetTotalEnergy();
395 G4double minE = e1 + e2;
396 /*
397 G4cout << "Z= " << Z << " A= " << A << " Z1= " << Z1 << " A1= " << A1
398 << " Z2= " << Z2 << " A2= " << A2 << G4endl;
399 */
400 // check if this is the list of stable pairs
401 G4FermiPair* fpair = nullptr;
402 // check unphysics list
403 G4int kmax = list_g[A].size();
404 for(G4int k=0; k<kmax; ++k) {
405 const G4FermiFragment* f3 = (list_g[A])[k];
406 /*
407 if(Z == f3->GetZ())
408 G4cout << " Unst+ST k= " << k << " Z= " << f3->GetZ()
409 << " isStab " << f3->IsStable()
410 << " Nch= " << (list_d[A])[k]->GetNumberOfChannels()
411 << " de= " << f3->GetTotalEnergy() - minE
412 << G4endl;
413 */
414 if(Z == f3->GetZ() &&
415 (list_d[A])[k]->GetNumberOfChannels() < nmin &&
416 f3->GetTotalEnergy() - minE + tolerance >= 0.0)
417 {
418 fpair = new G4FermiPair(f1, f2);
419 list_u[A].push_back(fpair);
420 (list_d[A])[k]->AddChannel(fpair);
421 // G4cout << " isAdded Unstable" << G4endl;
422 break;
423 }
424 }
425 }
426 }
427
428 // compute static probabilities
429 for(G4int A=1; A<maxA; ++A) {
430 for(size_t j=0; j<list_c[A].size(); ++j) {
431 G4FermiChannels* ch = (list_c[A])[j];
432 const G4FermiFragment* frag = (list_f[A])[j];
433 size_t nch = ch->GetNumberOfChannels();
434 if(1 < nch) {
435 std::vector<G4double>& prob = ch->GetProbabilities();
436 const std::vector<const G4FermiPair*>& pairs = ch->GetChannels();
437 G4double ptot = 0.0;
438 for(size_t i=0; i<nch; ++i) {
439 ptot += theDecay.ComputeProbability(frag->GetZ(), frag->GetA(),
440 frag->GetSpin(),
441 frag->GetTotalEnergy(),
442 pairs[i]->GetFragment1(),
443 pairs[i]->GetFragment2());
444 prob[i] = ptot;
445 }
446 if(0.0 == ptot) {
447 prob[0] = 1.0;
448 } else {
449 ptot = 1./ptot;
450 for(size_t i=0; i<nch-1; ++i) { prob[i] *= ptot; }
451 prob[nch-1] = 1.0;
452 }
453 }
454 }
455 }
456 for(G4int A=1; A<maxA; ++A) {
457 for(size_t j=0; j<list_d[A].size(); ++j) {
458 G4FermiChannels* ch = (list_d[A])[j];
459 const G4FermiFragment* frag = (list_g[A])[j];
460 size_t nch = ch->GetNumberOfChannels();
461 if(1 < nch) {
462 std::vector<G4double>& prob = ch->GetProbabilities();
463 const std::vector<const G4FermiPair*>& pairs = ch->GetChannels();
464 G4double ptot = 0.0;
465 for(size_t i=0; i<nch; ++i) {
466 ptot += theDecay.ComputeProbability(frag->GetZ(), frag->GetA(),
467 frag->GetSpin(),
468 frag->GetTotalEnergy(),
469 pairs[i]->GetFragment1(),
470 pairs[i]->GetFragment2());
471 prob[i] = ptot;
472 }
473 if(0.0 == ptot) {
474 prob[0] = 1.0;
475 } else {
476 ptot = 1./ptot;
477 for(size_t i=0; i<nch-1; ++i) { prob[i] *= ptot; }
478 prob[nch-1] = 1.0;
479 }
480 }
481 }
482 }
483}
