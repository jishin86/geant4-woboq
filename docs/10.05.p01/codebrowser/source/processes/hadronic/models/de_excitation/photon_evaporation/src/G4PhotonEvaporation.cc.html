63G4PhotonEvaporation::G4PhotonEvaporation(G4GammaTransition* p)
64 : fLevelManager(nullptr), fTransition(p), fPolarization(nullptr),
65 fVerbose(0), fPoints(0), vShellNumber(-1), fIndex(0),
66 fMaxLifeTime(DBL_MAX),
67 fICM(true), fRDM(false), fSampleTime(true),
68 fCorrelatedGamma(false), isInitialised(false)
69{
70 //G4cout << "### New G4PhotonEvaporation() " << this << G4endl;
71 fNuclearLevelData = G4NuclearLevelData::GetInstance();
72 fNucPStore = G4NuclearPolarizationStore::GetInstance();
73 Tolerance = 20*CLHEP::eV;
74
75 if(!fTransition) { fTransition = new G4GammaTransition(); }
76
77 theA = theZ = fCode = 0;
78 fLevelEnergyMax = fStep = fExcEnergy = fProbability = 0.0;
79
80 for(G4int i=0; i<MAXDEPOINT; ++i) { fCummProbability[i] = 0.0; }
81 if(0.0f == GREnergy[1]) { InitialiseGRData(); }
82}

84G4PhotonEvaporation::~G4PhotonEvaporation()
85{
86 delete fTransition;
87}

89void G4PhotonEvaporation::Initialise()
90{
91 if(isInitialised) { return; }
92 isInitialised = true;
93
94 if(fVerbose > 0) {
95 G4cout << "### G4PhotonEvaporation is initialized " << this << G4endl;
96 }
97 G4DeexPrecoParameters* param = fNuclearLevelData->GetParameters();
98 Tolerance = param->GetMinExcitation();
99 fMaxLifeTime = param->GetMaxLifeTime();
100 fCorrelatedGamma = param->CorrelatedGamma();
101 fICM = param->GetInternalConversionFlag();
102
103 fTransition->SetPolarizationFlag(fCorrelatedGamma);
104 fTransition->SetTwoJMAX(param->GetTwoJMAX());
105 fTransition->SetVerbose(fVerbose);
106}

108void G4PhotonEvaporation::InitialiseGRData()
109{
110#ifdef G4MULTITHREADED
111 G4MUTEXLOCK(&G4PhotonEvaporation::PhotonEvaporationMutex);
112#endif
113 if(0.0f == GREnergy[1]) {
114 G4Pow* g4calc = G4Pow::GetInstance();
115 static const G4float GRWfactor = 0.3f;
116 for (G4int A=1; A<MAXGRDATA; ++A) {
117 GREnergy[A] = (G4float)(40.3*CLHEP::MeV/g4calc->powZ(A,0.2));
118 GRWidth[A] = GRWfactor*GREnergy[A];
119 }
120 }
121#ifdef G4MULTITHREADED
122 G4MUTEXUNLOCK(&G4PhotonEvaporation::PhotonEvaporationMutex);
123#endif
124}

126G4Fragment*
127G4PhotonEvaporation::EmittedFragment(G4Fragment* nucleus)
128{
129 if(!isInitialised) { Initialise(); }
130 fSampleTime = (fRDM) ? false : true;
131
132 // potentially external code may set initial polarization
133 // but only for radioactive decay nuclear polarization is considered
134 if(fCorrelatedGamma && fRDM) {
135 if(nucleus->GetNuclearPolarization()) {
136 fNucPStore->RemoveMe(nucleus->GetNuclearPolarization());
137 delete nucleus->GetNuclearPolarization();
138 }
139 fPolarization = fNucPStore->FindOrBuild(nucleus->GetZ_asInt(),
140 nucleus->GetA_asInt(),
141 nucleus->GetExcitationEnergy());
142 nucleus->SetNuclearPolarization(fPolarization);
143 }
144 if(fVerbose > 1) {
145 G4cout << "G4PhotonEvaporation::EmittedFragment: "
146 << *nucleus << G4endl;
147 if(fPolarization) { G4cout << "NucPolar: " << fPolarization << G4endl; }
148 G4cout << " CorrGamma: " << fCorrelatedGamma << " RDM: " << fRDM
149 << " fPolarization: " << fPolarization << G4endl;
150 }
151 G4Fragment* gamma = GenerateGamma(nucleus);
152
153 // remove G4NuclearPolarizaton when reach ground state
154 if(fPolarization && 0 == fIndex) {
155 if(fVerbose > 1) {
156 G4cout << "G4PhotonEvaporation::EmittedFragment: remove "
157 << fPolarization << G4endl;
158 }
159 fNucPStore->RemoveMe(fPolarization);
160 fPolarization = nullptr;
161 nucleus->SetNuclearPolarization(fPolarization);
162 }
163
164 if(fVerbose > 1) {
165 G4cout << "G4PhotonEvaporation::EmittedFragment: RDM= "
166 << fRDM << " done:" << G4endl;
167 if(gamma) { G4cout << *gamma << G4endl; }
168 G4cout << " Residual: " << *nucleus << G4endl;
169 }
170 return gamma;
171}

173G4FragmentVector*
174G4PhotonEvaporation::BreakItUp(const G4Fragment& nucleus)
175{
176 if(fVerbose > 0) {
177 G4cout << "G4PhotonEvaporation::BreakItUp" << G4endl;
178 }
179 G4Fragment* aNucleus = new G4Fragment(nucleus);
180 G4FragmentVector* products = new G4FragmentVector();
181 BreakUpChain(products, aNucleus);
182 products->push_back(aNucleus);
183 return products;
184}

186G4bool G4PhotonEvaporation::BreakUpChain(G4FragmentVector* products,
187 G4Fragment* nucleus)
188{
189 if(!isInitialised) { Initialise(); }
190 if(fVerbose > 0) {
191 G4cout << "G4PhotonEvaporation::BreakUpChain RDM= " << fRDM << " "
192 << *nucleus << G4endl;
193 }
194 G4Fragment* gamma = nullptr;
195 fSampleTime = (fRDM) ? false : true;
196
197 // start decay chain from unpolarized state
198 if(fCorrelatedGamma) {
199 fPolarization = new G4NuclearPolarization(nucleus->GetZ_asInt(),
200 nucleus->GetA_asInt(),
201 nucleus->GetExcitationEnergy());
202 nucleus->SetNuclearPolarization(fPolarization);
203 }
204
205 do {
206 gamma = GenerateGamma(nucleus);
207 if(gamma) {
208 products->push_back(gamma);
209 if(fVerbose > 0) {
210 G4cout << "G4PhotonEvaporation::BreakUpChain: "
211 << *gamma << G4endl;
212 G4cout << " Residual: " << *nucleus << G4endl;
213 }
214 // for next decays in the chain always sample time
215 fSampleTime = true;
216 }
217 // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
218 } while(gamma);
219
220 // clear nuclear polarization end of chain
221 if(fPolarization) {
222 delete fPolarization;
223 fPolarization = nullptr;
224 nucleus->SetNuclearPolarization(fPolarization);
225 }
226 return false;
227}

229G4double
230G4PhotonEvaporation::GetEmissionProbability(G4Fragment* nucleus)
231{
232 if(!isInitialised) { Initialise(); }
233 fProbability = 0.0;
234 fExcEnergy = nucleus->GetExcitationEnergy();
235 G4int Z = nucleus->GetZ_asInt();
236 G4int A = nucleus->GetA_asInt();
237 fCode = 1000*Z + A;
238 if(fVerbose > 1) {
239 G4cout << "G4PhotonEvaporation::GetEmissionProbability: Z="
240 << Z << " A=" << A << " Eexc(MeV)= " << fExcEnergy << G4endl;
241 }
242 // ignore gamma de-excitation for exotic fragments
243 // and for very low excitations
244 if(0 >= Z || 1 >= A || Z == A || Tolerance >= fExcEnergy)
245 { return fProbability; }
246
247 // ignore gamma de-excitation for highly excited levels
248 if(A >= MAXGRDATA) { A = MAXGRDATA-1; }
249 //G4cout<<" GREnergy= "<< GREnergy[A]<<" GRWidth= "<<GRWidth[A]<<G4endl;
250
251 static const G4float GREfactor = 5.0f;
252 if(fExcEnergy >= (G4double)(GREfactor*GRWidth[A] + GREnergy[A])) {
253 return fProbability;
254 }
255 // probability computed assuming continium transitions
256 // VI: continium transition are limited only to final states
257 // below Fermi energy (this approach needs further evaluation)
258 G4double emax = std::max(0.0, nucleus->ComputeGroundStateMass(Z, A-1)
259 + CLHEP::neutron_mass_c2 - nucleus->GetGroundStateMass());
260
261 // max energy level for continues transition
262 emax = std::min(emax, fExcEnergy);
263 const G4double eexcfac = 0.99;
264 if(0.0 == emax || fExcEnergy*eexcfac <= emax) { emax = fExcEnergy*eexcfac; }
265
266 fStep = emax;
267 static const G4double MaxDeltaEnergy = CLHEP::MeV;
268 fPoints = std::min((G4int)(fStep/MaxDeltaEnergy) + 2, MAXDEPOINT);
269 fStep /= ((G4double)(fPoints - 1));
270 if(fVerbose > 1) {
271 G4cout << "Emax= " << emax << " Npoints= " << fPoints
272 << " Eex= " << fExcEnergy << G4endl;
273 }
274 // integrate probabilities
275 G4double eres = (G4double)GREnergy[A];
276 G4double wres = (G4double)GRWidth[A];
277 G4double eres2= eres*eres;
278 G4double wres2= wres*wres;
279 G4double levelDensity = fNuclearLevelData->GetLevelDensity(Z,A,fExcEnergy);
280 G4double xsqr = std::sqrt(levelDensity*fExcEnergy);
281
282 G4double egam = fExcEnergy;
283 G4double gammaE2 = egam*egam;
284 G4double gammaR2 = gammaE2*wres2;
285 G4double egdp2 = gammaE2 - eres2;
286
287 G4double p0 = G4Exp(-2.0*xsqr)*gammaR2*gammaE2/(egdp2*egdp2 + gammaR2);
288 G4double p1(0.0);
289
290 for(G4int i=1; i<fPoints; ++i) {
291 egam -= fStep;
292 gammaE2 = egam*egam;
293 gammaR2 = gammaE2*wres2;
294 egdp2 = gammaE2 - eres2;
295 p1 = G4Exp(2.0*(std::sqrt(levelDensity*std::abs(fExcEnergy - egam)) - xsqr))
296 *gammaR2*gammaE2/(egdp2*egdp2 + gammaR2);
297 fProbability += (p1 + p0);
298 fCummProbability[i] = fProbability;
299 if(fVerbose > 2) {
300 G4cout << "Egamma= " << egam << " Eex= " << fExcEnergy
301 << " p0= " << p0 << " p1= " << p1 << " sum= "
302 << fCummProbability[i] <<G4endl;
303 }
304 p0 = p1;
305 }
306
307 static const G4double NormC = 1.25*CLHEP::millibarn
308 /(CLHEP::pi2*CLHEP::hbarc*CLHEP::hbarc);
309 fProbability *= fStep*NormC*A;
310 if(fVerbose > 1) { G4cout << "prob= " << fProbability << G4endl; }
311 return fProbability;
312}

314G4double
315G4PhotonEvaporation::GetFinalLevelEnergy(G4int Z, G4int A, G4double energy)
316{
317 G4double E = energy;
318 InitialiseLevelManager(Z, A);
319 if(fLevelManager) {
320 E = fLevelManager->NearestLevelEnergy(energy, fIndex);
321 if(E > fLevelEnergyMax + Tolerance) { E = energy; }
322 }
323 return E;
324}

326G4double G4PhotonEvaporation::GetUpperLevelEnergy(G4int Z, G4int A)
327{
328 InitialiseLevelManager(Z, A);
329 return fLevelEnergyMax;

332G4Fragment*
333G4PhotonEvaporation::GenerateGamma(G4Fragment* nucleus)
334{
335 if(!isInitialised) { Initialise(); }
336 G4Fragment* result = nullptr;
337 G4double eexc = nucleus->GetExcitationEnergy();
338 if(eexc <= Tolerance) { return result; }
339
340 InitialiseLevelManager(nucleus->GetZ_asInt(), nucleus->GetA_asInt());
341
342 G4double time = nucleus->GetCreationTime();
343
344 G4double efinal = 0.0;
345 G4double ratio = 0.0;
346 vShellNumber = -1;
347 G4int JP1 = 0;
348 G4int JP2 = 0;
349 G4int multiP = 0;
350 G4bool isGamma = true;
351 G4bool isDiscrete = false;
352
353 const G4NucLevel* level = nullptr;
354 size_t ntrans = 0;
355
356 if(fVerbose > 1) {
357 G4cout << "GenerateGamma: " << " Eex= " << eexc
358 << " Eexmax= " << fLevelEnergyMax << G4endl;
359 }
360 // initial discrete state
361 if(fLevelManager && eexc <= fLevelEnergyMax + Tolerance) {
362 fIndex = fLevelManager->NearestLevelIndex(eexc, fIndex);
363 isDiscrete = true;
364 if(fVerbose > 1) {
365 G4cout << " index= " << fIndex
366 << " lTime= " << fLevelManager->LifeTime(fIndex) << G4endl;
367 }
368 if(0 < fIndex) {
369 // for discrete transition
370 level = fLevelManager->GetLevel(fIndex);
371 if(level) {
372 ntrans = level->NumberOfTransitions();
373 JP1 = fLevelManager->SpinTwo(fIndex);
374 if(fVerbose > 2) {
375 G4cout << " ntrans= " << ntrans << " JP= " << JP1
376 << " RDM: " << fRDM << G4endl;
377 }
378 if(0 == ntrans && fLevelManager->FloatingLevel(fIndex) > 0) {
379 --fIndex;
380 level = fLevelManager->GetLevel(fIndex);
381 ntrans = level->NumberOfTransitions();
382 JP1 = fLevelManager->SpinTwo(fIndex);
383 }
384 }
385 }
386 }
387 if(fVerbose > 1) {
388 G4int prec = G4cout.precision(4);
389 G4cout << "GenerateGamma: Z= " << nucleus->GetZ_asInt()
390 << " A= " << nucleus->GetA_asInt()
391 << " Exc= " << eexc << " Emax= "
392 << fLevelEnergyMax << " idx= " << fIndex
393 << " fCode= " << fCode << " fPoints= " << fPoints
394 << " Ntr= " << ntrans << " discrete: " << isDiscrete
395 << " fProb= " << fProbability << G4endl;
396 G4cout.precision(prec);
397 }
398
399 // continues part
400 if(!isDiscrete) {
401 // we compare current excitation versus value used for probability
402 // computation and also Z and A used for probability computation
403 if(fCode != 1000*theZ + theA || eexc != fExcEnergy) {
404 GetEmissionProbability(nucleus);
405 }
406 //if(fProbability == 0.0) { return result; }
407 if(fProbability == 0.0) {
408 fPoints = 1;
409 efinal = 0.0;
410 } else {
411 G4double y = fCummProbability[fPoints-1]*G4UniformRand();
412 for(G4int i=1; i<fPoints; ++i) {
413 if(fVerbose > 2) {
414 G4cout << "y= " << y << " cummProb= " << fCummProbability[i]
415 << " fPoints= " << fPoints << " fStep= " << fStep << G4endl;
416 }
417 if(y <= fCummProbability[i]) {
418 efinal = fStep*((i - 1) + (y - fCummProbability[i-1])
419 /(fCummProbability[i] - fCummProbability[i-1]));
420 break;
421 }
422 }
423 }
424 // final discrete level
425 if(fVerbose > 2) {
426 G4cout << "Continues Efinal= " << efinal << G4endl;
427 }
428 if(fLevelManager) {
429 if(efinal < fLevelEnergyMax) {
430 fIndex = fLevelManager->NearestLevelIndex(efinal, fIndex);
431 efinal = fLevelManager->LevelEnergy(fIndex);
432 // protection - take level below
433 if(efinal >= eexc && 0 < fIndex) {
434 --fIndex;
435 efinal = fLevelManager->LevelEnergy(fIndex);
436 }
437 nucleus->SetFloatingLevelNumber(fLevelManager->FloatingLevel(fIndex));
438
439 // not allowed to have final energy above max energy
440 // if G4LevelManager exist
441 } else {
442 efinal = fLevelEnergyMax;
443 fIndex = fLevelManager->NearestLevelIndex(efinal, fIndex);
444 }
445 }
446 if(fVerbose > 1) {
447 G4cout << "Continues emission efinal(MeV)= " << efinal << G4endl;
448 }
449 //discrete part ground state
450 } else if(0 == fIndex) {
451 return result;
452
453 //discrete part no transitions defined
454 } else if(0 == ntrans) {
455 for(size_t ii=fIndex-1; ii>0; --ii) {
456 level = fLevelManager->GetLevel(ii);
457 if(level) {
458 ntrans = level->NumberOfTransitions();
459 // transition identified
460 if(0 < ntrans) {
461 efinal = fLevelManager->LevelEnergy(ii);
462 fIndex = ii;
463 break;
464 }
465 }
466 }
467
468 //discrete part for levels with transitions
469 } else {
470
471 if(fVerbose > 1) {
472 G4cout << "Discrete emission from level Index= " << fIndex
473 << " Elevel= " << fLevelManager->LevelEnergy(fIndex)
474 << " Ltime= " << fLevelManager->LifeTime(fIndex)
475 << " LtimeMax= " << fMaxLifeTime
476 << " RDM= " << fRDM << " ICM= " << fICM << G4endl;
477 }
478
479 // stable fragment has life time -1 or above the limit
480 // if is called from the radioactive decay the life time is not checked
481 G4double ltime = fLevelManager->LifeTime(fIndex);
482 if(ltime < 0.0 || (!fRDM && ltime > fMaxLifeTime)) { return result; }
483
484 size_t idx = 0;
485 if(1 < ntrans) {
486 idx = level->SampleGammaTransition(G4UniformRand());
487 }
488 if(fVerbose > 1) {
489 G4cout << "Ntrans= " << ntrans << " idx= " << idx
490 << " ICM= " << fICM << " JP1= " << JP1 << G4endl;
491 }
492 G4double prob = (G4double)level->GammaProbability(idx);
493 // prob = 0 means that there is only internal conversion
494 if(fICM && prob < 1.0) {
495 G4double rndm = G4UniformRand();
496 if(rndm > prob) {
497 isGamma = false;
498 rndm = (rndm - prob)/(1.0 - prob);
499 vShellNumber = level->SampleShell(idx, rndm);
500 }
501 }
502 // it is discrete transition with possible gamma correlation
503 ratio = level->MultipolarityRatio(idx);
504 multiP = level->TransitionType(idx);
505 fIndex = level->FinalExcitationIndex(idx);
506 JP2 = fLevelManager->SpinTwo(fIndex);
507
508 // final energy and time
509 efinal = fLevelManager->LevelEnergy(fIndex);
510 if(fSampleTime && ltime > 0.0) {
511 time -= ltime*G4Log(G4UniformRand());
512 }
513 nucleus->SetFloatingLevelNumber(fLevelManager->FloatingLevel(fIndex));
514 }
515 // protection for floating levels
516 if(std::abs(efinal - eexc) <= Tolerance) { return result; }
517
518 result = fTransition->SampleTransition(nucleus, efinal, ratio, JP1,
519 JP2, multiP, vShellNumber,
520 isDiscrete, isGamma);
521 if(result) { result->SetCreationTime(time); }
522
523 // updated residual nucleus
524 nucleus->SetCreationTime(time);
525 nucleus->SetSpin(0.5*JP2);
526 if(fPolarization) { fPolarization->SetExcitationEnergy(efinal); }
527
528 // ignore the floating levels with zero energy and create ground state
529 if(efinal == 0.0 && fIndex > 0) {
530 fIndex = 0;
531 nucleus->SetFloatingLevelNumber(fLevelManager->FloatingLevel(fIndex));
532 }
533
534 if(fVerbose > 1) {
535 G4cout << "Final level E= " << efinal << " time= " << time
536 << " idxFinal= " << fIndex << " isDiscrete: " << isDiscrete
537 << " isGamma: " << isGamma << " multiP= " << multiP
538 << " shell= " << vShellNumber
539 << " JP1= " << JP1 << " JP2= " << JP2 << G4endl;
540 }
541 return result;
542}
