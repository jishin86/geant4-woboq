49G4LundStringFragmentation::G4LundStringFragmentation()
50{
51 SetMassCut(160.*MeV); // For LightFragmentationTest it is required
52 // that no one pi-meson can be produced.
53
54 SigmaQT = 0.435 * GeV;
55
56 SetStringTensionParameter(1.*GeV/fermi);
57 SetDiquarkBreakProbability(0.5);
58 SetStrangenessSuppression((1.0 - 0.13)/2.0);
59
60 SetDiquarkSuppression(0.15);
61
62 SetMinMasses(); // For treating of small string decays
63
64 //... pspin_meson is probability to create pseudo-scalar meson
65 pspin_meson = 0.5; SetVectorMesonProbability(pspin_meson);
66
67 //... pspin_barion is probability to create 1/2 barion
68 pspin_barion = 0.5; SetSpinThreeHalfBarionProbability(pspin_barion);
69
70 //... vectorMesonMix[] is quark mixing parameters for vector mesons (Variable spin = 3)
71 vectorMesonMix[0] = 0.0;
72 vectorMesonMix[1] = 0.5;
73 vectorMesonMix[2] = 0.0;
74 vectorMesonMix[3] = 0.5;
75 vectorMesonMix[4] = 1.0;
76 vectorMesonMix[5] = 1.0;
77
78 SetVectorMesonMixings(vectorMesonMix);
79}

82G4KineticTrackVector* G4LundStringFragmentation::FragmentString(const G4ExcitedString& theString)
83{
84 // Can no longer modify Parameters for Fragmentation.
85
86 PastInitPhase=true;
87
88 G4FragmentingString aString(theString);
89 SetMinimalStringMass(&aString);
90
91 #ifdef debug_LUNDfragmentation
92 G4cout<<G4endl<<"LUND StringFragmentation ------------------------------------"<<G4endl;
93 G4cout<<G4endl<<"LUND StringFragm: String Mass "
94 <<theString.Get4Momentum().mag()<<G4endl
95 <<" 4Mom "<<theString.Get4Momentum()
96 <<"------------------------------------"<<G4endl;
97 G4cout<<"String ends Direct "<<theString.GetLeftParton()->GetPDGcode()<<" "
98 <<theString.GetRightParton()->GetPDGcode()<<" "
99 <<theString.GetDirection()<< G4endl;
100 G4cout<<"Left mom "<<theString.GetLeftParton()->Get4Momentum()<<G4endl;
101 G4cout<<"Right mom "<<theString.GetRightParton()->Get4Momentum()<<G4endl;
102 G4cout<<"Check for Fragmentation "<<G4endl;
103 #endif
104
105 G4KineticTrackVector * LeftVector(0);
106
107 if (!aString.FourQuarkString() && !IsFragmentable(&aString))
108 {
109 #ifdef debug_LUNDfragmentation
110 G4cout<<"Non fragmentable - the string is converted to one hadron "<<G4endl;
111 #endif
112 // SetMassCut(160.*MeV); // For LightFragmentationTest it is required
113 // that no one pi-meson can be produced.
114
115 G4double Mcut = GetMassCut();
116 SetMassCut(10000.*MeV);
117 LeftVector=LightFragmentationTest(&theString);
118 SetMassCut(Mcut);
119
120 LeftVector->operator[](0)->SetFormationTime(theString.GetTimeOfCreation());
121 LeftVector->operator[](0)->SetPosition(theString.GetPosition());
122
123 if(LeftVector->size() > 1)
124 {
125 // 2 hadrons created from qq-qqbar are stored
126 LeftVector->operator[](1)->SetFormationTime(theString.GetTimeOfCreation());
127 LeftVector->operator[](1)->SetPosition(theString.GetPosition());
128 }
129 return LeftVector;
130 }
131
132 #ifdef debug_LUNDfragmentation
133 G4cout<<"The string will be fragmented. "<<G4endl;
134 #endif
135
136 // The string can fragment. At least two particles can be produced.
137 LeftVector =new G4KineticTrackVector;
138 G4KineticTrackVector * RightVector=new G4KineticTrackVector;
139
140 G4bool success = Loop_toFragmentString(theString, LeftVector, RightVector);
141
142 if ( ! success )
143 {
144 std::for_each(LeftVector->begin(), LeftVector->end(), DeleteKineticTrack());
145 LeftVector->clear();
146 std::for_each(RightVector->begin(), RightVector->end(), DeleteKineticTrack());
147 delete RightVector;
148 return LeftVector;
149 }
150
151 // Join Left- and RightVector into LeftVector in correct order.
152 while(!RightVector->empty())
153 {
154 LeftVector->push_back(RightVector->back());
155 RightVector->erase(RightVector->end()-1);
156 }
157 delete RightVector;
158
159 return LeftVector;
160}

163G4bool G4LundStringFragmentation::IsFragmentable(const G4FragmentingString * const string)
164{
165 SetMinimalStringMass(string);
166 // G4cout<<"MinM StrM "<<MinimalStringMass<<" "<< string->Get4Momentum().mag()<<G4endl;
167 return MinimalStringMass < string->Get4Momentum().mag();
168}

170//----------------------------------------------------------------------------------------
171G4bool G4LundStringFragmentation::Loop_toFragmentString(
172 const G4ExcitedString &theString,
173 G4KineticTrackVector * & LeftVector,
174 G4KineticTrackVector * & RightVector )
175{
176 #ifdef debug_LUNDfragmentation
177 G4cout<<"Loop_toFrag "<<theString.GetLeftParton()->GetPDGcode()<<" "
178 <<theString.GetLeftParton()->Get4Momentum()<<G4endl
179 <<" "<<theString.GetRightParton()->GetPDGcode()<<" "
180 <<theString.GetRightParton()->Get4Momentum()<<G4endl
181 <<"Direction "<<theString.GetDirection()<< G4endl;
182 #endif
183
184 G4bool final_success=false;
185 G4bool inner_success=true;
186
187 G4int attempt=0;
188
189 while ( ! final_success && attempt++ < StringLoopInterrupt )
190 { // If the string fragmentation does not be happend,
191 // repeat the fragmentation.
192
193 G4FragmentingString *currentString=new G4FragmentingString(theString);
194 G4LorentzRotation toCms, toObserverFrame;
195
196 //G4cout<<"Main loop start whilecounter "<<attempt<<G4endl;
197
198 // Cleaning up the previously produced hadrons
199 std::for_each(LeftVector->begin(), LeftVector->end(), DeleteKineticTrack());
200 LeftVector->clear();
201 std::for_each(RightVector->begin(), RightVector->end(), DeleteKineticTrack());
202 RightVector->clear();
203
204 // Main fragmentation loop until the string will not be able to fragment
205 inner_success=true; // set false on failure.
206 const G4int maxNumberOfLoops = 1000;
207 G4int loopCounter = -1;
208
209 while ( (! StopFragmenting(currentString)) && ++loopCounter < maxNumberOfLoops )
210 { // Split current string into hadron + new string
211 #ifdef debug_LUNDfragmentation
212 G4cout<<"The string can fragment. "<<G4endl;;
213 //G4cout<<"1 "<<currentString->GetDecayDirection()<<G4endl;
214 #endif
215 G4FragmentingString *newString=0; // used as output from SplitUp.
216
217 toCms=currentString->TransformToAlignedCms();
218 toObserverFrame= toCms.inverse();
219 #ifdef debug_LUNDfragmentation
220 G4cout<<"CMS Left mom "<<currentString->GetPleft()<<G4endl;
221 G4cout<<"CMS Right mom "<<currentString->GetPright()<<G4endl;
222 G4cout<<"CMS String M "<<currentString->GetPstring()<<G4endl;
223 #endif
224
225 G4KineticTrack * Hadron=Splitup(currentString,newString);
226
227 if ( Hadron != 0 ) // Store the hadron
228 {
229 #ifdef debug_LUNDfragmentation
230 G4cout<<"Hadron prod at fragm. "<<Hadron->GetDefinition()->GetParticleName()<<G4endl;
231 //G4cout<<"2 "<<currentString->GetDecayDirection()<<G4endl;
232 #endif
233
234 Hadron->Set4Momentum(toObserverFrame*Hadron->Get4Momentum());
235
236 G4double TimeOftheStringCreation=theString.GetTimeOfCreation();
237 G4ThreeVector PositionOftheStringCreation(theString.GetPosition());
238
239 G4LorentzVector Coordinate(Hadron->GetPosition(), Hadron->GetFormationTime());
240 G4LorentzVector Momentum = toObserverFrame*Coordinate;
241 Hadron->SetFormationTime(TimeOftheStringCreation + Momentum.e() - fermi/c_light);
242 G4ThreeVector aPosition(Momentum.vect());
243 Hadron->SetPosition(PositionOftheStringCreation+aPosition);
244
245 // Open to protect hadron production at fragmentation
246 if ( currentString->GetDecayDirection() > 0 )
247 {
248 LeftVector->push_back(Hadron);
249 } else
250 {
251 RightVector->push_back(Hadron);
252 }
253 delete currentString;
254 currentString=newString;
255 } else {
256 if ( newString ) delete newString;
257 }
258
259 currentString->LorentzRotate(toObserverFrame);
260 };
261
262 if ( loopCounter >= maxNumberOfLoops ) {
263 inner_success=false;
264 }
265
266 // Split remaining string into 2 final hadrons.
267 #ifdef debug_LUNDfragmentation
268 if(inner_success) G4cout<<"Split remaining string into 2 final hadrons."<<G4endl;
269 #endif
270 if ( inner_success && SplitLast(currentString, LeftVector, RightVector) ) // Close to protect Last Str. Decay
271 {
272 final_success = true;
273 }
274
275 delete currentString;
276 } // End of the loop where we try to fragment the string.
277 return final_success;
278}

280//----------------------------------------------------------------------------------------
281G4bool G4LundStringFragmentation::StopFragmenting(const G4FragmentingString * const string)
282{
283 SetMinimalStringMass(string);
284 if (string->FourQuarkString())
285 {
286 return G4UniformRand() < G4Exp(-0.0005*(string->Mass() - MinimalStringMass));
287 } else {
288
289 G4bool Result = G4UniformRand() <
290 G4Exp(-0.66e-6*(string->Mass()*string->Mass() - MinimalStringMass*MinimalStringMass));
291 // G4bool Result = string->Mass() < MinimalStringMass + 150.*MeV*G4UniformRand(); // a'la LUND
292
293 #ifdef debug_LUNDfragmentation
294 G4cout<<"StopFragmenting MinimalStringMass string->Mass() "<<MinimalStringMass<<" "<<string->Mass()<<G4endl;
295 G4cout<<"StopFragmenting - Yes/No "<<Result<<G4endl;
296 #endif
297 return Result;
298 }

301//-----------------------------------------------------------------------------
302G4KineticTrack * G4LundStringFragmentation::Splitup(
303 G4FragmentingString *string,
304 G4FragmentingString *&newString)
305{
306 #ifdef debug_LUNDfragmentation
307 G4cout<<G4endl;
308 G4cout<<"Start SplitUP (G4VLongitudinalStringDecay) ========================="<<G4endl;
309 G4cout<<"String partons: " <<string->GetLeftParton()->GetPDGEncoding()<<" "
310 <<string->GetRightParton()->GetPDGEncoding()<<" "
311 <<"Direction " <<string->GetDecayDirection()<<G4endl;
312 #endif
313
314 //... random choice of string end to use for creating the hadron (decay)
315 G4int SideOfDecay = (G4UniformRand() < 0.5)? 1: -1;
316 if (SideOfDecay < 0)
317 {
318 string->SetLeftPartonStable();
319 } else
320 {
321 string->SetRightPartonStable();
322 }
323
324 G4ParticleDefinition *newStringEnd;
325 G4ParticleDefinition * HadronDefinition;
326
327 G4double StringMass=string->Mass();
328 G4double HadronMass=0.;
329 G4double MinMass =0.;
330 G4int Iter(0), MaxIter(1);
331
332 G4double ProbDqADq = GetDiquarkSuppress();
333 G4double ProbSaS = 1.0 - 2.0 * GetStrangeSuppress();
334
335 #ifdef debug_LUNDfragmentation
336 G4cout<<"StrMass DiquarkSuppression "<<StringMass<<" "<<GetDiquarkSuppress()<<G4endl;
337 #endif
338
339 G4int NumberOfpossibleBaryons = 2;
340
341 if(string->GetLeftParton()->GetParticleSubType() != "quark") NumberOfpossibleBaryons++;
342 if(string->GetRightParton()->GetParticleSubType() != "quark") NumberOfpossibleBaryons++;
343
344 G4double ActualProb = ProbDqADq ;
345 ActualProb *= (1.0-sqr(NumberOfpossibleBaryons*1400.0/StringMass));
346
347 SetDiquarkSuppression(ActualProb);
348
349 G4double Mth = 1250.0; // 2 Mk + Mpi
350 if( NumberOfpossibleBaryons == 3 ){Mth = 2520.0;} // Mlambda/Msigma + Mk + Mpi
351 else if( NumberOfpossibleBaryons == 4 ){Mth = 2380.0;} // 2 Mlambda/Msigma + Mk + Mpi
352 else {}
353
354 ActualProb = ProbSaS * (1.0 - G4Pow::GetInstance()->powA( Mth/StringMass , 4.0 ));
355 if( ActualProb < 0.0 ) ActualProb = 0.0;
356 SetStrangenessSuppression((1.0-ActualProb)/2.0);
357
358 #ifdef debug_LUNDfragmentation
359 G4cout<<"StrMass DiquarkSuppression Pow "<<StringMass<<" "<<GetDiquarkSuppress()<<G4endl;
360 #endif
361
362 do {
363
364 if (string->DecayIsQuark())
365 {
366 HadronDefinition= QuarkSplitup(string->GetDecayParton(), newStringEnd);
367 } else {
368 HadronDefinition= DiQuarkSplitup(string->GetDecayParton(), newStringEnd);
369 }
370
371 #ifdef debug_LUNDfragmentation
372 G4cout<<"Iter "<<Iter<<G4endl;
373 G4cout<<"The parton "<<string->GetDecayParton()->GetPDGEncoding()<<" "
374 <<" produces hadron "<<HadronDefinition->GetParticleName()
375 <<" and is transformed to "<<newStringEnd->GetPDGEncoding()<<G4endl;
376 G4cout<<"The side of the string decay Left/Right (1/-1) "<<SideOfDecay<<G4endl;
377 #endif
378 // create new String from old, ie. keep Left and Right order, but replace decay
379
380 if ( newString ) delete newString;
381
382 newString=new G4FragmentingString(*string,newStringEnd); // To store possible
383 // quark containt of new string
384 SetMinimalStringMass(newString); MinMass = MinimalStringMass;
385 HadronMass = HadronDefinition->GetPDGMass();
386
387 if (Iter >= MaxIter) { SetDiquarkSuppression(ProbDqADq);
388 SetStrangenessSuppression((1.0-ProbSaS)/2.0);
389 G4KineticTrack * Hadron =0;
390 return Hadron; }
391 Iter++;
392
393 } while (HadronMass + MinMass > StringMass);
394
395 SetDiquarkSuppression(ProbDqADq);
396 SetStrangenessSuppression((1.0-ProbSaS)/2.0);
397
398 #ifdef debug_LUNDfragmentation
399 G4cout<<"An attempt to determine its energy (SplitEandP)"<<G4endl;
400 #endif
401 G4LorentzVector* HadronMomentum=SplitEandP(HadronDefinition, string, newString);
402
403 delete newString; newString=0;
404
405 G4KineticTrack * Hadron =0;
406 if ( HadronMomentum != 0 ) {
407
408 #ifdef debug_LUNDfragmentation
409 G4cout<<"The attempt was successful"<<G4endl;
410 #endif
411 G4ThreeVector Pos;
412 Hadron = new G4KineticTrack(HadronDefinition, 0,Pos, *HadronMomentum);
413
414 if ( newString ) delete newString;
415
416 newString=new G4FragmentingString(*string,newStringEnd,
417 HadronMomentum);
418 delete HadronMomentum;
419 }
420 else
421 {
422 #ifdef debug_LUNDfragmentation
423 G4cout<<"The attempt was not successful !!!"<<G4endl;
424 #endif
425 }
426
427 #ifdef debug_LUNDfragmentation
428 G4cout<<"End SplitUP (G4VLongitudinalStringDecay) ====================="<<G4endl;
429 #endif
430
431 return Hadron;
432}

434//-----------------------------------------------------------------------------
435G4ParticleDefinition * G4LundStringFragmentation::DiQuarkSplitup(
436 G4ParticleDefinition* decay,
437 G4ParticleDefinition *&created)
438{
439 G4double StrSup=GetStrangeSuppress();
440 G4double ProbQQbar = (1.0 - 2.0*StrSup);
441
442 //... can Diquark break or not?
443 if (G4UniformRand() < DiquarkBreakProb ){
444
445 //... Diquark break
446 G4int stableQuarkEncoding = decay->GetPDGEncoding()/1000;
447 G4int decayQuarkEncoding = (decay->GetPDGEncoding()/100)%10;
448 if (G4UniformRand() < 0.5)
449 {
450 G4int Swap = stableQuarkEncoding;
451 stableQuarkEncoding = decayQuarkEncoding;
452 decayQuarkEncoding = Swap;
453 }
454
455 G4int IsParticle=(decayQuarkEncoding>0) ? -1 : +1; // if we have a quark, we need antiquark)
456
457 pDefPair QuarkPair = CreatePartonPair(IsParticle,false); // no diquarks wanted
458
459 //... Build new Diquark
460 G4int QuarkEncoding=QuarkPair.second->GetPDGEncoding();
461 G4int i10 = std::max(std::abs(QuarkEncoding), std::abs(stableQuarkEncoding));
462 G4int i20 = std::min(std::abs(QuarkEncoding), std::abs(stableQuarkEncoding));
463 G4int spin = (i10 != i20 && G4UniformRand() <= 0.5)? 1 : 3;
464 G4int NewDecayEncoding = -1*IsParticle*(i10 * 1000 + i20 * 100 + spin);
465 created = FindParticle(NewDecayEncoding);
466 G4ParticleDefinition * decayQuark=FindParticle(decayQuarkEncoding);
467 G4ParticleDefinition * had=hadronizer->Build(QuarkPair.first, decayQuark);
468 StrangeSuppress=StrSup;
469
470 return had;
471
472 } else {
473 //... Diquark does not break
474
475 G4int IsParticle=(decay->GetPDGEncoding()>0) ? +1 : -1; // if we have a diquark, we need quark
476
477 StrangeSuppress=(1.0 - ProbQQbar * 0.9)/2.0;
478 pDefPair QuarkPair = CreatePartonPair(IsParticle,false); // no diquarks wanted
479
480 created = QuarkPair.second;
481
482 G4ParticleDefinition * had=hadronizer->Build(QuarkPair.first, decay);
483 StrangeSuppress=StrSup;
484
485 return had;
486 }
487
488}

490//-----------------------------------------------------------------------------
491G4LorentzVector * G4LundStringFragmentation::SplitEandP(G4ParticleDefinition * pHadron,
492 G4FragmentingString * string,
493 G4FragmentingString * newString)
494{
495 G4LorentzVector String4Momentum=string->Get4Momentum();
496 G4double StringMT2=string->MassT2();
497 G4double StringMT =std::sqrt(StringMT2);
498
499 G4double HadronMass = pHadron->GetPDGMass();
500
501 SetMinimalStringMass(newString);
502
503 #ifdef debug_LUNDfragmentation
504 G4cout<<G4endl<<"Start LUND SplitEandP "<<G4endl;
505 G4cout<<"String 4 mom, String M and Mt "<<String4Momentum<<" "<<String4Momentum.mag()<<" "<<std::sqrt(StringMT2)<<G4endl;
506 G4cout<<"Hadron "<<pHadron->GetParticleName()<<G4endl;
507 G4cout<<"HadM MinimalStringMassLeft StringM hM+sM "<<HadronMass<<" "<<MinimalStringMass<<" "
508 <<String4Momentum.mag()<<" "<<HadronMass+MinimalStringMass<<G4endl;
509 #endif
510
511 if(HadronMass + MinimalStringMass > string->Mass())
512 {
513 #ifdef debug_LUNDfragmentation
514 G4cout<<"Mass of the string is not sufficient to produce the hadron!"<<G4endl;
515 #endif
516 return 0;
517 } // have to start all over!
518
519 String4Momentum.setPz(0.);
520 G4ThreeVector StringPt=String4Momentum.vect();
521
522 // calculate and assign hadron transverse momentum component HadronPx and HadronPy
523 G4ThreeVector HadronPt , RemSysPt;
524 G4double HadronMassT2, ResidualMassT2;
525 G4double HadronMt, Pt, Pt2, phi;
526 //... sample Pt of the hadron
527 G4int attempt=0;
528 do
529 {
530 attempt++; if(attempt > StringLoopInterrupt) {return 0;}
531
532 HadronMt = HadronMass - 300.0*G4Log(G4UniformRand());
533 Pt2 = sqr(HadronMt)-sqr(HadronMass); Pt=std::sqrt(Pt2);
534 phi = 2.*pi*G4UniformRand();
535 G4ThreeVector SampleQuarkPtw= G4ThreeVector(Pt * std::cos(phi),Pt * std::sin(phi),0);
536
537 HadronPt =SampleQuarkPtw + string->DecayPt();
538 HadronPt.setZ(0);
539 RemSysPt = StringPt - HadronPt;
540
541 HadronMassT2 = sqr(HadronMass) + HadronPt.mag2();
542 ResidualMassT2=sqr(MinimalStringMass) + RemSysPt.mag2();
543
544 } while (std::sqrt(HadronMassT2) + std::sqrt(ResidualMassT2) > StringMT);
545
546 //... sample z to define hadron longitudinal momentum and energy
547 //... but first check the available phase space
548
549 G4double Pz2 = (sqr(StringMT2 - HadronMassT2 - ResidualMassT2) -
550 4*HadronMassT2 * ResidualMassT2)/4./StringMT2;
551
552 if (Pz2 < 0 ) {return 0;} // have to start all over!
553
554 //... then compute allowed z region z_min <= z <= z_max
555
556 G4double Pz = std::sqrt(Pz2);
557 G4double zMin = (std::sqrt(HadronMassT2+Pz2) - Pz)/std::sqrt(StringMT2);
558 // G4double zMin = (std::sqrt(HadronMassT2+Pz2) - 0.)/std::sqrt(StringMT2); // For testing purposes
559 G4double zMax = (std::sqrt(HadronMassT2+Pz2) + Pz)/std::sqrt(StringMT2);
560
561 if (zMin >= zMax) return 0; // have to start all over!
562
563 G4double z = GetLightConeZ(zMin, zMax,
564 string->GetDecayParton()->GetPDGEncoding(), pHadron,
565 HadronPt.x(), HadronPt.y());
566
567 //... now compute hadron longitudinal momentum and energy
568 // longitudinal hadron momentum component HadronPz
569
570 HadronPt.setZ(0.5* string->GetDecayDirection() *
571 (z * string->LightConeDecay() -
572 HadronMassT2/(z * string->LightConeDecay())));
573 G4double HadronE = 0.5* (z * string->LightConeDecay() +
574 HadronMassT2/(z * string->LightConeDecay()));
575
576 G4LorentzVector * a4Momentum= new G4LorentzVector(HadronPt,HadronE);
577
578 #ifdef debug_LUNDfragmentation
579 G4cout<<G4endl<<" string->GetDecayDirection() "<<string->GetDecayDirection()<<G4endl<<G4endl;
580 G4cout<<"string->LightConeDecay() "<<string->LightConeDecay()<<G4endl;
581 G4cout<<"HadronPt,HadronE "<<HadronPt<<" "<<HadronE<<G4endl;
582 G4cout<<"String4Momentum "<<String4Momentum<<G4endl;
583 G4cout<<"Out of LUND SplitEandP "<<G4endl<<G4endl;
584 #endif
585
586 return a4Momentum;
587}

589//-----------------------------------------------------------------------------------------
590G4double G4LundStringFragmentation::GetLightConeZ(G4double zmin, G4double zmax,
591 G4int PDGEncodingOfDecayParton,
592 G4ParticleDefinition* pHadron,
593 G4double Px, G4double Py)
594{
595 G4double Mass = pHadron->GetPDGMass();
596 G4int HadronEncoding=std::abs(pHadron->GetPDGEncoding());
597
598 G4double Mt2 = Px*Px + Py*Py + Mass*Mass;
599
600 G4double Alund, Blund;
601 G4double zOfMaxyf(0.), maxYf(1.), z(0.), yf(1.);
602
603 if (!((std::abs(PDGEncodingOfDecayParton) > 1000) && (HadronEncoding > 1000)) )
604 { // ---------------- Quark fragmentation and qq-> meson ----------------------
605 Alund=1.;
606 Blund=0.7/GeV/GeV;
607
608 G4double BMt2 = Blund*Mt2;
609 if (Alund == 1.0) {
610 zOfMaxyf=BMt2/(Blund*Mt2 + 1.);}
611 else {
612 zOfMaxyf = ((1.0+BMt2) - std::sqrt(sqr(1.0-BMt2) + 4.0*BMt2*Alund))/2.0/(1.-Alund);
613 }
614
615 if (zOfMaxyf < zmin) {zOfMaxyf=zmin;}
616 if (zOfMaxyf > zmax) {zOfMaxyf=zmax;}
617 maxYf=(1-zOfMaxyf)/zOfMaxyf * G4Exp(-Blund*Mt2/zOfMaxyf);
618
619 const G4int maxNumberOfLoops = 1000;
620 G4int loopCounter = 0;
621 do
622 {
623 z = zmin + G4UniformRand()*(zmax-zmin);
624 //yf = (1-z)/z * G4Exp(-Blund*Mt2/z);
625 yf = G4Pow::GetInstance()->powA(1.0-z,Alund)/z*G4Exp(-BMt2/z);
626 }
627 while ( (G4UniformRand()*maxYf > yf) && ++loopCounter < maxNumberOfLoops );
628 if ( loopCounter >= maxNumberOfLoops ) {
629 z = 0.5*(zmin + zmax); // Just a value between zmin and zmax, no physics considerations at all!
630 }
631 return z;
632 }
633
634 if(std::abs(PDGEncodingOfDecayParton) > 1000)
635 {
636 G4double an = 2.5;
637 an +=(sqr(Px)+sqr(Py))/sqr(GeV)-0.5;
638 z=zmin + (zmax-zmin)*G4Pow::GetInstance()->powA(G4UniformRand(),1./an);
639 }
640
641 return z;
642}

1183void G4LundStringFragmentation::Sample4Momentum(G4LorentzVector* Mom, G4double Mass,
1184 G4LorentzVector* AntiMom, G4double AntiMass,
1185 G4double InitialMass)
1186{
1187 // ------ Sampling of momenta of 2 last produced hadrons --------------------
1188 G4ThreeVector Pt;
1189 G4double MassMt, AntiMassMt;
1190 G4double AvailablePz, AvailablePz2;
1191 #ifdef debug_LUNDfragmentation
1192 G4cout<<"Sampling of momenta of 2 last produced hadrons ----------------"<<G4endl;
1193 G4cout<<"Init Mass "<<InitialMass<<" FirstM "<<Mass<<" SecondM "<<AntiMass<<" ProbIsotropy "<<G4endl;
1194 #endif
1195
1196 G4double r_val = sqr(InitialMass*InitialMass - Mass*Mass - AntiMass*AntiMass) -
1197 sqr(2.*Mass*AntiMass);
1198 G4double Pabs = (r_val > 0.)? std::sqrt(r_val)/(2.*InitialMass) : 0;
1199
1200 const G4int maxNumberOfLoops = 1000;
1201 G4double SigmaQTw=SigmaQT;
1202 if (Mass > 930. || AntiMass > 930.) SigmaQT *=(1.0-0.55*sqr((Mass+AntiMass)/InitialMass));
1203
1204 G4int loopCounter = 0;
1205 do
1206 {
1207 Pt=SampleQuarkPt(Pabs); Pt.setZ(0); G4double Pt2=Pt.mag2();
1208 MassMt = std::sqrt( Mass * Mass + Pt2);
1209 AntiMassMt= std::sqrt(AntiMass * AntiMass + Pt2);
1210 }
