98inline G4Parton* G4QGSMSplitableHadron::GetNextParton()
99{
100 if(Color.size()==0) return 0;
101
102 G4Parton * result = Color.operator[](iP);
103 iP++; if(iP == Color.size()) iP=0;
104 return result;
105}
106
107inline G4Parton* G4QGSMSplitableHadron::GetNextAntiParton()
108{
109 if(AntiColor.size() == 0) return 0;
110
111 G4Parton * result = AntiColor.operator[](iAP);
112 iAP++; if(iAP == AntiColor.size()) iAP=0;
113 return result;
114}
115

117{PDGcode++;}
118inline void G4QGSMSplitableHadron::SetSecondParton(G4int PDGcode)
119{PDGcode++;}
