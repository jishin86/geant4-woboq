26#ifndef G4PartonPair_h
27#define G4PartonPair_h 1
28
29#include "globals.hh"
30#include "G4ThreeVector.hh"
31#include "G4LorentzVector.hh"
32#include "G4Parton.hh"
33#include "G4PartonVector.hh"
34
35class G4PartonPair
36{
37 public:
38 enum {
39 DIFFRACTIVE = 1,
40 SOFT = 2,
41 HARD = 3
42 };
43 enum {
44 PROJECTILE = 1,
45 TARGET = -1
46 };
47 public:
48 G4PartonPair(G4Parton* P1, G4Parton* P2, G4int Type, G4int Direction);
49 ~G4PartonPair();
50
51 private:
52 G4PartonPair(const G4PartonPair &right);
53 G4bool operator==(const G4PartonPair &right) const;
54 G4bool operator!=(const G4PartonPair &right) const;
55
56 public:
57 void SetPartons(G4Parton* P1, G4Parton* P2);
58 void SetCollisionType(G4int Type);
59 G4int GetCollisionType();
60 G4Parton* GetParton1(void);
61 G4Parton* GetParton2(void);
62 G4int GetDirection();
63
64 private:
65 G4Parton* Parton1;
66 G4Parton* Parton2;
67 G4int CollisionType;
68 G4int Direction;
69};
70
71inline G4Parton* G4PartonPair::GetParton1(void)
72{
73 return Parton1;
74}
75
76inline G4Parton* G4PartonPair::GetParton2(void)
77{
78 return Parton2;
79}
80
81inline void G4PartonPair::SetCollisionType(G4int Type)
82{
83 CollisionType = Type;
84}
85
86inline G4int G4PartonPair::GetCollisionType()
87{
88 return CollisionType;
89}
90
91inline G4int G4PartonPair::GetDirection()
92{
93 return Direction;
94}
