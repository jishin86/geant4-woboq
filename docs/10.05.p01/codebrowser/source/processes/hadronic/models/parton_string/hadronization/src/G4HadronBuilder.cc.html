42G4HadronBuilder::G4HadronBuilder(G4double mesonMix, G4double barionMix,
43 std::vector<double> scalarMesonMix,
44 std::vector<double> vectorMesonMix)
45{
46 mesonSpinMix=mesonMix;
47 barionSpinMix=barionMix;
48 scalarMesonMixings=scalarMesonMix;
49 vectorMesonMixings=vectorMesonMix;
50}
51
52G4ParticleDefinition * G4HadronBuilder::Build(G4ParticleDefinition * black, G4ParticleDefinition * white)
53{
54
55 if (black->GetParticleSubType()== "di_quark" || white->GetParticleSubType()== "di_quark" ) {
56 // Barion
57 Spin spin = (G4UniformRand() < barionSpinMix) ? SpinHalf : SpinThreeHalf;
58 return Barion(black,white,spin);
59
60 } else {
61 // Meson
62 Spin spin = (G4UniformRand() < mesonSpinMix) ? SpinZero : SpinOne;
63 return Meson(black,white,spin);
64 }
65}

69G4ParticleDefinition * G4HadronBuilder::BuildLowSpin(G4ParticleDefinition * black, G4ParticleDefinition * white)
70{
71 if ( black->GetParticleSubType()== "quark" && white->GetParticleSubType()== "quark" ) {
72 return Meson(black,white, SpinZero);
73 } else {
74 // will return a SpinThreeHalf Barion if all quarks the same
75 return Barion(black,white, SpinHalf);
76 }
77}

81G4ParticleDefinition * G4HadronBuilder::BuildHighSpin(G4ParticleDefinition * black, G4ParticleDefinition * white)
82{
83 if ( black->GetParticleSubType()== "quark" && white->GetParticleSubType()== "quark" ) {
84 return Meson(black,white, SpinOne);
85 } else {
86 return Barion(black,white,SpinThreeHalf);
87 }
88}

92G4ParticleDefinition * G4HadronBuilder::Meson(G4ParticleDefinition * black,
93 G4ParticleDefinition * white, Spin theSpin)
94{
95 #ifdef G4VERBOSE
96 // Verify Input Charge
97 G4double charge = black->GetPDGCharge() + white->GetPDGCharge();
98 if (std::abs(charge) > 2 || std::abs(3.*charge - 3*G4int(charge*1.001)) > perCent ) // 1.001 to avoid int(.9999) -> 0
99 {
100 G4cerr << " G4HadronBuilder::Build()" << G4endl;
101 G4cerr << " Invalid total charge found for on input: "
102 << charge<< G4endl;
103 G4cerr << " PGDcode input quark1/quark2 : " <<
104 black->GetPDGEncoding() << " / "<<
105 white->GetPDGEncoding() << G4endl;
106 G4cerr << G4endl;
107 }
108 #endif
109
110 G4int id1= black->GetPDGEncoding();
111 G4int id2= white->GetPDGEncoding();
112 // G4int ifl1= std::max(std::abs(id1), std::abs(id2));
113 if ( std::abs(id1) < std::abs(id2) )
114 {
115 G4int xchg = id1;
116 id1 = id2;
117 id2 = xchg;
118 }
119
120 if (std::abs(id1) > 3 )
121 throw G4HadronicException(__FILE__, __LINE__, "G4HadronBuilder::Meson : Illegal Quark content as input");
122
123 G4int PDGEncoding=0;
124
125 if (id1 + id2 == 0) {
126 G4double rmix = G4UniformRand();
127 G4int imix = 2*std::abs(id1) - 1;
128 if (theSpin == SpinZero) {
129 PDGEncoding = 110*(1 + (G4int)(rmix + scalarMesonMixings[imix - 1])
130 + (G4int)(rmix + scalarMesonMixings[imix])
131 ) + theSpin;
132 } else {
133 PDGEncoding = 110*(1 + (G4int)(rmix + vectorMesonMixings[imix - 1])
134 + (G4int)(rmix + vectorMesonMixings[imix])
135 ) + theSpin;
136 }
137 } else {
138 PDGEncoding = 100 * std::abs(id1) + 10 * std::abs(id2) + theSpin;
139 G4bool IsUp = (std::abs(id1)&1) == 0; // quark 1 up type quark (u or c)
140 G4bool IsAnti = id1 < 0; // quark 1 is antiquark?
141 if( (IsUp && IsAnti ) || (!IsUp && !IsAnti ) )
142 PDGEncoding = - PDGEncoding;
143 }
144
145 G4ParticleDefinition * MesonDef=
146 G4ParticleTable::GetParticleTable()->FindParticle(PDGEncoding);
147
148 #ifdef G4VERBOSE
149 if (MesonDef == 0 ) {
150 G4cerr << " G4HadronBuilder - Warning: No particle for PDGcode= "
151 << PDGEncoding << G4endl;
152 } else if ( ( black->GetPDGCharge() + white->GetPDGCharge()
153 - MesonDef->GetPDGCharge() ) > perCent ) {
154 G4cerr << " G4HadronBuilder - Warning: Incorrect Charge : "
155 << " Quark1/2 = "
156 << black->GetParticleName() << " / "
157 << white->GetParticleName()
158 << " resulting Hadron " << MesonDef->GetParticleName()
159 << G4endl;
160 }
161 #endif
162
163 return MesonDef;
164}

167G4ParticleDefinition * G4HadronBuilder::Barion(G4ParticleDefinition * black,
168 G4ParticleDefinition * white,Spin theSpin)
169{
170 #ifdef G4VERBOSE
171 // Verify Input Charge
172 G4double charge = black->GetPDGCharge() + white->GetPDGCharge();
173 if (std::abs(charge) > 2 || std::abs(3.*charge - 3*G4int(charge*1.001)) > perCent )
174 {
175 G4cerr << " G4HadronBuilder::Build()" << G4endl;
176 G4cerr << " Invalid total charge found for on input: "
177 << charge<< G4endl;
178 G4cerr << " PGDcode input quark1/quark2 : " <<
179 black->GetPDGEncoding() << " / "<<
180 white->GetPDGEncoding() << G4endl;
181 G4cerr << G4endl;
182 }
183 #endif
184
185 G4int id1= black->GetPDGEncoding();
186 G4int id2= white->GetPDGEncoding();
187 if ( std::abs(id1) < std::abs(id2) )
188 {
189 G4int xchg = id1;
190 id1 = id2;
191 id2 = xchg;
192 }
193
194 if (std::abs(id1) < 1000 || std::abs(id2) > 3 )
195 throw G4HadronicException(__FILE__, __LINE__, "G4HadronBuilder::Barion: Illegal quark content as input");
196
197 G4int ifl1= std::abs(id1)/1000;
198 G4int ifl2 = (std::abs(id1) - ifl1 * 1000)/100;
199 G4int diquarkSpin = std::abs(id1)%10;
200 G4int ifl3 = id2;
201 if (id1 < 0)
202 {
203 ifl1 = - ifl1;
204 ifl2 = - ifl2;
205 }
206 //... Construct barion, distinguish Lambda and Sigma barions.
207 G4int kfla = std::abs(ifl1);
208 G4int kflb = std::abs(ifl2);
209 G4int kflc = std::abs(ifl3);
210
211 G4int kfld = std::max(kfla,kflb);
212 kfld = std::max(kfld,kflc);
213 G4int kflf = std::min(kfla,kflb);
214 kflf = std::min(kflf,kflc);
215
216 G4int kfle = kfla + kflb + kflc - kfld - kflf;
217
218 //... barion with content uuu or ddd or sss has always spin = 3/2
219 theSpin = (kfla == kflb && kflb == kflc)? SpinThreeHalf : theSpin;
220
221 G4int kfll = 0;
222 if (theSpin == SpinHalf && kfld > kfle && kfle > kflf) {
223 // Spin J=1/2 and all three quarks different
224 // Two states exist: (uds -> lambda or sigma0)
225 // - lambda: s(ud)0 s : 3122; ie. reverse the two lighter quarks
226 // - sigma0: s(ud)1 s : 3212
227 if(diquarkSpin == 1 ) {
228 if ( kfla == kfld) { // heaviest quark in diquark
229 kfll = 1;
230 } else {
231 kfll = (G4int)(0.25 + G4UniformRand());
232 }
233 }
234 if(diquarkSpin == 3 && kfla != kfld)
235 kfll = (G4int)(0.75 + G4UniformRand());
236 }
237
238 G4int PDGEncoding;
239 if (kfll == 1)
240 PDGEncoding = 1000 * kfld + 100 * kflf + 10 * kfle + theSpin;
241 else
242 PDGEncoding = 1000 * kfld + 100 * kfle + 10 * kflf + theSpin;
243
244 if (id1 < 0)
245 PDGEncoding = -PDGEncoding;
246
247 G4ParticleDefinition * BarionDef=
248 G4ParticleTable::GetParticleTable()->FindParticle(PDGEncoding);
249
250 #ifdef G4VERBOSE
251 if (BarionDef == 0 ) {
252 G4cerr << " G4HadronBuilder - Warning: No particle for PDGcode= "
253 << PDGEncoding << G4endl;
254 } else if ( ( black->GetPDGCharge() + white->GetPDGCharge()
255 - BarionDef->GetPDGCharge() ) > perCent ) {
256 G4cerr << " G4HadronBuilder - Warning: Incorrect Charge : "
257 << " DiQuark/Quark = "
258 << black->GetParticleName() << " / "
259 << white->GetParticleName()
260 << " resulting Hadron " << BarionDef->GetParticleName()
261 << G4endl;
262 }
263 #endif
264
265 return BarionDef;
266}
