469void G4FTFParameters::InitForInteraction( const G4ParticleDefinition* particle,
470 G4int theA, G4int theZ, G4double PlabPerParticle )
471{
472 Reset();
473
474 G4int ProjectilePDGcode = particle->GetPDGEncoding();
475 G4int ProjectileabsPDGcode = std::abs( ProjectilePDGcode );
476 G4double ProjectileMass = particle->GetPDGMass();
477 G4double ProjectileMass2 = ProjectileMass * ProjectileMass;
478
479 G4int ProjectileBaryonNumber( 0 ), AbsProjectileBaryonNumber( 0 ), AbsProjectileCharge( 0 );
480 G4bool ProjectileIsNucleus = false;
481
482 if ( std::abs( particle->GetBaryonNumber() ) > 1 ) { // The projectile is a nucleus
483 ProjectileIsNucleus = true;
484 ProjectileBaryonNumber = particle->GetBaryonNumber();
485 AbsProjectileBaryonNumber = std::abs( ProjectileBaryonNumber );
486 AbsProjectileCharge = G4int( particle->GetPDGCharge() );
487 if ( ProjectileBaryonNumber > 1 ) {
488 ProjectilePDGcode = 2212; ProjectileabsPDGcode = 2212; // Proton
489 } else {
490 ProjectilePDGcode = -2212; ProjectileabsPDGcode = 2212; // Anti-Proton
491 }
492 ProjectileMass = G4Proton::Proton()->GetPDGMass();
493 ProjectileMass2 = sqr( ProjectileMass );
494 }
495
496 G4double TargetMass = G4Proton::Proton()->GetPDGMass();
497 G4double TargetMass2 = TargetMass * TargetMass;
498
499 G4double Plab = PlabPerParticle;
500 G4double Elab = std::sqrt( Plab*Plab + ProjectileMass2 );
501 G4double KineticEnergy = Elab - ProjectileMass;
502
503 G4double S = ProjectileMass2 + TargetMass2 + 2.0*TargetMass*Elab;
504
505 #ifdef debugFTFparams
506 G4cout << "--------- FTF Parameters --------------" << G4endl << "Proj Plab "
507 << ProjectilePDGcode << " " << Plab << G4endl << "Mass KinE " << ProjectileMass
508 << " " << KineticEnergy << G4endl << " A Z " << theA << " " << theZ << G4endl;
509 #endif
510
511 G4double Ylab, Xtotal, Xelastic, Xannihilation;
512 G4int NumberOfTargetNucleons;
513
514 Ylab = 0.5 * G4Log( (Elab + Plab)/(Elab - Plab) );
515
516 G4double ECMSsqr = S/GeV/GeV;
517 G4double SqrtS = std::sqrt( S )/GeV;
518
519 #ifdef debugFTFparams
520 G4cout << "Sqrt(s) " << SqrtS << G4endl;
521 #endif
522
523 TargetMass /= GeV; TargetMass2 /= (GeV*GeV);
524 ProjectileMass /= GeV; ProjectileMass2 /= (GeV*GeV);
525
526 Plab /= GeV;
527 G4double Xftf = 0.0;
