50class G4FTFParticipants : public G4VParticipants {
51 public:
52 G4FTFParticipants();
53 const G4FTFParticipants& operator=( const G4FTFParticipants& right );
54 ~G4FTFParticipants();
55 G4bool operator==( const G4FTFParticipants& right ) const;
56 G4bool operator!=( const G4FTFParticipants& right ) const;
57
58 void GetList( const G4ReactionProduct& thePrimary, G4FTFParameters* theParameters );
59 void StartLoop();
60 G4bool Next();
61 void SortInteractionsIncT();
62 void ShiftInteractionTime();
63 G4InteractionContent& GetInteraction();
64 void Clean();
65 std::vector< G4InteractionContent* > theInteractions;
66
67 private:
68 G4FTFParticipants( const G4FTFParticipants& right );
69 G4int currentInteraction;
70};
71
72
73inline void G4FTFParticipants::StartLoop() {
74 currentInteraction = -1;
75}
76
77
78inline G4bool G4FTFParticipants::Next() {
79 return ++currentInteraction < static_cast< G4int >( theInteractions.size() );
80}
81
82
83inline G4InteractionContent& G4FTFParticipants::GetInteraction() {
84 return *theInteractions[ currentInteraction ];
85}
