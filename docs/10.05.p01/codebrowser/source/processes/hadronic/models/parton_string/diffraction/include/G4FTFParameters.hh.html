52 // parameters of excitation

54 // Proc=0 --> Qexchg w/o excitation

57 double GetProc0B1() const { return fProc0B1; }

63 //
64 // Proc=1 --> Qexchg w/excitation
65 //
66 // Proc=2 & Proc=3 for the case ( AbsProjectileBaryonNumber > 1 || NumberOfTargetNucleons > 1 )

69 bool IsProjDiffDissociation() const { return fProjDiffDissociation; }
70 bool IsTgtDiffDissociation() const { return fTgtDiffDissociation; }
71 //

132 // parameters of excitation
133 //
134 //
135 // these are for Inelastic interactions, i.e. Xinelastic=(Xtotal-Xelastix)>0.
136 // for elastic, all the A's & B's, Atop & Ymin are zeros
137 // general formula: Pp = A1*exp(B1*Y) + A2*exp(B2*Y) + A3
138 // but if Y<Ymin, then Pp=max(0.,Atop)
139 // for details, see also G4FTFParameters::GetProcProb( ProcN, y )
140 //
141 // Proc=0 --> Qexchg w/o excitation
142 double fProc0A1; // D=13.71
143 double fProc0B1; // D=1.75
144 double fProc0A2; // D=-30.69 (or -214.5 as in Doc ?)
145 double fProc0B2; // D=3. ( or 4. as in Doc ?)
146 double fProc0A3; // D=0.
147 double fProc0Atop; // D=1. ( or 0.5 as in Doc ?)
148 double fProc0Ymin; // D=0.93 (or 1.1 as in Doc ?)
149 // Proc=1 --> Qexchg w/excitation
150 double fProc1A1; // D=25.
151 double fProc1B1; // D=1.
152 double fProc1A2; // D=-50.34
153 double fProc1B2; // D=1.5
154 double fProc1A3; // D=0.
155 double fProc1Atop; // D=0.
156 double fProc1Ymin; // D=1.4
157 //
158 // NOTE: Proc #2 & 3 are projectile & target diffraction
159 // they have more complex definition of A1 & A2
160 // (see around line 540 or so)
161 // SetParams( 2, 6.0/Xinel, 0.0 ,-6.0/Xinel*16.28, 3.0 , 0.0, 0.0 , 0.93);// Projectile diffraction
162 // SetParams( 3, 6.0/Xinel, 0.0 ,-6.0/Xinel*16.28, 3.0 , 0.0, 0.0 , 0.93);// Target diffraction
163 //
164 // Also, for ( AbsProjectileBaryonNumber > 1 || NumberOfTargetNucleons > 1 )
165 // projectile and/or target diffraction (dissociation) may be switched ON/OFF
166 bool fProjDiffDissociation;
167 bool fTgtDiffDissociation;
168 //
169 // Proc=4 --> Qexchg w/additional multiplier in excitation
170 double fProc4A1; // D=0.6 (or 1. as in Doc ?)
171 double fProc4B1; // D=0.
172 double fProc4A2; // D=-1.2 (or -2.01 as in Doc ?)
173 double fProc4B2; // D=0.5
174 double fProc4A3; // D=0.
175 double fProc4Atop; // D=0.
176 double fProc4Ymin; // D=1.4
177 //
178 // parameters of participating baryon excitation
179 //
180 double fDeltaProbAtQuarkExchange; // D=0.
181 double fProbOfSameQuarkExchange; // D=0. if A<=26, otherwise D=1.
182 double fProjMinDiffMass; // projectile, D=1.16GeV
183 double fProjMinNonDiffMass; // projectile, D=1.16GeV
184 double fTgtMinDiffMass; // target, D=1.16GeV
185 double fTgtMinNonDiffMass; // target, D=1.16GeV
186 double fAveragePt2; // D=0.3GeV**2 ( or 0.15 as in the Doc ???)
187 double fProbLogDistrPrD; // D=0.6 (or 0.3 ???)
188 double fProbLogDistr; // D=0.6 (or 0.3 ???)
189
190 // parameters of nuclear distruction
191 //
192 // NOTE (JVY): there're 3 cases here:
193 // * baryon projectile
194 // * anti-baryon projectile
195 // * meson projectile
196 //
197 // double fBaryonMaxNumberOfCollisions; // D=2.
198 // void SetBaryonProbOfInteraction( const double ); // ??? this is prob. of inelastic interaction
199 // that is set internally based on certain conditions...
200 // general (i.e. for used for baryons,anti-baryons, and mesons)
201 // NOTE: these parameters have stayed THE SAME for quite a while
202 double fNuclearProjDestructP1; // D=0.00481 in 10.3.ref04 !!!
203 // BUT !!! In 10.3.ref04 as well as in 10.2-seriesit's multiplied of AbsProjectileBaryonNumber
204 // which somehow is 0 for the proton projectile (see in 10.3.ref04 around lines 130-140 In G4FTFParameters.cc).
205 // For the target destr. it's multipled by the number of target nucleons (12 for Carbon).
206 // In 10.3.p01 it's set to 1. FLAT OUT for both projectile & target, no multiplications, etc.
207 // Now, make default at 1.
208 bool fNuclearProjDestructP1_NBRNDEP;
209 double fNuclearTgtDestructP1; // Make D=1. as in 10.3.p01
210 bool fNuclearTgtDestructP1_ADEP;
211 double fNuclearProjDestructP2; // D=4.0
212 double fNuclearProjDestructP3; // D=2.1
213 double fNuclearTgtDestructP2; // D=4.0
214 double fNuclearTgtDestructP3; // D=2.1
215 //
216 double fPt2NuclearDestructP1; // D=0.035
217 double fPt2NuclearDestructP2; // D=0.04
218 double fPt2NuclearDestructP3; // D=4.0
219 double fPt2NuclearDestructP4; // D=2.5
220 // baryons
221 double fR2ofNuclearDestruct; // D=1.5*fermi*fermi
222 double fExciEnergyPerWoundedNucleon; // D=40MeV
223 double fDofNuclearDestruct; // D=0.3
224 // NOTE: this parameter has changed from 1. to 9. between 10.2 and 10.4.ref04 !!!
225 double fMaxPt2ofNuclearDestruct; // D=9GeV**2
226
227 private:
228
229 void Reset();
230};
