30#ifndef G4ParticleHPDataPoint_h
31#define G4ParticleHPDataPoint_h 1
32
33#include "globals.hh"
34
35class G4ParticleHPDataPoint
36{
37 public:
38
39 G4ParticleHPDataPoint(){energy = 0; xSec = 0;}
40 G4ParticleHPDataPoint(G4double e, G4double x){ energy = e; xSec = x;}
41
42 void operator= (const G4ParticleHPDataPoint & aSet)
43 {
44 if(&aSet!=this)
45 {
46 energy = aSet.GetEnergy();
47 xSec = aSet.GetXsection();
48 }
49 }
50
51// ~G4ParticleHPDataPoint(){}
52
53 inline G4double GetEnergy() const {return energy;}
54 inline G4double GetXsection() const {return xSec;}
55
56 inline void SetEnergy(G4double e) {energy = e;}
57 inline void SetXsection(G4double x){xSec = x;}
58
59 inline G4double GetX() const {return energy;}
60 inline G4double GetY() const {return xSec;}
61
62 inline void SetX(G4double e) {energy = e;}
63 inline void SetY(G4double x) {xSec = x;}
64
65 inline void SetData(G4double e, G4double x) {energy = e; xSec = x;}
66
67 private:
68
69 G4double energy;
70 G4double xSec;
71};

73#endif
