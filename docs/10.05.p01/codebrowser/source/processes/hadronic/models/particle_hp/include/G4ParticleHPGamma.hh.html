29#ifndef G4ParticleHPGamma_h
30#define G4ParticleHPGamma_h 1
31
32#include "globals.hh"
33#include "G4ios.hh"
34#include <fstream>
35#include "G4DynamicParticleVector.hh"
36#include "G4DynamicParticle.hh"
37#include "G4Gamma.hh"
38#include "G4ParticleHPLevel.hh"
39
40class G4ParticleHPGamma
41{
42 public:
43
44 G4ParticleHPGamma();
45 ~G4ParticleHPGamma();
46
47 G4bool Init(std::istream & aDataFile);
48
49 inline void SetNext(G4ParticleHPLevel * aLevel)
50 {
51 next = aLevel;
52 }
53
54 G4DynamicParticleVector * GetDecayGammas()
55 {
56 G4DynamicParticleVector * theResult;
57 if(next == 0)
58 {
59 theResult = new G4DynamicParticleVector;
60 }
61 else
62 {
63 theResult = next->GetDecayGammas();
64 }
65 G4DynamicParticle * theNew = new G4DynamicParticle;
66 theNew->SetDefinition(G4Gamma::Gamma());
67 theNew->SetKineticEnergy(gammaEnergy);
68 theResult->push_back(theNew);
69 return theResult;
70 }
71
72 inline G4double GetLevelEnergy()
73 {
74 return levelEnergy;
75 }
76
77 inline G4double GetGammaEnergy()
78 {
79 return gammaEnergy;
80 }
81
82 inline G4double GetWeight()
83 {
84 return probability;
85 }
86
87 private:
88
89 G4double levelEnergy;
90 G4double gammaEnergy;
91 G4double probability;
92
93 G4ParticleHPLevel * next;
94 static G4ThreadLocal int instancecount;
95};
96
97#endif
