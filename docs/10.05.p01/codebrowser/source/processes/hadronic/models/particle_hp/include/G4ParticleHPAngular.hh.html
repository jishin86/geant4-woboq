32#ifndef G4ParticleHPAngular_h
33#define G4ParticleHPAngular_h 1
34
35#include <fstream>
36
37#include "globals.hh"
38#include "G4ios.hh"
39#include "G4ReactionProduct.hh"
40#include "Randomize.hh"
41#include "G4ParticleHPLegendreStore.hh"
42#include "G4ParticleHPPartial.hh"
43#include "G4Cache.hh"
44
45class G4ParticleHPAngular
46{
47 struct toBeCached
48 {
49 const G4ReactionProduct* theProjectileRP;
50 const G4ReactionProduct* theTarget;
51 toBeCached() : theProjectileRP(0),theTarget(0) {}
52 };
53
54public:
55
56 G4ParticleHPAngular()
57 {
58 theIsoFlag = true;
59 theCoefficients = 0;
60 theProbArray = 0;
61 toBeCached val;
62 fCache.Put( val );
63
64 theAngularDistributionType = 0;
65 frameFlag = 0;
66 targetMass = 0.0;
67 }
68
69 ~G4ParticleHPAngular()
70 {
71 delete theCoefficients;
72 delete theProbArray;
73 }
74
75 void Init(std::istream & aDataFile);
76
77 void SampleAndUpdate(G4ReactionProduct & anIncidentParticle);
78
79 void SetTarget(const G4ReactionProduct & aTarget)
80 {
81 fCache.Get().theTarget = &aTarget;
82 }
83
84 void SetProjectileRP(const G4ReactionProduct & anIncidentParticleRP)
85 {
86 fCache.Get().theProjectileRP = &anIncidentParticleRP;
87 }
88
89 inline G4double GetTargetMass()
90 {
91 return targetMass;
92 }
93
94private:
95
96 // the type of distribution; currently
97 // isotropic (0),
98 // and legendre representation (1)
99 // probability distribution (2)
100 // are supported
101
102 G4int theAngularDistributionType;
103 G4int frameFlag; // 1=Lab, 2=CMS
104
105 G4bool theIsoFlag; // isotropic or not?
106
107 G4ParticleHPLegendreStore * theCoefficients; // the legendre coefficients
108
109 G4ParticleHPPartial * theProbArray;
110 // the probability array p,costh for energy
111
112 G4double targetMass;
113
114 G4Cache<toBeCached> fCache;
115};
116
117#endif
