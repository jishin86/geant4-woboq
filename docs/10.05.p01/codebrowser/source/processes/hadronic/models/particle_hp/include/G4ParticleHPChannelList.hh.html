34#ifndef G4ParticleHPChannelList_h
35#define G4ParticleHPChannelList_h 1
36
37#include "globals.hh"
38#include "G4ParticleHPChannel.hh"
39#include "G4StableIsotopes.hh"
40
41class G4Element;
42class G4HadFinalState;
43class G4HadProjectile;
44class G4ParticleHPFinalState;
45class G4ParticleDefinition;
46
47class G4ParticleHPChannelList
48{
49
50 public:
51
52 G4ParticleHPChannelList(G4int n, G4ParticleDefinition* projectile);
53
54 G4ParticleHPChannelList();
55
56 void Init(G4int n);
57
58 ~G4ParticleHPChannelList();
59
60 G4HadFinalState * ApplyYourself(const G4Element * theElement, const G4HadProjectile & aTrack);
61
62 void Init(G4Element * anElement, const G4String & dirName, G4ParticleDefinition* projectile);
63
64 void Register(G4ParticleHPFinalState *theFS, const G4String &aName);
65
66 inline G4double GetXsec(G4double anEnergy)
67 {
68 G4double result=0;
69 G4int i;
70 for(i=0; i<nChannels; i++)
71 {
72 result+=std::max(0., theChannels[i]->GetXsec(anEnergy));
73 }
74 return result;
75 }
76
77 inline G4int GetNumberOfChannels() { return nChannels; }
78
79 inline G4bool HasDataInAnyFinalState()
80 {
81 G4bool result = false;
82 G4int i;
83 for(i=0; i<nChannels; i++)
84 {
85 if(theChannels[i]->HasDataInAnyFinalState()) result = true;
86 }
87 return result;
88 }
89 inline void RestartRegistration()
90 {
91 allChannelsCreated = true;
92 theInitCount = 0;
93 }
94
95 void DumpInfo();
96
97
98private:
99
100 static G4ThreadLocal G4int trycounter;
101 G4ParticleHPChannel ** theChannels;
102 G4int nChannels;
103 G4String theDir;
104 G4Element * theElement;
105
106 G4bool allChannelsCreated;
107 G4int theInitCount;
108
109 G4StableIsotopes theStableOnes;
110
111 G4ParticleDefinition* theProjectile;
112
113 G4HadFinalState unChanged;
114
115};
116
117#endif
