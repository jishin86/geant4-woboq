57class G4ParticleHPPhotonDist
58{
59public:
60
61 G4ParticleHPPhotonDist()
62 : repFlag( 0 )
63 , targetMass( 0.0 )
64 , nDiscrete( 0 )
65 , isoFlag( 0 )
66 , tabulationType( 0 )
67 , nDiscrete2( 0 )
68 , nIso( 0 )
69 , nPartials( 0 )
70 , theInternalConversionFlag( 0 )
71 , nGammaEnergies( 0 )
72 , theBaseEnergy( 0.0 )
73 {
74
75 disType = 0;
76 energy = 0;
77 theYield = 0;
78 thePartialXsec = 0;
79 theReactionXsec = 0;
80 isPrimary = 0;
81 theShells = 0;
82 theGammas = 0;
83 nNeu = 0;
84 theLegendre = 0;
85 theAngular = 0;
86 distribution = 0;
87 probs = 0;
88 partials = 0;
89 actualMult.Put( 0 );
90
91 theLevelEnergies = 0;
92 theTransitionProbabilities = 0;
93 thePhotonTransitionFraction = 0;
94
95 }
96
97 ~G4ParticleHPPhotonDist()
98 {
99 delete [] disType;
100 delete [] energy;
101 delete [] theYield;
102 delete [] thePartialXsec;
103 delete [] theReactionXsec;
104 delete [] isPrimary;
105 delete [] theShells;
106 delete [] theGammas;
107 delete [] nNeu;
108 delete [] theAngular;
109 delete [] distribution;
110 delete [] probs;
111
112 if ( theLegendre != 0 )
113 {
114 for ( G4int i = 0 ; i < (nDiscrete2-nIso) ; i++ )
115 if ( theLegendre[i] != 0 ) delete[] theLegendre[i];
116
117 delete [] theLegendre;
118 }
119
120 if ( partials != 0 )
121 {
122 for ( G4int i = 0 ; i < nPartials ; i++ )
123 { delete partials[i]; }
124
125 delete [] partials;
126 }
127
128 delete [] theLevelEnergies;
129 delete [] theTransitionProbabilities;
130 delete [] thePhotonTransitionFraction;
131 if (actualMult.Get() != 0) delete actualMult.Get();
132 }
133
134 G4bool InitMean(std::istream & aDataFile);
135
136 void InitAngular(std::istream & aDataFile);
137
138 void InitEnergies(std::istream & aDataFile);
139
140 void InitPartials(std::istream& aDataFile, G4ParticleHPVector* theXsec = 0);
141
142 G4ReactionProductVector * GetPhotons(G4double anEnergy);
143
144 inline G4double GetTargetMass() {return targetMass;}
145
146 inline G4bool NeedsCascade() {return repFlag==2;}
147
148 inline G4double GetLevelEnergy() {return theBaseEnergy;}
149
150private:
151
152 G4int repFlag; //representation as multiplicities or transition probability arrays.
153 G4double targetMass;
154
155 G4int nDiscrete; //number of discrete photons
156 G4int* disType; // discrete, or continuum photons
157 G4double* energy; // photon energies
158 G4ParticleHPVector* theYield; // multiplicity as a function of neutron energy.
159 G4ParticleHPVector theTotalXsec;
160 G4ParticleHPVector* thePartialXsec;
161 G4ParticleHPVector* theReactionXsec;
162 G4int* isPrimary;
163
164 G4int isoFlag; // isotropic or not?
165 G4int tabulationType;
166 G4int nDiscrete2;
167 G4int nIso;
168 G4double * theShells;
169 G4double * theGammas;
170 G4int * nNeu;
171 G4InterpolationManager theLegendreManager;
172 G4ParticleHPLegendreTable ** theLegendre;
173 G4ParticleHPAngularP ** theAngular;
174
175 G4int * distribution; // not used for the moment.
176 G4int nPartials;
177 G4ParticleHPVector * probs; // probabilities for the partial distributions.
178 G4ParticleHPPartial ** partials; // the partials, parallel to the above
179
180 G4Cache< std::vector<G4int>* > actualMult;
181
182 // for transition prob arrays start
183 G4int theInternalConversionFlag;
184 G4int nGammaEnergies;
185 G4double theBaseEnergy;
186 G4double * theLevelEnergies;
187 G4double * theTransitionProbabilities;
188 G4double * thePhotonTransitionFraction;
189 // for transition prob arrays end
190
191 G4ParticleHPFastLegendre theLegend; // fast look-up for leg-integrals
192 G4ParticleHPInterpolator theInt; // interpolation
193};
