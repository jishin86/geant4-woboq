29#ifndef G4ParticleHPFSFissionFS_h
30#define G4ParticleHPFSFissionFS_h 1
31
32#include "globals.hh"
33#include "G4HadProjectile.hh"
34#include "G4DynamicParticleVector.hh"
35#include "G4ParticleHPFinalState.hh"
36#include "G4ParticleHPNames.hh"
37#include "G4ParticleHPParticleYield.hh"
38#include "G4ParticleHPVector.hh"
39#include "G4ParticleHPFissionERelease.hh"
40#include "G4ParticleHPEnergyDistribution.hh"
41#include "G4ParticleHPPhotonDist.hh"
42#include "G4ParticleHPAngular.hh"
43#include "G4Cache.hh"
44
45class G4ParticleHPFSFissionFS : public G4ParticleHPFinalState
46{
47 struct toBeCached
48 {
49 const G4ReactionProduct* theNeutronRP;
50 const G4ReactionProduct* theTarget;
51 toBeCached() : theNeutronRP(0),theTarget(0) {}
52 };
53
54public:
55
56 G4ParticleHPFSFissionFS() { hasXsec = true; }
57 ~G4ParticleHPFSFissionFS() {}
58
59 void Init (G4double A, G4double Z, G4int M,
60 G4String & dirName, G4String & aFSType, G4ParticleDefinition*);
61
62 G4DynamicParticleVector * ApplyYourself(G4int Prompt, G4int delayed,
63 G4double *decayconst);
64
65 G4ParticleHPFinalState* New()
66 {
67 G4ParticleHPFSFissionFS * theNew = new G4ParticleHPFSFissionFS;
68 return theNew;
69 }
70
71 inline G4double GetMass()
72 {
73 return theFinalStateNeutrons.GetTargetMass();
74 }
75
76 void SampleNeutronMult(G4int&all,
77 G4int&Prompt,
78 G4int&delayed,
79 G4double energy,
80 G4int off);
81
82 inline void SetNeutronRP(const G4ReactionProduct & aNeutron)
83 {
84 fCache.Get().theNeutronRP = &aNeutron;
85 theNeutronAngularDis.SetProjectileRP(aNeutron);
86 }
87
88 inline void SetTarget(const G4ReactionProduct & aTarget)
89 {
90 fCache.Get().theTarget = &aTarget;
91 theNeutronAngularDis.SetTarget(aTarget);
92 }
93
94 G4DynamicParticleVector * GetPhotons();
95
96 inline G4ParticleHPFissionERelease * GetEnergyRelease()
97 {
98 return &theEnergyRelease;
99 }
100
101private:
102
103 G4HadFinalState * ApplyYourself(const G4HadProjectile & ) { return 0; }
104
105 G4ParticleHPParticleYield theFinalStateNeutrons;
106 G4ParticleHPEnergyDistribution thePromptNeutronEnDis;
107 G4ParticleHPEnergyDistribution theDelayedNeutronEnDis;
108 G4ParticleHPAngular theNeutronAngularDis;
109
110 G4ParticleHPPhotonDist theFinalStatePhotons;
111 G4ParticleHPFissionERelease theEnergyRelease;
112
113 G4Cache<toBeCached> fCache;
114 G4ParticleHPNames theNames;
115};
116
117#endif
