91G4bool G4ParticleHPThermalScatteringNames::IsThisThermalElement( G4String aname)
92{
93 G4bool result = false;
94 if ( names.find ( aname ) != names.end() ) result = true;
95 return result;
96}
97
98G4bool G4ParticleHPThermalScatteringNames::IsThisThermalElement( G4String material , G4String element )
99{
100 G4bool result = false;
101 if ( nist_names.find ( std::pair<G4String,G4String>(material,element) ) != nist_names.end() ) result = true;
102 return result;
103}

106void G4ParticleHPThermalScatteringNames::AddThermalElement ( G4String nameG4Element , G4String filename)
107{
108 if ( names.find ( nameG4Element ) == names.end() ) names.insert( std::pair<G4String,G4String>( nameG4Element , filename ) );
109}
