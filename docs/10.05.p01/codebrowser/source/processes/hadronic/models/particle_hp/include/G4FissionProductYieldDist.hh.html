34#define G4FISSIONPRODUCTYIELDDIST_HH
35
36#include "G4Ions.hh"
37#include "G4Gamma.hh"
38#include "G4IonTable.hh"
39#include "G4ParticleHPNames.hh"
40#include "globals.hh"
41#include "G4DynamicParticle.hh"
42#include "G4DynamicParticleVector.hh"
43#include "G4ReactionProduct.hh"
44
45#include "G4ENDFTapeRead.hh"
46#include "G4ENDFYieldDataContainer.hh"
47#include "G4FFGEnumerations.hh"
48#include "G4FPYNubarValues.hh"
49#include "G4FPYSamplingOps.hh"

52/** G4FissionProductYieldDist is the base class for storing all the fission
53 * data and generating fission events. */
54class G4FissionProductYieldDist
55{
56public:
57// Constructor definition
58 /** Default constructor
59 * - Usage:
60 * - \p WhichIsotope: Isotope number of the element in ZZZAAA form
61 * - \p WhichMetaState: \p GROUND_STATE, \p META_1, or \p META_2
62 * - \p WhichCause: \p SPONTANEOUS or \p N_INDUCED
63 * - \p WhichYieldType: \p INDEPENDENT or \p CUMULATIVE
64 *
65 * - Notes:
66 */
67 G4FissionProductYieldDist( G4int WhichIsotope,
68 G4FFGEnumerations::MetaState WhichMetaState,
69 G4FFGEnumerations::FissionCause WhichCause,
70 G4FFGEnumerations::YieldType WhichYieldType,
71 std::istringstream& dataStream);
72 /** Overloaded constructor
73 * - Usage:
74 * - \p WhichIsotope: Isotope number of the element in ZZZAAA form
75 * - \p WhichMetaState: \p GROUND_STATE, \p META_1, or \p META_2
76 * - \p WhichCause: \p SPONTANEOUS or \p N_INDUCED
77 * - \p WhichYieldType: \p INDEPENDENT or \p CUMULATIVE
78 * - \p Verbosity: Verbosity level
79 *
80 * - Notes:
81 */
82 G4FissionProductYieldDist( G4int WhichIsotope,
83 G4FFGEnumerations::MetaState WhichMetaState,
84 G4FFGEnumerations::FissionCause WhichCause,
85 G4FFGEnumerations::YieldType WhichYieldType,
86 G4int Verbosity,
87 std::istringstream& dataStream);
88private:
89 /** Initialize is a common function called by all constructors. */
90 void Initialize( std::istringstream& dataStream );
91
92public:
93// Functions
94 /** Generates a fission event using default sampling and returns the pointer
95 * to that fission event.
96 * - Usage: No arguments required
97 *
98 * - Notes:
99 * - The fission products are loaded into FissionContainer in the
100 * following order:
101 * - First daughter product
102 * - Second daughter product
103 * - Alpha particles
104 * - Neutrons
105 * - Gamma rays
106 * - The last particle will have a NULL NextFragment pointer
107 */
108 G4DynamicParticleVector* G4GetFission( void );
109 /** Selects a fission fragment at random from the probability tree and
110 * returns the \p G4Ions pointer.
111 * - Usage: No arguments required
112 *
113 * - Notes:
114 */
115 G4Ions* G4GetFissionProduct( void );
116 /** Set the alpha production behavior for fission event generation.
117 * - Usage:
118 * - if \p AlphaProduction is negative then alpha particles are sampled
119 * randomly.
120 *
121 * - Notes:
122 * - The maximum number of alpha particles that may be created is
123 * physically limited by the nucleons present in the parent nucleus.
124 * Setting the AlphaProduction too high will have unpredictable
125 * results on the sampling of the fission products.
126 */
127 void G4SetAlphaProduction( G4double WhatAlphaProduction );
128 /** Sets the energy of the incident particle
129 * - Usage:
130 * - \p WhatIncidentEnergy: Kinetic energy, if any, of the incident
131 * neutron in GeV
132 *
133 * - Notes:
134 */
135 void G4SetEnergy( G4double WhatIncidentEnergy );
136 /** Sets the probability of ternary fission
137 * - Usage:
138 * - \p WhatTernaryProbability: Probability of generating a ternary
139 * fission event.
140 *
141 * - Notes:
142 */
143 void G4SetTernaryProbability( G4double TernaryProbability );
144 /** Sets the verbosity levels
145 * - Usage:
146 * - \p WhichVerbosity: Combination of levels
147 *
148 * - Notes:
149 * - \p SILENT: All verbose output is repressed
150 * - \p UPDATES: Only high-level internal changes are reported
151 * - \p DAUGHTER_INFO: Displays information about daughter product sampling
152 * - \p NEUTRON_INFO: Displays information about neutron sampling
153 * - \p GAMMA_INFO: Displays information about gamma sampling
154 * - \p ALPHA_INFO: Displays information about alpha sampling
155 * - \p MOMENTUM_INFO: Displays information about momentum balancing
156 * - \p EXTRAPOLATION_INTERPOLATION_INFO: Displays information about any data extrapolation or interpolation that occurs
157 * - \p DEBUG: Reports program flow as it steps through functions
158 * - \p PRINT_ALL: Displays any and all output
159 */
160 void G4SetVerbosity( G4int WhatVerbosity );
161
162protected:
163// Variables
164 // Class descriptor variables
165 /** Number in ZZZAAA format of the isotope that
166 * G4FissionProductYieldDist references
167 */
168 const G4int Isotope_;
169 /** MetaState information of the isotope that G4FissionProductYieldDist
170 * references
171 * \n Possible values are \p GROUND_STATE, \p META_1, or \p META_2
172 */
173 const G4FFGEnumerations::MetaState MetaState_;
174 /** The cause of fission: \p SPONTANEOUS or \p N_INDUCED. */
175 const G4FFGEnumerations::FissionCause Cause_;
176 /** The type of yield to be used: \p INDEPENDET or \p CUMULATIVE */
177 const G4FFGEnumerations::YieldType YieldType_;
178
179 // Datafile variables
180 /** Name of the fission yield product data file that
181 * G4FissionProductYieldDist references
182 */
183 G4ENDFTapeRead* ENDFData_;
184
185 // Fission generation variables
186 /** Contains the \p G4Ions pointer to an alpha particle */
187 G4Ions* AlphaDefinition_;
188 /** Controls whether alpha particles are emitted, and how many */
189 G4double AlphaProduction_;
190 /** Sets the ternary fission probability. Valid ranges are [0, 1] */
191 G4double TernaryProbability_;
192 /** Contains the \p g4ParticleDefinition pointer to a gamma particle */
193 G4Gamma* GammaDefinition_;
194 /** Kinetic energy, if any, of the incident particle in GeV. */
195 G4double IncidentEnergy_;
196 /** Sets the mean gamma energy, in MeV, produced by the fission of the
197 * isotope described by Isotope_
198 */
199 G4double MeanGammaEnergy_;
200 /** Contains the G4ParticleDefinition pointer to a neutron, cast as a
201 * G4Ion for compatibility*/
202 G4Ions* NeutronDefinition_;
203 /** Nubar for the isotope and incident neutron energy that
204 * G4FissionProductYieldDist references.
205 */
206 G4double Nubar_;
207 /** Width of the gaussian distribution that samples nubar for the
208 * isotope and incident neutron energy that G4FissionProductYieldDist
209 * references.
210 */
211 G4double NubarWidth_;
212 /** Counter for the number of protons available to the fission event */
213 G4int RemainingZ_;
214 /** Counter for the number of nucleons available to the fission event */
215 G4int RemainingA_;
216 /** Container for the energy remaining to be assigned in the fission generation */
217 G4double RemainingEnergy_;
218 /** Verbosity level */
219 G4int Verbosity_;
220
221 // Pointers to the field of trees and relevant normalization data
222 /** An array, or 'field', of the probability trees */
223 ProbabilityTree* Trees_;
224 /** Defines the smallest Z particle in the field of trees */
225 G4Ions* SmallestZ_;
226 /** Defines the smallest A particle in the field of trees */
227 G4Ions* SmallestA_;
228 /** Defines the largest Z particle in the field of trees. */
229 G4Ions* LargestZ_;
230 /** Defines the largest Z particle in the field of trees */
231 G4Ions* LargestA_;
232 /** Number of specific energy groups */
233 G4int YieldEnergyGroups_;
234 /** Energy values of each energy */
235 G4double* YieldEnergies_;
236 /** Variable for ensuring that the input data is normalized */
237 G4double* MaintainNormalizedData_;
238 /** A running total of all the probabilities */
239 G4double* DataTotal_;
240 /** The number of trees in the field */
241 G4int TreeCount_;
242 /** A run-time counter for the total number of branches stored */
243 G4int BranchCount_;
244
245 // Pointers to runtime classes that G4FissionProductYieldDist needs to
246 // function properly
247 /** Pointer to \p G4IonTable
248 * \n All \p G4Ions are created using
249 * \p G4IonTable
250 */
251 G4IonTable* IonTable_;
252 /** Pointer to \p G4NeutronHPNames
253 * \n Provides access to the list of element names included in Geant4
254 */
255 G4ParticleHPNames* ElementNames_;
256 /** Pointer to the \p CLHEP library random engine */
257 G4FPYSamplingOps* RandomEngine_;
258
259//Functions
260 /** Checks to make sure that alpha overpopulation will not occur, which
261 * could result in an unsolvable zero momentum in the LAB system.
262 */
263 void CheckAlphaSanity( void );
264 /** Returns the \p G4Ions definitions pointer for the particle whose
265 * probability segment contains the (0, 1] random number \p RandomParticle
266 */
267 G4Ions* FindParticle( G4double RandomParticle );
268 /** Returns the \p G4Ions definitions pointer for the particle whose
269 * probability segment contains the (0, 1] random number \p RandomParticle
270 * by extrapolating values using the current data set.
271 * This function exists so that that different models of extrapolation
272 * may be more easily implemented in the future.
273 */
274 G4Ions* FindParticleExtrapolation( G4double RandomParticle,
275 G4bool LowerEnergyGroupExists );
276 /** Returns the \p G4Ions definitions pointer for the particle whose
277 * probability segment contains the (0, 1] random number \p RandomParticle
278 * by interpolating values in the current data set.
279 * This function exists so that that different models of interpolation
280 * may be more easily implemented in the future.
281 */
282 G4Ions* FindParticleInterpolation( G4double RandomParticle,
283 G4int LowerEnergyGroup );
284 /** Returns the \p G4Ions definitions pointer for the particle whose
285 * probability segment contains the (0, 1] random number \p RandomParticle
286 * by searching through a branch. Both the extrapolation and interpolation
287 * schemes currently use this function to identify the particle.
288 */
289 G4Ions* FindParticleBranchSearch( ProbabilityBranch* Branch,
290 G4double RandomParticle,
291 G4int EnergyGroup1,
292 G4int EnergyGroup2 );
293 /** Generates a \p G4DynamicParticleVector with the fission alphas
294 */
295 virtual void GenerateAlphas( std::vector< G4ReactionProduct* >* Alphas );
296 /** Generate a linked chain of neutrons and return the pointer to the last
297 * neutron in the chain.
298 */
299 virtual void GenerateNeutrons( std::vector< G4ReactionProduct* >* Neutrons );
300 /** Selects a fission product from the probability tree, limited by the
301 * number of nucleons available to the system
302 */
303 virtual G4Ions* GetFissionProduct( void ) = 0;
304 /** Returns the \p G4Ions definition pointer to the isotope defined by
305 * \p Product and \p MetaState.
306 * Searches the \p ParticleTable for the particle defined by \p Product
307 * (ZZZAAA) and \p MetaState and returns the \p G4Ions
308 * pointer to that particle. If the particle does not exist then it is
309 * created in \p G4ParticleTable and the pointer to the new particle is
310 * returned.
311 */
312 G4Ions* GetParticleDefinition( G4int Product,
313 G4FFGEnumerations::MetaState MetaState );
314 /** Generates the directory location for the data file referenced by
315 * G4FissionProductYieldDist
316 */
317 G4String MakeDirectoryName( void );
318 /** Generates the appropriate file name for the isotope requested */
319 G4String MakeFileName( G4int Isotope,
320 G4FFGEnumerations::MetaState MetaState );
321 /** Creates a \p G4DynamicParticle from an existing \p G4ReactionProduct */
322 G4DynamicParticle* MakeG4DynamicParticle( G4ReactionProduct* );
323 /** Generates the unique name for an isotope/isomer defined by \p Isotope\
324 * and \p MetaState in the following format: ZZZ_AAAmX_NAME
325 */
326 G4String MakeIsotopeName( G4int Isotope,
327 G4FFGEnumerations::MetaState MetaState );
328 /** Dynamically allocates and initializes the 'field' of 'trees' with the
329 * 'trunks'
330 */
331 virtual void MakeTrees( void );
332 /** Reads in the probability data from the data file */
333 virtual void ReadProbabilities( void );
334 /** Renormalizes the data in a ProbabilityTree.
335 * Traverses the tree structure and renormalizes all the probability data
336 * into probability segments, ensuring that no segment overlaps the
337 * other.
338 */
339 void Renormalize( ProbabilityBranch* Branch );
340 /** Sample the energy of the alpha particles. The energy used by the alpha
341 * particles is subtracted from the available energy
342 */
343 void SampleAlphaEnergies( std::vector< G4ReactionProduct* >* Alphas );
344 /** Samples the energy of the gamma rays */
345 void SampleGammaEnergies( std::vector< G4ReactionProduct* >* Gammas );
346 /** Sample the energy of the neutrons using the Watt fission spectrum. The
347 * kinetic energy consumed is returned.
348 */
349 void SampleNeutronEnergies( std::vector< G4ReactionProduct* >* Neutrons );
350 /** Sets the nubar values for the isotope referenced by
351 * G4FissionProductYieldDistdefined from the data sets defined in
352 * SpecialOps.hh
353 */
354 void SetNubar( void );
355 /** Sorts information for a potential new particle into the correct tree */
356 virtual void SortProbability( G4ENDFYieldDataContainer* YieldData );
357
358// Destructor function(s)
359public:
360 /** Default deconstructor. It is a virtual function since
361 * G4FissionProductYieldDist is a parent class
362 */
363 virtual ~G4FissionProductYieldDist( void );
364protected:
365 /** Recursively burns each branch in a probability tree. */
366 void BurnTree( ProbabilityBranch* Branch );
367};
368
369#endif /* G4FISSIONPRODUCTYIELDDIST_HH */
