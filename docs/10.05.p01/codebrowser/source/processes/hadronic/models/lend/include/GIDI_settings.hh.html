91class GIDI_settings_flux_order {
92
93 private:
94 int mOrder; /**< The Legendre order of the flux. */
95 std::vector<double> mEnergies; /**< List of flux energies. */
96 std::vector<double> mFluxes; /**< List of flux values - one for each element of mEnergies. */
97
98 public:
99 GIDI_settings_flux_order( int order /**< The Legendre order for this flux data. */ );
100 GIDI_settings_flux_order( int order /**< The Legendre order for this flux data. */,
101 int length /**< The number or values in energies and fluxes. */,
102 double const *energies /**< List of energies where flux is given. */,
103 double const *fluxes /**< List of flux value for each energies value. */ );
104 GIDI_settings_flux_order( int order /**< The Legendre order for this flux data. */,
105 std::vector<double> const &energies /**< List of energies where flux is given. */,
106 std::vector<double> const &fluxes /**< List of flux value for each energies value. */ );
107 GIDI_settings_flux_order( GIDI_settings_flux_order const &fluxOrder /**< Legendre flux order to copy. */ );
108 ~GIDI_settings_flux_order( );
109
110 inline int getOrder( void ) const { return( mOrder ); }
111 inline int size( void ) const { return( (int) mEnergies.size( ) ); }
112 inline double const *getEnergies( void ) const { return( &(mEnergies[0]) ); }
113 inline double const *getFluxes( void ) const { return( &(mFluxes[0]) ); }
114 void print( int valuesPerLine = 10 ) const;
115
116 private:
117 void initialize( int order, int length, double const *energies, double const *fluxes );
118};
