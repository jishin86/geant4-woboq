33#ifndef G4VPreCompoundTransitions_hh
34#define G4VPreCompoundTransitions_hh 1
35
36#include "G4Fragment.hh"
37
38class G4VPreCompoundTransitions
39{
40public:
41
42 G4VPreCompoundTransitions();
43 virtual ~G4VPreCompoundTransitions();
44
45 virtual G4double CalculateProbability(const G4Fragment& aFragment) = 0;
46 virtual void PerformTransition(G4Fragment& aFragment) = 0;
47
48 inline G4double GetTransitionProb1() const;
49
50 inline G4double GetTransitionProb2() const;
51
52 inline G4double GetTransitionProb3() const;
53
54 // for never go back hypothesis (if useNGB=true, default=false)
55 inline void UseNGB(G4bool use){useNGB=use;}
56 //for use of CEM transition probabilities (if useCEMtr=true, defaut false)
57 inline void UseCEMtr(G4bool use){useCEMtr=use;}
58
59private:
60
61 G4VPreCompoundTransitions(const G4VPreCompoundTransitions &);
62 const G4VPreCompoundTransitions& operator=(const G4VPreCompoundTransitions &right);
63 G4bool operator==(const G4VPreCompoundTransitions &right) const;
64 G4bool operator!=(const G4VPreCompoundTransitions &right) const;
65
66protected:
67
68 G4bool useNGB;
69 G4bool useCEMtr;
70
71 G4double TransitionProb1;
72 G4double TransitionProb2;
73 G4double TransitionProb3;
74};
