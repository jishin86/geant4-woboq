41G4NeutrinoElectronTotXsc::G4NeutrinoElectronTotXsc()
42 : G4VCrossSectionDataSet("NuElectronTotXsc")
43{
44 fCcXsc = new G4NeutrinoElectronCcXsc();
45 fNcXsc = new G4NeutrinoElectronNcXsc();
46
47 fCutEnergy = 0.; // default value
48 fBiasingFactor = 1.; // default as physics
49 fCcRatio = 0.5;
50}
51
52G4NeutrinoElectronTotXsc::~G4NeutrinoElectronTotXsc()
53{}

56
57G4bool
58G4NeutrinoElectronTotXsc::IsElementApplicable( const G4DynamicParticle* aPart, G4int i, const G4Material* mat)
59{
60 G4bool result = false;
61
62 G4bool apCc = fCcXsc->IsElementApplicable( aPart,i, mat);
63 G4bool apNc = fNcXsc->IsElementApplicable( aPart, i, mat);
64
65 if( apCc || apNc) result = true;
66
67 return result;
68}
69
70////////////////////////////////////////////////////
71
72G4double G4NeutrinoElectronTotXsc::
73GetElementCrossSection(const G4DynamicParticle* aPart, G4int ZZ,
74 const G4Material* mat)
75{
76 G4double result = 0.;
77
78 G4double ccXsc = fCcXsc->GetElementCrossSection( aPart, ZZ, mat);
79 G4double ncXsc = fNcXsc->GetElementCrossSection( aPart, ZZ, mat);
80
81 result = ccXsc + ncXsc;
82
83 if (result > 0.) fCcRatio = ccXsc/result;
84 else fCcRatio = 0.;
85
86 return result;
87}
