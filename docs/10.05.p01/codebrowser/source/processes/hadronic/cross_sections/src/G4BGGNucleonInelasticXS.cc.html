64G4BGGNucleonInelasticXS::G4BGGNucleonInelasticXS(const G4ParticleDefinition* p)
65 : G4VCrossSectionDataSet("Barashenkov-Glauber")
66{
67 verboseLevel = 0;
68 fGlauberEnergy = 91.*GeV;
69 fLowEnergy = 14.*MeV;
70 fHighEnergy = 5.*GeV;
71 fSAIDHighEnergyLimit = 1.3*GeV;
72 fLowestXSection = millibarn;
73 for (G4int i = 0; i < 93; ++i) {
74 theGlauberFac[i] = 0.0;
75 theCoulombFac[i] = 0.0;
76 theA[i] = 1;
77 }
78 fNucleon = nullptr;
79 fGlauber = nullptr;
80 fHadron = nullptr;
81 fSAID = nullptr;
82
83 particle = p;
84 theProton= G4Proton::Proton();
85 isProton = (theProton == p) ? true : false;
86 isInitialized = false;
87 SetForAllAtomsAndEnergies(true);
88}

117G4double
118G4BGGNucleonInelasticXS::GetElementCrossSection(const G4DynamicParticle* dp,
119 G4int ZZ, const G4Material*)
120{
121 G4double cross = 0.0;
122 G4double ekin = dp->GetKineticEnergy();
123 G4int Z = std::min(ZZ, 92);
124 if(1 == Z) {
125 cross = 1.0115*GetIsoCrossSection(dp,1,1);
126 } else if(2 == Z) {
127 if(ekin > fGlauberEnergy) {
128 cross = theGlauberFac[Z]*fGlauber->GetInelasticGlauberGribov(dp, Z, theA[Z]);
129 } else {
130 cross = fNucleon->GetElementCrossSection(dp, Z);
131 }
132 } else {
133 if(ekin <= fLowEnergy) {
134 cross = theCoulombFac[Z]*CoulombFactor(ekin, Z);
135 } else if(ekin > fGlauberEnergy) {
136 cross = theGlauberFac[Z]*fGlauber->GetInelasticGlauberGribov(dp, Z, theA[Z]);
137 } else {
138 cross = fNucleon->GetElementCrossSection(dp, Z);
139 }
140 }
141
142 if(verboseLevel > 1) {
143 G4cout << "G4BGGNucleonInelasticXS::GetCrossSection for "
144 << dp->GetDefinition()->GetParticleName()
145 << " Ekin(GeV)= " << dp->GetKineticEnergy()/CLHEP::GeV
146 << " in nucleus Z= " << Z << " A= " << theA[Z]
147 << " XS(b)= " << cross/barn
148 << G4endl;
149 }
150 return cross;
151}

155G4double
156G4BGGNucleonInelasticXS::GetIsoCrossSection(const G4DynamicParticle* dp,
157 G4int Z, G4int A,
158 const G4Isotope*,
159 const G4Element*,
160 const G4Material*)
161{
162 // this method should be called only for Z = 1
163
164 G4double cross = 0.0;
165 if(1 == Z) {
166 fHadron->GetHadronNucleonXscNS(dp, theProton);
167 cross = fHadron->GetInelasticHadronNucleonXsc();
168 }
169 cross *= A;
170 if(verboseLevel > 1) {
171 G4cout << "G4BGGNucleonInelasticXS::GetIsoCrossSection for "
172 << dp->GetDefinition()->GetParticleName()
173 << " Ekin(GeV)= " << dp->GetKineticEnergy()/CLHEP::GeV
174 << " in nucleus Z= " << Z << " A= " << theA[Z]
175 << " XS(b)= " << cross/barn
176 << G4endl;
177 }
178 return cross;
179}

183void G4BGGNucleonInelasticXS::BuildPhysicsTable(const G4ParticleDefinition& p)
184{
185 if(&p == theProton || &p == G4Neutron::Neutron()) {
186 particle = &p;
187 } else {
188 G4cout << "### G4BGGNucleonInelasticXS WARNING: is not applicable to "
189 << p.GetParticleName()
190 << G4endl;
191 throw G4HadronicException(__FILE__, __LINE__,
192 "G4BGGNucleonElasticXS::BuildPhysicsTable is used for wrong particle");
193 return;
194 }
195
196 if(isInitialized) { return; }
197 isInitialized = true;
198
199 fNucleon = (G4NucleonNuclearCrossSection*)G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4NucleonNuclearCrossSection::Default_Name());
200 fGlauber = new G4ComponentGGHadronNucleusXsc();
201 fHadron = new G4HadronNucleonXsc();
202
203 fNucleon->BuildPhysicsTable(*particle);
204 fGlauber->BuildPhysicsTable(*particle);
205
206 G4ThreeVector mom(0.0,0.0,1.0);
207 G4DynamicParticle dp(particle, mom, fGlauberEnergy);
208
209 G4NistManager* nist = G4NistManager::Instance();
210 G4int A;
211
212 G4double csup, csdn;
213
214 if(verboseLevel > 0) {
215 G4cout << "### G4BGGNucleonInelasticXS::Initialise for "
216 << particle->GetParticleName() << G4endl;
217 }
218 for(G4int iz=2; iz<93; iz++) {
219
220 A = G4lrint(nist->GetAtomicMassAmu(iz));
221 theA[iz] = A;
222
223 csup = fGlauber->GetInelasticGlauberGribov(&dp, iz, A);
224 csdn = fNucleon->GetElementCrossSection(&dp, iz);
225
226 theGlauberFac[iz] = csdn/csup;
227 if(verboseLevel > 0) {
228 G4cout << "Z= " << iz << " A= " << A
229 << " GlauberFactor= " << theGlauberFac[iz] << G4endl;
230 }
231 }
232 theCoulombFac[1] = theCoulombFac[2] = 1.0;
233 dp.SetKineticEnergy(fLowEnergy);
234 for(G4int iz=3; iz<93; ++iz) {
235 theCoulombFac[iz] =
236 fNucleon->GetElementCrossSection(&dp, iz)/CoulombFactor(fLowEnergy, iz);
237
238 if(verboseLevel > 0) {
239 G4cout << "Z= " << iz << " A= " << theA[iz]
240 << " CoulombFactor= " << theCoulombFac[iz] << G4endl;
241 }
242 }
243}

247G4double G4BGGNucleonInelasticXS::CoulombFactor(G4double kinEnergy, G4int Z)
248{
249 G4double res= 0.0;
250 if(kinEnergy <= 0.0) { return res; }
251 else if (Z <= 1) { return kinEnergy*kinEnergy; }
252
253 G4double elog = G4Log(kinEnergy/GeV)/llog10;
254 G4double aa = theA[Z];
255
256 // from G4ProtonInelasticCrossSection
257 if(isProton) {
258
259 G4double ff1 = 5.6 - 0.016*aa; // slope of the drop at medium energies.
260 G4double ff2 = 1.37 + 1.37/aa; // start of the slope.
261 G4double ff3 = 0.8 + 18./aa - 0.002*aa; // stephight
262 res = 1.0 + ff3*(1.0 - (1.0/(1+G4Exp(-ff1*(elog + ff2)))));
263
264 ff1 = 8. - 8./aa - 0.008*aa; // slope of the rise
265 ff2 = 2.34 - 5.4/aa - 0.0028*aa; // start of the rise
266 res /= (1.0 + G4Exp(-ff1*(elog + ff2)));
267
268 } else {
269
270 // from G4NeutronInelasticCrossSection
271 G4double p3 = 0.6 + 13./aa - 0.0005*aa;
272 G4double p4 = 7.2449 - 0.018242*aa;
273 G4double p5 = 1.36 + 1.8/aa + 0.0005*aa;
274 G4double p6 = 1. + 200./aa + 0.02*aa;
275 G4double p7 = 3.0 - (aa-70.)*(aa-200.)/11000.;
276
277 G4double firstexp = G4Exp(-p4*(elog + p5));
278 G4double secondexp = G4Exp(-p6*(elog + p7));
279
280 res = (1.+p3*firstexp/(1. + firstexp))/(1. + secondexp);
281
282 }
283 return res;
284}

288void G4BGGNucleonInelasticXS::CrossSectionDescription(std::ostream& outFile) const
289{
290 outFile << "The Barashenkov-Glauber-Gribov cross section calculates inelastic\n"
291 << "scattering of protons and neutrons from nuclei using the\n"
292 << "Barashenkov parameterization below 91 GeV and the Glauber-Gribov\n"
293 << "parameterization above 91 GeV. It uses the G4HadronNucleonXsc\n"
294 << "cross section component for hydrogen targets, and the\n"
295 << "G4ComponentGGHadronNucleusXsc component for other targets.\n";
296}
