56G4BGGPionElasticXS::G4BGGPionElasticXS(const G4ParticleDefinition* p)
57 : G4VCrossSectionDataSet("Barashenkov-Glauber")
58{
59 verboseLevel = 0;
60 fGlauberEnergy = 91.*GeV;
61 fLowEnergy = 20.*MeV;
62 fSAIDHighEnergyLimit = 2.6*GeV;
63 SetMinKinEnergy(0.0);
64 SetMaxKinEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
65
66 for (G4int i = 0; i < 93; i++) {
67 theGlauberFac[i] = 0.0;
68 theCoulombFac[i] = 0.0;
69 theA[i] = 1;
70 }
71 fPion = nullptr;
72 fGlauber = nullptr;
73 fHadron = nullptr;
74 fSAID = nullptr;
75 particle = p;
76 theProton= G4Proton::Proton();
77 isPiplus = (p == G4PionPlus::PionPlus()) ? true : false;
78 isInitialized = false;
79 SetForAllAtomsAndEnergies(true);
80}

109G4double
110G4BGGPionElasticXS::GetElementCrossSection(const G4DynamicParticle* dp,
111 G4int ZZ, const G4Material*)
112{
113 // this method should be called only for Z > 1
114
115 G4double cross = 0.0;
116 G4double ekin = dp->GetKineticEnergy();
117 G4int Z = std::min(ZZ, 92);
118 if(1 == Z) {
119 cross = 1.0115*GetIsoCrossSection(dp,1,1);
120 } else {
121 if(ekin <= fLowEnergy) {
122 cross = theCoulombFac[Z];
123 } else if(ekin > fGlauberEnergy) {
124 cross = theGlauberFac[Z]*fGlauber->GetElasticGlauberGribov(dp, Z, theA[Z]);
125 } else {
126 cross = fPion->GetElasticCrossSection(dp, Z, theA[Z]);
127 }
128 }
129 if(verboseLevel > 1) {
130 G4cout << "G4BGGPionElasticXS::GetElementCrossSection for "
131 << dp->GetDefinition()->GetParticleName()
132 << " Ekin(GeV)= " << dp->GetKineticEnergy()
133 << " in nucleus Z= " << Z << " A= " << theA[Z]
134 << " XS(b)= " << cross/barn
135 << G4endl;
136 }
137 return cross;
138}

140G4double
141G4BGGPionElasticXS::GetIsoCrossSection(const G4DynamicParticle* dp,
142 G4int Z, G4int A,
143 const G4Isotope*,
144 const G4Element*,
145 const G4Material*)
146{
147 // this method should be called only for Z = 1
148 G4double cross = 0.0;
149 if(1 == Z) {
150 G4double ekin = std::max(dp->GetKineticEnergy(), fLowEnergy);
151 fHadron->HadronNucleonXscNS(dp->GetDefinition(), theProton, ekin);
152 cross = fHadron->GetElasticHadronNucleonXsc();
153 }
154 cross *= A;
155 if(verboseLevel > 1) {
156 G4cout << "G4BGGPionElasticXS::GetIsoCrossSection for "
157 << dp->GetDefinition()->GetParticleName()
158 << " Ekin(GeV)= " << dp->GetKineticEnergy()
159 << " in nucleus Z= " << Z << " A= " << A
160 << " XS(b)= " << cross/barn
161 << G4endl;
162 }
163 return cross;
164}

168void G4BGGPionElasticXS::BuildPhysicsTable(const G4ParticleDefinition& p)
169{
170 if(&p == G4PionPlus::PionPlus() || &p == G4PionMinus::PionMinus()) {
171 particle = &p;
172 } else {
173 G4cout << "### G4BGGPionElasticXS WARNING: is not applicable to "
174 << p.GetParticleName()
175 << G4endl;
176 throw G4HadronicException(__FILE__, __LINE__,
177 "G4BGGPionElasticXS::BuildPhysicsTable is used for wrong particle");
178 return;
179 }
180
181 if(isInitialized) { return; }
182 isInitialized = true;
183
184 fPion = new G4UPiNuclearCrossSection();
185 fGlauber = new G4ComponentGGHadronNucleusXsc();
186 fHadron = new G4HadronNucleonXsc();
187
188 fPion->BuildPhysicsTable(*particle);
189 fGlauber->BuildPhysicsTable(*particle);
190
191 G4ThreeVector mom(0.0,0.0,1.0);
192 G4DynamicParticle dp(particle, mom, fGlauberEnergy);
193
194 G4NistManager* nist = G4NistManager::Instance();
195
196 G4double csup, csdn;
197 for(G4int iz=2; iz<93; iz++) {
198
199 G4int A = G4lrint(nist->GetAtomicMassAmu(iz));
200 theA[iz] = A;
201
202 csup = fGlauber->GetElasticGlauberGribov(&dp, iz, A);
203 csdn = fPion->GetElasticCrossSection(&dp, iz, A);
204
205 theGlauberFac[iz] = csdn/csup;
206 if(verboseLevel > 0) {
207 G4cout << "Z= " << iz << " A= " << A
208 << " factor= " << theGlauberFac[iz] << G4endl;
209 }
210 }
211 theCoulombFac[1] = 1.0;
212 dp.SetKineticEnergy(fLowEnergy);
213 for(G4int iz=2; iz<93; iz++) {
214 theCoulombFac[iz] = fPion->GetElasticCrossSection(&dp, iz, theA[iz]);
215 if(verboseLevel > 0) {
216 G4cout << "Z= " << iz << " A= " << theA[iz]
217 << " CoulombFactor= " << theCoulombFac[iz] << G4endl;
218 }
219 }
220}

224{
225 outFile << "The Barashenkov-Glauber-Gribov cross section handles elastic\n"

228 << "Glauber-Gribov parameterization is used above 91 GeV.\n";
229}
