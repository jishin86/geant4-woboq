39// Class Description:
40// This is a base class for n,p,d,t,he3,he4 inelastic hadronic cross
41// section based on data files from G4PARTICLEXSDATA data set
42// and Glauber-Gribov model for high energy
43//

48#include "G4VCrossSectionDataSet.hh"
49#include "globals.hh"

51#include "G4Threading.hh"

53#include <iostream>

57class G4DynamicParticle;
58class G4ParticleDefinition;

78 virtual

80 const G4Element*, const G4Material*);
81
82 virtual
83 G4double GetElementCrossSection(const G4DynamicParticle*,

87 G4double GetIsoCrossSection(const G4DynamicParticle*, G4int Z, G4int A,

90 const G4Material* mat);

92 virtual const G4Isotope* SelectIsotope(const G4Element*, G4double kinEnergy);

94 virtual

97 virtual void CrossSectionDescription(std::ostream&) const;
98
99 G4double IsoCrossSection(G4double ekin, G4int Z, G4int A);
100
101private:
102
103 void Initialise(G4int Z, G4DynamicParticle* dp, const char*);
104
105 G4PhysicsVector* RetrieveVector(std::ostringstream& in, G4bool warn);
106
107 G4ParticleInelasticXS & operator=(const G4ParticleInelasticXS &right);

117
118 G4String particleName;
119
120 G4bool isMaster;
121
122 G4double emax;
123 std::vector<G4double> temp;
124
125 static G4ElementData* data;
126
127 static G4double coeff[MAXZINELP];
128
129 static const G4int amin[MAXZINELP];

132#ifdef G4MULTITHREADED
133 static G4Mutex particleInelasticXSMutex;
134#endif
135};
