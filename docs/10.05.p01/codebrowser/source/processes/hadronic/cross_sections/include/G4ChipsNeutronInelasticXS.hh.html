37#ifndef G4ChipsNeutronInelasticXS_h
38#define G4ChipsNeutronInelasticXS_h 1
39
40#include "G4ParticleTable.hh"
41#include "G4NucleiProperties.hh"
42#include <vector>
43#include "G4VCrossSectionDataSet.hh"
44
45class G4ChipsNeutronInelasticXS : public G4VCrossSectionDataSet
46{
47
48
49public:
50
51 G4ChipsNeutronInelasticXS();
52
53 ~G4ChipsNeutronInelasticXS();
54
55 static const char* Default_Name() {return "ChipsNeutronInelasticXS";}
56
57 virtual void CrossSectionDescription(std::ostream&) const;
58
59 // At present momentum (pMom) in MeV/c, CS in mb (@@ Units)
60
61 virtual G4bool IsIsoApplicable(const G4DynamicParticle* Pt, G4int Z, G4int A,
62 const G4Element* elm,
63 const G4Material* mat );
64
65 // At present momentum (pMom) in MeV/c, CS in mb (@@ Units)
66 virtual G4double GetIsoCrossSection(const G4DynamicParticle*, G4int tgZ, G4int A,
67 const G4Isotope* iso = 0,
68 const G4Element* elm = 0,
69 const G4Material* mat = 0);
70
71 virtual G4double GetChipsCrossSection(G4double momentum, G4int Z, G4int N, G4int pdg);
72
73private:
74 G4double CalculateCrossSection(G4int F, G4int I, G4int PDG, G4int Z,
75 G4int N, G4double Momentum);
76
77 G4double CrossSectionLin(G4int targZ, G4int targN, G4double P);
78 G4double CrossSectionLog(G4int targZ, G4int targN, G4double lP);
79 G4double CrossSectionFormula(G4int targZ, G4int targN, G4double P, G4double lP);
80 G4double EquLinearFit(G4double X, G4int N, G4double X0, G4double DX, G4double* Y);
81
82// Body
83private:
84 G4double* lastLEN; // Pointer to the last array of LowEnergy cross sections
85 G4double* lastHEN; // Pointer to the last array of HighEnergy cross sections
86 G4int lastN; // The last N of calculated nucleus
87 G4int lastZ; // The last Z of calculated nucleus
88 G4double lastP; // Last used in the cross section Momentum
89 G4double lastTH; // Last value of the Momentum Threshold
90 G4double lastCS; // Last value of the Cross Section
91 G4int lastI; // The last position in the DAMDB
92 G4double HEthresh; // High energy threshold parameter for the CS calculation
93 std::vector<G4double*>* LEN; // Vector of pointers to LowEnNeutronCrossSection
94 std::vector<G4double*>* HEN; // Vector of pointers to HighEnNeutronCrossSection
95
96 G4int j; // A#0f Z/N-records already tested in AMDB
97 std::vector <G4int> colN; // Vector of N for calculated nuclei (isotops)
98 std::vector <G4int> colZ; // Vector of Z for calculated nuclei (isotops)
99 std::vector <G4double> colP; // Vector of last momenta for the reaction
100 std::vector <G4double> colTH; // Vector of energy thresholds for the reaction
101 std::vector <G4double> colCS; // Vector of last cross sections for the reaction
102
103};
104
105#endif
