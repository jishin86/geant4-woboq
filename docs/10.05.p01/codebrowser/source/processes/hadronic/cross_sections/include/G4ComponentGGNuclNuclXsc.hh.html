27// elastic and quasi-elastic cross-sections

29// cross-sections in

34// G4GlauberGribovCrossSection

171inline G4double
172G4ComponentGGNuclNuclXsc::GetElasticGlauberGribov(const G4DynamicParticle* dp,
173 G4int Z, G4int A)
174{
175 ComputeCrossSections(dp->GetDefinition(), dp->GetKineticEnergy(), Z, A);
176 return fElasticXsc;
177}
178
179inline G4double
180G4ComponentGGNuclNuclXsc::GetInelasticGlauberGribov(const G4DynamicParticle* dp,
181 G4int Z, G4int A)
182{
183 ComputeCrossSections(dp->GetDefinition(), dp->GetKineticEnergy(), Z, A);
184 return fInelasticXsc;
185}
186
187inline G4double
188G4ComponentGGNuclNuclXsc::GetElementCrossSection(const G4DynamicParticle* dp,
189 G4int Z, const G4Material*)
190{
191 G4int A = G4lrint(fNist->GetAtomicMassAmu(Z));
192 ComputeCrossSections(dp->GetDefinition(), dp->GetKineticEnergy(), Z, A);
193 return fInelasticXsc;
194}
195
196inline G4double
197G4ComponentGGNuclNuclXsc::GetZandACrossSection(const G4DynamicParticle* dp,
198 G4int Z, G4int A)
199{
200 ComputeCrossSections(dp->GetDefinition(), dp->GetKineticEnergy(), Z, A);
201 return fInelasticXsc;
202}
203
204inline G4double
205G4ComponentGGNuclNuclXsc::GetCoulombBarier(const G4DynamicParticle* dp,
206 G4double Z, G4double A,
207 G4double pR, G4double tR)
208{
209 return ComputeCoulombBarier(dp->GetDefinition(), dp->GetKineticEnergy(),
210 G4lrint(Z), G4lrint(A), pR, tR);
211}
