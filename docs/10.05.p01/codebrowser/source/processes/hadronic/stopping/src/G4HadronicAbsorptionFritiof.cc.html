59// Constructor
60G4HadronicAbsorptionFritiof::
61G4HadronicAbsorptionFritiof( G4ParticleDefinition* pdef )
62 : G4HadronStoppingProcess( "hFritiofCaptureAtRest" ),
63 pdefApplicable( pdef ) {
64
65 G4TheoFSGenerator * theModel = new G4TheoFSGenerator( "FTFP" );
66 G4FTFModel * theStringModel = new G4FTFModel;
67 theLund = new G4LundStringFragmentation;
68 theStringDecay = new G4ExcitedStringDecay( theLund );
69 theStringModel->SetFragmentationModel( theStringDecay );
70
71 // Not a cascade - goes straight to Preco
72 G4HadronicInteraction* p =
73 G4HadronicInteractionRegistry::Instance()->FindModel("PRECO");
74 G4VPreCompoundModel * thePreEquilib = static_cast<G4VPreCompoundModel*>(p);
75 if(! thePreEquilib) { thePreEquilib = new G4PreCompoundModel; }
76
77 G4GeneratorPrecompoundInterface * theCascade =
78 new G4GeneratorPrecompoundInterface( thePreEquilib );
79
80 theModel->SetHighEnergyGenerator( theStringModel );
81 theModel->SetTransport( theCascade );
82
83 G4double theMin = 0.0*GeV;
84 G4double theMax = G4HadronicParameters::Instance()->GetMaxEnergy();
85 theModel->SetMinEnergy( theMin );
86 theModel->SetMaxEnergy( theMax );
87
88 RegisterMe( theModel );
89}
90
91
92G4HadronicAbsorptionFritiof::~G4HadronicAbsorptionFritiof() {
93 delete theLund;
94 delete theStringDecay;
95}
96
97
98// Applies to constructor-specified particle, or to all known cases
99G4bool G4HadronicAbsorptionFritiof::
100IsApplicable( const G4ParticleDefinition& particle ) {
101 return ( ( 0 == pdefApplicable &&
102 ( &particle == G4AntiProton::Definition() ||
103 &particle == G4AntiSigmaPlus::Definition() ||
104 particle.GetBaryonNumber() < -1 ) ) // Anti-nuclei
105 || ( &particle == pdefApplicable ) );
106}

111ProcessDescription( std::ostream& os ) const {
112 os << "Stopping and absorption of anti_protons, anti_sigma+, and \n"
113 << "all anti-nuclei using Fritiof (FTF) string model.\n"
114 << "Geant4 PreCompound model is used for nuclear de-excitation."
115 << std::endl;
116}
