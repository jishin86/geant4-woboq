63G4MuonMinusAtomicCapture::G4MuonMinusAtomicCapture(const G4String& name)
64 : G4VRestProcess(name, fHadronic),
65 // : G4HadronicProcess(name, fHadronAtRest),// name, process type
66 fElementSelector(new G4ElementSelector()),
67 fEmCascade(new G4EmCaptureCascade()), // Owned by InteractionRegistry
68 theTotalResult(new G4ParticleChange())
69{
70 // Modify G4VProcess flags to emulate G4VRest instead of G4VDiscrete
71 // enableAtRestDoIt = true;
72 // enablePostStepDoIt = false;
73 SetProcessSubType(fMuAtomicCapture);
74 G4HadronicProcessStore::Instance()->RegisterExtraProcess(this);
75}

83}

117G4VParticleChange* G4MuonMinusAtomicCapture::AtRestDoIt(const G4Track& track,
118 const G4Step&)
119{
120 // if primary is not Alive then do nothing (how?)
121 theTotalResult->Initialize(track);
122
123 G4Nucleus* nucleus = &targetNucleus;
124 // the call below actually sets the nucleus params;
125 // G4Nucleus targetNucleus; is a member of G4HadronicProcess
126 // G4Element* elm =
127 fElementSelector->SelectZandA(track, nucleus);
128
129 thePro.Initialise(track); // thePro was G4HadProjectile from G4HadronicProcess
130
131 // save track time an dstart capture from zero time
132 thePro.SetGlobalTime(0.0);
133 G4double time0 = track.GetGlobalTime();
134
135 // Do the electromagnetic cascade in the nuclear field.
136 // EM cascade should keep G4HadFinalState object,
137 // because it will not be deleted at the end of this method
138 //
139 result = fEmCascade->ApplyYourself(thePro, *nucleus);
140 G4double ebound = result->GetLocalEnergyDeposit(); // may need to carry this over; review
141 G4double edep = 0.0;
142 G4int nSecondaries = result->GetNumberOfSecondaries();
143 thePro.SetBoundEnergy(ebound);
144
145 // creating the muonic atom
146 ++nSecondaries;
147
148 G4IonTable* itp = G4IonTable::GetIonTable();
149 G4ParticleDefinition* muonicAtom = itp->GetMuonicAtom(nucleus->GetZ_asInt(),
150 nucleus->GetA_asInt());
151
152 G4DynamicParticle* dp = new G4DynamicParticle(muonicAtom,G4RandomDirection(),0.);
153 G4HadSecondary hadSec(dp);
154 hadSec.SetTime(time0);
155 result->AddSecondary(hadSec);
156
157 // Fill results
158 //
159 theTotalResult->ProposeTrackStatus(fStopAndKill);
160 theTotalResult->ProposeLocalEnergyDeposit(edep);
161 theTotalResult->SetNumberOfSecondaries(nSecondaries);
162 G4double w = track.GetWeight();
163 theTotalResult->ProposeWeight(w);
164
165#ifdef G4VERBOSE
166 if (GetVerboseLevel() > 1) {
167 G4cout << __func__
168 << " nSecondaries "
169 << nSecondaries
170 << G4endl;
171 }
172#endif
173
174 for(G4int i=0; i<nSecondaries; ++i) {
175 G4HadSecondary* sec = result->GetSecondary(i);
176
177 // add track global time to the reaction time
178 G4double time = sec->GetTime();
179 if(time < 0.0) { time = 0.0; }
180 time += time0;
181
182#ifdef G4VERBOSE
183 if (GetVerboseLevel() > 1) {
184 G4cout << __func__
185 << " "
186 << i
187 << " Resulting secondary "
188 << sec->GetParticle()->GetPDGcode()
189 << " "
190 << sec->GetParticle()->GetDefinition()->GetParticleName()
191 << G4endl;
192 }
193#endif
194
195 // create secondary track
196 G4Track* t = new G4Track(sec->GetParticle(),
197 time,
198 track.GetPosition());
199 t->SetWeight(w*sec->GetWeight());
200
201 t->SetTouchableHandle(track.GetTouchableHandle());
202 theTotalResult->AddSecondary(t);
203 }
204 result->Clear();
205
206 // fixme: needs to be done at the MuonicAtom level
207 // if (epReportLevel != 0) { // G4HadronicProcess::
208 // CheckEnergyMomentumConservation(track, *nucleus);
209 // }
210 return theTotalResult;
211}
