36 template <class A> class G4SortHelperPtr
37 {
38 public:
39 G4SortHelperPtr(A * aA){theA = aA;}
40 G4bool operator<(G4SortHelperPtr<A> right) const
41 { return *theA < (*right); }
42 A & operator * () {return *theA;}
43 private:
44 A * theA;
45 };
46#include <algorithm>
47template<class A> void G4PtrSort(std::vector<A *> * aList)
48{
49 std::vector<G4SortHelperPtr<A> > helper;
50 for(size_t i=0; i<aList->size(); i++) helper.push_back(aList->operator[](i));
51 std::sort(helper.begin(), helper.end());
52 for(size_t j=0; j<helper.size(); j++) aList->operator[](j)=&(*helper[j]);
53}
