26#ifndef G4HadronicException_h
27#define G4HadronicException_h
28
29#include <exception>
30#include <iostream>
31#include "globals.hh"
32
33class G4HadronicException : public std::exception
34{
35 public:
36 G4HadronicException(G4String in, G4int at, G4String mess)
37 {
38 theMessage = mess;
39 theName = in;
40 theLine = at;
41
42 Report(G4cout);
43
44 if(getenv("DumpCoreOnHadronicException") )
45 {
46 G4Exception("G4HadronicException", "007", FatalException,
47 "Fatal problem in above location");
48 }
49
50 }
51 virtual ~G4HadronicException() throw () {}
52
53 void Report(std::ostream & aS)
54 {
55 aS<< "In " <<theName<<", line "<<theLine<<": "<<std::endl;
56 aS<< "===> "<<theMessage<<std::endl;
57 }
58
59 private:
60 G4String theMessage;
61 G4String theName;
62 G4int theLine;
63};
64
65#endif
