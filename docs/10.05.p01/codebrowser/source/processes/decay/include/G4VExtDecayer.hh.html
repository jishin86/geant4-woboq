37#ifndef G4VExtDecayer_h
38#define G4VExtDecayer_h 1
39
40#include "G4ios.hh"
41#include "globals.hh"
42#include "G4DecayProducts.hh"
43class G4Track;
44
45class G4VExtDecayer
46{
47 // Class Description
48 // This class is a Abstract class for external decayer
49 // G4VExtDecayer has one pure virtual method of
50 // ImportDecayProducts which return decay products
51
52 public: //With Description
53 // Constructors
54 G4VExtDecayer(const G4String& name ="");
55
56 // Destructor
57 virtual ~G4VExtDecayer(){}
58
59 private:
60 // copy constructor
61 G4VExtDecayer(const G4VExtDecayer &){}
62
63 // Assignment Operation (generated)
64 G4VExtDecayer & operator=(const G4VExtDecayer&){return *this;};
65
66 public: //With Description
67 virtual G4DecayProducts* ImportDecayProducts(
68 const G4Track& aTrack
69 ) = 0;
70
71 const G4String& GetName() const;
72
73 protected:
74 G4String decayerName;
75};
76
77inline
78 G4VExtDecayer::G4VExtDecayer(const G4String& name):
79 decayerName(name)
80{
81}
82
83inline
84 const G4String& G4VExtDecayer::GetName() const
85{
86 return decayerName;
87}
88
89#endif
