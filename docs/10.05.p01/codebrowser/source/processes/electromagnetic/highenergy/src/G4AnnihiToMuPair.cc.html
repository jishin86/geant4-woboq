58G4AnnihiToMuPair::G4AnnihiToMuPair(const G4String& processName,
59 G4ProcessType type):G4VDiscreteProcess (processName, type)
60{
61 //e+ Energy threshold
62 const G4double Mu_massc2 = G4MuonPlus::MuonPlus()->GetPDGMass();
63 LowestEnergyLimit = 2.*Mu_massc2*Mu_massc2/electron_mass_c2 - electron_mass_c2;
64
65 //modele ok up to 1000 TeV due to neglected Z-interference
66 HighestEnergyLimit = 1000.*TeV;
67
68 CurrentSigma = 0.0;
69 CrossSecFactor = 1.;
70 SetProcessSubType(6);
71
72}

108G4double G4AnnihiToMuPair::ComputeCrossSectionPerAtom(G4double Epos, G4double Z)
109// Calculates the microscopic cross section in GEANT4 internal units.
110// It gives a good description from threshold to 1000 GeV
111{
112 static const G4double Mmuon = G4MuonPlus::MuonPlus()->GetPDGMass();
113 static const G4double Rmuon = CLHEP::elm_coupling/Mmuon; //classical particle radius
114 static const G4double Sig0 = CLHEP::pi*Rmuon*Rmuon/3.; //constant in crossSection
115 static const G4double pia = CLHEP::pi * CLHEP::fine_structure_const; // pi * alphaQED
116
117 G4double CrossSection = 0.;
118 if (Epos < LowestEnergyLimit) return CrossSection;
119
120 G4double xi = LowestEnergyLimit/Epos;
121 G4double piaxi = pia * sqrt(xi);
122 G4double SigmaEl = Sig0 * xi * (1.+xi/2.) * piaxi;
123 if( Epos>LowestEnergyLimit+1.e-5 ) SigmaEl /= (1.-std::exp( -piaxi/std::sqrt(1-xi) ));
124 CrossSection = SigmaEl*Z; // SigmaEl per electron * number of electrons per atom
125 return CrossSection;
126}

130G4double G4AnnihiToMuPair::CrossSectionPerVolume(G4double PositronEnergy,
131 const G4Material* aMaterial)
132{
133 const G4ElementVector* theElementVector = aMaterial->GetElementVector();
134 const G4double* NbOfAtomsPerVolume = aMaterial->GetVecNbOfAtomsPerVolume();
135
136 G4double SIGMA = 0.0;
137
138 for ( size_t i=0 ; i < aMaterial->GetNumberOfElements() ; ++i )
139 {
140 G4double AtomicZ = (*theElementVector)[i]->GetZ();
141 SIGMA += NbOfAtomsPerVolume[i] *
142 ComputeCrossSectionPerAtom(PositronEnergy,AtomicZ);
143 }
144 return SIGMA;
145}

149G4double G4AnnihiToMuPair::GetMeanFreePath(const G4Track& aTrack,
150 G4double, G4ForceCondition*)
151
152// returns the positron mean free path in GEANT4 internal units
153
154{
155 const G4DynamicParticle* aDynamicPositron = aTrack.GetDynamicParticle();
156 G4double PositronEnergy = aDynamicPositron->GetKineticEnergy()
157 +electron_mass_c2;
158 G4Material* aMaterial = aTrack.GetMaterial();
159 CurrentSigma = CrossSectionPerVolume(PositronEnergy, aMaterial);
160
161 // increase the CrossSection by CrossSecFactor (default 1)
162 G4double mfp = DBL_MAX;
163 if(CurrentSigma > DBL_MIN) mfp = 1.0/(CurrentSigma*CrossSecFactor);
164
165 return mfp;
166}

170G4VParticleChange* G4AnnihiToMuPair::PostStepDoIt(const G4Track& aTrack,
171 const G4Step& aStep)
172//
173// generation of e+e- -> mu+mu-
174//
175{
176
177 aParticleChange.Initialize(aTrack);
178 static const G4double Mele=electron_mass_c2;
179 static const G4double Mmuon=G4MuonPlus::MuonPlus()->GetPDGMass();
180
181 // current Positron energy and direction, return if energy too low
182 const G4DynamicParticle *aDynamicPositron = aTrack.GetDynamicParticle();
183 G4double Epos = aDynamicPositron->GetKineticEnergy() + Mele;
184
185 // test of cross section
186 if(CurrentSigma*G4UniformRand() >
187 CrossSectionPerVolume(Epos, aTrack.GetMaterial()))
188 {
189 return G4VDiscreteProcess::PostStepDoIt(aTrack,aStep);
190 }
191
192 if (Epos < LowestEnergyLimit) {
193 return G4VDiscreteProcess::PostStepDoIt(aTrack,aStep);
194 }
195
196 G4ParticleMomentum PositronDirection =
197 aDynamicPositron->GetMomentumDirection();
198 G4double xi = LowestEnergyLimit/Epos; // xi is always less than 1,
199 // goes to 0 at high Epos
200
201 // generate cost
202 //
203 G4double cost;
204 do { cost = 2.*G4UniformRand()-1.; }
205 // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
206 while (2.*G4UniformRand() > 1.+xi+cost*cost*(1.-xi) );
207 //1+cost**2 at high Epos
208 G4double sint = sqrt(1.-cost*cost);
209
210 // generate phi
211 //
212 G4double phi=2.*pi*G4UniformRand();
213
214 G4double Ecm = sqrt(0.5*Mele*(Epos+Mele));
215 G4double Pcm = sqrt(Ecm*Ecm-Mmuon*Mmuon);
216 G4double beta = sqrt((Epos-Mele)/(Epos+Mele));
217 G4double gamma = Ecm/Mele; // =sqrt((Epos+Mele)/(2.*Mele));
218 G4double Pt = Pcm*sint;
219
220 // energy and momentum of the muons in the Lab
221 //
222 G4double EmuPlus = gamma*( Ecm+cost*beta*Pcm);
223 G4double EmuMinus = gamma*( Ecm-cost*beta*Pcm);
224 G4double PmuPlusZ = gamma*(beta*Ecm+cost* Pcm);
225 G4double PmuMinusZ = gamma*(beta*Ecm-cost* Pcm);
226 G4double PmuPlusX = Pt*cos(phi);
227 G4double PmuPlusY = Pt*sin(phi);
228 G4double PmuMinusX =-Pt*cos(phi);
229 G4double PmuMinusY =-Pt*sin(phi);
230 // absolute momenta
231 G4double PmuPlus = sqrt(Pt*Pt+PmuPlusZ *PmuPlusZ );
232 G4double PmuMinus = sqrt(Pt*Pt+PmuMinusZ*PmuMinusZ);
233
234 // mu+ mu- directions for Positron in z-direction
235 //
236 G4ThreeVector
237 MuPlusDirection ( PmuPlusX/PmuPlus, PmuPlusY/PmuPlus, PmuPlusZ/PmuPlus );
238 G4ThreeVector
239 MuMinusDirection(PmuMinusX/PmuMinus,PmuMinusY/PmuMinus,PmuMinusZ/PmuMinus);
240
241 // rotate to actual Positron direction
242 //
243 MuPlusDirection.rotateUz(PositronDirection);
244 MuMinusDirection.rotateUz(PositronDirection);
245
246 aParticleChange.SetNumberOfSecondaries(2);
247 // create G4DynamicParticle object for the particle1
248 G4DynamicParticle* aParticle1= new G4DynamicParticle(
249 G4MuonPlus::MuonPlus(),MuPlusDirection,EmuPlus-Mmuon);
250 aParticleChange.AddSecondary(aParticle1);
251 // create G4DynamicParticle object for the particle2
252 G4DynamicParticle* aParticle2= new G4DynamicParticle(
253 G4MuonMinus::MuonMinus(),MuMinusDirection,EmuMinus-Mmuon);
254 aParticleChange.AddSecondary(aParticle2);
255
256 // Kill the incident positron
257 //
258 aParticleChange.ProposeEnergy(0.);
259 aParticleChange.ProposeTrackStatus(fStopAndKill);
260
261 return &aParticleChange;
262}

266void G4AnnihiToMuPair::PrintInfoDefinition()
267{
268 G4String comments ="e+e->mu+mu- annihilation, atomic e- at rest, SubType=.";
269 G4cout << G4endl << GetProcessName() << ": " << comments
270 << GetProcessSubType() << G4endl;
271 G4cout << " threshold at " << LowestEnergyLimit/GeV << " GeV"
272 << " good description up to "
273 << HighestEnergyLimit/TeV << " TeV for all Z." << G4endl;
274}
