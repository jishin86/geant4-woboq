598inline void
599G4VEnergyLossProcess::DefineMaterial(const G4MaterialCutsCouple* couple)
600{
601 if(couple != currentCouple) {
602 currentCouple = couple;
603 currentMaterial = couple->GetMaterial();
604 currentCoupleIndex = couple->GetIndex();
605 basedCoupleIndex = (*theDensityIdx)[currentCoupleIndex];
606 fFactor = chargeSqRatio*biasFactor*(*theDensityFactor)[currentCoupleIndex];
607 reduceFactor = 1.0/(fFactor*massRatio);
608 mfpKinEnergy = DBL_MAX;
609 idxLambda = idxSubLambda = 0;
610 }
611}

615inline void G4VEnergyLossProcess::SetDynamicMassCharge(G4double massratio,
616 G4double charge2ratio)
617{
618 massRatio = massratio;
619 fFactor = charge2ratio*biasFactor*(*theDensityFactor)[currentCoupleIndex];
620 chargeSqRatio = charge2ratio;
621 reduceFactor = 1.0/(fFactor*massRatio);
622}

624//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
625
626inline G4double G4VEnergyLossProcess::GetDEDXForScaledEnergy(G4double e)
627{
628 /*
629 G4cout << "G4VEnergyLossProcess::GetDEDX: Idx= "
630 << basedCoupleIndex << " E(MeV)= " << e
631 << " Emin= " << minKinEnergy << " Factor= " << fFactor
632 << " " << theDEDXTable << G4endl; */
633 G4double x = fFactor*(*theDEDXTable)[basedCoupleIndex]->Value(e, idxDEDX);
634 if(e < minKinEnergy) { x *= std::sqrt(e/minKinEnergy); }
635 return x;
636}

643 fFactor*(*theDEDXSubTable)[basedCoupleIndex]->Value(e, idxDEDXSub);
644 if(e < minKinEnergy) { x *= std::sqrt(e/minKinEnergy); }
645 return x;
646}

669//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
670
671inline G4double G4VEnergyLossProcess::GetScaledRangeForScaledEnergy(G4double e)
672{
673 //G4cout << "G4VEnergyLossProcess::GetScaledRange: Idx= "
674 // << basedCoupleIndex << " E(MeV)= " << e
675 // << " lastIdx= " << lastIdx << " " << theRangeTableForLoss << G4endl;
676 if(basedCoupleIndex != lastIdx || preStepRangeEnergy != e) {
677 lastIdx = basedCoupleIndex;
678 preStepRangeEnergy = e;
679 computedRange =
680 ((*theRangeTableForLoss)[basedCoupleIndex])->Value(e, idxRange);
681 if(e < minKinEnergy) { computedRange *= std::sqrt(e/minKinEnergy); }
682 }
683 //G4cout << "G4VEnergyLossProcess::GetScaledRange: Idx= "
684 // << basedCoupleIndex << " E(MeV)= " << e
685 // << " R= " << fRange << " " << theRangeTableForLoss << G4endl;
686
687 return computedRange;
688}

690//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
691
692inline G4double
693G4VEnergyLossProcess::GetLimitScaledRangeForScaledEnergy(G4double e)
694{
695 G4double x;
696 if (e < maxKinEnergyCSDA) {
697 x = ((*theCSDARangeTable)[basedCoupleIndex])->Value(e, idxCSDA);
698 if(e < minKinEnergy) { x *= std::sqrt(e/minKinEnergy); }
699 } else {
700 x = theRangeAtMaxEnergy[basedCoupleIndex] +
701 (e - maxKinEnergyCSDA)/theDEDXAtMaxEnergy[basedCoupleIndex];
702 }
703 return x;
704}

708inline G4double G4VEnergyLossProcess::ScaledKinEnergyForLoss(G4double r)

710 //G4cout << "G4VEnergyLossProcess::GetEnergy: Idx= "
711 // << basedCoupleIndex << " R(mm)= " << r << " "
712 // << theInverseRangeTable << G4endl;
713 G4PhysicsVector* v = (*theInverseRangeTable)[basedCoupleIndex];
714 G4double rmin = v->Energy(0);
715 G4double e = 0.0;
716 if(r >= rmin) { e = v->Value(r, idxInverseRange); }
717 else if(r > 0.0) {
718 G4double x = r/rmin;
719 e = minKinEnergy*x*x;
720 }
721 return e;
722}
