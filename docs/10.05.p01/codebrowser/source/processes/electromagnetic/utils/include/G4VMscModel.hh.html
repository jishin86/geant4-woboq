266inline G4double
267G4VMscModel::GetDEDX(const G4ParticleDefinition* part,
268 G4double kinEnergy, const G4MaterialCutsCouple* couple)
269{
270 G4double x;
271 if(ionisation) { x = ionisation->GetDEDX(kinEnergy, couple); }
272 else {
273 G4double q = part->GetPDGCharge()*inveplus;
274 x = dedx*q*q;
275 }
276 return x;
277}

281inline G4double
282G4VMscModel::GetRange(const G4ParticleDefinition* part,
283 G4double kinEnergy, const G4MaterialCutsCouple* couple)
284{
285 //G4cout << "G4VMscModel::GetRange E(MeV)= " << kinEnergy << " "
286 // << ionisation << " " << part->GetParticleName()
287 // << G4endl;
288 localtkin = kinEnergy;
289 if(ionisation) {
290 localrange = ionisation->GetRangeForLoss(kinEnergy, couple);
291 } else {
292 G4double q = part->GetPDGCharge()*inveplus;
293 localrange = kinEnergy/(dedx*q*q*couple->GetMaterial()->GetDensity());
294 }
295 //G4cout << "R(mm)= " << localrange << " " << ionisation << G4endl;
296 return localrange;
297}

299//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo......

301inline G4double
302G4VMscModel::GetEnergy(const G4ParticleDefinition* part,
303 G4double range, const G4MaterialCutsCouple* couple)
304{
305 G4double e;
306 //G4cout << "G4VMscModel::GetEnergy R(mm)= " << range << " " << ionisation
307 // << " Rlocal(mm)= " << localrange << " Elocal(MeV)= " << localtkin
308 // << G4endl;
309 if(ionisation) { e = ionisation->GetKineticEnergy(range, couple); }
310 else {
311 e = localtkin;
312 if(localrange > range) {
313 G4double q = part->GetPDGCharge()*inveplus;
314 e -= (localrange - range)*dedx*q*q*couple->GetMaterial()->GetDensity();
315 }
316 }
317 return e;
318}

338inline G4double
339G4VMscModel::GetTransportMeanFreePath(const G4ParticleDefinition* part,
340 G4double ekin)
341{
342 G4double x;
343 if(xSectionTable) {
344 G4int idx = CurrentCouple()->GetIndex();
345 x = (*xSectionTable)[(*theDensityIdx)[idx]]->Value(ekin, idxTable)
346 *(*theDensityFactor)[idx]/(ekin*ekin);
347 } else {
348 x = CrossSectionPerVolume(CurrentCouple()->GetMaterial(), part, ekin,
349 0.0, DBL_MAX);
350 }
351 return (x > 0.0) ? 1.0/x : DBL_MAX;
352}
