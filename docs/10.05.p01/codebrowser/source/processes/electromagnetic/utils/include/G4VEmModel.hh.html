494//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
495
496inline G4double G4VEmModel::ComputeDEDX(const G4MaterialCutsCouple* couple,
497 const G4ParticleDefinition* part,
498 G4double kinEnergy,
499 G4double cutEnergy)
500{
501 SetCurrentCouple(couple);
502 return ComputeDEDXPerVolume(couple->GetMaterial(),part,kinEnergy,cutEnergy);
503}
504
505//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
506
507inline G4double G4VEmModel::CrossSection(const G4MaterialCutsCouple* couple,
508 const G4ParticleDefinition* part,
509 G4double kinEnergy,
510 G4double cutEnergy,
511 G4double maxEnergy)
512{
513 SetCurrentCouple(couple);
514 return CrossSectionPerVolume(couple->GetMaterial(),part,kinEnergy,
515 cutEnergy,maxEnergy);
516}
517
518//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
519
520inline
521G4double G4VEmModel::ComputeMeanFreePath(const G4ParticleDefinition* part,
522 G4double ekin,
523 const G4Material* material,
524 G4double emin,
525 G4double emax)
526{
527 G4double cross = CrossSectionPerVolume(material,part,ekin,emin,emax);
528 return (cross > 0.0) ? 1./cross : DBL_MAX;
529}
530
531//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
532
533inline G4double
534G4VEmModel::ComputeCrossSectionPerAtom(const G4ParticleDefinition* part,
535 const G4Element* elm,
536 G4double kinEnergy,
537 G4double cutEnergy,
538 G4double maxEnergy)
539{
540 SetCurrentElement(elm);
541 return ComputeCrossSectionPerAtom(part,kinEnergy,elm->GetZ(),elm->GetN(),
542 cutEnergy,maxEnergy);
543}
544
545//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
546
547inline const G4Element*
548G4VEmModel::SelectRandomAtom(const G4MaterialCutsCouple* couple,
549 const G4ParticleDefinition* part,
550 G4double kinEnergy,
551 G4double cutEnergy,
552 G4double maxEnergy)
553{
554 fCurrentCouple = couple;
555 fCurrentElement = (nSelectors > 0) ?
556 ((*elmSelectors)[couple->GetIndex()])->SelectRandomAtom(kinEnergy) :
557 SelectRandomAtom(couple->GetMaterial(),part,kinEnergy,cutEnergy,maxEnergy);
558 fCurrentIsotope = nullptr;
559 return fCurrentElement;
560}
