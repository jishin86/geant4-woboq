69G4EmParameters* G4EmParameters::Instance()
70{
71 if(nullptr == theInstance) {
72#ifdef G4MULTITHREADED
73 G4MUTEXLOCK(&emParametersMutex);
74 if(nullptr == theInstance) {
75#endif
76 static G4EmParameters manager;
77 theInstance = &manager;
78#ifdef G4MULTITHREADED
79 }
80 G4MUTEXUNLOCK(&emParametersMutex);
81#endif
82 }
83 return theInstance;
84}

457void G4EmParameters::SetGeneralProcessActive(G4bool val)
458{
459 if(IsLocked()) { return; }
460 gener = val;
461 // if general interaction is enabled then sub-cutoff and
462 // force interaction options should be disabled
463 if(gener) {
464 m_regnamesForced.clear();
465 m_procForced.clear();
466 m_lengthForced.clear();
467 m_weightForced.clear();
468 m_regnamesSubCut.clear();
469 m_subCuts.clear();
470 }
471}

533void G4EmParameters::SetMinEnergy(G4double val)
534{
535 if(IsLocked()) { return; }
536 if(val > 1.e-3*eV && val < maxKinEnergy) {
537 minKinEnergy = val;
538 nbins = nbinsPerDecade*G4lrint(std::log10(maxKinEnergy/minKinEnergy));
539 } else {
540 G4ExceptionDescription ed;
541 ed << "Value of MinKinEnergy is out of range: " << val/MeV
542 << " MeV is ignored";
543 PrintWarning(ed);
544 }
545}

862void G4EmParameters::SetNumberOfBins(G4int val)
863{
864 if(IsLocked()) { return; }
865 if(val >= 5 && val < 10000000) {
866 nbins = val;
867 nbinsPerDecade = G4lrint(nbins/std::log10(maxKinEnergy/minKinEnergy));
868 } else {
869 G4ExceptionDescription ed;

1261void G4EmParameters::DefineRegParamForLoss(G4VEnergyLossProcess* ptr,
1262 G4bool isElectron) const
1263{
1264 if(isElectron) { ptr->SetStepFunction(dRoverRange, finalRange, false); }
1265 else { ptr->SetStepFunction(dRoverRangeMuHad, finalRangeMuHad, false); }
1266
1267 G4RegionStore* regionStore = G4RegionStore::GetInstance();
1268 G4int n = m_regnamesSubCut.size();
1269 for(G4int i=0; i<n; ++i) {
1270 const G4Region* reg = regionStore->GetRegion(m_regnamesSubCut[i], false);
1271 if(reg) { ptr->ActivateSubCutoff(m_subCuts[i], reg); }
1272 }
1273 n = m_procBiasedXS.size();
1274 for(G4int i=0; i<n; ++i) {
1275 if(ptr->GetProcessName() == m_procBiasedXS[i]) {
1276 ptr->SetCrossSectionBiasingFactor(m_factBiasedXS[i],
1277 m_weightBiasedXS[i]);
1278 break;
1279 }
1280 }
1281 n = m_procForced.size();
1282 for(G4int i=0; i<n; ++i) {
1283 if(ptr->GetProcessName() == m_procForced[i]) {
1284 ptr->ActivateForcedInteraction(m_lengthForced[i],
1285 m_regnamesForced[i],
1286 m_weightForced[i]);
1287 break;
1288 }
1289 }
1290 n = m_procBiasedSec.size();
1291 for(G4int i=0; i<n; ++i) {
1292 if(ptr->GetProcessName() == m_procBiasedSec[i]) {
1293 ptr->ActivateSecondaryBiasing(m_regnamesBiasedSec[i],
1294 m_factBiasedSec[i],
1295 m_elimBiasedSec[i]);
1296 break;
1297 }
1298 }
