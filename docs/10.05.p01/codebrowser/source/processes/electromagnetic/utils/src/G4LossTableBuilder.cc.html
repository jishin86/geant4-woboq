94void
95G4LossTableBuilder::BuildDEDXTable(G4PhysicsTable* dedxTable,
96 const std::vector<G4PhysicsTable*>& list)
97{
98 size_t n_processes = list.size();
99 //G4cout << "Nproc= " << n_processes << " Ncoup= "
100 //<< dedxTable->size() << G4endl;
101 if(1 >= n_processes) { return; }
102
103 size_t nCouples = dedxTable->size();
104 if(0 >= nCouples) { return; }
105
106 for (size_t i=0; i<nCouples; ++i) {
107 // if ((*theFlag)[i]) {
108 G4PhysicsLogVector* pv0 =
109 static_cast<G4PhysicsLogVector*>((*(list[0]))[i]);
110 if(pv0) {
111 size_t npoints = pv0->GetVectorLength();
112 G4PhysicsLogVector* pv = new G4PhysicsLogVector(*pv0);
113 pv->SetSpline(splineFlag);
114 for (size_t j=0; j<npoints; ++j) {
115 G4double dedx = 0.0;
116 for (size_t k=0; k<n_processes; ++k) {
117 G4PhysicsVector* pv1 = (*(list[k]))[i];
118 dedx += (*pv1)[j];
119 }
120 pv->PutValue(j, dedx);
121 }
122 if(splineFlag) { pv->FillSecondDerivatives(); }
123 G4PhysicsTableHelper::SetPhysicsVector(dedxTable, i, pv);
124 }
125 }
126}

130void G4LossTableBuilder::BuildRangeTable(const G4PhysicsTable* dedxTable,
131 G4PhysicsTable* rangeTable,
132 G4bool isIonisation)
133// Build range table from the energy loss table
134{
135 size_t nCouples = dedxTable->size();
136 if(0 >= nCouples) { return; }
137
138 size_t n = 100;
139 G4double del = 1.0/(G4double)n;
140
141 for (size_t i=0; i<nCouples; ++i) {
142 if(isIonisation) {
143 if( !(*theFlag)[i] ) { continue; }
144 }
145 G4PhysicsLogVector* pv = static_cast<G4PhysicsLogVector*>((*dedxTable)[i]);
146 size_t npoints = pv->GetVectorLength();
147 size_t bin0 = 0;
148 G4double elow = pv->Energy(0);
149 G4double ehigh = pv->Energy(npoints-1);
150 G4double dedx1 = (*pv)[0];
151
152 //G4cout << "i= " << i << "npoints= " << npoints << " dedx1= "
153 //<< dedx1 << G4endl;
154
155 // protection for specific cases dedx=0
156 if(dedx1 == 0.0) {
157 for (size_t k=1; k<npoints; ++k) {
158 bin0++;
159 elow = pv->Energy(k);
160 dedx1 = (*pv)[k];
161 if(dedx1 > 0.0) { break; }
162 }
163 npoints -= bin0;
164 }
165 //G4cout<<"New Range vector" << G4endl;
166 //G4cout<<"nbins= "<<npoints-1<<" elow= "<<elow<<" ehigh= "<<ehigh
167 // <<" bin0= " << bin0 <<G4endl;
168
169 // initialisation of a new vector
170 if(npoints < 2) { npoints = 2; }
171
172 delete (*rangeTable)[i];
173 G4PhysicsLogVector* v;
174 if(0 == bin0) { v = new G4PhysicsLogVector(*pv); }
175 else { v = new G4PhysicsLogVector(elow, ehigh, npoints-1); }
176
177 // dedx is exact zero cannot build range table
178 if(2 == npoints) {
179 v->PutValue(0,1000.);
180 v->PutValue(1,2000.);
181 G4PhysicsTableHelper::SetPhysicsVector(rangeTable, i, v);
182 return;
183 }
184 v->SetSpline(splineFlag);
185
186 // assumed dedx proportional to beta
187 G4double energy1 = v->Energy(0);
188 G4double range = 2.*energy1/dedx1;
189 //G4cout << "range0= " << range << G4endl;
190 v->PutValue(0,range);
191
192 for (size_t j=1; j<npoints; ++j) {
193
194 G4double energy2 = v->Energy(j);
195 G4double de = (energy2 - energy1) * del;
196 G4double energy = energy2 + de*0.5;
197 G4double sum = 0.0;
198 //G4cout << "j= " << j << " e1= " << energy1 << " e2= " << energy2
199 // << " n= " << n << G4endl;
200 for (size_t k=0; k<n; ++k) {
201 energy -= de;
202 dedx1 = pv->Value(energy);
203 if(dedx1 > 0.0) { sum += de/dedx1; }
204 }
205 range += sum;
206 v->PutValue(j,range);
207 energy1 = energy2;
208 }
209 if(splineFlag) { v->FillSecondDerivatives(); }
210 G4PhysicsTableHelper::SetPhysicsVector(rangeTable, i, v);
211 }
212}

216void
217G4LossTableBuilder::BuildInverseRangeTable(const G4PhysicsTable* rangeTable,
218 G4PhysicsTable* invRangeTable,
219 G4bool isIonisation)
220// Build inverse range table from the energy loss table
221{
222 size_t nCouples = rangeTable->size();
223 if(0 >= nCouples) { return; }
224
225 for (size_t i=0; i<nCouples; ++i) {
226
227 if(isIonisation) {
228 if( !(*theFlag)[i] ) { continue; }
229 }
230 G4PhysicsVector* pv = (*rangeTable)[i];
231 size_t npoints = pv->GetVectorLength();
232 G4double rlow = (*pv)[0];
233 G4double rhigh = (*pv)[npoints-1];
234
235 delete (*invRangeTable)[i];
236 G4LPhysicsFreeVector* v = new G4LPhysicsFreeVector(npoints,rlow,rhigh);
237 v->SetSpline(splineFlag);
238
239 for (size_t j=0; j<npoints; ++j) {
240 G4double e = pv->Energy(j);
241 G4double r = (*pv)[j];
242 v->PutValues(j,r,e);
243 }
244 if(splineFlag) { v->FillSecondDerivatives(); }
245
246 G4PhysicsTableHelper::SetPhysicsVector(invRangeTable, i, v);
247 }

252void
253G4LossTableBuilder::InitialiseBaseMaterials(G4PhysicsTable* table)
254{
255 size_t nCouples = table->size();
256 size_t nFlags = theFlag->size();
257
258 if(nCouples == nFlags && isInitialized) { return; }
259
260 isInitialized = true;
261
262 //G4cout << "%%%%%% G4LossTableBuilder::InitialiseBaseMaterials Ncouples= "
263 // << nCouples << " FlagSize= " << nFlags << G4endl;
264
265 // variable density check
266 const G4ProductionCutsTable* theCoupleTable=
267 G4ProductionCutsTable::GetProductionCutsTable();
268
269 /*
270 for(size_t i=0; i<nFlags; ++i) {
271 G4cout << "CoupleIdx= " << i << " Flag= " << (*theFlag)[i]
272 << " tableFlag= " << table->GetFlag(i) << " "
273 << theCoupleTable->GetMaterialCutsCouple(i)->GetMaterial()->GetName()
274 << G4endl;
275 }
276 */
277
278 // expand vectors
279 if(nFlags < nCouples) {
280 for(size_t i=nFlags; i<nCouples; ++i) {
281 theDensityFactor->push_back(1.0);
282 }
283 for(size_t i=nFlags; i<nCouples; ++i) { theDensityIdx->push_back(-1); }
284 for(size_t i=nFlags; i<nCouples; ++i) { theFlag->push_back(true); }
285 }
286 for(size_t i=0; i<nCouples; ++i) {
287
288 // base material is needed only for a couple which is not
289 // initialised and for which tables will be computed
290 (*theFlag)[i] = table->GetFlag(i);
291 if ((*theDensityIdx)[i] < 0) {
292 (*theDensityIdx)[i] = i;
293 const G4MaterialCutsCouple* couple =
294 theCoupleTable->GetMaterialCutsCouple(i);
295 const G4ProductionCuts* pcuts = couple->GetProductionCuts();
296 const G4Material* mat = couple->GetMaterial();
297 const G4Material* bmat = mat->GetBaseMaterial();
298
299 // base material exists - find it and check if it can be reused
300 if(bmat) {
301 for(size_t j=0; j<nCouples; ++j) {
302
303 if(j == i) { continue; }
304 const G4MaterialCutsCouple* bcouple =
305 theCoupleTable->GetMaterialCutsCouple(j);
306
307 if(bcouple->GetMaterial() == bmat &&
308 bcouple->GetProductionCuts() == pcuts) {
309
310 // based couple exist in the same region
311 (*theDensityIdx)[i] = j;
312 (*theDensityFactor)[i] = mat->GetDensity()/bmat->GetDensity();
313 (*theFlag)[i] = false;
314
315 // ensure that there will no double initialisation
316 (*theDensityIdx)[j] = j;
317 (*theDensityFactor)[j] = 1.0;
318 (*theFlag)[j] = true;
319 break;
320 }
321 }
322 }
323 }
324 }
325 /*
326 for(size_t i=0; i<nCouples; ++i) {
327 G4cout << "CoupleIdx= " << i << " Flag= " << (*theFlag)[i]
328 << " TableFlag= " << table->GetFlag(i) << " "
329 << theCoupleTable->GetMaterialCutsCouple(i)->GetMaterial()->GetName()
330 << G4endl;
331 }
332 G4cout << "%%%%%% G4LossTableBuilder::InitialiseBaseMaterials end"
333 << G4endl;
334 */
335}

339void G4LossTableBuilder::InitialiseCouples()
340{
341 isInitialized = true;
342
343 // variable density initialisation for the cas without tables
344 const G4ProductionCutsTable* theCoupleTable=
345 G4ProductionCutsTable::GetProductionCutsTable();
346 size_t nCouples = theCoupleTable->GetTableSize();
347 //G4cout << "%%%%%% G4LossTableBuilder::InitialiseCouples() nCouples= "
348 // << nCouples << G4endl;
349
350 theDensityFactor->resize(nCouples, 1.0);
351 theDensityIdx->resize(nCouples, -1);
352 theFlag->resize(nCouples, true);
353
354 for(size_t i=0; i<nCouples; ++i) {
355
356 // base material is needed only for a couple which is not
357 // initialised and for which tables will be computed
358 if ((*theDensityIdx)[i] < 0) {
359 (*theDensityIdx)[i] = i;
360 const G4MaterialCutsCouple* couple =
361 theCoupleTable->GetMaterialCutsCouple(i);
362 const G4ProductionCuts* pcuts = couple->GetProductionCuts();
363 const G4Material* mat = couple->GetMaterial();
364 const G4Material* bmat = mat->GetBaseMaterial();
365
366 // base material exists - find it and check if it can be reused
367 if(bmat) {
368 for(size_t j=0; j<nCouples; ++j) {
369
370 if(j == i) { continue; }
371 const G4MaterialCutsCouple* bcouple =
372 theCoupleTable->GetMaterialCutsCouple(j);
373
374 if(bcouple->GetMaterial() == bmat &&

376
377 // based couple exist in the same region

380 (*theFlag)[i] = false;

405G4PhysicsTable*
406G4LossTableBuilder::BuildTableForModel(G4PhysicsTable* aTable,
407 G4VEmModel* model,
408 const G4ParticleDefinition* part,
409 G4double emin, G4double emax,
410 G4bool spline)
411{
412 // check input
413 G4PhysicsTable* table = G4PhysicsTableHelper::PreparePhysicsTable(aTable);
414 if(!table) { return table; }
415 if(emin >= emax) {
416 table->clearAndDestroy();
417 delete table;
418 table = 0;
419 return table;
420 }
421 InitialiseBaseMaterials(table);
422
423 G4int nbins = theParameters->NumberOfBinsPerDecade();
424
425 // Access to materials
426 const G4ProductionCutsTable* theCoupleTable=
427 G4ProductionCutsTable::GetProductionCutsTable();
428 size_t numOfCouples = theCoupleTable->GetTableSize();
429
430 G4PhysicsLogVector* aVector = 0;
431
432 for(size_t i=0; i<numOfCouples; ++i) {
433
434 //G4cout<< "i= " << i << " Flag= " << GetFlag(i) << G4endl;
435
436 if (GetFlag(i)) {
437
438 // create physics vector and fill it
439 const G4MaterialCutsCouple* couple =
440 theCoupleTable->GetMaterialCutsCouple(i);
441 delete (*table)[i];
442
443 // if start from zero then change the scale
444
445 const G4Material* mat = couple->GetMaterial();
446
447 G4double tmin = std::max(emin,model->MinPrimaryEnergy(mat,part));
448 if(0.0 >= tmin) { tmin = eV; }
449 G4int n = nbins;
450
451 if(tmin >= emax) {
452 aVector = 0;
453 } else {
454 n *= G4int(std::log10(emax/tmin) + 0.5);
455 if(n < 3) { n = 3; }
456 aVector = new G4PhysicsLogVector(tmin, emax, n);
457 }
458
459 if(aVector) {
460 aVector->SetSpline(spline);
461 for(G4int j=0; j<=n; ++j) {
462 aVector->PutValue(j, model->Value(couple, part,
463 aVector->Energy(j)));
464 }
465 if(spline) { aVector->FillSecondDerivatives(); }
466 }
467 G4PhysicsTableHelper::SetPhysicsVector(table, i, aVector);
468 }
469 }
470 /*
471 G4cout << "G4LossTableBuilder::BuildTableForModel done for "
472 << part->GetParticleName() << " and "<< model->GetName()
473 << " " << table << G4endl;
474 */
475 return table;
476}
