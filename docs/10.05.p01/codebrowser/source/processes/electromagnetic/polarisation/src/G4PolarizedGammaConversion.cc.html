51G4PolarizedGammaConversion::G4PolarizedGammaConversion(const G4String& processName,
52 G4ProcessType type):G4VEmProcess (processName, type),
53 isInitialised(false)
54{
55 SetMinKinEnergy(2.0*electron_mass_c2);
56 SetLambdaBinning(220);
57 //SetMaxKinEnergy(100.0*GeV);
58 SetProcessSubType(fGammaConversion);
59 SetBuildTableFlag(true);
60 SetSecondaryParticle(G4Electron::Electron());
61}

70G4bool G4PolarizedGammaConversion::IsApplicable(const G4ParticleDefinition& p)
71{
72 return (&p == G4Gamma::Gamma());
73}
74
75//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
76
77void G4PolarizedGammaConversion::InitialiseProcess(const G4ParticleDefinition*)
78{
79 if(!isInitialised) {
80 isInitialised = true;
81 G4EmParameters* param = G4EmParameters::Instance();
82 G4double emin = std::max(param->MinKinEnergy(), 2*electron_mass_c2);
83 G4double emax = param->MaxKinEnergy();
84 if(!EmModel(0)) { SetEmModel(new G4PolarizedGammaConversionModel()); }
85 EmModel(0)->SetLowEnergyLimit(emin);
86 EmModel(0)->SetHighEnergyLimit(emax);
87 AddEmModel(1, EmModel(0));
88 }
89}
