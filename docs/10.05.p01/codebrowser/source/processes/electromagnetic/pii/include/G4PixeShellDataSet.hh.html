45#ifndef G4PIXESHELLDATASET_HH
46#define G4PIXESHELLDATASET_HH 1
47
48#include <vector>
49#include <CLHEP/Units/SystemOfUnits.h>
50
51#include "globals.hh"
52#include "G4IDataSet.hh"
53
54class G4IInterpolator;
55
56class G4PixeShellDataSet : public G4IDataSet
57{
58public:
59 G4PixeShellDataSet(G4int Z,
60 G4IInterpolator* algo,
61 //const G4String& particleTpye="proton",
62 const G4String& modelK="ecpssr",
63 const G4String& modelL="ecpssr",
64 const G4String& modelM="ecpssr",
65 G4double eUnit=CLHEP::MeV,
66 G4double dataUnit=CLHEP::barn);
67
68 virtual ~G4PixeShellDataSet();
69
70 virtual G4double FindValue(G4double energy, G4int componentId=0) const;
71
72 virtual void PrintData(void) const;
73
74 virtual const G4IDataSet* GetComponent(G4int componentId) const
75 { return components[componentId]; }
76
77 virtual void AddComponent(G4IDataSet* dataSet)
78 { components.push_back(dataSet); }
79
80 virtual size_t NumberOfComponents(void) const
81 { return components.size(); }
82
83 virtual const G4DataVector& GetEnergies(G4int componentId) const
84 { return GetComponent(componentId)->GetEnergies(0); }
85
86 virtual const G4DataVector& GetData(G4int componentId) const
87 { return GetComponent(componentId)->GetData(0); }
88
89 virtual void SetEnergiesData(G4DataVector* energies,
90 G4DataVector* data,
91 G4int componentId);
92
93 virtual G4bool LoadData(const G4String& fileName);
94 virtual G4bool SaveData(const G4String& fileName) const;
95
96 virtual G4double RandomSelect(G4int /*componentId = 0*/) const {return -1.;};
97
98protected:
99
100 G4double GetUnitEnergies() const { return unitEnergies; }
101 G4double GetUnitData() const { return unitData; }
102 const G4IInterpolator* GetAlgorithm() const { return algorithm; }
103
104 void CleanUpComponents(void);
105
106private:
107
108 G4String FullFileName(const G4String& particleType,
109 const G4String& subShell) const;
110 G4int TranslateShell(const G4String& subShell) const;
111
112 // Hide copy constructor and assignment operator
113 G4PixeShellDataSet();
114 G4PixeShellDataSet(const G4PixeShellDataSet& copy);
115 G4PixeShellDataSet& operator=(const G4PixeShellDataSet& right);
116
117 std::vector<G4IDataSet*> components; // Owned pointers
118
119 G4int z;
120 G4IInterpolator* algorithm; // Owned pointer
121 // G4String particle;
122 // G4String crossModelK;
123 // G4String crossModelL;
124 // G4String crossModelM;
125 std::vector<G4String> crossModel;
126 G4double unitEnergies;
127 G4double unitData;
128 std::vector<G4String> shellName;
129 std::vector<G4String> subShellName;
130
131};
132#endif /* G4PIXESHELLDATASET_HH */
