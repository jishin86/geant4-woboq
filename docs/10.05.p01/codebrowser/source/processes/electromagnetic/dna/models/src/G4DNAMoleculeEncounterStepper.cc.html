57G4DNAMoleculeEncounterStepper::Utils::Utils(const G4Track& tA,
58 G4MolecularConfiguration* mB) :
59 trackA(tA), moleculeB(mB)
60{
61 moleculeA = GetMolecule(tA);
62 DA = moleculeA->GetDiffusionCoefficient();
63 DB = moleculeB->GetDiffusionCoefficient();
64 Constant = 8 * (DA + DB + 2 * sqrt(DA * DB));
65}

141G4double
142G4DNAMoleculeEncounterStepper::CalculateStep(const G4Track& trackA,
143 const G4double& userMinTimeStep)
144{
145 // DEBUG
146// G4cout << "G4MoleculeEncounterStepper::CalculateStep, time :"
147// << G4ITTrackHolder::Instance()->GetGlobalTime() << G4endl;
148
149 G4Molecule* moleculeA = GetMolecule(trackA);
150 InitializeForNewTrack();
151 fUserMinTimeStep = userMinTimeStep;
152
153#ifdef G4VERBOSE
154 if (fVerbose)
155 {
156 G4cout
157 << "_______________________________________________________________________"
158 << G4endl;
159 G4cout << "G4DNAMoleculeEncounterStepper::CalculateStep" << G4endl;
160 G4cout << "Check done for molecule : " << moleculeA->GetName()
161 << " (" << trackA.GetTrackID() << ") "
162 << G4endl;
163 }
164#endif
165
166 //__________________________________________________________________
167 // Retrieve general informations for making reactions
168 G4MolecularConfiguration* molConfA = moleculeA->GetMolecularConfiguration();
169
170 const vector<G4MolecularConfiguration*>* reactivesVector = fMolecularReactionTable
171 ->CanReactWith(molConfA);
172
173 if (!reactivesVector)
174 {
175#ifdef G4VERBOSE
176 // DEBUG
177 if (fVerbose > 1)
178 {
179 G4cout << "!!!!!!!!!!!!!!!!!!!!" << G4endl;
180 G4cout << "!!! WARNING" << G4endl;
181 G4cout << "G4MoleculeEncounterStepper::CalculateStep will return infinity "
182 "for the reaction because the molecule "
183 << moleculeA->GetName()
184 << " does not have any reactants given in the reaction table."
185 << G4endl;
186 G4cout << "!!!!!!!!!!!!!!!!!!!!"<<G4endl;
187 }
188#endif
189 return DBL_MAX;
190 }
191
192 G4int nbReactives = reactivesVector->size();
193
194 if (nbReactives == 0)
195 {
196#ifdef G4VERBOSE
197 // DEBUG
198 if (fVerbose)
199 {
200 // TODO replace with the warning mode of G4Exception
201 G4cout << "!!!!!!!!!!!!!!!!!!!!" << G4endl;
202 G4cout << "!!! WARNING" << G4endl;
203 G4cout << "G4MoleculeEncounterStepper::CalculateStep will return infinity "
204 "for the reaction because the molecule "
205 << moleculeA->GetName()
206 << " does not have any reactants given in the reaction table."
207 << "This message can also result from a wrong implementation of the reaction table."
208 << G4endl;
209 G4cout << "!!!!!!!!!!!!!!!!!!!!"<<G4endl;
210 }
211#endif
212 return DBL_MAX;
213 }
214 // DEBUG
215 // else
216 // {
217 // G4cout << "nb reactants : " << nbReactives << " pour mol "<< moleculeA -> GetName () << G4endl;
218 // for(int k=0 ; k < nbReactives ; k++)
219 // {
220 // G4cout << (*reactivesVector)[k]->GetName() << G4endl;
221 // }
222 // }
223
224// fReactants = new vector<G4Track*>();
225 fReactants.reset(new vector<G4Track*>());
226 fReactionModel->Initialise(molConfA, trackA);
227
228 //__________________________________________________________________
229 // Start looping on possible reactants
230 for (G4int i = 0; i < nbReactives; i++)
231 {
232 G4MolecularConfiguration* moleculeB = (*reactivesVector)[i];
233
234 //______________________________________________________________
235 // Retrieve reaction range
236 const G4double R = fReactionModel->GetReactionRadius(i);
237
238 //G4cout << "Reaction range = " << G4BestUnit(R, "Length") << G4endl;
239
240 //______________________________________________________________
241 // Use KdTree algorithm to find closest reactants
242 G4KDTreeResultHandle resultsNearest(
243 G4MoleculeFinder::Instance()->FindNearest(moleculeA,
244 moleculeB->GetMoleculeID()));
245
246 if (resultsNearest == 0) continue;
247
248 G4double r2 = resultsNearest->GetDistanceSqr();
249 Utils utils(trackA, moleculeB);
250
251 if (r2 <= R * R) // ==> Record in range
252 {
253 // Entering in this condition may due to the fact that molecules are very close
254 // to each other
255 // Therefore, if we only take the nearby reactant into account, it might have already
256 // reacted. Instead, we will take all possible reactants that satisfy the condition r<R
257
258 if (fHasAlreadyReachedNullTime == false)
259 {
260 fReactants->clear();
261 fHasAlreadyReachedNullTime = true;
262 }
263
264 fSampledMinTimeStep = 0.;
265 G4KDTreeResultHandle resultsInRange(
266 G4MoleculeFinder::Instance()->FindNearestInRange(moleculeA,
267 moleculeB->GetMoleculeID(),
268 R));
269 CheckAndRecordResults(utils,
270#ifdef G4VERBOSE
271 R,
272#endif
273 resultsInRange);
274 }
275 else
276 {
277 G4double r = sqrt(r2);
278 G4double tempMinET = pow(r - R, 2) / utils.Constant;
279 // constant = 16 * (DA + DB + 2*sqrt(DA*DB))
280
281 //G4cout << tempMinET << G4endl;
282// G4cout << "fSampledMinTimeStep =" << fSampledMinTimeStep << G4endl;
283// G4cout << "fUserMinTimeStep =" << fUserMinTimeStep
284// << " isInf = "<< IsInf(fUserMinTimeStep) << G4endl;
285
286 if (tempMinET <= fSampledMinTimeStep)
287 {
288 if (fUserMinTimeStep < DBL_MAX/*IsInf(fUserMinTimeStep) == false*/
289 && tempMinET <= fUserMinTimeStep) // ==> Record in range
290 {
291 if(fSampledMinTimeStep > fUserMinTimeStep)
292 {
293 fReactants->clear();
294 }
295
296 fSampledMinTimeStep = fUserMinTimeStep;
297
298 G4double range = R + sqrt(fUserMinTimeStep*utils.Constant);
299
300 G4KDTreeResultHandle resultsInRange (
301 G4MoleculeFinder::Instance()->
302 FindNearestInRange(moleculeA,
303 moleculeB->GetMoleculeID(),
304 range));
305
306 CheckAndRecordResults(utils,
307#ifdef G4VERBOSE
308 range,
309#endif
310 resultsInRange);
311 }
312 else // ==> Record nearest
313 {
314 if(tempMinET < fSampledMinTimeStep)
315 // to avoid cases where fSampledMinTimeStep == tempMinET
316 {
317 fSampledMinTimeStep = tempMinET;
318 fReactants->clear();
319 }
320
321 CheckAndRecordResults(utils,
322#ifdef G4VERBOSE
323 R,
324#endif
325 resultsNearest);
326 }
327 }
328 }
329
330 // DEBUG
331// if(bool(fReactants))
332// {
333// G4cout << "Potential reactions :" << G4endl;
334// G4cout << GetMolecule(trackA)->GetName()
335// << " ("<< trackA.GetTrackID()<< ") " << " + ..." << G4endl;
336// //<< " | " << trackB->GetTrackID() << G4endl;
337//
338// for(int j = 0 ; j < (int) fReactants->size() ; j++)
339// {
340// G4cout << GetMolecule(fReactants->at(j) )->GetName()
341// <<" ("<< fReactants->at(j)->GetTrackID() << ")" << G4endl;
342// }
343// }
344
345 }
346
347#ifdef G4VERBOSE
348 // DEBUG
349 if (fVerbose)
350 {
351 G4cout << "G4MoleculeEncounterStepper::CalculateStep will finally return :"
352 << G4BestUnit(fSampledMinTimeStep, "Time") << G4endl;
353
354 if(fVerbose > 1)
355 {
356 G4cout << "Selected reactants for trackA: " << moleculeA->GetName()
357 << " (" << trackA.GetTrackID() << ") are: ";
358
359 vector<G4Track*>::iterator it;
360 for(it = fReactants->begin(); it != fReactants->end(); it++)
361 {
362 G4Track* trackB = *it;
363 G4cout << GetMolecule(trackB)->GetName() << " ("
364 << trackB->GetTrackID() << ") \t ";
365 }
366 G4cout << G4endl;
367 }
368 }
369#endif
370 return fSampledMinTimeStep;
371}

373void G4DNAMoleculeEncounterStepper::CheckAndRecordResults(const Utils& utils,
374#ifdef G4VERBOSE
375 const G4double R,
376#endif
377 G4KDTreeResultHandle& results)
378{
379 if (results == 0)
380 {
381#ifdef G4VERBOSE
382 // DEBUG
383 if (fVerbose > 1)
384 {
385 G4cout << "No molecule " << utils.moleculeB->GetName()
386 << " found to react with " << utils.moleculeA->GetName()
387 << G4endl;
388 }
389#endif
390 return;
391 }
392
393 for (results->Rewind(); !results->End(); results->Next())
394 {
395
396 G4IT* reactiveB = results->GetItem<G4IT>();
397
398 if (reactiveB == 0)
399 {
400 // DEBUG
401 // G4cout<<"Continue 1"<<G4endl;
402 continue;
403 }
404
405 G4Track *trackB = reactiveB->GetTrack();
406
407 if (trackB == 0)
408 {
409 G4ExceptionDescription exceptionDescription;
410 exceptionDescription
411 <<"The reactant B found using the MoleculeFinder does not have a valid "
412 "track attached to it. If this is done on purpose, please do "
413 "not record this molecule in the MoleculeFinder."
414 << G4endl;
415 G4Exception("G4DNAMoleculeEncounterStepper::RetrieveResults",
416 "MoleculeEncounterStepper001", FatalErrorInArgument,
417 exceptionDescription);
418 continue;
419 }
420
421 if (trackB->GetTrackStatus() != fAlive)
422 {
423// G4ExceptionDescription exceptionDescription;
424// exceptionDescription
425// << "The track status of one of the nearby reactants is not fAlive"
426// << G4endl;
427// exceptionDescription << "The incomming trackID "
428// << "(trackA entering in G4DNAMoleculeEncounterStepper and "
429// << "for which you are looking reactant for) is : "
430// << utils.trackA.GetTrackID() << "("
431// << GetMolecule(utils.trackA)->GetName() << ")" << G4endl;
432// exceptionDescription << "And the trackID of the reactant (trackB) is: "
433// << trackB->GetTrackID() << "(" << GetMolecule(trackB)->GetName()
434// << ")" << G4endl;
435// G4Exception("G4DNAMoleculeEncounterStepper::RetrieveResults",
436// "MoleculeEncounterStepper002", FatalErrorInArgument,
437// exceptionDescription);
438 continue;
439 }
440
441 if (trackB == &utils.trackA)
442 {
443 // DEBUG
444 G4ExceptionDescription exceptionDescription;
445 exceptionDescription
446 << "A track is reacting with itself (which is impossible) ie trackA == trackB"
447 << G4endl;
448 exceptionDescription << "Molecule A (and B) is of type : "
449 << utils.moleculeA->GetName() << " with trackID : "
450 << utils.trackA.GetTrackID() << G4endl;
451
452 G4Exception("G4DNAMoleculeEncounterStepper::RetrieveResults",
453 "MoleculeEncounterStepper003", FatalErrorInArgument,
454 exceptionDescription);
455
456 }
457
458 if (fabs(trackB->GetGlobalTime() - utils.trackA.GetGlobalTime())
459 > utils.trackA.GetGlobalTime() * (1 - 1 / 100))
460 {
461 // DEBUG
462 G4ExceptionDescription exceptionDescription;
463 exceptionDescription
464 << "The interacting tracks are not synchronized in time" << G4endl;
465 exceptionDescription
466 << "trackB->GetGlobalTime() != trackA.GetGlobalTime()" << G4endl;
467
468 exceptionDescription << "trackA : trackID : " << utils.trackA.GetTrackID()
469 << "\t Name :" << utils.moleculeA->GetName()
470 << "\t trackA->GetGlobalTime() = "
471 << G4BestUnit(utils.trackA.GetGlobalTime(), "Time") << G4endl;
472
473 exceptionDescription << "trackB : trackID : " << trackB->GetTrackID()
474 << "\t Name :" << utils.moleculeB->GetName()
475 << "\t trackB->GetGlobalTime() = "
476 << G4BestUnit(trackB->GetGlobalTime(), "Time") << G4endl;
477
478 G4Exception("G4DNAMoleculeEncounterStepper::RetrieveResults",
479 "MoleculeEncounterStepper004", FatalErrorInArgument,
480 exceptionDescription);
481 }
482
483#ifdef G4VERBOSE
484 if(fVerbose > 1)
485 {
486 G4double r2 = results->GetDistanceSqr();
487 G4cout << "\t ************************************************** " << G4endl;
488 G4cout <<"\t Reaction between "
489 << utils.moleculeA->GetName() << " (" << utils.trackA.GetTrackID() << ") "
490 << " & " << utils.moleculeB->GetName() << " (" << trackB->GetTrackID() << "), "
491 << "Interaction Range = "
492 << G4BestUnit(R, "Length")<<G4endl;
493 G4cout <<"\t Real distance between reactants = "
494 << G4BestUnit((utils.trackA.GetPosition() - trackB->GetPosition()).mag(), "Length")<<G4endl;
495 G4cout <<"\t Distance between reactants calculated by nearest neighbor algorithm = "
496 << G4BestUnit(sqrt(r2), "Length")<<G4endl;
497 // G4cout << " ***** " << G4endl;
498 }
499#endif
500
501 fReactants->push_back(trackB);
502 }
