50#include "AddClone_def.hh"
51#include "G4VDNAReactionModel.hh"
52#include <vector>
53
54class G4DNAMolecularReactionData;
55
56/**
57 * G4DNASmoluchowskiReactionModel should be used
58 * for very fast reactions (high reaction rate) : the reactions between
59 * reactants occuring at encounter.
60 * When the time step is constrained this model
61 * uses brownian bridge : "Absorbing (Smoluchowski) boundary condition"
62 * Reference : On the simulation of diffusion processes close to boundaries,
63 * N. J. B. Green, Molecular Physics, 65: 6, 1399 — 1408(1988)
64 */
65
66class G4DNASmoluchowskiReactionModel : public G4VDNAReactionModel
67{
68public :
69 G4DNASmoluchowskiReactionModel();
70 virtual ~G4DNASmoluchowskiReactionModel();
71
72 G4DNASmoluchowskiReactionModel(const G4DNASmoluchowskiReactionModel&);
73
74 G4IT_ADD_CLONE(G4VDNAReactionModel, G4DNASmoluchowskiReactionModel)
75
76 virtual void Initialise(G4MolecularConfiguration*, const G4Track&) ;
77 virtual void InitialiseToPrint(G4MolecularConfiguration*) ;
78 virtual G4double GetReactionRadius(G4MolecularConfiguration*,
79 G4MolecularConfiguration*);
80 virtual G4double GetReactionRadius(const G4int);
81
82 virtual G4bool FindReaction(const G4Track&,
83 const G4Track&,
84 const G4double /*reactionRadius*/,
85 G4double& /*separationDistance*/,
86 const G4bool /*alongStepInteraction*/) ;
87
88private :
89 const std::vector<const G4DNAMolecularReactionData*>* fReactionData ;
90 G4DNASmoluchowskiReactionModel& operator=(const G4DNASmoluchowskiReactionModel&);
91};
