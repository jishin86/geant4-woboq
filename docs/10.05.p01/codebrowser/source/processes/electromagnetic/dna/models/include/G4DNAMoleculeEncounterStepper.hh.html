46#ifndef G4MOLECULEENCOUNTERSTEPPER_H
47#define G4MOLECULEENCOUNTERSTEPPER_H
48
49#include "G4VITTimeStepComputer.hh"
50#include "G4KDTreeResult.hh"
51
52class G4VDNAReactionModel;
53class G4DNAMolecularReactionTable;
54class G4MolecularConfiguration;
55
56class G4Molecule;

58/**
59 * Given a molecule G4DNAMoleculeEncounterStepper will calculate for its possible reactants
60 * what will be the minimum encounter time and the associated molecules.*
61 *
62 * This model includes dynamical time steps as explained in
63 * "Computer-Aided Stochastic Modeling of the Radiolysis of Liquid Water",
64 * V. Michalik, M. Begusová, E. A. Bigildeev,
65 * Radiation Research, Vol. 149, No. 3 (Mar., 1998), pp. 224-236
66 *
67 */
68
69class G4DNAMoleculeEncounterStepper : public G4VITTimeStepComputer
70{
71public:
72 G4DNAMoleculeEncounterStepper();
73 virtual ~G4DNAMoleculeEncounterStepper();
74 G4DNAMoleculeEncounterStepper(const G4DNAMoleculeEncounterStepper&);
75 G4IT_ADD_CLONE(G4VITTimeStepComputer,G4DNAMoleculeEncounterStepper)
76
77 virtual void Prepare();
78 // virtual void PrepareForAllProcessors();
79 virtual G4double CalculateStep(const G4Track&, const G4double&);
80
81 inline void SetReactionModel(G4VDNAReactionModel*);
82 inline G4VDNAReactionModel* GetReactionModel();
83
84 inline void SetVerbose(int);
85 // Final time returned when reaction is avalaible in the reaction table = 1
86 // All details = 2
87
88private:
89 void InitializeForNewTrack();
90
91 class Utils;
92 void CheckAndRecordResults(const Utils&,
93#ifdef G4VERBOSE
94 const G4double reactionRange,
95#endif
96 G4KDTreeResultHandle&);
97
98 G4bool fHasAlreadyReachedNullTime;
99
100 G4DNAMoleculeEncounterStepper& operator=(const G4DNAMoleculeEncounterStepper&);
101 const G4DNAMolecularReactionTable*& fMolecularReactionTable;
102 G4VDNAReactionModel* fReactionModel;
103 G4int fVerbose;
104
105 class Utils
106 {
107 public:
108 Utils(const G4Track& tA, G4MolecularConfiguration* mB);
109 ~Utils(){;}
110
111 G4double GetConstant() const
112 {
113 return Constant;
114 }
115
116 const G4Track& trackA;
117 G4MolecularConfiguration* moleculeB;
118 const G4Molecule* moleculeA;
119 G4double DA;
120 G4double DB;
121 G4double Constant;
122 };
123
124};
125
126inline void G4DNAMoleculeEncounterStepper::SetReactionModel(G4VDNAReactionModel* reactionModel)
127{
128 fReactionModel = reactionModel;
129}
130
131inline G4VDNAReactionModel* G4DNAMoleculeEncounterStepper::GetReactionModel()
132{
133 return fReactionModel;
134}
135
136inline void G4DNAMoleculeEncounterStepper::SetVerbose(int flag)
137{
138 fVerbose = flag;
139}
