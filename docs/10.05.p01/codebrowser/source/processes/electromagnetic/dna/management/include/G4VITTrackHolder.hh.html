47#ifndef G4VITTRACKHOLDER_HH
48#define G4VITTRACKHOLDER_HH
49
50#include "G4Types.hh"
51
52class G4Track;
53
54/**
55 * G4ITTrackHolder is an empty interface that permits to push tracks to the IT system
56 * without actually depending on the IT tracking system.
57 * However, G4ITTrackHolder does not permit to retrieve any track.
58 */
59
60class G4VITTrackHolder
61{
62protected :
63 G4VITTrackHolder();
64 virtual ~G4VITTrackHolder();
65 static G4ThreadLocal G4VITTrackHolder* fInstance;
66
67public:
68 static G4VITTrackHolder* Instance();
69 virtual void Push(G4Track*);
70 inline virtual size_t GetNTracks()
71 {
72 return 0;
73 }
74};
75
76#endif // G4ITTRACKHOLDER_HH
