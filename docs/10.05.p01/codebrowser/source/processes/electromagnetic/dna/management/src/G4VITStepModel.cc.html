35#include "G4VITStepModel.hh"
36
37G4VITStepModel::G4VITStepModel(const G4String& aName)
38{
39 //ctor
40 fpTimeStepper = 0;
41 fpReactionProcess = 0;
42 fpReactionTable = 0;
43
44 fType1 = -1;
45 fType2 = -1;
46 fName = aName;
47}
48
49G4VITStepModel::~G4VITStepModel()
50{
51 //dtor
52 if(fpTimeStepper) delete fpTimeStepper;
53 if(fpReactionProcess) delete fpReactionProcess;
54 //if(fReactionTable) delete fReactionTable;
55 // Let the concrete class delete the reactionTable
56}
57
58G4VITStepModel::G4VITStepModel(const G4VITStepModel& right)
59{
60 //copy ctor
61 fName = right.fName;
62 fType1 = right.fType1;
63 fType2 = right.fType2;
64 fpReactionTable = 0;
65 fpTimeStepper = right.fpTimeStepper->Clone();
66 fpReactionProcess = right.fpReactionProcess->Clone();
67}
68
69// should not be used
70G4VITStepModel& G4VITStepModel::operator=(const G4VITStepModel& right)
71{
72 if (this == &right) return *this; // handle self assignment
73
74 fName = right.fName;
75 fType1 = right.fType1;
76 fType2 = right.fType2;
77 fpReactionTable = 0;
78 if(fpTimeStepper) delete fpTimeStepper;
79 fpTimeStepper = right.fpTimeStepper->Clone();
80 if(fpReactionProcess) delete fpReactionProcess;
81 fpReactionProcess = right.fpReactionProcess->Clone();
82
83 //assignment operator
84 return *this;
85}
86
87void G4VITStepModel::IsApplicable(G4ITType& type1, G4ITType& type2)
88{
89 type1 = fType1;
90 type2 = fType2;
91 PrintInfo();
92}
93
94void G4VITStepModel::Initialize()
95{
96 fpReactionProcess->SetReactionTable(fpReactionTable);
97 fpTimeStepper->SetReactionTable(fpReactionTable);
98 fpTimeStepper->Initialize();
99 fpReactionProcess->Initialize();
100}
