35#include "G4ITType.hh"
36#include "G4AutoLock.hh"
37
38/*G4ThreadLocal*/ G4ITTypeManager* G4ITTypeManager::fgInstance = 0;
39G4ThreadLocal G4ITTypeManager* G4ITTypeManager::fgInstance_local = 0;
40
41G4Mutex deleteMutex=G4MUTEX_INITIALIZER;
42G4Mutex ressourceMutex=G4MUTEX_INITIALIZER;
43
44// static method
45size_t G4ITType::size()
46{
47 return G4ITTypeManager::Instance()->size();
48}
49
50G4ITType& G4ITType::operator=(const G4ITType & rhs)
51{
52 if (this == &rhs) return *this;
53 fValue = rhs.fValue;
54 return *this;
55}
56
57G4ITTypeManager* G4ITTypeManager::Instance()
58{
59 if(fgInstance == 0)
60 {
61 fgInstance = new G4ITTypeManager();
62 }
63 return fgInstance;
64}
65
66void G4ITTypeManager::DeleteInstance()
67{
68 G4AutoLock lock(&deleteMutex);
69 if(fgInstance)
70 {
71 delete fgInstance ;
72 fgInstance = 0;
73 }
74}
75
76void G4ITTypeManager::ReserveRessource()
77{
78 G4AutoLock lock(&ressourceMutex);
79 fRessource++;
80}
81
82void G4ITTypeManager::ReleaseRessource()
83{
84 G4AutoLock lock(&ressourceMutex);
85 fRessource--;
86
87 if(fRessource <= 0) DeleteInstance();
88}
89
90G4ITTypeManager::G4ITTypeManager()
91{
92 fLastType = 0;
93 fRessource = 0;
94}
95
96G4ITTypeManager::~G4ITTypeManager()
97{;}
98
99size_t G4ITTypeManager::size() const
100{
101 return fLastType;
102}
103
104G4ITType G4ITTypeManager::NewType()
105{
106 G4ITType newType = fLastType;
107 fLastType++;
108 return newType;
109}
