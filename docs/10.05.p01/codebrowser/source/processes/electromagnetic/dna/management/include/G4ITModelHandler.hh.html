46#ifndef G4ITModelHandler_H
47#define G4ITModelHandler_H
48
49#include "G4Types.hh"
50#include "G4ITType.hh"
51#include "G4ITModelManager.hh"
52
53/**
54 * G4ITModelHandler holds for two IT types the corresponding model manager
55 */
56class G4ITModelHandler
57{
58public:
59 G4ITModelHandler();
60 G4ITModelHandler(const G4ITModelHandler& other);
61 G4ITModelHandler& operator=(const G4ITModelHandler& rhs);
62
63 /** Default destructor */
64 ~G4ITModelHandler();
65
66 void Initialize();
67
68 // Register a model at a starting time (time1)
69 // if a second model is registered at a later time (time2);
70 // the second model will be considered from
71 // time2 to the end of simulation
72 void RegisterModel(G4VITStepModel* aModel, const G4double globalTime);
73
74 // Model applying for type 1 and type 2
75 inline G4ITModelManager* GetModelManager(G4ITType, G4ITType);
76 void SetModel(G4ITType,
77 G4ITType,
78 G4VITStepModel* aModel,
79 G4double startingTime);
80 G4VITStepModel* GetModel(G4ITType, G4ITType, const G4double globalTime);
81
82 //
83 inline const std::vector<std::vector<G4ITModelManager*> >* GetAllModelManager()
84 {
85 return &fModelManager;
86 }
87
88 inline bool GetTimeStepComputerFlag()
89 {
90 return fTimeStepComputerFlag;
91 }
92 inline bool GetReactionProcessFlag()
93 {
94 return fReactionProcessFlag;
95 }
96
97protected:
98 G4bool fIsInitialized;
99 std::vector<std::vector<G4ITModelManager*> > fModelManager;
100
101 G4bool fTimeStepComputerFlag; // Set true if a computer is registered
102 G4bool fReactionProcessFlag; // Set true if a reaction process is registered
103};
104
105inline G4ITModelManager* G4ITModelHandler::GetModelManager(G4ITType type1,
106 G4ITType type2)
107{
108 if (fModelManager.empty())
109 {
110 return 0;
111 }
112
113 if ((int) fModelManager.size() < type1) return 0;
114
115 std::vector<G4ITModelManager*>* v = &(fModelManager.at(type1));
116
117 if ((int) v->size() < type2) return 0;
118
119 return v->at(type2);
120}
121
122#endif // G4ITModelHandler_H
