71void G4ITReactionChange::Initialize(const G4Track& trackA,
72 const G4Track& trackB,
73 G4VParticleChange* particleChangeA,
74 G4VParticleChange* particleChangeB)
75{
76 fParticleChange.clear();
77 fParticleChange[&trackA] = particleChangeA;
78 fParticleChange[&trackB] = particleChangeB;
79
80 if (particleChangeA || particleChangeB)
81 {
82 G4bool test = particleChangeA && particleChangeB;
83
84 if (test == false)
85 {
86 G4ExceptionDescription exceptionDescription;
87 exceptionDescription << "If you give for one track a particleChange, ";
88 exceptionDescription
89 << "G4ITReactionChange is expecting that you give for both ";
90 exceptionDescription << "reacting tracks a particleChange.";
91 G4Exception("G4ITReactionChange::Initialize", "ITReactionChange001",
92 FatalErrorInArgument, exceptionDescription);
93 }
94
95 fParticleChangeIsSet = true;
96
97 fParticleChange[&trackA]->Initialize(trackA);
98 fParticleChange[&trackB]->Initialize(trackB);
99 ;
100
101 }
102
103 fSecondaries = 0;
104 fNumberOfSecondaries = 0;
105 fKillParents = false;
106}
107
108void G4ITReactionChange::AddSecondary(G4Track* aTrack)
109{
110 if (fSecondaries == 0) fSecondaries = new std::vector<G4Track*>();
111 fSecondaries->push_back(aTrack);
112 fNumberOfSecondaries++;
113}

115void G4ITReactionChange::UpdateStepInfo(G4Step* stepA, G4Step* stepB)
116{
117 fParticleChange[stepA->GetTrack()]->UpdateStepForPostStep(stepA);
118 fParticleChange[stepB->GetTrack()]->UpdateStepForPostStep(stepB);
119}

121G4VParticleChange* G4ITReactionChange::GetParticleChange(const G4Track* track)
122{
123 std::map<const G4Track*, G4VParticleChange*>::iterator it = fParticleChange
124 .find(track);
125
126 if (it == fParticleChange.end()) return 0;
127 else return it->second;
128}
129
130const G4Track* G4ITReactionChange::GetTrackA()
131{
132 std::map<const G4Track*, G4VParticleChange*>::iterator it = fParticleChange
133 .begin();
134 if (it != fParticleChange.end())
135 {
136 return it->first;
137 }
138
139 G4ExceptionDescription exceptionDescription;
140 exceptionDescription
141 << "No track A found ! Have you initialized the ReactionChange ?";
142 G4Exception("G4ITReactionChange::GetTrackA", "ITReactionChange001",
143 FatalErrorInArgument, exceptionDescription);
144 return 0;
145}
146
147const G4Track* G4ITReactionChange::GetTrackB()
148{
149 std::map<const G4Track*, G4VParticleChange*>::iterator it = fParticleChange
150 .begin();
151 std::map<const G4Track*, G4VParticleChange*>::iterator next = it++;
152 if (next == fParticleChange.end())
153 {
154 G4ExceptionDescription exceptionDescription;
155 exceptionDescription
156 << "No track B found ! Have you initialized the ReactionChange ?";
157 G4Exception("G4ITReactionChange::GetTrackB", "ITReactionChange002",
158 FatalErrorInArgument, exceptionDescription);
159 }
160
161 return it->first;
162}
