61/**
62 * G4DNAMolecularReactionData contains the information
63 * relative to a given reaction (eg : °OH + °OH -> H2O2)
64 */
65
66class G4DNAMolecularReactionData
67{
68 public :
69 //----------------------------------------------------------------------------
70
71 G4DNAMolecularReactionData(G4double reactionRate,
72 G4MolecularConfiguration* reactive1,
73 G4MolecularConfiguration* reactive2);
74
75 G4DNAMolecularReactionData(G4double reactionRate,
76 const G4String& reactive1,
77 const G4String& reactive2);
78 ~G4DNAMolecularReactionData();
79
80
81 //----------------------------------------------------------------------------
82 inline int GetReactionID() const { return fReactionID; }
83 inline void SetReactionID(int ID) { fReactionID = ID; }
84
85 //----------------------------------------------------------------------------
86 inline std::pair<G4MolecularConfiguration*, G4MolecularConfiguration*>
87 GetReactants()
88 {
89 return std::make_pair(fReactant1, fReactant2);
90 }
91
92 inline G4MolecularConfiguration* GetReactant1() const
93 {
94 return fReactant1;
95 }
96 inline G4MolecularConfiguration* GetReactant2() const
97 {
98 return fReactant2;
99 }
100
101 inline void SetObservedReactionRateConstant(G4double rate)
102 {
103 fObservedReactionRate = rate;
104 }
105
106 inline G4double GetObservedReactionRateConstant() const
107 {
108 return fObservedReactionRate;
109 }
110
111 inline G4double GetEffectiveReactionRadius() const
112 {
113 return fEffectiveReactionRadius;
114 }
115
116 inline void SetEffectiveReactionRadius(G4double radius)
117 {
118 fEffectiveReactionRadius = radius;
119 }
120
121 //_____________________________________________________
122
123 void SetReactant1(G4MolecularConfiguration* reactive) ;
124 void SetReactant2(G4MolecularConfiguration* reactive) ;
125
126 void SetReactants(G4MolecularConfiguration* reactive1,
127 G4MolecularConfiguration* reactive2);
128
129 void AddProduct(G4MolecularConfiguration* molecule);
130
131 void SetReactant1(const G4String& reactive) ;
132 void SetReactant2(const G4String& reactive) ;
133 void SetReactants(const G4String& reactive1, const G4String& reactive2);
134 void AddProduct(const G4String& molecule);
135
136 inline G4int GetNbProducts() const
137 {
138 if(fProducts) return fProducts->size();
139 return 0;
140 }
141
142 inline G4MolecularConfiguration* GetProduct(G4int i) const
143 {
144 if(fProducts) return (*fProducts)[i];
145 return 0;
146 }
147
148 inline const std::vector<G4MolecularConfiguration*>* GetProducts() const
149 {
150 return fProducts;
151 }
152
153 inline void RemoveProducts()
154 {
155 if(fProducts)
156 {
157 fProducts->clear();
158 delete fProducts;
159 }
160 }
161
162 //----------------------------------------------------------------------------
163 // Temperature scaling
164 typedef std::function<double(double)> RateParam;
165
166 static double PolynomialParam(double temp_K, std::vector<double> P);
167 static double ArrehniusParam(double temp_K, std::vector<double> P);
168 static double ScaledParameterization(double temp_K,
169 double temp_init,
170 double rateCste_init);
171
172 void SetPolynomialParameterization(const std::vector<double>& P);
173
174 void SetArrehniusParameterization(double A0, double E_R);
175 void SetScaledParameterization(double temperature_K,
176 double rateCste);
177
178 void ScaleForNewTemperature(double temp_K);
179
180 protected :
181 G4DNAMolecularReactionData();
182 G4MolecularConfiguration* fReactant1;
183 G4MolecularConfiguration* fReactant2;
184 G4double fObservedReactionRate;
185 G4double fEffectiveReactionRadius;
186
187 std::vector<G4MolecularConfiguration*>* fProducts;
188 // G4DNAReactionType fReactionType;
189 RateParam fRateParam;
190 int fReactionID;
191};

193/**
194 * G4DNAMolecularReactionTable sorts out the G4DNAMolecularReactionData
195 * for bimolecular reaction
196 */
197
198class G4DNAMolecularReactionTable : public G4ITReactionTable
199{
200protected:
201 G4DNAMolecularReactionTable();
202 static G4DNAMolecularReactionTable* fInstance;
203
204 public :
205 static G4DNAMolecularReactionTable* GetReactionTable();
206 static G4DNAMolecularReactionTable* Instance();
207 static void DeleteInstance();
208 virtual ~G4DNAMolecularReactionTable();
209
210 /**
211 * Define a reaction :
212 * First argument : reaction rate
213 * Second argument : reactant 1
214 * Third argument : reactant 2
215 * Fourth argument : a std::vector holding the molecular products
216 * if this last argument is NULL then it will be interpreted as
217 * a reaction giving no products
218 */
219 void SetReaction(G4double observedReactionRate,
220 G4MolecularConfiguration* reactive1,
221 G4MolecularConfiguration* reactive2);
222
223 void SetReaction(G4DNAMolecularReactionData*);
224
225 const G4DNAMolecularReactionData* GetReactionData(G4MolecularConfiguration*,
226 G4MolecularConfiguration*) const;
227
228 const G4DNAMolecularReactionData* GetReactionData(const G4String&,
229 const G4String&) const;
230
231 const G4DNAMolecularReactionData* GetReaction(int reactionID) const;
232
233 size_t GetNReactions() const
234 { return fVectorOfReactionData.size(); }
235
236 //_________________________________________________________________
237 /**
238 * Given a molecule's type, it returns with which a reaction is allowed
239 */
240 const std::vector<G4MolecularConfiguration*>*
241 CanReactWith(G4MolecularConfiguration*) const ;
242
243 const std::map<G4MolecularConfiguration*, const G4DNAMolecularReactionData*>*
244 GetReativesNData(G4MolecularConfiguration*) const;
245
246 const std::vector<const G4DNAMolecularReactionData*>*
247 GetReactionData(G4MolecularConfiguration*) const;
248
249 inline const std::map<G4MolecularConfiguration*,
250 std::map<G4MolecularConfiguration*,
251 const G4DNAMolecularReactionData*> >&
252 GetAllReactionData()
253 {
254 return fReactionData;
255 }
256
257 inline const std::vector<const G4DNAMolecularReactionData*>&
258 GetVectorOfReactionData()
259 {
260 return fVectorOfReactionData;
261 }
262
263 void ScaleReactionRateForNewTemperature(double temp_K);
264
265 //_________________________________________________________________
266 void PrintTable(G4VDNAReactionModel* = 0);
267
268 protected :
269 G4bool fVerbose;
270
271 //_________________________________________________

273 std::map<G4MolecularConfiguration*,
274 const G4DNAMolecularReactionData*> > ReactionDataMap;

276 std::vector<G4MolecularConfiguration*> > ReactivesMV;

281 ReactivesMV fReactantsMV;
282 ReactionDataMV fReactionDataMV;
283 std::vector<const G4DNAMolecularReactionData*> fVectorOfReactionData;
284 G4ReactionTableMessenger* fpMessenger;
285};
