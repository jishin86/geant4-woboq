48//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
49void G4DNAPositronium::InitialiseProcess(const G4ParticleDefinition*)
50{
51 if(!isInitialised)
52 {
53 isInitialised = true;
54 SetBuildTableFlag(false);
55 if(!EmModel()) SetEmModel(new G4LEPTSPositroniumModel);
56 EmModel()->SetLowEnergyLimit(0.1*CLHEP::eV);
57 EmModel()->SetHighEnergyLimit(15.*CLHEP::MeV);
58 AddEmModel(1, EmModel());
59
60 }
61}
