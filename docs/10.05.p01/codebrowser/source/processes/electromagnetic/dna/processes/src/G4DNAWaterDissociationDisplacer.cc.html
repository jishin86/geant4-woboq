71#ifdef _WATER_DISPLACER_USE_KREIPL_
72// This function is used to generate the following density distribution:
73// f(r) := 4*r * exp(-2*r)
74// reference:
75// Kreipl, M. S., Friedland, W. & Paretzke, H. G.
76// Time- and space-resolved Monte Carlo study of water radiolysis for photon,
77// electron and ion irradiation.
78// Radiat. Environ. Biophys. 48, 11–20 (2009).
79G4double G4DNAWaterDissociationDisplacer::ElectronProbaDistribution(G4double r)
80{
81 return (2.*r+1.)*G4Exp(-2.*r);
82}
83#endif
84
85//------------------------------------------------------------------------------
86#ifdef _WATER_DISPLACER_USE_TERRISOL_
87// This function is used to generate the following density distribution:
88// f(r) := 4*x^2/(sqrt(%pi)*(b)^3)*exp(-x^2/(b)^2)
89// with b=27.22 for 7 eV
90// reference
91// Terrissol M, Beaudre A (1990) Simulation of space and time evolution
92// of radiolytic species induced by electrons in water.
93// Radiat Prot Dosimetry 31:171–175
94
95G4double G4DNAWaterDissociationDisplacer::ElectronProbaDistribution(G4double r)
96{
97#define b 27.22 //*nanometer
98 static constexpr double sqrt_pi=1.77245; // sqrt(CLHEP::pi);
99 static constexpr double b_to3 = 20168.1; // pow(b,3.);
100 static constexpr double b_to2 = 740.928; // pow(b,2.);
101 static constexpr double inverse_b_to2 = 1./b_to2;
102
103 static constexpr double main_factor=4./(sqrt_pi*b_to3);
104 static constexpr double factorA=sqrt_pi*b_to3/4.;
105 static constexpr double factorB=b_to2/2.;
106
107 return (main_factor*
108 (factorA*erf(r/b)
109 - factorB*r*G4Exp(-pow(r,2.)*inverse_b_to2)));
110}
111#endif

114G4DNAWaterDissociationDisplacer::G4DNAWaterDissociationDisplacer() :
115 G4VMolecularDecayDisplacer(),
116#ifdef _WATER_DISPLACER_USE_KREIPL_
117 fFastElectronDistrib(0., 5., 0.001)
118#elif defined _WATER_DISPLACER_USE_TERRISOL_
119 fFastElectronDistrib(0., 100., 0.001)
120#endif
121{
122 fProba1DFunction =
123 std::bind((G4double(*)(G4double))
124 &G4DNAWaterDissociationDisplacer::ElectronProbaDistribution,
125 std::placeholders::_1);
126
127 size_t nBins = 500;
128 fElectronThermalization.reserve(nBins);
129 double eps = 1./((int)nBins);
130 double proba = eps;
131
132 fElectronThermalization.push_back(0.);
133
134 for(size_t i = 1 ; i < nBins ; ++i){
135 double r = fFastElectronDistrib.Revert(proba, fProba1DFunction);
136 fElectronThermalization.push_back(r*nanometer);
137 proba+=eps;
138// G4cout << G4BestUnit(r*nanometer, "Length") << G4endl;
139 }
140// SetVerbose(1);
141}

150G4ThreeVector
151G4DNAWaterDissociationDisplacer::
152GetMotherMoleculeDisplacement(const G4MolecularDissociationChannel*
153 theDecayChannel) const
154{
155 G4int decayType = theDecayChannel->GetDisplacementType();
156 G4double RMSMotherMoleculeDisplacement(0.);
157
158 switch(decayType){
159 case Ionisation_DissociationDecay:
160 RMSMotherMoleculeDisplacement = 2.0 * nanometer;
161 break;
162 case A1B1_DissociationDecay:
163 RMSMotherMoleculeDisplacement = 0. * nanometer;
164 break;
165 case B1A1_DissociationDecay:
166 RMSMotherMoleculeDisplacement = 0. * nanometer;
167 break;
168 case AutoIonisation:
169 RMSMotherMoleculeDisplacement = 2.0 * nanometer;
170 break;
171 case DissociativeAttachment:
172 RMSMotherMoleculeDisplacement = 0. * nanometer;
173 break;
174 }
175
176 if(RMSMotherMoleculeDisplacement == 0){
177 return G4ThreeVector(0, 0, 0);
178 }
179 auto RandDirection =
180 radialDistributionOfProducts(RMSMotherMoleculeDisplacement);
181
182 return RandDirection;
183}

186
187vector<G4ThreeVector>
188G4DNAWaterDissociationDisplacer::
189GetProductsDisplacement(const G4MolecularDissociationChannel*
190 theDecayChannel) const
191{
192 G4int nbProducts = theDecayChannel->GetNbProducts();
193 vector<G4ThreeVector> theProductDisplacementVector(nbProducts);
194
195 typedef map<const G4MoleculeDefinition*, G4double> RMSmap;
196 RMSmap theRMSmap;
197
198 G4int decayType = theDecayChannel->GetDisplacementType();
199
200 switch(decayType){
201 case Ionisation_DissociationDecay:
202 {
203 if (fVerbose){
204 G4cout << "Ionisation_DissociationDecay" << G4endl;
205 G4cout << "Channel's name: " << theDecayChannel->GetName() << G4endl;
206 }
207 G4double RdmValue = G4UniformRand();
208
209 if(RdmValue< 0.5){
210 // H3O
211 theRMSmap[G4H3O::Definition()] = 0.* nanometer;
212 // OH
213 theRMSmap[G4OH::Definition()] = 0.8* nanometer;
214 }
215 else{
216 // H3O
217 theRMSmap[G4H3O::Definition()] = 0.8* nanometer;
218 // OH
219 theRMSmap[G4OH::Definition()] = 0.* nanometer;
220 }
221
222 for(int i = 0; i < nbProducts; i++){
223 G4MolecularConfiguration* product = theDecayChannel->GetProduct(i);
224 G4double theRMSDisplacement = theRMSmap[product->GetDefinition()];
225
226 if(theRMSDisplacement==0.){
227 theProductDisplacementVector[i] = G4ThreeVector();
228 }
229 else{
230 auto RandDirection = radialDistributionOfProducts(theRMSDisplacement);
231 theProductDisplacementVector[i] = RandDirection;
232 }
233 }
234 break;
235 }
236 case A1B1_DissociationDecay:
237 {
238 if(fVerbose){
239 G4cout<<"A1B1_DissociationDecay"<<G4endl;
240 G4cout << "Channel's name: " << theDecayChannel->GetName() << G4endl;
241 }
242 G4double theRMSDisplacement = 2.4 * nanometer;
243 auto RandDirection =
244 radialDistributionOfProducts(theRMSDisplacement);
245
246 for(G4int i =0; i < nbProducts; i++){
247 G4MolecularConfiguration* product = theDecayChannel->GetProduct(i);
248
249 if(product->GetDefinition()== G4OH::Definition()){
250 // OH
251 theProductDisplacementVector[i] = -1./18.*RandDirection;
252 }
253 else if(product->GetDefinition() == G4Hydrogen::Definition()){
254 // H
255 theProductDisplacementVector[i] = +17./18.*RandDirection;
256 }
257 }
258 break;
259 }
260 case B1A1_DissociationDecay:
261 {
262 if(fVerbose){
263 G4cout<<"B1A1_DissociationDecay"<<G4endl;
264 G4cout << "Channel's name: " << theDecayChannel->GetName() << G4endl;
265 }
266
267 G4double theRMSDisplacement = 0.8 * nanometer;
268 auto RandDirection =
269 radialDistributionOfProducts(theRMSDisplacement);
270
271 G4int NbOfOH = 0;
272 for(G4int i =0; i < nbProducts; ++i)
273 {
274 G4MolecularConfiguration* product = theDecayChannel->GetProduct(i);
275 if(product->GetDefinition() == G4H2::Definition()){
276 // H2
277 theProductDisplacementVector[i] = -2./18.*RandDirection;
278 }
279 else if(product->GetDefinition() == G4OH::Definition()){
280 // OH
281 G4ThreeVector OxygenDisplacement = +16./18.*RandDirection;
282 G4double OHRMSDisplacement = 1.1 * nanometer;
283
284 auto OHDisplacement =
285 radialDistributionOfProducts(OHRMSDisplacement);
286
287 if(NbOfOH==0){
288 OHDisplacement = 0.5*OHDisplacement;
289 }
290 else{
291 OHDisplacement = -0.5*OHDisplacement;
292 }
293
294 theProductDisplacementVector[i] =
295 OHDisplacement + OxygenDisplacement;
296
297 ++NbOfOH;
298 }
299 }
300 break;
301 }
302 case AutoIonisation:
303 {
304 if(fVerbose){
305 G4cout<<"AutoIonisation"<<G4endl;
306 G4cout << "Channel's name: " << theDecayChannel->GetName() << G4endl;
307 }
308
309 G4double RdmValue = G4UniformRand();
310
311 if(RdmValue< 0.5){
312 // H3O
313 theRMSmap[G4H3O::Definition()] = 0.* nanometer;
314 // OH
315 theRMSmap[G4OH::Definition()] = 0.8* nanometer;
316 }
317 else{
318 // H3O
319 theRMSmap[G4H3O::Definition()] = 0.8* nanometer;
320 // OH
321 theRMSmap[G4OH::Definition()] = 0.* nanometer;
322 }
323
324 for(G4int i =0; i < nbProducts; i++){
325 G4MolecularConfiguration* product = theDecayChannel->GetProduct(i);
326 auto theRMSDisplacement = theRMSmap[product->GetDefinition()];
327
328 if(theRMSDisplacement==0){
329 theProductDisplacementVector[i] = G4ThreeVector();
330 }
331 else{
332 auto RandDirection =
333 radialDistributionOfProducts(theRMSDisplacement);
334 theProductDisplacementVector[i] = RandDirection;
335 }
336 if(product->GetDefinition() == G4Electron_aq::Definition()){
337 theProductDisplacementVector[i]=radialDistributionOfElectron();
338 }
339 }
340 break;
341 }
342 case DissociativeAttachment:
343 {
344 if(fVerbose){
345 G4cout<<"DissociativeAttachment"<<G4endl;
346 G4cout << "Channel's name: " << theDecayChannel->GetName() << G4endl;
347 }
348 G4double theRMSDisplacement = 0.8 * nanometer;
349 auto RandDirection =
350 radialDistributionOfProducts(theRMSDisplacement);
351
352 G4int NbOfOH = 0;
353 for(G4int i =0; i < nbProducts; ++i){
354 G4MolecularConfiguration* product = theDecayChannel->GetProduct(i);
355 if(product->GetDefinition() == G4H2::Definition()){
356 theProductDisplacementVector[i] = -2./18.*RandDirection;
357 }
358 else if(product->GetDefinition() == G4OH::Definition()){
359 G4ThreeVector OxygenDisplacement = +16./18.*RandDirection;
360 G4double OHRMSDisplacement = 1.1 * nanometer;
361
362 auto OHDisplacement =
363 radialDistributionOfProducts(OHRMSDisplacement);
364
365 if(NbOfOH==0){
366 OHDisplacement = 0.5*OHDisplacement;
367 }
368 else{
369 OHDisplacement = -0.5*OHDisplacement;
370 }
371 theProductDisplacementVector[i] = OHDisplacement +
372 OxygenDisplacement;
373 ++NbOfOH;
374 }
375 }
376 break;
377 }
378 }
379 return theProductDisplacementVector;
380}

384G4ThreeVector
385G4DNAWaterDissociationDisplacer::
386radialDistributionOfProducts(G4double Rrms) const
387{
388 static const double inverse_sqrt_3 = 1./sqrt(3.);
389 double sigma = Rrms*inverse_sqrt_3;
390 double x = G4RandGauss::shoot(0.,sigma);
391 double y = G4RandGauss::shoot(0.,sigma);
392 double z = G4RandGauss::shoot(0.,sigma);
393 return G4ThreeVector(x,y,z);
394}

398G4ThreeVector
399G4DNAWaterDissociationDisplacer::radialDistributionOfElectron() const
400{
401 G4double rand_value = G4UniformRand();
402 size_t nBins = fElectronThermalization.size();
403 size_t bin = size_t(floor(rand_value*nBins));
404 size_t bin_p1 = min(bin+1,nBins-1);
405
406 return (fElectronThermalization[bin]*(1.-rand_value)
407 + fElectronThermalization[bin_p1]*rand_value)*
408 G4RandomDirection();
409// return fElectronThermalization[bin]*G4RandomDirection();
410}
