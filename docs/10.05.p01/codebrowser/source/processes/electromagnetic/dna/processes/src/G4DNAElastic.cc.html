50G4bool G4DNAElastic::IsApplicable(const G4ParticleDefinition& p)
51{
52 G4DNAGenericIonsManager *instance;
53 instance = G4DNAGenericIonsManager::Instance();
54
55 return (&p == G4Electron::Electron() || &p == G4Positron::Positron()
56 || &p == G4Proton::Proton() || &p == instance->GetIon("hydrogen")
57 || &p == instance->GetIon("alpha++")
58 || &p == instance->GetIon("alpha+")
59 || &p == instance->GetIon("helium"));
60
61}

65void G4DNAElastic::InitialiseProcess(const G4ParticleDefinition* p)
66{
67 if(!isInitialised)
68 {
69 isInitialised = true;
70 SetBuildTableFlag(false);
71
72 G4String name = p->GetParticleName();
73
74 if(name == "e-")
75 {
76 if(!EmModel())
77 {
78 SetEmModel(new G4DNAScreenedRutherfordElasticModel);
79 EmModel()->SetLowEnergyLimit(0 * eV);
80 EmModel()->SetHighEnergyLimit(1. * MeV);
81 }
82 AddEmModel(1, EmModel());
83 }
84
85 else if(name == "proton" || name == "hydrogen")
86 {
87 if(!EmModel())
88 {
89 SetEmModel(new G4DNAIonElasticModel);
90 EmModel()->SetLowEnergyLimit(0 * eV);
91 EmModel()->SetHighEnergyLimit(1 * MeV);
92 }
93 AddEmModel(1, EmModel());
94 }
95
96 // "alpha" must be explicitely used, not alpha++
97 else if(name == "helium" || name == "alpha" || name == "alpha+")
98 {
99 if(!EmModel())
100 {
101 SetEmModel(new G4DNAIonElasticModel);
102 EmModel()->SetLowEnergyLimit(0 * eV);
103 EmModel()->SetHighEnergyLimit(1 * MeV);
104 }
105 AddEmModel(1, EmModel());
106 }
107 }
108}
