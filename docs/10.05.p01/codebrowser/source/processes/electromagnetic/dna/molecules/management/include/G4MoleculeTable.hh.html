52class G4MoleculeDefinition;
53class G4MolecularConfiguration;
54
55typedef G4MoleculeIterator<G4MoleculeDefinition> G4MoleculeDefinitionIterator;
56typedef G4MoleculeIterator<G4MolecularConfiguration> G4ConfigurationIterator;

58class G4MoleculeTable
59{
60public:
61 static G4MoleculeTable* Instance();
62 static G4MoleculeTable* GetMoleculeTable();
63 virtual ~G4MoleculeTable();
64
65 //____________________________________________________________________________
66 // The methods below enable to create G4MoleculeDefinition &
67 // G4MolecularConfiguration with a user identifier so that they can be retrieved
68 // from this molecule table
69 //
70
71 //____________________________________________________________________________
72
73 G4MoleculeDefinition* CreateMoleculeDefinition(const G4String& userIdentifier,
74 double diffusion_coefficient);
75
76 //____________________________________________________________________________
77
78 G4MolecularConfiguration*
79 CreateConfiguration(const G4String& userIdentifier,
80 const G4MoleculeDefinition* molDef,
81 const G4String& configurationLabel,
82 const G4ElectronOccupancy& eOcc);
83
84 G4MolecularConfiguration*
85 CreateConfiguration(const G4String& userIdentifier,
86 G4MoleculeDefinition*,
87 int charge,
88 double diffusion_coefficient = -1);
89
90 G4MolecularConfiguration*
91 CreateConfiguration(const G4String& userIdentifier,
92 G4MoleculeDefinition*);
93
94 G4MolecularConfiguration*
95 CreateConfiguration(const G4String& userIdentifier,
96 G4MoleculeDefinition*,
97 const G4String& configurationLabel,
98 int charge = 0);
99
100 //____________________________________________________________________________
101
102 G4MoleculeDefinition* GetMoleculeDefinition(const G4String&,
103 bool mustExist = true);
104
105 G4MolecularConfiguration* GetConfiguration(const G4String&,
106 bool mustExist = true);
107 G4MolecularConfiguration* GetConfiguration(G4int id);
108
109 //____________________________________________________________________________
110
111 void Insert(G4MoleculeDefinition*);
112 void Finalize(G4MoleculeDefinition*){}
113 void Finalize();
114 //____________________________________________________________________________
115
116 G4MoleculeDefinitionIterator GetDefintionIterator()
117 {
118 return G4MoleculeDefinitionIterator(this->fMoleculeDefTable);
119 }
120
121 G4ConfigurationIterator GetConfigurationIterator();
122
123 void PrepareMolecularConfiguration();
124
125 int GetNumberOfDefinedSpecies();
126
127protected:
128 G4MoleculeTable();
129
130 static G4MoleculeTable* fpgMoleculeTable;
131 typedef std::map<G4String, G4MoleculeDefinition*> MoleculeDefTable;
132
133 MoleculeDefTable fMoleculeDefTable;
134};
