96G4bool G4MoleculeCounter::SearchTimeMap(G4MolecularConfiguration* molecule)
97{
98 if (fpLastSearch.get() == 0)
99 {
100 fpLastSearch.reset(new Search());
101 }
102 else
103 {
104 if (fpLastSearch->fLowerBoundSet &&
105 fpLastSearch->fLastMoleculeSearched->first == molecule)
106 return true;
107 }
108
109 CounterMapType::iterator mol_it = fCounterMap.find(molecule);
110 fpLastSearch->fLastMoleculeSearched = mol_it;
111
112 if (mol_it != fCounterMap.end()) // TODO
113 {
114 fpLastSearch->fLowerBoundTime = fpLastSearch->fLastMoleculeSearched->second
115 .end();
116 fpLastSearch->fLowerBoundSet = true;
117 }
118 else
119 {
120 fpLastSearch->fLowerBoundSet = false;
121 }
122
123 return false;
124}

126//------------------------------------------------------------------------------
127
128int G4MoleculeCounter::SearchUpperBoundTime(double time,
129 bool sameTypeOfMolecule)
130{
131 CounterMapType::iterator mol_it = fpLastSearch->fLastMoleculeSearched;
132 if (mol_it == fCounterMap.end()) return 0; // RETURN
133
134 NbMoleculeAgainstTime& timeMap = mol_it->second;
135 if (timeMap.empty()) return 0;
136
137 NbMoleculeAgainstTime::iterator end_time = timeMap.end();
138
139 if (sameTypeOfMolecule == true)
140 {
141 //G4cout << "SAME MOLECULE" << G4endl;
142 if (fpLastSearch->fLowerBoundSet && fpLastSearch->fLowerBoundTime
143 != end_time)
144 {
145 if (fpLastSearch->fLowerBoundTime->first < time)
146 {
147 NbMoleculeAgainstTime::iterator upperToLast = fpLastSearch
148 ->fLowerBoundTime;
149 upperToLast++;
150
151 if (upperToLast == end_time)
152 {
153 return fpLastSearch->fLowerBoundTime->second;
154 }
155
156 if (upperToLast->first > time)
157 {
158 return fpLastSearch->fLowerBoundTime->second;
159 }
160 }
161 }
162 }
163 /*
164 else
165 {
166 G4cout << "--> Molecule has changed" << G4endl;
167 }
168 */
169 //G4cout << "Searching" << G4endl;
170 // With upper bound
171 NbMoleculeAgainstTime::iterator up_time_it = timeMap.upper_bound(time);
172
173 if (up_time_it == end_time)
174 {
175 NbMoleculeAgainstTime::reverse_iterator last_time = timeMap.rbegin();
176
177// {
178 //G4cout << "RETURN LAST : " << G4BestUnit(time, "Time") << G4endl;
179 return last_time->second;
180// }
181
182// G4cout << "RETURN 0 (1)" << G4endl;
183// return 0; // RETURN
184 }
185 if (up_time_it == timeMap.begin())
186 {
187// G4cout << "RETURN 0 (2)" << G4endl;
188 return 0; // RETURN
189 }
190
191 //G4cout << "Going back : " << up_time_it->first << "-->";
192
193 up_time_it--;
194
195// G4cout << up_time_it->first << G4endl;
196
197 fpLastSearch->fLowerBoundTime = up_time_it;
198 fpLastSearch->fLowerBoundSet = true;
199
200// G4cout << "returning : " << fpLastSearch->fLowerBoundTime->second << G4endl;
201
202 return fpLastSearch->fLowerBoundTime->second;
203}

216void G4MoleculeCounter::AddAMoleculeAtTime(G4MolecularConfiguration* molecule,
217 G4double time,
218 const G4ThreeVector* /*position*/,
219 int number)
220{
221 if (fDontRegister[molecule->GetDefinition()]) return;
222
223 if (fVerbose){
224 G4cout << "G4MoleculeCounter::AddAMoleculeAtTime : " << molecule->GetName()
225 << " at time : " << G4BestUnit(time, "Time") << G4endl;
226 }
227
228 CounterMapType::iterator counterMap_i =
229 fCounterMap.find(molecule);
230
231 if (counterMap_i == fCounterMap.end()){
232 fCounterMap[molecule][time] = number;
233 }
234 else if (counterMap_i->second.empty()){
235 counterMap_i->second[time] = number;
236 }
237 else{
238 NbMoleculeAgainstTime::reverse_iterator end = counterMap_i->second.rbegin();
239
240 if (end->first <= time ||
241 fabs(end->first - time) <= compDoubleWithPrecision::fPrecision)
242 // Case 1 = new time comes after last recorded data
243 // Case 2 = new time is about the same as the last recorded one
244 {
245 double newValue = end->second + number;
246 counterMap_i->second[time] = newValue;
247 }
248 else
249 {
250// if(fabs(time - G4Scheduler::Instance()->GetGlobalTime()) >
251// G4Scheduler::Instance()->GetTimeTolerance())
252 {
253 G4ExceptionDescription errMsg;
254 errMsg << "Time of species "
255 << molecule->GetName() << " is "
256 << G4BestUnit(time, "Time") << " while "
257 << " global time is "
258 << G4BestUnit(G4Scheduler::Instance()->GetGlobalTime(), "Time")
259 << G4endl;
260 G4Exception("G4MoleculeCounter::RemoveAMoleculeAtTime",
261 "TIME_DONT_MATCH",
262 FatalException, errMsg);
263 }
264 }
265 }
266}

268//------------------------------------------------------------------------------
269
270void
271G4MoleculeCounter::RemoveAMoleculeAtTime(G4MolecularConfiguration* molecule,
272 G4double time,
273 const G4ThreeVector* /*position*/,
274 int number)
275{
276 if (fDontRegister[molecule->GetDefinition()]) return;
277
278 if (fVerbose)
279 {
280 G4cout << "G4MoleculeCounter::RemoveAMoleculeAtTime : "
281 << molecule->GetName() << " at time : " << G4BestUnit(time, "Time")
282 << G4endl;
283 }
284
285 if(fCheckTimeIsConsistentWithScheduler)
286 {
287 if(fabs(time - G4Scheduler::Instance()->GetGlobalTime()) >
288 G4Scheduler::Instance()->GetTimeTolerance())
289 {
290 G4ExceptionDescription errMsg;
291 errMsg << "Time of species "
292 << molecule->GetName() << " is "
293 << G4BestUnit(time, "Time") << " while "
294 << " global time is "
295 << G4BestUnit(G4Scheduler::Instance()->GetGlobalTime(), "Time")
296 << G4endl;
297 G4Exception("G4MoleculeCounter::RemoveAMoleculeAtTime",
298 "TIME_DONT_MATCH",
299 FatalException, errMsg);
300 }
301 }
302
303 NbMoleculeAgainstTime& nbMolPerTime = fCounterMap[molecule];
304
305 if (nbMolPerTime.empty())
306 {
307 molecule->PrintState();
308 Dump();
309 G4String errMsg =
310 "You are trying to remove molecule " + molecule->GetName()
311 + " from the counter while this kind of molecules has not been registered yet";
312 G4Exception("G4MoleculeCounter::RemoveAMoleculeAtTime", "",
313 FatalErrorInArgument, errMsg);
314
315 return;
316 }
317 else
318 {
319 NbMoleculeAgainstTime::reverse_iterator it = nbMolPerTime.rbegin();
320
321 if (it == nbMolPerTime.rend()){
322 it--;
323
324 G4String errMsg = "There was no " + molecule->GetName()
325 + " recorded at the time or even before the time asked";
326 G4Exception("G4MoleculeCounter::RemoveAMoleculeAtTime", "",
327 FatalErrorInArgument, errMsg);
328 }
329
330 if (time - it->first < -compDoubleWithPrecision::fPrecision){
331 Dump();
332 G4ExceptionDescription errMsg;
333 errMsg << "Is time going back?? " << molecule->GetName()
334 << " is being removed at time " << G4BestUnit(time, "Time")
335 << " while last recorded time was "
336 << G4BestUnit(it->first, "Time") << ".";
337 G4Exception("G4MoleculeCounter::RemoveAMoleculeAtTime",
338 "RETURN_TO_THE_FUTUR",
339 FatalErrorInArgument,
340 errMsg);
341 }
342
343 double finalN = it->second - number;
344
345 if(finalN < 0){
346 Dump();
347 G4ExceptionDescription errMsg;
348 errMsg << "After removal of " << number << " species of "
349 << molecule->GetName() << " the final number at time "
350 << G4BestUnit(time, "Time") << " is less than zero and so not valid."
351 << " Global time is "
352 << G4BestUnit(G4Scheduler::Instance()->GetGlobalTime(), "Time")
353 << ". Previous selected time is "
354 << G4BestUnit(it->first, "Time")
355 << G4endl;
356 G4Exception("G4MoleculeCounter::RemoveAMoleculeAtTime",
357 "N_INF_0",
358 FatalException, errMsg);
359 }
360
361 nbMolPerTime[time] = finalN;
362 }
363}

367G4MoleculeCounter::RecordedMolecules G4MoleculeCounter::GetRecordedMolecules()
368{
369 if (fVerbose > 1)
370 {
371 G4cout << "Entering in G4MoleculeCounter::RecordMolecules" << G4endl;
372 }
373
374 CounterMapType::iterator it;
375 RecordedMolecules output (new vector<G4MolecularConfiguration*>);
376
377 for(it = fCounterMap.begin(); it != fCounterMap.end(); it++)
378 {
379 output->push_back(it->first);
380 }
381 return output;
382}

384//------------------------------------------------------------------------------
385
386RecordedTimes G4MoleculeCounter::GetRecordedTimes()
387{
388 RecordedTimes output(new std::set<G4double>);
389
390 //G4double time;
391
392 CounterMapType::iterator it;
393 CounterMapType::const_iterator ite;
394
395 NbMoleculeAgainstTime::iterator it2;
396 NbMoleculeAgainstTime::const_iterator ite2;
397
398 // iterate on each molecule
399 for (it = fCounterMap.begin(), ite = fCounterMap.end(); it != ite; ++it)
400 {
401 // iterate on each time
402 for (it2 = (it->second).begin(), ite2 = (it->second).end(); it2 != ite2;
403 ++it2)
404 {
405 //time = it2->first;
406 output->insert(it2->first);
407 }
408 }
409
410 return output;
411}
