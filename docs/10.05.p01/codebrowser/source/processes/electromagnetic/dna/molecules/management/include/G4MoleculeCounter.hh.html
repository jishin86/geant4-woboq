54//------------------------------------------------------------------------------
55
56struct compDoubleWithPrecision{
57 bool operator()(const double& a, const double& b) const{
58 if (std::fabs(a - b) < fPrecision){
59 return false;
60 }
61 else{
62 return a < b;
63 }
64 }
65
66 static G4ThreadLocal double fPrecision;
67};
68
69//------------------------------------------------------------------------------
70
71typedef std::map<G4double, G4int, compDoubleWithPrecision>
72 NbMoleculeAgainstTime;
73
74typedef std::unique_ptr<std::set<G4double> > RecordedTimes;
75typedef std::set<G4double>::iterator RecordedTimesIterator;

77//------------------------------------------------------------------------------
78
79class G4MoleculeCounter : public G4VMoleculeCounter
80{
81 //----------------------------------------------------------------------------
82public:
83 typedef std::map<G4MolecularConfiguration*,
84 NbMoleculeAgainstTime> CounterMapType;
85 typedef std::unique_ptr<std::vector<G4MolecularConfiguration*> >
86 RecordedMolecules;
87
88 //----------------------------------------------------------------------------
89protected:
90 G4MoleculeCounter();
91 virtual ~G4MoleculeCounter();
92
93 CounterMapType fCounterMap;
94 std::map<const G4MoleculeDefinition*, G4bool> fDontRegister;
95
96 G4int fVerbose;
97 G4bool fCheckTimeIsConsistentWithScheduler;
98
99 struct Search
100 {
101 Search(){
102 fLowerBoundSet = false;
103 }
104 CounterMapType::iterator fLastMoleculeSearched;
105 NbMoleculeAgainstTime::iterator fLowerBoundTime;
106 bool fLowerBoundSet;
107 };
108
109 std::unique_ptr<Search> fpLastSearch;
110
111#ifdef MOLECULE_COUNTER_TESTING
112public:
113#else
114protected:
115#endif
116
117 friend class G4Molecule;
118 friend class G4VMoleculeCounter;
119
120 //----------------------------------------------------------------------------
121public:
122 static G4MoleculeCounter* Instance();
123
124 //----------------------------------------------------------------------------
125public:
126 void Initialize() override;
127
128 void ResetCounter() override;
129
130 void AddAMoleculeAtTime(G4MolecularConfiguration*,
131 G4double time,
132 const G4ThreeVector* position = nullptr,
133 int number = 1) override;
134 void RemoveAMoleculeAtTime(G4MolecularConfiguration*,
135 G4double time,
136 const G4ThreeVector* position = nullptr,
137 int number = 1) override;
138
139 /* The dynamics of the given molecule won't be saved into memory.*/
140 inline void DontRegister(const G4MoleculeDefinition*) override;
141 inline bool IsRegistered(const G4MoleculeDefinition*) override;
142 inline void RegisterAll() override;
143
144 //----------------------------------------------------------------------------
145public:
146 int GetNMoleculesAtTime(G4MolecularConfiguration* molecule,
147 double time);
148 inline const NbMoleculeAgainstTime&
149 GetNbMoleculeAgainstTime(G4MolecularConfiguration* molecule);
150
151 RecordedMolecules GetRecordedMolecules();
152 RecordedTimes GetRecordedTimes();
153
154 inline void SetVerbose(G4int);
155 inline G4int GetVerbose();
156
157 /*
158 * It sets the min time difference in between two time slices.
159 */
160 void SetTimeSlice(double);
161
162
163 void Dump();
164
165 inline G4bool IsTimeCheckedForConsistency() const
166 {
167 return fCheckTimeIsConsistentWithScheduler;
168 }
169
170 inline void CheckTimeForConsistency(G4bool flag)
171 {
172 fCheckTimeIsConsistentWithScheduler = flag;
173 }
174
175 //----------------------------------------------------------------------------
176protected:
177 G4bool SearchTimeMap(G4MolecularConfiguration* molecule);
178 int SearchUpperBoundTime(double time, bool sameTypeOfMolecule);
179};
180
181//------------------------------------------------------------------------------
182
183inline void G4MoleculeCounter::ResetCounter()
184{
185 if(fVerbose)
186 {
187 G4cout << " ---> G4MoleculeCounter::ResetCounter" << G4endl;
188 }
189 fCounterMap.clear();
190 fpLastSearch.reset(0);
191}
192
193//------------------------------------------------------------------------------
194
195inline const NbMoleculeAgainstTime&
196G4MoleculeCounter::GetNbMoleculeAgainstTime(G4MolecularConfiguration* molecule)
197{
198 return fCounterMap[molecule];
199}
200
201//------------------------------------------------------------------------------
202
203inline void G4MoleculeCounter::SetVerbose(G4int level)
204{
205 fVerbose = level;
206}
207
208//------------------------------------------------------------------------------
209
210inline G4int G4MoleculeCounter::GetVerbose()
211{
212 return fVerbose;
213}
214
215//------------------------------------------------------------------------------
216
217inline void
218G4MoleculeCounter::DontRegister(const G4MoleculeDefinition* molDef)
219{
220 fDontRegister[molDef] = true;
221}
222
223//------------------------------------------------------------------------------
224
225inline bool
226G4MoleculeCounter::IsRegistered(const G4MoleculeDefinition* molDef)
227{
228 if(fDontRegister.find(molDef) == fDontRegister.end()) return true;
229 return false;
230}
231
232//------------------------------------------------------------------------------
233
234inline void G4MoleculeCounter::RegisterAll()
235{
236 fDontRegister.clear();
237}
