92/** Class Description
93 * The dynamic molecule holds all the data that change for a molecule
94 * It has a pointer to G4MoleculeDefinition object, which holds
95 * all the "ground level" information.
96 */
97
98class G4Molecule : public G4IT
99{
100
101public:
102 // With Description
103
104ITDef(G4Molecule)
105
106 //From G4VUserTrackInformation
107 void Print() const;
108
109 // new/delete operators are overloded to use G4Allocator
110 inline void *operator new(size_t);
111#ifdef __IBMCPP__
112 inline void *operator new(size_t sz, void* p)
113 { return p;}
114#endif
115 inline void operator delete(void*);
116
117 G4Molecule(const G4Molecule&);
118 G4Molecule & operator=(const G4Molecule &right);
119 G4bool operator==(const G4Molecule &right) const;
120 G4bool operator!=(const G4Molecule &right) const;
121 G4bool operator<(const G4Molecule &right) const;
122
123 operator int() const
124 {
125 return GetMoleculeID();
126 }
127
128 virtual G4ITType GetITSubType() const
129 {
130 return GetMoleculeID();
131 }
132
133public:
134 //------ Constructors --------------------------
135 /** To build a molecule at ground state according to a given
136 * G4MoleculeDefinition that can be obtained from G4GenericMoleculeManager
137 */
138 G4Molecule(G4MoleculeDefinition * molecule);
139
140 G4Molecule(G4MoleculeDefinition* molDef, int charge);
141
142 /** To build a molecule at a specific excitation/ionisation state according
143 * to a ground state that can be obtained from G4GenericMoleculeManager
144 */
145 G4Molecule(G4MoleculeDefinition * molecule, G4int, G4int);
146
147 /** Specific builder for water molecules to be used in Geant4-DNA,
148 * the last option Excitation is true if the molecule is excited, is
149 * false is the molecule is ionized.
150 */
151 G4Molecule(G4MoleculeDefinition * molecule, G4int, G4bool);
152
153 G4Molecule(G4MolecularConfiguration*);
154
155 virtual ~G4Molecule();
156
157 //-------- Methods -------------------------------
158 //Get from static definition
159 /** Returns the name of the molecule
160 */
161 const G4String& GetName() const;
162
163 /** Returns the formated name of the molecule
164 */
165 const G4String& GetFormatedName() const;
166
167 /** Returns the nomber of atoms compouning the molecule
168 */
169 G4int GetAtomsNumber() const;
170
171 /** Will set up the correct molecularConfiguration given
172 * an electron configuration
173 */
174 void SetElectronOccupancy(const G4ElectronOccupancy*);
175
176 /** Method used in Geant4-DNA to excite water molecules
177 */
178 void ExciteMolecule(G4int);
179
180 /** Method used in Geant4-DNA to ionize water molecules
181 */
182 void IonizeMolecule(G4int);
183
184 /** Add n electrons to a given orbit.
185 * Note : You can add as many electrons to a given orbit, the result
186 * may be unrealist.
187 */
188 void AddElectron(G4int orbit, G4int n = 1);
189
190 /** Remove n electrons to a given orbit.
191 */
192 void RemoveElectron(G4int, G4int number = 1);
193
194 /** Move one electron from an orbit to another.
195 */
196 void MoveOneElectron(G4int /*orbit*/, G4int /*orbit*/);
197
198 /** Returns the number of electron.
199 */
200 G4double GetNbElectrons() const; //This method can be used to check if the electron s number is physical
201
202 /** Show the electronic state of the molecule.
203 */
204 void PrintState() const;
205
206 G4Track * BuildTrack(G4double globalTime, const G4ThreeVector& Position);
207
208 G4double GetKineticEnergy() const;
209
210 G4double GetDiffusionVelocity() const;
211
212 const std::vector<const G4MolecularDissociationChannel*>* GetDecayChannel() const;
213
214 G4int GetFakeParticleID() const;
215 G4int GetMoleculeID() const;
216
217 //-------------Inline functions ---------------------
218 /** Get molecule definition. This G4MoleculeDefinition has the ground
219 * electronic state of the molecule.
220 */
221 const G4MoleculeDefinition* GetDefinition() const;
222
223 //methods to set/get changing parameters
224
225/////////////////////////////////////////////////////////////////////////////
226 /** Sets the diffusion coefficient D of the molecule used in diffusion
227 * processes to calculate the mean square jump distance between two
228 * changes of direction. In three dimension : <x^2> = 6 D t where t is
229 * the mean jump time between two changes of direction.
230 */
231 void SetDiffusionCoefficient(G4double);
232
233 /** Returns the diffusion coefficient D.
234 */
235 G4double GetDiffusionCoefficient() const;
236
237 /** Returns the diffusion coefficient D.
238 */
239 G4double GetDiffusionCoefficient(const G4Material*,
240 double temperature) const;
241
242 /** Set the decay time of the molecule.
243 */
244 void SetDecayTime(G4double);
245
246 /** Returns the decay time of the molecule.
247 */
248 G4double GetDecayTime() const;
249
250 /** The Van Der Valls Radius of the molecule
251 */
252 void SetVanDerVaalsRadius(G4double);
253 G4double GetVanDerVaalsRadius() const;
254
255 /** Returns the object ElectronOccupancy describing the electronic
256 * configuration of the molecule.
257 */
258 const G4ElectronOccupancy* GetElectronOccupancy() const;
259
260 /** Returns the charge of molecule.
261 */
262 G4int GetCharge() const;
263
264 /** Set the total mass of the molecule.
265 */
266 void SetMass(G4double);
267
268 /** Returns the total mass of the molecule.
269 */
270 G4double GetMass() const;
271
272 /** Returns the label of the molecule configuration
273 */
274 const G4String& GetLabel() const;
275
276 void SetLabel(const G4String& label);
277
278 void ChangeConfigurationToLabel(const G4String& label);
279
280////////////////////////////////////////////////////////////////////////
281
282 G4MolecularConfiguration* GetMolecularConfiguration() const;
283
284// static void SetGlobalTemperature(G4double);
285// static G4double GetGlobalTemperature();
286
287 static G4Molecule* GetMolecule(const G4Track*);
288
289private:
290 /** Default molecule builder
291 */
292 G4Molecule();
293
294 G4MolecularConfiguration* fpMolecularConfiguration;
295
296// double fCachedDiffusionCoefficient;
297
298// static /*G4ThreadLocal*/double fgTemperature;
299};

307//////////////////////////
308inline void * G4Molecule::operator new(size_t)
309//////////////////////////
310{
311 if (!aMoleculeAllocator()) aMoleculeAllocator() = new G4Allocator<G4Molecule>;
312 return (void *) aMoleculeAllocator()->MallocSingle();
313}
314
315//////////////////////////
316inline void G4Molecule::operator delete(void * aMolecule)
317//////////////////////////
318{
319 aMoleculeAllocator()->FreeSingle((G4Molecule *) aMolecule);
320}
