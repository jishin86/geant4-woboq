74 G4double e1, G4double e2)

118const G4PhysicsTable*
119G4TablesForExtrapolator::GetPhysicsTable(ExtTableType type) const
120{
121 const G4PhysicsTable* table = nullptr;
122 switch (type)
123 {
124 case fDedxElectron:
125 table = dedxElectron;
126 break;
127 case fDedxPositron:
128 table = dedxPositron;
129 break;
130 case fDedxProton:
131 table = dedxProton;
132 break;
133 case fDedxMuon:
134 table = dedxMuon;
135 break;
136 case fRangeElectron:
137 table = rangeElectron;
138 break;
139 case fRangePositron:
140 table = rangePositron;
141 break;
142 case fRangeProton:
143 table = rangeProton;
144 break;
145 case fRangeMuon:
146 table = rangeMuon;
147 break;
148 case fInvRangeElectron:
149 table = invRangeElectron;
150 break;
151 case fInvRangePositron:
152 table = invRangePositron;
153 break;
154 case fInvRangeProton:
155 table = invRangeProton;
156 break;
157 case fInvRangeMuon:
158 table = invRangeMuon;
159 break;
160 case fMscElectron:
161 table = mscElectron;
162 }
163 return table;
164}

168void G4TablesForExtrapolator::Initialisation()
169{
170 if(verbose>1) {
171 G4cout << "### G4TablesForExtrapolator::Initialisation" << G4endl;
172 }
173 currentParticle = nullptr;
174
175 electron = G4Electron::Electron();
176 positron = G4Positron::Positron();
177 proton = G4Proton::Proton();
178 muonPlus = G4MuonPlus::MuonPlus();
179 muonMinus= G4MuonMinus::MuonMinus();
180
181 mass = charge2 = 0.0;
182
183 nmat = G4Material::GetNumberOfMaterials();
184 const G4MaterialTable* mtable = G4Material::GetMaterialTable();
185 pcuts = new G4ProductionCuts();
186
187 couples.resize(nmat,0);
188 cuts.resize(nmat,DBL_MAX);
189
190 for(G4int i=0; i<nmat; ++i) {
191 couples[i] = new G4MaterialCutsCouple((*mtable)[i],pcuts);
192 }
193
194 splineFlag = G4EmParameters::Instance()->Spline();
195
196 dedxElectron = PrepareTable();
197 dedxPositron = PrepareTable();
198 dedxMuon = PrepareTable();
199 dedxProton = PrepareTable();
200 rangeElectron = PrepareTable();
201 rangePositron = PrepareTable();
202 rangeMuon = PrepareTable();
203 rangeProton = PrepareTable();
204 invRangeElectron = PrepareTable();
205 invRangePositron = PrepareTable();
206 invRangeMuon = PrepareTable();
207 invRangeProton = PrepareTable();
208 mscElectron = PrepareTable();
209
210 G4LossTableBuilder builder;
211
212 if(verbose>1) {
213 G4cout << "### G4TablesForExtrapolator Builds electron tables"
214 << G4endl;
215 }
216 ComputeElectronDEDX(electron, dedxElectron);
217 builder.BuildRangeTable(dedxElectron,rangeElectron);
218 builder.BuildInverseRangeTable(rangeElectron, invRangeElectron);
219
220 if(verbose>1) {
221 G4cout << "### G4TablesForExtrapolator Builds positron tables"
222 << G4endl;
223 }
224 ComputeElectronDEDX(positron, dedxPositron);
225 builder.BuildRangeTable(dedxPositron, rangePositron);
226 builder.BuildInverseRangeTable(rangePositron, invRangePositron);
227
228 if(verbose>1) {
229 G4cout << "### G4TablesForExtrapolator Builds muon tables" << G4endl;
230 }
231 ComputeMuonDEDX(muonPlus, dedxMuon);
232 builder.BuildRangeTable(dedxMuon, rangeMuon);
233 builder.BuildInverseRangeTable(rangeMuon, invRangeMuon);
234 /*
235 G4cout << "DEDX MUON" << G4endl
236 G4cout << *dedxMuon << G4endl;
237 G4cout << "RANGE MUON" << G4endl
238 G4cout << *rangeMuon << G4endl;
239 G4cout << "INVRANGE MUON" << G4endl
240 G4cout << *invRangeMuon << G4endl;
241 */
242 if(verbose>1) {
243 G4cout << "### G4TablesForExtrapolator Builds proton tables"
244 << G4endl;
245 }
246 ComputeProtonDEDX(proton, dedxProton);
247 builder.BuildRangeTable(dedxProton, rangeProton);
248 builder.BuildInverseRangeTable(rangeProton, invRangeProton);
249
250 ComputeTrasportXS(electron, mscElectron);
251}

255G4PhysicsTable* G4TablesForExtrapolator::PrepareTable()
256{
257 G4PhysicsTable* table = new G4PhysicsTable();
258
259 for(G4int i=0; i<nmat; ++i) {
260
261 G4PhysicsVector* v = new G4PhysicsLogVector(emin, emax, nbins);
262 v->SetSpline(splineFlag);
263 table->push_back(v);
264 }
265 return table;
266}

270void G4TablesForExtrapolator::ComputeElectronDEDX(
271 const G4ParticleDefinition* part,
272 G4PhysicsTable* table)
273{
274 G4MollerBhabhaModel* ioni = new G4MollerBhabhaModel();
275 G4eBremsstrahlungRelModel* brem = new G4eBremsstrahlungRelModel();
276 ioni->Initialise(part, cuts);
277 brem->Initialise(part, cuts);
278
279 mass = electron_mass_c2;
280 charge2 = 1.0;
281 currentParticle = part;
282
283 const G4MaterialTable* mtable = G4Material::GetMaterialTable();
284 if(0<verbose) {
285 G4cout << "G4TablesForExtrapolator::ComputeElectronDEDX for "
286 << part->GetParticleName()
287 << G4endl;
288 }
289 for(G4int i=0; i<nmat; ++i) {
290
291 const G4Material* mat = (*mtable)[i];
292 if(1<verbose) {
293 G4cout << "i= " << i << " mat= " << mat->GetName() << G4endl;
294 }
295 const G4MaterialCutsCouple* couple = couples[i];
296 G4PhysicsVector* aVector = (*table)[i];
297
298 for(G4int j=0; j<=nbins; ++j) {
299
300 G4double e = aVector->Energy(j);
301 G4double dedx = ioni->ComputeDEDX(couple,part,e,e)
302 + brem->ComputeDEDX(couple,part,e,e);
303 if(1<verbose) {
304 G4cout << "j= " << j
305 << " e(MeV)= " << e/MeV
306 << " dedx(Mev/cm)= " << dedx*cm/MeV
307 << " dedx(Mev.cm2/g)= "
308 << dedx/((MeV*mat->GetDensity())/(g/cm2))
309 << G4endl;
310 }
311 aVector->PutValue(j,dedx);
312 }
313 if(splineFlag) { aVector->FillSecondDerivatives(); }
314 }
315 delete ioni;
316 delete brem;
317}

321void
322G4TablesForExtrapolator::ComputeMuonDEDX(const G4ParticleDefinition* part,
323 G4PhysicsTable* table)
324{
325 G4BetheBlochModel* ioni = new G4BetheBlochModel();
326 G4MuPairProductionModel* pair = new G4MuPairProductionModel();
327 G4MuBremsstrahlungModel* brem = new G4MuBremsstrahlungModel();
328 ioni->Initialise(part, cuts);
329 pair->Initialise(part, cuts);
330 brem->Initialise(part, cuts);
331
332 mass = part->GetPDGMass();
333 charge2 = 1.0;
334 currentParticle = part;
335
336 const G4MaterialTable* mtable = G4Material::GetMaterialTable();
337
338 if(0<verbose) {
339 G4cout << "G4TablesForExtrapolator::ComputeMuonDEDX for "
340 << part->GetParticleName()
341 << G4endl;
342 }
343
344 for(G4int i=0; i<nmat; ++i) {
345
346 const G4Material* mat = (*mtable)[i];
347 if(1<verbose) {
348 G4cout << "i= " << i << " mat= " << mat->GetName() << G4endl;
349 }
350 const G4MaterialCutsCouple* couple = couples[i];
351 G4PhysicsVector* aVector = (*table)[i];
352 for(G4int j=0; j<=nbins; j++) {
353
354 G4double e = aVector->Energy(j);
355 G4double dedx = ioni->ComputeDEDX(couple,part,e,e) +
356 pair->ComputeDEDX(couple,part,e,e) +
357 brem->ComputeDEDX(couple,part,e,e);
358 aVector->PutValue(j,dedx);
359 if(1<verbose) {
360 G4cout << "j= " << j
361 << " e(MeV)= " << e/MeV
362 << " dedx(Mev/cm)= " << dedx*cm/MeV
363 << " dedx(Mev/(g/cm2)= "
364 << dedx/((MeV*mat->GetDensity())/(g/cm2))
365 << G4endl;
366 }
367 }
368 if(splineFlag) { aVector->FillSecondDerivatives(); }
369 }
370 delete ioni;
371}

421void
422G4TablesForExtrapolator::ComputeTrasportXS(const G4ParticleDefinition* part,
423 G4PhysicsTable* table)
424{
425 G4WentzelVIModel* msc = new G4WentzelVIModel();
426 msc->SetPolarAngleLimit(CLHEP::pi);
427 msc->Initialise(part, cuts);
428
429 mass = part->GetPDGMass();
430 charge2 = 1.0;
431 currentParticle = part;
432
433 const G4MaterialTable* mtable = G4Material::GetMaterialTable();
434
435 if(0<verbose) {
436 G4cout << "G4TablesForExtrapolator::ComputeProtonDEDX for "
437 << part->GetParticleName()
438 << G4endl;
439 }
440
441 for(G4int i=0; i<nmat; ++i) {
442
443 const G4Material* mat = (*mtable)[i];
444 msc->SetCurrentCouple(couples[i]);
445 if(1<verbose)
446 G4cout << "i= " << i << " mat= " << mat->GetName() << G4endl;
447 G4PhysicsVector* aVector = (*table)[i];
448 for(G4int j=0; j<=nbins; j++) {
449
450 G4double e = aVector->Energy(j);
451 G4double xs = msc->CrossSectionPerVolume(mat,part,e);
452 aVector->PutValue(j,xs);
453 if(1<verbose) {
454 G4cout << "j= " << j << " e(MeV)= " << e/MeV
455 << " xs(1/mm)= " << xs*mm << G4endl;
456 }
457 }
458 if(splineFlag) { aVector->FillSecondDerivatives(); }
459 }
460 delete msc;
461}
