73using namespace std;

75G4MuBremsstrahlung::G4MuBremsstrahlung(const G4String& name)
76 : G4VEnergyLossProcess(name),
77 lowestKinEnergy(1.*GeV),
78 isInitialised(false)
79{
80 SetProcessSubType(fBremsstrahlung);
81 SetSecondaryParticle(G4Gamma::Gamma());
82 SetIonisation(false);
83}
84
85//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
86
87G4bool G4MuBremsstrahlung::IsApplicable(const G4ParticleDefinition& p)
88{
89 return (p.GetPDGCharge() != 0.0 && p.GetPDGMass() > 10.0*MeV);
90}
91
92//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
93
94G4double G4MuBremsstrahlung::MinPrimaryEnergy(const G4ParticleDefinition*,
95 const G4Material*,
96 G4double)
97{
98 return lowestKinEnergy;
99}

101//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
102
103void G4MuBremsstrahlung::InitialiseEnergyLossProcess(
104 const G4ParticleDefinition*,
105 const G4ParticleDefinition*)
106{
107 if(!isInitialised) {
108
109 isInitialised = true;
110
111 if (!EmModel()) { SetEmModel(new G4MuBremsstrahlungModel()); }
112
113 G4VEmFluctuationModel* fm = nullptr;
114 G4EmParameters* param = G4EmParameters::Instance();
115 EmModel()->SetLowEnergyLimit(param->MinKinEnergy());
116 EmModel()->SetHighEnergyLimit(param->MaxKinEnergy());
117 AddEmModel(1, EmModel(), fm);
118 }
119}
