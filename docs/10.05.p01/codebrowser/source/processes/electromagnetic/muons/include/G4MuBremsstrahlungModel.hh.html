163inline void G4MuBremsstrahlungModel::SetLowestKineticEnergy(G4double e)
164{
165 lowestKinEnergy = e;
166}
167
168//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo......
169
170inline
171void G4MuBremsstrahlungModel::SetParticle(const G4ParticleDefinition* p)
172{
173 if(!particle) {
174 particle = p;
175 mass = particle->GetPDGMass();
176 rmass=mass/CLHEP::electron_mass_c2 ;
177 cc=CLHEP::classic_electr_radius/rmass ;
178 coeff= 16.*CLHEP::fine_structure_const*cc*cc/3. ;
179 }
180}
