64G4bool G4AdjointhMultipleScattering::IsApplicable (const G4ParticleDefinition& p)
65{
66 return (p.GetPDGCharge() != 0.0 && !p.IsShortLived());
67}

69//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo......
70
71void G4AdjointhMultipleScattering::InitialiseProcess(const G4ParticleDefinition*)
72{
73 if(isInitialized) { return; }
74 AddEmModel(1, new G4UrbanMscModel());
75 isInitialized = true;
76}
