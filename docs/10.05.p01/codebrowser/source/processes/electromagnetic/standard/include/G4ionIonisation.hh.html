77class G4ionIonisation : public G4VEnergyLossProcess
78{
79public:
80
81 explicit G4ionIonisation(const G4String& name = "ionIoni");
82
83 virtual ~G4ionIonisation();
84
85 virtual G4bool IsApplicable(const G4ParticleDefinition& p) final;
86
87 void AddStoppingData(G4int Z, G4int A, const G4String& materialName,
88 G4PhysicsVector* dVector);
89
90 void ActivateStoppingData(G4bool);
91
92 // print documentation in html format
93 virtual void ProcessDescription(std::ostream&) const override;
94
95protected:
96
97 // Print out of the class parameters
98 virtual void StreamProcessInfo(std::ostream& outFile) const override;
99
100 virtual void
101 InitialiseEnergyLossProcess(const G4ParticleDefinition*,
102 const G4ParticleDefinition*) override;
103
104 virtual G4double MinPrimaryEnergy(const G4ParticleDefinition* p,
105 const G4Material*, G4double cut) final;
106
107 inline G4double BetheBlochEnergyThreshold();
108
109private:
110
111 // hide assignment operator
112 G4ionIonisation & operator=(const G4ionIonisation &right) = delete;
113 G4ionIonisation(const G4ionIonisation&) = delete;
114
115 G4EmCorrections* corr;
116
117 const G4ParticleDefinition* theParticle;
118
119 G4double eth;
120
121 G4bool isInitialised;
122 G4bool stopDataActive;
123};
