75G4eSingleCoulombScatteringModel::G4eSingleCoulombScatteringModel(const G4String& nam)
76 : G4VEmModel(nam),
77 cosThetaMin(1.0)
78{
79 fNistManager = G4NistManager::Instance();
80 theIonTable = G4ParticleTable::GetParticleTable()->GetIonTable();
81 fParticleChange = nullptr;
82
83 pCuts=nullptr;
84 currentMaterial = nullptr;
85 currentElement = nullptr;
86 currentCouple = nullptr;
87
88 lowEnergyLimit = 0*keV;
89 recoilThreshold = 0.*eV;
90 XSectionModel = 1;
91 FormFactor = 0;
92 particle = nullptr;
93 mass=0.0;
94 currentMaterialIndex = -1;
95
96 Mottcross = new G4ScreeningMottCrossSection();
97}

108void G4eSingleCoulombScatteringModel::Initialise(const G4ParticleDefinition* p,
109 const G4DataVector& cuts)
110{
111 G4EmParameters* param = G4EmParameters::Instance();
112
113 SetupParticle(p);
114 currentCouple = nullptr;
115 currentMaterialIndex = -1;
116 //cosThetaMin = cos(PolarAngleLimit());
117 Mottcross->Initialise(p,cosThetaMin);
118
119 pCuts = &cuts;
120 //G4ProductionCutsTable::GetProductionCutsTable()->GetEnergyCutsVector(3);
121
122 /*
123 G4cout << "!!! G4eSingleCoulombScatteringModel::Initialise for "
124 << part->GetParticleName() << " cos(TetMin)= " << cosThetaMin
125 << " cos(TetMax)= " << cosThetaMax <<G4endl;
126 G4cout << "cut= " << (*pCuts)[0] << " cut1= " << (*pCuts)[1] << G4endl;
127 */
128
129 if(!fParticleChange) {
130 fParticleChange = GetParticleChangeForGamma();
131 }
132
133 if(IsMaster()) {
134 InitialiseElementSelectors(p,cuts);
135 }
136
137 FormFactor=param->NuclearFormfactorType();
138
139 //G4cout<<"NUCLEAR FORM FACTOR: "<<FormFactor<<G4endl;
140}

153void G4eSingleCoulombScatteringModel::SetXSectionModel(const G4String& model)
154{
155 if(model == "Fast" || model == "fast") { XSectionModel=1; }
156 else if(model == "Precise" || model == "precise") { XSectionModel=0; }
157 else {
158 G4cout<<"G4eSingleCoulombScatteringModel WARNING: "<<model
159 <<" is not a valid model name"<<G4endl;
160 }
161}

165G4double G4eSingleCoulombScatteringModel::ComputeCrossSectionPerAtom(
166 const G4ParticleDefinition* p,
167 G4double kinEnergy,
168 G4double Z,
169 G4double ,
170 G4double,
171 G4double )
172{
173 SetupParticle(p);
174
175 G4double cross =0.0;
176 if(kinEnergy < lowEnergyLimit) return cross;
177
178 DefineMaterial(CurrentCouple());
179
180 //Total Cross section
181 Mottcross->SetupKinematic(kinEnergy, Z);
182 cross = Mottcross->NuclearCrossSection(FormFactor,XSectionModel);
183
184 //cout<< "Compute Cross Section....cross "<<G4BestUnit(cross,"Surface") << " cm2 "<< cross/cm2 <<" Z: "<<Z<<" kinEnergy: "<<kinEnergy<<endl;
185
186 //G4cout<<"Energy: "<<kinEnergy/MeV<<" Total Cross: "<<cross<<G4endl;
187 return cross;
188}

192void G4eSingleCoulombScatteringModel::SampleSecondaries(
193 std::vector<G4DynamicParticle*>* fvect,
194 const G4MaterialCutsCouple* couple,
195 const G4DynamicParticle* dp,
196 G4double cutEnergy,
197 G4double)
198{
199 G4double kinEnergy = dp->GetKineticEnergy();
200 //cout<<"--- kinEnergy "<<kinEnergy<<endl;
201
202 if(kinEnergy < lowEnergyLimit) return;
203
204 DefineMaterial(couple);
205 SetupParticle(dp->GetDefinition());
206
207 // Choose nucleus
208 //last two :cutEnergy= min e kinEnergy=max
209 currentElement = SelectRandomAtom(couple, particle, kinEnergy,
210 cutEnergy, kinEnergy);
211 G4double Z = currentElement->GetZ();
212 G4int iz = G4int(Z);
213 G4int ia = SelectIsotopeNumber(currentElement);
214 G4double mass2 = G4NucleiProperties::GetNuclearMass(ia, iz);
215
216 //G4cout<<"..Z: "<<Z<<" ..iz: "<<iz<<" ..ia: "<<ia<<" ..mass2: "<<mass2<<G4endl;
217
218 Mottcross->SetupKinematic(kinEnergy, Z);
219 G4double cross= Mottcross->NuclearCrossSection(FormFactor,XSectionModel);
220 if(cross == 0.0) { return; }
221 //cout<< "Energy: "<<kinEnergy/MeV<<" Z: "<<Z<<"....cross "<<G4BestUnit(cross,"Surface") << " cm2 "<< cross/cm2 <<endl;
222
223 G4double z1 = Mottcross->GetScatteringAngle(FormFactor,XSectionModel);
224 G4double sint = sin(z1);
225 G4double cost = cos(z1);
226 G4double phi = twopi* G4UniformRand();
227
228 // kinematics in the Lab system
229 G4double ptot = sqrt(kinEnergy*(kinEnergy + 2.0*mass));
230 G4double e1 = mass + kinEnergy;
231
232 // Lab. system kinematics along projectile direction
233 G4LorentzVector v0 = G4LorentzVector(0, 0, ptot, e1+mass2);
234 G4LorentzVector v1 = G4LorentzVector(0, 0, ptot, e1);
235 G4ThreeVector bst = v0.boostVector();
236 v1.boost(-bst);
237 // CM projectile
238 G4double momCM = v1.pz();
239
240 // Momentum after scattering of incident particle
241 v1.setX(momCM*sint*cos(phi));
242 v1.setY(momCM*sint*sin(phi));
243 v1.setZ(momCM*cost);
244
245 // CM--->Lab
246 v1.boost(bst);
247
248 // Rotate to global system
249 G4ThreeVector dir = dp->GetMomentumDirection();
250 G4ThreeVector newDirection = v1.vect().unit();
251 newDirection.rotateUz(dir);
252
253 fParticleChange->ProposeMomentumDirection(newDirection);
254
255 // recoil
256 v0 -= v1;
257 G4double trec = std::max(v0.e() - mass2, 0.0);
258 G4double edep = 0.0;
259
260 G4double tcut = recoilThreshold;
261
262 //G4cout<<" Energy Transfered: "<<trec/eV<<G4endl;
263
264 if(pCuts) {
265 tcut= std::max(tcut,(*pCuts)[currentMaterialIndex]);
266 //G4cout<<"Cuts: "<<(*pCuts)[currentMaterialIndex]/eV<<" eV"<<G4endl;
267 //G4cout<<"Threshold: "<<tcut/eV<<" eV"<<G4endl;
268 }
269
270 if(trec > tcut) {
271 G4ParticleDefinition* ion = theIonTable->GetIon(iz, ia, 0);
272 newDirection = v0.vect().unit();
273 newDirection.rotateUz(dir);
274 G4DynamicParticle* newdp = new G4DynamicParticle(ion, newDirection, trec);
275 fvect->push_back(newdp);
276 } else if(trec > 0.0) {
277 edep = trec;
278 fParticleChange->ProposeNonIonizingEnergyDeposit(edep);
279 }
280
281 // finelize primary energy and energy balance
282 G4double finalT = v1.e() - mass;
283 //G4cout<<"Final Energy: "<<finalT/eV<<G4endl;
284 if(finalT <= lowEnergyLimit) {
285 edep += finalT;
286 finalT = 0.0;
287 }
288 edep = std::max(edep, 0.0);
289 fParticleChange->SetProposedKineticEnergy(finalT);
290 fParticleChange->ProposeLocalEnergyDeposit(edep);
291
292}
