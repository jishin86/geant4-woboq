344G4double G4ScreeningMottCrossSection::SetDifferentialXSection(
345 G4double angles, G4double step, G4int form)
346{
347 G4double Xsec=0;
348 G4double R=0;
349 G4double F2=0;
350
351 if(form==0){F2=1;}
352 if(form==1){F2=FormFactor2ExpHof(angles);}
353 if(form==2){F2=FormFactor2Gauss(angles);}
354 if(form==3){F2=FormFactor2UniformHelm(angles);}
355
356 // if (coeffb[0][0]!=0){
357 //G4cout<<" Mott....targetZ "<< targetZ<<G4endl;
358 R=RatioMottRutherford(angles);
359 //} else if (coeffb[0][0]==0){
360 //G4cout<<" McF.... targetZ "<< targetZ<<G4endl;
361 // R=McFcorrection(angles);
362 //}
363
364 G4double den=2.*As+2.*fG4pow->powN(sin(angles*0.5),2);
365 G4double func=1./(den*den);
366
367 G4double fatt= (G4double)targetZ/(mu_rel*gamma*beta*beta);
368 G4double sigma=e2*e2*fatt*fatt*func;
369 G4double pi2sintet=twopi*sin(angles);
370
371 Xsec=std::max(pi2sintet*F2*R*sigma*step, 0.);
372 return Xsec;
373}

379 fTotalCross=0.;
380 if(cosTetMaxNuc >= cosTetMinNuc) return fTotalCross;
381
382 //G4cout<<"MODEL: "<<FastComputation<<G4endl;

388 //G4cout<<i<<" Angle: "<<angle[i]<<" tet: "<<tet[i]<<" Cross: "<<cross[i]<G4endl;
389 fTotalCross+=cross[i];
390 }
391//**************** FAST COMPUTATION
392 }else if(FastComputation==1){
393
394 G4double p0 = electron_mass_c2*classic_electr_radius;
395 G4double coeff = twopi*p0*p0;
396
397 G4double fac = coeff*targetZ*targetZ*invbeta2/mom2;
398
399 G4double x = 1.0 - cosTetMinNuc;
400 G4double x1 = x + 2*As;
401
402 // scattering with nucleus
403 fTotalCross = fac*(cosTetMinNuc - cosTetMaxNuc)/
404 (x1*(1.0 - cosTetMaxNuc + 2*As));
405 }
406
407 // G4cout<<"Energy: "<<tkinLab/MeV<<" Total Cross: "<<fTotalCross<<G4endl;
408 return fTotalCross;
