57G4eIonisation::G4eIonisation(const G4String& name)
58 : G4VEnergyLossProcess(name),
59 theElectron(G4Electron::Electron()),
60 isElectron(true),
61 isInitialised(false)
62{
63 SetProcessSubType(fIonisation);
64 SetSecondaryParticle(theElectron);
65}

74G4double G4eIonisation::MinPrimaryEnergy(const G4ParticleDefinition*,
75 const G4Material*,
76 G4double cut)
77{
78 G4double x = cut;
79 if(isElectron) x += cut;
80 return x;
81}

85G4bool G4eIonisation::IsApplicable(const G4ParticleDefinition& p)
86{
87 return (&p == theElectron || &p == G4Positron::Positron());
88}

91
92void G4eIonisation::InitialiseEnergyLossProcess(
93 const G4ParticleDefinition* part,
94 const G4ParticleDefinition*)
95{
96 if(!isInitialised) {
97 if(part != theElectron) { isElectron = false; }
98 if (!EmModel(0)) { SetEmModel(new G4MollerBhabhaModel()); }
99 G4EmParameters* param = G4EmParameters::Instance();
100 EmModel(0)->SetLowEnergyLimit(param->MinKinEnergy());
101 EmModel(0)->SetHighEnergyLimit(param->MaxKinEnergy());
102 if (!FluctModel()) { SetFluctModel(new G4UniversalFluctuation()); }
103
104 AddEmModel(1, EmModel(), FluctModel());
105 isInitialised = true;
106 }
107}
