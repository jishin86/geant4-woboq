72G4PAIPhotModel::G4PAIPhotModel(const G4ParticleDefinition* p, const G4String& nam)
73 : G4VEmModel(nam),G4VEmFluctuationModel(nam),
74 fVerbose(0),
75 fModelData(nullptr),
76 fParticle(nullptr)
77{
78 fElectron = G4Electron::Electron();
79 fPositron = G4Positron::Positron();
80
81 fParticleChange = nullptr;
82
83 if(p) { SetParticle(p); }
84 else { SetParticle(fElectron); }
85
86 // default generator
87 SetAngularDistribution(new G4DeltaAngle());
88 fLowestTcut = 12.5*CLHEP::eV;
89}

101void G4PAIPhotModel::Initialise(const G4ParticleDefinition* p,
102 const G4DataVector& cuts)
103{
104 if(fVerbose > 0)
105 {
106 G4cout<<"G4PAIPhotModel::Initialise for "<<p->GetParticleName()<<G4endl;
107 }
108 SetParticle(p);
109 fParticleChange = GetParticleChangeForLoss();
110
111 if( IsMaster() )
112 {
113
114 InitialiseElementSelectors(p, cuts);
115
116 delete fModelData;
117 fMaterialCutsCoupleVector.clear();
118
119 G4double tmin = LowEnergyLimit()*fRatio;
120 G4double tmax = HighEnergyLimit()*fRatio;
121 fModelData = new G4PAIPhotData(tmin, tmax, fVerbose);
122
123 // Prepare initialization
124 const G4MaterialTable* theMaterialTable = G4Material::GetMaterialTable();
125 size_t numOfMat = G4Material::GetNumberOfMaterials();
126 size_t numRegions = fPAIRegionVector.size();
127
128 // protect for unit tests
129 if(0 == numRegions) {
130 G4Exception("G4PAIModel::Initialise()","em0106",JustWarning,
131 "no G4Regions are registered for the PAI model - World is used");
132 fPAIRegionVector.push_back(G4RegionStore::GetInstance()
133 ->GetRegion("DefaultRegionForTheWorld", false));
134 numRegions = 1;
135 }
136
137 for( size_t iReg = 0; iReg < numRegions; ++iReg )
138 {
139 const G4Region* curReg = fPAIRegionVector[iReg];
140 G4Region* reg = const_cast<G4Region*>(curReg);
141
142 for(size_t jMat = 0; jMat < numOfMat; ++jMat)
143 {
144 G4Material* mat = (*theMaterialTable)[jMat];
145 const G4MaterialCutsCouple* cutCouple = reg->FindCouple(mat);
146 //G4cout << "Couple <" << fCutCouple << G4endl;
147 if(cutCouple)
148 {
149 if(fVerbose>0)
150 {
151 G4cout << "Reg <" <<curReg->GetName() << "> mat <"
152 << mat->GetName() << "> fCouple= "
153 << cutCouple << ", idx= " << cutCouple->GetIndex()
154 <<" " << p->GetParticleName()
155 <<", cuts.size() = " << cuts.size() << G4endl;
156 }
157 // check if this couple is not already initialized
158
159 size_t n = fMaterialCutsCoupleVector.size();
160
161 G4bool isnew = true;
162 if( 0 < n )
163 {
164 for(size_t i=0; i<fMaterialCutsCoupleVector.size(); ++i)
165 {
166 if(cutCouple == fMaterialCutsCoupleVector[i]) {
167 isnew = false;
168 break;
169 }
170 }
171 }
172 // initialise data banks
173 if(isnew) {
174 fMaterialCutsCoupleVector.push_back(cutCouple);
175 G4double deltaCutInKinEnergy = cuts[cutCouple->GetIndex()];
176 fModelData->Initialise(cutCouple, deltaCutInKinEnergy, this);
177 }
178 }
179 }
180 }
181 }
182}

204G4double G4PAIPhotModel::ComputeDEDXPerVolume(const G4Material*,
205 const G4ParticleDefinition* p,
206 G4double kineticEnergy,
207 G4double cutEnergy)
208{
209 G4int coupleIndex = FindCoupleIndex(CurrentCouple());
210 if(0 > coupleIndex) { return 0.0; }
211
212 G4double cut = std::min(MaxSecondaryEnergy(p, kineticEnergy), cutEnergy);
213
214 G4double scaledTkin = kineticEnergy*fRatio;
215
216 return fChargeSquare*fModelData->DEDXPerVolume(coupleIndex, scaledTkin, cut);
217}

219/////////////////////////////////////////////////////////////////////////
220
221G4double G4PAIPhotModel::CrossSectionPerVolume( const G4Material*,
222 const G4ParticleDefinition* p,
223 G4double kineticEnergy,
224 G4double cutEnergy,
225 G4double maxEnergy )
226{
227 G4int coupleIndex = FindCoupleIndex(CurrentCouple());
228
229 if(0 > coupleIndex) return 0.0;
230
231 G4double tmax = std::min(MaxSecondaryEnergy(p, kineticEnergy), maxEnergy);
232
233 if(tmax <= cutEnergy) return 0.0;
234
235 G4double scaledTkin = kineticEnergy*fRatio;
236 G4double xsc = fChargeSquare*fModelData->CrossSectionPerVolume(coupleIndex,
237 scaledTkin,
238 cutEnergy, tmax);
239 return xsc;
240}

243//
244// It is analog of PostStepDoIt in terms of secondary electron.
245//
246
247void G4PAIPhotModel::SampleSecondaries(std::vector<G4DynamicParticle*>* vdp,
248 const G4MaterialCutsCouple* matCC,
249 const G4DynamicParticle* dp,
250 G4double tmin,
251 G4double maxEnergy)
252{
253 G4int coupleIndex = FindCoupleIndex(matCC);
254 if(0 > coupleIndex) { return; }
255
256 SetParticle(dp->GetDefinition());
257
258 G4double kineticEnergy = dp->GetKineticEnergy();
259
260 G4double tmax = MaxSecondaryEnergy(fParticle, kineticEnergy);
261
262 if( maxEnergy < tmax) tmax = maxEnergy;
263 if( tmin >= tmax) return;
264
265 G4ThreeVector direction = dp->GetMomentumDirection();
266 G4double scaledTkin = kineticEnergy*fRatio;
267 G4double totalEnergy = kineticEnergy + fMass;
268 G4double totalMomentum = sqrt(kineticEnergy*(totalEnergy + fMass));
269 G4double plRatio = fModelData->GetPlasmonRatio(coupleIndex, scaledTkin);
270
271 if( G4UniformRand() <= plRatio )
272 {
273 G4double deltaTkin = fModelData->SamplePostStepPlasmonTransfer(coupleIndex, scaledTkin);
274
275 // G4cout<<"G4PAIPhotModel::SampleSecondaries; dp "<<dp->GetParticleDefinition()->GetParticleName()
276 // <<"; Tkin = "<<kineticEnergy/keV<<" keV; transfer = "<<deltaTkin/keV<<" keV "<<G4endl;
277
278 if( deltaTkin <= 0. && fVerbose > 0)
279 {
280 G4cout<<"G4PAIPhotModel::SampleSecondary e- deltaTkin = "<<deltaTkin<<G4endl;
281 }
282 if( deltaTkin <= 0.) return;
283
284 if( deltaTkin > tmax) deltaTkin = tmax;
285
286 const G4Element* anElement = SelectRandomAtom(matCC,fParticle,kineticEnergy);
287 G4int Z = G4lrint(anElement->GetZ());
288
289 G4DynamicParticle* deltaRay = new G4DynamicParticle(fElectron,
290 GetAngularDistribution()->SampleDirection(dp, deltaTkin,
291 Z, matCC->GetMaterial()),
292 deltaTkin);
293
294 // primary change
295
296 kineticEnergy -= deltaTkin;
297
298 if( kineticEnergy <= 0. ) // kill primary as local? energy deposition
299 {
300 fParticleChange->SetProposedKineticEnergy(0.0);
301 fParticleChange->ProposeLocalEnergyDeposit(kineticEnergy+deltaTkin);
302 // fParticleChange->ProposeTrackStatus(fStopAndKill);
303 return;
304 }
305 else
306 {
307 G4ThreeVector dir = totalMomentum*direction - deltaRay->GetMomentum();
308 direction = dir.unit();
309 fParticleChange->SetProposedKineticEnergy(kineticEnergy);
310 fParticleChange->SetProposedMomentumDirection(direction);
311 vdp->push_back(deltaRay);
312 }
313 }
314 else // secondary X-ray CR photon
315 {
316 G4double deltaTkin = fModelData->SamplePostStepPhotonTransfer(coupleIndex, scaledTkin);
317
318 // G4cout<<"PAIPhotonModel PhotonPostStepTransfer = "<<deltaTkin/keV<<" keV"<<G4endl;
319
320 if( deltaTkin <= 0. )
321 {
322 G4cout<<"G4PAIPhotonModel::SampleSecondary gamma deltaTkin = "<<deltaTkin<<G4endl;
323 }
324 if( deltaTkin <= 0.) return;
325
326 if( deltaTkin >= kineticEnergy ) // stop primary
327 {
328 deltaTkin = kineticEnergy;
329 kineticEnergy = 0.0;
330 }
331 G4double costheta = 0.; // G4UniformRand(); // VG: ??? for start only
332 G4double sintheta = sqrt((1.+costheta)*(1.-costheta));
333
334 // direction of the 'Cherenkov' photon
335 G4double phi = twopi*G4UniformRand();
336 G4double dirx = sintheta*cos(phi), diry = sintheta*sin(phi), dirz = costheta;
337
338 G4ThreeVector deltaDirection(dirx,diry,dirz);
339 deltaDirection.rotateUz(direction);
340
341 if( kineticEnergy > 0.) // primary change
342 {
343 kineticEnergy -= deltaTkin;
344 fParticleChange->SetProposedKineticEnergy(kineticEnergy);
345 }
346 else // stop primary, but pass X-ray CR
347 {
348 // fParticleChange->ProposeLocalEnergyDeposit(deltaTkin);
349 fParticleChange->SetProposedKineticEnergy(0.0);
350 }
351 // create G4DynamicParticle object for photon ray
352
353 G4DynamicParticle* photonRay = new G4DynamicParticle;
354 photonRay->SetDefinition( G4Gamma::Gamma() );
355 photonRay->SetKineticEnergy( deltaTkin );
356 photonRay->SetMomentumDirection(deltaDirection);
357
358 vdp->push_back(photonRay);
359 }
360 return;
361}

365G4double G4PAIPhotModel::SampleFluctuations( const G4MaterialCutsCouple* matCC,
366 const G4DynamicParticle* aParticle,
367 G4double, G4double step,
368 G4double eloss)
369{
370 // return 0.;
371 G4int coupleIndex = FindCoupleIndex(matCC);
372 if(0 > coupleIndex) { return eloss; }
373
374 SetParticle(aParticle->GetDefinition());
375
376
377 // G4cout << "G4PAIPhotModel::SampleFluctuations step(mm)= "<< step/mm
378 // << " Eloss(keV)= " << eloss/keV << " in "
379 // << matCC->GetMaterial()->GetName() << G4endl;
380
381
382 G4double Tkin = aParticle->GetKineticEnergy();
383 G4double scaledTkin = Tkin*fRatio;
384
385 G4double loss = fModelData->SampleAlongStepPhotonTransfer(coupleIndex, Tkin,
386 scaledTkin,
387 step*fChargeSquare);
388 loss += fModelData->SampleAlongStepPlasmonTransfer(coupleIndex, Tkin,
389 scaledTkin,
390 step*fChargeSquare);
391
392
393 // G4cout<<" PAIPhotModel::SampleFluctuations loss = "<<loss/keV<<" keV, on step = "
394 // <<step/mm<<" mm"<<G4endl;
395 return loss;
396
397}

404
405G4double G4PAIPhotModel::Dispersion( const G4Material* material,
406 const G4DynamicParticle* aParticle,
407 G4double tmax,
408 G4double step )
409{
410 G4double particleMass = aParticle->GetMass();
411 G4double electronDensity = material->GetElectronDensity();
412 G4double kineticEnergy = aParticle->GetKineticEnergy();
413 G4double q = aParticle->GetCharge()/eplus;
414 G4double etot = kineticEnergy + particleMass;
415 G4double beta2 = kineticEnergy*(kineticEnergy + 2.0*particleMass)/(etot*etot);
416 G4double siga = (1.0/beta2 - 0.5) * twopi_mc2_rcl2 * tmax * step
417 * electronDensity * q * q;
418
419 return siga;
420 /*
421 G4double loss, sumLoss=0., sumLoss2=0., sigma2, meanLoss=0.;
422 for(G4int i = 0; i < fMeanNumber; i++)
423 {
424 loss = SampleFluctuations(material,aParticle,tmax,step,meanLoss);
425 sumLoss += loss;
426 sumLoss2 += loss*loss;
427 }
428 meanLoss = sumLoss/fMeanNumber;
429 sigma2 = meanLoss*meanLoss + (sumLoss2-2*sumLoss*meanLoss)/fMeanNumber;
430 return sigma2;
431 */
432}

434/////////////////////////////////////////////////////////////////////
435
436G4double G4PAIPhotModel::MaxSecondaryEnergy( const G4ParticleDefinition* p,
437 G4double kinEnergy)
438{
439 SetParticle(p);
440 G4double tmax = kinEnergy;
441 if(p == fElectron) { tmax *= 0.5; }
442 else if(p != fPositron) {
443 G4double ratio= electron_mass_c2/fMass;
444 G4double gamma= kinEnergy/fMass + 1.0;
445 tmax = 2.0*electron_mass_c2*(gamma*gamma - 1.) /
446 (1. + 2.0*gamma*ratio + ratio*ratio);
447 }
448 return tmax;
449}
