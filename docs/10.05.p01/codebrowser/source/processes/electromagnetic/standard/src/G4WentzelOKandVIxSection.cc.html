70G4WentzelOKandVIxSection::G4WentzelOKandVIxSection(G4bool comb) :
71 temp(0.,0.,0.),
72 numlimit(0.1),
73 nwarnings(0),
74 nwarnlimit(50),
75 isCombined(comb),
76 cosThetaMax(-1.0),
77 alpha2(fine_structure_const*fine_structure_const)
78{
79 fNistManager = G4NistManager::Instance();
80 fG4pow = G4Pow::GetInstance();
81 fMottXSection = nullptr;
82
83 theElectron = G4Electron::Electron();
84 thePositron = G4Positron::Positron();
85 theProton = G4Proton::Proton();
86
87 lowEnergyLimit = 1.0*eV;
88 G4double p0 = electron_mass_c2*classic_electr_radius;
89 coeff = twopi*p0*p0;
90 particle = nullptr;
91
92 fNucFormfactor = fExponentialNF;
93
94 currentMaterial = nullptr;
95 factB = factD = formfactA = screenZ = 0.0;
96 cosTetMaxElec = cosTetMaxNuc = invbeta2 = kinFactor = fMottFactor
97 = gam0pcmp = pcmp2 = 1.0;
98
99 factB1= 0.5*CLHEP::pi*fine_structure_const;
100
101 tkin = mom2 = momCM2 = factorA2 = mass = spin = chargeSquare = charge3 = 0.0;
102 ecut = etag = DBL_MAX;
103 targetZ = 0;
104 targetMass = CLHEP::proton_mass_c2;
105}

116void G4WentzelOKandVIxSection::Initialise(const G4ParticleDefinition* p,
117 G4double cosThetaLim)
118{
119 SetupParticle(p);
120 tkin = mom2 = momCM2 = 0.0;
121 ecut = etag = DBL_MAX;
122 targetZ = 0;
123
124 // cosThetaMax is below 1.0 only when MSC is combined with SS
125 if(isCombined) { cosThetaMax = cosThetaLim; }
126
127 G4double a = G4EmParameters::Instance()->FactorForAngleLimit()
128 *CLHEP::hbarc/CLHEP::fermi;
129 factorA2 = 0.5*a*a;
130 currentMaterial = nullptr;
131
132 fNucFormfactor = G4EmParameters::Instance()->NuclearFormfactorType();
133 if(0.0 == ScreenRSquare[0]) { InitialiseA(); }
134
135 // Mott corrections
136 if((p == theElectron || p == thePositron) && !fMottXSection) {
137 fMottXSection = new G4ScreeningMottCrossSection();
138 fMottXSection->Initialise(p, 1.0);
139 }
140}

144void G4WentzelOKandVIxSection::InitialiseA()
145{
146 // Thomas-Fermi screening radii
147 // Formfactors from A.V. Butkevich et al., NIM A 488 (2002) 282
148#ifdef G4MULTITHREADED
149 G4MUTEXLOCK(&G4WentzelOKandVIxSection::WentzelOKandVIxSectionMutex);
150#endif
151 if(0.0 == ScreenRSquare[0]) {
152 G4double a0 = electron_mass_c2/0.88534;
153 G4double constn = 6.937e-6/(MeV*MeV);
154 G4double fct = G4EmParameters::Instance()->ScreeningFactor();
155
156 G4double afact = fct*0.5*alpha2*a0*a0;
157 ScreenRSquare[0] = afact;
158 ScreenRSquare[1] = afact;
159 ScreenRSquareElec[1] = afact;
160 FormFactor[1] = 3.097e-6/(MeV*MeV);
161
162 for(G4int j=2; j<100; ++j) {
163 G4double x = fG4pow->Z13(j);
164 ScreenRSquare[j] = afact*(1 + G4Exp(-j*j*0.001))*x*x;
165 ScreenRSquareElec[j] = afact*x*x;
166 x = fNistManager->GetA27(j);
167 FormFactor[j] = constn*x*x;
168 }
169 }
170#ifdef G4MULTITHREADED
171 G4MUTEXUNLOCK(&G4WentzelOKandVIxSection::WentzelOKandVIxSectionMutex);
172#endif
173
174 //G4cout << "G4WentzelOKandVIxSection::Initialise mass= " << mass
175 // << " " << p->GetParticleName()
176 // << " cosThetaMax= " << cosThetaMax << G4endl;
177
178}

182void G4WentzelOKandVIxSection::SetupParticle(const G4ParticleDefinition* p)
183{
184 /*
185 G4cout << "G4WentzelOKandVIxSection::SetupParticle " << p
186 << " " << particle << " " << this << G4endl;
187 G4cout << this << " " << p->GetParticleName() << G4endl;
188 */
189 particle = p;
190 mass = particle->GetPDGMass();
191 spin = particle->GetPDGSpin();
192 if(0.0 != spin) { spin = 0.5; }
193 G4double q = std::abs(particle->GetPDGCharge()/eplus);
194 chargeSquare = q*q;
195 charge3 = chargeSquare*q;
196 tkin = 0.0;
197 currentMaterial = nullptr;
198 targetZ = 0;
199}

203G4double
204G4WentzelOKandVIxSection::SetupKinematic(G4double ekin, const G4Material* mat)
205{
206 if(ekin != tkin || mat != currentMaterial) {
207 currentMaterial = mat;
208 tkin = ekin;
209 mom2 = tkin*(tkin + 2.0*mass);
210 invbeta2 = 1.0 + mass*mass/mom2;
211 factB = spin/invbeta2;
212 cosTetMaxNuc = isCombined ?
213 std::max(cosThetaMax, 1.-factorA2*mat->GetIonisation()->GetInvA23()/mom2)
214 : cosThetaMax;
215 }
216 return cosTetMaxNuc;
217}

219//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
220
221G4double
222G4WentzelOKandVIxSection::SetupTarget(G4int Z, G4double cut)
223{
224 G4double cosTetMaxNuc2 = cosTetMaxNuc;
225 if(Z != targetZ || tkin != etag) {
226 etag = tkin;
227 targetZ = std::min(Z, 99);
228 G4double massT = (1 == Z) ? CLHEP::proton_mass_c2 :
229 fNistManager->GetAtomicMassAmu(Z)*CLHEP::amu_c2;
230 SetTargetMass(massT);
231
232 kinFactor = coeff*Z*chargeSquare*invbeta2/mom2;
233 if(particle == theElectron && fMottXSection) {
234 fMottFactor = (1.0 + 2.0e-4*Z*Z);
235 }
236
237 if(1 == Z) {
238 screenZ = ScreenRSquare[targetZ]/mom2;
239 } else if(mass > MeV) {
240 screenZ = std::min(Z*1.13,1.13 +3.76*Z*Z*invbeta2*alpha2*chargeSquare)*
241 ScreenRSquare[targetZ]/mom2;
242 } else {
243 G4double tau = tkin/mass;
244 screenZ = std::min(Z*1.13,(1.13 +3.76*Z*Z
245 *invbeta2*alpha2*std::sqrt(tau/(tau + fG4pow->Z23(targetZ)))))*
246 ScreenRSquareElec[targetZ]/mom2;
247 }
248 if(targetZ == 1 && particle == theProton && cosTetMaxNuc2 < 0.0) {
249 cosTetMaxNuc2 = 0.0;
250 }
251 formfactA = FormFactor[targetZ]*mom2;
252
253 cosTetMaxElec = 1.0;
254 ComputeMaxElectronScattering(cut);
255 }
256 return cosTetMaxNuc2;
257}

261G4double
262G4WentzelOKandVIxSection::ComputeTransportCrossSectionPerAtom(G4double cosTMax)
263{
264 G4double xSection = 0.0;
265 if(cosTMax >= 1.0) { return xSection; }
266
267 G4double costm = std::max(cosTMax,cosTetMaxElec);
268 G4double fb = screenZ*factB;
269
270 // scattering off electrons
271 if(costm < 1.0) {
272 G4double x = (1.0 - costm)/screenZ;
273 if(x < numlimit) {
274 G4double x2 = 0.5*x*x;
275 xSection = x2*((1.0 - 1.3333333*x + 3*x2) - fb*x*(0.6666667 - x));
276 } else {
277 G4double x1= x/(1 + x);
278 G4double xlog = G4Log(1.0 + x);
279 xSection = xlog - x1 - fb*(x + x1 - 2*xlog);
280 }
281
282 if(xSection < 0.0) {
283 ++nwarnings;
284 if(nwarnings < nwarnlimit) {
285 G4cout << "G4WentzelOKandVIxSection::ComputeTransportCrossSectionPerAtom"
286 << " scattering on e- <0"
287 << G4endl;
288 G4cout << "cross= " << xSection
289 << " e(MeV)= " << tkin << " p(MeV/c)= " << sqrt(mom2)
290 << " Z= " << targetZ << " "
291 << particle->GetParticleName() << G4endl;
292 G4cout << " 1-costm= " << 1.0-costm << " screenZ= " << screenZ
293 << " x= " << x << G4endl;
294 }
295 xSection = 0.0;
296 }
297 }
298 /*
299 G4cout << "G4WentzelOKandVIxSection::ComputeTransportCrossSectionPerAtom: \n"
300 << " Z= " << targetZ
301 << " e(MeV)= " << tkin/MeV << " XSel= " << xSection
302 << " zmaxE= " << (1.0 - cosTetMaxElec)/screenZ
303 << " zmaxN= " << (1.0 - cosThetaMax)/screenZ
304 << " 1-costm= " << 1.0 - cosThetaMax << G4endl;
305 */
306 // scattering off nucleus
307 if(cosTMax < 1.0) {
308 G4double x = (1.0 - cosTMax)/screenZ;
309 G4double y;
310 if(x < numlimit) {
311 G4double x2 = 0.5*x*x;
312 y = x2*((1.0 - 1.3333333*x + 3*x2) - fb*x*(0.6666667 - x));
313 } else {
314 G4double x1= x/(1 + x);
315 G4double xlog = G4Log(1.0 + x);
316 y = xlog - x1 - fb*(x + x1 - 2*xlog);
317 }
318
319 if(y < 0.0) {
320 ++nwarnings;
321 if(nwarnings < nwarnlimit) {
322 G4cout << "G4WentzelOKandVIxSection::ComputeTransportCrossSectionPerAtom"
323 << " scattering on nucleus <0"
324 << G4endl;
325 G4cout << "y= " << y
326 << " e(MeV)= " << tkin << " Z= " << targetZ << " "
327 << particle->GetParticleName() << G4endl;
328 G4cout << " formfactA= " << formfactA << " screenZ= " << screenZ
329 << " x= " << x <<G4endl;
330 }
331 y = 0.0;
332 }
333 xSection += y*targetZ;
334 }
335 xSection *= kinFactor;
336
337 /*
338 G4cout << "Z= " << targetZ << " XStot= " << xSection/barn
339 << " screenZ= " << screenZ << " formF= " << formfactA
340 << " for " << particle->GetParticleName()
341 << " m= " << mass << " 1/v= " << sqrt(invbeta2)
342 << " p= " << sqrt(mom2)
343 << " x= " << x << G4endl;
344 */
345 return xSection;
346}

350G4ThreeVector&
351G4WentzelOKandVIxSection::SampleSingleScattering(G4double cosTMin,
352 G4double cosTMax,
353 G4double elecRatio)
354{
355 temp.set(0.0,0.0,1.0);
356 CLHEP::HepRandomEngine* rndmEngineMod = G4Random::getTheEngine();
357
358 G4double formf = formfactA;
359 G4double cost1 = cosTMin;
360 G4double cost2 = cosTMax;
361 if(elecRatio > 0.0) {
362 if(rndmEngineMod->flat() <= elecRatio) {
363 formf = 0.0;
364 cost1 = std::max(cost1,cosTetMaxElec);
365 cost2 = std::max(cost2,cosTetMaxElec);
366 }
367 }
368 if(cost1 > cost2) {
369
370 G4double w1 = 1. - cost1 + screenZ;
371 G4double w2 = 1. - cost2 + screenZ;
372 G4double z1 = w1*w2/(w1 + rndmEngineMod->flat()*(w2 - w1)) - screenZ;
373
374 G4double fm = 1.0;
375 if(fNucFormfactor == fExponentialNF) {
376 fm += formf*z1;
377 fm = 1.0/(fm*fm);
378 } else if(fNucFormfactor == fGaussianNF) {
379 fm = G4Exp(-2*formf*z1);
380 } else if(fNucFormfactor == fFlatNF) {
381 static const G4double ccoef = 0.00508/MeV;
382 G4double x = std::sqrt(2.*mom2*z1)*ccoef*2.;
383 fm = FlatFormfactor(x);
384 fm *= FlatFormfactor(x*0.6
385 *fG4pow->A13(fNistManager->GetAtomicMassAmu(targetZ)));
386 }
387 G4double grej;
388 if(fMottXSection) {
389 fMottXSection->SetupKinematic(tkin, (G4double)targetZ);
390 grej = fMottXSection->RatioMottRutherfordCosT(std::sqrt(z1))*fm*fm;
391 } else {
392 grej = (1. - z1*factB + factB1*targetZ*sqrt(z1*factB)*(2. - z1))
393 *fm*fm/(1.0 + z1*factD);
394 }
395 //G4cout << "SampleSingleScattering: E= " << tkin << " z1= "
396 // << z1 << " grej= " << grej << G4endl;
397 if(fMottFactor*rndmEngineMod->flat() <= grej ) {
398 // exclude "false" scattering due to formfactor and spin effect
399 G4double cost = 1.0 - z1;
400 if(cost > 1.0) { cost = 1.0; }
401 else if(cost < -1.0) { cost =-1.0; }
402 G4double sint = sqrt((1.0 - cost)*(1.0 + cost));
403 //G4cout << "sint= " << sint << G4endl;
404 G4double phi = twopi*rndmEngineMod->flat();
405 temp.set(sint*cos(phi),sint*sin(phi),cost);
406 }
407 }
408 return temp;
409}

413void
414G4WentzelOKandVIxSection::ComputeMaxElectronScattering(G4double cutEnergy)
415{
416 if(mass > MeV) {
417 G4double ratio = electron_mass_c2/mass;
418 G4double tau = tkin/mass;
419 G4double tmax = 2.0*electron_mass_c2*tau*(tau + 2.)/
420 (1.0 + 2.0*ratio*(tau + 1.0) + ratio*ratio);
421 cosTetMaxElec = 1.0 - std::min(cutEnergy, tmax)*electron_mass_c2/mom2;
422 } else {
423
424 G4double tmax = (particle == theElectron) ? 0.5*tkin : tkin;
425 G4double t = std::min(cutEnergy, tmax);
426 G4double mom21 = t*(t + 2.0*electron_mass_c2);
427 G4double t1 = tkin - t;
428 //G4cout <<"tkin=" <<tkin<<" tmax= "<<tmax<<" t= "
429 //<<t<< " t1= "<<t1<<" cut= "<<ecut<<G4endl;
430 if(t1 > 0.0) {
431 G4double mom22 = t1*(t1 + 2.0*mass);
432 G4double ctm = (mom2 + mom22 - mom21)*0.5/sqrt(mom2*mom22);
433 if(ctm < 1.0) { cosTetMaxElec = ctm; }
434 if(particle == theElectron && cosTetMaxElec < 0.0) {
435 cosTetMaxElec = 0.0;
436 }
437 }
438 }
439}
