276 static const G4double hecorr[15] = {
277 120.70, 117.50, 105.00, 92.92, 79.23, 74.510, 68.29,
278 57.39, 41.97, 36.14, 24.53, 10.21, -7.855, -16.84,
279 -22.30};
280
281 G4double sigma;
282 SetParticle(part);
283
284 Z23 = G4Pow::GetInstance()->Z23(G4lrint(AtomicNumber));
285
286 // correction if particle .ne. e-/e+
287 // compute equivalent kinetic energy
288 // lambda depends on p*beta ....
289
290 G4double eKineticEnergy = KineticEnergy;
291
292 if(mass > electron_mass_c2)
293 {
294 G4double TAU = KineticEnergy/mass ;
295 G4double c = mass*TAU*(TAU+2.)/(electron_mass_c2*(TAU+1.)) ;
296 G4double w = c-2. ;
297 G4double tau = 0.5*(w+sqrt(w*w+4.*c)) ;
298 eKineticEnergy = electron_mass_c2*tau ;
299 }
300
301 G4double eTotalEnergy = eKineticEnergy + electron_mass_c2 ;
302 G4double beta2 = eKineticEnergy*(eTotalEnergy+electron_mass_c2)
303 /(eTotalEnergy*eTotalEnergy);
304 G4double bg2 = eKineticEnergy*(eTotalEnergy+electron_mass_c2)
305 /(electron_mass_c2*electron_mass_c2);
306
307 static const G4double epsfactor = 2.*CLHEP::electron_mass_c2*
308 CLHEP::electron_mass_c2*CLHEP::Bohr_radius*CLHEP::Bohr_radius
309 /(CLHEP::hbarc*CLHEP::hbarc);
310 G4double eps = epsfactor*bg2/Z23;
311
312 if (eps<epsmin) sigma = 2.*eps*eps;
313 else if(eps<epsmax) sigma = G4Log(1.+2.*eps)-2.*eps/(1.+2.*eps);
314 else sigma = G4Log(2.*eps)-1.+1./eps;
315
316 sigma *= ChargeSquare*AtomicNumber*AtomicNumber/(beta2*bg2);
317
318 // interpolate in AtomicNumber and beta2
319 G4double c1,c2,cc1,cc2,corr;
320
321 // get bin number in Z
322 G4int iZ = 14;
323 // Loop checking, 03-Aug-2015, Vladimir Ivanchenko
324 while ((iZ>=0)&&(Zdat[iZ]>=AtomicNumber)) iZ -= 1;
325 if (iZ==14) iZ = 13;
326 if (iZ==-1) iZ = 0 ;
327
328 G4double ZZ1 = Zdat[iZ];
329 G4double ZZ2 = Zdat[iZ+1];
330 G4double ratZ = (AtomicNumber-ZZ1)*(AtomicNumber+ZZ1)/
331 ((ZZ2-ZZ1)*(ZZ2+ZZ1));
332
333 static const G4double Tlim = 10.*CLHEP::MeV;
334 static const G4double sigmafactor =
335 CLHEP::twopi*CLHEP::classic_electr_radius*CLHEP::classic_electr_radius;
336 static const G4double beta2lim = Tlim*(Tlim+2.*CLHEP::electron_mass_c2)/
337 ((Tlim+CLHEP::electron_mass_c2)*(Tlim+CLHEP::electron_mass_c2));
338 static const G4double bg2lim = Tlim*(Tlim+2.*CLHEP::electron_mass_c2)/
339 (CLHEP::electron_mass_c2*CLHEP::electron_mass_c2);
340
341 static const G4double sig0[15] = {
342 0.2672*CLHEP::barn, 0.5922*CLHEP::barn, 2.653*CLHEP::barn, 6.235*CLHEP::barn,
343 11.69*CLHEP::barn , 13.24*CLHEP::barn , 16.12*CLHEP::barn, 23.00*CLHEP::barn,
344 35.13*CLHEP::barn , 39.95*CLHEP::barn , 50.85*CLHEP::barn, 67.19*CLHEP::barn,
345 91.15*CLHEP::barn , 104.4*CLHEP::barn , 113.1*CLHEP::barn};
346
347 static const G4double Tdat[22] = {
348 100*CLHEP::eV, 200*CLHEP::eV, 400*CLHEP::eV, 700*CLHEP::eV,
349 1*CLHEP::keV, 2*CLHEP::keV, 4*CLHEP::keV, 7*CLHEP::keV,
350 10*CLHEP::keV, 20*CLHEP::keV, 40*CLHEP::keV, 70*CLHEP::keV,
351 100*CLHEP::keV, 200*CLHEP::keV, 400*CLHEP::keV, 700*CLHEP::keV,
352 1*CLHEP::MeV, 2*CLHEP::MeV, 4*CLHEP::MeV, 7*CLHEP::MeV,
353 10*CLHEP::MeV, 20*CLHEP::MeV};
354
355 if(eKineticEnergy <= Tlim)
356 {
357 // get bin number in T (beta2)
358 G4int iT = 21;
359 // Loop checking, 03-Aug-2015, Vladimir Ivanchenko
360 while ((iT>=0)&&(Tdat[iT]>=eKineticEnergy)) iT -= 1;
361 if(iT==21) iT = 20;
362 if(iT==-1) iT = 0 ;
363
364 // calculate betasquare values
365 G4double T = Tdat[iT], E = T + electron_mass_c2;
366 G4double b2small = T*(E+electron_mass_c2)/(E*E);
367
368 T = Tdat[iT+1]; E = T + electron_mass_c2;
369 G4double b2big = T*(E+electron_mass_c2)/(E*E);
370 G4double ratb2 = (beta2-b2small)/(b2big-b2small);
371
372 if (charge < 0.)
373 {
374 c1 = celectron[iZ][iT];
375 c2 = celectron[iZ+1][iT];
376 cc1 = c1+ratZ*(c2-c1);
377
378 c1 = celectron[iZ][iT+1];
379 c2 = celectron[iZ+1][iT+1];
380 cc2 = c1+ratZ*(c2-c1);
381
382 corr = cc1+ratb2*(cc2-cc1);
383
384 sigma *= sigmafactor/corr;
385 }
386 else
387 {
388 c1 = cpositron[iZ][iT];
389 c2 = cpositron[iZ+1][iT];
390 cc1 = c1+ratZ*(c2-c1);
391
392 c1 = cpositron[iZ][iT+1];
393 c2 = cpositron[iZ+1][iT+1];
394 cc2 = c1+ratZ*(c2-c1);
395
396 corr = cc1+ratb2*(cc2-cc1);
397
398 sigma *= sigmafactor/corr;
399 }
400 }
401 else
402 {
403 c1 = bg2lim*sig0[iZ]*(1.+hecorr[iZ]*(beta2-beta2lim))/bg2;
404 c2 = bg2lim*sig0[iZ+1]*(1.+hecorr[iZ+1]*(beta2-beta2lim))/bg2;
405 if((AtomicNumber >= ZZ1) && (AtomicNumber <= ZZ2))
406 sigma = c1+ratZ*(c2-c1) ;
407 else if(AtomicNumber < ZZ1)
408 sigma = AtomicNumber*AtomicNumber*c1/(ZZ1*ZZ1);
409 else if(AtomicNumber > ZZ2)
410 sigma = AtomicNumber*AtomicNumber*c2/(ZZ2*ZZ2);
411 }
412 // low energy correction based on theory
413 sigma *= 1.+0.30/(1.+sqrt(1000.*eKineticEnergy));
414
415 return sigma;
416

436G4double G4UrbanMscModel::ComputeTruePathLengthLimit(
437 const G4Track& track,
438 G4double& currentMinimalStep)
439{
440 tPathLength = currentMinimalStep;
441 const G4DynamicParticle* dp = track.GetDynamicParticle();
442
443 G4StepPoint* sp = track.GetStep()->GetPreStepPoint();
444 G4StepStatus stepStatus = sp->GetStepStatus();
445 couple = track.GetMaterialCutsCouple();
446 SetCurrentCouple(couple);
447 currentMaterialIndex = couple->GetIndex();
448 currentKinEnergy = dp->GetKineticEnergy();
449 currentRange = GetRange(particle,currentKinEnergy,couple);
450 lambda0 = GetTransportMeanFreePath(particle,currentKinEnergy);
451 tPathLength = min(tPathLength,currentRange);
452 /*
453 G4cout << "G4Urban::StepLimit tPathLength= " << tPathLength
454 << " range= " <<currentRange<< " lambda= "<<lambda0
455 <<G4endl;
456 */
457 // set flag to default values
458 Zeff = couple->GetMaterial()->GetIonisation()->GetZeffective();
459
460 if(Zold != Zeff)
461 UpdateCache();
462
463 // stop here if small step
464 if(tPathLength < tlimitminfix) {
465 latDisplasment = false;
466 return ConvertTrueToGeom(tPathLength, currentMinimalStep);
467 }
468
469 // upper limit for the straight line distance the particle can travel
470 // for electrons and positrons
471 G4double distance = currentRange;
472 // for muons, hadrons
473 if(mass > masslimite) {
474 distance *= (1.15-9.76e-4*Zeff);
475 } else {
476 distance *= (1.20-Zeff*(1.62e-2-9.22e-5*Zeff));
477 }
478 presafety = sp->GetSafety();
479 /*
480 G4cout << "G4Urban::StepLimit tPathLength= "
481 <<tPathLength<<" safety= " << presafety
482 << " range= " <<currentRange<< " lambda= "<<lambda0
483 << " Alg: " << steppingAlgorithm <<G4endl;
484 */
485 // far from geometry boundary
486 if(distance < presafety)
487 {
488 latDisplasment = false;
489 return ConvertTrueToGeom(tPathLength, currentMinimalStep);
490 }
491
492 latDisplasment = latDisplasmentbackup;
493 static const G4double invmev = 1.0/CLHEP::MeV;
494 // standard version
495 //
496 if (steppingAlgorithm == fUseDistanceToBoundary)
497 {
498 //compute geomlimit and presafety
499 geomlimit = ComputeGeomLimit(track, presafety, currentRange);
500 /*
501 G4cout << "G4Urban::Distance to boundary geomlimit= "
502 <<geomlimit<<" safety= " << presafety<<G4endl;
503 */
504
505 // is it far from boundary ?
506 if(distance < presafety)
507 {
508 latDisplasment = false;
509 return ConvertTrueToGeom(tPathLength, currentMinimalStep);
510 }
511
512 smallstep += 1.;
513 insideskin = false;
514
515 // initialisation at firs step and at the boundary
516 if(firstStep || (stepStatus == fGeomBoundary))
517 {
518 rangeinit = currentRange;
519 if(!firstStep) { smallstep = 1.; }
520
521 //define stepmin here (it depends on lambda!)
522 //rough estimation of lambda_elastic/lambda_transport
523 G4double rat = currentKinEnergy*invmev;
524 rat = 1.e-3/(2.e-3+rat*(stepmina+stepminb*rat));
525 //stepmin ~ lambda_elastic
526 stepmin = rat*lambda0;
527 skindepth = skin*stepmin;
528 tlimitmin = max(0.7*sqrt(Zeff)*stepmin,tlimitminfix);
529 /*
530 G4cout << "rangeinit= " << rangeinit << " stepmin= " << stepmin
531 << " tlimitmin= " << tlimitmin << " geomlimit= "
532 << geomlimit <<G4endl;
533 */
534 // constraint from the geometry
535
536 if((geomlimit < geombig) && (geomlimit > geommin))
537 {
538 // geomlimit is a geometrical step length
539 // transform it to true path length (estimation)
540 if((1.-geomlimit/lambda0) > 0.)
541 geomlimit = -lambda0*G4Log(1.-geomlimit/lambda0)+tlimitmin ;
542
543 if(stepStatus == fGeomBoundary)
544 tgeom = geomlimit/facgeom;
545 else
546 tgeom = 2.*geomlimit/facgeom;
547 }
548 else
549 tgeom = geombig;
550 }
551
552 //step limit
553 tlimit = facrange*rangeinit;
554
555 //lower limit for tlimit
556 tlimit = max(tlimit,tlimitmin);
557 tlimit = min(tlimit,tgeom);
558 /*
559 G4cout << "tgeom= " << tgeom << " geomlimit= " << geomlimit
560 << " tlimit= " << tlimit << " presafety= " << presafety << G4endl;
561 */
562 // shortcut
563 if((tPathLength < tlimit) && (tPathLength < presafety) &&
564 (smallstep > skin) && (tPathLength < geomlimit-0.999*skindepth))
565 {
566 return ConvertTrueToGeom(tPathLength, currentMinimalStep);
567 }
568
569 // step reduction near to boundary
570 if(smallstep <= skin)
571 {
572 tlimit = stepmin;
573 insideskin = true;
574 }
575 else if(geomlimit < geombig)
576 {
577 if(geomlimit > skindepth)
578 {
579 tlimit = min(tlimit, geomlimit-0.999*skindepth);
580 }
581 else
582 {
583 insideskin = true;
584 tlimit = min(tlimit, stepmin);
585 }
586 }
587
588 tlimit = max(tlimit, stepmin);
589
590 // randomise if not 'small' step and step determined by msc
591 if((tlimit < tPathLength) && (smallstep > skin) && !insideskin)
592 {
593 tPathLength = min(tPathLength, Randomizetlimit());
594 }
595 else
596 {
597 tPathLength = min(tPathLength, tlimit);
598 }
599
600 }
601 // for 'normal' simulation with or without magnetic field
602 // there no small step/single scattering at boundaries
603 else if(steppingAlgorithm == fUseSafety)
604 {
605 if(stepStatus != fGeomBoundary) {
606 presafety = ComputeSafety(sp->GetPosition(),tPathLength);
607 }
608 /*
609 G4cout << "presafety= " << presafety
610 << " firstStep= " << firstStep
611 << " stepStatus= " << stepStatus
612 << G4endl;
613 */
614 // is far from boundary
615 if(distance < presafety)
616 {
617 latDisplasment = false;
618 return ConvertTrueToGeom(tPathLength, currentMinimalStep);
619 }
620
621 if(firstStep || (stepStatus == fGeomBoundary)) {
622 rangeinit = currentRange;
623 fr = facrange;
624 // 9.1 like stepping for e+/e- only (not for muons,hadrons)
625 if(mass < masslimite)
626 {
627 rangeinit = max(rangeinit, lambda0);
628 if(lambda0 > lambdalimit) {
629 fr *= (0.75+0.25*lambda0/lambdalimit);
630 }
631 }
632 //lower limit for tlimit
633 G4double rat = currentKinEnergy*invmev;
634 rat = 1.e-3/(2.e-3+rat*(stepmina+stepminb*rat));
635 stepmin = lambda0*rat;
636 tlimitmin = max(0.7*sqrt(Zeff)*stepmin,tlimitminfix);
637 }
638
639 //step limit
640 tlimit = max(fr*rangeinit, facsafety*presafety);
641
642 //lower limit for tlimit
643 tlimit = max(tlimit, tlimitmin);
644
645 // randomise if step determined by msc
646 if(tlimit < tPathLength)
647 {
648 tPathLength = min(tPathLength, Randomizetlimit());
649 }
650 else { tPathLength = min(tPathLength, tlimit); }
651 }
652 // new stepping mode UseSafetyPlus
653 else if(steppingAlgorithm == fUseSafetyPlus)
654 {
655 if(stepStatus != fGeomBoundary) {
656 presafety = ComputeSafety(sp->GetPosition(),tPathLength);
657 }
658 /*
659 G4cout << "presafety= " << presafety
660 << " firstStep= " << firstStep
661 << " stepStatus= " << stepStatus
662 << G4endl;
663 */
664 // is far from boundary
665 if(distance < presafety)
666 {
667 latDisplasment = false;
668 return ConvertTrueToGeom(tPathLength, currentMinimalStep);
669 }
670
671 if(firstStep || (stepStatus == fGeomBoundary)) {
672 rangeinit = currentRange;
673 fr = facrange;
674 rangecut = geombig;
675 if(mass < masslimite)
676 {
677 G4int index = 1;
678 if(charge > 0.) index = 2;
679 rangecut = couple->GetProductionCuts()->GetProductionCut(index);
680 if(lambda0 > lambdalimit) {
681 fr *= (0.84+0.16*lambda0/lambdalimit);
682 }
683 }
684 //lower limit for tlimit
685 G4double rat = currentKinEnergy*invmev;
686 rat = 1.e-3/(2.e-3+rat*(stepmina+stepminb*rat));
687 stepmin = lambda0*rat;
688 tlimitmin = max(0.7*sqrt(Zeff)*stepmin,tlimitminfix);
689 }
690 //step limit
691 tlimit = max(fr*rangeinit, facsafety*presafety);
692
693 //lower limit for tlimit
694 tlimit = max(tlimit, tlimitmin);
695
696 // condition for tPathLength from drr and finalr
697 if(currentRange > finalr) {
698 G4double tmax = drr*currentRange+
699 finalr*(1.-drr)*(2.-finalr/currentRange);
700 tPathLength = min(tPathLength,tmax);
701 }
702
703 // condition safety
704 if(currentRange > rangecut) {
705 if(firstStep) {
706 tPathLength = min(tPathLength,facsafety*presafety);
707 } else if(stepStatus != fGeomBoundary && presafety > stepmin) {
708 tPathLength = min(tPathLength,presafety);
709 }
710 }
711
712 // randomise if step determined by msc
713 if(tPathLength < tlimit)
714 {
715 tPathLength = min(tPathLength, Randomizetlimit());
716 }
717 else { tPathLength = min(tPathLength, tlimit); }
718 }
719
720 // version similar to 7.1 (needed for some experiments)
721 else
722 {
723 if (stepStatus == fGeomBoundary)
724 {
725 if (currentRange > lambda0) { tlimit = facrange*currentRange; }
726 else { tlimit = facrange*lambda0; }
727
728 tlimit = max(tlimit, tlimitmin);
729 }
730 // randomise if step determined by msc
731 if(tlimit < tPathLength)
732 {
733 tPathLength = min(tPathLength, Randomizetlimit());
734 }
735 else { tPathLength = min(tPathLength, tlimit); }
736 }
737 firstStep = false;
738 return ConvertTrueToGeom(tPathLength, currentMinimalStep);
739}

743G4double G4UrbanMscModel::ComputeGeomPathLength(G4double)
744{
745 lambdaeff = lambda0;
746 par1 = -1. ;
747 par2 = par3 = 0. ;
748
749 // this correction needed to run MSC with eIoni and eBrem inactivated
750 // and makes no harm for a normal run
751 tPathLength = std::min(tPathLength,currentRange);
752
753 // do the true -> geom transformation
754 zPathLength = tPathLength;
755
756 // z = t for very small tPathLength
757 if(tPathLength < tlimitminfix2) return zPathLength;
758
759 // VI: it is already checked
760 // if(tPathLength > currentRange)
761 // tPathLength = currentRange ;
762 /*
763 G4cout << "ComputeGeomPathLength: tpl= " << tPathLength
764 << " R= " << currentRange << " L0= " << lambda0
765 << " E= " << currentKinEnergy << " "
766 << particle->GetParticleName() << G4endl;
767 */
768 G4double tau = tPathLength/lambda0 ;
769
770 if ((tau <= tausmall) || insideskin) {
771 zPathLength = min(tPathLength, lambda0);
772
773 } else if (tPathLength < currentRange*dtrl) {
774 if(tau < taulim) zPathLength = tPathLength*(1.-0.5*tau) ;
775 else zPathLength = lambda0*(1.-G4Exp(-tau));
776
777 } else if(currentKinEnergy < mass || tPathLength == currentRange) {
778 par1 = 1./currentRange ;
779 par2 = 1./(par1*lambda0) ;
780 par3 = 1.+par2 ;
781 if(tPathLength < currentRange) {
782 zPathLength =
783 (1.-G4Exp(par3*G4Log(1.-tPathLength/currentRange)))/(par1*par3);
784 } else {
785 zPathLength = 1./(par1*par3);
786 }
787
788 } else {
789 G4double rfin = max(currentRange-tPathLength, 0.01*currentRange);
790 G4double T1 = GetEnergy(particle,rfin,couple);
791 G4double lambda1 = GetTransportMeanFreePath(particle,T1);
792
793 par1 = (lambda0-lambda1)/(lambda0*tPathLength);
794 //G4cout << "par1= " << par1 << " L1= " << lambda1 << G4endl;
795 par2 = 1./(par1*lambda0);
796 par3 = 1.+par2 ;
797 zPathLength = (1.-G4Exp(par3*G4Log(lambda1/lambda0)))/(par1*par3);
798 }
799
800 zPathLength = min(zPathLength, lambda0);
801 //G4cout<< "zPathLength= "<< zPathLength<< " L0= " << lambda0 << G4endl;
802 return zPathLength;
803}

851G4ThreeVector&
852G4UrbanMscModel::SampleScattering(const G4ThreeVector& oldDirection,
853 G4double /*safety*/)
854{
855 fDisplacement.set(0.0,0.0,0.0);
856 G4double kineticEnergy = currentKinEnergy;
857 if (tPathLength > currentRange*dtrl) {
858 kineticEnergy = GetEnergy(particle,currentRange-tPathLength,couple);
859 } else {
860 kineticEnergy -= tPathLength*GetDEDX(particle,currentKinEnergy,couple);
861 }
862
863 if((kineticEnergy <= eV) || (tPathLength <= tlimitminfix) ||
864 (tPathLength < tausmall*lambda0)) { return fDisplacement; }
865
866 G4double cth = SampleCosineTheta(tPathLength,kineticEnergy);
867
868 // protection against 'bad' cth values
869 if(std::abs(cth) >= 1.0) { return fDisplacement; }
870
871 /*
872 if(cth < 1.0 - 1000*tPathLength/lambda0 && cth < 0.5 &&
873 kineticEnergy > 20*MeV) {
874 G4cout << "### G4UrbanMscModel::SampleScattering for "
875 << particle->GetParticleName()
876 << " E(MeV)= " << kineticEnergy/MeV
877 << " Step(mm)= " << tPathLength/mm
878 << " in " << CurrentCouple()->GetMaterial()->GetName()
879 << " CosTheta= " << cth << G4endl;
880 }
881 */
882 G4double sth = sqrt((1.0 - cth)*(1.0 + cth));
883 G4double phi = twopi*rndmEngineMod->flat();
884 G4double dirx = sth*cos(phi);
885 G4double diry = sth*sin(phi);
886
887 G4ThreeVector newDirection(dirx,diry,cth);
888 newDirection.rotateUz(oldDirection);
889
890 fParticleChange->ProposeMomentumDirection(newDirection);
891 /*
892 G4cout << "G4UrbanMscModel::SampleSecondaries: e(MeV)= " << kineticEnergy
893 << " sinTheta= " << sth << " safety(mm)= " << safety
894 << " trueStep(mm)= " << tPathLength
895 << " geomStep(mm)= " << zPathLength
896 << G4endl;
897 */
898
899 if (latDisplasment && currentTau >= tausmall) {
900 if(dispAlg96) { SampleDisplacement(sth, phi); }
901 else { SampleDisplacementNew(cth, phi); }
902 fDisplacement.rotateUz(oldDirection);
903 }
904 return fDisplacement;
905}

909G4double G4UrbanMscModel::SampleCosineTheta(G4double trueStepLength,
910 G4double KineticEnergy)
911{
912 G4double cth = 1. ;
913 G4double tau = trueStepLength/lambda0;
914 currentTau = tau;
915 lambdaeff = lambda0;
916
917 G4double lambda1 = GetTransportMeanFreePath(particle,KineticEnergy);
918 if(std::abs(lambda1 - lambda0) > lambda0*0.01 && lambda1 > 0.)
919 {
920 // mean tau value
921 tau = trueStepLength*G4Log(lambda0/lambda1)/(lambda0-lambda1);
922 }
923
924 currentTau = tau ;
925 lambdaeff = trueStepLength/currentTau;
926 currentRadLength = couple->GetMaterial()->GetRadlen();
927
928 if (tau >= taubig) { cth = -1.+2.*rndmEngineMod->flat(); }
929 else if (tau >= tausmall) {
930 static const G4double numlim = 0.01;
931 static const G4double onethird = 1./3.;
932 G4double xmeanth, x2meanth;
933 if(tau < numlim) {
934 xmeanth = 1.0 - tau*(1.0 - 0.5*tau);
935 x2meanth= 1.0 - tau*(5.0 - 6.25*tau)*onethird;
936 } else {
937 xmeanth = G4Exp(-tau);
938 x2meanth = (1.+2.*G4Exp(-2.5*tau))*onethird;
939 }
940
941 // too large step of low-energy particle
942 G4double relloss = 1. - KineticEnergy/currentKinEnergy;
943 static const G4double rellossmax= 0.50;
944 if(relloss > rellossmax) {
945 return SimpleScattering(xmeanth,x2meanth);
946 }
947 // is step extreme small ?
948 G4bool extremesmallstep = false ;
949 G4double tsmall = std::min(tlimitmin,lambdalimit);
950 G4double theta0 = 0.;
951 if(trueStepLength > tsmall) {
952 theta0 = ComputeTheta0(trueStepLength,KineticEnergy);
953 } else {
954 theta0 = sqrt(trueStepLength/tsmall)*ComputeTheta0(tsmall,KineticEnergy);
955 extremesmallstep = true ;
956 }
957
958 static const G4double onesixth = 1./6.;
959 static const G4double theta0max = CLHEP::pi*onesixth;
960 //G4cout << "Theta0= " << theta0 << " theta0max= " << theta0max
961 // << " sqrt(tausmall)= " << sqrt(tausmall) << G4endl;
962
963 // protection for very small angles
964 G4double theta2 = theta0*theta0;
965
966 if(theta2 < tausmall) { return cth; }
967
968 if(theta0 > theta0max) {
969 return SimpleScattering(xmeanth,x2meanth);
970 }
971
972 G4double x = theta2*(1.0 - theta2/12.);
973 if(theta2 > numlim) {
974 G4double sth = 2*sin(0.5*theta0);
975 x = sth*sth;
976 }
977
978 // parameter for tail
979 G4double ltau= G4Log(tau);
980 G4double u = extremesmallstep
981 ? G4Exp(G4Log(tsmall/lambda0)*onesixth)
982 : G4Exp(ltau*onesixth);
983 G4double xx = G4Log(lambdaeff/currentRadLength);
984 G4double xsi = coeffc1+u*(coeffc2+coeffc3*u)+coeffc4*xx;
985
986 // tail should not be too big
987 xsi = std::max(xsi, 1.9);
988 /*
989 if(KineticEnergy > 20*MeV && xsi < 1.6) {
990 G4cout << "G4UrbanMscModel::SampleCosineTheta: E(GeV)= "
991 << KineticEnergy/GeV
992 << " !!** c= " << xsi
993 << " **!! length(mm)= " << trueStepLength << " Zeff= " << Zeff
994 << " " << couple->GetMaterial()->GetName()
995 << " tau= " << tau << G4endl;
996 }
997 */
998
999 G4double c = xsi;
1000
1001 if(std::abs(c-3.) < 0.001) { c = 3.001; }
1002 else if(std::abs(c-2.) < 0.001) { c = 2.001; }
1003
1004 G4double c1 = c-1.;
1005
1006 G4double ea = G4Exp(-xsi);
1007 G4double eaa = 1.-ea ;
1008 G4double xmean1 = 1.-(1.-(1.+xsi)*ea)*x/eaa;
1009 G4double x0 = 1. - xsi*x;
1010
1011 // G4cout << " xmean1= " << xmean1 << " xmeanth= " << xmeanth << G4endl;
1012
1013 if(xmean1 <= 0.999*xmeanth) {
1014 return SimpleScattering(xmeanth,x2meanth);
1015 }
1016 //from continuity of derivatives
1017 G4double b = 1.+(c-xsi)*x;
1018
1019 G4double b1 = b+1.;
1020 G4double bx = c*x;
1021
1022 G4double eb1 = G4Exp(G4Log(b1)*c1);
1023 G4double ebx = G4Exp(G4Log(bx)*c1);
1024 G4double d = ebx/eb1;
1025
1026 G4double xmean2 = (x0 + d - (bx - b1*d)/(c-2.))/(1. - d);
1027
1028 G4double f1x0 = ea/eaa;
1029 G4double f2x0 = c1/(c*(1. - d));
1030 G4double prob = f2x0/(f1x0+f2x0);
1031
1032 G4double qprob = xmeanth/(prob*xmean1+(1.-prob)*xmean2);
1033
1034 // sampling of costheta
1035 //G4cout << "c= " << c << " qprob= " << qprob << " eb1= " << eb1
1036 // << " c1= " << c1 << " b1= " << b1 << " bx= " << bx << " eb1= " << eb1
1037 // << G4endl;
1038 if(rndmEngineMod->flat() < qprob)
1039 {
1040 G4double var = 0;
1041 if(rndmEngineMod->flat() < prob) {
1042 cth = 1.+G4Log(ea+rndmEngineMod->flat()*eaa)*x;
1043 } else {
1044 var = (1.0 - d)*rndmEngineMod->flat();
1045 if(var < numlim*d) {
1046 var /= (d*c1);
1047 cth = -1.0 + var*(1.0 - 0.5*var*c)*(2. + (c - xsi)*x);
1048 } else {
1049 cth = 1. + x*(c - xsi - c*G4Exp(-G4Log(var + d)/c1));
1050 }
1051 }
1052 /*
1053 if(KineticEnergy > 5*GeV && cth < 0.9) {
1054 G4cout << "G4UrbanMscModel::SampleCosineTheta: E(GeV)= "
1055 << KineticEnergy/GeV
1056 << " 1-cosT= " << 1 - cth
1057 << " length(mm)= " << trueStepLength << " Zeff= " << Zeff
1058 << " tau= " << tau
1059 << " prob= " << prob << " var= " << var << G4endl;
1060 G4cout << " c= " << c << " qprob= " << qprob << " eb1= " << eb1
1061 << " ebx= " << ebx
1062 << " c1= " << c1 << " b= " << b << " b1= " << b1
1063 << " bx= " << bx << " d= " << d
1064 << " ea= " << ea << " eaa= " << eaa << G4endl;
1065 }
1066 */
1067 }
1068 else {
1069 cth = -1.+2.*rndmEngineMod->flat();
1070 /*
1071 if(KineticEnergy > 5*GeV) {
1072 G4cout << "G4UrbanMscModel::SampleCosineTheta: E(GeV)= "
1073 << KineticEnergy/GeV
1074 << " length(mm)= " << trueStepLength << " Zeff= " << Zeff
1075 << " qprob= " << qprob << G4endl;
1076 }
1077 */
1078 }
1079 }
1080 return cth ;
1081}

1084
1085G4double G4UrbanMscModel::ComputeTheta0(G4double trueStepLength,
1086 G4double KineticEnergy)
1087{
1088 // for all particles take the width of the central part
1089 // from a parametrization similar to the Highland formula
1090 // ( Highland formula: Particle Physics Booklet, July 2002, eq. 26.10)
1091 G4double invbetacp = std::sqrt((currentKinEnergy+mass)*(KineticEnergy+mass)/
1092 (currentKinEnergy*(currentKinEnergy+2.*mass)*
1093 KineticEnergy*(KineticEnergy+2.*mass)));
1094 G4double y = trueStepLength/currentRadLength;
1095
1096 if(particle == positron)
1097 {
1098 static const G4double xl= 0.6;
1099 static const G4double xh= 0.9;
1100 static const G4double e = 113.0;
1101 G4double corr;
1102
1103 G4double tau = std::sqrt(currentKinEnergy*KineticEnergy)/mass;
1104 G4double x = std::sqrt(tau*(tau+2.)/((tau+1.)*(tau+1.)));
1105 G4double a = 0.994-4.08e-3*Zeff;
1106 G4double b = 7.16+(52.6+365./Zeff)/Zeff;
1107 G4double c = 1.000-4.47e-3*Zeff;
1108 G4double d = 1.21e-3*Zeff;
1109 if(x < xl) {
1110 corr = a*(1.-G4Exp(-b*x));
1111 } else if(x > xh) {
1112 corr = c+d*G4Exp(e*(x-1.));
1113 } else {
1114 G4double yl = a*(1.-G4Exp(-b*xl));
1115 G4double yh = c+d*G4Exp(e*(xh-1.));
1116 G4double y0 = (yh-yl)/(xh-xl);
1117 G4double y1 = yl-y0*xl;
1118 corr = y0*x+y1;
1119 }
1120 //==================================================================
1121 y *= corr*(1.+Zeff*(1.84035e-4*Zeff-1.86427e-2)+0.41125);
1122 }
1123
1124 static const G4double c_highland = 13.6*CLHEP::MeV;
1125 G4double theta0 = c_highland*std::abs(charge)*std::sqrt(y)*invbetacp;
1126
1127 // correction factor from e- scattering data
1128 theta0 *= (coeffth1+coeffth2*G4Log(y));
1129 return theta0;
1130}

1135void G4UrbanMscModel::SampleDisplacement(G4double , G4double phi)
1136{
1137 //simple distribution for u=r/rmax
1138 // based on single scattering results
1139 // ~(u/u0)**p1 for u < u0
1140 // ~((1-u)/(1-u0))**p2 for u >= u0
1141
1142 G4double rmax = sqrt((tPathLength-zPathLength)*(tPathLength+zPathLength));
1143 G4double r = 0.;
1144 if(rmax > 0.)
1145 {
1146 static const G4double su0 = 0.851549;
1147 static const G4double sp1 = 3.02549;
1148 static const G4double sp2 = 1.84108;
1149 static const G4double su1 = 1-su0;
1150 static const G4double sp11 = sp1+1;
1151 static const G4double sp21 = sp2+1;
1152 static const G4double sweight = 0.802110;
1153 G4double u;
1154
1155 if(rndmEngineMod->flat() < sweight) {
1156 u = su0*G4Exp(G4Log(rndmEngineMod->flat())/sp11);
1157 } else {
1158 u = 1-su1*G4Exp(G4Log(1-rndmEngineMod->flat())/sp21);
1159 }
1160 r = rmax*u ;
1161 }
1162
1163 //simple distribution for v=Phi-phi=psi ~exp(-beta*v)
1164 // alpha determined from the requirement that distribution should give
1165 // the same mean value than that obtained from the ss simulation
1166 if(r > 0.)
1167 {
1168 static const G4double cbeta = 1.933 ;
1169 static const G4double cbeta1 = 1.-exp(-cbeta*CLHEP::pi);
1170 G4double psi = -G4Log(1.-rndmEngineMod->flat()*cbeta1)/cbeta;
1171 G4double Phi = (rndmEngineMod->flat() < 0.5) ? phi+psi : phi-psi;
1172 fDisplacement.set(r*std::cos(Phi),r*std::sin(Phi),0.0);
1173 }
1174}

1178void G4UrbanMscModel::SampleDisplacementNew(G4double , G4double phi)
1179{
1180 //sample displacement r
1181
1182 G4double rmax = sqrt((tPathLength-zPathLength)*(tPathLength+zPathLength));
1183 G4double r = 0.;
1184 G4double u = r/rmax;
1185 if(rmax > 0.)
1186 {
1187 G4double rej;
1188 G4int count = 0;
1189
1190 static const G4double reps = 1.e-6;
1191 static const G4double rp1 = 1.61385e+1;
1192 static const G4double rp2 = 3.26646e+0;
1193 static const G4double rp3 =-3.35702e+0;
1194 static const G4double rp4 = 7.38037e+1;
1195 static const G4double rp5 =-1.12829e+2;
1196 static const G4double rp6 = 4.63974e+1;
1197 static const G4double ymax = 2.88900e+1;
1198
1199 do {
1200 u = reps+(1.-2.*reps)*rndmEngineMod->flat();
1201 G4double v = 1.-u ;
1202 G4double v2= v*v;
1203 G4double v4= v2*v2;
1204 G4double v6= v4*v2;
1205 G4double v8= v6*v2;
1206 rej = G4Exp(rp1*u*u+rp2*G4Log(v))*u*v*(1+rp3*v2+rp4*v4+rp5*v6+rp6*v8);
1207 }
1208 // Loop checking, 15-Sept-2015, Vladimir Ivanchenko
1209 while (ymax*rndmEngineMod->flat() > rej && ++count < 1000);
1210 r = rmax*u;
1211 }
1212
1213 if(r > 0.)
1214 {
1215 // sample Phi using lateral correlation
1216 // and r/rmax - (Phi-phi) correlation
1217 // v = Phi-phi = acos(latcorr/(r*sth))
1218 // from SS simulation f(v) = a0*exp(-a1*v)+a2
1219 G4double v, rej;
1220 G4int count(0);
1221
1222 static const G4double a1phi[10] = {4.508e-1,6.132e-1,1.180e+0,1.357e+0,1.582e+0,
1223 1.863e+0,2.217e+0,2.739e+0,3.652e+0,5.149e+0};
1224 static const G4double a2phi[10] = {1.556e+0,3.571e-1,6.480e-2,3.964e-2,2.733e-2,
1225 1.571e-2,8.546e-3,3.308e-3,6.464e-4,4.194e-5};
1226 static const G4double a3phi[10] = {3.631e-2,1.300e-1,8.899e-1,8.396e-1,7.362e-1,
1227 6.782e-1,5.613e-1,4.568e-1,4.296e-1,4.067e-1};
1228 static const G4double gmphi[10] = {3.8455,1.5860,2.0190,1.4924,1.1711,
1229 1.0158,1.0086,1.0034,1.0007,1.0001};
1230
1231 G4int iphi = u*10.;
1232 if(iphi < 0) { iphi = 0; }
1233 else if(iphi > 9) { iphi = 9; }
1234 G4double a1 = a1phi[iphi];
1235 G4double a2 = a2phi[iphi];
1236 G4double a3 = a3phi[iphi];
1237 G4double rejmax = gmphi[iphi];
1238 G4double wphi = 1-G4Exp(-0.5*a1*CLHEP::pi);
1239
1240 do {
1241 v = -2*G4Log(1-wphi*rndmEngineMod->flat())/a1;
1242 G4double exav = G4Exp(-0.5*a1*v);
1243 rej = (1+G4Exp(a3*G4Log(v)))*(exav*exav+a2)/(exav*rejmax);
1244 }
1245 // Loop checking, 5-March-2018, Vladimir Ivanchenko
1246 while (rndmEngineMod->flat() > rej && ++count < 1000);
1247
1248 G4double Phi = (rndmEngineMod->flat() < 0.5) ? phi+v : phi-v;
1249
1250 fDisplacement.set(r*std::cos(Phi),r*std::sin(Phi),0.0);
1251 }
1252}
