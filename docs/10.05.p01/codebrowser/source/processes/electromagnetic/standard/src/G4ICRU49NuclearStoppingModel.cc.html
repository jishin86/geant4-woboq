120G4double
121G4ICRU49NuclearStoppingModel::ComputeDEDXPerVolume(
122 const G4Material* mat,
123 const G4ParticleDefinition* p,
124 G4double kinEnergy,
125 G4double)
126{
127 G4double nloss = 0.0;
128 if(kinEnergy <= 0.0) { return nloss; }
129
130 // projectile
131 G4double mass1 = p->GetPDGMass();
132 G4double z1 = std::abs(p->GetPDGCharge()/eplus);
133
134 if(kinEnergy*proton_mass_c2/mass1 > z1*z1*MeV) { return nloss; }
135
136 // Projectile nucleus
137 mass1 /= amu_c2;
138
139 // loop for the elements in the material
140 G4int numberOfElements = mat->GetNumberOfElements();
141 const G4ElementVector* theElementVector = mat->GetElementVector();
142 const G4double* atomDensity = mat->GetAtomicNumDensityVector();
143
144 for (G4int iel=0; iel<numberOfElements; ++iel) {
145 const G4Element* element = (*theElementVector)[iel] ;
146 G4double z2 = element->GetZ();
147 G4double mass2 = element->GetN();
148 nloss += (NuclearStoppingPower(kinEnergy, z1, z2, mass1, mass2))
149 * atomDensity[iel];
150 }
151 nloss *= theZieglerFactor;
152 //G4cout << " nloss= " << nloss << G4endl;
153 return nloss;
154}

166 G4int iz1 = std::min(99, G4lrint(z1));

169 G4double rm;
170 if(z1 > 1.5) {
171 rm = (mass1 + mass2)*(Z23[iz1] + Z23[iz2]);
172 } else {
173 rm = (mass1 + mass2)*g4calc->Z13(G4lrint(z2));
174 }
175 G4double er = 32.536 * mass2 * energy / ( z12 * rm ) ; // reduced energy
176 /*

300 else {
301 // the table is inverse in energy
302 for (G4int i=102; i>=0; --i) {
303 G4double edi = nuca[i][0];
304 if (er <= edi) {
305 G4double edi1 = nuca[i+1][0];
306 G4double ai = nuca[i][1];

308 nloss = (ai - ai1)*(er - edi1)/(edi - edi1) + ai1;

310 }

314 // Stragling
315 if(lossFlucFlag) {
316 G4double sig = 4.0 * mass1 * mass2 / ((mass1 + mass2)*(mass1 + mass2)*
317 (4.0 + 0.197/(er*er) + 6.584/er));
318
319 nloss *= G4RandGauss::shoot(1.0,sig);
320 }
321
322 nloss *= 8.462 * z12 * mass1 / rm; // Return to [ev/(10^15 atoms/cm^2]
323
324 nloss = std::max(nloss, 0.0);
