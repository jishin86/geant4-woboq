45G4MicroElecElastic::G4MicroElecElastic(const G4String& processName,
46 G4ProcessType type):G4VEmProcess (processName, type),
47 isInitialised(false)
48{
49 SetProcessSubType(51);
50}

58
59G4bool G4MicroElecElastic::IsApplicable(const G4ParticleDefinition& p)
60{
61 return (&p == G4Electron::Electron());
62}
63
64//....oooOO0OOooo........oooOO0OOooo........oooOO0OOooo........oooOO0OOooo....
65
66void G4MicroElecElastic::InitialiseProcess(const G4ParticleDefinition*)
67{
68 if(!isInitialised)
69 {
70 isInitialised = true;
71 SetBuildTableFlag(false);
72 if(!EmModel()) SetEmModel(new G4DummyModel());
73 AddEmModel(2, EmModel());
74 }
75}
