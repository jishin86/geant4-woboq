44#ifndef G4FluoTransition_h
45#define G4FluoTransition_h 1
46
47#include "G4DataVector.hh"
48#include "globals.hh"
49#include <vector>
50
51class G4FluoTransition {
52
53public:
54
55 G4FluoTransition(G4int,const std::vector<G4int>&,const G4DataVector&,
56 const G4DataVector&);
57
58 ~G4FluoTransition();
59
60 // All the data stored and provided by this class are relative to a
61 // given vacancy, whose identity is provided by the FinalShellId() method,
62 // in an atom of a given material
63
64 // Returns the identities of the originating shells for the transitions
65 const std::vector<G4int>& OriginatingShellIds() const;
66
67 // Return the energies of the transitions
68 const G4DataVector& TransitionEnergies() const;
69
70 // Return the probabilities of the transitions
71 const G4DataVector& TransitionProbabilities() const;
72
73 // Return the identity if the vacancy
74 G4int FinalShellId() const;
75
76 // Given the index of the originating shells returns its identity
77 G4int OriginatingShellId(G4int index) const;
78
79 // Given the index of the originating shells returns the energy
80 // of the transition starting from it
81 G4double TransitionEnergy(G4int index) const;
82
83 // Given the index of the originating shells returns the probability
84 // of the transition starting from it
85 G4double TransitionProbability(G4int index) const;
86
87private:
88
89 G4int finalShellId;
90 std::vector<G4int> originatingShellIds;
91 G4DataVector transitionEnergies;
92 G4DataVector transitionProbabilities;
93
94};
95
96#endif
