48#ifndef G4SHELLEMDATASET_HH
49#define G4SHELLEMDATASET_HH 1
50
51#include <vector>
52#include <CLHEP/Units/SystemOfUnits.h>
53
54#include "globals.hh"
55#include "G4VEMDataSet.hh"
56
57class G4VDataSetAlgorithm;
58
59class G4ShellEMDataSet : public G4VEMDataSet
60{
61public:
62 G4ShellEMDataSet(G4int Z,
63 G4VDataSetAlgorithm* algo,
64 G4double eUnit=CLHEP::MeV,
65 G4double dataUnit=CLHEP::barn);
66
67 virtual ~G4ShellEMDataSet();
68
69 virtual G4double FindValue(G4double energy, G4int componentId=0) const;
70
71 virtual void PrintData(void) const;
72
73 virtual const G4VEMDataSet* GetComponent(G4int componentId) const { return components[componentId]; }
74 virtual void AddComponent(G4VEMDataSet* dataSet) { components.push_back(dataSet); }
75 virtual size_t NumberOfComponents(void) const { return components.size(); }
76
77 virtual const G4DataVector& GetEnergies(G4int componentId) const { return GetComponent(componentId)->GetEnergies(0); }
78 virtual const G4DataVector& GetData(G4int componentId) const { return GetComponent(componentId)->GetData(0); }
79 virtual const G4DataVector& GetLogEnergies(G4int componentId) const { return GetComponent(componentId)->GetLogEnergies(0); }
80 virtual const G4DataVector& GetLogData(G4int componentId) const { return GetComponent(componentId)->GetLogData(0); }
81
82 virtual void SetEnergiesData(G4DataVector* energies, G4DataVector* data, G4int componentId);
83 virtual void SetLogEnergiesData(G4DataVector* energies,
84 G4DataVector* data,
85 G4DataVector* log_energies,
86 G4DataVector* log_data,
87 G4int componentId);
88
89 virtual G4bool LoadData(const G4String& fileName);
90 virtual G4bool LoadNonLogData(const G4String& fileName);
91
92 virtual G4bool SaveData(const G4String& fileName) const;
93
94 virtual G4double RandomSelect(G4int /*componentId = 0*/) const { return -1.; };
95
96protected:
97
98 G4double GetUnitEnergies() const { return unitEnergies; }
99 G4double GetUnitData() const { return unitData; }
100 const G4VDataSetAlgorithm* GetAlgorithm() const { return algorithm; }
101
102 void CleanUpComponents(void);
103
104private:
105
106 G4String FullFileName(const G4String& fileName) const;
107
108 // Hide copy constructor and assignment operator
109 G4ShellEMDataSet();
110 G4ShellEMDataSet(const G4ShellEMDataSet& copy);
111 G4ShellEMDataSet& operator=(const G4ShellEMDataSet& right);
112
113 std::vector<G4VEMDataSet*> components; // Owned pointers
114
115 G4int z;
116
117 G4VDataSetAlgorithm* algorithm; // Owned pointer
118
119 G4double unitEnergies;
120 G4double unitData;
121};
122#endif /* G4SHELLEMDATASET_HH */
