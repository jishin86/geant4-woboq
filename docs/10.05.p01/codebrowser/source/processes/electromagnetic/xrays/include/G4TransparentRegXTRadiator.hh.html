31// Process describing a radiator of X-ray transition radiation.
32// Thicknesses of plates and gas gaps are fixed.
33// We suppose that:
34// formation zone ~ mean thickness << absorption length
35// for each material and in the range 1-100 keV. This allows us to simplify
36// interference effects in radiator stack (GetStackFactor method).
37//
38//
39// History:
40//
41// 05.04.05 V. Grichine, first version
42//
43
44

46#define G4TransparentRegXTRadiator_h 1
47
48#include "G4VXTRenergyLoss.hh"
49
50class G4TransparentRegXTRadiator : public G4VXTRenergyLoss
51{
52public:
53
54 explicit G4TransparentRegXTRadiator (G4LogicalVolume *anEnvelope,
55 G4Material*,G4Material*, G4double,G4double,G4int,
56 const G4String & processName = "TransparentRegXTRadiator");
57 ~G4TransparentRegXTRadiator ();
58
59 // reimplementation of base class function in analytical way
60
61 G4double SpectralXTRdEdx(G4double energy) override;
62
63 // Pure virtual function from base class
64
65 G4double GetStackFactor( G4double energy, G4double gamma, G4double varAngle) override;
66};
67
68#endif
