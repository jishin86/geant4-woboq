34// ---------------- G4ProcessManager -----------------
35// Class Description
36// It collects all physics a particle can undertake as seven vectors.
37// These vectors are
38// one vector for all processes (called as "process List")
39// two vectors for processes with AtRestGetPhysicalInteractionLength
40// and AtRestDoIt
41// two vectors for processes with AlongStepGetPhysicalInteractionLength
42// and AlongStepDoIt
43// two vectors for processes with PostStepGetPhysicalInteractionLength
44// and PostStepDoIt
45// The tracking will message three types of GetPhysicalInteractionLength
46// in order to limit the Step and select the occurrence of processes.
47// It will message the corresponding DoIt() to apply the selected
48// processes. In addition, the Tracking will limit the Step
49// and select the occurrence of the processes according to
50// the shortest physical interaction length computed (except for
51// processes at rest, for which the Tracking will select the
52// occurrence of the process which returns the shortest mean
53// life-time from the GetPhysicalInteractionLength()).
54//
55// History:
56// revised by G.Cosmo, 06 May 1996
57// Added vector of processes at rest, 06 May 1996
58// ------------------------------------------------------------
59// New Physics scheme 8 Jan. 1997 H.Kurahige
60// Add SetProcessOrdering methods 27 Mar 1998 H.Kurahige

63// Use STL vector instead of RW vector 1. Mar 00 H.Kurashige

65// Add
66// ------------------------------------------------------------

82// Indexes for ProcessVector
83enum G4ProcessVectorTypeIndex
84{
85 typeGPIL = 0, // for GetPhysicalInteractionLength
86 typeDoIt =1 // for DoIt
87};
88enum G4ProcessVectorDoItIndex
89{
90 idxAll = -1, // for all DoIt/GPIL
91 idxAtRest = 0, // for AtRestDoIt/GPIL
92 idxAlongStep = 1, // for AlongStepDoIt/GPIL
93 idxPostStep =2, // for AlongSTepDoIt/GPIL
94 NDoit =3
95};

97// enumeration for Ordering Parameter
98enum G4ProcessVectorOrdering
99{
100 ordInActive = -1, // ordering parameter to indicate InActive DoIt
101 ordDefault = 1000, // default ordering parameter
102 ordLast = 9999 // ordering parameter to indicate the last DoIt
103};

191 // Add a process to the process List
192 // return values are index to the List. Negative return value
193 // indicates that the process has not be added due to some errors
194 // The first argument is a pointer to process.
195 // Following arguments are ordering parameters of the process in
196 // process vectors. If value is negative, the process is
197 // not added to the corresponding process vector.
198
199 // following methods are provided for simple processes
200 // AtRestProcess has only AtRestDoIt
201 // ContinuousProcess has only AlongStepDoIt
202 // DiscreteProcess has only PostStepDoIt
203 // If the ording parameter is not specified, the process is
204 // added at the end of List of process vectors
205 // If a process with same ordering parameter exists,
206 // this new process will be added just after processes
207 // with same ordering parameter
208 // (except for processes assigned to LAST explicitly )
209 // for both DoIt and GetPhysicalInteractionLength
210 //
