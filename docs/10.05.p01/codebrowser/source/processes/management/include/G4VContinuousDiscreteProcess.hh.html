46#ifndef G4VContinuousDiscreteProcess_h
47#define G4VContinuousDiscreteProcess_h 1
48
49#include "globals.hh"
50#include "G4ios.hh"
51
52#include "G4VProcess.hh"
53
54class G4VContinuousDiscreteProcess : public G4VProcess
55{
56 // Abstract class which defines the public behavior of
57 // discrete physics interactions.
58 public:
59
60 G4VContinuousDiscreteProcess(const G4String& ,
61 G4ProcessType aType = fNotDefined );
62 G4VContinuousDiscreteProcess(G4VContinuousDiscreteProcess &);
63
64 virtual ~G4VContinuousDiscreteProcess();
65
66 public :// with description
67 virtual G4double PostStepGetPhysicalInteractionLength(
68 const G4Track& track,
69 G4double previousStepSize,
70 G4ForceCondition* condition
71 );
72
73 virtual G4VParticleChange* PostStepDoIt(
74 const G4Track& ,
75 const G4Step&
76 );
77
78 virtual G4double AlongStepGetPhysicalInteractionLength(
79 const G4Track&,
80 G4double previousStepSize,
81 G4double currentMinimumStep,
82 G4double& currentSafety,
83 G4GPILSelection* selection
84 );
85
86 virtual G4VParticleChange* AlongStepDoIt(
87 const G4Track& ,
88 const G4Step&
89 );
90
91 // no operation in AtRestDoIt
92 virtual G4double AtRestGetPhysicalInteractionLength(
93 const G4Track& ,
94 G4ForceCondition*
95 ) { return -1.0; };
96
97 // no operation in AtRestDoIt
98 virtual G4VParticleChange* AtRestDoIt(
99 const G4Track& ,
100 const G4Step&
101 ) {return 0;};
102
103 protected:// with description
104 virtual G4double GetMeanFreePath(const G4Track& aTrack,
105 G4double previousStepSize,
106 G4ForceCondition* condition
107 )=0;
108 // Calculates from the macroscopic cross section a mean
109 // free path, the value is returned in units of distance.
110
111 protected:// with description
112 virtual G4double GetContinuousStepLimit(const G4Track& aTrack,
113 G4double previousStepSize,
114 G4double currentMinimumStep,
115 G4double& currentSafety
116 )=0;
117 private:
118 // this is the returnd value of G4GPILSelection in
119 // the arguments of AlongStepGPIL()
120 G4GPILSelection valueGPILSelection;
121
122 protected:// with description
123 // these two methods are set/get methods for valueGPILSelection
124 void SetGPILSelection(G4GPILSelection selection)
125 { valueGPILSelection = selection;};
126
127 G4GPILSelection GetGPILSelection() const{return valueGPILSelection;};
128
129 private:
130 // hide default constructor and assignment operator as private
131 G4VContinuousDiscreteProcess();
132 G4VContinuousDiscreteProcess & operator=(const G4VContinuousDiscreteProcess &right);
133
134};
135
136#endif
