67G4Transportation::G4Transportation( G4int verbosity )
68 : G4VProcess( G4String("Transportation"), fTransportation ),
69 // fTransportEndPosition( 0.0, 0.0, 0.0 ),
70 // fTransportEndMomentumDir( 0.0, 0.0, 0.0 ),
71 // fTransportEndKineticEnergy( 0.0 ),
72 // fTransportEndSpin( 0.0, 0.0, 0.0 ),
73 // fMomentumChanged(true),
74 // fEndGlobalTimeComputed(false),
75 // fCandidateEndGlobalTime(0.0),
76 // fParticleIsLooping( false ),
77 // fNewTrack( true ),
78 // fFirstStepInVolume( true ),
79 // fLastStepInVolume( false ),
80 // fGeometryLimitedStep(true),
81 fFieldExertedForce( false ),
82 fPreviousSftOrigin( 0.,0.,0. ),
83 fPreviousSafety( 0.0 ),
84 fEndPointDistance( -1.0 ),
85 fShortStepOptimisation( false ) // Old default: true (=fast short steps)
86{
87 SetProcessSubType(static_cast<G4int>(TRANSPORTATION));
88 pParticleChange= &fParticleChange; // Required to conform to G4VProcess
89 SetVerboseLevel(verbosity);
90
91 G4TransportationManager* transportMgr ;
92
93 transportMgr = G4TransportationManager::GetTransportationManager() ;
94
95 fLinearNavigator = transportMgr->GetNavigatorForTracking() ;
96
97 fFieldPropagator = transportMgr->GetPropagatorInField() ;
98
99 fpSafetyHelper = transportMgr->GetSafetyHelper(); // New
100
101 fpLogger = new G4TransportationLogger("G4Transportation", verbosity);
102
103 SetHighLooperThresholds();
104 // Use the old defaults: Warning = 100 MeV, Important = 250 MeV, No Trials = 10;
105
106 PushThresholdsToLogger();
107 // Should be done by Set methods in SetHighLooperThresholds -- making sure
108
109 // Cannot determine whether a field exists here, as it would
110 // depend on the relative order of creating the detector's
111 // field and this process. That order is not guaranted.
112 // Instead later the method DoesGlobalFieldExist() is called
113
114 static G4ThreadLocal G4TouchableHandle* pNullTouchableHandle = 0;
115 if ( !pNullTouchableHandle)
116 {
117 pNullTouchableHandle = new G4TouchableHandle;
118 }
119 fCurrentTouchableHandle = *pNullTouchableHandle;
120 // Points to (G4VTouchable*) 0
121
122#ifdef G4VERBOSE
123 if( verboseLevel > 0)
124 {
125 G4cout << " G4Transportation constructor> set fShortStepOptimisation to ";
126 if ( fShortStepOptimisation ) { G4cout << "true" << G4endl; }
127 else { G4cout << "false" << G4endl; }
128 }
129#endif
130}

182//
183// Responsibilities:
184// Find whether the geometry limits the Step, and to what length
185// Calculate the new value of the safety and return it.
186// Store the final time, position and momentum.
187
188G4double G4Transportation::
189AlongStepGetPhysicalInteractionLength( const G4Track& track,
190 G4double, // previousStepSize
191 G4double currentMinimumStep,
192 G4double& currentSafety,
193 G4GPILSelection* selection )
194{
195 G4double geometryStepLength= -1.0, newSafety= -1.0;
196 fParticleIsLooping = false ;
197
198 // Initial actions moved to StartTrack()
199 // --------------------------------------
200 // Note: in case another process changes touchable handle
201 // it will be necessary to add here (for all steps)
202 // fCurrentTouchableHandle = aTrack->GetTouchableHandle();
203
204 // GPILSelection is set to defaule value of CandidateForSelection
205 // It is a return value
206 //
207 *selection = CandidateForSelection ;
208
209 fFirstStepInVolume= fNewTrack || fLastStepInVolume;
210 fLastStepInVolume= false;
211 fNewTrack = false;
212
213 fParticleChange.ProposeFirstStepInVolume(fFirstStepInVolume);
214
215 // Get initial Energy/Momentum of the track
216 //
217 const G4DynamicParticle* pParticle = track.GetDynamicParticle() ;
218 const G4ParticleDefinition* pParticleDef = pParticle->GetDefinition() ;
219 G4ThreeVector startMomentumDir = pParticle->GetMomentumDirection() ;
220 G4ThreeVector startPosition = track.GetPosition() ;
221
222 // The Step Point safety can be limited by other geometries and/or the
223 // assumptions of any process - it's not always the geometrical safety.
224 // We calculate the starting point's isotropic safety here.
225 //
226 G4ThreeVector OriginShift = startPosition - fPreviousSftOrigin ;
227 G4double MagSqShift = OriginShift.mag2() ;
228 if( MagSqShift >= sqr(fPreviousSafety) )
229 {
230 currentSafety = 0.0 ;
231 }
232 else
233 {
234 currentSafety = fPreviousSafety - std::sqrt(MagSqShift) ;
235 }
236
237 // Is the particle charged or has it a magnetic moment?
238 //
239 G4double particleCharge = pParticle->GetCharge() ;
240 G4double magneticMoment = pParticle->GetMagneticMoment() ;
241 G4double restMass = pParticle->GetMass() ;
242
243 fGeometryLimitedStep = false ;
244
245 // There is no need to locate the current volume. It is Done elsewhere:
246 // On track construction
247 // By the tracking, after all AlongStepDoIts, in "Relocation"
248
249 // Check if the particle has a force, EM or gravitational, exerted on it
250 //
251 G4FieldManager* fieldMgr=0;
252 G4bool fieldExertsForce = false ;
253
254 G4bool gravityOn = false;
255 G4bool fieldExists = false; // Field is not 0 (null pointer)
256
257 fieldMgr = fFieldPropagator->FindAndSetFieldManager( track.GetVolume() );
258 if( fieldMgr != 0 )
259 {
260 // Message the field Manager, to configure it for this track
261 //
262 fieldMgr->ConfigureForTrack( &track );
263
264 // Is here to allow a transition from no-field pointer
265 // to finite field (non-zero pointer).
266 // If the field manager has no field ptr, the field is zero
267 // by definition ( = there is no field ! )
268 //
269 const G4Field* ptrField= fieldMgr->GetDetectorField();
270 fieldExists = (ptrField!=0) ;
271 if( fieldExists )
272 {
273 gravityOn= ptrField->IsGravityActive();
274
275 if( (particleCharge != 0.0)
276 || (fUseMagneticMoment && (magneticMoment != 0.0) )
277 || (gravityOn && (restMass != 0.0) ) )
278 {
279 fieldExertsForce = fieldExists;
280 }
281 }
282 }
283 fFieldExertedForce = fieldExertsForce;
284
285 if( !fieldExertsForce )
286 {
287 G4double linearStepLength ;
288 if( fShortStepOptimisation && (currentMinimumStep <= currentSafety) )
289 {
290 // The Step is guaranteed to be taken
291 //
292 geometryStepLength = currentMinimumStep ;
293 fGeometryLimitedStep = false ;
294 }
295 else
296 {
297 // Find whether the straight path intersects a volume
298 //
299 linearStepLength = fLinearNavigator->ComputeStep( startPosition,
300 startMomentumDir,
301 currentMinimumStep,
302 newSafety) ;
303 // Remember last safety origin & value.
304 //
305 fPreviousSftOrigin = startPosition ;
306 fPreviousSafety = newSafety ;
307 fpSafetyHelper->SetCurrentSafety( newSafety, startPosition);
308
309 currentSafety = newSafety ;
310
311 fGeometryLimitedStep= (linearStepLength <= currentMinimumStep);
312 if( fGeometryLimitedStep )
313 {
314 // The geometry limits the Step size (an intersection was found.)
315 geometryStepLength = linearStepLength ;
316 }
317 else
318 {
319 // The full Step is taken.
320 geometryStepLength = currentMinimumStep ;
321 }
322 }
323 fEndPointDistance = geometryStepLength ;
324
325 // Calculate final position
326 //
327 fTransportEndPosition = startPosition+geometryStepLength*startMomentumDir;
328
329 // Momentum direction, energy and polarisation are unchanged by transport
330 //
331 fTransportEndMomentumDir = startMomentumDir ;
332 fTransportEndKineticEnergy = track.GetKineticEnergy() ;
333 fTransportEndSpin = track.GetPolarization();
334 fParticleIsLooping = false ;
335 fMomentumChanged = false ;
336 fEndGlobalTimeComputed = false ;
337 }
338 else // A field exerts force
339 {
340 G4double momentumMagnitude = pParticle->GetTotalMomentum() ;
341 G4ThreeVector EndUnitMomentum ;
342 G4double lengthAlongCurve ;
343
344 // The charge can change (dynamic)
345 //
346 G4ChargeState chargeState(particleCharge,
347 magneticMoment,
348 pParticleDef->GetPDGSpin() );
349
350 auto equationOfMotion = fFieldPropagator->GetCurrentEquationOfMotion();
351
352 equationOfMotion->SetChargeMomentumMass( chargeState,
353 momentumMagnitude,
354 restMass);
355
356 G4FieldTrack aFieldTrack = G4FieldTrack( startPosition,
357 track.GetGlobalTime(), // Lab.
358 track.GetMomentumDirection(),
359 track.GetKineticEnergy(),
360 restMass,
361 particleCharge,
362 track.GetPolarization(),
363 pParticleDef->GetPDGMagneticMoment(),
364 0.0, // Length along track
365 pParticleDef->GetPDGSpin() );
366
367 if( currentMinimumStep > 0 )
368 {
369 // Do the Transport in the field (non recti-linear)
370 //
371 lengthAlongCurve = fFieldPropagator->ComputeStep( aFieldTrack,
372 currentMinimumStep,
373 currentSafety,
374 track.GetVolume() );
375
376 fGeometryLimitedStep= fFieldPropagator->IsLastStepInVolume();
377 //
378 // It is possible that step was reduced in PropagatorInField due to
379 // previous zero steps. To cope with case that reduced step is taken
380 // in full, we must rely on PiF to obtain this value
381
382 geometryStepLength = std::min( lengthAlongCurve, currentMinimumStep );
383
384 // Remember last safety origin & value.
385 //
386 fPreviousSftOrigin = startPosition ;
387 fPreviousSafety = currentSafety ;
388 fpSafetyHelper->SetCurrentSafety( currentSafety, startPosition);
389 }
390 else
391 {
392 geometryStepLength = lengthAlongCurve= 0.0 ;
393 fGeometryLimitedStep = false ;
394 }
395
396 // Get the End-Position and End-Momentum (Dir-ection)
397 //
398 fTransportEndPosition = aFieldTrack.GetPosition() ;
399
400 // Momentum: Magnitude and direction can be changed too now ...
401 //
402 fMomentumChanged = true ;
403 fTransportEndMomentumDir = aFieldTrack.GetMomentumDir() ;
404
405 fTransportEndKineticEnergy = aFieldTrack.GetKineticEnergy() ;
406
407 if( fFieldPropagator->GetCurrentFieldManager()->DoesFieldChangeEnergy() )
408 {
409 // If the field can change energy, then the time must be integrated
410 // - so this should have been updated
411 //
412 fCandidateEndGlobalTime = aFieldTrack.GetLabTimeOfFlight();
413 fEndGlobalTimeComputed = true;
414
415 // was ( fCandidateEndGlobalTime != track.GetGlobalTime() );
416 // a cleaner way is to have FieldTrack knowing whether time is updated.
417 }
418 else
419 {
420 // The energy should be unchanged by field transport,
421 // - so the time changed will be calculated elsewhere
422 //
423 fEndGlobalTimeComputed = false;
424
425 // Check that the integration preserved the energy
426 // - and if not correct this!
427 G4double startEnergy= track.GetKineticEnergy();
428 G4double endEnergy= fTransportEndKineticEnergy;
429
430 static G4ThreadLocal G4int no_inexact_steps=0, no_large_ediff;
431 G4double absEdiff = std::fabs(startEnergy- endEnergy);
432 if( absEdiff > perMillion * endEnergy )
433 {
434 no_inexact_steps++;
435 // Possible statistics keeping here ...
436 }
437 if( verboseLevel > 1 )
438 {
439 if( std::fabs(startEnergy- endEnergy) > perThousand * endEnergy )
440 {
441 static G4ThreadLocal G4int no_warnings= 0, warnModulo=1,
442 moduloFactor= 10;
443 no_large_ediff ++;
444 if( (no_large_ediff% warnModulo) == 0 )
445 {
446 no_warnings++;
447 std::ostringstream message;
448 message << "Energy change in Step is above 1^-3 relative value. "
449 << G4endl
450 << " Relative change in 'tracking' step = "
451 << std::setw(15) << (endEnergy-startEnergy)/startEnergy
452 << G4endl
453 << " Starting E= " << std::setw(12) << startEnergy / MeV
454 << " MeV " << G4endl
455 << " Ending E= " << std::setw(12) << endEnergy / MeV
456 << " MeV " << G4endl
457 << "Energy has been corrected -- however, review"
458 << " field propagation parameters for accuracy." << G4endl;
459 if ( (verboseLevel > 2 ) || (no_warnings<4)
460 || (no_large_ediff == warnModulo * moduloFactor) )
461 {
462 message << "These include EpsilonStepMax(/Min) in G4FieldManager " << G4endl
463 << "which determine fractional error per step for integrated quantities. " << G4endl
464 << "Note also the influence of the permitted number of integration steps."
465 << G4endl;
466 }
467 message << "Bad 'endpoint'. Energy change detected and corrected."
468 << G4endl
469 << "Has occurred already " << no_large_ediff << " times.";
470 G4Exception("G4Transportation::AlongStepGetPIL()",
471 "EnergyChange", JustWarning, message);
472 if( no_large_ediff == warnModulo * moduloFactor )
473 {
474 warnModulo *= moduloFactor;
475 }
476 }
477 }
478 } // end of if (verboseLevel)
479
480 // Correct the energy for fields that conserve it
481 // This - hides the integration error
482 // - but gives a better physical answer
483 //
484 fTransportEndKineticEnergy= track.GetKineticEnergy();
485 }
486
487 fTransportEndSpin = aFieldTrack.GetSpin();
488 fParticleIsLooping = fFieldPropagator->IsParticleLooping() ;
489 fEndPointDistance = (fTransportEndPosition - startPosition).mag() ;
490 }
491
492 // If we are asked to go a step length of 0, and we are on a boundary
493 // then a boundary will also limit the step -> we must flag this.
494 //
495 if( currentMinimumStep == 0.0 )
496 {
497 if( currentSafety == 0.0 ) { fGeometryLimitedStep = true; }
498 }
499
500 // Update the safety starting from the end-point,
501 // if it will become negative at the end-point.
502 //
503 if( currentSafety < fEndPointDistance )
504 {
505 if( particleCharge != 0.0 )
506 {
507 G4double endSafety =
508 fLinearNavigator->ComputeSafety( fTransportEndPosition) ;
509 currentSafety = endSafety ;
510 fPreviousSftOrigin = fTransportEndPosition ;
511 fPreviousSafety = currentSafety ;
512 fpSafetyHelper->SetCurrentSafety(currentSafety, fTransportEndPosition);
513
514 // Because the Stepping Manager assumes it is from the start point,
515 // add the StepLength
516 //
517 currentSafety += fEndPointDistance ;
518
519#ifdef G4DEBUG_TRANSPORT
520 G4cout.precision(12) ;
521 G4cout << "***G4Transportation::AlongStepGPIL ** " << G4endl ;
522 G4cout << " Called Navigator->ComputeSafety at " << fTransportEndPosition
523 << " and it returned safety= " << endSafety << G4endl ;
524 G4cout << " Adding endpoint distance " << fEndPointDistance
525 << " to obtain pseudo-safety= " << currentSafety << G4endl ;
526 }
527 else
528 {
529 G4cout << "***G4Transportation::AlongStepGPIL ** " << G4endl ;
530 G4cout << " Avoiding call to ComputeSafety : " << G4endl;
531 G4cout << " charge = " << particleCharge << G4endl;
532 G4cout << " mag moment = " << magneticMoment << G4endl;
533#endif
534 }
535 }
536
537 fParticleChange.ProposeTrueStepLength(geometryStepLength) ;
538
539 return geometryStepLength ;
540}

542//////////////////////////////////////////////////////////////////////////
543//
544// Initialize ParticleChange (by setting all its members equal
545// to corresponding members in G4Track)
546
547G4VParticleChange* G4Transportation::AlongStepDoIt( const G4Track& track,
548 const G4Step& stepData )
549{
550 static G4ThreadLocal G4long noCallsASDI=0;
551 const char *methodName= "AlongStepDoIt";
552 noCallsASDI++;
553
554 fParticleChange.Initialize(track) ;
555
556 // Code for specific process
557 //
558 fParticleChange.ProposePosition(fTransportEndPosition) ;
559 fParticleChange.ProposeMomentumDirection(fTransportEndMomentumDir) ;
560 fParticleChange.ProposeEnergy(fTransportEndKineticEnergy) ;
561 fParticleChange.SetMomentumChanged(fMomentumChanged) ;
562
563 fParticleChange.ProposePolarization(fTransportEndSpin);
564
565 G4double deltaTime = 0.0 ;
566
567 // Calculate Lab Time of Flight (ONLY if field Equations used it!)
568 // G4double endTime = fCandidateEndGlobalTime;
569 // G4double delta_time = endTime - startTime;
570
571 G4double startTime = track.GetGlobalTime() ;
572
573 if (!fEndGlobalTimeComputed)
574 {
575 // The time was not integrated .. make the best estimate possible
576 //
577 G4double initialVelocity = stepData.GetPreStepPoint()->GetVelocity();
578 G4double stepLength = track.GetStepLength();
579
580 deltaTime= 0.0; // in case initialVelocity = 0
581 if ( initialVelocity > 0.0 ) { deltaTime = stepLength/initialVelocity; }
582
583 fCandidateEndGlobalTime = startTime + deltaTime ;
584 fParticleChange.ProposeLocalTime( track.GetLocalTime() + deltaTime) ;
585 }
586 else
587 {
588 deltaTime = fCandidateEndGlobalTime - startTime ;
589 fParticleChange.ProposeGlobalTime( fCandidateEndGlobalTime ) ;
590 }
591
592
593 // Now Correct by Lorentz factor to get delta "proper" Time
594
595 G4double restMass = track.GetDynamicParticle()->GetMass() ;
596 G4double deltaProperTime = deltaTime*( restMass/track.GetTotalEnergy() ) ;
597
598 fParticleChange.ProposeProperTime(track.GetProperTime() + deltaProperTime) ;
599 //fParticleChange.ProposeTrueStepLength( track.GetStepLength() ) ;
600
601 // If the particle is caught looping or is stuck (in very difficult
602 // boundaries) in a magnetic field (doing many steps) THEN can kill it ...
603 //
604 if ( fParticleIsLooping )
605 {
606 G4double endEnergy= fTransportEndKineticEnergy;
607 fNoLooperTrials ++;
608 auto particleType= track.GetDynamicParticle()->GetParticleDefinition();
609
610 G4bool stable = particleType->GetPDGStable();
611 G4bool candidateForEnd = (endEnergy < fThreshold_Important_Energy)
612 || (fNoLooperTrials >= fThresholdTrials) ;
613 G4bool unstableAndKillable = !stable && ( fAbandonUnstableTrials != 0);
614 G4bool unstableForEnd = (endEnergy < fThreshold_Important_Energy)
615 && (fNoLooperTrials >= fAbandonUnstableTrials) ;
616 if( (candidateForEnd && stable) || (unstableAndKillable && unstableForEnd) )
617 {
618 // Kill the looping particle
619 //
620 fParticleChange.ProposeTrackStatus( fStopAndKill ) ;
621 G4int particlePDG= particleType->GetPDGEncoding();
622 const G4int electronPDG= 11; // G4Electron::G4Electron()->GetPDGEncoding();
623
624 // Simple statistics
625 fSumEnergyKilled += endEnergy;
626 fSumEnerSqKilled = endEnergy * endEnergy;
627 fNumLoopersKilled++;
628
629 if( endEnergy > fMaxEnergyKilled ) {
630 fMaxEnergyKilled = endEnergy;
631 fMaxEnergyKilledPDG = particlePDG;
632 }
633 if( particleType->GetPDGEncoding() != electronPDG )
634 {
635 fSumEnergyKilled_NonElectron += endEnergy;
636 fSumEnerSqKilled_NonElectron += endEnergy * endEnergy;
637 fNumLoopersKilled_NonElectron++;
638
639 if( endEnergy > fMaxEnergyKilled_NonElectron )
640 {
641 fMaxEnergyKilled_NonElectron = endEnergy;
642 fMaxEnergyKilled_NonElecPDG = particlePDG;
643 }
644 }
645
646 if( endEnergy > fThreshold_Warning_Energy )
647 {
648 fpLogger->ReportLoopingTrack( track, stepData, fNoLooperTrials,
649 noCallsASDI, methodName );
650 }
651 fNoLooperTrials=0;
652 }
653 else
654 {
655 fMaxEnergySaved = std::max( endEnergy, fMaxEnergySaved);
656 if( fNoLooperTrials == 1 ) {
657 fSumEnergySaved += endEnergy;
658 if ( !stable )
659 fSumEnergyUnstableSaved += endEnergy;
660 }
661#ifdef G4VERBOSE
662 if( verboseLevel > 2 )
663 {
664 G4cout << " " << methodName
665 << " Particle is looping but is saved ..." << G4endl
666 << " Number of trials = " << fNoLooperTrials << G4endl
667 << " No of calls to = " << noCallsASDI << G4endl;
668 }
669#endif
670 }
671 }
672 else
673 {
674 fNoLooperTrials=0;
675 }
676
677 // Another (sometimes better way) is to use a user-limit maximum Step size
678 // to alleviate this problem ..
679
680 // Introduce smooth curved trajectories to particle-change
681 //
682 fParticleChange.SetPointerToVectorOfAuxiliaryPoints
683 (fFieldPropagator->GimmeTrajectoryVectorAndForgetIt() );
684
685 return &fParticleChange ;
686}

704/////////////////////////////////////////////////////////////////////////////
705//
706
707G4VParticleChange* G4Transportation::PostStepDoIt( const G4Track& track,
708 const G4Step& )
709{
710 G4TouchableHandle retCurrentTouchable ; // The one to return
711 G4bool isLastStep= false;
712
713 // Initialize ParticleChange (by setting all its members equal
714 // to corresponding members in G4Track)
715 // fParticleChange.Initialize(track) ; // To initialise TouchableChange
716
717 fParticleChange.ProposeTrackStatus(track.GetTrackStatus()) ;
718
719 // If the Step was determined by the volume boundary,
720 // logically relocate the particle
721
722 if(fGeometryLimitedStep)
723 {
724 // fCurrentTouchable will now become the previous touchable,
725 // and what was the previous will be freed.
726 // (Needed because the preStepPoint can point to the previous touchable)
727
728 fLinearNavigator->SetGeometricallyLimitedStep() ;
729 fLinearNavigator->
730 LocateGlobalPointAndUpdateTouchableHandle( track.GetPosition(),
731 track.GetMomentumDirection(),
732 fCurrentTouchableHandle,
733 true ) ;
734 // Check whether the particle is out of the world volume
735 // If so it has exited and must be killed.
736 //
737 if( fCurrentTouchableHandle->GetVolume() == 0 )
738 {
739 fParticleChange.ProposeTrackStatus( fStopAndKill ) ;
740 }
741 retCurrentTouchable = fCurrentTouchableHandle ;
742 fParticleChange.SetTouchableHandle( fCurrentTouchableHandle ) ;
743
744 // Update the Step flag which identifies the Last Step in a volume
745 if( !fFieldExertedForce )
746 isLastStep = fLinearNavigator->ExitedMotherVolume()
747 | fLinearNavigator->EnteredDaughterVolume() ;
748 else
749 isLastStep = fFieldPropagator->IsLastStepInVolume();
750 }
751 else // fGeometryLimitedStep is false
752 {
753 // This serves only to move the Navigator's location
754 //
755 fLinearNavigator->LocateGlobalPointWithinVolume( track.GetPosition() ) ;
756
757 // The value of the track's current Touchable is retained.
758 // (and it must be correct because we must use it below to
759 // overwrite the (unset) one in particle change)
760 // It must be fCurrentTouchable too ??
761 //
762 fParticleChange.SetTouchableHandle( track.GetTouchableHandle() ) ;
763 retCurrentTouchable = track.GetTouchableHandle() ;
764
765 isLastStep= false;
766 } // endif ( fGeometryLimitedStep )
767 fLastStepInVolume= isLastStep;
768
769 fParticleChange.ProposeFirstStepInVolume(fFirstStepInVolume);
770 fParticleChange.ProposeLastStepInVolume(isLastStep);
771
772 const G4VPhysicalVolume* pNewVol = retCurrentTouchable->GetVolume() ;
773 const G4Material* pNewMaterial = 0 ;
774 const G4VSensitiveDetector* pNewSensitiveDetector = 0 ;
775
776 if( pNewVol != 0 )
777 {
778 pNewMaterial= pNewVol->GetLogicalVolume()->GetMaterial();
779 pNewSensitiveDetector= pNewVol->GetLogicalVolume()->GetSensitiveDetector();
780 }
781
782 fParticleChange.SetMaterialInTouchable( (G4Material *) pNewMaterial ) ;
783 fParticleChange.SetSensitiveDetectorInTouchable( (G4VSensitiveDetector *) pNewSensitiveDetector ) ;
784
785 const G4MaterialCutsCouple* pNewMaterialCutsCouple = 0;
786 if( pNewVol != 0 )
787 {
788 pNewMaterialCutsCouple=pNewVol->GetLogicalVolume()->GetMaterialCutsCouple();
789 }
790
791 if ( pNewVol!=0 && pNewMaterialCutsCouple!=0
792 && pNewMaterialCutsCouple->GetMaterial()!=pNewMaterial )
793 {
794 // for parametrized volume
795 //
796 pNewMaterialCutsCouple =
797 G4ProductionCutsTable::GetProductionCutsTable()
798 ->GetMaterialCutsCouple(pNewMaterial,
799 pNewMaterialCutsCouple->GetProductionCuts());
800 }
801 fParticleChange.SetMaterialCutsCoupleInTouchable( pNewMaterialCutsCouple );
802
803 // temporarily until Get/Set Material of ParticleChange,
804 // and StepPoint can be made const.
805 // Set the touchable in ParticleChange
806 // this must always be done because the particle change always
807 // uses this value to overwrite the current touchable pointer.
808 //
809 fParticleChange.SetTouchableHandle(retCurrentTouchable) ;
810
811 return &fParticleChange ;
812}

816// G4Transportation object at the start of a new track or the resumption
817// of a suspended track.

820void
821G4Transportation::StartTracking(G4Track* aTrack)
822{
823 G4VProcess::StartTracking(aTrack);
824 fNewTrack= true;
825 fFirstStepInVolume= true;
826 fLastStepInVolume= false;
827
828 // The actions here are those that were taken in AlongStepGPIL
829 // when track.GetCurrentStepNumber()==1
830
831 // reset safety value and center
832 //
833 fPreviousSafety = 0.0 ;
834 fPreviousSftOrigin = G4ThreeVector(0.,0.,0.) ;
835
836 // reset looping counter -- for motion in field
837 fNoLooperTrials= 0;
838 // Must clear this state .. else it depends on last track's value
839 // --> a better solution would set this from state of suspended track TODO ?
840 // Was if( aTrack->GetCurrentStepNumber()==1 ) { .. }
841
842 // ChordFinder reset internal state
843 //
844 if( DoesGlobalFieldExist() )
845 {
846 fFieldPropagator->ClearPropagatorState();
847 // Resets all state of field propagator class (ONLY) including safety
848 // values (in case of overlaps and to wipe for first track).
849 }
850
851 // Make sure to clear the chord finders of all fields (i.e. managers)
852 //
853 G4FieldManagerStore* fieldMgrStore = G4FieldManagerStore::GetInstance();
854 fieldMgrStore->ClearAllChordFindersState();
855
856 // Update the current touchable handle (from the track's)
857 //
858 fCurrentTouchableHandle = aTrack->GetTouchableHandle();
859
860 // Inform field propagator of new track
861 //
862 fFieldPropagator->PrepareNewTrack();
863}
