60G4CoupledTransportation::G4CoupledTransportation( G4int verbosity )
61 : G4VProcess( G4String("CoupledTransportation"), fTransportation ),
62 fTransportEndPosition(0.0, 0.0, 0.0),
63 fTransportEndMomentumDir(0.0, 0.0, 0.0),
64 fTransportEndKineticEnergy(0.0),
65 fTransportEndSpin(0.0, 0.0, 0.0),
66 fMomentumChanged(false),
67 fEndGlobalTimeComputed(false),
68 fCandidateEndGlobalTime(0.0),
69 fParticleIsLooping( false ),
70 fNewTrack( true ),
71 fPreviousSftOrigin( 0.,0.,0. ),
72 fPreviousMassSafety( 0.0 ),
73 fPreviousFullSafety( 0.0 ),
74 fMassGeometryLimitedStep( false ),
75 fAnyGeometryLimitedStep( false ),
76 fEndpointDistance( -1.0 ),
77 fSumEnergyKilled( 0.0 ), fMaxEnergyKilled( 0.0 ),
78 fFirstStepInMassVolume( true ),
79 fFirstStepInAnyVolume( true )
80{
81 SetProcessSubType(static_cast<G4int>(COUPLED_TRANSPORTATION));
82 SetVerboseLevel(verbosity);
83
84 G4TransportationManager* transportMgr ;
85
86 transportMgr = G4TransportationManager::GetTransportationManager() ;
87
88 fMassNavigator = transportMgr->GetNavigatorForTracking() ;
89 fFieldPropagator = transportMgr->GetPropagatorInField() ;
90 // fGlobalFieldMgr = transportMgr->GetFieldManager() ;
91 fNavigatorId= transportMgr->ActivateNavigator( fMassNavigator );
92 if( verboseLevel > 0 )
93 {
94 G4cout << " G4CoupledTransportation constructor: ----- " << G4endl;
95 G4cout << " Verbose level is " << verboseLevel << G4endl;
96 G4cout << " Navigator Id obtained in G4CoupledTransportation constructor "
97 << fNavigatorId << G4endl;
98 G4cout << " Reports First/Last in "
99 << (fSignifyStepInAnyVolume ? " any " : " mass " )
100 << " geometry " << G4endl;
101 }
102 fPathFinder= G4PathFinder::GetInstance();
103 fpSafetyHelper = transportMgr->GetSafetyHelper(); // New
104
105 fpLogger = new G4TransportationLogger("G4Transportation", verbosity);
106
107 SetHighLooperThresholds();
108 // Use the old defaults: Warning = 100 MeV, Important = 250 MeV, No Trials = 10;
109
110 PushThresholdsToLogger();
111 // Should be done by Set methods in SetHighLooperThresholds -- making sure
112
113 static G4ThreadLocal G4TouchableHandle* pNullTouchableHandle = 0;
114 if ( !pNullTouchableHandle) { pNullTouchableHandle = new G4TouchableHandle; }
115 fCurrentTouchableHandle = *pNullTouchableHandle;
116 // Points to (G4VTouchable*) 0
117
118 G4FieldManager *globalFieldMgr= transportMgr->GetFieldManager();
119 fGlobalFieldExists= globalFieldMgr ? globalFieldMgr->GetDetectorField() : 0 ;
120}

157//
158// Responsibilities:
159// Find whether the geometry limits the Step, and to what length
160// Calculate the new value of the safety and return it.
161// Store the final time, position and momentum.
162
163G4double G4CoupledTransportation::
164AlongStepGetPhysicalInteractionLength( const G4Track& track,
165 G4double, // previousStepSize
166 G4double currentMinimumStep,
167 G4double& proposedSafetyForStart,
168 G4GPILSelection* selection )
169{
170 G4double geometryStepLength;
171 G4double startMassSafety= 0.0; // estimated safety for start point (mass geometry)
172 G4double startFullSafety= 0.0; // estimated safety for start point (all geometries)
173 G4double safetyProposal= -1.0; // local copy of proposal
174
175 G4ThreeVector EndUnitMomentum ;
176 G4double lengthAlongCurve = 0.0 ;
177
178 fParticleIsLooping = false ;
179
180 // Initial actions moved to StartTrack()
181 // --------------------------------------
182 // Note: in case another process changes touchable handle
183 // it will be necessary to add here (for all steps)
184 // fCurrentTouchableHandle = aTrack->GetTouchableHandle();
185
186 // GPILSelection is set to defaule value of CandidateForSelection
187 // It is a return value
188 //
189 *selection = CandidateForSelection ;
190
191 fFirstStepInMassVolume = fNewTrack | fMassGeometryLimitedStep ;
192 fFirstStepInAnyVolume = fNewTrack | fAnyGeometryLimitedStep ;
193
194#ifdef G4DEBUG_TRANSPORT
195 G4cout << " CoupledTransport::AlongStep GPIL: "
196 << " 1st-step: any= " <<fFirstStepInAnyVolume << " ( geom= "
197 << fAnyGeometryLimitedStep << " ) "
198 << " mass= " << fFirstStepInMassVolume << " ( geom= "
199 << fMassGeometryLimitedStep << " ) "
200 << " newTrack= " << fNewTrack << G4endl;
201#endif
202
203 // fLastStepInVolume= false;
204 fNewTrack = false;
205
206 // Get initial Energy/Momentum of the track
207 //
208 const G4DynamicParticle* pParticle = track.GetDynamicParticle() ;
209 const G4ParticleDefinition* pParticleDef = pParticle->GetDefinition() ;
210 G4ThreeVector startMomentumDir = pParticle->GetMomentumDirection() ;
211 G4ThreeVector startPosition = track.GetPosition() ;
212 G4VPhysicalVolume* currentVolume= track.GetVolume();
213
214#ifdef G4DEBUG_TRANSPORT
215 if( verboseLevel > 1 )
216 {
217 G4cout << "G4CoupledTransportation::AlongStepGPIL> called in volume "
218 << currentVolume->GetName() << G4endl;
219 }
220#endif
221 // G4double theTime = track.GetGlobalTime() ;
222
223 // The Step Point safety can be limited by other geometries and/or the
224 // assumptions of any process - it's not always the geometrical safety.
225 // We calculate the starting point's isotropic safety here.
226 //
227 G4ThreeVector OriginShift = startPosition - fPreviousSftOrigin ;
228 G4double MagSqShift = OriginShift.mag2() ;
229 startMassSafety = 0.0;
230 startFullSafety= 0.0;
231
232 // Recall that FullSafety <= MassSafety
233 // Original: if( MagSqShift < sqr(fPreviousMassSafety) ) {
234 if( MagSqShift < sqr(fPreviousFullSafety) )
235 {
236 G4double mag_shift= std::sqrt(MagSqShift);
237 startMassSafety = std::max( (fPreviousMassSafety - mag_shift), 0.0);
238 startFullSafety = std::max( (fPreviousFullSafety - mag_shift), 0.0);
239 // Need to be consistent between full safety with Mass safety
240 // in order reproduce results in simple case
241 // --> use same calculation method
242
243 // Only compute full safety if massSafety > 0. Else it remains 0
244 // startFullSafety = fPathFinder->ComputeSafety( startPosition );
245 }
246
247 // Is the particle charged or has it a magnetic moment?
248 //
249 G4double particleCharge = pParticle->GetCharge() ;
250 G4double magneticMoment = pParticle->GetMagneticMoment() ;
251 G4double restMass = pParticle->GetMass() ;
252
253 fMassGeometryLimitedStep = false ; // Set default - alex
254 fAnyGeometryLimitedStep = false;
255
256 // There is no need to locate the current volume. It is Done elsewhere:
257 // On track construction
258 // By the tracking, after all AlongStepDoIts, in "Relocation"
259
260 // Check if the particle has a force, EM or gravitational, exerted on it
261 //
262 G4FieldManager* fieldMgr=0;
263 G4bool fieldExertsForce = false ;
264
265 G4bool gravityOn = false;
266 const G4Field* ptrField= 0;
267
268 fieldMgr = fFieldPropagator->FindAndSetFieldManager( track.GetVolume() );
269 if( fieldMgr != 0 )
270 {
271 // Message the field Manager, to configure it for this track
272 //
273 fieldMgr->ConfigureForTrack( &track );
274
275 // Here it can transition from a null field-ptr to a finite field.
276 // If the field manager has no field ptr, the field is zero
277 // by definition ( = there is no field ! )
278 //
279 ptrField= fieldMgr->GetDetectorField();
280
281 if( ptrField != 0)
282 {
283 gravityOn= ptrField->IsGravityActive();
284 if( (particleCharge != 0.0)
285 || (fUseMagneticMoment && (magneticMoment != 0.0) )
286 || (gravityOn && (restMass != 0.0)) )
287 {
288 fieldExertsForce = true;
289 }
290 }
291 }
292 G4double momentumMagnitude = pParticle->GetTotalMomentum() ;
293
294 if( fieldExertsForce )
295 {
296 auto equationOfMotion= fFieldPropagator->GetCurrentEquationOfMotion();
297
298 G4ChargeState chargeState(particleCharge, // Charge can change (dynamic)
299 magneticMoment,
300 pParticleDef->GetPDGSpin() );
301 if( equationOfMotion )
302 {
303 equationOfMotion->SetChargeMomentumMass( chargeState,
304 momentumMagnitude,
305 restMass );
306 }
307#ifdef G4DEBUG_TRANSPORT
308 else
309 {
310 G4cerr << " ERROR in G4CoupledTransportation> "
311 << "Cannot find valid Equation of motion: " << G4endl;
312 << " Unable to pass Charge, Momentum and Mass " << G4endl;
313 }
314#endif
315 }
316
317 G4ThreeVector polarizationVec = track.GetPolarization() ;
318 G4FieldTrack aFieldTrack = G4FieldTrack(startPosition,
319 track.GetGlobalTime(), // Lab.
320 track.GetMomentumDirection(),
321 track.GetKineticEnergy(),
322 restMass,
323 particleCharge,
324 polarizationVec,
325 pParticleDef->GetPDGMagneticMoment(),
326 0.0, // Length along track
327 pParticleDef->GetPDGSpin() ) ;
328 G4int stepNo= track.GetCurrentStepNumber();
329
330 ELimited limitedStep;
331 G4FieldTrack endTrackState('a'); // Default values
332
333 fMassGeometryLimitedStep = false ; // default
334 fAnyGeometryLimitedStep = false ;
335 if( currentMinimumStep > 0 )
336 {
337 G4double newMassSafety= 0.0; // temp. for recalculation
338
339 // Do the Transport in the field (non recti-linear)
340 //
341 lengthAlongCurve = fPathFinder->ComputeStep( aFieldTrack,
342 currentMinimumStep,
343 fNavigatorId,
344 stepNo,
345 newMassSafety,
346 limitedStep,
347 endTrackState,
348 currentVolume ) ;
349
350 G4double newFullSafety= fPathFinder->GetCurrentSafety();
351 // this was estimated already in step above
352
353 if( limitedStep == kUnique || limitedStep == kSharedTransport )
354 {
355 fMassGeometryLimitedStep = true ;
356 }
357
358 fAnyGeometryLimitedStep = (fPathFinder->GetNumberGeometriesLimitingStep() != 0);
359
360#ifdef G4DEBUG_TRANSPORT
361 if( fMassGeometryLimitedStep && !fAnyGeometryLimitedStep )
362 {
363 std::ostringstream message;
364 message << " ERROR in determining geometries limiting the step" << G4endl;
365 message << " Limiting: mass=" << fMassGeometryLimitedStep
366 << " any= " << fAnyGeometryLimitedStep << G4endl;
367 message << "Incompatible conditions - by which geometries was it limited ?";
368 G4Exception("G4CoupledTransportation::AlongStepGetPhysicalInteractionLength()",
369 "PathFinderConfused", FatalException, message);
370 }
371#endif
372
373 geometryStepLength = std::min( lengthAlongCurve, currentMinimumStep);
374
375 // Momentum: Magnitude and direction can be changed too now ...
376 //
377 fMomentumChanged = true ;
378 fTransportEndMomentumDir = endTrackState.GetMomentumDir() ;
379
380 // Remember last safety origin & value.
381 fPreviousSftOrigin = startPosition ;
382 fPreviousMassSafety = newMassSafety ;
383 fPreviousFullSafety = newFullSafety ;
384 // fpSafetyHelper->SetCurrentSafety( newFullSafety, startPosition);
385
386#ifdef G4DEBUG_TRANSPORT
387 if( verboseLevel > 1 )
388 {
389 G4cout << "G4Transport:CompStep> "
390 << " called the pathfinder for a new step at " << startPosition
391 << " and obtained step = " << lengthAlongCurve << G4endl;
392 G4cout << " New safety (preStep) = " << newMassSafety
393 << " versus precalculated = " << startMassSafety << G4endl;
394 }
395#endif
396
397 // Store as best estimate value
398 startMassSafety = newMassSafety ;
399 startFullSafety = newFullSafety ;
400
401 // Get the End-Position and End-Momentum (Dir-ection)
402 fTransportEndPosition = endTrackState.GetPosition() ;
403 fTransportEndKineticEnergy = endTrackState.GetKineticEnergy() ;
404 }
405 else
406 {
407 geometryStepLength = lengthAlongCurve= 0.0 ;
408 fMomentumChanged = false ;
409 // fMassGeometryLimitedStep = false ; // --- ???
410 // fAnyGeometryLimitedStep = true;
411 fTransportEndMomentumDir = track.GetMomentumDirection();
412 fTransportEndKineticEnergy = track.GetKineticEnergy();
413
414 fTransportEndPosition = startPosition;
415
416 endTrackState= aFieldTrack; // Ensures that time is updated
417
418 // If the step length requested is 0, and we are on a boundary
419 // then a boundary will also limit the step.
420 if( startMassSafety == 0.0 )
421 {
422 fMassGeometryLimitedStep = true ;
423 fAnyGeometryLimitedStep = true;
424 }
425 // TODO: Add explicit logical status for being at a boundary
426 }
427 // G4FieldTrack aTrackState(endTrackState);
428
429 if( !fieldExertsForce )
430 {
431 fParticleIsLooping = false ;
432 fMomentumChanged = false ;
433 fEndGlobalTimeComputed = false ;
434 }
435 else
436 {
437
438#ifdef G4DEBUG_TRANSPORT
439 if( verboseLevel > 1 )
440 {
441 G4cout << " G4CT::CS End Position = "
442 << fTransportEndPosition << G4endl;
443 G4cout << " G4CT::CS End Direction = "
444 << fTransportEndMomentumDir << G4endl;
445 }
446#endif
447 if( fFieldPropagator->GetCurrentFieldManager()->DoesFieldChangeEnergy() )
448 {
449 // If the field can change energy, then the time must be integrated
450 // - so this should have been updated
451 //
452 fCandidateEndGlobalTime = endTrackState.GetLabTimeOfFlight();
453 fEndGlobalTimeComputed = true;
454
455 // was ( fCandidateEndGlobalTime != track.GetGlobalTime() );
456 // a cleaner way is to have FieldTrack knowing whether time
457 // is updated
458 }
459 else
460 {
461 // The energy should be unchanged by field transport,
462 // - so the time changed will be calculated elsewhere
463 //
464 fEndGlobalTimeComputed = false;
465
466 // Check that the integration preserved the energy
467 // - and if not correct this!
468 G4double startEnergy= track.GetKineticEnergy();
469 G4double endEnergy= fTransportEndKineticEnergy;
470
471 static G4ThreadLocal G4int no_inexact_steps=0; // , no_large_ediff;
472 G4double absEdiff = std::fabs(startEnergy- endEnergy);
473 if( absEdiff > perMillion * endEnergy )
474 {
475 no_inexact_steps++;
476 // Possible statistics keeping here ...
477 }
478#ifdef G4VERBOSE
479 if( (verboseLevel > 1) && ( absEdiff > perThousand * endEnergy) )
480 {
481 ReportInexactEnergy(startEnergy, endEnergy);
482 } // end of if (verboseLevel)
483#endif
484 // Correct the energy for fields that conserve it
485 // This - hides the integration error
486 // - but gives a better physical answer
487 fTransportEndKineticEnergy= track.GetKineticEnergy();
488 }
489 }
490
491 fEndpointDistance = (fTransportEndPosition - startPosition).mag() ;
492 fParticleIsLooping = fFieldPropagator->IsParticleLooping() ;
493
494 fTransportEndSpin = endTrackState.GetSpin();
495
496 // Calculate the safety
497
498 safetyProposal= startFullSafety; // used to be startMassSafety
499 // Changed to accomodate processes that cannot update the safety
500
501 // Update safety for the end-point, if becomes negative at the end-point.
502
503 if( (startFullSafety < fEndpointDistance )
504 && ( particleCharge != 0.0 ) ) // Only needed to prepare for MSC
505 // && !fAnyGeometryLimitedStep ) // To-Try: No safety update if at boundary
506 {
507 G4double endFullSafety =
508 fPathFinder->ComputeSafety( fTransportEndPosition);
509 // Expected mission -- only mass geometry's safety
510 // fMassNavigator->ComputeSafety( fTransportEndPosition) ;
511 // Yet discrete processes only have poststep -- and this cannot
512 // currently revise the safety
513 // ==> so we use the all-geometry safety as a precaution
514
515 fpSafetyHelper->SetCurrentSafety( endFullSafety, fTransportEndPosition);
516 // Pushing safety to Helper avoids recalculation at this point
517
518 G4ThreeVector centerPt= G4ThreeVector(0.0, 0.0, 0.0); // Used for return value
519 G4double endMassSafety= fPathFinder->ObtainSafety( fNavigatorId, centerPt);
520 // Retrieves the mass value from PathFinder (it calculated it)
521
522 fPreviousMassSafety = endMassSafety ;
523 fPreviousFullSafety = endFullSafety;
524 fPreviousSftOrigin = fTransportEndPosition ;
525
526 // The convention (Stepping Manager's) is safety from the start point
527 //
528 safetyProposal = endFullSafety + fEndpointDistance;
529 // --> was endMassSafety
530 // Changed to accomodate processes that cannot update the safety
531
532#ifdef G4DEBUG_TRANSPORT
533 G4int prec= G4cout.precision(12) ;
534 G4cout << "***CoupledTransportation::AlongStepGPIL ** " << G4endl ;
535 G4cout << " Revised Safety at endpoint " << fTransportEndPosition
536 << " give safety values: Mass= " << endMassSafety
537 << " All= " << endFullSafety << G4endl ;
538 G4cout << " Adding endpoint distance " << fEndpointDistance
539 << " to obtain pseudo-safety= " << safetyProposal << G4endl ;
540 G4cout.precision(prec);
541 }
542 else
543 {
544 G4int prec= G4cout.precision(12) ;
545 G4cout << "***CoupledTransportation::AlongStepGPIL ** " << G4endl ;
546 G4cout << " Quick Safety estimate at endpoint "
547 << fTransportEndPosition
548 << " gives safety endpoint value = "
549 << startFullSafety - fEndpointDistance
550 << " using start-point value " << startFullSafety
551 << " and endpointDistance " << fEndpointDistance << G4endl;
552 G4cout.precision(prec);
553#endif
554 }
555
556 proposedSafetyForStart= safetyProposal;
557 fParticleChange.ProposeTrueStepLength(geometryStepLength) ;
558
559 return geometryStepLength ;
560}

562//////////////////////////////////////////////////////////////////////////
563
564G4VParticleChange*
565G4CoupledTransportation::AlongStepDoIt( const G4Track& track,
566 const G4Step& stepData )
567{
568 static G4ThreadLocal G4long noCallsCT_ASDI=0;
569 const char *methodName= "AlongStepDoIt";
570
571 noCallsCT_ASDI++;
572
573 fParticleChange.Initialize(track) ;
574 // sets all its members to the value of corresponding members in G4Track
575
576 // Code specific for Transport
577 //
578 fParticleChange.ProposePosition(fTransportEndPosition) ;
579 fParticleChange.ProposeMomentumDirection(fTransportEndMomentumDir) ;
580 fParticleChange.ProposeEnergy(fTransportEndKineticEnergy) ;
581 fParticleChange.SetMomentumChanged(fMomentumChanged) ;
582
583 fParticleChange.ProposePolarization(fTransportEndSpin);
584
585 G4double deltaTime = 0.0 ;
586
587 // Calculate Lab Time of Flight (ONLY if field Equations used it!)
588 // G4double endTime = fCandidateEndGlobalTime;
589 // G4double delta_time = endTime - startTime;
590
591 G4double startTime = track.GetGlobalTime() ;
592
593 if (!fEndGlobalTimeComputed)
594 {
595 G4double finalInverseVel= DBL_MAX, initialInverseVel=DBL_MAX;
596
597 // The time was not integrated .. make the best estimate possible
598 //
599 G4double finalVelocity = track.GetVelocity() ;
600 if( finalVelocity > 0.0 ) { finalInverseVel= 1.0 / finalVelocity; }
601 G4double initialVelocity = stepData.GetPreStepPoint()->GetVelocity() ;
602 if( initialVelocity > 0.0 ) { initialInverseVel= 1.0 / initialVelocity; }
603 G4double stepLength = track.GetStepLength() ;
604
605 if (finalVelocity > 0.0)
606 {
607 // deltaTime = stepLength/finalVelocity ;
608 G4double meanInverseVelocity = 0.5
609 * (initialInverseVel + finalInverseVel);
610 deltaTime = stepLength * meanInverseVelocity ;
611 }
612 else
613 {
614 deltaTime = stepLength * initialInverseVel ;
615 } // Could do with better estimate for final step (finalVelocity = 0) ?
616
617 fCandidateEndGlobalTime = startTime + deltaTime ;
618 fParticleChange.ProposeLocalTime( track.GetLocalTime() + deltaTime) ;
619 }
620 else
621 {
622 deltaTime = fCandidateEndGlobalTime - startTime ;
623 fParticleChange.ProposeGlobalTime( fCandidateEndGlobalTime ) ;
624 }
625
626 // Now Correct by Lorentz factor to get "proper" deltaTime
627
628 G4double restMass = track.GetDynamicParticle()->GetMass() ;
629 G4double deltaProperTime = deltaTime*( restMass/track.GetTotalEnergy() ) ;
630
631 fParticleChange.ProposeProperTime(track.GetProperTime() + deltaProperTime) ;
632 // fParticleChange. ProposeTrueStepLength( track.GetStepLength() ) ;
633
634 // If the particle is caught looping or is stuck (in very difficult
635 // boundaries) in a magnetic field (doing many steps) THEN this kills it ...
636 //
637 if ( fParticleIsLooping )
638 {
639 G4double endEnergy= fTransportEndKineticEnergy;
640
641 const G4ParticleDefinition* particleType=
642 track.GetDynamicParticle() -> GetParticleDefinition();
643 G4bool stable = particleType->GetPDGStable();
644
645 G4bool candidateForEnd = (endEnergy < fThreshold_Important_Energy)
646 || (fNoLooperTrials >= fThresholdTrials) ;
647
648 if( candidateForEnd && stable )
649 {
650 const G4int electronPDG= 11; // G4Electron::G4Electron()->GetPDGEncoding();
651 G4int particlePDG= particleType->GetPDGEncoding();
652
653 // Kill the looping particle
654 //
655 fParticleChange.ProposeTrackStatus( fStopAndKill ) ;
656
657 // Simple statistics
658 fSumEnergyKilled += endEnergy;
659 fSumEnerSqKilled = endEnergy * endEnergy;
660 fNumLoopersKilled++;
661
662 if( endEnergy > fMaxEnergyKilled ) {
663 fMaxEnergyKilled = endEnergy;
664 fMaxEnergyKilledPDG = particlePDG;
665 }
666
667 if( particleType->GetPDGEncoding() != electronPDG )
668 {
669 fSumEnergyKilled_NonElectron += endEnergy;
670 fSumEnerSqKilled_NonElectron += endEnergy * endEnergy;
671 fNumLoopersKilled_NonElectron++;
672
673 if( endEnergy > fMaxEnergyKilled_NonElectron )
674 {
675 fMaxEnergyKilled_NonElectron = endEnergy;
676 fMaxEnergyKilled_NonElecPDG = particlePDG;
677 }
678 }
679
680 if( endEnergy > fThreshold_Warning_Energy )
681 {
682 fpLogger->ReportLoopingTrack( track, stepData, fNoLooperTrials,
683 noCallsCT_ASDI, methodName );
684 }
685
686 fNoLooperTrials=0;
687 }
688 else
689 {
690 fNoLooperTrials ++;
691
692 fMaxEnergySaved = std::max( endEnergy, fMaxEnergySaved);
693 if( fNoLooperTrials == 1 ) {
694 fSumEnergySaved += endEnergy;
695 if ( !stable )
696 fSumEnergyUnstableSaved += endEnergy;
697 }
698#ifdef G4VERBOSE
699 if( verboseLevel > 2 )
700 {
701 G4cout << " ** G4CoupledTransportation::AlongStepDoIt():"
702 << " Particle is looping but is saved ..." << G4endl
703 << " Number of trials (this track) = " << fNoLooperTrials
704 << G4endl
705 << " Steps by this track: " << track.GetCurrentStepNumber()
706 << G4endl
707 << " Total no of calls to this method (all tracks) = "
708 << noCallsCT_ASDI << G4endl;
709 }
710#endif
711 }
712 }
713 else
714 {
715 fNoLooperTrials=0;
716 }
717
718 // Another (sometimes better way) is to use a user-limit maximum Step size
719 // to alleviate this problem ..
720 // Add smooth curved trajectories to particle-change
721 //
722 // fParticleChange.SetPointerToVectorOfAuxiliaryPoints
723 // (fFieldPropagator->GimmeTrajectoryVectorAndForgetIt() );
724
725 return &fParticleChange ;

766G4VParticleChange* G4CoupledTransportation::PostStepDoIt( const G4Track& track,
767 const G4Step& )
768{
769 G4TouchableHandle retCurrentTouchable ; // The one to return
770
771 // Initialize ParticleChange (by setting all its members equal
772 // to corresponding members in G4Track)
773 // fParticleChange.Initialize(track) ; // To initialise TouchableChange
774
775 fParticleChange.ProposeTrackStatus(track.GetTrackStatus()) ;
776
777 if( fSignifyStepInAnyVolume )
778 {
779 fParticleChange.ProposeFirstStepInVolume( fFirstStepInAnyVolume );
780 }
781 else
782 {
783 fParticleChange.ProposeFirstStepInVolume( fFirstStepInMassVolume );
784 }
785
786 // Check that the end position and direction are preserved
787 // since call to AlongStepDoIt
788
789#ifdef G4DEBUG_TRANSPORT
790 if( ( verboseLevel > 0 )
791 && ((fTransportEndPosition - track.GetPosition()).mag2() >= 1.0e-16) )
792 {
793 ReportMove( track.GetPosition(), fTransportEndPosition,
794 "End of Step Position" );
795 G4cerr << " Problem in G4CoupledTransportation::PostStepDoIt " << G4endl;
796 }
797
798 // If the Step was determined by the volume boundary, relocate the particle
799 // The pathFinder will know that the geometry limited the step (!?)
800
801 if( verboseLevel > 0 )
802 {
803 G4cout << " Calling PathFinder::Locate() from "
804 << " G4CoupledTransportation::PostStepDoIt " << G4endl;
805 G4cout << " fAnyGeometryLimitedStep is " << fAnyGeometryLimitedStep
806 << G4endl;
807
808 }
809#endif
810
811 if(fAnyGeometryLimitedStep)
812 {
813 fPathFinder->Locate( track.GetPosition(),
814 track.GetMomentumDirection(),
815 true);
816
817 // fCurrentTouchable will now become the previous touchable,
818 // and what was the previous will be freed.
819 // (Needed because the preStepPoint can point to the previous touchable)
820
821 fCurrentTouchableHandle=
822 fPathFinder->CreateTouchableHandle( fNavigatorId );
823
824#ifdef G4DEBUG_TRANSPORT
825 if( verboseLevel > 0 )
826 {
827 G4cout << "G4CoupledTransportation::PostStepDoIt --- fNavigatorId = "
828 << fNavigatorId << G4endl;
829 }
830 if( verboseLevel > 1 )
831 {
832 G4VPhysicalVolume* vol= fCurrentTouchableHandle->GetVolume();
833 G4cout << "CHECK !!!!!!!!!!! fCurrentTouchableHandle->GetVolume() = "
834 << vol;
835 if( vol ) { G4cout << "Name=" << vol->GetName(); }
836 G4cout << G4endl;
837 }
838#endif
839
840 // Check whether the particle is out of the world volume
841 // If so it has exited and must be killed.
842 //
843 if( fCurrentTouchableHandle->GetVolume() == 0 )
844 {
845 fParticleChange.ProposeTrackStatus( fStopAndKill ) ;
846 }
847 retCurrentTouchable = fCurrentTouchableHandle ;
848 // fParticleChange.SetTouchableHandle( fCurrentTouchableHandle ) ;
849 }
850 else // fAnyGeometryLimitedStep is false
851 {
852#ifdef G4DEBUG_TRANSPORT
853 if( verboseLevel > 1 )
854 {
855 G4cout << "G4CoupledTransportation::PostStepDoIt -- "
856 << " fAnyGeometryLimitedStep = " << fAnyGeometryLimitedStep
857 << " must be false " << G4endl;
858 }
859#endif
860 // This serves only to move each of the Navigator's location
861 //
862 // fLinearNavigator->LocateGlobalPointWithinVolume( track.GetPosition() ) ;
863
864 fPathFinder->ReLocate( track.GetPosition() );
865 // track.GetMomentumDirection() );
866
867 // Keep the value of the track's current Touchable is retained,
868 // and use it to overwrite the (unset) one in particle change.
869 // Expect this must be fCurrentTouchable too
870 // - could it be different, eg at the start of a step ?
871 //
872 retCurrentTouchable = track.GetTouchableHandle() ;
873 // fParticleChange.SetTouchableHandle( track.GetTouchableHandle() ) ;
874 } // endif ( fAnyGeometryLimitedStep )
875
876#ifdef G4DEBUG_NAVIGATION
877 G4cout << " CoupledTransport::AlongStep GPIL: "
878 << " last-step: any= " << fAnyGeometryLimitedStep
879 << " . ..... x . "
880 << " mass= " << fMassGeometryLimitedStep
881 << G4endl;
882#endif
883
884 if( fSignifyStepInAnyVolume )
885 fParticleChange.ProposeLastStepInVolume(fAnyGeometryLimitedStep);
886 else
887 fParticleChange.ProposeLastStepInVolume(fMassGeometryLimitedStep);
888
889 const G4VPhysicalVolume* pNewVol = retCurrentTouchable->GetVolume() ;
890 const G4Material* pNewMaterial = 0 ;
891 const G4VSensitiveDetector* pNewSensitiveDetector = 0 ;
892
893 if( pNewVol != 0 )
894 {
895 pNewMaterial= pNewVol->GetLogicalVolume()->GetMaterial();
896 pNewSensitiveDetector= pNewVol->GetLogicalVolume()->GetSensitiveDetector();
897 }
898
899 // ( const_cast<G4Material *> pNewMaterial ) ;
900 // ( const_cast<G4VSensitiveDetetor *> pNewSensitiveDetector) ;
901
902 fParticleChange.SetMaterialInTouchable( (G4Material *) pNewMaterial ) ;
903 fParticleChange.SetSensitiveDetectorInTouchable( (G4VSensitiveDetector *) pNewSensitiveDetector ) ;
904 // "temporarily" until Get/Set Material of ParticleChange,
905 // and StepPoint can be made const.
906
907 const G4MaterialCutsCouple* pNewMaterialCutsCouple = 0;
908 if( pNewVol != 0 )
909 {
910 pNewMaterialCutsCouple=pNewVol->GetLogicalVolume()->GetMaterialCutsCouple();
911 if( pNewMaterialCutsCouple!=0
912 && pNewMaterialCutsCouple->GetMaterial()!=pNewMaterial )
913 {
914 // for parametrized volume
915 //
916 pNewMaterialCutsCouple = G4ProductionCutsTable::GetProductionCutsTable()
917 ->GetMaterialCutsCouple(pNewMaterial,
918 pNewMaterialCutsCouple->GetProductionCuts());
919 }
920 }
921 fParticleChange.SetMaterialCutsCoupleInTouchable( pNewMaterialCutsCouple );
922
923 // Must always set the touchable in ParticleChange, whether relocated or not
924 //
925 fParticleChange.SetTouchableHandle(retCurrentTouchable) ;
926
927 return &fParticleChange ;
928}

1039 << " MeV " << G4endl
