118inline void G4Transportation::ResetKilledStatistics(G4int report)
119{
120 // Statistics for tracks killed (currently due to looping in field)
121
122 if( report )
123 {
124 G4cout << " G4Transportation: Statistics for looping particles "
125 << G4endl;
126 G4cout << " Sum of energy of loopers killed: " << fSumEnergyKilled
127 << G4endl;
128 G4cout << " Max energy of loopers killed: " << fMaxEnergyKilled
129 << G4endl;
130 }
131
132 fSumEnergyKilled= 0;
133 fMaxEnergyKilled= -1.0*CLHEP::GeV;
134}
