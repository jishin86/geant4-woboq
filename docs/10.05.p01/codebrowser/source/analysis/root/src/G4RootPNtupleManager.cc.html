63//_____________________________________________________________________________
64G4RootPNtupleDescription*
65G4RootPNtupleManager::GetNtupleDescriptionInFunction(
66 G4int id, G4String functionName, G4bool warn) const
67{
68 auto index = id - fFirstId;
69 if ( index < 0 || index >= G4int(fNtupleDescriptionVector.size()) ) {
70 if ( warn) {
71 G4String inFunction = "G4RootPNtupleManager::";
72 inFunction += functionName;
73 G4ExceptionDescription description;
74 description << " " << "ntuple " << id << " does not exist.";
75 G4Exception(inFunction, "Analysis_W011", JustWarning, description);
76 }
77 return nullptr;
78 }
79
80 return fNtupleDescriptionVector[index];
81}

83//_____________________________________________________________________________
84tools::wroot::base_pntuple*
85 G4RootPNtupleManager::GetNtupleInFunction(
86 G4int id, G4String functionName, G4bool warn) const
87{
88 auto ntupleDescription = GetNtupleDescriptionInFunction(id, functionName);
89 if ( ! ntupleDescription ) return nullptr;
90
91 if ( ! ntupleDescription->fBasePNtuple ) {
92 if ( warn ) {
93 G4String inFunction = "G4RootPNtupleManager::";
94 inFunction += functionName;
95 G4ExceptionDescription description;
96 description << " " << "ntupleId " << id << " does not exist.";
97 G4Exception(inFunction, "Analysis_W011", JustWarning, description);
98 }
99 return nullptr;
100 }
101 return ntupleDescription->fBasePNtuple;
102}

132//_____________________________________________________________________________
133void G4RootPNtupleManager::CreateNtuple(G4RootPNtupleDescription* ntupleDescription,
134 tools::wroot::ntuple* mainNtuple)
135{
136#ifdef G4VERBOSE
137 if ( fState.GetVerboseL4() )
138 fState.GetVerboseL4()
139 ->Message("create from main", "pntuple", mainNtuple->name());
140#endif
141
142 auto rfile = fMainNtupleManager->GetNtupleFile();
143 // auto directory = fMainNtupleManager->GetNtupleDirectory();
144
145 // Get parameters from main ntuple
146 auto mainBranch = mainNtuple->get_row_wise_branch();
147 auto rowWise = mainBranch ? true : false;
148
149 ntupleDescription->fFile = rfile.get();
150 mainNtuple->get_branches(ntupleDescription->fMainBranches);
151
152 G4bool verbose = true;
153 if ( rowWise ) {
154 tools::wroot::mt_ntuple_row_wise* mtNtuple
155 = new tools::wroot::mt_ntuple_row_wise(
156 G4cout, rfile->byte_swap(), rfile->compression(),
157 mainNtuple->dir().seek_directory(),
158 *mainBranch, mainBranch->basket_size(),
159 ntupleDescription->fNtupleBooking, verbose);
160
161 ntupleDescription->fNtuple
162 = static_cast<tools::wroot::imt_ntuple*>(mtNtuple);
163 ntupleDescription->fBasePNtuple
164 = static_cast<tools::wroot::base_pntuple*>(mtNtuple);
165 }
166 else {
167 std::vector<tools::uint32> basketSizes;
168 tools_vforcit(tools::wroot::branch*, ntupleDescription->fMainBranches, it) {
169 basketSizes.push_back((*it)->basket_size());
170 }
171
172 tools::wroot::mt_ntuple_column_wise* mtNtuple =
173 new tools::wroot::mt_ntuple_column_wise(
174 G4cout, rfile->byte_swap(), rfile->compression(),
175 mainNtuple->dir().seek_directory(),
176 ntupleDescription->fMainBranches, basketSizes,
177 ntupleDescription->fNtupleBooking, verbose);
178
179 ntupleDescription->fNtuple
180 = static_cast<tools::wroot::imt_ntuple*>(mtNtuple);
181 ntupleDescription->fBasePNtuple
182 = static_cast<tools::wroot::base_pntuple*>(mtNtuple);
183 }
184
185 ntupleDescription->fIsNtupleOwner = true;
186 // // pntuple object is not deleted automatically
187 fNtupleVector.push_back(ntupleDescription->fNtuple);
188
189
190#ifdef G4VERBOSE
191 if ( fState.GetVerboseL3() )
192 fState.GetVerboseL3()
193 ->Message("create from main", "pntuple", mainNtuple->name());
194#endif
195}

197//_____________________________________________________________________________
198void G4RootPNtupleManager::CreateNtuplesFromMain()
199{
200// Create ntuple from booking (if not yet done) and main ntuple
201// This function is called from G4AnalysisManager::OpenFile.
202
203 if ( fCreateMode == G4PNtupleCreateMode::kUndefined ) {
204 if ( fNtupleDescriptionVector.size() ) {
205 fCreateMode = G4PNtupleCreateMode::kSlaveBeforeOpen;
206 // G4cout << "Create mode: kSlaveBeforeOpen" << G4endl;
207 } else {
208 fCreateMode = G4PNtupleCreateMode::kSlaveAfterOpen;
209 // G4cout << "Create mode: kSlaveAfterOpen" << G4endl;
210 }
211 }
212
213 if ( fCreateMode == G4PNtupleCreateMode::kSlaveAfterOpen ) {
214 // ntuples are not yet booked
215 // G4cout << "Ntuples are not yet booked ?" << G4endl;
216 return;
217 }
218
219 auto& mainNtupleVector
220 = fMainNtupleManager->GetNtupleVector();
221
222 G4int lcounter = 0;
223 for ( auto mainNtuple : mainNtupleVector ) {
224
225 auto& ntupleDescription
226 = fNtupleDescriptionVector[lcounter++];
227 CreateNtuple(ntupleDescription, mainNtuple);
228 }
229}

231//_____________________________________________________________________________
232G4int G4RootPNtupleManager::CreateNtuple(
233 const G4String& name, const G4String& title)
234{
235// Create pntuple description with ntuple_booking
236
237#ifdef G4VERBOSE
238 if ( fState.GetVerboseL4() )
239 fState.GetVerboseL4()->Message("create", "pntuple booking", name);
240#endif
241
242 // Set create mode if not yet defined
243 if ( fCreateMode == G4PNtupleCreateMode::kUndefined ) {
244 if ( fMainNtupleManager->GetNtupleFile() ) {
245 fCreateMode = G4PNtupleCreateMode::kSlaveAfterOpen;
246 } else {
247 fCreateMode = G4PNtupleCreateMode::kSlaveBeforeOpen;
248 }
249 }
250
251 // Create ntuple description
252 auto index = fNtupleDescriptionVector.size();
253 auto ntupleDescription = new G4RootPNtupleDescription();
254 fNtupleDescriptionVector.push_back(ntupleDescription);
255
256 // Save name & title in ntuple booking
257 ntupleDescription->fNtupleBooking.set_name(name);
258 ntupleDescription->fNtupleBooking.set_title(title);
259
260 fLockFirstId = true;
261
262#ifdef G4VERBOSE
263 if ( fState.GetVerboseL2() ) {
264 G4ExceptionDescription description;
265 description << name << " ntupleId " << index + fFirstId;
266 fState.GetVerboseL2()->Message("create", "pntuple booking", description);
267 }
268#endif
269
270 return index + fFirstId;
271}

304{

307 // if ( fMainNtupleManager->GetNtupleFile() ) {

348//_____________________________________________________________________________
349G4bool G4RootPNtupleManager::AddNtupleRow(G4int ntupleId)
350{
351 if ( fState.GetIsActivation() && ( ! GetActivation(ntupleId) ) ) {
352 //G4cout << "Skipping AddNtupleRow for " << ntupleId << G4endl;
353 return false;
354 }
355
356#ifdef G4VERBOSE
357 if ( fState.GetVerboseL4() ) {
358 G4ExceptionDescription description;
359 description << " ntupleId " << ntupleId;
360 fState.GetVerboseL4()->Message("add", "pntuple row", description);
361 }
362#endif
363
364 auto ntupleDescription = GetNtupleDescriptionInFunction(ntupleId, "AddNtupleRow");
365 if ( ! ntupleDescription ) return false;
366
367 G4AutoLock lock(&pntupleMutex);
368 lock.unlock();
369 mutex toolsLock(lock);
370 auto result
371 = ntupleDescription->fNtuple
372 ->add_row(toolsLock, *ntupleDescription->fFile);
373
374 if ( ! result ) {
375 G4ExceptionDescription description;
376 description << " " << " ntupleId " << ntupleId
377 << "adding row has failed.";
378 G4Exception("G4RootPNtupleManager::AddNtupleRow()",
379 "Analysis_W002", JustWarning, description);
380 }
381
382#ifdef G4VERBOSE
383 if ( fState.GetVerboseL3() ) {
384 G4ExceptionDescription description;
385 description << " ntupleId " << ntupleId;
386 fState.GetVerboseL3()->Message("add", "pntuple row", description);
387 }
388#endif
389
390 return true;
391}

393//_____________________________________________________________________________
394G4bool G4RootPNtupleManager::Merge()
395{
396 for ( auto ntupleDescription : fNtupleDescriptionVector) {
397
398 // skip inactivated ntuples
399 if ( ! ntupleDescription->fActivation ) continue;
400
401#ifdef G4VERBOSE
402 if ( fState.GetVerboseL4() ) {
403 fState.GetVerboseL4()
404 ->Message("merge", "pntuple", ntupleDescription->fNtupleBooking.name());
405 }
406#endif
407
408 G4AutoLock lock(&pntupleMutex);
409 lock.unlock();
410 mutex toolsLock(lock);
411 auto result
412 = ntupleDescription->fNtuple
413 ->end_fill(toolsLock, *ntupleDescription->fFile);
414
415 if ( ! result ) {
416 G4ExceptionDescription description;
417 description << " " << " ntuple " << ntupleDescription->fNtupleBooking.name()
418 << "end fill has failed.";
419 G4Exception("G4RootPNtupleManager::Merge()",
420 "Analysis_W002", JustWarning, description);
421 }
422
423 delete ntupleDescription->fNtuple;
424 ntupleDescription->fNtuple = nullptr;
425
426#ifdef G4VERBOSE
427 if ( fState.GetVerboseL3() ) {
428 fState.GetVerboseL3()
429 ->Message("merge", "pntuple", ntupleDescription->fNtupleBooking.name());
430 }
431#endif
432 }
433 return true;
434
435}

437//_____________________________________________________________________________
438G4bool G4RootPNtupleManager::Reset(G4bool deleteNtuple)
439{
440 for ( auto ntupleDescription : fNtupleDescriptionVector ) {
441 if ( deleteNtuple ) {
442 delete ntupleDescription->fNtuple;
443 }
444 ntupleDescription->fNtuple = nullptr;
445 }
446
447 fNtupleVector.clear();
448
449 return true;
450}
