27// Manager class for CSV ntuples.
28//
29// Author: Ivana Hrivnacova, 18/06/2013 (ivana@ipno.in2p3.fr)
30
31#ifndef G4CsvNtupleManager_h
32#define G4CsvNtupleManager_h 1
33
34#include "G4TNtupleManager.hh"
35#include "globals.hh"
36
37#include "tools/wcsv_ntuple"
38
39#include <memory>
40
41class G4CsvFileManager;
42
43class G4CsvNtupleManager : public G4TNtupleManager<tools::wcsv::ntuple>
44{
45 friend class G4CsvAnalysisManager;
46
47 public:
48 explicit G4CsvNtupleManager(const G4AnalysisManagerState& state);
49 ~G4CsvNtupleManager();
50
51 private:
52 // Types alias
53 using NtupleType = tools::wcsv::ntuple;
54 using NtupleDescriptionType = G4TNtupleDescription<NtupleType>;
55
56 // Functions specific to the output type
57
58 // Set methods
59 void SetFileManager(std::shared_ptr<G4CsvFileManager> fileManager);
60
61 // Access to ntuple vector (needed for Write())
62 const std::vector<NtupleDescriptionType*>& GetNtupleDescriptionVector() const;
63
64 void SetIsCommentedHeader(G4bool isCommentedHeader);
65 void SetIsHippoHeader(G4bool isHippoHeader);
66
67 // Methods from the templated base class
68 //
69 virtual void CreateTNtuple(
70 NtupleDescriptionType* ntupleDescription,
71 const G4String& name, const G4String& title) final;
72 virtual void CreateTNtupleFromBooking(
73 NtupleDescriptionType* ntupleDescription) final;
74
75 virtual void FinishTNtuple(
76 NtupleDescriptionType* ntupleDescription,
77 G4bool fromBooking) final;
78
79 G4bool WriteHeader(NtupleType* ntuple) const;
80
81 // data members
82 std::shared_ptr<G4CsvFileManager> fFileManager;
83 G4bool fIsCommentedHeader;
84 G4bool fIsHippoHeader;
85};
86
87// inline functions
88
89inline void
90G4CsvNtupleManager::SetFileManager(std::shared_ptr<G4CsvFileManager> fileManager)
91{ fFileManager = fileManager; }
92
93inline const std::vector<G4TNtupleDescription<tools::wcsv::ntuple>*>&
94G4CsvNtupleManager::GetNtupleDescriptionVector() const
95{ return fNtupleDescriptionVector; }
96
97inline void G4CsvNtupleManager::SetIsCommentedHeader(G4bool isCommentedHeader)
98{ fIsCommentedHeader = isCommentedHeader; }
99
100inline void G4CsvNtupleManager::SetIsHippoHeader(G4bool isHippoHeader)
101{ fIsHippoHeader = isHippoHeader; }
102
103#endif
