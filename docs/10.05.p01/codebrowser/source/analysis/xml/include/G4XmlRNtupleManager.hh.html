32#ifndef G4XmlRNtupleManager_h
33#define G4XmlRNtupleManager_h 1
34
35#include "G4TRNtupleManager.hh"
36#include "globals.hh"
37
38#include "tools/raxml"
39
40#include <vector>
41
42struct G4XmlRNtupleDescription;
43
44class G4XmlRNtupleManager : public G4TRNtupleManager<tools::aida::ntuple>
45{
46 friend class G4XmlAnalysisReader;
47
48 protected:
49 G4XmlRNtupleManager(const G4AnalysisManagerState& state);
50 virtual ~G4XmlRNtupleManager();
51
52 // Methods from base classes
53 using G4BaseRNtupleManager::SetNtupleIColumn;
54 using G4BaseRNtupleManager::SetNtupleFColumn;
55 using G4BaseRNtupleManager::SetNtupleDColumn;
56 using G4TRNtupleManager<tools::aida::ntuple>::SetNtupleIColumn;
57 using G4TRNtupleManager<tools::aida::ntuple>::SetNtupleFColumn;
58 using G4TRNtupleManager<tools::aida::ntuple>::SetNtupleDColumn;
59
60 // Override base class functions for vector columns
61 // in order to fill the maps
62
63 virtual G4bool SetNtupleIColumn(G4int ntupleId, const G4String& columnName,
64 std::vector<G4int>& vector) final;
65 virtual G4bool SetNtupleFColumn(G4int ntupleId, const G4String& columnName,
66 std::vector<G4float>& vector) final;
67 virtual G4bool SetNtupleDColumn(G4int ntupleId, const G4String& columnName,
68 std::vector<G4double>& vector) final;
69
70 private:
71 // Methods from the templated base class
72 //
73 virtual G4bool GetTNtupleRow(G4TRNtupleDescription<tools::aida::ntuple>* ntupleDescription) final;
74};
