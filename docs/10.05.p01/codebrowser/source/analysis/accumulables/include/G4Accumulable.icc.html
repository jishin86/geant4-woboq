37//_____________________________________________________________________________
38template <typename T>
39G4Accumulable<T>::G4Accumulable(const G4String& name, T initValue, G4MergeMode mergeMode)
40 : G4VAccumulable(name),
41 fValue(initValue),
42 fInitValue(initValue),
43 fMergeMode(mergeMode),
44 fMergeFunction(GetMergeFunction<T>(mergeMode))
45{}
46
47//_____________________________________________________________________________
48template <typename T>

54 fMergeFunction(GetMergeFunction<T>(mergeMode))

72//_____________________________________________________________________________
73template <typename T>
74G4Accumulable<T>&
75G4Accumulable<T>::operator=(const G4Accumulable<T>& rhs)
76{
77 // check assignment to self
78 if (this == &rhs) return *this;
79
80 // base class assignment
81 G4VAccumulable::operator=(rhs);
82
83 // this class data assignment
84 fValue = rhs.fValue;
85 fInitValue = rhs.fInitValue;
86 fMergeMode = rhs.fMergeMode;
87 fMergeFunction = rhs.fMergeFunction;
88
89 return *this;
90}

146{

163//_____________________________________________________________________________
164template <typename T>
165void G4Accumulable<T>::Merge(const G4VAccumulable& other)
166{
167 // G4cout << "Merging other: " << other.GetName() << " " << static_cast<const G4Accumulable<T>&>(other).fValue << G4endl;
168 // G4cout << " to master: " << fName << " " << fValue << G4endl;
169 fValue = fMergeFunction(fValue, static_cast<const G4Accumulable<T>&>(other).fValue);
170 // G4cout << " new value: " << fName << " " << fValue << G4endl;
171}

173//_____________________________________________________________________________
174template <typename T>
175void G4Accumulable<T>::Reset()
176{
177 fValue = fInitValue;
178}

180//_____________________________________________________________________________
181template <typename T>
182T G4Accumulable<T>::GetValue() const
183{
184 return fValue;
