49G4H1Messenger::G4H1Messenger(G4VAnalysisManager* manager)
50 : G4UImessenger(),
51 fManager(manager),
52 fHelper(nullptr),
53 fDirectory(nullptr),
54 fCreateH1Cmd(nullptr),
55 fSetH1Cmd(nullptr),
56 fSetH1TitleCmd(nullptr),
57 fSetH1XAxisCmd(nullptr),
58 fSetH1YAxisCmd(nullptr)
59{
60 fHelper = G4Analysis::make_unique<G4AnalysisMessengerHelper>("h1");
61
62 fDirectory = fHelper->CreateHnDirectory();
63
64 CreateH1Cmd();
65
66 SetH1Cmd();
67 fSetH1XCmd = fHelper->CreateSetBinsCommand("x", this);
68
69
70 fSetH1TitleCmd = fHelper->CreateSetTitleCommand(this);
71 fSetH1XAxisCmd = fHelper->CreateSetAxisCommand("x", this);
72 fSetH1YAxisCmd = fHelper->CreateSetAxisCommand("y", this);
73}

84void G4H1Messenger::CreateH1Cmd()
85{
86 auto h1Name = new G4UIparameter("name", 's', false);
87 h1Name->SetGuidance("Histogram name (label)");
88
89 auto h1Title = new G4UIparameter("title", 's', false);
90 h1Title->SetGuidance("Histogram title");
91
92 auto h1Nbins0 = new G4UIparameter("nbins0", 'i', true);
93 h1Nbins0->SetGuidance("Number of bins (default = 100)");
94 h1Nbins0->SetGuidance("Can be reset with /analysis/h1/set command");
95 h1Nbins0->SetDefaultValue(100);
96
97 auto h1ValMin0 = new G4UIparameter("valMin0", 'd', true);
98 h1ValMin0->SetGuidance("Minimum value, expressed in unit (default = 0.)");
99 h1ValMin0->SetGuidance("Can be reset with /analysis/h1/set command");
100 h1ValMin0->SetDefaultValue(0.);
101
102 auto h1ValMax0 = new G4UIparameter("valMax0", 'd', true);
103 h1ValMax0->SetGuidance("Maximum value, expressed in unit (default = 1.)");
104 h1ValMax0->SetGuidance("Can be reset with /analysis/h1/set command");
105 h1ValMax0->SetDefaultValue(1.);
106
107 auto h1ValUnit0 = new G4UIparameter("valUnit0", 's', true);
108 h1ValUnit0->SetGuidance("The unit applied to filled values and valMin0, valMax0");
109 h1ValUnit0->SetDefaultValue("none");
110
111 auto h1ValFcn0 = new G4UIparameter("valFcn0", 's', true);
112 G4String fcnGuidance = "The function applied to filled values (log, log10, exp).\n";
113 fcnGuidance += "Note that the unit parameter cannot be omitted in this case,\n";
114 fcnGuidance += "but none value should be used instead.";
115 h1ValFcn0->SetGuidance(fcnGuidance);
116 h1ValFcn0->SetParameterCandidates("log log10 exp none");
117 h1ValFcn0->SetDefaultValue("none");
118
119 auto h1ValBinScheme0 = new G4UIparameter("valBinScheme0", 's', true);
120 G4String binSchemeGuidance = "The binning scheme (linear, log).\n";
121 h1ValBinScheme0->SetParameterCandidates("linear log");
122 binSchemeGuidance
123 += "Note that the unit and fcn parameters cannot be omitted in this case,\n";
124 binSchemeGuidance += "but none value should be used instead.";
125 h1ValBinScheme0->SetGuidance(binSchemeGuidance);
126 h1ValBinScheme0->SetDefaultValue("linear");
127
128 fCreateH1Cmd = G4Analysis::make_unique<G4UIcommand>("/analysis/h1/create", this);
129 fCreateH1Cmd->SetGuidance("Create 1D histogram");
130 fCreateH1Cmd->SetParameter(h1Name);
131 fCreateH1Cmd->SetParameter(h1Title);
132 fCreateH1Cmd->SetParameter(h1Nbins0);
133 fCreateH1Cmd->SetParameter(h1ValMin0);
134 fCreateH1Cmd->SetParameter(h1ValMax0);
135 fCreateH1Cmd->SetParameter(h1ValUnit0);
136 fCreateH1Cmd->SetParameter(h1ValFcn0);
137 fCreateH1Cmd->SetParameter(h1ValBinScheme0);
138 fCreateH1Cmd->AvailableForStates(G4State_PreInit, G4State_Idle);
139}

142//_____________________________________________________________________________
143void G4H1Messenger::SetH1Cmd()
144{
145 auto h1Id = new G4UIparameter("id", 'i', false);
146 h1Id->SetGuidance("Histogram id");
147 h1Id->SetParameterRange("id>=0");
148
149 auto h1Nbins = new G4UIparameter("nbins", 'i', false);
150 h1Nbins->SetGuidance("Number of bins");
151
152 auto h1ValMin = new G4UIparameter("valMin", 'd', false);
153 h1ValMin->SetGuidance("Minimum value, expressed in unit");
154
155 auto h1ValMax = new G4UIparameter("valMax", 'd', false);
156 h1ValMax->SetGuidance("Maximum value, expressed in unit");
157
158 auto h1ValUnit = new G4UIparameter("valUnit", 's', true);
159 h1ValUnit->SetGuidance("The unit applied to filled values and valMin, valMax");
160 h1ValUnit->SetDefaultValue("none");
161
162 auto h1ValFcn = new G4UIparameter("valFcn", 's', true);
163 h1ValFcn->SetParameterCandidates("log log10 exp none");
164 G4String fcnGuidance = "The function applied to filled values (log, log10, exp, none).\n";
165 fcnGuidance += "Note that the unit parameter cannot be omitted in this case,\n";
166 fcnGuidance += "but none value should be used instead.";
167 h1ValFcn->SetGuidance(fcnGuidance);
168 h1ValFcn->SetDefaultValue("none");
169
170 auto h1ValBinScheme = new G4UIparameter("valBinScheme", 's', true);
171 h1ValBinScheme->SetParameterCandidates("linear log");
172 G4String binSchemeGuidance = "The binning scheme (linear, log).\n";
173 binSchemeGuidance
174 += "Note that the unit and fcn parameters cannot be omitted in this case,\n";
175 binSchemeGuidance += "but none value should be used instead.";
176 h1ValBinScheme->SetGuidance(binSchemeGuidance);
177 h1ValBinScheme->SetDefaultValue("linear");
178
179 fSetH1Cmd = G4Analysis::make_unique<G4UIcommand>("/analysis/h1/set", this);
180 fSetH1Cmd->SetGuidance("Set parameters for the 1D histogram of given id:");
181 fSetH1Cmd->SetGuidance(" nbins; valMin; valMax; unit; function; binScheme");
182 fSetH1Cmd->SetParameter(h1Id);
183 fSetH1Cmd->SetParameter(h1Nbins);
184 fSetH1Cmd->SetParameter(h1ValMin);
185 fSetH1Cmd->SetParameter(h1ValMax);
186 fSetH1Cmd->SetParameter(h1ValUnit);
187 fSetH1Cmd->SetParameter(h1ValFcn);
188 fSetH1Cmd->SetParameter(h1ValBinScheme);
189 fSetH1Cmd->AvailableForStates(G4State_PreInit, G4State_Idle);
190}

197void G4H1Messenger::SetNewValue(G4UIcommand* command, G4String newValues)
198{
199 // tokenize parameters in a vector
200 std::vector<G4String> parameters;
201 G4Analysis::Tokenize(newValues, parameters);
202 // check consistency
203 if ( G4int(parameters.size()) != command->GetParameterEntries() ) {
204 // Should never happen but let's check anyway for consistency
205 fHelper->WarnAboutParameters(command, parameters.size());
206 return;
207 }
208
209 if ( command == fCreateH1Cmd.get() ) {
210 auto counter = 0;
211 auto name = parameters[counter++];
212 auto title = parameters[counter++];
213 G4AnalysisMessengerHelper::BinData xdata;
214 fHelper->GetBinData(xdata, parameters, counter);
215 auto unit = GetUnitValue(xdata.fSunit);
216 fManager->CreateH1(name, title,
217 xdata.fNbins, xdata.fVmin*unit, xdata.fVmax*unit,
218 xdata.fSunit, xdata.fSfcn, xdata.fSbinScheme);
219 }
220 else if ( command == fSetH1Cmd.get() ) {
221 auto counter = 0;
222 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
223 G4AnalysisMessengerHelper::BinData xdata;
224 fHelper->GetBinData(xdata, parameters, counter);
225 auto unit = GetUnitValue(xdata.fSunit);
226 fManager->SetH1(id,
227 xdata.fNbins, xdata.fVmin*unit, xdata.fVmax*unit,
228 xdata.fSunit, xdata.fSfcn, xdata.fSbinScheme);
229 }
230 else if ( command == fSetH1XCmd.get() ) {
231 auto counter = 0;
232 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
233 G4AnalysisMessengerHelper::BinData xdata;
234 fHelper->GetBinData(xdata, parameters, counter);
235 auto unit = GetUnitValue(xdata.fSunit);
236 fManager->SetH1(id,
237 xdata.fNbins, xdata.fVmin*unit, xdata.fVmax*unit,
238 xdata.fSunit, xdata.fSfcn, xdata.fSbinScheme);
239 }
240 else if ( command == fSetH1TitleCmd.get() ) {
241 auto counter = 0;
242 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
243 auto title = parameters[counter++];
244 fManager->SetH1Title(id, title);
245 }
246 else if ( command == fSetH1XAxisCmd.get() ) {
247 auto counter = 0;
248 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
249 auto xaxis = parameters[counter++];
250 fManager->SetH1XAxisTitle(id, xaxis);
251 }
252 else if ( command == fSetH1YAxisCmd.get() ) {
253 auto counter = 0;
254 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
255 auto yaxis = parameters[counter++];
256 fManager->SetH1YAxisTitle(id, yaxis);
257 }
