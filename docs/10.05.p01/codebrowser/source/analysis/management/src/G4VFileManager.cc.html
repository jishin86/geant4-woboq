54G4bool G4VFileManager::SetFileName(const G4String& fileName)
55{
56 if ( fLockFileName ) {
57 G4ExceptionDescription description;
58 description
59 << "Cannot set File name as its value was already used.";
60 G4Exception("G4VFileManager::SetFileName()",
61 "Analysis_W012", JustWarning, description);
62 return false;
63 }
64
65 return G4BaseFileManager::SetFileName(fileName);
66}

69G4bool G4VFileManager::SetHistoDirectoryName(const G4String& dirName)
70{
71 if ( fLockHistoDirectoryName ) {
72 G4ExceptionDescription description;
73 description
74 << "Cannot set Histo directory name as its value was already used.";
75 G4Exception("G4VFileManager::SetHistoDirectoryName()",
76 "Analysis_W012", JustWarning, description);
77 return false;
78 }
79
80 fHistoDirectoryName = dirName;
81 return true;
82}
