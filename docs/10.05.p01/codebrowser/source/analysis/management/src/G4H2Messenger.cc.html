42G4H2Messenger::G4H2Messenger(G4VAnalysisManager* manager)
43 : G4UImessenger(),
44 fManager(manager),
45 fHelper(nullptr),
46 fDirectory(nullptr),
47 fCreateH2Cmd(nullptr),
48 fSetH2Cmd(nullptr),
49 fSetH2XCmd(nullptr),
50 fSetH2YCmd(nullptr),
51 fSetH2TitleCmd(nullptr),
52 fSetH2XAxisCmd(nullptr),
53 fSetH2YAxisCmd(nullptr),
54 fXId(-1),
55 fXData()
56{
57 fHelper = G4Analysis::make_unique<G4AnalysisMessengerHelper>("h2");
58
59 fDirectory = fHelper->CreateHnDirectory();
60
61 CreateH2Cmd();
62
63 SetH2Cmd();
64 fSetH2XCmd = fHelper->CreateSetBinsCommand("x", this);
65 fSetH2YCmd = fHelper->CreateSetBinsCommand("y", this);
66
67 fSetH2TitleCmd = fHelper->CreateSetTitleCommand(this);
68 fSetH2XAxisCmd = fHelper->CreateSetAxisCommand("x", this);
69 fSetH2YAxisCmd = fHelper->CreateSetAxisCommand("y", this);
70 fSetH2ZAxisCmd = fHelper->CreateSetAxisCommand("z", this);
71}

82void G4H2Messenger::CreateH2Cmd()
83{
84 auto h2Name = new G4UIparameter("name", 's', false);
85 h2Name->SetGuidance("Histogram name (label)");
86
87 auto h2Title = new G4UIparameter("title", 's', false);
88 h2Title->SetGuidance("Histogram title");
89
90 auto h2xNbins0 = new G4UIparameter("xnbins0", 'i', true);
91 h2xNbins0->SetGuidance("Number of x-bins (default = 100)");
92 h2xNbins0->SetGuidance("Can be reset with /analysis/h2/set command");
93 h2xNbins0->SetDefaultValue(100);
94
95 auto h2xValMin0 = new G4UIparameter("xvalMin0", 'd', true);
96 h2xValMin0->SetGuidance("Minimum x-value, expressed in unit (default = 0.)");
97 h2xValMin0->SetGuidance("Can be reset with /analysis/h2/set command");
98 h2xValMin0->SetDefaultValue(0.);
99
100 auto h2xValMax0 = new G4UIparameter("xvalMax0", 'd', true);
101 h2xValMax0->SetGuidance("Maximum x-value, expressed in unit (default = 1.)");
102 h2xValMax0->SetGuidance("Can be reset with /analysis/h2/set command");
103 h2xValMax0->SetDefaultValue(1.);
104
105 auto h2xValUnit0 = new G4UIparameter("xvalUnit0", 's', true);
106 h2xValUnit0->SetGuidance("The unit applied to filled x-values and xvalMin0, xvalMax0");
107 h2xValUnit0->SetDefaultValue("none");
108
109 auto h2xValFcn0 = new G4UIparameter("xvalFcn0", 's', true);
110 G4String fcnxGuidance = "The function applied to filled x-values (log, log10, exp, none).";
111 h2xValFcn0->SetGuidance(fcnxGuidance);
112 h2xValFcn0->SetParameterCandidates("log log10 exp none");
113 h2xValFcn0->SetDefaultValue("none");
114
115 auto h2xValBinScheme0 = new G4UIparameter("xvalBinScheme0", 's', true);
116 G4String xbinSchemeGuidance = "The binning scheme (linear, log).";
117 h2xValBinScheme0->SetParameterCandidates("linear log");
118 h2xValBinScheme0->SetGuidance(xbinSchemeGuidance);
119 h2xValBinScheme0->SetDefaultValue("linear");
120
121 auto h2yNbins0 = new G4UIparameter("ynbins0", 'i', true);
122 h2yNbins0->SetGuidance("Number of y-bins (default = 100)");
123 h2yNbins0->SetGuidance("Can be reset with /analysis/h2/set command");
124 h2yNbins0->SetDefaultValue(100);
125
126 auto h2yValMin0 = new G4UIparameter("yvalMin0", 'd', true);
127 h2yValMin0->SetGuidance("Minimum y-value, expressed in unit (default = 0.)");
128 h2yValMin0->SetGuidance("Can be reset with /analysis/h2/set command");
129 h2yValMin0->SetDefaultValue(0.);
130
131 auto h2yValMax0 = new G4UIparameter("yvalMax0", 'd', true);
132 h2yValMax0->SetGuidance("Maximum y-value, expressed in unit (default = 1.)");
133 h2yValMax0->SetGuidance("Can be reset with /analysis/h2/set command");
134 h2yValMax0->SetDefaultValue(1.);
135
136 auto h2yValUnit0 = new G4UIparameter("yvalUnit0", 's', true);
137 h2yValUnit0->SetGuidance("The unit applied to filled y-values and yvalMin0, yvalMax0");
138 h2yValUnit0->SetDefaultValue("none");
139
140 auto h2yValFcn0 = new G4UIparameter("yvalFcn0", 's', true);
141 G4String fcnyGuidance = "The function applied to filled y-values (log, log10, exp, none).";
142 h2yValFcn0->SetGuidance(fcnyGuidance);
143 h2yValFcn0->SetParameterCandidates("log log10 exp none");
144 h2yValFcn0->SetDefaultValue("none");
145
146 auto h2yValBinScheme0 = new G4UIparameter("yvalBinScheme0", 's', true);
147 G4String ybinSchemeGuidance = "The binning scheme (linear, log).";
148 h2yValBinScheme0->SetParameterCandidates("linear log");
149 h2yValBinScheme0->SetGuidance(ybinSchemeGuidance);
150 h2yValBinScheme0->SetDefaultValue("linear");
151
152 fCreateH2Cmd = G4Analysis::make_unique<G4UIcommand>("/analysis/h2/create", this);
153 fCreateH2Cmd->SetGuidance("Create 2D histogram");
154 fCreateH2Cmd->SetParameter(h2Name);
155 fCreateH2Cmd->SetParameter(h2Title);
156 fCreateH2Cmd->SetParameter(h2xNbins0);
157 fCreateH2Cmd->SetParameter(h2xValMin0);
158 fCreateH2Cmd->SetParameter(h2xValMax0);
159 fCreateH2Cmd->SetParameter(h2xValUnit0);
160 fCreateH2Cmd->SetParameter(h2xValFcn0);
161 fCreateH2Cmd->SetParameter(h2xValBinScheme0);
162 fCreateH2Cmd->SetParameter(h2yNbins0);
163 fCreateH2Cmd->SetParameter(h2yValMin0);
164 fCreateH2Cmd->SetParameter(h2yValMax0);
165 fCreateH2Cmd->SetParameter(h2yValUnit0);
166 fCreateH2Cmd->SetParameter(h2yValFcn0);
167 fCreateH2Cmd->SetParameter(h2yValBinScheme0);
168 fCreateH2Cmd->AvailableForStates(G4State_PreInit, G4State_Idle);
169}

255void G4H2Messenger::SetNewValue(G4UIcommand* command, G4String newValues)
256{
257 // tokenize parameters in a vector
258 std::vector<G4String> parameters;
259 G4Analysis::Tokenize(newValues, parameters);
260 // check consistency
261 if ( G4int(parameters.size()) != command->GetParameterEntries() ) {
262 // Should never happen but let's check anyway for consistency
263 fHelper->WarnAboutParameters(command, parameters.size());
264 return;
265 }
266
267 if ( command == fCreateH2Cmd.get() ) {
268 auto counter = 0;
269 auto name = parameters[counter++];
270 auto title = parameters[counter++];
271 G4AnalysisMessengerHelper::BinData xdata;
272 fHelper->GetBinData(xdata, parameters, counter);
273 auto xunit = GetUnitValue(xdata.fSunit);
274 G4AnalysisMessengerHelper::BinData ydata;
275 fHelper->GetBinData(ydata, parameters, counter);
276 auto yunit = GetUnitValue(ydata.fSunit);
277 fManager->CreateH2(name, title,
278 xdata.fNbins, xdata.fVmin*xunit, xdata.fVmax*xunit,
279 ydata.fNbins, ydata.fVmin*yunit, ydata.fVmax*yunit,
280 xdata.fSunit, ydata.fSunit,
281 xdata.fSfcn, ydata.fSfcn,
282 xdata.fSbinScheme, ydata.fSbinScheme);
283 }
284 else if ( command == fSetH2Cmd.get() ) {
285 auto counter = 0;
286 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
287 G4AnalysisMessengerHelper::BinData xdata;
288 fHelper->GetBinData(xdata, parameters, counter);
289 auto xunit = GetUnitValue(xdata.fSunit);
290 G4AnalysisMessengerHelper::BinData ydata;
291 fHelper->GetBinData(ydata, parameters, counter);
292 auto yunit = GetUnitValue(ydata.fSunit);
293 fManager->SetH2(id,
294 xdata.fNbins, xdata.fVmin*xunit, xdata.fVmax*xunit,
295 ydata.fNbins, ydata.fVmin*yunit, ydata.fVmax*yunit,
296 xdata.fSunit, ydata.fSunit,
297 xdata.fSfcn, ydata.fSfcn,
298 xdata.fSbinScheme, ydata.fSbinScheme);
299 }
300 else if ( command == fSetH2XCmd.get() ) {
301 // Only save values
302 auto counter = 0;
303 fXId = G4UIcommand::ConvertToInt(parameters[counter++]);
304 fHelper->GetBinData(fXData, parameters, counter);
305 }
306 else if ( command == fSetH2YCmd.get() ) {
307 // Check if setX command was called
308 auto counter = 0;
309 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
310 if ( fXId == -1 || fXId != id ) {
311 fHelper->WarnAboutSetCommands();
312 return;
313 }
314 auto xunit = GetUnitValue(fXData.fSunit);
315 G4AnalysisMessengerHelper::BinData ydata;
316 fHelper->GetBinData(ydata, parameters, counter);
317 auto yunit = GetUnitValue(ydata.fSunit);
318 fManager->SetH2(id,
319 fXData.fNbins, fXData.fVmin*xunit, fXData.fVmax*xunit,
320 ydata.fNbins, ydata.fVmin*yunit, ydata.fVmax*yunit,
321 fXData.fSunit, ydata.fSunit,
322 fXData.fSfcn, ydata.fSfcn,
323 fXData.fSbinScheme, ydata.fSbinScheme);
324 fXId = -1;
325 }
326 else if ( command == fSetH2TitleCmd.get() ) {
327 auto counter = 0;
328 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
329 auto title = parameters[counter++];
330 fManager->SetH2Title(id, title);
331 }
332 else if ( command == fSetH2XAxisCmd.get() ) {
333 auto counter = 0;
334 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
335 auto xaxis = parameters[counter++];
336 fManager->SetH2XAxisTitle(id, xaxis);
337 }
338 else if ( command == fSetH2YAxisCmd.get() ) {
339 auto counter = 0;
340 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
341 auto yaxis = parameters[counter++];
342 fManager->SetH2YAxisTitle(id, yaxis);
343 }
344 else if ( command == fSetH2ZAxisCmd.get() ) {
345 auto counter = 0;
346 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
347 auto zaxis = parameters[counter++];
348 fManager->SetH2ZAxisTitle(id, zaxis);
349 }
