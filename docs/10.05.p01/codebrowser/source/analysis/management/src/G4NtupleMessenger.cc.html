41//_____________________________________________________________________________
42G4NtupleMessenger::G4NtupleMessenger(G4VAnalysisManager* manager)
43 : G4UImessenger(),
44 fManager(manager),
45 fSetActivationCmd(nullptr),
46 fSetActivationAllCmd(nullptr)
47{
48 fNtupleDir = G4Analysis::make_unique<G4UIdirectory>("/analysis/ntuple/");
49 fNtupleDir->SetGuidance("ntuple control");
50
51 SetActivationCmd();
52 SetActivationToAllCmd();
53}

57{
58}

64//_____________________________________________________________________________
65void G4NtupleMessenger::SetActivationCmd()
66{
67 auto ntupleId = new G4UIparameter("NtupleId", 'i', false);
68 ntupleId->SetGuidance("Ntuple id");
69 ntupleId->SetParameterRange("NtupleId>=0");
70
71 auto ntupleActivation = new G4UIparameter("NtupleActivation", 's', true);
72 ntupleActivation->SetGuidance("Ntuple activation");
73 ntupleActivation->SetDefaultValue("none");
74
75 fSetActivationCmd = G4Analysis::make_unique<G4UIcommand>("/analysis/ntuple/setActivation", this);
76 G4String guidance("Set activation for the ntuple of given id");
77
78 fSetActivationCmd->SetGuidance(guidance);
79 fSetActivationCmd->SetParameter(ntupleId);
80 fSetActivationCmd->SetParameter(ntupleActivation);
81 fSetActivationCmd->AvailableForStates(G4State_PreInit, G4State_Idle);
82}
83
84//_____________________________________________________________________________
85void G4NtupleMessenger::SetActivationToAllCmd()
86{
87 fSetActivationAllCmd
88 = G4Analysis::make_unique<G4UIcmdWithABool>("/analysis/ntuple/setActivationToAll", this);
89 G4String guidance("Set activation to all ntuples");
90 fSetActivationAllCmd->SetGuidance(guidance);
91 fSetActivationAllCmd->SetParameterName("AllNtupleActivation",false);
92}

98//_____________________________________________________________________________
99void G4NtupleMessenger::SetNewValue(G4UIcommand* command, G4String newValues)
100{
101 if ( command == fSetActivationCmd.get() ) {
102 // tokenize parameters in a vector
103 std::vector<G4String> parameters;
104 G4Analysis::Tokenize(newValues, parameters);
105 // check consistency
106 if ( G4int(parameters.size()) == command->GetParameterEntries() ) {
107 auto counter = 0;
108 auto id = G4UIcommand::ConvertToInt(parameters[counter++]);
109 auto activation = G4UIcommand::ConvertToBool(parameters[counter++]);
110 fManager->SetNtupleActivation(id, activation);
111 }
112 else {
113 // Should never happen but let's check anyway for consistency
114 G4ExceptionDescription description;
115 description
116 << "Got wrong number of \"" << command->GetCommandName()
117 << "\" parameters: " << parameters.size()
118 << " instead of " << command->GetParameterEntries()
119 << " expected" << G4endl;
120 G4Exception("G4NtupleMessenger::SetNewValue",
121 "Analysis_W013", JustWarning, description);
122 }
123 }
124 else if ( command == fSetActivationAllCmd.get() ) {
125 auto activation = fSetActivationAllCmd->GetNewBoolValue(newValues);
126 fManager->SetNtupleActivation(activation);
127 }
128}
