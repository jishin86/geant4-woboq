27// The helper class for histogram and profiles messengers.
28// It implements reusable commands in /analysis/[hn|pn] directory.
29//
30// Author: Ivana Hrivnacova, 05/05/2015 (ivana@ipno.in2p3.fr)
31
32#ifndef G4AnalysisMessengerHelper_h
33#define G4AnalysisMessengerHelper_h 1
34
35#include "G4UImessenger.hh"
36#include "globals.hh"
37
38#include <memory>
39
40class G4VAnalysisManager;
41class G4UIdirectory;
42class G4UIcommand;
43
44class G4AnalysisMessengerHelper
45{
46 public:
47 // types
48 struct BinData {
49 BinData() :
50 fNbins(0),
51 fVmin(0.),
52 fVmax(0.),
53 fSunit(""),
54 fSfcn(""),
55 fSbinScheme("") {}
56 G4int fNbins;
57 G4double fVmin;
58 G4double fVmax;
59 G4String fSunit;
60 G4String fSfcn;
61 G4String fSbinScheme;
62 };
63 // types
64 struct ValueData {
65 ValueData() :
66 fVmin(0.),
67 fVmax(0.),
68 fSunit(""),
69 fSfcn("") {}
70 G4double fVmin;
71 G4double fVmax;
72 G4String fSunit;
73 G4String fSfcn;
74 };
75
76 public:
77 // Make available utility Update method
78 friend class G4HnMessenger;
79
80 public:
81 explicit G4AnalysisMessengerHelper(const G4String& hnType);
82 ~G4AnalysisMessengerHelper();
83
84 // methods to create commands
85 std::unique_ptr<G4UIdirectory> CreateHnDirectory() const;
86
87 std::unique_ptr<G4UIcommand> CreateSetTitleCommand(
88 G4UImessenger* messenger) const;
89 std::unique_ptr<G4UIcommand> CreateSetBinsCommand(const G4String& axis,
90 G4UImessenger* messenger) const;
91 std::unique_ptr<G4UIcommand> CreateSetValuesCommand(const G4String& axis,
92 G4UImessenger* messenger) const;
93 std::unique_ptr<G4UIcommand> CreateSetAxisCommand(const G4String& axis,
94 G4UImessenger* messenger) const;
95
96 // methods to read command paremeters
97 void GetBinData(BinData& data, std::vector<G4String>& parameters,
98 G4int& counter) const;
99 void GetValueData(ValueData& data, std::vector<G4String>& parameters,
100 G4int& counter) const;
101
102 // warnings
103 void WarnAboutParameters(G4UIcommand* command, G4int nofParameters) const;
104 void WarnAboutSetCommands() const;
105
106 private:
107 // methods
108 G4String Update(const G4String& str, const G4String& axis = "") const;
109
110 // data members
111 G4String fHnType;
112};
113
114#endif
