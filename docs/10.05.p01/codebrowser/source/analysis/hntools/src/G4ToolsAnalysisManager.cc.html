39G4ToolsAnalysisManager::G4ToolsAnalysisManager(const G4String& type, G4bool isMaster)
40 : G4VAnalysisManager(type, isMaster),
41 fH1Manager(nullptr),
42 fH2Manager(nullptr),
43 fH3Manager(nullptr),
44 fP1Manager(nullptr),
45 fP2Manager(nullptr)
46{
47 // Create managers
48 fH1Manager = new G4H1ToolsManager(fState);
49 fH2Manager = new G4H2ToolsManager(fState);
50 fH3Manager = new G4H3ToolsManager(fState);
51 fP1Manager = new G4P1ToolsManager(fState);
52 fP2Manager = new G4P2ToolsManager(fState);
53 // The managers will be deleted by the base class
54
55 // Set managers to base class which takes then their ownership
56 SetH1Manager(fH1Manager);
57 SetH2Manager(fH2Manager);
58 SetH3Manager(fH3Manager);
59 SetP1Manager(fP1Manager);
60 SetP2Manager(fP2Manager);
61
62 // Plot manager
63 SetPlotManager(std::make_shared<G4PlotManager>(fState));
64}

75G4bool G4ToolsAnalysisManager::PlotImpl()
76{
77
78 // Only master thread performs plotting
79 if ( G4Threading::IsWorkerThread() ) return true;
80
81 auto finalResult = true;
82
83 // Open output file
84 fPlotManager->OpenFile(fVFileManager->GetPlotFileName());
85
86 // H1
87 auto result
88 = fPlotManager->PlotAndWrite<tools::histo::h1d>(fH1Manager->GetH1Vector(),
89 fH1Manager->GetHnVector());
90 finalResult = finalResult && result;
91
92 // H2
93 result
94 = fPlotManager->PlotAndWrite<tools::histo::h2d>(fH2Manager->GetH2Vector(),
95 fH2Manager->GetHnVector());
96 finalResult = finalResult && result;
97
98 // H3
99 // not yet available in tools
100
101 // P1
102 result
103 = fPlotManager->PlotAndWrite<tools::histo::p1d>(fP1Manager->GetP1Vector(),
104 fP1Manager->GetHnVector());
105 finalResult = finalResult && result;
106
107 // P2
108 // not yet available in tools
109
110 // Close output file
111 result = fPlotManager->CloseFile();
112 finalResult = finalResult && result;
113
114 return finalResult;
115}

117//_____________________________________________________________________________
118G4bool G4ToolsAnalysisManager::MergeImpl(tools::histo::hmpi* hmpi)
119{
120
121 // if ( G4Threading::IsWorkerThread() ) return true;
122
123 if ( ! hmpi ) return false;
124
125 G4bool finalResult = true;
126
127 // Create MPI manager
128 G4MPIToolsManager mpiToolsManager(fState, hmpi);
129
130 // H1
131 G4bool result
132 = mpiToolsManager.Merge<tools::histo::h1d>(fH1Manager->GetH1Vector(),
133 fH1Manager->GetHnVector());
134 finalResult = finalResult && result;
135
136 // H2
137 result
138 = mpiToolsManager.Merge<tools::histo::h2d>(fH2Manager->GetH2Vector(),
139 fH2Manager->GetHnVector());
140 finalResult = finalResult && result;
141
142 // H3
143 result
144 = mpiToolsManager.Merge<tools::histo::h3d>(fH3Manager->GetH3Vector(),
145 fH3Manager->GetHnVector());
146 finalResult = finalResult && result;
147
148 // P1
149 result
150 = mpiToolsManager.Merge<tools::histo::p1d>(fP1Manager->GetP1Vector(),
151 fP1Manager->GetHnVector());
152 finalResult = finalResult && result;
153
154 // P2
155 result
156 = mpiToolsManager.Merge<tools::histo::p2d>(fP2Manager->GetP2Vector(),
157 fP2Manager->GetHnVector());
158 finalResult = finalResult && result;
159
160 return finalResult;
161}

163//_____________________________________________________________________________
164G4bool G4ToolsAnalysisManager::Reset()
165{
166// Reset histograms and profiles
167
168 auto finalResult = true;
169
170 auto result = fH1Manager->Reset();
171 finalResult = finalResult && result;
172
173 result = fH2Manager->Reset();
174 finalResult = finalResult && result;
175
176 result = fH3Manager->Reset();
177 finalResult = finalResult && result;
178
179 result = fP1Manager->Reset();
180 finalResult = finalResult && result;
181
182 result = fP2Manager->Reset();
183 finalResult = finalResult && result;
184
185 return finalResult;
186}
