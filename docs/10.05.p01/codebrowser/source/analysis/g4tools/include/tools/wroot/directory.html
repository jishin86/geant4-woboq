89 directory* a_parent, //assume a_parent not nul.

91 const std::string& a_title)
92 :m_file(a_file)
93 ,m_parent(a_parent)
94 ,m_is_valid(false)
95 ,m_name(a_name)
96 ,m_title(a_title)
97 ,m_nbytes_keys(0)
98 ,m_nbytes_name(0)
99 ,m_seek_directory(0)
100 ,m_seek_parent(0)
101 ,m_seek_keys(0)
102 {
103#ifdef TOOLS_MEM
104 mem::increment(s_class().c_str());
105#endif
106 m_date_C = get_date();
107 m_date_M = get_date();
108
109 if(m_name.empty()) {
110 m_file.out() << "tools::wroot::directory::directory :"
111 << " directory name cannot be \"\"."
112 << std::endl;
113 return; //FIXME : throw
114 }
115 if(m_name.find('/')!=std::string::npos) {
116 m_file.out() << "tools::wroot::directory::directory :"
117 << " directory name " << sout(m_name)
118 << " cannot contain a slash."
119 << std::endl;
120 return; //FIXME : throw
121 }
122
123 if(m_title.empty()) m_title = m_name;
124
125 if(m_parent->find_key(m_name)) {
126 m_file.out() << "tools::wroot::directory::directory :"
127 << " directory " << sout(m_name) << " exists already."
128 << std::endl;
129 return; //FIXME : throw
130 }
131
132 m_seek_parent = m_parent->seek_directory();
133 uint32 nbytes = record_size();
134
135 wroot::key* key = new wroot::key(m_file.out(),m_file,m_parent->seek_directory(),
136 m_name,m_title,"TDirectory",nbytes); // It does a m_file.set_END().
137 m_nbytes_name = key->key_length();
138 m_seek_directory = key->seek_key(); //at EOF
139 if(!m_seek_directory) {
140 m_file.out() << "tools::wroot::directory::directory :"
141 << " bad key."
142 << std::endl;
143 delete key;
144 return; //FIXME : throw
145 }
146 {char* buffer = key->data_buffer();
147 wbuf wb(m_file.out(),m_file.byte_swap(),key->eob(),buffer);
148 if(!to_buffer(wb)) {
149 m_file.out() << "tools::wroot::directory::directory :"
150 << " directory name " << sout(m_name)
151 << " cannot fill buffer."
152 << std::endl;
153 delete key;
154 return; //FIXME : throw
155 }}
156 uint16 cycle = m_parent->append_key(key);
157 key->set_cycle(cycle);
158 if(!key->write_self(m_file)) {
159 m_file.out() << "tools::wroot::directory::directory :"
160 << " key.write_self() failed."
161 << std::endl;
162 return; //FIXME : throw
163 }
164 uint32 n;
165 if(!key->write_file(m_file,n)) {
166 m_file.out() << "tools::wroot::directory::directory :"
167 << " directory name " << sout(m_name)
168 << " cannot write key to file."
169 << std::endl;
170 return; //FIXME : throw
171 }
172
173 m_is_valid = true;

201 directory* mkdir(const std::string& a_name,const std::string& a_title = ""){
202 // Create a sub-directory and return a pointer to the created directory.
203 // Note that the directory name cannot contain slashes.
204 if(a_name.empty()) {
205 m_file.out() << "tools::wroot::directory::mkdir :"
206 << " directory name cannot be \"\"."
207 << std::endl;
208 return 0;
209 }
210 if(a_name.find('/')!=std::string::npos) {
211 m_file.out() << "tools::wroot::directory::mkdir :"
212 << " " << sout(a_name)
213 << " cannot contain a slash."
214 << std::endl;
215 return 0;
216 }
217 directory* dir =
218 new directory(m_file,this,a_name,a_title.empty()?a_name:a_title);
219 if(!dir->is_valid()) {
220 m_file.out() << "tools::wroot::directory::mkdir :"
221 << " directory badly created."
222 << std::endl;
223 delete dir;
224 return 0;
225 }
226 m_dirs.push_back(dir);
227 return dir;
228 }

233 uint32 record_size() const {
234 uint32 nbytes = sizeof(short);
235 nbytes += sizeof(date); //m_date_C.record_size();
236 nbytes += sizeof(date); //m_date_M.record_size();
237 nbytes += sizeof(m_nbytes_keys);
238 nbytes += sizeof(m_nbytes_name);
239 //ROOT version >= 40000:
240 nbytes += sizeof(seek);
241 nbytes += sizeof(seek);
242 nbytes += sizeof(seek);
243 return nbytes;

254 bool to_buffer(wbuf& a_wb){
255 // Decode input buffer.
256 // (Name, title) are stored in the (name, title) of the associated key.
257 short version = class_version();
258 version += big_file_version_tag(); //GB : enforce writing on seek (and not seek32).
259 if(!a_wb.write(version)) return false;
260 if(!a_wb.write(m_date_C)) return false;
261 if(!a_wb.write(m_date_M)) return false;
262 if(!a_wb.write(m_nbytes_keys)) return false;
263 if(!a_wb.write(m_nbytes_name)) return false;
264
265 if(!a_wb.write(m_seek_directory)) return false;
266 if(!a_wb.write(m_seek_parent)) return false;
267 if(!a_wb.write(m_seek_keys)) return false;
268
269 if(m_file.verbose()) {
270 m_file.out() << "tools::wroot::key::to_buffer :"
271 << " nbytes keys : " << m_nbytes_keys
272 << ", pos keys : " << m_seek_keys
273 << std::endl;
274 }
275 return true;
276 }

278 bool write(uint32& a_nbytes){
279 // Write all objects in memory to disk.
280 // Loop on all objects in memory (including subdirectories).
281 // A new key is created in the m_keys linked list for each object.
282 // For allowed options see TObject::Write().
283 // The directory header info is rewritten on the directory header record
284 a_nbytes = 0;
285 if(m_file.verbose()) {
286 m_file.out() << "tools::wroot::directory::write :"
287 << " " << sout(m_name)
288 << " : " << m_dirs.size()
289 << " : " << m_objs.size()
290 << " objects."
291 << std::endl;
292 }
293
294 uint32 nbytes = 0;
295
296 {tools_vforcit(directory*,m_dirs,it) {
297 uint32 n;
298 if(!(*it)->write(n)) return false;
299 nbytes += n;
300 }}
301
302 {tools_vforit(iobject*,m_objs,it) {
303 uint32 n;
304 if(!write_object(*(*it),n)) {
305 m_file.out() << "tools::wroot::directory::write :"
306 << " for directory " << sout(m_name)
307 << ", write_object " << sout((*it)->name())
308 << " failed."
309 << std::endl;
310 return false;
311 }
312 nbytes += n;
313 }}
314
315 if(!save_self()) {
316 m_file.out() << "tools::wroot::directory::write :"
317 << " for directory " << sout(m_name)
318 << ", save_self failed."
319 << std::endl;
320 return false; //it will write keys of objects.
321 }
322
323 a_nbytes = nbytes;
324 return true;
325 }

381 uint16 append_key(key* a_key){ //take ownership of a_key
382 tools_lforit(key*,m_keys,it) {
383 if((*it)->object_name()==a_key->object_name()) {
384 m_keys.insert(it,a_key); //a_key will be before *it.
385 return ((*it)->cycle() + 1);
386 }
387 }
388 // Not found :
389 m_keys.push_back(a_key);
390 return 1;

393 bool write_keys(){
394 // The list of keys (m_keys) is written as a single data record
395 // Delete the old keys structure if it exists
396
397 //if(fSeekKeys) {
398 // if(!fFile.makeFreeSegment
399 // (fSeekKeys, fSeekKeys + fNbytesKeys -1)) return false;
400 //}
401
402 // Write new keys record :
403 uint32 nkeys = uint32(m_keys.size());
404
405 // Compute size of all keys
406 uint32 nbytes = sizeof(nkeys);
407
408 {tools_lforit(key*,m_keys,it) {
409 nbytes += (*it)->key_length();
410 }}
411
412 key headerkey(m_file.out(),m_file,m_seek_directory,m_name,m_title,"TDirectory",nbytes); // m_file.set_END().
413 if(!headerkey.seek_key()) return false;
414
415 {char* buffer = headerkey.data_buffer();
416 wbuf wb(m_file.out(),m_file.byte_swap(),headerkey.eob(),buffer);
417 if(!wb.write(nkeys)) return false;
418 {tools_lforit(key*,m_keys,it) {
419 if(!((*it)->to_buffer(wb,m_file.verbose()))) return false;
420 }}}
421
422 m_seek_keys = headerkey.seek_key();
423 m_nbytes_keys = headerkey.number_of_bytes();
424
425 if(m_file.verbose()) {
426 m_file.out() << "tools::wroot::directory::write_keys :"
427 << " write header key"
428 << " " << sout(m_name)
429 << " " << sout(m_title)
430 << " (" << nkeys
431 << ", " << nbytes
432 << ", " << m_seek_keys
433 << ", " << m_nbytes_keys
434 << "):"
435 << std::endl;
436 }
437
438 headerkey.set_cycle(1);
439 if(!headerkey.write_self(m_file)) {
440 m_file.out() << "tools::wroot::directory::write_keys :"
441 << " key.write_self() failed."
442 << std::endl;
443 return false;
444 }
445
446 uint32 n;
447 return headerkey.write_file(m_file,n);

450 bool write_header(){
451 // Overwrite the Directory header record.
452 uint32 nbytes = record_size();
453 char* header = new char[nbytes];
454 char* buffer = header;
455 m_date_M = get_date();
456 wbuf wb(m_file.out(),m_file.byte_swap(),header+nbytes,buffer);
457 if(!to_buffer(wb)) {
458 delete [] header;
459 return false;
460 }
461 // do not overwrite the name/title part
462 seek pointer = m_seek_directory + m_nbytes_name;
463 //fModified = false;
464 if(!m_file.set_pos(pointer)) {
465 delete [] header;
466 return false;
467 }
468 if(!m_file.write_buffer(header,nbytes)) {
469 delete [] header;
470 return false;
471 }
472 if(!m_file.synchronize()) {
473 delete [] header;
474 return false;
475 }
476 delete [] header;
477 return true;
478 }

480 bool write_object(iobject& a_obj,uint32& a_nbytes){
481 buffer bref(m_file.out(),m_file.byte_swap(),256*128); //32768
482 if(!a_obj.stream(bref)) {
483 m_file.out() << "tools::wroot::directory::write_object :"
484 << " cannot stream object of store class name "
485 << " " << sout(a_obj.store_class_name()) << "."
486 << std::endl;
487 a_nbytes = 0;
488 return false;
489 }
490
491 std::string name = a_obj.name();
492 strip(name);
493
494 //first create the key to get key_length();
495
496 wroot::key* key = new wroot::key(m_file.out(),m_file,m_seek_directory,
497 name,
498 a_obj.title(),a_obj.store_class_name(),
499 bref.length()); // It does a m_file.set_END().
500
501 if(!key->seek_key()) {
502 delete key;
503 return false;
504 }
505
506 if(!bref.displace_mapped(key->key_length())) { //done before compression.
507 delete key;
508 return false;
509 }
510
511 char* kbuf = 0;
512 uint32 klen = 0;
513 bool kdelete = false;
514 m_file.compress_buffer(bref,kbuf,klen,kdelete);
515
516 ::memcpy(key->data_buffer(),kbuf,klen);
517 if(kdelete) delete [] kbuf;
518
519 {uint32 nkey = key->key_length()+klen;
520 m_file.set_END(key->seek_key()+nkey);
521 key->set_number_of_bytes(nkey);}
522
523 uint16 cycle = append_key(key);
524 key->set_cycle(cycle);
525
526 if(!key->write_self(m_file)) {
527 m_file.out() << "tools::wroot::directory::write_object :"
528 << " key.write_self() failed."
529 << std::endl;
530 return false;
531 }
532
533 //FIXME m_file.sumBuffer(key->object_size()); //uncompressed data size.
534
535 if(m_file.verbose()) {
536 m_file.out() << "tools::wroot::directory::_write_buffer :"
537 << " " << sout(a_obj.name()) << "."
538 << std::endl;
539 }
540
541 return key->write_file(m_file,a_nbytes);
542 }
