11inline void suffix(const std::string& a_string,std::string& a_value,bool a_back = true) {
12 // If a_string = dir0/dir1/dir2/dir3/name.xxx
13 // a_value = xxx
14 std::string::size_type pos = a_back?a_string.rfind('.'):a_string.find('.');
15 if(pos==std::string::npos) {a_value.clear();return;}
16 pos++;
17 a_value = a_string.substr(pos,a_string.size()-pos);
18}

20inline void nosuffix(const std::string& a_string,std::string& a_value,bool a_back = true){
21 // If a_string = dir0/dir1/dir2/dir3/name.xxx
22 // a_value = name
23 // Start searching after the last / (or last \ for Windows).
24 std::string::size_type pos = a_string.rfind('/');
25 if(pos==std::string::npos) pos = a_string.rfind('\\');
26 if(pos==std::string::npos) pos = 0;
27 else pos++;
28 std::string s = a_string.substr(pos,a_string.size()-pos);
29 std::string::size_type dot_pos = a_back?s.rfind('.'):s.find('.');
30 if(dot_pos==std::string::npos) {
31 a_value = s;
32 } else {
33 a_value = s.substr(0,dot_pos);
34 }
35}

37inline void base_name(const std::string& a_path,std::string& a_value) {
38 std::string::size_type pos_slash = a_path.rfind('/');
39 std::string::size_type pos_bslash = a_path.rfind('\\');
40 std::string::size_type pos = 0;
41 if(pos_slash==std::string::npos) {
42 if(pos_bslash==std::string::npos) {
43 pos = std::string::npos;
44 } else {
45 pos = pos_bslash;
46 }
47 } else {
48 if(pos_bslash==std::string::npos) {
49 pos = pos_slash;
50 } else {
51 if(pos_slash<=pos_bslash) {
52 pos = pos_bslash;
53 } else {
54 pos = pos_slash;
55 }
56 }
57 }
58 if(pos==std::string::npos) {a_value = a_path;return;}
59 pos++;
60 a_value = a_path.substr(pos,a_path.size()-pos);
61}

81inline bool is_absolute_path(const std::string& a_path) {
82 if(a_path.find('\\')!=std::string::npos) { //Windows path.
83 if(a_path.find(':')!=std::string::npos) return true;
84 return (a_path.size()&&(a_path[0]=='\\')?true:false);
85 } else { //UNIX path
86 return (a_path.size()&&(a_path[0]=='/')?true:false);
87 }
88}

90inline bool path_name_suffix(const std::string& a_string,std::string& a_path,std::string& a_name,std::string& a_suffix){
91 // If a_string = dir0/dir1/dir2/dir3/name.xxx
92 // a_path = dir0/dir1/dir2/dir3
93 // a_name = name.xxx
94 // a_suffix = xxx
95 // If a_string = dir0/name.xxx
96 // a_path = dir0
97 // a_name = name.xxx
98 // a_suffix = xxx
99 // If a_string = name.xxx
100 // a_path.clear()
101 // a_name = name.xxx
102 // a_suffix = xxx
103 // If a_string = /name.xxx
104 // a_path = "/"
105 // a_name = name.xxx
106 // a_suffix = xxx
107 // If a_string = .
108 // a_path = "."
109 // a_name.clear()
110 // a_suffix.clear()
111 // If a_string = ..
112 // a_path = ".."
113 // a_name.clear()
114 // a_suffix.clear()
115 if(a_string==".") {
116 a_path = ".";
117 a_name.clear();
118 a_suffix.clear();
119 return true;
120 } else if(a_string=="..") {
121 a_path = "..";
122 a_name.clear();
123 a_suffix.clear();
124 return true;
125 }
126
127 std::string::size_type pos_slash = a_string.rfind('/');
128 std::string::size_type pos_bslash = a_string.rfind('\\');
129 std::string::size_type pos = 0;
130 if(pos_slash==std::string::npos) {
131 if(pos_bslash==std::string::npos) {
132 pos = std::string::npos;
133 } else {
134 pos = pos_bslash;
135 }
136 } else {
137 if(pos_bslash==std::string::npos) {
138 pos = pos_slash;
139 } else {
140 if(pos_slash<=pos_bslash) {
141 pos = pos_bslash;
142 } else {
143 pos = pos_slash;
144 }
145 }
146 }
147
148 if(pos==std::string::npos) {
149 a_path.clear();
150 pos = 0;
151 } else if(pos==0) {
152 a_path = "/";
153 pos++;
154 } else {
155 a_path = a_string.substr(0,pos);
156 pos++;
157 }
158 std::string s = a_string.substr(pos,a_string.size()-pos);
159 pos = s.rfind('.');
160 if(pos==std::string::npos) {
161 a_name = s;
162 a_suffix.clear();
163 } else {
164 a_name = s;
165 pos++;
166 a_suffix = s.substr(pos,s.size()-pos);
167 }
168 return true;
169}

171inline std::string dir_name(const std::string& a_path,unsigned int a_num = 1){
172 std::string path = a_path;
173 for(unsigned int index=0;index<a_num;index++) {
174 std::string p,n,s;
175 path_name_suffix(path,p,n,s);
176 path = p;
177 }
178 return path;
179}

182inline bool is_f77(const std::string& a_path){
183 std::string sfx = suffix(a_path);
184
185 //tolowercase(sfx);
186 for(std::string::iterator it=sfx.begin();it!=sfx.end();++it) {
187 char c = *it;
188 *it = ((c) >= 'A' && (c) <= 'Z' ? c - 'A' + 'a' : c);
189 }
190
191 if(sfx=="f") return true; //the standard.
192 if(sfx=="for") return true; //for opaw. Known by g77.
193 if(sfx=="ftn") return true; //for opaw.
194 if(sfx=="fortran") return true; //for opaw.
195 if(sfx=="f77") return true;
196 return false;
197}
