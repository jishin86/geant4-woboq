14template <class TO>
15inline void* cmp_cast(const TO* a_this,const std::string& a_class) {
16 if(!rcmp(a_class,TO::s_class())) return 0;
17 return (void*)static_cast<const TO*>(a_this);
18}
19
20template <class TO>
21inline void* cmp_cast(const TO* a_this,cid a_id) {
22 if(TO::id_class()!=a_id) return 0;
23 return (void*)static_cast<const TO*>(a_this);
24}
