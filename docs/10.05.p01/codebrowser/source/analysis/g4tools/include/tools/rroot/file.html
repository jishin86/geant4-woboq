43 virtual bool set_pos(seek a_offset = 0,from a_from = begin){
44 int whence = 0;
45 switch(a_from) {
46 case begin:
47 whence = SEEK_SET;
48 break;
49 case current:
50 whence = SEEK_CUR;
51 break;
52 case end:
53 whence = SEEK_END;
54 break;
55 }
56
57#if defined(__linux__) && (__GLIBC__ == 2) && (__GLIBC_MINOR__ >= 2)
58 if (::lseek64(m_file, a_offset, whence) < 0) {
59#elif defined(_MSC_VER)
60 if (::_lseeki64(m_file, a_offset, whence) < 0) {
61#else
62 if (::lseek(m_file, a_offset, whence) < 0) {
63#endif
64 m_out << "tools::rroot::file::set_pos :"
65 << " cannot set position " << a_offset
66 << " in file " << sout(m_path) << "."
67 << std::endl;
68 return false;
69 }
70 return true;
71 }
72 virtual bool read_buffer(char* a_buffer,uint32 a_length) {
73 // Read a buffer from the file.
74 // This is the basic low level read operation.
75#ifdef _MSC_VER
76 typedef int ssize_t;
77#endif
78 ssize_t siz;
79 while ((siz = ::read(m_file,a_buffer,a_length)) < 0 &&
80 error_number() == EINTR) reset_error_number();
81 if (siz < 0) {
82 m_out << "tools::rroot::file::read_buffer :"
83 << " error reading from file " << sout(m_path) << "."
84 << std::endl;
85 return false;
86 }
87 if (siz != ssize_t(a_length)) {
88 m_out << "tools::rroot::file::read_buffer :"
89 << " error reading all requested bytes from file "
90 << sout(m_path) << ", got " << long_out(siz)
91 << " of " << a_length
92 << std::endl;
93 return false;
94 }
95 m_bytes_read += siz;
96 return true;
97 }
98 virtual bool unziper(char a_key,decompress_func& a_func) const {
99 std::map<char,decompress_func>::const_iterator it = m_unzipers.find(a_key);
100 if(it==m_unzipers.end()) {
101 a_func = 0;
102 return false;
103 }
104 a_func = (*it).second;
105 return true;
106 }

202 bool initialize() {
203 if(!read_header()) {
204 m_out << "tools::rroot::file::initialize :"
205 << " can't read header."
206 << std::endl;
207 return false;
208 }
209/*
210 fRootDirectory->setSeekDirectory(fBEGIN);
211 // Read Free segments structure if file is writable :
212 if (fWritable) {
213 if (fSeekFree > fBEGIN) {
214 if(!readFreeSegments()) {
215 m_out << "tools::rroot::file::initialize : Cannot read free segments."
216 << std::endl;
217 return false;
218 }
219 } else {
220 m_out << "tools::rroot::file::initialize : file \"" << fName
221 << "\" probably not closed, cannot read free segments" << std::endl;
222 }
223 }
224*/
225 // Read Directory info :
226 uint32 nbytes = m_nbytes_name + m_root_directory.record_size(m_version);
227 char* header = new char[nbytes];
228 char* buffer = header;
229 if(!set_pos(m_BEGIN)) {
230 m_out << "tools::rroot::file::initialize :"
231 << " can't set position."
232 << std::endl;
233 delete [] header;
234 return false;
235 }
236 if(!read_buffer(buffer,nbytes)) {
237 m_out << "tools::rroot::file::initialize :"
238 << " can't read buffer."
239 << std::endl;
240 delete [] header;
241 return false;
242 }
243 buffer = header+m_nbytes_name;
244 const char* eob = header+nbytes;
245 if(!m_root_directory.from_buffer(eob,buffer)) {
246 m_out << "tools::rroot::file::initialize :"
247 << " can't read buffer (2)."
248 << std::endl;
249 delete [] header;
250 return false;
251 }
252 uint32 nk = //size of Key
253 sizeof(int) + //Key::fNumberOfBytes
254 sizeof(short) + //Key::fVersion
255 2*sizeof(int) + //Key::fObjectSize, Date
256 2*sizeof(short) + //Key::fKeyLength,fCycle
257 2*sizeof(seek32); //Key::fSeekKey,fSeekParentDirectory
258 //WARNING : the upper is seek32 since at begin of file.
259 buffer = header+nk;
260 std::string cname;
261 rbuf rb(m_out,byte_swap(),eob,buffer);
262 // Should be "TFile".
263 if(!rb.read(cname)) {
264 m_out << "tools::rroot::file::initialize :"
265 << " can't read buffer (3)."
266 << std::endl;
267 delete [] header;
268 return false;
269 }
270 if(cname!="TFile") {
271 m_out << "tools::rroot::file::initialize : TFile expected." << std::endl;
272 delete [] header;
273 return false;
274 }
275 if(m_verbose) {
276 m_out << "tools::rroot::file::initialize :"
277 << " " << sout("TFile") << " found."
278 << std::endl;
279 }
280 if(!rb.read(cname)) {
281 m_out << "tools::rroot::file::initialize :"
282 << " can't read buffer (4)."
283 << std::endl;
284 delete [] header;
285 return false;
286 }
287 if(m_verbose) {
288 m_out << "tools::rroot::file::initialize :"
289 << " found file name " << sout(cname)
290 << std::endl;
291 }
292 if(!rb.read(m_title)) {
293 m_out << "tools::rroot::file::initialize :"
294 << " can't read buffer (5)."
295 << std::endl;
296 delete [] header;
297 return false;
298 }
299 delete [] header;
300 if(m_verbose) {
301 m_out << "tools::rroot::file::initialize :"
302 << " found title " << sout(m_title)
303 << std::endl;
304 }
305 uint32 dirNbytesName = m_root_directory.nbytes_name();
306 if (dirNbytesName < 10 || dirNbytesName > 1000) {
307 m_out << "tools::rroot::file::initialize :"
308 << " can't read directory info."
309 << std::endl;
310 return false;
311 }
312 // Read keys of the top directory :
313 if(m_root_directory.seek_keys() > m_BEGIN) {
314 uint32 n;
315 if(!m_root_directory.read_keys(n)) {
316 m_out << "tools::rroot::file::initialize :"
317 << " can't read keys."
318 << std::endl;

320 }
321 } else {
322 m_out << "tools::rroot::file::initialize :"
323 << " file " << sout(m_path)
324 << " probably not closed."
325 << std::endl;
326 return false;
327 }
328
329 // Create StreamerInfo index
330 if(m_seek_info > m_BEGIN) {
331 if(!read_streamer_infos()) {
332 m_out << "tools::rroot::file::initialize :"
333 << " read_streamer_infos() failed."

339 << " file " << sout(m_path)

341 << std::endl;
342 return false;
343 }
344
345 return true;

347 bool read_header() {
348 static const uint32 kBegin = 64;
349 char header[kBegin];
350 if(!set_pos()) return false;
351 if(!read_buffer(header,kBegin)) return false;
352 // make sure this is a root file
353 if(::strncmp(header, "root", 4)) {
354 m_out << "tools::rroot::file::read_header :"
355 << " " << sout(m_path) << " not a file at the CERN-ROOT format."
356 << std::endl;
357 return false;
358 }
359 if(m_verbose) {
360 m_out << "tools::rroot::file::read_header :"
361 << " file signature is " << sout("root")
362 << std::endl;
363 }
364 char* buffer = header + 4; // skip the "root" file identifier
365 const char* eob = header + kBegin;
366 rbuf rb(m_out,byte_swap(),eob,buffer);
367 {int v;
368 if(!rb.read(v)) return false;
369 m_version = v;}
370 {seek32 i;
371 if(!rb.read(i)) return false;
372 m_BEGIN = i;}
373 if(m_version>1000000) {
374 if(!rb.read(m_END)) return false;
375 if(!rb.read(m_seek_free)) return false;
376 } else {
377 {seek32 i;
378 if(!rb.read(i)) return false;
379 m_END = i;}
380 {seek32 i;
381 if(!rb.read(i)) return false;
382 m_seek_free = i;}
383 }
384 if(m_verbose) {
385 m_out << "tools::rroot::file::read_header :"
386 << " begin " << m_BEGIN
387 << " end " << m_END
388 << std::endl;
389 }
390 {int v;
391 if(!rb.read(v)) return false;
392 m_nbytes_free = v;}
393 int nfree = 0;
394 if(!rb.read(nfree)) return false;
395 {int v;
396 if(!rb.read(v)) return false;
397 m_nbytes_name = v;}
398 //m_out << "debug : 1002 " << m_nbytes_name << std::endl;
399 {char fUnits;
400 if(!rb.read(fUnits)) return false;}
401 {int fCompress;
402 if(!rb.read(fCompress)) return false;}
403 if(m_version>1000000) {
404 if(!rb.read(m_seek_info)) return false;
405 } else {
406 {seek32 i;
407 if(!rb.read(i)) return false;
408 m_seek_info = i;}
409 }
410 if(!rb.read(m_nbytes_info)) return false;
411 //m_out << "debug : seek_info " << m_seek_info << " nbytes_info " << m_nbytes_info << std::endl;
412 return true;
413 }

416 // Read the list of StreamerInfo from this file

418 // The corresponding TClass objects are updated.
419 if(m_seek_info<=0) return false;

426 delete [] buffer;
427 return false;

469//doc
470//
471// A ROOT file is a suite of consecutive data records with the following
472// format (see also the TKey class);
473// TKey ---------------------
474// byte 1->4 Nbytes = Length of compressed object (in bytes)
475// 5->6 Version = TKey version identifier
476// 7->10 ObjLen = Length of uncompressed object
477// 11->14 Datime = Date and time when object was written to file
478// 15->16 KeyLen = Length of the key structure (in bytes)
479// 17->18 Cycle = Cycle of key
480// 19->22 SeekKey = Pointer to record itself (consistency check)
481// 23->26 SeekPdir = Pointer to directory header
482// 27->27 lname = Number of bytes in the class name
483// 28->.. ClassName = Object Class Name
484// ..->.. lname = Number of bytes in the object name
485// ..->.. Name = lName bytes with the name of the object
486// ..->.. lTitle = Number of bytes in the object title
487// ..->.. Title = Title of the object
488// -----> DATA = Data bytes associated to the object
489//
490// The first data record starts at byte fBEGIN (currently set to kBegin)
491// Bytes 1->kBegin contain the file description:
492// byte 1->4 "root" = Root file identifier
493// 5->8 fVersion = File format version
494// 9->12 fBEGIN = Pointer to first data record
495// 13->16 fEND = Pointer to first free word at the EOF
496// 17->20 fSeekFree = Pointer to FREE data record
497// 21->24 fNbytesFree = Number of bytes in FREE data record
498// 25->28 nfree = Number of free data records
499// 29->32 fNbytesName = Number of bytes in TNamed at creation time
500// 33->33 fUnits = Number of bytes for file pointers
501// 34->37 fCompress = Zip compression level
502//
