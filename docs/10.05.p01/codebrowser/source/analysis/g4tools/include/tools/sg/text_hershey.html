106 std::vector<float> gsto_data;
107
108 {size_t npts = m_segs.size()/2; //2 coords

111 gsto_data.resize(sz+ngsto);

292 void get_segments(std::vector<float>& a_segs) const {
293
294 float Y = 0;
295 if( (vjust.value()==sg::middle) ||
296 (vjust.value()==sg::top) ){
297 float mn_x,mn_y,mn_z;
298 float mx_x,mx_y,mx_z;
299 get_bounds(height.value(),mn_x,mn_y,mn_z,mx_x,mx_y,mx_z);
300 float szy = mx_y - mn_y;
301
302 if(vjust.value()==sg::middle) {
303 Y -= 0.5F * szy;
304 } else if(vjust.value()==sg::top) {
305 Y -= szy;
306 }
307 }
308
309 float HEIGHT = height.value();
310 float Y_ADVANCE = 2 * HEIGHT;
311
312 const std::string& encod = encoding.value();
313
314 const std::vector<std::string>& ss = strings.values();
315 tools_vforcit(std::string,ss,it) {
316
317 float X = 0;
318 if( (hjust.value()==sg::center) ||
319 (hjust.value()==sg::right) ){
320 float mn_x,mn_y,mn_z;
321 float mx_x,mx_y,mx_z;
322 get_bounds(height,encoding.value(),font,*it,
323 mn_x,mn_y,mn_z,mx_x,mx_y,mx_z);
324 float szx = mx_x - mn_x;
325
326 if(hjust.value()==sg::center) {
327 X -= 0.5F * szx;
328 } else if(hjust.value()==sg::right) {
329 X -= szx;
330 }
331 }
332
333 string_segs(true,*it,HEIGHT,encod,font.value(),X,Y,a_segs,true);
334 Y -= Y_ADVANCE;
335 }
336 }
337protected:
338 static float _descent(float a_height) {
339 // '/' seems to be the char with the max descent.
340 float mn_x,mn_y,mn_z;
341 float mx_x,mx_y,mx_z;
342 get_char_bound('/',sg::latin,a_height,false,
343 mn_x,mn_y,mn_z,
344 mx_x,mx_y,mx_z);
345 return -mn_y; //return then a positive number.
346 }

348 static bool _truncate(const std::string& a_string,
349 float a_height,
350 font_type a_font,float a_cut_width,
351 std::string& a_out) {
352 //It does not take into account encoding.
353
354 a_out.clear();
355
356 float width = 0;
357
358 const unsigned int mx_poly = 4;
359 const unsigned int mx_point = 160;
360
361 int max_point[mx_poly];
362 float xp[mx_point];
363 float yp[mx_point];
364
365 tools_sforcit(a_string,it) {
366
367 float cwidth;
368 int number;
369 if (a_font==sg::greek) {
370 hershey::greek_char_points(*it,a_height,number,max_point,xp,yp,cwidth);
371 } else if (a_font==sg::special) {
372 hershey::special_char_points(*it,a_height,number,max_point,xp,yp,cwidth);
373 } else {
374 hershey::latin_char_points(*it,a_height,number,max_point,xp,yp,cwidth);
375 }
376
377 float advance = cwidth + a_height * 0.01F;
378
379 if((width+cwidth)>=a_cut_width) return true;
380 a_out += *it;
381 width += advance;
382 }
383
384 return true;
385 }

387 static void get_char_bound(char a_char,
388 font_type a_font,
389 float a_scale,bool a_bar,
390 float& a_mn_x,float& a_mn_y,float& a_mn_z,
391 float& a_mx_x,float& a_mx_y,float& a_mx_z){
392 box_3f_make_empty(a_mn_x,a_mn_y,a_mn_z,a_mx_x,a_mx_y,a_mx_z);
393
394 const unsigned int mx_poly = 4;
395 const unsigned int mx_point = 160;
396
397 int max_point[mx_poly];
398 float xp[mx_point];
399 float yp[mx_point];
400
401 int number;
402 float width;
403 if (a_font==sg::greek) {
404 hershey::greek_char_points(a_char,a_scale,number,max_point,xp,yp,width);
405 } else if (a_font==sg::special) {
406 hershey::special_char_points(a_char,a_scale,number,max_point,xp,yp,width);
407 } else {
408 hershey::latin_char_points(a_char,a_scale,number,max_point,xp,yp,width);
409 }
410
411 float ymax = 0;
412
413 int ipoint = 0;
414 for (int ipoly=0;ipoly<number;ipoly++) {
415 int pointn = max_point[ipoly];
416 if(pointn>0) {
417 for(int count=0;count<pointn-1;count++) {
418 ymax = mx<float>(ymax,yp[ipoint]);
419 box_3f_extend_by(a_mn_x,a_mn_y,a_mn_z,a_mx_x,a_mx_y,a_mx_z,xp[ipoint],yp[ipoint],0);
420
421 ymax = mx<float>(ymax,yp[ipoint+1]);
422 box_3f_extend_by(a_mn_x,a_mn_y,a_mn_z,a_mx_x,a_mx_y,a_mx_z,xp[ipoint+1],yp[ipoint+1],0);
423
424 ipoint ++;
425 }
426 ipoint ++;
427 }
428 }
429
430 if(a_bar) { //Draw a bar on top of the character.
431 float xbar = 0;
432 float ybar = ymax * 1.3F;
433 box_3f_extend_by(a_mn_x,a_mn_y,a_mn_z,a_mx_x,a_mx_y,a_mx_z,xbar,ybar,0);
434 box_3f_extend_by(a_mn_x,a_mn_y,a_mn_z,a_mx_x,a_mx_y,a_mx_z,xbar+width,ybar,0);
435 }

438 static void string_segs(
439 bool aGEN_POINTS // false = advance only.
440 ,const std::string& aString
441 ,float a_height
442 ,const std::string& a_encoding
443 ,font_type a_font
444 ,float& aX
445 ,float& aY
446 ,std::vector<float>& a_segs
447 ,bool a_fill_segs
448 ){
449 float oldX = 0;
450 float HEIGHT = a_height;
451
452 bool encod_PAW = (a_encoding==encoding_PAW()?true:false);
453
454 sencoded sed = encod_PAW ? decode_PAW(aString) : decode_plain(aString);
455
456 tools_vforcit(hchar,sed,it) {
457 const hchar& hc = *it;
458
459 font_type hershey_font = hc.m_font;
460 if(encod_PAW) {
461 hershey_font = hc.m_font;
462 } else {
463 hershey_font = a_font;
464 }
465
466 float scale = HEIGHT;
467 float ymove = 0;
468 if(hc.m_y_move==hchar::up) {
469 scale = HEIGHT*0.6F;
470 ymove = HEIGHT*0.6F;
471 } else if(hc.m_y_move==hchar::down) {
472 scale = HEIGHT*0.6F;
473 ymove = -HEIGHT*0.6F;
474 }
475 if(hc.m_back) aX = oldX;
476 oldX = aX;
477 //FIXME : bar
478 aY += ymove;
479
480 float advance = char_segs(aGEN_POINTS,hc.m_char,hershey_font,scale,hc.m_bar,aX,aY,a_segs,a_fill_segs) + HEIGHT * 0.01F;
481
482 aX += advance;
483
484 aY -= ymove;
485 }
486 }

488 static float char_segs(
489 bool aGEN_POINTS // false = advance only.
490 ,char a_char
491 ,font_type a_font
492 ,float a_scale
493 ,bool aBar
494 ,float aX
495 ,float aY
496 ,std::vector<float>& a_segs
497 ,bool a_fill_segs
498 ){
499 const unsigned int mx_poly = 8;
500 const unsigned int mx_point = 160;
501
502 int max_point[mx_poly];
503 float xp[mx_point];
504 float yp[mx_point];
505
506 int number;
507 float width;
508 if (a_font==sg::greek) {
509 hershey::greek_char_points(a_char,a_scale,number,max_point,xp,yp,width);
510 } else if (a_font==sg::special) {
511 hershey::special_char_points(a_char,a_scale,number,max_point,xp,yp,width);
512 } else {
513 hershey::latin_char_points(a_char,a_scale,number,max_point,xp,yp,width);
514 }
515 if(!aGEN_POINTS) return width;
516
517 float ymax = 0;
518
519 int ipoint = 0;
520 int pointn;
521 for (int ipoly=0;ipoly<number;ipoly++) {
522 pointn = max_point[ipoly];
523 if(pointn>0) {
524 for(int count=0;count<pointn-1;count++) {
525 ymax = mx<float>(ymax,yp[ipoint]);
526 if(a_fill_segs) {
527 a_segs.push_back(aX+xp[ipoint]);
528 a_segs.push_back(aY+yp[ipoint]);
529 }
530 ymax = mx<float>(ymax,yp[ipoint+1]);
531 if(a_fill_segs) {
532 a_segs.push_back(aX+xp[ipoint+1]);
533 a_segs.push_back(aY+yp[ipoint+1]);
534 }
535 ipoint ++;
536 }
537 ipoint ++;
538 }
539 }
540
541 if(aBar) { //Draw a bar on top of the character.
542 float xbar = 0;
543 float ybar = ymax * 1.3F;
544
545 if(a_fill_segs) {
546 a_segs.push_back(aX+xbar);
547 a_segs.push_back(aY+ybar);
548
549 a_segs.push_back(aX+xbar+width);
550 a_segs.push_back(aY+ybar);
551 }
552 }
553
554 return width;
555 }

559 static sencoded decode_plain(const std::string& a_s){
560 sencoded sed;
561 tools_sforcit(a_s,it) {
562 hchar hc;
563 hc.m_char = *it;
564 sed.push_back(hc);
565 }
566 if(sed.size()) sed[sed.size()-1].m_cr = true;
567 return sed;
568 }
569 ///////////////////////////////////////////////////////////////////////////
570 /// PAW encoding //////////////////////////////////////////////////////////
571 ///////////////////////////////////////////////////////////////////////////
572 // PAW control characters :
573 // [ go to greek (roman = default)
574 // ] end of greek
575 // " go to special
576 // # end of special
577 // ! go to normal level of script
578 // ^ go to superscript
579 // ? go to subscript
580 // & backscpace one charachter
581 // Extension :
582 // | draw a bar over one character
583 ///////////////////////////////////////////////////////////////////////////
584 static sencoded decode_PAW(const std::string& a_s){
585 sencoded sed;
586
587 font_type font = sg::latin;
588 hchar::e_move move = hchar::none;
589 bool back = false;
590 bool bar = false;
591
592 tools_sforcit(a_s,it) {
593 char c = *it;
594 // Control characters :
595 if(c=='[') {
596 font = sg::greek;
597 continue;
598 } else if(c==']') {
599 font = sg::latin;
600 continue;
601 } else if(c=='"') {
602 font = sg::special;
603 continue;
604 } else if(c=='#') {
605 font = sg::latin;
606 continue;
607 } else if(c=='!') {
608 move = hchar::none;
609 continue;
610 } else if(c=='^') {
611 move = hchar::up;
612 continue;
613 } else if(c=='?') {
614 move = hchar::down;
615 continue;
616 } else if(c=='&') {
617 back = true;
618 continue;
619 } else if(c=='|') {
620 bar = true;
621 continue;
622 }
623
624 hchar hc;
625 hc.m_y_move = move;
626 hc.m_back = back;
627 hc.m_bar = bar;
628
629 if(font==sg::special) {
630 hc.m_font = sg::latin;
631 //if(c=='A') {
632 // hc.m_char = '';
633 //} else
634 if(c=='B') {
635 hc.m_char = '|';
636 //} else if(c=='C') {
637 // hc.m_char = '';
638 } else if(c=='D') {
639 hc.m_char = '$';
640 } else if(c=='E') {
641 hc.m_char = '!';
642 } else if(c=='F') {
643 hc.m_char = '#';
644 } else if(c=='G') {
645 hc.m_char = '>';
646 } else if(c=='H') {
647 hc.m_char = '?';
648 //} else if(c=='I') {
649 // hc.m_char = '';
650 } else if(c=='J') {
651 hc.m_char = ':';
652 } else if(c=='K') {
653 hc.m_char = ';';
654 } else if(c=='L') {
655 hc.m_char = '<';
656 } else if(c=='M') {
657 hc.m_char = '[';
658 } else if(c=='N') {
659 hc.m_char = ']';
660
661 } else if(c=='W') {
662 hc.m_char = '&';
663 } else if(c=='Y') {
664 hc.m_char = '%';
665 } else {
666 //FIXME : handle other characters.
667 hc.m_char = c;
668 }
669
670 } else {
671 hc.m_font = font;
672 hc.m_char = c;
673 }
674
675
676 sed.push_back(hc);
677
678 back = false;
679 bar = false;
680 }
681
682 if(sed.size()) sed[sed.size()-1].m_cr = true;
683 return sed;
684 }

686protected:
687 std::vector<float> m_segs; //list of [begin,end]
688 size_t m_gsto_sz;
689};
