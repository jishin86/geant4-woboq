20template <class T>
21class mf : public bmf<T> {
22 typedef bmf<T> parent;
23public:
24 static const std::string& s_class() {
25 static const std::string s_v("tools::sg::mf<"+stype(T())+">");
26 return s_v;
27 }
28 virtual void* cast(const std::string& a_class) const {
29 if(void* p = cmp_cast< mf<T> >(this,a_class)) {return p;}
30 return parent::cast(a_class);
31 }
32 virtual const std::string& s_cls() const {return s_class();}
33public:
34 virtual bool write(io::iwbuf& a_buffer) {
35 const std::vector<T>& vec = parent::m_values;
36 return a_buffer.write_vec((uint32)vec.size(),vec_data(vec));
37 }
38 virtual bool read(io::irbuf& a_buffer) {
39 std::vector<T>& vec = parent::m_values;
40 return a_buffer.read_std_vec(vec);
41 }

43 const std::vector<T>& vec = parent::m_values;
44 a_out << "size : " << vec.size() << std::endl;
45 typedef typename std::vector<T>::const_iterator cit_t;
46 for(cit_t it=vec.begin();it!=vec.end();++it) {
47 a_out << " " << (*it) << std::endl;
48 }
49 return true;
50 }
51public:
52 mf(){}
53 mf(const T& a_v):parent(a_v){}
54 mf(const std::vector<T>& a_v):parent(a_v){}
55 virtual ~mf(){}
56public:

59 //typedef typename parent::iterator bmf_t;
60 parent::operator=(a_from);
61 return *this;
62 }
63public:

65 parent::operator=(a_from);
66 return *this;
67 }
68 mf& operator=(const T& a_v){
69 parent::operator=(a_v);

71 }

117template <class T>
118class mf_enum : public bmf<T> {
119 typedef bmf<T> parent;
120public:
121 static const std::string& s_class() {
122 static const std::string s_v("tools::sg::mf_enum");
123 return s_v;
124 }
125 virtual void* cast(const std::string& a_class) const {
126 if(void* p = cmp_cast< mf_enum<T> >(this,a_class)) {return p;}
127 return parent::cast(a_class);
128 }
129 virtual const std::string& s_cls() const {return s_class();}
130public:
131 virtual bool write(io::iwbuf& a_buffer) {
132 const std::vector<T>& vec = parent::m_values;
133 std::vector<int16> v; //an enum can be negative.
134 typedef typename std::vector<T>::const_iterator cit_t;
135 for(cit_t it=vec.begin();it!=vec.end();++it) v.push_back(*it);
136 return a_buffer.write_vec((uint32)v.size(),vec_data(v));
137 }
138 virtual bool read(io::irbuf& a_buffer) {
139 std::vector<int16> v; //an enum can be negative.
140 if(!a_buffer.read_std_vec(v)) return false;
141 std::vector<T>& vec = parent::m_values;
142 vec.clear();
143 std::vector<int16>::const_iterator it;
144 for(it=v.begin();it!=v.end();++it) vec.push_back((T)(*it));
145 return true;
146 }
147 virtual bool dump(std::ostream& a_out) {
148 const std::vector<T>& vec = parent::m_values;
149 a_out << "size : " << vec.size() << std::endl;
150 typedef typename std::vector<T>::const_iterator cit_t;
151 for(cit_t it=vec.begin();it!=vec.end();++it) {
152 a_out << " " << (*it) << std::endl;
153 }
154 return true;
155 }
156public:
157 mf_enum():parent(){}
158 mf_enum(const T& a_v):parent(a_v){}
159 mf_enum(const std::vector<T>& a_v):parent(a_v){}
160 virtual ~mf_enum(){}
161public:
162 mf_enum(const mf_enum& a_from):parent(a_from){}
163 mf_enum& operator=(const mf_enum& a_from){
164 parent::operator=(a_from);
165 return *this;
166 }
167};

193template <class T,class TT>
194class mf_vec : public bmf<T> {
195 typedef bmf<T> parent;
196public:
197 static const std::string& s_class() {
198 static const std::string s_v("tools::sg::mf_vec<"+stype(T())+","+stype(TT())+">");
199 return s_v;
200 }
201 virtual void* cast(const std::string& a_class) const {
202 if(void* p = cmp_cast< mf_vec<T,TT> >(this,a_class)) {return p;}
203 return parent::cast(a_class);
204 }
205 virtual const std::string& s_cls() const {return s_class();}
206public:
207 virtual bool write(io::iwbuf& a_buffer) {
208 const std::vector<T>& vec = parent::m_values;
209 typedef typename std::vector<TT> vec_t;
210 std::vector<vec_t> vec_vec;
211 typedef typename std::vector<T>::const_iterator cit_t;
212 for(cit_t it=vec.begin();it!=vec.end();++it) {
213
214 const T& v = (*it);
215 size_t num = v.size();
216 const TT* d = get_data(v);
217
218 std::vector<TT> std_vec(num);
219 for(size_t i=0;i<num;i++) std_vec[i] = d[i];
220
221 vec_vec.push_back(std_vec);
222 }
223 return a_buffer.write_std_vec_vec(vec_vec);
224 }
225 virtual bool read(io::irbuf& a_buffer) {
226 std::vector<T>& vec = parent::m_values;
227 vec.clear();
228 typedef typename std::vector<TT> vec_t;
229 std::vector<vec_t> vec_vec;
230 if(!a_buffer.read_std_vec_vec(vec_vec)) return false;
231 typedef typename std::vector<vec_t>::iterator _it_t;
232 for(_it_t it=vec_vec.begin();it!=vec_vec.end();++it) {
233 T x;
234 // x colorf, *it = std::vector<float>
235 // x vecs, *it = std::vector<std::string>
236 if(!set_from_vec(x,*it)) {vec.clear();return false;}
237 vec.push_back(x);
238 }
239 return true;
240 }
241 virtual bool dump(std::ostream& a_out) {
242 const std::vector<T>& vec = parent::m_values;
243 a_out << "size : " << vec.size() << std::endl;
244 typedef typename std::vector<T>::const_iterator cit_t;
245 for(cit_t it=vec.begin();it!=vec.end();++it) {
246 a_out << " " << (*it) << std::endl;
247 }
248 return true;
249 }
250public:
251 mf_vec():parent(){}
252 mf_vec(const T& a_v):parent(a_v){}
253 mf_vec(const std::vector<T>& a_v):parent(a_v){}
254 virtual ~mf_vec(){}
255public:
256 mf_vec(const mf_vec& a_from):parent(a_from){}
257 mf_vec& operator=(const mf_vec& a_from){
258 parent::operator=(a_from);
259 return *this;
260 }
261};
