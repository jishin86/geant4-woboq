92template <class T>
93class num_out : public std::string {
94 typedef std::string parent;
95private:
96 //typedef typename std::enable_if<std::is_floating_point<T>::value, T>::type type_t;
97 //typedef typename std::enable_if<std::is_integral<T>::value, T>::type type_t;
98 //typedef typename std::enable_if<std::is_arithmetic<T>::value, T>::type type_t;
99public:
100 num_out(const T& a_value) {
101 parent::operator+=("\"");
102 if(!numas(a_value,*this)) {} //throw
103 parent::operator+=("\"");
104 }
105public:
106 num_out(const num_out& a_from):parent(a_from){}
107 num_out& operator=(const num_out& a_from){parent::operator=(a_from);return *this;}
108};

116template <class VEC>
117inline bool nums2s(const VEC& a_vals,std::string& a_s,const std::string& a_sep = "\n",bool a_sep_at_end = false) {
118 a_s.clear();
119 typename VEC::size_type number = a_vals.size();
120 if(number<=0) return true; //it is ok.
121 number--;
122 std::string stmp;
123 bool status = true;
124 for(typename VEC::size_type index=0;index<number;index++) {
125 if(!num2s(a_vals[index],stmp)) status = false; //continue.
126 a_s += stmp;
127 a_s += a_sep;
128 }
129 if(!num2s(a_vals[number],stmp)) status = false;
130 a_s += stmp;
131 if(a_sep_at_end) a_s += a_sep;
132 return status;
133}
134
135template <class T>
136inline bool nums2s(const std::vector<T>& a_vals,std::string& a_s,const std::string& a_sep = "\n",bool a_sep_at_end = false) {
137 return nums2s< std::vector<T> >(a_vals,a_s,a_sep,a_sep_at_end);
138}
