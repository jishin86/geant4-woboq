4#ifndef tools_sg_draw_style
5#define tools_sg_draw_style
6
7#include "node"
8
9#include "lpat"
10#include "sf_enum"
11#include "render_action"
12
13namespace tools {
14namespace sg {
15
16class draw_style : public node {
17 TOOLS_NODE(draw_style,tools::sg::draw_style,node)
18public:
19 enum e_style {
20 filled,
21 lines,
22 points,
23 invisible
24 };
25
26 sf_enum<e_style> style;
27 sf<float> line_width;
28 sf<lpat> line_pattern;
29 sf<float> point_size;
30 sf<bool> cull_face;
31 sf<bool> winding_ccw;
32
33public:
34 virtual const std::vector<field_desc>& node_fields() const {
35 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::draw_style)
36 static std::vector<field_desc> s_v;
37 if(s_v.empty()) {
38 s_v = parent::node_fields();
39 TOOLS_ADD_FIELD_DESC(style)
40 TOOLS_ADD_FIELD_DESC(line_width)
41 TOOLS_ADD_FIELD_DESC(line_pattern)
42 TOOLS_ADD_FIELD_DESC(point_size)
43 TOOLS_ADD_FIELD_DESC(cull_face)
44 TOOLS_ADD_FIELD_DESC(winding_ccw)
45 }
46 return s_v;
47 }
48private:
49 void add_fields(){
50 add_field(&style);
51 add_field(&line_width);
52 add_field(&line_pattern);
53 add_field(&point_size);
54 add_field(&cull_face);
55 add_field(&winding_ccw);
56 }
57public:
58 virtual void render(render_action& a_action) {
59 state& state = a_action.state();
60 state.m_line_width = line_width;
61 state.m_line_pattern = line_pattern;
62 state.m_point_size = point_size;
63 state.m_GL_CULL_FACE = cull_face;
64 state.m_winding = winding_ccw.value()?sg::winding_ccw:sg::winding_cw;
65
66 if(style.value()==lines) {
67 //no LINE_STIPPLE in GLES.
68 //::glEnable(GL_LINE_STIPPLE); //done in viewer::render()
69 //::glLineStipple(1,line_pattern.value());
70 a_action.line_width(state.m_line_width);
71 } else if(style.value()==points) {
72 a_action.point_size(state.m_point_size);
73 } else if(style.value()==filled) {
74 a_action.set_cull_face(state.m_GL_CULL_FACE);
75 a_action.set_winding(state.m_winding);
76 }
77 }
78public:
79 draw_style()
80 :parent()
81 ,style(filled)
82 ,line_width(1) //NOTE : 0 induces a 501 gl error.
83 ,line_pattern(line_solid)
84 ,point_size(1)
85 ,cull_face(true)
86 ,winding_ccw(true)
87 {
88 add_fields();
89 }
90 virtual ~draw_style(){}
91public:
92 draw_style(const draw_style& a_from)
93 :parent(a_from)
94 ,style(a_from.style)
95 ,line_width(a_from.line_width)
96 ,line_pattern(a_from.line_pattern)
97 ,point_size(a_from.point_size)
98 ,cull_face(a_from.cull_face)
99 ,winding_ccw(a_from.winding_ccw)
100 {
101 add_fields();
102 }
103 draw_style& operator=(const draw_style& a_from){
104 parent::operator=(a_from);
105
106 style = a_from.style;
107 line_width = a_from.line_width;
108 line_pattern = a_from.line_pattern;
109 point_size = a_from.point_size;
110 cull_face = a_from.cull_face;
111 winding_ccw = a_from.winding_ccw;
112
113 return *this;
114 }
115};
116
117}}

119#endif
