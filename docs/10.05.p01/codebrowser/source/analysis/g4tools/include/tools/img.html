201 bool equal(const img& a_from) const {
202 if(m_w!=a_from.m_w) return false;
203 if(m_h!=a_from.m_h) return false;
204 if(m_n!=a_from.m_n) return false;
205 //don't test ownership.
206 unsigned int sz = m_w*m_h*m_n;
207 T* pos = m_buffer;
208 T* fpos = a_from.m_buffer;
209 for(unsigned int index=0;index<sz;index++,pos++,fpos++) {
210 if((*pos)!=(*fpos)) return false;
211 }
212 return true;

222 bool pixel(unsigned int a_i,unsigned a_j,std::vector<T>& a_pixel) const {
223 if((!m_w)||(!m_h)||(a_i>=m_w)||(a_j>=m_h)) {
224 a_pixel.clear();
225 return false;
226 }
227 a_pixel.resize(m_n);
228 T* pos = m_buffer + a_j * (m_w * m_n) + a_i*m_n;
229 for(unsigned int ipix=0;ipix<m_n;ipix++) {
230 a_pixel[ipix] = *(pos+ipix);
231 }
232 return true;

235 bool expand(unsigned int a_factor,img<T>& a_res,bool a_res_force_owner = true) const {
236 if(a_factor==1) {
237 if(a_res_force_owner) {
238 a_res.copy(m_w,m_h,m_n,m_buffer);
239 } else {
240 a_res.set(m_w,m_h,m_n,m_buffer,false);
241 }
242 return true;
243 }
244
245 unsigned int nw = m_w*a_factor;
246 unsigned int nh = m_h*a_factor;
247 unsigned int sz = nh*nw*m_n;
248 if(!sz) {
249 a_res.make_empty();
250 return false;
251 }
252
253 T* nb = new T[sz];
254 if(!nb) {
255 a_res.make_empty();
256 return false;
257 }
258
259 for(unsigned int j=0;j<m_h;j++) {
260 for(unsigned int i=0;i<m_w;i++) {
261 //position in the original image.
262 T* pos = m_buffer + j * (m_w * m_n) + i*m_n;
263
264 for(unsigned int fr=0;fr<a_factor;fr++) {
265 for(unsigned int fc=0;fc<a_factor;fc++) {
266 //position in the new image.
267 T* npos = nb + (j*a_factor+fr) * (nw * m_n) + (i*a_factor+fc)*m_n;
268 for(unsigned int ipix=0;ipix<m_n;ipix++) {
269 *(npos+ipix) = *(pos+ipix);
270 }
271 }
272 }
273
274 }
275 }
276
277 a_res.set(nw,nh,m_n,nb,true);
278 return true;
279 }

281 bool contract_raw(unsigned int a_w,unsigned int a_h,img<T>& a_res,bool a_force_res_owner = true) const {
282 if((a_w==m_w)&&(a_h==m_h)) {
283 if(a_force_res_owner) {
284 a_res.copy(m_w,m_h,m_n,m_buffer);
285 } else {
286 a_res.set(m_w,m_h,m_n,m_buffer,false);
287 }
288 return true;
289 }
290
291 unsigned int sz = a_h*a_w*m_n;
292 if(!sz) {
293 a_res.make_empty();
294 return false;
295 }
296
297 T* rb = new T[sz];
298 if(!rb) {
299 a_res.make_empty();
300 return false;
301 }
302
303 double* pixels = new double[m_n]; //for mean value.
304 if(!pixels) {
305 delete [] rb;
306 a_res.make_empty();
307 return false;
308 }
309
310 unsigned int wfac = double(m_w)/double(a_w);
311 unsigned int hfac = double(m_h)/double(a_h);
312 if(!wfac) wfac = 1;
313 if(!hfac) hfac = 1;
314
315 unsigned int wfac_hfac = wfac*hfac;
316
317 T* hpos;T* pos;
318 for(unsigned int j=0;j<a_h;j++) {
319 for(unsigned int i=0;i<a_w;i++) {
320
321 // take mean value of wfac*hfac pixels :
322 {for(unsigned int ipix=0;ipix<m_n;ipix++) pixels[ipix] = 0;}
323
324 for(unsigned int fr=0;fr<hfac;fr++) {
325 hpos = m_buffer + (j*hfac+fr)*(m_w*m_n);
326 for(unsigned int fc=0;fc<wfac;fc++) {
327 pos = hpos + (i*wfac+fc)*m_n;
328 for(unsigned int ipix=0;ipix<m_n;ipix++) {
329 pixels[ipix] += double(*pos)/double(wfac_hfac);pos++;
330 }
331 }
332 }
333
334 //position in the result image.
335 T* rpos = rb + j * (a_w * m_n) + i*m_n;
336 {for(unsigned int ipix=0;ipix<m_n;ipix++) {*rpos = T(pixels[ipix]);rpos++;}}
337 }
338 }
339
340 delete [] pixels;
341
342 a_res.set(a_w,a_h,m_n,rb,true);
343 return true;
344 }

346 bool contract(unsigned int a_w,unsigned int a_h,img<T>& a_res,bool a_force_res_owner = true) const {
347 //optimized version of contract_raw().
348
349 if((a_w==m_w)&&(a_h==m_h)) {
350 if(a_force_res_owner) {
351 a_res.copy(m_w,m_h,m_n,m_buffer);
352 } else {
353 a_res.set(m_w,m_h,m_n,m_buffer,false);
354 }
355 return true;
356 }
357
358 size_t sz = a_h*a_w*m_n;
359 if(!sz) {
360 a_res.make_empty();
361 return false;
362 }
363
364 T* rb = new T[sz];
365 if(!rb) {
366 a_res.make_empty();
367 return false;
368 }
369
370 double* pixels = new double[m_n]; //for mean value.
371 if(!pixels) {
372 delete [] rb;
373 a_res.make_empty();
374 return false;
375 }
376 {for(unsigned int ipix=0;ipix<m_n;ipix++) pixels[ipix] = 0;}
377
378 unsigned int wfac = (unsigned int)(double(m_w)/double(a_w));
379 unsigned int hfac = (unsigned int)(double(m_h)/double(a_h));
380 if(!wfac) wfac = 1;
381 if(!hfac) hfac = 1;
382
383 double wfac_hfac = wfac*hfac;
384
385 //::printf("debug : %d %d, %d %d\n",a_h,a_w,hfac,wfac);
386
387 T* hpos;T* pos;T* hrpos;T* rpos;T* hhpos;T* _pos;double* ppos;
388 unsigned int i,j,fr,fc,ipix,i0;
389 unsigned int astride = a_w * m_n;
390 unsigned int mstride = m_w * m_n;
391 unsigned int wfacstride = wfac * m_n;
392
393 for(j=0;j<a_h;j++) {
394 hrpos = rb + j * astride;
395 hhpos = m_buffer + j*hfac*mstride;
396 for(i=0;i<a_w;i++) {
397
398 // take mean value of wfac*hfac pixels :
399
400 i0 = i*wfacstride;
401
402 hpos = hhpos;
403 for(fr=0;fr<hfac;fr++,hpos+=mstride) {
404 _pos = hpos + i0;
405 for(fc=0;fc<wfac;fc++,_pos+=m_n) {
406 pos = _pos;
407 ppos = pixels;
408 for(ipix=0;ipix<m_n;ipix++,pos++,ppos++) {
409 *ppos += double(*pos)/wfac_hfac;
410// *ppos += double(*pos); //NOTE : doing the wfac_hfac division in the below loop is slower !
411 }
412 }
413 }
414
415 //position in the result image.
416 rpos = hrpos + i*m_n;
417 ppos = pixels;
418 for(ipix=0;ipix<m_n;ipix++,rpos++,ppos++) {
419 *rpos = T(*ppos);
420// *rpos = T((*ppos)/wfac_hfac); //slower !
421 *ppos = 0;
422 }
423 }
424 }
425
426 delete [] pixels;
427
428 a_res.set(a_w,a_h,m_n,rb,true);
429 return true;
430 }

439 template <class TTO>
440 bool convert(img<TTO>& a_res) const {
441 a_res.make_empty();
442
443 unsigned int sz = m_w*m_h*m_n;
444 if(!sz) return false;
445
446 TTO* _buffer = new TTO[sz];
447 if(!_buffer) return false;
448
449 unsigned int i,j,ipix,imn;
450 unsigned int mwn = m_w*m_n;
451 T* _pos;T* pos;
452 TTO* _rpos;TTO* rpos;
453
454 for(j=0;j<m_h;j++) {
455 _pos = m_buffer + j*mwn;
456 _rpos = _buffer + j*mwn;
457 for(i=0;i<m_w;i++) {
458 imn = i*m_n;
459 pos = _pos + imn;
460 rpos = _rpos + imn;
461 for(ipix=0;ipix<m_n;ipix++,pos++,rpos++) *rpos = *pos;
462 }
463 }
464
465 a_res.set(m_w,m_h,m_n,_buffer,true);
466 return true;
467 }

469 bool get_part(unsigned int a_sx,unsigned int a_sy,unsigned int a_sw,unsigned int a_sh,img<T>& a_res) const {
470
471 if((a_sx>=m_w)||(a_sy>=m_h)){
472 a_res.make_empty();
473 return false;
474 }
475
476 // 012345
477 unsigned int rw = min_of<unsigned int>(m_w-a_sx,a_sw);
478 unsigned int rh = min_of<unsigned int>(m_h-a_sy,a_sh);
479 unsigned int sz = rh*rw*m_n;
480 if(!sz) {
481 a_res.make_empty();
482 return false;
483 }
484
485 T* rb = new T[sz];
486 if(!rb) {
487 a_res.make_empty();
488 return false;
489 }
490
491 unsigned int rstride = rw * m_n;
492 T* rpos = rb;
493
494 unsigned int stride = m_w * m_n;
495 T* pos = m_buffer+a_sy*stride+a_sx*m_n;
496
497 //T* mx = m_buffer+size();
498 //T* rmx = rb+sz*sizeof(T);
499
500 for(unsigned int j=0;j<rh;j++,rpos+=rstride,pos+=stride) {//j=0 -> bottom.

504 delete [] rb;

522 bool to_texture(bool a_expand,
523 const T a_pixel[], //size shoulde be a_img.m_n.
524 img<T>& a_res,bool a_res_force_owner = true) const {
525
526 //NOTE : pixels of the original image are not expanded or shrinked.
527
528 if((!m_w)||(!m_h)) {
529 a_res.make_empty();
530 return false;
531 }
532
533 // in case (m_w==1)||(m_h==1), expand the pixel
534 // up to the closest power of 2 ?
535
536 if((m_w==1)||(m_h==1)||a_expand) {
537 // find closest power of two upper than m_w, m_h :
538 unsigned int rw = 2;
539 while(true) {if(rw>=m_w) break;rw *=2;}
540 unsigned int rh = 2;
541 while(true) {if(rh>=m_h) break;rh *=2;}
542
543 if((rw==m_w)&&(rh==m_h)) { //exact match.
544 if(a_res_force_owner) {
545 a_res.copy(m_w,m_h,m_n,m_buffer);
546 } else {
547 a_res.set(m_w,m_h,m_n,m_buffer,false); //WARNING owner=false.
548 }
549 return true;
550 }
551
552 // we expand the image and fill new spaces with a_pixel.
553
554 T* rb = 0;
555 bool res_set = true;
556 if(a_res.owner()&&(a_res.size()==(rh*rw*m_n))) {
557 // a_res has already the right allocation.
558 rb = a_res.buffer();
559 res_set = false;
560 } else {
561 rb = new T[rh*rw*m_n];
562 if(!rb) {
563 a_res.make_empty();
564 return false;
565 }
566 }
567
568 unsigned int num = rw*m_n;
569
570 // initialize with given color :
571 {T* pos = rb;
572 for(unsigned int i=0;i<rw;i++,pos+=m_n) {
573 ::memcpy(pos,a_pixel,m_n*sizeof(T));
574 }
575 unsigned int sz = num*sizeof(T);
576 for(unsigned int j=1;j<rh;j++,pos+=num) { //j=0 -> bottom.
577 ::memcpy(pos,rb,sz);
578 }}
579
580 // center :
581 unsigned int col = (rw-m_w)/2;
582 unsigned int row = (rh-m_h)/2;
583
584 unsigned int mnum = m_w*m_n;
585
586 // copy original image in a centered part of the new one :
587 {T* pos = m_buffer;
588 T* rpos = rb+row*num+col*m_n;
589 unsigned int sz = mnum*sizeof(T);
590 for(unsigned int j=0;j<m_h;j++,pos+=mnum,rpos+=num) {
591 ::memcpy(rpos,pos,sz);
592 }}
593
594 if(res_set) a_res.set(rw,rh,m_n,rb,true);
595
596 return true;
597 } else {
598 // then m_w>=2 and m_h>=2
599
600 // find closest power of two lower than m_w, m_h :
601 unsigned int sw = 2;
602 while(true) {if((sw*2)>m_w) break;sw *=2;}
603 unsigned int sh = 2;
604 while(true) {if((sh*2)>m_h) break;sh *=2;}
605
606 if((sw==m_w)&&(sh==m_h)) { //exact match.
607 if(a_res_force_owner) {
608 a_res.copy(m_w,m_h,m_n,m_buffer);
609 } else {
610 a_res.set(m_w,m_h,m_n,m_buffer,false); //WARNING owner=false.
611 }
612 return true;
613 }
614
615 unsigned int sx = (m_w-sw)/2;
616 unsigned int sy = (m_h-sh)/2;
617
618 return get_part(sx,sy,sw,sh,a_res);
619 }

623 bool check_gl_limit(unsigned int a_GL_MAX_TEXTURE_SIZE,img<T>& a_res) const {
624 // if ret true and a_res.is_empty(), "this" does not exceeds the limit.
625 // if ret true and !a_res.is_empty(), "this" exceeds the limit and a new fitting image is returned in a_res.
626 // if ret false, "this" exceeds the limit but something went wrong in building a_res.
627 unsigned int tw = m_w;
628 unsigned int th = m_h;
629 if((tw<=a_GL_MAX_TEXTURE_SIZE)&&(th<=a_GL_MAX_TEXTURE_SIZE)) {
630 a_res.make_empty();
631 return true;
632 }
633 unsigned int fac = 2;
634 while(true) {
635 unsigned int pw = tw/fac;
636 unsigned int ph = th/fac;
637 if((pw<=a_GL_MAX_TEXTURE_SIZE)&&(ph<=a_GL_MAX_TEXTURE_SIZE)) {
638 //unsigned int sx = (tw-pw)/2;
639 //unsigned int sy = (th-ph)/2;
640 //if(!get_part(sx,sy,pw,ph,a_res)) {
641 if(!contract(fac,a_res)) {
642 a_res.make_empty();
643 return false;
644 }
645 return true;
646 }
647 fac *= 2;
648 }
649 a_res.make_empty();
650 return false;
651 }

653 bool bw2x(unsigned int a_n,img<T>& a_res) const {
654 //expect a bw img.
655 if(m_n!=1) return false;
656
657 a_res.make_empty();
658 if(a_n<m_n) return false;
659 unsigned int sz = m_w*m_h*a_n;
660 if(!sz) return false;
661
662 a_res.m_buffer = new T[sz];
663 if(!a_res.m_buffer) return false;
664 a_res.m_owner = true;
665 a_res.m_w = m_w;
666 a_res.m_h = m_h;
667 a_res.m_n = a_n;
668
669 for(unsigned int j=0;j<m_h;j++) {
670 for(unsigned int i=0;i<m_w;i++) {
671 //position in the original image.
672 T* pos = m_buffer + j * (m_w * m_n) + i*m_n;
673
674 T* rpos = a_res.m_buffer + j * (m_w * a_n) + i*a_n;
675
676 for(unsigned int ipix=0;ipix<a_n;ipix++) {
677 *(rpos+ipix) = *pos;
678 }
679

695

697

699 T* pos = m_buffer + j * stride;
700 T* rpos = a_res.m_buffer + (m_h-j-1) * stride;
701 ::memcpy(rpos,pos,stride*sizeof(T));
702 }
703
704 return true;
