17class iros : public virtual iro,public std::vector<iro*> {
18 static const std::string& s_store_class() {
19 static const std::string s_v("TObjArray");
20 return s_v;
21 }
22public:
23 static const std::string& s_class() {
24 static const std::string s_v("tools::rroot::iros");
25 return s_v;
26 }
27public: //iro
28 virtual void* cast(const std::string& a_class) const {
29 if(void* p = cmp_cast<iros>(this,a_class)) return p;
30 return 0;
31 }
32 virtual const std::string& s_cls() const {return s_class();}
33public:
34 static cid id_class() {return obj_list_cid();}
35 virtual void* cast(cid a_class) const {
36 if(void* p = cmp_cast<iros>(this,a_class)) {return p;}
37 else return 0;
38 }
39public:
40 virtual iro* copy() const {return new iros(*this);}
41 virtual bool stream(buffer& a_buffer) {
42 ifac::args args;
43 bool accept_null = false;
44 return stream(a_buffer,args,accept_null);
45 }
46public:
47 iros(ifac& a_fac,bool a_owner,bool a_warn /*= true*/) //a_warn used if a_owner=false.
48 :m_fac(a_fac)
49 ,m_owner(a_owner)
50 ,m_warn(a_warn)
51 {
52#ifdef TOOLS_MEM
53 mem::increment(s_class().c_str());
54#endif
55 }
56 virtual ~iros(){
57 _clear();
58#ifdef TOOLS_MEM
59 mem::decrement(s_class().c_str());
60#endif
61 }
62public:
63 iros(const iros& a_from)
64 :iro(a_from)
65 ,std::vector<iro*>()
66 ,m_fac(a_from.m_fac)
67 ,m_owner(a_from.m_owner)
68 ,m_warn(a_from.m_warn)
69 {
70#ifdef TOOLS_MEM
71 mem::increment(s_class().c_str());
72#endif
73 if(a_from.m_owner) {
74 tools_vforcit(iro*,a_from,it) push_back((*it)->copy());
75 } else {
76 tools_vforcit(iro*,a_from,it) push_back((*it));
77 }
78 }
79 iros& operator=(const iros& a_from){
80 if(&a_from==this) return *this;
81
82 _clear();
83
84 m_owner = a_from.m_owner;
85 m_warn = a_from.m_warn;
86
87 if(a_from.m_owner) {
88 tools_vforcit(iro*,a_from,it) push_back((*it)->copy());
89 } else {
90 tools_vforcit(iro*,a_from,it) push_back((*it));
91 }
92
93 return *this;
94 }
95public:
96 void cleanup() {_clear();}
97 void dump(std::ostream& a_out) {
98 a_out << " iros : size " << size() << std::endl;
99 tools_vforcit(iro*,*this,it) {
100 a_out << " class " << (*it)->s_cls() << std::endl;
101 }
102 }
103public:
104 bool stream(buffer& a_buffer,const ifac::args& a_args,bool a_accept_null = false) {
105 _clear();
106
107 short v;
108 unsigned int s, c;
109 if(!a_buffer.read_version(v,s,c)) return false;
110
111 //::printf("debug : iros::stream : version %d, byte count %d\n",v,c);
112
113 {uint32 id,bits;
114 if(!Object_stream(a_buffer,id,bits)) return false;}
115 std::string name;
116 if(!a_buffer.read(name)) return false;
117 int nobjects;
118 if(!a_buffer.read(nobjects)) return false;
119 int lowerBound;
120 if(!a_buffer.read(lowerBound)) return false;
121
122 //::printf("debug : iros : name \"%s\", nobject %d, lowerBound %d\n",
123 // name.c_str(),nobjects,lowerBound);
124
125 for (int i=0;i<nobjects;i++) {
126 //::printf("debug : iros : n=%d i=%d ...\n",nobjects,i);
127
128 iro* obj;
129 bool created;
130 if(!a_buffer.read_object(m_fac,a_args,obj,created)){
131 a_buffer.out() << "tools::rroot::iros::stream : can't read object." << std::endl;
132 return false;
133 }
134 //::printf("debug : iros : n=%d i=%d : ok\n",nobjects,i);
135 if(obj) {
136 if(created) {
137 if(m_owner) {
138 push_back(obj);
139 } else { // Who manage this object ?
140 if(m_warn) {
141 a_buffer.out() << "tools::rroot::iros::stream :"
142 << " warning : created object of class " << sout(obj->s_cls()) << " not managed."
143 << std::endl;
144 }
145 push_back(obj);
146 }
147 } else { //someone else may manage this object.
148 if(m_owner) {
149 a_buffer.out() << "tools::rroot::iros::stream : warning : not created object can't be manage here." << std::endl;
150 return false;
151 } else {
152 push_back(obj);
153 }
154 }
155 } else {
156 //a_accept_null for branch::stream m_baskets.
157 if(a_accept_null) push_back(0);
158 }
159 }
160
161 return a_buffer.check_byte_count(s,c,s_store_class());
162 }
163protected:
164 void _clear() {
165 if(m_owner) {
166 safe_clear<iro>(*this);
167 } else {
168 std::vector<iro*>::clear();
169 }
170 }
171protected:
172 ifac& m_fac;
173 bool m_owner;
174 bool m_warn;
175};
