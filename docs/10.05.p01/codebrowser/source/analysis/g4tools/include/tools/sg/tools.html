21inline void add_string(
22 separator& a_sep
23,const std::string& a_font
24,const std::string& a_encoding
25,bool //a_smoothing
26,const std::string& a_string
27,float a_x,float a_y,float a_z
28,const vec3f& a_X
29,const vec3f& a_Y
30,float a_size
31,hjust a_hjust
32,vjust a_vjust
33,const base_freetype& a_ttf
34){
35 if(a_string.empty()) return;
36
37 matrix* tsf = new matrix;
38
39 {tsf->mul_translate(a_x,a_y,a_z);
40 vec3f X = a_X;
41 vec3f Y = a_Y;
42 X.normalize();
43 Y.normalize();
44 vec3f Z;X.cross(Y,Z);
45 Z.cross(X,Y);
46 mat4f r(X.v0(),Y.v0(),Z.v0(),0, //first row
47 X.v1(),Y.v1(),Z.v1(),0,
48 X.v2(),Y.v2(),Z.v2(),0,
49 0,0,0,1);
50 tsf->mul_mtx(r);
51 tsf->mul_scale(a_size,a_size,1);} //applied first on GL
52
53 a_sep.add(tsf);
54
55 if(a_font==font_hershey()) {
56
57 text_hershey* text = new text_hershey;
58 text->encoding.value(a_encoding);
59 text->strings.add(a_string);
60 text->hjust.value(a_hjust);
61 text->vjust.value(a_vjust);
62 a_sep.add(text);
63
64 } else {
65
66 base_freetype* text = base_freetype::create(a_ttf);
67
68 text->font = a_font;
69
70 //text->modeling.value(font_outline);
71 //text->encoding.value(a_encoding);
72 //text->smooting.value(a_smoothing);
73 text->strings.add(a_string);
74 text->hjust.value(a_hjust);
75 text->vjust.value(a_vjust);
76 //text->modeling = font_outline;
77 a_sep.add(text);
78
79 }
