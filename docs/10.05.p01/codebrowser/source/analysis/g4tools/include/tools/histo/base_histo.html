102 TW equivalent_bin_entries() const {
103 TW sw = 0;
104 TW sw2 = 0;
105 for(TO ibin=0;ibin<parent::m_bin_number;ibin++) {
106 if(!histo::is_out(parent::m_axes,ibin)) {
107 sw += parent::m_bin_Sw[ibin];

238 bool configure(dim_t a_dim,
239 const std::vector<bn_t>& aNumbers,
240 const std::vector<TC>& aMins,
241 const std::vector<TC>& aMaxs) {
242 // Clear :
243 parent::m_bin_entries.clear();
244 parent::m_bin_Sw.clear();
245 parent::m_bin_Sw2.clear();
246 parent::m_bin_Sxw.clear();
247 parent::m_bin_Sx2w.clear();
248 parent::m_in_range_Sxw.clear();
249 parent::m_in_range_Sx2w.clear();
250 parent::m_axes.clear();
251 parent::m_in_range_plane_Sxyw.clear();
252 parent::m_annotations.clear();
253
254 parent::m_bin_number = 0;
255 parent::m_dimension = 0;
256 parent::m_all_entries = 0;
257 parent::m_in_range_entries = 0;
258 parent::m_in_range_Sw = 0;
259 parent::m_in_range_Sw2 = 0;
260 parent::m_in_range_Sxw.resize(a_dim,0);
261 parent::m_in_range_Sx2w.resize(a_dim,0);
262
263 // Some checks :
264 if(!a_dim) return false;
265 parent::m_axes.resize(a_dim);
266 // Setup axes :
267 for(dim_t iaxis=0;iaxis<a_dim;iaxis++) {
268 if(!parent::m_axes[iaxis].configure(aNumbers[iaxis],aMins[iaxis],aMaxs[iaxis])) {
269 // do not do :
270 // m_axes.clear()
271 // so that :
272 // b1::axis(),b2::axis_[x,y]()
273 // do not crash in case of a bad booking.
274 //m_axes.clear();
275 return false;
276 }
277 }
278
279 parent::m_dimension = a_dim;
280
281 base_allocate(); //set m_bin_number.
282
283 return true;
284 }

326 void base_reset() {
327 // Reset content (different of clear that deallocate all internal things).
328 for(TO ibin=0;ibin<parent::m_bin_number;ibin++) {
329 parent::m_bin_entries[ibin] = 0;
330 parent::m_bin_Sw[ibin] = 0;
331 parent::m_bin_Sw2[ibin] = 0;
332 for(dim_t iaxis=0;iaxis<parent::m_dimension;iaxis++) {
333 parent::m_bin_Sxw[ibin][iaxis] = 0;
334 parent::m_bin_Sx2w[ibin][iaxis] = 0;
335 }
336 }
337 parent::m_in_range_plane_Sxyw.assign(dim_planes(parent::m_dimension),0);
338 //profile not done here.
339 parent::reset_fast_getters();
340 }

342protected:
343 void base_allocate() {
344 dim_t iaxis;
345 // Add two bins for the [under,out]flow data.
346 TO n_bin = 1;
347 for(iaxis=0;iaxis<parent::m_dimension;iaxis++) {
348 n_bin *= (parent::m_axes[iaxis].bins() + 2);
349 }
350
351 parent::m_bin_entries.resize(n_bin,0);
352 parent::m_bin_Sw.resize(n_bin,0);
353 parent::m_bin_Sw2.resize(n_bin,0);
354
355 std::vector<TC> empty;
356 empty.resize(parent::m_dimension,0);
357 parent::m_bin_Sxw.resize(n_bin,empty);
358 parent::m_bin_Sx2w.resize(n_bin,empty);
359
360 parent::m_bin_number = n_bin; // All bins : [in-range, underflow, outflow] bins.
361
362 parent::m_axes[0].m_offset = 1;
363 for(iaxis=1;iaxis<parent::m_dimension;iaxis++) {
364 parent::m_axes[iaxis].m_offset = parent::m_axes[iaxis-1].m_offset * (parent::m_axes[iaxis-1].bins()+2);
365 }
366
367 parent::m_in_range_plane_Sxyw.resize(dim_planes(parent::m_dimension),0);
368 }

438 void base_subtract(const base_histo& a_histo) {
439 //ill-defined operation. We keep that because of the "ill-defined past".
440 // We build a new histo with one entry in each bin.
441 for(TO ibin=0;ibin<parent::m_bin_number;ibin++) {
442 parent::m_bin_entries[ibin] = 1;
443
444 parent::m_bin_Sw[ibin] -= a_histo.m_bin_Sw[ibin];
445 // Yes, it is a += in the below.

458 bool base_multiply(const base_histo& a_histo) {
459 //ill-defined operation. We keep that because of the "ill-defined past".
460
461 // We build a new histo with one entry in each bin of weight :
462 // this.w * a_histo.w
463 // The current histo is overriden with this new histo.
464 // The m_bin_Sw2 computation is consistent with FreeHEP and CERN-ROOT.
465
466 if(!is_compatible(a_histo)) return false;
467
468 std::vector<int> is(parent::m_dimension);
469 for(TO ibin=0;ibin<parent::m_bin_number;ibin++) {
470 TW swa = parent::m_bin_Sw[ibin];
471 TW sw2a = parent::m_bin_Sw2[ibin];
472 TW swb = a_histo.m_bin_Sw[ibin];
473 TW sw2b = a_histo.m_bin_Sw2[ibin];
474 TW sw = swa * swb;
475 parent::m_bin_entries[ibin] = 1;
476 parent::m_bin_Sw[ibin] = sw;
477 parent::m_bin_Sw2[ibin] = sw2a * swb * swb + sw2b * swa * swa;
478 histo::get_indices(parent::m_axes,ibin,is);
479 for(dim_t iaxis=0;iaxis<parent::m_dimension;iaxis++) {
480 TC x = parent::m_axes[iaxis].bin_center(is[iaxis]);
481 parent::m_bin_Sxw[ibin][iaxis] = x * sw;
482 parent::m_bin_Sx2w[ibin][iaxis] = x * x * sw;
483 }
484 }
485
486 //{for(dim_t iplane=0;iplane<nplane;iplane++) parent::m_in_range_plane_Sxyw[iplane] ??? a_histo.m_in_range_plane_Sxyw[iplane];}
487
488 parent::update_fast_getters();
489 return true;
490 }

492 bool base_divide(const base_histo& a_histo) {
493 //ill-defined operation. We keep that because of the "ill-defined past".
494
495 // We build a new histo with one entry in each bin of weight :
496 // this.w / a_histo.w
497 // The current histo is overriden with this new histo.
498 // The m_bin_Sw2 computation is consistent with FreeHEP and ROOT.
499
500 if(!is_compatible(a_histo)) return false;
501
502 std::vector<int> is(parent::m_dimension);
503 for(TO ibin=0;ibin<parent::m_bin_number;ibin++) {
504 histo::get_indices(parent::m_axes,ibin,is);
505 TW swa = parent::m_bin_Sw[ibin];
506 TW swb = a_histo.m_bin_Sw[ibin];
507 TW sw2a = parent::m_bin_Sw2[ibin];
508 TW sw2b = a_histo.m_bin_Sw2[ibin];
509 if(swb!=0) {
510 parent::m_bin_entries[ibin] = 1;
511 TW sw = swa / swb;
512 parent::m_bin_Sw[ibin] = sw;
513 TW swb2 = swb * swb;
514 parent::m_bin_Sw2[ibin] = sw2a / swb2 + sw2b * swa * swa /(swb2*swb2);
515 for(dim_t iaxis=0;iaxis<parent::m_dimension;iaxis++) {
516 TC x = parent::m_axes[iaxis].bin_center(is[iaxis]);
517 parent::m_bin_Sxw[ibin][iaxis] = x * sw;
518 parent::m_bin_Sx2w[ibin][iaxis] = x * x * sw;
519 }
520 } else {
521 parent::m_bin_entries[ibin] = 0;
522 parent::m_bin_Sw[ibin] = 0;
523 parent::m_bin_Sw2[ibin] = 0;
524 for(dim_t iaxis=0;iaxis<parent::m_dimension;iaxis++) {
525 parent::m_bin_Sxw[ibin][iaxis] = 0;
526 parent::m_bin_Sx2w[ibin][iaxis] = 0;
527 }
528 }
529 }
530
531 //{for(dim_t iplane=0;iplane<nplane;iplane++) parent::m_in_range_plane_Sxyw[iplane] ??? a_histo.m_in_range_plane_Sxyw[iplane];}
532
533 parent::update_fast_getters();
534 return true;
535 }

539 TW factor2 = a_factor * a_factor;

544 parent::m_bin_Sxw[ibin][iaxis] *= a_factor;
545 parent::m_bin_Sx2w[ibin][iaxis] *= a_factor;
546 }

549 for(size_t iplane=0;iplane<nplane;iplane++) parent::m_in_range_plane_Sxyw[iplane] *= a_factor;}
550 parent::update_fast_getters();

554 bool get_ith_axis_mean(dim_t a_axis,TC& a_value) const {
555 a_value = 0;
556 if(a_axis>=parent::m_dimension) return false;
557 TW sw = 0;
558 TC sxw = 0;
559 for(TO ibin=0;ibin<parent::m_bin_number;ibin++) {
560 if(!histo::is_out(parent::m_axes,ibin)) {
561 sw += parent::m_bin_Sw[ibin];
562 sxw += parent::m_bin_Sxw[ibin][a_axis];
563 }
564 }
565 if(sw==0) return false;
566 a_value = sxw/sw;
567 return true;
568 }
569
570 bool get_ith_axis_rms(dim_t a_axis,TC& a_value) const {
571 a_value = 0;
572 if(a_axis>=parent::m_dimension) return false;
573 TW sw = 0;
574 TC sxw = 0;
575 TC sx2w = 0;
576 for(TO ibin=0;ibin<parent::m_bin_number;ibin++) {
577 if(!histo::is_out(parent::m_axes,ibin)) {
578 sw += parent::m_bin_Sw[ibin];
579 sxw += parent::m_bin_Sxw[ibin][a_axis];
580 sx2w += parent::m_bin_Sx2w[ibin][a_axis];
581 }
582 }
583 if(sw==0) return false;
584 TC mean = sxw/sw;
585 a_value = ::sqrt(::fabs((sx2w / sw) - mean * mean));
586 return true;
