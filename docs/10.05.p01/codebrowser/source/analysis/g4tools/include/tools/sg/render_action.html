212 void dump_vertex_array_xy(std::ostream& a_out,gl::mode_t /*a_mode*/,size_t a_floatn,const float* a_xys) {
213 size_t num = a_floatn/2;
214 if(!num) return;
215 a_out << "dump_vertex_array_xy : begin : " << num << std::endl;
216 for(size_t index=0;index<num;index++) {
217 a_out << "xy : " << index
218 << " " << a_xys[2*index]
219 << " " << a_xys[2*index+1]
220 << std::endl;
221 }
222 a_out << "dump_vertex_array_xy : end." << std::endl;
223 }
