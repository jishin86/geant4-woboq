63public: //helpers
64 template <class T>
65 bool read_std_vec(std::vector<T>& a_x) {
66 uint32 n;
67 T* v;
68 if(!read_vec(n,v)) return false;
69 a_x.resize(n);
70 for(uint32 index=0;index<n;index++) a_x[index] = v[index];
71 delete [] v;
72#ifdef TOOLS_MEM
73 mem::decrement(s_new().c_str());
74#endif
75 return true;
76 }
