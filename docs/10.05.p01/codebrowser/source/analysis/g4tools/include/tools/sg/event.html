260inline bool get_event_xy(const sg::event& a_event,int& a_x,int& a_y) {
261 typedef sg::event evt_t;
262 typedef sg::up_event up_evt_t;
263 typedef sg::down_event dn_evt_t;
264 if(const dn_evt_t* devt = safe_cast<evt_t,dn_evt_t>(a_event)) {
265 a_x = devt->x();
266 a_y = devt->y();
267 return true;
268
269 } else if(const up_evt_t* uevt = safe_cast<evt_t,up_evt_t>(a_event)) {
270 a_x = uevt->x();
271 a_y = uevt->y();
272 return true;
273
274 }
275 a_x = 0;
276 a_y = 0;
277 return false;
278}
