16template <class T>
17class obj_array : public virtual iro,public std::vector<T*> {
18 static const std::string& s_store_class() {
19 static const std::string s_v("TObjArray");
20 return s_v;
21 }
22public:
23 static const std::string& s_class() {
24 static const std::string s_v("tools::rroot::obj_array<"+T::s_class()+">");
25 return s_v;
26 }
27public: //iro
28 virtual void* cast(const std::string& a_class) const {
29 if(void* p = cmp_cast< obj_array<T> >(this,a_class)) return p;
30 return 0;
31 }
32 virtual const std::string& s_cls() const {return s_class();}
33public:
34 static cid id_class() {return obj_array_cid()+T::id_class();}
35 virtual void* cast(cid a_class) const {
36 if(void* p = cmp_cast<obj_array>(this,a_class)) {return p;}
37 return 0;
38 }
39 virtual iro* copy() const {return new obj_array<T>(*this);}
40 virtual bool stream(buffer& a_buffer) {
41 ifac::args args;
42 bool accept_null = false;
43 return stream(a_buffer,args,accept_null);
44 }
45public:
46 obj_array(ifac& a_fac,bool a_owner,bool a_warn) //a_warn used if a_owner=false.
47 :m_fac(a_fac)
48 ,m_owner(a_owner)
49 ,m_warn(a_warn)
50 {
51#ifdef TOOLS_MEM
52 mem::increment(s_class().c_str());
53#endif
54 }
55 virtual ~obj_array(){
56 _clear();
57#ifdef TOOLS_MEM
58 mem::decrement(s_class().c_str());
59#endif
60 }
61public:
62 obj_array(const obj_array& a_from)
63 :iro(a_from)
64 ,std::vector<T*>()
65 ,m_fac(a_from.m_fac)
66 ,m_owner(a_from.m_owner)
67 ,m_warn(a_from.m_warn)
68 {
69#ifdef TOOLS_MEM
70 mem::increment(s_class().c_str());
71#endif
72 if(a_from.m_owner) {
73 typedef typename std::vector<T*>::const_iterator it_t;
74 for(it_t it=a_from.begin();it!=a_from.end();++it) {
75 if(!(*it)) {
76 std::vector<T*>::push_back(0);
77 } else {
78 iro* _obj = (*it)->copy();
79 T* obj = safe_cast<iro,T>(*_obj);
80 if(!obj) {
81 //a_buffer.out() << "tools::rroot::obj_array::obj_array :"
82 // << " inlib::cast failed."
83 // << std::endl;
84 delete _obj;
85 } else {
86 std::vector<T*>::push_back(obj);
87 }
88 }
89 }
90 } else {
91 std::vector<T*>::operator=(a_from);
92 }
93 }
94 obj_array& operator=(const obj_array& a_from){
95 if(&a_from==this) return *this;
96
97 _clear();
98
99 m_owner = a_from.m_owner;
100 m_warn = a_from.m_warn;
101
102 if(a_from.m_owner) {
103 typedef typename std::vector<T*>::const_iterator it_t;
104 for(it_t it=a_from.begin();it!=a_from.end();++it) {
105 if(!(*it)) {
106 std::vector<T*>::push_back(0);
107 } else {
108 iro* _obj = (*it)->copy();
109 T* obj = safe_cast<iro,T>(*_obj);
110 if(!obj) {
111 //a_buffer.out() << "tools::rroot::obj_array::operator= :"
112 // << " inlib::cast failed."
113 // << std::endl;
114 delete _obj;
115 } else {
116 std::vector<T*>::push_back(obj);
117 }
118 }
119 }
120 } else {
121 std::vector<T*>::operator=(a_from);
122 }
123
124 return *this;
125 }
126public:
127 void cleanup() {_clear();}
128public:
129 bool stream(buffer& a_buffer,const ifac::args& a_args,bool a_accept_null = false) {
130 _clear();
131
132 short v;
133 unsigned int sp, bc;
134 if(!a_buffer.read_version(v,sp,bc)) return false;
135
136 //::printf("debug : obj_array::stream : version %d, byte count %d\n",v,c);
137
138 {uint32 id,bits;
139 if(!Object_stream(a_buffer,id,bits)) return false;}
140 std::string name;
141 if(!a_buffer.read(name)) return false;
142 int nobjects;
143 if(!a_buffer.read(nobjects)) return false;
144 int lowerBound;
145 if(!a_buffer.read(lowerBound)) return false;
146
147 //::printf("debug : obj_array : name \"%s\", nobject %d, lowerBound %d\n",
148 // name.c_str(),nobjects,lowerBound);
149
150 for (int i=0;i<nobjects;i++) {
151 //::printf("debug : obj_array : n=%d i=%d ...\n",nobjects,i);
152
153 iro* obj;
154 bool created;
155 if(!a_buffer.read_object(m_fac,a_args,obj,created)){
156 a_buffer.out() << "tools::rroot::obj_array::stream : can't read object." << std::endl;
157 return false;
158 }
159 //::printf("debug : obj_array : n=%d i=%d : ok\n",nobjects,i);
160 if(obj) {
161 T* to = safe_cast<iro,T>(*obj);
162 if(!to) {
163 a_buffer.out() << "tools::rroot::obj_array::stream :"
164 << " inlib::cast failed."
165 << " " << obj->s_cls() << " is not a " << T::s_class() << "."
166 << std::endl;
167 if(created) {
168 if(a_buffer.map_objs()) a_buffer.remove_in_map(obj);
169 delete obj;
170 }
171 } else {
172 if(created) {
173 if(m_owner) {
174 std::vector<T*>::push_back(to);
175 } else { //who manage this object ?
176 if(m_warn) {
177 a_buffer.out() << "tools::rroot::obj_array::stream :"
178 << " warning : created object of class " << sout(obj->s_cls()) << " not managed."
179 << std::endl;
180 }
181 std::vector<T*>::push_back(to);
182 }
183 } else { //someone else may manage this object.
184 if(m_owner) {
185 a_buffer.out() << "tools::rroot::obj_array::stream : not created object can't be manage here." << std::endl;
186 return false;
187 } else {
188 std::vector<T*>::push_back(to);
189 }
190 }
191 }
192 } else {
193 //a_accept_null for branch::stream m_baskets.
194 if(a_accept_null) std::vector<T*>::push_back(0);
195 }
196 }
197
198 return a_buffer.check_byte_count(sp,bc,s_store_class());
199 }
200protected:
201 void _clear() {
202 if(m_owner) {
203 safe_clear<T>(*this);
204 } else {
205 std::vector<T*>::clear();
206 }
207 }
208protected:
209 ifac& m_fac;
210 bool m_owner;
211 bool m_warn;
212};
