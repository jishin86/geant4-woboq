58
59 base_pntuple_column_wise(std::ostream& a_out,bool a_byte_swap,uint32 a_compression,seek a_seek_directory,
60 const std::vector<uint32>& a_basket_sizes,
61 const ntuple_booking& a_bkg,bool a_verbose)
62 :parent(a_out,a_seek_directory,a_bkg.name(),a_bkg.title())
63 ,m_file(a_out,a_byte_swap,a_compression,a_verbose)
64 {
65 const std::vector<column_booking>& cols = a_bkg.columns();
66
67 if(a_basket_sizes.size()!=cols.size()) {
68 a_out << "tools::wroot::base_pntuple_column_wise :"
69 << " a_basket_sizes.size() (" << a_basket_sizes.size() << ") != "
70 << "a_bkg.columns().size() (" << a_bkg.columns().size() << ")."
71 << std::endl;
72 return;
73 }
74 std::vector<uint32>::const_iterator itb = a_basket_sizes.begin();
75
76 tools_vforcit(column_booking,cols,it){
77
78#define TOOLS_WROOT_PNTUPLE_CREATE_COL(a__type) \
79 if((*it).cls_id()==_cid(a__type())) {\
80 a__type* user = (a__type*)(*it).user_obj();\
81 if(user) {\
82 if(!create_column_ref<a__type>(*itb,(*it).name(),*user)) {\
83 a_out << "tools::wroot::base_pntuple_column_wise : create_column_ref(" << (*it).name() << ") failed." << std::endl;\
84 safe_clear<icol>(m_cols);\
85 safe_clear<branch>(m_branches);\
86 return;\
87 }\
88 itb++;\
89 } else {\
90 if(!create_column<a__type>(*itb,(*it).name())) {\
91 a_out << "tools::wroot::base_pntuple_column_wise : create_column(" << (*it).name() << ") failed." << std::endl;\
92 safe_clear<icol>(m_cols);\
93 safe_clear<branch>(m_branches);\
94 return;\
95 }\
96 itb++;\
97 }\
98 }
99
100#define TOOLS_WROOT_PNTUPLE_CREATE_VEC_COL(a__type) \
101 if((*it).cls_id()==_cid_std_vector<a__type>()) {\
102 std::vector<a__type>* vec = (std::vector<a__type>*)(*it).user_obj();\
103 if(vec) {\
104 if(!create_column_vector_ref<a__type>(*itb,(*it).name(),*vec)) {\
105 a_out << "tools::wroot::base_pntuple_column_wise :"\
106 << " create_column failed for std::vector column_ref " << sout((*it).name()) << "."\
107 << std::endl;\
108 safe_clear<icol>(m_cols);\
109 safe_clear<branch>(m_branches);\
110 return;\
111 }\
112 itb++;\
113 } else {\
114 if(!create_column_vector<a__type>(*itb,(*it).name())) {\
115 a_out << "tools::wroot::base_pntuple_column_wise :"\
116 << " create_column failed for std::vector column " << sout((*it).name()) << "."\
117 << std::endl;\
118 safe_clear<icol>(m_cols);\
119 safe_clear<branch>(m_branches);\
120 return;\
121 }\
122 itb++;\
123 }\
124 }
125
126 TOOLS_WROOT_PNTUPLE_CREATE_COL(char)
127 else TOOLS_WROOT_PNTUPLE_CREATE_COL(short)
128 else TOOLS_WROOT_PNTUPLE_CREATE_COL(int)
129 else TOOLS_WROOT_PNTUPLE_CREATE_COL(float)
130 else TOOLS_WROOT_PNTUPLE_CREATE_COL(double)
131
132 else if((*it).cls_id()==_cid(std::string())) {
133 std::string* user = (std::string*)(*it).user_obj();
134 if(user) {
135 if(!create_column_string_ref(*itb,(*it).name(),*user)) {
136 a_out << "tools::wroot::base_pntuple_column_wise : create_column_string_ref(" << (*it).name() << ") failed."
137 << std::endl;
138 safe_clear<icol>(m_cols);
139 safe_clear<branch>(m_branches);
140 return;
141 }
142 itb++;
143 } else {
144 if(!create_column_string(*itb,(*it).name())) {
145 a_out << "tools::wroot::base_pntuple_column_wise : create_column_string(" << (*it).name() << ") failed." << std::endl;
146 safe_clear<icol>(m_cols);
147 safe_clear<branch>(m_branches);
148 return;
149 }
150 itb++;
151 }
152 }
153
154 else TOOLS_WROOT_PNTUPLE_CREATE_VEC_COL(char)
155 else TOOLS_WROOT_PNTUPLE_CREATE_VEC_COL(short)
156 else TOOLS_WROOT_PNTUPLE_CREATE_VEC_COL(int)
157 else TOOLS_WROOT_PNTUPLE_CREATE_VEC_COL(float)
158 else TOOLS_WROOT_PNTUPLE_CREATE_VEC_COL(double)
159
160 else if((*it).cls_id()==_cid_std_vector<std::string>()) {\
161 std::vector<std::string>* user = (std::vector<std::string>*)(*it).user_obj();
162 char sep = '\n';
163 if(user) {
164 if(!create_column_vector_string_ref(*itb,(*it).name(),*user,sep)) {
165 a_out << "tools::wroot::base_pntuple_column_wise :"
166 << " create_column_vector_string_ref(" << (*it).name() << ") failed." << std::endl;
167 safe_clear<icol>(m_cols);
168 safe_clear<branch>(m_branches);
169 return;
170 }
171 itb++;
172 } else {
173 if(!create_column_vector_string(*itb,(*it).name(),std::vector<std::string>(),sep)) {
174 a_out << "tools::wroot::base_pntuple_column_wise :"
175 << " create_column_vector_string(" << (*it).name() << ") failed." << std::endl;
176 safe_clear<icol>(m_cols);
177 safe_clear<branch>(m_branches);
178 return;
179 }
180 itb++;
181 }
182 }
183
184 // no leaf_store_class() defined for the other types.
185
186 else {
187 a_out << "tools::wroot::base_pntuple_column_wise :"
188 << " for column " << sout((*it).name())
189 << ", type with cid " << (*it).cls_id() << " not yet handled."
190 << std::endl;
191 //throw
192 safe_clear<icol>(m_cols);
193 safe_clear<branch>(m_branches);
194 return;
195 }
196
197#undef TOOLS_WROOT_PNTUPLE_CREATE_COL
198#undef TOOLS_WROOT_PNTUPLE_CREATE_VEC_COL
199
200 }
201 }

208 template <class T>
209 column_ref<T>* create_column_ref(uint32 a_basket_size,const std::string& a_name,const T& a_ref) {
210 if(find_named<icol>(m_cols,a_name)) return 0;
211 branch* _branch = new branch(m_file.out(),m_file.byte_swap(),m_file.compression(),
212 m_seek_directory,a_name,m_name,m_file.verbose());
213 _branch->set_basket_size(a_basket_size);
214 column_ref<T>* col = new column_ref<T>(*_branch,a_name,a_ref);
215 if(!col) {delete _branch;return 0;}
216 m_branches.push_back(_branch);
217 m_cols.push_back(col);
218 return col;
219 }

300 template <class T>
301 std_vector_column<T>* create_column_vector(uint32 a_basket_size,const std::string& a_name,const std::vector<T>& a_def = std::vector<T>()) {
302 if(find_named<icol>(m_cols,a_name)) return 0;
303 std_vector_be_pointer<T>* _branch = new std_vector_be_pointer<T>(m_file.out(),m_file.byte_swap(),m_file.compression(),
304 m_seek_directory,a_name,m_name,0,m_file.verbose());
305 _branch->set_basket_size(a_basket_size);
306 std_vector_column<T>* col = new std_vector_column<T>(*_branch,a_name,a_def);
307 if(!col) {delete _branch;return 0;}
308 _branch->set_pointer(&(col->variable()));
309 m_branches.push_back(_branch);
310 m_cols.push_back(col);
311 return col;
312 }
