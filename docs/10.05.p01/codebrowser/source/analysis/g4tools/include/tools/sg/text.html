102public:
103 virtual void render(render_action& a_action) {
104 if(touched()) {
105 update_sg();
106 reset_touched();
107 }
108 if(back_visible.value()) m_back_sep.render(a_action);
109 m_sep.render(a_action);
110 }
111 virtual void pick(pick_action& a_action) {
112 if(touched()) {
113 update_sg();
114 reset_touched();
115 }
116 if(back_visible.value()) {
117 nodekit_pick(a_action,m_back_sep,this);
118 }
119 //m_sep.pick(a_action);
120 }
121 virtual void search(search_action& a_action) {
122 node::search(a_action);
123 if(a_action.done()) return;
124 if(touched()) {
125 update_sg();
126 reset_touched();
127 }
128 if(a_action.do_path()) a_action.path_push(this);
129 if(back_visible.value()) {
130 m_back_sep.search(a_action);
131 if(a_action.done()) return;
132 }
133 m_sep.search(a_action);
134 if(a_action.done()) return;
135 if(a_action.do_path()) a_action.path_pop();
136 }
137 virtual void bbox(bbox_action& a_action) {
138 if(touched()) {
139 update_sg();
140 reset_touched();
141 }
142 if(back_visible.value()) m_back_sep.bbox(a_action);
143 m_sep.bbox(a_action);

248public:
249 void update_sg() {
250 parent::update_sg();
251
252 m_sep.clear();
253 m_base_text = 0;
254
255 if(width.value()<=0) return;
256 if(height.value()<=0) return;
257 if(is_empty()) return;
258
259 rgba* mat = new rgba();
260 mat->color = color;
261 m_sep.add(mat);
262
263 matrix* tsf = new matrix;
264 m_sep.add(tsf);
265
266 float fw = width * wmargin_factor;
267
268 //sf<float> zfront ?
269 float zz = back_visible.value()?0.01f:0;
270
271 if(font==font_hershey()) {
272
273 draw_style* ds = new draw_style;
274 ds->style = draw_style::lines;
275 ds->line_width = line_width;
276 m_sep.add(ds);
277
278 text_hershey* _text = new text_hershey;
279 m_base_text = _text;
280 _text->encoding = encoding;
281 _text->strings = strings;
282 m_sep.add(_text);
283
284 } else {
285
286 m_base_text = m_TT_text;
287 //ttf/arialbd 11
288 //01234567890
289 m_TT_text->font = font;
290 m_TT_text->strings = strings;
291
292 //_text->modeling.value(font_outline);
293
294 m_sep.add(new noderef(*m_TT_text));
295
296 }
297
298 if(enforce_front_height) {
299 m_base_text->height = front_height;
300
301 float mn_x,mn_y,mn_z;
302 float mx_x,mx_y,mx_z;
303 m_base_text->get_bounds(front_height,mn_x,mn_y,mn_z,mx_x,mx_y,mx_z);
304
305 float bxw = mx_x-mn_x;
306 float xtrans = -fw*0.5f+bxw*0.5f; //left justified
307 if(hjust==center) {
308 xtrans = 0;
309 } else if(hjust==right) {
310 xtrans = fw*0.5f-bxw*0.5f;
311 }
312 float xx = -(mn_x+mx_x)*0.5F+xtrans;
313 float yy = -(mn_y+mx_y)*0.5F;
314 tsf->set_translate(xx,yy,zz);
315
316 return;
317 }
318
319 //various automatic text height strategies :
320
321 if(confine) {
322 // code common to freetype and hershey :
323
324 float fh = height * hmargin_factor;
325
326 // try to adjust text within fw x fh (by attempting to be smart !).
327 // the below assumes that text bounds are linear according
328 // "text height".
329 {float mn_x,mn_y,mn_z;
330 float mx_x,mx_y,mx_z;
331
332 float th = fh;
333 m_base_text->get_bounds(th,mn_x,mn_y,mn_z,mx_x,mx_y,mx_z);
334 float bxh = mx_y-mn_y;
335 // adjust box height :
336 // fh -> bxh then to have fh :
337 if(bxh>0) {
338 th = fh*fh/bxh;
339 m_base_text->get_bounds(th,mn_x,mn_y,mn_z,mx_x,mx_y,mx_z);
340 }
341
342 //float bxw = box.mx()[0]-box.mn()[0];
343 float bxw = mx_x-mn_x;
344 bxh = mx_y-mn_y;
345 if((fh>0)&&(bxh>0)){
346 float fasp = fw/fh;
347 float basp = bxw/bxh;
348 if(fasp>=basp) {
349 } else {
350 // adjust box width :
351 // th -> bxw then to have fw :
352 if(bxw>0) {
353 th = th*fw/bxw;
354 m_base_text->get_bounds(th,mn_x,mn_y,mn_z,mx_x,mx_y,mx_z);
355 }
356 }
357 }
358
359 m_base_text->height = th;
360
361 //bxw = box.mx()[0]-box.mn()[0];
362 bxw = mx_x-mn_x;
363 float xtrans = -fw*0.5f+bxw*0.5f; //left justified
364 if(hjust==center) {
365 xtrans = 0;
366 } else if(hjust==right) {
367 xtrans = fw*0.5f-bxw*0.5f;
368 }
369 float xx = -(mn_x+mx_x)*0.5F+xtrans;
370 float yy = -(mn_y+mx_y)*0.5F;
371 tsf->set_translate(xx,yy,zz);
372
373 }
374
375 } else {
376 float fh = height * hmargin_factor;
377
378 // we arrange yy so that two aside texts
379 // with same height will have their text base lines aligned.
380 // The max height is given by ascent+descent (with descent>0).
381 float th = fh;
382
383 float mxh = m_base_text->ascent(th)+
384 m_base_text->y_advance(th)*(strings.size()-1)+
385 m_base_text->descent(th);
386
387 //{box3f box;
388 // m_base_text->get_bounds(th,box);
389 // float bxh = box.mx()[1]-box.mn()[1]; //should be idem mxh.
390 // mxh = bxh;}
391
392 if(mxh) th = fh*fh/mxh; //end/final height.
393
394 m_base_text->height = th; //then all chars will fit into th.
395
396 mxh = m_base_text->ascent(th)+
397 m_base_text->y_advance(th)*(strings.size()-1)+
398 m_base_text->descent(th);
399
400 float yy = -fh*0.5f+m_base_text->descent(th)+
401 m_base_text->y_advance(th)*(strings.size()-1);
402
403 //float xx = -fw*0.5F; //left justified.
404 //tsf->set_translate(xx,yy,zz);
405 {float mn_x,mn_y,mn_z;
406 float mx_x,mx_y,mx_z;
407 m_base_text->get_bounds(th,mn_x,mn_y,mn_z,mx_x,mx_y,mx_z);
408 float bxw = mx_x-mn_x;
409 float xtrans = -fw*0.5f+bxw*0.5f; //left justified
410 if(hjust==center) {
411 xtrans = 0;
412 } else if(hjust==right) {
413 xtrans = fw*0.5f-bxw*0.5f;
414 }
415 float xx = -(mn_x+mx_x)*0.5F+xtrans;
416 tsf->set_translate(xx,yy,zz);}
417
418 // truncate text at right if out of border :
419 {std::vector<std::string> labcut;
420 tools_vforcit(std::string,strings.values(),it) {
421 std::string scut;
422 m_base_text->truncate(*it,th,fw,scut);
423 labcut.push_back(scut);
424 }
425 m_base_text->strings = labcut;}
426 }
