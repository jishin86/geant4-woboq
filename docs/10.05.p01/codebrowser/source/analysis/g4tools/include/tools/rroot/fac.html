28 virtual iro* create(const std::string& a_class,const args&) {
29 //m_out << "tools::rroot::fac::create :"
30 // << " create object of class " << a_class << "..."
31 // << std::endl;
32 if(a_class=="TBranch") {
33 return new branch(m_out,*this);
34 } else if(a_class=="TBranchElement") {
35 return new branch_element(m_out,*this);
36 } else if(a_class=="TBranchObject") {
37 return new branch_object(m_out,*this);
38
39 } else if(a_class=="TLeafB") {
40 return new leaf<char>(m_out,*this);
41
42 } else if(a_class=="TLeafS") {
43 return new leaf<short>(m_out,*this);
44
45 } else if(a_class=="TLeafI") {
46 return new leaf<int>(m_out,*this);
47
48 } else if(a_class=="TLeafF") {
49 return new leaf<float>(m_out,*this);
50
51 } else if(a_class=="TLeafD") {
52 return new leaf<double>(m_out,*this);
53
54 } else if(a_class=="TLeafO") {
55 return new leaf<bool>(m_out,*this);
56
57 } else if(a_class=="TLeafC") {
58 return new leaf_string(m_out,*this);
59
60 } else if(a_class=="TLeafElement") {
61 return new leaf_element(m_out,*this);
62
63 } else if(a_class=="TLeafObject") {
64 return new leaf_object(m_out,*this);
65
66 } else if(a_class=="TBasket") {
67 return new basket(m_out);
68
69 // L.Duflot ATLAS file :
70 } else if(a_class=="TTreeIndex") {
71 return new tree_index();
72
73 } else if(a_class=="TList") {
74 return new obj_list(*this,true,true);
75 } else if(a_class=="TNamed") {
76 return new named();
77
78 } else if(a_class=="vector<unsigned short>") {
79 return new stl_vector<unsigned short>();
80 } else if(a_class=="vector<short>") {
81 return new stl_vector<short>();
82 } else if(a_class=="vector<unsigned int>") {
83 return new stl_vector<unsigned int>();
84 } else if(a_class=="vector<int>") {
85 return new stl_vector<int>();
86 } else if(a_class=="vector<float>") {
87 return new stl_vector<float>();
88 } else if(a_class=="vector<double>") {
89 return new stl_vector<double>();
90
91 } else if(a_class=="vector<unsigned long>") { //beurk
92 return new stl_vector<uint64>(); //is it ok to map to an uint64 ?
93
94 } else if(a_class=="vector<string>") {
95 return new stl_vector_string();
96
97 } else if(a_class=="vector<vector<unsigned short> >") {
98 return new stl_vector_vector<unsigned short>();
99 } else if(a_class=="vector<vector<short> >") {
100 return new stl_vector_vector<short>();
101 } else if(a_class=="vector<vector<unsigned int> >") {
102 return new stl_vector_vector<unsigned int>();
103 } else if(a_class=="vector<vector<int> >") {
104 return new stl_vector_vector<int>();
105 } else if(a_class=="vector<vector<float> >") {
106 return new stl_vector_vector<float>();
107 } else if(a_class=="vector<vector<double> >") {
108 return new stl_vector_vector<double>();
109
110 } else if(a_class=="TBranchRef") {
111 return new dummy();
112
113 } else {
114 m_out << "tools::rroot::fac::create :"
115 << " unknown class " << sout(a_class) << "."
116 << " Create a tools::rroot::dummy object."
117 << std::endl;
118 return new dummy();
119 }
120 }
