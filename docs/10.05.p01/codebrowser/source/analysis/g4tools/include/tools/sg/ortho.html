4#ifndef tools_sg_ortho
5#define tools_sg_ortho
6
7#include "base_camera"
8
9namespace tools {
10namespace sg {
11
12class ortho : public base_camera {
13 TOOLS_NODE(ortho,tools::sg::ortho,base_camera)
14public:
15 virtual float near_height() const {return height.value();}
16 virtual void zoom(float a_fac) {
17 //for exa :
18 // a_fac = 0.99f is a zoom in
19 // a_fac = 1.01f is a zoom out
20 height.value(height.value()*a_fac);
21 }
22 virtual camera_type type() const {return camera_ortho;}
23public:
24 sf<float> height;
25public:
26 virtual const std::vector<field_desc>& node_fields() const {
27 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::ortho)
28 static std::vector<field_desc> s_v;
29 if(s_v.empty()) {
30 s_v = parent::node_fields();
31 TOOLS_ADD_FIELD_DESC(height)
32 }
33 return s_v;
34 }
35private:
36 void add_fields(){
37 add_field(&height);
38 }
39public:
40 ortho()
41 :parent()
42 ,height(2)
43 {
44#ifdef TOOLS_MEM
45 mem::increment(s_class().c_str());
46#endif
47 add_fields();
48 }
49 virtual ~ortho(){
50#ifdef TOOLS_MEM
51 mem::decrement(s_class().c_str());
52#endif
53 }
54public:
55 ortho(const ortho& a_from)
56 :parent(a_from)
57 ,height(a_from.height)
58 {
59#ifdef TOOLS_MEM
60 mem::increment(s_class().c_str());
61#endif
62 add_fields();
63 }
64 ortho& operator=(const ortho& a_from){
65 parent::operator=(a_from);
66 height = a_from.height;
67 return *this;
68 }
69public: //operators:
70 bool operator==(const ortho& a_from) const{
71 if(!parent::operator==(a_from)) return false;
72 if(height!=a_from.height) return false;
73 return true;
74 }
75 bool operator!=(const ortho& a_from) const {
76 return !operator==(a_from);
77 }
78public:
79 void dump(std::ostream& a_out) {
80 parent::dump(a_out);
81 a_out << " height " << height.value() << std::endl;
82 }
83protected:
84 virtual void get_lrbt(unsigned int a_ww,unsigned int a_wh,
85 float& a_l,float& a_r,float& a_b,float& a_t) {
86 float aspect = float(a_ww)/float(a_wh);
87 //landscape : float aspect = float(a_action.wh())/float(a_action.ww());
88 float hh = height.value()*0.5f;
89 a_l = -aspect*hh;
90 a_r = aspect*hh;
91 a_b = -hh;
92 a_t = hh;
93 }
94};
95
96}}

98#endif
