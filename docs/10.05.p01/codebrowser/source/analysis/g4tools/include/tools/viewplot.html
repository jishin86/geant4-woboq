146 bool write(const std::string& a_file,bool a_anonymous = false) {
147 sg::render_zb action(m_mgr,m_out,width(),height());
148 colorf clear_color = colorf_white();
149 action.zbuffer().clear_color_buffer(0);
150 action.add_color(clear_color.r(),clear_color.g(),clear_color.b());
151 action.zbuffer().clear_depth_buffer();
152 sg().render(action);
153 wps wps(m_out);
154 if(!wps.open_file(a_file,a_anonymous)) {
155 m_out << "tools::viewplot::write : can't open " << a_file << "." << std::endl;
156 return false;
157 }
158 wps.PS_BEGIN_PAGE();
159 wps.PS_PAGE_SCALE(float(width()),float(height()));
160 wps.PS_IMAGE(width(),height(),wps::rgb_4,sg::render_zb::get_rgb,&action);
161 wps.PS_END_PAGE();
162 wps.close_file();
163 return true;
164 }

191#ifdef TOOLS_USE_PNG
192 bool write_png(const std::string& a_file) {
193 sg::render_zb action(m_mgr,m_out,m_ww,m_wh);
194 colorf clear_color = colorf_white();
195 action.zbuffer().clear_color_buffer(0);
196 action.add_color(clear_color.r(),clear_color.g(),clear_color.b());
197 action.zbuffer().clear_depth_buffer();
198 sg().render(action);
199
200 unsigned int bpp = 3;
201 uchar* buffer = new unsigned char[m_ww*m_wh*bpp];
202 if(!buffer) {
203 m_out << "tools::viewplot::write_png : can't alloc buffer." << std::endl;
204 return false;
205 }
206 unsigned char* pos = buffer;
207 sg::render_zb::VCol r,g,b;
208 for(unsigned int row=0;row<m_wh;row++) {
209 for(unsigned int col=0;col<m_ww;col++) {
210 sg::render_zb::get_rgb(&action,col,m_wh-row-1,r,g,b);
211 *pos = (uchar)(255.0F*r);pos++;
212 *pos = (uchar)(255.0F*g);pos++;
213 *pos = (uchar)(255.0F*b);pos++;
214 }
215 }
216
217 bool status = exlib::png::write(m_out,a_file,buffer,m_ww,m_wh,bpp);
218 if(!status) {
219 m_out << "tools::viewplot::write_png : can't write " << a_file << "." << std::endl;
220 }
221 delete [] buffer;
222 return status;
223 }
224#endif

226protected:
227 void create_sg(unsigned int a_cols,unsigned int a_rows) {
228 m_sg.clear();
229
230 m_camera.height = 1;
231 float z = 10*1;
232 m_camera.znear = 0.1f*z;
233 m_camera.zfar = 10*z; //100*z induces problems with lego rendering.
234 m_camera.position = vec3f(0,0,z);
235 m_camera.orientation = rotf(vec3f(0,0,1),0);
236 m_camera.focal = z;
237 m_sg.add(new sg::noderef(m_camera));
238
239 m_plots.cols = a_cols;
240 m_plots.rows = a_rows;
241 m_plots.adjust_size(width(),height());
242
243 m_sg.add(new sg::noderef(m_plots));
244 }

246 void style_from_res(const std::string& a_path,sg::axis& a_axis) {
247 {typedef xml::styles::style_t style_t;
248 const style_t* style = m_styles.find_style(a_path);
249 if(style){
250 a_axis.set_from_style(m_out,*style);
251 } else {
252 //m_out << "tools::viewplot::style_from_res(axis) :"
253 // << " style " << sout(a_path) << " not found."
254 // << std::endl;
255 }}
256
257 m_styles.res_sg_style<sg::line_style>(a_path+".line_style",a_axis.line_style());
258 m_styles.res_sg_style<sg::line_style>(a_path+".ticks_style",a_axis.ticks_style());
259 m_styles.res_sg_style<sg::text_style>(a_path+".labels_style",a_axis.labels_style());
260 m_styles.res_sg_style<sg::text_style>(a_path+".mag_style",a_axis.mag_style());
261 m_styles.res_sg_style<sg::text_style>(a_path+".title_style",a_axis.title_style());
262 }

264 void style_from_res(const std::string& a_path,sg::plotter& a_plotter,bool a_verbose) {
265 //sg::fields of plotter :
266 {typedef xml::styles::style_t style_t;
267 const style_t* _style = m_styles.find_style(a_path);
268 if(_style){
269 a_plotter.set_from_style(m_out,*_style);
270 } else {
271 if(a_verbose) {
272 m_out << "tools::viewplot::style_from_res :"
273 << " style " << sout(a_path) << " not found."
274 << std::endl;
275 }
276 }}
277
278 //sg::fields of various xml::styles :
279 m_styles.res_sg_style<sg::style>(a_path+".bins_style.0",a_plotter.bins_style(0));
280 m_styles.res_sg_style<sg::style>(a_path+".errors_style.0",a_plotter.errors_style(0));
281 m_styles.res_sg_style<sg::style>(a_path+".func_style.0",a_plotter.func_style(0));
282 m_styles.res_sg_style<sg::style>(a_path+".points_style.0",a_plotter.points_style(0));
283
284 m_styles.res_sg_style<sg::style>(a_path+".background_style",a_plotter.background_style());
285 m_styles.res_sg_style<sg::text_style>(a_path+".title_style",a_plotter.title_style());
286 m_styles.res_sg_style<sg::text_style>(a_path+".infos_style",a_plotter.infos_style());
287 m_styles.res_sg_style<sg::text_style>(a_path+".title_box_style",a_plotter.title_box_style());
288 m_styles.res_sg_style<sg::style>(a_path+".inner_frame_style",a_plotter.inner_frame_style());
289 m_styles.res_sg_style<sg::style>(a_path+".grid_style",a_plotter.grid_style());
290
291 style_from_res(a_path+".x_axis",a_plotter.x_axis());
292 style_from_res(a_path+".y_axis",a_plotter.y_axis());
293 style_from_res(a_path+".z_axis",a_plotter.z_axis());
294 style_from_res(a_path+".colormap_axis",a_plotter.colormap_axis());
295 }
