15class obj_list : public virtual iro, public std::vector<iro*> {
16 static const std::string& s_store_class() {
17 static const std::string s_v("TList");
18 return s_v;
19 }
20public:
21 static const std::string& s_class() {
22 static const std::string s_v("tools::rroot::obj_list");
23 return s_v;
24 }
25public: //iro
26 virtual void* cast(const std::string& a_class) const {
27 if(void* p = cmp_cast<obj_list>(this,a_class)) return p;
28 return 0;
29 }
30 virtual const std::string& s_cls() const {return s_class();}
31public:
32 static cid id_class() {return obj_list_cid();}
33 virtual void* cast(cid a_class) const {
34 if(void* p = cmp_cast<obj_list>(this,a_class)) {return p;}
35 else return 0;
36 }
37 //virtual void* cast(cid) const {return obj_list_cid();}
38public:
39 virtual iro* copy() const {return new obj_list(*this);}
40 virtual bool stream(buffer& a_buffer) {
41 _clear();
42
43 short v;
44 unsigned int s, c;
45 if(!a_buffer.read_version(v,s,c)) return false;
46
47 //::printf("debug : obj_list::stream : version %d, byte count %d\n",v,c);
48
49 {uint32 id,bits;
50 if(!Object_stream(a_buffer,id,bits)) return false;}
51
52 std::string name;
53 if(!a_buffer.read(name)) return false;
54 int nobjects;
55 if(!a_buffer.read(nobjects)) return false;
56
57 //::printf("debug : obj_list : name \"%s\", nobject %d\n",
58 // name.c_str(),nobjects);
59
60 ifac::args args;
61 for (int i=0;i<nobjects;i++) {
62 //::printf("debug : obj_list : n=%d i=%d ...\n",nobjects,i);
63
64 iro* obj;
65 bool created;
66 if(!a_buffer.read_object(m_fac,args,obj,created)){
67 a_buffer.out() << "tools::rroot::obj_list::stream : can't read object." << std::endl;
68 return false;
69 }
70
71 unsigned char nch;
72 if(!a_buffer.read(nch)) return false;
73 if(nch) {
74 char readOption[256];
75 if(!a_buffer.read_fast_array(readOption,nch)) return false;
76 readOption[nch] = 0;
77 }
78 if(obj) {
79 if(created) {
80 if(m_owner) {
81 push_back(obj);
82 } else { //who manage this object ?
83 if(m_warn) {
84 a_buffer.out() << "tools::rroot::obj_list::stream :"
85 << " warning : created object of class " << sout(obj->s_cls()) << " not managed."
86 << std::endl;
87 }
88 push_back(obj);
89 }
90 } else { //someone else may manage this object.
91 if(m_owner) {
92 a_buffer.out() << "tools::rroot::obj_list::stream : not created object can't be manage here." << std::endl;
93 return false;
94 } else {
95 push_back(obj);
96 }
97 }
98 }
99 }
100
101 return a_buffer.check_byte_count(s,c,s_store_class());
102 }
103public:
104 obj_list(ifac& a_fac,bool a_owner,bool a_warn) //a_warn used if a_owner=false.
105 :m_fac(a_fac)
106 ,m_owner(a_owner)
107 ,m_warn(a_warn)
108 {
109#ifdef TOOLS_MEM
110 mem::increment(s_class().c_str());
111#endif
112 }
113 virtual ~obj_list(){
114 _clear();
115#ifdef TOOLS_MEM
116 mem::decrement(s_class().c_str());
117#endif
118 }
119protected:
120 obj_list(const obj_list& a_from)
121 :iro(a_from)
122 ,std::vector<iro*>()
123 ,m_fac(a_from.m_fac)
124 ,m_owner(a_from.m_owner)
125 ,m_warn(a_from.m_warn)
126 {
127#ifdef TOOLS_MEM
128 mem::increment(s_class().c_str());
129#endif
130 }
131 obj_list& operator=(const obj_list&){return *this;}
132protected:
133 void _clear() {
134 if(m_owner) {
135 safe_clear<iro>(*this);
136 } else {
137 std::vector<iro*>::clear();
138 }
139 }
140protected:
141 ifac& m_fac;
142 bool m_owner;
143 bool m_warn;
144};

146class hash_list : public obj_list {
147 static const std::string& s_store_class() {
148 static const std::string s_v("THashList");
149 return s_v;
150 }
151public:
152 static const std::string& s_class() {
153 static const std::string s_v("tools::rroot::hash_list");
154 return s_v;
155 }
156public: //iro
157 virtual void* cast(const std::string& a_class) const {
158 if(void* p = cmp_cast<hash_list>(this,a_class)) return p;
159 return obj_list::cast(a_class);
160 }
161 virtual const std::string& s_cls() const {return s_class();}
162public:
163 static cid id_class() {return hash_list_cid();}
164 virtual void* cast(cid a_class) const {
165 if(void* p = cmp_cast<hash_list>(this,a_class)) {return p;}
166 return obj_list::cast(a_class);
167 }
168public:
169 virtual bool stream(buffer& a_buffer) {
170 uint32 startpos = a_buffer.length();
171 unsigned int s, c;
172 short v;
173 if(!a_buffer.read_version(v,s,c)) return false;
174
175 //if(!obj_list::stream(a_buffer)) return false;
176
177 a_buffer.set_offset(startpos+c+sizeof(unsigned int));
178 if(!a_buffer.check_byte_count(s,c,s_store_class())) return false;
179 return true;
180 }
181
182public:
183 hash_list(ifac& a_fac,bool a_owner,bool a_warn) //a_warn used if a_owner=false.
184 :obj_list(a_fac,a_owner,a_warn)
185 {
186#ifdef TOOLS_MEM
187 mem::increment(s_class().c_str());
188#endif
189 }
190 virtual ~hash_list(){
191#ifdef TOOLS_MEM
192 mem::decrement(s_class().c_str());
193#endif
194 }
195protected:
196 hash_list(const hash_list& a_from):iro(a_from),obj_list(a_from){}
197 hash_list& operator=(const hash_list&){return *this;}
198};
