27inline int fround(float a_x) {
28 // From CoinGL/src/base/SbViewportRegion.cpp.
29 if (a_x == (float) (int(a_x))) return int(a_x);
30 else return (a_x>0.0f) ? int(a_x+0.5f) : -int(0.5f-a_x);
31}
