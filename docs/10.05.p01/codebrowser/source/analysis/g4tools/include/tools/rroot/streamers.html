39inline bool AttAxis_stream(buffer& a_buffer){
40 int fNdivisions = 510; //Number of divisions(10000*n3 + 100*n2 + n1)
41 short fAxisColor = 1; //color of the line axis
42 short fLabelColor = 1; //color of labels
43 short fLabelFont = 62; //font for labels
44 float fLabelOffset = 0.005F; //offset of labels
45 float fLabelSize = 0.04F; //size of labels
46 float fTickLength = 0.03F; //length of tick marks
47 float fTitleOffset = 1; //offset of axis title
48 float fTitleSize = 0.04F; //size of axis title
49 short fTitleColor = 1; //color of axis title
50 short fTitleFont = 62; //font for axis title
51
52 // Version 4 streaming (ROOT/v3-00-6).
53 short v;
54 unsigned int s, c;
55 if(!a_buffer.read_version(v,s,c)) return false;
56
57 if(!a_buffer.read(fNdivisions)) return false;
58 if(!a_buffer.read(fAxisColor)) return false;
59 if(!a_buffer.read(fLabelColor)) return false;
60 if(!a_buffer.read(fLabelFont)) return false;
61 if(!a_buffer.read(fLabelOffset)) return false;
62 if(!a_buffer.read(fLabelSize)) return false;
63 if(!a_buffer.read(fTickLength)) return false;
64 if(!a_buffer.read(fTitleOffset)) return false;
65 if(!a_buffer.read(fTitleSize)) return false;
66 if(!a_buffer.read(fTitleColor)) return false;
67 if(!a_buffer.read(fTitleFont)) return false;
68
69 if(!a_buffer.check_byte_count(s, c,"TAttAxis")) return false;
70 return true;
71}

73class dummy_fac : public virtual ifac {
74public: //ifac
75 virtual iro* create(const std::string& a_class,const args& a_args) {
76 if(rcmp(a_class,"TGraph")) { //for TH_read_1D/List.
77 return new graph();
78
79 // for read_sinfos() :
80 } else if(rcmp(a_class,"TStreamerInfo")) {
81 return new StreamerInfo(*this);
82 } else if(rcmp(a_class,"TObjArray")) {
83 std::string* sc = ifac::arg_class(a_args);
84 if(sc) {
85 if((*sc)==streamer_element::s_class()){
86 return new obj_array<streamer_element>(*this,true,true);
87 } else {
88 m_out << "tools::rroot::dummy_fac::create :"
89 << " Can't create TObjArray of " << *sc << "."
90 << std::endl;
91 return 0;
92 }
93 } else {
94 return new iros(*this,true,true);
95 }
96
97 } else if(rcmp(a_class,"TStreamerBase")
98 ||rcmp(a_class,"TStreamerBasicType")
99 ||rcmp(a_class,"TStreamerBasicPointer")
100 ||rcmp(a_class,"TStreamerObjectAny")
101 ||rcmp(a_class,"TStreamerObject")
102 ||rcmp(a_class,"TStreamerObjectPointer")
103 ||rcmp(a_class,"TStreamerString")
104 ||rcmp(a_class,"TStreamerSTL")
105 ||rcmp(a_class,"TStreamerLoop")
106 ||rcmp(a_class,"TList")
107 ) {
108 return new dummy_streamer_element();
109
110 } else {
111 m_out << "tools::rroot::dummy_fac::create :"
112 << " dummy. Can't create object of class " << sout(a_class) << "."
113 << std::endl;
114 }
115 return 0;
116 }
117 virtual void destroy(iro*& a_obj) {delete a_obj;a_obj = 0;}
118public:
119 dummy_fac(std::ostream& a_out):m_out(a_out){}
120 virtual ~dummy_fac(){}
121protected:
122 dummy_fac(const dummy_fac& a_from): ifac(a_from),m_out(a_from.m_out){}
123 dummy_fac& operator=(const dummy_fac&){return *this;}
124protected:
125 std::ostream& m_out;
126};

128inline bool Axis_stream(buffer& a_buffer,histo::axis<double,unsigned int>& a_fAxis){
129 // Version 6 streaming (ROOT/v3-00-6).
130 short v;
131 unsigned int s, c;
132 if(!a_buffer.read_version(v,s,c)) return false;
133
134 std::string name;
135 std::string title;
136 if(!Named_stream(a_buffer,name,title)) return false;
137
138 if(!AttAxis_stream(a_buffer)) return false;
139
140 int number;
141 if(!a_buffer.read(number)) return false;
142 double min;
143 if(!a_buffer.read(min)) return false;
144 double max;
145 if(!a_buffer.read(max)) return false;
146 //::printf("debug : BatchLab::RioTH::streamTAxis : %d %g %g\n",
147 // number,min,max);
148
149 std::vector<double> edges;
150 if(!Array_stream<double>(a_buffer,edges)) return false; //fXbins TArrayD
151
152 size_t edgen = edges.size();
153 if(!edgen) {
154 a_fAxis.configure(number,min,max);
155 } else {
156 std::vector<double> vedges;
157 for(size_t index=0;index<edgen;index++) {
158 vedges.push_back(edges[index]);
159 }
160 a_fAxis.configure(vedges);
161 }
162
163 int First;
164 if(!a_buffer.read(First)) return false;
165 int Last;
166 if(!a_buffer.read(Last)) return false;
167
168 if(v>=8) { //fBits2.
169 unsigned short dummy;
170 if(!a_buffer.read(dummy)) return false;
171 }
172
173 //Bool_t
174 unsigned char TimeDisplay;
175 if(!a_buffer.read(TimeDisplay)) return false;
176
177 //TString
178 std::string TimeFormat;
179 if(!a_buffer.read(TimeFormat)) return false;
180
181 if(v>=7) {
182 //THashList*
183 dummy_fac fac(a_buffer.out());
184 if(!dummy_TXxx_pointer_stream(a_buffer,fac)) return false;
185 }
186
187 if(!a_buffer.check_byte_count(s,c,"TAxis")) return false;
188 return true;
189}

195inline bool TH_read_1D(buffer& a_buffer,hd_data& a_data,
196 double& a_entries,double& a_Sw,double& a_Sw2,double& a_Sxw,double& a_Sx2w){
197 a_entries = 0;
198 a_Sw = 0;
199 a_Sw2 = 0;
200 a_Sxw = 0;
201 a_Sx2w = 0;
202
203 unsigned int s, c;
204 short vers;
205 if(!a_buffer.read_version(vers,s,c)) return false;
206
207 //::printf("debug : BatchLab::Rio::TH::streamTH1 : version %d\n",vers);
208
209 // Version 3 streaming (ROOT/v3-00-6).
210
211 std::string name;
212 std::string title;
213 if(!Named_stream(a_buffer,name,title)) return false;
214
215 a_data.m_title = title;
216
217 {short color,style,width;
218 if(!AttLine_stream(a_buffer,color,style,width)) return false;}
219 {short color,style;
220 if(!AttFill_stream(a_buffer,color,style)) return false;}
221 if(!AttMarker_stream(a_buffer)) return false;
222
223 int Ncells;
224 if(!a_buffer.read(Ncells)) return false;
225
226 //fXAxis
227 if(!Axis_stream(a_buffer,a_data.m_axes[0])) return false;
228 a_data.m_axes[0].m_offset = 1;
229
230 if(a_data.m_dimension==3) {
231 if(!Axis_stream(a_buffer,a_data.m_axes[1])) return false; //fYAxis
232 a_data.m_axes[1].m_offset = a_data.m_axes[0].m_offset * (a_data.m_axes[0].bins()+2);
233
234 if(!Axis_stream(a_buffer,a_data.m_axes[2])) return false; //fZAxis
235 a_data.m_axes[2].m_offset = a_data.m_axes[1].m_offset * (a_data.m_axes[1].bins()+2);
236
237 } else if(a_data.m_dimension==2) {
238 if(!Axis_stream(a_buffer,a_data.m_axes[1])) return false; //fYAxis
239 a_data.m_axes[1].m_offset = a_data.m_axes[0].m_offset * (a_data.m_axes[0].bins()+2);
240
241 histo::axis<double,unsigned int> dummy;
242 if(!Axis_stream(a_buffer,dummy)) return false; //fZAxis
243 } else {
244 histo::axis<double,unsigned int> dummy;
245 if(!Axis_stream(a_buffer,dummy)) return false; //fYAxis
246 if(!Axis_stream(a_buffer,dummy)) return false; //fZAxis
247 }
248
249 short barOffset;
250 if(!a_buffer.read(barOffset)) return false;
251
252 short barWidth;
253 if(!a_buffer.read(barWidth)) return false;
254
255 if(!a_buffer.read(a_entries)) return false;
256
257 if(!a_buffer.read(a_Sw)) return false; //fTsumw
258 //::printf("debug : BatchLab::Rio::TH::streamTH1 : \"%s\" %g %g\n",
259 // a_data.m_title.c_str(),fEntries,fSw);
260
261 if(!a_buffer.read(a_Sw2)) return false;
262
263 if(!a_buffer.read(a_Sxw)) return false;
264
265 if(!a_buffer.read(a_Sx2w)) return false;
266
267 double max;
268 if(!a_buffer.read(max)) return false;
269
270 double min;
271 if(!a_buffer.read(min)) return false;
272
273 double NormFactor;
274 if(!a_buffer.read(NormFactor)) return false;
275
276 {std::vector<double> v;
277 if(!Array_stream<double>(a_buffer,v)) return false;} //fContour TArrayD
278
279 std::vector<double> sumw2; //fSumw2 TArrayD
280 if(!Array_stream<double>(a_buffer,sumw2)) return false;
281
282 {std::string opt;
283 if(!a_buffer.read(opt)) return false; //TString fOption
284 //look if it is an "annotation trick" :
285 //if(opt.size()&&(opt[0]==0)) {
286 // fAnnotation = opt.substr(1,opt.size()-1);
287 //}
288 }
289
290 {dummy_fac fac(a_buffer.out());
291 obj_list dummy(fac,true,true);
292 if(!dummy.stream(a_buffer)) {
293 a_buffer.out() << "tools::rroot::TH_read_1D :"
294 << " obj_list stream failed."
295 << std::endl;
296 return false;
297 }} //Functions
298
299 if(vers>=4) {
300 int BufferSize;
301 if(!a_buffer.read(BufferSize)) return false;
302
303 //Double_t* Buffer; //[fBufferSize]
304 if(!dummy_array_stream<double>(a_buffer,BufferSize)) return false;
305 }
306
307 if(vers>=7) {
308 //EBinErrorOpt fBinStatErrOpt;
309 int dummy;
310 if(!a_buffer.read(dummy)) return false;
311 }
312
313 // Add two for outflows.
314 if(a_data.m_dimension==1) {
315 a_data.m_bin_number = a_data.m_axes[0].m_number_of_bins + 2;
316 } else if(a_data.m_dimension==2) {
317 a_data.m_bin_number =
318 (a_data.m_axes[0].m_number_of_bins + 2) *
319 (a_data.m_axes[1].m_number_of_bins + 2);
320 } else if(a_data.m_dimension==3) {
321 a_data.m_bin_number =
322 (a_data.m_axes[0].m_number_of_bins + 2) *
323 (a_data.m_axes[1].m_number_of_bins + 2) *
324 (a_data.m_axes[2].m_number_of_bins + 2);
325 }
326
327 unsigned int binn = a_data.m_bin_number;
328 a_data.m_bin_Sw2.resize(binn);
329 if(binn==sumw2.size()) {
330 for(unsigned int index=0;index<binn;index++){
331 a_data.m_bin_Sw2[index] = sumw2[index];
332 }
333 } else {
334 a_data.m_bin_Sw2.assign(binn,0);
335 }
336
337 if(!a_buffer.check_byte_count(s,c,"TH")) return false;
338
339 return true;
340}

467
468 histo::h1d* h = new histo::h1d("",10,0,1);
469 h->copy_from_data(data);
470 return h; //give ownership to caller.
471}

742 double ymax;
743 if(!a_buffer.read(ymax)) return 0;

746 double sumwy;

748 double sumwy2;
749 if(!a_buffer.read(sumwy2)) return 0;

753 if(!Array_stream<double>(a_buffer,bins_sumw2)) return 0;

755
756 if(!a_buffer.check_byte_count(s,c,"TProfile")) return 0;
757
758 data.m_is_profile = true;
759 data.m_cut_v = true;
760 data.m_min_v = ymin;
761 data.m_max_v = ymax;

770 double sw = bins[index];

777 }
778
779 histo::p1d* p = new histo::p1d("",10,0,1);
