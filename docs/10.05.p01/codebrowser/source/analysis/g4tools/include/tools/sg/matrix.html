16namespace tools {
17namespace sg {
18
19class matrix : public node {
20 TOOLS_NODE(matrix,tools::sg::matrix,node)
21public:
22 sf_mat4f mtx;
23public:
24 virtual const std::vector<field_desc>& node_fields() const {
25 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::matrix)
26 static std::vector<field_desc> s_v;
27 if(s_v.empty()) {
28 s_v = parent::node_fields();
29 TOOLS_ADD_FIELD_DESC(mtx)
30 }
31 return s_v;
32 }
33private:
34 void add_fields(){
35 add_field(&mtx);
36 }
37public:
38 virtual void render(render_action& a_action) {
39 a_action.model_matrix().mul_mtx(mtx.value(),m_tmp);
40 a_action.state().m_model = a_action.model_matrix();
41 a_action.load_model_matrix(a_action.model_matrix());
42 }
43 virtual void pick(pick_action& a_action) {
44 a_action.model_matrix().mul_mtx(mtx.value(),m_tmp);
45 a_action.state().m_model = a_action.model_matrix();
46 }
47 virtual void bbox(bbox_action& a_action) {
48 a_action.model_matrix().mul_mtx(mtx.value(),m_tmp);
49 a_action.state().m_model = a_action.model_matrix();
50 }
51 virtual void event(event_action& a_action) {
52 a_action.model_matrix().mul_mtx(mtx.value(),m_tmp);
53 a_action.state().m_model = a_action.model_matrix();
54 }
55 virtual void get_matrix(get_matrix_action& a_action) {
56 a_action.model_matrix().mul_mtx(mtx.value(),m_tmp);
57 a_action.state().m_model = a_action.model_matrix();
58 }
59 virtual void is_visible(visible_action& a_action) {
60 a_action.model_matrix().mul_mtx(mtx.value(),m_tmp);
61 a_action.state().m_model = a_action.model_matrix();
62 }
63public:
64 matrix():parent(),mtx(mat4f()) {
65#ifdef TOOLS_MEM
66 mem::increment(s_class().c_str());
67#endif
68 add_fields();
69 mtx.set_identity();
70 }
71 virtual ~matrix(){
72#ifdef TOOLS_MEM
73 mem::decrement(s_class().c_str());
74#endif
75 }
76public:
77 matrix(const matrix& a_from):parent(a_from),mtx(a_from.mtx) {
78#ifdef TOOLS_MEM
79 mem::increment(s_class().c_str());
80#endif
81 add_fields();
82 }
83 matrix& operator=(const matrix& a_from){
84 parent::operator=(a_from);
85 mtx = a_from.mtx;
86 return *this;
87 }
88public:
89 // shortcuts :
90 void set_identity() {mtx.set_identity();}
91
92 void set_translate(float a_x,float a_y,float a_z) {
93 mtx.set_translate(a_x,a_y,a_z);
94 }
95 void set_translate(const vec3f& a_v) {mtx.set_translate(a_v);}
96
97 void set_scale(float a_x,float a_y,float a_z) {mtx.set_scale(a_x,a_y,a_z);}
98 void set_scale(float a_s) {mtx.set_scale(a_s);}
99
100 void set_rotate(float a_x,float a_y,float a_z,float a_angle) {
101 mtx.set_rotate(a_x,a_y,a_z,a_angle);
102 }
103 void set_rotate(const vec3f& a_v,float a_angle) {
104 mtx.set_rotate(a_v,a_angle);
105 }
106
107 void mul_mtx(const mat4f& a_m) {mtx.mul_mtx(a_m,m_tmp);}
108
109 void mul_translate(float a_x,float a_y,float a_z) {
110 mtx.mul_translate(a_x,a_y,a_z);
111 }
112 void mul_translate(const vec3f& a_v) {mtx.mul_translate(a_v);}
113 void mul_scale(float a_x,float a_y,float a_z) {mtx.mul_scale(a_x,a_y,a_z);}
114 void mul_scale(float a_s) {mtx.mul_scale(a_s);}
115
116 void mul_rotate(float a_x,float a_y,float a_z,float a_angle) {
117 mtx.mul_rotate(a_x,a_y,a_z,a_angle);
118 }
119
120 void mul_rotate(const vec3f& a_v,float a_angle) {mtx.mul_rotate(a_v,a_angle);}
121
122 void left_mul_rotate(float a_x,float a_y,float a_z,float a_angle) {
123 mtx.left_mul_rotate(a_x,a_y,a_z,a_angle);
124 }
125
126 void left_mul_scale(float a_x,float a_y,float a_z) {
127 mtx.left_mul_scale(a_x,a_y,a_z);
128 }
129
130 void left_mul_translate(float a_x,float a_y,float a_z) {
131 mtx.left_mul_translate(a_x,a_y,a_z);
132 }
133
134 void left_mul_translate(const vec3f& a_v) {mtx.left_mul_translate(a_v);}
135
136 bool mul_rotate(const vec3f& a_from,const vec3f& a_to) {
137 return mtx.mul_rotate(a_from,a_to,m_tmp);
138 }
139protected:
140 float m_tmp[16]; //OPTIMIZATION
141};
