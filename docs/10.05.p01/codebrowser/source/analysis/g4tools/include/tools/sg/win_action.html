4#ifndef tools_sg_win_action
5#define tools_sg_win_action
6
7#include "action"
8
9namespace tools {
10namespace sg {
11
12class win_action : public action {
13public:
14 win_action(std::ostream& a_out,unsigned int a_ww,unsigned int a_wh)
15 :action(a_out)
16 ,m_ww(a_ww) //WARNING : we assume that we receive a not zero value.
17 ,m_wh(a_wh) //WARNING : we assume that we receive a not zero value.
18 {}
19 virtual ~win_action(){}
20protected:
21 win_action(const win_action& a_from)
22 :action(a_from)
23 ,m_ww(a_from.m_ww)
24 ,m_wh(a_from.m_wh)
25 {}
26 win_action& operator=(const win_action& a_from){
27 action::operator=(a_from);
28 m_ww = a_from.m_ww;
29 m_wh = a_from.m_wh;
30 return *this;
31 }
32public:
33 unsigned int ww() const {return m_ww;}
34 unsigned int wh() const {return m_wh;}
35protected:
36 unsigned int m_ww; //window width
37 unsigned int m_wh; //window height
38};
39
40}}
