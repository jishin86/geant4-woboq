47 const style_t* find_style(const std::string& a_name) const {

71 void remove_styles(const std::string& a_pattern) {
72 std::vector<named_style_t>::iterator it;
73 for(it=m_named_styles.begin();it!=m_named_styles.end();) {
74 if(match((*it).first,a_pattern)) {
75 it = m_named_styles.erase(it);
76 } else {
77 it++;
78 }
79 }

105 template <class T>
106 bool res_value(const style_t& a_sty,const std::string& a_key,T& a_v,const std::string& a_msg) const {
107 //NOTE : if ret false, we do not set a_v to something.
108
109 std::string _s;
110 if(!find(a_sty,a_key,_s)) {
111 if(a_msg.size()) {
112 m_out << "tools::xml::styles::res_value :"
113 << " key " << sout(a_key) << " not found"
114 << " in style " << sout(a_msg) << "."
115 << std::endl;
116 }
117 return false;
118 }
119 T v;
120 if(!to(_s,v)) {
121 m_out << "tools::xml::styles::res_value :"
122 << " for key " << sout(a_key) << " not found"
123 << " in style " << sout(a_msg) << "."
124 << " can't convert " << sout(_s) << " to value."
125 << std::endl;
126 return false;
127 }
128 a_v = v;
129 return true;
130 }

132 template <class T>
133 bool res_value(const std::string& a_style,const std::string& a_key,T& a_v) const {
134 //NOTE : if ret false, we do not set a_v to something.
135
136 //look if a_style is an alias :
137 std::string style;
138 is_alias(a_style,style);
139
140 const style_t* sty = find_style(style);
141 if(!sty) {
142 //m_out << "tools::xml::styles::res_value :"
143 // << " style " << sout(style) << " not found."
144 // << std::endl;
145 return false;
146 }
147 T v;
148 if(!res_value<T>(*sty,a_key,v,style)) return false;
149 a_v = v;
150 return true;
151 }

322inline void load_style(styles& a_styles,const tree& a_tree) {
323 std::string name;
324 a_tree.attribute_value("name",name);
325 if(name.empty()) {
326 a_styles.out() << "tools::sg::gui_viewer::load_style :"
327 << " <style> without name."
328 << std::endl;
329 return;
330 }
331
332 styles::style_t sty;
333
334 {looper _for(a_tree);
335 while(element* _elem = _for.next_element()) {
336
337 if(_elem->name()=="copy") {
338
339 std::string from;
340 _elem->attribute_value("from",from); //expect another style name.
341 if(from.empty()) {
342 a_styles.out() << "tools::sg::gui_viewer::load_style :"
343 << " <copy> without from."
344 << std::endl;
345 return;
346 }
347
348 const styles::style_t* csty = a_styles.find_style(from);
349 if(!csty) {
350 a_styles.out() << "tools::sg::gui_viewer::load_style :"
351 << " <copy> : from " << sout(from) << " not found."
352 << std::endl;
353 return;
354 }
355
356 append(sty,*csty);
357
358 } else {
359 sty.push_back(styles::style_item_t(_elem->name(),_elem->value()));
360 }
361
362 }}
363
364 if(sty.size()) a_styles.add_style(name,sty);
365}

367inline void load_plotter_style(styles& a_styles,const tree& a_tree) {
368 std::string pname;
369 a_tree.attribute_value("name",pname);
370 if(pname.empty()) {
371 a_styles.out() << "tools::sg::gui_viewer::load_plotter_style :"
372 << " <plotter_style> without name."
373 << std::endl;
374 return;
375 }
376
377 {styles::style_t sty;
378 sty.push_back(styles::style_item_t("tag","plotter_style"));
379
380 {looper _for(a_tree);
381 while(element* _elem = _for.next_element()) {
382
383 if(_elem->name()=="copy") {
384
385 std::string from;
386 _elem->attribute_value("from",from); //expect a plotter_style name.
387 if(from.empty()) {
388 a_styles.out() << "tools::sg::gui_viewer::load_plotter_style :"
389 << " <copy> without from."
390 << std::endl;
391 return;
392 }
393
394 const styles::style_t* csty = a_styles.find_style(from);
395 if(!csty) {
396 a_styles.out() << "tools::sg::gui_viewer::load_plotter_style :"
397 << " <copy> : from " << sout(from) << " not found."
398 << std::endl;
399 return;
400 }
401 if(csty->size()) a_styles.add_style(pname,*csty);
402
403 if(styles::is_plotter_style(*csty)) {
404 //search all <from>.<sub_style> styles :
405 std::string head = from+".";
406 std::string::size_type l = head.size();
407
408 styles sts(a_styles.out());
409
410 const std::vector<styles::named_style_t>& nss =
411 a_styles.named_styles();
412 tools_vforcit(styles::named_style_t,nss,it){
413 const std::string& name = (*it).first;
414 if(name.substr(0,l)==head) {
415 std::string tail = name.substr(l,name.size()-l);
416 const styles::style_t& ssty = (*it).second;
417 if(ssty.size()) {
418 sts.add_style(pname+"."+tail,ssty);
419 }
420 }
421 }
422
423 a_styles.append(sts);
424
425 }
426
427
428 } else {
429 sty.push_back(styles::style_item_t(_elem->name(),_elem->value()));
430 }
431
432 }}
433
434 if(sty.size()) a_styles.add_style(pname,sty);}
435
436 {looper _for(a_tree);
437 while(tree* _tree = _for.next_tree()) {
438
439 const std::string& tag = _tree->tag_name();
440 if(tag=="style") {
441
442 std::string name;
443 _tree->attribute_value("name",name);
444 if(name.empty()) {
445 a_styles.out() << "tools::sg::gui_viewer::load_plotter_style :"
446 << " <style> without name."
447 << std::endl;
448 return;
449 }
450
451 {styles::style_t sty;
452
453 {looper _for2(*_tree);
454 while(element* _elem = _for2.next_element()) {
455 sty.push_back(styles::style_item_t(_elem->name(),_elem->value()));
456 }}
457
458 if(sty.size()) {
459 std::string path = pname+"."+name;
460 a_styles.add_style(path,sty);
461 }}
462
463 } else {
464 a_styles.out() << "tools::sg::gui_viewer::load_plotter_style :"
465 << " unexpected tag " << sout(tag) << "."
466 << std::endl;
467 }
468
469 }}

473inline bool scan_style_tree(styles& a_styles,const tree& a_tree) {
474
475 if(a_tree.tag_name()!="styles") return false;
476
477 // look for aliases :
478 {looper _for(a_tree);
479 while(element* _elem = _for.next_element()) {
480
481 std::string name;
482 _elem->attribute_value("name",name);
483 if(name.empty()) {
484 a_styles.out() << "tools::sg::gui_viewer::load_style :"
485 << " <alias> without name."
486 << std::endl;
487 continue;
488 }
489 add<std::string,std::string>(a_styles.aliases(),name,_elem->value());
490 }}
491
492 // scan children :
493 {looper _for(a_tree);
494 while(tree* _tree = _for.next_tree()) {
495
496 const std::string& tag = _tree->tag_name();
497 if(tag=="style") {
498 load_style(a_styles,*_tree);
499 } else if(tag=="plotter_style") {
500 load_plotter_style(a_styles,*_tree);
501 }
502
503 }}
504
505 return true;
506}
