42 class point_writer : public virtual writer {
43 public:
44 virtual void write(ZPos a_x,ZPos a_y,ZZ a_z) {
45 if(m_size>1) {
46 ZPos x,y;
47 for(int i=-int(m_size);i<=int(m_size);i++) {
48 x = a_x + i;
49 for(int j=-int(m_size);j<=int(m_size);j++) {
50 y = a_y + j;
51 _write(x,y,a_z);
52 }
53 }
54 } else {
55 _write(a_x,a_y,a_z);
56 }
57 }
58 public:
59 point_writer(ZPixel a_pixel,buffer& a_buffer,unsigned int a_size)
60 :writer(a_pixel)
61 ,m_buffer(a_buffer)
62 ,m_size(a_size)
63 {}
64 virtual ~point_writer(){}
65 public:
66 point_writer(const point_writer& a_from)
67 :writer(a_from)
68 ,m_buffer(a_from.m_buffer)
69 ,m_size(a_from.m_size)
70 {}
71 point_writer& operator=(const point_writer& a_from){
72 writer::operator=(a_from);
73 m_size = a_from.m_size;
74 return *this;
75 }
76 protected:
77 void _write(ZPos a_x,ZPos a_y,ZZ a_z) {
78 if((a_x<m_buffer.m_begX) || (a_x>m_buffer.m_endX)) return;
79 if((a_y<m_buffer.m_begY) || (a_y>m_buffer.m_endY)) return;
80
81 ZReal zpoint = (ZReal)a_z;
82 unsigned long offset = a_y * m_buffer.m_zbw + a_x;
83 ZReal* zbuff = m_buffer.m_zbuffer + offset;
84
85 if(m_buffer.m_depth_test) {if(zpoint<*zbuff) return;}
86
87 ZPixel* zimage = m_buffer.m_zimage + offset;
88
89 /* transparency :
90 ZPixel old_pix = *zimage;
91 // need the alpha of m_pixel !
92 */
93
94 *zbuff = zpoint;
95 *zimage = m_pixel;
96 }
97 protected:
98 buffer& m_buffer;
99 unsigned int m_size;
100 };

178 bool change_size(unsigned int a_width,unsigned int a_height){
179 if(!a_width||!a_height) return false;
180
181 if(m_zbuffer && (m_zbw==a_width) && (m_zbh==a_height) ) return true;
182
183 if(m_zbuffer){
184 cmem_free(m_zbuffer);
185 cmem_free(m_zimage);
186 }
187
188 //printf ("debug:ZBufferChangeSize:%d %d\n",a_width,a_height);
189 m_zbw = a_width;
190 m_zbh = a_height;
191 m_zbuffer = cmem_alloc<ZReal>(m_zbw*m_zbh);
192 if(!m_zbuffer){
193 m_zbw = 0;
194 m_zbh = 0;
195 return false;
196 }
197
198 m_zimage = cmem_alloc<ZPixel>(m_zbw*m_zbh);
199 if(!m_zimage){
200 cmem_free(m_zbuffer);
201 m_zbw = 0;
202 m_zbh = 0;
203 return false;
204 }

229 void clear_color_buffer(ZPixel a_pixel) {
230 // Erase acoording clip region.
231 ZPos row,col;
232 for(row=m_begY;row<=m_endY;row++){
233 ZPixel* zimage = m_zimage + row * m_zbw + m_begX;
234 for(col=m_begX;col<=m_endX;col++,zimage++) *zimage = a_pixel;
235 }

449 static void WriteScanLine(void* a_tag,int a_beg,int a_end,int a_y){
450 buffer& a_buffer = *((buffer*)a_tag);
451
452 if((a_y<a_buffer.m_begY) || (a_y>a_buffer.m_endY)) return;
453 if(a_end<=a_beg) return;
454
455 if(a_beg>a_buffer.m_endX) return;
456 if(a_end<a_buffer.m_begX) return;
457
458 // border clip :
459 int beg = mx(a_beg,(int)a_buffer.m_begX);
460 int end = mn(a_end,(int)a_buffer.m_endX);
461
462 unsigned long offset = a_y * a_buffer.m_zbw + beg;
463 ZReal* zbuff = a_buffer.m_zbuffer + offset;
464 ZPixel* zimage = a_buffer.m_zimage + offset;
465
466 ZReal zpoint;
467 for(int x=beg;x<=end;x++){
468 // Computing must be the same as in edge_point_writer routine.
469 zpoint =
470 (ZReal)(- a_buffer.m_planeDC
471 - a_buffer.m_planeAC * x
472 - a_buffer.m_planeBC * a_y);
473 if(a_buffer.m_depth_test) {
474 if((zpoint>=(*zbuff))
475// &&(zpoint>=a_buffer.m_zmin) //for plane quite perpandicular to screen.
476// &&(zpoint<=a_buffer.m_zmax)
477 ){
478 *zbuff = zpoint;
479 *zimage = a_buffer.m_scan_pixel;
480 }
481 } else {
482 *zbuff = zpoint;
483 *zimage = a_buffer.m_scan_pixel;
484 }
485 zbuff ++;
486 zimage ++;
487 }
488 }

490 static void WriteLine(const point& a_beg,
491 const point& a_end,
492 writer& a_writer){
493 ZPos dx = a_end.x - a_beg.x;
494 ZPos dy = a_end.y - a_beg.y;
495 ZZ dz = a_end.z - a_beg.z;
496
497 // 6 2
498 // 5 1
499 // 7 3
500 // 8 4
501 scan_writer_1 sw1(a_writer);
502 scan_writer_2 sw2(a_writer);
503 scan_writer_3 sw3(a_writer);
504 scan_writer_4 sw4(a_writer);
505
506 if( (dx==0) && (dy==0) ) {
507 a_writer.write(a_beg.x,a_beg.y,a_beg.z);
508 a_writer.write(a_end.x,a_end.y,a_end.z);
509
510 } else if(dx==0) {
511 if(dy>0)
512 ScanLine ( a_beg.y, a_beg.x,a_beg.z, dy, dx, dz,sw2);
513 else
514 ScanLine ( a_end.y, a_end.x,a_end.z,-dy, dx,-dz,sw2);
515
516 } else if(dx>0) {
517 if((0<=dy) && (dy<=dx)) /*1*/
518 ScanLine ( a_beg.x, a_beg.y,a_beg.z, dx, dy, dz,sw1);
519 else if(dx<dy) /*2*/
520 ScanLine ( a_beg.y, a_beg.x,a_beg.z, dy, dx, dz,sw2);
521 else if((-dx<=dy) && (dy<0) ) /*3*/
522 ScanLine ( a_beg.x, a_beg.y,a_beg.z, dx,-dy, dz,sw3);
523 else if(dy<-dx) /*4*/
524 ScanLine ( a_end.y, a_end.x,a_end.z,-dy, dx,-dz,sw4);
525
526 } else { //dx<0
527 if((0<=dy) && (dy<=-dx)) /*5*/
528 ScanLine ( a_end.x, a_end.y,a_end.z,-dx, dy,-dz,sw3);
529 else if(-dx<dy) /*6*/
530 ScanLine ( a_beg.y, a_beg.x,a_beg.z, dy,-dx, dz,sw4);
531 else if((dx<=dy) && (dy<0) ) /*7*/
532 ScanLine ( a_end.x, a_end.y,a_end.z,-dx,-dy,-dz,sw1);
533 else if(dy<dx) /*8*/
534 ScanLine ( a_end.y, a_end.x,a_end.z,-dy,-dx,-dz,sw2);
535 }

539 static void ScanLine(ZPos a_x,ZPos a_y,ZZ a_z,
540 ZPos a_dx,ZPos a_dy,ZZ a_dz,
541 scan_writer& a_proc){
542 // Mid point algorithm
543 // assume 0<dx 0<=dy<=dx
544
545 ZPos end = a_x + a_dx;
546 ZPos beg = a_y;
547 ZZ incz = a_dz/(ZZ)a_dx;
548 if(a_dy==0) {
549 a_proc.write(a_x,a_y,a_z,beg);
550 while(a_x<end){
551 a_x++;
552 a_z += incz;
553 a_proc.write(a_x,a_y,a_z,beg);
554 }
555 } else if(a_dy==a_dx) {
556 a_proc.write(a_x,a_y,a_z,beg);
557 while(a_x<end){
558 a_x++;
559 a_y++;
560 a_z += incz;
561 a_proc.write(a_x,a_y,a_z,beg);
562 }
563 } else {
564 ZPos d = 2 * a_dy - a_dx;
565 ZPos incrE = 2 * a_dy;
566 ZPos incrNE = 2 * ( a_dy - a_dx);
567 a_proc.write(a_x,a_y,a_z,beg);
568 while(a_x<end){
569 if(d<=0){
570 d += incrE;
571 a_x++;
572 }else{
573 d += incrNE;
574 a_x++;
575 a_y++;
576 }
577 a_z += incz;
578 a_proc.write(a_x,a_y,a_z,beg);
579 }
580 }
