72 void set_ortho(const T& a_l,const T& a_r, //left,right
73 const T& a_b,const T& a_t, //bottom,top
74 const T& a_n,const T& a_f) { //znear,zfar
75 // from man glOrtho.
76 T tx = -(a_r+a_l)/(a_r-a_l);
77 T ty = -(a_t+a_b)/(a_t-a_b);
78 T tz = -(a_f+a_n)/(a_f-a_n);
79
80 T a_00,a_01,a_02,a_03;
81 T a_10,a_11,a_12,a_13;
82 T a_20,a_21,a_22,a_23;
83 T a_30,a_31,a_32,a_33;
84 a_00 = 2/(a_r-a_l);a_01 = 0;a_02 = 0;a_03 = tx;
85 a_10 = 0;a_11 = 2/(a_t-a_b);a_12 = 0;a_13 = ty;
86 a_20 = 0;a_21 = 0;a_22 = -2/(a_f-a_n);a_23 = tz;
87 a_30 = 0;a_31 = 0;a_32 = 0;a_33 = 1;
88 set_matrix(a_00,a_01,a_02,a_03, //1 row
89 a_10,a_11,a_12,a_13, //2 row
90 a_20,a_21,a_22,a_23, //3 row
91 a_30,a_31,a_32,a_33); //4 row
92
93 //NOTE : Z(x,y, z,1) = -2z/(f-n)+tz = [-2z-f-n]/(f-n)
94 // W(x,y, z,1) = 1 -> Z/W = Z
95 // Z(x,y,-n,1) = -1
96 // Z(x,y,-f,1) = 1
97
98 // Z(x,y,-(f+n)/2,1) = 0
99
100 // X(x,0, z,1) = 2x/(r-l)+tx = [2x-r-l]/(r-l)
101 // X(r,0, z,1) = 1
102
103 // the view direction is then (0,0,1) in the final projection.
104 }
105 void set_frustum(const T& a_l,const T& a_r, //left,right
106 const T& a_b,const T& a_t, //bottom,top
107 const T& a_n,const T& a_f) { //znear,zfar
108 // from man glFrustum.
109 T A = (a_r+a_l)/(a_r-a_l);
110 T B = (a_t+a_b)/(a_t-a_b);
111 T C = -(a_f+a_n)/(a_f-a_n);
112 T D = -(2*a_f*a_n)/(a_f-a_n);
113
114 T a_00,a_01,a_02,a_03;
115 T a_10,a_11,a_12,a_13;
116 T a_20,a_21,a_22,a_23;
117 T a_30,a_31,a_32,a_33;
118 a_00 = (2*a_n)/(a_r-a_l);a_01 = 0;a_02 = A;a_03 = 0;
119 a_10 = 0;a_11 = (2*a_n)/(a_t-a_b);a_12 = B;a_13 = 0;
120 a_20 = 0;a_21 = 0;a_22 = C;a_23 = D;
121 a_30 = 0;a_31 = 0;a_32 = -1;a_33 = 0;
122 set_matrix(a_00,a_01,a_02,a_03, //1 row
123 a_10,a_11,a_12,a_13, //2 row
124 a_20,a_21,a_22,a_23, //3 row
125 a_30,a_31,a_32,a_33); //4 row
126
127 //NOTE : Z(x,y, z,1) = C*z+D = -[(f+n)*z+2*f*n]/(f-n)
128
129 // Z(x,y,-n,1) = -[-fn-nn+2fn]/(f-n) = -[fn-nn]/(f-n) = -n
130 // W(x,y,-n,1) = n
131 // -> Z/W(x,y,-n,1) = -1
132
133 // Z(x,y,-2fn/(f+n),1) = 0
134 // -> Z/W = 0
135
136 // Z(x,y,-f,1) = -[-ff-fn+2fn]/(f-n) = -[fn-ff]/(f-n) = f
137 // W(x,y,-f,1) = f
138 // -> Z/W(x,y,-f,1) = 1
139
140 // X(x,0, z,1) = 2nx/(r-l)+z(r+l)/(r-l) = [2nx+zr+zl]/(r-l)
141 // X(r,0,-n,1) = [nr-nl]/(r-l) = n
142 // W(r,0,-n,1) = n
143 // -> X/W(r,0,-n,1) = 1
144
145 // X(l,0,-n,1) = (2nl-n(r+l))/(r-l) = -n
146 // W(l,0,-n,1) = n
147 // -> X/W(l,0,-n,1) = -1
148
149 // lrbt corners are in plane z=-1 at xy=+/-1.
150
151 // eye ?
152 // eye before ? (0,0,z,1) -> (zA,zB,zC+D,-z) /W -> (-A,-B,-(C+D/z),1)
153 // z=0 -> (0,0,D=-2fn(f-n),0)
154
155 }

170 void mul_4(T& a_x,T& a_y,T& a_z,T& a_p) const {
171 // a_[x,y,z,p] = this * a_[x,y,z,p]
172 //pr::m_vec[R + C * 4];
173 //pr::m_vec[0]= 00;pr::m_vec[4] = 01;pr::m_vec[ 8] = 02;pr::m_vec[12] = 03;
174 //pr::m_vec[1]= 10;pr::m_vec[5] = 11;pr::m_vec[ 9] = 12;pr::m_vec[13] = 13;
175 //pr::m_vec[2]= 20;pr::m_vec[6] = 21;pr::m_vec[10] = 22;pr::m_vec[14] = 23;
176 //pr::m_vec[3]= 30;pr::m_vec[7] = 31;pr::m_vec[11] = 32;pr::m_vec[15] = 33;
177 T x= pr::m_vec[0]*a_x+pr::m_vec[4]*a_y+pr::m_vec[ 8]*a_z+pr::m_vec[12]*a_p;
178 T y= pr::m_vec[1]*a_x+pr::m_vec[5]*a_y+pr::m_vec[ 9]*a_z+pr::m_vec[13]*a_p;
179 T z= pr::m_vec[2]*a_x+pr::m_vec[6]*a_y+pr::m_vec[10]*a_z+pr::m_vec[14]*a_p;
180 T p= pr::m_vec[3]*a_x+pr::m_vec[7]*a_y+pr::m_vec[11]*a_z+pr::m_vec[15]*a_p;
181 a_x = x;
182 a_y = y;
183 a_z = z;
184 a_p = p;
185 }

362 static void _set_rotate(const T& a_x,const T& a_y,const T& a_z,const T& a_angle,T v[],T(*a_sin)(T),T(*a_cos)(T)) {
363 //WARNING : (a_x,a_y,a_z) must be a normalized vector.
364 T si = a_sin(a_angle);
365 T co = a_cos(a_angle);
366 T x = a_x;
367 T y = a_y;
368 T z = a_z;
369 T x2 = x*x;
370 T y2 = y*y;
371 T z2 = z*z;
372 T xy = x*y;
373 T xz = x*z;
374 T yz = y*z;
375 v[0] = x2+(1-x2)*co;v[4] = xy*(1-co)-z*si;v[ 8] = xz*(1-co)+y*si;v[12] = 0;
376 v[1] = xy*(1-co)+z*si;v[5] = y2+(1-y2)*co;v[ 9] = yz*(1-co)-x*si;v[13] = 0;
377 v[2] = xz*(1-co)-y*si;v[6] = yz*(1-co)+x*si;v[10] = z2+(1-z2)*co;v[14] = 0;
378 v[3] = 0;v[7] = 0;v[11] = 0;v[15] = 1;
379
380 // If :
381 // n =(x,y,z)
382 // n2 = x2+y2+z2 = 1
383 // n.E = x*E1+y*E2+z*E3
384 // with :
385 // E1 E2 E3
386 // 0 0 0 0 0 -1 0 1 0
387 // 0 0 1 0 0 0 -1 0 0
388 // 0 -1 0 1 0 0 0 0 0
389 //
390 // R(r,c) = cos(theta)*Id(r,c)+(1-cos(theta))*nr*nc-sin(theta)*(n.E)(r,c)
391 //
392 // R = exp(-theta*(n.E))
393
394 }

397 void mul_mtx_rot_root(const T& a_00,const T& a_01,const T& a_02, //1 row
398 const T& a_10,const T& a_11,const T& a_12, //2 row
399 const T& a_20,const T& a_21,const T& a_22) //3 row
