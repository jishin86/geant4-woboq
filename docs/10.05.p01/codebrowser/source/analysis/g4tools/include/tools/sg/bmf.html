18namespace sg {

20template <class T>
21class bmf : public field {
22 typedef field parent;
23public:
24 static const std::string& s_class() {
25 //we do not use stype(T()).
26 static const std::string s_v("tools::sg::bmf");
27 return s_v;
28 }
29 virtual void* cast(const std::string& a_class) const {
30 if(void* p = cmp_cast< bmf<T> >(this,a_class)) {return p;}
31 return parent::cast(a_class);
32 }
33 virtual const std::string& s_cls() const {return s_class();}
34public:
35 bmf(){}
36 bmf(const T& a_value) {m_values.push_back(a_value);}
37 bmf(const std::vector<T>& a_v) {m_values = a_v;}
38 virtual ~bmf(){m_values.clear();}
39public:
40 bmf(const bmf& a_from):parent(a_from),m_values(a_from.m_values){}
41 bmf& operator=(const bmf& a_from){
42 parent::operator=(a_from);
43 if(a_from.m_values!=m_values) m_touched = true;
44 m_values = a_from.m_values;
45 return *this;
46 }
47public:
48 bmf& operator=(const std::vector<T>& a_from){
49 if(a_from!=m_values) m_touched = true;
50 m_values = a_from;
51 return *this;
52 }
53 bool operator==(const bmf& a_from) const {
54 return m_values==a_from.m_values;
55 }
56 bool operator!=(const bmf& a_from) const {
57 return !operator==(a_from);
58 }
59 const T& operator[](size_t a_index) const {
60 //WARNING : no check is done on a_index.
61 return m_values[a_index];
62 }
63 T& operator[](size_t a_index) {
64 //WARNING : no check is done on a_index.
65 return m_values[a_index];
66 }
67public:
68 size_t size() const {return m_values.size();}
69 bool empty() const {return m_values.empty();}
70 const std::vector<T>& values() const {return m_values;}
71 std::vector<T>& values() {return m_values;}
72 void add(const T& a_value) {
73 m_values.push_back(a_value);
74 m_touched = true;
75 }
76 void add(const std::vector<T>& a_vals) {
77 if(a_vals.empty()) return;
78 typedef typename std::vector<T>::const_iterator const_it_t;
79 for(const_it_t it=a_vals.begin();it!=a_vals.end();++it){
80 m_values.push_back(*it);
81 }
82 m_touched = true;
83 }
84 typedef typename std::vector<T>::iterator it_t;
85 void insert(const it_t& a_it,const T& a_value) {
86 m_values.insert(a_it,a_value);
87 m_touched = true;
88 }
89 bool set_value(size_t a_index,const T& a_value) {
90 if(a_index>=m_values.size()) return false;
91 if(m_values[a_index]!=a_value) m_touched = true;
92 m_values[a_index] = a_value;
93 return true;
94 }
95 bool get_value(size_t a_index,T& a_value) {
96 if(a_index>=m_values.size()) {a_value=T();return false;}
97 a_value = m_values[a_index];
98 return true;
99 }
100 void clear() {
101 if(m_values.size()) m_touched = true;
102 m_values.clear();
103 }
104
105 void set_values(const std::vector<T>& a_values) {
106 if(a_values!=m_values) m_touched = true;
107 m_values = a_values;
108 }
109 void set_value(const T& a_value) { //used in ArcheryTune.
110 bool to_resize = m_values.size()==1?false:true;
111 bool is_eq = ( (m_values.size()>=1) && (m_values[0]==a_value) ) ? true : false;
112 if(to_resize) m_values.resize(1);
113 if(to_resize || !is_eq) m_touched = true;
114 m_values[0] = a_value;
115 }
116
117public: //for iv2sg
118 //bool setValues(size_t a_index,size_t a_num,const T* a_vs) {
119 // for(size_t index=0;index<a_num;index++) {
120 // if(!set1Value(a_index+index,a_vs[index])) return false;
121 // }
122 // return true;
123 //}
124
125 bool setValues(size_t a_index,size_t a_num,const T* a_vs) {
126 // 012345678
127 // 234
128 if((a_index+a_num)>=m_values.size()) m_values.resize(a_index+a_num);
129 for(size_t index=0;index<a_num;index++) {
130 if(a_vs[index]!=m_values[a_index+index]) m_touched = true;
131 m_values[a_index+index] = a_vs[index];
132 }
133 return true;
134 }
135
136 bool set1Value(size_t a_index,const T& a_value) {
137 if(a_index>=m_values.size()) m_values.resize(a_index+1);
138 if(m_values[a_index]!=a_value) m_touched = true;
139 m_values[a_index] = a_value;
140 return true;
141 }
142 bool setValue(const T& a_value) {set_value(a_value);return true;}
143
144 bmf& operator=(const T& a_v){
145 if(!setValue(a_v)) {}
146 return *this;
147 }
148 size_t getNum() const {return m_values.size();}
149 T* getValues(size_t a_start) { //for gopaw.
150 if(a_start>=(m_values.size()+1)) return 0;
151 T* data = vec_data(m_values);
152 return data+a_start;
153 }
154protected:
155 std::vector<T> m_values;
156};
157
158}}
