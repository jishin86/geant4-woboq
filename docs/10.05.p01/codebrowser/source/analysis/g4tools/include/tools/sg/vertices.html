23public:
24 virtual const std::vector<field_desc>& node_fields() const {
25 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::vertices)
26 static std::vector<field_desc> s_v;
27 if(s_v.empty()) {
28 s_v = parent::node_fields();
29 TOOLS_ADD_FIELD_DESC(mode)
30 TOOLS_ADD_FIELD_DESC(xyzs)

34private:
35 void add_fields(){
36 add_field(&mode);
37 add_field(&xyzs);

39public:
40 virtual void render(render_action& a_action) {
41 if(gl::is_line(mode.value())) {
42 //Same logic as Inventor SoLightModel.model = BASE_COLOR.
43 const state& state = a_action.state();
44 a_action.set_lighting(false);
45 a_action.draw_vertex_array(mode.value(),xyzs.values());
46 a_action.set_lighting(state.m_GL_LIGHTING);
47 } else {
48 a_action.draw_vertex_array(mode.value(),xyzs.values());
49 }
50 }
51 virtual void pick(pick_action& a_action) {
52 a_action.add__primitive(*this,mode.value(),xyzs.values(),true);
53 }
54
55 virtual void bbox(bbox_action& a_action) {
56 a_action.add_points(xyzs.values());
57 }
58 virtual void is_visible(visible_action& a_action) {
59 if(_is_visible(a_action)) a_action.increment();
60 }

130 bool add_dashed_line(float a_bx,float a_by,float a_bz,
131 float a_ex,float a_ey,float a_ez,
132 unsigned int a_num_dash) {
133 //optimized version.
134 if(!a_num_dash) return false;
135 // there is a dash at beg and end of line.
136
137 float fac = 1.0f/float(2*a_num_dash-1);
138 float sx = (a_ex-a_bx)*fac;
139 float sy = (a_ey-a_by)*fac;
140 float sz = (a_ez-a_bz)*fac;
141
142 float two_sx = sx*2.0f;
143 float two_sy = sy*2.0f;
144 float two_sz = sz*2.0f;
145
146 float px = a_bx;
147 float py = a_by;
148 float pz = a_bz;
149
150 for(unsigned int idash=0;idash<a_num_dash;idash++) {
151 add(px,py,pz);
152 add(px+sx,py+sy,pz+sz);
153 px += two_sx;
154 py += two_sy;
155 pz += two_sz;
156 }
157 return true;
158 }
159protected:
160 bool _is_visible(const matrix_action& a_action) {
161 if(xyzs.empty()) return false;
162 const state& _state = a_action.state();
163 pick_action action(a_action.out(),_state.m_ww,_state.m_wh,0,float(_state.m_ww),0,float(_state.m_wh));
164 action.set_win_size(_state.m_ww,_state.m_wh);
165 action.set_area(0,float(_state.m_ww),0,float(_state.m_wh));
166 action.set_stop_at_first(true);
167 action.matrix_action::operator=(a_action); //IMPORTANT.
168 int old_cur = action.cur(); //not 0.
169 action.add__primitive(*this,mode.value(),xyzs.values(),true);
170 if(action.cur()!=old_cur) return false;
171 if(!action.node()) return false;
172 return true;
173 }
