13inline bool size(const std::string& a_file,long& a_size){
14 FILE* file = ::fopen(a_file.c_str(),"rb");
15 if(!file) {
16 a_size = 0L;
17 return false;
18 }
19 //::rewind(file);
20 ::fseek(file,0L,SEEK_END);
21 a_size = ::ftell(file);
22 ::fclose(file);
23 return true;
24}
25
26inline bool is_empty(const std::string& a_file){
27 long sz;
28 if(!size(a_file,sz)) return true; //if not existing, consider it empty.
29 return (sz==0L)?true:false;
30}
