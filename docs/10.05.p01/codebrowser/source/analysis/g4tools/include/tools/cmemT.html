15template <class T>
16inline void cmem_free(T*& a_p){
17 if(!a_p) return;
18 ::free(a_p);
19 a_p = NULL;
20#ifdef TOOLS_MEM
21 mem::decrement("cmem");
22#endif
23}
24
25template <class T>
26inline T* cmem_alloc(size_t a_num){
27 if(a_num<=0) return 0;
28 T* p = (T*)::malloc(a_num*sizeof(T));
29 if(!p) return 0;
30#ifdef TOOLS_MEM
31 mem::increment("cmem");
32#endif
33 return p;
34}
35
36template <class T>
37inline T* cmem_alloc_copy(const T* a_from,size_t a_num){
38 if(a_num<=0) return 0;
39 T* p = (T*)::malloc(a_num*sizeof(T));
40 if(!p) return 0;
41#ifdef TOOLS_MEM
42 mem::increment("cmem");
43#endif
44 //::memcpy(p,a_from,a_num*sizeof(T));
45 for(size_t i=0;i<a_num;i++) p[i] = a_from[i];
46 return p;
47}
