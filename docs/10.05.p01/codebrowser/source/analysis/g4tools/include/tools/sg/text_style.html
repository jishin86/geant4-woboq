192public:
193 bool from_string(std::ostream& a_out,const cmaps_t& a_cmaps,
194 const std::string& a_s){
195 style_parser sp;
196
197 sp.visible(visible.value());
198 sp.color(color.value());
199 //sp.transparency(transparency.value());
200 sp.back_color(back_color.value());
201 //sp.back_transparency(back_transparency.value());
202 sp.back_shadow(back_shadow.value());
203 sp.modeling(modeling.value());
204 sp.font(font.value());
205 sp.font_size(font_size.value());
206 sp.encoding(encoding.value());
207 sp.smoothing(smoothing.value());
208 sp.hinting(hinting.value());
209 sp.scale(scale.value());
210 //sp.angle(angle.value());
211 //sp.x_orientation(x_orientation.value());
212 //sp.y_orientation(y_orientation.value());
213 //sp.rotated(rotated.value());
214 sp.line_width(line_width.value());
215 sp.line_pattern(line_pattern.value());
216 sp.enforced(enforced.value());
217 sp.translation(translation.value());
218 sp.front_face(front_face.value());
219 sp.options(options.value());
220
221 if(!sp.parse(a_out,a_cmaps,a_s)) {
222 a_out << "tools::sg::text_style::from_string :"
223 << " parse failed."
224 << std::endl;
225 return false;
226 }
227
228 visible.value(sp.visible());
229 color.value(sp.color());
230 //transparency.value(sp.transparency());
231 back_color.value(sp.back_color());
232 //back_transparency.value(sp.back_transparency());
233 back_shadow.value(sp.back_shadow());
234 modeling.value(sp.modeling());
235 font.value(sp.font());
236 font_size.value(sp.font_size());
237 encoding.value(sp.encoding());
238 smoothing.value(sp.smoothing());
239 hinting.value(sp.hinting());
240 scale.value(sp.scale());
241 //angle.value(sp.angle());
242 //x_orientation.value(sp.x_orientation());
243 //y_orientation.value(sp.y_orientation());
244 //rotated.value(sp.rotated());
245 line_width.value(sp.line_width());
246 line_pattern.value(sp.line_pattern());
247 enforced.value(sp.enforced());
248 translation.value(sp.translation());
249 front_face = sp.front_face();
250 options = sp.options();
251
252 return true;
253 }
