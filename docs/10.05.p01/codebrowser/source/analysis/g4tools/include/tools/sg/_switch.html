4#ifndef tools_sg__switch
5#define tools_sg__switch
6
7// NOTE : "switch" is a C/C++ keyword. Have "_switch" as a class name.
8
9#include "group"
10#include "sf"
11
12namespace tools {
13namespace sg {
14
15class _switch : public group {
16 TOOLS_NODE(_switch,tools::sg::_switch,group)
17public:
18 sf<int> which;
19public:
20 virtual const std::vector<field_desc>& node_fields() const {
21 TOOLS_FIELD_DESC_NODE_CLASS(tools::sg::_switch)
22 static std::vector<field_desc> s_v;
23 if(s_v.empty()) {
24 s_v = parent::node_fields();
25 TOOLS_ADD_FIELD_DESC(which)
26 }
27 return s_v;
28 }
29private:
30 void add_fields(){
31 add_field(&which);
32 }
33public: //node
34 virtual void render(render_action& a_action) {
35 int index = which.value();
36 if(index==(-1)) {
37 group::render(a_action);
38 } else if((index>=0)&&(index<(int)m_children.size())) {
39 m_children[index]->render(a_action);
40 }
41 }
42 virtual void pick(pick_action& a_action) {
43 int index = which.value();
44 if(index==(-1)) {
45 group::pick(a_action);
46 } else if((index>=0)&&(index<(int)m_children.size())) {
47 m_children[index]->pick(a_action);
48 }
49 }
50 virtual void bbox(bbox_action& a_action) {
51 int index = which.value();
52 if(index==(-1)) {
53 group::bbox(a_action);
54 } else if((index>=0)&&(index<(int)m_children.size())) {
55 m_children[index]->bbox(a_action);
56 }
57 }
58 virtual void search(search_action& a_action) {
59 int index = which.value();
60 if(index==(-1)) {
61 group::search(a_action);
62 } else if((index>=0)&&(index<(int)m_children.size())) {
63 if(a_action.do_path()) a_action.path_push(this);
64 m_children[index]->search(a_action);
65 if(a_action.done()) return;
66 if(a_action.do_path()) a_action.path_pop();
67 }
68 }
69 virtual void event(event_action& a_action) {
70 if(a_action.do_switch_children()) {
71 group::event(a_action);
72 } else {
73 int index = which.value();
74 if(index==(-1)){
75 group::event(a_action);
76 } else if((index>=0)&&(index<(int)m_children.size())) {
77 m_children[index]->event(a_action);
78 }
79 }
80 }
81 virtual bool write(write_action& a_action) {
82 if(!a_action.beg_node(*this)) return false;
83 if(!write_fields(a_action)) return false;
84
85 if(a_action.switch_do_all_children()) { //for exam write_exsg
86 if(!write_children(a_action)) return false;
87 } else { //for exa write_gl2ps
88 int index = which.value();
89 if(index==(-1)) {
90 if(!write_children(a_action)) return false;
91 } else if((index>=0)&&(index<(int)m_children.size())) {
92 m_children[index]->write(a_action);
93 }
94 }
95 if(!a_action.end_node(*this)) return false;
96 return true;
97 }
98public:
99 _switch()
100 :group()
101 ,which(-1)
102 {
103 add_fields();
104 }
105 virtual ~_switch(){}
106public:
107 _switch(const _switch& a_from)
108 :group(a_from)
109 ,which(a_from.which)
110 {
111 add_fields();
112 }
113 _switch& operator=(const _switch& a_from){
114 group::operator=(a_from);
115 which = a_from.which;
116 return *this;
117 }
118public:
119 node* which_node() const {
120 int index = which.value();
121 if(index<0) return 0;
122 if(index>=(int)m_children.size()) return 0;
123 return m_children[index];
124 }
125};
126
127}}
128
129#endif
