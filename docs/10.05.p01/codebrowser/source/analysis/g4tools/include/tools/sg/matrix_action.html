13namespace sg {

15class matrix_action : public win_action,public states {
16public:
17 matrix_action(std::ostream& a_out,unsigned int a_ww,unsigned int a_wh)
18 :win_action(a_out,a_ww,a_wh)
19 ,states(a_ww,a_wh)
20 ,m_cur(0)
21 ,m_landscape(true)
22 {
23 m_projs.resize(5);
24 m_models.resize(5);
25 reset();
26 m_identity.set_identity();
27 }
28 virtual ~matrix_action(){}
29public:
30 matrix_action(const matrix_action& a_from)
31 :win_action(a_from)
32 ,states(a_from)
33 ,m_projs(a_from.m_projs)
34 ,m_models(a_from.m_models)
35 ,m_cur(a_from.m_cur)
36 ,m_landscape(a_from.m_landscape)
37 {
38 m_identity.set_identity();
39 }
40 matrix_action& operator=(const matrix_action& a_from){
41 win_action::operator=(a_from);
42 states::operator=(a_from);
43 m_projs = a_from.m_projs;
44 m_models = a_from.m_models;
45 m_cur = a_from.m_cur;
46 m_landscape = a_from.m_landscape;
47 return *this;
48 }
49public:
50 void push_matrices() {
51 if((m_cur+1)>=(int)m_projs.size()) {
52 m_projs.resize(m_projs.size()+5);
53 m_models.resize(m_models.size()+5);
54 }
55 m_cur++;
56 m_projs[m_cur].set_matrix(m_projs[m_cur-1]);
57 m_models[m_cur].set_matrix(m_models[m_cur-1]);
58//OPTIMIZATION : in fact the two lines below are not needed !
59// m_state.m_proj = m_projs[m_cur];
60// m_state.m_model = m_models[m_cur];
61 }
62
63 //WARNING : in the three below methods, there is no
64 // protection against m_cur<0 being zero here.
65 void pop_matrices() {
66 m_cur--;
67//OPTIMIZATION : in fact the two lines below are not needed !
68// m_state.m_proj = m_projs[m_cur];
69// m_state.m_model = m_models[m_cur];
70 }
71 mat4f& projection_matrix() {return m_projs[m_cur];}
72 mat4f& model_matrix() {return m_models[m_cur];}
73
74 bool end() const {return m_cur==0?true:false;}
75 int cur() const {return m_cur;}
76
77 bool project_point(float& a_x,float& a_y,float& a_z,float& a_w) {
78 a_w = 1;
79 model_matrix().mul_4f(a_x,a_y,a_z,a_w);
80 projection_matrix().mul_4f(a_x,a_y,a_z,a_w);
81 if(a_w==0.0F) return false;
82 a_x /= a_w;
83 a_y /= a_w;
84 a_z /= a_w;
85 return true;
86 }
87
88 void model_point(float& a_x,float& a_y,float& a_z,float& a_w) {
89 a_w = 1;
90 model_matrix().mul_4f(a_x,a_y,a_z,a_w);
91 }
92
93 // for marker rendering :
94 void projected_origin(float& a_x,float& a_y,float& a_z) {
95 a_x = 0;
96 a_y = 0;
97 a_z = 0;
98 float w;
99 project_point(a_x,a_y,a_z,w); //if render : in [-1,1][-1,1]
100 }
101
102protected:
103 void reset() {
104 m_cur = 0;
105 if(m_landscape) {
106 m_projs[m_cur].set_identity();
107 } else {
108 m_projs[m_cur].set_matrix(0,-1,0,0,
109 1, 0,0,0,
110 0, 0,1,0,
111 0, 0,0,1);
112 }
113 m_models[m_cur].set_identity();
114 m_state.m_proj = m_projs[m_cur];
115 m_state.m_model = m_models[m_cur];
116 }
117protected:
118 std::vector<mat4f> m_projs;
119 std::vector<mat4f> m_models;
120 int m_cur;
121 mat4f m_identity;
122private:
123 bool m_landscape; //false not yet ready.
124};
125
126}}

128#endif
