5#define tools_sg_states

7#include "state"

9namespace tools {
10namespace sg {
11
12class states {
13public:
14 states(unsigned int a_ww,unsigned int a_wh):m_states(8){
15 m_state.m_ww = a_ww;
16 m_state.m_wh = a_wh;
17 }
18 virtual ~states(){}
19protected:
20 states(const states& a_from)
21 :m_state(a_from.m_state)
22 ,m_states(a_from.m_states)
23 ,m_saved_state(a_from.m_saved_state)
24 {}
25 states& operator=(const states& a_from){
26 m_state = a_from.m_state;
27 m_states = a_from.m_states;
28 m_saved_state = a_from.m_saved_state;
29 return *this;
30 }
31public:
32 const sg::state& state() const {return m_state;}
33 sg::state& state() {return m_state;}
34 void pop_state() {
35 //if(!m_states.size()) return; //throw.
36 m_state = m_states.back();
37 m_states.pop_back();
38 }
39 void push_state() {m_states.push_back(m_state);}
40
41 void save_state(const sg::state& a_state) {m_saved_state = a_state;}
42 const sg::state& saved_state() const {return m_saved_state;}
43 sg::state& saved_state() {return m_saved_state;}
44protected:
45 sg::state m_state;
46 std::vector<sg::state> m_states;
47 sg::state m_saved_state;
48};
49
50}}
