77 virtual void render(render_action& a_action) {
78 const state& state = a_action.state();
79
80 bool draw_edges = state.m_GL_LIGHTING?false:true;
81
82 if(state.m_use_gsto) {
83 unsigned int _id = get_gsto_id(a_action.out(),a_action.gl_manager());
84 if(_id) {
85 bufpos pxyzs = 0;
86 bufpos ptris = m_gsto_segs_sz*sizeof(float);
87 bufpos pnms = ptris+m_gsto_tris_sz*sizeof(float);
88
89 a_action.begin_gsto(_id);
90 if(draw_edges) {
91 //a_action.set_lighting(false); //NOTE : we should do that if style==draw_fill !
92 a_action.color4f(0,0,0,1); //if lighten, then rendered grey.
93 a_action.line_width(1);
94
95 a_action.draw_gsto_v(gl::lines(),m_gsto_segs_sz/3,pxyzs);
96
97 //pushes back the filled polygons to avoid z-fighting with lines
98 a_action.set_polygon_offset(true);
99
100 a_action.color4f(state.m_color);
101 a_action.line_width(state.m_line_width);
102 //a_action.set_lighting(state.m_GL_LIGHTING);
103 }
104 a_action.draw_gsto_vn(gl::triangles(),m_gsto_tris_sz/3,ptris,pnms);
105 a_action.set_polygon_offset(state.m_GL_POLYGON_OFFSET_FILL);
106 a_action.end_gsto();
107 return;
108
109 } else { //!_id
110 // use immediate rendering.
111 }
112
113 } else {
114 clean_gstos(&a_action.gl_manager());
115 }
116
117 // immediate rendering :
118 if(draw_edges) {
119 a_action.color4f(0,0,0,1); //if lighten, then rendered grey.
120 a_action.line_width(1);
121
122 floatp lines;
123 _lines(lines);
124 a_action.draw_vertex_array(gl::lines(),144,lines);
125
126 a_action.set_polygon_offset(true);
127
128 a_action.color4f(state.m_color);
129 a_action.line_width(state.m_line_width);
130 }
131
132 floatp tris,nms;
133 _tris(tris,nms);
134 a_action.draw_vertex_normal_array(gl::triangles(),108,tris,nms);
135 a_action.set_polygon_offset(state.m_GL_POLYGON_OFFSET_FILL);
136 }
